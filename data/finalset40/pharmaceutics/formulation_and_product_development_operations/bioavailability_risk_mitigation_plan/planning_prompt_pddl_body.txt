  (:action register_development_entity
    :parameters (?development_entity - development_entity)
    :precondition
      (and
        (not
          (entity_registered ?development_entity)
        )
        (not
          (entity_ready_for_next_stage ?development_entity)
        )
      )
    :effect (entity_registered ?development_entity)
  )
  (:action assign_mitigation_strategy_to_development_entity
    :parameters (?development_entity - development_entity ?mitigation_strategy - mitigation_strategy)
    :precondition
      (and
        (entity_registered ?development_entity)
        (not
          (entity_mitigation_allocated ?development_entity)
        )
        (mitigation_strategy_available ?mitigation_strategy)
      )
    :effect
      (and
        (entity_mitigation_allocated ?development_entity)
        (mitigation_assigned_to_entity ?development_entity ?mitigation_strategy)
        (not
          (mitigation_strategy_available ?mitigation_strategy)
        )
      )
  )
  (:action assign_bioavailability_assay_to_development_entity
    :parameters (?development_entity - development_entity ?bioavailability_assay - bioavailability_assay)
    :precondition
      (and
        (entity_registered ?development_entity)
        (entity_mitigation_allocated ?development_entity)
        (assay_available ?bioavailability_assay)
      )
    :effect
      (and
        (assay_allocated_to_entity ?development_entity ?bioavailability_assay)
        (not
          (assay_available ?bioavailability_assay)
        )
      )
  )
  (:action confirm_development_entity_screening
    :parameters (?development_entity - development_entity ?bioavailability_assay - bioavailability_assay)
    :precondition
      (and
        (entity_registered ?development_entity)
        (entity_mitigation_allocated ?development_entity)
        (assay_allocated_to_entity ?development_entity ?bioavailability_assay)
        (not
          (entity_screened ?development_entity)
        )
      )
    :effect (entity_screened ?development_entity)
  )
  (:action release_bioavailability_assay
    :parameters (?development_entity - development_entity ?bioavailability_assay - bioavailability_assay)
    :precondition
      (and
        (assay_allocated_to_entity ?development_entity ?bioavailability_assay)
      )
    :effect
      (and
        (assay_available ?bioavailability_assay)
        (not
          (assay_allocated_to_entity ?development_entity ?bioavailability_assay)
        )
      )
  )
  (:action assign_formulation_expert_to_development_entity
    :parameters (?development_entity - development_entity ?formulation_expert - formulation_expert)

        )
      )
  )
  (:action assign_development_material_to_project

    :parameters (?oral_formulation_variant - oral_formulation_variant ?critical_quality_attribute - critical_quality_attribute ?bioavailability_assay - bioavailability_assay)

      (and
        (entity_screened ?oral_formulation_variant)

        (not
          (cqa_flagged ?critical_quality_attribute)
        )
        (not
          (cqa_addressed_by_excipient ?critical_quality_attribute)
        )
      )
    :effect (cqa_flagged ?critical_quality_attribute)
  )
  (:action prioritize_variant_with_expert_input
    :parameters (?oral_formulation_variant - oral_formulation_variant ?critical_quality_attribute - critical_quality_attribute ?formulation_expert - formulation_expert)
    :precondition
      (and
        (entity_screened ?oral_formulation_variant)
        (expert_assigned_to_entity ?oral_formulation_variant ?formulation_expert)
        (cqa_associated_with_oral_variant ?oral_formulation_variant ?critical_quality_attribute)
        (cqa_flagged ?critical_quality_attribute)
        (not
          (oral_variant_ready_for_assembly ?oral_formulation_variant)
        )

        (oral_variant_ready_for_assembly ?oral_formulation_variant)

  )

          (oral_variant_prioritized ?oral_formulation_variant)

    :effect
      (and
        (cqa_flagged ?critical_quality_attribute)
        (oral_variant_prioritized ?oral_formulation_variant)
        (excipient_available ?excipient_option)
        (not
          (excipient_assigned_to_oral_variant ?oral_formulation_variant ?excipient_option)
        )
      )
  )

          (prototype_cqa_verified ?prototype_batch)

      (and
        (project_regulatory_consultation_engaged ?development_project)
