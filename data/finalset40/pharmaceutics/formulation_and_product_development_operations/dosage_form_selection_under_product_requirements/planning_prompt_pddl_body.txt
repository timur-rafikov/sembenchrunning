  (:action shortlist_dosage_form_candidate
    :parameters (?candidate - dosage_form_candidate)
    :precondition
      (and
        (not
          (shortlisted ?candidate)
        )
        (not
          (selected ?candidate)
        )
      )
    :effect (shortlisted ?candidate)
  )
  (:action associate_candidate_with_platform
    :parameters (?candidate - dosage_form_candidate ?platform - manufacturing_platform)
    :precondition
      (and
        (shortlisted ?candidate)
        (not
          (platform_option_confirmed ?candidate)
        )
        (platform_available ?platform)
      )
    :effect
      (and
        (platform_option_confirmed ?candidate)
        (platform_reserved_for ?candidate ?platform)
        (not
          (platform_available ?platform)
        )
      )
  )
  (:action assign_analytical_method_to_candidate
    :parameters (?candidate - dosage_form_candidate ?analytical_method - analytical_method)
    :precondition
      (and
        (shortlisted ?candidate)
        (platform_option_confirmed ?candidate)
        (analytical_method_available ?analytical_method)
      )
    :effect
      (and
        (analytical_method_assigned ?candidate ?analytical_method)
        (not
          (analytical_method_available ?analytical_method)
        )
      )
  )
  (:action validate_candidate_by_analytical_method
    :parameters (?candidate - dosage_form_candidate ?analytical_method - analytical_method)
    :precondition
      (and
        (shortlisted ?candidate)
        (platform_option_confirmed ?candidate)
        (analytical_method_assigned ?candidate ?analytical_method)
        (not
          (validated ?candidate)
        )
      )
    :effect (validated ?candidate)
  )
  (:action release_analytical_method
    :parameters (?candidate - dosage_form_candidate ?analytical_method - analytical_method)

      (and
        (analytical_method_assigned ?candidate ?analytical_method)
      )
    :effect
      (and
        (analytical_method_available ?analytical_method)
        (not
          (analytical_method_assigned ?candidate ?analytical_method)
        )
      )
  )
  (:action assign_evaluation_team_to_candidate
    :parameters (?candidate - dosage_form_candidate ?evaluation_team - evaluation_team)

        (validated ?candidate)

        (evaluation_team_assigned ?candidate ?evaluation_team)

          (product_assigned_safety_assessment ?product_concept ?safety_assessment)

  (:action schedule_unit_operation_test_for_formulation
    :parameters (?formulation_track - formulation_track_candidate ?unit_operation - unit_operation ?analytical_method - analytical_method)
    :precondition
      (and
        (validated ?formulation_track)
        (analytical_method_assigned ?formulation_track ?analytical_method)
        (formulation_unit_operation_link ?formulation_track ?unit_operation)
        (not
          (unit_operation_ready ?unit_operation)
        )
        (not
          (unit_operation_experiment_recorded ?unit_operation)
        )
      )
    :effect (unit_operation_ready ?unit_operation)
  )

    :parameters (?formulation_track - formulation_track_candidate ?unit_operation - unit_operation ?evaluation_team - evaluation_team)
    :precondition
      (and
        (validated ?formulation_track)
        (evaluation_team_assigned ?formulation_track ?evaluation_team)
        (formulation_unit_operation_link ?formulation_track ?unit_operation)
        (unit_operation_ready ?unit_operation)
        (not
          (formulation_ready ?formulation_track)
        )
      )
    :effect
      (and
        (formulation_ready ?formulation_track)
        (formulation_evaluated ?formulation_track)
      )
  )
  (:action conduct_excipient_screening_for_formulation
    :parameters (?formulation_track - formulation_track_candidate ?unit_operation - unit_operation ?excipient_set - excipient_set)
    :precondition
      (and
        (validated ?formulation_track)
        (formulation_unit_operation_link ?formulation_track ?unit_operation)
        (excipient_available ?excipient_set)
        (not
          (formulation_ready ?formulation_track)
        )
      )
    :effect
      (and
        (unit_operation_experiment_recorded ?unit_operation)
        (formulation_ready ?formulation_track)
        (formulation_excipient_association ?formulation_track ?excipient_set)
        (not
          (excipient_available ?excipient_set)
        )
      )
  )

    :parameters (?formulation_track - formulation_track_candidate ?unit_operation - unit_operation ?analytical_method - analytical_method ?excipient_set - excipient_set)

        (validated ?formulation_track)

        (formulation_unit_operation_link ?formulation_track ?unit_operation)
        (unit_operation_experiment_recorded ?unit_operation)
        (formulation_excipient_association ?formulation_track ?excipient_set)
        (not
          (formulation_evaluated ?formulation_track)
        )
      )
    :effect

        (unit_operation_ready ?unit_operation)
        (formulation_evaluated ?formulation_track)

          (formulation_excipient_association ?formulation_track ?excipient_set)
        )
      )
  )

    :parameters (?process_track - process_track_candidate ?route - route_of_administration ?analytical_method - analytical_method)
    :precondition
      (and
        (validated ?process_track)
        (analytical_method_assigned ?process_track ?analytical_method)
        (process_route_assignment ?process_track ?route)
        (not
          (route_ready ?route)
        )
        (not
          (route_excipient_assigned ?route)
        )
      )

  )
  (:action perform_process_route_team_evaluation

        (route_excipient_assigned ?route)
        (process_ready ?process_track)
        (process_excipient_association ?process_track ?excipient_set)
        (not
          (excipient_available ?excipient_set)
        )
      )
  )
  (:action finalize_process_excipient_experiment
    :parameters (?process_track - process_track_candidate ?route - route_of_administration ?analytical_method - analytical_method ?excipient_set - excipient_set)

        (stability_plan_unit_operation_link ?stability_plan ?unit_operation)

          (stability_plan_available ?stability_plan)

          (stability_plan_available ?stability_plan)

          (stability_plan_includes_accelerated_study ?stability_plan)

          (product_study_ready_for_assessment ?product_concept)

  (:action prepare_product_study_for_assessment_accelerated
    :parameters (?product_concept - product_concept ?compatibility_test - compatibility_test ?evaluation_team - evaluation_team ?stability_study - stability_study ?stability_plan - stability_plan)
