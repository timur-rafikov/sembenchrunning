  (:action register_field_alert_case
    :parameters (?field_alert_case - field_alert_case)
    :precondition
      (and
        (not
          (field_alert_registered ?field_alert_case)
        )
        (not
          (field_alert_closed ?field_alert_case)
        )
      )
    :effect (field_alert_registered ?field_alert_case)
  )
  (:action assign_investigator_to_case
    :parameters (?field_alert_case - field_alert_case ?investigator - investigator)
    :precondition
      (and
        (field_alert_registered ?field_alert_case)
        (not
          (field_alert_investigator_assigned ?field_alert_case)
        )
        (investigator_available ?investigator)
      )
    :effect
      (and
        (field_alert_investigator_assigned ?field_alert_case)
        (investigator_assigned_to_case ?field_alert_case ?investigator)
        (not
          (investigator_available ?investigator)
        )
      )
  )
  (:action attach_sample_to_case
    :parameters (?field_alert_case - field_alert_case ?sample - sample)
    :precondition
      (and
        (field_alert_registered ?field_alert_case)
        (field_alert_investigator_assigned ?field_alert_case)
        (sample_available ?sample)
      )
    :effect
      (and
        (field_alert_linked_to_sample ?field_alert_case ?sample)
        (not
          (sample_available ?sample)
        )
      )
  )
  (:action confirm_case_investigation
    :parameters (?field_alert_case - field_alert_case ?sample - sample)
    :precondition
      (and
        (field_alert_registered ?field_alert_case)
        (field_alert_investigator_assigned ?field_alert_case)
        (field_alert_linked_to_sample ?field_alert_case ?sample)
        (not
          (field_alert_confirmed ?field_alert_case)
        )
      )
    :effect (field_alert_confirmed ?field_alert_case)
  )
  (:action release_sample_from_case
    :parameters (?field_alert_case - field_alert_case ?sample - sample)
    :precondition
      (and
        (field_alert_linked_to_sample ?field_alert_case ?sample)
      )
    :effect
      (and
        (sample_available ?sample)
        (not
          (field_alert_linked_to_sample ?field_alert_case ?sample)
        )
      )
  )
  (:action allocate_quarantine_site_to_case
    :parameters (?field_alert_case - field_alert_case ?quarantine_site - quarantine_site)
    :precondition
      (and
        (field_alert_confirmed ?field_alert_case)
        (quarantine_site_available ?quarantine_site)
      )
    :effect
      (and
        (field_alert_allocated_to_quarantine_site ?field_alert_case ?quarantine_site)
        (not
          (quarantine_site_available ?quarantine_site)
        )
      )
  )
  (:action deallocate_quarantine_site_from_case
    :parameters (?field_alert_case - field_alert_case ?quarantine_site - quarantine_site)
    :precondition
      (and
        (field_alert_allocated_to_quarantine_site ?field_alert_case ?quarantine_site)
      )
    :effect
      (and
        (quarantine_site_available ?quarantine_site)
        (not
          (field_alert_allocated_to_quarantine_site ?field_alert_case ?quarantine_site)
        )
      )
  )
  (:action assign_logistics_asset_to_team
    :parameters (?crisis_management_team - crisis_management_team ?logistics_asset - logistics_asset)
    :precondition
      (and
        (field_alert_confirmed ?crisis_management_team)
        (logistics_asset_available ?logistics_asset)
      )

      (and
        (logistics_asset_assigned_to_team ?crisis_management_team ?logistics_asset)

          (logistics_asset_available ?logistics_asset)
        )
      )

    :parameters (?crisis_management_team - crisis_management_team ?logistics_asset - logistics_asset)

      (and
        (logistics_asset_assigned_to_team ?crisis_management_team ?logistics_asset)
      )
    :effect
      (and
        (logistics_asset_available ?logistics_asset)
        (not
          (logistics_asset_assigned_to_team ?crisis_management_team ?logistics_asset)
        )
      )
  )
  (:action link_regulatory_agency_to_team
    :parameters (?crisis_management_team - crisis_management_team ?regulatory_agency - regulatory_agency)
    :precondition
      (and
        (field_alert_confirmed ?crisis_management_team)
        (regulatory_agency_contact_available ?regulatory_agency)
      )
    :effect
      (and
        (regulatory_agency_linked_to_team ?crisis_management_team ?regulatory_agency)
        (not
          (regulatory_agency_contact_available ?regulatory_agency)
        )
      )
  )
  (:action unlink_regulatory_agency_from_team
    :parameters (?crisis_management_team - crisis_management_team ?regulatory_agency - regulatory_agency)

      (and
        (regulatory_agency_linked_to_team ?crisis_management_team ?regulatory_agency)

        (regulatory_agency_contact_available ?regulatory_agency)

          (regulatory_agency_linked_to_team ?crisis_management_team ?regulatory_agency)
        )
      )
  )
  (:action flag_distribution_node_primary
    :parameters (?site_quality_team - site_quality_team ?distribution_node - distribution_node ?sample - sample)
    :precondition
      (and
        (field_alert_confirmed ?site_quality_team)
        (field_alert_linked_to_sample ?site_quality_team ?sample)
        (site_linked_to_distribution_node ?site_quality_team ?distribution_node)
        (not
          (distribution_node_flagged_primary ?distribution_node)
        )
        (not
          (distribution_node_flagged_secondary ?distribution_node)
        )
      )
    :effect (distribution_node_flagged_primary ?distribution_node)
  )
  (:action record_site_containment_actions
    :parameters (?site_quality_team - site_quality_team ?distribution_node - distribution_node ?quarantine_site - quarantine_site)
    :precondition
      (and
        (field_alert_confirmed ?site_quality_team)
        (field_alert_allocated_to_quarantine_site ?site_quality_team ?quarantine_site)
        (site_linked_to_distribution_node ?site_quality_team ?distribution_node)
        (distribution_node_flagged_primary ?distribution_node)
        (not
          (site_containment_completed ?site_quality_team)
        )
      )
    :effect
      (and
        (site_containment_completed ?site_quality_team)
        (site_containment_recorded ?site_quality_team)
      )
  )
  (:action apply_mitigation_resource_to_site
    :parameters (?site_quality_team - site_quality_team ?distribution_node - distribution_node ?mitigation_resource - mitigation_resource)

      (and
        (field_alert_confirmed ?site_quality_team)

        (not
          (site_containment_completed ?site_quality_team)
        )
      )
    :effect
      (and
        (distribution_node_flagged_secondary ?distribution_node)
        (site_containment_completed ?site_quality_team)
        (mitigation_resource_assigned_to_site ?site_quality_team ?mitigation_resource)
        (not
          (mitigation_resource_available ?mitigation_resource)
        )
      )
  )
  (:action finalize_site_containment_and_release_resource
    :parameters (?site_quality_team - site_quality_team ?distribution_node - distribution_node ?sample - sample ?mitigation_resource - mitigation_resource)

      (and
        (field_alert_confirmed ?site_quality_team)
        (field_alert_linked_to_sample ?site_quality_team ?sample)
        (site_linked_to_distribution_node ?site_quality_team ?distribution_node)
        (distribution_node_flagged_secondary ?distribution_node)
        (mitigation_resource_assigned_to_site ?site_quality_team ?mitigation_resource)
        (not
          (site_containment_recorded ?site_quality_team)
        )
      )
    :effect
      (and
        (distribution_node_flagged_primary ?distribution_node)
        (site_containment_recorded ?site_quality_team)
        (mitigation_resource_available ?mitigation_resource)
        (not
          (mitigation_resource_assigned_to_site ?site_quality_team ?mitigation_resource)
        )
      )

          (shipment_flagged_secondary ?shipment)

  (:action record_supply_chain_containment_actions
    :parameters (?supply_chain_team - supply_chain_team ?shipment - shipment ?quarantine_site - quarantine_site)

          (supply_chain_containment_completed ?supply_chain_team)

      (and
        (supply_chain_containment_completed ?supply_chain_team)
        (supply_chain_containment_recorded ?supply_chain_team)

  (:action allocate_mitigation_to_shipment
    :parameters (?supply_chain_team - supply_chain_team ?shipment - shipment ?mitigation_resource - mitigation_resource)

      (and
        (field_alert_confirmed ?supply_chain_team)
        (supply_chain_linked_to_shipment ?supply_chain_team ?shipment)
        (mitigation_resource_available ?mitigation_resource)

          (supply_chain_containment_completed ?supply_chain_team)

          (mitigation_resource_available ?mitigation_resource)
        )

  (:action finalize_supply_chain_containment_and_release_resource
    :parameters (?supply_chain_team - supply_chain_team ?shipment - shipment ?sample - sample ?mitigation_resource - mitigation_resource)
    :precondition

        (not
          (supply_chain_containment_recorded ?supply_chain_team)

    :effect
      (and
        (shipment_flagged_primary ?shipment)
        (supply_chain_containment_recorded ?supply_chain_team)

          (mitigation_resource_assigned_to_supply_chain ?supply_chain_team ?mitigation_resource)

      (and
        (site_containment_completed ?site_quality_team)
        (supply_chain_containment_completed ?supply_chain_team)

        (supply_chain_linked_to_shipment ?supply_chain_team ?shipment)
        (distribution_node_flagged_primary ?distribution_node)
        (shipment_flagged_primary ?shipment)
        (site_containment_recorded ?site_quality_team)
        (supply_chain_containment_recorded ?supply_chain_team)
        (recall_notice_draft ?recall_notice)
      )
    :effect
      (and
        (recall_notice_issued ?recall_notice)
        (recall_targets_distribution_node ?recall_notice ?distribution_node)
        (recall_targets_shipment ?recall_notice ?shipment)
        (not
          (recall_notice_draft ?recall_notice)
        )
      )
  )
  (:action issue_recall_notice_with_primary_evidence

    :precondition
      (and
        (site_containment_completed ?site_quality_team)
        (supply_chain_containment_completed ?supply_chain_team)
        (site_linked_to_distribution_node ?site_quality_team ?distribution_node)
        (supply_chain_linked_to_shipment ?supply_chain_team ?shipment)
        (distribution_node_flagged_secondary ?distribution_node)
        (shipment_flagged_primary ?shipment)
        (not
          (site_containment_recorded ?site_quality_team)

        (supply_chain_containment_recorded ?supply_chain_team)
        (recall_notice_draft ?recall_notice)

    :effect

        (recall_notice_issued ?recall_notice)
        (recall_targets_distribution_node ?recall_notice ?distribution_node)

        (not
          (recall_notice_draft ?recall_notice)
        )
      )
  )
  (:action issue_recall_notice_with_secondary_evidence

        (supply_chain_linked_to_shipment ?supply_chain_team ?shipment)
        (distribution_node_flagged_primary ?distribution_node)
        (shipment_flagged_secondary ?shipment)
        (site_containment_recorded ?site_quality_team)
        (not
          (supply_chain_containment_recorded ?supply_chain_team)

        (recall_notice_draft ?recall_notice)

        (recall_notice_issued ?recall_notice)

          (recall_notice_draft ?recall_notice)

        (recall_has_primary_evidence ?recall_notice)
        (recall_has_secondary_evidence ?recall_notice)
        (not
          (recall_notice_draft ?recall_notice)

          (recall_confirmed_effective ?recall_notice)

        (team_has_lab_batch_custody ?crisis_management_team ?lab_batch)
        (lab_batch_available ?lab_batch)
        (recall_notice_issued ?recall_notice)
        (recall_confirmed_effective ?recall_notice)

          (lab_batch_locked ?lab_batch)

        (lab_batch_locked ?lab_batch)

          (lab_batch_available ?lab_batch)
        )
      )
  )
  (:action mark_analysis_complete_for_team
    :parameters (?crisis_management_team - crisis_management_team ?lab_batch - lab_batch ?recall_notice - recall_notice ?sample - sample)
    :precondition
      (and
        (field_alert_confirmed ?crisis_management_team)
        (team_has_lab_batch_custody ?crisis_management_team ?lab_batch)
        (lab_batch_locked ?lab_batch)
        (lab_batch_linked_to_recall ?lab_batch ?recall_notice)

        (not
          (recall_has_primary_evidence ?recall_notice)
        )
        (not
          (team_analysis_complete ?crisis_management_team)
        )
      )
    :effect (team_analysis_complete ?crisis_management_team)
  )
  (:action assign_regulatory_document_to_team
    :parameters (?crisis_management_team - crisis_management_team ?regulatory_document - regulatory_document)
    :precondition
      (and
        (field_alert_confirmed ?crisis_management_team)
        (regulatory_document_available ?regulatory_document)
        (not
          (regulatory_document_locked ?crisis_management_team)
        )
      )
    :effect
      (and
        (regulatory_document_locked ?crisis_management_team)
        (regulatory_document_assigned_to_team ?crisis_management_team ?regulatory_document)
        (not
