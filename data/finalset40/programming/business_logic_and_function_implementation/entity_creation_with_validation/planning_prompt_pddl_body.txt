## Planning domain (PDDL)

(define (domain entity_creation_workflow)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource - object asset_group - object assembly_group - object creation_subject - object creation_target - creation_subject identifier_token - resource input_field - resource assignee - resource template_variant - resource feature_option - resource audit_token - resource extension_a - resource extension_c - resource attachable_asset - asset_group module_component - asset_group extension_b_group - asset_group creator_capability_slot - assembly_group reviewer_capability_slot - assembly_group composite_package - assembly_group participant_role - creation_target entity_group - creation_target creator - participant_role reviewer - participant_role entity_instance - entity_group)
  (:predicates
    (draft_created ?creation_target - creation_target)
    (validated ?creation_target - creation_target)
    (identifier_assigned ?creation_target - creation_target)
    (committed ?creation_target - creation_target)
    (ready_for_commit ?creation_target - creation_target)
    (audit_attached ?creation_target - creation_target)
    (identifier_token_available ?identifier_token - identifier_token)
    (has_identifier_token ?creation_target - creation_target ?identifier_token - identifier_token)
    (input_field_available ?input_field - input_field)
    (bound_input_field ?creation_target - creation_target ?input_field - input_field)
    (assignee_available ?assignee - assignee)
    (assigned_to ?creation_target - creation_target ?assignee - assignee)
    (attachable_asset_available ?attachable_asset - attachable_asset)
    (creator_has_asset ?creator - creator ?attachable_asset - attachable_asset)
    (reviewer_has_asset ?reviewer - reviewer ?attachable_asset - attachable_asset)
    (creator_has_capability_slot ?creator - creator ?creator_capability_slot - creator_capability_slot)
    (creator_slot_selected ?creator_capability_slot - creator_capability_slot)
    (creator_slot_enriched ?creator_capability_slot - creator_capability_slot)
    (creator_confirmed ?creator - creator)
    (reviewer_has_capability_slot ?reviewer - reviewer ?reviewer_capability_slot - reviewer_capability_slot)
    (reviewer_slot_selected ?reviewer_capability_slot - reviewer_capability_slot)
    (reviewer_slot_enriched ?reviewer_capability_slot - reviewer_capability_slot)
    (reviewer_confirmed ?reviewer - reviewer)
    (package_template_available ?composite_package - composite_package)
    (package_prepared ?composite_package - composite_package)
    (package_bound_creator_slot ?composite_package - composite_package ?creator_capability_slot - creator_capability_slot)
    (package_bound_reviewer_slot ?composite_package - composite_package ?reviewer_capability_slot - reviewer_capability_slot)
    (package_variant_a ?composite_package - composite_package)
    (package_variant_b ?composite_package - composite_package)
    (package_finalized ?composite_package - composite_package)
    (instance_linked_creator ?entity_instance - entity_instance ?creator - creator)
    (instance_linked_reviewer ?entity_instance - entity_instance ?reviewer - reviewer)
    (instance_has_package ?entity_instance - entity_instance ?composite_package - composite_package)
    (module_component_available ?module_component - module_component)
    (instance_has_component ?entity_instance - entity_instance ?module_component - module_component)
    (module_component_reserved ?module_component - module_component)
    (component_bound_to_package ?module_component - module_component ?composite_package - composite_package)
    (instance_component_provision_in_progress ?entity_instance - entity_instance)
    (instance_component_provision_ready ?entity_instance - entity_instance)
    (instance_component_enrichment_ready ?entity_instance - entity_instance)
    (instance_template_variant_selected ?entity_instance - entity_instance)
    (instance_template_variant_applied ?entity_instance - entity_instance)
    (instance_component_enriched_flag ?entity_instance - entity_instance)
    (instance_component_provision_finalized ?entity_instance - entity_instance)
    (extension_b_available ?extension_b - extension_b_group)
    (instance_has_extension_b ?entity_instance - entity_instance ?extension_b - extension_b_group)
    (instance_extension_b_applied ?entity_instance - entity_instance)
    (instance_extension_b_enrichment ?entity_instance - entity_instance)
    (instance_extension_b_finalized ?entity_instance - entity_instance)
    (template_variant_available ?template_variant - template_variant)
    (instance_has_template_variant ?entity_instance - entity_instance ?template_variant - template_variant)
    (feature_option_available ?feature_option - feature_option)
    (instance_has_feature_option ?entity_instance - entity_instance ?feature_option - feature_option)
    (extension_a_available ?extension_a - extension_a)
    (instance_has_extension_a ?entity_instance - entity_instance ?extension_a - extension_a)
    (extension_c_available ?extension_c - extension_c)
    (instance_has_extension_c ?entity_instance - entity_instance ?extension_c - extension_c)
    (audit_token_available ?audit_token - audit_token)
    (bound_audit_token ?creation_target - creation_target ?audit_token - audit_token)
    (creator_ready ?creator - creator)
    (reviewer_ready ?reviewer - reviewer)
    (instance_commit_locked ?entity_instance - entity_instance)
  )
  (:action create_draft
    :parameters (?creation_target - creation_target)
    :precondition
      (and
        (not
          (draft_created ?creation_target)
        )
        (not
          (committed ?creation_target)
        )
      )
    :effect (draft_created ?creation_target)
  )
  (:action reserve_identifier
    :parameters (?creation_target - creation_target ?identifier_token - identifier_token)
    :precondition
      (and
        (draft_created ?creation_target)
        (not
          (identifier_assigned ?creation_target)
        )
        (identifier_token_available ?identifier_token)
      )
    :effect
      (and
        (identifier_assigned ?creation_target)
        (has_identifier_token ?creation_target ?identifier_token)
        (not
          (identifier_token_available ?identifier_token)
        )
      )
  )
  (:action bind_input_field
    :parameters (?creation_target - creation_target ?input_field - input_field)
    :precondition
      (and
        (draft_created ?creation_target)
        (identifier_assigned ?creation_target)
        (input_field_available ?input_field)
      )
    :effect
      (and
        (bound_input_field ?creation_target ?input_field)
        (not
          (input_field_available ?input_field)
        )
      )
  )
  (:action validate_inputs
    :parameters (?creation_target - creation_target ?input_field - input_field)
    :precondition
      (and
        (draft_created ?creation_target)
        (identifier_assigned ?creation_target)
        (bound_input_field ?creation_target ?input_field)
        (not
          (validated ?creation_target)
        )
      )
    :effect (validated ?creation_target)
  )
  (:action unbind_input_field
    :parameters (?creation_target - creation_target ?input_field - input_field)
    :precondition
      (and
        (bound_input_field ?creation_target ?input_field)
      )
    :effect
      (and
        (input_field_available ?input_field)
        (not
          (bound_input_field ?creation_target ?input_field)
        )
      )
  )
  (:action assign_assignee
    :parameters (?creation_target - creation_target ?assignee - assignee)
    :precondition
      (and
        (validated ?creation_target)
        (assignee_available ?assignee)
      )
    :effect
      (and
        (assigned_to ?creation_target ?assignee)
        (not
          (assignee_available ?assignee)
        )
      )
  )
  (:action unassign_assignee
    :parameters (?creation_target - creation_target ?assignee - assignee)
    :precondition
      (and
        (assigned_to ?creation_target ?assignee)
      )
    :effect
      (and
        (assignee_available ?assignee)
        (not
          (assigned_to ?creation_target ?assignee)
        )
      )
  )
  (:action attach_extension_a
    :parameters (?entity_instance - entity_instance ?extension_a - extension_a)
    :precondition
      (and
        (validated ?entity_instance)
        (extension_a_available ?extension_a)
      )
    :effect
      (and
        (instance_has_extension_a ?entity_instance ?extension_a)
        (not
          (extension_a_available ?extension_a)
        )
      )
  )
  (:action detach_extension_a
    :parameters (?entity_instance - entity_instance ?extension_a - extension_a)
    :precondition
      (and
        (instance_has_extension_a ?entity_instance ?extension_a)
      )
    :effect
      (and
        (extension_a_available ?extension_a)
        (not
          (instance_has_extension_a ?entity_instance ?extension_a)
        )
      )
  )
  (:action attach_extension_c
    :parameters (?entity_instance - entity_instance ?extension_c - extension_c)
    :precondition
      (and
        (validated ?entity_instance)
        (extension_c_available ?extension_c)
      )
    :effect
      (and
        (instance_has_extension_c ?entity_instance ?extension_c)
        (not
          (extension_c_available ?extension_c)
        )
      )
  )
  (:action detach_extension_c
    :parameters (?entity_instance - entity_instance ?extension_c - extension_c)
    :precondition
      (and
        (instance_has_extension_c ?entity_instance ?extension_c)
      )
    :effect
      (and
        (extension_c_available ?extension_c)
        (not
          (instance_has_extension_c ?entity_instance ?extension_c)
        )
      )
  )
  (:action claim_creator_slot
    :parameters (?creator - creator ?creator_capability_slot - creator_capability_slot ?input_field - input_field)
    :precondition
      (and
        (validated ?creator)
        (bound_input_field ?creator ?input_field)
        (creator_has_capability_slot ?creator ?creator_capability_slot)
        (not
          (creator_slot_selected ?creator_capability_slot)
        )
        (not
          (creator_slot_enriched ?creator_capability_slot)
        )
      )
    :effect (creator_slot_selected ?creator_capability_slot)
  )
  (:action confirm_creator_slot
    :parameters (?creator - creator ?creator_capability_slot - creator_capability_slot ?assignee - assignee)
    :precondition
      (and
        (validated ?creator)
        (assigned_to ?creator ?assignee)
        (creator_has_capability_slot ?creator ?creator_capability_slot)
        (creator_slot_selected ?creator_capability_slot)
        (not
          (creator_ready ?creator)
        )
      )
    :effect
      (and
        (creator_ready ?creator)
        (creator_confirmed ?creator)
      )
  )
  (:action attach_asset_to_creator_slot
    :parameters (?creator - creator ?creator_capability_slot - creator_capability_slot ?attachable_asset - attachable_asset)
    :precondition
      (and
        (validated ?creator)
        (creator_has_capability_slot ?creator ?creator_capability_slot)
        (attachable_asset_available ?attachable_asset)
        (not
          (creator_ready ?creator)
        )
      )
    :effect
      (and
        (creator_slot_enriched ?creator_capability_slot)
        (creator_ready ?creator)
        (creator_has_asset ?creator ?attachable_asset)
        (not
          (attachable_asset_available ?attachable_asset)
        )
      )
  )
  (:action finalize_creator_slot
    :parameters (?creator - creator ?creator_capability_slot - creator_capability_slot ?input_field - input_field ?attachable_asset - attachable_asset)
    :precondition
      (and
        (validated ?creator)
        (bound_input_field ?creator ?input_field)
        (creator_has_capability_slot ?creator ?creator_capability_slot)
        (creator_slot_enriched ?creator_capability_slot)
        (creator_has_asset ?creator ?attachable_asset)
        (not
          (creator_confirmed ?creator)
        )
      )
    :effect
      (and
        (creator_slot_selected ?creator_capability_slot)
        (creator_confirmed ?creator)
        (attachable_asset_available ?attachable_asset)
        (not
          (creator_has_asset ?creator ?attachable_asset)
        )
      )
  )
  (:action claim_reviewer_slot
    :parameters (?reviewer - reviewer ?reviewer_capability_slot - reviewer_capability_slot ?input_field - input_field)
    :precondition
      (and
        (validated ?reviewer)
        (bound_input_field ?reviewer ?input_field)
        (reviewer_has_capability_slot ?reviewer ?reviewer_capability_slot)
        (not
          (reviewer_slot_selected ?reviewer_capability_slot)
        )
        (not
          (reviewer_slot_enriched ?reviewer_capability_slot)
        )
      )
    :effect (reviewer_slot_selected ?reviewer_capability_slot)
  )
  (:action confirm_reviewer_slot
    :parameters (?reviewer - reviewer ?reviewer_capability_slot - reviewer_capability_slot ?assignee - assignee)
    :precondition
      (and
        (validated ?reviewer)
        (assigned_to ?reviewer ?assignee)
        (reviewer_has_capability_slot ?reviewer ?reviewer_capability_slot)
        (reviewer_slot_selected ?reviewer_capability_slot)
        (not
          (reviewer_ready ?reviewer)
        )
      )
    :effect
      (and
        (reviewer_ready ?reviewer)
        (reviewer_confirmed ?reviewer)
      )
  )
  (:action attach_asset_to_reviewer_slot
    :parameters (?reviewer - reviewer ?reviewer_capability_slot - reviewer_capability_slot ?attachable_asset - attachable_asset)
    :precondition
      (and
        (validated ?reviewer)
        (reviewer_has_capability_slot ?reviewer ?reviewer_capability_slot)
        (attachable_asset_available ?attachable_asset)
        (not
          (reviewer_ready ?reviewer)
        )
      )
    :effect
      (and
        (reviewer_slot_enriched ?reviewer_capability_slot)
        (reviewer_ready ?reviewer)
        (reviewer_has_asset ?reviewer ?attachable_asset)
        (not
          (attachable_asset_available ?attachable_asset)
        )
      )
  )
  (:action finalize_reviewer_slot
    :parameters (?reviewer - reviewer ?reviewer_capability_slot - reviewer_capability_slot ?input_field - input_field ?attachable_asset - attachable_asset)
    :precondition
      (and
        (validated ?reviewer)
        (bound_input_field ?reviewer ?input_field)
        (reviewer_has_capability_slot ?reviewer ?reviewer_capability_slot)
        (reviewer_slot_enriched ?reviewer_capability_slot)
        (reviewer_has_asset ?reviewer ?attachable_asset)
        (not
          (reviewer_confirmed ?reviewer)
        )
      )
    :effect
      (and
        (reviewer_slot_selected ?reviewer_capability_slot)
        (reviewer_confirmed ?reviewer)
        (attachable_asset_available ?attachable_asset)
        (not
          (reviewer_has_asset ?reviewer ?attachable_asset)
        )
      )
  )
  (:action assemble_package
    :parameters (?creator - creator ?reviewer - reviewer ?creator_capability_slot - creator_capability_slot ?reviewer_capability_slot - reviewer_capability_slot ?composite_package - composite_package)
    :precondition
      (and
        (creator_ready ?creator)
        (reviewer_ready ?reviewer)
        (creator_has_capability_slot ?creator ?creator_capability_slot)
        (reviewer_has_capability_slot ?reviewer ?reviewer_capability_slot)
        (creator_slot_selected ?creator_capability_slot)
        (reviewer_slot_selected ?reviewer_capability_slot)
        (creator_confirmed ?creator)
        (reviewer_confirmed ?reviewer)
        (package_template_available ?composite_package)
      )
    :effect
      (and
        (package_prepared ?composite_package)
        (package_bound_creator_slot ?composite_package ?creator_capability_slot)
        (package_bound_reviewer_slot ?composite_package ?reviewer_capability_slot)
        (not
          (package_template_available ?composite_package)
        )
      )
  )
  (:action assemble_package_variant_a
    :parameters (?creator - creator ?reviewer - reviewer ?creator_capability_slot - creator_capability_slot ?reviewer_capability_slot - reviewer_capability_slot ?composite_package - composite_package)
    :precondition
      (and
        (creator_ready ?creator)
        (reviewer_ready ?reviewer)
        (creator_has_capability_slot ?creator ?creator_capability_slot)
        (reviewer_has_capability_slot ?reviewer ?reviewer_capability_slot)
        (creator_slot_enriched ?creator_capability_slot)
        (reviewer_slot_selected ?reviewer_capability_slot)
        (not
          (creator_confirmed ?creator)
        )
        (reviewer_confirmed ?reviewer)
        (package_template_available ?composite_package)
      )
    :effect
      (and
        (package_prepared ?composite_package)
        (package_bound_creator_slot ?composite_package ?creator_capability_slot)
        (package_bound_reviewer_slot ?composite_package ?reviewer_capability_slot)
        (package_variant_a ?composite_package)
        (not
          (package_template_available ?composite_package)
        )
      )
  )
  (:action assemble_package_variant_b
    :parameters (?creator - creator ?reviewer - reviewer ?creator_capability_slot - creator_capability_slot ?reviewer_capability_slot - reviewer_capability_slot ?composite_package - composite_package)
    :precondition
      (and
        (creator_ready ?creator)
        (reviewer_ready ?reviewer)
        (creator_has_capability_slot ?creator ?creator_capability_slot)
        (reviewer_has_capability_slot ?reviewer ?reviewer_capability_slot)
        (creator_slot_selected ?creator_capability_slot)
        (reviewer_slot_enriched ?reviewer_capability_slot)
        (creator_confirmed ?creator)
        (not
          (reviewer_confirmed ?reviewer)
        )
        (package_template_available ?composite_package)
      )
    :effect
      (and
        (package_prepared ?composite_package)
        (package_bound_creator_slot ?composite_package ?creator_capability_slot)
        (package_bound_reviewer_slot ?composite_package ?reviewer_capability_slot)
        (package_variant_b ?composite_package)
        (not
          (package_template_available ?composite_package)
        )
      )
  )
  (:action assemble_package_all_variants
    :parameters (?creator - creator ?reviewer - reviewer ?creator_capability_slot - creator_capability_slot ?reviewer_capability_slot - reviewer_capability_slot ?composite_package - composite_package)
    :precondition
      (and
        (creator_ready ?creator)
        (reviewer_ready ?reviewer)
        (creator_has_capability_slot ?creator ?creator_capability_slot)
        (reviewer_has_capability_slot ?reviewer ?reviewer_capability_slot)
        (creator_slot_enriched ?creator_capability_slot)
        (reviewer_slot_enriched ?reviewer_capability_slot)
        (not
          (creator_confirmed ?creator)
        )
        (not
          (reviewer_confirmed ?reviewer)
        )
        (package_template_available ?composite_package)
      )
    :effect
      (and
        (package_prepared ?composite_package)
        (package_bound_creator_slot ?composite_package ?creator_capability_slot)
        (package_bound_reviewer_slot ?composite_package ?reviewer_capability_slot)
        (package_variant_a ?composite_package)
        (package_variant_b ?composite_package)
        (not
          (package_template_available ?composite_package)
        )
      )
  )
  (:action finalize_package
    :parameters (?composite_package - composite_package ?creator - creator ?input_field - input_field)
    :precondition
      (and
        (package_prepared ?composite_package)
        (creator_ready ?creator)
        (bound_input_field ?creator ?input_field)
        (not
          (package_finalized ?composite_package)
        )
      )
    :effect (package_finalized ?composite_package)
  )
  (:action reserve_module_component
    :parameters (?entity_instance - entity_instance ?module_component - module_component ?composite_package - composite_package)
    :precondition
      (and
        (validated ?entity_instance)
        (instance_has_package ?entity_instance ?composite_package)
        (instance_has_component ?entity_instance ?module_component)
        (module_component_available ?module_component)
        (package_prepared ?composite_package)
        (package_finalized ?composite_package)
        (not
          (module_component_reserved ?module_component)
        )
      )
    :effect
      (and
        (module_component_reserved ?module_component)
        (component_bound_to_package ?module_component ?composite_package)
        (not
          (module_component_available ?module_component)
        )
      )
  )
  (:action prepare_component_for_enrichment
    :parameters (?entity_instance - entity_instance ?module_component - module_component ?composite_package - composite_package ?input_field - input_field)
    :precondition
      (and
        (validated ?entity_instance)
        (instance_has_component ?entity_instance ?module_component)
        (module_component_reserved ?module_component)
        (component_bound_to_package ?module_component ?composite_package)
        (bound_input_field ?entity_instance ?input_field)
        (not
          (package_variant_a ?composite_package)
        )
        (not
          (instance_component_provision_in_progress ?entity_instance)
        )
      )
    :effect (instance_component_provision_in_progress ?entity_instance)
  )
  (:action apply_template_variant
    :parameters (?entity_instance - entity_instance ?template_variant - template_variant)
    :precondition
      (and
        (validated ?entity_instance)
        (template_variant_available ?template_variant)
        (not
          (instance_template_variant_selected ?entity_instance)
        )
      )
    :effect
      (and
        (instance_template_variant_selected ?entity_instance)
        (instance_has_template_variant ?entity_instance ?template_variant)
        (not
          (template_variant_available ?template_variant)
        )
      )
  )
  (:action prepare_component_with_variant
    :parameters (?entity_instance - entity_instance ?module_component - module_component ?composite_package - composite_package ?input_field - input_field ?template_variant - template_variant)
    :precondition
      (and
        (validated ?entity_instance)
        (instance_has_component ?entity_instance ?module_component)
        (module_component_reserved ?module_component)
        (component_bound_to_package ?module_component ?composite_package)
        (bound_input_field ?entity_instance ?input_field)
        (package_variant_a ?composite_package)
        (instance_template_variant_selected ?entity_instance)
        (instance_has_template_variant ?entity_instance ?template_variant)
        (not
          (instance_component_provision_in_progress ?entity_instance)
        )
      )
    :effect
      (and
        (instance_component_provision_in_progress ?entity_instance)
        (instance_template_variant_applied ?entity_instance)
      )
  )
  (:action provision_component_primary
    :parameters (?entity_instance - entity_instance ?extension_a - extension_a ?assignee - assignee ?module_component - module_component ?composite_package - composite_package)
    :precondition
      (and
        (instance_component_provision_in_progress ?entity_instance)
        (instance_has_extension_a ?entity_instance ?extension_a)
        (assigned_to ?entity_instance ?assignee)
        (instance_has_component ?entity_instance ?module_component)
        (component_bound_to_package ?module_component ?composite_package)
        (not
          (package_variant_b ?composite_package)
        )
        (not
          (instance_component_provision_ready ?entity_instance)
        )
      )
    :effect (instance_component_provision_ready ?entity_instance)
  )
  (:action provision_component_secondary
    :parameters (?entity_instance - entity_instance ?extension_a - extension_a ?assignee - assignee ?module_component - module_component ?composite_package - composite_package)
    :precondition
      (and
        (instance_component_provision_in_progress ?entity_instance)
        (instance_has_extension_a ?entity_instance ?extension_a)
        (assigned_to ?entity_instance ?assignee)
        (instance_has_component ?entity_instance ?module_component)
        (component_bound_to_package ?module_component ?composite_package)
        (package_variant_b ?composite_package)
        (not
          (instance_component_provision_ready ?entity_instance)
        )
      )
    :effect (instance_component_provision_ready ?entity_instance)
  )
  (:action begin_component_enrichment_primary
    :parameters (?entity_instance - entity_instance ?extension_c - extension_c ?module_component - module_component ?composite_package - composite_package)
    :precondition
      (and
        (instance_component_provision_ready ?entity_instance)
        (instance_has_extension_c ?entity_instance ?extension_c)
        (instance_has_component ?entity_instance ?module_component)
        (component_bound_to_package ?module_component ?composite_package)
        (not
          (package_variant_a ?composite_package)
        )
        (not
          (package_variant_b ?composite_package)
        )
        (not
          (instance_component_enrichment_ready ?entity_instance)
        )
      )
    :effect (instance_component_enrichment_ready ?entity_instance)
  )
  (:action begin_component_enrichment_with_variant_a
    :parameters (?entity_instance - entity_instance ?extension_c - extension_c ?module_component - module_component ?composite_package - composite_package)
    :precondition
      (and
        (instance_component_provision_ready ?entity_instance)
        (instance_has_extension_c ?entity_instance ?extension_c)
        (instance_has_component ?entity_instance ?module_component)
        (component_bound_to_package ?module_component ?composite_package)
        (package_variant_a ?composite_package)
        (not
          (package_variant_b ?composite_package)
        )
        (not
          (instance_component_enrichment_ready ?entity_instance)
        )
      )
    :effect
      (and
        (instance_component_enrichment_ready ?entity_instance)
        (instance_component_enriched_flag ?entity_instance)
      )
  )
  (:action begin_component_enrichment_with_variant_b

        (instance_has_component ?entity_instance ?module_component)
        (component_bound_to_package ?module_component ?composite_package)
        (package_variant_a ?composite_package)

        (not
          (instance_component_enrichment_ready ?entity_instance)

    :effect
      (and
        (instance_component_enrichment_ready ?entity_instance)
        (instance_component_enriched_flag ?entity_instance)

    :parameters (?entity_instance - entity_instance)

      (and

        (not
          (instance_component_enriched_flag ?entity_instance)
        )
        (not
          (instance_commit_locked ?entity_instance)

    :effect
      (and
        (instance_commit_locked ?entity_instance)
        (ready_for_commit ?entity_instance)

    :parameters (?entity_instance - entity_instance ?feature_option - feature_option)

        (instance_component_enriched_flag ?entity_instance)

      (and
        (instance_has_feature_option ?entity_instance ?feature_option)
        (not
          (feature_option_available ?feature_option)

      )

  (:action finalize_enrichment
    :parameters (?entity_instance - entity_instance ?creator - creator ?reviewer - reviewer ?input_field - input_field ?feature_option - feature_option)

        (instance_component_enriched_flag ?entity_instance)
        (instance_has_feature_option ?entity_instance ?feature_option)
        (instance_linked_creator ?entity_instance ?creator)
        (instance_linked_reviewer ?entity_instance ?reviewer)
        (creator_confirmed ?creator)
        (reviewer_confirmed ?reviewer)
        (bound_input_field ?entity_instance ?input_field)
        (not
          (instance_component_provision_finalized ?entity_instance)
        )
      )
    :effect (instance_component_provision_finalized ?entity_instance)

        (instance_component_enrichment_ready ?entity_instance)
        (instance_component_provision_finalized ?entity_instance)
        (not
          (instance_commit_locked ?entity_instance)
        )

    :effect
      (and
        (instance_commit_locked ?entity_instance)
        (ready_for_commit ?entity_instance)

        (instance_has_extension_b ?entity_instance ?extension_b)
        (not
          (instance_extension_b_applied ?entity_instance)

      (and
        (instance_extension_b_applied ?entity_instance)

          (extension_b_available ?extension_b)

    :parameters (?entity_instance - entity_instance ?assignee - assignee)
    :precondition
      (and
        (instance_extension_b_applied ?entity_instance)
        (assigned_to ?entity_instance ?assignee)
        (not
          (instance_extension_b_enrichment ?entity_instance)
        )
      )
    :effect (instance_extension_b_enrichment ?entity_instance)
  )
  (:action apply_extension_c_after_b
    :parameters (?entity_instance - entity_instance ?extension_c - extension_c)

        (not
          (instance_extension_b_finalized ?entity_instance)
        )

  )
  (:action finalize_extension_b_and_mark_ready
    :parameters (?entity_instance - entity_instance)
    :precondition
      (and
        (instance_extension_b_finalized ?entity_instance)
        (not
          (instance_commit_locked ?entity_instance)
        )
      )
    :effect
      (and
        (instance_commit_locked ?entity_instance)
        (ready_for_commit ?entity_instance)
      )

    :parameters (?creator - creator ?composite_package - composite_package)
    :precondition
      (and
        (creator_ready ?creator)
        (creator_confirmed ?creator)
        (package_prepared ?composite_package)
        (package_finalized ?composite_package)
        (not
          (ready_for_commit ?creator)
        )
      )
    :effect (ready_for_commit ?creator)

  (:action confirm_reviewer_ready_for_commit
    :parameters (?reviewer - reviewer ?composite_package - composite_package)
    :precondition
      (and
        (reviewer_ready ?reviewer)
        (reviewer_confirmed ?reviewer)
        (package_prepared ?composite_package)
        (package_finalized ?composite_package)
        (not
          (ready_for_commit ?reviewer)
        )
      )
    :effect (ready_for_commit ?reviewer)
  )
  (:action attach_audit_token

        (not
          (audit_attached ?creation_target)

      (and
        (audit_attached ?creation_target)
        (bound_audit_token ?creation_target ?audit_token)
        (not
          (audit_token_available ?audit_token)
        )
      )
  )
  (:action commit_creator
    :parameters (?creator - creator ?identifier_token - identifier_token ?audit_token - audit_token)
    :precondition
      (and
        (audit_attached ?creator)
        (has_identifier_token ?creator ?identifier_token)
        (bound_audit_token ?creator ?audit_token)
        (not
          (committed ?creator)
        )
      )
    :effect
      (and
        (committed ?creator)
        (identifier_token_available ?identifier_token)
        (audit_token_available ?audit_token)
      )
  )
  (:action commit_reviewer
    :parameters (?reviewer - reviewer ?identifier_token - identifier_token ?audit_token - audit_token)
    :precondition
      (and
        (audit_attached ?reviewer)
        (has_identifier_token ?reviewer ?identifier_token)
        (bound_audit_token ?reviewer ?audit_token)
        (not
          (committed ?reviewer)
        )
      )
    :effect
      (and
        (committed ?reviewer)
        (identifier_token_available ?identifier_token)
        (audit_token_available ?audit_token)
      )
  )
  (:action commit_entity_instance
    :parameters (?entity_instance - entity_instance ?identifier_token - identifier_token ?audit_token - audit_token)
    :precondition
      (and
        (audit_attached ?entity_instance)
        (has_identifier_token ?entity_instance ?identifier_token)
        (bound_audit_token ?entity_instance ?audit_token)
        (not
          (committed ?entity_instance)
        )
      )
    :effect
      (and
        (committed ?entity_instance)
        (identifier_token_available ?identifier_token)
        (audit_token_available ?audit_token)
      )
  )
