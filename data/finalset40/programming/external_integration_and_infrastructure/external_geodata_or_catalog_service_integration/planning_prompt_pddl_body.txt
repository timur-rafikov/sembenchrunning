    :parameters (?integration_context - integration_context)
    :precondition
      (and

  )

    :parameters (?integration_context - integration_context ?external_endpoint - external_endpoint)

        (context_endpoint_bound ?integration_context)
        (context_endpoint_binding ?integration_context ?external_endpoint)

          (endpoint_available ?external_endpoint)

      )
