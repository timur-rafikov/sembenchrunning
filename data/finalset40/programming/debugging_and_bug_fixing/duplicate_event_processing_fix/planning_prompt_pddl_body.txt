  (:action register_processing_entity
    :parameters (?event - processing_entity)
    :precondition
      (and
        (not
          (processing_entity_registered ?event)
        )
        (not
          (committed ?event)
        )
      )
    :effect (processing_entity_registered ?event)
  )
  (:action assign_handler_slot_to_processing_entity
    :parameters (?event - processing_entity ?handler_slot - handler_slot)
    :precondition
      (and
        (processing_entity_registered ?event)
        (not
          (handler_claimed ?event)
        )
        (handler_slot_available ?handler_slot)
      )
    :effect
      (and
        (handler_claimed ?event)
        (handler_assigned ?event ?handler_slot)
        (not
          (handler_slot_available ?handler_slot)
        )
      )
  )
  (:action bind_payload_to_processing_entity
    :parameters (?event - processing_entity ?payload_descriptor - payload_descriptor)
    :precondition
      (and
        (processing_entity_registered ?event)
        (handler_claimed ?event)
        (payload_descriptor_available ?payload_descriptor)
      )
    :effect
      (and
        (payload_attached ?event ?payload_descriptor)
        (not
          (payload_descriptor_available ?payload_descriptor)
        )
      )
  )
  (:action mark_processing_entity_ready_for_processing
    :parameters (?event - processing_entity ?payload_descriptor - payload_descriptor)
    :precondition
      (and
        (processing_entity_registered ?event)
        (handler_claimed ?event)
        (payload_attached ?event ?payload_descriptor)
        (not
          (ready_for_processing ?event)
        )
      )
    :effect (ready_for_processing ?event)
  )
  (:action release_payload_descriptor
    :parameters (?event - processing_entity ?payload_descriptor - payload_descriptor)
    :precondition
      (and
        (payload_attached ?event ?payload_descriptor)
      )
    :effect
      (and
        (payload_descriptor_available ?payload_descriptor)
        (not
          (payload_attached ?event ?payload_descriptor)
        )
      )
  )
  (:action attach_processing_resource_to_processing_entity
    :parameters (?event - processing_entity ?processing_resource - processing_resource)
    :precondition
      (and
        (ready_for_processing ?event)
        (processing_resource_available ?processing_resource)
      )
    :effect
      (and
        (resource_attached ?event ?processing_resource)
        (not
          (processing_resource_available ?processing_resource)
        )
      )
  )
  (:action detach_processing_resource_from_processing_entity
    :parameters (?event - processing_entity ?processing_resource - processing_resource)

        (resource_attached ?event ?processing_resource)
      )
    :effect

        (processing_resource_available ?processing_resource)
        (not
          (resource_attached ?event ?processing_resource)
        )
      )
  )
