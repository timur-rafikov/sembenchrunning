## Planning domain (PDDL)

(define (domain destination_selection_planning_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types trip_resource_type - object itinerary_enrichment_type - object access_group_type - object planning_entity_type - object planning_option_type - planning_entity_type transport_option_type - trip_resource_type activity_option_type - trip_resource_type accommodation_option_type - trip_resource_type itinerary_theme_type - trip_resource_type traveler_profile_type - trip_resource_type trip_constraint_type - trip_resource_type service_addon_type - trip_resource_type time_window_type - trip_resource_type auxiliary_addon_type - itinerary_enrichment_type activity_bundle_type - itinerary_enrichment_type promotion_option_type - itinerary_enrichment_type access_option_type - access_group_type local_access_option_type - access_group_type destination_proposal_type - access_group_type day_segment_base_type - planning_option_type itinerary_variant_base_type - planning_option_type primary_day_segment_type - day_segment_base_type secondary_day_segment_type - day_segment_base_type itinerary_variant_type - itinerary_variant_base_type)
  (:predicates
    (planning_entity_drafted_state ?planning_option_var - planning_option_type)
    (planning_entity_ready_state ?planning_option_var - planning_option_type)
    (planning_entity_transport_assigned_state ?planning_option_var - planning_option_type)
    (planning_entity_finalized_state ?planning_option_var - planning_option_type)
    (planning_entity_chosen_state ?planning_option_var - planning_option_type)
    (planning_entity_constraint_applied_state ?planning_option_var - planning_option_type)
    (transport_available_state ?transport_option_var - transport_option_type)
    (planning_entity_transport_link ?planning_option_var - planning_option_type ?transport_option_var - transport_option_type)
    (activity_available_state ?activity_option_var - activity_option_type)
    (planning_entity_activity_link ?planning_option_var - planning_option_type ?activity_option_var - activity_option_type)
    (accommodation_available_state ?accommodation_option_var - accommodation_option_type)
    (planning_entity_accommodation_link ?planning_option_var - planning_option_type ?accommodation_option_var - accommodation_option_type)
    (addon_available_state ?auxiliary_addon_var - auxiliary_addon_type)
    (primary_addon_link ?primary_day_segment_var - primary_day_segment_type ?auxiliary_addon_var - auxiliary_addon_type)
    (secondary_addon_link ?secondary_day_segment_var - secondary_day_segment_type ?auxiliary_addon_var - auxiliary_addon_type)
    (primary_access_link ?primary_day_segment_var - primary_day_segment_type ?access_option_var - access_option_type)
    (preferred_access_state ?access_option_var - access_option_type)
    (alternate_access_state ?access_option_var - access_option_type)
    (primary_segment_ready_state ?primary_day_segment_var - primary_day_segment_type)
    (secondary_local_access_link ?secondary_day_segment_var - secondary_day_segment_type ?local_access_option_var - local_access_option_type)
    (preferred_local_access_state ?local_access_option_var - local_access_option_type)
    (alternate_local_access_state ?local_access_option_var - local_access_option_type)
    (secondary_segment_ready_state ?secondary_day_segment_var - secondary_day_segment_type)
    (proposal_draft_state ?destination_proposal_var - destination_proposal_type)
    (proposal_assembled_state ?destination_proposal_var - destination_proposal_type)
    (proposal_access_link ?destination_proposal_var - destination_proposal_type ?access_option_var - access_option_type)
    (proposal_local_access_link ?destination_proposal_var - destination_proposal_type ?local_access_option_var - local_access_option_type)
    (primary_gap_state ?destination_proposal_var - destination_proposal_type)
    (secondary_gap_state ?destination_proposal_var - destination_proposal_type)
    (proposal_validated_state ?destination_proposal_var - destination_proposal_type)
    (variant_primary_segment_link ?itinerary_variant_var - itinerary_variant_type ?primary_day_segment_var - primary_day_segment_type)
    (variant_secondary_segment_link ?itinerary_variant_var - itinerary_variant_type ?secondary_day_segment_var - secondary_day_segment_type)
    (variant_proposal_link ?itinerary_variant_var - itinerary_variant_type ?destination_proposal_var - destination_proposal_type)
    (activity_bundle_available_state ?activity_bundle_var - activity_bundle_type)
    (variant_bundle_link ?itinerary_variant_var - itinerary_variant_type ?activity_bundle_var - activity_bundle_type)
    (bundle_committed_state ?activity_bundle_var - activity_bundle_type)
    (bundle_proposal_link ?activity_bundle_var - activity_bundle_type ?destination_proposal_var - destination_proposal_type)
    (variant_structured_state ?itinerary_variant_var - itinerary_variant_type)
    (variant_enriched_state ?itinerary_variant_var - itinerary_variant_type)
    (variant_scheduled_state ?itinerary_variant_var - itinerary_variant_type)
    (theme_applied_state ?itinerary_variant_var - itinerary_variant_type)
    (theme_refined_state ?itinerary_variant_var - itinerary_variant_type)
    (variant_profile_aligned_state ?itinerary_variant_var - itinerary_variant_type)
    (sequence_validated_state ?itinerary_variant_var - itinerary_variant_type)
    (promotion_available_state ?promotion_option_var - promotion_option_type)
    (variant_promotion_link ?itinerary_variant_var - itinerary_variant_type ?promotion_option_var - promotion_option_type)
    (promotion_applied_state ?itinerary_variant_var - itinerary_variant_type)
    (accommodation_integrated_state ?itinerary_variant_var - itinerary_variant_type)
    (variant_ready_state ?itinerary_variant_var - itinerary_variant_type)
    (theme_available_state ?itinerary_theme_var - itinerary_theme_type)
    (variant_theme_link ?itinerary_variant_var - itinerary_variant_type ?itinerary_theme_var - itinerary_theme_type)
    (traveler_profile_available_state ?traveler_profile_var - traveler_profile_type)
    (variant_traveler_profile_link ?itinerary_variant_var - itinerary_variant_type ?traveler_profile_var - traveler_profile_type)
    (service_addon_available_state ?service_addon_var - service_addon_type)
    (variant_service_addon_link ?itinerary_variant_var - itinerary_variant_type ?service_addon_var - service_addon_type)
    (time_window_available_state ?time_window_var - time_window_type)
    (variant_time_window_link ?itinerary_variant_var - itinerary_variant_type ?time_window_var - time_window_type)
    (trip_constraint_available_state ?trip_constraint_var - trip_constraint_type)
    (planning_entity_constraint_link ?planning_option_var - planning_option_type ?trip_constraint_var - trip_constraint_type)
    (primary_segment_open_state ?primary_day_segment_var - primary_day_segment_type)
    (secondary_segment_open_state ?secondary_day_segment_var - secondary_day_segment_type)
    (variant_confirmed_state ?itinerary_variant_var - itinerary_variant_type)
  )
  (:action initialize_planning_option
    :parameters (?planning_option_var - planning_option_type)
    :precondition
      (and
        (not
          (planning_entity_drafted_state ?planning_option_var)
        )
        (not
          (planning_entity_finalized_state ?planning_option_var)
        )
      )
    :effect (planning_entity_drafted_state ?planning_option_var)
  )
  (:action assign_transport_option
    :parameters (?planning_option_var - planning_option_type ?transport_option_var - transport_option_type)
    :precondition
      (and
        (planning_entity_drafted_state ?planning_option_var)
        (not
          (planning_entity_transport_assigned_state ?planning_option_var)
        )
        (transport_available_state ?transport_option_var)
      )
    :effect
      (and
        (planning_entity_transport_assigned_state ?planning_option_var)
        (planning_entity_transport_link ?planning_option_var ?transport_option_var)
        (not
          (transport_available_state ?transport_option_var)
        )
      )
  )
  (:action propose_activity_option
    :parameters (?planning_option_var - planning_option_type ?activity_option_var - activity_option_type)
    :precondition
      (and
        (planning_entity_drafted_state ?planning_option_var)
        (planning_entity_transport_assigned_state ?planning_option_var)
        (activity_available_state ?activity_option_var)
      )
    :effect
      (and
        (planning_entity_activity_link ?planning_option_var ?activity_option_var)
        (not
          (activity_available_state ?activity_option_var)
        )
      )
  )
  (:action confirm_activity_option
    :parameters (?planning_option_var - planning_option_type ?activity_option_var - activity_option_type)
    :precondition
      (and
        (planning_entity_drafted_state ?planning_option_var)
        (planning_entity_transport_assigned_state ?planning_option_var)
        (planning_entity_activity_link ?planning_option_var ?activity_option_var)
        (not
          (planning_entity_ready_state ?planning_option_var)
        )
      )
    :effect (planning_entity_ready_state ?planning_option_var)
  )
  (:action withdraw_activity_option
    :parameters (?planning_option_var - planning_option_type ?activity_option_var - activity_option_type)
    :precondition
      (and
        (planning_entity_activity_link ?planning_option_var ?activity_option_var)
      )
    :effect
      (and
        (activity_available_state ?activity_option_var)
        (not
          (planning_entity_activity_link ?planning_option_var ?activity_option_var)
        )
      )
  )
  (:action reserve_accommodation_option
    :parameters (?planning_option_var - planning_option_type ?accommodation_option_var - accommodation_option_type)
    :precondition
      (and
        (planning_entity_ready_state ?planning_option_var)
        (accommodation_available_state ?accommodation_option_var)
      )
    :effect
      (and
        (planning_entity_accommodation_link ?planning_option_var ?accommodation_option_var)
        (not
          (accommodation_available_state ?accommodation_option_var)
        )
      )
  )
  (:action release_accommodation_option
    :parameters (?planning_option_var - planning_option_type ?accommodation_option_var - accommodation_option_type)
    :precondition
      (and
        (planning_entity_accommodation_link ?planning_option_var ?accommodation_option_var)
      )
    :effect
      (and
        (accommodation_available_state ?accommodation_option_var)
        (not
          (planning_entity_accommodation_link ?planning_option_var ?accommodation_option_var)
        )
      )
  )
  (:action assign_service_addon
    :parameters (?itinerary_variant_var - itinerary_variant_type ?service_addon_var - service_addon_type)
    :precondition
      (and
        (planning_entity_ready_state ?itinerary_variant_var)
        (service_addon_available_state ?service_addon_var)
      )
    :effect
      (and
        (variant_service_addon_link ?itinerary_variant_var ?service_addon_var)
        (not
          (service_addon_available_state ?service_addon_var)
        )
      )
  )
  (:action release_service_addon
    :parameters (?itinerary_variant_var - itinerary_variant_type ?service_addon_var - service_addon_type)
    :precondition
      (and
        (variant_service_addon_link ?itinerary_variant_var ?service_addon_var)
      )
    :effect
      (and
        (service_addon_available_state ?service_addon_var)
        (not
          (variant_service_addon_link ?itinerary_variant_var ?service_addon_var)
        )
      )
  )
  (:action assign_time_window
    :parameters (?itinerary_variant_var - itinerary_variant_type ?time_window_var - time_window_type)
    :precondition
      (and
        (planning_entity_ready_state ?itinerary_variant_var)
        (time_window_available_state ?time_window_var)
      )
    :effect
      (and
        (variant_time_window_link ?itinerary_variant_var ?time_window_var)
        (not
          (time_window_available_state ?time_window_var)
        )
      )
  )
  (:action release_time_window
    :parameters (?itinerary_variant_var - itinerary_variant_type ?time_window_var - time_window_type)
    :precondition
      (and
        (variant_time_window_link ?itinerary_variant_var ?time_window_var)
      )
    :effect
      (and
        (time_window_available_state ?time_window_var)
        (not
          (variant_time_window_link ?itinerary_variant_var ?time_window_var)
        )
      )
  )
  (:action set_primary_access
    :parameters (?primary_day_segment_var - primary_day_segment_type ?access_option_var - access_option_type ?activity_option_var - activity_option_type)
    :precondition
      (and
        (planning_entity_ready_state ?primary_day_segment_var)
        (planning_entity_activity_link ?primary_day_segment_var ?activity_option_var)
        (primary_access_link ?primary_day_segment_var ?access_option_var)
        (not
          (preferred_access_state ?access_option_var)
        )
        (not
          (alternate_access_state ?access_option_var)
        )
      )
    :effect (preferred_access_state ?access_option_var)
  )
  (:action prepare_primary_segment
    :parameters (?primary_day_segment_var - primary_day_segment_type ?access_option_var - access_option_type ?accommodation_option_var - accommodation_option_type)
    :precondition
      (and
        (planning_entity_ready_state ?primary_day_segment_var)
        (planning_entity_accommodation_link ?primary_day_segment_var ?accommodation_option_var)
        (primary_access_link ?primary_day_segment_var ?access_option_var)
        (preferred_access_state ?access_option_var)
        (not
          (primary_segment_open_state ?primary_day_segment_var)
        )
      )
    :effect
      (and
        (primary_segment_open_state ?primary_day_segment_var)
        (primary_segment_ready_state ?primary_day_segment_var)
      )
  )
  (:action attach_primary_addon_route
    :parameters (?primary_day_segment_var - primary_day_segment_type ?access_option_var - access_option_type ?auxiliary_addon_var - auxiliary_addon_type)
    :precondition
      (and
        (planning_entity_ready_state ?primary_day_segment_var)
        (primary_access_link ?primary_day_segment_var ?access_option_var)
        (addon_available_state ?auxiliary_addon_var)
        (not
          (primary_segment_open_state ?primary_day_segment_var)
        )
      )
    :effect
      (and
        (alternate_access_state ?access_option_var)
        (primary_segment_open_state ?primary_day_segment_var)
        (primary_addon_link ?primary_day_segment_var ?auxiliary_addon_var)
        (not
          (addon_available_state ?auxiliary_addon_var)
        )
      )
  )
  (:action restore_primary_segment
    :parameters (?primary_day_segment_var - primary_day_segment_type ?access_option_var - access_option_type ?activity_option_var - activity_option_type ?auxiliary_addon_var - auxiliary_addon_type)
    :precondition
      (and
        (planning_entity_ready_state ?primary_day_segment_var)
        (planning_entity_activity_link ?primary_day_segment_var ?activity_option_var)
        (primary_access_link ?primary_day_segment_var ?access_option_var)
        (alternate_access_state ?access_option_var)
        (primary_addon_link ?primary_day_segment_var ?auxiliary_addon_var)
        (not
          (primary_segment_ready_state ?primary_day_segment_var)
        )
      )
    :effect
      (and
        (preferred_access_state ?access_option_var)
        (primary_segment_ready_state ?primary_day_segment_var)
        (addon_available_state ?auxiliary_addon_var)
        (not
          (primary_addon_link ?primary_day_segment_var ?auxiliary_addon_var)
        )
      )
  )
  (:action set_secondary_access
    :parameters (?secondary_day_segment_var - secondary_day_segment_type ?local_access_option_var - local_access_option_type ?activity_option_var - activity_option_type)
    :precondition
      (and
        (planning_entity_ready_state ?secondary_day_segment_var)
        (planning_entity_activity_link ?secondary_day_segment_var ?activity_option_var)
        (secondary_local_access_link ?secondary_day_segment_var ?local_access_option_var)
        (not
          (preferred_local_access_state ?local_access_option_var)
        )
        (not
          (alternate_local_access_state ?local_access_option_var)
        )
      )
    :effect (preferred_local_access_state ?local_access_option_var)
  )
  (:action prepare_secondary_segment
    :parameters (?secondary_day_segment_var - secondary_day_segment_type ?local_access_option_var - local_access_option_type ?accommodation_option_var - accommodation_option_type)
    :precondition
      (and
        (planning_entity_ready_state ?secondary_day_segment_var)
        (planning_entity_accommodation_link ?secondary_day_segment_var ?accommodation_option_var)
        (secondary_local_access_link ?secondary_day_segment_var ?local_access_option_var)
        (preferred_local_access_state ?local_access_option_var)
        (not
          (secondary_segment_open_state ?secondary_day_segment_var)
        )
      )
    :effect
      (and
        (secondary_segment_open_state ?secondary_day_segment_var)
        (secondary_segment_ready_state ?secondary_day_segment_var)
      )
  )
  (:action attach_secondary_addon_route
    :parameters (?secondary_day_segment_var - secondary_day_segment_type ?local_access_option_var - local_access_option_type ?auxiliary_addon_var - auxiliary_addon_type)
    :precondition
      (and
        (planning_entity_ready_state ?secondary_day_segment_var)
        (secondary_local_access_link ?secondary_day_segment_var ?local_access_option_var)
        (addon_available_state ?auxiliary_addon_var)
        (not
          (secondary_segment_open_state ?secondary_day_segment_var)
        )
      )
    :effect
      (and
        (alternate_local_access_state ?local_access_option_var)
        (secondary_segment_open_state ?secondary_day_segment_var)
        (secondary_addon_link ?secondary_day_segment_var ?auxiliary_addon_var)
        (not
          (addon_available_state ?auxiliary_addon_var)
        )
      )
  )
  (:action restore_secondary_segment
    :parameters (?secondary_day_segment_var - secondary_day_segment_type ?local_access_option_var - local_access_option_type ?activity_option_var - activity_option_type ?auxiliary_addon_var - auxiliary_addon_type)
    :precondition
      (and
        (planning_entity_ready_state ?secondary_day_segment_var)
        (planning_entity_activity_link ?secondary_day_segment_var ?activity_option_var)
        (secondary_local_access_link ?secondary_day_segment_var ?local_access_option_var)
        (alternate_local_access_state ?local_access_option_var)
        (secondary_addon_link ?secondary_day_segment_var ?auxiliary_addon_var)
        (not
          (secondary_segment_ready_state ?secondary_day_segment_var)
        )
      )
    :effect
      (and
        (preferred_local_access_state ?local_access_option_var)
        (secondary_segment_ready_state ?secondary_day_segment_var)
        (addon_available_state ?auxiliary_addon_var)
        (not
          (secondary_addon_link ?secondary_day_segment_var ?auxiliary_addon_var)
        )
      )
  )
  (:action assemble_destination_proposal_clean
    :parameters (?primary_day_segment_var - primary_day_segment_type ?secondary_day_segment_var - secondary_day_segment_type ?access_option_var - access_option_type ?local_access_option_var - local_access_option_type ?destination_proposal_var - destination_proposal_type)
    :precondition
      (and
        (primary_segment_open_state ?primary_day_segment_var)
        (secondary_segment_open_state ?secondary_day_segment_var)
        (primary_access_link ?primary_day_segment_var ?access_option_var)
        (secondary_local_access_link ?secondary_day_segment_var ?local_access_option_var)
        (preferred_access_state ?access_option_var)
        (preferred_local_access_state ?local_access_option_var)
        (primary_segment_ready_state ?primary_day_segment_var)
        (secondary_segment_ready_state ?secondary_day_segment_var)
        (proposal_draft_state ?destination_proposal_var)
      )
    :effect
      (and
        (proposal_assembled_state ?destination_proposal_var)
        (proposal_access_link ?destination_proposal_var ?access_option_var)
        (proposal_local_access_link ?destination_proposal_var ?local_access_option_var)
        (not
          (proposal_draft_state ?destination_proposal_var)
        )
      )
  )
  (:action assemble_destination_proposal_primary_gap
    :parameters (?primary_day_segment_var - primary_day_segment_type ?secondary_day_segment_var - secondary_day_segment_type ?access_option_var - access_option_type ?local_access_option_var - local_access_option_type ?destination_proposal_var - destination_proposal_type)
    :precondition
      (and
        (primary_segment_open_state ?primary_day_segment_var)
        (secondary_segment_open_state ?secondary_day_segment_var)
        (primary_access_link ?primary_day_segment_var ?access_option_var)
        (secondary_local_access_link ?secondary_day_segment_var ?local_access_option_var)
        (alternate_access_state ?access_option_var)
        (preferred_local_access_state ?local_access_option_var)
        (not
          (primary_segment_ready_state ?primary_day_segment_var)
        )
        (secondary_segment_ready_state ?secondary_day_segment_var)
        (proposal_draft_state ?destination_proposal_var)
      )
    :effect
      (and
        (proposal_assembled_state ?destination_proposal_var)
        (proposal_access_link ?destination_proposal_var ?access_option_var)
        (proposal_local_access_link ?destination_proposal_var ?local_access_option_var)
        (primary_gap_state ?destination_proposal_var)
        (not
          (proposal_draft_state ?destination_proposal_var)
        )
      )
  )
  (:action assemble_destination_proposal_secondary_gap
    :parameters (?primary_day_segment_var - primary_day_segment_type ?secondary_day_segment_var - secondary_day_segment_type ?access_option_var - access_option_type ?local_access_option_var - local_access_option_type ?destination_proposal_var - destination_proposal_type)
    :precondition
      (and
        (primary_segment_open_state ?primary_day_segment_var)
        (secondary_segment_open_state ?secondary_day_segment_var)
        (primary_access_link ?primary_day_segment_var ?access_option_var)
        (secondary_local_access_link ?secondary_day_segment_var ?local_access_option_var)
        (preferred_access_state ?access_option_var)
        (alternate_local_access_state ?local_access_option_var)
        (primary_segment_ready_state ?primary_day_segment_var)
        (not
          (secondary_segment_ready_state ?secondary_day_segment_var)
        )
        (proposal_draft_state ?destination_proposal_var)
      )
    :effect
      (and
        (proposal_assembled_state ?destination_proposal_var)
        (proposal_access_link ?destination_proposal_var ?access_option_var)
        (proposal_local_access_link ?destination_proposal_var ?local_access_option_var)
        (secondary_gap_state ?destination_proposal_var)
        (not
          (proposal_draft_state ?destination_proposal_var)
        )
      )
  )
  (:action assemble_destination_proposal_full_gap
    :parameters (?primary_day_segment_var - primary_day_segment_type ?secondary_day_segment_var - secondary_day_segment_type ?access_option_var - access_option_type ?local_access_option_var - local_access_option_type ?destination_proposal_var - destination_proposal_type)
    :precondition
      (and
        (primary_segment_open_state ?primary_day_segment_var)
        (secondary_segment_open_state ?secondary_day_segment_var)
        (primary_access_link ?primary_day_segment_var ?access_option_var)
        (secondary_local_access_link ?secondary_day_segment_var ?local_access_option_var)
        (alternate_access_state ?access_option_var)
        (alternate_local_access_state ?local_access_option_var)
        (not
          (primary_segment_ready_state ?primary_day_segment_var)
        )
        (not
          (secondary_segment_ready_state ?secondary_day_segment_var)
        )
        (proposal_draft_state ?destination_proposal_var)
      )
    :effect
      (and
        (proposal_assembled_state ?destination_proposal_var)
        (proposal_access_link ?destination_proposal_var ?access_option_var)
        (proposal_local_access_link ?destination_proposal_var ?local_access_option_var)
        (primary_gap_state ?destination_proposal_var)
        (secondary_gap_state ?destination_proposal_var)
        (not
          (proposal_draft_state ?destination_proposal_var)
        )
      )
