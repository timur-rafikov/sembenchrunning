## Planning domain (PDDL)

(define (domain airport_transfer_breakdown_recovery)
  (:requirements :strips :typing :negative-preconditions)
  (:types operational_resource - object service_component - object logistical_entity - object case_root - object passenger_case - case_root replacement_vehicle - operational_resource transport_option - operational_resource third_party_provider - operational_resource policy_document - operational_resource authorization_credential - operational_resource payment_instruction - operational_resource maintenance_resource - operational_resource safety_report - operational_resource compensation_offer - service_component itinerary_segment - service_component insurer_claim - service_component pickup_location - logistical_entity dropoff_location - logistical_entity recovery_plan - logistical_entity transfer_provider_category - passenger_case transfer_provider_category_alt - passenger_case local_transfer_unit - transfer_provider_category partner_transfer_unit - transfer_provider_category recovery_coordinator - transfer_provider_category_alt)
  (:predicates
    (incident_logged ?passenger_case - passenger_case)
    (entity_assignment_confirmed ?passenger_case - passenger_case)
    (entity_replacement_reserved ?passenger_case - passenger_case)
    (entity_closure_recorded ?passenger_case - passenger_case)
    (entity_execution_confirmed ?passenger_case - passenger_case)
    (entity_payment_instruction_created ?passenger_case - passenger_case)
    (replacement_available ?replacement_vehicle - replacement_vehicle)
    (entity_assigned_replacement ?passenger_case - passenger_case ?replacement_vehicle - replacement_vehicle)
    (transport_option_available ?transport_option - transport_option)
    (entity_assigned_transport ?passenger_case - passenger_case ?transport_option - transport_option)
    (third_party_provider_available ?third_party_provider - third_party_provider)
    (entity_engaged_provider ?passenger_case - passenger_case ?third_party_provider - third_party_provider)
    (compensation_offer_available ?compensation_offer - compensation_offer)
    (local_unit_assigned_compensation ?local_transfer_unit - local_transfer_unit ?compensation_offer - compensation_offer)
    (partner_unit_assigned_compensation ?partner_transfer_unit - partner_transfer_unit ?compensation_offer - compensation_offer)
    (local_unit_assigned_pickup ?local_transfer_unit - local_transfer_unit ?pickup_location - pickup_location)
    (pickup_confirmed ?pickup_location - pickup_location)
    (pickup_compensation_flag ?pickup_location - pickup_location)
    (local_unit_ready ?local_transfer_unit - local_transfer_unit)
    (partner_unit_assigned_dropoff ?partner_transfer_unit - partner_transfer_unit ?dropoff_location - dropoff_location)
    (dropoff_confirmed ?dropoff_location - dropoff_location)
    (dropoff_compensation_flag ?dropoff_location - dropoff_location)
    (partner_unit_ready ?partner_transfer_unit - partner_transfer_unit)
    (plan_pending ?recovery_plan - recovery_plan)
    (plan_created ?recovery_plan - recovery_plan)
    (plan_includes_pickup ?recovery_plan - recovery_plan ?pickup_location - pickup_location)
    (plan_includes_dropoff ?recovery_plan - recovery_plan ?dropoff_location - dropoff_location)
    (plan_needs_local_followup ?recovery_plan - recovery_plan)
    (plan_needs_partner_followup ?recovery_plan - recovery_plan)
    (plan_finalised ?recovery_plan - recovery_plan)
    (coordinator_allocated_local_unit ?recovery_coordinator - recovery_coordinator ?local_transfer_unit - local_transfer_unit)
    (coordinator_allocated_partner_unit ?recovery_coordinator - recovery_coordinator ?partner_transfer_unit - partner_transfer_unit)
    (coordinator_linked_recovery_plan ?recovery_coordinator - recovery_coordinator ?recovery_plan - recovery_plan)
    (itinerary_segment_pending ?itinerary_segment - itinerary_segment)
    (coordinator_has_itinerary_segment ?recovery_coordinator - recovery_coordinator ?itinerary_segment - itinerary_segment)
    (itinerary_segment_committed ?itinerary_segment - itinerary_segment)
    (itinerary_segment_assigned_to_plan ?itinerary_segment - itinerary_segment ?recovery_plan - recovery_plan)
    (coordinator_documents_prepared ?recovery_coordinator - recovery_coordinator)
    (coordinator_maintenance_authorized ?recovery_coordinator - recovery_coordinator)
    (coordinator_activation_prepared ?recovery_coordinator - recovery_coordinator)
    (coordinator_policy_attached ?recovery_coordinator - recovery_coordinator)
    (coordinator_policy_verified ?recovery_coordinator - recovery_coordinator)
    (coordinator_has_authorizations ?recovery_coordinator - recovery_coordinator)
    (coordinator_validated ?recovery_coordinator - recovery_coordinator)
    (insurer_claim_available ?insurer_claim - insurer_claim)
    (coordinator_linked_claim ?recovery_coordinator - recovery_coordinator ?insurer_claim - insurer_claim)
    (coordinator_claim_attached ?recovery_coordinator - recovery_coordinator)
    (coordinator_claim_acknowledged ?recovery_coordinator - recovery_coordinator)
    (claim_safety_attached ?recovery_coordinator - recovery_coordinator)
    (policy_document_available ?policy_document - policy_document)
    (coordinator_linked_policy ?recovery_coordinator - recovery_coordinator ?policy_document - policy_document)
    (authorization_credential_available ?authorization_credential - authorization_credential)
    (coordinator_attached_authorization ?recovery_coordinator - recovery_coordinator ?authorization_credential - authorization_credential)
    (maintenance_resource_available ?maintenance_resource - maintenance_resource)
    (coordinator_allocated_maintenance_resource ?recovery_coordinator - recovery_coordinator ?maintenance_resource - maintenance_resource)
    (safety_report_available ?safety_report - safety_report)
    (coordinator_attached_safety_report ?recovery_coordinator - recovery_coordinator ?safety_report - safety_report)
    (payment_instruction_available ?payment_instruction - payment_instruction)
    (entity_linked_payment_instruction ?passenger_case - passenger_case ?payment_instruction - payment_instruction)
    (local_unit_checked_in ?local_transfer_unit - local_transfer_unit)
    (partner_unit_checked_in ?partner_transfer_unit - partner_transfer_unit)
    (activation_recorded ?recovery_coordinator - recovery_coordinator)
  )
  (:action log_incident_case
    :parameters (?passenger_case - passenger_case)
    :precondition
      (and
        (not
          (incident_logged ?passenger_case)
        )
        (not
          (entity_closure_recorded ?passenger_case)
        )
      )
    :effect (incident_logged ?passenger_case)
  )
  (:action reserve_replacement_vehicle
    :parameters (?passenger_case - passenger_case ?replacement_vehicle - replacement_vehicle)
    :precondition
      (and
        (incident_logged ?passenger_case)
        (not
          (entity_replacement_reserved ?passenger_case)
        )
        (replacement_available ?replacement_vehicle)
      )
    :effect
      (and
        (entity_replacement_reserved ?passenger_case)
        (entity_assigned_replacement ?passenger_case ?replacement_vehicle)
        (not
          (replacement_available ?replacement_vehicle)
        )
      )
  )
  (:action assign_transport_option
    :parameters (?passenger_case - passenger_case ?transport_option - transport_option)
    :precondition
      (and
        (incident_logged ?passenger_case)
        (entity_replacement_reserved ?passenger_case)
        (transport_option_available ?transport_option)
      )
    :effect
      (and
        (entity_assigned_transport ?passenger_case ?transport_option)
        (not
          (transport_option_available ?transport_option)
        )
      )
  )
  (:action confirm_transport_assignment
    :parameters (?passenger_case - passenger_case ?transport_option - transport_option)
    :precondition
      (and
        (incident_logged ?passenger_case)
        (entity_replacement_reserved ?passenger_case)
        (entity_assigned_transport ?passenger_case ?transport_option)
        (not
          (entity_assignment_confirmed ?passenger_case)
        )
      )
    :effect (entity_assignment_confirmed ?passenger_case)
  )
  (:action release_transport_option
    :parameters (?passenger_case - passenger_case ?transport_option - transport_option)
    :precondition
      (and
        (entity_assigned_transport ?passenger_case ?transport_option)
      )
    :effect
      (and
        (transport_option_available ?transport_option)
        (not
          (entity_assigned_transport ?passenger_case ?transport_option)
        )
      )
  )
  (:action engage_third_party_provider
    :parameters (?passenger_case - passenger_case ?third_party_provider - third_party_provider)
    :precondition
      (and
        (entity_assignment_confirmed ?passenger_case)
        (third_party_provider_available ?third_party_provider)
      )
    :effect
      (and
        (entity_engaged_provider ?passenger_case ?third_party_provider)
        (not
          (third_party_provider_available ?third_party_provider)
        )
      )
  )
  (:action release_third_party_provider
    :parameters (?passenger_case - passenger_case ?third_party_provider - third_party_provider)
    :precondition
      (and
        (entity_engaged_provider ?passenger_case ?third_party_provider)
      )
    :effect
      (and
        (third_party_provider_available ?third_party_provider)
        (not
          (entity_engaged_provider ?passenger_case ?third_party_provider)
        )
      )
  )
  (:action assign_maintenance_resource
    :parameters (?recovery_coordinator - recovery_coordinator ?maintenance_resource - maintenance_resource)
    :precondition
      (and
        (entity_assignment_confirmed ?recovery_coordinator)
        (maintenance_resource_available ?maintenance_resource)
      )
    :effect
      (and
        (coordinator_allocated_maintenance_resource ?recovery_coordinator ?maintenance_resource)
        (not
          (maintenance_resource_available ?maintenance_resource)
        )
      )
  )
  (:action release_maintenance_resource
    :parameters (?recovery_coordinator - recovery_coordinator ?maintenance_resource - maintenance_resource)
    :precondition
      (and
        (coordinator_allocated_maintenance_resource ?recovery_coordinator ?maintenance_resource)
      )
    :effect
      (and
        (maintenance_resource_available ?maintenance_resource)
        (not
          (coordinator_allocated_maintenance_resource ?recovery_coordinator ?maintenance_resource)
        )
      )
  )
  (:action attach_safety_report
    :parameters (?recovery_coordinator - recovery_coordinator ?safety_report - safety_report)
    :precondition
      (and
        (entity_assignment_confirmed ?recovery_coordinator)
        (safety_report_available ?safety_report)
      )
    :effect
      (and
        (coordinator_attached_safety_report ?recovery_coordinator ?safety_report)
        (not
          (safety_report_available ?safety_report)
        )
      )
  )
  (:action detach_safety_report
    :parameters (?recovery_coordinator - recovery_coordinator ?safety_report - safety_report)
    :precondition
      (and
        (coordinator_attached_safety_report ?recovery_coordinator ?safety_report)
      )
    :effect
      (and
        (safety_report_available ?safety_report)
        (not
          (coordinator_attached_safety_report ?recovery_coordinator ?safety_report)
        )
      )
  )
  (:action confirm_local_pickup
    :parameters (?local_transfer_unit - local_transfer_unit ?pickup_location - pickup_location ?transport_option - transport_option)
    :precondition
      (and
        (entity_assignment_confirmed ?local_transfer_unit)
        (entity_assigned_transport ?local_transfer_unit ?transport_option)
        (local_unit_assigned_pickup ?local_transfer_unit ?pickup_location)
        (not
          (pickup_confirmed ?pickup_location)
        )
        (not
          (pickup_compensation_flag ?pickup_location)
        )
      )
    :effect (pickup_confirmed ?pickup_location)
  )
  (:action confirm_local_unit_ready
    :parameters (?local_transfer_unit - local_transfer_unit ?pickup_location - pickup_location ?third_party_provider - third_party_provider)
    :precondition
      (and
        (entity_assignment_confirmed ?local_transfer_unit)
        (entity_engaged_provider ?local_transfer_unit ?third_party_provider)
        (local_unit_assigned_pickup ?local_transfer_unit ?pickup_location)
        (pickup_confirmed ?pickup_location)
        (not
          (local_unit_checked_in ?local_transfer_unit)
        )
      )
    :effect
      (and
        (local_unit_checked_in ?local_transfer_unit)
        (local_unit_ready ?local_transfer_unit)
      )
  )
  (:action assign_compensation_offer_to_local_unit
    :parameters (?local_transfer_unit - local_transfer_unit ?pickup_location - pickup_location ?compensation_offer - compensation_offer)
    :precondition
      (and
        (entity_assignment_confirmed ?local_transfer_unit)
        (local_unit_assigned_pickup ?local_transfer_unit ?pickup_location)
        (compensation_offer_available ?compensation_offer)
        (not
          (local_unit_checked_in ?local_transfer_unit)
        )
      )
    :effect
      (and
        (pickup_compensation_flag ?pickup_location)
        (local_unit_checked_in ?local_transfer_unit)
        (local_unit_assigned_compensation ?local_transfer_unit ?compensation_offer)
        (not
          (compensation_offer_available ?compensation_offer)
        )
      )
  )
  (:action finalize_local_pickup
    :parameters (?local_transfer_unit - local_transfer_unit ?pickup_location - pickup_location ?transport_option - transport_option ?compensation_offer - compensation_offer)
    :precondition
      (and
        (entity_assignment_confirmed ?local_transfer_unit)
        (entity_assigned_transport ?local_transfer_unit ?transport_option)
        (local_unit_assigned_pickup ?local_transfer_unit ?pickup_location)
        (pickup_compensation_flag ?pickup_location)
        (local_unit_assigned_compensation ?local_transfer_unit ?compensation_offer)
        (not
          (local_unit_ready ?local_transfer_unit)
        )
      )
    :effect
      (and
        (pickup_confirmed ?pickup_location)
        (local_unit_ready ?local_transfer_unit)
        (compensation_offer_available ?compensation_offer)
        (not
          (local_unit_assigned_compensation ?local_transfer_unit ?compensation_offer)
        )
      )
  )
  (:action confirm_partner_dropoff
    :parameters (?partner_transfer_unit - partner_transfer_unit ?dropoff_location - dropoff_location ?transport_option - transport_option)
    :precondition
      (and
        (entity_assignment_confirmed ?partner_transfer_unit)
        (entity_assigned_transport ?partner_transfer_unit ?transport_option)
        (partner_unit_assigned_dropoff ?partner_transfer_unit ?dropoff_location)
        (not
          (dropoff_confirmed ?dropoff_location)
        )
        (not
          (dropoff_compensation_flag ?dropoff_location)
        )
      )
    :effect (dropoff_confirmed ?dropoff_location)
  )
  (:action confirm_partner_unit_ready
    :parameters (?partner_transfer_unit - partner_transfer_unit ?dropoff_location - dropoff_location ?third_party_provider - third_party_provider)
    :precondition
      (and
        (entity_assignment_confirmed ?partner_transfer_unit)
        (entity_engaged_provider ?partner_transfer_unit ?third_party_provider)
        (partner_unit_assigned_dropoff ?partner_transfer_unit ?dropoff_location)
        (dropoff_confirmed ?dropoff_location)
        (not
          (partner_unit_checked_in ?partner_transfer_unit)
        )
      )
    :effect
      (and
        (partner_unit_checked_in ?partner_transfer_unit)
        (partner_unit_ready ?partner_transfer_unit)
      )
  )
  (:action assign_compensation_offer_to_partner_unit
    :parameters (?partner_transfer_unit - partner_transfer_unit ?dropoff_location - dropoff_location ?compensation_offer - compensation_offer)
    :precondition
      (and
        (entity_assignment_confirmed ?partner_transfer_unit)
        (partner_unit_assigned_dropoff ?partner_transfer_unit ?dropoff_location)
        (compensation_offer_available ?compensation_offer)
        (not
          (partner_unit_checked_in ?partner_transfer_unit)
        )
      )
    :effect
      (and
        (dropoff_compensation_flag ?dropoff_location)
        (partner_unit_checked_in ?partner_transfer_unit)
        (partner_unit_assigned_compensation ?partner_transfer_unit ?compensation_offer)
        (not
          (compensation_offer_available ?compensation_offer)
        )
      )
  )
  (:action finalize_partner_dropoff
    :parameters (?partner_transfer_unit - partner_transfer_unit ?dropoff_location - dropoff_location ?transport_option - transport_option ?compensation_offer - compensation_offer)
    :precondition
      (and
        (entity_assignment_confirmed ?partner_transfer_unit)
        (entity_assigned_transport ?partner_transfer_unit ?transport_option)
        (partner_unit_assigned_dropoff ?partner_transfer_unit ?dropoff_location)
        (dropoff_compensation_flag ?dropoff_location)
        (partner_unit_assigned_compensation ?partner_transfer_unit ?compensation_offer)
        (not
          (partner_unit_ready ?partner_transfer_unit)
        )
      )
    :effect
      (and
        (dropoff_confirmed ?dropoff_location)
        (partner_unit_ready ?partner_transfer_unit)
        (compensation_offer_available ?compensation_offer)
        (not
          (partner_unit_assigned_compensation ?partner_transfer_unit ?compensation_offer)
        )
      )
  )
  (:action compose_recovery_plan
    :parameters (?local_transfer_unit - local_transfer_unit ?partner_transfer_unit - partner_transfer_unit ?pickup_location - pickup_location ?dropoff_location - dropoff_location ?recovery_plan - recovery_plan)
    :precondition
      (and
        (local_unit_checked_in ?local_transfer_unit)
        (partner_unit_checked_in ?partner_transfer_unit)
        (local_unit_assigned_pickup ?local_transfer_unit ?pickup_location)
        (partner_unit_assigned_dropoff ?partner_transfer_unit ?dropoff_location)
        (pickup_confirmed ?pickup_location)
        (dropoff_confirmed ?dropoff_location)
        (local_unit_ready ?local_transfer_unit)
        (partner_unit_ready ?partner_transfer_unit)
        (plan_pending ?recovery_plan)
      )
    :effect
      (and
        (plan_created ?recovery_plan)
        (plan_includes_pickup ?recovery_plan ?pickup_location)
        (plan_includes_dropoff ?recovery_plan ?dropoff_location)
        (not
          (plan_pending ?recovery_plan)
        )
      )
  )
  (:action compose_recovery_plan_local_followup
    :parameters (?local_transfer_unit - local_transfer_unit ?partner_transfer_unit - partner_transfer_unit ?pickup_location - pickup_location ?dropoff_location - dropoff_location ?recovery_plan - recovery_plan)
    :precondition
      (and
        (local_unit_checked_in ?local_transfer_unit)
        (partner_unit_checked_in ?partner_transfer_unit)
        (local_unit_assigned_pickup ?local_transfer_unit ?pickup_location)
        (partner_unit_assigned_dropoff ?partner_transfer_unit ?dropoff_location)
        (pickup_compensation_flag ?pickup_location)
        (dropoff_confirmed ?dropoff_location)
        (not
          (local_unit_ready ?local_transfer_unit)
        )
        (partner_unit_ready ?partner_transfer_unit)
        (plan_pending ?recovery_plan)
      )
    :effect
      (and
        (plan_created ?recovery_plan)
        (plan_includes_pickup ?recovery_plan ?pickup_location)
        (plan_includes_dropoff ?recovery_plan ?dropoff_location)
        (plan_needs_local_followup ?recovery_plan)
        (not
          (plan_pending ?recovery_plan)
        )
      )
  )
  (:action compose_recovery_plan_partner_followup
    :parameters (?local_transfer_unit - local_transfer_unit ?partner_transfer_unit - partner_transfer_unit ?pickup_location - pickup_location ?dropoff_location - dropoff_location ?recovery_plan - recovery_plan)
    :precondition
      (and
        (local_unit_checked_in ?local_transfer_unit)
        (partner_unit_checked_in ?partner_transfer_unit)
        (local_unit_assigned_pickup ?local_transfer_unit ?pickup_location)
        (partner_unit_assigned_dropoff ?partner_transfer_unit ?dropoff_location)
        (pickup_confirmed ?pickup_location)
        (dropoff_compensation_flag ?dropoff_location)
        (local_unit_ready ?local_transfer_unit)
        (not
          (partner_unit_ready ?partner_transfer_unit)
        )
        (plan_pending ?recovery_plan)
      )
    :effect
      (and
        (plan_created ?recovery_plan)
        (plan_includes_pickup ?recovery_plan ?pickup_location)
        (plan_includes_dropoff ?recovery_plan ?dropoff_location)
        (plan_needs_partner_followup ?recovery_plan)
        (not
          (plan_pending ?recovery_plan)
        )
      )
  )
  (:action compose_recovery_plan_full_followup
    :parameters (?local_transfer_unit - local_transfer_unit ?partner_transfer_unit - partner_transfer_unit ?pickup_location - pickup_location ?dropoff_location - dropoff_location ?recovery_plan - recovery_plan)
    :precondition
      (and
        (local_unit_checked_in ?local_transfer_unit)
        (partner_unit_checked_in ?partner_transfer_unit)
        (local_unit_assigned_pickup ?local_transfer_unit ?pickup_location)
        (partner_unit_assigned_dropoff ?partner_transfer_unit ?dropoff_location)
        (pickup_compensation_flag ?pickup_location)
        (dropoff_compensation_flag ?dropoff_location)
        (not
          (local_unit_ready ?local_transfer_unit)
        )
        (not
          (partner_unit_ready ?partner_transfer_unit)
        )
        (plan_pending ?recovery_plan)
      )
    :effect
      (and
        (plan_created ?recovery_plan)
        (plan_includes_pickup ?recovery_plan ?pickup_location)
        (plan_includes_dropoff ?recovery_plan ?dropoff_location)
        (plan_needs_local_followup ?recovery_plan)
        (plan_needs_partner_followup ?recovery_plan)
        (not
          (plan_pending ?recovery_plan)
        )
      )
  )
  (:action finalize_recovery_plan
    :parameters (?recovery_plan - recovery_plan ?local_transfer_unit - local_transfer_unit ?transport_option - transport_option)
    :precondition
      (and
        (plan_created ?recovery_plan)
        (local_unit_checked_in ?local_transfer_unit)
        (entity_assigned_transport ?local_transfer_unit ?transport_option)
        (not
          (plan_finalised ?recovery_plan)
        )
      )
    :effect (plan_finalised ?recovery_plan)
  )
  (:action attach_itinerary_segment_to_plan
    :parameters (?recovery_coordinator - recovery_coordinator ?itinerary_segment - itinerary_segment ?recovery_plan - recovery_plan)
    :precondition
      (and
        (entity_assignment_confirmed ?recovery_coordinator)
        (coordinator_linked_recovery_plan ?recovery_coordinator ?recovery_plan)
        (coordinator_has_itinerary_segment ?recovery_coordinator ?itinerary_segment)
        (itinerary_segment_pending ?itinerary_segment)
        (plan_created ?recovery_plan)
        (plan_finalised ?recovery_plan)
        (not
          (itinerary_segment_committed ?itinerary_segment)
        )
      )
    :effect
      (and
        (itinerary_segment_committed ?itinerary_segment)
        (itinerary_segment_assigned_to_plan ?itinerary_segment ?recovery_plan)
        (not
          (itinerary_segment_pending ?itinerary_segment)
        )
      )
  )
  (:action prepare_coordinator_for_activation
    :parameters (?recovery_coordinator - recovery_coordinator ?itinerary_segment - itinerary_segment ?recovery_plan - recovery_plan ?transport_option - transport_option)
    :precondition
      (and
        (entity_assignment_confirmed ?recovery_coordinator)
        (coordinator_has_itinerary_segment ?recovery_coordinator ?itinerary_segment)
        (itinerary_segment_committed ?itinerary_segment)
        (itinerary_segment_assigned_to_plan ?itinerary_segment ?recovery_plan)
        (entity_assigned_transport ?recovery_coordinator ?transport_option)
        (not
          (plan_needs_local_followup ?recovery_plan)
        )
        (not
          (coordinator_documents_prepared ?recovery_coordinator)
        )
      )
    :effect (coordinator_documents_prepared ?recovery_coordinator)
  )
  (:action attach_policy_document_to_coordinator
    :parameters (?recovery_coordinator - recovery_coordinator ?policy_document - policy_document)
    :precondition
      (and
        (entity_assignment_confirmed ?recovery_coordinator)
        (policy_document_available ?policy_document)
        (not
          (coordinator_policy_attached ?recovery_coordinator)
        )
      )
    :effect
      (and
        (coordinator_policy_attached ?recovery_coordinator)
        (coordinator_linked_policy ?recovery_coordinator ?policy_document)
        (not
          (policy_document_available ?policy_document)
        )
      )
  )
  (:action verify_policy_and_prepare_coordinator
    :parameters (?recovery_coordinator - recovery_coordinator ?itinerary_segment - itinerary_segment ?recovery_plan - recovery_plan ?transport_option - transport_option ?policy_document - policy_document)
    :precondition
      (and
        (entity_assignment_confirmed ?recovery_coordinator)
        (coordinator_has_itinerary_segment ?recovery_coordinator ?itinerary_segment)
        (itinerary_segment_committed ?itinerary_segment)
        (itinerary_segment_assigned_to_plan ?itinerary_segment ?recovery_plan)
        (entity_assigned_transport ?recovery_coordinator ?transport_option)
        (plan_needs_local_followup ?recovery_plan)
        (coordinator_policy_attached ?recovery_coordinator)
        (coordinator_linked_policy ?recovery_coordinator ?policy_document)
        (not
          (coordinator_documents_prepared ?recovery_coordinator)
        )
      )
    :effect
