## Planning domain (PDDL)

(define (domain pharmaceutics_batch_hold_point_decisioning)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_category - object resource_category_level2 - object resource_category_level3 - object batch_root_type - object batch_instance - batch_root_type production_line - resource_category operation_task - resource_category equipment_unit - resource_category technical_reviewer - resource_category qa_checklist - resource_category hold_reason_ticket - resource_category calibration_certificate - resource_category final_approval_document - resource_category material_lot - resource_category_level2 sample_or_test - resource_category_level2 release_document - resource_category_level2 upstream_hold_point - resource_category_level3 downstream_hold_point - resource_category_level3 batch_record - resource_category_level3 batch_subtype_a - batch_instance batch_subtype_b - batch_instance upstream_batch_unit - batch_subtype_a downstream_batch_unit - batch_subtype_a manufacturing_order - batch_subtype_b)
  (:predicates
    (unit_initiated ?batch - batch_instance)
    (unit_operator_assignment_confirmed ?batch - batch_instance)
    (unit_scheduled ?batch - batch_instance)
    (release_authorized_for_item ?batch - batch_instance)
    (release_candidate_for_item ?batch - batch_instance)
    (release_requested_for_item ?batch - batch_instance)
    (production_line_available ?production_line - production_line)
    (unit_scheduled_on_production_line ?batch - batch_instance ?production_line - production_line)
    (operation_task_available ?operation_task - operation_task)
    (operation_task_assigned_to_unit ?batch - batch_instance ?operation_task - operation_task)
    (equipment_available ?equipment - equipment_unit)
    (equipment_reserved_for_unit ?batch - batch_instance ?equipment - equipment_unit)
    (material_lot_available ?material_lot - material_lot)
    (material_allocated_to_upstream_unit ?upstream_unit - upstream_batch_unit ?material_lot - material_lot)
    (material_allocated_to_downstream_unit ?downstream_unit - downstream_batch_unit ?material_lot - material_lot)
    (upstream_unit_assigned_to_hold_point ?upstream_unit - upstream_batch_unit ?upstream_hold_point - upstream_hold_point)
    (upstream_hold_point_signalled ?upstream_hold_point - upstream_hold_point)
    (upstream_hold_point_material_reserved ?upstream_hold_point - upstream_hold_point)
    (upstream_unit_processing_complete ?upstream_unit - upstream_batch_unit)
    (downstream_unit_assigned_to_hold_point ?downstream_unit - downstream_batch_unit ?downstream_hold_point - downstream_hold_point)
    (downstream_hold_point_signalled ?downstream_hold_point - downstream_hold_point)
    (downstream_hold_point_material_reserved ?downstream_hold_point - downstream_hold_point)
    (downstream_unit_processing_complete ?downstream_unit - downstream_batch_unit)
    (batch_record_draft_available ?batch_record - batch_record)
    (batch_record_assembled ?batch_record - batch_record)
    (batch_record_linked_to_upstream_hold_point ?batch_record - batch_record ?upstream_hold_point - upstream_hold_point)
    (batch_record_linked_to_downstream_hold_point ?batch_record - batch_record ?downstream_hold_point - downstream_hold_point)
    (batch_record_requires_technical_review ?batch_record - batch_record)
    (batch_record_requires_final_signoff ?batch_record - batch_record)
    (batch_record_validated ?batch_record - batch_record)
    (manufacturing_order_linked_to_upstream_unit ?manufacturing_order - manufacturing_order ?upstream_unit - upstream_batch_unit)
    (manufacturing_order_linked_to_downstream_unit ?manufacturing_order - manufacturing_order ?downstream_unit - downstream_batch_unit)
    (manufacturing_order_linked_to_record ?manufacturing_order - manufacturing_order ?batch_record - batch_record)
    (sample_available ?sample - sample_or_test)
    (manufacturing_order_has_sample ?manufacturing_order - manufacturing_order ?sample - sample_or_test)
    (sample_attached ?sample - sample_or_test)
    (sample_linked_to_record ?sample - sample_or_test ?batch_record - batch_record)
    (manufacturing_order_evidence_attached ?manufacturing_order - manufacturing_order)
    (manufacturing_order_tests_completed ?manufacturing_order - manufacturing_order)
    (manufacturing_order_ready_for_review ?manufacturing_order - manufacturing_order)
    (manufacturing_order_reviewer_assigned ?manufacturing_order - manufacturing_order)
    (manufacturing_order_documented_evidence_flag ?manufacturing_order - manufacturing_order)
    (manufacturing_order_qa_checklist_attached ?manufacturing_order - manufacturing_order)
    (manufacturing_order_qa_evaluation_complete ?manufacturing_order - manufacturing_order)
    (release_document_available ?release_document - release_document)
    (manufacturing_order_associated_with_release_document ?manufacturing_order - manufacturing_order ?release_document - release_document)
    (manufacturing_order_release_document_attached ?manufacturing_order - manufacturing_order)
    (manufacturing_order_review_in_progress ?manufacturing_order - manufacturing_order)
    (manufacturing_order_technical_review_completed ?manufacturing_order - manufacturing_order)
    (technical_reviewer_available ?technical_reviewer - technical_reviewer)
    (manufacturing_order_assigned_to_reviewer ?manufacturing_order - manufacturing_order ?technical_reviewer - technical_reviewer)
    (qa_checklist_available ?qa_checklist - qa_checklist)
    (qa_checklist_attached_to_manufacturing_order ?manufacturing_order - manufacturing_order ?qa_checklist - qa_checklist)
    (calibration_certificate_available ?calibration_certificate - calibration_certificate)
    (manufacturing_order_has_calibration_certificate ?manufacturing_order - manufacturing_order ?calibration_certificate - calibration_certificate)
    (final_approval_document_available ?final_approval_document - final_approval_document)
    (manufacturing_order_has_final_approval_document ?manufacturing_order - manufacturing_order ?final_approval_document - final_approval_document)
    (hold_reason_ticket_available ?hold_reason_ticket - hold_reason_ticket)
    (unit_linked_to_hold_reason_ticket ?batch - batch_instance ?hold_reason_ticket - hold_reason_ticket)
    (upstream_unit_prepared_for_handover ?upstream_unit - upstream_batch_unit)
    (downstream_unit_prepared_for_handover ?downstream_unit - downstream_batch_unit)
    (manufacturing_order_release_flag_set ?manufacturing_order - manufacturing_order)
  )
  (:action initialize_unit_instance
    :parameters (?batch - batch_instance)
    :precondition
      (and
        (not
          (unit_initiated ?batch)
        )
        (not
          (release_authorized_for_item ?batch)
        )
      )
    :effect (unit_initiated ?batch)
  )
  (:action schedule_unit_to_production_line
    :parameters (?batch - batch_instance ?production_line - production_line)
    :precondition
      (and
        (unit_initiated ?batch)
        (not
          (unit_scheduled ?batch)
        )
        (production_line_available ?production_line)
      )
    :effect
      (and
        (unit_scheduled ?batch)
        (unit_scheduled_on_production_line ?batch ?production_line)
        (not
          (production_line_available ?production_line)
        )
      )
  )
  (:action assign_operation_task_to_unit
    :parameters (?batch - batch_instance ?operation_task - operation_task)
    :precondition
      (and
        (unit_initiated ?batch)
        (unit_scheduled ?batch)
        (operation_task_available ?operation_task)
      )
    :effect
      (and
        (operation_task_assigned_to_unit ?batch ?operation_task)
        (not
          (operation_task_available ?operation_task)
        )
      )
  )
  (:action confirm_operator_assignment
    :parameters (?batch - batch_instance ?operation_task - operation_task)
    :precondition
      (and
        (unit_initiated ?batch)
        (unit_scheduled ?batch)
        (operation_task_assigned_to_unit ?batch ?operation_task)
        (not
          (unit_operator_assignment_confirmed ?batch)
        )
      )
    :effect (unit_operator_assignment_confirmed ?batch)
  )
  (:action unassign_operation_task_from_unit
    :parameters (?batch - batch_instance ?operation_task - operation_task)
    :precondition
      (and
        (operation_task_assigned_to_unit ?batch ?operation_task)
      )
    :effect
      (and
        (operation_task_available ?operation_task)
        (not
          (operation_task_assigned_to_unit ?batch ?operation_task)
        )
      )
  )
  (:action reserve_equipment_for_unit
    :parameters (?batch - batch_instance ?equipment - equipment_unit)
    :precondition
      (and
        (unit_operator_assignment_confirmed ?batch)
        (equipment_available ?equipment)
      )
    :effect
      (and
        (equipment_reserved_for_unit ?batch ?equipment)
        (not
          (equipment_available ?equipment)
        )
      )
  )
  (:action release_equipment_from_unit
    :parameters (?batch - batch_instance ?equipment - equipment_unit)
    :precondition
      (and
        (equipment_reserved_for_unit ?batch ?equipment)
      )
    :effect
      (and
        (equipment_available ?equipment)
        (not
          (equipment_reserved_for_unit ?batch ?equipment)
        )
      )
  )
  (:action attach_calibration_certificate_to_order
    :parameters (?manufacturing_order - manufacturing_order ?calibration_certificate - calibration_certificate)
    :precondition
      (and
        (unit_operator_assignment_confirmed ?manufacturing_order)
        (calibration_certificate_available ?calibration_certificate)
      )
    :effect
      (and
        (manufacturing_order_has_calibration_certificate ?manufacturing_order ?calibration_certificate)
        (not
          (calibration_certificate_available ?calibration_certificate)
        )
      )
  )
  (:action detach_calibration_certificate_from_order
    :parameters (?manufacturing_order - manufacturing_order ?calibration_certificate - calibration_certificate)
    :precondition
      (and
        (manufacturing_order_has_calibration_certificate ?manufacturing_order ?calibration_certificate)
      )
    :effect
      (and
        (calibration_certificate_available ?calibration_certificate)
        (not
          (manufacturing_order_has_calibration_certificate ?manufacturing_order ?calibration_certificate)
        )
      )
  )
  (:action attach_final_approval_document_to_order
    :parameters (?manufacturing_order - manufacturing_order ?final_approval_document - final_approval_document)
    :precondition
      (and
        (unit_operator_assignment_confirmed ?manufacturing_order)
        (final_approval_document_available ?final_approval_document)
      )
    :effect
      (and
        (manufacturing_order_has_final_approval_document ?manufacturing_order ?final_approval_document)
        (not
          (final_approval_document_available ?final_approval_document)
        )
      )
  )
  (:action detach_final_approval_document_from_order
    :parameters (?manufacturing_order - manufacturing_order ?final_approval_document - final_approval_document)
    :precondition
      (and
        (manufacturing_order_has_final_approval_document ?manufacturing_order ?final_approval_document)
      )
    :effect
      (and
        (final_approval_document_available ?final_approval_document)
        (not
          (manufacturing_order_has_final_approval_document ?manufacturing_order ?final_approval_document)
        )
      )
  )
  (:action signal_upstream_hold_point
    :parameters (?upstream_unit - upstream_batch_unit ?upstream_hold_point - upstream_hold_point ?operation_task - operation_task)
    :precondition
      (and
        (unit_operator_assignment_confirmed ?upstream_unit)
        (operation_task_assigned_to_unit ?upstream_unit ?operation_task)
        (upstream_unit_assigned_to_hold_point ?upstream_unit ?upstream_hold_point)
        (not
          (upstream_hold_point_signalled ?upstream_hold_point)
        )
        (not
          (upstream_hold_point_material_reserved ?upstream_hold_point)
        )
      )
    :effect (upstream_hold_point_signalled ?upstream_hold_point)
  )
  (:action start_upstream_equipment_operation
    :parameters (?upstream_unit - upstream_batch_unit ?upstream_hold_point - upstream_hold_point ?equipment - equipment_unit)
    :precondition
      (and
        (unit_operator_assignment_confirmed ?upstream_unit)
        (equipment_reserved_for_unit ?upstream_unit ?equipment)
        (upstream_unit_assigned_to_hold_point ?upstream_unit ?upstream_hold_point)
        (upstream_hold_point_signalled ?upstream_hold_point)
        (not
          (upstream_unit_prepared_for_handover ?upstream_unit)
        )
      )
    :effect
      (and
        (upstream_unit_prepared_for_handover ?upstream_unit)
        (upstream_unit_processing_complete ?upstream_unit)
      )
  )
  (:action allocate_material_to_upstream_unit
    :parameters (?upstream_unit - upstream_batch_unit ?upstream_hold_point - upstream_hold_point ?material_lot - material_lot)
    :precondition
      (and
        (unit_operator_assignment_confirmed ?upstream_unit)
        (upstream_unit_assigned_to_hold_point ?upstream_unit ?upstream_hold_point)
        (material_lot_available ?material_lot)
        (not
          (upstream_unit_prepared_for_handover ?upstream_unit)
        )
      )
    :effect
      (and
        (upstream_hold_point_material_reserved ?upstream_hold_point)
        (upstream_unit_prepared_for_handover ?upstream_unit)
        (material_allocated_to_upstream_unit ?upstream_unit ?material_lot)
        (not
          (material_lot_available ?material_lot)
        )
      )
  )
  (:action process_upstream_material_and_release
    :parameters (?upstream_unit - upstream_batch_unit ?upstream_hold_point - upstream_hold_point ?operation_task - operation_task ?material_lot - material_lot)
    :precondition
      (and
        (unit_operator_assignment_confirmed ?upstream_unit)
        (operation_task_assigned_to_unit ?upstream_unit ?operation_task)
        (upstream_unit_assigned_to_hold_point ?upstream_unit ?upstream_hold_point)
        (upstream_hold_point_material_reserved ?upstream_hold_point)
        (material_allocated_to_upstream_unit ?upstream_unit ?material_lot)
        (not
          (upstream_unit_processing_complete ?upstream_unit)
        )
      )
    :effect
      (and
        (upstream_hold_point_signalled ?upstream_hold_point)
        (upstream_unit_processing_complete ?upstream_unit)
        (material_lot_available ?material_lot)
        (not
          (material_allocated_to_upstream_unit ?upstream_unit ?material_lot)
        )
      )
  )
  (:action signal_downstream_hold_point
    :parameters (?downstream_unit - downstream_batch_unit ?downstream_hold_point - downstream_hold_point ?operation_task - operation_task)
    :precondition
      (and
        (unit_operator_assignment_confirmed ?downstream_unit)
        (operation_task_assigned_to_unit ?downstream_unit ?operation_task)
        (downstream_unit_assigned_to_hold_point ?downstream_unit ?downstream_hold_point)
        (not
          (downstream_hold_point_signalled ?downstream_hold_point)
        )
        (not
          (downstream_hold_point_material_reserved ?downstream_hold_point)
        )
      )
    :effect (downstream_hold_point_signalled ?downstream_hold_point)
  )
  (:action start_downstream_equipment_operation
    :parameters (?downstream_unit - downstream_batch_unit ?downstream_hold_point - downstream_hold_point ?equipment - equipment_unit)
    :precondition
      (and
        (unit_operator_assignment_confirmed ?downstream_unit)
        (equipment_reserved_for_unit ?downstream_unit ?equipment)
        (downstream_unit_assigned_to_hold_point ?downstream_unit ?downstream_hold_point)
        (downstream_hold_point_signalled ?downstream_hold_point)
        (not
          (downstream_unit_prepared_for_handover ?downstream_unit)
        )
      )
    :effect
      (and
        (downstream_unit_prepared_for_handover ?downstream_unit)
        (downstream_unit_processing_complete ?downstream_unit)
      )
  )
  (:action allocate_material_to_downstream_unit
    :parameters (?downstream_unit - downstream_batch_unit ?downstream_hold_point - downstream_hold_point ?material_lot - material_lot)
    :precondition
      (and
        (unit_operator_assignment_confirmed ?downstream_unit)
        (downstream_unit_assigned_to_hold_point ?downstream_unit ?downstream_hold_point)
        (material_lot_available ?material_lot)
        (not
          (downstream_unit_prepared_for_handover ?downstream_unit)
        )
      )
    :effect
      (and
        (downstream_hold_point_material_reserved ?downstream_hold_point)
        (downstream_unit_prepared_for_handover ?downstream_unit)
        (material_allocated_to_downstream_unit ?downstream_unit ?material_lot)
        (not
          (material_lot_available ?material_lot)
        )
      )
  )
  (:action process_downstream_material_and_release
    :parameters (?downstream_unit - downstream_batch_unit ?downstream_hold_point - downstream_hold_point ?operation_task - operation_task ?material_lot - material_lot)
    :precondition
      (and
        (unit_operator_assignment_confirmed ?downstream_unit)
        (operation_task_assigned_to_unit ?downstream_unit ?operation_task)
        (downstream_unit_assigned_to_hold_point ?downstream_unit ?downstream_hold_point)
        (downstream_hold_point_material_reserved ?downstream_hold_point)
        (material_allocated_to_downstream_unit ?downstream_unit ?material_lot)
        (not
          (downstream_unit_processing_complete ?downstream_unit)
        )
      )
    :effect
      (and
        (downstream_hold_point_signalled ?downstream_hold_point)
        (downstream_unit_processing_complete ?downstream_unit)
        (material_lot_available ?material_lot)
        (not
          (material_allocated_to_downstream_unit ?downstream_unit ?material_lot)
        )
      )
  )
  (:action assemble_batch_record_and_link_hold_points
    :parameters (?upstream_unit - upstream_batch_unit ?downstream_unit - downstream_batch_unit ?upstream_hold_point - upstream_hold_point ?downstream_hold_point - downstream_hold_point ?batch_record - batch_record)
    :precondition
      (and
        (upstream_unit_prepared_for_handover ?upstream_unit)
        (downstream_unit_prepared_for_handover ?downstream_unit)
        (upstream_unit_assigned_to_hold_point ?upstream_unit ?upstream_hold_point)
        (downstream_unit_assigned_to_hold_point ?downstream_unit ?downstream_hold_point)
        (upstream_hold_point_signalled ?upstream_hold_point)
        (downstream_hold_point_signalled ?downstream_hold_point)
        (upstream_unit_processing_complete ?upstream_unit)
        (downstream_unit_processing_complete ?downstream_unit)
        (batch_record_draft_available ?batch_record)
      )
    :effect
      (and
        (batch_record_assembled ?batch_record)
        (batch_record_linked_to_upstream_hold_point ?batch_record ?upstream_hold_point)
        (batch_record_linked_to_downstream_hold_point ?batch_record ?downstream_hold_point)
        (not
          (batch_record_draft_available ?batch_record)
        )
      )
  )
  (:action assemble_batch_record_with_upstream_attachment
    :parameters (?upstream_unit - upstream_batch_unit ?downstream_unit - downstream_batch_unit ?upstream_hold_point - upstream_hold_point ?downstream_hold_point - downstream_hold_point ?batch_record - batch_record)
    :precondition
      (and
        (upstream_unit_prepared_for_handover ?upstream_unit)
        (downstream_unit_prepared_for_handover ?downstream_unit)
        (upstream_unit_assigned_to_hold_point ?upstream_unit ?upstream_hold_point)
        (downstream_unit_assigned_to_hold_point ?downstream_unit ?downstream_hold_point)
        (upstream_hold_point_material_reserved ?upstream_hold_point)
        (downstream_hold_point_signalled ?downstream_hold_point)
        (not
          (upstream_unit_processing_complete ?upstream_unit)
        )
        (downstream_unit_processing_complete ?downstream_unit)
        (batch_record_draft_available ?batch_record)
      )
    :effect
      (and
        (batch_record_assembled ?batch_record)
        (batch_record_linked_to_upstream_hold_point ?batch_record ?upstream_hold_point)
        (batch_record_linked_to_downstream_hold_point ?batch_record ?downstream_hold_point)
        (batch_record_requires_technical_review ?batch_record)
        (not
          (batch_record_draft_available ?batch_record)
        )
      )
  )
  (:action assemble_batch_record_with_downstream_attachment
    :parameters (?upstream_unit - upstream_batch_unit ?downstream_unit - downstream_batch_unit ?upstream_hold_point - upstream_hold_point ?downstream_hold_point - downstream_hold_point ?batch_record - batch_record)
    :precondition
      (and
        (upstream_unit_prepared_for_handover ?upstream_unit)
        (downstream_unit_prepared_for_handover ?downstream_unit)
        (upstream_unit_assigned_to_hold_point ?upstream_unit ?upstream_hold_point)
        (downstream_unit_assigned_to_hold_point ?downstream_unit ?downstream_hold_point)
        (upstream_hold_point_signalled ?upstream_hold_point)
        (downstream_hold_point_material_reserved ?downstream_hold_point)
        (upstream_unit_processing_complete ?upstream_unit)
        (not
          (downstream_unit_processing_complete ?downstream_unit)
        )
        (batch_record_draft_available ?batch_record)
      )
    :effect
      (and
        (batch_record_assembled ?batch_record)
        (batch_record_linked_to_upstream_hold_point ?batch_record ?upstream_hold_point)
        (batch_record_linked_to_downstream_hold_point ?batch_record ?downstream_hold_point)
        (batch_record_requires_final_signoff ?batch_record)
        (not
          (batch_record_draft_available ?batch_record)
        )
      )
  )
  (:action assemble_batch_record_with_both_attachments
    :parameters (?upstream_unit - upstream_batch_unit ?downstream_unit - downstream_batch_unit ?upstream_hold_point - upstream_hold_point ?downstream_hold_point - downstream_hold_point ?batch_record - batch_record)
    :precondition
      (and
        (upstream_unit_prepared_for_handover ?upstream_unit)
        (downstream_unit_prepared_for_handover ?downstream_unit)
        (upstream_unit_assigned_to_hold_point ?upstream_unit ?upstream_hold_point)
        (downstream_unit_assigned_to_hold_point ?downstream_unit ?downstream_hold_point)
        (upstream_hold_point_material_reserved ?upstream_hold_point)
        (downstream_hold_point_material_reserved ?downstream_hold_point)
        (not
          (upstream_unit_processing_complete ?upstream_unit)
        )
        (not
          (downstream_unit_processing_complete ?downstream_unit)
        )
        (batch_record_draft_available ?batch_record)
      )
    :effect
      (and
        (batch_record_assembled ?batch_record)
        (batch_record_linked_to_upstream_hold_point ?batch_record ?upstream_hold_point)
        (batch_record_linked_to_downstream_hold_point ?batch_record ?downstream_hold_point)
        (batch_record_requires_technical_review ?batch_record)
        (batch_record_requires_final_signoff ?batch_record)
        (not
          (batch_record_draft_available ?batch_record)
        )
      )
  )
  (:action finalize_batch_record_validation
    :parameters (?batch_record - batch_record ?upstream_unit - upstream_batch_unit ?operation_task - operation_task)
    :precondition
      (and
        (batch_record_assembled ?batch_record)
        (upstream_unit_prepared_for_handover ?upstream_unit)
        (operation_task_assigned_to_unit ?upstream_unit ?operation_task)
        (not
          (batch_record_validated ?batch_record)
        )
      )
    :effect (batch_record_validated ?batch_record)
  )
  (:action attach_sample_to_record
    :parameters (?manufacturing_order - manufacturing_order ?sample - sample_or_test ?batch_record - batch_record)
    :precondition
      (and
        (unit_operator_assignment_confirmed ?manufacturing_order)
        (manufacturing_order_linked_to_record ?manufacturing_order ?batch_record)
        (manufacturing_order_has_sample ?manufacturing_order ?sample)
        (sample_available ?sample)
        (batch_record_assembled ?batch_record)
        (batch_record_validated ?batch_record)
        (not
          (sample_attached ?sample)
        )
      )
    :effect
      (and
        (sample_attached ?sample)
        (sample_linked_to_record ?sample ?batch_record)
        (not
          (sample_available ?sample)
        )
      )
  )
  (:action flag_order_for_sample_testing
    :parameters (?manufacturing_order - manufacturing_order ?sample - sample_or_test ?batch_record - batch_record ?operation_task - operation_task)
    :precondition
      (and
        (unit_operator_assignment_confirmed ?manufacturing_order)
        (manufacturing_order_has_sample ?manufacturing_order ?sample)
        (sample_attached ?sample)
        (sample_linked_to_record ?sample ?batch_record)
        (operation_task_assigned_to_unit ?manufacturing_order ?operation_task)
        (not
          (batch_record_requires_technical_review ?batch_record)
        )
        (not
          (manufacturing_order_evidence_attached ?manufacturing_order)
        )
      )
    :effect (manufacturing_order_evidence_attached ?manufacturing_order)
  )
  (:action assign_technical_reviewer_to_order
    :parameters (?manufacturing_order - manufacturing_order ?technical_reviewer - technical_reviewer)
    :precondition
      (and
        (unit_operator_assignment_confirmed ?manufacturing_order)
        (technical_reviewer_available ?technical_reviewer)
        (not
          (manufacturing_order_reviewer_assigned ?manufacturing_order)
        )
      )
    :effect
      (and
