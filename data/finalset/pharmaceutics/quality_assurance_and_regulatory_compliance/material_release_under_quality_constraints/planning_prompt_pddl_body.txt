## Planning domain (PDDL)

(define (domain material_release_under_quality_constraints_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_entity - object document_category - object operational_activity_category - object release_candidate_root - object release_candidate - release_candidate_root approver - resource_entity analytical_test_result - resource_entity qa_representative - resource_entity regulatory_checklist - resource_entity training_record - resource_entity regulatory_evidence - resource_entity technical_specialist_approval - resource_entity validation_specialist - resource_entity capa_item - document_category certificate_of_analysis - document_category stability_expert_approval - document_category validation_task - operational_activity_category inspection_activity - operational_activity_category release_record - operational_activity_category manufacturing_batch_group - release_candidate dossier_group - release_candidate production_batch - manufacturing_batch_group inbound_batch - manufacturing_batch_group release_dossier - dossier_group)
  (:predicates
    (release_requested ?release_candidate - release_candidate)
    (entity_under_quality_review ?release_candidate - release_candidate)
    (approver_assigned ?release_candidate - release_candidate)
    (release_authorized ?release_candidate - release_candidate)
    (final_readiness_confirmed ?release_candidate - release_candidate)
    (entity_has_regulatory_evidence ?release_candidate - release_candidate)
    (approver_available ?approver - approver)
    (assigned_approver ?release_candidate - release_candidate ?approver - approver)
    (test_result_available ?analytical_test_result - analytical_test_result)
    (test_result_linked ?release_candidate - release_candidate ?analytical_test_result - analytical_test_result)
    (qa_representative_available ?qa_representative - qa_representative)
    (qa_representative_assigned ?release_candidate - release_candidate ?qa_representative - qa_representative)
    (capa_item_available ?capa_item - capa_item)
    (capa_assigned_to_production_batch ?production_batch - production_batch ?capa_item - capa_item)
    (capa_assigned_to_inbound_batch ?inbound_batch - inbound_batch ?capa_item - capa_item)
    (validation_task_assigned_to_batch ?production_batch - production_batch ?validation_task - validation_task)
    (validation_task_declared ?validation_task - validation_task)
    (validation_task_requires_capa ?validation_task - validation_task)
    (batch_validated ?production_batch - production_batch)
    (inspection_task_assigned_to_inbound_batch ?inbound_batch - inbound_batch ?inspection_activity - inspection_activity)
    (inspection_task_declared ?inspection_activity - inspection_activity)
    (inspection_task_requires_capa ?inspection_activity - inspection_activity)
    (inbound_batch_inspection_passed ?inbound_batch - inbound_batch)
    (release_record_available ?release_record - release_record)
    (release_record_assembled ?release_record - release_record)
    (release_record_contains_validation ?release_record - release_record ?validation_task - validation_task)
    (release_record_contains_inspection ?release_record - release_record ?inspection_activity - inspection_activity)
    (release_record_technical_review_complete ?release_record - release_record)
    (release_record_stability_review_complete ?release_record - release_record)
    (release_record_quality_verified ?release_record - release_record)
    (dossier_covers_production_batch ?release_dossier - release_dossier ?production_batch - production_batch)
    (dossier_covers_inbound_batch ?release_dossier - release_dossier ?inbound_batch - inbound_batch)
    (dossier_linked_to_release_record ?release_dossier - release_dossier ?release_record - release_record)
    (coa_available ?certificate_of_analysis - certificate_of_analysis)
    (dossier_has_coa_attached ?release_dossier - release_dossier ?certificate_of_analysis - certificate_of_analysis)
    (coa_processed ?certificate_of_analysis - certificate_of_analysis)
    (coa_linked_to_release_record ?certificate_of_analysis - certificate_of_analysis ?release_record - release_record)
    (dossier_documents_reviewed ?release_dossier - release_dossier)
    (dossier_technical_reviewed ?release_dossier - release_dossier)
    (dossier_ready_for_final_signoff ?release_dossier - release_dossier)
    (regulatory_checklist_assigned ?release_dossier - release_dossier)
    (dossier_document_review_completed ?release_dossier - release_dossier)
    (dossier_training_evidence_present ?release_dossier - release_dossier)
    (dossier_all_checks_passed ?release_dossier - release_dossier)
    (stability_approval_available ?stability_expert_approval - stability_expert_approval)
    (dossier_has_stability_approval ?release_dossier - release_dossier ?stability_expert_approval - stability_expert_approval)
    (dossier_stability_reviewed ?release_dossier - release_dossier)
    (dossier_stability_review_ready ?release_dossier - release_dossier)
    (dossier_stability_approval_recorded ?release_dossier - release_dossier)
    (regulatory_checklist_available ?regulatory_checklist - regulatory_checklist)
    (dossier_has_regulatory_checklist ?release_dossier - release_dossier ?regulatory_checklist - regulatory_checklist)
    (training_record_available ?training_record - training_record)
    (dossier_has_training_record ?release_dossier - release_dossier ?training_record - training_record)
    (technical_specialist_approval_available ?technical_specialist_approval - technical_specialist_approval)
    (dossier_has_technical_specialist_approval ?release_dossier - release_dossier ?technical_specialist_approval - technical_specialist_approval)
    (validation_specialist_available ?validation_specialist - validation_specialist)
    (dossier_has_validation_specialist ?release_dossier - release_dossier ?validation_specialist - validation_specialist)
    (regulatory_evidence_available ?regulatory_evidence - regulatory_evidence)
    (candidate_linked_to_regulatory_evidence ?release_candidate - release_candidate ?regulatory_evidence - regulatory_evidence)
    (validation_evidence_recorded ?production_batch - production_batch)
    (inspection_evidence_recorded ?inbound_batch - inbound_batch)
    (final_signoff_recorded ?release_dossier - release_dossier)
  )
  (:action initiate_release_request
    :parameters (?release_candidate - release_candidate)
    :precondition
      (and
        (not
          (release_requested ?release_candidate)
        )
        (not
          (release_authorized ?release_candidate)
        )
      )
    :effect (release_requested ?release_candidate)
  )
  (:action assign_approver_to_candidate
    :parameters (?release_candidate - release_candidate ?approver - approver)
    :precondition
      (and
        (release_requested ?release_candidate)
        (not
          (approver_assigned ?release_candidate)
        )
        (approver_available ?approver)
      )
    :effect
      (and
        (approver_assigned ?release_candidate)
        (assigned_approver ?release_candidate ?approver)
        (not
          (approver_available ?approver)
        )
      )
  )
  (:action attach_test_result_to_candidate
    :parameters (?release_candidate - release_candidate ?analytical_test_result - analytical_test_result)
    :precondition
      (and
        (release_requested ?release_candidate)
        (approver_assigned ?release_candidate)
        (test_result_available ?analytical_test_result)
      )
    :effect
      (and
        (test_result_linked ?release_candidate ?analytical_test_result)
        (not
          (test_result_available ?analytical_test_result)
        )
      )
  )
  (:action start_qa_review
    :parameters (?release_candidate - release_candidate ?analytical_test_result - analytical_test_result)
    :precondition
      (and
        (release_requested ?release_candidate)
        (approver_assigned ?release_candidate)
        (test_result_linked ?release_candidate ?analytical_test_result)
        (not
          (entity_under_quality_review ?release_candidate)
        )
      )
    :effect (entity_under_quality_review ?release_candidate)
  )
  (:action unassign_test_result_from_candidate
    :parameters (?release_candidate - release_candidate ?analytical_test_result - analytical_test_result)
    :precondition
      (and
        (test_result_linked ?release_candidate ?analytical_test_result)
      )
    :effect
      (and
        (test_result_available ?analytical_test_result)
        (not
          (test_result_linked ?release_candidate ?analytical_test_result)
        )
      )
  )
  (:action assign_qa_representative
    :parameters (?release_candidate - release_candidate ?qa_representative - qa_representative)
    :precondition
      (and
        (entity_under_quality_review ?release_candidate)
        (qa_representative_available ?qa_representative)
      )
    :effect
      (and
        (qa_representative_assigned ?release_candidate ?qa_representative)
        (not
          (qa_representative_available ?qa_representative)
        )
      )
  )
  (:action unassign_qa_representative
    :parameters (?release_candidate - release_candidate ?qa_representative - qa_representative)
    :precondition
      (and
        (qa_representative_assigned ?release_candidate ?qa_representative)
      )
    :effect
      (and
        (qa_representative_available ?qa_representative)
        (not
          (qa_representative_assigned ?release_candidate ?qa_representative)
        )
      )
  )
  (:action assign_technical_specialist_approval
    :parameters (?release_dossier - release_dossier ?technical_specialist_approval - technical_specialist_approval)
    :precondition
      (and
        (entity_under_quality_review ?release_dossier)
        (technical_specialist_approval_available ?technical_specialist_approval)
      )
    :effect
      (and
        (dossier_has_technical_specialist_approval ?release_dossier ?technical_specialist_approval)
        (not
          (technical_specialist_approval_available ?technical_specialist_approval)
        )
      )
  )
  (:action revoke_technical_specialist_approval
    :parameters (?release_dossier - release_dossier ?technical_specialist_approval - technical_specialist_approval)
    :precondition
      (and
        (dossier_has_technical_specialist_approval ?release_dossier ?technical_specialist_approval)
      )
    :effect
      (and
        (technical_specialist_approval_available ?technical_specialist_approval)
        (not
          (dossier_has_technical_specialist_approval ?release_dossier ?technical_specialist_approval)
        )
      )
  )
  (:action assign_validation_specialist_to_dossier
    :parameters (?release_dossier - release_dossier ?validation_specialist - validation_specialist)
    :precondition
      (and
        (entity_under_quality_review ?release_dossier)
        (validation_specialist_available ?validation_specialist)
      )
    :effect
      (and
        (dossier_has_validation_specialist ?release_dossier ?validation_specialist)
        (not
          (validation_specialist_available ?validation_specialist)
        )
      )
  )
  (:action revoke_validation_specialist_from_dossier
    :parameters (?release_dossier - release_dossier ?validation_specialist - validation_specialist)
    :precondition
      (and
        (dossier_has_validation_specialist ?release_dossier ?validation_specialist)
      )
    :effect
      (and
        (validation_specialist_available ?validation_specialist)
        (not
          (dossier_has_validation_specialist ?release_dossier ?validation_specialist)
        )
      )
  )
  (:action declare_validation_task
    :parameters (?production_batch - production_batch ?validation_task - validation_task ?analytical_test_result - analytical_test_result)
    :precondition
      (and
        (entity_under_quality_review ?production_batch)
        (test_result_linked ?production_batch ?analytical_test_result)
        (validation_task_assigned_to_batch ?production_batch ?validation_task)
        (not
          (validation_task_declared ?validation_task)
        )
        (not
          (validation_task_requires_capa ?validation_task)
        )
      )
    :effect (validation_task_declared ?validation_task)
  )
  (:action complete_validation_activity
    :parameters (?production_batch - production_batch ?validation_task - validation_task ?qa_representative - qa_representative)
    :precondition
      (and
        (entity_under_quality_review ?production_batch)
        (qa_representative_assigned ?production_batch ?qa_representative)
        (validation_task_assigned_to_batch ?production_batch ?validation_task)
        (validation_task_declared ?validation_task)
        (not
          (validation_evidence_recorded ?production_batch)
        )
      )
    :effect
      (and
        (validation_evidence_recorded ?production_batch)
        (batch_validated ?production_batch)
      )
  )
  (:action open_capa_for_validation
    :parameters (?production_batch - production_batch ?validation_task - validation_task ?capa_item - capa_item)
    :precondition
      (and
        (entity_under_quality_review ?production_batch)
        (validation_task_assigned_to_batch ?production_batch ?validation_task)
        (capa_item_available ?capa_item)
        (not
          (validation_evidence_recorded ?production_batch)
        )
      )
    :effect
      (and
        (validation_task_requires_capa ?validation_task)
        (validation_evidence_recorded ?production_batch)
        (capa_assigned_to_production_batch ?production_batch ?capa_item)
        (not
          (capa_item_available ?capa_item)
        )
      )
  )
  (:action resolve_capa_and_revalidate
    :parameters (?production_batch - production_batch ?validation_task - validation_task ?analytical_test_result - analytical_test_result ?capa_item - capa_item)
    :precondition
      (and
        (entity_under_quality_review ?production_batch)
        (test_result_linked ?production_batch ?analytical_test_result)
        (validation_task_assigned_to_batch ?production_batch ?validation_task)
        (validation_task_requires_capa ?validation_task)
        (capa_assigned_to_production_batch ?production_batch ?capa_item)
        (not
          (batch_validated ?production_batch)
        )
      )
    :effect
      (and
        (validation_task_declared ?validation_task)
        (batch_validated ?production_batch)
        (capa_item_available ?capa_item)
        (not
          (capa_assigned_to_production_batch ?production_batch ?capa_item)
        )
      )
  )
  (:action declare_inspection_task
    :parameters (?inbound_batch - inbound_batch ?inspection_activity - inspection_activity ?analytical_test_result - analytical_test_result)
    :precondition
      (and
        (entity_under_quality_review ?inbound_batch)
        (test_result_linked ?inbound_batch ?analytical_test_result)
        (inspection_task_assigned_to_inbound_batch ?inbound_batch ?inspection_activity)
        (not
          (inspection_task_declared ?inspection_activity)
        )
        (not
          (inspection_task_requires_capa ?inspection_activity)
        )
      )
    :effect (inspection_task_declared ?inspection_activity)
  )
  (:action assign_qa_and_complete_inspection
    :parameters (?inbound_batch - inbound_batch ?inspection_activity - inspection_activity ?qa_representative - qa_representative)
    :precondition
      (and
        (entity_under_quality_review ?inbound_batch)
        (qa_representative_assigned ?inbound_batch ?qa_representative)
        (inspection_task_assigned_to_inbound_batch ?inbound_batch ?inspection_activity)
        (inspection_task_declared ?inspection_activity)
        (not
          (inspection_evidence_recorded ?inbound_batch)
        )
      )
    :effect
      (and
        (inspection_evidence_recorded ?inbound_batch)
        (inbound_batch_inspection_passed ?inbound_batch)
      )
  )
  (:action open_capa_for_inspection
    :parameters (?inbound_batch - inbound_batch ?inspection_activity - inspection_activity ?capa_item - capa_item)
    :precondition
      (and
        (entity_under_quality_review ?inbound_batch)
        (inspection_task_assigned_to_inbound_batch ?inbound_batch ?inspection_activity)
        (capa_item_available ?capa_item)
        (not
          (inspection_evidence_recorded ?inbound_batch)
        )
      )
    :effect
      (and
        (inspection_task_requires_capa ?inspection_activity)
        (inspection_evidence_recorded ?inbound_batch)
        (capa_assigned_to_inbound_batch ?inbound_batch ?capa_item)
        (not
          (capa_item_available ?capa_item)
        )
      )
  )
  (:action resolve_inspection_capa_and_reassess
    :parameters (?inbound_batch - inbound_batch ?inspection_activity - inspection_activity ?analytical_test_result - analytical_test_result ?capa_item - capa_item)
    :precondition
      (and
        (entity_under_quality_review ?inbound_batch)
        (test_result_linked ?inbound_batch ?analytical_test_result)
        (inspection_task_assigned_to_inbound_batch ?inbound_batch ?inspection_activity)
        (inspection_task_requires_capa ?inspection_activity)
        (capa_assigned_to_inbound_batch ?inbound_batch ?capa_item)
        (not
          (inbound_batch_inspection_passed ?inbound_batch)
        )
      )
    :effect
      (and
        (inspection_task_declared ?inspection_activity)
        (inbound_batch_inspection_passed ?inbound_batch)
        (capa_item_available ?capa_item)
        (not
          (capa_assigned_to_inbound_batch ?inbound_batch ?capa_item)
        )
      )
  )
  (:action assemble_release_record_standard
    :parameters (?production_batch - production_batch ?inbound_batch - inbound_batch ?validation_task - validation_task ?inspection_activity - inspection_activity ?release_record - release_record)
    :precondition
      (and
        (validation_evidence_recorded ?production_batch)
        (inspection_evidence_recorded ?inbound_batch)
        (validation_task_assigned_to_batch ?production_batch ?validation_task)
        (inspection_task_assigned_to_inbound_batch ?inbound_batch ?inspection_activity)
        (validation_task_declared ?validation_task)
        (inspection_task_declared ?inspection_activity)
        (batch_validated ?production_batch)
        (inbound_batch_inspection_passed ?inbound_batch)
        (release_record_available ?release_record)
      )
    :effect
      (and
        (release_record_assembled ?release_record)
        (release_record_contains_validation ?release_record ?validation_task)
        (release_record_contains_inspection ?release_record ?inspection_activity)
        (not
          (release_record_available ?release_record)
        )
      )
  )
  (:action assemble_release_record_with_technical_review
    :parameters (?production_batch - production_batch ?inbound_batch - inbound_batch ?validation_task - validation_task ?inspection_activity - inspection_activity ?release_record - release_record)
    :precondition
      (and
        (validation_evidence_recorded ?production_batch)
        (inspection_evidence_recorded ?inbound_batch)
        (validation_task_assigned_to_batch ?production_batch ?validation_task)
        (inspection_task_assigned_to_inbound_batch ?inbound_batch ?inspection_activity)
        (validation_task_requires_capa ?validation_task)
        (inspection_task_declared ?inspection_activity)
        (not
          (batch_validated ?production_batch)
        )
        (inbound_batch_inspection_passed ?inbound_batch)
        (release_record_available ?release_record)
      )
    :effect
      (and
        (release_record_assembled ?release_record)
        (release_record_contains_validation ?release_record ?validation_task)
        (release_record_contains_inspection ?release_record ?inspection_activity)
        (release_record_technical_review_complete ?release_record)
        (not
          (release_record_available ?release_record)
        )
      )
  )
  (:action assemble_release_record_with_stability_review
    :parameters (?production_batch - production_batch ?inbound_batch - inbound_batch ?validation_task - validation_task ?inspection_activity - inspection_activity ?release_record - release_record)
    :precondition
      (and
        (validation_evidence_recorded ?production_batch)
        (inspection_evidence_recorded ?inbound_batch)
        (validation_task_assigned_to_batch ?production_batch ?validation_task)
        (inspection_task_assigned_to_inbound_batch ?inbound_batch ?inspection_activity)
        (validation_task_declared ?validation_task)
        (inspection_task_requires_capa ?inspection_activity)
        (batch_validated ?production_batch)
        (not
          (inbound_batch_inspection_passed ?inbound_batch)
        )
        (release_record_available ?release_record)
      )
    :effect
      (and
        (release_record_assembled ?release_record)
        (release_record_contains_validation ?release_record ?validation_task)
        (release_record_contains_inspection ?release_record ?inspection_activity)
        (release_record_stability_review_complete ?release_record)
        (not
          (release_record_available ?release_record)
        )
      )
  )
  (:action assemble_release_record_full_review
    :parameters (?production_batch - production_batch ?inbound_batch - inbound_batch ?validation_task - validation_task ?inspection_activity - inspection_activity ?release_record - release_record)
    :precondition
      (and
        (validation_evidence_recorded ?production_batch)
        (inspection_evidence_recorded ?inbound_batch)
        (validation_task_assigned_to_batch ?production_batch ?validation_task)
        (inspection_task_assigned_to_inbound_batch ?inbound_batch ?inspection_activity)
        (validation_task_requires_capa ?validation_task)
        (inspection_task_requires_capa ?inspection_activity)
        (not
          (batch_validated ?production_batch)
        )
        (not
          (inbound_batch_inspection_passed ?inbound_batch)
        )
        (release_record_available ?release_record)
      )
    :effect
      (and
        (release_record_assembled ?release_record)
        (release_record_contains_validation ?release_record ?validation_task)
        (release_record_contains_inspection ?release_record ?inspection_activity)
        (release_record_technical_review_complete ?release_record)
        (release_record_stability_review_complete ?release_record)
        (not
          (release_record_available ?release_record)
        )
      )
  )
  (:action verify_release_record_quality
    :parameters (?release_record - release_record ?production_batch - production_batch ?analytical_test_result - analytical_test_result)
    :precondition
      (and
        (release_record_assembled ?release_record)
        (validation_evidence_recorded ?production_batch)
        (test_result_linked ?production_batch ?analytical_test_result)
        (not
          (release_record_quality_verified ?release_record)
        )
      )
    :effect (release_record_quality_verified ?release_record)
  )
  (:action attach_coa_and_mark_processed
    :parameters (?release_dossier - release_dossier ?certificate_of_analysis - certificate_of_analysis ?release_record - release_record)
    :precondition
      (and
        (entity_under_quality_review ?release_dossier)
        (dossier_linked_to_release_record ?release_dossier ?release_record)
        (dossier_has_coa_attached ?release_dossier ?certificate_of_analysis)
        (coa_available ?certificate_of_analysis)
        (release_record_assembled ?release_record)
        (release_record_quality_verified ?release_record)
        (not
          (coa_processed ?certificate_of_analysis)
        )
      )
    :effect
      (and
        (coa_processed ?certificate_of_analysis)
        (coa_linked_to_release_record ?certificate_of_analysis ?release_record)
        (not
          (coa_available ?certificate_of_analysis)
        )
      )
  )
  (:action verify_coa_and_mark_dossier_review
    :parameters (?release_dossier - release_dossier ?certificate_of_analysis - certificate_of_analysis ?release_record - release_record ?analytical_test_result - analytical_test_result)
    :precondition
      (and
        (entity_under_quality_review ?release_dossier)
        (dossier_has_coa_attached ?release_dossier ?certificate_of_analysis)
        (coa_processed ?certificate_of_analysis)
        (coa_linked_to_release_record ?certificate_of_analysis ?release_record)
        (test_result_linked ?release_dossier ?analytical_test_result)
        (not
          (release_record_technical_review_complete ?release_record)
        )
        (not
          (dossier_documents_reviewed ?release_dossier)
        )
      )
    :effect (dossier_documents_reviewed ?release_dossier)
  )
  (:action assign_regulatory_checklist_to_dossier
    :parameters (?release_dossier - release_dossier ?regulatory_checklist - regulatory_checklist)
    :precondition
      (and
        (entity_under_quality_review ?release_dossier)
        (regulatory_checklist_available ?regulatory_checklist)
        (not
          (regulatory_checklist_assigned ?release_dossier)
        )
      )
    :effect
      (and
        (regulatory_checklist_assigned ?release_dossier)
        (dossier_has_regulatory_checklist ?release_dossier ?regulatory_checklist)
        (not
          (regulatory_checklist_available ?regulatory_checklist)
        )
      )
  )
  (:action apply_regulatory_checklist_and_record_review
    :parameters (?release_dossier - release_dossier ?certificate_of_analysis - certificate_of_analysis ?release_record - release_record ?analytical_test_result - analytical_test_result ?regulatory_checklist - regulatory_checklist)
    :precondition
      (and
        (entity_under_quality_review ?release_dossier)
        (dossier_has_coa_attached ?release_dossier ?certificate_of_analysis)
        (coa_processed ?certificate_of_analysis)
        (coa_linked_to_release_record ?certificate_of_analysis ?release_record)
        (test_result_linked ?release_dossier ?analytical_test_result)
        (release_record_technical_review_complete ?release_record)
        (regulatory_checklist_assigned ?release_dossier)
