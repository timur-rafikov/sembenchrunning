    :parameters (?medication_request - medication_request)
    :precondition
      (and
        (not
          (item_registered ?medication_request)

          (item_finalized ?medication_request)

    :effect (item_registered ?medication_request)
  )

    :parameters (?medication_request - medication_request ?inventory_stock_unit - inventory_stock_unit)
    :precondition
      (and
        (item_registered ?medication_request)
        (not
          (item_stock_allocated ?medication_request)

        (stock_available ?inventory_stock_unit)

    :effect
      (and
        (item_stock_allocated ?medication_request)
