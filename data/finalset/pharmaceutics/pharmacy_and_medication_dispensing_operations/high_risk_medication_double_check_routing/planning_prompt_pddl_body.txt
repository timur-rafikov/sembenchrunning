## Planning domain (PDDL)

(define (domain pharmacy_high_risk_double_check_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types domain_entity - object physical_item - object supply_item - object operational_entity - object workflow_entity - operational_entity verifier_slot - domain_entity medication_product - domain_entity checker_slot - domain_entity ancillary_supply - domain_entity patient_education_material - domain_entity counseling_topic - domain_entity clinical_check_flag - domain_entity override_token - domain_entity component_item - physical_item label_template - physical_item approval_token - physical_item dosage_form - supply_item package_type - supply_item medication_batch - supply_item preparer_role - workflow_entity checker_role - workflow_entity preparer_technician - preparer_role independent_check_technician - preparer_role dispensing_task - checker_role)
  (:predicates
    (prescription_review_task ?prescription_order - workflow_entity)
    (clinical_clearance_granted ?prescription_order - workflow_entity)
    (verifier_assigned ?prescription_order - workflow_entity)
    (entity_released_to_patient ?prescription_order - workflow_entity)
    (final_release_recorded ?prescription_order - workflow_entity)
    (release_initiated ?prescription_order - workflow_entity)
    (verifier_slot_available ?verifier_slot - verifier_slot)
    (task_verifier_link ?prescription_order - workflow_entity ?verifier_slot - verifier_slot)
    (medication_product_available ?medication_product - medication_product)
    (medication_selected ?prescription_order - workflow_entity ?medication_product - medication_product)
    (checker_slot_available ?checker_slot - checker_slot)
    (task_checker_link ?prescription_order - workflow_entity ?checker_slot - checker_slot)
    (component_available ?component_item - component_item)
    (preparer_allocated_component ?preparer_technician - preparer_technician ?component_item - component_item)
    (checker_allocated_component ?independent_check_technician - independent_check_technician ?component_item - component_item)
    (preparer_selected_dosage_form ?preparer_technician - preparer_technician ?dosage_form - dosage_form)
    (dosage_form_staged ?dosage_form - dosage_form)
    (dosage_form_component_bound ?dosage_form - dosage_form)
    (preparer_component_confirmed ?preparer_technician - preparer_technician)
    (checker_assigned_package_type ?independent_check_technician - independent_check_technician ?package_type - package_type)
    (package_type_staged ?package_type - package_type)
    (package_type_component_bound ?package_type - package_type)
    (checker_component_confirmed ?independent_check_technician - independent_check_technician)
    (batch_pending ?medication_batch - medication_batch)
    (batch_marked_for_packaging ?medication_batch - medication_batch)
    (batch_dosage_form_link ?medication_batch - medication_batch ?dosage_form - dosage_form)
    (batch_package_type_link ?medication_batch - medication_batch ?package_type - package_type)
    (batch_requires_ancillary_supply ?medication_batch - medication_batch)
    (batch_requires_patient_education ?medication_batch - medication_batch)
    (batch_label_ready ?medication_batch - medication_batch)
    (task_assigned_preparer ?dispensing_task - dispensing_task ?preparer_technician - preparer_technician)
    (task_assigned_checker ?dispensing_task - dispensing_task ?independent_check_technician - independent_check_technician)
    (task_assigned_batch ?dispensing_task - dispensing_task ?medication_batch - medication_batch)
    (label_template_available ?label_template - label_template)
    (task_assigned_label_template ?dispensing_task - dispensing_task ?label_template - label_template)
    (label_template_bound ?label_template - label_template)
    (label_template_assigned_to_batch ?label_template - label_template ?medication_batch - medication_batch)
    (task_packaging_in_progress ?dispensing_task - dispensing_task)
    (task_packaging_completed ?dispensing_task - dispensing_task)
    (task_ready_for_final_check ?dispensing_task - dispensing_task)
    (task_ancillary_reserved ?dispensing_task - dispensing_task)
    (task_ancillary_added ?dispensing_task - dispensing_task)
    (task_patient_education_attached ?dispensing_task - dispensing_task)
    (task_independent_check_completed ?dispensing_task - dispensing_task)
    (approval_token_available ?approval_token - approval_token)
    (task_has_approval_token ?dispensing_task - dispensing_task ?approval_token - approval_token)
    (task_approval_recorded ?dispensing_task - dispensing_task)
    (task_authorization_granted ?dispensing_task - dispensing_task)
    (task_override_applied ?dispensing_task - dispensing_task)
    (ancillary_supply_available ?ancillary_supply - ancillary_supply)
    (task_assigned_ancillary_supply ?dispensing_task - dispensing_task ?ancillary_supply - ancillary_supply)
    (patient_education_available ?patient_education_material - patient_education_material)
    (task_assigned_patient_education ?dispensing_task - dispensing_task ?patient_education_material - patient_education_material)
    (clinical_check_flag_available ?clinical_check_flag - clinical_check_flag)
    (task_assigned_clinical_check_flag ?dispensing_task - dispensing_task ?clinical_check_flag - clinical_check_flag)
    (override_token_available ?override_token - override_token)
    (task_override_link ?dispensing_task - dispensing_task ?override_token - override_token)
    (counseling_topic_available ?counseling_topic - counseling_topic)
    (prescription_counseling_link ?prescription_order - workflow_entity ?counseling_topic - counseling_topic)
    (preparer_ready ?preparer_technician - preparer_technician)
    (checker_ready ?independent_check_technician - independent_check_technician)
    (task_released ?dispensing_task - dispensing_task)
  )
  (:action create_prescription_review_task
    :parameters (?prescription_order - workflow_entity)
    :precondition
      (and
        (not
          (prescription_review_task ?prescription_order)
        )
        (not
          (entity_released_to_patient ?prescription_order)
        )
      )
    :effect (prescription_review_task ?prescription_order)
  )
  (:action allocate_verifier_slot_to_prescription
    :parameters (?prescription_order - workflow_entity ?verifier_slot - verifier_slot)
    :precondition
      (and
        (prescription_review_task ?prescription_order)
        (not
          (verifier_assigned ?prescription_order)
        )
        (verifier_slot_available ?verifier_slot)
      )
    :effect
      (and
        (verifier_assigned ?prescription_order)
        (task_verifier_link ?prescription_order ?verifier_slot)
        (not
          (verifier_slot_available ?verifier_slot)
        )
      )
  )
  (:action select_medication_for_prescription
    :parameters (?prescription_order - workflow_entity ?medication_product - medication_product)
    :precondition
      (and
        (prescription_review_task ?prescription_order)
        (verifier_assigned ?prescription_order)
        (medication_product_available ?medication_product)
      )
    :effect
      (and
        (medication_selected ?prescription_order ?medication_product)
        (not
          (medication_product_available ?medication_product)
        )
      )
  )
  (:action authorize_prescription_clinical
    :parameters (?prescription_order - workflow_entity ?medication_product - medication_product)
    :precondition
      (and
        (prescription_review_task ?prescription_order)
        (verifier_assigned ?prescription_order)
        (medication_selected ?prescription_order ?medication_product)
        (not
          (clinical_clearance_granted ?prescription_order)
        )
      )
    :effect (clinical_clearance_granted ?prescription_order)
  )
  (:action release_medication_allocation
    :parameters (?prescription_order - workflow_entity ?medication_product - medication_product)
    :precondition
      (and
        (medication_selected ?prescription_order ?medication_product)
      )
    :effect
      (and
        (medication_product_available ?medication_product)
        (not
          (medication_selected ?prescription_order ?medication_product)
        )
      )
  )
  (:action allocate_checker_slot_to_prescription
    :parameters (?prescription_order - workflow_entity ?checker_slot - checker_slot)
    :precondition
      (and
        (clinical_clearance_granted ?prescription_order)
        (checker_slot_available ?checker_slot)
      )
    :effect
      (and
        (task_checker_link ?prescription_order ?checker_slot)
        (not
          (checker_slot_available ?checker_slot)
        )
      )
  )
  (:action release_checker_slot_from_prescription
    :parameters (?prescription_order - workflow_entity ?checker_slot - checker_slot)
    :precondition
      (and
        (task_checker_link ?prescription_order ?checker_slot)
      )
    :effect
      (and
        (checker_slot_available ?checker_slot)
        (not
          (task_checker_link ?prescription_order ?checker_slot)
        )
      )
  )
  (:action assign_clinical_check_flag_to_task
    :parameters (?dispensing_task - dispensing_task ?clinical_check_flag - clinical_check_flag)
    :precondition
      (and
        (clinical_clearance_granted ?dispensing_task)
        (clinical_check_flag_available ?clinical_check_flag)
      )
    :effect
      (and
        (task_assigned_clinical_check_flag ?dispensing_task ?clinical_check_flag)
        (not
          (clinical_check_flag_available ?clinical_check_flag)
        )
      )
  )
  (:action release_clinical_check_flag_from_task
    :parameters (?dispensing_task - dispensing_task ?clinical_check_flag - clinical_check_flag)
    :precondition
      (and
        (task_assigned_clinical_check_flag ?dispensing_task ?clinical_check_flag)
      )
    :effect
      (and
        (clinical_check_flag_available ?clinical_check_flag)
        (not
          (task_assigned_clinical_check_flag ?dispensing_task ?clinical_check_flag)
        )
      )
  )
  (:action assign_override_token_to_task
    :parameters (?dispensing_task - dispensing_task ?override_token - override_token)
    :precondition
      (and
        (clinical_clearance_granted ?dispensing_task)
        (override_token_available ?override_token)
      )
    :effect
      (and
        (task_override_link ?dispensing_task ?override_token)
        (not
          (override_token_available ?override_token)
        )
      )
  )
  (:action release_override_token_from_task
    :parameters (?dispensing_task - dispensing_task ?override_token - override_token)
    :precondition
      (and
        (task_override_link ?dispensing_task ?override_token)
      )
    :effect
      (and
        (override_token_available ?override_token)
        (not
          (task_override_link ?dispensing_task ?override_token)
        )
      )
  )
  (:action stage_dosage_form_for_preparation
    :parameters (?preparer_technician - preparer_technician ?dosage_form - dosage_form ?medication_product - medication_product)
    :precondition
      (and
        (clinical_clearance_granted ?preparer_technician)
        (medication_selected ?preparer_technician ?medication_product)
        (preparer_selected_dosage_form ?preparer_technician ?dosage_form)
        (not
          (dosage_form_staged ?dosage_form)
        )
        (not
          (dosage_form_component_bound ?dosage_form)
        )
      )
    :effect (dosage_form_staged ?dosage_form)
  )
  (:action preparer_confirm_dosage_staging
    :parameters (?preparer_technician - preparer_technician ?dosage_form - dosage_form ?checker_slot - checker_slot)
    :precondition
      (and
        (clinical_clearance_granted ?preparer_technician)
        (task_checker_link ?preparer_technician ?checker_slot)
        (preparer_selected_dosage_form ?preparer_technician ?dosage_form)
        (dosage_form_staged ?dosage_form)
        (not
          (preparer_ready ?preparer_technician)
        )
      )
    :effect
      (and
        (preparer_ready ?preparer_technician)
        (preparer_component_confirmed ?preparer_technician)
      )
  )
  (:action preparer_allocate_component
    :parameters (?preparer_technician - preparer_technician ?dosage_form - dosage_form ?component_item - component_item)
    :precondition
      (and
        (clinical_clearance_granted ?preparer_technician)
        (preparer_selected_dosage_form ?preparer_technician ?dosage_form)
        (component_available ?component_item)
        (not
          (preparer_ready ?preparer_technician)
        )
      )
    :effect
      (and
        (dosage_form_component_bound ?dosage_form)
        (preparer_ready ?preparer_technician)
        (preparer_allocated_component ?preparer_technician ?component_item)
        (not
          (component_available ?component_item)
        )
      )
  )
  (:action preparer_finalize_component_selection
    :parameters (?preparer_technician - preparer_technician ?dosage_form - dosage_form ?medication_product - medication_product ?component_item - component_item)
    :precondition
      (and
        (clinical_clearance_granted ?preparer_technician)
        (medication_selected ?preparer_technician ?medication_product)
        (preparer_selected_dosage_form ?preparer_technician ?dosage_form)
        (dosage_form_component_bound ?dosage_form)
        (preparer_allocated_component ?preparer_technician ?component_item)
        (not
          (preparer_component_confirmed ?preparer_technician)
        )
      )
    :effect
      (and
        (dosage_form_staged ?dosage_form)
        (preparer_component_confirmed ?preparer_technician)
        (component_available ?component_item)
        (not
          (preparer_allocated_component ?preparer_technician ?component_item)
        )
      )
  )
  (:action stage_package_for_checker
    :parameters (?independent_check_technician - independent_check_technician ?package_type - package_type ?medication_product - medication_product)
    :precondition
      (and
        (clinical_clearance_granted ?independent_check_technician)
        (medication_selected ?independent_check_technician ?medication_product)
        (checker_assigned_package_type ?independent_check_technician ?package_type)
        (not
          (package_type_staged ?package_type)
        )
        (not
          (package_type_component_bound ?package_type)
        )
      )
    :effect (package_type_staged ?package_type)
  )
  (:action checker_confirm_package_staging
    :parameters (?independent_check_technician - independent_check_technician ?package_type - package_type ?checker_slot - checker_slot)
    :precondition
      (and
        (clinical_clearance_granted ?independent_check_technician)
        (task_checker_link ?independent_check_technician ?checker_slot)
        (checker_assigned_package_type ?independent_check_technician ?package_type)
        (package_type_staged ?package_type)
        (not
          (checker_ready ?independent_check_technician)
        )
      )
    :effect
      (and
        (checker_ready ?independent_check_technician)
        (checker_component_confirmed ?independent_check_technician)
      )
  )
  (:action checker_allocate_component
    :parameters (?independent_check_technician - independent_check_technician ?package_type - package_type ?component_item - component_item)
    :precondition
      (and
        (clinical_clearance_granted ?independent_check_technician)
        (checker_assigned_package_type ?independent_check_technician ?package_type)
        (component_available ?component_item)
        (not
          (checker_ready ?independent_check_technician)
        )
      )
    :effect
      (and
        (package_type_component_bound ?package_type)
        (checker_ready ?independent_check_technician)
        (checker_allocated_component ?independent_check_technician ?component_item)
        (not
          (component_available ?component_item)
        )
      )
  )
  (:action checker_finalize_component_selection
    :parameters (?independent_check_technician - independent_check_technician ?package_type - package_type ?medication_product - medication_product ?component_item - component_item)
    :precondition
      (and
        (clinical_clearance_granted ?independent_check_technician)
        (medication_selected ?independent_check_technician ?medication_product)
        (checker_assigned_package_type ?independent_check_technician ?package_type)
        (package_type_component_bound ?package_type)
        (checker_allocated_component ?independent_check_technician ?component_item)
        (not
          (checker_component_confirmed ?independent_check_technician)
        )
      )
    :effect
      (and
        (package_type_staged ?package_type)
        (checker_component_confirmed ?independent_check_technician)
        (component_available ?component_item)
        (not
          (checker_allocated_component ?independent_check_technician ?component_item)
        )
      )
  )
  (:action route_task_to_packaging
    :parameters (?preparer_technician - preparer_technician ?independent_check_technician - independent_check_technician ?dosage_form - dosage_form ?package_type - package_type ?medication_batch - medication_batch)
    :precondition
      (and
        (preparer_ready ?preparer_technician)
        (checker_ready ?independent_check_technician)
        (preparer_selected_dosage_form ?preparer_technician ?dosage_form)
        (checker_assigned_package_type ?independent_check_technician ?package_type)
        (dosage_form_staged ?dosage_form)
        (package_type_staged ?package_type)
        (preparer_component_confirmed ?preparer_technician)
        (checker_component_confirmed ?independent_check_technician)
        (batch_pending ?medication_batch)
      )
    :effect
      (and
        (batch_marked_for_packaging ?medication_batch)
        (batch_dosage_form_link ?medication_batch ?dosage_form)
        (batch_package_type_link ?medication_batch ?package_type)
        (not
          (batch_pending ?medication_batch)
        )
      )
  )
  (:action route_task_to_packaging_with_ancillary
    :parameters (?preparer_technician - preparer_technician ?independent_check_technician - independent_check_technician ?dosage_form - dosage_form ?package_type - package_type ?medication_batch - medication_batch)
    :precondition
      (and
        (preparer_ready ?preparer_technician)
        (checker_ready ?independent_check_technician)
        (preparer_selected_dosage_form ?preparer_technician ?dosage_form)
        (checker_assigned_package_type ?independent_check_technician ?package_type)
        (dosage_form_component_bound ?dosage_form)
        (package_type_staged ?package_type)
        (not
          (preparer_component_confirmed ?preparer_technician)
        )
        (checker_component_confirmed ?independent_check_technician)
        (batch_pending ?medication_batch)
      )
    :effect
      (and
        (batch_marked_for_packaging ?medication_batch)
        (batch_dosage_form_link ?medication_batch ?dosage_form)
        (batch_package_type_link ?medication_batch ?package_type)
        (batch_requires_ancillary_supply ?medication_batch)
        (not
          (batch_pending ?medication_batch)
        )
      )
  )
  (:action route_task_to_packaging_with_education
    :parameters (?preparer_technician - preparer_technician ?independent_check_technician - independent_check_technician ?dosage_form - dosage_form ?package_type - package_type ?medication_batch - medication_batch)
    :precondition
      (and
        (preparer_ready ?preparer_technician)
        (checker_ready ?independent_check_technician)
        (preparer_selected_dosage_form ?preparer_technician ?dosage_form)
        (checker_assigned_package_type ?independent_check_technician ?package_type)
        (dosage_form_staged ?dosage_form)
        (package_type_component_bound ?package_type)
        (preparer_component_confirmed ?preparer_technician)
        (not
          (checker_component_confirmed ?independent_check_technician)
        )
        (batch_pending ?medication_batch)
      )
    :effect
      (and
        (batch_marked_for_packaging ?medication_batch)
        (batch_dosage_form_link ?medication_batch ?dosage_form)
        (batch_package_type_link ?medication_batch ?package_type)
        (batch_requires_patient_education ?medication_batch)
        (not
          (batch_pending ?medication_batch)
        )
      )
  )
  (:action route_task_to_packaging_with_ancillary_and_education
    :parameters (?preparer_technician - preparer_technician ?independent_check_technician - independent_check_technician ?dosage_form - dosage_form ?package_type - package_type ?medication_batch - medication_batch)
    :precondition
      (and
        (preparer_ready ?preparer_technician)
        (checker_ready ?independent_check_technician)
        (preparer_selected_dosage_form ?preparer_technician ?dosage_form)
        (checker_assigned_package_type ?independent_check_technician ?package_type)
        (dosage_form_component_bound ?dosage_form)
        (package_type_component_bound ?package_type)
        (not
          (preparer_component_confirmed ?preparer_technician)
        )
        (not
          (checker_component_confirmed ?independent_check_technician)
        )
        (batch_pending ?medication_batch)
      )
    :effect
      (and
        (batch_marked_for_packaging ?medication_batch)
        (batch_dosage_form_link ?medication_batch ?dosage_form)
        (batch_package_type_link ?medication_batch ?package_type)
        (batch_requires_ancillary_supply ?medication_batch)
        (batch_requires_patient_education ?medication_batch)
        (not
          (batch_pending ?medication_batch)
        )
      )
  )
  (:action mark_batch_label_ready
    :parameters (?medication_batch - medication_batch ?preparer_technician - preparer_technician ?medication_product - medication_product)
    :precondition
      (and
        (batch_marked_for_packaging ?medication_batch)
        (preparer_ready ?preparer_technician)
        (medication_selected ?preparer_technician ?medication_product)
        (not
          (batch_label_ready ?medication_batch)
        )
      )
    :effect (batch_label_ready ?medication_batch)
  )
  (:action bind_label_template_to_batch_and_task
    :parameters (?dispensing_task - dispensing_task ?label_template - label_template ?medication_batch - medication_batch)
    :precondition
      (and
        (clinical_clearance_granted ?dispensing_task)
        (task_assigned_batch ?dispensing_task ?medication_batch)
        (task_assigned_label_template ?dispensing_task ?label_template)
        (label_template_available ?label_template)
        (batch_marked_for_packaging ?medication_batch)
        (batch_label_ready ?medication_batch)
        (not
          (label_template_bound ?label_template)
        )
      )
    :effect
      (and
        (label_template_bound ?label_template)
        (label_template_assigned_to_batch ?label_template ?medication_batch)
        (not
          (label_template_available ?label_template)
        )
      )
  )
  (:action start_task_packaging
    :parameters (?dispensing_task - dispensing_task ?label_template - label_template ?medication_batch - medication_batch ?medication_product - medication_product)
    :precondition
      (and
        (clinical_clearance_granted ?dispensing_task)
        (task_assigned_label_template ?dispensing_task ?label_template)
        (label_template_bound ?label_template)
        (label_template_assigned_to_batch ?label_template ?medication_batch)
        (medication_selected ?dispensing_task ?medication_product)
        (not
          (batch_requires_ancillary_supply ?medication_batch)
        )
        (not
          (task_packaging_in_progress ?dispensing_task)
        )
      )
    :effect (task_packaging_in_progress ?dispensing_task)
  )
  (:action assign_ancillary_supply_to_task
    :parameters (?dispensing_task - dispensing_task ?ancillary_supply - ancillary_supply)
    :precondition
      (and
        (clinical_clearance_granted ?dispensing_task)
        (ancillary_supply_available ?ancillary_supply)
        (not
          (task_ancillary_reserved ?dispensing_task)
        )
      )
    :effect
      (and
        (task_ancillary_reserved ?dispensing_task)
        (task_assigned_ancillary_supply ?dispensing_task ?ancillary_supply)
        (not
          (ancillary_supply_available ?ancillary_supply)
        )
      )
  )
  (:action attach_ancillary_supply_and_start_packaging
    :parameters (?dispensing_task - dispensing_task ?label_template - label_template ?medication_batch - medication_batch ?medication_product - medication_product ?ancillary_supply - ancillary_supply)
    :precondition
      (and
        (clinical_clearance_granted ?dispensing_task)
        (task_assigned_label_template ?dispensing_task ?label_template)
        (label_template_bound ?label_template)
        (label_template_assigned_to_batch ?label_template ?medication_batch)
        (medication_selected ?dispensing_task ?medication_product)
        (batch_requires_ancillary_supply ?medication_batch)
        (task_ancillary_reserved ?dispensing_task)
        (task_assigned_ancillary_supply ?dispensing_task ?ancillary_supply)
        (not
          (task_packaging_in_progress ?dispensing_task)
        )
      )
    :effect
      (and
        (task_packaging_in_progress ?dispensing_task)
        (task_ancillary_added ?dispensing_task)
      )
  )
  (:action complete_packaging_check_with_clinical_flag
    :parameters (?dispensing_task - dispensing_task ?clinical_check_flag - clinical_check_flag ?checker_slot - checker_slot ?label_template - label_template ?medication_batch - medication_batch)
    :precondition
      (and
        (task_packaging_in_progress ?dispensing_task)
        (task_assigned_clinical_check_flag ?dispensing_task ?clinical_check_flag)
