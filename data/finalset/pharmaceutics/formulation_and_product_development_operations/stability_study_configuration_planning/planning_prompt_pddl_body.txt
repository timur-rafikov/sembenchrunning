## Planning domain (PDDL)

(define (domain stability_study_configuration_planning)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity - object resource_category - entity material_category - entity configuration_category - entity lifecycle_class - entity development_item - lifecycle_class analytical_resource - resource_category analytical_method - resource_category reviewer - resource_category regulatory_document_type - resource_category analytical_package - resource_category study_tag - resource_category acceptance_criterion - resource_category change_control_record - resource_category packaging_option - material_category sample_set - material_category quality_attribute - material_category storage_condition - configuration_category packaging_configuration - configuration_category study_arm - configuration_category production_unit - development_item product_instance_group - development_item manufacturing_batch - production_unit formulation_variant - production_unit product_candidate - product_instance_group)

  (:predicates
    (development_item_registered ?development_item - development_item)
    (development_item_ready_for_review ?development_item - development_item)
    (development_item_has_reserved_resource ?development_item - development_item)
    (configuration_finalized ?development_item - development_item)
    (approved_for_execution ?development_item - development_item)
    (development_item_has_study_tag_flag ?development_item - development_item)
    (analytical_resource_available ?analytical_resource - analytical_resource)
    (reserved_resource_for_development_item ?development_item - development_item ?analytical_resource - analytical_resource)
    (analytical_method_available ?analytical_method - analytical_method)
    (assigned_method_for_development_item ?development_item - development_item ?analytical_method - analytical_method)
    (reviewer_available ?reviewer - reviewer)
    (assigned_reviewer_to_development_item ?development_item - development_item ?reviewer - reviewer)
    (packaging_option_available ?packaging_option - packaging_option)
    (batch_packaging_assignment ?manufacturing_batch - manufacturing_batch ?packaging_option - packaging_option)
    (variant_packaging_assignment ?formulation_variant - formulation_variant ?packaging_option - packaging_option)
    (batch_storage_condition_compatibility ?manufacturing_batch - manufacturing_batch ?storage_condition - storage_condition)
    (storage_condition_validated ?storage_condition - storage_condition)
    (storage_condition_associated_with_packaging ?storage_condition - storage_condition)
    (batch_qualified ?manufacturing_batch - manufacturing_batch)
    (variant_packaging_configuration_assignment ?formulation_variant - formulation_variant ?packaging_configuration - packaging_configuration)
    (packaging_configuration_validated ?packaging_configuration - packaging_configuration)
    (packaging_configuration_associated ?packaging_configuration - packaging_configuration)
    (variant_qualified ?formulation_variant - formulation_variant)
    (study_arm_unassigned ?study_arm - study_arm)
    (study_arm_created ?study_arm - study_arm)
    (study_arm_assigned_storage_condition ?study_arm - study_arm ?storage_condition - storage_condition)
    (study_arm_assigned_packaging_configuration ?study_arm - study_arm ?packaging_configuration - packaging_configuration)
    (study_arm_storage_verified ?study_arm - study_arm)
    (study_arm_packaging_verified ?study_arm - study_arm)
    (study_arm_samples_allocated ?study_arm - study_arm)
    (product_candidate_batch_link ?product_candidate - product_candidate ?manufacturing_batch - manufacturing_batch)
    (product_candidate_variant_link ?product_candidate - product_candidate ?formulation_variant - formulation_variant)
    (product_candidate_assigned_to_study_arm ?product_candidate - product_candidate ?study_arm - study_arm)
    (sample_set_available ?sample_set - sample_set)
    (product_candidate_has_sample_set ?product_candidate - product_candidate ?sample_set - sample_set)
    (sample_set_reserved ?sample_set - sample_set)
    (sample_set_assigned_to_study_arm ?sample_set - sample_set ?study_arm - study_arm)
    (product_candidate_qualification_flag ?product_candidate - product_candidate)
    (product_candidate_ready_for_governance ?product_candidate - product_candidate)
    (product_candidate_governance_started ?product_candidate - product_candidate)
    (regulatory_document_attached_to_product_candidate ?product_candidate - product_candidate)
    (regulatory_document_verified_for_product_candidate ?product_candidate - product_candidate)
    (product_candidate_eligible_for_analytical_package ?product_candidate - product_candidate)
    (product_candidate_cross_discipline_clearance ?product_candidate - product_candidate)
    (quality_attribute_available ?quality_attribute - quality_attribute)
    (product_candidate_quality_attribute_association ?product_candidate - product_candidate ?quality_attribute - quality_attribute)
    (product_candidate_quality_attribute_bound ?product_candidate - product_candidate)
    (product_candidate_change_control_ready ?product_candidate - product_candidate)
    (product_candidate_change_control_reviewed ?product_candidate - product_candidate)
    (regulatory_document_type_available ?regulatory_document_type - regulatory_document_type)
    (product_candidate_regulatory_document_link ?product_candidate - product_candidate ?regulatory_document_type - regulatory_document_type)
    (analytical_package_available ?analytical_package - analytical_package)
    (product_candidate_analytical_package_assigned ?product_candidate - product_candidate ?analytical_package - analytical_package)
    (acceptance_criterion_available ?acceptance_criterion - acceptance_criterion)
    (product_candidate_acceptance_criterion_assigned ?product_candidate - product_candidate ?acceptance_criterion - acceptance_criterion)
    (change_control_available ?change_control_record - change_control_record)
    (product_candidate_change_control_assigned ?product_candidate - product_candidate ?change_control_record - change_control_record)
    (study_tag_available ?study_tag - study_tag)
    (development_item_has_study_tag ?development_item - development_item ?study_tag - study_tag)
    (batch_ready_for_assembly ?manufacturing_batch - manufacturing_batch)
    (variant_ready_for_assembly ?formulation_variant - formulation_variant)
    (product_candidate_final_approval ?product_candidate - product_candidate)
  )
  (:action register_development_item
    :parameters (?development_item - development_item)
    :precondition
      (and
        (not
          (development_item_registered ?development_item)
        )
        (not
          (configuration_finalized ?development_item)
        )
      )
    :effect (development_item_registered ?development_item)
  )
  (:action assign_analytical_resource_to_item
    :parameters (?development_item - development_item ?analytical_resource - analytical_resource)
    :precondition
      (and
        (development_item_registered ?development_item)
        (not
          (development_item_has_reserved_resource ?development_item)
        )
        (analytical_resource_available ?analytical_resource)
      )
    :effect
      (and
        (development_item_has_reserved_resource ?development_item)
        (reserved_resource_for_development_item ?development_item ?analytical_resource)
        (not
          (analytical_resource_available ?analytical_resource)
        )
      )
  )
  (:action assign_analytical_method_to_item
    :parameters (?development_item - development_item ?analytical_method - analytical_method)
    :precondition
      (and
        (development_item_registered ?development_item)
        (development_item_has_reserved_resource ?development_item)
        (analytical_method_available ?analytical_method)
      )
    :effect
      (and
        (assigned_method_for_development_item ?development_item ?analytical_method)
        (not
          (analytical_method_available ?analytical_method)
        )
      )
  )
  (:action submit_item_for_review
    :parameters (?development_item - development_item ?analytical_method - analytical_method)
    :precondition
      (and
        (development_item_registered ?development_item)
        (development_item_has_reserved_resource ?development_item)
        (assigned_method_for_development_item ?development_item ?analytical_method)
        (not
          (development_item_ready_for_review ?development_item)
        )
      )
    :effect (development_item_ready_for_review ?development_item)
  )
  (:action release_analytical_method_from_item
    :parameters (?development_item - development_item ?analytical_method - analytical_method)
    :precondition
      (and
        (assigned_method_for_development_item ?development_item ?analytical_method)
      )
    :effect
      (and
        (analytical_method_available ?analytical_method)
        (not
          (assigned_method_for_development_item ?development_item ?analytical_method)
        )
      )
  )
  (:action assign_reviewer_to_item
    :parameters (?development_item - development_item ?reviewer - reviewer)
    :precondition
      (and
        (development_item_ready_for_review ?development_item)
        (reviewer_available ?reviewer)
      )
    :effect
      (and
        (assigned_reviewer_to_development_item ?development_item ?reviewer)
        (not
          (reviewer_available ?reviewer)
        )
      )
  )
  (:action release_reviewer_from_item
    :parameters (?development_item - development_item ?reviewer - reviewer)
    :precondition
      (and
        (assigned_reviewer_to_development_item ?development_item ?reviewer)
      )
    :effect
      (and
        (reviewer_available ?reviewer)
        (not
          (assigned_reviewer_to_development_item ?development_item ?reviewer)
        )
      )
  )
  (:action attach_acceptance_criterion_to_product_candidate
    :parameters (?product_candidate - product_candidate ?acceptance_criterion - acceptance_criterion)
    :precondition
      (and
        (development_item_ready_for_review ?product_candidate)
        (acceptance_criterion_available ?acceptance_criterion)
      )
    :effect
      (and
        (product_candidate_acceptance_criterion_assigned ?product_candidate ?acceptance_criterion)
        (not
          (acceptance_criterion_available ?acceptance_criterion)
        )
      )
  )
  (:action detach_acceptance_criterion_from_product_candidate
    :parameters (?product_candidate - product_candidate ?acceptance_criterion - acceptance_criterion)
    :precondition
      (and
        (product_candidate_acceptance_criterion_assigned ?product_candidate ?acceptance_criterion)
      )
    :effect
      (and
        (acceptance_criterion_available ?acceptance_criterion)
        (not
          (product_candidate_acceptance_criterion_assigned ?product_candidate ?acceptance_criterion)
        )
      )
  )
  (:action attach_change_control_to_product_candidate
    :parameters (?product_candidate - product_candidate ?change_control_record - change_control_record)
    :precondition
      (and
        (development_item_ready_for_review ?product_candidate)
        (change_control_available ?change_control_record)
      )
    :effect
      (and
        (product_candidate_change_control_assigned ?product_candidate ?change_control_record)
        (not
          (change_control_available ?change_control_record)
        )
      )
  )
  (:action detach_change_control_from_product_candidate
    :parameters (?product_candidate - product_candidate ?change_control_record - change_control_record)
    :precondition
      (and
        (product_candidate_change_control_assigned ?product_candidate ?change_control_record)
      )
    :effect
      (and
        (change_control_available ?change_control_record)
        (not
          (product_candidate_change_control_assigned ?product_candidate ?change_control_record)
        )
      )
  )
  (:action validate_storage_condition_for_batch
    :parameters (?manufacturing_batch - manufacturing_batch ?storage_condition - storage_condition ?analytical_method - analytical_method)
    :precondition
      (and
        (development_item_ready_for_review ?manufacturing_batch)
        (assigned_method_for_development_item ?manufacturing_batch ?analytical_method)
        (batch_storage_condition_compatibility ?manufacturing_batch ?storage_condition)
        (not
          (storage_condition_validated ?storage_condition)
        )
        (not
          (storage_condition_associated_with_packaging ?storage_condition)
        )
      )
    :effect (storage_condition_validated ?storage_condition)
  )
  (:action review_batch_storage_condition
    :parameters (?manufacturing_batch - manufacturing_batch ?storage_condition - storage_condition ?reviewer - reviewer)
    :precondition
      (and
        (development_item_ready_for_review ?manufacturing_batch)
        (assigned_reviewer_to_development_item ?manufacturing_batch ?reviewer)
        (batch_storage_condition_compatibility ?manufacturing_batch ?storage_condition)
        (storage_condition_validated ?storage_condition)
        (not
          (batch_ready_for_assembly ?manufacturing_batch)
        )
      )
    :effect
      (and
        (batch_ready_for_assembly ?manufacturing_batch)
        (batch_qualified ?manufacturing_batch)
      )
  )
  (:action assign_packaging_option_to_batch
    :parameters (?manufacturing_batch - manufacturing_batch ?storage_condition - storage_condition ?packaging_option - packaging_option)
    :precondition
      (and
        (development_item_ready_for_review ?manufacturing_batch)
        (batch_storage_condition_compatibility ?manufacturing_batch ?storage_condition)
        (packaging_option_available ?packaging_option)
        (not
          (batch_ready_for_assembly ?manufacturing_batch)
        )
      )
    :effect
      (and
        (storage_condition_associated_with_packaging ?storage_condition)
        (batch_ready_for_assembly ?manufacturing_batch)
        (batch_packaging_assignment ?manufacturing_batch ?packaging_option)
        (not
          (packaging_option_available ?packaging_option)
        )
      )
  )
  (:action finalize_batch_condition_and_packaging
    :parameters (?manufacturing_batch - manufacturing_batch ?storage_condition - storage_condition ?analytical_method - analytical_method ?packaging_option - packaging_option)
    :precondition
      (and
        (development_item_ready_for_review ?manufacturing_batch)
        (assigned_method_for_development_item ?manufacturing_batch ?analytical_method)
        (batch_storage_condition_compatibility ?manufacturing_batch ?storage_condition)
        (storage_condition_associated_with_packaging ?storage_condition)
        (batch_packaging_assignment ?manufacturing_batch ?packaging_option)
        (not
          (batch_qualified ?manufacturing_batch)
        )
      )
    :effect
      (and
        (storage_condition_validated ?storage_condition)
        (batch_qualified ?manufacturing_batch)
        (packaging_option_available ?packaging_option)
        (not
          (batch_packaging_assignment ?manufacturing_batch ?packaging_option)
        )
      )
  )
  (:action initiate_variant_storage_qualification
    :parameters (?formulation_variant - formulation_variant ?packaging_configuration - packaging_configuration ?analytical_method - analytical_method)
    :precondition
      (and
        (development_item_ready_for_review ?formulation_variant)
        (assigned_method_for_development_item ?formulation_variant ?analytical_method)
        (variant_packaging_configuration_assignment ?formulation_variant ?packaging_configuration)
        (not
          (packaging_configuration_validated ?packaging_configuration)
        )
        (not
          (packaging_configuration_associated ?packaging_configuration)
        )
      )
    :effect (packaging_configuration_validated ?packaging_configuration)
  )
  (:action review_variant_qualification
    :parameters (?formulation_variant - formulation_variant ?packaging_configuration - packaging_configuration ?reviewer - reviewer)
    :precondition
      (and
        (development_item_ready_for_review ?formulation_variant)
        (assigned_reviewer_to_development_item ?formulation_variant ?reviewer)
        (variant_packaging_configuration_assignment ?formulation_variant ?packaging_configuration)
        (packaging_configuration_validated ?packaging_configuration)
        (not
          (variant_ready_for_assembly ?formulation_variant)
        )
      )
    :effect
      (and
        (variant_ready_for_assembly ?formulation_variant)
        (variant_qualified ?formulation_variant)
      )
  )
  (:action assign_packaging_option_to_variant
    :parameters (?formulation_variant - formulation_variant ?packaging_configuration - packaging_configuration ?packaging_option - packaging_option)
    :precondition
      (and
        (development_item_ready_for_review ?formulation_variant)
        (variant_packaging_configuration_assignment ?formulation_variant ?packaging_configuration)
        (packaging_option_available ?packaging_option)
        (not
          (variant_ready_for_assembly ?formulation_variant)
        )
      )
    :effect
      (and
        (packaging_configuration_associated ?packaging_configuration)
        (variant_ready_for_assembly ?formulation_variant)
        (variant_packaging_assignment ?formulation_variant ?packaging_option)
        (not
          (packaging_option_available ?packaging_option)
        )
      )
  )
  (:action finalize_variant_condition_and_packaging
    :parameters (?formulation_variant - formulation_variant ?packaging_configuration - packaging_configuration ?analytical_method - analytical_method ?packaging_option - packaging_option)
    :precondition
      (and
        (development_item_ready_for_review ?formulation_variant)
        (assigned_method_for_development_item ?formulation_variant ?analytical_method)
        (variant_packaging_configuration_assignment ?formulation_variant ?packaging_configuration)
        (packaging_configuration_associated ?packaging_configuration)
        (variant_packaging_assignment ?formulation_variant ?packaging_option)
        (not
          (variant_qualified ?formulation_variant)
        )
      )
    :effect
      (and
        (packaging_configuration_validated ?packaging_configuration)
        (variant_qualified ?formulation_variant)
        (packaging_option_available ?packaging_option)
        (not
          (variant_packaging_assignment ?formulation_variant ?packaging_option)
        )
      )
  )
  (:action assemble_study_arm
    :parameters (?manufacturing_batch - manufacturing_batch ?formulation_variant - formulation_variant ?storage_condition - storage_condition ?packaging_configuration - packaging_configuration ?study_arm - study_arm)
    :precondition
      (and
        (batch_ready_for_assembly ?manufacturing_batch)
        (variant_ready_for_assembly ?formulation_variant)
        (batch_storage_condition_compatibility ?manufacturing_batch ?storage_condition)
        (variant_packaging_configuration_assignment ?formulation_variant ?packaging_configuration)
        (storage_condition_validated ?storage_condition)
        (packaging_configuration_validated ?packaging_configuration)
        (batch_qualified ?manufacturing_batch)
        (variant_qualified ?formulation_variant)
        (study_arm_unassigned ?study_arm)
      )
    :effect
      (and
        (study_arm_created ?study_arm)
        (study_arm_assigned_storage_condition ?study_arm ?storage_condition)
        (study_arm_assigned_packaging_configuration ?study_arm ?packaging_configuration)
        (not
          (study_arm_unassigned ?study_arm)
        )
      )
  )
  (:action assemble_study_arm_with_packaging_validation
    :parameters (?manufacturing_batch - manufacturing_batch ?formulation_variant - formulation_variant ?storage_condition - storage_condition ?packaging_configuration - packaging_configuration ?study_arm - study_arm)
    :precondition
      (and
        (batch_ready_for_assembly ?manufacturing_batch)
        (variant_ready_for_assembly ?formulation_variant)
        (batch_storage_condition_compatibility ?manufacturing_batch ?storage_condition)
        (variant_packaging_configuration_assignment ?formulation_variant ?packaging_configuration)
        (storage_condition_associated_with_packaging ?storage_condition)
        (packaging_configuration_validated ?packaging_configuration)
        (not
          (batch_qualified ?manufacturing_batch)
        )
        (variant_qualified ?formulation_variant)
        (study_arm_unassigned ?study_arm)
      )
    :effect
      (and
        (study_arm_created ?study_arm)
        (study_arm_assigned_storage_condition ?study_arm ?storage_condition)
        (study_arm_assigned_packaging_configuration ?study_arm ?packaging_configuration)
        (study_arm_storage_verified ?study_arm)
        (not
          (study_arm_unassigned ?study_arm)
        )
      )
  )
  (:action assemble_study_arm_with_storage_validation
    :parameters (?manufacturing_batch - manufacturing_batch ?formulation_variant - formulation_variant ?storage_condition - storage_condition ?packaging_configuration - packaging_configuration ?study_arm - study_arm)
    :precondition
      (and
        (batch_ready_for_assembly ?manufacturing_batch)
        (variant_ready_for_assembly ?formulation_variant)
        (batch_storage_condition_compatibility ?manufacturing_batch ?storage_condition)
        (variant_packaging_configuration_assignment ?formulation_variant ?packaging_configuration)
        (storage_condition_validated ?storage_condition)
        (packaging_configuration_associated ?packaging_configuration)
        (batch_qualified ?manufacturing_batch)
        (not
          (variant_qualified ?formulation_variant)
        )
        (study_arm_unassigned ?study_arm)
      )
    :effect
      (and
        (study_arm_created ?study_arm)
        (study_arm_assigned_storage_condition ?study_arm ?storage_condition)
        (study_arm_assigned_packaging_configuration ?study_arm ?packaging_configuration)
        (study_arm_packaging_verified ?study_arm)
        (not
          (study_arm_unassigned ?study_arm)
        )
      )
  )
  (:action assemble_study_arm_full_validation
    :parameters (?manufacturing_batch - manufacturing_batch ?formulation_variant - formulation_variant ?storage_condition - storage_condition ?packaging_configuration - packaging_configuration ?study_arm - study_arm)
    :precondition
      (and
        (batch_ready_for_assembly ?manufacturing_batch)
        (variant_ready_for_assembly ?formulation_variant)
        (batch_storage_condition_compatibility ?manufacturing_batch ?storage_condition)
        (variant_packaging_configuration_assignment ?formulation_variant ?packaging_configuration)
        (storage_condition_associated_with_packaging ?storage_condition)
        (packaging_configuration_associated ?packaging_configuration)
        (not
          (batch_qualified ?manufacturing_batch)
        )
        (not
          (variant_qualified ?formulation_variant)
        )
        (study_arm_unassigned ?study_arm)
      )
    :effect
      (and
        (study_arm_created ?study_arm)
        (study_arm_assigned_storage_condition ?study_arm ?storage_condition)
        (study_arm_assigned_packaging_configuration ?study_arm ?packaging_configuration)
        (study_arm_storage_verified ?study_arm)
        (study_arm_packaging_verified ?study_arm)
        (not
          (study_arm_unassigned ?study_arm)
        )
      )
  )
  (:action activate_study_arm
    :parameters (?study_arm - study_arm ?manufacturing_batch - manufacturing_batch ?analytical_method - analytical_method)
    :precondition
      (and
        (study_arm_created ?study_arm)
        (batch_ready_for_assembly ?manufacturing_batch)
        (assigned_method_for_development_item ?manufacturing_batch ?analytical_method)
        (not
          (study_arm_samples_allocated ?study_arm)
        )
      )
    :effect (study_arm_samples_allocated ?study_arm)
  )
  (:action allocate_sample_set_to_product_candidate_arm
    :parameters (?product_candidate - product_candidate ?sample_set - sample_set ?study_arm - study_arm)
    :precondition
      (and
        (development_item_ready_for_review ?product_candidate)
        (product_candidate_assigned_to_study_arm ?product_candidate ?study_arm)
        (product_candidate_has_sample_set ?product_candidate ?sample_set)
        (sample_set_available ?sample_set)
        (study_arm_created ?study_arm)
        (study_arm_samples_allocated ?study_arm)
        (not
          (sample_set_reserved ?sample_set)
        )
      )
    :effect
      (and
        (sample_set_reserved ?sample_set)
        (sample_set_assigned_to_study_arm ?sample_set ?study_arm)
        (not
          (sample_set_available ?sample_set)
        )
      )
  )
  (:action activate_sample_set_for_product_candidate_arm
    :parameters (?product_candidate - product_candidate ?sample_set - sample_set ?study_arm - study_arm ?analytical_method - analytical_method)
    :precondition
      (and
        (development_item_ready_for_review ?product_candidate)
        (product_candidate_has_sample_set ?product_candidate ?sample_set)
        (sample_set_reserved ?sample_set)
        (sample_set_assigned_to_study_arm ?sample_set ?study_arm)
        (assigned_method_for_development_item ?product_candidate ?analytical_method)
        (not
          (study_arm_storage_verified ?study_arm)
        )
        (not
          (product_candidate_qualification_flag ?product_candidate)
        )
      )
    :effect (product_candidate_qualification_flag ?product_candidate)
  )
  (:action attach_regulatory_document_to_product_candidate
    :parameters (?product_candidate - product_candidate ?regulatory_document_type - regulatory_document_type)
    :precondition
      (and
        (development_item_ready_for_review ?product_candidate)
        (regulatory_document_type_available ?regulatory_document_type)
        (not
          (regulatory_document_attached_to_product_candidate ?product_candidate)
        )
      )
    :effect
      (and
        (regulatory_document_attached_to_product_candidate ?product_candidate)
        (product_candidate_regulatory_document_link ?product_candidate ?regulatory_document_type)
        (not
          (regulatory_document_type_available ?regulatory_document_type)
        )
      )
  )
  (:action verify_regulatory_document_and_enable_product_candidate
    :parameters (?product_candidate - product_candidate ?sample_set - sample_set ?study_arm - study_arm ?analytical_method - analytical_method ?regulatory_document_type - regulatory_document_type)
    :precondition
      (and
        (development_item_ready_for_review ?product_candidate)
        (product_candidate_has_sample_set ?product_candidate ?sample_set)
        (sample_set_reserved ?sample_set)
        (sample_set_assigned_to_study_arm ?sample_set ?study_arm)
        (assigned_method_for_development_item ?product_candidate ?analytical_method)
        (study_arm_storage_verified ?study_arm)
        (regulatory_document_attached_to_product_candidate ?product_candidate)
        (product_candidate_regulatory_document_link ?product_candidate ?regulatory_document_type)
        (not
          (product_candidate_qualification_flag ?product_candidate)
        )
      )
    :effect
      (and
        (product_candidate_qualification_flag ?product_candidate)
        (regulatory_document_verified_for_product_candidate ?product_candidate)
