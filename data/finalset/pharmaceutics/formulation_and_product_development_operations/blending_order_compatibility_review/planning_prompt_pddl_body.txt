  (:action register_formulation_candidate
    :parameters (?formulation_candidate - entity_instance)
    :precondition
      (and
        (not
          (entity_registered ?formulation_candidate)
        )
        (not
          (entity_compatibility_confirmed ?formulation_candidate)
        )
      )
    :effect (entity_registered ?formulation_candidate)
  )
  (:action assign_material_source_to_candidate
    :parameters (?formulation_candidate - entity_instance ?material_source - material_source)
    :precondition
      (and
        (entity_registered ?formulation_candidate)
        (not
          (entity_material_assigned ?formulation_candidate)
        )
        (material_source_available ?material_source)
      )
    :effect
      (and
        (entity_material_assigned ?formulation_candidate)
        (entity_material_source_association ?formulation_candidate ?material_source)
        (not
          (material_source_available ?material_source)
        )
      )
  )
  (:action assign_analytical_method_to_candidate
    :parameters (?formulation_candidate - entity_instance ?analytical_method - analytical_method)
    :precondition
      (and
        (entity_registered ?formulation_candidate)
        (entity_material_assigned ?formulation_candidate)
        (analytical_method_available ?analytical_method)
      )
    :effect
      (and
        (entity_method_assigned ?formulation_candidate ?analytical_method)
        (not
          (analytical_method_available ?analytical_method)
        )
      )
  )
  (:action finalize_candidate_onboarding
    :parameters (?formulation_candidate - entity_instance ?analytical_method - analytical_method)
    :precondition
      (and
        (entity_registered ?formulation_candidate)
        (entity_material_assigned ?formulation_candidate)
        (entity_method_assigned ?formulation_candidate ?analytical_method)
        (not
          (entity_onboarded ?formulation_candidate)
        )
      )
    :effect (entity_onboarded ?formulation_candidate)
  )
  (:action release_analytical_method
    :parameters (?formulation_candidate - entity_instance ?analytical_method - analytical_method)
    :precondition
      (and
        (entity_method_assigned ?formulation_candidate ?analytical_method)
      )
    :effect
      (and
        (analytical_method_available ?analytical_method)
        (not
          (entity_method_assigned ?formulation_candidate ?analytical_method)
        )
      )
  )
  (:action assign_equipment_to_candidate
    :parameters (?formulation_candidate - entity_instance ?equipment_unit - equipment_unit)

      (and
        (entity_onboarded ?formulation_candidate)
        (equipment_available ?equipment_unit)

        (entity_equipment_assigned ?formulation_candidate ?equipment_unit)

        )
      )
  )

        (case_has_risk_assessment ?compatibility_review_case ?risk_assessment)

    :parameters (?blend_sequence_a - blend_sequence_a ?blend_order_profile_a - blend_order_profile_a ?analytical_method - analytical_method)

      (and
        (entity_onboarded ?blend_sequence_a)
        (entity_method_assigned ?blend_sequence_a ?analytical_method)

        (not
          (profile_a_prepared ?blend_order_profile_a)
        )

          (profile_a_has_assigned_sample ?blend_order_profile_a)

  )
  (:action allocate_equipment_and_mark_sequence_a_ready
    :parameters (?blend_sequence_a - blend_sequence_a ?blend_order_profile_a - blend_order_profile_a ?equipment_unit - equipment_unit)
    :precondition
      (and
        (entity_onboarded ?blend_sequence_a)
        (entity_equipment_assigned ?blend_sequence_a ?equipment_unit)
        (sequence_a_has_profile ?blend_sequence_a ?blend_order_profile_a)
        (profile_a_prepared ?blend_order_profile_a)
        (not
          (sequence_a_samples_allocated ?blend_sequence_a)

      (and
        (sequence_a_samples_allocated ?blend_sequence_a)
        (sequence_a_ready ?blend_sequence_a)
      )

          (sequence_a_samples_allocated ?blend_sequence_a)

          (sequence_a_has_sample ?blend_sequence_a ?sample)

  (:action assemble_experiment_batch_standard

    :precondition
      (and
        (sequence_a_samples_allocated ?blend_sequence_a)

    :effect
      (and
        (experiment_batch_prepared ?experiment_batch)

        (batch_includes_profile_b ?experiment_batch ?blend_order_profile_b)
        (not
          (experiment_batch_available ?experiment_batch)
        )

      (and
        (experiment_batch_prepared ?experiment_batch)

          (stability_condition_available ?stability_condition)

        (stability_condition_applied_to_batch ?stability_condition ?experiment_batch)

        (not
          (batch_requires_profile_a_instrument ?experiment_batch)
        )
        (not
          (case_has_initial_results ?compatibility_review_case)
        )
      )
    :effect (case_has_initial_results ?compatibility_review_case)
  )

    :precondition
      (and
        (entity_onboarded ?compatibility_review_case)
