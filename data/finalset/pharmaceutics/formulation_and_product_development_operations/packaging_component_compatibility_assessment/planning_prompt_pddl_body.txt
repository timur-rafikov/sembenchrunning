  (:action register_packaging_component_candidate
    :parameters (?packaging_component_candidate - packaging_component_candidate)
    :precondition
      (and
        (not
          (registered ?packaging_component_candidate)

        (not
          (compatibility_approval_granted ?packaging_component_candidate)

      (and
        (registered ?packaging_component_candidate)
        (not
          (sample_allocated ?packaging_component_candidate)

        (material_sample_available ?material_sample)
      )
    :effect
      (and
        (sample_allocated ?packaging_component_candidate)
        (assigned_material_sample ?packaging_component_candidate ?material_sample)
        (not
          (material_sample_available ?material_sample)

      )
