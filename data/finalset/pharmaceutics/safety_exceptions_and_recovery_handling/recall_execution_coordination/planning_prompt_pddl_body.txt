## Planning domain (PDDL)

(define (domain recall_execution_coordination)
  (:requirements :strips :typing :negative-preconditions)
  (:types operational_resource - object product_asset - object recall_artifact - object asset - object recall_entity - asset recall_agent - operational_resource laboratory_test - operational_resource site_contact - operational_resource regulatory_clearance - operational_resource operational_checklist - operational_resource notification_recipient_group - operational_resource logistics_provider - operational_resource quality_subject_matter_expert - operational_resource mitigation_option - product_asset inspection_sample - product_asset regulatory_agency - product_asset manufacturing_lot - recall_artifact distribution_lot - recall_artifact recall_notice - recall_artifact site - recall_entity response_unit - recall_entity manufacturing_site - site distribution_site - site response_team - response_unit)
  (:predicates
    (entity_reported ?recall_case - recall_entity)
    (entity_validated ?recall_case - recall_entity)
    (entity_agent_allocated ?recall_case - recall_entity)
    (entity_closed ?recall_case - recall_entity)
    (entity_action_completed ?recall_case - recall_entity)
    (final_notification_sent ?recall_case - recall_entity)
    (recall_agent_available ?recall_agent - recall_agent)
    (recall_agent_assigned_to_entity ?recall_case - recall_entity ?recall_agent - recall_agent)
    (lab_test_available ?laboratory_test - laboratory_test)
    (entity_assigned_test ?recall_case - recall_entity ?laboratory_test - laboratory_test)
    (site_contact_available ?site_contact - site_contact)
    (entity_assigned_site_contact ?recall_case - recall_entity ?site_contact - site_contact)
    (mitigation_option_available ?mitigation_option - mitigation_option)
    (manufacturing_site_applied_mitigation ?manufacturing_site - manufacturing_site ?mitigation_option - mitigation_option)
    (distribution_site_applied_mitigation ?distribution_site - distribution_site ?mitigation_option - mitigation_option)
    (lot_manufactured_at_site ?manufacturing_site - manufacturing_site ?manufacturing_lot - manufacturing_lot)
    (manufacturing_lot_quarantined ?manufacturing_lot - manufacturing_lot)
    (manufacturing_lot_mitigation_applied ?manufacturing_lot - manufacturing_lot)
    (manufacturing_site_verified ?manufacturing_site - manufacturing_site)
    (distribution_lot_at_site ?distribution_site - distribution_site ?distribution_lot - distribution_lot)
    (distribution_lot_quarantined ?distribution_lot - distribution_lot)
    (distribution_lot_mitigation_applied ?distribution_lot - distribution_lot)
    (distribution_site_verified ?distribution_site - distribution_site)
    (recall_notice_created ?recall_notice - recall_notice)
    (recall_notice_finalized ?recall_notice - recall_notice)
    (recall_notice_includes_manufacturing_lot ?recall_notice - recall_notice ?manufacturing_lot - manufacturing_lot)
    (recall_notice_includes_distribution_lot ?recall_notice - recall_notice ?distribution_lot - distribution_lot)
    (recall_notice_includes_manufacturing_mitigation_flag ?recall_notice - recall_notice)
    (recall_notice_includes_distribution_mitigation_flag ?recall_notice - recall_notice)
    (recall_notice_ready_for_sample_attachment ?recall_notice - recall_notice)
    (team_deployed_to_manufacturing_site ?response_team - response_team ?manufacturing_site - manufacturing_site)
    (team_deployed_to_distribution_site ?response_team - response_team ?distribution_site - distribution_site)
    (team_assigned_recall_notice ?response_team - response_team ?recall_notice - recall_notice)
    (inspection_sample_available ?inspection_sample - inspection_sample)
    (team_assigned_inspection_sample ?response_team - response_team ?inspection_sample - inspection_sample)
    (inspection_sample_collected ?inspection_sample - inspection_sample)
    (inspection_sample_attached_to_notice ?inspection_sample - inspection_sample ?recall_notice - recall_notice)
    (team_ready_for_dispatch ?response_team - response_team)
    (team_dispatched_for_verification ?response_team - response_team)
    (team_verification_complete ?response_team - response_team)
    (team_has_regulatory_clearance ?response_team - response_team)
    (team_notification_payload_prepared ?response_team - response_team)
    (team_quality_check_completed ?response_team - response_team)
    (team_logistics_and_readiness_confirmed ?response_team - response_team)
    (regulatory_agency_available ?regulatory_agency - regulatory_agency)
    (team_linked_to_regulatory_agency ?response_team - response_team ?regulatory_agency - regulatory_agency)
    (team_regulatory_acknowledged ?response_team - response_team)
    (team_initiated_site_closure_prep ?response_team - response_team)
    (team_sme_review_complete ?response_team - response_team)
    (regulatory_clearance_available ?regulatory_clearance - regulatory_clearance)
    (team_associated_with_regulatory_clearance ?response_team - response_team ?regulatory_clearance - regulatory_clearance)
    (operational_checklist_available ?operational_checklist - operational_checklist)
    (team_assigned_operational_checklist ?response_team - response_team ?operational_checklist - operational_checklist)
    (logistics_provider_available ?logistics_provider - logistics_provider)
    (team_assigned_logistics_provider ?response_team - response_team ?logistics_provider - logistics_provider)
    (quality_sme_available ?quality_subject_matter_expert - quality_subject_matter_expert)
    (team_assigned_quality_sme ?response_team - response_team ?quality_subject_matter_expert - quality_subject_matter_expert)
    (notification_recipient_group_available ?notification_recipient_group - notification_recipient_group)
    (entity_notification_sent_to_group ?recall_case - recall_entity ?notification_recipient_group - notification_recipient_group)
    (manufacturing_site_quarantine_active ?manufacturing_site - manufacturing_site)
    (distribution_site_quarantine_active ?distribution_site - distribution_site)
    (team_close_ready ?response_team - response_team)
  )
  (:action report_recall_entity
    :parameters (?recall_case - recall_entity)
    :precondition
      (and
        (not
          (entity_reported ?recall_case)
        )
        (not
          (entity_closed ?recall_case)
        )
      )
    :effect (entity_reported ?recall_case)
  )
  (:action assign_recall_agent_to_entity
    :parameters (?recall_case - recall_entity ?recall_agent - recall_agent)
    :precondition
      (and
        (entity_reported ?recall_case)
        (not
          (entity_agent_allocated ?recall_case)
        )
        (recall_agent_available ?recall_agent)
      )
    :effect
      (and
        (entity_agent_allocated ?recall_case)
        (recall_agent_assigned_to_entity ?recall_case ?recall_agent)
        (not
          (recall_agent_available ?recall_agent)
        )
      )
  )
  (:action assign_lab_test_to_entity
    :parameters (?recall_case - recall_entity ?laboratory_test - laboratory_test)
    :precondition
      (and
        (entity_reported ?recall_case)
        (entity_agent_allocated ?recall_case)
        (lab_test_available ?laboratory_test)
      )
    :effect
      (and
        (entity_assigned_test ?recall_case ?laboratory_test)
        (not
          (lab_test_available ?laboratory_test)
        )
      )
  )
  (:action validate_recall_entity
    :parameters (?recall_case - recall_entity ?laboratory_test - laboratory_test)
    :precondition
      (and
        (entity_reported ?recall_case)
        (entity_agent_allocated ?recall_case)
        (entity_assigned_test ?recall_case ?laboratory_test)
        (not
          (entity_validated ?recall_case)
        )
      )
    :effect (entity_validated ?recall_case)
  )
  (:action release_lab_test_from_entity
    :parameters (?recall_case - recall_entity ?laboratory_test - laboratory_test)
    :precondition
      (and
        (entity_assigned_test ?recall_case ?laboratory_test)
      )
    :effect
      (and
        (lab_test_available ?laboratory_test)
        (not
          (entity_assigned_test ?recall_case ?laboratory_test)
        )
      )
  )
  (:action assign_site_contact_to_entity
    :parameters (?recall_case - recall_entity ?site_contact - site_contact)
    :precondition
      (and
        (entity_validated ?recall_case)
        (site_contact_available ?site_contact)
      )
    :effect
      (and
        (entity_assigned_site_contact ?recall_case ?site_contact)
        (not
          (site_contact_available ?site_contact)
        )
      )
  )
  (:action release_site_contact_from_entity
    :parameters (?recall_case - recall_entity ?site_contact - site_contact)
    :precondition
      (and
        (entity_assigned_site_contact ?recall_case ?site_contact)
      )
    :effect
      (and
        (site_contact_available ?site_contact)
        (not
          (entity_assigned_site_contact ?recall_case ?site_contact)
        )
      )
  )
  (:action assign_logistics_provider_to_team
    :parameters (?response_team - response_team ?logistics_provider - logistics_provider)
    :precondition
      (and
        (entity_validated ?response_team)
        (logistics_provider_available ?logistics_provider)
      )
    :effect
      (and
        (team_assigned_logistics_provider ?response_team ?logistics_provider)
        (not
          (logistics_provider_available ?logistics_provider)
        )
      )
  )
  (:action release_logistics_provider_from_team
    :parameters (?response_team - response_team ?logistics_provider - logistics_provider)
    :precondition
      (and
        (team_assigned_logistics_provider ?response_team ?logistics_provider)
      )
    :effect
      (and
        (logistics_provider_available ?logistics_provider)
        (not
          (team_assigned_logistics_provider ?response_team ?logistics_provider)
        )
      )
  )
  (:action assign_quality_sme_to_team
    :parameters (?response_team - response_team ?quality_subject_matter_expert - quality_subject_matter_expert)
    :precondition
      (and
        (entity_validated ?response_team)
        (quality_sme_available ?quality_subject_matter_expert)
      )
    :effect
      (and
        (team_assigned_quality_sme ?response_team ?quality_subject_matter_expert)
        (not
          (quality_sme_available ?quality_subject_matter_expert)
        )
      )
  )
  (:action release_quality_sme_from_team
    :parameters (?response_team - response_team ?quality_subject_matter_expert - quality_subject_matter_expert)
    :precondition
      (and
        (team_assigned_quality_sme ?response_team ?quality_subject_matter_expert)
      )
    :effect
      (and
        (quality_sme_available ?quality_subject_matter_expert)
        (not
          (team_assigned_quality_sme ?response_team ?quality_subject_matter_expert)
        )
      )
  )
  (:action flag_manufacturing_lot_for_quarantine
    :parameters (?manufacturing_site - manufacturing_site ?manufacturing_lot - manufacturing_lot ?laboratory_test - laboratory_test)
    :precondition
      (and
        (entity_validated ?manufacturing_site)
        (entity_assigned_test ?manufacturing_site ?laboratory_test)
        (lot_manufactured_at_site ?manufacturing_site ?manufacturing_lot)
        (not
          (manufacturing_lot_quarantined ?manufacturing_lot)
        )
        (not
          (manufacturing_lot_mitigation_applied ?manufacturing_lot)
        )
      )
    :effect (manufacturing_lot_quarantined ?manufacturing_lot)
  )
  (:action confirm_manufacturing_site_quarantine
    :parameters (?manufacturing_site - manufacturing_site ?manufacturing_lot - manufacturing_lot ?site_contact - site_contact)
    :precondition
      (and
        (entity_validated ?manufacturing_site)
        (entity_assigned_site_contact ?manufacturing_site ?site_contact)
        (lot_manufactured_at_site ?manufacturing_site ?manufacturing_lot)
        (manufacturing_lot_quarantined ?manufacturing_lot)
        (not
          (manufacturing_site_quarantine_active ?manufacturing_site)
        )
      )
    :effect
      (and
        (manufacturing_site_quarantine_active ?manufacturing_site)
        (manufacturing_site_verified ?manufacturing_site)
      )
  )
  (:action apply_mitigation_option_at_manufacturing_site
    :parameters (?manufacturing_site - manufacturing_site ?manufacturing_lot - manufacturing_lot ?mitigation_option - mitigation_option)
    :precondition
      (and
        (entity_validated ?manufacturing_site)
        (lot_manufactured_at_site ?manufacturing_site ?manufacturing_lot)
        (mitigation_option_available ?mitigation_option)
        (not
          (manufacturing_site_quarantine_active ?manufacturing_site)
        )
      )
    :effect
      (and
        (manufacturing_lot_mitigation_applied ?manufacturing_lot)
        (manufacturing_site_quarantine_active ?manufacturing_site)
        (manufacturing_site_applied_mitigation ?manufacturing_site ?mitigation_option)
        (not
          (mitigation_option_available ?mitigation_option)
        )
      )
  )
  (:action finalize_manufacturing_mitigation
    :parameters (?manufacturing_site - manufacturing_site ?manufacturing_lot - manufacturing_lot ?laboratory_test - laboratory_test ?mitigation_option - mitigation_option)
    :precondition
      (and
        (entity_validated ?manufacturing_site)
        (entity_assigned_test ?manufacturing_site ?laboratory_test)
        (lot_manufactured_at_site ?manufacturing_site ?manufacturing_lot)
        (manufacturing_lot_mitigation_applied ?manufacturing_lot)
        (manufacturing_site_applied_mitigation ?manufacturing_site ?mitigation_option)
        (not
          (manufacturing_site_verified ?manufacturing_site)
        )
      )
    :effect
      (and
        (manufacturing_lot_quarantined ?manufacturing_lot)
        (manufacturing_site_verified ?manufacturing_site)
        (mitigation_option_available ?mitigation_option)
        (not
          (manufacturing_site_applied_mitigation ?manufacturing_site ?mitigation_option)
        )
      )
  )
  (:action flag_distribution_lot_for_quarantine
    :parameters (?distribution_site - distribution_site ?distribution_lot - distribution_lot ?laboratory_test - laboratory_test)
    :precondition
      (and
        (entity_validated ?distribution_site)
        (entity_assigned_test ?distribution_site ?laboratory_test)
        (distribution_lot_at_site ?distribution_site ?distribution_lot)
        (not
          (distribution_lot_quarantined ?distribution_lot)
        )
        (not
          (distribution_lot_mitigation_applied ?distribution_lot)
        )
      )
    :effect (distribution_lot_quarantined ?distribution_lot)
  )
  (:action confirm_distribution_site_quarantine
    :parameters (?distribution_site - distribution_site ?distribution_lot - distribution_lot ?site_contact - site_contact)
    :precondition
      (and
        (entity_validated ?distribution_site)

        (distribution_lot_at_site ?distribution_site ?distribution_lot)

        )
      )
    :effect
      (and
        (distribution_site_quarantine_active ?distribution_site)
        (distribution_site_verified ?distribution_site)
      )
  )
  (:action apply_mitigation_option_at_distribution_site
    :parameters (?distribution_site - distribution_site ?distribution_lot - distribution_lot ?mitigation_option - mitigation_option)
    :precondition
      (and
        (entity_validated ?distribution_site)
        (distribution_lot_at_site ?distribution_site ?distribution_lot)
        (mitigation_option_available ?mitigation_option)
        (not
          (distribution_site_quarantine_active ?distribution_site)
        )
      )
    :effect
      (and
        (distribution_lot_mitigation_applied ?distribution_lot)
        (distribution_site_quarantine_active ?distribution_site)
        (distribution_site_applied_mitigation ?distribution_site ?mitigation_option)
        (not
          (mitigation_option_available ?mitigation_option)
        )
      )
  )
  (:action finalize_distribution_mitigation
    :parameters (?distribution_site - distribution_site ?distribution_lot - distribution_lot ?laboratory_test - laboratory_test ?mitigation_option - mitigation_option)
    :precondition
      (and
        (entity_validated ?distribution_site)
        (entity_assigned_test ?distribution_site ?laboratory_test)
        (distribution_lot_at_site ?distribution_site ?distribution_lot)
        (distribution_lot_mitigation_applied ?distribution_lot)
        (distribution_site_applied_mitigation ?distribution_site ?mitigation_option)
        (not
          (distribution_site_verified ?distribution_site)
        )
      )
    :effect
      (and
        (distribution_lot_quarantined ?distribution_lot)
        (distribution_site_verified ?distribution_site)
        (mitigation_option_available ?mitigation_option)
        (not
          (distribution_site_applied_mitigation ?distribution_site ?mitigation_option)
        )
      )
  )
  (:action finalize_recall_notice_binding
    :parameters (?manufacturing_site - manufacturing_site ?distribution_site - distribution_site ?manufacturing_lot - manufacturing_lot ?distribution_lot - distribution_lot ?recall_notice - recall_notice)
    :precondition
      (and
        (manufacturing_site_quarantine_active ?manufacturing_site)
        (distribution_site_quarantine_active ?distribution_site)
        (lot_manufactured_at_site ?manufacturing_site ?manufacturing_lot)
        (distribution_lot_at_site ?distribution_site ?distribution_lot)
        (manufacturing_lot_quarantined ?manufacturing_lot)
        (distribution_lot_quarantined ?distribution_lot)
        (manufacturing_site_verified ?manufacturing_site)
        (distribution_site_verified ?distribution_site)
        (recall_notice_created ?recall_notice)
      )
    :effect
      (and
        (recall_notice_finalized ?recall_notice)
        (recall_notice_includes_manufacturing_lot ?recall_notice ?manufacturing_lot)
        (recall_notice_includes_distribution_lot ?recall_notice ?distribution_lot)
        (not
          (recall_notice_created ?recall_notice)
        )
      )
  )
  (:action finalize_recall_notice_binding_with_manufacturing_mitigation
    :parameters (?manufacturing_site - manufacturing_site ?distribution_site - distribution_site ?manufacturing_lot - manufacturing_lot ?distribution_lot - distribution_lot ?recall_notice - recall_notice)
    :precondition
      (and
        (manufacturing_site_quarantine_active ?manufacturing_site)
        (distribution_site_quarantine_active ?distribution_site)
        (lot_manufactured_at_site ?manufacturing_site ?manufacturing_lot)
        (distribution_lot_at_site ?distribution_site ?distribution_lot)
        (manufacturing_lot_mitigation_applied ?manufacturing_lot)
        (distribution_lot_quarantined ?distribution_lot)
        (not
          (manufacturing_site_verified ?manufacturing_site)
        )
        (distribution_site_verified ?distribution_site)
        (recall_notice_created ?recall_notice)
      )
    :effect
      (and
        (recall_notice_finalized ?recall_notice)
        (recall_notice_includes_manufacturing_lot ?recall_notice ?manufacturing_lot)
        (recall_notice_includes_distribution_lot ?recall_notice ?distribution_lot)
        (recall_notice_includes_manufacturing_mitigation_flag ?recall_notice)
        (not
          (recall_notice_created ?recall_notice)
        )
      )
  )
  (:action finalize_recall_notice_binding_with_distribution_mitigation
    :parameters (?manufacturing_site - manufacturing_site ?distribution_site - distribution_site ?manufacturing_lot - manufacturing_lot ?distribution_lot - distribution_lot ?recall_notice - recall_notice)
    :precondition
      (and
        (manufacturing_site_quarantine_active ?manufacturing_site)
        (distribution_site_quarantine_active ?distribution_site)
        (lot_manufactured_at_site ?manufacturing_site ?manufacturing_lot)
        (distribution_lot_at_site ?distribution_site ?distribution_lot)
        (manufacturing_lot_quarantined ?manufacturing_lot)
        (distribution_lot_mitigation_applied ?distribution_lot)
        (manufacturing_site_verified ?manufacturing_site)
        (not
          (distribution_site_verified ?distribution_site)
        )
        (recall_notice_created ?recall_notice)
      )
    :effect
      (and
        (recall_notice_finalized ?recall_notice)
        (recall_notice_includes_manufacturing_lot ?recall_notice ?manufacturing_lot)
        (recall_notice_includes_distribution_lot ?recall_notice ?distribution_lot)
        (recall_notice_includes_distribution_mitigation_flag ?recall_notice)
        (not
          (recall_notice_created ?recall_notice)
        )
      )
  )
  (:action finalize_recall_notice_binding_with_both_mitigations
    :parameters (?manufacturing_site - manufacturing_site ?distribution_site - distribution_site ?manufacturing_lot - manufacturing_lot ?distribution_lot - distribution_lot ?recall_notice - recall_notice)
    :precondition
      (and
        (manufacturing_site_quarantine_active ?manufacturing_site)
        (distribution_site_quarantine_active ?distribution_site)
        (lot_manufactured_at_site ?manufacturing_site ?manufacturing_lot)
        (distribution_lot_at_site ?distribution_site ?distribution_lot)
        (manufacturing_lot_mitigation_applied ?manufacturing_lot)
        (distribution_lot_mitigation_applied ?distribution_lot)
        (not
          (manufacturing_site_verified ?manufacturing_site)
        )
        (not
          (distribution_site_verified ?distribution_site)
        )
        (recall_notice_created ?recall_notice)
      )
    :effect
      (and
        (recall_notice_finalized ?recall_notice)
        (recall_notice_includes_manufacturing_lot ?recall_notice ?manufacturing_lot)
        (recall_notice_includes_distribution_lot ?recall_notice ?distribution_lot)
        (recall_notice_includes_manufacturing_mitigation_flag ?recall_notice)
        (recall_notice_includes_distribution_mitigation_flag ?recall_notice)
        (not
          (recall_notice_created ?recall_notice)
        )
      )
  )
  (:action verify_recall_notice_for_sample_attachment
    :parameters (?recall_notice - recall_notice ?manufacturing_site - manufacturing_site ?laboratory_test - laboratory_test)
    :precondition
      (and
        (recall_notice_finalized ?recall_notice)
        (manufacturing_site_quarantine_active ?manufacturing_site)
        (entity_assigned_test ?manufacturing_site ?laboratory_test)
        (not
          (recall_notice_ready_for_sample_attachment ?recall_notice)
        )
      )
    :effect (recall_notice_ready_for_sample_attachment ?recall_notice)
  )
  (:action attach_inspection_sample_to_notice
    :parameters (?response_team - response_team ?inspection_sample - inspection_sample ?recall_notice - recall_notice)
    :precondition
      (and
        (entity_validated ?response_team)
        (team_assigned_recall_notice ?response_team ?recall_notice)
        (team_assigned_inspection_sample ?response_team ?inspection_sample)
        (inspection_sample_available ?inspection_sample)
        (recall_notice_finalized ?recall_notice)
        (recall_notice_ready_for_sample_attachment ?recall_notice)
        (not
          (inspection_sample_collected ?inspection_sample)
        )
      )
    :effect
      (and
        (inspection_sample_collected ?inspection_sample)
        (inspection_sample_attached_to_notice ?inspection_sample ?recall_notice)
        (not
          (inspection_sample_available ?inspection_sample)
        )
      )
  )
  (:action authorize_team_for_sample_dispatch
    :parameters (?response_team - response_team ?inspection_sample - inspection_sample ?recall_notice - recall_notice ?laboratory_test - laboratory_test)
    :precondition
      (and
        (entity_validated ?response_team)
        (team_assigned_inspection_sample ?response_team ?inspection_sample)
        (inspection_sample_collected ?inspection_sample)
        (inspection_sample_attached_to_notice ?inspection_sample ?recall_notice)
        (entity_assigned_test ?response_team ?laboratory_test)
        (not
          (recall_notice_includes_manufacturing_mitigation_flag ?recall_notice)
        )
        (not
          (team_ready_for_dispatch ?response_team)
        )
      )
    :effect (team_ready_for_dispatch ?response_team)
  )
  (:action attach_regulatory_clearance_to_team
    :parameters (?response_team - response_team ?regulatory_clearance - regulatory_clearance)
    :precondition
      (and
        (entity_validated ?response_team)
        (regulatory_clearance_available ?regulatory_clearance)
        (not
          (team_has_regulatory_clearance ?response_team)
        )
      )
    :effect
      (and
        (team_has_regulatory_clearance ?response_team)
        (team_associated_with_regulatory_clearance ?response_team ?regulatory_clearance)
        (not
          (regulatory_clearance_available ?regulatory_clearance)
        )
      )
  )
  (:action prepare_team_notification_payload
    :parameters (?response_team - response_team ?inspection_sample - inspection_sample ?recall_notice - recall_notice ?laboratory_test - laboratory_test ?regulatory_clearance - regulatory_clearance)
    :precondition
      (and
        (entity_validated ?response_team)
        (team_assigned_inspection_sample ?response_team ?inspection_sample)
        (inspection_sample_collected ?inspection_sample)
        (inspection_sample_attached_to_notice ?inspection_sample ?recall_notice)
        (entity_assigned_test ?response_team ?laboratory_test)
        (recall_notice_includes_manufacturing_mitigation_flag ?recall_notice)
        (team_has_regulatory_clearance ?response_team)
        (team_associated_with_regulatory_clearance ?response_team ?regulatory_clearance)
        (not
          (team_ready_for_dispatch ?response_team)
        )
      )
    :effect
      (and
        (team_ready_for_dispatch ?response_team)
        (team_notification_payload_prepared ?response_team)
      )
