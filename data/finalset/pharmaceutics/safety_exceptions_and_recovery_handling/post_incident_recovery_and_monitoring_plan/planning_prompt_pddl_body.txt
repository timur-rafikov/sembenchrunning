## Planning domain (PDDL)

(define (domain pharm_post_incident_recovery_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types asset_category - object supply_element - object operational_unit - object node_supertype - object supply_chain_node - node_supertype logistics_asset - asset_category sample_test - asset_category site_agent - asset_category communication_channel - asset_category technical_auditor - asset_category regulatory_document - asset_category remediation_method - asset_category independent_lab_result - asset_category alternative_supplier - supply_element inspection_protocol - supply_element surveillance_feed - supply_element supply_batch - operational_unit distribution_route - operational_unit recovery_plan - operational_unit production_node - supply_chain_node distribution_node - supply_chain_node manufacturing_site - production_node distribution_center - production_node response_team - distribution_node)
  (:predicates
    (incident_identified ?supply_chain_node - supply_chain_node)
    (entity_triage_completed ?supply_chain_node - supply_chain_node)
    (asset_allocated ?supply_chain_node - supply_chain_node)
    (node_operational ?supply_chain_node - supply_chain_node)
    (entity_release_ready ?supply_chain_node - supply_chain_node)
    (regulatory_submitted ?supply_chain_node - supply_chain_node)
    (asset_available ?logistics_asset - logistics_asset)
    (asset_assigned_to_entity ?supply_chain_node - supply_chain_node ?logistics_asset - logistics_asset)
    (sample_available ?sample_test - sample_test)
    (sample_linked_to_entity ?supply_chain_node - supply_chain_node ?sample_test - sample_test)
    (agent_available ?site_agent - site_agent)
    (agent_assigned_to_entity ?supply_chain_node - supply_chain_node ?site_agent - site_agent)
    (alternate_supplier_available ?alternative_supplier - alternative_supplier)
    (manufacturing_site_has_alternate_supplier ?manufacturing_site - manufacturing_site ?alternative_supplier - alternative_supplier)
    (distribution_center_has_alternate_supplier ?distribution_center - distribution_center ?alternative_supplier - alternative_supplier)
    (manufacturing_site_contains_batch ?manufacturing_site - manufacturing_site ?supply_batch - supply_batch)
    (batch_quarantined ?supply_batch - supply_batch)
    (batch_marked_for_alternate_supply ?supply_batch - supply_batch)
    (site_quarantine_confirmed ?manufacturing_site - manufacturing_site)
    (distribution_center_has_route ?distribution_center - distribution_center ?distribution_route - distribution_route)
    (route_quarantined ?distribution_route - distribution_route)
    (route_marked_for_alternate_supply ?distribution_route - distribution_route)
    (distribution_quarantine_confirmed ?distribution_center - distribution_center)
    (recovery_plan_available ?recovery_plan - recovery_plan)
    (recovery_plan_instantiated ?recovery_plan - recovery_plan)
    (recovery_plan_covers_batch ?recovery_plan - recovery_plan ?supply_batch - supply_batch)
    (recovery_plan_covers_route ?recovery_plan - recovery_plan ?distribution_route - distribution_route)
    (recovery_plan_flag_stage_one ?recovery_plan - recovery_plan)
    (recovery_plan_flag_stage_two ?recovery_plan - recovery_plan)
    (recovery_plan_testing_scheduled ?recovery_plan - recovery_plan)
    (team_assigned_to_manufacturing_site ?response_team - response_team ?manufacturing_site - manufacturing_site)
    (team_assigned_to_distribution_center ?response_team - response_team ?distribution_center - distribution_center)
    (team_assigned_to_recovery_plan ?response_team - response_team ?recovery_plan - recovery_plan)
    (inspection_protocol_available ?inspection_protocol - inspection_protocol)
    (team_uses_inspection_protocol ?response_team - response_team ?inspection_protocol - inspection_protocol)
    (inspection_protocol_in_use ?inspection_protocol - inspection_protocol)
    (protocol_applied_to_recovery_plan ?inspection_protocol - inspection_protocol ?recovery_plan - recovery_plan)
    (team_validation_ready ?response_team - response_team)
    (team_testing_completed ?response_team - response_team)
    (team_verification_complete ?response_team - response_team)
    (team_has_comm_channel ?response_team - response_team)
    (team_comm_ready ?response_team - response_team)
    (team_audit_cleared ?response_team - response_team)
    (team_internal_approval ?response_team - response_team)
    (surveillance_feed_available ?surveillance_feed - surveillance_feed)
    (team_monitors_surveillance_feed ?response_team - response_team ?surveillance_feed - surveillance_feed)
    (team_received_surveillance_alert ?response_team - response_team)
    (team_in_field_deployment ?response_team - response_team)
    (team_field_action_verified ?response_team - response_team)
    (comm_channel_available ?communication_channel - communication_channel)
    (team_assigned_comm_channel ?response_team - response_team ?communication_channel - communication_channel)
    (auditor_available ?technical_auditor - technical_auditor)
    (team_engaged_technical_auditor ?response_team - response_team ?technical_auditor - technical_auditor)
    (remediation_method_available ?remediation_method - remediation_method)
    (remediation_method_assigned_to_team ?response_team - response_team ?remediation_method - remediation_method)
    (independent_lab_result_available ?independent_lab_result - independent_lab_result)
    (team_has_independent_lab_result ?response_team - response_team ?independent_lab_result - independent_lab_result)
    (regulatory_document_available ?regulatory_document - regulatory_document)
    (regulatory_document_linked_to_entity ?supply_chain_node - supply_chain_node ?regulatory_document - regulatory_document)
    (site_quarantine_active ?manufacturing_site - manufacturing_site)
    (distribution_center_quarantine_active ?distribution_center - distribution_center)
    (final_approval_documented ?response_team - response_team)
  )
  (:action identify_incident_node
    :parameters (?supply_chain_node - supply_chain_node)
    :precondition
      (and
        (not
          (incident_identified ?supply_chain_node)
        )
        (not
          (node_operational ?supply_chain_node)
        )
      )
    :effect (incident_identified ?supply_chain_node)
  )
  (:action allocate_logistics_asset_to_node
    :parameters (?supply_chain_node - supply_chain_node ?logistics_asset - logistics_asset)
    :precondition
      (and
        (incident_identified ?supply_chain_node)
        (not
          (asset_allocated ?supply_chain_node)
        )
        (asset_available ?logistics_asset)
      )
    :effect
      (and
        (asset_allocated ?supply_chain_node)
        (asset_assigned_to_entity ?supply_chain_node ?logistics_asset)
        (not
          (asset_available ?logistics_asset)
        )
      )
  )
  (:action collect_sample_from_node
    :parameters (?supply_chain_node - supply_chain_node ?sample_test - sample_test)
    :precondition
      (and
        (incident_identified ?supply_chain_node)
        (asset_allocated ?supply_chain_node)
        (sample_available ?sample_test)
      )
    :effect
      (and
        (sample_linked_to_entity ?supply_chain_node ?sample_test)
        (not
          (sample_available ?sample_test)
        )
      )
  )
  (:action complete_initial_triage
    :parameters (?supply_chain_node - supply_chain_node ?sample_test - sample_test)
    :precondition
      (and
        (incident_identified ?supply_chain_node)
        (asset_allocated ?supply_chain_node)
        (sample_linked_to_entity ?supply_chain_node ?sample_test)
        (not
          (entity_triage_completed ?supply_chain_node)
        )
      )
    :effect (entity_triage_completed ?supply_chain_node)
  )
  (:action release_sample_to_pool
    :parameters (?supply_chain_node - supply_chain_node ?sample_test - sample_test)
    :precondition
      (and
        (sample_linked_to_entity ?supply_chain_node ?sample_test)
      )
    :effect
      (and
        (sample_available ?sample_test)
        (not
          (sample_linked_to_entity ?supply_chain_node ?sample_test)
        )
      )
  )
  (:action deploy_site_agent_to_node
    :parameters (?supply_chain_node - supply_chain_node ?site_agent - site_agent)
    :precondition
      (and
        (entity_triage_completed ?supply_chain_node)
        (agent_available ?site_agent)
      )
    :effect
      (and
        (agent_assigned_to_entity ?supply_chain_node ?site_agent)
        (not
          (agent_available ?site_agent)
        )
      )
  )
  (:action withdraw_site_agent_from_node
    :parameters (?supply_chain_node - supply_chain_node ?site_agent - site_agent)
    :precondition
      (and
        (agent_assigned_to_entity ?supply_chain_node ?site_agent)
      )
    :effect
      (and
        (agent_available ?site_agent)
        (not
          (agent_assigned_to_entity ?supply_chain_node ?site_agent)
        )
      )
  )
  (:action assign_remediation_method_to_team
    :parameters (?response_team - response_team ?remediation_method - remediation_method)
    :precondition
      (and
        (entity_triage_completed ?response_team)
        (remediation_method_available ?remediation_method)
      )
    :effect
      (and
        (remediation_method_assigned_to_team ?response_team ?remediation_method)
        (not
          (remediation_method_available ?remediation_method)
        )
      )
  )
  (:action unassign_remediation_method_from_team
    :parameters (?response_team - response_team ?remediation_method - remediation_method)
    :precondition
      (and
        (remediation_method_assigned_to_team ?response_team ?remediation_method)
      )
    :effect
      (and
        (remediation_method_available ?remediation_method)
        (not
          (remediation_method_assigned_to_team ?response_team ?remediation_method)
        )
      )
  )
  (:action assign_independent_lab_result_to_team
    :parameters (?response_team - response_team ?independent_lab_result - independent_lab_result)
    :precondition
      (and
        (entity_triage_completed ?response_team)
        (independent_lab_result_available ?independent_lab_result)
      )
    :effect
      (and
        (team_has_independent_lab_result ?response_team ?independent_lab_result)
        (not
          (independent_lab_result_available ?independent_lab_result)
        )
      )
  )
  (:action unassign_independent_lab_result_from_team
    :parameters (?response_team - response_team ?independent_lab_result - independent_lab_result)
    :precondition
      (and
        (team_has_independent_lab_result ?response_team ?independent_lab_result)
      )
    :effect
      (and
        (independent_lab_result_available ?independent_lab_result)
        (not
          (team_has_independent_lab_result ?response_team ?independent_lab_result)
        )
      )
  )
  (:action quarantine_batch_at_site
    :parameters (?manufacturing_site - manufacturing_site ?supply_batch - supply_batch ?sample_test - sample_test)
    :precondition
      (and
        (entity_triage_completed ?manufacturing_site)
        (sample_linked_to_entity ?manufacturing_site ?sample_test)
        (manufacturing_site_contains_batch ?manufacturing_site ?supply_batch)
        (not
          (batch_quarantined ?supply_batch)
        )
        (not
          (batch_marked_for_alternate_supply ?supply_batch)
        )
      )
    :effect (batch_quarantined ?supply_batch)
  )
  (:action confirm_site_quarantine_with_agent
    :parameters (?manufacturing_site - manufacturing_site ?supply_batch - supply_batch ?site_agent - site_agent)
    :precondition
      (and
        (entity_triage_completed ?manufacturing_site)
        (agent_assigned_to_entity ?manufacturing_site ?site_agent)
        (manufacturing_site_contains_batch ?manufacturing_site ?supply_batch)
        (batch_quarantined ?supply_batch)
        (not
          (site_quarantine_active ?manufacturing_site)
        )
      )
    :effect
      (and
        (site_quarantine_active ?manufacturing_site)
        (site_quarantine_confirmed ?manufacturing_site)
      )
  )
  (:action assign_alternate_supplier_for_batch
    :parameters (?manufacturing_site - manufacturing_site ?supply_batch - supply_batch ?alternative_supplier - alternative_supplier)
    :precondition
      (and
        (entity_triage_completed ?manufacturing_site)
        (manufacturing_site_contains_batch ?manufacturing_site ?supply_batch)
        (alternate_supplier_available ?alternative_supplier)
        (not
          (site_quarantine_active ?manufacturing_site)
        )
      )
    :effect
      (and
        (batch_marked_for_alternate_supply ?supply_batch)
        (site_quarantine_active ?manufacturing_site)
        (manufacturing_site_has_alternate_supplier ?manufacturing_site ?alternative_supplier)
        (not
          (alternate_supplier_available ?alternative_supplier)
        )
      )
  )
  (:action process_batch_test_outcome
    :parameters (?manufacturing_site - manufacturing_site ?supply_batch - supply_batch ?sample_test - sample_test ?alternative_supplier - alternative_supplier)
    :precondition
      (and
        (entity_triage_completed ?manufacturing_site)
        (sample_linked_to_entity ?manufacturing_site ?sample_test)
        (manufacturing_site_contains_batch ?manufacturing_site ?supply_batch)
        (batch_marked_for_alternate_supply ?supply_batch)
        (manufacturing_site_has_alternate_supplier ?manufacturing_site ?alternative_supplier)
        (not
          (site_quarantine_confirmed ?manufacturing_site)
        )
      )
    :effect
      (and
        (batch_quarantined ?supply_batch)
        (site_quarantine_confirmed ?manufacturing_site)
        (alternate_supplier_available ?alternative_supplier)
        (not
          (manufacturing_site_has_alternate_supplier ?manufacturing_site ?alternative_supplier)
        )
      )
  )
  (:action quarantine_distribution_route
    :parameters (?distribution_center - distribution_center ?distribution_route - distribution_route ?sample_test - sample_test)
    :precondition
      (and
        (entity_triage_completed ?distribution_center)
        (sample_linked_to_entity ?distribution_center ?sample_test)
        (distribution_center_has_route ?distribution_center ?distribution_route)
        (not
          (route_quarantined ?distribution_route)
        )
        (not
          (route_marked_for_alternate_supply ?distribution_route)
        )
      )
    :effect (route_quarantined ?distribution_route)
  )
  (:action confirm_route_quarantine_with_agent
    :parameters (?distribution_center - distribution_center ?distribution_route - distribution_route ?site_agent - site_agent)
    :precondition
      (and
        (entity_triage_completed ?distribution_center)
        (agent_assigned_to_entity ?distribution_center ?site_agent)
        (distribution_center_has_route ?distribution_center ?distribution_route)
        (route_quarantined ?distribution_route)
        (not
          (distribution_center_quarantine_active ?distribution_center)
        )
      )
    :effect
      (and
        (distribution_center_quarantine_active ?distribution_center)
        (distribution_quarantine_confirmed ?distribution_center)
      )
  )
  (:action assign_alternate_supplier_for_route
    :parameters (?distribution_center - distribution_center ?distribution_route - distribution_route ?alternative_supplier - alternative_supplier)
    :precondition
      (and
        (entity_triage_completed ?distribution_center)
        (distribution_center_has_route ?distribution_center ?distribution_route)
        (alternate_supplier_available ?alternative_supplier)
        (not
          (distribution_center_quarantine_active ?distribution_center)
        )
      )
    :effect
      (and
        (route_marked_for_alternate_supply ?distribution_route)
        (distribution_center_quarantine_active ?distribution_center)
        (distribution_center_has_alternate_supplier ?distribution_center ?alternative_supplier)
        (not
          (alternate_supplier_available ?alternative_supplier)
        )
      )
  )
  (:action process_route_test_outcome
    :parameters (?distribution_center - distribution_center ?distribution_route - distribution_route ?sample_test - sample_test ?alternative_supplier - alternative_supplier)
    :precondition
      (and
        (entity_triage_completed ?distribution_center)
        (sample_linked_to_entity ?distribution_center ?sample_test)
        (distribution_center_has_route ?distribution_center ?distribution_route)
        (route_marked_for_alternate_supply ?distribution_route)
        (distribution_center_has_alternate_supplier ?distribution_center ?alternative_supplier)
        (not
          (distribution_quarantine_confirmed ?distribution_center)
        )
      )
    :effect
      (and
        (route_quarantined ?distribution_route)
        (distribution_quarantine_confirmed ?distribution_center)
        (alternate_supplier_available ?alternative_supplier)
        (not
          (distribution_center_has_alternate_supplier ?distribution_center ?alternative_supplier)
        )
      )
  )
  (:action instantiate_recovery_plan
    :parameters (?manufacturing_site - manufacturing_site ?distribution_center - distribution_center ?supply_batch - supply_batch ?distribution_route - distribution_route ?recovery_plan - recovery_plan)
    :precondition
      (and
        (site_quarantine_active ?manufacturing_site)
        (distribution_center_quarantine_active ?distribution_center)
        (manufacturing_site_contains_batch ?manufacturing_site ?supply_batch)
        (distribution_center_has_route ?distribution_center ?distribution_route)
        (batch_quarantined ?supply_batch)
        (route_quarantined ?distribution_route)
        (site_quarantine_confirmed ?manufacturing_site)
        (distribution_quarantine_confirmed ?distribution_center)
        (recovery_plan_available ?recovery_plan)
      )
    :effect
      (and
        (recovery_plan_instantiated ?recovery_plan)
        (recovery_plan_covers_batch ?recovery_plan ?supply_batch)
        (recovery_plan_covers_route ?recovery_plan ?distribution_route)
        (not
          (recovery_plan_available ?recovery_plan)
        )
      )
  )
  (:action instantiate_recovery_plan_stage_one
    :parameters (?manufacturing_site - manufacturing_site ?distribution_center - distribution_center ?supply_batch - supply_batch ?distribution_route - distribution_route ?recovery_plan - recovery_plan)
    :precondition
      (and
        (site_quarantine_active ?manufacturing_site)
        (distribution_center_quarantine_active ?distribution_center)
        (manufacturing_site_contains_batch ?manufacturing_site ?supply_batch)
        (distribution_center_has_route ?distribution_center ?distribution_route)
        (batch_marked_for_alternate_supply ?supply_batch)
        (route_quarantined ?distribution_route)
        (not
          (site_quarantine_confirmed ?manufacturing_site)
        )
        (distribution_quarantine_confirmed ?distribution_center)
        (recovery_plan_available ?recovery_plan)
      )
    :effect
      (and
        (recovery_plan_instantiated ?recovery_plan)
        (recovery_plan_covers_batch ?recovery_plan ?supply_batch)
        (recovery_plan_covers_route ?recovery_plan ?distribution_route)
        (recovery_plan_flag_stage_one ?recovery_plan)
        (not
          (recovery_plan_available ?recovery_plan)
        )
      )
  )
  (:action instantiate_recovery_plan_stage_two
    :parameters (?manufacturing_site - manufacturing_site ?distribution_center - distribution_center ?supply_batch - supply_batch ?distribution_route - distribution_route ?recovery_plan - recovery_plan)
    :precondition
      (and
        (site_quarantine_active ?manufacturing_site)
        (distribution_center_quarantine_active ?distribution_center)
        (manufacturing_site_contains_batch ?manufacturing_site ?supply_batch)
        (distribution_center_has_route ?distribution_center ?distribution_route)
        (batch_quarantined ?supply_batch)
        (route_marked_for_alternate_supply ?distribution_route)
        (site_quarantine_confirmed ?manufacturing_site)
        (not
          (distribution_quarantine_confirmed ?distribution_center)
        )
        (recovery_plan_available ?recovery_plan)
      )
    :effect
      (and
        (recovery_plan_instantiated ?recovery_plan)
        (recovery_plan_covers_batch ?recovery_plan ?supply_batch)
        (recovery_plan_covers_route ?recovery_plan ?distribution_route)
        (recovery_plan_flag_stage_two ?recovery_plan)
        (not
          (recovery_plan_available ?recovery_plan)
        )
      )
  )
  (:action instantiate_recovery_plan_full_stage
    :parameters (?manufacturing_site - manufacturing_site ?distribution_center - distribution_center ?supply_batch - supply_batch ?distribution_route - distribution_route ?recovery_plan - recovery_plan)
    :precondition
      (and
        (site_quarantine_active ?manufacturing_site)
        (distribution_center_quarantine_active ?distribution_center)
        (manufacturing_site_contains_batch ?manufacturing_site ?supply_batch)
        (distribution_center_has_route ?distribution_center ?distribution_route)
        (batch_marked_for_alternate_supply ?supply_batch)
        (route_marked_for_alternate_supply ?distribution_route)
        (not
          (site_quarantine_confirmed ?manufacturing_site)
        )
        (not
          (distribution_quarantine_confirmed ?distribution_center)
        )
        (recovery_plan_available ?recovery_plan)
      )
    :effect
      (and
        (recovery_plan_instantiated ?recovery_plan)
        (recovery_plan_covers_batch ?recovery_plan ?supply_batch)
        (recovery_plan_covers_route ?recovery_plan ?distribution_route)
        (recovery_plan_flag_stage_one ?recovery_plan)
        (recovery_plan_flag_stage_two ?recovery_plan)
        (not
          (recovery_plan_available ?recovery_plan)
        )
      )
  )
  (:action schedule_plan_testing
    :parameters (?recovery_plan - recovery_plan ?manufacturing_site - manufacturing_site ?sample_test - sample_test)
    :precondition
      (and
        (recovery_plan_instantiated ?recovery_plan)
        (site_quarantine_active ?manufacturing_site)
        (sample_linked_to_entity ?manufacturing_site ?sample_test)
        (not
          (recovery_plan_testing_scheduled ?recovery_plan)
        )
      )
    :effect (recovery_plan_testing_scheduled ?recovery_plan)
  )
  (:action engage_inspection_protocol
    :parameters (?response_team - response_team ?inspection_protocol - inspection_protocol ?recovery_plan - recovery_plan)
    :precondition
      (and
        (entity_triage_completed ?response_team)
        (team_assigned_to_recovery_plan ?response_team ?recovery_plan)
        (team_uses_inspection_protocol ?response_team ?inspection_protocol)
        (inspection_protocol_available ?inspection_protocol)
        (recovery_plan_instantiated ?recovery_plan)
        (recovery_plan_testing_scheduled ?recovery_plan)
        (not
          (inspection_protocol_in_use ?inspection_protocol)
        )
      )
    :effect
      (and
        (inspection_protocol_in_use ?inspection_protocol)
        (protocol_applied_to_recovery_plan ?inspection_protocol ?recovery_plan)
        (not
          (inspection_protocol_available ?inspection_protocol)
        )
      )
  )
  (:action complete_inspection_protocol_validation
    :parameters (?response_team - response_team ?inspection_protocol - inspection_protocol ?recovery_plan - recovery_plan ?sample_test - sample_test)
    :precondition
      (and
        (entity_triage_completed ?response_team)
        (team_uses_inspection_protocol ?response_team ?inspection_protocol)
        (inspection_protocol_in_use ?inspection_protocol)
        (protocol_applied_to_recovery_plan ?inspection_protocol ?recovery_plan)
        (sample_linked_to_entity ?response_team ?sample_test)
        (not
          (recovery_plan_flag_stage_one ?recovery_plan)
        )
        (not
          (team_validation_ready ?response_team)
        )
      )
    :effect (team_validation_ready ?response_team)
  )
  (:action assign_communication_channel_to_team
    :parameters (?response_team - response_team ?communication_channel - communication_channel)
    :precondition
      (and
        (entity_triage_completed ?response_team)
        (comm_channel_available ?communication_channel)
        (not
          (team_has_comm_channel ?response_team)
        )
      )
    :effect
      (and
        (team_has_comm_channel ?response_team)
        (team_assigned_comm_channel ?response_team ?communication_channel)
        (not
          (comm_channel_available ?communication_channel)
        )
      )
  )
  (:action prepare_team_for_communication_and_validation
    :parameters (?response_team - response_team ?inspection_protocol - inspection_protocol ?recovery_plan - recovery_plan ?sample_test - sample_test ?communication_channel - communication_channel)
    :precondition
      (and
        (entity_triage_completed ?response_team)
        (team_uses_inspection_protocol ?response_team ?inspection_protocol)
        (inspection_protocol_in_use ?inspection_protocol)
        (protocol_applied_to_recovery_plan ?inspection_protocol ?recovery_plan)
        (sample_linked_to_entity ?response_team ?sample_test)
        (recovery_plan_flag_stage_one ?recovery_plan)
        (team_has_comm_channel ?response_team)
        (team_assigned_comm_channel ?response_team ?communication_channel)
        (not
          (team_validation_ready ?response_team)
        )
      )
    :effect
      (and
        (team_validation_ready ?response_team)
        (team_comm_ready ?response_team)
      )
  )
  (:action perform_remediation_and_initiate_team_testing
    :parameters (?response_team - response_team ?remediation_method - remediation_method ?site_agent - site_agent ?inspection_protocol - inspection_protocol ?recovery_plan - recovery_plan)
    :precondition
