## Planning domain (PDDL)

(define (domain pharmaceutics_post_market_signal_escalation)
  (:requirements :strips :typing :negative-preconditions)
  (:types clinical_safety_resource - object operational_resource - object supply_chain_asset - object case_supertype - object signal_case - case_supertype triage_resource - clinical_safety_resource investigator - clinical_safety_resource containment_team - clinical_safety_resource external_stakeholder - clinical_safety_resource logistics_asset - clinical_safety_resource notification_template - clinical_safety_resource mitigation_resource - clinical_safety_resource regulatory_notice - clinical_safety_resource alternate_supplier - operational_resource quality_document - operational_resource external_expert - operational_resource manufacturing_lot - supply_chain_asset distribution_batch - supply_chain_asset remedial_action_item - supply_chain_asset site_subtype - signal_case product_subtype - signal_case manufacturing_site - site_subtype distribution_site - site_subtype product - product_subtype)
  (:predicates
    (case_registered ?signal_case - signal_case)
    (entity_assessed ?signal_case - signal_case)
    (triage_assigned ?signal_case - signal_case)
    (entity_final_resolution_applied ?signal_case - signal_case)
    (entity_recovery_initiated ?signal_case - signal_case)
    (case_closed ?signal_case - signal_case)
    (triage_resource_available ?triage_resource - triage_resource)
    (triage_assignment_for_entity ?signal_case - signal_case ?triage_resource - triage_resource)
    (investigator_available ?investigator - investigator)
    (investigator_assignment_for_entity ?signal_case - signal_case ?investigator - investigator)
    (containment_team_available ?containment_team - containment_team)
    (containment_assignment_for_entity ?signal_case - signal_case ?containment_team - containment_team)
    (alternate_supplier_available ?alternate_supplier - alternate_supplier)
    (manufacturing_site_alternate_supplier ?manufacturing_site - manufacturing_site ?alternate_supplier - alternate_supplier)
    (distribution_site_alternate_supplier ?distribution_site - distribution_site ?alternate_supplier - alternate_supplier)
    (site_lot_link ?manufacturing_site - manufacturing_site ?manufacturing_lot - manufacturing_lot)
    (lot_active_mitigation ?manufacturing_lot - manufacturing_lot)
    (lot_alternate_mitigation ?manufacturing_lot - manufacturing_lot)
    (site_mitigation_active ?manufacturing_site - manufacturing_site)
    (site_batch_link ?distribution_site - distribution_site ?distribution_batch - distribution_batch)
    (batch_active_mitigation ?distribution_batch - distribution_batch)
    (batch_alternate_mitigation ?distribution_batch - distribution_batch)
    (distribution_site_mitigation_active ?distribution_site - distribution_site)
    (remedial_action_ready ?remedial_action - remedial_action_item)
    (remedial_action_initiated ?remedial_action - remedial_action_item)
    (remedial_action_targets_lot ?remedial_action - remedial_action_item ?manufacturing_lot - manufacturing_lot)
    (remedial_action_targets_batch ?remedial_action - remedial_action_item ?distribution_batch - distribution_batch)
    (remedial_action_requires_external_consultation ?remedial_action - remedial_action_item)
    (remedial_action_requires_regulatory_clearance ?remedial_action - remedial_action_item)
    (remedial_action_executed ?remedial_action - remedial_action_item)
    (product_manufacturing_site_link ?product - product ?manufacturing_site - manufacturing_site)
    (product_distribution_site_link ?product - product ?distribution_site - distribution_site)
    (product_remedial_action_link ?product - product ?remedial_action - remedial_action_item)
    (document_available ?quality_document - quality_document)
    (product_document_link ?product - product ?quality_document - quality_document)
    (document_applied_flag ?quality_document - quality_document)
    (document_applied_to_remedial_action ?quality_document - quality_document ?remedial_action - remedial_action_item)
    (product_documentation_verified ?product - product)
    (product_approval_initiated ?product - product)
    (release_authorized ?product - product)
    (external_stakeholder_assigned ?product - product)
    (external_stakeholder_approval_obtained ?product - product)
    (additional_approval_obtained ?product - product)
    (quality_release_validated ?product - product)
    (external_expert_available ?external_expert - external_expert)
    (product_external_expert_assignment ?product - product ?external_expert - external_expert)
    (external_expert_engaged ?product - product)
    (external_expert_review_completed ?product - product)
    (external_expert_signoff_obtained ?product - product)
    (external_stakeholder_available ?external_stakeholder - external_stakeholder)
    (product_stakeholder_link ?product - product ?external_stakeholder - external_stakeholder)
    (logistics_asset_available ?logistics_asset - logistics_asset)
    (product_logistics_asset_assignment ?product - product ?logistics_asset - logistics_asset)
    (mitigation_resource_available ?mitigation_resource - mitigation_resource)
    (product_mitigation_resource_assignment ?product - product ?mitigation_resource - mitigation_resource)
    (regulatory_notice_available ?regulatory_notice - regulatory_notice)
    (product_regulatory_notice_assignment ?product - product ?regulatory_notice - regulatory_notice)
    (notification_template_available ?notification_template - notification_template)
    (notification_template_assignment_for_entity ?signal_case - signal_case ?notification_template - notification_template)
    (site_ready ?manufacturing_site - manufacturing_site)
    (distribution_site_ready ?distribution_site - distribution_site)
    (final_release_confirmed ?product - product)
  )
  (:action register_signal_case
    :parameters (?signal_case - signal_case)
    :precondition
      (and
        (not
          (case_registered ?signal_case)
        )
        (not
          (entity_final_resolution_applied ?signal_case)
        )
      )
    :effect (case_registered ?signal_case)
  )
  (:action assign_triage_resource
    :parameters (?signal_case - signal_case ?triage_resource - triage_resource)
    :precondition
      (and
        (case_registered ?signal_case)
        (not
          (triage_assigned ?signal_case)
        )
        (triage_resource_available ?triage_resource)
      )
    :effect
      (and
        (triage_assigned ?signal_case)
        (triage_assignment_for_entity ?signal_case ?triage_resource)
        (not
          (triage_resource_available ?triage_resource)
        )
      )
  )
  (:action allocate_investigator
    :parameters (?signal_case - signal_case ?investigator - investigator)
    :precondition
      (and
        (case_registered ?signal_case)
        (triage_assigned ?signal_case)
        (investigator_available ?investigator)
      )
    :effect
      (and
        (investigator_assignment_for_entity ?signal_case ?investigator)
        (not
          (investigator_available ?investigator)
        )
      )
  )
  (:action complete_assessment
    :parameters (?signal_case - signal_case ?investigator - investigator)
    :precondition
      (and
        (case_registered ?signal_case)
        (triage_assigned ?signal_case)
        (investigator_assignment_for_entity ?signal_case ?investigator)
        (not
          (entity_assessed ?signal_case)
        )
      )
    :effect (entity_assessed ?signal_case)
  )
  (:action release_investigator
    :parameters (?signal_case - signal_case ?investigator - investigator)
    :precondition
      (and
        (investigator_assignment_for_entity ?signal_case ?investigator)
      )
    :effect
      (and
        (investigator_available ?investigator)
        (not
          (investigator_assignment_for_entity ?signal_case ?investigator)
        )
      )
  )
  (:action assign_containment_team
    :parameters (?signal_case - signal_case ?containment_team - containment_team)
    :precondition
      (and
        (entity_assessed ?signal_case)
        (containment_team_available ?containment_team)
      )
    :effect
      (and
        (containment_assignment_for_entity ?signal_case ?containment_team)
        (not
          (containment_team_available ?containment_team)
        )
      )
  )
  (:action release_containment_team
    :parameters (?signal_case - signal_case ?containment_team - containment_team)
    :precondition
      (and
        (containment_assignment_for_entity ?signal_case ?containment_team)
      )
    :effect
      (and
        (containment_team_available ?containment_team)
        (not
          (containment_assignment_for_entity ?signal_case ?containment_team)
        )
      )
  )
  (:action assign_mitigation_resource_to_product
    :parameters (?product - product ?mitigation_resource - mitigation_resource)
    :precondition
      (and
        (entity_assessed ?product)
        (mitigation_resource_available ?mitigation_resource)
      )
    :effect
      (and
        (product_mitigation_resource_assignment ?product ?mitigation_resource)
        (not
          (mitigation_resource_available ?mitigation_resource)
        )
      )
  )
  (:action release_mitigation_resource_from_product
    :parameters (?product - product ?mitigation_resource - mitigation_resource)
    :precondition
      (and
        (product_mitigation_resource_assignment ?product ?mitigation_resource)
      )
    :effect
      (and
        (mitigation_resource_available ?mitigation_resource)
        (not
          (product_mitigation_resource_assignment ?product ?mitigation_resource)
        )
      )
  )
  (:action assign_regulatory_notice_to_product
    :parameters (?product - product ?regulatory_notice - regulatory_notice)
    :precondition
      (and
        (entity_assessed ?product)
        (regulatory_notice_available ?regulatory_notice)
      )
    :effect
      (and
        (product_regulatory_notice_assignment ?product ?regulatory_notice)
        (not
          (regulatory_notice_available ?regulatory_notice)
        )
      )
  )
  (:action release_regulatory_notice_from_product
    :parameters (?product - product ?regulatory_notice - regulatory_notice)
    :precondition
      (and
        (product_regulatory_notice_assignment ?product ?regulatory_notice)
      )
    :effect
      (and
        (regulatory_notice_available ?regulatory_notice)
        (not
          (product_regulatory_notice_assignment ?product ?regulatory_notice)
        )
      )
  )
  (:action initiate_lot_detection
    :parameters (?manufacturing_site - manufacturing_site ?manufacturing_lot - manufacturing_lot ?investigator - investigator)
    :precondition
      (and
        (entity_assessed ?manufacturing_site)
        (investigator_assignment_for_entity ?manufacturing_site ?investigator)
        (site_lot_link ?manufacturing_site ?manufacturing_lot)
        (not
          (lot_active_mitigation ?manufacturing_lot)
        )
        (not
          (lot_alternate_mitigation ?manufacturing_lot)
        )
      )
    :effect (lot_active_mitigation ?manufacturing_lot)
  )
  (:action start_site_mitigation
    :parameters (?manufacturing_site - manufacturing_site ?manufacturing_lot - manufacturing_lot ?containment_team - containment_team)
    :precondition
      (and
        (entity_assessed ?manufacturing_site)
        (containment_assignment_for_entity ?manufacturing_site ?containment_team)
        (site_lot_link ?manufacturing_site ?manufacturing_lot)
        (lot_active_mitigation ?manufacturing_lot)
        (not
          (site_ready ?manufacturing_site)
        )
      )
    :effect
      (and
        (site_ready ?manufacturing_site)
        (site_mitigation_active ?manufacturing_site)
      )
  )
  (:action assign_alternate_supplier_to_site
    :parameters (?manufacturing_site - manufacturing_site ?manufacturing_lot - manufacturing_lot ?alternate_supplier - alternate_supplier)
    :precondition
      (and
        (entity_assessed ?manufacturing_site)
        (site_lot_link ?manufacturing_site ?manufacturing_lot)
        (alternate_supplier_available ?alternate_supplier)
        (not
          (site_ready ?manufacturing_site)
        )
      )
    :effect
      (and
        (lot_alternate_mitigation ?manufacturing_lot)
        (site_ready ?manufacturing_site)
        (manufacturing_site_alternate_supplier ?manufacturing_site ?alternate_supplier)
        (not
          (alternate_supplier_available ?alternate_supplier)
        )
      )
  )
  (:action finalize_site_mitigation
    :parameters (?manufacturing_site - manufacturing_site ?manufacturing_lot - manufacturing_lot ?investigator - investigator ?alternate_supplier - alternate_supplier)
    :precondition
      (and
        (entity_assessed ?manufacturing_site)
        (investigator_assignment_for_entity ?manufacturing_site ?investigator)
        (site_lot_link ?manufacturing_site ?manufacturing_lot)
        (lot_alternate_mitigation ?manufacturing_lot)
        (manufacturing_site_alternate_supplier ?manufacturing_site ?alternate_supplier)
        (not
          (site_mitigation_active ?manufacturing_site)
        )
      )
    :effect
      (and
        (lot_active_mitigation ?manufacturing_lot)
        (site_mitigation_active ?manufacturing_site)
        (alternate_supplier_available ?alternate_supplier)
        (not
          (manufacturing_site_alternate_supplier ?manufacturing_site ?alternate_supplier)
        )
      )
  )
  (:action initiate_batch_detection
    :parameters (?distribution_site - distribution_site ?distribution_batch - distribution_batch ?investigator - investigator)
    :precondition
      (and
        (entity_assessed ?distribution_site)
        (investigator_assignment_for_entity ?distribution_site ?investigator)
        (site_batch_link ?distribution_site ?distribution_batch)
        (not
          (batch_active_mitigation ?distribution_batch)
        )
        (not
          (batch_alternate_mitigation ?distribution_batch)
        )
      )
    :effect (batch_active_mitigation ?distribution_batch)
  )
  (:action start_distribution_site_mitigation
    :parameters (?distribution_site - distribution_site ?distribution_batch - distribution_batch ?containment_team - containment_team)
    :precondition
      (and
        (entity_assessed ?distribution_site)
        (containment_assignment_for_entity ?distribution_site ?containment_team)
        (site_batch_link ?distribution_site ?distribution_batch)
        (batch_active_mitigation ?distribution_batch)
        (not
          (distribution_site_ready ?distribution_site)
        )
      )
    :effect
      (and
        (distribution_site_ready ?distribution_site)
        (distribution_site_mitigation_active ?distribution_site)
      )
  )
  (:action assign_alternate_supplier_to_distribution_site
    :parameters (?distribution_site - distribution_site ?distribution_batch - distribution_batch ?alternate_supplier - alternate_supplier)
    :precondition
      (and
        (entity_assessed ?distribution_site)
        (site_batch_link ?distribution_site ?distribution_batch)
        (alternate_supplier_available ?alternate_supplier)
        (not
          (distribution_site_ready ?distribution_site)
        )
      )
    :effect
      (and
        (batch_alternate_mitigation ?distribution_batch)
        (distribution_site_ready ?distribution_site)
        (distribution_site_alternate_supplier ?distribution_site ?alternate_supplier)
        (not
          (alternate_supplier_available ?alternate_supplier)
        )
      )
  )
  (:action finalize_distribution_site_mitigation
    :parameters (?distribution_site - distribution_site ?distribution_batch - distribution_batch ?investigator - investigator ?alternate_supplier - alternate_supplier)
    :precondition
      (and
        (entity_assessed ?distribution_site)
        (investigator_assignment_for_entity ?distribution_site ?investigator)
        (site_batch_link ?distribution_site ?distribution_batch)
        (batch_alternate_mitigation ?distribution_batch)
        (distribution_site_alternate_supplier ?distribution_site ?alternate_supplier)
        (not
          (distribution_site_mitigation_active ?distribution_site)
        )
      )
    :effect
      (and
        (batch_active_mitigation ?distribution_batch)
        (distribution_site_mitigation_active ?distribution_site)
        (alternate_supplier_available ?alternate_supplier)
        (not
          (distribution_site_alternate_supplier ?distribution_site ?alternate_supplier)
        )
      )
  )
  (:action prepare_remedial_action
    :parameters (?manufacturing_site - manufacturing_site ?distribution_site - distribution_site ?manufacturing_lot - manufacturing_lot ?distribution_batch - distribution_batch ?remedial_action - remedial_action_item)
    :precondition
      (and
        (site_ready ?manufacturing_site)
        (distribution_site_ready ?distribution_site)
        (site_lot_link ?manufacturing_site ?manufacturing_lot)
        (site_batch_link ?distribution_site ?distribution_batch)
        (lot_active_mitigation ?manufacturing_lot)
        (batch_active_mitigation ?distribution_batch)
        (site_mitigation_active ?manufacturing_site)
        (distribution_site_mitigation_active ?distribution_site)
        (remedial_action_ready ?remedial_action)
      )
    :effect
      (and
        (remedial_action_initiated ?remedial_action)
        (remedial_action_targets_lot ?remedial_action ?manufacturing_lot)
        (remedial_action_targets_batch ?remedial_action ?distribution_batch)
        (not
          (remedial_action_ready ?remedial_action)
        )
      )
  )
  (:action prepare_remedial_action_with_consultation
    :parameters (?manufacturing_site - manufacturing_site ?distribution_site - distribution_site ?manufacturing_lot - manufacturing_lot ?distribution_batch - distribution_batch ?remedial_action - remedial_action_item)
    :precondition
      (and
        (site_ready ?manufacturing_site)
        (distribution_site_ready ?distribution_site)
        (site_lot_link ?manufacturing_site ?manufacturing_lot)
        (site_batch_link ?distribution_site ?distribution_batch)
        (lot_alternate_mitigation ?manufacturing_lot)
        (batch_active_mitigation ?distribution_batch)
        (not
          (site_mitigation_active ?manufacturing_site)
        )
        (distribution_site_mitigation_active ?distribution_site)
        (remedial_action_ready ?remedial_action)
      )
    :effect
      (and
        (remedial_action_initiated ?remedial_action)
        (remedial_action_targets_lot ?remedial_action ?manufacturing_lot)
        (remedial_action_targets_batch ?remedial_action ?distribution_batch)
        (remedial_action_requires_external_consultation ?remedial_action)
        (not
          (remedial_action_ready ?remedial_action)
        )
      )
  )
  (:action prepare_remedial_action_with_regulatory_clearance
    :parameters (?manufacturing_site - manufacturing_site ?distribution_site - distribution_site ?manufacturing_lot - manufacturing_lot ?distribution_batch - distribution_batch ?remedial_action - remedial_action_item)
    :precondition
      (and
        (site_ready ?manufacturing_site)
        (distribution_site_ready ?distribution_site)
        (site_lot_link ?manufacturing_site ?manufacturing_lot)
        (site_batch_link ?distribution_site ?distribution_batch)
        (lot_active_mitigation ?manufacturing_lot)
        (batch_alternate_mitigation ?distribution_batch)
        (site_mitigation_active ?manufacturing_site)
        (not
          (distribution_site_mitigation_active ?distribution_site)
        )
        (remedial_action_ready ?remedial_action)
      )
    :effect
      (and
        (remedial_action_initiated ?remedial_action)
        (remedial_action_targets_lot ?remedial_action ?manufacturing_lot)
        (remedial_action_targets_batch ?remedial_action ?distribution_batch)
        (remedial_action_requires_regulatory_clearance ?remedial_action)
        (not
          (remedial_action_ready ?remedial_action)
        )
      )
  )
  (:action prepare_remedial_action_with_consultation_and_regulatory_clearance
    :parameters (?manufacturing_site - manufacturing_site ?distribution_site - distribution_site ?manufacturing_lot - manufacturing_lot ?distribution_batch - distribution_batch ?remedial_action - remedial_action_item)
    :precondition
      (and
        (site_ready ?manufacturing_site)
        (distribution_site_ready ?distribution_site)
        (site_lot_link ?manufacturing_site ?manufacturing_lot)
        (site_batch_link ?distribution_site ?distribution_batch)
        (lot_alternate_mitigation ?manufacturing_lot)
        (batch_alternate_mitigation ?distribution_batch)
        (not
          (site_mitigation_active ?manufacturing_site)
        )
        (not
          (distribution_site_mitigation_active ?distribution_site)
        )
        (remedial_action_ready ?remedial_action)
      )
    :effect
      (and
        (remedial_action_initiated ?remedial_action)
        (remedial_action_targets_lot ?remedial_action ?manufacturing_lot)
        (remedial_action_targets_batch ?remedial_action ?distribution_batch)
        (remedial_action_requires_external_consultation ?remedial_action)
        (remedial_action_requires_regulatory_clearance ?remedial_action)
        (not
          (remedial_action_ready ?remedial_action)
        )
      )
  )
  (:action execute_remedial_action
    :parameters (?remedial_action - remedial_action_item ?manufacturing_site - manufacturing_site ?investigator - investigator)
    :precondition
      (and
        (remedial_action_initiated ?remedial_action)
        (site_ready ?manufacturing_site)
        (investigator_assignment_for_entity ?manufacturing_site ?investigator)
        (not
          (remedial_action_executed ?remedial_action)
        )
      )
    :effect (remedial_action_executed ?remedial_action)
  )
  (:action attach_quality_document_to_remedial_action
    :parameters (?product - product ?quality_document - quality_document ?remedial_action - remedial_action_item)
    :precondition
      (and
        (entity_assessed ?product)
        (product_remedial_action_link ?product ?remedial_action)
        (product_document_link ?product ?quality_document)
        (document_available ?quality_document)
        (remedial_action_initiated ?remedial_action)
        (remedial_action_executed ?remedial_action)
        (not
          (document_applied_flag ?quality_document)
        )
      )
    :effect
      (and
        (document_applied_flag ?quality_document)
        (document_applied_to_remedial_action ?quality_document ?remedial_action)
        (not
          (document_available ?quality_document)
        )
      )
  )
  (:action verify_product_documentation
    :parameters (?product - product ?quality_document - quality_document ?remedial_action - remedial_action_item ?investigator - investigator)
    :precondition
      (and
        (entity_assessed ?product)
        (product_document_link ?product ?quality_document)
        (document_applied_flag ?quality_document)
        (document_applied_to_remedial_action ?quality_document ?remedial_action)
        (investigator_assignment_for_entity ?product ?investigator)
        (not
          (remedial_action_requires_external_consultation ?remedial_action)
        )
        (not
          (product_documentation_verified ?product)
        )
      )
    :effect (product_documentation_verified ?product)
  )
  (:action assign_external_stakeholder_to_product
    :parameters (?product - product ?external_stakeholder - external_stakeholder)
    :precondition
      (and
        (entity_assessed ?product)
        (external_stakeholder_available ?external_stakeholder)
        (not
          (external_stakeholder_assigned ?product)
        )
      )
    :effect
      (and
        (external_stakeholder_assigned ?product)
        (product_stakeholder_link ?product ?external_stakeholder)
        (not
          (external_stakeholder_available ?external_stakeholder)
        )
      )
  )
  (:action obtain_stakeholder_approval_and_verify_documentation
    :parameters (?product - product ?quality_document - quality_document ?remedial_action - remedial_action_item ?investigator - investigator ?external_stakeholder - external_stakeholder)
    :precondition
      (and
        (entity_assessed ?product)
        (product_document_link ?product ?quality_document)
        (document_applied_flag ?quality_document)
        (document_applied_to_remedial_action ?quality_document ?remedial_action)
        (investigator_assignment_for_entity ?product ?investigator)
        (remedial_action_requires_external_consultation ?remedial_action)
        (external_stakeholder_assigned ?product)
        (product_stakeholder_link ?product ?external_stakeholder)
        (not
          (product_documentation_verified ?product)
        )
      )
    :effect
      (and
        (product_documentation_verified ?product)
        (external_stakeholder_approval_obtained ?product)
      )
  )
  (:action initiate_product_approval
    :parameters (?product - product ?mitigation_resource - mitigation_resource ?containment_team - containment_team ?quality_document - quality_document ?remedial_action - remedial_action_item)
    :precondition
      (and
        (product_documentation_verified ?product)
        (product_mitigation_resource_assignment ?product ?mitigation_resource)
        (containment_assignment_for_entity ?product ?containment_team)
        (product_document_link ?product ?quality_document)
        (document_applied_to_remedial_action ?quality_document ?remedial_action)
        (not
          (remedial_action_requires_regulatory_clearance ?remedial_action)
        )
        (not
          (product_approval_initiated ?product)
        )
      )
    :effect (product_approval_initiated ?product)
  )
  (:action initiate_product_approval_with_regulatory_consideration
    :parameters (?product - product ?mitigation_resource - mitigation_resource ?containment_team - containment_team ?quality_document - quality_document ?remedial_action - remedial_action_item)
    :precondition
      (and
        (product_documentation_verified ?product)
        (product_mitigation_resource_assignment ?product ?mitigation_resource)
        (containment_assignment_for_entity ?product ?containment_team)
        (product_document_link ?product ?quality_document)
        (document_applied_to_remedial_action ?quality_document ?remedial_action)
        (remedial_action_requires_regulatory_clearance ?remedial_action)
        (not
          (product_approval_initiated ?product)
