## Planning domain (PDDL)

(define (domain pharmaceutics_medication_error_follow_up)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_group - object supply_resource_group - object asset_group - object entity_abstract - object incident_entity - entity_abstract operational_resource - resource_group investigator - resource_group subject_matter_expert - resource_group notification_message - resource_group stakeholder - resource_group evidence_item - resource_group remediation_step - resource_group recovery_plan - resource_group mitigation_option - supply_resource_group inventory_item - supply_resource_group external_authority - supply_resource_group supply_node - asset_group supply_batch - asset_group recall_notice - asset_group affected_unit_group - incident_entity response_entity_group - incident_entity pharmacy_location - affected_unit_group clinical_unit - affected_unit_group response_coordinator - response_entity_group)
  (:predicates
    (entity_reported ?medication_error_case - incident_entity)
    (entity_ready_for_assignment ?medication_error_case - incident_entity)
    (entity_triaged ?medication_error_case - incident_entity)
    (containment_executed ?medication_error_case - incident_entity)
    (final_verification_passed_for_entity ?medication_error_case - incident_entity)
    (entity_requires_resource_allocation ?medication_error_case - incident_entity)
    (operational_resource_available ?operational_resource - operational_resource)
    (entity_assigned_to_resource ?medication_error_case - incident_entity ?operational_resource - operational_resource)
    (investigator_available ?investigator - investigator)
    (investigator_linked ?medication_error_case - incident_entity ?investigator - investigator)
    (subject_matter_expert_available ?subject_matter_expert - subject_matter_expert)
    (sme_linked ?medication_error_case - incident_entity ?subject_matter_expert - subject_matter_expert)
    (mitigation_option_available ?mitigation_option - mitigation_option)
    (mitigation_assigned_to_location ?pharmacy_location - pharmacy_location ?mitigation_option - mitigation_option)
    (mitigation_assigned_to_unit ?clinical_unit - clinical_unit ?mitigation_option - mitigation_option)
    (location_supply_link ?pharmacy_location - pharmacy_location ?supply_node - supply_node)
    (supply_node_quarantine_declared ?supply_node - supply_node)
    (supply_node_mitigation_declared ?supply_node - supply_node)
    (location_containment_confirmed ?pharmacy_location - pharmacy_location)
    (unit_supply_link ?clinical_unit - clinical_unit ?supply_batch - supply_batch)
    (supply_batch_quarantine_declared ?supply_batch - supply_batch)
    (supply_batch_mitigation_declared ?supply_batch - supply_batch)
    (unit_containment_confirmed ?clinical_unit - clinical_unit)
    (recall_notice_active ?recall_notice - recall_notice)
    (recall_notice_generated ?recall_notice - recall_notice)
    (recall_notice_targets_supply_node ?recall_notice - recall_notice ?supply_node - supply_node)
    (recall_notice_targets_supply_batch ?recall_notice - recall_notice ?supply_batch - supply_batch)
    (recall_notice_escalation_level_1 ?recall_notice - recall_notice)
    (recall_notice_escalation_level_2 ?recall_notice - recall_notice)
    (recall_notice_acknowledged ?recall_notice - recall_notice)
    (coordinator_linked_to_location ?response_coordinator - response_coordinator ?pharmacy_location - pharmacy_location)
    (coordinator_linked_to_unit ?response_coordinator - response_coordinator ?clinical_unit - clinical_unit)
    (coordinator_linked_to_recall_notice ?response_coordinator - response_coordinator ?recall_notice - recall_notice)
    (inventory_item_available ?inventory_item - inventory_item)
    (coordinator_linked_to_inventory_item ?response_coordinator - response_coordinator ?inventory_item - inventory_item)
    (inventory_item_quarantined ?inventory_item - inventory_item)
    (inventory_item_recall_link ?inventory_item - inventory_item ?recall_notice - recall_notice)
    (coordinator_inventory_verified ?response_coordinator - response_coordinator)
    (coordinator_analysis_completed ?response_coordinator - response_coordinator)
    (coordinator_ready_for_verification ?response_coordinator - response_coordinator)
    (notification_generated_for_coordinator ?response_coordinator - response_coordinator)
    (coordinator_notification_acknowledged ?response_coordinator - response_coordinator)
    (coordinator_has_pending_stakeholder_action ?response_coordinator - response_coordinator)
    (coordinator_final_checks_done ?response_coordinator - response_coordinator)
    (external_authority_available ?external_authority - external_authority)
    (coordinator_linked_to_external_authority ?response_coordinator - response_coordinator ?external_authority - external_authority)
    (external_authority_notified ?response_coordinator - response_coordinator)
    (external_authority_acknowledged ?response_coordinator - response_coordinator)
    (external_authority_approval_received ?response_coordinator - response_coordinator)
    (notification_message_available ?notification_message - notification_message)
    (coordinator_linked_to_notification_message ?response_coordinator - response_coordinator ?notification_message - notification_message)
    (stakeholder_available ?stakeholder - stakeholder)
    (coordinator_linked_to_stakeholder ?response_coordinator - response_coordinator ?stakeholder - stakeholder)
    (remediation_step_available ?remediation_step - remediation_step)
    (remediation_assigned_to_coordinator ?response_coordinator - response_coordinator ?remediation_step - remediation_step)
    (recovery_plan_available ?recovery_plan - recovery_plan)
    (recovery_plan_assigned_to_coordinator ?response_coordinator - response_coordinator ?recovery_plan - recovery_plan)
    (evidence_item_available ?evidence_item - evidence_item)
    (evidence_linked_to_case ?medication_error_case - incident_entity ?evidence_item - evidence_item)
    (location_containment_initiated ?pharmacy_location - pharmacy_location)
    (unit_containment_initiated ?clinical_unit - clinical_unit)
    (coordinator_final_verification_recorded ?response_coordinator - response_coordinator)
  )
  (:action register_medication_error_case
    :parameters (?medication_error_case - incident_entity)
    :precondition
      (and
        (not
          (entity_reported ?medication_error_case)
        )
        (not
          (containment_executed ?medication_error_case)
        )
      )
    :effect (entity_reported ?medication_error_case)
  )
  (:action assign_operational_resource_to_case
    :parameters (?medication_error_case - incident_entity ?operational_resource - operational_resource)
    :precondition
      (and
        (entity_reported ?medication_error_case)
        (not
          (entity_triaged ?medication_error_case)
        )
        (operational_resource_available ?operational_resource)
      )
    :effect
      (and
        (entity_triaged ?medication_error_case)
        (entity_assigned_to_resource ?medication_error_case ?operational_resource)
        (not
          (operational_resource_available ?operational_resource)
        )
      )
  )
  (:action link_investigator_to_case
    :parameters (?medication_error_case - incident_entity ?investigator - investigator)
    :precondition
      (and
        (entity_reported ?medication_error_case)
        (entity_triaged ?medication_error_case)
        (investigator_available ?investigator)
      )
    :effect
      (and
        (investigator_linked ?medication_error_case ?investigator)
        (not
          (investigator_available ?investigator)
        )
      )
  )
  (:action mark_case_ready_for_assignment
    :parameters (?medication_error_case - incident_entity ?investigator - investigator)
    :precondition
      (and
        (entity_reported ?medication_error_case)
        (entity_triaged ?medication_error_case)
        (investigator_linked ?medication_error_case ?investigator)
        (not
          (entity_ready_for_assignment ?medication_error_case)
        )
      )
    :effect (entity_ready_for_assignment ?medication_error_case)
  )
  (:action release_investigator
    :parameters (?medication_error_case - incident_entity ?investigator - investigator)
    :precondition
      (and
        (investigator_linked ?medication_error_case ?investigator)
      )
    :effect
      (and
        (investigator_available ?investigator)
        (not
          (investigator_linked ?medication_error_case ?investigator)
        )
      )
  )
  (:action assign_sme_to_case
    :parameters (?medication_error_case - incident_entity ?subject_matter_expert - subject_matter_expert)
    :precondition
      (and
        (entity_ready_for_assignment ?medication_error_case)
        (subject_matter_expert_available ?subject_matter_expert)
      )
    :effect
      (and
        (sme_linked ?medication_error_case ?subject_matter_expert)
        (not
          (subject_matter_expert_available ?subject_matter_expert)
        )
      )
  )
  (:action release_sme
    :parameters (?medication_error_case - incident_entity ?subject_matter_expert - subject_matter_expert)
    :precondition
      (and
        (sme_linked ?medication_error_case ?subject_matter_expert)
      )
    :effect
      (and
        (subject_matter_expert_available ?subject_matter_expert)
        (not
          (sme_linked ?medication_error_case ?subject_matter_expert)
        )
      )
  )
  (:action assign_remediation_step_to_coordinator
    :parameters (?response_coordinator - response_coordinator ?remediation_step - remediation_step)
    :precondition
      (and
        (entity_ready_for_assignment ?response_coordinator)
        (remediation_step_available ?remediation_step)
      )
    :effect
      (and
        (remediation_assigned_to_coordinator ?response_coordinator ?remediation_step)
        (not
          (remediation_step_available ?remediation_step)
        )
      )
  )
  (:action complete_remediation_step
    :parameters (?response_coordinator - response_coordinator ?remediation_step - remediation_step)
    :precondition
      (and
        (remediation_assigned_to_coordinator ?response_coordinator ?remediation_step)
      )
    :effect
      (and
        (remediation_step_available ?remediation_step)
        (not
          (remediation_assigned_to_coordinator ?response_coordinator ?remediation_step)
        )
      )
  )
  (:action assign_recovery_plan_to_coordinator
    :parameters (?response_coordinator - response_coordinator ?recovery_plan - recovery_plan)
    :precondition
      (and
        (entity_ready_for_assignment ?response_coordinator)
        (recovery_plan_available ?recovery_plan)
      )
    :effect
      (and
        (recovery_plan_assigned_to_coordinator ?response_coordinator ?recovery_plan)
        (not
          (recovery_plan_available ?recovery_plan)
        )
      )
  )
  (:action complete_recovery_plan
    :parameters (?response_coordinator - response_coordinator ?recovery_plan - recovery_plan)
    :precondition
      (and
        (recovery_plan_assigned_to_coordinator ?response_coordinator ?recovery_plan)
      )
    :effect
      (and
        (recovery_plan_available ?recovery_plan)
        (not
          (recovery_plan_assigned_to_coordinator ?response_coordinator ?recovery_plan)
        )
      )
  )
  (:action flag_supply_node_for_quarantine
    :parameters (?pharmacy_location - pharmacy_location ?supply_node - supply_node ?investigator - investigator)
    :precondition
      (and
        (entity_ready_for_assignment ?pharmacy_location)
        (investigator_linked ?pharmacy_location ?investigator)
        (location_supply_link ?pharmacy_location ?supply_node)
        (not
          (supply_node_quarantine_declared ?supply_node)
        )
        (not
          (supply_node_mitigation_declared ?supply_node)
        )
      )
    :effect (supply_node_quarantine_declared ?supply_node)
  )
  (:action confirm_location_containment
    :parameters (?pharmacy_location - pharmacy_location ?supply_node - supply_node ?subject_matter_expert - subject_matter_expert)
    :precondition
      (and
        (entity_ready_for_assignment ?pharmacy_location)
        (sme_linked ?pharmacy_location ?subject_matter_expert)
        (location_supply_link ?pharmacy_location ?supply_node)
        (supply_node_quarantine_declared ?supply_node)
        (not
          (location_containment_initiated ?pharmacy_location)
        )
      )
    :effect
      (and
        (location_containment_initiated ?pharmacy_location)
        (location_containment_confirmed ?pharmacy_location)
      )
  )
  (:action apply_mitigation_option_to_location
    :parameters (?pharmacy_location - pharmacy_location ?supply_node - supply_node ?mitigation_option - mitigation_option)
    :precondition
      (and
        (entity_ready_for_assignment ?pharmacy_location)
        (location_supply_link ?pharmacy_location ?supply_node)
        (mitigation_option_available ?mitigation_option)
        (not
          (location_containment_initiated ?pharmacy_location)
        )
      )
    :effect
      (and
        (supply_node_mitigation_declared ?supply_node)
        (location_containment_initiated ?pharmacy_location)
        (mitigation_assigned_to_location ?pharmacy_location ?mitigation_option)
        (not
          (mitigation_option_available ?mitigation_option)
        )
      )
  )
  (:action finalize_location_mitigation
    :parameters (?pharmacy_location - pharmacy_location ?supply_node - supply_node ?investigator - investigator ?mitigation_option - mitigation_option)
    :precondition
      (and
        (entity_ready_for_assignment ?pharmacy_location)
        (investigator_linked ?pharmacy_location ?investigator)
        (location_supply_link ?pharmacy_location ?supply_node)
        (supply_node_mitigation_declared ?supply_node)
        (mitigation_assigned_to_location ?pharmacy_location ?mitigation_option)
        (not
          (location_containment_confirmed ?pharmacy_location)
        )
      )
    :effect
      (and
        (supply_node_quarantine_declared ?supply_node)
        (location_containment_confirmed ?pharmacy_location)
        (mitigation_option_available ?mitigation_option)
        (not
          (mitigation_assigned_to_location ?pharmacy_location ?mitigation_option)
        )
      )
  )
  (:action flag_supply_batch_for_quarantine
    :parameters (?clinical_unit - clinical_unit ?supply_batch - supply_batch ?investigator - investigator)
    :precondition
      (and
        (entity_ready_for_assignment ?clinical_unit)
        (investigator_linked ?clinical_unit ?investigator)
        (unit_supply_link ?clinical_unit ?supply_batch)
        (not
          (supply_batch_quarantine_declared ?supply_batch)
        )
        (not
          (supply_batch_mitigation_declared ?supply_batch)
        )
      )
    :effect (supply_batch_quarantine_declared ?supply_batch)
  )
  (:action confirm_unit_containment
    :parameters (?clinical_unit - clinical_unit ?supply_batch - supply_batch ?subject_matter_expert - subject_matter_expert)
    :precondition
      (and
        (entity_ready_for_assignment ?clinical_unit)
        (sme_linked ?clinical_unit ?subject_matter_expert)
        (unit_supply_link ?clinical_unit ?supply_batch)
        (supply_batch_quarantine_declared ?supply_batch)
        (not
          (unit_containment_initiated ?clinical_unit)
        )
      )
    :effect
      (and
        (unit_containment_initiated ?clinical_unit)
        (unit_containment_confirmed ?clinical_unit)
      )
  )
  (:action apply_mitigation_option_to_unit
    :parameters (?clinical_unit - clinical_unit ?supply_batch - supply_batch ?mitigation_option - mitigation_option)
    :precondition
      (and
        (entity_ready_for_assignment ?clinical_unit)
        (unit_supply_link ?clinical_unit ?supply_batch)
        (mitigation_option_available ?mitigation_option)
        (not
          (unit_containment_initiated ?clinical_unit)
        )
      )
    :effect
      (and
        (supply_batch_mitigation_declared ?supply_batch)
        (unit_containment_initiated ?clinical_unit)
        (mitigation_assigned_to_unit ?clinical_unit ?mitigation_option)
        (not
          (mitigation_option_available ?mitigation_option)
        )
      )
  )
  (:action finalize_unit_mitigation
    :parameters (?clinical_unit - clinical_unit ?supply_batch - supply_batch ?investigator - investigator ?mitigation_option - mitigation_option)
    :precondition
      (and
        (entity_ready_for_assignment ?clinical_unit)
        (investigator_linked ?clinical_unit ?investigator)
        (unit_supply_link ?clinical_unit ?supply_batch)
        (supply_batch_mitigation_declared ?supply_batch)
        (mitigation_assigned_to_unit ?clinical_unit ?mitigation_option)
        (not
          (unit_containment_confirmed ?clinical_unit)
        )
      )
    :effect
      (and
        (supply_batch_quarantine_declared ?supply_batch)
        (unit_containment_confirmed ?clinical_unit)
        (mitigation_option_available ?mitigation_option)
        (not
          (mitigation_assigned_to_unit ?clinical_unit ?mitigation_option)
        )
      )
  )
  (:action issue_recall_notice_standard
    :parameters (?pharmacy_location - pharmacy_location ?clinical_unit - clinical_unit ?supply_node - supply_node ?supply_batch - supply_batch ?recall_notice - recall_notice)
    :precondition
      (and
        (location_containment_initiated ?pharmacy_location)
        (unit_containment_initiated ?clinical_unit)
        (location_supply_link ?pharmacy_location ?supply_node)
        (unit_supply_link ?clinical_unit ?supply_batch)
        (supply_node_quarantine_declared ?supply_node)
        (supply_batch_quarantine_declared ?supply_batch)
        (location_containment_confirmed ?pharmacy_location)
        (unit_containment_confirmed ?clinical_unit)
        (recall_notice_active ?recall_notice)
      )
    :effect
      (and
        (recall_notice_generated ?recall_notice)
        (recall_notice_targets_supply_node ?recall_notice ?supply_node)
        (recall_notice_targets_supply_batch ?recall_notice ?supply_batch)
        (not
          (recall_notice_active ?recall_notice)
        )
      )
  )
  (:action issue_recall_notice_escalation_level_1
    :parameters (?pharmacy_location - pharmacy_location ?clinical_unit - clinical_unit ?supply_node - supply_node ?supply_batch - supply_batch ?recall_notice - recall_notice)
    :precondition
      (and
        (location_containment_initiated ?pharmacy_location)
        (unit_containment_initiated ?clinical_unit)
        (location_supply_link ?pharmacy_location ?supply_node)
        (unit_supply_link ?clinical_unit ?supply_batch)
        (supply_node_mitigation_declared ?supply_node)
        (supply_batch_quarantine_declared ?supply_batch)
        (not
          (location_containment_confirmed ?pharmacy_location)
        )
        (unit_containment_confirmed ?clinical_unit)
        (recall_notice_active ?recall_notice)
      )
    :effect
      (and
        (recall_notice_generated ?recall_notice)
        (recall_notice_targets_supply_node ?recall_notice ?supply_node)
        (recall_notice_targets_supply_batch ?recall_notice ?supply_batch)
        (recall_notice_escalation_level_1 ?recall_notice)
        (not
          (recall_notice_active ?recall_notice)
        )
      )
  )
  (:action issue_recall_notice_escalation_level_2
    :parameters (?pharmacy_location - pharmacy_location ?clinical_unit - clinical_unit ?supply_node - supply_node ?supply_batch - supply_batch ?recall_notice - recall_notice)
    :precondition
      (and
        (location_containment_initiated ?pharmacy_location)
        (unit_containment_initiated ?clinical_unit)
        (location_supply_link ?pharmacy_location ?supply_node)
        (unit_supply_link ?clinical_unit ?supply_batch)
        (supply_node_quarantine_declared ?supply_node)
        (supply_batch_mitigation_declared ?supply_batch)
        (location_containment_confirmed ?pharmacy_location)
        (not
          (unit_containment_confirmed ?clinical_unit)
        )
        (recall_notice_active ?recall_notice)
      )
    :effect
      (and
        (recall_notice_generated ?recall_notice)
        (recall_notice_targets_supply_node ?recall_notice ?supply_node)
        (recall_notice_targets_supply_batch ?recall_notice ?supply_batch)
        (recall_notice_escalation_level_2 ?recall_notice)
        (not
          (recall_notice_active ?recall_notice)
        )
      )
  )
  (:action issue_recall_notice_escalation_levels_1_and_2
    :parameters (?pharmacy_location - pharmacy_location ?clinical_unit - clinical_unit ?supply_node - supply_node ?supply_batch - supply_batch ?recall_notice - recall_notice)
    :precondition
      (and
        (location_containment_initiated ?pharmacy_location)
        (unit_containment_initiated ?clinical_unit)
        (location_supply_link ?pharmacy_location ?supply_node)
        (unit_supply_link ?clinical_unit ?supply_batch)
        (supply_node_mitigation_declared ?supply_node)
        (supply_batch_mitigation_declared ?supply_batch)
        (not
          (location_containment_confirmed ?pharmacy_location)
        )
        (not
          (unit_containment_confirmed ?clinical_unit)
        )
        (recall_notice_active ?recall_notice)
      )
    :effect
      (and
        (recall_notice_generated ?recall_notice)
        (recall_notice_targets_supply_node ?recall_notice ?supply_node)
        (recall_notice_targets_supply_batch ?recall_notice ?supply_batch)
        (recall_notice_escalation_level_1 ?recall_notice)
        (recall_notice_escalation_level_2 ?recall_notice)
        (not
          (recall_notice_active ?recall_notice)
        )
      )
  )
  (:action acknowledge_recall_notice_by_location
    :parameters (?recall_notice - recall_notice ?pharmacy_location - pharmacy_location ?investigator - investigator)
    :precondition
      (and
        (recall_notice_generated ?recall_notice)
        (location_containment_initiated ?pharmacy_location)
        (investigator_linked ?pharmacy_location ?investigator)
        (not
          (recall_notice_acknowledged ?recall_notice)
        )
      )
    :effect (recall_notice_acknowledged ?recall_notice)
  )
  (:action quarantine_inventory_item
    :parameters (?response_coordinator - response_coordinator ?inventory_item - inventory_item ?recall_notice - recall_notice)
    :precondition
      (and
        (entity_ready_for_assignment ?response_coordinator)
        (coordinator_linked_to_recall_notice ?response_coordinator ?recall_notice)
        (coordinator_linked_to_inventory_item ?response_coordinator ?inventory_item)
        (inventory_item_available ?inventory_item)
        (recall_notice_generated ?recall_notice)
        (recall_notice_acknowledged ?recall_notice)
        (not
          (inventory_item_quarantined ?inventory_item)
        )
      )
    :effect
      (and
        (inventory_item_quarantined ?inventory_item)
        (inventory_item_recall_link ?inventory_item ?recall_notice)
        (not
          (inventory_item_available ?inventory_item)
        )
      )
  )
  (:action start_inventory_verification
    :parameters (?response_coordinator - response_coordinator ?inventory_item - inventory_item ?recall_notice - recall_notice ?investigator - investigator)
    :precondition
      (and
        (entity_ready_for_assignment ?response_coordinator)
        (coordinator_linked_to_inventory_item ?response_coordinator ?inventory_item)
        (inventory_item_quarantined ?inventory_item)
        (inventory_item_recall_link ?inventory_item ?recall_notice)
        (investigator_linked ?response_coordinator ?investigator)
        (not
          (recall_notice_escalation_level_1 ?recall_notice)
        )
        (not
          (coordinator_inventory_verified ?response_coordinator)
        )
      )
    :effect (coordinator_inventory_verified ?response_coordinator)
  )
  (:action generate_notification_message_for_coordinator
    :parameters (?response_coordinator - response_coordinator ?notification_message - notification_message)
    :precondition
      (and
        (entity_ready_for_assignment ?response_coordinator)
        (notification_message_available ?notification_message)
        (not
          (notification_generated_for_coordinator ?response_coordinator)
        )
      )
    :effect
      (and
        (notification_generated_for_coordinator ?response_coordinator)
        (coordinator_linked_to_notification_message ?response_coordinator ?notification_message)
        (not
          (notification_message_available ?notification_message)
        )
      )
  )
  (:action process_notification_and_initiate_verification
    :parameters (?response_coordinator - response_coordinator ?inventory_item - inventory_item ?recall_notice - recall_notice ?investigator - investigator ?notification_message - notification_message)
    :precondition
      (and
        (entity_ready_for_assignment ?response_coordinator)
        (coordinator_linked_to_inventory_item ?response_coordinator ?inventory_item)
        (inventory_item_quarantined ?inventory_item)
        (inventory_item_recall_link ?inventory_item ?recall_notice)
        (investigator_linked ?response_coordinator ?investigator)
        (recall_notice_escalation_level_1 ?recall_notice)
        (notification_generated_for_coordinator ?response_coordinator)
        (coordinator_linked_to_notification_message ?response_coordinator ?notification_message)
