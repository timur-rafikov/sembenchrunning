## Planning domain (PDDL)

(define (domain module_responsibility_partitioning)
  (:requirements :strips :typing :negative-preconditions)
  (:types auxiliary_resource_group - object design_artifact_group - object runtime_artifact_group - object named_entity - object module - named_entity interface_spec - auxiliary_resource_group responsibility_token - auxiliary_resource_group stakeholder - auxiliary_resource_group extension_plugin - auxiliary_resource_group test_asset - auxiliary_resource_group nonfunctional_constraint - auxiliary_resource_group capability_descriptor - auxiliary_resource_group compliance_record - auxiliary_resource_group design_rationale - design_artifact_group configuration_asset - design_artifact_group review_record - design_artifact_group staging_runtime_context - runtime_artifact_group production_runtime_context - runtime_artifact_group deployment_artifact - runtime_artifact_group module_category - module implementation_category - module service_module - module_category client_module - module_category implementation_component - implementation_category)
  (:predicates
    (module_declared ?module - module)
    (module_validated ?module - module)
    (interface_assigned ?module - module)
    (module_deployed ?module - module)
    (ready_for_deployment ?module - module)
    (module_approved_for_deployment ?module - module)
    (interface_available ?interface_spec - interface_spec)
    (module_interface_binding ?module - module ?interface_spec - interface_spec)
    (responsibility_token_available ?responsibility_token - responsibility_token)
    (module_responsibility_bound ?module - module ?responsibility_token - responsibility_token)
    (stakeholder_available ?stakeholder - stakeholder)
    (module_has_stakeholder ?module - module ?stakeholder - stakeholder)
    (design_rationale_available ?design_rationale - design_rationale)
    (service_has_design_rationale ?service_module - service_module ?design_rationale - design_rationale)
    (client_has_design_rationale ?client_module - client_module ?design_rationale - design_rationale)
    (service_bound_to_staging_runtime_context ?service_module - service_module ?staging_runtime_context - staging_runtime_context)
    (staging_runtime_context_ready ?staging_runtime_context - staging_runtime_context)
    (staging_runtime_context_allocated ?staging_runtime_context - staging_runtime_context)
    (service_ready ?service_module - service_module)
    (client_bound_to_production_runtime_context ?client_module - client_module ?production_runtime_context - production_runtime_context)
    (production_runtime_context_ready ?production_runtime_context - production_runtime_context)
    (production_runtime_context_allocated ?production_runtime_context - production_runtime_context)
    (client_ready ?client_module - client_module)
    (artifact_available ?deployment_artifact - deployment_artifact)
    (artifact_packaged ?deployment_artifact - deployment_artifact)
    (artifact_bound_to_staging_runtime_context ?deployment_artifact - deployment_artifact ?staging_runtime_context - staging_runtime_context)
    (artifact_bound_to_production_runtime_context ?deployment_artifact - deployment_artifact ?production_runtime_context - production_runtime_context)
    (artifact_includes_staging_context ?deployment_artifact - deployment_artifact)
    (artifact_includes_production_context ?deployment_artifact - deployment_artifact)
    (artifact_validated ?deployment_artifact - deployment_artifact)
    (component_assigned_to_service ?implementation_component - implementation_component ?service_module - service_module)
    (component_assigned_to_client ?implementation_component - implementation_component ?client_module - client_module)
    (component_in_deployment_artifact ?implementation_component - implementation_component ?deployment_artifact - deployment_artifact)
    (configuration_asset_available ?configuration_asset - configuration_asset)
    (component_has_configuration_asset ?implementation_component - implementation_component ?configuration_asset - configuration_asset)
    (configuration_asset_committed ?configuration_asset - configuration_asset)
    (configuration_bound_to_artifact ?configuration_asset - configuration_asset ?deployment_artifact - deployment_artifact)
    (component_feature_gate_prepared ?implementation_component - implementation_component)
    (component_feature_gate_started ?implementation_component - implementation_component)
    (component_features_enabled ?implementation_component - implementation_component)
    (component_extension_registered ?implementation_component - implementation_component)
    (component_extension_configured ?implementation_component - implementation_component)
    (component_test_ready ?implementation_component - implementation_component)
    (component_integrated ?implementation_component - implementation_component)
    (review_record_available ?review_record - review_record)
    (component_has_review_record ?implementation_component - implementation_component ?review_record - review_record)
    (component_under_review ?implementation_component - implementation_component)
    (component_review_validated ?implementation_component - implementation_component)
    (component_review_approved ?implementation_component - implementation_component)
    (extension_plugin_available ?extension_plugin - extension_plugin)
    (component_extension_assigned ?implementation_component - implementation_component ?extension_plugin - extension_plugin)
    (test_asset_available ?test_asset - test_asset)
    (component_has_test_asset ?implementation_component - implementation_component ?test_asset - test_asset)
    (capability_descriptor_available ?capability_descriptor - capability_descriptor)
    (component_has_capability_descriptor ?implementation_component - implementation_component ?capability_descriptor - capability_descriptor)
    (compliance_record_available ?compliance_record - compliance_record)
    (component_has_compliance_record ?implementation_component - implementation_component ?compliance_record - compliance_record)
    (nonfunctional_constraint_available ?nonfunctional_constraint - nonfunctional_constraint)
    (module_has_nonfunctional_constraint ?module - module ?nonfunctional_constraint - nonfunctional_constraint)
    (service_packaging_flag ?service_module - service_module)
    (client_packaging_flag ?client_module - client_module)
    (component_release_staged ?implementation_component - implementation_component)
  )
  (:action declare_module
    :parameters (?module - module)
    :precondition
      (and
        (not
          (module_declared ?module)
        )
        (not
          (module_deployed ?module)
        )
      )
    :effect (module_declared ?module)
  )
  (:action allocate_interface_to_module
    :parameters (?module - module ?interface_spec - interface_spec)
    :precondition
      (and
        (module_declared ?module)
        (not
          (interface_assigned ?module)
        )
        (interface_available ?interface_spec)
      )
    :effect
      (and
        (interface_assigned ?module)
        (module_interface_binding ?module ?interface_spec)
        (not
          (interface_available ?interface_spec)
        )
      )
  )
  (:action assign_responsibility
    :parameters (?module - module ?responsibility_token - responsibility_token)
    :precondition
      (and
        (module_declared ?module)
        (interface_assigned ?module)
        (responsibility_token_available ?responsibility_token)
      )
    :effect
      (and
        (module_responsibility_bound ?module ?responsibility_token)
        (not
          (responsibility_token_available ?responsibility_token)
        )
      )
  )
  (:action validate_responsibility
    :parameters (?module - module ?responsibility_token - responsibility_token)
    :precondition
      (and
        (module_declared ?module)
        (interface_assigned ?module)
        (module_responsibility_bound ?module ?responsibility_token)
        (not
          (module_validated ?module)
        )
      )
    :effect (module_validated ?module)
  )
  (:action release_responsibility
    :parameters (?module - module ?responsibility_token - responsibility_token)
    :precondition
      (and
        (module_responsibility_bound ?module ?responsibility_token)
      )
    :effect
      (and
        (responsibility_token_available ?responsibility_token)
        (not
          (module_responsibility_bound ?module ?responsibility_token)
        )
      )
  )
  (:action assign_stakeholder_to_module
    :parameters (?module - module ?stakeholder - stakeholder)
    :precondition
      (and
        (module_validated ?module)
        (stakeholder_available ?stakeholder)
      )
    :effect
      (and
        (module_has_stakeholder ?module ?stakeholder)
        (not
          (stakeholder_available ?stakeholder)
        )
      )
  )
  (:action unassign_stakeholder_from_module
    :parameters (?module - module ?stakeholder - stakeholder)
    :precondition
      (and
        (module_has_stakeholder ?module ?stakeholder)
      )
    :effect
      (and
        (stakeholder_available ?stakeholder)
        (not
          (module_has_stakeholder ?module ?stakeholder)
        )
      )
  )
  (:action attach_capability_descriptor
    :parameters (?implementation_component - implementation_component ?capability_descriptor - capability_descriptor)
    :precondition
      (and
        (module_validated ?implementation_component)
        (capability_descriptor_available ?capability_descriptor)
      )
    :effect
      (and
        (component_has_capability_descriptor ?implementation_component ?capability_descriptor)
        (not
          (capability_descriptor_available ?capability_descriptor)
        )
      )
  )
  (:action detach_capability_descriptor
    :parameters (?implementation_component - implementation_component ?capability_descriptor - capability_descriptor)
    :precondition
      (and
        (component_has_capability_descriptor ?implementation_component ?capability_descriptor)
      )
    :effect
      (and
        (capability_descriptor_available ?capability_descriptor)
        (not
          (component_has_capability_descriptor ?implementation_component ?capability_descriptor)
        )
      )
  )
  (:action attach_compliance_record
    :parameters (?implementation_component - implementation_component ?compliance_record - compliance_record)
    :precondition
      (and
        (module_validated ?implementation_component)
        (compliance_record_available ?compliance_record)
      )
    :effect
      (and
        (component_has_compliance_record ?implementation_component ?compliance_record)
        (not
          (compliance_record_available ?compliance_record)
        )
      )
  )
  (:action detach_compliance_record
    :parameters (?implementation_component - implementation_component ?compliance_record - compliance_record)
    :precondition
      (and
        (component_has_compliance_record ?implementation_component ?compliance_record)
      )
    :effect
      (and
        (compliance_record_available ?compliance_record)
        (not
          (component_has_compliance_record ?implementation_component ?compliance_record)
        )
      )
  )
  (:action stage_service_staging_runtime_context
    :parameters (?service_module - service_module ?staging_runtime_context - staging_runtime_context ?responsibility_token - responsibility_token)
    :precondition
      (and
        (module_validated ?service_module)
        (module_responsibility_bound ?service_module ?responsibility_token)
        (service_bound_to_staging_runtime_context ?service_module ?staging_runtime_context)
        (not
          (staging_runtime_context_ready ?staging_runtime_context)
        )
        (not
          (staging_runtime_context_allocated ?staging_runtime_context)
        )
      )
    :effect (staging_runtime_context_ready ?staging_runtime_context)
  )
  (:action confirm_service_context_ready
    :parameters (?service_module - service_module ?staging_runtime_context - staging_runtime_context ?stakeholder - stakeholder)
    :precondition
      (and
        (module_validated ?service_module)
        (module_has_stakeholder ?service_module ?stakeholder)
        (service_bound_to_staging_runtime_context ?service_module ?staging_runtime_context)
        (staging_runtime_context_ready ?staging_runtime_context)
        (not
          (service_packaging_flag ?service_module)
        )
      )
    :effect
      (and
        (service_packaging_flag ?service_module)
        (service_ready ?service_module)
      )
  )
  (:action apply_rationale_to_service_context
    :parameters (?service_module - service_module ?staging_runtime_context - staging_runtime_context ?design_rationale - design_rationale)
    :precondition
      (and
        (module_validated ?service_module)
        (service_bound_to_staging_runtime_context ?service_module ?staging_runtime_context)
        (design_rationale_available ?design_rationale)
        (not
          (service_packaging_flag ?service_module)
        )
      )
    :effect
      (and
        (staging_runtime_context_allocated ?staging_runtime_context)
        (service_packaging_flag ?service_module)
        (service_has_design_rationale ?service_module ?design_rationale)
        (not
          (design_rationale_available ?design_rationale)
        )
      )
  )
  (:action synchronize_service_context_and_rationale
    :parameters (?service_module - service_module ?staging_runtime_context - staging_runtime_context ?responsibility_token - responsibility_token ?design_rationale - design_rationale)
    :precondition
      (and
        (module_validated ?service_module)
        (module_responsibility_bound ?service_module ?responsibility_token)
        (service_bound_to_staging_runtime_context ?service_module ?staging_runtime_context)
        (staging_runtime_context_allocated ?staging_runtime_context)
        (service_has_design_rationale ?service_module ?design_rationale)
        (not
          (service_ready ?service_module)
        )
      )
    :effect
      (and
        (staging_runtime_context_ready ?staging_runtime_context)
        (service_ready ?service_module)
        (design_rationale_available ?design_rationale)
        (not
          (service_has_design_rationale ?service_module ?design_rationale)
        )
      )
  )
  (:action stage_client_production_runtime_context

      (and
        (module_validated ?client_module)

  (:action confirm_client_context_ready

        )

        (production_runtime_context_allocated ?production_runtime_context)

          (design_rationale_available ?design_rationale)
        )

  )
  (:action synchronize_client_context_and_rationale
    :parameters (?client_module - client_module ?production_runtime_context - production_runtime_context ?responsibility_token - responsibility_token ?design_rationale - design_rationale)

  (:action assemble_deployment_artifact

      (and
        (service_packaging_flag ?service_module)
        (client_packaging_flag ?client_module)
        (service_bound_to_staging_runtime_context ?service_module ?staging_runtime_context)
        (client_bound_to_production_runtime_context ?client_module ?production_runtime_context)
        (staging_runtime_context_ready ?staging_runtime_context)
        (production_runtime_context_ready ?production_runtime_context)
        (service_ready ?service_module)
        (client_ready ?client_module)
        (artifact_available ?deployment_artifact)
      )
    :effect
      (and
        (artifact_packaged ?deployment_artifact)
        (artifact_bound_to_staging_runtime_context ?deployment_artifact ?staging_runtime_context)
        (artifact_bound_to_production_runtime_context ?deployment_artifact ?production_runtime_context)
        (not
          (artifact_available ?deployment_artifact)
        )
      )
  )
  (:action assemble_deployment_artifact_with_staging_context
    :parameters (?service_module - service_module ?client_module - client_module ?staging_runtime_context - staging_runtime_context ?production_runtime_context - production_runtime_context ?deployment_artifact - deployment_artifact)
    :precondition
      (and
        (service_packaging_flag ?service_module)
        (client_packaging_flag ?client_module)

    :effect
      (and
        (artifact_packaged ?deployment_artifact)

    :effect

        (artifact_packaged ?deployment_artifact)

        (artifact_includes_production_context ?deployment_artifact)
        (not
          (artifact_available ?deployment_artifact)
        )
      )
  )
  (:action assemble_deployment_artifact_with_both_contexts
    :parameters (?service_module - service_module ?client_module - client_module ?staging_runtime_context - staging_runtime_context ?production_runtime_context - production_runtime_context ?deployment_artifact - deployment_artifact)
    :precondition
      (and

          (client_ready ?client_module)

        (artifact_includes_production_context ?deployment_artifact)
        (not
          (artifact_available ?deployment_artifact)
        )

      (and
        (component_feature_gate_prepared ?implementation_component)

          (artifact_includes_production_context ?deployment_artifact)

    :precondition
      (and

        (artifact_includes_production_context ?deployment_artifact)

          (component_feature_gate_started ?implementation_component)

  (:action enable_features_stage_1
    :parameters (?implementation_component - implementation_component ?compliance_record - compliance_record ?configuration_asset - configuration_asset ?deployment_artifact - deployment_artifact)
    :precondition
