    (orchestration_operation_requested ?long_running_operation - long_running_operation)
    (entity_activated ?long_running_operation - long_running_operation)
    (orchestration_operation_has_resource_claim ?long_running_operation - long_running_operation)

    (entity_completion_published ?long_running_operation - long_running_operation)
    (entity_completion_signaled ?long_running_operation - long_running_operation)

    (orchestration_operation_bound_to_resource ?long_running_operation - long_running_operation ?resource_token - resource_token)

    (orchestration_operation_bound_to_task ?long_running_operation - long_running_operation ?task_definition - task_definition)
    (connector_available ?external_connector - external_connector)
    (orchestration_operation_bound_to_connector ?long_running_operation - long_running_operation ?external_connector - external_connector)
    (data_artifact_available ?data_artifact - data_artifact)
    (coordinator_has_data_artifact ?coordinator_instance - coordinator_instance ?data_artifact - data_artifact)
    (worker_pool_has_data_artifact ?worker_pool - worker_pool ?data_artifact - data_artifact)
    (coordinator_has_input_channel ?coordinator_instance - coordinator_instance ?input_channel - input_channel)

    (input_channel_populated ?input_channel - input_channel)
    (coordinator_input_finalized ?coordinator_instance - coordinator_instance)
    (worker_pool_has_output_channel ?worker_pool - worker_pool ?output_channel - output_channel)

    (work_unit_output_artifact_attached ?work_unit - work_unit)

    (workflow_engine_bound_to_worker_pool ?workflow_engine - workflow_engine ?worker_pool - worker_pool)

    (package_available ?package - package)
    (workflow_engine_has_package ?workflow_engine - workflow_engine ?package - package)
    (package_locked ?package - package)

    (extension_engaged ?workflow_engine - workflow_engine)

    (execution_triggered ?workflow_engine - workflow_engine)

  (:action create_orchestration_operation_request
    :parameters (?long_running_operation - long_running_operation)
    :precondition
      (and
        (not
          (orchestration_operation_requested ?long_running_operation)
        )
        (not
          (entity_finalized ?long_running_operation)
        )
      )
    :effect (orchestration_operation_requested ?long_running_operation)
  )
  (:action assign_resource_token_to_orchestration_operation
    :parameters (?long_running_operation - long_running_operation ?resource_token - resource_token)
    :precondition
      (and
        (orchestration_operation_requested ?long_running_operation)
        (not
          (orchestration_operation_has_resource_claim ?long_running_operation)
        )
        (resource_token_available ?resource_token)
      )
    :effect
      (and
        (orchestration_operation_has_resource_claim ?long_running_operation)
        (orchestration_operation_bound_to_resource ?long_running_operation ?resource_token)
        (not
          (resource_token_available ?resource_token)
        )
      )
  )
  (:action bind_task_definition_to_orchestration_operation
    :parameters (?long_running_operation - long_running_operation ?task_definition - task_definition)
    :precondition
      (and
        (orchestration_operation_requested ?long_running_operation)
        (orchestration_operation_has_resource_claim ?long_running_operation)
        (task_definition_available ?task_definition)
      )
    :effect
      (and
        (orchestration_operation_bound_to_task ?long_running_operation ?task_definition)
        (not
          (task_definition_available ?task_definition)
        )
      )
  )
  (:action activate_orchestration_operation
    :parameters (?long_running_operation - long_running_operation ?task_definition - task_definition)
    :precondition
      (and
        (orchestration_operation_requested ?long_running_operation)
        (orchestration_operation_has_resource_claim ?long_running_operation)
        (orchestration_operation_bound_to_task ?long_running_operation ?task_definition)
        (not
          (entity_activated ?long_running_operation)
        )
      )
    :effect (entity_activated ?long_running_operation)
  )
  (:action release_task_definition
    :parameters (?long_running_operation - long_running_operation ?task_definition - task_definition)
    :precondition
      (and
        (orchestration_operation_bound_to_task ?long_running_operation ?task_definition)
      )
    :effect
      (and
        (task_definition_available ?task_definition)
        (not
          (orchestration_operation_bound_to_task ?long_running_operation ?task_definition)
        )
      )
  )
  (:action attach_external_connector_to_orchestration_operation
    :parameters (?long_running_operation - long_running_operation ?external_connector - external_connector)
    :precondition
      (and
        (entity_activated ?long_running_operation)
        (connector_available ?external_connector)
      )
    :effect
      (and
        (orchestration_operation_bound_to_connector ?long_running_operation ?external_connector)
        (not
          (connector_available ?external_connector)
        )
      )
  )
  (:action detach_external_connector_from_orchestration_operation
    :parameters (?long_running_operation - long_running_operation ?external_connector - external_connector)
    :precondition
      (and
        (orchestration_operation_bound_to_connector ?long_running_operation ?external_connector)
      )
    :effect
      (and
        (connector_available ?external_connector)
        (not
          (orchestration_operation_bound_to_connector ?long_running_operation ?external_connector)
        )
      )
  )
  (:action attach_credential_to_engine
    :parameters (?workflow_engine - workflow_engine ?credential - credential)
    :precondition
      (and
        (entity_activated ?workflow_engine)
        (credential_available ?credential)
      )
    :effect
      (and
        (workflow_engine_has_credential ?workflow_engine ?credential)
        (not
          (credential_available ?credential)
        )
      )
  )
  (:action detach_credential_from_engine
    :parameters (?workflow_engine - workflow_engine ?credential - credential)
    :precondition
      (and
        (workflow_engine_has_credential ?workflow_engine ?credential)
      )
    :effect
      (and
        (credential_available ?credential)
        (not
          (workflow_engine_has_credential ?workflow_engine ?credential)
        )
      )
  )
  (:action attach_plugin_to_engine
    :parameters (?workflow_engine - workflow_engine ?plugin - plugin)
    :precondition
      (and
        (entity_activated ?workflow_engine)
        (plugin_available ?plugin)
      )
    :effect
      (and
        (workflow_engine_has_plugin ?workflow_engine ?plugin)
        (not
          (plugin_available ?plugin)
        )
      )
  )
  (:action detach_plugin_from_engine
    :parameters (?workflow_engine - workflow_engine ?plugin - plugin)
    :precondition
      (and
        (workflow_engine_has_plugin ?workflow_engine ?plugin)
      )
    :effect
      (and
        (plugin_available ?plugin)
        (not
          (workflow_engine_has_plugin ?workflow_engine ?plugin)
        )
      )
  )
  (:action coordinator_claim_input_channel
    :parameters (?coordinator_instance - coordinator_instance ?input_channel - input_channel ?task_definition - task_definition)
    :precondition
      (and
        (entity_activated ?coordinator_instance)
        (orchestration_operation_bound_to_task ?coordinator_instance ?task_definition)
        (coordinator_has_input_channel ?coordinator_instance ?input_channel)
        (not
          (input_channel_claimed ?input_channel)
        )
        (not
          (input_channel_populated ?input_channel)
        )
      )
    :effect (input_channel_claimed ?input_channel)
  )
  (:action coordinator_confirm_connector_and_lock
    :parameters (?coordinator_instance - coordinator_instance ?input_channel - input_channel ?external_connector - external_connector)
    :precondition
      (and
        (entity_activated ?coordinator_instance)
        (orchestration_operation_bound_to_connector ?coordinator_instance ?external_connector)
        (coordinator_has_input_channel ?coordinator_instance ?input_channel)
        (input_channel_claimed ?input_channel)
        (not
          (coordinator_ready ?coordinator_instance)
        )
      )
    :effect
      (and
        (coordinator_ready ?coordinator_instance)
        (coordinator_input_finalized ?coordinator_instance)
      )
  )
  (:action coordinator_attach_data_artifact
    :parameters (?coordinator_instance - coordinator_instance ?input_channel - input_channel ?data_artifact - data_artifact)
    :precondition
      (and
        (entity_activated ?coordinator_instance)
        (coordinator_has_input_channel ?coordinator_instance ?input_channel)
        (data_artifact_available ?data_artifact)
        (not
          (coordinator_ready ?coordinator_instance)
        )
      )
    :effect
      (and
        (input_channel_populated ?input_channel)
        (coordinator_ready ?coordinator_instance)
        (coordinator_has_data_artifact ?coordinator_instance ?data_artifact)
        (not
          (data_artifact_available ?data_artifact)
        )
      )
  )
  (:action coordinator_finalize_input
    :parameters (?coordinator_instance - coordinator_instance ?input_channel - input_channel ?task_definition - task_definition ?data_artifact - data_artifact)
    :precondition
      (and
        (entity_activated ?coordinator_instance)
        (orchestration_operation_bound_to_task ?coordinator_instance ?task_definition)
        (coordinator_has_input_channel ?coordinator_instance ?input_channel)
        (input_channel_populated ?input_channel)
        (coordinator_has_data_artifact ?coordinator_instance ?data_artifact)
        (not
          (coordinator_input_finalized ?coordinator_instance)
        )
      )
    :effect
      (and
        (input_channel_claimed ?input_channel)
        (coordinator_input_finalized ?coordinator_instance)
        (data_artifact_available ?data_artifact)
        (not
          (coordinator_has_data_artifact ?coordinator_instance ?data_artifact)
        )
      )
  )
  (:action worker_pool_claim_output_channel
    :parameters (?worker_pool - worker_pool ?output_channel - output_channel ?task_definition - task_definition)
    :precondition
      (and
        (entity_activated ?worker_pool)
        (orchestration_operation_bound_to_task ?worker_pool ?task_definition)
        (worker_pool_has_output_channel ?worker_pool ?output_channel)
        (not
          (output_channel_claimed ?output_channel)
        )
        (not
          (output_channel_populated ?output_channel)
        )
      )
    :effect (output_channel_claimed ?output_channel)
  )
  (:action worker_pool_confirm_connector_and_lock
    :parameters (?worker_pool - worker_pool ?output_channel - output_channel ?external_connector - external_connector)
    :precondition
      (and
        (entity_activated ?worker_pool)
        (orchestration_operation_bound_to_connector ?worker_pool ?external_connector)
        (worker_pool_has_output_channel ?worker_pool ?output_channel)
        (output_channel_claimed ?output_channel)
        (not
          (worker_pool_ready ?worker_pool)
        )
      )
    :effect
      (and
        (worker_pool_ready ?worker_pool)
        (worker_pool_output_finalized ?worker_pool)
      )
  )
  (:action worker_pool_attach_data_artifact
    :parameters (?worker_pool - worker_pool ?output_channel - output_channel ?data_artifact - data_artifact)
    :precondition
      (and
        (entity_activated ?worker_pool)
        (worker_pool_has_output_channel ?worker_pool ?output_channel)
        (data_artifact_available ?data_artifact)
        (not
          (worker_pool_ready ?worker_pool)
        )
      )
    :effect
      (and
        (output_channel_populated ?output_channel)
        (worker_pool_ready ?worker_pool)
        (worker_pool_has_data_artifact ?worker_pool ?data_artifact)
        (not
          (data_artifact_available ?data_artifact)
        )
      )
  )
  (:action worker_pool_finalize_output
    :parameters (?worker_pool - worker_pool ?output_channel - output_channel ?task_definition - task_definition ?data_artifact - data_artifact)
    :precondition
      (and
        (entity_activated ?worker_pool)
        (orchestration_operation_bound_to_task ?worker_pool ?task_definition)
        (worker_pool_has_output_channel ?worker_pool ?output_channel)
        (output_channel_populated ?output_channel)
        (worker_pool_has_data_artifact ?worker_pool ?data_artifact)
        (not
          (worker_pool_output_finalized ?worker_pool)
        )
      )
    :effect
      (and
        (output_channel_claimed ?output_channel)
        (worker_pool_output_finalized ?worker_pool)
        (data_artifact_available ?data_artifact)
        (not
          (worker_pool_has_data_artifact ?worker_pool ?data_artifact)
        )
      )
  )
  (:action compose_work_unit
    :parameters (?coordinator_instance - coordinator_instance ?worker_pool - worker_pool ?input_channel - input_channel ?output_channel - output_channel ?work_unit - work_unit)
    :precondition
      (and
        (coordinator_ready ?coordinator_instance)
        (worker_pool_ready ?worker_pool)
        (coordinator_has_input_channel ?coordinator_instance ?input_channel)
        (worker_pool_has_output_channel ?worker_pool ?output_channel)
        (input_channel_claimed ?input_channel)
        (output_channel_claimed ?output_channel)
        (coordinator_input_finalized ?coordinator_instance)
        (worker_pool_output_finalized ?worker_pool)
        (work_unit_allocated ?work_unit)
      )
    :effect
      (and
        (work_unit_composed ?work_unit)
        (work_unit_input_channel ?work_unit ?input_channel)
        (work_unit_output_channel ?work_unit ?output_channel)
        (not
          (work_unit_allocated ?work_unit)
        )
      )
  )
  (:action compose_work_unit_with_input_attachment
    :parameters (?coordinator_instance - coordinator_instance ?worker_pool - worker_pool ?input_channel - input_channel ?output_channel - output_channel ?work_unit - work_unit)
    :precondition
      (and
        (coordinator_ready ?coordinator_instance)
        (worker_pool_ready ?worker_pool)
        (coordinator_has_input_channel ?coordinator_instance ?input_channel)
        (worker_pool_has_output_channel ?worker_pool ?output_channel)
        (input_channel_populated ?input_channel)
        (output_channel_claimed ?output_channel)
        (not
          (coordinator_input_finalized ?coordinator_instance)
        )
        (worker_pool_output_finalized ?worker_pool)
        (work_unit_allocated ?work_unit)
      )
    :effect
      (and
        (work_unit_composed ?work_unit)
        (work_unit_input_channel ?work_unit ?input_channel)
        (work_unit_output_channel ?work_unit ?output_channel)
        (work_unit_input_artifact_attached ?work_unit)
        (not
          (work_unit_allocated ?work_unit)
        )
      )
  )
  (:action compose_work_unit_with_output_attachment
    :parameters (?coordinator_instance - coordinator_instance ?worker_pool - worker_pool ?input_channel - input_channel ?output_channel - output_channel ?work_unit - work_unit)
    :precondition
      (and
        (coordinator_ready ?coordinator_instance)
        (worker_pool_ready ?worker_pool)
        (coordinator_has_input_channel ?coordinator_instance ?input_channel)
        (worker_pool_has_output_channel ?worker_pool ?output_channel)
        (input_channel_claimed ?input_channel)
        (output_channel_populated ?output_channel)
        (coordinator_input_finalized ?coordinator_instance)
        (not
          (worker_pool_output_finalized ?worker_pool)
        )
        (work_unit_allocated ?work_unit)
      )
    :effect
      (and
        (work_unit_composed ?work_unit)
        (work_unit_input_channel ?work_unit ?input_channel)
        (work_unit_output_channel ?work_unit ?output_channel)
        (work_unit_output_artifact_attached ?work_unit)
        (not
          (work_unit_allocated ?work_unit)
        )
      )
  )
  (:action compose_work_unit_with_both_attachments
    :parameters (?coordinator_instance - coordinator_instance ?worker_pool - worker_pool ?input_channel - input_channel ?output_channel - output_channel ?work_unit - work_unit)
    :precondition
      (and
        (coordinator_ready ?coordinator_instance)
        (worker_pool_ready ?worker_pool)
        (coordinator_has_input_channel ?coordinator_instance ?input_channel)
        (worker_pool_has_output_channel ?worker_pool ?output_channel)
        (input_channel_populated ?input_channel)
        (output_channel_populated ?output_channel)
        (not
          (coordinator_input_finalized ?coordinator_instance)
        )
        (not
          (worker_pool_output_finalized ?worker_pool)
        )
        (work_unit_allocated ?work_unit)
      )
    :effect
      (and
        (work_unit_composed ?work_unit)
        (work_unit_input_channel ?work_unit ?input_channel)
        (work_unit_output_channel ?work_unit ?output_channel)
        (work_unit_input_artifact_attached ?work_unit)
        (work_unit_output_artifact_attached ?work_unit)
        (not
          (work_unit_allocated ?work_unit)
        )
      )
  )
  (:action lock_work_unit_for_execution
    :parameters (?work_unit - work_unit ?coordinator_instance - coordinator_instance ?task_definition - task_definition)
    :precondition
      (and
        (work_unit_composed ?work_unit)
        (coordinator_ready ?coordinator_instance)
        (orchestration_operation_bound_to_task ?coordinator_instance ?task_definition)
        (not
          (work_unit_locked ?work_unit)
        )
      )
    :effect (work_unit_locked ?work_unit)
  )
  (:action attach_package_to_workflow_engine_and_lock
    :parameters (?workflow_engine - workflow_engine ?package - package ?work_unit - work_unit)
    :precondition
      (and
        (entity_activated ?workflow_engine)
        (workflow_engine_bound_to_work_unit ?workflow_engine ?work_unit)
        (workflow_engine_has_package ?workflow_engine ?package)
        (package_available ?package)
        (work_unit_composed ?work_unit)
        (work_unit_locked ?work_unit)
        (not
          (package_locked ?package)
        )
      )
    :effect
      (and
        (package_locked ?package)
        (package_assigned_to_work_unit ?package ?work_unit)
        (not
          (package_available ?package)
        )
      )
  )
  (:action validate_package_on_workflow_engine_and_mark
    :parameters (?workflow_engine - workflow_engine ?package - package ?work_unit - work_unit ?task_definition - task_definition)
    :precondition
      (and
        (entity_activated ?workflow_engine)
        (workflow_engine_has_package ?workflow_engine ?package)
        (package_locked ?package)
        (package_assigned_to_work_unit ?package ?work_unit)
        (orchestration_operation_bound_to_task ?workflow_engine ?task_definition)
        (not
          (work_unit_input_artifact_attached ?work_unit)
        )
        (not
          (workflow_engine_validated_package ?workflow_engine)
        )
      )
    :effect (workflow_engine_validated_package ?workflow_engine)
  )
  (:action bind_policy_to_workflow_engine
    :parameters (?workflow_engine - workflow_engine ?policy - policy)
    :precondition
      (and
        (entity_activated ?workflow_engine)
        (policy_available ?policy)
        (not
          (workflow_engine_policy_applied ?workflow_engine)
        )
      )
    :effect
      (and
        (workflow_engine_policy_applied ?workflow_engine)
        (workflow_engine_has_policy ?workflow_engine ?policy)
        (not
          (policy_available ?policy)
        )
      )
  )
  (:action apply_policy_and_finalize_package
    :parameters (?workflow_engine - workflow_engine ?package - package ?work_unit - work_unit ?task_definition - task_definition ?policy - policy)
    :precondition
      (and
        (entity_activated ?workflow_engine)
        (workflow_engine_has_package ?workflow_engine ?package)
        (package_locked ?package)
        (package_assigned_to_work_unit ?package ?work_unit)
        (orchestration_operation_bound_to_task ?workflow_engine ?task_definition)
        (work_unit_input_artifact_attached ?work_unit)
        (workflow_engine_policy_applied ?workflow_engine)
        (workflow_engine_has_policy ?workflow_engine ?policy)
        (not
          (workflow_engine_validated_package ?workflow_engine)
        )
      )
    :effect
      (and
        (workflow_engine_validated_package ?workflow_engine)
        (workflow_engine_policy_validated ?workflow_engine)
      )
  )
  (:action apply_credential_and_prepare_workflow_engine_stage1
    :parameters (?workflow_engine - workflow_engine ?credential - credential ?external_connector - external_connector ?package - package ?work_unit - work_unit)
    :precondition
      (and
        (workflow_engine_validated_package ?workflow_engine)
        (workflow_engine_has_credential ?workflow_engine ?credential)
        (orchestration_operation_bound_to_connector ?workflow_engine ?external_connector)
        (workflow_engine_has_package ?workflow_engine ?package)
        (package_assigned_to_work_unit ?package ?work_unit)
        (not
          (work_unit_output_artifact_attached ?work_unit)
        )
        (not
          (workflow_engine_artifacts_attached ?workflow_engine)
        )
      )
    :effect (workflow_engine_artifacts_attached ?workflow_engine)
  )
  (:action apply_credential_and_prepare_workflow_engine_stage2
    :parameters (?workflow_engine - workflow_engine ?credential - credential ?external_connector - external_connector ?package - package ?work_unit - work_unit)
    :precondition
      (and
        (workflow_engine_validated_package ?workflow_engine)
        (workflow_engine_has_credential ?workflow_engine ?credential)
        (orchestration_operation_bound_to_connector ?workflow_engine ?external_connector)
        (workflow_engine_has_package ?workflow_engine ?package)
        (package_assigned_to_work_unit ?package ?work_unit)
        (work_unit_output_artifact_attached ?work_unit)
        (not
          (workflow_engine_artifacts_attached ?workflow_engine)
        )
      )
