  (:action create_query_session
    :parameters (?entity - entity)
    :precondition
      (and
        (not
          (session_created ?entity)
        )
        (not
          (selected ?entity)
        )
      )
    :effect (session_created ?entity)
  )
  (:action attach_filter_to_session
    :parameters (?entity - entity ?filter_token - filter_token)
    :precondition
      (and
        (session_created ?entity)
        (not
          (filter_attached ?entity)
        )
        (filter_token_available ?filter_token)
      )
    :effect
      (and
        (filter_attached ?entity)
        (subject_filter_binding ?entity ?filter_token)
        (not
          (filter_token_available ?filter_token)
        )
      )
  )
  (:action apply_attribute_constraint
    :parameters (?entity - entity ?attribute_constraint - attribute_constraint)
    :precondition
      (and
        (session_created ?entity)
        (filter_attached ?entity)
        (attribute_constraint_available ?attribute_constraint)
      )
    :effect
      (and
        (subject_matches_attribute ?entity ?attribute_constraint)
        (not
          (attribute_constraint_available ?attribute_constraint)
        )
      )
  )
  (:action validate_session
    :parameters (?entity - entity ?attribute_constraint - attribute_constraint)
    :precondition
      (and
        (session_created ?entity)
        (filter_attached ?entity)
        (subject_matches_attribute ?entity ?attribute_constraint)
        (not
          (validated ?entity)
        )
      )
    :effect (validated ?entity)
  )
  (:action release_attribute_constraint
    :parameters (?entity - entity ?attribute_constraint - attribute_constraint)
    :precondition
      (and
        (subject_matches_attribute ?entity ?attribute_constraint)
      )
    :effect
      (and
        (attribute_constraint_available ?attribute_constraint)
        (not
          (subject_matches_attribute ?entity ?attribute_constraint)
        )
      )
  )
  (:action attach_credential_to_session
    :parameters (?entity - entity ?auth_credential - auth_credential)

        (validated ?entity)

          (credential_available ?auth_credential)
        )
      )
  )
  (:action release_credential_from_session
    :parameters (?entity - entity ?auth_credential - auth_credential)
    :precondition
      (and
        (subject_credential_binding ?entity ?auth_credential)
      )
    :effect
      (and
        (credential_available ?auth_credential)
        (not
          (subject_credential_binding ?entity ?auth_credential)
