## Planning domain (PDDL)

(define (domain internal_data_export_formatting_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types record_subtype - object attribute_type - object asset_type - object domain_root - object source_record - domain_root format_slot - record_subtype field_definition - record_subtype binding_resource - record_subtype attachment_template - record_subtype presentation_style - record_subtype target_channel - record_subtype processing_module - record_subtype policy_module - record_subtype auxiliary_asset - attribute_type component_template - attribute_type policy_tag - attribute_type format_profile - asset_type layout_profile - asset_type export_file - asset_type record_category - source_record job_category - source_record producer_context - record_category consumer_context - record_category export_job - job_category)
  (:predicates
    (entity_registered ?source_record - source_record)
    (entity_prepared ?source_record - source_record)
    (entity_slot_assigned ?source_record - source_record)
    (entity_committed ?source_record - source_record)
    (entity_published ?source_record - source_record)
    (entity_scheduled_for_target ?source_record - source_record)
    (format_slot_available ?format_slot - format_slot)
    (entity_assigned_to_format_slot ?source_record - source_record ?format_slot - format_slot)
    (field_definition_available ?field_definition - field_definition)
    (entity_field_mapped ?source_record - source_record ?field_definition - field_definition)
    (binding_resource_available ?binding_resource - binding_resource)
    (entity_bound_to_resource ?source_record - source_record ?binding_resource - binding_resource)
    (auxiliary_asset_available ?auxiliary_asset - auxiliary_asset)
    (producer_has_auxiliary_asset ?producer_context - producer_context ?auxiliary_asset - auxiliary_asset)
    (consumer_has_auxiliary_asset ?consumer_context - consumer_context ?auxiliary_asset - auxiliary_asset)
    (context_assigned_format_profile ?producer_context - producer_context ?format_profile - format_profile)
    (format_profile_reserved ?format_profile - format_profile)
    (format_profile_populated ?format_profile - format_profile)
    (producer_ready_for_assembly ?producer_context - producer_context)
    (consumer_assigned_layout_profile ?consumer_context - consumer_context ?layout_profile - layout_profile)
    (layout_profile_reserved ?layout_profile - layout_profile)
    (layout_profile_populated ?layout_profile - layout_profile)
    (consumer_ready_for_assembly ?consumer_context - consumer_context)
    (export_file_available ?export_file - export_file)
    (export_file_reserved ?export_file - export_file)
    (export_file_associated_format_profile ?export_file - export_file ?format_profile - format_profile)
    (export_file_associated_layout_profile ?export_file - export_file ?layout_profile - layout_profile)
    (export_file_has_producer_content ?export_file - export_file)
    (export_file_has_consumer_content ?export_file - export_file)
    (export_file_sealed ?export_file - export_file)
    (job_linked_producer_context ?export_job - export_job ?producer_context - producer_context)
    (job_linked_consumer_context ?export_job - export_job ?consumer_context - consumer_context)
    (job_associated_export_file ?export_job - export_job ?export_file - export_file)
    (component_template_available ?component_template - component_template)
    (job_has_component_template ?export_job - export_job ?component_template - component_template)
    (component_template_in_use ?component_template - component_template)
    (component_template_rendered_to_file ?component_template - component_template ?export_file - export_file)
    (job_component_check_passed ?export_job - export_job)
    (job_module_attached ?export_job - export_job)
    (job_composition_ready ?export_job - export_job)
    (job_attachment_template_selected ?export_job - export_job)
    (job_attachment_linked ?export_job - export_job)
    (job_presentation_style_applied ?export_job - export_job)
    (job_components_confirmed ?export_job - export_job)
    (policy_tag_available ?policy_tag - policy_tag)
    (job_has_policy_tag ?export_job - export_job ?policy_tag - policy_tag)
    (job_policy_tag_selected ?export_job - export_job)
    (job_attachment_template_prepared ?export_job - export_job)
    (job_attachment_ready ?export_job - export_job)
    (attachment_template_available ?attachment_template - attachment_template)
    (job_has_attachment_template ?export_job - export_job ?attachment_template - attachment_template)
    (presentation_style_available ?presentation_style - presentation_style)
    (job_has_presentation_style ?export_job - export_job ?presentation_style - presentation_style)
    (processing_module_available ?processing_module - processing_module)
    (job_has_processing_module ?export_job - export_job ?processing_module - processing_module)
    (policy_module_available ?policy_module - policy_module)
    (job_has_policy_module ?export_job - export_job ?policy_module - policy_module)
    (target_channel_available ?target_channel - target_channel)
    (entity_assigned_to_target_channel ?source_record - source_record ?target_channel - target_channel)
    (producer_availability_signaled ?producer_context - producer_context)
    (consumer_availability_signaled ?consumer_context - consumer_context)
    (job_finalized ?export_job - export_job)
  )
  (:action register_source_record
    :parameters (?source_record - source_record)
    :precondition
      (and
        (not
          (entity_registered ?source_record)
        )
        (not
          (entity_committed ?source_record)
        )
      )
    :effect (entity_registered ?source_record)
  )
  (:action assign_format_slot_to_record
    :parameters (?source_record - source_record ?format_slot - format_slot)
    :precondition
      (and
        (entity_registered ?source_record)
        (not
          (entity_slot_assigned ?source_record)
        )
        (format_slot_available ?format_slot)
      )
    :effect
      (and
        (entity_slot_assigned ?source_record)
        (entity_assigned_to_format_slot ?source_record ?format_slot)
        (not
          (format_slot_available ?format_slot)
        )
      )
  )
  (:action map_field_definition_to_record
    :parameters (?source_record - source_record ?field_definition - field_definition)
    :precondition
      (and
        (entity_registered ?source_record)
        (entity_slot_assigned ?source_record)
        (field_definition_available ?field_definition)
      )
    :effect
      (and
        (entity_field_mapped ?source_record ?field_definition)
        (not
          (field_definition_available ?field_definition)
        )
      )
  )
  (:action confirm_record_mapping
    :parameters (?source_record - source_record ?field_definition - field_definition)
    :precondition
      (and
        (entity_registered ?source_record)
        (entity_slot_assigned ?source_record)
        (entity_field_mapped ?source_record ?field_definition)
        (not
          (entity_prepared ?source_record)
        )
      )
    :effect (entity_prepared ?source_record)
  )
  (:action unmap_field_from_record
    :parameters (?source_record - source_record ?field_definition - field_definition)
    :precondition
      (and
        (entity_field_mapped ?source_record ?field_definition)
      )
    :effect
      (and
        (field_definition_available ?field_definition)
        (not
          (entity_field_mapped ?source_record ?field_definition)
        )
      )
  )
  (:action attach_binding_resource_to_record
    :parameters (?source_record - source_record ?binding_resource - binding_resource)
    :precondition
      (and
        (entity_prepared ?source_record)
        (binding_resource_available ?binding_resource)
      )
    :effect
      (and
        (entity_bound_to_resource ?source_record ?binding_resource)
        (not
          (binding_resource_available ?binding_resource)
        )
      )
  )
  (:action detach_binding_resource_from_record
    :parameters (?source_record - source_record ?binding_resource - binding_resource)
    :precondition
      (and
        (entity_bound_to_resource ?source_record ?binding_resource)
      )
    :effect
      (and
        (binding_resource_available ?binding_resource)
        (not
          (entity_bound_to_resource ?source_record ?binding_resource)
        )
      )
  )
  (:action attach_processing_module_to_job
    :parameters (?export_job - export_job ?processing_module - processing_module)
    :precondition
      (and
        (entity_prepared ?export_job)
        (processing_module_available ?processing_module)
      )
    :effect
      (and
        (job_has_processing_module ?export_job ?processing_module)
        (not
          (processing_module_available ?processing_module)
        )
      )
  )
  (:action detach_processing_module_from_job
    :parameters (?export_job - export_job ?processing_module - processing_module)
    :precondition
      (and
        (job_has_processing_module ?export_job ?processing_module)
      )
    :effect
      (and
        (processing_module_available ?processing_module)
        (not
          (job_has_processing_module ?export_job ?processing_module)
        )
      )
  )
  (:action attach_policy_module_to_job
    :parameters (?export_job - export_job ?policy_module - policy_module)
    :precondition
      (and
        (entity_prepared ?export_job)
        (policy_module_available ?policy_module)
      )
    :effect
      (and
        (job_has_policy_module ?export_job ?policy_module)
        (not
          (policy_module_available ?policy_module)
        )
      )
  )
  (:action detach_policy_module_from_job
    :parameters (?export_job - export_job ?policy_module - policy_module)
    :precondition
      (and
        (job_has_policy_module ?export_job ?policy_module)
      )
    :effect
      (and
        (policy_module_available ?policy_module)
        (not
          (job_has_policy_module ?export_job ?policy_module)
        )
      )
  )
  (:action reserve_format_profile_for_producer_context
    :parameters (?producer_context - producer_context ?format_profile - format_profile ?field_definition - field_definition)
    :precondition
      (and
        (entity_prepared ?producer_context)
        (entity_field_mapped ?producer_context ?field_definition)
        (context_assigned_format_profile ?producer_context ?format_profile)
        (not
          (format_profile_reserved ?format_profile)
        )
        (not
          (format_profile_populated ?format_profile)
        )
      )
    :effect (format_profile_reserved ?format_profile)
  )
  (:action signal_producer_availability_claim_binding_resource
    :parameters (?producer_context - producer_context ?format_profile - format_profile ?binding_resource - binding_resource)
    :precondition
      (and
        (entity_prepared ?producer_context)
        (entity_bound_to_resource ?producer_context ?binding_resource)
        (context_assigned_format_profile ?producer_context ?format_profile)
        (format_profile_reserved ?format_profile)
        (not
          (producer_availability_signaled ?producer_context)
        )
      )
    :effect
      (and
        (producer_availability_signaled ?producer_context)
        (producer_ready_for_assembly ?producer_context)
      )
  )
  (:action collect_auxiliary_asset_for_producer_context
    :parameters (?producer_context - producer_context ?format_profile - format_profile ?auxiliary_asset - auxiliary_asset)
    :precondition
      (and
        (entity_prepared ?producer_context)
        (context_assigned_format_profile ?producer_context ?format_profile)
        (auxiliary_asset_available ?auxiliary_asset)
        (not
          (producer_availability_signaled ?producer_context)
        )
      )
    :effect
      (and
        (format_profile_populated ?format_profile)
        (producer_availability_signaled ?producer_context)
        (producer_has_auxiliary_asset ?producer_context ?auxiliary_asset)
        (not
          (auxiliary_asset_available ?auxiliary_asset)
        )
      )
  )
  (:action finalize_producer_asset_collection
    :parameters (?producer_context - producer_context ?format_profile - format_profile ?field_definition - field_definition ?auxiliary_asset - auxiliary_asset)
    :precondition
      (and
        (entity_prepared ?producer_context)
        (entity_field_mapped ?producer_context ?field_definition)
        (context_assigned_format_profile ?producer_context ?format_profile)
        (format_profile_populated ?format_profile)
        (producer_has_auxiliary_asset ?producer_context ?auxiliary_asset)
        (not
          (producer_ready_for_assembly ?producer_context)
        )
      )
    :effect
      (and
        (format_profile_reserved ?format_profile)
        (producer_ready_for_assembly ?producer_context)
        (auxiliary_asset_available ?auxiliary_asset)
        (not
          (producer_has_auxiliary_asset ?producer_context ?auxiliary_asset)
        )
      )
  )
  (:action reserve_layout_profile_for_consumer_context
    :parameters (?consumer_context - consumer_context ?layout_profile - layout_profile ?field_definition - field_definition)
    :precondition
      (and
        (entity_prepared ?consumer_context)
        (entity_field_mapped ?consumer_context ?field_definition)
        (consumer_assigned_layout_profile ?consumer_context ?layout_profile)
        (not
          (layout_profile_reserved ?layout_profile)
        )
        (not
          (layout_profile_populated ?layout_profile)
        )
      )
    :effect (layout_profile_reserved ?layout_profile)
  )
  (:action signal_consumer_availability_claim_binding_resource
    :parameters (?consumer_context - consumer_context ?layout_profile - layout_profile ?binding_resource - binding_resource)
    :precondition
      (and
        (entity_prepared ?consumer_context)
        (entity_bound_to_resource ?consumer_context ?binding_resource)
        (consumer_assigned_layout_profile ?consumer_context ?layout_profile)
        (layout_profile_reserved ?layout_profile)
        (not
          (consumer_availability_signaled ?consumer_context)
        )
      )
    :effect
      (and
        (consumer_availability_signaled ?consumer_context)
        (consumer_ready_for_assembly ?consumer_context)
      )
  )
  (:action collect_auxiliary_asset_for_consumer_context
    :parameters (?consumer_context - consumer_context ?layout_profile - layout_profile ?auxiliary_asset - auxiliary_asset)
    :precondition
      (and
        (entity_prepared ?consumer_context)
        (consumer_assigned_layout_profile ?consumer_context ?layout_profile)
        (auxiliary_asset_available ?auxiliary_asset)
        (not
          (consumer_availability_signaled ?consumer_context)
        )
      )
    :effect
      (and
        (layout_profile_populated ?layout_profile)
        (consumer_availability_signaled ?consumer_context)
        (consumer_has_auxiliary_asset ?consumer_context ?auxiliary_asset)
        (not
          (auxiliary_asset_available ?auxiliary_asset)
        )
      )
  )
  (:action finalize_consumer_asset_collection
    :parameters (?consumer_context - consumer_context ?layout_profile - layout_profile ?field_definition - field_definition ?auxiliary_asset - auxiliary_asset)
    :precondition
      (and
        (entity_prepared ?consumer_context)
        (entity_field_mapped ?consumer_context ?field_definition)
        (consumer_assigned_layout_profile ?consumer_context ?layout_profile)
        (layout_profile_populated ?layout_profile)
        (consumer_has_auxiliary_asset ?consumer_context ?auxiliary_asset)
        (not
          (consumer_ready_for_assembly ?consumer_context)
        )
      )
    :effect
      (and
        (layout_profile_reserved ?layout_profile)
        (consumer_ready_for_assembly ?consumer_context)
        (auxiliary_asset_available ?auxiliary_asset)
        (not
          (consumer_has_auxiliary_asset ?consumer_context ?auxiliary_asset)
        )
      )
  )
  (:action reserve_and_associate_export_file
    :parameters (?producer_context - producer_context ?consumer_context - consumer_context ?format_profile - format_profile ?layout_profile - layout_profile ?export_file - export_file)
    :precondition
      (and
        (producer_availability_signaled ?producer_context)
        (consumer_availability_signaled ?consumer_context)
        (context_assigned_format_profile ?producer_context ?format_profile)
        (consumer_assigned_layout_profile ?consumer_context ?layout_profile)
        (format_profile_reserved ?format_profile)
        (layout_profile_reserved ?layout_profile)
        (producer_ready_for_assembly ?producer_context)
        (consumer_ready_for_assembly ?consumer_context)
        (export_file_available ?export_file)
      )
    :effect
      (and
        (export_file_reserved ?export_file)
        (export_file_associated_format_profile ?export_file ?format_profile)
        (export_file_associated_layout_profile ?export_file ?layout_profile)
        (not
          (export_file_available ?export_file)
        )
      )
  )
  (:action reserve_and_populate_export_file_with_producer_content
    :parameters (?producer_context - producer_context ?consumer_context - consumer_context ?format_profile - format_profile ?layout_profile - layout_profile ?export_file - export_file)
    :precondition
      (and
        (producer_availability_signaled ?producer_context)
        (consumer_availability_signaled ?consumer_context)
        (context_assigned_format_profile ?producer_context ?format_profile)
        (consumer_assigned_layout_profile ?consumer_context ?layout_profile)
        (format_profile_populated ?format_profile)
        (layout_profile_reserved ?layout_profile)
        (not
          (producer_ready_for_assembly ?producer_context)
        )
        (consumer_ready_for_assembly ?consumer_context)
        (export_file_available ?export_file)
      )
    :effect
      (and
        (export_file_reserved ?export_file)
        (export_file_associated_format_profile ?export_file ?format_profile)
        (export_file_associated_layout_profile ?export_file ?layout_profile)
        (export_file_has_producer_content ?export_file)
        (not
          (export_file_available ?export_file)
        )
      )
  )
  (:action reserve_and_populate_export_file_with_consumer_content
    :parameters (?producer_context - producer_context ?consumer_context - consumer_context ?format_profile - format_profile ?layout_profile - layout_profile ?export_file - export_file)
    :precondition
      (and
        (producer_availability_signaled ?producer_context)
        (consumer_availability_signaled ?consumer_context)
        (context_assigned_format_profile ?producer_context ?format_profile)
        (consumer_assigned_layout_profile ?consumer_context ?layout_profile)
        (format_profile_reserved ?format_profile)
        (layout_profile_populated ?layout_profile)
        (producer_ready_for_assembly ?producer_context)
        (not
          (consumer_ready_for_assembly ?consumer_context)
        )
        (export_file_available ?export_file)
      )
    :effect
      (and
        (export_file_reserved ?export_file)
        (export_file_associated_format_profile ?export_file ?format_profile)
        (export_file_associated_layout_profile ?export_file ?layout_profile)
        (export_file_has_consumer_content ?export_file)
        (not
          (export_file_available ?export_file)
        )
      )
  )
  (:action reserve_and_populate_export_file_with_both_contents
    :parameters (?producer_context - producer_context ?consumer_context - consumer_context ?format_profile - format_profile ?layout_profile - layout_profile ?export_file - export_file)
    :precondition
      (and
        (producer_availability_signaled ?producer_context)
        (consumer_availability_signaled ?consumer_context)
        (context_assigned_format_profile ?producer_context ?format_profile)
        (consumer_assigned_layout_profile ?consumer_context ?layout_profile)
        (format_profile_populated ?format_profile)
        (layout_profile_populated ?layout_profile)
        (not
          (producer_ready_for_assembly ?producer_context)
        )
        (not
          (consumer_ready_for_assembly ?consumer_context)
        )
        (export_file_available ?export_file)
      )
    :effect
      (and
        (export_file_reserved ?export_file)
        (export_file_associated_format_profile ?export_file ?format_profile)
        (export_file_associated_layout_profile ?export_file ?layout_profile)
        (export_file_has_producer_content ?export_file)
        (export_file_has_consumer_content ?export_file)
        (not
          (export_file_available ?export_file)
        )
      )
  )
  (:action seal_export_file
    :parameters (?export_file - export_file ?producer_context - producer_context ?field_definition - field_definition)
    :precondition
      (and
        (export_file_reserved ?export_file)
        (producer_availability_signaled ?producer_context)
        (entity_field_mapped ?producer_context ?field_definition)
        (not
          (export_file_sealed ?export_file)
        )
      )
    :effect (export_file_sealed ?export_file)
  )
  (:action instantiate_component_template_into_file
    :parameters (?export_job - export_job ?component_template - component_template ?export_file - export_file)
    :precondition
      (and
        (entity_prepared ?export_job)
        (job_associated_export_file ?export_job ?export_file)
        (job_has_component_template ?export_job ?component_template)
        (component_template_available ?component_template)
        (export_file_reserved ?export_file)
        (export_file_sealed ?export_file)
        (not
          (component_template_in_use ?component_template)
        )
      )
    :effect
      (and
        (component_template_in_use ?component_template)
        (component_template_rendered_to_file ?component_template ?export_file)
        (not
          (component_template_available ?component_template)
        )
      )
  )
  (:action check_component_template_composition
    :parameters (?export_job - export_job ?component_template - component_template ?export_file - export_file ?field_definition - field_definition)
    :precondition
      (and
        (entity_prepared ?export_job)
        (job_has_component_template ?export_job ?component_template)
        (component_template_in_use ?component_template)
        (component_template_rendered_to_file ?component_template ?export_file)
        (entity_field_mapped ?export_job ?field_definition)
        (not
          (export_file_has_producer_content ?export_file)
        )
        (not
          (job_component_check_passed ?export_job)
        )
      )
    :effect (job_component_check_passed ?export_job)
  )
  (:action claim_attachment_template_for_job
    :parameters (?export_job - export_job ?attachment_template - attachment_template)
    :precondition
      (and
        (entity_prepared ?export_job)
        (attachment_template_available ?attachment_template)
        (not
          (job_attachment_template_selected ?export_job)
        )
      )
    :effect
      (and
        (job_attachment_template_selected ?export_job)
        (job_has_attachment_template ?export_job ?attachment_template)
        (not
          (attachment_template_available ?attachment_template)
        )
      )
  )
  (:action compose_component_with_attachment_and_mark
    :parameters (?export_job - export_job ?component_template - component_template ?export_file - export_file ?field_definition - field_definition ?attachment_template - attachment_template)
    :precondition
      (and
        (entity_prepared ?export_job)
        (job_has_component_template ?export_job ?component_template)
        (component_template_in_use ?component_template)
        (component_template_rendered_to_file ?component_template ?export_file)
        (entity_field_mapped ?export_job ?field_definition)
        (export_file_has_producer_content ?export_file)
        (job_attachment_template_selected ?export_job)
        (job_has_attachment_template ?export_job ?attachment_template)
        (not
          (job_component_check_passed ?export_job)
        )
      )
    :effect
      (and
        (job_component_check_passed ?export_job)
        (job_attachment_linked ?export_job)
      )
  )
  (:action attach_processing_module_and_apply_to_component
    :parameters (?export_job - export_job ?processing_module - processing_module ?binding_resource - binding_resource ?component_template - component_template ?export_file - export_file)
    :precondition
      (and
        (job_component_check_passed ?export_job)
        (job_has_processing_module ?export_job ?processing_module)
        (entity_bound_to_resource ?export_job ?binding_resource)
        (job_has_component_template ?export_job ?component_template)
        (component_template_rendered_to_file ?component_template ?export_file)
        (not
          (export_file_has_consumer_content ?export_file)
        )
        (not
          (job_module_attached ?export_job)
        )
      )
    :effect (job_module_attached ?export_job)
  )
  (:action finalize_processing_module_application
    :parameters (?export_job - export_job ?processing_module - processing_module ?binding_resource - binding_resource ?component_template - component_template ?export_file - export_file)
    :precondition
      (and
        (job_component_check_passed ?export_job)
        (job_has_processing_module ?export_job ?processing_module)
        (entity_bound_to_resource ?export_job ?binding_resource)
        (job_has_component_template ?export_job ?component_template)
        (component_template_rendered_to_file ?component_template ?export_file)
        (export_file_has_consumer_content ?export_file)
        (not
          (job_module_attached ?export_job)
        )
      )
    :effect (job_module_attached ?export_job)
  )
  (:action apply_policy_module_and_mark_component
    :parameters (?export_job - export_job ?policy_module - policy_module ?component_template - component_template ?export_file - export_file)
    :precondition
      (and
        (job_module_attached ?export_job)
        (job_has_policy_module ?export_job ?policy_module)
        (job_has_component_template ?export_job ?component_template)
        (component_template_rendered_to_file ?component_template ?export_file)
        (not
          (export_file_has_producer_content ?export_file)
        )
        (not
          (export_file_has_consumer_content ?export_file)
        )
        (not

        (consumer_ready_for_assembly ?consumer_context)

          (job_finalized ?export_job)

      (and
        (job_finalized ?export_job)
        (entity_published ?export_job)

          (job_finalized ?export_job)
