## Planning domain (PDDL)

(define (domain payment_provider_integration)
  (:requirements :strips :typing :negative-preconditions)
  (:types integration_resource_type - object artifact - object endpoint_type - object lifecycle_subject - object lifecycle_instance - lifecycle_subject connector_instance - integration_resource_type api_client_binding - integration_resource_type service_account - integration_resource_type feature_toggle - integration_resource_type operational_policy - integration_resource_type payment_instrument - integration_resource_type provider_capability - integration_resource_type partner_credential - integration_resource_type one_time_token - artifact certificate_artifact - artifact third_party_credential - artifact client_endpoint - endpoint_type server_endpoint - endpoint_type transaction_record - endpoint_type checkout_instance_class - lifecycle_instance service_instance_class - lifecycle_instance primary_checkout_instance - checkout_instance_class secondary_checkout_instance - checkout_instance_class checkout_service - service_instance_class)
  (:predicates
    (lifecycle_created ?lifecycle_subject - lifecycle_instance)
    (lifecycle_configured ?lifecycle_subject - lifecycle_instance)
    (lifecycle_connector_reserved ?lifecycle_subject - lifecycle_instance)
    (lifecycle_finalized ?lifecycle_subject - lifecycle_instance)
    (lifecycle_completed ?lifecycle_subject - lifecycle_instance)
    (lifecycle_reconciled ?lifecycle_subject - lifecycle_instance)
    (connector_available ?connector - connector_instance)
    (lifecycle_assigned_connector ?lifecycle_subject - lifecycle_instance ?connector - connector_instance)
    (api_client_available ?api_client - api_client_binding)
    (lifecycle_bound_api_client ?lifecycle_subject - lifecycle_instance ?api_client - api_client_binding)
    (service_account_available ?service_account - service_account)
    (lifecycle_bound_service_account ?lifecycle_subject - lifecycle_instance ?service_account - service_account)
    (one_time_token_available ?one_time_token - one_time_token)
    (primary_checkout_has_token ?primary_checkout - primary_checkout_instance ?one_time_token - one_time_token)
    (secondary_checkout_has_token ?secondary_checkout - secondary_checkout_instance ?one_time_token - one_time_token)
    (primary_checkout_client_endpoint ?primary_checkout - primary_checkout_instance ?client_endpoint - client_endpoint)
    (client_endpoint_ready ?client_endpoint - client_endpoint)
    (client_endpoint_token_bound ?client_endpoint - client_endpoint)
    (primary_checkout_client_endpoint_confirmed ?primary_checkout - primary_checkout_instance)
    (secondary_checkout_server_endpoint ?secondary_checkout - secondary_checkout_instance ?server_endpoint - server_endpoint)
    (server_endpoint_ready ?server_endpoint - server_endpoint)
    (server_endpoint_token_bound ?server_endpoint - server_endpoint)
    (secondary_checkout_server_endpoint_confirmed ?secondary_checkout - secondary_checkout_instance)
    (transaction_pending ?transaction - transaction_record)
    (transaction_dispatched ?transaction - transaction_record)
    (transaction_attached_client_endpoint ?transaction - transaction_record ?client_endpoint - client_endpoint)
    (transaction_attached_server_endpoint ?transaction - transaction_record ?server_endpoint - server_endpoint)
    (transaction_client_processed ?transaction - transaction_record)
    (transaction_server_processed ?transaction - transaction_record)
    (transaction_committed ?transaction - transaction_record)
    (service_associated_with_primary_checkout ?checkout_service - checkout_service ?primary_checkout - primary_checkout_instance)
    (service_associated_with_secondary_checkout ?checkout_service - checkout_service ?secondary_checkout - secondary_checkout_instance)
    (service_has_transaction ?checkout_service - checkout_service ?transaction - transaction_record)
    (certificate_available ?certificate - certificate_artifact)
    (service_has_certificate ?checkout_service - checkout_service ?certificate - certificate_artifact)
    (certificate_installed ?certificate - certificate_artifact)
    (certificate_attached_to_transaction ?certificate - certificate_artifact ?transaction - transaction_record)
    (service_certificate_validated ?checkout_service - checkout_service)
    (service_certificate_provisioned ?checkout_service - checkout_service)
    (service_capability_binding_ready ?checkout_service - checkout_service)
    (service_feature_applied ?checkout_service - checkout_service)
    (service_feature_committed ?checkout_service - checkout_service)
    (service_policy_attached ?checkout_service - checkout_service)
    (service_provisioning_finalized ?checkout_service - checkout_service)
    (third_party_credential_available ?third_party_credential - third_party_credential)
    (service_has_third_party_credential ?checkout_service - checkout_service ?third_party_credential - third_party_credential)
    (service_third_party_credential_bound ?checkout_service - checkout_service)
    (service_third_party_credential_verified ?checkout_service - checkout_service)
    (service_third_party_credential_committed ?checkout_service - checkout_service)
    (feature_toggle_available ?feature_toggle - feature_toggle)
    (service_has_feature_toggle ?checkout_service - checkout_service ?feature_toggle - feature_toggle)
    (operational_policy_available ?operational_policy - operational_policy)
    (service_has_operational_policy ?checkout_service - checkout_service ?operational_policy - operational_policy)
    (provider_capability_available ?provider_capability - provider_capability)
    (service_has_provider_capability ?checkout_service - checkout_service ?provider_capability - provider_capability)
    (partner_credential_available ?partner_credential - partner_credential)
    (service_has_partner_credential ?checkout_service - checkout_service ?partner_credential - partner_credential)
    (payment_instrument_available ?payment_instrument - payment_instrument)
    (lifecycle_has_payment_instrument ?lifecycle_subject - lifecycle_instance ?payment_instrument - payment_instrument)
    (primary_checkout_endpoint_processed ?primary_checkout - primary_checkout_instance)
    (secondary_checkout_endpoint_processed ?secondary_checkout - secondary_checkout_instance)
    (service_provisioned ?checkout_service - checkout_service)
  )
  (:action create_lifecycle_instance
    :parameters (?lifecycle_subject - lifecycle_instance)
    :precondition
      (and
        (not
          (lifecycle_created ?lifecycle_subject)
        )
        (not
          (lifecycle_finalized ?lifecycle_subject)
        )
      )
    :effect (lifecycle_created ?lifecycle_subject)
  )
  (:action reserve_connector_for_lifecycle
    :parameters (?lifecycle_subject - lifecycle_instance ?connector - connector_instance)
    :precondition
      (and
        (lifecycle_created ?lifecycle_subject)
        (not
          (lifecycle_connector_reserved ?lifecycle_subject)
        )
        (connector_available ?connector)
      )
    :effect
      (and
        (lifecycle_connector_reserved ?lifecycle_subject)
        (lifecycle_assigned_connector ?lifecycle_subject ?connector)
        (not
          (connector_available ?connector)
        )
      )
  )
  (:action bind_api_client_to_lifecycle
    :parameters (?lifecycle_subject - lifecycle_instance ?api_client - api_client_binding)
    :precondition
      (and
        (lifecycle_created ?lifecycle_subject)
        (lifecycle_connector_reserved ?lifecycle_subject)
        (api_client_available ?api_client)
      )
    :effect
      (and
        (lifecycle_bound_api_client ?lifecycle_subject ?api_client)
        (not
          (api_client_available ?api_client)
        )
      )
  )
  (:action mark_lifecycle_configured
    :parameters (?lifecycle_subject - lifecycle_instance ?api_client - api_client_binding)
    :precondition
      (and
        (lifecycle_created ?lifecycle_subject)
        (lifecycle_connector_reserved ?lifecycle_subject)
        (lifecycle_bound_api_client ?lifecycle_subject ?api_client)
        (not
          (lifecycle_configured ?lifecycle_subject)
        )
      )
    :effect (lifecycle_configured ?lifecycle_subject)
  )
  (:action release_api_client_binding
    :parameters (?lifecycle_subject - lifecycle_instance ?api_client - api_client_binding)
    :precondition
      (and
        (lifecycle_bound_api_client ?lifecycle_subject ?api_client)
      )
    :effect
      (and
        (api_client_available ?api_client)
        (not
          (lifecycle_bound_api_client ?lifecycle_subject ?api_client)
        )
      )
  )
  (:action bind_service_account_to_lifecycle
    :parameters (?lifecycle_subject - lifecycle_instance ?service_account - service_account)

  (:action start_client_endpoint_negotiation
    :parameters (?primary_checkout - primary_checkout_instance ?client_endpoint - client_endpoint ?api_client - api_client_binding)

        (lifecycle_configured ?primary_checkout)
        (lifecycle_bound_api_client ?primary_checkout ?api_client)
        (primary_checkout_client_endpoint ?primary_checkout ?client_endpoint)
        (not
          (client_endpoint_ready ?client_endpoint)
        )
        (not
          (client_endpoint_token_bound ?client_endpoint)
        )
      )
    :effect (client_endpoint_ready ?client_endpoint)
  )
  (:action complete_client_endpoint_handshake
    :parameters (?primary_checkout - primary_checkout_instance ?client_endpoint - client_endpoint ?service_account - service_account)
    :precondition

        (lifecycle_configured ?primary_checkout)
        (lifecycle_bound_service_account ?primary_checkout ?service_account)
        (primary_checkout_client_endpoint ?primary_checkout ?client_endpoint)
        (client_endpoint_ready ?client_endpoint)
        (not
          (primary_checkout_endpoint_processed ?primary_checkout)
        )
      )
    :effect
      (and
        (primary_checkout_endpoint_processed ?primary_checkout)
        (primary_checkout_client_endpoint_confirmed ?primary_checkout)
      )
  )
  (:action bind_one_time_token_to_client_endpoint
    :parameters (?primary_checkout - primary_checkout_instance ?client_endpoint - client_endpoint ?one_time_token - one_time_token)

        )

    :effect
      (and
        (client_endpoint_token_bound ?client_endpoint)
        (primary_checkout_endpoint_processed ?primary_checkout)

          (one_time_token_available ?one_time_token)

          (primary_checkout_has_token ?primary_checkout ?one_time_token)

  (:action route_transaction_standard
    :parameters (?primary_checkout - primary_checkout_instance ?secondary_checkout - secondary_checkout_instance ?client_endpoint - client_endpoint ?server_endpoint - server_endpoint ?transaction - transaction_record)
    :precondition
      (and
        (primary_checkout_endpoint_processed ?primary_checkout)
        (secondary_checkout_endpoint_processed ?secondary_checkout)
        (primary_checkout_client_endpoint ?primary_checkout ?client_endpoint)
        (secondary_checkout_server_endpoint ?secondary_checkout ?server_endpoint)
        (client_endpoint_ready ?client_endpoint)
        (server_endpoint_ready ?server_endpoint)
        (primary_checkout_client_endpoint_confirmed ?primary_checkout)
        (secondary_checkout_server_endpoint_confirmed ?secondary_checkout)
        (transaction_pending ?transaction)
      )
    :effect
      (and
        (transaction_dispatched ?transaction)
        (transaction_attached_client_endpoint ?transaction ?client_endpoint)
        (transaction_attached_server_endpoint ?transaction ?server_endpoint)
        (not
          (transaction_pending ?transaction)
        )
      )
  )
  (:action route_transaction_client_token
    :parameters (?primary_checkout - primary_checkout_instance ?secondary_checkout - secondary_checkout_instance ?client_endpoint - client_endpoint ?server_endpoint - server_endpoint ?transaction - transaction_record)
    :precondition
      (and
        (primary_checkout_endpoint_processed ?primary_checkout)
        (secondary_checkout_endpoint_processed ?secondary_checkout)

        (secondary_checkout_server_endpoint ?secondary_checkout ?server_endpoint)
        (client_endpoint_token_bound ?client_endpoint)
        (server_endpoint_ready ?server_endpoint)
        (not
          (primary_checkout_client_endpoint_confirmed ?primary_checkout)
        )
        (secondary_checkout_server_endpoint_confirmed ?secondary_checkout)
        (transaction_pending ?transaction)
      )
    :effect
      (and
        (transaction_dispatched ?transaction)
        (transaction_attached_client_endpoint ?transaction ?client_endpoint)
        (transaction_attached_server_endpoint ?transaction ?server_endpoint)
        (transaction_client_processed ?transaction)
        (not
          (transaction_pending ?transaction)
        )
      )
  )

      (and
        (lifecycle_configured ?checkout_service)

        (certificate_installed ?certificate)
        (certificate_attached_to_transaction ?certificate ?transaction)
        (lifecycle_bound_api_client ?checkout_service ?api_client)
        (transaction_client_processed ?transaction)
        (service_feature_applied ?checkout_service)
        (service_has_feature_toggle ?checkout_service ?feature_toggle)
        (not
          (service_certificate_validated ?checkout_service)

    :effect
      (and
        (service_certificate_validated ?checkout_service)

  )

    :parameters (?checkout_service - checkout_service ?provider_capability - provider_capability ?service_account - service_account ?certificate - certificate_artifact ?transaction - transaction_record)
    :precondition
      (and
        (service_certificate_validated ?checkout_service)
        (service_has_provider_capability ?checkout_service ?provider_capability)
        (lifecycle_bound_service_account ?checkout_service ?service_account)
        (service_has_certificate ?checkout_service ?certificate)
        (certificate_attached_to_transaction ?certificate ?transaction)
        (not
          (transaction_server_processed ?transaction)
        )
        (not
          (service_certificate_provisioned ?checkout_service)
        )
      )
    :effect (service_certificate_provisioned ?checkout_service)
  )
  (:action provision_service_with_capability_alt
    :parameters (?checkout_service - checkout_service ?provider_capability - provider_capability ?service_account - service_account ?certificate - certificate_artifact ?transaction - transaction_record)
    :precondition
      (and
        (service_certificate_validated ?checkout_service)
