## Planning domain (PDDL)

(define (domain top_k_selection_by_scoring_function_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types candidate_group - object worker_group - object score_dimension - object input_set - object candidate - input_set resource_token - candidate_group score_function - candidate_group feature_extractor - candidate_group preference_label - candidate_group tie_breaker - candidate_group metadata - candidate_group attribute_primary - candidate_group attribute_secondary - candidate_group temporary_score - worker_group priority_tag - worker_group provenance - worker_group score_key_primary - score_dimension score_key_secondary - score_dimension aggregated_slot - score_dimension candidate_subgroup - candidate candidate_bucket - candidate partition_primary - candidate_subgroup partition_secondary - candidate_subgroup result_container - candidate_bucket)
  (:predicates
    (entity_registered ?candidate - candidate)
    (entity_scored ?candidate - candidate)
    (entity_has_token ?candidate - candidate)
    (published ?candidate - candidate)
    (adjudicated ?candidate - candidate)
    (commit_ready ?candidate - candidate)
    (token_available ?resource_token - resource_token)
    (assigned_token ?candidate - candidate ?resource_token - resource_token)
    (score_function_available ?score_function - score_function)
    (assigned_score_function ?candidate - candidate ?score_function - score_function)
    (feature_extractor_available ?feature_extractor - feature_extractor)
    (bound_feature_extractor ?candidate - candidate ?feature_extractor - feature_extractor)
    (temporary_score_available ?temporary_score - temporary_score)
    (assigned_temp_score_primary ?partition_a - partition_primary ?temporary_score - temporary_score)
    (assigned_temp_score_secondary ?partition_b - partition_secondary ?temporary_score - temporary_score)
    (partition_has_score_key_primary ?partition_a - partition_primary ?score_key_a - score_key_primary)
    (score_key_primary_flag_primary ?score_key_a - score_key_primary)
    (score_key_primary_flag_secondary ?score_key_a - score_key_primary)
    (partition_primary_ready ?partition_a - partition_primary)
    (partition_has_score_key_secondary ?partition_b - partition_secondary ?score_key_b - score_key_secondary)
    (score_key_secondary_flag_primary ?score_key_b - score_key_secondary)
    (score_key_secondary_flag_secondary ?score_key_b - score_key_secondary)
    (partition_secondary_ready ?partition_b - partition_secondary)
    (slot_available ?aggregated_slot - aggregated_slot)
    (slot_reserved ?aggregated_slot - aggregated_slot)
    (slot_has_score_key_primary ?aggregated_slot - aggregated_slot ?score_key_a - score_key_primary)
    (slot_has_score_key_secondary ?aggregated_slot - aggregated_slot ?score_key_b - score_key_secondary)
    (slot_mark_primary ?aggregated_slot - aggregated_slot)
    (slot_mark_secondary ?aggregated_slot - aggregated_slot)
    (slot_validated ?aggregated_slot - aggregated_slot)
    (container_associated_with_partition_primary ?result_container - result_container ?partition_a - partition_primary)
    (container_associated_with_partition_secondary ?result_container - result_container ?partition_b - partition_secondary)
    (container_has_aggregated_slot ?result_container - result_container ?aggregated_slot - aggregated_slot)
    (priority_tag_available ?priority_tag - priority_tag)
    (container_has_priority_tag ?result_container - result_container ?priority_tag - priority_tag)
    (priority_tag_consumed ?priority_tag - priority_tag)
    (priority_tag_assigned_to_slot ?priority_tag - priority_tag ?aggregated_slot - aggregated_slot)
    (container_stage1_locked ?result_container - result_container)
    (container_stage2_locked ?result_container - result_container)
    (container_adjudication_candidate ?result_container - result_container)
    (preference_label_attached ?result_container - result_container)
    (preference_label_confirmed ?result_container - result_container)
    (adjudication_marker_set ?result_container - result_container)
    (adjudication_lock_acquired ?result_container - result_container)
    (provenance_available ?provenance - provenance)
    (container_has_provenance ?result_container - result_container ?provenance - provenance)
    (provenance_bound ?result_container - result_container)
    (provenance_acknowledged ?result_container - result_container)
    (provenance_validated ?result_container - result_container)
    (preference_label_available ?preference_label - preference_label)
    (container_has_preference_label ?result_container - result_container ?preference_label - preference_label)
    (tie_breaker_available ?tie_breaker - tie_breaker)
    (container_has_tie_breaker ?result_container - result_container ?tie_breaker - tie_breaker)
    (attribute_primary_available ?attribute_a - attribute_primary)
    (container_has_attribute_primary ?result_container - result_container ?attribute_a - attribute_primary)
    (attribute_secondary_available ?attribute_b - attribute_secondary)
    (container_has_attribute_secondary ?result_container - result_container ?attribute_b - attribute_secondary)
    (metadata_available ?metadata - metadata)
    (entity_has_metadata ?candidate - candidate ?metadata - metadata)
    (partition_primary_processed ?partition_a - partition_primary)
    (partition_secondary_processed ?partition_b - partition_secondary)
    (container_committed ?result_container - result_container)
  )
  (:action register_candidate
    :parameters (?candidate - candidate)
    :precondition
      (and
        (not
          (entity_registered ?candidate)
        )
        (not
          (published ?candidate)
        )
      )
    :effect (entity_registered ?candidate)
  )
  (:action allocate_resource_token
    :parameters (?candidate - candidate ?resource_token - resource_token)
    :precondition
      (and
        (entity_registered ?candidate)
        (not
          (entity_has_token ?candidate)
        )
        (token_available ?resource_token)
      )
    :effect
      (and
        (entity_has_token ?candidate)
        (assigned_token ?candidate ?resource_token)
        (not
          (token_available ?resource_token)
        )
      )
  )
  (:action assign_score_function
    :parameters (?candidate - candidate ?score_function - score_function)
    :precondition
      (and
        (entity_registered ?candidate)
        (entity_has_token ?candidate)
        (score_function_available ?score_function)
      )
    :effect
      (and
        (assigned_score_function ?candidate ?score_function)
        (not
          (score_function_available ?score_function)
        )
      )
  )
  (:action confirm_score
    :parameters (?candidate - candidate ?score_function - score_function)
    :precondition
      (and
        (entity_registered ?candidate)
        (entity_has_token ?candidate)
        (assigned_score_function ?candidate ?score_function)
        (not
          (entity_scored ?candidate)
        )
      )
    :effect (entity_scored ?candidate)
  )
  (:action release_score_function
    :parameters (?candidate - candidate ?score_function - score_function)
    :precondition
      (and
        (assigned_score_function ?candidate ?score_function)
      )
    :effect
      (and
        (score_function_available ?score_function)
        (not
          (assigned_score_function ?candidate ?score_function)
        )
      )
  )
  (:action bind_feature_extractor
    :parameters (?candidate - candidate ?feature_extractor - feature_extractor)
    :precondition
      (and
        (entity_scored ?candidate)
        (feature_extractor_available ?feature_extractor)
      )
    :effect
      (and
        (bound_feature_extractor ?candidate ?feature_extractor)
        (not
          (feature_extractor_available ?feature_extractor)
        )
      )
  )
  (:action unbind_feature_extractor
    :parameters (?candidate - candidate ?feature_extractor - feature_extractor)
    :precondition
      (and
        (bound_feature_extractor ?candidate ?feature_extractor)
      )
    :effect
      (and
        (feature_extractor_available ?feature_extractor)
        (not
          (bound_feature_extractor ?candidate ?feature_extractor)
        )
      )
  )
  (:action reserve_attribute_a
    :parameters (?result_container - result_container ?attribute_a - attribute_primary)
    :precondition
      (and
        (entity_scored ?result_container)
        (attribute_primary_available ?attribute_a)
      )
    :effect
      (and
        (container_has_attribute_primary ?result_container ?attribute_a)
        (not
          (attribute_primary_available ?attribute_a)
        )
      )
  )
  (:action release_attribute_a
    :parameters (?result_container - result_container ?attribute_a - attribute_primary)
    :precondition
      (and
        (container_has_attribute_primary ?result_container ?attribute_a)
      )
    :effect
      (and
        (attribute_primary_available ?attribute_a)
        (not
          (container_has_attribute_primary ?result_container ?attribute_a)
        )
      )
  )
  (:action reserve_attribute_b
    :parameters (?result_container - result_container ?attribute_b - attribute_secondary)
    :precondition
      (and
        (entity_scored ?result_container)
        (attribute_secondary_available ?attribute_b)
      )
    :effect
      (and
        (container_has_attribute_secondary ?result_container ?attribute_b)
        (not
          (attribute_secondary_available ?attribute_b)
        )
      )
  )
  (:action release_attribute_b
    :parameters (?result_container - result_container ?attribute_b - attribute_secondary)
    :precondition
      (and
        (container_has_attribute_secondary ?result_container ?attribute_b)
      )
    :effect
      (and
        (attribute_secondary_available ?attribute_b)
        (not
          (container_has_attribute_secondary ?result_container ?attribute_b)
        )
      )
  )
  (:action claim_score_key_primary
    :parameters (?partition_a - partition_primary ?score_key_a - score_key_primary ?score_function - score_function)
    :precondition
      (and
        (entity_scored ?partition_a)
        (assigned_score_function ?partition_a ?score_function)
        (partition_has_score_key_primary ?partition_a ?score_key_a)
        (not
          (score_key_primary_flag_primary ?score_key_a)
        )
        (not
          (score_key_primary_flag_secondary ?score_key_a)
        )
      )
    :effect (score_key_primary_flag_primary ?score_key_a)
  )
  (:action confirm_partition_primary_readiness
    :parameters (?partition_a - partition_primary ?score_key_a - score_key_primary ?feature_extractor - feature_extractor)
    :precondition
      (and
        (entity_scored ?partition_a)
        (bound_feature_extractor ?partition_a ?feature_extractor)
        (partition_has_score_key_primary ?partition_a ?score_key_a)
        (score_key_primary_flag_primary ?score_key_a)
        (not
          (partition_primary_processed ?partition_a)
        )
      )
    :effect
      (and
        (partition_primary_processed ?partition_a)
        (partition_primary_ready ?partition_a)
      )
  )
  (:action apply_temp_score_to_partition_primary
    :parameters (?partition_a - partition_primary ?score_key_a - score_key_primary ?temporary_score - temporary_score)
    :precondition
      (and
        (entity_scored ?partition_a)
        (partition_has_score_key_primary ?partition_a ?score_key_a)
        (temporary_score_available ?temporary_score)
        (not
          (partition_primary_processed ?partition_a)
        )
      )
    :effect
      (and
        (score_key_primary_flag_secondary ?score_key_a)
        (partition_primary_processed ?partition_a)
        (assigned_temp_score_primary ?partition_a ?temporary_score)
        (not
          (temporary_score_available ?temporary_score)
        )
      )
  )
  (:action integrate_temp_score_partition_primary
    :parameters (?partition_a - partition_primary ?score_key_a - score_key_primary ?score_function - score_function ?temporary_score - temporary_score)
    :precondition
      (and
        (entity_scored ?partition_a)
        (assigned_score_function ?partition_a ?score_function)
        (partition_has_score_key_primary ?partition_a ?score_key_a)
        (score_key_primary_flag_secondary ?score_key_a)
        (assigned_temp_score_primary ?partition_a ?temporary_score)
        (not
          (partition_primary_ready ?partition_a)
        )
      )
    :effect
      (and
        (score_key_primary_flag_primary ?score_key_a)
        (partition_primary_ready ?partition_a)
        (temporary_score_available ?temporary_score)
        (not
          (assigned_temp_score_primary ?partition_a ?temporary_score)
        )
      )
  )
  (:action claim_score_key_secondary
    :parameters (?partition_b - partition_secondary ?score_key_b - score_key_secondary ?score_function - score_function)
    :precondition
      (and
        (entity_scored ?partition_b)
        (assigned_score_function ?partition_b ?score_function)
        (partition_has_score_key_secondary ?partition_b ?score_key_b)
        (not
          (score_key_secondary_flag_primary ?score_key_b)
        )
        (not
          (score_key_secondary_flag_secondary ?score_key_b)
        )
      )
    :effect (score_key_secondary_flag_primary ?score_key_b)
  )
  (:action confirm_partition_secondary_readiness
    :parameters (?partition_b - partition_secondary ?score_key_b - score_key_secondary ?feature_extractor - feature_extractor)
    :precondition
      (and
        (entity_scored ?partition_b)
        (bound_feature_extractor ?partition_b ?feature_extractor)
        (partition_has_score_key_secondary ?partition_b ?score_key_b)
        (score_key_secondary_flag_primary ?score_key_b)
        (not
          (partition_secondary_processed ?partition_b)
        )
      )
    :effect
      (and
        (partition_secondary_processed ?partition_b)
        (partition_secondary_ready ?partition_b)
      )
  )
  (:action apply_temp_score_to_partition_secondary
    :parameters (?partition_b - partition_secondary ?score_key_b - score_key_secondary ?temporary_score - temporary_score)
    :precondition
      (and
        (entity_scored ?partition_b)
        (partition_has_score_key_secondary ?partition_b ?score_key_b)
        (temporary_score_available ?temporary_score)
        (not
          (partition_secondary_processed ?partition_b)
        )
      )
    :effect
      (and
        (score_key_secondary_flag_secondary ?score_key_b)
        (partition_secondary_processed ?partition_b)
        (assigned_temp_score_secondary ?partition_b ?temporary_score)
        (not
          (temporary_score_available ?temporary_score)
        )
      )
  )
  (:action integrate_temp_score_partition_secondary
    :parameters (?partition_b - partition_secondary ?score_key_b - score_key_secondary ?score_function - score_function ?temporary_score - temporary_score)
    :precondition
      (and
        (entity_scored ?partition_b)
        (assigned_score_function ?partition_b ?score_function)
        (partition_has_score_key_secondary ?partition_b ?score_key_b)
        (score_key_secondary_flag_secondary ?score_key_b)
        (assigned_temp_score_secondary ?partition_b ?temporary_score)
        (not
          (partition_secondary_ready ?partition_b)
        )
      )
    :effect
      (and
        (score_key_secondary_flag_primary ?score_key_b)
        (partition_secondary_ready ?partition_b)
        (temporary_score_available ?temporary_score)
        (not
          (assigned_temp_score_secondary ?partition_b ?temporary_score)
        )
      )
  )
  (:action allocate_aggregated_slot
    :parameters (?partition_a - partition_primary ?partition_b - partition_secondary ?score_key_a - score_key_primary ?score_key_b - score_key_secondary ?aggregated_slot - aggregated_slot)
    :precondition
      (and
        (partition_primary_processed ?partition_a)
        (partition_secondary_processed ?partition_b)
        (partition_has_score_key_primary ?partition_a ?score_key_a)
        (partition_has_score_key_secondary ?partition_b ?score_key_b)
        (score_key_primary_flag_primary ?score_key_a)
        (score_key_secondary_flag_primary ?score_key_b)
        (partition_primary_ready ?partition_a)
        (partition_secondary_ready ?partition_b)
        (slot_available ?aggregated_slot)
      )
    :effect
      (and
        (slot_reserved ?aggregated_slot)
        (slot_has_score_key_primary ?aggregated_slot ?score_key_a)
        (slot_has_score_key_secondary ?aggregated_slot ?score_key_b)
        (not
          (slot_available ?aggregated_slot)
        )
      )
  )
  (:action allocate_aggregated_slot_with_mark_primary
    :parameters (?partition_a - partition_primary ?partition_b - partition_secondary ?score_key_a - score_key_primary ?score_key_b - score_key_secondary ?aggregated_slot - aggregated_slot)
    :precondition
      (and
        (partition_primary_processed ?partition_a)
        (partition_secondary_processed ?partition_b)
        (partition_has_score_key_primary ?partition_a ?score_key_a)
        (partition_has_score_key_secondary ?partition_b ?score_key_b)
        (score_key_primary_flag_secondary ?score_key_a)
        (score_key_secondary_flag_primary ?score_key_b)
        (not
          (partition_primary_ready ?partition_a)
        )
        (partition_secondary_ready ?partition_b)
        (slot_available ?aggregated_slot)
      )
    :effect
      (and
        (slot_reserved ?aggregated_slot)
        (slot_has_score_key_primary ?aggregated_slot ?score_key_a)
        (slot_has_score_key_secondary ?aggregated_slot ?score_key_b)
        (slot_mark_primary ?aggregated_slot)
        (not
          (slot_available ?aggregated_slot)
        )
      )
  )
  (:action allocate_aggregated_slot_with_mark_secondary
    :parameters (?partition_a - partition_primary ?partition_b - partition_secondary ?score_key_a - score_key_primary ?score_key_b - score_key_secondary ?aggregated_slot - aggregated_slot)
    :precondition
      (and
        (partition_primary_processed ?partition_a)
        (partition_secondary_processed ?partition_b)
        (partition_has_score_key_primary ?partition_a ?score_key_a)
        (partition_has_score_key_secondary ?partition_b ?score_key_b)
        (score_key_primary_flag_primary ?score_key_a)
        (score_key_secondary_flag_secondary ?score_key_b)
        (partition_primary_ready ?partition_a)
        (not
          (partition_secondary_ready ?partition_b)
        )
        (slot_available ?aggregated_slot)
      )
    :effect
      (and
        (slot_reserved ?aggregated_slot)
        (slot_has_score_key_primary ?aggregated_slot ?score_key_a)
        (slot_has_score_key_secondary ?aggregated_slot ?score_key_b)
        (slot_mark_secondary ?aggregated_slot)
        (not
          (slot_available ?aggregated_slot)
        )
      )
  )
  (:action allocate_aggregated_slot_with_marks_primary_secondary
    :parameters (?partition_a - partition_primary ?partition_b - partition_secondary ?score_key_a - score_key_primary ?score_key_b - score_key_secondary ?aggregated_slot - aggregated_slot)
    :precondition
      (and
        (partition_primary_processed ?partition_a)
        (partition_secondary_processed ?partition_b)
        (partition_has_score_key_primary ?partition_a ?score_key_a)
        (partition_has_score_key_secondary ?partition_b ?score_key_b)
        (score_key_primary_flag_secondary ?score_key_a)
        (score_key_secondary_flag_secondary ?score_key_b)
        (not
          (partition_primary_ready ?partition_a)
        )
        (not
          (partition_secondary_ready ?partition_b)
        )
        (slot_available ?aggregated_slot)
      )
    :effect
      (and
        (slot_reserved ?aggregated_slot)
        (slot_has_score_key_primary ?aggregated_slot ?score_key_a)
        (slot_has_score_key_secondary ?aggregated_slot ?score_key_b)
        (slot_mark_primary ?aggregated_slot)
        (slot_mark_secondary ?aggregated_slot)
        (not
          (slot_available ?aggregated_slot)
        )
      )
  )
  (:action validate_slot_with_partition_primary
    :parameters (?aggregated_slot - aggregated_slot ?partition_a - partition_primary ?score_function - score_function)
    :precondition
      (and
        (slot_reserved ?aggregated_slot)
        (partition_primary_processed ?partition_a)
        (assigned_score_function ?partition_a ?score_function)
        (not
          (slot_validated ?aggregated_slot)
        )
      )
    :effect (slot_validated ?aggregated_slot)
  )
  (:action assign_priority_tag_to_slot
    :parameters (?result_container - result_container ?priority_tag - priority_tag ?aggregated_slot - aggregated_slot)
    :precondition
      (and
        (entity_scored ?result_container)
        (container_has_aggregated_slot ?result_container ?aggregated_slot)
        (container_has_priority_tag ?result_container ?priority_tag)
        (priority_tag_available ?priority_tag)
        (slot_reserved ?aggregated_slot)
        (slot_validated ?aggregated_slot)
        (not
          (priority_tag_consumed ?priority_tag)
        )
      )
    :effect
      (and
        (priority_tag_consumed ?priority_tag)
        (priority_tag_assigned_to_slot ?priority_tag ?aggregated_slot)
        (not
          (priority_tag_available ?priority_tag)
        )
      )
  )
  (:action lock_container_stage1
    :parameters (?result_container - result_container ?priority_tag - priority_tag ?aggregated_slot - aggregated_slot ?score_function - score_function)
    :precondition
      (and
        (entity_scored ?result_container)
        (container_has_priority_tag ?result_container ?priority_tag)
        (priority_tag_consumed ?priority_tag)
        (priority_tag_assigned_to_slot ?priority_tag ?aggregated_slot)
        (assigned_score_function ?result_container ?score_function)
        (not
          (slot_mark_primary ?aggregated_slot)
        )
        (not
          (container_stage1_locked ?result_container)
        )
      )
    :effect (container_stage1_locked ?result_container)
  )
  (:action attach_preference_label
    :parameters (?result_container - result_container ?preference_label - preference_label)
    :precondition
      (and
        (entity_scored ?result_container)
        (preference_label_available ?preference_label)
        (not
          (preference_label_attached ?result_container)
        )
      )
    :effect
      (and
        (preference_label_attached ?result_container)
        (container_has_preference_label ?result_container ?preference_label)
        (not
          (preference_label_available ?preference_label)
        )
      )
  )
  (:action attach_preference_label_with_validation
    :parameters (?result_container - result_container ?priority_tag - priority_tag ?aggregated_slot - aggregated_slot ?score_function - score_function ?preference_label - preference_label)
    :precondition
      (and
        (entity_scored ?result_container)
        (container_has_priority_tag ?result_container ?priority_tag)
        (priority_tag_consumed ?priority_tag)
        (priority_tag_assigned_to_slot ?priority_tag ?aggregated_slot)
        (assigned_score_function ?result_container ?score_function)
        (slot_mark_primary ?aggregated_slot)
        (preference_label_attached ?result_container)
        (container_has_preference_label ?result_container ?preference_label)
        (not
          (container_stage1_locked ?result_container)
        )
      )
    :effect
      (and
        (container_stage1_locked ?result_container)
        (preference_label_confirmed ?result_container)
      )
  )
  (:action acquire_stage2_lock_primary
    :parameters (?result_container - result_container ?attribute_a - attribute_primary ?feature_extractor - feature_extractor ?priority_tag - priority_tag ?aggregated_slot - aggregated_slot)
    :precondition
      (and
        (container_stage1_locked ?result_container)
        (container_has_attribute_primary ?result_container ?attribute_a)
        (bound_feature_extractor ?result_container ?feature_extractor)
        (container_has_priority_tag ?result_container ?priority_tag)
        (priority_tag_assigned_to_slot ?priority_tag ?aggregated_slot)
        (not
          (slot_mark_secondary ?aggregated_slot)
        )
        (not
          (container_stage2_locked ?result_container)
        )
      )
    :effect (container_stage2_locked ?result_container)
  )
  (:action acquire_stage2_lock_secondary
    :parameters (?result_container - result_container ?attribute_a - attribute_primary ?feature_extractor - feature_extractor ?priority_tag - priority_tag ?aggregated_slot - aggregated_slot)
    :precondition
      (and
        (container_stage1_locked ?result_container)
        (container_has_attribute_primary ?result_container ?attribute_a)
        (bound_feature_extractor ?result_container ?feature_extractor)
        (container_has_priority_tag ?result_container ?priority_tag)
        (priority_tag_assigned_to_slot ?priority_tag ?aggregated_slot)
        (slot_mark_secondary ?aggregated_slot)
        (not
          (container_stage2_locked ?result_container)
        )
      )
    :effect (container_stage2_locked ?result_container)
  )
  (:action mark_container_adjudication_candidate
    :parameters (?result_container - result_container ?attribute_b - attribute_secondary ?priority_tag - priority_tag ?aggregated_slot - aggregated_slot)
    :precondition
      (and
        (container_stage2_locked ?result_container)
        (container_has_attribute_secondary ?result_container ?attribute_b)
        (container_has_priority_tag ?result_container ?priority_tag)
        (priority_tag_assigned_to_slot ?priority_tag ?aggregated_slot)
        (not
          (slot_mark_primary ?aggregated_slot)
        )
        (not
          (slot_mark_secondary ?aggregated_slot)
