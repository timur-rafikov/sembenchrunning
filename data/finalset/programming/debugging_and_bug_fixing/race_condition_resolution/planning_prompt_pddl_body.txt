## Planning domain (PDDL)

(define (domain race_condition_resolution_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity - object software_entity - entity artifact - entity resource - entity incident_report - entity investigation_target - incident_report engineer - software_entity reproducer_test - software_entity fix_template - software_entity external_dependency - software_entity reviewer - software_entity validation_token - software_entity analysis_tool - software_entity monitoring_alert - software_entity lock_token - artifact test_artifact - artifact configuration_option - artifact shared_memory_location_x - resource shared_memory_location_y - resource ci_test_job - resource execution_context_group - investigation_target repository_group - investigation_target execution_context_a - execution_context_group execution_context_b - execution_context_group code_repository - repository_group)

  (:predicates
    (target_open ?report - investigation_target)
    (target_under_investigation ?report - investigation_target)
    (triaged ?report - investigation_target)
    (fixed ?report - investigation_target)
    (target_ready_for_validation ?report - investigation_target)
    (target_validation_attested ?report - investigation_target)
    (engineer_available ?engineer - engineer)
    (target_assigned_engineer ?report - investigation_target ?engineer - engineer)
    (reproducer_available ?reproducer_test - reproducer_test)
    (reproducer_attached ?report - investigation_target ?reproducer_test - reproducer_test)
    (fix_template_available ?fix_template - fix_template)
    (fix_template_attached ?report - investigation_target ?fix_template - fix_template)
    (lock_token_available ?lock_token - lock_token)
    (context_a_lock_held ?execution_context_a - execution_context_a ?lock_token - lock_token)
    (context_b_lock_held ?execution_context_b - execution_context_b ?lock_token - lock_token)
    (context_a_accesses_shared_x ?execution_context_a - execution_context_a ?shared_memory_location_x - shared_memory_location_x)
    (shared_x_race_flagged ?shared_memory_location_x - shared_memory_location_x)
    (shared_x_lock_held ?shared_memory_location_x - shared_memory_location_x)
    (context_a_synchronized ?execution_context_a - execution_context_a)
    (context_b_accesses_shared_y ?execution_context_b - execution_context_b ?shared_memory_location_y - shared_memory_location_y)
    (shared_y_race_flagged ?shared_memory_location_y - shared_memory_location_y)
    (shared_y_lock_held ?shared_memory_location_y - shared_memory_location_y)
    (context_b_synchronized ?execution_context_b - execution_context_b)
    (ci_job_pending ?ci_test_job - ci_test_job)
    (ci_job_created ?ci_test_job - ci_test_job)
    (ci_job_targets_shared_x ?ci_test_job - ci_test_job ?shared_memory_location_x - shared_memory_location_x)
    (ci_job_targets_shared_y ?ci_test_job - ci_test_job ?shared_memory_location_y - shared_memory_location_y)
    (ci_job_variant_a_ready ?ci_test_job - ci_test_job)
    (ci_job_variant_b_ready ?ci_test_job - ci_test_job)
    (ci_job_executed ?ci_test_job - ci_test_job)
    (repository_links_context_a ?repository - code_repository ?execution_context_a - execution_context_a)
    (repository_links_context_b ?repository - code_repository ?execution_context_b - execution_context_b)
    (repository_links_ci_job ?repository - code_repository ?ci_test_job - ci_test_job)
    (test_artifact_available ?test_artifact - test_artifact)
    (repository_links_test_artifact ?repository - code_repository ?test_artifact - test_artifact)
    (test_artifact_promoted ?test_artifact - test_artifact)
    (test_artifact_linked_to_job ?test_artifact - test_artifact ?ci_test_job - ci_test_job)
    (repository_candidate_ready ?repository - code_repository)
    (repository_review_drafted ?repository - code_repository)
    (repository_review_in_progress ?repository - code_repository)
    (repository_dependency_registered ?repository - code_repository)
    (repository_dependency_reviewed ?repository - code_repository)
    (manual_review_required ?repository - code_repository)
    (repository_cross_validated ?repository - code_repository)
    (configuration_option_available ?configuration_option - configuration_option)
    (repository_links_configuration_option ?repository - code_repository ?configuration_option - configuration_option)
    (configuration_applied ?repository - code_repository)
    (configuration_verified ?repository - code_repository)
    (configuration_confirmed ?repository - code_repository)
    (external_dependency_available ?external_dependency - external_dependency)
    (repository_links_external_dependency ?repository - code_repository ?external_dependency - external_dependency)
    (reviewer_available ?reviewer - reviewer)
    (repository_links_reviewer ?repository - code_repository ?reviewer - reviewer)
    (analysis_tool_available ?analysis_tool - analysis_tool)
    (repository_links_analysis_tool ?repository - code_repository ?analysis_tool - analysis_tool)
    (monitoring_alert_available ?monitoring_alert - monitoring_alert)
    (repository_links_monitoring_alert ?repository - code_repository ?monitoring_alert - monitoring_alert)
    (validation_token_available ?validation_token - validation_token)
    (target_links_validation_token ?report - investigation_target ?validation_token - validation_token)
    (context_a_instrumented ?execution_context_a - execution_context_a)
    (context_b_instrumented ?execution_context_b - execution_context_b)
    (repository_ready_for_validation ?repository - code_repository)
  )
  (:action open_bug_report
    :parameters (?report - investigation_target)
    :precondition
      (and
        (not
          (target_open ?report)
        )
        (not
          (fixed ?report)
        )
      )
    :effect (target_open ?report)
  )
  (:action triage_bug_report
    :parameters (?report - investigation_target ?engineer - engineer)
    :precondition
      (and
        (target_open ?report)
        (not
          (triaged ?report)
        )
        (engineer_available ?engineer)
      )
    :effect
      (and
        (triaged ?report)
        (target_assigned_engineer ?report ?engineer)
        (not
          (engineer_available ?engineer)
        )
      )
  )
  (:action attach_reproducer_test
    :parameters (?report - investigation_target ?reproducer_test - reproducer_test)
    :precondition
      (and
        (target_open ?report)
        (triaged ?report)
        (reproducer_available ?reproducer_test)
      )
    :effect
      (and
        (reproducer_attached ?report ?reproducer_test)
        (not
          (reproducer_available ?reproducer_test)
        )
      )
  )
  (:action begin_investigation
    :parameters (?report - investigation_target ?reproducer_test - reproducer_test)
    :precondition
      (and
        (target_open ?report)
        (triaged ?report)
        (reproducer_attached ?report ?reproducer_test)
        (not
          (target_under_investigation ?report)
        )
      )
    :effect (target_under_investigation ?report)
  )
  (:action detach_reproducer_test
    :parameters (?report - investigation_target ?reproducer_test - reproducer_test)
    :precondition
      (and
        (reproducer_attached ?report ?reproducer_test)
      )
    :effect
      (and
        (reproducer_available ?reproducer_test)
        (not
          (reproducer_attached ?report ?reproducer_test)
        )
      )
  )
  (:action attach_fix_template
    :parameters (?report - investigation_target ?fix_template - fix_template)
    :precondition
      (and
        (target_under_investigation ?report)
        (fix_template_available ?fix_template)
      )
    :effect
      (and
        (fix_template_attached ?report ?fix_template)
        (not
          (fix_template_available ?fix_template)
        )
      )
  )
  (:action detach_fix_template
    :parameters (?report - investigation_target ?fix_template - fix_template)
    :precondition
      (and
        (fix_template_attached ?report ?fix_template)
      )
    :effect
      (and
        (fix_template_available ?fix_template)
        (not
          (fix_template_attached ?report ?fix_template)
        )
      )
  )
  (:action attach_analysis_tool
    :parameters (?repository - code_repository ?analysis_tool - analysis_tool)
    :precondition
      (and
        (target_under_investigation ?repository)
        (analysis_tool_available ?analysis_tool)
      )
    :effect
      (and
        (repository_links_analysis_tool ?repository ?analysis_tool)
        (not
          (analysis_tool_available ?analysis_tool)
        )
      )
  )
  (:action detach_analysis_tool
    :parameters (?repository - code_repository ?analysis_tool - analysis_tool)
    :precondition
      (and
        (repository_links_analysis_tool ?repository ?analysis_tool)
      )
    :effect
      (and
        (analysis_tool_available ?analysis_tool)
        (not
          (repository_links_analysis_tool ?repository ?analysis_tool)
        )
      )
  )
  (:action attach_monitoring_alert
    :parameters (?repository - code_repository ?monitoring_alert - monitoring_alert)
    :precondition
      (and
        (target_under_investigation ?repository)
        (monitoring_alert_available ?monitoring_alert)
      )
    :effect
      (and
        (repository_links_monitoring_alert ?repository ?monitoring_alert)
        (not
          (monitoring_alert_available ?monitoring_alert)
        )
      )
  )
  (:action detach_monitoring_alert
    :parameters (?repository - code_repository ?monitoring_alert - monitoring_alert)
    :precondition
      (and
        (repository_links_monitoring_alert ?repository ?monitoring_alert)
      )
    :effect
      (and
        (monitoring_alert_available ?monitoring_alert)
        (not
          (repository_links_monitoring_alert ?repository ?monitoring_alert)
        )
      )
  )
  (:action detect_shared_x_race
    :parameters (?execution_context_a - execution_context_a ?shared_memory_location_x - shared_memory_location_x ?reproducer_test - reproducer_test)
    :precondition
      (and
        (target_under_investigation ?execution_context_a)
        (reproducer_attached ?execution_context_a ?reproducer_test)
        (context_a_accesses_shared_x ?execution_context_a ?shared_memory_location_x)
        (not
          (shared_x_race_flagged ?shared_memory_location_x)
        )
        (not
          (shared_x_lock_held ?shared_memory_location_x)
        )
      )
    :effect (shared_x_race_flagged ?shared_memory_location_x)
  )
  (:action instrument_context_a
    :parameters (?execution_context_a - execution_context_a ?shared_memory_location_x - shared_memory_location_x ?fix_template - fix_template)
    :precondition
      (and
        (target_under_investigation ?execution_context_a)
        (fix_template_attached ?execution_context_a ?fix_template)
        (context_a_accesses_shared_x ?execution_context_a ?shared_memory_location_x)
        (shared_x_race_flagged ?shared_memory_location_x)
        (not
          (context_a_instrumented ?execution_context_a)
        )
      )
    :effect
      (and
        (context_a_instrumented ?execution_context_a)
        (context_a_synchronized ?execution_context_a)
      )
  )
  (:action acquire_context_a_lock
    :parameters (?execution_context_a - execution_context_a ?shared_memory_location_x - shared_memory_location_x ?lock_token - lock_token)
    :precondition
      (and
        (target_under_investigation ?execution_context_a)
        (context_a_accesses_shared_x ?execution_context_a ?shared_memory_location_x)
        (lock_token_available ?lock_token)
        (not
          (context_a_instrumented ?execution_context_a)
        )
      )
    :effect
      (and
        (shared_x_lock_held ?shared_memory_location_x)
        (context_a_instrumented ?execution_context_a)
        (context_a_lock_held ?execution_context_a ?lock_token)
        (not
          (lock_token_available ?lock_token)
        )
      )
  )
  (:action release_context_a_lock
    :parameters (?execution_context_a - execution_context_a ?shared_memory_location_x - shared_memory_location_x ?reproducer_test - reproducer_test ?lock_token - lock_token)
    :precondition
      (and
        (target_under_investigation ?execution_context_a)
        (reproducer_attached ?execution_context_a ?reproducer_test)
        (context_a_accesses_shared_x ?execution_context_a ?shared_memory_location_x)
        (shared_x_lock_held ?shared_memory_location_x)
        (context_a_lock_held ?execution_context_a ?lock_token)
        (not
          (context_a_synchronized ?execution_context_a)
        )
      )
    :effect
      (and
        (shared_x_race_flagged ?shared_memory_location_x)
        (context_a_synchronized ?execution_context_a)
        (lock_token_available ?lock_token)
        (not
          (context_a_lock_held ?execution_context_a ?lock_token)
        )
      )
  )
  (:action detect_shared_y_race
    :parameters (?execution_context_b - execution_context_b ?shared_memory_location_y - shared_memory_location_y ?reproducer_test - reproducer_test)
    :precondition
      (and
        (target_under_investigation ?execution_context_b)
        (reproducer_attached ?execution_context_b ?reproducer_test)
        (context_b_accesses_shared_y ?execution_context_b ?shared_memory_location_y)
        (not
          (shared_y_race_flagged ?shared_memory_location_y)
        )
        (not
          (shared_y_lock_held ?shared_memory_location_y)
        )
      )
    :effect (shared_y_race_flagged ?shared_memory_location_y)
  )
  (:action instrument_context_b
    :parameters (?execution_context_b - execution_context_b ?shared_memory_location_y - shared_memory_location_y ?fix_template - fix_template)
    :precondition
      (and
        (target_under_investigation ?execution_context_b)
        (fix_template_attached ?execution_context_b ?fix_template)
        (context_b_accesses_shared_y ?execution_context_b ?shared_memory_location_y)
        (shared_y_race_flagged ?shared_memory_location_y)
        (not
          (context_b_instrumented ?execution_context_b)
        )
      )
    :effect
      (and
        (context_b_instrumented ?execution_context_b)
        (context_b_synchronized ?execution_context_b)
      )
  )
  (:action acquire_context_b_lock
    :parameters (?execution_context_b - execution_context_b ?shared_memory_location_y - shared_memory_location_y ?lock_token - lock_token)
    :precondition
      (and
        (target_under_investigation ?execution_context_b)
        (context_b_accesses_shared_y ?execution_context_b ?shared_memory_location_y)
        (lock_token_available ?lock_token)
        (not
          (context_b_instrumented ?execution_context_b)
        )
      )
    :effect
      (and
        (shared_y_lock_held ?shared_memory_location_y)
        (context_b_instrumented ?execution_context_b)
        (context_b_lock_held ?execution_context_b ?lock_token)
        (not
          (lock_token_available ?lock_token)
        )
      )
  )
  (:action release_context_b_lock
    :parameters (?execution_context_b - execution_context_b ?shared_memory_location_y - shared_memory_location_y ?reproducer_test - reproducer_test ?lock_token - lock_token)
    :precondition
      (and
        (target_under_investigation ?execution_context_b)
        (reproducer_attached ?execution_context_b ?reproducer_test)
        (context_b_accesses_shared_y ?execution_context_b ?shared_memory_location_y)
        (shared_y_lock_held ?shared_memory_location_y)
        (context_b_lock_held ?execution_context_b ?lock_token)
        (not
          (context_b_synchronized ?execution_context_b)
        )
      )
    :effect
      (and
        (shared_y_race_flagged ?shared_memory_location_y)
        (context_b_synchronized ?execution_context_b)
        (lock_token_available ?lock_token)
        (not
          (context_b_lock_held ?execution_context_b ?lock_token)
        )
      )
  )
  (:action create_ci_race_job
    :parameters (?execution_context_a - execution_context_a ?execution_context_b - execution_context_b ?shared_memory_location_x - shared_memory_location_x ?shared_memory_location_y - shared_memory_location_y ?ci_test_job - ci_test_job)
    :precondition
      (and
        (context_a_instrumented ?execution_context_a)
        (context_b_instrumented ?execution_context_b)
        (context_a_accesses_shared_x ?execution_context_a ?shared_memory_location_x)
        (context_b_accesses_shared_y ?execution_context_b ?shared_memory_location_y)
        (shared_x_race_flagged ?shared_memory_location_x)
        (shared_y_race_flagged ?shared_memory_location_y)
        (context_a_synchronized ?execution_context_a)
        (context_b_synchronized ?execution_context_b)
        (ci_job_pending ?ci_test_job)
      )
    :effect
      (and
        (ci_job_created ?ci_test_job)
        (ci_job_targets_shared_x ?ci_test_job ?shared_memory_location_x)
        (ci_job_targets_shared_y ?ci_test_job ?shared_memory_location_y)
        (not
          (ci_job_pending ?ci_test_job)
        )
      )
  )
  (:action create_ci_race_job_variant_a
    :parameters (?execution_context_a - execution_context_a ?execution_context_b - execution_context_b ?shared_memory_location_x - shared_memory_location_x ?shared_memory_location_y - shared_memory_location_y ?ci_test_job - ci_test_job)
    :precondition
      (and
        (context_a_instrumented ?execution_context_a)
        (context_b_instrumented ?execution_context_b)
        (context_a_accesses_shared_x ?execution_context_a ?shared_memory_location_x)
        (context_b_accesses_shared_y ?execution_context_b ?shared_memory_location_y)
        (shared_x_lock_held ?shared_memory_location_x)
        (shared_y_race_flagged ?shared_memory_location_y)
        (not
          (context_a_synchronized ?execution_context_a)
        )
        (context_b_synchronized ?execution_context_b)
        (ci_job_pending ?ci_test_job)
      )
    :effect
      (and
        (ci_job_created ?ci_test_job)
        (ci_job_targets_shared_x ?ci_test_job ?shared_memory_location_x)
        (ci_job_targets_shared_y ?ci_test_job ?shared_memory_location_y)
        (ci_job_variant_a_ready ?ci_test_job)
        (not
          (ci_job_pending ?ci_test_job)
        )
      )
  )
  (:action create_ci_race_job_variant_b
    :parameters (?execution_context_a - execution_context_a ?execution_context_b - execution_context_b ?shared_memory_location_x - shared_memory_location_x ?shared_memory_location_y - shared_memory_location_y ?ci_test_job - ci_test_job)
    :precondition
      (and
        (context_a_instrumented ?execution_context_a)
        (context_b_instrumented ?execution_context_b)
        (context_a_accesses_shared_x ?execution_context_a ?shared_memory_location_x)
        (context_b_accesses_shared_y ?execution_context_b ?shared_memory_location_y)
        (shared_x_race_flagged ?shared_memory_location_x)
        (shared_y_lock_held ?shared_memory_location_y)
        (context_a_synchronized ?execution_context_a)
        (not
          (context_b_synchronized ?execution_context_b)
        )
        (ci_job_pending ?ci_test_job)
      )
    :effect
      (and
        (ci_job_created ?ci_test_job)
        (ci_job_targets_shared_x ?ci_test_job ?shared_memory_location_x)
        (ci_job_targets_shared_y ?ci_test_job ?shared_memory_location_y)
        (ci_job_variant_b_ready ?ci_test_job)
        (not
          (ci_job_pending ?ci_test_job)
        )
      )
  )
  (:action create_ci_race_job_variant_ab
    :parameters (?execution_context_a - execution_context_a ?execution_context_b - execution_context_b ?shared_memory_location_x - shared_memory_location_x ?shared_memory_location_y - shared_memory_location_y ?ci_test_job - ci_test_job)
    :precondition
      (and
        (context_a_instrumented ?execution_context_a)
        (context_b_instrumented ?execution_context_b)
        (context_a_accesses_shared_x ?execution_context_a ?shared_memory_location_x)
        (context_b_accesses_shared_y ?execution_context_b ?shared_memory_location_y)
        (shared_x_lock_held ?shared_memory_location_x)
        (shared_y_lock_held ?shared_memory_location_y)
        (not
          (context_a_synchronized ?execution_context_a)
        )
        (not
          (context_b_synchronized ?execution_context_b)
        )
        (ci_job_pending ?ci_test_job)
      )
    :effect
      (and
        (ci_job_created ?ci_test_job)
        (ci_job_targets_shared_x ?ci_test_job ?shared_memory_location_x)
        (ci_job_targets_shared_y ?ci_test_job ?shared_memory_location_y)
        (ci_job_variant_a_ready ?ci_test_job)
        (ci_job_variant_b_ready ?ci_test_job)
        (not
          (ci_job_pending ?ci_test_job)
        )
      )
  )
  (:action execute_ci_race_job
    :parameters (?ci_test_job - ci_test_job ?execution_context_a - execution_context_a ?reproducer_test - reproducer_test)
    :precondition
      (and
        (ci_job_created ?ci_test_job)
        (context_a_instrumented ?execution_context_a)
        (reproducer_attached ?execution_context_a ?reproducer_test)
        (not
          (ci_job_executed ?ci_test_job)
        )
      )
    :effect (ci_job_executed ?ci_test_job)
  )
  (:action collect_test_artifact
    :parameters (?repository - code_repository ?test_artifact - test_artifact ?ci_test_job - ci_test_job)
    :precondition
      (and
        (target_under_investigation ?repository)
        (repository_links_ci_job ?repository ?ci_test_job)
        (repository_links_test_artifact ?repository ?test_artifact)
        (test_artifact_available ?test_artifact)
        (ci_job_created ?ci_test_job)
        (ci_job_executed ?ci_test_job)
        (not
          (test_artifact_promoted ?test_artifact)
        )
      )
    :effect
      (and
        (test_artifact_promoted ?test_artifact)
        (test_artifact_linked_to_job ?test_artifact ?ci_test_job)
        (not
          (test_artifact_available ?test_artifact)
        )
      )
  )
  (:action promote_test_artifact
    :parameters (?repository - code_repository ?test_artifact - test_artifact ?ci_test_job - ci_test_job ?reproducer_test - reproducer_test)
    :precondition
      (and
        (target_under_investigation ?repository)
        (repository_links_test_artifact ?repository ?test_artifact)
        (test_artifact_promoted ?test_artifact)
        (test_artifact_linked_to_job ?test_artifact ?ci_test_job)
        (reproducer_attached ?repository ?reproducer_test)
        (not
          (ci_job_variant_a_ready ?ci_test_job)
        )
        (not
          (repository_candidate_ready ?repository)
        )
      )
    :effect (repository_candidate_ready ?repository)
  )
  (:action register_external_dependency
    :parameters (?repository - code_repository ?external_dependency - external_dependency)
    :precondition
      (and
        (target_under_investigation ?repository)
        (external_dependency_available ?external_dependency)
        (not
          (repository_dependency_registered ?repository)
        )
      )
    :effect
      (and
        (repository_dependency_registered ?repository)
        (repository_links_external_dependency ?repository ?external_dependency)
        (not
          (external_dependency_available ?external_dependency)
        )
      )
  )
  (:action prepare_dependency_review_candidate
    :parameters (?repository - code_repository ?test_artifact - test_artifact ?ci_test_job - ci_test_job ?reproducer_test - reproducer_test ?external_dependency - external_dependency)
    :precondition
      (and
        (target_under_investigation ?repository)
        (repository_links_test_artifact ?repository ?test_artifact)
        (test_artifact_promoted ?test_artifact)
        (test_artifact_linked_to_job ?test_artifact ?ci_test_job)
        (reproducer_attached ?repository ?reproducer_test)
        (ci_job_variant_a_ready ?ci_test_job)
        (repository_dependency_registered ?repository)
        (repository_links_external_dependency ?repository ?external_dependency)
        (not
          (repository_candidate_ready ?repository)
        )
      )
    :effect
      (and
        (repository_candidate_ready ?repository)
        (repository_dependency_reviewed ?repository)
      )
  )
  (:action draft_repository_review
    :parameters (?repository - code_repository ?analysis_tool - analysis_tool ?fix_template - fix_template ?test_artifact - test_artifact ?ci_test_job - ci_test_job)
    :precondition
      (and
        (repository_candidate_ready ?repository)
        (repository_links_analysis_tool ?repository ?analysis_tool)
        (fix_template_attached ?repository ?fix_template)
        (repository_links_test_artifact ?repository ?test_artifact)
        (test_artifact_linked_to_job ?test_artifact ?ci_test_job)
        (not
