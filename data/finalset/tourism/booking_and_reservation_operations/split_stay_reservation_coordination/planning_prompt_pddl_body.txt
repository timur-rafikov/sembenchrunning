  (:action create_split_stay_request
    :parameters (?split_stay_request - split_stay_request)
    :precondition
      (and
        (not
          (booking_created ?split_stay_request)
        )
        (not
          (booking_confirmed_with_supplier ?split_stay_request)

      )
    :effect (booking_created ?split_stay_request)
  )

    :parameters (?split_stay_request - split_stay_request ?supplier_offer - supplier_offer)

      (and
        (booking_created ?split_stay_request)

          (offer_selected ?split_stay_request)

      )
    :effect

        (booking_selected_offer ?split_stay_request ?supplier_offer)

          (supplier_offer_available ?supplier_offer)
        )
      )
  )
