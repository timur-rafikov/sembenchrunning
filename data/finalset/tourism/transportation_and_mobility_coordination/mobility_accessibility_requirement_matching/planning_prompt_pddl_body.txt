## Planning domain (PDDL)

(define (domain mobility_accessibility_requirement_matching)
  (:requirements :strips :typing :negative-preconditions)
  (:types transport_resource_category - object infrastructure_category - object equipment_category - object trip_component - object passenger_request - trip_component transport_resource - transport_resource_category accessibility_requirement - transport_resource_category local_assistance - transport_resource_category passenger_preference - transport_resource_category operator_credential - transport_resource_category accessibility_tag - transport_resource_category vehicle_feature - transport_resource_category service_attribute - transport_resource_category auxiliary_item_type - infrastructure_category equipment_instance - infrastructure_category policy_document - infrastructure_category transfer_node_inbound - equipment_category transfer_node_outbound - equipment_category vehicle_unit - equipment_category request_variant_inbound - passenger_request request_variant_outbound - passenger_request boarding_activity - request_variant_inbound alighting_activity - request_variant_inbound itinerary_segment - request_variant_outbound)
  (:predicates
    (request_registered ?passenger_request - passenger_request)
    (assignment_confirmed_flag ?passenger_request - passenger_request)
    (reservation_tentative ?passenger_request - passenger_request)
    (assignment_finalized ?passenger_request - passenger_request)
    (assignment_ready ?passenger_request - passenger_request)
    (allocation_committed ?passenger_request - passenger_request)
    (resource_available ?transport_resource - transport_resource)
    (reserved_for_resource ?passenger_request - passenger_request ?transport_resource - transport_resource)
    (requirement_available ?accessibility_requirement - accessibility_requirement)
    (bound_to_requirement ?passenger_request - passenger_request ?accessibility_requirement - accessibility_requirement)
    (assistance_available ?local_assistance - local_assistance)
    (assistance_assigned ?passenger_request - passenger_request ?local_assistance - local_assistance)
    (aux_item_type_available ?auxiliary_item_type - auxiliary_item_type)
    (boarding_activity_equipment_allocated ?boarding_activity - boarding_activity ?auxiliary_item_type - auxiliary_item_type)
    (alighting_activity_equipment_allocated ?alighting_activity - alighting_activity ?auxiliary_item_type - auxiliary_item_type)
    (activity_inbound_node ?boarding_activity - boarding_activity ?transfer_node_inbound - transfer_node_inbound)
    (inbound_node_ready ?transfer_node_inbound - transfer_node_inbound)
    (inbound_node_equipment_staged ?transfer_node_inbound - transfer_node_inbound)
    (boarding_activity_ready ?boarding_activity - boarding_activity)
    (activity_outbound_node ?alighting_activity - alighting_activity ?transfer_node_outbound - transfer_node_outbound)
    (outbound_node_ready ?transfer_node_outbound - transfer_node_outbound)
    (outbound_node_equipment_staged ?transfer_node_outbound - transfer_node_outbound)
    (alighting_activity_ready ?alighting_activity - alighting_activity)
    (vehicle_available ?vehicle_unit - vehicle_unit)
    (vehicle_reserved ?vehicle_unit - vehicle_unit)
    (vehicle_at_inbound_node ?vehicle_unit - vehicle_unit ?transfer_node_inbound - transfer_node_inbound)
    (vehicle_at_outbound_node ?vehicle_unit - vehicle_unit ?transfer_node_outbound - transfer_node_outbound)
    (vehicle_timing_buffer_applied ?vehicle_unit - vehicle_unit)
    (vehicle_extra_buffer_applied ?vehicle_unit - vehicle_unit)
    (vehicle_locked ?vehicle_unit - vehicle_unit)
    (segment_has_boarding_activity ?itinerary_segment - itinerary_segment ?boarding_activity - boarding_activity)
    (segment_has_alighting_activity ?itinerary_segment - itinerary_segment ?alighting_activity - alighting_activity)
    (segment_associated_vehicle ?itinerary_segment - itinerary_segment ?vehicle_unit - vehicle_unit)
    (equipment_available ?equipment_instance - equipment_instance)
    (segment_equipment_association ?itinerary_segment - itinerary_segment ?equipment_instance - equipment_instance)
    (equipment_checked_in ?equipment_instance - equipment_instance)
    (equipment_staged_on_vehicle ?equipment_instance - equipment_instance ?vehicle_unit - vehicle_unit)
    (segment_equipment_ready ?itinerary_segment - itinerary_segment)
    (segment_equipment_allocated ?itinerary_segment - itinerary_segment)
    (segment_attributes_present ?itinerary_segment - itinerary_segment)
    (segment_preference_applied ?itinerary_segment - itinerary_segment)
    (segment_preference_secondary_flag ?itinerary_segment - itinerary_segment)
    (segment_attributes_verified ?itinerary_segment - itinerary_segment)
    (segment_ready_for_handoff ?itinerary_segment - itinerary_segment)
    (policy_document_available ?policy_document - policy_document)
    (segment_policy_document ?itinerary_segment - itinerary_segment ?policy_document - policy_document)
    (policy_approved_for_segment ?itinerary_segment - itinerary_segment)
    (policy_intermediate_approved ?itinerary_segment - itinerary_segment)
    (policy_fully_approved ?itinerary_segment - itinerary_segment)
    (preference_available ?passenger_preference - passenger_preference)
    (segment_preference_assigned ?itinerary_segment - itinerary_segment ?passenger_preference - passenger_preference)
    (operator_credential_available ?operator_credential - operator_credential)
    (segment_operator_credential_assigned ?itinerary_segment - itinerary_segment ?operator_credential - operator_credential)
    (vehicle_feature_available ?vehicle_feature - vehicle_feature)
    (segment_vehicle_feature_assigned ?itinerary_segment - itinerary_segment ?vehicle_feature - vehicle_feature)
    (service_attribute_available ?service_attribute - service_attribute)
    (segment_service_attribute_assigned ?itinerary_segment - itinerary_segment ?service_attribute - service_attribute)
    (accessibility_tag_available ?accessibility_tag - accessibility_tag)
    (tag_assigned ?passenger_request - passenger_request ?accessibility_tag - accessibility_tag)
    (boarding_preparation_initiated ?boarding_activity - boarding_activity)
    (alighting_preparation_initiated ?alighting_activity - alighting_activity)
    (handoff_confirmed ?itinerary_segment - itinerary_segment)
  )
  (:action register_passenger_request
    :parameters (?passenger_request - passenger_request)
    :precondition
      (and
        (not
          (request_registered ?passenger_request)
        )
        (not
          (assignment_finalized ?passenger_request)
        )
      )
    :effect (request_registered ?passenger_request)
  )
  (:action propose_transport_reservation
    :parameters (?passenger_request - passenger_request ?transport_resource - transport_resource)
    :precondition
      (and
        (request_registered ?passenger_request)
        (not
          (reservation_tentative ?passenger_request)
        )
        (resource_available ?transport_resource)
      )
    :effect
      (and
        (reservation_tentative ?passenger_request)
        (reserved_for_resource ?passenger_request ?transport_resource)
        (not
          (resource_available ?transport_resource)
        )
      )
  )
  (:action assign_accessibility_requirement_to_request
    :parameters (?passenger_request - passenger_request ?accessibility_requirement - accessibility_requirement)
    :precondition
      (and
        (request_registered ?passenger_request)
        (reservation_tentative ?passenger_request)
        (requirement_available ?accessibility_requirement)
      )
    :effect
      (and
        (bound_to_requirement ?passenger_request ?accessibility_requirement)
        (not
          (requirement_available ?accessibility_requirement)
        )
      )
  )
  (:action confirm_resource_reservation
    :parameters (?passenger_request - passenger_request ?accessibility_requirement - accessibility_requirement)
    :precondition
      (and
        (request_registered ?passenger_request)
        (reservation_tentative ?passenger_request)
        (bound_to_requirement ?passenger_request ?accessibility_requirement)
        (not
          (assignment_confirmed_flag ?passenger_request)
        )
      )
    :effect (assignment_confirmed_flag ?passenger_request)
  )
  (:action rollback_requirement_assignment
    :parameters (?passenger_request - passenger_request ?accessibility_requirement - accessibility_requirement)
    :precondition
      (and
        (bound_to_requirement ?passenger_request ?accessibility_requirement)
      )
    :effect
      (and
        (requirement_available ?accessibility_requirement)
        (not
          (bound_to_requirement ?passenger_request ?accessibility_requirement)
        )
      )
  )
  (:action assign_local_assistance_to_request
    :parameters (?passenger_request - passenger_request ?local_assistance - local_assistance)
    :precondition
      (and
        (assignment_confirmed_flag ?passenger_request)
        (assistance_available ?local_assistance)
      )
    :effect
      (and
        (assistance_assigned ?passenger_request ?local_assistance)
        (not
          (assistance_available ?local_assistance)
        )
      )
  )
  (:action release_local_assistance_from_request
    :parameters (?passenger_request - passenger_request ?local_assistance - local_assistance)
    :precondition
      (and
        (assistance_assigned ?passenger_request ?local_assistance)
      )
    :effect
      (and
        (assistance_available ?local_assistance)
        (not
          (assistance_assigned ?passenger_request ?local_assistance)
        )
      )
  )
  (:action allocate_vehicle_feature_to_segment
    :parameters (?itinerary_segment - itinerary_segment ?vehicle_feature - vehicle_feature)
    :precondition
      (and
        (assignment_confirmed_flag ?itinerary_segment)
        (vehicle_feature_available ?vehicle_feature)
      )
    :effect
      (and
        (segment_vehicle_feature_assigned ?itinerary_segment ?vehicle_feature)
        (not
          (vehicle_feature_available ?vehicle_feature)
        )
      )
  )
  (:action release_vehicle_feature_from_segment
    :parameters (?itinerary_segment - itinerary_segment ?vehicle_feature - vehicle_feature)
    :precondition
      (and
        (segment_vehicle_feature_assigned ?itinerary_segment ?vehicle_feature)
      )
    :effect
      (and
        (vehicle_feature_available ?vehicle_feature)
        (not
          (segment_vehicle_feature_assigned ?itinerary_segment ?vehicle_feature)
        )
      )
  )
  (:action allocate_service_attribute_to_segment
    :parameters (?itinerary_segment - itinerary_segment ?service_attribute - service_attribute)
    :precondition
      (and
        (assignment_confirmed_flag ?itinerary_segment)
        (service_attribute_available ?service_attribute)
      )
    :effect
      (and
        (segment_service_attribute_assigned ?itinerary_segment ?service_attribute)
        (not
          (service_attribute_available ?service_attribute)
        )
      )
  )
  (:action release_service_attribute_from_segment
    :parameters (?itinerary_segment - itinerary_segment ?service_attribute - service_attribute)
    :precondition
      (and
        (segment_service_attribute_assigned ?itinerary_segment ?service_attribute)
      )
    :effect
      (and
        (service_attribute_available ?service_attribute)
        (not
          (segment_service_attribute_assigned ?itinerary_segment ?service_attribute)
        )
      )
  )
  (:action confirm_inbound_node_readiness
    :parameters (?boarding_activity - boarding_activity ?transfer_node_inbound - transfer_node_inbound ?accessibility_requirement - accessibility_requirement)
    :precondition
      (and
        (assignment_confirmed_flag ?boarding_activity)
        (bound_to_requirement ?boarding_activity ?accessibility_requirement)
        (activity_inbound_node ?boarding_activity ?transfer_node_inbound)
        (not
          (inbound_node_ready ?transfer_node_inbound)
        )
        (not
          (inbound_node_equipment_staged ?transfer_node_inbound)
        )
      )
    :effect (inbound_node_ready ?transfer_node_inbound)
  )
  (:action assign_local_assistance_to_boarding_activity
    :parameters (?boarding_activity - boarding_activity ?transfer_node_inbound - transfer_node_inbound ?local_assistance - local_assistance)
    :precondition
      (and
        (assignment_confirmed_flag ?boarding_activity)
        (assistance_assigned ?boarding_activity ?local_assistance)
        (activity_inbound_node ?boarding_activity ?transfer_node_inbound)
        (inbound_node_ready ?transfer_node_inbound)
        (not
          (boarding_preparation_initiated ?boarding_activity)
        )
      )
    :effect
      (and
        (boarding_preparation_initiated ?boarding_activity)
        (boarding_activity_ready ?boarding_activity)
      )
  )
  (:action stage_auxiliary_equipment_to_boarding_activity
    :parameters (?boarding_activity - boarding_activity ?transfer_node_inbound - transfer_node_inbound ?auxiliary_item_type - auxiliary_item_type)
    :precondition
      (and
        (assignment_confirmed_flag ?boarding_activity)
        (activity_inbound_node ?boarding_activity ?transfer_node_inbound)
        (aux_item_type_available ?auxiliary_item_type)
        (not
          (boarding_preparation_initiated ?boarding_activity)
        )
      )
    :effect
      (and
        (inbound_node_equipment_staged ?transfer_node_inbound)
        (boarding_preparation_initiated ?boarding_activity)
        (boarding_activity_equipment_allocated ?boarding_activity ?auxiliary_item_type)
        (not
          (aux_item_type_available ?auxiliary_item_type)
        )
      )
  )
  (:action finalize_boarding_equipment_binding
    :parameters (?boarding_activity - boarding_activity ?transfer_node_inbound - transfer_node_inbound ?accessibility_requirement - accessibility_requirement ?auxiliary_item_type - auxiliary_item_type)
    :precondition
      (and
        (assignment_confirmed_flag ?boarding_activity)
        (bound_to_requirement ?boarding_activity ?accessibility_requirement)
        (activity_inbound_node ?boarding_activity ?transfer_node_inbound)
        (inbound_node_equipment_staged ?transfer_node_inbound)
        (boarding_activity_equipment_allocated ?boarding_activity ?auxiliary_item_type)
        (not
          (boarding_activity_ready ?boarding_activity)
        )
      )
    :effect
      (and
        (inbound_node_ready ?transfer_node_inbound)
        (boarding_activity_ready ?boarding_activity)
        (aux_item_type_available ?auxiliary_item_type)
        (not
          (boarding_activity_equipment_allocated ?boarding_activity ?auxiliary_item_type)
        )
      )
  )
  (:action confirm_outbound_node_readiness
    :parameters (?alighting_activity - alighting_activity ?transfer_node_outbound - transfer_node_outbound ?accessibility_requirement - accessibility_requirement)
    :precondition
      (and
        (assignment_confirmed_flag ?alighting_activity)
        (bound_to_requirement ?alighting_activity ?accessibility_requirement)
        (activity_outbound_node ?alighting_activity ?transfer_node_outbound)
        (not
          (outbound_node_ready ?transfer_node_outbound)
        )
        (not
          (outbound_node_equipment_staged ?transfer_node_outbound)
        )
      )
    :effect (outbound_node_ready ?transfer_node_outbound)
  )
  (:action assign_local_assistance_to_alighting_activity
    :parameters (?alighting_activity - alighting_activity ?transfer_node_outbound - transfer_node_outbound ?local_assistance - local_assistance)
    :precondition
      (and
        (assignment_confirmed_flag ?alighting_activity)
        (assistance_assigned ?alighting_activity ?local_assistance)
        (activity_outbound_node ?alighting_activity ?transfer_node_outbound)
        (outbound_node_ready ?transfer_node_outbound)
        (not
          (alighting_preparation_initiated ?alighting_activity)
        )
      )
    :effect
      (and
        (alighting_preparation_initiated ?alighting_activity)
        (alighting_activity_ready ?alighting_activity)
      )
  )
  (:action stage_auxiliary_equipment_to_outbound_node
    :parameters (?alighting_activity - alighting_activity ?transfer_node_outbound - transfer_node_outbound ?auxiliary_item_type - auxiliary_item_type)
    :precondition
      (and
        (assignment_confirmed_flag ?alighting_activity)
        (activity_outbound_node ?alighting_activity ?transfer_node_outbound)
        (aux_item_type_available ?auxiliary_item_type)
        (not
          (alighting_preparation_initiated ?alighting_activity)
        )
      )
    :effect
      (and
        (outbound_node_equipment_staged ?transfer_node_outbound)
        (alighting_preparation_initiated ?alighting_activity)
        (alighting_activity_equipment_allocated ?alighting_activity ?auxiliary_item_type)
        (not
          (aux_item_type_available ?auxiliary_item_type)
        )
      )
  )
  (:action finalize_alighting_equipment_binding
    :parameters (?alighting_activity - alighting_activity ?transfer_node_outbound - transfer_node_outbound ?accessibility_requirement - accessibility_requirement ?auxiliary_item_type - auxiliary_item_type)
    :precondition
      (and
        (assignment_confirmed_flag ?alighting_activity)
        (bound_to_requirement ?alighting_activity ?accessibility_requirement)
        (activity_outbound_node ?alighting_activity ?transfer_node_outbound)
        (outbound_node_equipment_staged ?transfer_node_outbound)
        (alighting_activity_equipment_allocated ?alighting_activity ?auxiliary_item_type)
        (not
          (alighting_activity_ready ?alighting_activity)
        )
      )
    :effect
      (and
        (outbound_node_ready ?transfer_node_outbound)
        (alighting_activity_ready ?alighting_activity)
        (aux_item_type_available ?auxiliary_item_type)
        (not
          (alighting_activity_equipment_allocated ?alighting_activity ?auxiliary_item_type)
        )
      )
  )
  (:action create_vehicle_candidate_full_confirm
    :parameters (?boarding_activity - boarding_activity ?alighting_activity - alighting_activity ?transfer_node_inbound - transfer_node_inbound ?transfer_node_outbound - transfer_node_outbound ?vehicle_unit - vehicle_unit)
    :precondition
      (and
        (boarding_preparation_initiated ?boarding_activity)
        (alighting_preparation_initiated ?alighting_activity)
        (activity_inbound_node ?boarding_activity ?transfer_node_inbound)
        (activity_outbound_node ?alighting_activity ?transfer_node_outbound)
        (inbound_node_ready ?transfer_node_inbound)
        (outbound_node_ready ?transfer_node_outbound)
        (boarding_activity_ready ?boarding_activity)
        (alighting_activity_ready ?alighting_activity)
        (vehicle_available ?vehicle_unit)
      )
    :effect
      (and
        (vehicle_reserved ?vehicle_unit)
        (vehicle_at_inbound_node ?vehicle_unit ?transfer_node_inbound)
        (vehicle_at_outbound_node ?vehicle_unit ?transfer_node_outbound)
        (not
          (vehicle_available ?vehicle_unit)
        )
      )
  )
  (:action create_vehicle_candidate_inbound_equipped
    :parameters (?boarding_activity - boarding_activity ?alighting_activity - alighting_activity ?transfer_node_inbound - transfer_node_inbound ?transfer_node_outbound - transfer_node_outbound ?vehicle_unit - vehicle_unit)
    :precondition
      (and
        (boarding_preparation_initiated ?boarding_activity)
        (alighting_preparation_initiated ?alighting_activity)
        (activity_inbound_node ?boarding_activity ?transfer_node_inbound)
        (activity_outbound_node ?alighting_activity ?transfer_node_outbound)
        (inbound_node_equipment_staged ?transfer_node_inbound)
        (outbound_node_ready ?transfer_node_outbound)
        (not
          (boarding_activity_ready ?boarding_activity)
        )
        (alighting_activity_ready ?alighting_activity)
        (vehicle_available ?vehicle_unit)
      )
    :effect
      (and
        (vehicle_reserved ?vehicle_unit)
        (vehicle_at_inbound_node ?vehicle_unit ?transfer_node_inbound)
        (vehicle_at_outbound_node ?vehicle_unit ?transfer_node_outbound)
        (vehicle_timing_buffer_applied ?vehicle_unit)
        (not
          (vehicle_available ?vehicle_unit)
        )
      )
  )
  (:action create_vehicle_candidate_outbound_equipped
    :parameters (?boarding_activity - boarding_activity ?alighting_activity - alighting_activity ?transfer_node_inbound - transfer_node_inbound ?transfer_node_outbound - transfer_node_outbound ?vehicle_unit - vehicle_unit)
    :precondition
      (and
        (boarding_preparation_initiated ?boarding_activity)
        (alighting_preparation_initiated ?alighting_activity)
        (activity_inbound_node ?boarding_activity ?transfer_node_inbound)
        (activity_outbound_node ?alighting_activity ?transfer_node_outbound)
        (inbound_node_ready ?transfer_node_inbound)
        (outbound_node_equipment_staged ?transfer_node_outbound)
        (boarding_activity_ready ?boarding_activity)
        (not
          (alighting_activity_ready ?alighting_activity)
        )
        (vehicle_available ?vehicle_unit)
      )
    :effect
      (and
        (vehicle_reserved ?vehicle_unit)
        (vehicle_at_inbound_node ?vehicle_unit ?transfer_node_inbound)
        (vehicle_at_outbound_node ?vehicle_unit ?transfer_node_outbound)
        (vehicle_extra_buffer_applied ?vehicle_unit)
        (not
          (vehicle_available ?vehicle_unit)
        )
      )
  )
  (:action create_vehicle_candidate_both_equipped
    :parameters (?boarding_activity - boarding_activity ?alighting_activity - alighting_activity ?transfer_node_inbound - transfer_node_inbound ?transfer_node_outbound - transfer_node_outbound ?vehicle_unit - vehicle_unit)
    :precondition
      (and
        (boarding_preparation_initiated ?boarding_activity)
        (alighting_preparation_initiated ?alighting_activity)
        (activity_inbound_node ?boarding_activity ?transfer_node_inbound)
        (activity_outbound_node ?alighting_activity ?transfer_node_outbound)
        (inbound_node_equipment_staged ?transfer_node_inbound)
        (outbound_node_equipment_staged ?transfer_node_outbound)
        (not
          (boarding_activity_ready ?boarding_activity)
        )
        (not
          (alighting_activity_ready ?alighting_activity)
        )
        (vehicle_available ?vehicle_unit)
      )
    :effect
      (and
        (vehicle_reserved ?vehicle_unit)
        (vehicle_at_inbound_node ?vehicle_unit ?transfer_node_inbound)
        (vehicle_at_outbound_node ?vehicle_unit ?transfer_node_outbound)
        (vehicle_timing_buffer_applied ?vehicle_unit)
        (vehicle_extra_buffer_applied ?vehicle_unit)
        (not
          (vehicle_available ?vehicle_unit)
        )
      )
  )
  (:action lock_vehicle_for_boarding_activity
    :parameters (?vehicle_unit - vehicle_unit ?boarding_activity - boarding_activity ?accessibility_requirement - accessibility_requirement)
    :precondition
      (and
        (vehicle_reserved ?vehicle_unit)
        (boarding_preparation_initiated ?boarding_activity)
        (bound_to_requirement ?boarding_activity ?accessibility_requirement)
        (not
          (vehicle_locked ?vehicle_unit)
        )
      )
    :effect (vehicle_locked ?vehicle_unit)
  )
  (:action checkin_equipment_to_vehicle
    :parameters (?itinerary_segment - itinerary_segment ?equipment_instance - equipment_instance ?vehicle_unit - vehicle_unit)
    :precondition
      (and
        (assignment_confirmed_flag ?itinerary_segment)
        (segment_associated_vehicle ?itinerary_segment ?vehicle_unit)
        (segment_equipment_association ?itinerary_segment ?equipment_instance)
        (equipment_available ?equipment_instance)
        (vehicle_reserved ?vehicle_unit)
        (vehicle_locked ?vehicle_unit)
        (not
          (equipment_checked_in ?equipment_instance)
        )
      )
    :effect
      (and
        (equipment_checked_in ?equipment_instance)
        (equipment_staged_on_vehicle ?equipment_instance ?vehicle_unit)
        (not
          (equipment_available ?equipment_instance)
        )
      )
  )
  (:action finalize_equipment_checkin_for_segment
    :parameters (?itinerary_segment - itinerary_segment ?equipment_instance - equipment_instance ?vehicle_unit - vehicle_unit ?accessibility_requirement - accessibility_requirement)
    :precondition
      (and
        (assignment_confirmed_flag ?itinerary_segment)
        (segment_equipment_association ?itinerary_segment ?equipment_instance)
        (equipment_checked_in ?equipment_instance)
        (equipment_staged_on_vehicle ?equipment_instance ?vehicle_unit)
        (bound_to_requirement ?itinerary_segment ?accessibility_requirement)
        (not
          (vehicle_timing_buffer_applied ?vehicle_unit)
        )
        (not
          (segment_equipment_ready ?itinerary_segment)
        )
      )
    :effect (segment_equipment_ready ?itinerary_segment)
  )
  (:action apply_passenger_preference_to_segment
    :parameters (?itinerary_segment - itinerary_segment ?passenger_preference - passenger_preference)
    :precondition
      (and
        (assignment_confirmed_flag ?itinerary_segment)
        (preference_available ?passenger_preference)
        (not
          (segment_preference_applied ?itinerary_segment)
        )
      )
    :effect
      (and
        (segment_preference_applied ?itinerary_segment)
        (segment_preference_assigned ?itinerary_segment ?passenger_preference)
        (not
          (preference_available ?passenger_preference)
        )
      )
  )
  (:action apply_preference_with_vehicle_and_equipment
    :parameters (?itinerary_segment - itinerary_segment ?equipment_instance - equipment_instance ?vehicle_unit - vehicle_unit ?accessibility_requirement - accessibility_requirement ?passenger_preference - passenger_preference)
    :precondition
      (and
        (assignment_confirmed_flag ?itinerary_segment)
        (segment_equipment_association ?itinerary_segment ?equipment_instance)
        (equipment_checked_in ?equipment_instance)
        (equipment_staged_on_vehicle ?equipment_instance ?vehicle_unit)
        (bound_to_requirement ?itinerary_segment ?accessibility_requirement)
        (vehicle_timing_buffer_applied ?vehicle_unit)
        (segment_preference_applied ?itinerary_segment)
        (segment_preference_assigned ?itinerary_segment ?passenger_preference)
        (not
          (segment_equipment_ready ?itinerary_segment)
        )
      )
    :effect
      (and
        (segment_equipment_ready ?itinerary_segment)
        (segment_preference_secondary_flag ?itinerary_segment)
      )
