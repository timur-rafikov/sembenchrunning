## Planning domain (PDDL)

(define (domain loyalty_status_benefit_eligibility_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_item - object requirement_artifact - object case_context_component - object case_container - object case_entity - case_container policy_rule_document - resource_item document_instance - resource_item verifier_agent - resource_item promotional_offer - resource_item verification_method - resource_item exception_document - resource_item status_badge - resource_item conditional_term - resource_item supporting_evidence - requirement_artifact evidence_requirement - requirement_artifact policy_clause - requirement_artifact entry_requirement_context - case_context_component fare_component - case_context_component benefit_allocation_token - case_context_component traveler_subcase - case_entity member_subcase - case_entity primary_traveler - traveler_subcase companion_traveler - traveler_subcase loyalty_member_record - member_subcase)
  (:predicates
    (case_intake_recorded ?case_entity - case_entity)
    (policy_interpretation_completed ?case_entity - case_entity)
    (entry_requirements_checked_flag ?case_entity - case_entity)
    (eligibility_confirmed ?case_entity - case_entity)
    (final_decision_ready ?case_entity - case_entity)
    (status_recorded ?case_entity - case_entity)
    (rule_available ?policy_rule_document - policy_rule_document)
    (rule_assigned_to_case ?case_entity - case_entity ?policy_rule_document - policy_rule_document)
    (document_instance_available ?document_instance - document_instance)
    (document_attached_to_case ?case_entity - case_entity ?document_instance - document_instance)
    (verifier_available ?verifier_agent - verifier_agent)
    (verifier_bound_to_case ?case_entity - case_entity ?verifier_agent - verifier_agent)
    (supporting_evidence_available ?supporting_evidence - supporting_evidence)
    (evidence_recorded_for_subcase ?primary_traveler - primary_traveler ?supporting_evidence - supporting_evidence)
    (evidence_recorded_for_companion ?companion_traveler - companion_traveler ?supporting_evidence - supporting_evidence)
    (requirement_context_link ?primary_traveler - primary_traveler ?entry_requirement_context - entry_requirement_context)
    (context_flag_positive ?entry_requirement_context - entry_requirement_context)
    (context_flag_alternative ?entry_requirement_context - entry_requirement_context)
    (subcase_precheck_complete ?primary_traveler - primary_traveler)
    (fare_component_context_link ?companion_traveler - companion_traveler ?fare_component - fare_component)
    (fare_component_context_positive ?fare_component - fare_component)
    (fare_component_context_alternative ?fare_component - fare_component)
    (companion_precheck_complete ?companion_traveler - companion_traveler)
    (benefit_token_pending ?benefit_allocation_token - benefit_allocation_token)
    (benefit_token_ready ?benefit_allocation_token - benefit_allocation_token)
    (token_bound_to_context_entry ?benefit_allocation_token - benefit_allocation_token ?entry_requirement_context - entry_requirement_context)
    (token_bound_to_fare ?benefit_allocation_token - benefit_allocation_token ?fare_component - fare_component)
    (token_entry_alternative_flag ?benefit_allocation_token - benefit_allocation_token)
    (token_fare_alternative_flag ?benefit_allocation_token - benefit_allocation_token)
    (token_activated ?benefit_allocation_token - benefit_allocation_token)
    (member_link_to_primary_traveler ?loyalty_member_record - loyalty_member_record ?primary_traveler - primary_traveler)
    (member_link_to_companion_traveler ?loyalty_member_record - loyalty_member_record ?companion_traveler - companion_traveler)
    (member_bound_to_token ?loyalty_member_record - loyalty_member_record ?benefit_allocation_token - benefit_allocation_token)
    (evidence_requirement_defined ?evidence_requirement - evidence_requirement)
    (member_has_evidence_requirement ?loyalty_member_record - loyalty_member_record ?evidence_requirement - evidence_requirement)
    (evidence_requirement_fulfilled ?evidence_requirement - evidence_requirement)
    (evidence_requirement_bound_to_token ?evidence_requirement - evidence_requirement ?benefit_allocation_token - benefit_allocation_token)
    (member_progress_stage_one ?loyalty_member_record - loyalty_member_record)
    (member_progress_stage_two ?loyalty_member_record - loyalty_member_record)
    (member_ready_for_final_checks ?loyalty_member_record - loyalty_member_record)
    (offer_claimed_flag ?loyalty_member_record - loyalty_member_record)
    (offer_consumption_record ?loyalty_member_record - loyalty_member_record)
    (member_verified_additional_stage ?loyalty_member_record - loyalty_member_record)
    (member_final_checks_recorded ?loyalty_member_record - loyalty_member_record)
    (policy_clause_available ?policy_clause - policy_clause)
    (policy_clause_bound_to_member ?loyalty_member_record - loyalty_member_record ?policy_clause - policy_clause)
    (policy_clause_applied_flag ?loyalty_member_record - loyalty_member_record)
    (policy_clause_verified_stage ?loyalty_member_record - loyalty_member_record)
    (policy_clause_finalized ?loyalty_member_record - loyalty_member_record)
    (promotional_offer_available ?promotional_offer - promotional_offer)
    (offer_bound_to_member ?loyalty_member_record - loyalty_member_record ?promotional_offer - promotional_offer)
    (verification_method_available ?verification_method - verification_method)
    (verification_method_bound_to_member ?loyalty_member_record - loyalty_member_record ?verification_method - verification_method)
    (status_badge_available ?status_badge - status_badge)
    (status_badge_bound_to_member ?loyalty_member_record - loyalty_member_record ?status_badge - status_badge)
    (conditional_term_available ?conditional_term - conditional_term)
    (conditional_term_bound_to_member ?loyalty_member_record - loyalty_member_record ?conditional_term - conditional_term)
    (exception_document_available ?exception_document - exception_document)
    (exception_document_bound_to_case ?case_entity - case_entity ?exception_document - exception_document)
    (primary_subcase_ready ?primary_traveler - primary_traveler)
    (companion_subcase_ready ?companion_traveler - companion_traveler)
    (decision_recorded_marker ?loyalty_member_record - loyalty_member_record)
  )
  (:action record_case_intake
    :parameters (?case_entity - case_entity)
    :precondition
      (and
        (not
          (case_intake_recorded ?case_entity)
        )
        (not
          (eligibility_confirmed ?case_entity)
        )
      )
    :effect (case_intake_recorded ?case_entity)
  )
  (:action reserve_and_assign_policy_rule
    :parameters (?case_entity - case_entity ?policy_rule_document - policy_rule_document)
    :precondition
      (and
        (case_intake_recorded ?case_entity)
        (not
          (entry_requirements_checked_flag ?case_entity)
        )
        (rule_available ?policy_rule_document)
      )
    :effect
      (and
        (entry_requirements_checked_flag ?case_entity)
        (rule_assigned_to_case ?case_entity ?policy_rule_document)
        (not
          (rule_available ?policy_rule_document)
        )
      )
  )
  (:action attach_document_to_case
    :parameters (?case_entity - case_entity ?document_instance - document_instance)
    :precondition
      (and
        (case_intake_recorded ?case_entity)
        (entry_requirements_checked_flag ?case_entity)
        (document_instance_available ?document_instance)
      )
    :effect
      (and
        (document_attached_to_case ?case_entity ?document_instance)
        (not
          (document_instance_available ?document_instance)
        )
      )
  )
  (:action finalize_policy_interpretation
    :parameters (?case_entity - case_entity ?document_instance - document_instance)
    :precondition
      (and
        (case_intake_recorded ?case_entity)
        (entry_requirements_checked_flag ?case_entity)
        (document_attached_to_case ?case_entity ?document_instance)
        (not
          (policy_interpretation_completed ?case_entity)
        )
      )
    :effect (policy_interpretation_completed ?case_entity)
  )
  (:action release_attached_document
    :parameters (?case_entity - case_entity ?document_instance - document_instance)
    :precondition
      (and
        (document_attached_to_case ?case_entity ?document_instance)
      )
    :effect
      (and
        (document_instance_available ?document_instance)
        (not
          (document_attached_to_case ?case_entity ?document_instance)
        )
      )
  )
  (:action assign_verifier_to_case
    :parameters (?case_entity - case_entity ?verifier_agent - verifier_agent)
    :precondition
      (and
        (policy_interpretation_completed ?case_entity)
        (verifier_available ?verifier_agent)
      )
    :effect
      (and
        (verifier_bound_to_case ?case_entity ?verifier_agent)
        (not
          (verifier_available ?verifier_agent)
        )
      )
  )
  (:action unassign_verifier_from_case
    :parameters (?case_entity - case_entity ?verifier_agent - verifier_agent)
    :precondition
      (and
        (verifier_bound_to_case ?case_entity ?verifier_agent)
      )
    :effect
      (and
        (verifier_available ?verifier_agent)
        (not
          (verifier_bound_to_case ?case_entity ?verifier_agent)
        )
      )
  )
  (:action bind_status_badge_to_member
    :parameters (?loyalty_member_record - loyalty_member_record ?status_badge - status_badge)
    :precondition
      (and
        (policy_interpretation_completed ?loyalty_member_record)
        (status_badge_available ?status_badge)
      )
    :effect
      (and
        (status_badge_bound_to_member ?loyalty_member_record ?status_badge)
        (not
          (status_badge_available ?status_badge)
        )
      )
  )
  (:action unbind_status_badge_from_member
    :parameters (?loyalty_member_record - loyalty_member_record ?status_badge - status_badge)
    :precondition
      (and
        (status_badge_bound_to_member ?loyalty_member_record ?status_badge)
      )
    :effect
      (and
        (status_badge_available ?status_badge)
        (not
          (status_badge_bound_to_member ?loyalty_member_record ?status_badge)
        )
      )
  )
  (:action bind_conditional_term_to_member
    :parameters (?loyalty_member_record - loyalty_member_record ?conditional_term - conditional_term)
    :precondition
      (and
        (policy_interpretation_completed ?loyalty_member_record)
        (conditional_term_available ?conditional_term)
      )
    :effect
      (and
        (conditional_term_bound_to_member ?loyalty_member_record ?conditional_term)
        (not
          (conditional_term_available ?conditional_term)
        )
      )
  )
  (:action unbind_conditional_term_from_member
    :parameters (?loyalty_member_record - loyalty_member_record ?conditional_term - conditional_term)
    :precondition
      (and
        (conditional_term_bound_to_member ?loyalty_member_record ?conditional_term)
      )
    :effect
      (and
        (conditional_term_available ?conditional_term)
        (not
          (conditional_term_bound_to_member ?loyalty_member_record ?conditional_term)
        )
      )
  )
  (:action flag_entry_context_positive
    :parameters (?primary_traveler - primary_traveler ?entry_requirement_context - entry_requirement_context ?document_instance - document_instance)
    :precondition
      (and
        (policy_interpretation_completed ?primary_traveler)
        (document_attached_to_case ?primary_traveler ?document_instance)
        (requirement_context_link ?primary_traveler ?entry_requirement_context)
        (not
          (context_flag_positive ?entry_requirement_context)
        )
        (not
          (context_flag_alternative ?entry_requirement_context)
        )
      )
    :effect (context_flag_positive ?entry_requirement_context)
  )
  (:action complete_primary_subcase_precheck
    :parameters (?primary_traveler - primary_traveler ?entry_requirement_context - entry_requirement_context ?verifier_agent - verifier_agent)
    :precondition
      (and
        (policy_interpretation_completed ?primary_traveler)
        (verifier_bound_to_case ?primary_traveler ?verifier_agent)
        (requirement_context_link ?primary_traveler ?entry_requirement_context)
        (context_flag_positive ?entry_requirement_context)
        (not
          (primary_subcase_ready ?primary_traveler)
        )
      )
    :effect
      (and
        (primary_subcase_ready ?primary_traveler)
        (subcase_precheck_complete ?primary_traveler)
      )
  )
  (:action process_supporting_evidence_for_subcase
    :parameters (?primary_traveler - primary_traveler ?entry_requirement_context - entry_requirement_context ?supporting_evidence - supporting_evidence)
    :precondition
      (and
        (policy_interpretation_completed ?primary_traveler)
        (requirement_context_link ?primary_traveler ?entry_requirement_context)
        (supporting_evidence_available ?supporting_evidence)
        (not
          (primary_subcase_ready ?primary_traveler)
        )
      )
    :effect
      (and
        (context_flag_alternative ?entry_requirement_context)
        (primary_subcase_ready ?primary_traveler)
        (evidence_recorded_for_subcase ?primary_traveler ?supporting_evidence)
        (not
          (supporting_evidence_available ?supporting_evidence)
        )
      )
  )
  (:action resolve_subcase_evidence_and_update_context
    :parameters (?primary_traveler - primary_traveler ?entry_requirement_context - entry_requirement_context ?document_instance - document_instance ?supporting_evidence - supporting_evidence)
    :precondition
      (and
        (policy_interpretation_completed ?primary_traveler)
        (document_attached_to_case ?primary_traveler ?document_instance)
        (requirement_context_link ?primary_traveler ?entry_requirement_context)
        (context_flag_alternative ?entry_requirement_context)
        (evidence_recorded_for_subcase ?primary_traveler ?supporting_evidence)
        (not
          (subcase_precheck_complete ?primary_traveler)
        )
      )
    :effect
      (and
        (context_flag_positive ?entry_requirement_context)
        (subcase_precheck_complete ?primary_traveler)
        (supporting_evidence_available ?supporting_evidence)
        (not
          (evidence_recorded_for_subcase ?primary_traveler ?supporting_evidence)
        )
      )
  )
  (:action flag_fare_component_context_positive
    :parameters (?companion_traveler - companion_traveler ?fare_component - fare_component ?document_instance - document_instance)
    :precondition
      (and
        (policy_interpretation_completed ?companion_traveler)
        (document_attached_to_case ?companion_traveler ?document_instance)
        (fare_component_context_link ?companion_traveler ?fare_component)
        (not
          (fare_component_context_positive ?fare_component)
        )
        (not
          (fare_component_context_alternative ?fare_component)
        )
      )
    :effect (fare_component_context_positive ?fare_component)
  )
  (:action complete_companion_precheck
    :parameters (?companion_traveler - companion_traveler ?fare_component - fare_component ?verifier_agent - verifier_agent)
    :precondition
      (and
        (policy_interpretation_completed ?companion_traveler)
        (verifier_bound_to_case ?companion_traveler ?verifier_agent)
        (fare_component_context_link ?companion_traveler ?fare_component)
        (fare_component_context_positive ?fare_component)
        (not
          (companion_subcase_ready ?companion_traveler)
        )
      )
    :effect
      (and
        (companion_subcase_ready ?companion_traveler)
        (companion_precheck_complete ?companion_traveler)
      )
  )
  (:action process_supporting_evidence_for_companion
    :parameters (?companion_traveler - companion_traveler ?fare_component - fare_component ?supporting_evidence - supporting_evidence)
    :precondition
      (and
        (policy_interpretation_completed ?companion_traveler)
        (fare_component_context_link ?companion_traveler ?fare_component)
        (supporting_evidence_available ?supporting_evidence)
        (not
          (companion_subcase_ready ?companion_traveler)
        )
      )
    :effect
      (and
        (fare_component_context_alternative ?fare_component)
        (companion_subcase_ready ?companion_traveler)
        (evidence_recorded_for_companion ?companion_traveler ?supporting_evidence)
        (not
          (supporting_evidence_available ?supporting_evidence)
        )
      )
  )
  (:action resolve_companion_evidence_and_update_fare
    :parameters (?companion_traveler - companion_traveler ?fare_component - fare_component ?document_instance - document_instance ?supporting_evidence - supporting_evidence)
    :precondition
      (and
        (policy_interpretation_completed ?companion_traveler)
        (document_attached_to_case ?companion_traveler ?document_instance)
        (fare_component_context_link ?companion_traveler ?fare_component)
        (fare_component_context_alternative ?fare_component)
        (evidence_recorded_for_companion ?companion_traveler ?supporting_evidence)
        (not
          (companion_precheck_complete ?companion_traveler)
        )
      )
    :effect
      (and
        (fare_component_context_positive ?fare_component)
        (companion_precheck_complete ?companion_traveler)
        (supporting_evidence_available ?supporting_evidence)
        (not
          (evidence_recorded_for_companion ?companion_traveler ?supporting_evidence)
        )
      )
  )
  (:action assemble_benefit_token_positive_contexts
    :parameters (?primary_traveler - primary_traveler ?companion_traveler - companion_traveler ?entry_requirement_context - entry_requirement_context ?fare_component - fare_component ?benefit_allocation_token - benefit_allocation_token)
    :precondition
      (and
        (primary_subcase_ready ?primary_traveler)
        (companion_subcase_ready ?companion_traveler)
        (requirement_context_link ?primary_traveler ?entry_requirement_context)
        (fare_component_context_link ?companion_traveler ?fare_component)
        (context_flag_positive ?entry_requirement_context)
        (fare_component_context_positive ?fare_component)
        (subcase_precheck_complete ?primary_traveler)
        (companion_precheck_complete ?companion_traveler)
        (benefit_token_pending ?benefit_allocation_token)
      )
    :effect
      (and
        (benefit_token_ready ?benefit_allocation_token)
        (token_bound_to_context_entry ?benefit_allocation_token ?entry_requirement_context)
        (token_bound_to_fare ?benefit_allocation_token ?fare_component)
        (not
          (benefit_token_pending ?benefit_allocation_token)
        )
      )
  )
  (:action assemble_benefit_token_alternative_entry
    :parameters (?primary_traveler - primary_traveler ?companion_traveler - companion_traveler ?entry_requirement_context - entry_requirement_context ?fare_component - fare_component ?benefit_allocation_token - benefit_allocation_token)
    :precondition
      (and
        (primary_subcase_ready ?primary_traveler)
        (companion_subcase_ready ?companion_traveler)
        (requirement_context_link ?primary_traveler ?entry_requirement_context)
        (fare_component_context_link ?companion_traveler ?fare_component)
        (context_flag_alternative ?entry_requirement_context)
        (fare_component_context_positive ?fare_component)
        (not
          (subcase_precheck_complete ?primary_traveler)
        )
        (companion_precheck_complete ?companion_traveler)
        (benefit_token_pending ?benefit_allocation_token)
      )
    :effect
      (and
        (benefit_token_ready ?benefit_allocation_token)
        (token_bound_to_context_entry ?benefit_allocation_token ?entry_requirement_context)
        (token_bound_to_fare ?benefit_allocation_token ?fare_component)
        (token_entry_alternative_flag ?benefit_allocation_token)
        (not
          (benefit_token_pending ?benefit_allocation_token)
        )
      )
  )
  (:action assemble_benefit_token_companion_alternative_fare
    :parameters (?primary_traveler - primary_traveler ?companion_traveler - companion_traveler ?entry_requirement_context - entry_requirement_context ?fare_component - fare_component ?benefit_allocation_token - benefit_allocation_token)
    :precondition
      (and
        (primary_subcase_ready ?primary_traveler)
        (companion_subcase_ready ?companion_traveler)
        (requirement_context_link ?primary_traveler ?entry_requirement_context)
        (fare_component_context_link ?companion_traveler ?fare_component)
        (context_flag_positive ?entry_requirement_context)
        (fare_component_context_alternative ?fare_component)
        (subcase_precheck_complete ?primary_traveler)
        (not
          (companion_precheck_complete ?companion_traveler)
        )
        (benefit_token_pending ?benefit_allocation_token)
      )
    :effect
      (and
        (benefit_token_ready ?benefit_allocation_token)
        (token_bound_to_context_entry ?benefit_allocation_token ?entry_requirement_context)
        (token_bound_to_fare ?benefit_allocation_token ?fare_component)
        (token_fare_alternative_flag ?benefit_allocation_token)
        (not
          (benefit_token_pending ?benefit_allocation_token)
        )
      )
  )
  (:action assemble_benefit_token_full_conditional
    :parameters (?primary_traveler - primary_traveler ?companion_traveler - companion_traveler ?entry_requirement_context - entry_requirement_context ?fare_component - fare_component ?benefit_allocation_token - benefit_allocation_token)
    :precondition
      (and
        (primary_subcase_ready ?primary_traveler)
        (companion_subcase_ready ?companion_traveler)
        (requirement_context_link ?primary_traveler ?entry_requirement_context)
        (fare_component_context_link ?companion_traveler ?fare_component)
        (context_flag_alternative ?entry_requirement_context)
        (fare_component_context_alternative ?fare_component)
        (not
          (subcase_precheck_complete ?primary_traveler)
        )
        (not
          (companion_precheck_complete ?companion_traveler)
        )
        (benefit_token_pending ?benefit_allocation_token)
      )
    :effect
      (and
        (benefit_token_ready ?benefit_allocation_token)
        (token_bound_to_context_entry ?benefit_allocation_token ?entry_requirement_context)
        (token_bound_to_fare ?benefit_allocation_token ?fare_component)
        (token_entry_alternative_flag ?benefit_allocation_token)
        (token_fare_alternative_flag ?benefit_allocation_token)
        (not
          (benefit_token_pending ?benefit_allocation_token)
        )
      )
  )
  (:action activate_benefit_token
    :parameters (?benefit_allocation_token - benefit_allocation_token ?primary_traveler - primary_traveler ?document_instance - document_instance)
    :precondition
      (and
        (benefit_token_ready ?benefit_allocation_token)
        (primary_subcase_ready ?primary_traveler)
        (document_attached_to_case ?primary_traveler ?document_instance)
        (not
          (token_activated ?benefit_allocation_token)
        )
      )
    :effect (token_activated ?benefit_allocation_token)
  )
  (:action fulfill_evidence_requirement_and_bind_to_token
    :parameters (?loyalty_member_record - loyalty_member_record ?evidence_requirement - evidence_requirement ?benefit_allocation_token - benefit_allocation_token)
    :precondition
      (and
        (policy_interpretation_completed ?loyalty_member_record)
        (member_bound_to_token ?loyalty_member_record ?benefit_allocation_token)
        (member_has_evidence_requirement ?loyalty_member_record ?evidence_requirement)
        (evidence_requirement_defined ?evidence_requirement)
        (benefit_token_ready ?benefit_allocation_token)
        (token_activated ?benefit_allocation_token)
        (not
          (evidence_requirement_fulfilled ?evidence_requirement)
        )
      )
    :effect
      (and
        (evidence_requirement_fulfilled ?evidence_requirement)
        (evidence_requirement_bound_to_token ?evidence_requirement ?benefit_allocation_token)
        (not
          (evidence_requirement_defined ?evidence_requirement)
        )
      )
  )
  (:action advance_member_progress_stage_one
    :parameters (?loyalty_member_record - loyalty_member_record ?evidence_requirement - evidence_requirement ?benefit_allocation_token - benefit_allocation_token ?document_instance - document_instance)
    :precondition
      (and
        (policy_interpretation_completed ?loyalty_member_record)
        (member_has_evidence_requirement ?loyalty_member_record ?evidence_requirement)
        (evidence_requirement_fulfilled ?evidence_requirement)
        (evidence_requirement_bound_to_token ?evidence_requirement ?benefit_allocation_token)
        (document_attached_to_case ?loyalty_member_record ?document_instance)
        (not
          (token_entry_alternative_flag ?benefit_allocation_token)
        )
        (not
          (member_progress_stage_one ?loyalty_member_record)
        )
      )
    :effect (member_progress_stage_one ?loyalty_member_record)
  )
  (:action claim_promotional_offer_for_member
    :parameters (?loyalty_member_record - loyalty_member_record ?promotional_offer - promotional_offer)
    :precondition
      (and
        (policy_interpretation_completed ?loyalty_member_record)
        (promotional_offer_available ?promotional_offer)
        (not
          (offer_claimed_flag ?loyalty_member_record)
        )
      )
    :effect
      (and
        (offer_claimed_flag ?loyalty_member_record)
        (offer_bound_to_member ?loyalty_member_record ?promotional_offer)
        (not
          (promotional_offer_available ?promotional_offer)
        )
      )
