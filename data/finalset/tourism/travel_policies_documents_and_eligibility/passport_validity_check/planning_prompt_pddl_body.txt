## Planning domain (PDDL)

(define (domain passport_validity_check)
  (:requirements :strips :typing :negative-preconditions)
  (:types domain_resource - object document_supertype - object policy_element_supertype - object case_identifier - object case_record - case_identifier jurisdiction - domain_resource travel_date - domain_resource inspector - domain_resource exemption_type - domain_resource service_option - domain_resource policy_reference - domain_resource fee_item - domain_resource waiver_rule - domain_resource passport_document - document_supertype supporting_document_type - document_supertype mitigating_evidence - document_supertype validity_flag - policy_element_supertype entry_condition - policy_element_supertype advice_package - policy_element_supertype case_subtype - case_record case_subtype_alt - case_record primary_traveler - case_subtype secondary_traveler - case_subtype assessment_task - case_subtype_alt)
  (:predicates
    (case_created ?case_record - case_record)
    (case_ready ?case_record - case_record)
    (jurisdiction_attached ?case_record - case_record)
    (case_approved ?case_record - case_record)
    (ready_for_decision ?case_record - case_record)
    (decision_recorded ?case_record - case_record)
    (jurisdiction_available ?jurisdiction - jurisdiction)
    (case_has_jurisdiction ?case_record - case_record ?jurisdiction - jurisdiction)
    (travel_date_available ?travel_date - travel_date)
    (case_has_travel_date ?case_record - case_record ?travel_date - travel_date)
    (inspector_available ?inspector - inspector)
    (case_assigned_inspector ?case_record - case_record ?inspector - inspector)
    (passport_document_available ?passport_document - passport_document)
    (primary_traveler_has_passport ?primary_traveler - primary_traveler ?passport_document - passport_document)
    (secondary_traveler_has_passport ?secondary_traveler - secondary_traveler ?passport_document - passport_document)
    (primary_traveler_flag_link ?primary_traveler - primary_traveler ?validity_flag - validity_flag)
    (validity_flag_confirmed ?validity_flag - validity_flag)
    (validity_flag_support_required ?validity_flag - validity_flag)
    (primary_traveler_checks_passed ?primary_traveler - primary_traveler)
    (secondary_traveler_has_entry_condition ?secondary_traveler - secondary_traveler ?entry_condition - entry_condition)
    (entry_condition_confirmed ?entry_condition - entry_condition)
    (entry_condition_flagged ?entry_condition - entry_condition)
    (secondary_traveler_checks_passed ?secondary_traveler - secondary_traveler)
    (advice_package_available ?advice_package - advice_package)
    (advice_package_created ?advice_package - advice_package)
    (advice_package_has_validity_flag ?advice_package - advice_package ?validity_flag - validity_flag)
    (advice_package_has_entry_condition ?advice_package - advice_package ?entry_condition - entry_condition)
    (advice_package_flag_requires_exemption ?advice_package - advice_package)
    (advice_package_flag_requires_mitigation ?advice_package - advice_package)
    (advice_package_finalized ?advice_package - advice_package)
    (task_associated_primary_traveler ?assessment_task - assessment_task ?primary_traveler - primary_traveler)
    (task_associated_secondary_traveler ?assessment_task - assessment_task ?secondary_traveler - secondary_traveler)
    (task_associated_advice_package ?assessment_task - assessment_task ?advice_package - advice_package)
    (supporting_document_type_available ?supporting_document_type - supporting_document_type)
    (task_requires_supporting_document_type ?assessment_task - assessment_task ?supporting_document_type - supporting_document_type)
    (supporting_document_type_verified ?supporting_document_type - supporting_document_type)
    (supporting_document_type_linked_to_advice_package ?supporting_document_type - supporting_document_type ?advice_package - advice_package)
    (assessment_task_documents_verified ?assessment_task - assessment_task)
    (assessment_task_document_checks_passed ?assessment_task - assessment_task)
    (assessment_task_authorization_requested ?assessment_task - assessment_task)
    (task_exemption_applied ?assessment_task - assessment_task)
    (task_exemption_confirmed ?assessment_task - assessment_task)
    (task_authorization_granted ?assessment_task - assessment_task)
    (task_ready_for_finalization ?assessment_task - assessment_task)
    (mitigating_evidence_available ?mitigating_evidence - mitigating_evidence)
    (task_has_mitigating_evidence ?assessment_task - assessment_task ?mitigating_evidence - mitigating_evidence)
    (task_exception_flagged ?assessment_task - assessment_task)
    (task_targeted_inspection_initiated ?assessment_task - assessment_task)
    (task_targeted_inspection_completed ?assessment_task - assessment_task)
    (exemption_type_available ?exemption_type - exemption_type)
    (task_has_exemption_type ?assessment_task - assessment_task ?exemption_type - exemption_type)
    (service_option_available ?service_option - service_option)
    (task_has_service_option ?assessment_task - assessment_task ?service_option - service_option)
    (fee_item_available ?fee_item - fee_item)
    (task_has_fee_item ?assessment_task - assessment_task ?fee_item - fee_item)
    (waiver_rule_available ?waiver_rule - waiver_rule)
    (task_has_waiver_rule ?assessment_task - assessment_task ?waiver_rule - waiver_rule)
    (policy_reference_available ?policy_reference - policy_reference)
    (case_has_policy_reference ?case_record - case_record ?policy_reference - policy_reference)
    (primary_traveler_assessed ?primary_traveler - primary_traveler)
    (secondary_traveler_assessed ?secondary_traveler - secondary_traveler)
    (assessment_task_finalized ?assessment_task - assessment_task)
  )
  (:action create_case_record
    :parameters (?case_record - case_record)
    :precondition
      (and
        (not
          (case_created ?case_record)
        )
        (not
          (case_approved ?case_record)
        )
      )
    :effect (case_created ?case_record)
  )
  (:action attach_jurisdiction_to_case
    :parameters (?case_record - case_record ?jurisdiction - jurisdiction)
    :precondition
      (and
        (case_created ?case_record)
        (not
          (jurisdiction_attached ?case_record)
        )
        (jurisdiction_available ?jurisdiction)
      )
    :effect
      (and
        (jurisdiction_attached ?case_record)
        (case_has_jurisdiction ?case_record ?jurisdiction)
        (not
          (jurisdiction_available ?jurisdiction)
        )
      )
  )
  (:action attach_travel_date_to_case
    :parameters (?case_record - case_record ?travel_date - travel_date)
    :precondition
      (and
        (case_created ?case_record)
        (jurisdiction_attached ?case_record)
        (travel_date_available ?travel_date)
      )
    :effect
      (and
        (case_has_travel_date ?case_record ?travel_date)
        (not
          (travel_date_available ?travel_date)
        )
      )
  )
  (:action mark_case_ready_for_assessment
    :parameters (?case_record - case_record ?travel_date - travel_date)
    :precondition
      (and
        (case_created ?case_record)
        (jurisdiction_attached ?case_record)
        (case_has_travel_date ?case_record ?travel_date)
        (not
          (case_ready ?case_record)
        )
      )
    :effect (case_ready ?case_record)
  )
  (:action release_travel_date_from_case
    :parameters (?case_record - case_record ?travel_date - travel_date)
    :precondition
      (and
        (case_has_travel_date ?case_record ?travel_date)
      )
    :effect
      (and
        (travel_date_available ?travel_date)
        (not
          (case_has_travel_date ?case_record ?travel_date)
        )
      )
  )
  (:action assign_inspector_to_case
    :parameters (?case_record - case_record ?inspector - inspector)
    :precondition
      (and
        (case_ready ?case_record)
        (inspector_available ?inspector)
      )
    :effect
      (and
        (case_assigned_inspector ?case_record ?inspector)
        (not
          (inspector_available ?inspector)
        )
      )
  )
  (:action unassign_inspector_from_case
    :parameters (?case_record - case_record ?inspector - inspector)
    :precondition
      (and
        (case_assigned_inspector ?case_record ?inspector)
      )
    :effect
      (and
        (inspector_available ?inspector)
        (not
          (case_assigned_inspector ?case_record ?inspector)
        )
      )
  )
  (:action attach_fee_item_to_task
    :parameters (?assessment_task - assessment_task ?fee_item - fee_item)
    :precondition
      (and
        (case_ready ?assessment_task)
        (fee_item_available ?fee_item)
      )
    :effect
      (and
        (task_has_fee_item ?assessment_task ?fee_item)
        (not
          (fee_item_available ?fee_item)
        )
      )
  )
  (:action detach_fee_item_from_task
    :parameters (?assessment_task - assessment_task ?fee_item - fee_item)
    :precondition
      (and
        (task_has_fee_item ?assessment_task ?fee_item)
      )
    :effect
      (and
        (fee_item_available ?fee_item)
        (not
          (task_has_fee_item ?assessment_task ?fee_item)
        )
      )
  )
  (:action attach_waiver_rule_to_task
    :parameters (?assessment_task - assessment_task ?waiver_rule - waiver_rule)
    :precondition
      (and
        (case_ready ?assessment_task)
        (waiver_rule_available ?waiver_rule)
      )
    :effect
      (and
        (task_has_waiver_rule ?assessment_task ?waiver_rule)
        (not
          (waiver_rule_available ?waiver_rule)
        )
      )
  )
  (:action detach_waiver_rule_from_task
    :parameters (?assessment_task - assessment_task ?waiver_rule - waiver_rule)
    :precondition
      (and
        (task_has_waiver_rule ?assessment_task ?waiver_rule)
      )
    :effect
      (and
        (waiver_rule_available ?waiver_rule)
        (not
          (task_has_waiver_rule ?assessment_task ?waiver_rule)
        )
      )
  )
  (:action confirm_validity_flag_for_traveler
    :parameters (?primary_traveler - primary_traveler ?validity_flag - validity_flag ?travel_date - travel_date)
    :precondition
      (and
        (case_ready ?primary_traveler)
        (case_has_travel_date ?primary_traveler ?travel_date)
        (primary_traveler_flag_link ?primary_traveler ?validity_flag)
        (not
          (validity_flag_confirmed ?validity_flag)
        )
        (not
          (validity_flag_support_required ?validity_flag)
        )
      )
    :effect (validity_flag_confirmed ?validity_flag)
  )
  (:action apply_inspector_verification
    :parameters (?primary_traveler - primary_traveler ?validity_flag - validity_flag ?inspector - inspector)
    :precondition
      (and
        (case_ready ?primary_traveler)
        (case_assigned_inspector ?primary_traveler ?inspector)
        (primary_traveler_flag_link ?primary_traveler ?validity_flag)
        (validity_flag_confirmed ?validity_flag)
        (not
          (primary_traveler_assessed ?primary_traveler)
        )
      )
    :effect
      (and
        (primary_traveler_assessed ?primary_traveler)
        (primary_traveler_checks_passed ?primary_traveler)
      )
  )
  (:action attach_passport_document_to_primary_traveler
    :parameters (?primary_traveler - primary_traveler ?validity_flag - validity_flag ?passport_document - passport_document)
    :precondition
      (and
        (case_ready ?primary_traveler)
        (primary_traveler_flag_link ?primary_traveler ?validity_flag)
        (passport_document_available ?passport_document)
        (not
          (primary_traveler_assessed ?primary_traveler)
        )
      )
    :effect
      (and
        (validity_flag_support_required ?validity_flag)
        (primary_traveler_assessed ?primary_traveler)
        (primary_traveler_has_passport ?primary_traveler ?passport_document)
        (not
          (passport_document_available ?passport_document)
        )
      )
  )
  (:action confirm_flag_with_passport_document
    :parameters (?primary_traveler - primary_traveler ?validity_flag - validity_flag ?travel_date - travel_date ?passport_document - passport_document)
    :precondition
      (and
        (case_ready ?primary_traveler)
        (case_has_travel_date ?primary_traveler ?travel_date)
        (primary_traveler_flag_link ?primary_traveler ?validity_flag)
        (validity_flag_support_required ?validity_flag)
        (primary_traveler_has_passport ?primary_traveler ?passport_document)
        (not
          (primary_traveler_checks_passed ?primary_traveler)
        )
      )
    :effect
      (and
        (validity_flag_confirmed ?validity_flag)
        (primary_traveler_checks_passed ?primary_traveler)
        (passport_document_available ?passport_document)
        (not
          (primary_traveler_has_passport ?primary_traveler ?passport_document)
        )
      )
  )
  (:action confirm_entry_condition_for_secondary_traveler
    :parameters (?secondary_traveler - secondary_traveler ?entry_condition - entry_condition ?travel_date - travel_date)
    :precondition
      (and
        (case_ready ?secondary_traveler)
        (case_has_travel_date ?secondary_traveler ?travel_date)
        (secondary_traveler_has_entry_condition ?secondary_traveler ?entry_condition)
        (not
          (entry_condition_confirmed ?entry_condition)
        )
        (not
          (entry_condition_flagged ?entry_condition)
        )
      )
    :effect (entry_condition_confirmed ?entry_condition)
  )
  (:action apply_inspector_verification_secondary
    :parameters (?secondary_traveler - secondary_traveler ?entry_condition - entry_condition ?inspector - inspector)
    :precondition
      (and
        (case_ready ?secondary_traveler)
        (case_assigned_inspector ?secondary_traveler ?inspector)
        (secondary_traveler_has_entry_condition ?secondary_traveler ?entry_condition)
        (entry_condition_confirmed ?entry_condition)
        (not
          (secondary_traveler_assessed ?secondary_traveler)
        )
      )
    :effect
      (and
        (secondary_traveler_assessed ?secondary_traveler)
        (secondary_traveler_checks_passed ?secondary_traveler)
      )
  )
  (:action attach_passport_document_to_secondary_traveler
    :parameters (?secondary_traveler - secondary_traveler ?entry_condition - entry_condition ?passport_document - passport_document)
    :precondition
      (and
        (case_ready ?secondary_traveler)
        (secondary_traveler_has_entry_condition ?secondary_traveler ?entry_condition)
        (passport_document_available ?passport_document)
        (not
          (secondary_traveler_assessed ?secondary_traveler)
        )
      )
    :effect
      (and
        (entry_condition_flagged ?entry_condition)
        (secondary_traveler_assessed ?secondary_traveler)
        (secondary_traveler_has_passport ?secondary_traveler ?passport_document)
        (not
          (passport_document_available ?passport_document)
        )
      )
  )
  (:action confirm_entry_condition_with_passport_document_for_secondary
    :parameters (?secondary_traveler - secondary_traveler ?entry_condition - entry_condition ?travel_date - travel_date ?passport_document - passport_document)
    :precondition
      (and
        (case_ready ?secondary_traveler)
        (case_has_travel_date ?secondary_traveler ?travel_date)
        (secondary_traveler_has_entry_condition ?secondary_traveler ?entry_condition)
        (entry_condition_flagged ?entry_condition)
        (secondary_traveler_has_passport ?secondary_traveler ?passport_document)
        (not
          (secondary_traveler_checks_passed ?secondary_traveler)
        )
      )
    :effect
      (and
        (entry_condition_confirmed ?entry_condition)
        (secondary_traveler_checks_passed ?secondary_traveler)
        (passport_document_available ?passport_document)
        (not
          (secondary_traveler_has_passport ?secondary_traveler ?passport_document)
        )
      )
  )
  (:action assemble_advice_package
    :parameters (?primary_traveler - primary_traveler ?secondary_traveler - secondary_traveler ?validity_flag - validity_flag ?entry_condition - entry_condition ?advice_package - advice_package)
    :precondition
      (and
        (primary_traveler_assessed ?primary_traveler)
        (secondary_traveler_assessed ?secondary_traveler)
        (primary_traveler_flag_link ?primary_traveler ?validity_flag)
        (secondary_traveler_has_entry_condition ?secondary_traveler ?entry_condition)
        (validity_flag_confirmed ?validity_flag)
        (entry_condition_confirmed ?entry_condition)
        (primary_traveler_checks_passed ?primary_traveler)
        (secondary_traveler_checks_passed ?secondary_traveler)
        (advice_package_available ?advice_package)
      )
    :effect
      (and
        (advice_package_created ?advice_package)
        (advice_package_has_validity_flag ?advice_package ?validity_flag)
        (advice_package_has_entry_condition ?advice_package ?entry_condition)
        (not
          (advice_package_available ?advice_package)
        )
      )
  )
  (:action assemble_advice_package_with_exemption
    :parameters (?primary_traveler - primary_traveler ?secondary_traveler - secondary_traveler ?validity_flag - validity_flag ?entry_condition - entry_condition ?advice_package - advice_package)
    :precondition
      (and
        (primary_traveler_assessed ?primary_traveler)
        (secondary_traveler_assessed ?secondary_traveler)
        (primary_traveler_flag_link ?primary_traveler ?validity_flag)
        (secondary_traveler_has_entry_condition ?secondary_traveler ?entry_condition)
        (validity_flag_support_required ?validity_flag)
        (entry_condition_confirmed ?entry_condition)
        (not
          (primary_traveler_checks_passed ?primary_traveler)
        )
        (secondary_traveler_checks_passed ?secondary_traveler)
        (advice_package_available ?advice_package)
      )
    :effect
      (and
        (advice_package_created ?advice_package)
        (advice_package_has_validity_flag ?advice_package ?validity_flag)
        (advice_package_has_entry_condition ?advice_package ?entry_condition)
        (advice_package_flag_requires_exemption ?advice_package)
        (not
          (advice_package_available ?advice_package)
        )
      )
  )
  (:action assemble_advice_package_with_mitigation
    :parameters (?primary_traveler - primary_traveler ?secondary_traveler - secondary_traveler ?validity_flag - validity_flag ?entry_condition - entry_condition ?advice_package - advice_package)
    :precondition
      (and
        (primary_traveler_assessed ?primary_traveler)
        (secondary_traveler_assessed ?secondary_traveler)
        (primary_traveler_flag_link ?primary_traveler ?validity_flag)
        (secondary_traveler_has_entry_condition ?secondary_traveler ?entry_condition)
        (validity_flag_confirmed ?validity_flag)
        (entry_condition_flagged ?entry_condition)
        (primary_traveler_checks_passed ?primary_traveler)
        (not
          (secondary_traveler_checks_passed ?secondary_traveler)
        )
        (advice_package_available ?advice_package)
      )
    :effect
      (and
        (advice_package_created ?advice_package)
        (advice_package_has_validity_flag ?advice_package ?validity_flag)
        (advice_package_has_entry_condition ?advice_package ?entry_condition)
        (advice_package_flag_requires_mitigation ?advice_package)
        (not
          (advice_package_available ?advice_package)
        )
      )
  )
  (:action assemble_advice_package_with_exemption_and_mitigation
    :parameters (?primary_traveler - primary_traveler ?secondary_traveler - secondary_traveler ?validity_flag - validity_flag ?entry_condition - entry_condition ?advice_package - advice_package)
    :precondition
      (and
        (primary_traveler_assessed ?primary_traveler)
        (secondary_traveler_assessed ?secondary_traveler)
        (primary_traveler_flag_link ?primary_traveler ?validity_flag)
        (secondary_traveler_has_entry_condition ?secondary_traveler ?entry_condition)
        (validity_flag_support_required ?validity_flag)
        (entry_condition_flagged ?entry_condition)
        (not
          (primary_traveler_checks_passed ?primary_traveler)
        )
        (not
          (secondary_traveler_checks_passed ?secondary_traveler)
        )
        (advice_package_available ?advice_package)
      )
    :effect
      (and
        (advice_package_created ?advice_package)
        (advice_package_has_validity_flag ?advice_package ?validity_flag)
        (advice_package_has_entry_condition ?advice_package ?entry_condition)
        (advice_package_flag_requires_exemption ?advice_package)
        (advice_package_flag_requires_mitigation ?advice_package)
        (not
          (advice_package_available ?advice_package)
        )
      )
  )
  (:action finalize_advice_package
    :parameters (?advice_package - advice_package ?primary_traveler - primary_traveler ?travel_date - travel_date)
    :precondition
      (and
        (advice_package_created ?advice_package)
        (primary_traveler_assessed ?primary_traveler)
        (case_has_travel_date ?primary_traveler ?travel_date)
        (not
          (advice_package_finalized ?advice_package)
        )
      )
    :effect (advice_package_finalized ?advice_package)
  )
  (:action verify_supporting_document_type
    :parameters (?assessment_task - assessment_task ?supporting_document_type - supporting_document_type ?advice_package - advice_package)
    :precondition
      (and
        (case_ready ?assessment_task)
        (task_associated_advice_package ?assessment_task ?advice_package)
        (task_requires_supporting_document_type ?assessment_task ?supporting_document_type)
        (supporting_document_type_available ?supporting_document_type)
        (advice_package_created ?advice_package)
        (advice_package_finalized ?advice_package)
        (not
          (supporting_document_type_verified ?supporting_document_type)
        )
      )
    :effect
      (and
        (supporting_document_type_verified ?supporting_document_type)
        (supporting_document_type_linked_to_advice_package ?supporting_document_type ?advice_package)
        (not
          (supporting_document_type_available ?supporting_document_type)
        )
      )
  )
  (:action bind_verified_document_type_to_task
    :parameters (?assessment_task - assessment_task ?supporting_document_type - supporting_document_type ?advice_package - advice_package ?travel_date - travel_date)
    :precondition
      (and
        (case_ready ?assessment_task)
        (task_requires_supporting_document_type ?assessment_task ?supporting_document_type)
        (supporting_document_type_verified ?supporting_document_type)
        (supporting_document_type_linked_to_advice_package ?supporting_document_type ?advice_package)
        (case_has_travel_date ?assessment_task ?travel_date)
        (not
          (advice_package_flag_requires_exemption ?advice_package)
        )
        (not
          (assessment_task_documents_verified ?assessment_task)
        )
      )
    :effect (assessment_task_documents_verified ?assessment_task)
  )
  (:action apply_exemption_type_to_task
    :parameters (?assessment_task - assessment_task ?exemption_type - exemption_type)
    :precondition
      (and
        (case_ready ?assessment_task)
        (exemption_type_available ?exemption_type)
        (not
          (task_exemption_applied ?assessment_task)
        )
      )
    :effect
      (and
        (task_exemption_applied ?assessment_task)
        (task_has_exemption_type ?assessment_task ?exemption_type)
        (not
          (exemption_type_available ?exemption_type)
        )
      )
  )
  (:action resolve_exemption_with_document_for_task
    :parameters (?assessment_task - assessment_task ?supporting_document_type - supporting_document_type ?advice_package - advice_package ?travel_date - travel_date ?exemption_type - exemption_type)
    :precondition
      (and
        (case_ready ?assessment_task)
        (task_requires_supporting_document_type ?assessment_task ?supporting_document_type)
        (supporting_document_type_verified ?supporting_document_type)
        (supporting_document_type_linked_to_advice_package ?supporting_document_type ?advice_package)
        (case_has_travel_date ?assessment_task ?travel_date)
        (advice_package_flag_requires_exemption ?advice_package)
        (task_exemption_applied ?assessment_task)
        (task_has_exemption_type ?assessment_task ?exemption_type)
        (not
          (assessment_task_documents_verified ?assessment_task)
        )
      )
    :effect
      (and
        (assessment_task_documents_verified ?assessment_task)
        (task_exemption_confirmed ?assessment_task)
      )
  )
  (:action apply_fee_and_confirm_documents
    :parameters (?assessment_task - assessment_task ?fee_item - fee_item ?inspector - inspector ?supporting_document_type - supporting_document_type ?advice_package - advice_package)
    :precondition
      (and
        (assessment_task_documents_verified ?assessment_task)
        (task_has_fee_item ?assessment_task ?fee_item)
        (case_assigned_inspector ?assessment_task ?inspector)
        (task_requires_supporting_document_type ?assessment_task ?supporting_document_type)
