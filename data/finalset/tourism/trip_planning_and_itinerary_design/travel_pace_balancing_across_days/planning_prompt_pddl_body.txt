## Planning domain (PDDL)

(define (domain travel_pace_balancing)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_type - object temporal_type - object location_type - object day_base_type - object itinerary_element - day_base_type transport_option - resource_type activity - resource_type venue - resource_type optional_feature - resource_type scheduling_preference - resource_type global_constraint - resource_type service_option - resource_type supplier_availability - resource_type experience_tag - temporal_type activity_cluster - temporal_type special_request - temporal_type pace_bucket_day - location_type pace_bucket_variant - location_type visit_slot - location_type day_category - itinerary_element itinerary_container - itinerary_element travel_day - day_category travel_day_variant - day_category trip_plan - itinerary_container)
  (:predicates
    (itinerary_element_initialized ?itinerary_element - itinerary_element)
    (itinerary_element_time_confirmed ?itinerary_element - itinerary_element)
    (itinerary_element_transport_assigned ?itinerary_element - itinerary_element)
    (itinerary_element_committed ?itinerary_element - itinerary_element)
    (itinerary_element_finalized ?itinerary_element - itinerary_element)
    (itinerary_element_preferences_bound ?itinerary_element - itinerary_element)
    (transport_option_available ?transport_option - transport_option)
    (transport_assigned_to_element ?itinerary_element - itinerary_element ?transport_option - transport_option)
    (activity_available ?activity - activity)
    (activity_assigned_to_element ?itinerary_element - itinerary_element ?activity - activity)
    (venue_available ?venue - venue)
    (venue_assigned_to_element ?itinerary_element - itinerary_element ?venue - venue)
    (experience_tag_available ?experience_tag - experience_tag)
    (travel_day_tag_assigned ?travel_day - travel_day ?experience_tag - experience_tag)
    (travel_day_variant_tag_assigned ?travel_day_variant - travel_day_variant ?experience_tag - experience_tag)
    (travel_day_has_pace_bucket_day ?travel_day - travel_day ?pace_bucket_day - pace_bucket_day)
    (pace_bucket_day_primary ?pace_bucket_day - pace_bucket_day)
    (pace_bucket_day_secondary ?pace_bucket_day - pace_bucket_day)
    (travel_day_segment_confirmed ?travel_day - travel_day)
    (travel_day_variant_has_pace_bucket_variant ?travel_day_variant - travel_day_variant ?pace_bucket_variant - pace_bucket_variant)
    (pace_bucket_variant_primary ?pace_bucket_variant - pace_bucket_variant)
    (pace_bucket_variant_secondary ?pace_bucket_variant - pace_bucket_variant)
    (travel_day_variant_segment_confirmed ?travel_day_variant - travel_day_variant)
    (visit_slot_available ?visit_slot - visit_slot)
    (visit_slot_reserved ?visit_slot - visit_slot)
    (visit_slot_pace_bucket_day ?visit_slot - visit_slot ?pace_bucket_day - pace_bucket_day)
    (visit_slot_pace_bucket_variant ?visit_slot - visit_slot ?pace_bucket_variant - pace_bucket_variant)
    (visit_slot_variant_primary ?visit_slot - visit_slot)
    (visit_slot_variant_secondary ?visit_slot - visit_slot)
    (visit_slot_time_confirmed ?visit_slot - visit_slot)
    (trip_plan_includes_day ?trip_plan - trip_plan ?travel_day - travel_day)
    (trip_plan_includes_day_variant ?trip_plan - trip_plan ?travel_day_variant - travel_day_variant)
    (trip_plan_includes_visit_slot ?trip_plan - trip_plan ?visit_slot - visit_slot)
    (activity_cluster_available ?activity_cluster - activity_cluster)
    (trip_plan_includes_activity_cluster ?trip_plan - trip_plan ?activity_cluster - activity_cluster)
    (activity_cluster_consumed ?activity_cluster - activity_cluster)
    (activity_cluster_assigned_to_visit_slot ?activity_cluster - activity_cluster ?visit_slot - visit_slot)
    (trip_plan_services_integrated ?trip_plan - trip_plan)
    (trip_plan_service_option_locked ?trip_plan - trip_plan)
    (trip_plan_ready_for_sequence_finalization ?trip_plan - trip_plan)
    (trip_plan_optional_feature_requested ?trip_plan - trip_plan)
    (trip_plan_optional_feature_integrated ?trip_plan - trip_plan)
    (trip_plan_requires_preference_integration ?trip_plan - trip_plan)
    (trip_plan_sequence_finalized ?trip_plan - trip_plan)
    (special_request_available ?special_request - special_request)
    (trip_plan_has_special_request ?trip_plan - trip_plan ?special_request - special_request)
    (trip_plan_special_request_acknowledged ?trip_plan - trip_plan)
    (trip_plan_ready_for_special_request_service ?trip_plan - trip_plan)
    (trip_plan_special_request_processed ?trip_plan - trip_plan)
    (optional_feature_available ?optional_feature - optional_feature)
    (trip_plan_optional_feature_assigned ?trip_plan - trip_plan ?optional_feature - optional_feature)
    (scheduling_preference_available ?scheduling_preference - scheduling_preference)
    (trip_plan_has_scheduling_preference ?trip_plan - trip_plan ?scheduling_preference - scheduling_preference)
    (service_option_available ?service_option - service_option)
    (trip_plan_has_service_option ?trip_plan - trip_plan ?service_option - service_option)
    (supplier_availability_available ?supplier_availability - supplier_availability)
    (trip_plan_has_supplier_availability ?trip_plan - trip_plan ?supplier_availability - supplier_availability)
    (global_constraint_available ?global_constraint - global_constraint)
    (element_bound_global_constraint ?itinerary_element - itinerary_element ?global_constraint - global_constraint)
    (travel_day_has_segment_flag ?travel_day - travel_day)
    (travel_day_variant_has_segment_flag ?travel_day_variant - travel_day_variant)
    (trip_plan_sequence_locked ?trip_plan - trip_plan)
  )
  (:action initialize_itinerary_element
    :parameters (?itinerary_element - itinerary_element)
    :precondition
      (and
        (not
          (itinerary_element_initialized ?itinerary_element)
        )
        (not
          (itinerary_element_committed ?itinerary_element)
        )
      )
    :effect (itinerary_element_initialized ?itinerary_element)
  )
  (:action bind_transport_option
    :parameters (?itinerary_element - itinerary_element ?transport_option - transport_option)
    :precondition
      (and
        (itinerary_element_initialized ?itinerary_element)
        (not
          (itinerary_element_transport_assigned ?itinerary_element)
        )
        (transport_option_available ?transport_option)
      )
    :effect
      (and
        (itinerary_element_transport_assigned ?itinerary_element)
        (transport_assigned_to_element ?itinerary_element ?transport_option)
        (not
          (transport_option_available ?transport_option)
        )
      )
  )
  (:action provisionally_assign_activity
    :parameters (?itinerary_element - itinerary_element ?activity - activity)
    :precondition
      (and
        (itinerary_element_initialized ?itinerary_element)
        (itinerary_element_transport_assigned ?itinerary_element)
        (activity_available ?activity)
      )
    :effect
      (and
        (activity_assigned_to_element ?itinerary_element ?activity)
        (not
          (activity_available ?activity)
        )
      )
  )
  (:action confirm_activity_time
    :parameters (?itinerary_element - itinerary_element ?activity - activity)
    :precondition
      (and
        (itinerary_element_initialized ?itinerary_element)
        (itinerary_element_transport_assigned ?itinerary_element)
        (activity_assigned_to_element ?itinerary_element ?activity)
        (not
          (itinerary_element_time_confirmed ?itinerary_element)
        )
      )
    :effect (itinerary_element_time_confirmed ?itinerary_element)
  )
  (:action release_activity_assignment
    :parameters (?itinerary_element - itinerary_element ?activity - activity)
    :precondition
      (and
        (activity_assigned_to_element ?itinerary_element ?activity)
      )
    :effect
      (and
        (activity_available ?activity)
        (not
          (activity_assigned_to_element ?itinerary_element ?activity)
        )
      )
  )
  (:action assign_venue_to_element
    :parameters (?itinerary_element - itinerary_element ?venue - venue)
    :precondition
      (and
        (itinerary_element_time_confirmed ?itinerary_element)
        (venue_available ?venue)
      )
    :effect
      (and
        (venue_assigned_to_element ?itinerary_element ?venue)
        (not
          (venue_available ?venue)
        )
      )
  )
  (:action release_venue_from_element
    :parameters (?itinerary_element - itinerary_element ?venue - venue)
    :precondition
      (and
        (venue_assigned_to_element ?itinerary_element ?venue)
      )
    :effect
      (and
        (venue_available ?venue)
        (not
          (venue_assigned_to_element ?itinerary_element ?venue)
        )
      )
  )
  (:action reserve_service_option_for_plan
    :parameters (?trip_plan - trip_plan ?service_option - service_option)
    :precondition
      (and
        (itinerary_element_time_confirmed ?trip_plan)
        (service_option_available ?service_option)
      )
    :effect
      (and
        (trip_plan_has_service_option ?trip_plan ?service_option)
        (not
          (service_option_available ?service_option)
        )
      )
  )
  (:action release_service_option_from_plan
    :parameters (?trip_plan - trip_plan ?service_option - service_option)
    :precondition
      (and
        (trip_plan_has_service_option ?trip_plan ?service_option)
      )
    :effect
      (and
        (service_option_available ?service_option)
        (not
          (trip_plan_has_service_option ?trip_plan ?service_option)
        )
      )
  )
  (:action reserve_supplier_availability_for_plan
    :parameters (?trip_plan - trip_plan ?supplier_availability - supplier_availability)
    :precondition
      (and
        (itinerary_element_time_confirmed ?trip_plan)
        (supplier_availability_available ?supplier_availability)
      )
    :effect
      (and
        (trip_plan_has_supplier_availability ?trip_plan ?supplier_availability)
        (not
          (supplier_availability_available ?supplier_availability)
        )
      )
  )
  (:action release_supplier_availability_from_plan
    :parameters (?trip_plan - trip_plan ?supplier_availability - supplier_availability)
    :precondition
      (and
        (trip_plan_has_supplier_availability ?trip_plan ?supplier_availability)
      )
    :effect
      (and
        (supplier_availability_available ?supplier_availability)
        (not
          (trip_plan_has_supplier_availability ?trip_plan ?supplier_availability)
        )
      )
  )
  (:action evaluate_pace_bucket_day_for_day
    :parameters (?travel_day - travel_day ?pace_bucket_day - pace_bucket_day ?activity - activity)
    :precondition
      (and
        (itinerary_element_time_confirmed ?travel_day)
        (activity_assigned_to_element ?travel_day ?activity)
        (travel_day_has_pace_bucket_day ?travel_day ?pace_bucket_day)
        (not
          (pace_bucket_day_primary ?pace_bucket_day)
        )
        (not
          (pace_bucket_day_secondary ?pace_bucket_day)
        )
      )
    :effect (pace_bucket_day_primary ?pace_bucket_day)
  )
  (:action apply_pace_bucket_day_with_venue_to_day
    :parameters (?travel_day - travel_day ?pace_bucket_day - pace_bucket_day ?venue - venue)
    :precondition
      (and
        (itinerary_element_time_confirmed ?travel_day)
        (venue_assigned_to_element ?travel_day ?venue)
        (travel_day_has_pace_bucket_day ?travel_day ?pace_bucket_day)
        (pace_bucket_day_primary ?pace_bucket_day)
        (not
          (travel_day_has_segment_flag ?travel_day)
        )
      )
    :effect
      (and
        (travel_day_has_segment_flag ?travel_day)
        (travel_day_segment_confirmed ?travel_day)
      )
  )
  (:action assign_experience_tag_and_lock_day
    :parameters (?travel_day - travel_day ?pace_bucket_day - pace_bucket_day ?experience_tag - experience_tag)
    :precondition
      (and
        (itinerary_element_time_confirmed ?travel_day)
        (travel_day_has_pace_bucket_day ?travel_day ?pace_bucket_day)
        (experience_tag_available ?experience_tag)
        (not
          (travel_day_has_segment_flag ?travel_day)
        )
      )
    :effect
      (and
        (pace_bucket_day_secondary ?pace_bucket_day)
        (travel_day_has_segment_flag ?travel_day)
        (travel_day_tag_assigned ?travel_day ?experience_tag)
        (not
          (experience_tag_available ?experience_tag)
        )
      )
  )
  (:action finalize_day_segment_with_activity_and_tag
    :parameters (?travel_day - travel_day ?pace_bucket_day - pace_bucket_day ?activity - activity ?experience_tag - experience_tag)
    :precondition
      (and
        (itinerary_element_time_confirmed ?travel_day)
        (activity_assigned_to_element ?travel_day ?activity)
        (travel_day_has_pace_bucket_day ?travel_day ?pace_bucket_day)
        (pace_bucket_day_secondary ?pace_bucket_day)
        (travel_day_tag_assigned ?travel_day ?experience_tag)
        (not
          (travel_day_segment_confirmed ?travel_day)
        )
      )
    :effect
      (and
        (pace_bucket_day_primary ?pace_bucket_day)
        (travel_day_segment_confirmed ?travel_day)
        (experience_tag_available ?experience_tag)
        (not
          (travel_day_tag_assigned ?travel_day ?experience_tag)
        )
      )
  )
  (:action evaluate_pace_bucket_variant_for_variant
    :parameters (?travel_day_variant - travel_day_variant ?pace_bucket_variant - pace_bucket_variant ?activity - activity)
    :precondition
      (and
        (itinerary_element_time_confirmed ?travel_day_variant)
        (activity_assigned_to_element ?travel_day_variant ?activity)
        (travel_day_variant_has_pace_bucket_variant ?travel_day_variant ?pace_bucket_variant)
        (not
          (pace_bucket_variant_primary ?pace_bucket_variant)
        )
        (not
          (pace_bucket_variant_secondary ?pace_bucket_variant)
        )
      )
    :effect (pace_bucket_variant_primary ?pace_bucket_variant)
  )
  (:action apply_pace_bucket_variant_with_venue_to_variant
    :parameters (?travel_day_variant - travel_day_variant ?pace_bucket_variant - pace_bucket_variant ?venue - venue)
    :precondition
      (and
        (itinerary_element_time_confirmed ?travel_day_variant)
        (venue_assigned_to_element ?travel_day_variant ?venue)
        (travel_day_variant_has_pace_bucket_variant ?travel_day_variant ?pace_bucket_variant)
        (pace_bucket_variant_primary ?pace_bucket_variant)
        (not
          (travel_day_variant_has_segment_flag ?travel_day_variant)
        )
      )
    :effect
      (and
        (travel_day_variant_has_segment_flag ?travel_day_variant)
        (travel_day_variant_segment_confirmed ?travel_day_variant)
      )
  )
  (:action assign_experience_tag_and_lock_variant
    :parameters (?travel_day_variant - travel_day_variant ?pace_bucket_variant - pace_bucket_variant ?experience_tag - experience_tag)
    :precondition
      (and
        (itinerary_element_time_confirmed ?travel_day_variant)
        (travel_day_variant_has_pace_bucket_variant ?travel_day_variant ?pace_bucket_variant)
        (experience_tag_available ?experience_tag)
        (not
          (travel_day_variant_has_segment_flag ?travel_day_variant)
        )
      )
    :effect
      (and
        (pace_bucket_variant_secondary ?pace_bucket_variant)
        (travel_day_variant_has_segment_flag ?travel_day_variant)
        (travel_day_variant_tag_assigned ?travel_day_variant ?experience_tag)
        (not
          (experience_tag_available ?experience_tag)
        )
      )
  )
  (:action finalize_variant_segment_with_activity_and_tag
    :parameters (?travel_day_variant - travel_day_variant ?pace_bucket_variant - pace_bucket_variant ?activity - activity ?experience_tag - experience_tag)
    :precondition
      (and
        (itinerary_element_time_confirmed ?travel_day_variant)
        (activity_assigned_to_element ?travel_day_variant ?activity)
        (travel_day_variant_has_pace_bucket_variant ?travel_day_variant ?pace_bucket_variant)
        (pace_bucket_variant_secondary ?pace_bucket_variant)
        (travel_day_variant_tag_assigned ?travel_day_variant ?experience_tag)
        (not
          (travel_day_variant_segment_confirmed ?travel_day_variant)
        )
      )
    :effect
      (and
        (pace_bucket_variant_primary ?pace_bucket_variant)
        (travel_day_variant_segment_confirmed ?travel_day_variant)
        (experience_tag_available ?experience_tag)
        (not
          (travel_day_variant_tag_assigned ?travel_day_variant ?experience_tag)
        )
      )
  )
  (:action compose_visit_slot_from_day_segments
    :parameters (?travel_day - travel_day ?travel_day_variant - travel_day_variant ?pace_bucket_day - pace_bucket_day ?pace_bucket_variant - pace_bucket_variant ?visit_slot - visit_slot)
    :precondition
      (and
        (travel_day_has_segment_flag ?travel_day)
        (travel_day_variant_has_segment_flag ?travel_day_variant)
        (travel_day_has_pace_bucket_day ?travel_day ?pace_bucket_day)
        (travel_day_variant_has_pace_bucket_variant ?travel_day_variant ?pace_bucket_variant)
        (pace_bucket_day_primary ?pace_bucket_day)
        (pace_bucket_variant_primary ?pace_bucket_variant)
        (travel_day_segment_confirmed ?travel_day)
        (travel_day_variant_segment_confirmed ?travel_day_variant)
        (visit_slot_available ?visit_slot)
      )
    :effect
      (and
        (visit_slot_reserved ?visit_slot)
        (visit_slot_pace_bucket_day ?visit_slot ?pace_bucket_day)
        (visit_slot_pace_bucket_variant ?visit_slot ?pace_bucket_variant)
        (not
          (visit_slot_available ?visit_slot)
        )
      )
  )
  (:action compose_visit_slot_with_primary_option
    :parameters (?travel_day - travel_day ?travel_day_variant - travel_day_variant ?pace_bucket_day - pace_bucket_day ?pace_bucket_variant - pace_bucket_variant ?visit_slot - visit_slot)
    :precondition
      (and
        (travel_day_has_segment_flag ?travel_day)
        (travel_day_variant_has_segment_flag ?travel_day_variant)
        (travel_day_has_pace_bucket_day ?travel_day ?pace_bucket_day)
        (travel_day_variant_has_pace_bucket_variant ?travel_day_variant ?pace_bucket_variant)
        (pace_bucket_day_secondary ?pace_bucket_day)
        (pace_bucket_variant_primary ?pace_bucket_variant)
        (not
          (travel_day_segment_confirmed ?travel_day)
        )
        (travel_day_variant_segment_confirmed ?travel_day_variant)
        (visit_slot_available ?visit_slot)
      )
    :effect
      (and
        (visit_slot_reserved ?visit_slot)
        (visit_slot_pace_bucket_day ?visit_slot ?pace_bucket_day)
        (visit_slot_pace_bucket_variant ?visit_slot ?pace_bucket_variant)
        (visit_slot_variant_primary ?visit_slot)
        (not
          (visit_slot_available ?visit_slot)
        )
      )
  )
  (:action compose_visit_slot_with_secondary_option
    :parameters (?travel_day - travel_day ?travel_day_variant - travel_day_variant ?pace_bucket_day - pace_bucket_day ?pace_bucket_variant - pace_bucket_variant ?visit_slot - visit_slot)
    :precondition
      (and
        (travel_day_has_segment_flag ?travel_day)
        (travel_day_variant_has_segment_flag ?travel_day_variant)
        (travel_day_has_pace_bucket_day ?travel_day ?pace_bucket_day)
        (travel_day_variant_has_pace_bucket_variant ?travel_day_variant ?pace_bucket_variant)
        (pace_bucket_day_primary ?pace_bucket_day)
        (pace_bucket_variant_secondary ?pace_bucket_variant)
        (travel_day_segment_confirmed ?travel_day)
        (not
          (travel_day_variant_segment_confirmed ?travel_day_variant)
        )
        (visit_slot_available ?visit_slot)
      )
    :effect
      (and
        (visit_slot_reserved ?visit_slot)
        (visit_slot_pace_bucket_day ?visit_slot ?pace_bucket_day)
        (visit_slot_pace_bucket_variant ?visit_slot ?pace_bucket_variant)
        (visit_slot_variant_secondary ?visit_slot)
        (not
          (visit_slot_available ?visit_slot)
        )
      )
  )
  (:action compose_visit_slot_with_both_options
    :parameters (?travel_day - travel_day ?travel_day_variant - travel_day_variant ?pace_bucket_day - pace_bucket_day ?pace_bucket_variant - pace_bucket_variant ?visit_slot - visit_slot)
    :precondition
      (and
        (travel_day_has_segment_flag ?travel_day)
        (travel_day_variant_has_segment_flag ?travel_day_variant)
        (travel_day_has_pace_bucket_day ?travel_day ?pace_bucket_day)
        (travel_day_variant_has_pace_bucket_variant ?travel_day_variant ?pace_bucket_variant)
        (pace_bucket_day_secondary ?pace_bucket_day)
        (pace_bucket_variant_secondary ?pace_bucket_variant)
        (not
          (travel_day_segment_confirmed ?travel_day)
        )
        (not
          (travel_day_variant_segment_confirmed ?travel_day_variant)
        )
        (visit_slot_available ?visit_slot)
      )
    :effect
      (and
        (visit_slot_reserved ?visit_slot)
        (visit_slot_pace_bucket_day ?visit_slot ?pace_bucket_day)
        (visit_slot_pace_bucket_variant ?visit_slot ?pace_bucket_variant)
        (visit_slot_variant_primary ?visit_slot)
        (visit_slot_variant_secondary ?visit_slot)
        (not
          (visit_slot_available ?visit_slot)
        )
      )
  )
  (:action confirm_visit_slot_time
    :parameters (?visit_slot - visit_slot ?travel_day - travel_day ?activity - activity)
    :precondition
      (and
        (visit_slot_reserved ?visit_slot)
        (travel_day_has_segment_flag ?travel_day)
        (activity_assigned_to_element ?travel_day ?activity)
        (not
          (visit_slot_time_confirmed ?visit_slot)
        )
      )
    :effect (visit_slot_time_confirmed ?visit_slot)
  )
  (:action consume_activity_cluster_into_visit_slot
    :parameters (?trip_plan - trip_plan ?activity_cluster - activity_cluster ?visit_slot - visit_slot)
    :precondition
      (and
        (itinerary_element_time_confirmed ?trip_plan)
        (trip_plan_includes_visit_slot ?trip_plan ?visit_slot)
        (trip_plan_includes_activity_cluster ?trip_plan ?activity_cluster)
        (activity_cluster_available ?activity_cluster)
        (visit_slot_reserved ?visit_slot)
        (visit_slot_time_confirmed ?visit_slot)
        (not
          (activity_cluster_consumed ?activity_cluster)
        )
      )
    :effect
      (and
        (activity_cluster_consumed ?activity_cluster)
        (activity_cluster_assigned_to_visit_slot ?activity_cluster ?visit_slot)
        (not
          (activity_cluster_available ?activity_cluster)
        )
      )
  )
  (:action finalize_activity_cluster_assignment
    :parameters (?trip_plan - trip_plan ?activity_cluster - activity_cluster ?visit_slot - visit_slot ?activity - activity)
    :precondition
      (and
        (itinerary_element_time_confirmed ?trip_plan)
        (trip_plan_includes_activity_cluster ?trip_plan ?activity_cluster)
        (activity_cluster_consumed ?activity_cluster)
        (activity_cluster_assigned_to_visit_slot ?activity_cluster ?visit_slot)
        (activity_assigned_to_element ?trip_plan ?activity)
        (not
          (visit_slot_variant_primary ?visit_slot)
        )
        (not
          (trip_plan_services_integrated ?trip_plan)
        )
      )
    :effect (trip_plan_services_integrated ?trip_plan)
  )
  (:action request_optional_feature_for_plan
    :parameters (?trip_plan - trip_plan ?optional_feature - optional_feature)
    :precondition
      (and
        (itinerary_element_time_confirmed ?trip_plan)
        (optional_feature_available ?optional_feature)
        (not
          (trip_plan_optional_feature_requested ?trip_plan)
        )
      )
    :effect
      (and
        (trip_plan_optional_feature_requested ?trip_plan)
        (trip_plan_optional_feature_assigned ?trip_plan ?optional_feature)
        (not
          (optional_feature_available ?optional_feature)
        )
      )
  )
  (:action integrate_optional_feature_into_plan
    :parameters (?trip_plan - trip_plan ?activity_cluster - activity_cluster ?visit_slot - visit_slot ?activity - activity ?optional_feature - optional_feature)
    :precondition
      (and
        (itinerary_element_time_confirmed ?trip_plan)
        (trip_plan_includes_activity_cluster ?trip_plan ?activity_cluster)
        (activity_cluster_consumed ?activity_cluster)
        (activity_cluster_assigned_to_visit_slot ?activity_cluster ?visit_slot)
        (activity_assigned_to_element ?trip_plan ?activity)
        (visit_slot_variant_primary ?visit_slot)
        (trip_plan_optional_feature_requested ?trip_plan)
        (trip_plan_optional_feature_assigned ?trip_plan ?optional_feature)
        (not
          (trip_plan_services_integrated ?trip_plan)
        )
      )
    :effect
      (and
        (trip_plan_services_integrated ?trip_plan)
        (trip_plan_optional_feature_integrated ?trip_plan)
