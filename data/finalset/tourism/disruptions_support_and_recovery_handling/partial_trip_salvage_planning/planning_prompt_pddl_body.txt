    (salvage_open ?salvage_case - salvage_case)
    (salvage_validated ?salvage_case - salvage_case)

    (salvage_closed ?salvage_case - salvage_case)

    (alternative_service_option_available ?alternative_service_option - alternative_service_option)
    (salvage_assigned_alternative_service_option ?salvage_case - salvage_case ?alternative_service_option - alternative_service_option)
    (communication_channel_available ?communication_channel - communication_channel)
    (salvage_assigned_channel ?salvage_case - salvage_case ?communication_channel - communication_channel)
    (service_agent_available ?service_agent - service_agent)
    (salvage_assigned_agent ?salvage_case - salvage_case ?service_agent - service_agent)
    (compensation_package_available ?compensation_package - compensation_package)
    (passenger_case_allocated_compensation ?passenger_case_individual - passenger_case_individual ?compensation_package - compensation_package)
    (passenger_case_group_member_allocated_compensation ?passenger_case_group_member - passenger_case_group_member ?compensation_package - compensation_package)
    (case_impacted_itinerary_segment ?passenger_case_individual - passenger_case_individual ?itinerary_segment - itinerary_segment)
    (itinerary_segment_lock_pending ?itinerary_segment - itinerary_segment)
    (itinerary_segment_lock_confirmed ?itinerary_segment - itinerary_segment)
    (passenger_case_segment_confirmed ?passenger_case_individual - passenger_case_individual)

    (replacement_option_lock_confirmed ?replacement_segment_option - replacement_segment_option)
    (passenger_case_group_member_segment_confirmed ?passenger_case_group_member - passenger_case_group_member)
    (salvage_plan_slot_open ?salvage_plan_proposal - salvage_plan_proposal)
    (salvage_plan_proposed ?salvage_plan_proposal - salvage_plan_proposal)
    (salvage_plan_includes_segment ?salvage_plan_proposal - salvage_plan_proposal ?itinerary_segment - itinerary_segment)
    (salvage_plan_includes_replacement_option ?salvage_plan_proposal - salvage_plan_proposal ?replacement_segment_option - replacement_segment_option)
    (salvage_plan_vendor_confirmed ?salvage_plan_proposal - salvage_plan_proposal)
    (salvage_plan_funding_confirmed ?salvage_plan_proposal - salvage_plan_proposal)
    (salvage_plan_validated ?salvage_plan_proposal - salvage_plan_proposal)
    (operator_case_covers_passenger_case ?operator_case - operator_case ?passenger_case_individual - passenger_case_individual)
    (operator_case_covers_group_member ?operator_case - operator_case ?passenger_case_group_member - passenger_case_group_member)
    (operator_case_includes_plan_proposal ?operator_case - operator_case ?salvage_plan_proposal - salvage_plan_proposal)
    (service_package_available ?service_package - service_package)

    (operator_case_asset_allocated ?operator_case - operator_case)
    (operator_case_asset_confirmed ?operator_case - operator_case)
    (policy_clause_applied_to_operator_case ?operator_case - operator_case)
    (response_template_selected_for_operator_case ?operator_case - operator_case)

    (replacement_asset_available ?replacement_asset - replacement_asset)

    (operator_case_finalized_for_settlement ?operator_case - operator_case)

  (:action create_salvage_case
    :parameters (?salvage_case - salvage_case)
    :precondition
      (and
        (not
          (salvage_open ?salvage_case)
        )
        (not
          (salvage_closed ?salvage_case)
        )
      )
    :effect (salvage_open ?salvage_case)
  )
  (:action assign_alternative_option_to_case
    :parameters (?salvage_case - salvage_case ?alternative_service_option - alternative_service_option)
    :precondition
      (and
        (salvage_open ?salvage_case)
        (not
          (salvage_option_assigned ?salvage_case)
        )
        (alternative_service_option_available ?alternative_service_option)
      )
    :effect
      (and
        (salvage_option_assigned ?salvage_case)
        (salvage_assigned_alternative_service_option ?salvage_case ?alternative_service_option)
        (not
          (alternative_service_option_available ?alternative_service_option)
        )
      )
  )
  (:action attach_communication_channel_to_case
    :parameters (?salvage_case - salvage_case ?communication_channel - communication_channel)
    :precondition
      (and
        (salvage_open ?salvage_case)
        (salvage_option_assigned ?salvage_case)
        (communication_channel_available ?communication_channel)
      )
    :effect
      (and
        (salvage_assigned_channel ?salvage_case ?communication_channel)
        (not
          (communication_channel_available ?communication_channel)
        )
      )
  )
  (:action validate_salvage_case_with_channel
    :parameters (?salvage_case - salvage_case ?communication_channel - communication_channel)
    :precondition
      (and
        (salvage_open ?salvage_case)
        (salvage_option_assigned ?salvage_case)
        (salvage_assigned_channel ?salvage_case ?communication_channel)
        (not
          (salvage_validated ?salvage_case)
        )
      )
    :effect (salvage_validated ?salvage_case)
  )
  (:action release_communication_channel_from_case
    :parameters (?salvage_case - salvage_case ?communication_channel - communication_channel)
    :precondition
      (and
        (salvage_assigned_channel ?salvage_case ?communication_channel)
      )
    :effect
      (and
        (communication_channel_available ?communication_channel)
        (not
          (salvage_assigned_channel ?salvage_case ?communication_channel)
        )
      )
  )
  (:action assign_service_agent_to_case
    :parameters (?salvage_case - salvage_case ?service_agent - service_agent)
    :precondition
      (and
        (salvage_validated ?salvage_case)
        (service_agent_available ?service_agent)
      )
    :effect
      (and
        (salvage_assigned_agent ?salvage_case ?service_agent)
        (not
          (service_agent_available ?service_agent)
        )
      )
  )
  (:action release_service_agent_from_case
    :parameters (?salvage_case - salvage_case ?service_agent - service_agent)
    :precondition
      (and
        (salvage_assigned_agent ?salvage_case ?service_agent)
      )
    :effect
      (and
        (service_agent_available ?service_agent)
        (not
          (salvage_assigned_agent ?salvage_case ?service_agent)
        )
      )
  )
  (:action allocate_replacement_asset_to_operator_case
    :parameters (?operator_case - operator_case ?replacement_asset - replacement_asset)
    :precondition
      (and
        (salvage_validated ?operator_case)
        (replacement_asset_available ?replacement_asset)
      )
    :effect
      (and
        (operator_case_assigned_replacement_asset ?operator_case ?replacement_asset)
        (not
          (replacement_asset_available ?replacement_asset)
        )
      )
  )
  (:action release_replacement_asset_from_operator_case
    :parameters (?operator_case - operator_case ?replacement_asset - replacement_asset)
    :precondition
      (and
        (operator_case_assigned_replacement_asset ?operator_case ?replacement_asset)
      )
    :effect
      (and
        (replacement_asset_available ?replacement_asset)
        (not
          (operator_case_assigned_replacement_asset ?operator_case ?replacement_asset)
        )
      )
  )
  (:action apply_policy_clause_to_operator_case
    :parameters (?operator_case - operator_case ?policy_clause - policy_clause)
    :precondition
      (and
        (salvage_validated ?operator_case)
        (policy_clause_available ?policy_clause)
      )
    :effect
      (and
        (operator_case_applied_policy_clause ?operator_case ?policy_clause)
        (not
          (policy_clause_available ?policy_clause)
        )
      )
  )
  (:action remove_policy_clause_from_operator_case
    :parameters (?operator_case - operator_case ?policy_clause - policy_clause)
    :precondition
      (and
        (operator_case_applied_policy_clause ?operator_case ?policy_clause)
      )
    :effect
      (and
        (policy_clause_available ?policy_clause)
        (not
          (operator_case_applied_policy_clause ?operator_case ?policy_clause)
        )
      )
  )
  (:action request_itinerary_segment_lock
    :parameters (?passenger_case_individual - passenger_case_individual ?itinerary_segment - itinerary_segment ?communication_channel - communication_channel)
    :precondition
      (and
        (salvage_validated ?passenger_case_individual)
        (salvage_assigned_channel ?passenger_case_individual ?communication_channel)
        (case_impacted_itinerary_segment ?passenger_case_individual ?itinerary_segment)
        (not
          (itinerary_segment_lock_pending ?itinerary_segment)
        )
        (not
          (itinerary_segment_lock_confirmed ?itinerary_segment)
        )
      )
    :effect (itinerary_segment_lock_pending ?itinerary_segment)
  )
  (:action confirm_segment_lock_with_agent
    :parameters (?passenger_case_individual - passenger_case_individual ?itinerary_segment - itinerary_segment ?service_agent - service_agent)
    :precondition
      (and
        (salvage_validated ?passenger_case_individual)
        (salvage_assigned_agent ?passenger_case_individual ?service_agent)
        (case_impacted_itinerary_segment ?passenger_case_individual ?itinerary_segment)
        (itinerary_segment_lock_pending ?itinerary_segment)
        (not
          (passenger_case_segment_locked ?passenger_case_individual)
        )
      )
    :effect
      (and
        (passenger_case_segment_locked ?passenger_case_individual)
        (passenger_case_segment_confirmed ?passenger_case_individual)
      )
  )
  (:action reserve_compensation_and_lock_segment
    :parameters (?passenger_case_individual - passenger_case_individual ?itinerary_segment - itinerary_segment ?compensation_package - compensation_package)
    :precondition
      (and
        (salvage_validated ?passenger_case_individual)
        (case_impacted_itinerary_segment ?passenger_case_individual ?itinerary_segment)
        (compensation_package_available ?compensation_package)
        (not
          (passenger_case_segment_locked ?passenger_case_individual)
        )
      )
    :effect
      (and
        (itinerary_segment_lock_confirmed ?itinerary_segment)
        (passenger_case_segment_locked ?passenger_case_individual)
        (passenger_case_allocated_compensation ?passenger_case_individual ?compensation_package)
        (not
          (compensation_package_available ?compensation_package)
        )
      )
  )
  (:action finalize_segment_reservation
    :parameters (?passenger_case_individual - passenger_case_individual ?itinerary_segment - itinerary_segment ?communication_channel - communication_channel ?compensation_package - compensation_package)
    :precondition
      (and
        (salvage_validated ?passenger_case_individual)
        (salvage_assigned_channel ?passenger_case_individual ?communication_channel)
        (case_impacted_itinerary_segment ?passenger_case_individual ?itinerary_segment)
        (itinerary_segment_lock_confirmed ?itinerary_segment)
        (passenger_case_allocated_compensation ?passenger_case_individual ?compensation_package)
        (not
          (passenger_case_segment_confirmed ?passenger_case_individual)
        )
      )
    :effect
      (and
        (itinerary_segment_lock_pending ?itinerary_segment)
        (passenger_case_segment_confirmed ?passenger_case_individual)
        (compensation_package_available ?compensation_package)
        (not
          (passenger_case_allocated_compensation ?passenger_case_individual ?compensation_package)
        )
      )
  )
  (:action request_replacement_option_lock_for_group_case
    :parameters (?passenger_case_group_member - passenger_case_group_member ?replacement_segment_option - replacement_segment_option ?communication_channel - communication_channel)
    :precondition
      (and
        (salvage_validated ?passenger_case_group_member)
        (salvage_assigned_channel ?passenger_case_group_member ?communication_channel)
        (passenger_case_group_member_assigned_replacement_option ?passenger_case_group_member ?replacement_segment_option)
        (not
          (replacement_option_lock_pending ?replacement_segment_option)
        )
        (not
          (replacement_option_lock_confirmed ?replacement_segment_option)
        )
      )
    :effect (replacement_option_lock_pending ?replacement_segment_option)
  )
  (:action confirm_replacement_option_with_agent_for_group_case
    :parameters (?passenger_case_group_member - passenger_case_group_member ?replacement_segment_option - replacement_segment_option ?service_agent - service_agent)
    :precondition
      (and
        (salvage_validated ?passenger_case_group_member)
        (salvage_assigned_agent ?passenger_case_group_member ?service_agent)
        (passenger_case_group_member_assigned_replacement_option ?passenger_case_group_member ?replacement_segment_option)
        (replacement_option_lock_pending ?replacement_segment_option)
        (not
          (passenger_case_group_member_segment_locked ?passenger_case_group_member)
        )
      )
    :effect
      (and
        (passenger_case_group_member_segment_locked ?passenger_case_group_member)
        (passenger_case_group_member_segment_confirmed ?passenger_case_group_member)
      )
  )
  (:action reserve_compensation_for_group_replacement_option
    :parameters (?passenger_case_group_member - passenger_case_group_member ?replacement_segment_option - replacement_segment_option ?compensation_package - compensation_package)
    :precondition
      (and
        (salvage_validated ?passenger_case_group_member)
        (passenger_case_group_member_assigned_replacement_option ?passenger_case_group_member ?replacement_segment_option)
        (compensation_package_available ?compensation_package)
        (not
          (passenger_case_group_member_segment_locked ?passenger_case_group_member)
        )
      )
    :effect
      (and
        (replacement_option_lock_confirmed ?replacement_segment_option)
        (passenger_case_group_member_segment_locked ?passenger_case_group_member)
        (passenger_case_group_member_allocated_compensation ?passenger_case_group_member ?compensation_package)
        (not
          (compensation_package_available ?compensation_package)
        )
      )
  )
  (:action finalize_group_replacement_reservation
    :parameters (?passenger_case_group_member - passenger_case_group_member ?replacement_segment_option - replacement_segment_option ?communication_channel - communication_channel ?compensation_package - compensation_package)
    :precondition
      (and
        (salvage_validated ?passenger_case_group_member)
        (salvage_assigned_channel ?passenger_case_group_member ?communication_channel)
        (passenger_case_group_member_assigned_replacement_option ?passenger_case_group_member ?replacement_segment_option)
        (replacement_option_lock_confirmed ?replacement_segment_option)
        (passenger_case_group_member_allocated_compensation ?passenger_case_group_member ?compensation_package)
        (not
          (passenger_case_group_member_segment_confirmed ?passenger_case_group_member)
        )
      )
    :effect
      (and
        (replacement_option_lock_pending ?replacement_segment_option)
        (passenger_case_group_member_segment_confirmed ?passenger_case_group_member)
        (compensation_package_available ?compensation_package)
        (not
          (passenger_case_group_member_allocated_compensation ?passenger_case_group_member ?compensation_package)
        )
      )
  )
  (:action compile_salvage_plan_proposal
    :parameters (?passenger_case_individual - passenger_case_individual ?passenger_case_group_member - passenger_case_group_member ?itinerary_segment - itinerary_segment ?replacement_segment_option - replacement_segment_option ?salvage_plan_proposal - salvage_plan_proposal)
    :precondition
      (and
        (passenger_case_segment_locked ?passenger_case_individual)
        (passenger_case_group_member_segment_locked ?passenger_case_group_member)
        (case_impacted_itinerary_segment ?passenger_case_individual ?itinerary_segment)
        (passenger_case_group_member_assigned_replacement_option ?passenger_case_group_member ?replacement_segment_option)
        (itinerary_segment_lock_pending ?itinerary_segment)
        (replacement_option_lock_pending ?replacement_segment_option)
        (passenger_case_segment_confirmed ?passenger_case_individual)
        (passenger_case_group_member_segment_confirmed ?passenger_case_group_member)
        (salvage_plan_slot_open ?salvage_plan_proposal)
      )
    :effect
      (and
        (salvage_plan_proposed ?salvage_plan_proposal)
        (salvage_plan_includes_segment ?salvage_plan_proposal ?itinerary_segment)
        (salvage_plan_includes_replacement_option ?salvage_plan_proposal ?replacement_segment_option)
        (not
          (salvage_plan_slot_open ?salvage_plan_proposal)
        )
      )
  )
  (:action compile_salvage_plan_with_vendor_confirmation

    :precondition

        (itinerary_segment_lock_confirmed ?itinerary_segment)
        (replacement_option_lock_pending ?replacement_segment_option)
        (not
          (passenger_case_segment_confirmed ?passenger_case_individual)
        )
        (passenger_case_group_member_segment_confirmed ?passenger_case_group_member)
        (salvage_plan_slot_open ?salvage_plan_proposal)
      )
    :effect
      (and
        (salvage_plan_proposed ?salvage_plan_proposal)
        (salvage_plan_includes_segment ?salvage_plan_proposal ?itinerary_segment)
        (salvage_plan_includes_replacement_option ?salvage_plan_proposal ?replacement_segment_option)
        (salvage_plan_vendor_confirmed ?salvage_plan_proposal)
        (not
          (salvage_plan_slot_open ?salvage_plan_proposal)
        )
      )
  )
  (:action compile_salvage_plan_with_funding_confirmation

        (itinerary_segment_lock_pending ?itinerary_segment)
        (replacement_option_lock_confirmed ?replacement_segment_option)
        (passenger_case_segment_confirmed ?passenger_case_individual)
        (not
          (passenger_case_group_member_segment_confirmed ?passenger_case_group_member)

        (not
          (salvage_plan_slot_open ?salvage_plan_proposal)
        )
      )
  )
  (:action compile_complete_salvage_plan_proposal
    :parameters (?passenger_case_individual - passenger_case_individual ?passenger_case_group_member - passenger_case_group_member ?itinerary_segment - itinerary_segment ?replacement_segment_option - replacement_segment_option ?salvage_plan_proposal - salvage_plan_proposal)

      (and

        (itinerary_segment_lock_confirmed ?itinerary_segment)
        (replacement_option_lock_confirmed ?replacement_segment_option)
        (not
          (passenger_case_segment_confirmed ?passenger_case_individual)
        )
        (not
          (passenger_case_group_member_segment_confirmed ?passenger_case_group_member)
        )
        (salvage_plan_slot_open ?salvage_plan_proposal)

    :effect
      (and
        (salvage_plan_proposed ?salvage_plan_proposal)
        (salvage_plan_includes_segment ?salvage_plan_proposal ?itinerary_segment)
        (salvage_plan_includes_replacement_option ?salvage_plan_proposal ?replacement_segment_option)
        (salvage_plan_vendor_confirmed ?salvage_plan_proposal)
        (salvage_plan_funding_confirmed ?salvage_plan_proposal)
        (not
          (salvage_plan_slot_open ?salvage_plan_proposal)
        )
      )
  )
  (:action validate_and_lock_salvage_plan
    :parameters (?salvage_plan_proposal - salvage_plan_proposal ?passenger_case_individual - passenger_case_individual ?communication_channel - communication_channel)
    :precondition
      (and
        (salvage_plan_proposed ?salvage_plan_proposal)
        (passenger_case_segment_locked ?passenger_case_individual)
        (salvage_assigned_channel ?passenger_case_individual ?communication_channel)
        (not
          (salvage_plan_validated ?salvage_plan_proposal)
        )
      )
