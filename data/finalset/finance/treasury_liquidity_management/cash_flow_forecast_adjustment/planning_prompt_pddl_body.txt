  (:action register_adjustment_case
    :parameters (?adjustment_case - adjustment_case)
    :precondition
      (and
        (not
          (case_registered ?adjustment_case)
        )
        (not
          (routing_assigned ?adjustment_case)
        )
      )
    :effect (case_registered ?adjustment_case)
  )
  (:action associate_funding_candidate
    :parameters (?adjustment_case - adjustment_case ?funding_source_candidate - funding_source_candidate)
    :precondition
      (and
        (case_registered ?adjustment_case)
        (not
          (has_funding_candidate ?adjustment_case)
        )
        (candidate_available ?funding_source_candidate)
      )
    :effect
      (and
        (has_funding_candidate ?adjustment_case)
        (entity_funding_candidate_link ?adjustment_case ?funding_source_candidate)
        (not
          (candidate_available ?funding_source_candidate)
        )
      )
  )
  (:action reserve_time_bucket_for_case
    :parameters (?adjustment_case - adjustment_case ?time_period_bucket - time_period_bucket)
    :precondition
      (and
        (case_registered ?adjustment_case)

    :effect
      (and
        (entity_time_bucket_association ?adjustment_case ?time_period_bucket)
        (not
          (time_bucket_available ?time_period_bucket)
        )
      )
  )

  )

      )
    :effect

        (time_bucket_available ?time_period_bucket)
