## Planning domain (PDDL)

(define (domain short_term_investment_laddering_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types account_category - object counterparty_category - object instrument_category - object account_generic - object account_unit - account_generic counterparty_bank - account_category instrument_offer - account_category execution_channel - account_category approval_token - account_category allocation_policy - account_category liquidity_tag - account_category pricing_eligibility - account_category settlement_option - account_category buffer_tranche - counterparty_category security_id - counterparty_category deal_approval - counterparty_category maturity_bucket_short - instrument_category maturity_bucket_medium - instrument_category investment_lot - instrument_category account_subtype_a - account_unit account_subtype_b - account_unit entity_a_account - account_subtype_a entity_b_account - account_subtype_a treasury_pool - account_subtype_b)
  (:predicates
    (account_candidate ?account_unit - account_unit)
    (account_approved ?account_unit - account_unit)
    (counterparty_assigned_to_account ?account_unit - account_unit)
    (invested_status ?account_unit - account_unit)
    (processed_settlement_flag ?account_unit - account_unit)
    (investment_finalized_for_account ?account_unit - account_unit)
    (counterparty_available ?counterparty_bank - counterparty_bank)
    (account_to_counterparty_link ?account_unit - account_unit ?counterparty_bank - counterparty_bank)
    (instrument_offer_available ?instrument_offer - instrument_offer)
    (instrument_reserved_for_account ?account_unit - account_unit ?instrument_offer - instrument_offer)
    (execution_channel_available ?execution_channel - execution_channel)
    (execution_channel_booked ?account_unit - account_unit ?execution_channel - execution_channel)
    (buffer_tranche_available ?buffer_tranche - buffer_tranche)
    (entity_a_buffer_allocated ?entity_a_account - entity_a_account ?buffer_tranche - buffer_tranche)
    (entity_b_buffer_allocated ?entity_b_account - entity_b_account ?buffer_tranche - buffer_tranche)
    (entity_a_to_maturity_bucket_link ?entity_a_account - entity_a_account ?maturity_bucket_short - maturity_bucket_short)
    (bucket_signal_primary_short ?maturity_bucket_short - maturity_bucket_short)
    (bucket_signal_secondary_short ?maturity_bucket_short - maturity_bucket_short)
    (entity_a_ready_flag ?entity_a_account - entity_a_account)
    (entity_b_to_maturity_bucket_link ?entity_b_account - entity_b_account ?maturity_bucket_medium - maturity_bucket_medium)
    (bucket_signal_primary_medium ?maturity_bucket_medium - maturity_bucket_medium)
    (bucket_signal_secondary_medium ?maturity_bucket_medium - maturity_bucket_medium)
    (entity_b_ready_flag ?entity_b_account - entity_b_account)
    (investment_lot_available ?investment_lot - investment_lot)
    (investment_lot_allocated ?investment_lot - investment_lot)
    (lot_linked_to_bucket_short ?investment_lot - investment_lot ?maturity_bucket_short - maturity_bucket_short)
    (lot_linked_to_bucket_medium ?investment_lot - investment_lot ?maturity_bucket_medium - maturity_bucket_medium)
    (lot_attribute_first_leg ?investment_lot - investment_lot)
    (lot_attribute_second_leg ?investment_lot - investment_lot)
    (lot_attribute_settlement_ready ?investment_lot - investment_lot)
    (pool_to_entity_a_mapping ?treasury_pool - treasury_pool ?entity_a_account - entity_a_account)
    (pool_to_entity_b_mapping ?treasury_pool - treasury_pool ?entity_b_account - entity_b_account)
    (pool_to_lot_mapping ?treasury_pool - treasury_pool ?investment_lot - investment_lot)
    (security_id_unbound ?security_id - security_id)
    (pool_to_security_mapping ?treasury_pool - treasury_pool ?security_id - security_id)
    (security_id_bound ?security_id - security_id)
    (security_bound_to_lot ?security_id - security_id ?investment_lot - investment_lot)
    (pool_flag_stage1 ?treasury_pool - treasury_pool)
    (pool_flag_stage2 ?treasury_pool - treasury_pool)
    (pool_flag_approval_complete ?treasury_pool - treasury_pool)
    (pool_flag_approval_attached ?treasury_pool - treasury_pool)
    (pool_allocations_validated ?treasury_pool - treasury_pool)
    (pool_allocations_confirmed ?treasury_pool - treasury_pool)
    (pool_validation_done ?treasury_pool - treasury_pool)
    (deal_approval_available ?deal_approval - deal_approval)
    (pool_has_deal_approval ?treasury_pool - treasury_pool ?deal_approval - deal_approval)
    (pool_flag_deal_approval ?treasury_pool - treasury_pool)
    (pool_flag_deal_approval_stage ?treasury_pool - treasury_pool)
    (pool_flag_deal_approval_confirmed ?treasury_pool - treasury_pool)
    (approval_token_available ?approval_token - approval_token)
    (pool_has_approval_token ?treasury_pool - treasury_pool ?approval_token - approval_token)
    (allocation_policy_available ?allocation_policy - allocation_policy)
    (pool_has_allocation_policy ?treasury_pool - treasury_pool ?allocation_policy - allocation_policy)
    (pricing_eligibility_available ?pricing_eligibility - pricing_eligibility)
    (pool_has_pricing_eligibility ?treasury_pool - treasury_pool ?pricing_eligibility - pricing_eligibility)
    (settlement_option_available ?settlement_option - settlement_option)
    (pool_has_settlement_option ?treasury_pool - treasury_pool ?settlement_option - settlement_option)
    (liquidity_tag_available ?liquidity_tag - liquidity_tag)
    (account_has_liquidity_tag ?account_unit - account_unit ?liquidity_tag - liquidity_tag)
    (entity_a_processing_flag ?entity_a_account - entity_a_account)
    (entity_b_processing_flag ?entity_b_account - entity_b_account)
    (pool_postprocessing_flag ?treasury_pool - treasury_pool)
  )
  (:action identify_account_candidate
    :parameters (?account_unit - account_unit)
    :precondition
      (and
        (not
          (account_candidate ?account_unit)
        )
        (not
          (invested_status ?account_unit)
        )
      )
    :effect (account_candidate ?account_unit)
  )
  (:action assign_counterparty_to_account
    :parameters (?account_unit - account_unit ?counterparty_bank - counterparty_bank)
    :precondition
      (and
        (account_candidate ?account_unit)
        (not
          (counterparty_assigned_to_account ?account_unit)
        )
        (counterparty_available ?counterparty_bank)
      )
    :effect
      (and
        (counterparty_assigned_to_account ?account_unit)
        (account_to_counterparty_link ?account_unit ?counterparty_bank)
        (not
          (counterparty_available ?counterparty_bank)
        )
      )
  )
  (:action reserve_instrument_for_account
    :parameters (?account_unit - account_unit ?instrument_offer - instrument_offer)
    :precondition
      (and
        (account_candidate ?account_unit)
        (counterparty_assigned_to_account ?account_unit)
        (instrument_offer_available ?instrument_offer)
      )
    :effect
      (and
        (instrument_reserved_for_account ?account_unit ?instrument_offer)
        (not
          (instrument_offer_available ?instrument_offer)
        )
      )
  )
  (:action approve_account_for_investment
    :parameters (?account_unit - account_unit ?instrument_offer - instrument_offer)
    :precondition
      (and
        (account_candidate ?account_unit)
        (counterparty_assigned_to_account ?account_unit)
        (instrument_reserved_for_account ?account_unit ?instrument_offer)
        (not
          (account_approved ?account_unit)
        )
      )
    :effect (account_approved ?account_unit)
  )
  (:action release_instrument_reservation
    :parameters (?account_unit - account_unit ?instrument_offer - instrument_offer)
    :precondition
      (and
        (instrument_reserved_for_account ?account_unit ?instrument_offer)
      )
    :effect
      (and
        (instrument_offer_available ?instrument_offer)
        (not
          (instrument_reserved_for_account ?account_unit ?instrument_offer)
        )
      )
  )
  (:action book_execution_channel_for_account
    :parameters (?account_unit - account_unit ?execution_channel - execution_channel)
    :precondition
      (and
        (account_approved ?account_unit)
        (execution_channel_available ?execution_channel)
      )
    :effect
      (and
        (execution_channel_booked ?account_unit ?execution_channel)
        (not
          (execution_channel_available ?execution_channel)
        )
      )
  )
  (:action release_execution_channel_for_account
    :parameters (?account_unit - account_unit ?execution_channel - execution_channel)
    :precondition
      (and
        (execution_channel_booked ?account_unit ?execution_channel)
      )
    :effect
      (and
        (execution_channel_available ?execution_channel)
        (not
          (execution_channel_booked ?account_unit ?execution_channel)
        )
      )
  )
  (:action attach_pricing_eligibility_to_pool
    :parameters (?treasury_pool - treasury_pool ?pricing_eligibility - pricing_eligibility)
    :precondition
      (and
        (account_approved ?treasury_pool)
        (pricing_eligibility_available ?pricing_eligibility)
      )
    :effect
      (and
        (pool_has_pricing_eligibility ?treasury_pool ?pricing_eligibility)
        (not
          (pricing_eligibility_available ?pricing_eligibility)
        )
      )
  )
  (:action detach_pricing_eligibility_from_pool
    :parameters (?treasury_pool - treasury_pool ?pricing_eligibility - pricing_eligibility)
    :precondition
      (and
        (pool_has_pricing_eligibility ?treasury_pool ?pricing_eligibility)
      )
    :effect
      (and
        (pricing_eligibility_available ?pricing_eligibility)
        (not
          (pool_has_pricing_eligibility ?treasury_pool ?pricing_eligibility)
        )
      )
  )
  (:action attach_settlement_option_to_pool
    :parameters (?treasury_pool - treasury_pool ?settlement_option - settlement_option)
    :precondition
      (and
        (account_approved ?treasury_pool)
        (settlement_option_available ?settlement_option)
      )
    :effect
      (and
        (pool_has_settlement_option ?treasury_pool ?settlement_option)
        (not
          (settlement_option_available ?settlement_option)
        )
      )
  )
  (:action detach_settlement_option_from_pool
    :parameters (?treasury_pool - treasury_pool ?settlement_option - settlement_option)
    :precondition
      (and
        (pool_has_settlement_option ?treasury_pool ?settlement_option)
      )
    :effect
      (and
        (settlement_option_available ?settlement_option)
        (not
          (pool_has_settlement_option ?treasury_pool ?settlement_option)
        )
      )
  )
  (:action set_primary_short_bucket_signal_for_entity_a
    :parameters (?entity_a_account - entity_a_account ?maturity_bucket_short - maturity_bucket_short ?instrument_offer - instrument_offer)
    :precondition
      (and
        (account_approved ?entity_a_account)
        (instrument_reserved_for_account ?entity_a_account ?instrument_offer)
        (entity_a_to_maturity_bucket_link ?entity_a_account ?maturity_bucket_short)
        (not
          (bucket_signal_primary_short ?maturity_bucket_short)
        )
        (not
          (bucket_signal_secondary_short ?maturity_bucket_short)
        )
      )
    :effect (bucket_signal_primary_short ?maturity_bucket_short)
  )
  (:action confirm_entity_a_ready_for_assignment
    :parameters (?entity_a_account - entity_a_account ?maturity_bucket_short - maturity_bucket_short ?execution_channel - execution_channel)
    :precondition
      (and
        (account_approved ?entity_a_account)
        (execution_channel_booked ?entity_a_account ?execution_channel)
        (entity_a_to_maturity_bucket_link ?entity_a_account ?maturity_bucket_short)
        (bucket_signal_primary_short ?maturity_bucket_short)
        (not
          (entity_a_processing_flag ?entity_a_account)
        )
      )
    :effect
      (and
        (entity_a_processing_flag ?entity_a_account)
        (entity_a_ready_flag ?entity_a_account)
      )
  )
  (:action assign_buffer_tranche_to_entity_a
    :parameters (?entity_a_account - entity_a_account ?maturity_bucket_short - maturity_bucket_short ?buffer_tranche - buffer_tranche)
    :precondition
      (and
        (account_approved ?entity_a_account)
        (entity_a_to_maturity_bucket_link ?entity_a_account ?maturity_bucket_short)
        (buffer_tranche_available ?buffer_tranche)
        (not
          (entity_a_processing_flag ?entity_a_account)
        )
      )
    :effect
      (and
        (bucket_signal_secondary_short ?maturity_bucket_short)
        (entity_a_processing_flag ?entity_a_account)
        (entity_a_buffer_allocated ?entity_a_account ?buffer_tranche)
        (not
          (buffer_tranche_available ?buffer_tranche)
        )
      )
  )
  (:action process_entity_a_secondary_bucket_signal
    :parameters (?entity_a_account - entity_a_account ?maturity_bucket_short - maturity_bucket_short ?instrument_offer - instrument_offer ?buffer_tranche - buffer_tranche)
    :precondition
      (and
        (account_approved ?entity_a_account)
        (instrument_reserved_for_account ?entity_a_account ?instrument_offer)
        (entity_a_to_maturity_bucket_link ?entity_a_account ?maturity_bucket_short)
        (bucket_signal_secondary_short ?maturity_bucket_short)
        (entity_a_buffer_allocated ?entity_a_account ?buffer_tranche)
        (not
          (entity_a_ready_flag ?entity_a_account)
        )
      )
    :effect
      (and
        (bucket_signal_primary_short ?maturity_bucket_short)
        (entity_a_ready_flag ?entity_a_account)
        (buffer_tranche_available ?buffer_tranche)
        (not
          (entity_a_buffer_allocated ?entity_a_account ?buffer_tranche)
        )
      )
  )
  (:action set_primary_medium_bucket_signal_for_entity_b
    :parameters (?entity_b_account - entity_b_account ?maturity_bucket_medium - maturity_bucket_medium ?instrument_offer - instrument_offer)
    :precondition
      (and
        (account_approved ?entity_b_account)
        (instrument_reserved_for_account ?entity_b_account ?instrument_offer)
        (entity_b_to_maturity_bucket_link ?entity_b_account ?maturity_bucket_medium)
        (not
          (bucket_signal_primary_medium ?maturity_bucket_medium)
        )
        (not
          (bucket_signal_secondary_medium ?maturity_bucket_medium)
        )
      )
    :effect (bucket_signal_primary_medium ?maturity_bucket_medium)
  )
  (:action confirm_entity_b_ready_for_assignment
    :parameters (?entity_b_account - entity_b_account ?maturity_bucket_medium - maturity_bucket_medium ?execution_channel - execution_channel)
    :precondition
      (and
        (account_approved ?entity_b_account)
        (execution_channel_booked ?entity_b_account ?execution_channel)
        (entity_b_to_maturity_bucket_link ?entity_b_account ?maturity_bucket_medium)
        (bucket_signal_primary_medium ?maturity_bucket_medium)
        (not
          (entity_b_processing_flag ?entity_b_account)
        )
      )
    :effect
      (and
        (entity_b_processing_flag ?entity_b_account)
        (entity_b_ready_flag ?entity_b_account)
      )
  )
  (:action assign_buffer_tranche_to_entity_b
    :parameters (?entity_b_account - entity_b_account ?maturity_bucket_medium - maturity_bucket_medium ?buffer_tranche - buffer_tranche)
    :precondition
      (and
        (account_approved ?entity_b_account)
        (entity_b_to_maturity_bucket_link ?entity_b_account ?maturity_bucket_medium)
        (buffer_tranche_available ?buffer_tranche)
        (not
          (entity_b_processing_flag ?entity_b_account)
        )
      )
    :effect
      (and
        (bucket_signal_secondary_medium ?maturity_bucket_medium)
        (entity_b_processing_flag ?entity_b_account)
        (entity_b_buffer_allocated ?entity_b_account ?buffer_tranche)
        (not
          (buffer_tranche_available ?buffer_tranche)
        )
      )
  )
  (:action process_entity_b_secondary_bucket_signal
    :parameters (?entity_b_account - entity_b_account ?maturity_bucket_medium - maturity_bucket_medium ?instrument_offer - instrument_offer ?buffer_tranche - buffer_tranche)
    :precondition
      (and
        (account_approved ?entity_b_account)
        (instrument_reserved_for_account ?entity_b_account ?instrument_offer)
        (entity_b_to_maturity_bucket_link ?entity_b_account ?maturity_bucket_medium)
        (bucket_signal_secondary_medium ?maturity_bucket_medium)
        (entity_b_buffer_allocated ?entity_b_account ?buffer_tranche)
        (not
          (entity_b_ready_flag ?entity_b_account)
        )
      )
    :effect
      (and
        (bucket_signal_primary_medium ?maturity_bucket_medium)
        (entity_b_ready_flag ?entity_b_account)
        (buffer_tranche_available ?buffer_tranche)
        (not
          (entity_b_buffer_allocated ?entity_b_account ?buffer_tranche)
        )
      )
  )
  (:action allocate_investment_lot_to_buckets
    :parameters (?entity_a_account - entity_a_account ?entity_b_account - entity_b_account ?maturity_bucket_short - maturity_bucket_short ?maturity_bucket_medium - maturity_bucket_medium ?investment_lot - investment_lot)
    :precondition
      (and
        (entity_a_processing_flag ?entity_a_account)
        (entity_b_processing_flag ?entity_b_account)
        (entity_a_to_maturity_bucket_link ?entity_a_account ?maturity_bucket_short)
        (entity_b_to_maturity_bucket_link ?entity_b_account ?maturity_bucket_medium)
        (bucket_signal_primary_short ?maturity_bucket_short)
        (bucket_signal_primary_medium ?maturity_bucket_medium)
        (entity_a_ready_flag ?entity_a_account)
        (entity_b_ready_flag ?entity_b_account)
        (investment_lot_available ?investment_lot)
      )
    :effect
      (and
        (investment_lot_allocated ?investment_lot)
        (lot_linked_to_bucket_short ?investment_lot ?maturity_bucket_short)
        (lot_linked_to_bucket_medium ?investment_lot ?maturity_bucket_medium)
        (not
          (investment_lot_available ?investment_lot)
        )
      )
  )
  (:action allocate_investment_lot_with_first_leg
    :parameters (?entity_a_account - entity_a_account ?entity_b_account - entity_b_account ?maturity_bucket_short - maturity_bucket_short ?maturity_bucket_medium - maturity_bucket_medium ?investment_lot - investment_lot)
    :precondition
      (and
        (entity_a_processing_flag ?entity_a_account)
        (entity_b_processing_flag ?entity_b_account)
        (entity_a_to_maturity_bucket_link ?entity_a_account ?maturity_bucket_short)
        (entity_b_to_maturity_bucket_link ?entity_b_account ?maturity_bucket_medium)
        (bucket_signal_secondary_short ?maturity_bucket_short)
        (bucket_signal_primary_medium ?maturity_bucket_medium)
        (not
          (entity_a_ready_flag ?entity_a_account)
        )
        (entity_b_ready_flag ?entity_b_account)
        (investment_lot_available ?investment_lot)
      )
    :effect
      (and
        (investment_lot_allocated ?investment_lot)
        (lot_linked_to_bucket_short ?investment_lot ?maturity_bucket_short)
        (lot_linked_to_bucket_medium ?investment_lot ?maturity_bucket_medium)
        (lot_attribute_first_leg ?investment_lot)
        (not
          (investment_lot_available ?investment_lot)
        )
      )
  )
  (:action allocate_investment_lot_with_second_leg
    :parameters (?entity_a_account - entity_a_account ?entity_b_account - entity_b_account ?maturity_bucket_short - maturity_bucket_short ?maturity_bucket_medium - maturity_bucket_medium ?investment_lot - investment_lot)
    :precondition
      (and
        (entity_a_processing_flag ?entity_a_account)
        (entity_b_processing_flag ?entity_b_account)
        (entity_a_to_maturity_bucket_link ?entity_a_account ?maturity_bucket_short)
        (entity_b_to_maturity_bucket_link ?entity_b_account ?maturity_bucket_medium)
        (bucket_signal_primary_short ?maturity_bucket_short)
        (bucket_signal_secondary_medium ?maturity_bucket_medium)
        (entity_a_ready_flag ?entity_a_account)
        (not
          (entity_b_ready_flag ?entity_b_account)
        )
        (investment_lot_available ?investment_lot)
      )
    :effect
      (and
        (investment_lot_allocated ?investment_lot)
        (lot_linked_to_bucket_short ?investment_lot ?maturity_bucket_short)
        (lot_linked_to_bucket_medium ?investment_lot ?maturity_bucket_medium)
        (lot_attribute_second_leg ?investment_lot)
        (not
          (investment_lot_available ?investment_lot)
        )
      )
  )
  (:action allocate_investment_lot_with_both_legs
    :parameters (?entity_a_account - entity_a_account ?entity_b_account - entity_b_account ?maturity_bucket_short - maturity_bucket_short ?maturity_bucket_medium - maturity_bucket_medium ?investment_lot - investment_lot)
    :precondition
      (and
        (entity_a_processing_flag ?entity_a_account)
        (entity_b_processing_flag ?entity_b_account)
        (entity_a_to_maturity_bucket_link ?entity_a_account ?maturity_bucket_short)
        (entity_b_to_maturity_bucket_link ?entity_b_account ?maturity_bucket_medium)
        (bucket_signal_secondary_short ?maturity_bucket_short)
        (bucket_signal_secondary_medium ?maturity_bucket_medium)
        (not
          (entity_a_ready_flag ?entity_a_account)
        )
        (not
          (entity_b_ready_flag ?entity_b_account)
        )
        (investment_lot_available ?investment_lot)
      )
    :effect
      (and
        (investment_lot_allocated ?investment_lot)
        (lot_linked_to_bucket_short ?investment_lot ?maturity_bucket_short)
        (lot_linked_to_bucket_medium ?investment_lot ?maturity_bucket_medium)
        (lot_attribute_first_leg ?investment_lot)
        (lot_attribute_second_leg ?investment_lot)
        (not
          (investment_lot_available ?investment_lot)
        )
      )
  )
  (:action mark_lot_settlement_ready
    :parameters (?investment_lot - investment_lot ?entity_a_account - entity_a_account ?instrument_offer - instrument_offer)
    :precondition
      (and
        (investment_lot_allocated ?investment_lot)
        (entity_a_processing_flag ?entity_a_account)
        (instrument_reserved_for_account ?entity_a_account ?instrument_offer)
        (not
          (lot_attribute_settlement_ready ?investment_lot)
        )
      )
    :effect (lot_attribute_settlement_ready ?investment_lot)
  )
  (:action bind_security_id_to_lot
    :parameters (?treasury_pool - treasury_pool ?security_id - security_id ?investment_lot - investment_lot)
    :precondition
      (and
        (account_approved ?treasury_pool)
        (pool_to_lot_mapping ?treasury_pool ?investment_lot)
        (pool_to_security_mapping ?treasury_pool ?security_id)
        (security_id_unbound ?security_id)
        (investment_lot_allocated ?investment_lot)
        (lot_attribute_settlement_ready ?investment_lot)
        (not
          (security_id_bound ?security_id)
        )
      )
    :effect
      (and
        (security_id_bound ?security_id)
        (security_bound_to_lot ?security_id ?investment_lot)
        (not
          (security_id_unbound ?security_id)
        )
      )
  )
  (:action set_pool_flag_stage1
    :parameters (?treasury_pool - treasury_pool ?security_id - security_id ?investment_lot - investment_lot ?instrument_offer - instrument_offer)
    :precondition
      (and
        (account_approved ?treasury_pool)
        (pool_to_security_mapping ?treasury_pool ?security_id)
        (security_id_bound ?security_id)
        (security_bound_to_lot ?security_id ?investment_lot)
        (instrument_reserved_for_account ?treasury_pool ?instrument_offer)
        (not
          (lot_attribute_first_leg ?investment_lot)
        )
        (not
          (pool_flag_stage1 ?treasury_pool)
        )
      )
    :effect (pool_flag_stage1 ?treasury_pool)
  )
  (:action attach_approval_token_to_pool
    :parameters (?treasury_pool - treasury_pool ?approval_token - approval_token)
    :precondition
      (and
        (account_approved ?treasury_pool)
        (approval_token_available ?approval_token)
        (not
          (pool_flag_approval_attached ?treasury_pool)
        )
      )
    :effect
      (and
        (pool_flag_approval_attached ?treasury_pool)
        (pool_has_approval_token ?treasury_pool ?approval_token)
        (not
          (approval_token_available ?approval_token)
        )
      )
  )
  (:action advance_pool_after_security_and_approval
    :parameters (?treasury_pool - treasury_pool ?security_id - security_id ?investment_lot - investment_lot ?instrument_offer - instrument_offer ?approval_token - approval_token)
    :precondition
      (and
