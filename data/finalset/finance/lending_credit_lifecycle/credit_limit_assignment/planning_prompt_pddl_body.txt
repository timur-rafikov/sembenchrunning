## Planning domain (PDDL)

(define (domain credit_limit_assignment_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types domain_object - object resource_artifact - domain_object asset_document - domain_object assessment_component - domain_object primary_subject_group - domain_object borrower_entity - primary_subject_group underwriter_resource - resource_artifact scoring_model - resource_artifact reviewer_role - resource_artifact legal_document_template - resource_artifact pricing_scheme - resource_artifact authorization_token - resource_artifact pricing_parameter - resource_artifact portfolio_segment - resource_artifact collateral_package - asset_document security_document - asset_document risk_classification - asset_document borrower_risk_factor - assessment_component facility_pricing_factor - assessment_component limit_proposal - assessment_component borrower_profile_group - borrower_entity facility_profile_group - borrower_entity borrower_exposure_profile - borrower_profile_group facility_exposure_profile - borrower_profile_group credit_limit_case - facility_profile_group)

  (:predicates
    (borrower_intake_created ?borrower - borrower_entity)
    (subject_initialized ?borrower - borrower_entity)
    (underwriter_assignment_flag ?borrower - borrower_entity)
    (limit_assigned_for_subject ?borrower - borrower_entity)

    (assigned_scoring_model_for_subject ?borrower - borrower_entity ?scoring_model - scoring_model)
    (reviewer_available ?reviewer - reviewer_role)
    (assigned_reviewer_for_subject ?borrower - borrower_entity ?reviewer - reviewer_role)
    (collateral_package_available ?collateral_package - collateral_package)
    (borrower_exposure_has_collateral ?borrower_exposure_profile - borrower_exposure_profile ?collateral_package - collateral_package)
    (facility_exposure_has_collateral ?facility_exposure_profile - facility_exposure_profile ?collateral_package - collateral_package)
    (exposure_profile_has_risk_factor ?borrower_exposure_profile - borrower_exposure_profile ?borrower_risk_factor - borrower_risk_factor)
    (risk_factor_tagged ?borrower_risk_factor - borrower_risk_factor)
    (risk_factor_collateral_tag ?borrower_risk_factor - borrower_risk_factor)
    (borrower_exposure_verified ?borrower_exposure_profile - borrower_exposure_profile)
    (facility_exposure_has_pricing_factor ?facility_exposure_profile - facility_exposure_profile ?facility_pricing_factor - facility_pricing_factor)
    (facility_pricing_factor_tagged ?facility_pricing_factor - facility_pricing_factor)
    (facility_pricing_factor_collateral_tag ?facility_pricing_factor - facility_pricing_factor)
    (facility_exposure_verified ?facility_exposure_profile - facility_exposure_profile)
    (proposal_draft ?limit_proposal - limit_proposal)
    (proposal_staged ?limit_proposal - limit_proposal)
    (proposal_has_borrower_risk_factor ?limit_proposal - limit_proposal ?borrower_risk_factor - borrower_risk_factor)
    (proposal_has_facility_pricing_factor ?limit_proposal - limit_proposal ?facility_pricing_factor - facility_pricing_factor)
    (proposal_flag_borrower_risk ?limit_proposal - limit_proposal)
    (proposal_flag_facility_pricing ?limit_proposal - limit_proposal)
    (proposal_locked ?limit_proposal - limit_proposal)
    (case_includes_borrower_profile ?credit_limit_case - credit_limit_case ?borrower_exposure_profile - borrower_exposure_profile)
    (case_includes_facility_profile ?credit_limit_case - credit_limit_case ?facility_exposure_profile - facility_exposure_profile)
    (case_associated_proposal ?credit_limit_case - credit_limit_case ?limit_proposal - limit_proposal)
    (security_document_available ?security_document - security_document)
    (case_has_security_document ?credit_limit_case - credit_limit_case ?security_document - security_document)
    (security_document_registered ?security_document - security_document)
    (security_document_linked_to_proposal ?security_document - security_document ?limit_proposal - limit_proposal)
    (case_collateral_verified ?credit_limit_case - credit_limit_case)
    (case_pricing_verified ?credit_limit_case - credit_limit_case)
    (case_validation_complete ?credit_limit_case - credit_limit_case)
    (legal_template_attached ?credit_limit_case - credit_limit_case)
    (case_legal_attachment_completed ?credit_limit_case - credit_limit_case)
    (case_pricing_scheme_ready ?credit_limit_case - credit_limit_case)
    (case_approval_aggregated ?credit_limit_case - credit_limit_case)
    (risk_classification_available ?risk_classification - risk_classification)
    (case_has_risk_classification ?credit_limit_case - credit_limit_case ?risk_classification - risk_classification)
    (case_risk_classification_locked ?credit_limit_case - credit_limit_case)
    (case_legal_review_completed ?credit_limit_case - credit_limit_case)
    (case_portfolio_segment_attached ?credit_limit_case - credit_limit_case)
    (legal_template_available ?legal_document_template - legal_document_template)
    (case_attached_legal_template ?credit_limit_case - credit_limit_case ?legal_document_template - legal_document_template)
    (pricing_scheme_available ?pricing_scheme - pricing_scheme)
    (case_pricing_scheme_linked ?credit_limit_case - credit_limit_case ?pricing_scheme - pricing_scheme)
    (pricing_parameter_available ?pricing_parameter - pricing_parameter)
    (case_pricing_parameter_attached ?credit_limit_case - credit_limit_case ?pricing_parameter - pricing_parameter)
    (portfolio_segment_available ?portfolio_segment - portfolio_segment)
    (case_portfolio_segment_linked ?credit_limit_case - credit_limit_case ?portfolio_segment - portfolio_segment)

    (facility_exposure_ready ?facility_exposure_profile - facility_exposure_profile)
    (decision_marked ?credit_limit_case - credit_limit_case)

  (:action register_borrower_intake
    :parameters (?borrower - borrower_entity)
    :precondition
      (and
        (not
          (borrower_intake_created ?borrower)
        )
        (not
          (limit_assigned_for_subject ?borrower)
        )
      )
    :effect (borrower_intake_created ?borrower)
  )
  (:action assign_underwriter_to_borrower
    :parameters (?borrower - borrower_entity ?underwriter - underwriter_resource)
    :precondition
      (and
        (borrower_intake_created ?borrower)
        (not
          (underwriter_assignment_flag ?borrower)
        )
        (underwriter_available ?underwriter)
      )
    :effect
      (and
        (underwriter_assignment_flag ?borrower)
        (assigned_underwriter_for_subject ?borrower ?underwriter)
        (not
          (underwriter_available ?underwriter)
        )
      )
  )
  (:action apply_scoring_model_to_borrower
    :parameters (?borrower - borrower_entity ?scoring_model - scoring_model)
    :precondition
      (and
        (borrower_intake_created ?borrower)
        (underwriter_assignment_flag ?borrower)
        (scoring_model_available ?scoring_model)
      )
    :effect
      (and
        (assigned_scoring_model_for_subject ?borrower ?scoring_model)
        (not
          (scoring_model_available ?scoring_model)
        )
      )
  )
  (:action finalize_scoring_for_borrower
    :parameters (?borrower - borrower_entity ?scoring_model - scoring_model)
    :precondition
      (and
        (borrower_intake_created ?borrower)
        (underwriter_assignment_flag ?borrower)
        (assigned_scoring_model_for_subject ?borrower ?scoring_model)
        (not
          (subject_initialized ?borrower)
        )
      )
    :effect (subject_initialized ?borrower)
  )
  (:action release_scoring_model_from_borrower
    :parameters (?borrower - borrower_entity ?scoring_model - scoring_model)
    :precondition
      (and
        (assigned_scoring_model_for_subject ?borrower ?scoring_model)
      )
    :effect
      (and
        (scoring_model_available ?scoring_model)
        (not
          (assigned_scoring_model_for_subject ?borrower ?scoring_model)
        )
      )
  )
  (:action assign_reviewer_to_borrower
    :parameters (?borrower - borrower_entity ?reviewer - reviewer_role)
    :precondition
      (and
        (subject_initialized ?borrower)
        (reviewer_available ?reviewer)
      )
    :effect
      (and
        (assigned_reviewer_for_subject ?borrower ?reviewer)
        (not
          (reviewer_available ?reviewer)
        )
      )
  )
  (:action release_reviewer_from_borrower
    :parameters (?borrower - borrower_entity ?reviewer - reviewer_role)
    :precondition
      (and
        (assigned_reviewer_for_subject ?borrower ?reviewer)
      )
    :effect
      (and
        (reviewer_available ?reviewer)
        (not
          (assigned_reviewer_for_subject ?borrower ?reviewer)
        )
      )
  )
  (:action attach_pricing_parameter_to_case
    :parameters (?limit_case - credit_limit_case ?pricing_parameter - pricing_parameter)

        (subject_initialized ?limit_case)
        (pricing_parameter_available ?pricing_parameter)

        (case_pricing_parameter_attached ?limit_case ?pricing_parameter)

        (case_portfolio_segment_linked ?limit_case ?portfolio_segment)

    :effect
      (and
        (portfolio_segment_available ?portfolio_segment)
        (not
          (case_portfolio_segment_linked ?limit_case ?portfolio_segment)
        )
      )

  (:action flag_risk_factor_for_borrower_exposure
    :parameters (?borrower_exposure_profile - borrower_exposure_profile ?borrower_risk_factor - borrower_risk_factor ?scoring_model - scoring_model)
    :precondition
      (and
        (subject_initialized ?borrower_exposure_profile)

        (exposure_profile_has_risk_factor ?borrower_exposure_profile ?borrower_risk_factor)
        (not
          (risk_factor_tagged ?borrower_risk_factor)
        )
        (not
          (risk_factor_collateral_tag ?borrower_risk_factor)
        )
      )
    :effect (risk_factor_tagged ?borrower_risk_factor)
  )
  (:action verify_borrower_exposure
    :parameters (?borrower_exposure_profile - borrower_exposure_profile ?borrower_risk_factor - borrower_risk_factor ?reviewer - reviewer_role)
    :precondition
      (and
        (subject_initialized ?borrower_exposure_profile)
        (assigned_reviewer_for_subject ?borrower_exposure_profile ?reviewer)
        (exposure_profile_has_risk_factor ?borrower_exposure_profile ?borrower_risk_factor)
        (risk_factor_tagged ?borrower_risk_factor)
        (not
          (borrower_exposure_ready ?borrower_exposure_profile)
        )
      )
    :effect
      (and
        (borrower_exposure_ready ?borrower_exposure_profile)
        (borrower_exposure_verified ?borrower_exposure_profile)
      )
  )
