  (:requirements :strips :typing :negative-preconditions)

  (:predicates
    (case_registered ?returned_case - returned_payment_case)
    (entity_validated_for_processing ?returned_case - returned_payment_case)
    (channel_option_bound ?returned_case - returned_payment_case)
    (entity_released_for_remediation ?returned_case - returned_payment_case)
    (entity_ready_for_external_interaction ?returned_case - returned_payment_case)
    (entity_release_approved ?returned_case - returned_payment_case)
    (settlement_channel_option_available ?settlement_channel_option - settlement_channel_option)

    (processing_party_available ?processing_party - processing_party)
    (entity_assigned_processing_party ?returned_case - returned_payment_case ?processing_party - processing_party)
    (remediation_option_available ?remediation_option - remediation_option)
    (payer_account_remediation_assigned ?payer_account - payer_account ?remediation_option - remediation_option)
    (payee_account_remediation_assigned ?payee_account - payee_account ?remediation_option - remediation_option)
    (bound_payer_channel_endpoint ?payer_account - payer_account ?payer_channel_endpoint - payer_channel_endpoint)
    (payer_channel_endpoint_ready ?payer_channel_endpoint - payer_channel_endpoint)
    (payer_channel_endpoint_flagged_for_remediation ?payer_channel_endpoint - payer_channel_endpoint)

    (payee_channel_endpoint_ready ?payee_channel_endpoint - payee_channel_endpoint)
    (payee_channel_endpoint_flagged_for_remediation ?payee_channel_endpoint - payee_channel_endpoint)
    (payee_account_prepared_for_routing ?payee_account - payee_account)
    (settlement_message_available ?settlement_message - settlement_message)
    (settlement_message_selected ?settlement_message - settlement_message)
    (settlement_message_bound_to_payer_endpoint ?settlement_message - settlement_message ?payer_channel_endpoint - payer_channel_endpoint)
    (settlement_message_bound_to_payee_endpoint ?settlement_message - settlement_message ?payee_channel_endpoint - payee_channel_endpoint)
    (settlement_message_requires_payer_follow_up ?settlement_message - settlement_message)
    (settlement_message_requires_payee_follow_up ?settlement_message - settlement_message)
    (settlement_message_ready_for_transmission ?settlement_message - settlement_message)
    (work_item_associated_with_payer_account ?work_item - processing_work_item ?payer_account - payer_account)
    (work_item_associated_with_payee_account ?work_item - processing_work_item ?payee_account - payee_account)
    (work_item_assigned_settlement_message ?work_item - processing_work_item ?settlement_message - settlement_message)
    (supporting_document_available ?supporting_document - supporting_document)
    (work_item_attached_supporting_document ?work_item - processing_work_item ?supporting_document - supporting_document)

    (supporting_document_attached_to_message ?supporting_document - supporting_document ?settlement_message - settlement_message)
    (work_item_documents_ready ?work_item - processing_work_item)
    (work_item_resources_allocated ?work_item - processing_work_item)

    (routing_policy_locked_on_work_item ?work_item - processing_work_item)

    (processing_party_acknowledged_stakeholder_binding ?work_item - processing_work_item)

    (routing_policy_available ?routing_policy - routing_policy)
    (work_item_assigned_routing_policy ?work_item - processing_work_item ?routing_policy - routing_policy)
    (authorization_token_available ?authorization_token - authorization_token)
    (work_item_attached_authorization_token ?work_item - processing_work_item ?authorization_token - authorization_token)

    (work_item_finalized ?work_item - processing_work_item)

  (:action register_returned_payment_case
    :parameters (?returned_case - returned_payment_case)
    :precondition
      (and
        (not
          (case_registered ?returned_case)
        )
        (not
          (entity_released_for_remediation ?returned_case)
        )
      )
    :effect (case_registered ?returned_case)
  )
  (:action bind_settlement_channel_option_to_case
    :parameters (?returned_case - returned_payment_case ?settlement_channel_option - settlement_channel_option)
    :precondition
      (and
        (case_registered ?returned_case)
        (not
          (channel_option_bound ?returned_case)
        )
        (settlement_channel_option_available ?settlement_channel_option)
      )
    :effect
      (and
        (channel_option_bound ?returned_case)
        (bound_settlement_channel_option ?returned_case ?settlement_channel_option)
        (not
          (settlement_channel_option_available ?settlement_channel_option)
        )
      )
  )
  (:action apply_validation_rule_to_case
    :parameters (?returned_case - returned_payment_case ?validation_rule - validation_rule)
    :precondition
      (and
        (case_registered ?returned_case)
        (channel_option_bound ?returned_case)
        (validation_rule_available ?validation_rule)
      )
    :effect
      (and
        (bound_validation_rule_for_entity ?returned_case ?validation_rule)
        (not
          (validation_rule_available ?validation_rule)
        )
      )
  )
  (:action mark_case_validated
    :parameters (?returned_case - returned_payment_case ?validation_rule - validation_rule)
    :precondition
      (and
        (case_registered ?returned_case)
        (channel_option_bound ?returned_case)
        (bound_validation_rule_for_entity ?returned_case ?validation_rule)
        (not
          (entity_validated_for_processing ?returned_case)
        )
      )
    :effect (entity_validated_for_processing ?returned_case)
  )
  (:action release_validation_rule_from_case
    :parameters (?returned_case - returned_payment_case ?validation_rule - validation_rule)
    :precondition
      (and
        (bound_validation_rule_for_entity ?returned_case ?validation_rule)
      )
    :effect
      (and
        (validation_rule_available ?validation_rule)
        (not
          (bound_validation_rule_for_entity ?returned_case ?validation_rule)
        )
      )
  )
  (:action assign_processing_party_to_case
    :parameters (?returned_case - returned_payment_case ?processing_party - processing_party)
    :precondition
      (and
        (entity_validated_for_processing ?returned_case)
        (processing_party_available ?processing_party)
      )
    :effect
      (and
        (entity_assigned_processing_party ?returned_case ?processing_party)
        (not
          (processing_party_available ?processing_party)
        )
      )
  )
  (:action release_processing_party_from_case
    :parameters (?returned_case - returned_payment_case ?processing_party - processing_party)
    :precondition
      (and
        (entity_assigned_processing_party ?returned_case ?processing_party)
      )
    :effect
      (and
        (processing_party_available ?processing_party)
        (not
          (entity_assigned_processing_party ?returned_case ?processing_party)
        )
      )
  )
  (:action allocate_liquidity_reserve_to_work_item
    :parameters (?work_item - processing_work_item ?liquidity_reserve - liquidity_reserve)
    :precondition
      (and
        (entity_validated_for_processing ?work_item)
        (liquidity_reserve_available ?liquidity_reserve)
      )
    :effect
      (and
        (work_item_allocated_liquidity_reserve ?work_item ?liquidity_reserve)
        (not
          (liquidity_reserve_available ?liquidity_reserve)
        )
      )
  )
  (:action release_liquidity_reserve_from_work_item
    :parameters (?work_item - processing_work_item ?liquidity_reserve - liquidity_reserve)
    :precondition
      (and
        (work_item_allocated_liquidity_reserve ?work_item ?liquidity_reserve)
      )
    :effect
      (and
        (liquidity_reserve_available ?liquidity_reserve)
        (not
          (work_item_allocated_liquidity_reserve ?work_item ?liquidity_reserve)
        )
      )
  )
  (:action attach_regulatory_constraint_to_work_item
    :parameters (?work_item - processing_work_item ?regulatory_rule - regulatory_constraint)
    :precondition
      (and
        (entity_validated_for_processing ?work_item)
        (regulatory_rule_available ?regulatory_rule)
      )
    :effect
      (and
        (work_item_bound_to_regulatory_rule ?work_item ?regulatory_rule)
        (not
          (regulatory_rule_available ?regulatory_rule)
        )
      )
  )
  (:action detach_regulatory_constraint_from_work_item
    :parameters (?work_item - processing_work_item ?regulatory_rule - regulatory_constraint)
    :precondition
      (and
        (work_item_bound_to_regulatory_rule ?work_item ?regulatory_rule)
      )
    :effect
      (and
        (regulatory_rule_available ?regulatory_rule)
        (not
          (work_item_bound_to_regulatory_rule ?work_item ?regulatory_rule)
        )
      )
  )
  (:action verify_payer_channel_endpoint
    :parameters (?payer_account - payer_account ?payer_channel_endpoint - payer_channel_endpoint ?validation_rule - validation_rule)
    :precondition
      (and
        (entity_validated_for_processing ?payer_account)
        (bound_validation_rule_for_entity ?payer_account ?validation_rule)
        (bound_payer_channel_endpoint ?payer_account ?payer_channel_endpoint)
        (not
          (payer_channel_endpoint_ready ?payer_channel_endpoint)
        )
        (not
          (payer_channel_endpoint_flagged_for_remediation ?payer_channel_endpoint)
        )
      )
    :effect (payer_channel_endpoint_ready ?payer_channel_endpoint)
  )
  (:action confirm_payer_account_readiness
    :parameters (?payer_account - payer_account ?payer_channel_endpoint - payer_channel_endpoint ?processing_party - processing_party)
    :precondition
      (and
        (entity_validated_for_processing ?payer_account)
        (entity_assigned_processing_party ?payer_account ?processing_party)
        (bound_payer_channel_endpoint ?payer_account ?payer_channel_endpoint)
        (payer_channel_endpoint_ready ?payer_channel_endpoint)
        (not
          (payer_account_signaled_for_message_assembly ?payer_account)
        )
      )
    :effect
      (and
        (payer_account_signaled_for_message_assembly ?payer_account)
        (payer_account_prepared_for_routing ?payer_account)
      )
  )
  (:action reserve_remediation_option_for_payer_account
    :parameters (?payer_account - payer_account ?payer_channel_endpoint - payer_channel_endpoint ?remediation_option - remediation_option)
    :precondition
      (and
        (entity_validated_for_processing ?payer_account)
        (bound_payer_channel_endpoint ?payer_account ?payer_channel_endpoint)
        (remediation_option_available ?remediation_option)
        (not
          (payer_account_signaled_for_message_assembly ?payer_account)
        )
      )
    :effect
      (and
        (payer_channel_endpoint_flagged_for_remediation ?payer_channel_endpoint)
        (payer_account_signaled_for_message_assembly ?payer_account)
        (payer_account_remediation_assigned ?payer_account ?remediation_option)
        (not
          (remediation_option_available ?remediation_option)
        )
      )
  )
  (:action execute_payer_account_remediation
    :parameters (?payer_account - payer_account ?payer_channel_endpoint - payer_channel_endpoint ?validation_rule - validation_rule ?remediation_option - remediation_option)
    :precondition
      (and
        (entity_validated_for_processing ?payer_account)
        (bound_validation_rule_for_entity ?payer_account ?validation_rule)
        (bound_payer_channel_endpoint ?payer_account ?payer_channel_endpoint)
        (payer_channel_endpoint_flagged_for_remediation ?payer_channel_endpoint)
        (payer_account_remediation_assigned ?payer_account ?remediation_option)
        (not
          (payer_account_prepared_for_routing ?payer_account)
        )
      )
    :effect
      (and
        (payer_channel_endpoint_ready ?payer_channel_endpoint)
        (payer_account_prepared_for_routing ?payer_account)
        (remediation_option_available ?remediation_option)
        (not
          (payer_account_remediation_assigned ?payer_account ?remediation_option)
        )
      )
  )
  (:action verify_payee_channel_endpoint
    :parameters (?payee_account - payee_account ?payee_channel_endpoint - payee_channel_endpoint ?validation_rule - validation_rule)
    :precondition
      (and
        (entity_validated_for_processing ?payee_account)
        (bound_validation_rule_for_entity ?payee_account ?validation_rule)
        (bound_payee_channel_endpoint ?payee_account ?payee_channel_endpoint)
        (not
          (payee_channel_endpoint_ready ?payee_channel_endpoint)
        )
        (not
          (payee_channel_endpoint_flagged_for_remediation ?payee_channel_endpoint)
        )
      )
    :effect (payee_channel_endpoint_ready ?payee_channel_endpoint)
  )
  (:action confirm_payee_account_readiness
    :parameters (?payee_account - payee_account ?payee_channel_endpoint - payee_channel_endpoint ?processing_party - processing_party)
    :precondition
      (and
        (entity_validated_for_processing ?payee_account)
        (entity_assigned_processing_party ?payee_account ?processing_party)
        (bound_payee_channel_endpoint ?payee_account ?payee_channel_endpoint)
        (payee_channel_endpoint_ready ?payee_channel_endpoint)
        (not
          (payee_account_signaled_for_message_assembly ?payee_account)
        )
      )
    :effect
      (and
        (payee_account_signaled_for_message_assembly ?payee_account)
        (payee_account_prepared_for_routing ?payee_account)
      )
  )
  (:action reserve_remediation_option_for_payee_account
    :parameters (?payee_account - payee_account ?payee_channel_endpoint - payee_channel_endpoint ?remediation_option - remediation_option)
    :precondition
      (and
        (entity_validated_for_processing ?payee_account)
        (bound_payee_channel_endpoint ?payee_account ?payee_channel_endpoint)
        (remediation_option_available ?remediation_option)
        (not
          (payee_account_signaled_for_message_assembly ?payee_account)
        )
      )
    :effect
      (and
        (payee_channel_endpoint_flagged_for_remediation ?payee_channel_endpoint)
        (payee_account_signaled_for_message_assembly ?payee_account)
        (payee_account_remediation_assigned ?payee_account ?remediation_option)
        (not
          (remediation_option_available ?remediation_option)
        )
      )
  )
  (:action execute_payee_account_remediation
    :parameters (?payee_account - payee_account ?payee_channel_endpoint - payee_channel_endpoint ?validation_rule - validation_rule ?remediation_option - remediation_option)
    :precondition
      (and
        (entity_validated_for_processing ?payee_account)
        (bound_validation_rule_for_entity ?payee_account ?validation_rule)
        (bound_payee_channel_endpoint ?payee_account ?payee_channel_endpoint)
        (payee_channel_endpoint_flagged_for_remediation ?payee_channel_endpoint)
        (payee_account_remediation_assigned ?payee_account ?remediation_option)
        (not
          (payee_account_prepared_for_routing ?payee_account)
        )
      )
    :effect
      (and
        (payee_channel_endpoint_ready ?payee_channel_endpoint)
        (payee_account_prepared_for_routing ?payee_account)
        (remediation_option_available ?remediation_option)
        (not
          (payee_account_remediation_assigned ?payee_account ?remediation_option)
        )
      )
  )
  (:action select_and_bind_settlement_message
    :parameters (?payer_account - payer_account ?payee_account - payee_account ?payer_channel_endpoint - payer_channel_endpoint ?payee_channel_endpoint - payee_channel_endpoint ?settlement_message - settlement_message)
    :precondition
      (and
        (payer_account_signaled_for_message_assembly ?payer_account)
        (payee_account_signaled_for_message_assembly ?payee_account)
        (bound_payer_channel_endpoint ?payer_account ?payer_channel_endpoint)
        (bound_payee_channel_endpoint ?payee_account ?payee_channel_endpoint)
        (payer_channel_endpoint_ready ?payer_channel_endpoint)
        (payee_channel_endpoint_ready ?payee_channel_endpoint)
        (payer_account_prepared_for_routing ?payer_account)
        (payee_account_prepared_for_routing ?payee_account)
        (settlement_message_available ?settlement_message)
      )
    :effect
      (and
        (settlement_message_selected ?settlement_message)
        (settlement_message_bound_to_payer_endpoint ?settlement_message ?payer_channel_endpoint)
        (settlement_message_bound_to_payee_endpoint ?settlement_message ?payee_channel_endpoint)
        (not
          (settlement_message_available ?settlement_message)
        )
      )
  )
  (:action select_and_bind_settlement_message_requires_payer_follow_up
    :parameters (?payer_account - payer_account ?payee_account - payee_account ?payer_channel_endpoint - payer_channel_endpoint ?payee_channel_endpoint - payee_channel_endpoint ?settlement_message - settlement_message)
    :precondition
      (and
        (payer_account_signaled_for_message_assembly ?payer_account)
        (payee_account_signaled_for_message_assembly ?payee_account)
        (bound_payer_channel_endpoint ?payer_account ?payer_channel_endpoint)
        (bound_payee_channel_endpoint ?payee_account ?payee_channel_endpoint)
        (payer_channel_endpoint_flagged_for_remediation ?payer_channel_endpoint)
        (payee_channel_endpoint_ready ?payee_channel_endpoint)
        (not
          (payer_account_prepared_for_routing ?payer_account)
        )
        (payee_account_prepared_for_routing ?payee_account)
        (settlement_message_available ?settlement_message)
      )
    :effect
      (and
        (settlement_message_selected ?settlement_message)
        (settlement_message_bound_to_payer_endpoint ?settlement_message ?payer_channel_endpoint)
        (settlement_message_bound_to_payee_endpoint ?settlement_message ?payee_channel_endpoint)
        (settlement_message_requires_payer_follow_up ?settlement_message)
        (not
          (settlement_message_available ?settlement_message)
        )
      )
  )
  (:action select_and_bind_settlement_message_requires_payee_follow_up

      (and

        (settlement_message_requires_payee_follow_up ?settlement_message)
        (not
          (settlement_message_available ?settlement_message)
        )
      )
  )
  (:action select_and_bind_settlement_message_requires_both_follow_up

      (and

        (payer_channel_endpoint_flagged_for_remediation ?payer_channel_endpoint)
        (payee_channel_endpoint_flagged_for_remediation ?payee_channel_endpoint)
        (not
          (payer_account_prepared_for_routing ?payer_account)
        )
        (not
          (payee_account_prepared_for_routing ?payee_account)

        (settlement_message_requires_payee_follow_up ?settlement_message)
        (not
          (settlement_message_available ?settlement_message)
        )
      )
  )
  (:action finalize_settlement_message_for_transmission
    :parameters (?settlement_message - settlement_message ?payer_account - payer_account ?validation_rule - validation_rule)
    :precondition
      (and
        (settlement_message_selected ?settlement_message)
        (payer_account_signaled_for_message_assembly ?payer_account)
        (bound_validation_rule_for_entity ?payer_account ?validation_rule)
        (not
          (settlement_message_ready_for_transmission ?settlement_message)

    :effect (settlement_message_ready_for_transmission ?settlement_message)
  )
  (:action lock_and_attach_supporting_document_to_message
    :parameters (?work_item - processing_work_item ?supporting_document - supporting_document ?settlement_message - settlement_message)
    :precondition
      (and
        (entity_validated_for_processing ?work_item)
        (work_item_assigned_settlement_message ?work_item ?settlement_message)
        (work_item_attached_supporting_document ?work_item ?supporting_document)
        (supporting_document_available ?supporting_document)
        (settlement_message_selected ?settlement_message)
        (settlement_message_ready_for_transmission ?settlement_message)
        (not
          (supporting_document_locked ?supporting_document)
        )
      )
    :effect
