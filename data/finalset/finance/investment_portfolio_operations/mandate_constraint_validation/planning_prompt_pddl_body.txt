  (:action register_mandate_definition
    :parameters (?mandate_definition - mandate_definition)
    :precondition
      (and
        (not
          (registered ?mandate_definition)
        )
        (not
          (entity_finalized ?mandate_definition)
        )
      )
    :effect (registered ?mandate_definition)
  )
  (:action attach_actor_token
    :parameters (?mandate_definition - mandate_definition ?approval_actor_token - approval_actor_token)
    :precondition
      (and
        (registered ?mandate_definition)
        (not
          (mandate_actor_assigned ?mandate_definition)
        )
        (token_available ?approval_actor_token)
      )
    :effect
      (and
        (mandate_actor_assigned ?mandate_definition)
        (entity_assigned_actor_token ?mandate_definition ?approval_actor_token)
        (not
          (token_available ?approval_actor_token)
        )
      )
  )
  (:action claim_instrument_candidate
    :parameters (?mandate_definition - mandate_definition ?instrument_candidate - instrument_candidate)
    :precondition
      (and
        (registered ?mandate_definition)
        (mandate_actor_assigned ?mandate_definition)
        (instrument_available ?instrument_candidate)
      )
    :effect
      (and
        (entity_instrument_assigned ?mandate_definition ?instrument_candidate)
        (not
          (instrument_available ?instrument_candidate)
        )
      )
  )
  (:action authorize_governance_artifact
    :parameters (?mandate_definition - mandate_definition ?instrument_candidate - instrument_candidate)
    :precondition
      (and
        (registered ?mandate_definition)
        (mandate_actor_assigned ?mandate_definition)
        (entity_instrument_assigned ?mandate_definition ?instrument_candidate)
        (not
          (entity_authorized ?mandate_definition)
        )
      )
    :effect (entity_authorized ?mandate_definition)
  )
  (:action release_instrument_candidate
    :parameters (?mandate_definition - mandate_definition ?instrument_candidate - instrument_candidate)
    :precondition
      (and
        (entity_instrument_assigned ?mandate_definition ?instrument_candidate)
      )
    :effect
      (and
        (instrument_available ?instrument_candidate)
        (not
          (entity_instrument_assigned ?mandate_definition ?instrument_candidate)
        )
      )
  )
  (:action attach_reviewer_token
    :parameters (?mandate_definition - mandate_definition ?reviewer_token - reviewer_token)

      (and
        (entity_authorized ?mandate_definition)

          (primary_bucket_signaled ?primary_exposure_bucket)

          (primary_bucket_param_attached ?primary_exposure_bucket)
        )

  )
  (:action confirm_primary_segment_eligibility
    :parameters (?primary_segment - primary_segment ?primary_exposure_bucket - primary_exposure_bucket ?reviewer_token - reviewer_token)

        (entity_authorized ?primary_segment)
        (entity_assigned_reviewer_token ?primary_segment ?reviewer_token)
        (has_primary_exposure_bucket ?primary_segment ?primary_exposure_bucket)
        (primary_bucket_signaled ?primary_exposure_bucket)
        (not
          (primary_segment_flagged ?primary_segment)
        )
      )
    :effect
      (and
        (primary_segment_flagged ?primary_segment)
        (primary_segment_ready ?primary_segment)
      )
  )
  (:action bind_constraint_parameter_to_primary_segment
    :parameters (?primary_segment - primary_segment ?primary_exposure_bucket - primary_exposure_bucket ?constraint_parameter - constraint_parameter)

          (primary_segment_flagged ?primary_segment)

    :effect
      (and
        (primary_bucket_param_attached ?primary_exposure_bucket)
        (primary_segment_flagged ?primary_segment)

          (constraint_param_available ?constraint_parameter)
        )
      )
  )

          (primary_segment_ready ?primary_segment)

        (primary_bucket_signaled ?primary_exposure_bucket)

          (secondary_segment_flagged ?secondary_segment)

        (secondary_bucket_param_attached ?secondary_exposure_bucket)

          (constraint_param_available ?constraint_parameter)

  (:action finalize_secondary_bucket_parameter_binding

  (:action assemble_trade_basket_both_ready
    :parameters (?primary_segment - primary_segment ?secondary_segment - secondary_segment ?primary_exposure_bucket - primary_exposure_bucket ?secondary_exposure_bucket - secondary_exposure_bucket ?trade_basket - trade_basket)
    :precondition
      (and
        (primary_segment_flagged ?primary_segment)
        (secondary_segment_flagged ?secondary_segment)
        (has_primary_exposure_bucket ?primary_segment ?primary_exposure_bucket)
        (has_secondary_exposure_bucket ?secondary_segment ?secondary_exposure_bucket)
        (primary_bucket_signaled ?primary_exposure_bucket)
        (secondary_bucket_signaled ?secondary_exposure_bucket)
        (primary_segment_ready ?primary_segment)
        (secondary_segment_ready ?secondary_segment)
        (trade_basket_available ?trade_basket)
      )
    :effect
      (and
        (trade_basket_prepared ?trade_basket)
        (basket_primary_bucket_link ?trade_basket ?primary_exposure_bucket)
        (basket_secondary_bucket_link ?trade_basket ?secondary_exposure_bucket)
        (not
          (trade_basket_available ?trade_basket)
        )
      )
  )
  (:action assemble_trade_basket_primary_bound
    :parameters (?primary_segment - primary_segment ?secondary_segment - secondary_segment ?primary_exposure_bucket - primary_exposure_bucket ?secondary_exposure_bucket - secondary_exposure_bucket ?trade_basket - trade_basket)
    :precondition
      (and
        (primary_segment_flagged ?primary_segment)
        (secondary_segment_flagged ?secondary_segment)
        (has_primary_exposure_bucket ?primary_segment ?primary_exposure_bucket)
        (has_secondary_exposure_bucket ?secondary_segment ?secondary_exposure_bucket)
        (primary_bucket_param_attached ?primary_exposure_bucket)
        (secondary_bucket_signaled ?secondary_exposure_bucket)
        (not
          (primary_segment_ready ?primary_segment)
        )
        (secondary_segment_ready ?secondary_segment)
        (trade_basket_available ?trade_basket)
      )
    :effect
      (and
        (trade_basket_prepared ?trade_basket)
        (basket_primary_bucket_link ?trade_basket ?primary_exposure_bucket)
        (basket_secondary_bucket_link ?trade_basket ?secondary_exposure_bucket)
        (basket_primary_path ?trade_basket)
        (not
          (trade_basket_available ?trade_basket)
        )
      )
  )
  (:action assemble_trade_basket_secondary_bound
    :parameters (?primary_segment - primary_segment ?secondary_segment - secondary_segment ?primary_exposure_bucket - primary_exposure_bucket ?secondary_exposure_bucket - secondary_exposure_bucket ?trade_basket - trade_basket)
    :precondition
      (and
        (primary_segment_flagged ?primary_segment)
        (secondary_segment_flagged ?secondary_segment)
        (has_primary_exposure_bucket ?primary_segment ?primary_exposure_bucket)
        (has_secondary_exposure_bucket ?secondary_segment ?secondary_exposure_bucket)
        (primary_bucket_signaled ?primary_exposure_bucket)
        (secondary_bucket_param_attached ?secondary_exposure_bucket)
        (primary_segment_ready ?primary_segment)
        (not
          (secondary_segment_ready ?secondary_segment)
        )
        (trade_basket_available ?trade_basket)
      )
    :effect
      (and
        (trade_basket_prepared ?trade_basket)
        (basket_primary_bucket_link ?trade_basket ?primary_exposure_bucket)
        (basket_secondary_bucket_link ?trade_basket ?secondary_exposure_bucket)
        (basket_secondary_path ?trade_basket)
        (not
          (trade_basket_available ?trade_basket)
        )
      )
  )
  (:action assemble_trade_basket_both_bound
    :parameters (?primary_segment - primary_segment ?secondary_segment - secondary_segment ?primary_exposure_bucket - primary_exposure_bucket ?secondary_exposure_bucket - secondary_exposure_bucket ?trade_basket - trade_basket)
    :precondition
      (and
        (primary_segment_flagged ?primary_segment)
        (secondary_segment_flagged ?secondary_segment)
        (has_primary_exposure_bucket ?primary_segment ?primary_exposure_bucket)
        (has_secondary_exposure_bucket ?secondary_segment ?secondary_exposure_bucket)

        (secondary_bucket_param_attached ?secondary_exposure_bucket)
        (not
          (primary_segment_ready ?primary_segment)

        (not
          (secondary_segment_ready ?secondary_segment)

        (trade_basket_available ?trade_basket)

    :effect
      (and
        (trade_basket_prepared ?trade_basket)
        (basket_primary_bucket_link ?trade_basket ?primary_exposure_bucket)
        (basket_secondary_bucket_link ?trade_basket ?secondary_exposure_bucket)
        (basket_primary_path ?trade_basket)
        (basket_secondary_path ?trade_basket)
        (not
          (trade_basket_available ?trade_basket)

      )
  )

        (security_lot_reserved ?security_lot)

  )

          (strategy_tag_available ?strategy_tag)

        (strategy_tag_claimed ?portfolio_composite)
        (composite_strategy_tag_link ?portfolio_composite ?strategy_tag)

          (composite_lot_selected ?portfolio_composite)

        (composite_lot_selected ?portfolio_composite)

          (basket_secondary_path ?trade_basket)
