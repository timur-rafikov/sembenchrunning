## Planning domain (PDDL)

(define (domain financial_reporting_and_controls_control_exception_escalation_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types organizational_entity - object document - object control_element - object case_root - object exception_case - case_root escalation_recipient - organizational_entity analyst - organizational_entity manager - organizational_entity compliance_reference_document - organizational_entity final_signoff_document - organizational_entity audit_reference_document - organizational_entity remediation_task - organizational_entity audit_issue_record - organizational_entity evidence_document - document supporting_document - document regulatory_finding - document control_point - control_element control_procedure - control_element escalation_package - control_element scoped_exception_case - exception_case investigation_case - exception_case account_context_case - scoped_exception_case business_unit_context_case - scoped_exception_case investigation_record - investigation_case)
  (:predicates
    (case_logged ?exception_case - exception_case)
    (case_acknowledged ?exception_case - exception_case)
    (case_triaged ?exception_case - exception_case)
    (case_closed ?exception_case - exception_case)
    (remediation_authorized ?exception_case - exception_case)
    (case_remediated ?exception_case - exception_case)
    (recipient_available ?escalation_recipient - escalation_recipient)
    (case_assigned_to_recipient ?exception_case - exception_case ?escalation_recipient - escalation_recipient)
    (analyst_available ?analyst - analyst)
    (case_assigned_to_analyst ?exception_case - exception_case ?analyst - analyst)
    (manager_available ?manager - manager)
    (case_assigned_to_manager ?exception_case - exception_case ?manager - manager)
    (evidence_document_available ?evidence_document - evidence_document)
    (account_context_evidence_linked ?account_context_case - account_context_case ?evidence_document - evidence_document)
    (business_unit_context_evidence_linked ?business_unit_context_case - business_unit_context_case ?evidence_document - evidence_document)
    (account_context_control_point_linked ?account_context_case - account_context_case ?control_point - control_point)
    (control_point_flagged ?control_point - control_point)
    (control_point_documented ?control_point - control_point)
    (account_context_in_review ?account_context_case - account_context_case)
    (business_unit_context_control_procedure_linked ?business_unit_context_case - business_unit_context_case ?control_procedure - control_procedure)
    (control_procedure_flagged ?control_procedure - control_procedure)
    (control_procedure_documented ?control_procedure - control_procedure)
    (business_unit_context_in_review ?business_unit_context_case - business_unit_context_case)
    (package_draft_available ?escalation_package - escalation_package)
    (package_assembled ?escalation_package - escalation_package)
    (package_control_point_documented ?escalation_package - escalation_package ?control_point - control_point)
    (package_control_procedure_documented ?escalation_package - escalation_package ?control_procedure - control_procedure)
    (package_account_context_complete ?escalation_package - escalation_package)
    (package_business_unit_context_complete ?escalation_package - escalation_package)
    (package_analyst_attested ?escalation_package - escalation_package)
    (record_linked_to_account_context ?investigation_record - investigation_record ?account_context_case - account_context_case)
    (record_linked_to_business_unit_context ?investigation_record - investigation_record ?business_unit_context_case - business_unit_context_case)
    (record_linked_to_package ?investigation_record - investigation_record ?escalation_package - escalation_package)
    (supporting_document_available ?supporting_document - supporting_document)
    (record_linked_to_supporting_document ?investigation_record - investigation_record ?supporting_document - supporting_document)
    (supporting_document_attached ?supporting_document - supporting_document)
    (supporting_document_linked_to_package ?supporting_document - supporting_document ?escalation_package - escalation_package)
    (specialist_review_complete ?investigation_record - investigation_record)
    (manager_review_complete ?investigation_record - investigation_record)
    (senior_review_complete ?investigation_record - investigation_record)
    (compliance_reference_cited ?investigation_record - investigation_record)
    (compliance_basis_confirmed ?investigation_record - investigation_record)
    (final_signoff_ready ?investigation_record - investigation_record)
    (audit_support_compiled ?investigation_record - investigation_record)
    (regulatory_finding_available ?regulatory_finding - regulatory_finding)
    (record_linked_to_regulatory_finding ?investigation_record - investigation_record ?regulatory_finding - regulatory_finding)
    (regulatory_finding_logged ?investigation_record - investigation_record)
    (control_verification_complete ?investigation_record - investigation_record)
    (audit_issue_reviewed ?investigation_record - investigation_record)
    (compliance_reference_available ?compliance_reference - compliance_reference_document)
    (record_linked_to_compliance_reference ?investigation_record - investigation_record ?compliance_reference - compliance_reference_document)
    (final_signoff_available ?final_signoff - final_signoff_document)
    (record_linked_to_final_signoff ?investigation_record - investigation_record ?final_signoff - final_signoff_document)
    (remediation_action_available ?remediation_action - remediation_task)
    (record_linked_to_remediation_action ?investigation_record - investigation_record ?remediation_action - remediation_task)
    (audit_issue_available ?audit_issue - audit_issue_record)
    (record_linked_to_audit_issue ?investigation_record - investigation_record ?audit_issue - audit_issue_record)
    (audit_reference_available ?audit_reference - audit_reference_document)
    (case_linked_to_audit_reference ?exception_case - exception_case ?audit_reference - audit_reference_document)
    (account_context_active ?account_context_case - account_context_case)
    (business_unit_context_active ?business_unit_context_case - business_unit_context_case)
    (record_finalized ?investigation_record - investigation_record)
  )
  (:action record_control_exception
    :parameters (?exception_case - exception_case)
    :precondition
      (and
        (not
          (case_logged ?exception_case)
        )
        (not
          (case_closed ?exception_case)
        )
      )
    :effect (case_logged ?exception_case)
  )
  (:action assign_escalation_recipient
    :parameters (?exception_case - exception_case ?escalation_recipient - escalation_recipient)
    :precondition
      (and
        (case_logged ?exception_case)
        (not
          (case_triaged ?exception_case)
        )
        (recipient_available ?escalation_recipient)
      )
    :effect
      (and
        (case_triaged ?exception_case)
        (case_assigned_to_recipient ?exception_case ?escalation_recipient)
        (not
          (recipient_available ?escalation_recipient)
        )
      )
  )
  (:action assign_analyst
    :parameters (?exception_case - exception_case ?analyst - analyst)
    :precondition
      (and
        (case_logged ?exception_case)
        (case_triaged ?exception_case)
        (analyst_available ?analyst)
      )
    :effect
      (and
        (case_assigned_to_analyst ?exception_case ?analyst)
        (not
          (analyst_available ?analyst)
        )
      )
  )
  (:action acknowledge_case_assignment
    :parameters (?exception_case - exception_case ?analyst - analyst)
    :precondition
      (and
        (case_logged ?exception_case)
        (case_triaged ?exception_case)
        (case_assigned_to_analyst ?exception_case ?analyst)
        (not
          (case_acknowledged ?exception_case)
        )
      )
    :effect (case_acknowledged ?exception_case)
  )
  (:action release_analyst_assignment
    :parameters (?exception_case - exception_case ?analyst - analyst)
    :precondition
      (and
        (case_assigned_to_analyst ?exception_case ?analyst)
      )
    :effect
      (and
        (analyst_available ?analyst)
        (not
          (case_assigned_to_analyst ?exception_case ?analyst)
        )
      )
  )
  (:action assign_manager
    :parameters (?exception_case - exception_case ?manager - manager)
    :precondition
      (and
        (case_acknowledged ?exception_case)
        (manager_available ?manager)
      )
    :effect
      (and
        (case_assigned_to_manager ?exception_case ?manager)
        (not
          (manager_available ?manager)
        )
      )
  )
  (:action release_manager_assignment
    :parameters (?exception_case - exception_case ?manager - manager)
    :precondition
      (and
        (case_assigned_to_manager ?exception_case ?manager)
      )
    :effect
      (and
        (manager_available ?manager)
        (not
          (case_assigned_to_manager ?exception_case ?manager)
        )
      )
  )
  (:action assign_remediation_action
    :parameters (?investigation_record - investigation_record ?remediation_action - remediation_task)
    :precondition
      (and
        (case_acknowledged ?investigation_record)
        (remediation_action_available ?remediation_action)
      )
    :effect
      (and
        (record_linked_to_remediation_action ?investigation_record ?remediation_action)
        (not
          (remediation_action_available ?remediation_action)
        )
      )
  )
  (:action release_remediation_action
    :parameters (?investigation_record - investigation_record ?remediation_action - remediation_task)
    :precondition
      (and
        (record_linked_to_remediation_action ?investigation_record ?remediation_action)
      )
    :effect
      (and
        (remediation_action_available ?remediation_action)
        (not
          (record_linked_to_remediation_action ?investigation_record ?remediation_action)
        )
      )
  )
  (:action assign_audit_issue
    :parameters (?investigation_record - investigation_record ?audit_issue - audit_issue_record)
    :precondition
      (and
        (case_acknowledged ?investigation_record)
        (audit_issue_available ?audit_issue)
      )
    :effect
      (and
        (record_linked_to_audit_issue ?investigation_record ?audit_issue)
        (not
          (audit_issue_available ?audit_issue)
        )
      )
  )
  (:action release_audit_issue
    :parameters (?investigation_record - investigation_record ?audit_issue - audit_issue_record)
    :precondition
      (and
        (record_linked_to_audit_issue ?investigation_record ?audit_issue)
      )
    :effect
      (and
        (audit_issue_available ?audit_issue)
        (not
          (record_linked_to_audit_issue ?investigation_record ?audit_issue)
        )
      )
  )
  (:action flag_account_control_point
    :parameters (?account_context_case - account_context_case ?control_point - control_point ?analyst - analyst)
    :precondition
      (and
        (case_acknowledged ?account_context_case)
        (case_assigned_to_analyst ?account_context_case ?analyst)
        (account_context_control_point_linked ?account_context_case ?control_point)
        (not
          (control_point_flagged ?control_point)
        )
        (not
          (control_point_documented ?control_point)
        )
      )
    :effect (control_point_flagged ?control_point)
  )
  (:action initiate_account_review
    :parameters (?account_context_case - account_context_case ?control_point - control_point ?manager - manager)
    :precondition
      (and
        (case_acknowledged ?account_context_case)
        (case_assigned_to_manager ?account_context_case ?manager)
        (account_context_control_point_linked ?account_context_case ?control_point)
        (control_point_flagged ?control_point)
        (not
          (account_context_active ?account_context_case)
        )
      )
    :effect
      (and
        (account_context_active ?account_context_case)
        (account_context_in_review ?account_context_case)
      )
  )
  (:action attach_account_evidence
    :parameters (?account_context_case - account_context_case ?control_point - control_point ?evidence_document - evidence_document)
    :precondition
      (and
        (case_acknowledged ?account_context_case)
        (account_context_control_point_linked ?account_context_case ?control_point)
        (evidence_document_available ?evidence_document)
        (not
          (account_context_active ?account_context_case)
        )
      )
    :effect
      (and
        (control_point_documented ?control_point)
        (account_context_active ?account_context_case)
        (account_context_evidence_linked ?account_context_case ?evidence_document)
        (not
          (evidence_document_available ?evidence_document)
        )
      )
  )
  (:action advance_account_evidence_review
    :parameters (?account_context_case - account_context_case ?control_point - control_point ?analyst - analyst ?evidence_document - evidence_document)
    :precondition
      (and
        (case_acknowledged ?account_context_case)
        (case_assigned_to_analyst ?account_context_case ?analyst)
        (account_context_control_point_linked ?account_context_case ?control_point)
        (control_point_documented ?control_point)
        (account_context_evidence_linked ?account_context_case ?evidence_document)
        (not
          (account_context_in_review ?account_context_case)
        )
      )
    :effect
      (and
        (control_point_flagged ?control_point)
        (account_context_in_review ?account_context_case)
        (evidence_document_available ?evidence_document)
        (not
          (account_context_evidence_linked ?account_context_case ?evidence_document)
        )
      )
  )
  (:action flag_business_unit_control_procedure
    :parameters (?business_unit_context_case - business_unit_context_case ?control_procedure - control_procedure ?analyst - analyst)
    :precondition
      (and
        (case_acknowledged ?business_unit_context_case)
        (case_assigned_to_analyst ?business_unit_context_case ?analyst)
        (business_unit_context_control_procedure_linked ?business_unit_context_case ?control_procedure)
        (not
          (control_procedure_flagged ?control_procedure)
        )
        (not
          (control_procedure_documented ?control_procedure)
        )
      )
    :effect (control_procedure_flagged ?control_procedure)
  )
  (:action initiate_business_unit_review
    :parameters (?business_unit_context_case - business_unit_context_case ?control_procedure - control_procedure ?manager - manager)
    :precondition
      (and
        (case_acknowledged ?business_unit_context_case)
        (case_assigned_to_manager ?business_unit_context_case ?manager)
        (business_unit_context_control_procedure_linked ?business_unit_context_case ?control_procedure)
        (control_procedure_flagged ?control_procedure)
        (not
          (business_unit_context_active ?business_unit_context_case)
        )
      )
    :effect
      (and
        (business_unit_context_active ?business_unit_context_case)
        (business_unit_context_in_review ?business_unit_context_case)
      )
  )
  (:action attach_business_unit_evidence
    :parameters (?business_unit_context_case - business_unit_context_case ?control_procedure - control_procedure ?evidence_document - evidence_document)
    :precondition
      (and
        (case_acknowledged ?business_unit_context_case)
        (business_unit_context_control_procedure_linked ?business_unit_context_case ?control_procedure)
        (evidence_document_available ?evidence_document)
        (not
          (business_unit_context_active ?business_unit_context_case)
        )
      )
    :effect
      (and
        (control_procedure_documented ?control_procedure)
        (business_unit_context_active ?business_unit_context_case)
        (business_unit_context_evidence_linked ?business_unit_context_case ?evidence_document)
        (not
          (evidence_document_available ?evidence_document)
        )
      )
  )
  (:action advance_business_unit_evidence_review
    :parameters (?business_unit_context_case - business_unit_context_case ?control_procedure - control_procedure ?analyst - analyst ?evidence_document - evidence_document)
    :precondition
      (and
        (case_acknowledged ?business_unit_context_case)
        (case_assigned_to_analyst ?business_unit_context_case ?analyst)
        (business_unit_context_control_procedure_linked ?business_unit_context_case ?control_procedure)
        (control_procedure_documented ?control_procedure)
        (business_unit_context_evidence_linked ?business_unit_context_case ?evidence_document)
        (not
          (business_unit_context_in_review ?business_unit_context_case)
        )
      )
    :effect
      (and
        (control_procedure_flagged ?control_procedure)
        (business_unit_context_in_review ?business_unit_context_case)
        (evidence_document_available ?evidence_document)
        (not
          (business_unit_context_evidence_linked ?business_unit_context_case ?evidence_document)
        )
      )
  )
  (:action assemble_escalation_package
    :parameters (?account_context_case - account_context_case ?business_unit_context_case - business_unit_context_case ?control_point - control_point ?control_procedure - control_procedure ?escalation_package - escalation_package)
    :precondition
      (and
        (account_context_active ?account_context_case)
        (business_unit_context_active ?business_unit_context_case)
        (account_context_control_point_linked ?account_context_case ?control_point)
        (business_unit_context_control_procedure_linked ?business_unit_context_case ?control_procedure)
        (control_point_flagged ?control_point)
        (control_procedure_flagged ?control_procedure)
        (account_context_in_review ?account_context_case)
        (business_unit_context_in_review ?business_unit_context_case)
        (package_draft_available ?escalation_package)
      )
    :effect
      (and
        (package_assembled ?escalation_package)
        (package_control_point_documented ?escalation_package ?control_point)
        (package_control_procedure_documented ?escalation_package ?control_procedure)
        (not
          (package_draft_available ?escalation_package)
        )
      )
  )
  (:action assemble_escalation_package_with_account_confirmation
    :parameters (?account_context_case - account_context_case ?business_unit_context_case - business_unit_context_case ?control_point - control_point ?control_procedure - control_procedure ?escalation_package - escalation_package)
    :precondition
      (and
        (account_context_active ?account_context_case)
        (business_unit_context_active ?business_unit_context_case)
        (account_context_control_point_linked ?account_context_case ?control_point)
        (business_unit_context_control_procedure_linked ?business_unit_context_case ?control_procedure)
        (control_point_documented ?control_point)
        (control_procedure_flagged ?control_procedure)
        (not
          (account_context_in_review ?account_context_case)
        )
        (business_unit_context_in_review ?business_unit_context_case)
        (package_draft_available ?escalation_package)
      )
    :effect
      (and
        (package_assembled ?escalation_package)
        (package_control_point_documented ?escalation_package ?control_point)
        (package_control_procedure_documented ?escalation_package ?control_procedure)
        (package_account_context_complete ?escalation_package)
        (not
          (package_draft_available ?escalation_package)
        )
      )
  )
  (:action assemble_escalation_package_with_business_unit_confirmation
    :parameters (?account_context_case - account_context_case ?business_unit_context_case - business_unit_context_case ?control_point - control_point ?control_procedure - control_procedure ?escalation_package - escalation_package)
    :precondition
      (and
        (account_context_active ?account_context_case)
        (business_unit_context_active ?business_unit_context_case)
        (account_context_control_point_linked ?account_context_case ?control_point)
        (business_unit_context_control_procedure_linked ?business_unit_context_case ?control_procedure)
        (control_point_flagged ?control_point)
        (control_procedure_documented ?control_procedure)
        (account_context_in_review ?account_context_case)
        (not
          (business_unit_context_in_review ?business_unit_context_case)
        )
        (package_draft_available ?escalation_package)
      )
    :effect
      (and
        (package_assembled ?escalation_package)
        (package_control_point_documented ?escalation_package ?control_point)
        (package_control_procedure_documented ?escalation_package ?control_procedure)
        (package_business_unit_context_complete ?escalation_package)
        (not
          (package_draft_available ?escalation_package)
        )
      )
  )
  (:action assemble_escalation_package_with_full_confirmation
    :parameters (?account_context_case - account_context_case ?business_unit_context_case - business_unit_context_case ?control_point - control_point ?control_procedure - control_procedure ?escalation_package - escalation_package)
    :precondition
      (and
        (account_context_active ?account_context_case)
        (business_unit_context_active ?business_unit_context_case)
        (account_context_control_point_linked ?account_context_case ?control_point)
        (business_unit_context_control_procedure_linked ?business_unit_context_case ?control_procedure)
        (control_point_documented ?control_point)
        (control_procedure_documented ?control_procedure)
        (not
          (account_context_in_review ?account_context_case)
        )
        (not
          (business_unit_context_in_review ?business_unit_context_case)
        )
        (package_draft_available ?escalation_package)
      )
    :effect
      (and
        (package_assembled ?escalation_package)
        (package_control_point_documented ?escalation_package ?control_point)
        (package_control_procedure_documented ?escalation_package ?control_procedure)
        (package_account_context_complete ?escalation_package)
        (package_business_unit_context_complete ?escalation_package)
        (not
          (package_draft_available ?escalation_package)
        )
      )
  )
  (:action attest_escalation_package
    :parameters (?escalation_package - escalation_package ?account_context_case - account_context_case ?analyst - analyst)
    :precondition
      (and
        (package_assembled ?escalation_package)
        (account_context_active ?account_context_case)
        (case_assigned_to_analyst ?account_context_case ?analyst)
        (not
          (package_analyst_attested ?escalation_package)
        )
      )
    :effect (package_analyst_attested ?escalation_package)
  )
  (:action attach_supporting_document_to_package
    :parameters (?investigation_record - investigation_record ?supporting_document - supporting_document ?escalation_package - escalation_package)
    :precondition
      (and
        (case_acknowledged ?investigation_record)
        (record_linked_to_package ?investigation_record ?escalation_package)
        (record_linked_to_supporting_document ?investigation_record ?supporting_document)
        (supporting_document_available ?supporting_document)
        (package_assembled ?escalation_package)
        (package_analyst_attested ?escalation_package)
        (not
          (supporting_document_attached ?supporting_document)
        )
      )
    :effect
      (and
        (supporting_document_attached ?supporting_document)
        (supporting_document_linked_to_package ?supporting_document ?escalation_package)
        (not
          (supporting_document_available ?supporting_document)
        )
      )
  )
  (:action complete_specialist_review
    :parameters (?investigation_record - investigation_record ?supporting_document - supporting_document ?escalation_package - escalation_package ?analyst - analyst)
    :precondition
      (and
        (case_acknowledged ?investigation_record)
        (record_linked_to_supporting_document ?investigation_record ?supporting_document)
        (supporting_document_attached ?supporting_document)
        (supporting_document_linked_to_package ?supporting_document ?escalation_package)
        (case_assigned_to_analyst ?investigation_record ?analyst)
        (not
          (package_account_context_complete ?escalation_package)
        )
        (not
          (specialist_review_complete ?investigation_record)
        )
      )
    :effect (specialist_review_complete ?investigation_record)
  )
  (:action cite_compliance_reference
    :parameters (?investigation_record - investigation_record ?compliance_reference - compliance_reference_document)
    :precondition
      (and
        (case_acknowledged ?investigation_record)
        (compliance_reference_available ?compliance_reference)
        (not
          (compliance_reference_cited ?investigation_record)
        )
      )
    :effect
      (and
        (compliance_reference_cited ?investigation_record)
        (record_linked_to_compliance_reference ?investigation_record ?compliance_reference)
        (not
          (compliance_reference_available ?compliance_reference)
        )
      )
  )
  (:action confirm_compliance_basis
    :parameters (?investigation_record - investigation_record ?supporting_document - supporting_document ?escalation_package - escalation_package ?analyst - analyst ?compliance_reference - compliance_reference_document)
    :precondition
      (and
        (case_acknowledged ?investigation_record)
        (record_linked_to_supporting_document ?investigation_record ?supporting_document)
        (supporting_document_attached ?supporting_document)
        (supporting_document_linked_to_package ?supporting_document ?escalation_package)
        (case_assigned_to_analyst ?investigation_record ?analyst)
        (package_account_context_complete ?escalation_package)
        (compliance_reference_cited ?investigation_record)
        (record_linked_to_compliance_reference ?investigation_record ?compliance_reference)
        (not
          (specialist_review_complete ?investigation_record)
        )
      )
    :effect
      (and
        (specialist_review_complete ?investigation_record)
        (compliance_basis_confirmed ?investigation_record)
      )
  )
  (:action complete_manager_review
    :parameters (?investigation_record - investigation_record ?remediation_action - remediation_task ?manager - manager ?supporting_document - supporting_document ?escalation_package - escalation_package)
    :precondition
      (and
        (specialist_review_complete ?investigation_record)
        (record_linked_to_remediation_action ?investigation_record ?remediation_action)
