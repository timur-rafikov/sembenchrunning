## Planning domain (PDDL)

(define (domain tariff_classification_assignment)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity - object actor - entity document - entity classification_code - entity case_root - entity classification_case - case_root correspondent_agent - actor product_evidence - actor subject_matter_expert - actor classification_attribute - actor review_criterion - actor clearance_authorization_token - actor tariff_note - actor legal_opinion_record - actor supporting_document - document regulatory_reference - document external_ruling_record - document primary_classification_candidate - classification_code secondary_classification_candidate - classification_code compliance_submission_template - classification_code case_branch_import - classification_case case_branch_export - classification_case import_case - case_branch_import export_case - case_branch_import work_item - case_branch_export)

  (:predicates
    (entity_intake_opened ?classification_case - classification_case)
    (entity_intake_complete ?classification_case - classification_case)
    (entity_assigned ?classification_case - classification_case)
    (entity_authorized_for_filing ?classification_case - classification_case)
    (entity_ready_for_authorization ?classification_case - classification_case)
    (entity_authorization_issued ?classification_case - classification_case)
    (agent_available ?correspondent_agent - correspondent_agent)
    (entity_assigned_to_agent ?classification_case - classification_case ?correspondent_agent - correspondent_agent)
    (evidence_available ?product_evidence - product_evidence)
    (entity_has_evidence ?classification_case - classification_case ?product_evidence - product_evidence)
    (expert_available ?subject_matter_expert - subject_matter_expert)
    (entity_assigned_expert ?classification_case - classification_case ?subject_matter_expert - subject_matter_expert)
    (supporting_document_available ?supporting_document - supporting_document)
    (import_case_has_document ?import_case - import_case ?supporting_document - supporting_document)
    (export_case_has_document ?export_case - export_case ?supporting_document - supporting_document)
    (import_case_has_primary_candidate ?import_case - import_case ?primary_classification_candidate - primary_classification_candidate)
    (primary_candidate_marked ?primary_classification_candidate - primary_classification_candidate)
    (primary_candidate_documented ?primary_classification_candidate - primary_classification_candidate)
    (import_case_candidate_validated ?import_case - import_case)
    (export_case_has_secondary_candidate ?export_case - export_case ?secondary_classification_candidate - secondary_classification_candidate)
    (secondary_candidate_marked ?secondary_classification_candidate - secondary_classification_candidate)
    (secondary_candidate_documented ?secondary_classification_candidate - secondary_classification_candidate)
    (export_case_candidate_validated ?export_case - export_case)
    (compliance_submission_template_available ?compliance_submission_template - compliance_submission_template)
    (compliance_submission_template_prepared ?compliance_submission_template - compliance_submission_template)
    (compliance_submission_template_has_primary_candidate ?compliance_submission_template - compliance_submission_template ?primary_classification_candidate - primary_classification_candidate)
    (compliance_submission_template_has_secondary_candidate ?compliance_submission_template - compliance_submission_template ?secondary_classification_candidate - secondary_classification_candidate)
    (compliance_submission_template_requires_attribute_binding ?compliance_submission_template - compliance_submission_template)
    (compliance_submission_template_requires_legal_opinion_binding ?compliance_submission_template - compliance_submission_template)
    (compliance_submission_template_validated ?compliance_submission_template - compliance_submission_template)
    (work_item_linked_to_import_case ?work_item - work_item ?import_case - import_case)
    (work_item_linked_to_export_case ?work_item - work_item ?export_case - export_case)
    (work_item_bound_to_compliance_submission_template ?work_item - work_item ?compliance_submission_template - compliance_submission_template)
    (regulatory_reference_available ?regulatory_reference - regulatory_reference)
    (work_item_references_regulatory_reference ?work_item - work_item ?regulatory_reference - regulatory_reference)
    (regulatory_reference_consumed ?regulatory_reference - regulatory_reference)
    (regulatory_reference_used_in_compliance_submission_template ?regulatory_reference - regulatory_reference ?compliance_submission_template - compliance_submission_template)
    (work_item_submission_ready ?work_item - work_item)
    (work_item_attributes_attached ?work_item - work_item)
    (work_item_final_annotation_complete ?work_item - work_item)
    (work_item_attribute_assigned ?work_item - work_item)
    (work_item_attribute_locked ?work_item - work_item)
    (work_item_ready_for_review ?work_item - work_item)
    (work_item_finalized ?work_item - work_item)
    (external_ruling_available ?external_ruling_record - external_ruling_record)
    (work_item_has_external_ruling ?work_item - work_item ?external_ruling_record - external_ruling_record)
    (work_item_external_ruling_recorded ?work_item - work_item)
    (work_item_external_ruling_acknowledged ?work_item - work_item)
    (work_item_expert_validation_complete ?work_item - work_item)
    (classification_attribute_available ?classification_attribute - classification_attribute)
    (work_item_has_classification_attribute ?work_item - work_item ?classification_attribute - classification_attribute)
    (review_criterion_available ?review_criterion - review_criterion)
    (work_item_has_review_criterion ?work_item - work_item ?review_criterion - review_criterion)
    (tariff_note_available ?tariff_note - tariff_note)
    (work_item_has_tariff_note ?work_item - work_item ?tariff_note - tariff_note)
    (legal_opinion_available ?legal_opinion_record - legal_opinion_record)
    (work_item_has_legal_opinion ?work_item - work_item ?legal_opinion_record - legal_opinion_record)
    (clearance_token_available ?clearance_authorization_token - clearance_authorization_token)
    (clearance_token_assigned_to_entity ?classification_case - classification_case ?clearance_authorization_token - clearance_authorization_token)
    (import_case_processing_ready ?import_case - import_case)
    (export_case_processing_ready ?export_case - export_case)
    (work_item_signoff_recorded ?work_item - work_item)
  )
  (:action open_classification_case
    :parameters (?classification_case - classification_case)
    :precondition
      (and
        (not
          (entity_intake_opened ?classification_case)
        )
        (not
          (entity_authorized_for_filing ?classification_case)
        )
      )
    :effect (entity_intake_opened ?classification_case)
  )
  (:action assign_correspondent_agent_to_case
    :parameters (?classification_case - classification_case ?correspondent_agent - correspondent_agent)
    :precondition
      (and
        (entity_intake_opened ?classification_case)
        (not
          (entity_assigned ?classification_case)
        )
        (agent_available ?correspondent_agent)
      )
    :effect
      (and
        (entity_assigned ?classification_case)
        (entity_assigned_to_agent ?classification_case ?correspondent_agent)
        (not
          (agent_available ?correspondent_agent)
        )
      )
  )
  (:action attach_evidence_to_case
    :parameters (?classification_case - classification_case ?product_evidence - product_evidence)
    :precondition
      (and
        (entity_intake_opened ?classification_case)
        (entity_assigned ?classification_case)
        (evidence_available ?product_evidence)
      )
    :effect
      (and
        (entity_has_evidence ?classification_case ?product_evidence)
        (not
          (evidence_available ?product_evidence)
        )
      )
  )
  (:action complete_case_intake
    :parameters (?classification_case - classification_case ?product_evidence - product_evidence)
    :precondition
      (and
        (entity_intake_opened ?classification_case)
        (entity_assigned ?classification_case)
        (entity_has_evidence ?classification_case ?product_evidence)
        (not
          (entity_intake_complete ?classification_case)
        )
      )
    :effect (entity_intake_complete ?classification_case)
  )
  (:action release_evidence_from_case
    :parameters (?classification_case - classification_case ?product_evidence - product_evidence)
    :precondition
      (and
        (entity_has_evidence ?classification_case ?product_evidence)
      )
    :effect
      (and
        (evidence_available ?product_evidence)
        (not
          (entity_has_evidence ?classification_case ?product_evidence)
        )
      )
  )
  (:action assign_subject_matter_expert_to_case
    :parameters (?classification_case - classification_case ?subject_matter_expert - subject_matter_expert)
    :precondition
      (and
        (entity_intake_complete ?classification_case)
        (expert_available ?subject_matter_expert)
      )
    :effect
      (and
        (entity_assigned_expert ?classification_case ?subject_matter_expert)
        (not
          (expert_available ?subject_matter_expert)
        )
      )
  )
  (:action release_subject_matter_expert_from_case
    :parameters (?classification_case - classification_case ?subject_matter_expert - subject_matter_expert)
    :precondition
      (and
        (entity_assigned_expert ?classification_case ?subject_matter_expert)
      )
    :effect
      (and
        (expert_available ?subject_matter_expert)
        (not
          (entity_assigned_expert ?classification_case ?subject_matter_expert)
        )
      )
  )
  (:action attach_tariff_note_to_work_item
    :parameters (?work_item - work_item ?tariff_note - tariff_note)
    :precondition
      (and
        (entity_intake_complete ?work_item)
        (tariff_note_available ?tariff_note)
      )
    :effect
      (and
        (work_item_has_tariff_note ?work_item ?tariff_note)
        (not
          (tariff_note_available ?tariff_note)
        )
      )
  )
  (:action release_tariff_note_from_work_item
    :parameters (?work_item - work_item ?tariff_note - tariff_note)
    :precondition
      (and
        (work_item_has_tariff_note ?work_item ?tariff_note)
      )
    :effect
      (and
        (tariff_note_available ?tariff_note)
        (not
          (work_item_has_tariff_note ?work_item ?tariff_note)
        )
      )
  )
  (:action attach_legal_opinion_to_work_item
    :parameters (?work_item - work_item ?legal_opinion_record - legal_opinion_record)
    :precondition
      (and
        (entity_intake_complete ?work_item)
        (legal_opinion_available ?legal_opinion_record)
      )
    :effect
      (and
        (work_item_has_legal_opinion ?work_item ?legal_opinion_record)
        (not
          (legal_opinion_available ?legal_opinion_record)
        )
      )
  )
  (:action release_legal_opinion_from_work_item
    :parameters (?work_item - work_item ?legal_opinion_record - legal_opinion_record)
    :precondition
      (and
        (work_item_has_legal_opinion ?work_item ?legal_opinion_record)
      )
    :effect
      (and
        (legal_opinion_available ?legal_opinion_record)
        (not
          (work_item_has_legal_opinion ?work_item ?legal_opinion_record)
        )
      )
  )
  (:action propose_primary_candidate_for_import_case
    :parameters (?import_case - import_case ?primary_classification_candidate - primary_classification_candidate ?product_evidence - product_evidence)
    :precondition
      (and
        (entity_intake_complete ?import_case)
        (entity_has_evidence ?import_case ?product_evidence)
        (import_case_has_primary_candidate ?import_case ?primary_classification_candidate)
        (not
          (primary_candidate_marked ?primary_classification_candidate)
        )
        (not
          (primary_candidate_documented ?primary_classification_candidate)
        )
      )
    :effect (primary_candidate_marked ?primary_classification_candidate)
  )
  (:action expert_validate_primary_candidate
    :parameters (?import_case - import_case ?primary_classification_candidate - primary_classification_candidate ?subject_matter_expert - subject_matter_expert)
    :precondition
      (and
        (entity_intake_complete ?import_case)
        (entity_assigned_expert ?import_case ?subject_matter_expert)
        (import_case_has_primary_candidate ?import_case ?primary_classification_candidate)
        (primary_candidate_marked ?primary_classification_candidate)
        (not
          (import_case_processing_ready ?import_case)
        )
      )
    :effect
      (and
        (import_case_processing_ready ?import_case)
        (import_case_candidate_validated ?import_case)
      )
  )
  (:action attach_supporting_document_to_import_case
    :parameters (?import_case - import_case ?primary_classification_candidate - primary_classification_candidate ?supporting_document - supporting_document)
    :precondition
      (and
        (entity_intake_complete ?import_case)
        (import_case_has_primary_candidate ?import_case ?primary_classification_candidate)
        (supporting_document_available ?supporting_document)
        (not
          (import_case_processing_ready ?import_case)
        )
      )
    :effect
      (and
        (primary_candidate_documented ?primary_classification_candidate)
        (import_case_processing_ready ?import_case)
        (import_case_has_document ?import_case ?supporting_document)
        (not
          (supporting_document_available ?supporting_document)
        )
      )
  )
  (:action confirm_primary_candidate_with_documentation
    :parameters (?import_case - import_case ?primary_classification_candidate - primary_classification_candidate ?product_evidence - product_evidence ?supporting_document - supporting_document)
    :precondition
      (and
        (entity_intake_complete ?import_case)
        (entity_has_evidence ?import_case ?product_evidence)
        (import_case_has_primary_candidate ?import_case ?primary_classification_candidate)
        (primary_candidate_documented ?primary_classification_candidate)
        (import_case_has_document ?import_case ?supporting_document)
        (not
          (import_case_candidate_validated ?import_case)
        )
      )
    :effect
      (and
        (primary_candidate_marked ?primary_classification_candidate)
        (import_case_candidate_validated ?import_case)
        (supporting_document_available ?supporting_document)
        (not
          (import_case_has_document ?import_case ?supporting_document)
        )
      )
  )
  (:action propose_secondary_candidate_for_export_case
    :parameters (?export_case - export_case ?secondary_classification_candidate - secondary_classification_candidate ?product_evidence - product_evidence)
    :precondition
      (and
        (entity_intake_complete ?export_case)
        (entity_has_evidence ?export_case ?product_evidence)
        (export_case_has_secondary_candidate ?export_case ?secondary_classification_candidate)
        (not
          (secondary_candidate_marked ?secondary_classification_candidate)
        )
        (not
          (secondary_candidate_documented ?secondary_classification_candidate)
        )
      )
    :effect (secondary_candidate_marked ?secondary_classification_candidate)
  )
  (:action expert_validate_secondary_candidate
    :parameters (?export_case - export_case ?secondary_classification_candidate - secondary_classification_candidate ?subject_matter_expert - subject_matter_expert)
    :precondition
      (and
        (entity_intake_complete ?export_case)
        (entity_assigned_expert ?export_case ?subject_matter_expert)
        (export_case_has_secondary_candidate ?export_case ?secondary_classification_candidate)
        (secondary_candidate_marked ?secondary_classification_candidate)
        (not
          (export_case_processing_ready ?export_case)
        )
      )
    :effect
      (and
        (export_case_processing_ready ?export_case)
        (export_case_candidate_validated ?export_case)
      )
  )
  (:action attach_supporting_document_to_export_case
    :parameters (?export_case - export_case ?secondary_classification_candidate - secondary_classification_candidate ?supporting_document - supporting_document)
    :precondition
      (and
        (entity_intake_complete ?export_case)
        (export_case_has_secondary_candidate ?export_case ?secondary_classification_candidate)
        (supporting_document_available ?supporting_document)
        (not
          (export_case_processing_ready ?export_case)
        )
      )
    :effect
      (and
        (secondary_candidate_documented ?secondary_classification_candidate)
        (export_case_processing_ready ?export_case)
        (export_case_has_document ?export_case ?supporting_document)
        (not
          (supporting_document_available ?supporting_document)
        )
      )
  )
  (:action confirm_secondary_candidate_with_documentation
    :parameters (?export_case - export_case ?secondary_classification_candidate - secondary_classification_candidate ?product_evidence - product_evidence ?supporting_document - supporting_document)
    :precondition
      (and
        (entity_intake_complete ?export_case)
        (entity_has_evidence ?export_case ?product_evidence)
        (export_case_has_secondary_candidate ?export_case ?secondary_classification_candidate)
        (secondary_candidate_documented ?secondary_classification_candidate)
        (export_case_has_document ?export_case ?supporting_document)
        (not
          (export_case_candidate_validated ?export_case)
        )
      )
    :effect
      (and
        (secondary_candidate_marked ?secondary_classification_candidate)
        (export_case_candidate_validated ?export_case)
        (supporting_document_available ?supporting_document)
        (not
          (export_case_has_document ?export_case ?supporting_document)
        )
      )
  )
  (:action assemble_compliance_submission_template
    :parameters (?import_case - import_case ?export_case - export_case ?primary_classification_candidate - primary_classification_candidate ?secondary_classification_candidate - secondary_classification_candidate ?compliance_submission_template - compliance_submission_template)
    :precondition
      (and
        (import_case_processing_ready ?import_case)
        (export_case_processing_ready ?export_case)
        (import_case_has_primary_candidate ?import_case ?primary_classification_candidate)
        (export_case_has_secondary_candidate ?export_case ?secondary_classification_candidate)
        (primary_candidate_marked ?primary_classification_candidate)
        (secondary_candidate_marked ?secondary_classification_candidate)
        (import_case_candidate_validated ?import_case)
        (export_case_candidate_validated ?export_case)
        (compliance_submission_template_available ?compliance_submission_template)
      )
    :effect
      (and
        (compliance_submission_template_prepared ?compliance_submission_template)
        (compliance_submission_template_has_primary_candidate ?compliance_submission_template ?primary_classification_candidate)
        (compliance_submission_template_has_secondary_candidate ?compliance_submission_template ?secondary_classification_candidate)
        (not
          (compliance_submission_template_available ?compliance_submission_template)
        )
      )
  )
  (:action assemble_compliance_submission_template_with_attribute_binding
    :parameters (?import_case - import_case ?export_case - export_case ?primary_classification_candidate - primary_classification_candidate ?secondary_classification_candidate - secondary_classification_candidate ?compliance_submission_template - compliance_submission_template)
    :precondition
      (and
        (import_case_processing_ready ?import_case)
        (export_case_processing_ready ?export_case)
        (import_case_has_primary_candidate ?import_case ?primary_classification_candidate)
        (export_case_has_secondary_candidate ?export_case ?secondary_classification_candidate)
        (primary_candidate_documented ?primary_classification_candidate)
        (secondary_candidate_marked ?secondary_classification_candidate)
        (not
          (import_case_candidate_validated ?import_case)
        )
        (export_case_candidate_validated ?export_case)
        (compliance_submission_template_available ?compliance_submission_template)
      )
    :effect
      (and
        (compliance_submission_template_prepared ?compliance_submission_template)
        (compliance_submission_template_has_primary_candidate ?compliance_submission_template ?primary_classification_candidate)
        (compliance_submission_template_has_secondary_candidate ?compliance_submission_template ?secondary_classification_candidate)
        (compliance_submission_template_requires_attribute_binding ?compliance_submission_template)
        (not
          (compliance_submission_template_available ?compliance_submission_template)
        )
      )
  )
  (:action assemble_compliance_submission_template_with_legal_opinion_binding
    :parameters (?import_case - import_case ?export_case - export_case ?primary_classification_candidate - primary_classification_candidate ?secondary_classification_candidate - secondary_classification_candidate ?compliance_submission_template - compliance_submission_template)
    :precondition
      (and
        (import_case_processing_ready ?import_case)
        (export_case_processing_ready ?export_case)
        (import_case_has_primary_candidate ?import_case ?primary_classification_candidate)
        (export_case_has_secondary_candidate ?export_case ?secondary_classification_candidate)
        (primary_candidate_marked ?primary_classification_candidate)
        (secondary_candidate_documented ?secondary_classification_candidate)
        (import_case_candidate_validated ?import_case)
        (not
          (export_case_candidate_validated ?export_case)
        )
        (compliance_submission_template_available ?compliance_submission_template)
      )
    :effect
      (and
        (compliance_submission_template_prepared ?compliance_submission_template)
        (compliance_submission_template_has_primary_candidate ?compliance_submission_template ?primary_classification_candidate)
        (compliance_submission_template_has_secondary_candidate ?compliance_submission_template ?secondary_classification_candidate)
        (compliance_submission_template_requires_legal_opinion_binding ?compliance_submission_template)
        (not
          (compliance_submission_template_available ?compliance_submission_template)
        )
      )
  )
  (:action assemble_compliance_submission_template_with_full_bindings
    :parameters (?import_case - import_case ?export_case - export_case ?primary_classification_candidate - primary_classification_candidate ?secondary_classification_candidate - secondary_classification_candidate ?compliance_submission_template - compliance_submission_template)
    :precondition
      (and
        (import_case_processing_ready ?import_case)
        (export_case_processing_ready ?export_case)
        (import_case_has_primary_candidate ?import_case ?primary_classification_candidate)
        (export_case_has_secondary_candidate ?export_case ?secondary_classification_candidate)
        (primary_candidate_documented ?primary_classification_candidate)
        (secondary_candidate_documented ?secondary_classification_candidate)
        (not
          (import_case_candidate_validated ?import_case)
        )
        (not
          (export_case_candidate_validated ?export_case)
        )
        (compliance_submission_template_available ?compliance_submission_template)
      )
    :effect
      (and
        (compliance_submission_template_prepared ?compliance_submission_template)
        (compliance_submission_template_has_primary_candidate ?compliance_submission_template ?primary_classification_candidate)
        (compliance_submission_template_has_secondary_candidate ?compliance_submission_template ?secondary_classification_candidate)
        (compliance_submission_template_requires_attribute_binding ?compliance_submission_template)
        (compliance_submission_template_requires_legal_opinion_binding ?compliance_submission_template)
        (not
          (compliance_submission_template_available ?compliance_submission_template)
        )
      )
  )
  (:action validate_compliance_submission_template_for_case
    :parameters (?compliance_submission_template - compliance_submission_template ?import_case - import_case ?product_evidence - product_evidence)
    :precondition
      (and
        (compliance_submission_template_prepared ?compliance_submission_template)
        (import_case_processing_ready ?import_case)
        (entity_has_evidence ?import_case ?product_evidence)
        (not
          (compliance_submission_template_validated ?compliance_submission_template)
        )
      )
    :effect (compliance_submission_template_validated ?compliance_submission_template)
  )
  (:action bind_regulatory_reference_to_work_item
    :parameters (?work_item - work_item ?regulatory_reference - regulatory_reference ?compliance_submission_template - compliance_submission_template)
    :precondition
      (and
        (entity_intake_complete ?work_item)
        (work_item_bound_to_compliance_submission_template ?work_item ?compliance_submission_template)
        (work_item_references_regulatory_reference ?work_item ?regulatory_reference)
        (regulatory_reference_available ?regulatory_reference)
        (compliance_submission_template_prepared ?compliance_submission_template)
        (compliance_submission_template_validated ?compliance_submission_template)
        (not
          (regulatory_reference_consumed ?regulatory_reference)
        )
      )
    :effect
      (and
        (regulatory_reference_consumed ?regulatory_reference)
        (regulatory_reference_used_in_compliance_submission_template ?regulatory_reference ?compliance_submission_template)
        (not
          (regulatory_reference_available ?regulatory_reference)
        )
      )
  )
  (:action finalize_work_item_template_binding
    :parameters (?work_item - work_item ?regulatory_reference - regulatory_reference ?compliance_submission_template - compliance_submission_template ?product_evidence - product_evidence)
    :precondition
      (and
        (entity_intake_complete ?work_item)
        (work_item_references_regulatory_reference ?work_item ?regulatory_reference)
        (regulatory_reference_consumed ?regulatory_reference)
        (regulatory_reference_used_in_compliance_submission_template ?regulatory_reference ?compliance_submission_template)
        (entity_has_evidence ?work_item ?product_evidence)
        (not
          (compliance_submission_template_requires_attribute_binding ?compliance_submission_template)
        )
        (not
          (work_item_submission_ready ?work_item)
        )
      )
    :effect (work_item_submission_ready ?work_item)
  )
  (:action assign_classification_attribute_to_work_item
    :parameters (?work_item - work_item ?classification_attribute - classification_attribute)
    :precondition
      (and
        (entity_intake_complete ?work_item)
        (classification_attribute_available ?classification_attribute)
        (not
          (work_item_attribute_assigned ?work_item)
        )
      )
    :effect
      (and
        (work_item_attribute_assigned ?work_item)
        (work_item_has_classification_attribute ?work_item ?classification_attribute)
        (not
          (classification_attribute_available ?classification_attribute)
        )
      )
