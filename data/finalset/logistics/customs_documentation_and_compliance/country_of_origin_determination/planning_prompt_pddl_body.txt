## Planning domain (PDDL)

  (:requirements :strips :typing :negative-preconditions)
  (:types resource_or_actor_category - object document_category - object evidence_package_category - object case_root - object trade_entity - case_root broker_service - resource_or_actor_category classification_reference - resource_or_actor_category technical_reviewer - resource_or_actor_category legal_rule_reference - resource_or_actor_category regulatory_permit - resource_or_actor_category country_reference - resource_or_actor_category commercial_record - resource_or_actor_category supplier_declaration - resource_or_actor_category evidence_document - document_category product_line_item - document_category preference_program - document_category supplier_evidence_bundle_a - evidence_package_category supplier_evidence_bundle_b - evidence_package_category submission_package - evidence_package_category case_subject_group_a - trade_entity case_subject_group_b - trade_entity producer_entity - case_subject_group_a assembler_or_importer_entity - case_subject_group_a origin_assessment_case - case_subject_group_b)

    (reviewer_available ?technical_reviewer - technical_reviewer)
    (entity_assigned_reviewer ?trade_entity - trade_entity ?technical_reviewer - technical_reviewer)
    (evidence_document_available ?evidence_document - evidence_document)
    (entity_lineitem_has_document ?producer_entity - producer_entity ?evidence_document - evidence_document)
    (entity_lineitem_has_document_alt ?assembler_or_importer_entity - assembler_or_importer_entity ?evidence_document - evidence_document)
    (party_linked_evidence_bundle ?producer_entity - producer_entity ?supplier_evidence_bundle_a - supplier_evidence_bundle_a)
    (evidence_bundle_a_validated ?supplier_evidence_bundle_a - supplier_evidence_bundle_a)
    (evidence_bundle_a_pending ?supplier_evidence_bundle_a - supplier_evidence_bundle_a)
    (producer_entity_evidence_confirmed ?producer_entity - producer_entity)
    (party_linked_evidence_bundle_b ?assembler_or_importer_entity - assembler_or_importer_entity ?supplier_evidence_bundle_b - supplier_evidence_bundle_b)
    (evidence_bundle_b_validated ?supplier_evidence_bundle_b - supplier_evidence_bundle_b)
    (evidence_bundle_b_pending ?supplier_evidence_bundle_b - supplier_evidence_bundle_b)
    (assembler_entity_evidence_confirmed ?assembler_or_importer_entity - assembler_or_importer_entity)
    (submission_package_available ?submission_package - submission_package)
    (submission_package_prepared ?submission_package - submission_package)
    (package_contains_evidence_bundle_a ?submission_package - submission_package ?supplier_evidence_bundle_a - supplier_evidence_bundle_a)
    (package_contains_evidence_bundle_b ?submission_package - submission_package ?supplier_evidence_bundle_b - supplier_evidence_bundle_b)
    (package_contains_declaration_flag ?submission_package - submission_package)
    (package_contains_commercial_record_flag ?submission_package - submission_package)
    (package_contains_origin_statement ?submission_package - submission_package)
    (case_linked_producer ?origin_assessment_case - origin_assessment_case ?producer_entity - producer_entity)
    (case_linked_assembler ?origin_assessment_case - origin_assessment_case ?assembler_or_importer_entity - assembler_or_importer_entity)
    (case_associated_submission_package ?origin_assessment_case - origin_assessment_case ?submission_package - submission_package)
    (product_lineitem_spec_available ?product_line_item - product_line_item)
    (case_attached_lineitem_spec ?origin_assessment_case - origin_assessment_case ?product_line_item - product_line_item)
    (lineitem_spec_marked_processed ?product_line_item - product_line_item)
    (lineitem_spec_linked_package ?product_line_item - product_line_item ?submission_package - submission_package)
    (case_lineitem_documentation_ready ?origin_assessment_case - origin_assessment_case)
    (case_lineitem_reviewer_assigned ?origin_assessment_case - origin_assessment_case)

    (case_attached_preference_program ?origin_assessment_case - origin_assessment_case ?preference_program - preference_program)
    (preference_program_review_initiated ?origin_assessment_case - origin_assessment_case)
    (preference_program_reviewer_assigned ?origin_assessment_case - origin_assessment_case)
    (preference_program_reviewer_approved ?origin_assessment_case - origin_assessment_case)
    (legal_reference_available ?legal_rule_reference - legal_rule_reference)
    (case_linked_legal_reference ?origin_assessment_case - origin_assessment_case ?legal_rule_reference - legal_rule_reference)
    (regulatory_permit_available ?regulatory_permit - regulatory_permit)

    (assembler_entity_ready ?assembler_or_importer_entity - assembler_or_importer_entity)
    (case_record_locked_for_audit ?origin_assessment_case - origin_assessment_case)
  )
  (:action register_origin_determination_case
    :parameters (?trade_entity - trade_entity)
    :precondition
      (and
        (not
          (entity_registered ?trade_entity)
        )
        (not
          (entity_origin_determination_finalized ?trade_entity)
        )
      )
    :effect (entity_registered ?trade_entity)
  )
  (:action assign_broker_to_case
    :parameters (?trade_entity - trade_entity ?broker_service - broker_service)
    :precondition
      (and
        (entity_registered ?trade_entity)
        (not
          (broker_assignment_pending ?trade_entity)
        )
        (broker_available ?broker_service)
      )
    :effect
      (and
        (broker_assignment_pending ?trade_entity)
        (entity_assigned_broker ?trade_entity ?broker_service)
        (not
          (broker_available ?broker_service)
        )
      )
  )
  (:action attach_classification_reference_to_case
    :parameters (?trade_entity - trade_entity ?classification_reference - classification_reference)
    :precondition
      (and
        (entity_registered ?trade_entity)
        (broker_assignment_pending ?trade_entity)
        (classification_reference_available ?classification_reference)
      )
    :effect
      (and
        (entity_attached_classification ?trade_entity ?classification_reference)
        (not
          (classification_reference_available ?classification_reference)
        )
      )
  )
  (:action finalize_case_classification
    :parameters (?trade_entity - trade_entity ?classification_reference - classification_reference)
    :precondition
      (and
        (entity_registered ?trade_entity)
        (broker_assignment_pending ?trade_entity)
        (entity_attached_classification ?trade_entity ?classification_reference)
        (not
          (entity_has_classification ?trade_entity)
        )
      )
    :effect (entity_has_classification ?trade_entity)
  )
  (:action release_classification_reference_from_case
    :parameters (?trade_entity - trade_entity ?classification_reference - classification_reference)
    :precondition
      (and
        (entity_attached_classification ?trade_entity ?classification_reference)
      )
    :effect
      (and
        (classification_reference_available ?classification_reference)
        (not
          (entity_attached_classification ?trade_entity ?classification_reference)
        )
      )
  )
  (:action assign_reviewer_to_case
    :parameters (?trade_entity - trade_entity ?technical_reviewer - technical_reviewer)
    :precondition
      (and
        (entity_has_classification ?trade_entity)
        (reviewer_available ?technical_reviewer)
      )
    :effect
      (and
        (entity_assigned_reviewer ?trade_entity ?technical_reviewer)
        (not
          (reviewer_available ?technical_reviewer)
        )
      )
  )
  (:action unassign_reviewer_from_case
    :parameters (?trade_entity - trade_entity ?technical_reviewer - technical_reviewer)
    :precondition
      (and
        (entity_assigned_reviewer ?trade_entity ?technical_reviewer)
      )
    :effect
      (and
        (reviewer_available ?technical_reviewer)
        (not
          (entity_assigned_reviewer ?trade_entity ?technical_reviewer)
        )
      )
  )

          (case_attached_supplier_declaration ?origin_assessment_case ?supplier_declaration)
        )
      )

  (:action validate_evidence_bundle_a_for_producer
    :parameters (?producer_entity - producer_entity ?supplier_evidence_bundle_a - supplier_evidence_bundle_a ?classification_reference - classification_reference)
    :precondition
      (and
        (entity_has_classification ?producer_entity)
        (entity_attached_classification ?producer_entity ?classification_reference)
        (party_linked_evidence_bundle ?producer_entity ?supplier_evidence_bundle_a)
        (not
          (evidence_bundle_a_validated ?supplier_evidence_bundle_a)
        )
        (not
          (evidence_bundle_a_pending ?supplier_evidence_bundle_a)
        )
      )
    :effect (evidence_bundle_a_validated ?supplier_evidence_bundle_a)
  )
  (:action confirm_producer_evidence_collection
    :parameters (?producer_entity - producer_entity ?supplier_evidence_bundle_a - supplier_evidence_bundle_a ?technical_reviewer - technical_reviewer)
    :precondition
      (and
        (entity_has_classification ?producer_entity)
        (entity_assigned_reviewer ?producer_entity ?technical_reviewer)
        (party_linked_evidence_bundle ?producer_entity ?supplier_evidence_bundle_a)
        (evidence_bundle_a_validated ?supplier_evidence_bundle_a)
        (not
          (producer_entity_ready ?producer_entity)
        )
      )
    :effect
      (and
        (producer_entity_ready ?producer_entity)
        (producer_entity_evidence_confirmed ?producer_entity)
      )
  )
  (:action attach_evidence_document_to_producer_evidence_bundle
    :parameters (?producer_entity - producer_entity ?supplier_evidence_bundle_a - supplier_evidence_bundle_a ?evidence_document - evidence_document)
    :precondition
      (and

        (evidence_document_available ?evidence_document)
        (not
          (producer_entity_ready ?producer_entity)
        )
      )
    :effect
      (and
        (evidence_bundle_a_pending ?supplier_evidence_bundle_a)
        (producer_entity_ready ?producer_entity)
        (entity_lineitem_has_document ?producer_entity ?evidence_document)
        (not
          (evidence_document_available ?evidence_document)
        )
      )
  )

        (not
          (producer_entity_evidence_confirmed ?producer_entity)
