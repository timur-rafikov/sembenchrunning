    (alternative_has_replacement_lot ?alternative_warehouse - alternative_warehouse ?replacement_lot - replacement_lot)

    (agent_operates_at_alternative ?fulfillment_agent - fulfillment_agent ?alternative_warehouse - alternative_warehouse)

    (evidence_approved ?fulfillment_agent - fulfillment_agent)

  (:action record_fulfillment_case
    :parameters (?fulfillment_case - fulfillment_case)
    :precondition
      (and
        (not
          (entity_recorded ?fulfillment_case)
        )
        (not
          (operation_restored ?fulfillment_case)
        )
      )
    :effect (entity_recorded ?fulfillment_case)
  )
  (:action allocate_transport_resource_to_case
    :parameters (?fulfillment_case - fulfillment_case ?transport_resource - transport_resource)
    :precondition
      (and
        (entity_recorded ?fulfillment_case)
        (not
          (transport_allocated ?fulfillment_case)
        )
        (transport_available ?transport_resource)
      )
    :effect
      (and
        (transport_allocated ?fulfillment_case)
        (entity_transport_candidate ?fulfillment_case ?transport_resource)
        (not
          (transport_available ?transport_resource)
        )
      )
  )
  (:action associate_sku_with_case
    :parameters (?fulfillment_case - fulfillment_case ?item_sku - item_sku)
    :precondition
      (and
        (entity_recorded ?fulfillment_case)
        (transport_allocated ?fulfillment_case)
        (sku_available ?item_sku)
      )
    :effect
      (and
        (entity_has_sku ?fulfillment_case ?item_sku)
        (not
          (sku_available ?item_sku)
        )
      )
  )
  (:action validate_case_item_association
    :parameters (?fulfillment_case - fulfillment_case ?item_sku - item_sku)
    :precondition
      (and
        (entity_recorded ?fulfillment_case)
        (transport_allocated ?fulfillment_case)
        (entity_has_sku ?fulfillment_case ?item_sku)
        (not
          (entity_validated ?fulfillment_case)
        )
      )
    :effect (entity_validated ?fulfillment_case)
  )
  (:action release_sku_assignment
    :parameters (?fulfillment_case - fulfillment_case ?item_sku - item_sku)
    :precondition
      (and
        (entity_has_sku ?fulfillment_case ?item_sku)
      )
    :effect
      (and
        (sku_available ?item_sku)
        (not
          (entity_has_sku ?fulfillment_case ?item_sku)
        )
      )
  )
  (:action assign_staff_to_case
    :parameters (?fulfillment_case - fulfillment_case ?warehouse_staff - warehouse_staff)
    :precondition
      (and
        (entity_validated ?fulfillment_case)
        (staff_available ?warehouse_staff)
      )
    :effect
      (and
        (entity_has_assigned_staff ?fulfillment_case ?warehouse_staff)
        (not
          (staff_available ?warehouse_staff)
        )
      )
  )
  (:action release_staff_assignment
    :parameters (?fulfillment_case - fulfillment_case ?warehouse_staff - warehouse_staff)
    :precondition
      (and
        (entity_has_assigned_staff ?fulfillment_case ?warehouse_staff)
      )
    :effect
      (and
        (staff_available ?warehouse_staff)
        (not
          (entity_has_assigned_staff ?fulfillment_case ?warehouse_staff)
        )
      )
  )
  (:action reserve_repair_toolkit_for_agent
    :parameters (?fulfillment_agent - fulfillment_agent ?repair_toolkit - repair_toolkit)
    :precondition
      (and
        (entity_validated ?fulfillment_agent)

      )

        (agent_has_repair_toolkit ?fulfillment_agent ?repair_toolkit)

      (and
        (agent_has_repair_toolkit ?fulfillment_agent ?repair_toolkit)
      )

  (:action assign_external_approval_to_agent

        (agent_has_external_approval ?fulfillment_agent ?external_approval)

          (external_approval_available ?external_approval)
        )
      )
  )
  (:action release_external_approval_from_agent
    :parameters (?fulfillment_agent - fulfillment_agent ?external_approval - external_approval)
    :precondition
      (and
        (agent_has_external_approval ?fulfillment_agent ?external_approval)
      )
    :effect
      (and
        (external_approval_available ?external_approval)
        (not
          (agent_has_external_approval ?fulfillment_agent ?external_approval)
        )
      )
  )
  (:action evaluate_origin_leg_for_sourcing
    :parameters (?origin_warehouse - origin_warehouse ?origin_route_leg - origin_route_leg ?item_sku - item_sku)
    :precondition
      (and
        (entity_validated ?origin_warehouse)
        (entity_has_sku ?origin_warehouse ?item_sku)
        (origin_has_route_leg ?origin_warehouse ?origin_route_leg)
        (not
          (origin_leg_confirmed ?origin_route_leg)
        )
        (not
          (origin_leg_reserved ?origin_route_leg)
        )
      )
    :effect (origin_leg_confirmed ?origin_route_leg)
  )
  (:action mark_origin_warehouse_ready
    :parameters (?origin_warehouse - origin_warehouse ?origin_route_leg - origin_route_leg ?warehouse_staff - warehouse_staff)
    :precondition
      (and
        (entity_validated ?origin_warehouse)
        (entity_has_assigned_staff ?origin_warehouse ?warehouse_staff)
        (origin_has_route_leg ?origin_warehouse ?origin_route_leg)
        (origin_leg_confirmed ?origin_route_leg)
        (not
          (origin_ready_flag ?origin_warehouse)
        )
      )
    :effect
      (and
        (origin_ready_flag ?origin_warehouse)
        (origin_warehouse_ready ?origin_warehouse)
      )
  )
  (:action reserve_replacement_lot_for_origin
    :parameters (?origin_warehouse - origin_warehouse ?origin_route_leg - origin_route_leg ?replacement_lot - replacement_lot)
    :precondition
      (and
        (entity_validated ?origin_warehouse)
        (origin_has_route_leg ?origin_warehouse ?origin_route_leg)
        (replacement_lot_available ?replacement_lot)
        (not
          (origin_ready_flag ?origin_warehouse)
        )
      )
    :effect
      (and
        (origin_leg_reserved ?origin_route_leg)
        (origin_ready_flag ?origin_warehouse)
        (origin_has_replacement_lot ?origin_warehouse ?replacement_lot)
        (not
          (replacement_lot_available ?replacement_lot)
        )
      )
  )
  (:action fulfill_origin_restock_from_replacement

        (entity_validated ?origin_warehouse)
        (entity_has_sku ?origin_warehouse ?item_sku)
        (origin_has_route_leg ?origin_warehouse ?origin_route_leg)
        (origin_leg_reserved ?origin_route_leg)
        (origin_has_replacement_lot ?origin_warehouse ?replacement_lot)
        (not
          (origin_warehouse_ready ?origin_warehouse)
        )
      )
    :effect
      (and
        (origin_leg_confirmed ?origin_route_leg)
        (origin_warehouse_ready ?origin_warehouse)
        (replacement_lot_available ?replacement_lot)
        (not
          (origin_has_replacement_lot ?origin_warehouse ?replacement_lot)
        )
      )
  )

    :parameters (?alternative_warehouse - alternative_warehouse ?destination_route_leg - destination_route_leg ?item_sku - item_sku)
    :precondition
      (and
        (entity_validated ?alternative_warehouse)
        (entity_has_sku ?alternative_warehouse ?item_sku)
        (alternative_has_route_leg ?alternative_warehouse ?destination_route_leg)
        (not
          (destination_leg_confirmed ?destination_route_leg)
        )
        (not
          (destination_leg_reserved ?destination_route_leg)
        )

  (:action reserve_replacement_lot_for_alternative
    :parameters (?alternative_warehouse - alternative_warehouse ?destination_route_leg - destination_route_leg ?replacement_lot - replacement_lot)
    :precondition
      (and
        (entity_validated ?alternative_warehouse)
        (alternative_has_route_leg ?alternative_warehouse ?destination_route_leg)
        (replacement_lot_available ?replacement_lot)
        (not
          (alternative_ready_flag ?alternative_warehouse)
        )
      )
    :effect
      (and
        (destination_leg_reserved ?destination_route_leg)
        (alternative_ready_flag ?alternative_warehouse)

          (replacement_lot_available ?replacement_lot)

  (:action fulfill_alternative_restock_from_replacement
    :parameters (?alternative_warehouse - alternative_warehouse ?destination_route_leg - destination_route_leg ?item_sku - item_sku ?replacement_lot - replacement_lot)

      (and

        (destination_leg_reserved ?destination_route_leg)
        (alternative_has_replacement_lot ?alternative_warehouse ?replacement_lot)
        (not
          (alternative_warehouse_ready ?alternative_warehouse)
        )
      )
    :effect
      (and
        (destination_leg_confirmed ?destination_route_leg)
        (alternative_warehouse_ready ?alternative_warehouse)
        (replacement_lot_available ?replacement_lot)
        (not
          (alternative_has_replacement_lot ?alternative_warehouse ?replacement_lot)
        )

  )
  (:action create_shipment_proposal
    :parameters (?origin_warehouse - origin_warehouse ?alternative_warehouse - alternative_warehouse ?origin_route_leg - origin_route_leg ?destination_route_leg - destination_route_leg ?shipment_proposal - shipment_proposal)
    :precondition
      (and
        (origin_ready_flag ?origin_warehouse)
        (alternative_ready_flag ?alternative_warehouse)
        (origin_has_route_leg ?origin_warehouse ?origin_route_leg)
        (alternative_has_route_leg ?alternative_warehouse ?destination_route_leg)
        (origin_leg_confirmed ?origin_route_leg)
        (destination_leg_confirmed ?destination_route_leg)
        (origin_warehouse_ready ?origin_warehouse)
        (alternative_warehouse_ready ?alternative_warehouse)
        (shipment_proposal_created ?shipment_proposal)
      )
    :effect
      (and
        (shipment_proposal_locked ?shipment_proposal)
        (proposal_has_origin_leg ?shipment_proposal ?origin_route_leg)
        (proposal_has_destination_leg ?shipment_proposal ?destination_route_leg)
        (not
          (shipment_proposal_created ?shipment_proposal)
        )
      )
  )
  (:action create_shipment_proposal_with_origin_confirmation

    :precondition
      (and
        (origin_ready_flag ?origin_warehouse)
        (alternative_ready_flag ?alternative_warehouse)
        (origin_has_route_leg ?origin_warehouse ?origin_route_leg)
        (alternative_has_route_leg ?alternative_warehouse ?destination_route_leg)
        (origin_leg_reserved ?origin_route_leg)
        (destination_leg_confirmed ?destination_route_leg)
        (not
          (origin_warehouse_ready ?origin_warehouse)
        )
        (alternative_warehouse_ready ?alternative_warehouse)
        (shipment_proposal_created ?shipment_proposal)
      )
    :effect
      (and
        (shipment_proposal_locked ?shipment_proposal)
        (proposal_has_origin_leg ?shipment_proposal ?origin_route_leg)
        (proposal_has_destination_leg ?shipment_proposal ?destination_route_leg)
        (proposal_origin_confirmed ?shipment_proposal)
        (not
          (shipment_proposal_created ?shipment_proposal)
        )
      )
  )
  (:action create_shipment_proposal_with_destination_confirmation
    :parameters (?origin_warehouse - origin_warehouse ?alternative_warehouse - alternative_warehouse ?origin_route_leg - origin_route_leg ?destination_route_leg - destination_route_leg ?shipment_proposal - shipment_proposal)
    :precondition
      (and
        (origin_ready_flag ?origin_warehouse)
        (alternative_ready_flag ?alternative_warehouse)
        (origin_has_route_leg ?origin_warehouse ?origin_route_leg)
        (alternative_has_route_leg ?alternative_warehouse ?destination_route_leg)
        (origin_leg_confirmed ?origin_route_leg)
        (destination_leg_reserved ?destination_route_leg)
        (origin_warehouse_ready ?origin_warehouse)
        (not
          (alternative_warehouse_ready ?alternative_warehouse)
        )
        (shipment_proposal_created ?shipment_proposal)
      )
    :effect
      (and
        (shipment_proposal_locked ?shipment_proposal)
        (proposal_has_origin_leg ?shipment_proposal ?origin_route_leg)
        (proposal_has_destination_leg ?shipment_proposal ?destination_route_leg)
        (proposal_destination_confirmed ?shipment_proposal)
        (not
          (shipment_proposal_created ?shipment_proposal)
        )
      )
  )
  (:action create_shipment_proposal_with_full_confirmation
    :parameters (?origin_warehouse - origin_warehouse ?alternative_warehouse - alternative_warehouse ?origin_route_leg - origin_route_leg ?destination_route_leg - destination_route_leg ?shipment_proposal - shipment_proposal)
    :precondition

        (origin_ready_flag ?origin_warehouse)

        (shipment_proposal_locked ?shipment_proposal)

        (proposal_destination_confirmed ?shipment_proposal)

          (shipment_proposal_created ?shipment_proposal)

      )
  )

      (and
        (shipment_proposal_locked ?shipment_proposal)

    :precondition
      (and

        (agent_has_packing_unit ?fulfillment_agent ?packing_unit)
        (packing_unit_available ?packing_unit)

        (not
          (packing_unit_available ?packing_unit)
        )
      )
  )

        (packing_unit_assigned_to_proposal ?packing_unit ?shipment_proposal)

        (not
          (proposal_origin_confirmed ?shipment_proposal)
        )
        (not
          (agent_ready_for_tooling ?fulfillment_agent)
        )
      )
    :effect (agent_ready_for_tooling ?fulfillment_agent)
  )
  (:action assign_expedite_option_to_agent
    :parameters (?fulfillment_agent - fulfillment_agent ?expedite_option - expedite_option)
    :precondition
      (and
        (entity_validated ?fulfillment_agent)
        (expedite_option_available ?expedite_option)
        (not
          (agent_expedite_flag ?fulfillment_agent)
        )
      )
    :effect
      (and
        (agent_expedite_flag ?fulfillment_agent)
        (agent_expedite_assignment ?fulfillment_agent ?expedite_option)
        (not
          (expedite_option_available ?expedite_option)
        )
      )
  )
  (:action confirm_expedite_and_prepare_agent
    :parameters (?fulfillment_agent - fulfillment_agent ?packing_unit - packing_unit ?shipment_proposal - shipment_proposal ?item_sku - item_sku ?expedite_option - expedite_option)
    :precondition
      (and
        (entity_validated ?fulfillment_agent)
        (agent_has_packing_unit ?fulfillment_agent ?packing_unit)
        (packing_unit_reserved ?packing_unit)
        (packing_unit_assigned_to_proposal ?packing_unit ?shipment_proposal)
        (entity_has_sku ?fulfillment_agent ?item_sku)
        (proposal_origin_confirmed ?shipment_proposal)
        (agent_expedite_flag ?fulfillment_agent)
