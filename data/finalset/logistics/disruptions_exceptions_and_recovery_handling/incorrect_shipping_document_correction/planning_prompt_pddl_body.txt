  (:action open_shipping_document_case
    :parameters (?shipping_document_case - shipping_document_case)
    :precondition
      (and
        (not
          (entity_opened ?shipping_document_case)
        )
        (not
          (service_restored ?shipping_document_case)
        )
      )
    :effect (entity_opened ?shipping_document_case)
  )
  (:action assign_carrier_agent_to_case
    :parameters (?shipping_document_case - shipping_document_case ?carrier_agent - carrier_agent)
    :precondition
      (and
        (entity_opened ?shipping_document_case)
        (not
          (agent_assigned_to_case ?shipping_document_case)
        )
        (carrier_agent_available ?carrier_agent)
      )
    :effect
      (and
        (agent_assigned_to_case ?shipping_document_case)
        (entity_assigned_to_agent ?shipping_document_case ?carrier_agent)
        (not
          (carrier_agent_available ?carrier_agent)
        )
      )
  )
  (:action start_document_verification
    :parameters (?shipping_document_case - shipping_document_case ?document_verifier - document_verifier)
    :precondition
      (and
        (entity_opened ?shipping_document_case)
        (agent_assigned_to_case ?shipping_document_case)
        (verifier_available ?document_verifier)
      )
    :effect
      (and
        (entity_verifier_assigned ?shipping_document_case ?document_verifier)
        (not
          (verifier_available ?document_verifier)
        )
      )
  )
  (:action confirm_document_verification
    :parameters (?shipping_document_case - shipping_document_case ?document_verifier - document_verifier)

      (and
        (entity_opened ?shipping_document_case)
        (agent_assigned_to_case ?shipping_document_case)
        (entity_verifier_assigned ?shipping_document_case ?document_verifier)
        (not
          (entity_verified ?shipping_document_case)
        )
      )
    :effect (entity_verified ?shipping_document_case)
  )
  (:action rollback_document_verification
    :parameters (?shipping_document_case - shipping_document_case ?document_verifier - document_verifier)

        (entity_verifier_assigned ?shipping_document_case ?document_verifier)

    :effect

        (verifier_available ?document_verifier)

          (entity_verifier_assigned ?shipping_document_case ?document_verifier)
        )
      )
  )
