## Planning domain (PDDL)

(define (domain cross_border_transit_leg_planning_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_category - object cargo_category - object node_category - object planning_entity_root - object planning_entity - planning_entity_root transport_asset - resource_category time_slot - resource_category terminal_gate - resource_category routing_option - resource_category handling_profile - resource_category border_document - resource_category regulatory_certificate - resource_category inspection_manifest - resource_category cargo_unit - cargo_category container_asset - cargo_category transport_permit - cargo_category origin_transfer_point - node_category destination_transfer_point - node_category transit_leg_plan - node_category leg_category - planning_entity shipment_category - planning_entity origin_leg - leg_category destination_leg - leg_category shipment_order - shipment_category)
  (:predicates
    (planning_entity_initialized ?planning_entity - planning_entity)
    (planning_entity_confirmed ?planning_entity - planning_entity)
    (planning_entity_asset_reserved ?planning_entity - planning_entity)
    (planning_entity_dispatched ?planning_entity - planning_entity)
    (planning_entity_validated ?planning_entity - planning_entity)
    (planning_entity_cleared ?planning_entity - planning_entity)
    (transport_asset_available ?transport_asset - transport_asset)
    (asset_assigned_to_entity ?planning_entity - planning_entity ?transport_asset - transport_asset)
    (time_slot_available ?time_slot - time_slot)
    (entity_assigned_time_slot ?planning_entity - planning_entity ?time_slot - time_slot)
    (terminal_gate_available ?terminal_gate - terminal_gate)
    (entity_assigned_gate ?planning_entity - planning_entity ?terminal_gate - terminal_gate)
    (cargo_unit_available ?cargo_unit - cargo_unit)
    (origin_leg_loaded_with_cargo ?origin_leg - origin_leg ?cargo_unit - cargo_unit)
    (destination_leg_loaded_with_cargo ?destination_leg - destination_leg ?cargo_unit - cargo_unit)
    (origin_leg_at_transfer_point ?origin_leg - origin_leg ?origin_transfer_point - origin_transfer_point)
    (origin_transfer_point_reserved ?origin_transfer_point - origin_transfer_point)
    (origin_transfer_point_staged ?origin_transfer_point - origin_transfer_point)
    (origin_leg_ready_for_transit ?origin_leg - origin_leg)
    (destination_leg_at_transfer_point ?destination_leg - destination_leg ?destination_transfer_point - destination_transfer_point)
    (destination_transfer_point_reserved ?destination_transfer_point - destination_transfer_point)
    (destination_transfer_point_staged ?destination_transfer_point - destination_transfer_point)
    (destination_leg_ready_for_transit ?destination_leg - destination_leg)
    (transit_leg_plan_available ?transit_leg_plan - transit_leg_plan)
    (transit_leg_plan_committed ?transit_leg_plan - transit_leg_plan)
    (transit_leg_plan_origin_point ?transit_leg_plan - transit_leg_plan ?origin_transfer_point - origin_transfer_point)
    (transit_leg_plan_destination_point ?transit_leg_plan - transit_leg_plan ?destination_transfer_point - destination_transfer_point)
    (transit_leg_plan_origin_staged ?transit_leg_plan - transit_leg_plan)
    (transit_leg_plan_destination_staged ?transit_leg_plan - transit_leg_plan)
    (transit_leg_plan_activated ?transit_leg_plan - transit_leg_plan)
    (order_origin_leg ?shipment_order - shipment_order ?origin_leg - origin_leg)
    (order_destination_leg ?shipment_order - shipment_order ?destination_leg - destination_leg)
    (order_assigned_to_transit_leg ?shipment_order - shipment_order ?transit_leg_plan - transit_leg_plan)
    (container_available ?container_asset - container_asset)
    (order_has_container ?shipment_order - shipment_order ?container_asset - container_asset)
    (container_committed_to_order ?container_asset - container_asset)
    (container_assigned_to_transit_leg ?container_asset - container_asset ?transit_leg_plan - transit_leg_plan)
    (order_activated ?shipment_order - shipment_order)
    (order_certificates_linked ?shipment_order - shipment_order)
    (order_inspection_applied ?shipment_order - shipment_order)
    (order_routing_selected ?shipment_order - shipment_order)
    (order_routing_locked ?shipment_order - shipment_order)
    (order_compliance_confirmed ?shipment_order - shipment_order)
    (order_final_checks_passed ?shipment_order - shipment_order)
    (transport_permit_available ?transport_permit - transport_permit)
    (order_assigned_permit ?shipment_order - shipment_order ?transport_permit - transport_permit)
    (order_permit_assigned ?shipment_order - shipment_order)
    (order_permit_check_passed ?shipment_order - shipment_order)
    (order_permit_validated ?shipment_order - shipment_order)
    (routing_option_available ?routing_option - routing_option)
    (order_assigned_routing_option ?shipment_order - shipment_order ?routing_option - routing_option)
    (handling_profile_available ?handling_profile - handling_profile)
    (order_assigned_handling_profile ?shipment_order - shipment_order ?handling_profile - handling_profile)
    (regulatory_certificate_available ?regulatory_certificate - regulatory_certificate)
    (order_has_certificate ?shipment_order - shipment_order ?regulatory_certificate - regulatory_certificate)
    (inspection_manifest_available ?inspection_manifest - inspection_manifest)
    (order_has_inspection_manifest ?shipment_order - shipment_order ?inspection_manifest - inspection_manifest)
    (border_document_available ?border_document - border_document)
    (planning_entity_has_border_document ?planning_entity - planning_entity ?border_document - border_document)
    (origin_leg_loaded ?origin_leg - origin_leg)
    (destination_leg_loaded ?destination_leg - destination_leg)
    (order_finalized ?shipment_order - shipment_order)
  )
  (:action initialize_planning_entity
    :parameters (?planning_entity - planning_entity)
    :precondition
      (and
        (not
          (planning_entity_initialized ?planning_entity)
        )
        (not
          (planning_entity_dispatched ?planning_entity)
        )
      )
    :effect (planning_entity_initialized ?planning_entity)
  )
  (:action assign_transport_asset_to_entity
    :parameters (?planning_entity - planning_entity ?transport_asset - transport_asset)
    :precondition
      (and
        (planning_entity_initialized ?planning_entity)
        (not
          (planning_entity_asset_reserved ?planning_entity)
        )
        (transport_asset_available ?transport_asset)
      )
    :effect
      (and
        (planning_entity_asset_reserved ?planning_entity)
        (asset_assigned_to_entity ?planning_entity ?transport_asset)
        (not
          (transport_asset_available ?transport_asset)
        )
      )
  )
  (:action reserve_time_slot_for_entity
    :parameters (?planning_entity - planning_entity ?time_slot - time_slot)
    :precondition
      (and
        (planning_entity_initialized ?planning_entity)
        (planning_entity_asset_reserved ?planning_entity)
        (time_slot_available ?time_slot)
      )
    :effect
      (and
        (entity_assigned_time_slot ?planning_entity ?time_slot)
        (not
          (time_slot_available ?time_slot)
        )
      )
  )
  (:action confirm_entity_time_slot
    :parameters (?planning_entity - planning_entity ?time_slot - time_slot)
    :precondition
      (and
        (planning_entity_initialized ?planning_entity)
        (planning_entity_asset_reserved ?planning_entity)
        (entity_assigned_time_slot ?planning_entity ?time_slot)
        (not
          (planning_entity_confirmed ?planning_entity)
        )
      )
    :effect (planning_entity_confirmed ?planning_entity)
  )
  (:action release_time_slot_from_entity
    :parameters (?planning_entity - planning_entity ?time_slot - time_slot)
    :precondition
      (and
        (entity_assigned_time_slot ?planning_entity ?time_slot)
      )
    :effect
      (and
        (time_slot_available ?time_slot)
        (not
          (entity_assigned_time_slot ?planning_entity ?time_slot)
        )
      )
  )
  (:action assign_gate_to_entity
    :parameters (?planning_entity - planning_entity ?terminal_gate - terminal_gate)
    :precondition
      (and
        (planning_entity_confirmed ?planning_entity)
        (terminal_gate_available ?terminal_gate)
      )
    :effect
      (and
        (entity_assigned_gate ?planning_entity ?terminal_gate)
        (not
          (terminal_gate_available ?terminal_gate)
        )
      )
  )
  (:action release_gate_from_entity
    :parameters (?planning_entity - planning_entity ?terminal_gate - terminal_gate)
    :precondition
      (and
        (entity_assigned_gate ?planning_entity ?terminal_gate)
      )
    :effect
      (and
        (terminal_gate_available ?terminal_gate)
        (not
          (entity_assigned_gate ?planning_entity ?terminal_gate)
        )
      )
  )
  (:action attach_regulatory_certificate_to_order
    :parameters (?shipment_order - shipment_order ?regulatory_certificate - regulatory_certificate)
    :precondition
      (and
        (planning_entity_confirmed ?shipment_order)
        (regulatory_certificate_available ?regulatory_certificate)
      )
    :effect
      (and
        (order_has_certificate ?shipment_order ?regulatory_certificate)
        (not
          (regulatory_certificate_available ?regulatory_certificate)
        )
      )
  )
  (:action detach_regulatory_certificate_from_order
    :parameters (?shipment_order - shipment_order ?regulatory_certificate - regulatory_certificate)
    :precondition
      (and
        (order_has_certificate ?shipment_order ?regulatory_certificate)
      )
    :effect
      (and
        (regulatory_certificate_available ?regulatory_certificate)
        (not
          (order_has_certificate ?shipment_order ?regulatory_certificate)
        )
      )
  )
  (:action attach_inspection_manifest_to_order
    :parameters (?shipment_order - shipment_order ?inspection_manifest - inspection_manifest)
    :precondition
      (and
        (planning_entity_confirmed ?shipment_order)
        (inspection_manifest_available ?inspection_manifest)
      )
    :effect
      (and
        (order_has_inspection_manifest ?shipment_order ?inspection_manifest)
        (not
          (inspection_manifest_available ?inspection_manifest)
        )
      )
  )
  (:action detach_inspection_manifest_from_order
    :parameters (?shipment_order - shipment_order ?inspection_manifest - inspection_manifest)
    :precondition
      (and
        (order_has_inspection_manifest ?shipment_order ?inspection_manifest)
      )
    :effect
      (and
        (inspection_manifest_available ?inspection_manifest)
        (not
          (order_has_inspection_manifest ?shipment_order ?inspection_manifest)
        )
      )
  )
  (:action prepare_origin_transfer_point_for_leg
    :parameters (?origin_leg - origin_leg ?origin_transfer_point - origin_transfer_point ?time_slot - time_slot)
    :precondition
      (and
        (planning_entity_confirmed ?origin_leg)
        (entity_assigned_time_slot ?origin_leg ?time_slot)
        (origin_leg_at_transfer_point ?origin_leg ?origin_transfer_point)
        (not
          (origin_transfer_point_reserved ?origin_transfer_point)
        )
        (not
          (origin_transfer_point_staged ?origin_transfer_point)
        )
      )
    :effect (origin_transfer_point_reserved ?origin_transfer_point)
  )
  (:action assign_gate_and_mark_origin_leg_loaded
    :parameters (?origin_leg - origin_leg ?origin_transfer_point - origin_transfer_point ?terminal_gate - terminal_gate)
    :precondition
      (and
        (planning_entity_confirmed ?origin_leg)
        (entity_assigned_gate ?origin_leg ?terminal_gate)
        (origin_leg_at_transfer_point ?origin_leg ?origin_transfer_point)
        (origin_transfer_point_reserved ?origin_transfer_point)
        (not
          (origin_leg_loaded ?origin_leg)
        )
      )
    :effect
      (and
        (origin_leg_loaded ?origin_leg)
        (origin_leg_ready_for_transit ?origin_leg)
      )
  )
  (:action stage_cargo_on_origin_leg
    :parameters (?origin_leg - origin_leg ?origin_transfer_point - origin_transfer_point ?cargo_unit - cargo_unit)
    :precondition
      (and
        (planning_entity_confirmed ?origin_leg)
        (origin_leg_at_transfer_point ?origin_leg ?origin_transfer_point)
        (cargo_unit_available ?cargo_unit)
        (not
          (origin_leg_loaded ?origin_leg)
        )
      )
    :effect
      (and
        (origin_transfer_point_staged ?origin_transfer_point)
        (origin_leg_loaded ?origin_leg)
        (origin_leg_loaded_with_cargo ?origin_leg ?cargo_unit)
        (not
          (cargo_unit_available ?cargo_unit)
        )
      )
  )
  (:action complete_origin_leg_cargo_transfer
    :parameters (?origin_leg - origin_leg ?origin_transfer_point - origin_transfer_point ?time_slot - time_slot ?cargo_unit - cargo_unit)
    :precondition
      (and
        (planning_entity_confirmed ?origin_leg)
        (entity_assigned_time_slot ?origin_leg ?time_slot)
        (origin_leg_at_transfer_point ?origin_leg ?origin_transfer_point)
        (origin_transfer_point_staged ?origin_transfer_point)
        (origin_leg_loaded_with_cargo ?origin_leg ?cargo_unit)
        (not
          (origin_leg_ready_for_transit ?origin_leg)
        )
      )
    :effect
      (and
        (origin_transfer_point_reserved ?origin_transfer_point)
        (origin_leg_ready_for_transit ?origin_leg)
        (cargo_unit_available ?cargo_unit)
        (not
          (origin_leg_loaded_with_cargo ?origin_leg ?cargo_unit)
        )
      )
  )
  (:action prepare_destination_transfer_point_for_leg
    :parameters (?destination_leg - destination_leg ?destination_transfer_point - destination_transfer_point ?time_slot - time_slot)
    :precondition
      (and
        (planning_entity_confirmed ?destination_leg)
        (entity_assigned_time_slot ?destination_leg ?time_slot)
        (destination_leg_at_transfer_point ?destination_leg ?destination_transfer_point)
        (not
          (destination_transfer_point_reserved ?destination_transfer_point)
        )
        (not
          (destination_transfer_point_staged ?destination_transfer_point)
        )
      )
    :effect (destination_transfer_point_reserved ?destination_transfer_point)
  )
  (:action assign_gate_and_mark_destination_leg_loaded
    :parameters (?destination_leg - destination_leg ?destination_transfer_point - destination_transfer_point ?terminal_gate - terminal_gate)
    :precondition
      (and
        (planning_entity_confirmed ?destination_leg)
        (entity_assigned_gate ?destination_leg ?terminal_gate)
        (destination_leg_at_transfer_point ?destination_leg ?destination_transfer_point)
        (destination_transfer_point_reserved ?destination_transfer_point)
        (not
          (destination_leg_loaded ?destination_leg)
        )
      )
    :effect
      (and
        (destination_leg_loaded ?destination_leg)
        (destination_leg_ready_for_transit ?destination_leg)
      )
  )
  (:action stage_cargo_on_destination_leg
    :parameters (?destination_leg - destination_leg ?destination_transfer_point - destination_transfer_point ?cargo_unit - cargo_unit)
    :precondition
      (and
        (planning_entity_confirmed ?destination_leg)
        (destination_leg_at_transfer_point ?destination_leg ?destination_transfer_point)
        (cargo_unit_available ?cargo_unit)
        (not
          (destination_leg_loaded ?destination_leg)
        )
      )
    :effect
      (and
        (destination_transfer_point_staged ?destination_transfer_point)
        (destination_leg_loaded ?destination_leg)
        (destination_leg_loaded_with_cargo ?destination_leg ?cargo_unit)
        (not
          (cargo_unit_available ?cargo_unit)
        )
      )
  )
  (:action complete_destination_leg_cargo_transfer
    :parameters (?destination_leg - destination_leg ?destination_transfer_point - destination_transfer_point ?time_slot - time_slot ?cargo_unit - cargo_unit)
    :precondition
      (and
        (planning_entity_confirmed ?destination_leg)
        (entity_assigned_time_slot ?destination_leg ?time_slot)
        (destination_leg_at_transfer_point ?destination_leg ?destination_transfer_point)
        (destination_transfer_point_staged ?destination_transfer_point)
        (destination_leg_loaded_with_cargo ?destination_leg ?cargo_unit)
        (not
          (destination_leg_ready_for_transit ?destination_leg)
        )
      )
    :effect
      (and
        (destination_transfer_point_reserved ?destination_transfer_point)
        (destination_leg_ready_for_transit ?destination_leg)
        (cargo_unit_available ?cargo_unit)
        (not
          (destination_leg_loaded_with_cargo ?destination_leg ?cargo_unit)
        )
      )
  )
  (:action assemble_transit_leg_plan
    :parameters (?origin_leg - origin_leg ?destination_leg - destination_leg ?origin_transfer_point - origin_transfer_point ?destination_transfer_point - destination_transfer_point ?transit_leg_plan - transit_leg_plan)
    :precondition
      (and
        (origin_leg_loaded ?origin_leg)
        (destination_leg_loaded ?destination_leg)
        (origin_leg_at_transfer_point ?origin_leg ?origin_transfer_point)
        (destination_leg_at_transfer_point ?destination_leg ?destination_transfer_point)
        (origin_transfer_point_reserved ?origin_transfer_point)
        (destination_transfer_point_reserved ?destination_transfer_point)
        (origin_leg_ready_for_transit ?origin_leg)
        (destination_leg_ready_for_transit ?destination_leg)
        (transit_leg_plan_available ?transit_leg_plan)
      )
    :effect
      (and
        (transit_leg_plan_committed ?transit_leg_plan)
        (transit_leg_plan_origin_point ?transit_leg_plan ?origin_transfer_point)
        (transit_leg_plan_destination_point ?transit_leg_plan ?destination_transfer_point)
        (not
          (transit_leg_plan_available ?transit_leg_plan)
        )
      )
  )
  (:action assemble_transit_leg_plan_origin_staged
    :parameters (?origin_leg - origin_leg ?destination_leg - destination_leg ?origin_transfer_point - origin_transfer_point ?destination_transfer_point - destination_transfer_point ?transit_leg_plan - transit_leg_plan)
    :precondition
      (and
        (origin_leg_loaded ?origin_leg)
        (destination_leg_loaded ?destination_leg)
        (origin_leg_at_transfer_point ?origin_leg ?origin_transfer_point)
        (destination_leg_at_transfer_point ?destination_leg ?destination_transfer_point)
        (origin_transfer_point_staged ?origin_transfer_point)
        (destination_transfer_point_reserved ?destination_transfer_point)
        (not
          (origin_leg_ready_for_transit ?origin_leg)
        )
        (destination_leg_ready_for_transit ?destination_leg)
        (transit_leg_plan_available ?transit_leg_plan)
      )
    :effect
      (and
        (transit_leg_plan_committed ?transit_leg_plan)
        (transit_leg_plan_origin_point ?transit_leg_plan ?origin_transfer_point)
        (transit_leg_plan_destination_point ?transit_leg_plan ?destination_transfer_point)
        (transit_leg_plan_origin_staged ?transit_leg_plan)
        (not
          (transit_leg_plan_available ?transit_leg_plan)
        )
      )
  )
  (:action assemble_transit_leg_plan_destination_staged
    :parameters (?origin_leg - origin_leg ?destination_leg - destination_leg ?origin_transfer_point - origin_transfer_point ?destination_transfer_point - destination_transfer_point ?transit_leg_plan - transit_leg_plan)
    :precondition
      (and
        (origin_leg_loaded ?origin_leg)
        (destination_leg_loaded ?destination_leg)
        (origin_leg_at_transfer_point ?origin_leg ?origin_transfer_point)
        (destination_leg_at_transfer_point ?destination_leg ?destination_transfer_point)
        (origin_transfer_point_reserved ?origin_transfer_point)
        (destination_transfer_point_staged ?destination_transfer_point)
        (origin_leg_ready_for_transit ?origin_leg)
        (not
          (destination_leg_ready_for_transit ?destination_leg)
        )
        (transit_leg_plan_available ?transit_leg_plan)
      )
    :effect
      (and
        (transit_leg_plan_committed ?transit_leg_plan)
        (transit_leg_plan_origin_point ?transit_leg_plan ?origin_transfer_point)
        (transit_leg_plan_destination_point ?transit_leg_plan ?destination_transfer_point)
        (transit_leg_plan_destination_staged ?transit_leg_plan)
        (not
          (transit_leg_plan_available ?transit_leg_plan)
        )
      )
  )
  (:action assemble_transit_leg_plan_both_staged
    :parameters (?origin_leg - origin_leg ?destination_leg - destination_leg ?origin_transfer_point - origin_transfer_point ?destination_transfer_point - destination_transfer_point ?transit_leg_plan - transit_leg_plan)
    :precondition
      (and
        (origin_leg_loaded ?origin_leg)
        (destination_leg_loaded ?destination_leg)
        (origin_leg_at_transfer_point ?origin_leg ?origin_transfer_point)
        (destination_leg_at_transfer_point ?destination_leg ?destination_transfer_point)
        (origin_transfer_point_staged ?origin_transfer_point)
        (destination_transfer_point_staged ?destination_transfer_point)
        (not
          (origin_leg_ready_for_transit ?origin_leg)
        )
        (not
          (destination_leg_ready_for_transit ?destination_leg)
        )
        (transit_leg_plan_available ?transit_leg_plan)
      )
    :effect
      (and
        (transit_leg_plan_committed ?transit_leg_plan)
        (transit_leg_plan_origin_point ?transit_leg_plan ?origin_transfer_point)
        (transit_leg_plan_destination_point ?transit_leg_plan ?destination_transfer_point)
        (transit_leg_plan_origin_staged ?transit_leg_plan)
        (transit_leg_plan_destination_staged ?transit_leg_plan)
        (not
          (transit_leg_plan_available ?transit_leg_plan)
        )
      )
  )
  (:action activate_transit_leg_plan
    :parameters (?transit_leg_plan - transit_leg_plan ?origin_leg - origin_leg ?time_slot - time_slot)
    :precondition
      (and
        (transit_leg_plan_committed ?transit_leg_plan)
        (origin_leg_loaded ?origin_leg)
        (entity_assigned_time_slot ?origin_leg ?time_slot)
        (not
          (transit_leg_plan_activated ?transit_leg_plan)
        )
      )
    :effect (transit_leg_plan_activated ?transit_leg_plan)
  )
  (:action attach_container_to_transit_leg_for_order
    :parameters (?shipment_order - shipment_order ?container_asset - container_asset ?transit_leg_plan - transit_leg_plan)
    :precondition
      (and
        (planning_entity_confirmed ?shipment_order)
        (order_assigned_to_transit_leg ?shipment_order ?transit_leg_plan)
        (order_has_container ?shipment_order ?container_asset)
        (container_available ?container_asset)
        (transit_leg_plan_committed ?transit_leg_plan)
        (transit_leg_plan_activated ?transit_leg_plan)
        (not
          (container_committed_to_order ?container_asset)
        )
      )
    :effect
      (and
        (container_committed_to_order ?container_asset)
        (container_assigned_to_transit_leg ?container_asset ?transit_leg_plan)
        (not
          (container_available ?container_asset)
        )
      )
  )
  (:action finalize_order_activation_for_transit
    :parameters (?shipment_order - shipment_order ?container_asset - container_asset ?transit_leg_plan - transit_leg_plan ?time_slot - time_slot)
    :precondition
      (and
        (planning_entity_confirmed ?shipment_order)
        (order_has_container ?shipment_order ?container_asset)
        (container_committed_to_order ?container_asset)
        (container_assigned_to_transit_leg ?container_asset ?transit_leg_plan)
        (entity_assigned_time_slot ?shipment_order ?time_slot)
        (not
          (transit_leg_plan_origin_staged ?transit_leg_plan)
        )
        (not
          (order_activated ?shipment_order)
        )
      )
    :effect (order_activated ?shipment_order)
  )
  (:action assign_routing_option_to_order
    :parameters (?shipment_order - shipment_order ?routing_option - routing_option)
    :precondition
      (and
        (planning_entity_confirmed ?shipment_order)
        (routing_option_available ?routing_option)
        (not
          (order_routing_selected ?shipment_order)
        )
      )
    :effect
      (and
        (order_routing_selected ?shipment_order)
        (order_assigned_routing_option ?shipment_order ?routing_option)
        (not
          (routing_option_available ?routing_option)
        )
      )
  )
  (:action lock_routing_option_and_prepare_order
    :parameters (?shipment_order - shipment_order ?container_asset - container_asset ?transit_leg_plan - transit_leg_plan ?time_slot - time_slot ?routing_option - routing_option)
    :precondition
      (and
        (planning_entity_confirmed ?shipment_order)
        (order_has_container ?shipment_order ?container_asset)
        (container_committed_to_order ?container_asset)
        (container_assigned_to_transit_leg ?container_asset ?transit_leg_plan)
        (entity_assigned_time_slot ?shipment_order ?time_slot)
        (transit_leg_plan_origin_staged ?transit_leg_plan)
