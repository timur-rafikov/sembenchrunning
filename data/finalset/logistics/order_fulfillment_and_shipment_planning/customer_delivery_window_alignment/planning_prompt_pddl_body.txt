## Planning domain (PDDL)

(define (domain logistics_customer_delivery_window_alignment)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_category - object temporal_category - object option_category - object order_entity - object order - order_entity transport_resource - resource_category customer_time_window - resource_category loading_dock - resource_category vehicle_type - resource_category service_level_profile - resource_category delivery_time_slot - resource_category handling_requirement - resource_category carrier_shift - resource_category order_line - temporal_category pallet - temporal_category regulatory_clearance - temporal_category route_slot - option_category shipment_option - option_category shipment_unit - option_category fulfillment_group - order bundle_group - order fulfillment_unit - fulfillment_group order_segment - fulfillment_group shipment_plan - bundle_group)
  (:predicates
    (order_released_entity ?order - order)
    (entity_confirmed_for_shipment ?order - order)
    (resource_assigned_entity ?order - order)
    (ready_for_dispatch_entity ?order - order)
    (eligible_for_booking_entity ?order - order)
    (entity_delivery_slot_booked ?order - order)
    (transport_resource_available ?transport_resource - transport_resource)
    (assigned_to_transport_resource_entity ?order - order ?transport_resource - transport_resource)
    (customer_time_window_available ?customer_time_window - customer_time_window)
    (reserved_customer_window_entity ?order - order ?customer_time_window - customer_time_window)
    (dock_available ?loading_dock - loading_dock)
    (allocated_to_loading_dock_entity ?order - order ?loading_dock - loading_dock)
    (order_line_available ?order_line - order_line)
    (assigned_line_to_fulfillment_unit ?fulfillment_unit - fulfillment_unit ?order_line - order_line)
    (assigned_line_to_order_segment ?order_segment - order_segment ?order_line - order_line)
    (fulfillment_unit_route_slot_candidate ?fulfillment_unit - fulfillment_unit ?route_slot - route_slot)
    (route_slot_confirmed ?route_slot - route_slot)
    (route_slot_pending ?route_slot - route_slot)
    (ready_for_consolidation ?fulfillment_unit - fulfillment_unit)
    (segment_option_candidate ?order_segment - order_segment ?shipment_option - shipment_option)
    (shipment_option_confirmed ?shipment_option - shipment_option)
    (shipment_option_pending ?shipment_option - shipment_option)
    (segment_ready_for_consolidation ?order_segment - order_segment)
    (shipment_unit_available ?shipment_unit - shipment_unit)
    (shipment_unit_prepared ?shipment_unit - shipment_unit)
    (assigned_route_slot_to_shipment_unit ?shipment_unit - shipment_unit ?route_slot - route_slot)
    (assigned_shipment_option_to_shipment_unit ?shipment_unit - shipment_unit ?shipment_option - shipment_option)
    (shipment_unit_requires_vehicle_assignment ?shipment_unit - shipment_unit)
    (shipment_unit_requires_pallet_assignment ?shipment_unit - shipment_unit)
    (shipment_unit_ready_for_load_build ?shipment_unit - shipment_unit)
    (shipment_plan_includes_fulfillment_unit ?shipment_plan - shipment_plan ?fulfillment_unit - fulfillment_unit)
    (shipment_plan_includes_order_segment ?shipment_plan - shipment_plan ?order_segment - order_segment)
    (shipment_plan_assigned_to_shipment_unit ?shipment_plan - shipment_plan ?shipment_unit - shipment_unit)
    (pallet_available ?pallet - pallet)
    (assigned_pallet_to_shipment_plan ?shipment_plan - shipment_plan ?pallet - pallet)
    (pallet_in_use ?pallet - pallet)
    (pallet_assigned_to_shipment_unit ?pallet - pallet ?shipment_unit - shipment_unit)
    (load_build_confirmed ?shipment_plan - shipment_plan)
    (handling_assigned ?shipment_plan - shipment_plan)
    (carrier_shift_allocated ?shipment_plan - shipment_plan)
    (vehicle_type_assigned_flag ?shipment_plan - shipment_plan)
    (vehicle_assignment_confirmed ?shipment_plan - shipment_plan)
    (ready_for_service_assignment ?shipment_plan - shipment_plan)
    (dispatch_conditions_met ?shipment_plan - shipment_plan)
    (regulatory_clearance_available ?regulatory_clearance - regulatory_clearance)
    (assigned_regulatory_clearance_to_plan ?shipment_plan - shipment_plan ?regulatory_clearance - regulatory_clearance)
    (regulatory_clearance_applied_to_plan ?shipment_plan - shipment_plan)
    (plan_dock_verified ?shipment_plan - shipment_plan)
    (carrier_shift_confirmed ?shipment_plan - shipment_plan)
    (vehicle_type_available ?vehicle_type - vehicle_type)
    (plan_assigned_vehicle_type ?shipment_plan - shipment_plan ?vehicle_type - vehicle_type)
    (service_level_profile_available ?service_level_profile - service_level_profile)
    (plan_assigned_service_profile ?shipment_plan - shipment_plan ?service_level_profile - service_level_profile)
    (handling_requirement_available ?handling_requirement - handling_requirement)
    (plan_assigned_handling_requirement ?shipment_plan - shipment_plan ?handling_requirement - handling_requirement)
    (carrier_shift_available ?carrier_shift - carrier_shift)
    (assigned_carrier_shift_to_plan ?shipment_plan - shipment_plan ?carrier_shift - carrier_shift)
    (delivery_time_slot_available ?delivery_time_slot - delivery_time_slot)
    (entity_booked_delivery_time_slot ?order - order ?delivery_time_slot - delivery_time_slot)
    (fulfillment_unit_prepared ?fulfillment_unit - fulfillment_unit)
    (order_segment_prepared ?order_segment - order_segment)
    (shipment_plan_validated ?shipment_plan - shipment_plan)
  )
  (:action release_order
    :parameters (?order - order)
    :precondition
      (and
        (not
          (order_released_entity ?order)
        )
        (not
          (ready_for_dispatch_entity ?order)
        )
      )
    :effect (order_released_entity ?order)
  )
  (:action assign_transport_resource_to_order
    :parameters (?order - order ?transport_resource - transport_resource)
    :precondition
      (and
        (order_released_entity ?order)
        (not
          (resource_assigned_entity ?order)
        )
        (transport_resource_available ?transport_resource)
      )
    :effect
      (and
        (resource_assigned_entity ?order)
        (assigned_to_transport_resource_entity ?order ?transport_resource)
        (not
          (transport_resource_available ?transport_resource)
        )
      )
  )
  (:action reserve_customer_time_window_for_order
    :parameters (?order - order ?customer_time_window - customer_time_window)
    :precondition
      (and
        (order_released_entity ?order)
        (resource_assigned_entity ?order)
        (customer_time_window_available ?customer_time_window)
      )
    :effect
      (and
        (reserved_customer_window_entity ?order ?customer_time_window)
        (not
          (customer_time_window_available ?customer_time_window)
        )
      )
  )
  (:action confirm_customer_time_window_reservation_for_order
    :parameters (?order - order ?customer_time_window - customer_time_window)
    :precondition
      (and
        (order_released_entity ?order)
        (resource_assigned_entity ?order)
        (reserved_customer_window_entity ?order ?customer_time_window)
        (not
          (entity_confirmed_for_shipment ?order)
        )
      )
    :effect (entity_confirmed_for_shipment ?order)
  )
  (:action rollback_customer_time_window_reservation_for_order
    :parameters (?order - order ?customer_time_window - customer_time_window)
    :precondition
      (and
        (reserved_customer_window_entity ?order ?customer_time_window)
      )
    :effect
      (and
        (customer_time_window_available ?customer_time_window)
        (not
          (reserved_customer_window_entity ?order ?customer_time_window)
        )
      )
  )
  (:action allocate_loading_dock_to_order
    :parameters (?order - order ?loading_dock - loading_dock)
    :precondition
      (and
        (entity_confirmed_for_shipment ?order)
        (dock_available ?loading_dock)
      )
    :effect
      (and
        (allocated_to_loading_dock_entity ?order ?loading_dock)
        (not
          (dock_available ?loading_dock)
        )
      )
  )
  (:action release_loading_dock_from_order
    :parameters (?order - order ?loading_dock - loading_dock)
    :precondition
      (and
        (allocated_to_loading_dock_entity ?order ?loading_dock)
      )
    :effect
      (and
        (dock_available ?loading_dock)
        (not
          (allocated_to_loading_dock_entity ?order ?loading_dock)
        )
      )
  )
  (:action assign_handling_requirement_to_shipment_plan
    :parameters (?shipment_plan - shipment_plan ?handling_requirement - handling_requirement)
    :precondition
      (and
        (entity_confirmed_for_shipment ?shipment_plan)
        (handling_requirement_available ?handling_requirement)
      )
    :effect
      (and
        (plan_assigned_handling_requirement ?shipment_plan ?handling_requirement)
        (not
          (handling_requirement_available ?handling_requirement)
        )
      )
  )
  (:action unassign_handling_requirement_from_shipment_plan
    :parameters (?shipment_plan - shipment_plan ?handling_requirement - handling_requirement)
    :precondition
      (and
        (plan_assigned_handling_requirement ?shipment_plan ?handling_requirement)
      )
    :effect
      (and
        (handling_requirement_available ?handling_requirement)
        (not
          (plan_assigned_handling_requirement ?shipment_plan ?handling_requirement)
        )
      )
  )
  (:action assign_carrier_shift_to_shipment_plan
    :parameters (?shipment_plan - shipment_plan ?carrier_shift - carrier_shift)
    :precondition
      (and
        (entity_confirmed_for_shipment ?shipment_plan)
        (carrier_shift_available ?carrier_shift)
      )
    :effect
      (and
        (assigned_carrier_shift_to_plan ?shipment_plan ?carrier_shift)
        (not
          (carrier_shift_available ?carrier_shift)
        )
      )
  )
  (:action unassign_carrier_shift_from_shipment_plan
    :parameters (?shipment_plan - shipment_plan ?carrier_shift - carrier_shift)
    :precondition
      (and
        (assigned_carrier_shift_to_plan ?shipment_plan ?carrier_shift)
      )
    :effect
      (and
        (carrier_shift_available ?carrier_shift)
        (not
          (assigned_carrier_shift_to_plan ?shipment_plan ?carrier_shift)
        )
      )
  )
  (:action reserve_route_slot_for_fulfillment_unit
    :parameters (?fulfillment_unit - fulfillment_unit ?route_slot - route_slot ?customer_time_window - customer_time_window)
    :precondition
      (and
        (entity_confirmed_for_shipment ?fulfillment_unit)
        (reserved_customer_window_entity ?fulfillment_unit ?customer_time_window)
        (fulfillment_unit_route_slot_candidate ?fulfillment_unit ?route_slot)
        (not
          (route_slot_confirmed ?route_slot)
        )
        (not
          (route_slot_pending ?route_slot)
        )
      )
    :effect (route_slot_confirmed ?route_slot)
  )
  (:action allocate_dock_to_fulfillment_unit
    :parameters (?fulfillment_unit - fulfillment_unit ?route_slot - route_slot ?loading_dock - loading_dock)
    :precondition
      (and
        (entity_confirmed_for_shipment ?fulfillment_unit)
        (allocated_to_loading_dock_entity ?fulfillment_unit ?loading_dock)
        (fulfillment_unit_route_slot_candidate ?fulfillment_unit ?route_slot)
        (route_slot_confirmed ?route_slot)
        (not
          (fulfillment_unit_prepared ?fulfillment_unit)
        )
      )
    :effect
      (and
        (fulfillment_unit_prepared ?fulfillment_unit)
        (ready_for_consolidation ?fulfillment_unit)
      )
  )
  (:action assign_order_line_to_fulfillment_unit
    :parameters (?fulfillment_unit - fulfillment_unit ?route_slot - route_slot ?order_line - order_line)
    :precondition
      (and
        (entity_confirmed_for_shipment ?fulfillment_unit)
        (fulfillment_unit_route_slot_candidate ?fulfillment_unit ?route_slot)
        (order_line_available ?order_line)
        (not
          (fulfillment_unit_prepared ?fulfillment_unit)
        )
      )
    :effect
      (and
        (route_slot_pending ?route_slot)
        (fulfillment_unit_prepared ?fulfillment_unit)
        (assigned_line_to_fulfillment_unit ?fulfillment_unit ?order_line)
        (not
          (order_line_available ?order_line)
        )
      )
  )
  (:action finalize_line_assignment_and_confirm_route_slot_for_fulfillment_unit
    :parameters (?fulfillment_unit - fulfillment_unit ?route_slot - route_slot ?customer_time_window - customer_time_window ?order_line - order_line)
    :precondition
      (and
        (entity_confirmed_for_shipment ?fulfillment_unit)
        (reserved_customer_window_entity ?fulfillment_unit ?customer_time_window)
        (fulfillment_unit_route_slot_candidate ?fulfillment_unit ?route_slot)
        (route_slot_pending ?route_slot)
        (assigned_line_to_fulfillment_unit ?fulfillment_unit ?order_line)
        (not
          (ready_for_consolidation ?fulfillment_unit)
        )
      )
    :effect
      (and
        (route_slot_confirmed ?route_slot)
        (ready_for_consolidation ?fulfillment_unit)
        (order_line_available ?order_line)
        (not
          (assigned_line_to_fulfillment_unit ?fulfillment_unit ?order_line)
        )
      )
  )
  (:action reserve_shipment_option_for_order_segment
    :parameters (?order_segment - order_segment ?shipment_option - shipment_option ?customer_time_window - customer_time_window)
    :precondition
      (and
        (entity_confirmed_for_shipment ?order_segment)
        (reserved_customer_window_entity ?order_segment ?customer_time_window)
        (segment_option_candidate ?order_segment ?shipment_option)
        (not
          (shipment_option_confirmed ?shipment_option)
        )
        (not
          (shipment_option_pending ?shipment_option)
        )
      )
    :effect (shipment_option_confirmed ?shipment_option)
  )
  (:action allocate_dock_to_order_segment
    :parameters (?order_segment - order_segment ?shipment_option - shipment_option ?loading_dock - loading_dock)
    :precondition
      (and
        (entity_confirmed_for_shipment ?order_segment)
        (allocated_to_loading_dock_entity ?order_segment ?loading_dock)
        (segment_option_candidate ?order_segment ?shipment_option)
        (shipment_option_confirmed ?shipment_option)
        (not
          (order_segment_prepared ?order_segment)
        )
      )
    :effect
      (and
        (order_segment_prepared ?order_segment)
        (segment_ready_for_consolidation ?order_segment)
      )
  )
  (:action assign_order_line_to_order_segment
    :parameters (?order_segment - order_segment ?shipment_option - shipment_option ?order_line - order_line)
    :precondition
      (and
        (entity_confirmed_for_shipment ?order_segment)
        (segment_option_candidate ?order_segment ?shipment_option)
        (order_line_available ?order_line)
        (not
          (order_segment_prepared ?order_segment)
        )
      )
    :effect
      (and
        (shipment_option_pending ?shipment_option)
        (order_segment_prepared ?order_segment)
        (assigned_line_to_order_segment ?order_segment ?order_line)
        (not
          (order_line_available ?order_line)
        )
      )
  )
  (:action finalize_segment_line_assignment_and_confirm_option
    :parameters (?order_segment - order_segment ?shipment_option - shipment_option ?customer_time_window - customer_time_window ?order_line - order_line)
    :precondition
      (and
        (entity_confirmed_for_shipment ?order_segment)
        (reserved_customer_window_entity ?order_segment ?customer_time_window)
        (segment_option_candidate ?order_segment ?shipment_option)
        (shipment_option_pending ?shipment_option)
        (assigned_line_to_order_segment ?order_segment ?order_line)
        (not
          (segment_ready_for_consolidation ?order_segment)
        )
      )
    :effect
      (and
        (shipment_option_confirmed ?shipment_option)
        (segment_ready_for_consolidation ?order_segment)
        (order_line_available ?order_line)
        (not
          (assigned_line_to_order_segment ?order_segment ?order_line)
        )
      )
  )
  (:action consolidate_into_shipment_unit
    :parameters (?fulfillment_unit - fulfillment_unit ?order_segment - order_segment ?route_slot - route_slot ?shipment_option - shipment_option ?shipment_unit - shipment_unit)
    :precondition
      (and
        (fulfillment_unit_prepared ?fulfillment_unit)
        (order_segment_prepared ?order_segment)
        (fulfillment_unit_route_slot_candidate ?fulfillment_unit ?route_slot)
        (segment_option_candidate ?order_segment ?shipment_option)
        (route_slot_confirmed ?route_slot)
        (shipment_option_confirmed ?shipment_option)
        (ready_for_consolidation ?fulfillment_unit)
        (segment_ready_for_consolidation ?order_segment)
        (shipment_unit_available ?shipment_unit)
      )
    :effect
      (and
        (shipment_unit_prepared ?shipment_unit)
        (assigned_route_slot_to_shipment_unit ?shipment_unit ?route_slot)
        (assigned_shipment_option_to_shipment_unit ?shipment_unit ?shipment_option)
        (not
          (shipment_unit_available ?shipment_unit)
        )
      )
  )
  (:action consolidate_into_shipment_unit_with_vehicle_requirement
    :parameters (?fulfillment_unit - fulfillment_unit ?order_segment - order_segment ?route_slot - route_slot ?shipment_option - shipment_option ?shipment_unit - shipment_unit)
    :precondition
      (and
        (fulfillment_unit_prepared ?fulfillment_unit)
        (order_segment_prepared ?order_segment)
        (fulfillment_unit_route_slot_candidate ?fulfillment_unit ?route_slot)
        (segment_option_candidate ?order_segment ?shipment_option)
        (route_slot_pending ?route_slot)
        (shipment_option_confirmed ?shipment_option)
        (not
          (ready_for_consolidation ?fulfillment_unit)
        )
        (segment_ready_for_consolidation ?order_segment)
        (shipment_unit_available ?shipment_unit)
      )
    :effect
      (and
        (shipment_unit_prepared ?shipment_unit)
        (assigned_route_slot_to_shipment_unit ?shipment_unit ?route_slot)
        (assigned_shipment_option_to_shipment_unit ?shipment_unit ?shipment_option)
        (shipment_unit_requires_vehicle_assignment ?shipment_unit)
        (not
          (shipment_unit_available ?shipment_unit)
        )
      )
  )
  (:action consolidate_into_shipment_unit_with_pallet_requirement
    :parameters (?fulfillment_unit - fulfillment_unit ?order_segment - order_segment ?route_slot - route_slot ?shipment_option - shipment_option ?shipment_unit - shipment_unit)
    :precondition
      (and
        (fulfillment_unit_prepared ?fulfillment_unit)
        (order_segment_prepared ?order_segment)
        (fulfillment_unit_route_slot_candidate ?fulfillment_unit ?route_slot)
        (segment_option_candidate ?order_segment ?shipment_option)
        (route_slot_confirmed ?route_slot)
        (shipment_option_pending ?shipment_option)
        (ready_for_consolidation ?fulfillment_unit)
        (not
          (segment_ready_for_consolidation ?order_segment)
        )
        (shipment_unit_available ?shipment_unit)
      )
    :effect
      (and
        (shipment_unit_prepared ?shipment_unit)
        (assigned_route_slot_to_shipment_unit ?shipment_unit ?route_slot)
        (assigned_shipment_option_to_shipment_unit ?shipment_unit ?shipment_option)
        (shipment_unit_requires_pallet_assignment ?shipment_unit)
        (not
          (shipment_unit_available ?shipment_unit)
        )
      )
  )
  (:action consolidate_into_shipment_unit_with_vehicle_and_pallet_requirements
    :parameters (?fulfillment_unit - fulfillment_unit ?order_segment - order_segment ?route_slot - route_slot ?shipment_option - shipment_option ?shipment_unit - shipment_unit)
    :precondition
      (and
        (fulfillment_unit_prepared ?fulfillment_unit)
        (order_segment_prepared ?order_segment)
        (fulfillment_unit_route_slot_candidate ?fulfillment_unit ?route_slot)
        (segment_option_candidate ?order_segment ?shipment_option)
        (route_slot_pending ?route_slot)
        (shipment_option_pending ?shipment_option)
        (not
          (ready_for_consolidation ?fulfillment_unit)
        )
        (not
          (segment_ready_for_consolidation ?order_segment)
        )
        (shipment_unit_available ?shipment_unit)
      )
    :effect
      (and
        (shipment_unit_prepared ?shipment_unit)
        (assigned_route_slot_to_shipment_unit ?shipment_unit ?route_slot)
        (assigned_shipment_option_to_shipment_unit ?shipment_unit ?shipment_option)
        (shipment_unit_requires_vehicle_assignment ?shipment_unit)
        (shipment_unit_requires_pallet_assignment ?shipment_unit)
        (not
          (shipment_unit_available ?shipment_unit)
        )
      )
  )
  (:action flag_shipment_unit_ready_for_load_build
    :parameters (?shipment_unit - shipment_unit ?fulfillment_unit - fulfillment_unit ?customer_time_window - customer_time_window)
    :precondition
      (and
        (shipment_unit_prepared ?shipment_unit)
        (fulfillment_unit_prepared ?fulfillment_unit)
        (reserved_customer_window_entity ?fulfillment_unit ?customer_time_window)
        (not
          (shipment_unit_ready_for_load_build ?shipment_unit)
        )
      )
    :effect (shipment_unit_ready_for_load_build ?shipment_unit)
  )
  (:action assign_pallet_to_shipment_plan
    :parameters (?shipment_plan - shipment_plan ?pallet - pallet ?shipment_unit - shipment_unit)
    :precondition
      (and
        (entity_confirmed_for_shipment ?shipment_plan)
        (shipment_plan_assigned_to_shipment_unit ?shipment_plan ?shipment_unit)
        (assigned_pallet_to_shipment_plan ?shipment_plan ?pallet)
        (pallet_available ?pallet)
        (shipment_unit_prepared ?shipment_unit)
        (shipment_unit_ready_for_load_build ?shipment_unit)
        (not
          (pallet_in_use ?pallet)
        )
      )
    :effect
      (and
        (pallet_in_use ?pallet)
        (pallet_assigned_to_shipment_unit ?pallet ?shipment_unit)
        (not
          (pallet_available ?pallet)
        )
      )
  )
  (:action confirm_pallet_load_for_shipment_plan
    :parameters (?shipment_plan - shipment_plan ?pallet - pallet ?shipment_unit - shipment_unit ?customer_time_window - customer_time_window)
    :precondition
      (and
        (entity_confirmed_for_shipment ?shipment_plan)
        (assigned_pallet_to_shipment_plan ?shipment_plan ?pallet)
        (pallet_in_use ?pallet)
        (pallet_assigned_to_shipment_unit ?pallet ?shipment_unit)
        (reserved_customer_window_entity ?shipment_plan ?customer_time_window)
        (not
          (shipment_unit_requires_vehicle_assignment ?shipment_unit)
        )
        (not
          (load_build_confirmed ?shipment_plan)
        )
      )
    :effect (load_build_confirmed ?shipment_plan)
  )
  (:action assign_vehicle_type_to_shipment_plan
    :parameters (?shipment_plan - shipment_plan ?vehicle_type - vehicle_type)
    :precondition
      (and
        (entity_confirmed_for_shipment ?shipment_plan)
        (vehicle_type_available ?vehicle_type)
        (not
          (vehicle_type_assigned_flag ?shipment_plan)
        )
      )
    :effect
      (and
        (vehicle_type_assigned_flag ?shipment_plan)
        (plan_assigned_vehicle_type ?shipment_plan ?vehicle_type)
        (not
          (vehicle_type_available ?vehicle_type)
        )
      )
  )
  (:action complete_vehicle_assignment_for_shipment_plan
    :parameters (?shipment_plan - shipment_plan ?pallet - pallet ?shipment_unit - shipment_unit ?customer_time_window - customer_time_window ?vehicle_type - vehicle_type)
    :precondition
      (and
        (entity_confirmed_for_shipment ?shipment_plan)
        (assigned_pallet_to_shipment_plan ?shipment_plan ?pallet)
        (pallet_in_use ?pallet)
        (pallet_assigned_to_shipment_unit ?pallet ?shipment_unit)
        (reserved_customer_window_entity ?shipment_plan ?customer_time_window)
        (shipment_unit_requires_vehicle_assignment ?shipment_unit)
        (vehicle_type_assigned_flag ?shipment_plan)
        (plan_assigned_vehicle_type ?shipment_plan ?vehicle_type)
        (not
          (load_build_confirmed ?shipment_plan)
        )
      )
    :effect
      (and
        (load_build_confirmed ?shipment_plan)
        (vehicle_assignment_confirmed ?shipment_plan)
      )
  )
  (:action assign_handling_and_prepare_shipment_plan
    :parameters (?shipment_plan - shipment_plan ?handling_requirement - handling_requirement ?loading_dock - loading_dock ?pallet - pallet ?shipment_unit - shipment_unit)
    :precondition
      (and
        (load_build_confirmed ?shipment_plan)
