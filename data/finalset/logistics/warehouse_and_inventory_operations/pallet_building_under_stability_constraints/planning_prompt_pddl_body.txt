## Planning domain (PDDL)

(define (domain pallet_assembly)
  (:requirements :strips :typing :negative-preconditions)
  (:types warehouse_resource - object assembly_asset - object palleting_artifact - object order_record - object build_order - order_record packaging_resource - warehouse_resource sku - warehouse_resource operator - warehouse_resource packaging_configuration - warehouse_resource handling_equipment - warehouse_resource identity_tag - warehouse_resource stabilizer_material - warehouse_resource inspection_stamp - warehouse_resource case_unit - assembly_asset layer_template - assembly_asset inspection_tool - assembly_asset layer_stability_profile - palleting_artifact pallet_stability_profile - palleting_artifact pallet_unit - palleting_artifact work_unit - build_order assembly_context - build_order picker_work_unit - work_unit stager_work_unit - work_unit assembly_station - assembly_context)
  (:predicates
    (build_order_registered ?build_order - build_order)
    (entity_confirmed ?build_order - build_order)
    (build_order_packaging_allocated ?build_order - build_order)
    (handover_complete ?build_order - build_order)
    (entity_completed ?build_order - build_order)
    (entity_tagged ?build_order - build_order)
    (packaging_resource_available ?packaging_resource - packaging_resource)
    (entity_assigned_packaging_resource ?build_order - build_order ?packaging_resource - packaging_resource)
    (sku_available ?sku - sku)
    (entity_reserved_sku ?build_order - build_order ?sku - sku)
    (operator_available ?operator - operator)
    (entity_assigned_operator ?build_order - build_order ?operator - operator)
    (case_unit_available ?case_unit - case_unit)
    (work_unit_has_case_unit ?picker_work_unit - picker_work_unit ?case_unit - case_unit)
    (stager_work_unit_has_case_unit ?stager_work_unit - stager_work_unit ?case_unit - case_unit)
    (work_unit_assigned_layer_profile ?picker_work_unit - picker_work_unit ?layer_stability_profile - layer_stability_profile)
    (layer_stability_profile_ready ?layer_stability_profile - layer_stability_profile)
    (layer_stability_profile_populated ?layer_stability_profile - layer_stability_profile)
    (picker_work_unit_layer_complete ?picker_work_unit - picker_work_unit)
    (stager_assigned_pallet_profile ?stager_work_unit - stager_work_unit ?pallet_stability_profile - pallet_stability_profile)
    (pallet_stability_profile_ready ?pallet_stability_profile - pallet_stability_profile)
    (pallet_stability_profile_populated ?pallet_stability_profile - pallet_stability_profile)
    (stager_work_unit_layer_complete ?stager_work_unit - stager_work_unit)
    (pallet_unit_available ?pallet_unit - pallet_unit)
    (pallet_unit_selected ?pallet_unit - pallet_unit)
    (pallet_unit_has_layer_profile ?pallet_unit - pallet_unit ?layer_stability_profile - layer_stability_profile)
    (pallet_unit_has_pallet_profile ?pallet_unit - pallet_unit ?pallet_stability_profile - pallet_stability_profile)
    (pallet_unit_variant_flag_a ?pallet_unit - pallet_unit)
    (pallet_unit_variant_flag_b ?pallet_unit - pallet_unit)
    (pallet_unit_layer_composed ?pallet_unit - pallet_unit)
    (assembly_station_assigned_picker_work_unit ?assembly_station - assembly_station ?picker_work_unit - picker_work_unit)
    (assembly_station_assigned_stager_work_unit ?assembly_station - assembly_station ?stager_work_unit - stager_work_unit)
    (assembly_station_assigned_pallet_unit ?assembly_station - assembly_station ?pallet_unit - pallet_unit)
    (layer_template_available ?layer_template - layer_template)
    (assembly_station_has_layer_template ?assembly_station - assembly_station ?layer_template - layer_template)
    (layer_template_claimed ?layer_template - layer_template)
    (layer_template_assigned_to_pallet ?layer_template - layer_template ?pallet_unit - pallet_unit)
    (assembly_station_enabled ?assembly_station - assembly_station)
    (assembly_station_stabilizer_applied ?assembly_station - assembly_station)
    (assembly_station_inspection_applied ?assembly_station - assembly_station)
    (assembly_station_has_packaging_configuration ?assembly_station - assembly_station)
    (assembly_station_configuration_attached ?assembly_station - assembly_station)
    (assembly_station_ready_for_handling_equipment ?assembly_station - assembly_station)
    (assembly_station_attachment_finalized ?assembly_station - assembly_station)
    (inspection_tool_available ?inspection_tool - inspection_tool)
    (assembly_station_assigned_inspection_tool ?assembly_station - assembly_station ?inspection_tool - inspection_tool)
    (assembly_station_prepared_for_configuration ?assembly_station - assembly_station)
    (assembly_station_visual_inspection_recorded ?assembly_station - assembly_station)
    (assembly_station_inspection_stamp_recorded ?assembly_station - assembly_station)
    (packaging_configuration_available ?packaging_configuration - packaging_configuration)
    (assembly_station_assigned_packaging_configuration ?assembly_station - assembly_station ?packaging_configuration - packaging_configuration)
    (handling_equipment_available ?handling_equipment - handling_equipment)
    (assembly_station_assigned_handling_equipment ?assembly_station - assembly_station ?handling_equipment - handling_equipment)
    (stabilizer_material_available ?stabilizer_material - stabilizer_material)
    (assembly_station_assigned_stabilizer ?assembly_station - assembly_station ?stabilizer_material - stabilizer_material)
    (inspection_stamp_available ?inspection_stamp - inspection_stamp)
    (assembly_station_assigned_inspection_stamp ?assembly_station - assembly_station ?inspection_stamp - inspection_stamp)
    (identity_tag_available ?identity_tag - identity_tag)
    (entity_assigned_identity_tag ?build_order - build_order ?identity_tag - identity_tag)
    (picker_work_unit_prepared ?picker_work_unit - picker_work_unit)
    (stager_work_unit_prepared ?stager_work_unit - stager_work_unit)
    (assembly_station_handover_recorded ?assembly_station - assembly_station)
  )
  (:action register_build_order
    :parameters (?build_order - build_order)
    :precondition
      (and
        (not
          (build_order_registered ?build_order)
        )
        (not
          (handover_complete ?build_order)
        )
      )
    :effect (build_order_registered ?build_order)
  )
  (:action assign_packaging_to_order
    :parameters (?build_order - build_order ?packaging_resource - packaging_resource)
    :precondition
      (and
        (build_order_registered ?build_order)
        (not
          (build_order_packaging_allocated ?build_order)
        )
        (packaging_resource_available ?packaging_resource)
      )
    :effect
      (and
        (build_order_packaging_allocated ?build_order)
        (entity_assigned_packaging_resource ?build_order ?packaging_resource)
        (not
          (packaging_resource_available ?packaging_resource)
        )
      )
  )
  (:action reserve_sku_for_order
    :parameters (?build_order - build_order ?sku - sku)
    :precondition
      (and
        (build_order_registered ?build_order)
        (build_order_packaging_allocated ?build_order)
        (sku_available ?sku)
      )
    :effect
      (and
        (entity_reserved_sku ?build_order ?sku)
        (not
          (sku_available ?sku)
        )
      )
  )
  (:action confirm_order_sku_reservation
    :parameters (?build_order - build_order ?sku - sku)
    :precondition
      (and
        (build_order_registered ?build_order)
        (build_order_packaging_allocated ?build_order)
        (entity_reserved_sku ?build_order ?sku)
        (not
          (entity_confirmed ?build_order)
        )
      )
    :effect (entity_confirmed ?build_order)
  )
  (:action release_sku_reservation
    :parameters (?build_order - build_order ?sku - sku)
    :precondition
      (and
        (entity_reserved_sku ?build_order ?sku)
      )
    :effect
      (and
        (sku_available ?sku)
        (not
          (entity_reserved_sku ?build_order ?sku)
        )
      )
  )
  (:action assign_operator_to_order
    :parameters (?build_order - build_order ?operator - operator)
    :precondition
      (and
        (entity_confirmed ?build_order)
        (operator_available ?operator)
      )
    :effect
      (and
        (entity_assigned_operator ?build_order ?operator)
        (not
          (operator_available ?operator)
        )
      )
  )
  (:action release_operator_from_order
    :parameters (?build_order - build_order ?operator - operator)
    :precondition
      (and
        (entity_assigned_operator ?build_order ?operator)
      )
    :effect
      (and
        (operator_available ?operator)
        (not
          (entity_assigned_operator ?build_order ?operator)
        )
      )
  )
  (:action allocate_stabilizer_to_station
    :parameters (?assembly_station - assembly_station ?stabilizer_material - stabilizer_material)
    :precondition
      (and
        (entity_confirmed ?assembly_station)
        (stabilizer_material_available ?stabilizer_material)
      )
    :effect
      (and
        (assembly_station_assigned_stabilizer ?assembly_station ?stabilizer_material)
        (not
          (stabilizer_material_available ?stabilizer_material)
        )
      )
  )
  (:action release_stabilizer_from_station
    :parameters (?assembly_station - assembly_station ?stabilizer_material - stabilizer_material)
    :precondition
      (and
        (assembly_station_assigned_stabilizer ?assembly_station ?stabilizer_material)
      )
    :effect
      (and
        (stabilizer_material_available ?stabilizer_material)
        (not
          (assembly_station_assigned_stabilizer ?assembly_station ?stabilizer_material)
        )
      )
  )
  (:action assign_inspection_stamp_to_station
    :parameters (?assembly_station - assembly_station ?inspection_stamp - inspection_stamp)
    :precondition
      (and
        (entity_confirmed ?assembly_station)
        (inspection_stamp_available ?inspection_stamp)
      )
    :effect
      (and
        (assembly_station_assigned_inspection_stamp ?assembly_station ?inspection_stamp)
        (not
          (inspection_stamp_available ?inspection_stamp)
        )
      )
  )
  (:action release_inspection_stamp_from_station
    :parameters (?assembly_station - assembly_station ?inspection_stamp - inspection_stamp)
    :precondition
      (and
        (assembly_station_assigned_inspection_stamp ?assembly_station ?inspection_stamp)
      )
    :effect
      (and
        (inspection_stamp_available ?inspection_stamp)
        (not
          (assembly_station_assigned_inspection_stamp ?assembly_station ?inspection_stamp)
        )
      )
  )
  (:action reserve_layer_profile_for_picker_work_unit
    :parameters (?picker_work_unit - picker_work_unit ?layer_stability_profile - layer_stability_profile ?sku - sku)
    :precondition
      (and
        (entity_confirmed ?picker_work_unit)
        (entity_reserved_sku ?picker_work_unit ?sku)
        (work_unit_assigned_layer_profile ?picker_work_unit ?layer_stability_profile)
        (not
          (layer_stability_profile_ready ?layer_stability_profile)
        )
        (not
          (layer_stability_profile_populated ?layer_stability_profile)
        )
      )
    :effect (layer_stability_profile_ready ?layer_stability_profile)
  )
  (:action prepare_picker_work_unit_for_layer
    :parameters (?picker_work_unit - picker_work_unit ?layer_stability_profile - layer_stability_profile ?operator - operator)
    :precondition
      (and
        (entity_confirmed ?picker_work_unit)
        (entity_assigned_operator ?picker_work_unit ?operator)
        (work_unit_assigned_layer_profile ?picker_work_unit ?layer_stability_profile)
        (layer_stability_profile_ready ?layer_stability_profile)
        (not
          (picker_work_unit_prepared ?picker_work_unit)
        )
      )
    :effect
      (and
        (picker_work_unit_prepared ?picker_work_unit)
        (picker_work_unit_layer_complete ?picker_work_unit)
      )
  )
  (:action stage_case_unit_on_picker_work_unit
    :parameters (?picker_work_unit - picker_work_unit ?layer_stability_profile - layer_stability_profile ?case_unit - case_unit)
    :precondition
      (and
        (entity_confirmed ?picker_work_unit)
        (work_unit_assigned_layer_profile ?picker_work_unit ?layer_stability_profile)
        (case_unit_available ?case_unit)
        (not
          (picker_work_unit_prepared ?picker_work_unit)
        )
      )
    :effect
      (and
        (layer_stability_profile_populated ?layer_stability_profile)
        (picker_work_unit_prepared ?picker_work_unit)
        (work_unit_has_case_unit ?picker_work_unit ?case_unit)
        (not
          (case_unit_available ?case_unit)
        )
      )
  )
  (:action complete_picker_layer_preparation
    :parameters (?picker_work_unit - picker_work_unit ?layer_stability_profile - layer_stability_profile ?sku - sku ?case_unit - case_unit)
    :precondition
      (and
        (entity_confirmed ?picker_work_unit)
        (entity_reserved_sku ?picker_work_unit ?sku)
        (work_unit_assigned_layer_profile ?picker_work_unit ?layer_stability_profile)
        (layer_stability_profile_populated ?layer_stability_profile)
        (work_unit_has_case_unit ?picker_work_unit ?case_unit)
        (not
          (picker_work_unit_layer_complete ?picker_work_unit)
        )
      )
    :effect
      (and
        (layer_stability_profile_ready ?layer_stability_profile)
        (picker_work_unit_layer_complete ?picker_work_unit)
        (case_unit_available ?case_unit)
        (not
          (work_unit_has_case_unit ?picker_work_unit ?case_unit)
        )
      )
  )
  (:action reserve_pallet_profile_for_stager_work_unit
    :parameters (?stager_work_unit - stager_work_unit ?pallet_stability_profile - pallet_stability_profile ?sku - sku)
    :precondition
      (and
        (entity_confirmed ?stager_work_unit)
        (entity_reserved_sku ?stager_work_unit ?sku)
        (stager_assigned_pallet_profile ?stager_work_unit ?pallet_stability_profile)
        (not
          (pallet_stability_profile_ready ?pallet_stability_profile)
        )
        (not
          (pallet_stability_profile_populated ?pallet_stability_profile)
        )
      )
    :effect (pallet_stability_profile_ready ?pallet_stability_profile)
  )
  (:action prepare_stager_work_unit_for_layer
    :parameters (?stager_work_unit - stager_work_unit ?pallet_stability_profile - pallet_stability_profile ?operator - operator)
    :precondition
      (and
        (entity_confirmed ?stager_work_unit)
        (entity_assigned_operator ?stager_work_unit ?operator)
        (stager_assigned_pallet_profile ?stager_work_unit ?pallet_stability_profile)
        (pallet_stability_profile_ready ?pallet_stability_profile)
        (not
          (stager_work_unit_prepared ?stager_work_unit)
        )
      )
    :effect
      (and
        (stager_work_unit_prepared ?stager_work_unit)
        (stager_work_unit_layer_complete ?stager_work_unit)
      )
  )
  (:action stage_case_unit_on_stager_work_unit
    :parameters (?stager_work_unit - stager_work_unit ?pallet_stability_profile - pallet_stability_profile ?case_unit - case_unit)
    :precondition
      (and
        (entity_confirmed ?stager_work_unit)
        (stager_assigned_pallet_profile ?stager_work_unit ?pallet_stability_profile)
        (case_unit_available ?case_unit)
        (not
          (stager_work_unit_prepared ?stager_work_unit)
        )
      )
    :effect
      (and
        (pallet_stability_profile_populated ?pallet_stability_profile)
        (stager_work_unit_prepared ?stager_work_unit)
        (stager_work_unit_has_case_unit ?stager_work_unit ?case_unit)
        (not
          (case_unit_available ?case_unit)
        )
      )
  )
  (:action complete_stager_layer_preparation
    :parameters (?stager_work_unit - stager_work_unit ?pallet_stability_profile - pallet_stability_profile ?sku - sku ?case_unit - case_unit)
    :precondition
      (and
        (entity_confirmed ?stager_work_unit)
        (entity_reserved_sku ?stager_work_unit ?sku)
        (stager_assigned_pallet_profile ?stager_work_unit ?pallet_stability_profile)
        (pallet_stability_profile_populated ?pallet_stability_profile)
        (stager_work_unit_has_case_unit ?stager_work_unit ?case_unit)
        (not
          (stager_work_unit_layer_complete ?stager_work_unit)
        )
      )
    :effect
      (and
        (pallet_stability_profile_ready ?pallet_stability_profile)
        (stager_work_unit_layer_complete ?stager_work_unit)
        (case_unit_available ?case_unit)
        (not
          (stager_work_unit_has_case_unit ?stager_work_unit ?case_unit)
        )
      )
  )
  (:action compose_and_attach_layer_to_pallet_unit
    :parameters (?picker_work_unit - picker_work_unit ?stager_work_unit - stager_work_unit ?layer_stability_profile - layer_stability_profile ?pallet_stability_profile - pallet_stability_profile ?pallet_unit - pallet_unit)
    :precondition
      (and
        (picker_work_unit_prepared ?picker_work_unit)
        (stager_work_unit_prepared ?stager_work_unit)
        (work_unit_assigned_layer_profile ?picker_work_unit ?layer_stability_profile)
        (stager_assigned_pallet_profile ?stager_work_unit ?pallet_stability_profile)
        (layer_stability_profile_ready ?layer_stability_profile)
        (pallet_stability_profile_ready ?pallet_stability_profile)
        (picker_work_unit_layer_complete ?picker_work_unit)
        (stager_work_unit_layer_complete ?stager_work_unit)
        (pallet_unit_available ?pallet_unit)
      )
    :effect
      (and
        (pallet_unit_selected ?pallet_unit)
        (pallet_unit_has_layer_profile ?pallet_unit ?layer_stability_profile)
        (pallet_unit_has_pallet_profile ?pallet_unit ?pallet_stability_profile)
        (not
          (pallet_unit_available ?pallet_unit)
        )
      )
  )
  (:action compose_and_attach_layer_to_pallet_unit_variant_a
    :parameters (?picker_work_unit - picker_work_unit ?stager_work_unit - stager_work_unit ?layer_stability_profile - layer_stability_profile ?pallet_stability_profile - pallet_stability_profile ?pallet_unit - pallet_unit)
    :precondition
      (and
        (picker_work_unit_prepared ?picker_work_unit)
        (stager_work_unit_prepared ?stager_work_unit)
        (work_unit_assigned_layer_profile ?picker_work_unit ?layer_stability_profile)
        (stager_assigned_pallet_profile ?stager_work_unit ?pallet_stability_profile)
        (layer_stability_profile_populated ?layer_stability_profile)
        (pallet_stability_profile_ready ?pallet_stability_profile)
        (not
          (picker_work_unit_layer_complete ?picker_work_unit)
        )
        (stager_work_unit_layer_complete ?stager_work_unit)
        (pallet_unit_available ?pallet_unit)
      )
    :effect
      (and
        (pallet_unit_selected ?pallet_unit)
        (pallet_unit_has_layer_profile ?pallet_unit ?layer_stability_profile)
        (pallet_unit_has_pallet_profile ?pallet_unit ?pallet_stability_profile)
        (pallet_unit_variant_flag_a ?pallet_unit)
        (not
          (pallet_unit_available ?pallet_unit)
        )
      )
  )
  (:action compose_and_attach_layer_to_pallet_unit_variant_b
    :parameters (?picker_work_unit - picker_work_unit ?stager_work_unit - stager_work_unit ?layer_stability_profile - layer_stability_profile ?pallet_stability_profile - pallet_stability_profile ?pallet_unit - pallet_unit)
    :precondition
      (and
        (picker_work_unit_prepared ?picker_work_unit)
        (stager_work_unit_prepared ?stager_work_unit)
        (work_unit_assigned_layer_profile ?picker_work_unit ?layer_stability_profile)
        (stager_assigned_pallet_profile ?stager_work_unit ?pallet_stability_profile)
        (layer_stability_profile_ready ?layer_stability_profile)
        (pallet_stability_profile_populated ?pallet_stability_profile)
        (picker_work_unit_layer_complete ?picker_work_unit)
        (not
          (stager_work_unit_layer_complete ?stager_work_unit)
        )
        (pallet_unit_available ?pallet_unit)
      )
    :effect
      (and
        (pallet_unit_selected ?pallet_unit)
        (pallet_unit_has_layer_profile ?pallet_unit ?layer_stability_profile)
        (pallet_unit_has_pallet_profile ?pallet_unit ?pallet_stability_profile)
        (pallet_unit_variant_flag_b ?pallet_unit)
        (not
          (pallet_unit_available ?pallet_unit)
        )
      )
  )
  (:action compose_and_attach_layer_to_pallet_unit_variants_ab
    :parameters (?picker_work_unit - picker_work_unit ?stager_work_unit - stager_work_unit ?layer_stability_profile - layer_stability_profile ?pallet_stability_profile - pallet_stability_profile ?pallet_unit - pallet_unit)
    :precondition
      (and
        (picker_work_unit_prepared ?picker_work_unit)
        (stager_work_unit_prepared ?stager_work_unit)
        (work_unit_assigned_layer_profile ?picker_work_unit ?layer_stability_profile)
        (stager_assigned_pallet_profile ?stager_work_unit ?pallet_stability_profile)
        (layer_stability_profile_populated ?layer_stability_profile)
        (pallet_stability_profile_populated ?pallet_stability_profile)
        (not
          (picker_work_unit_layer_complete ?picker_work_unit)
        )
        (not
          (stager_work_unit_layer_complete ?stager_work_unit)
        )
        (pallet_unit_available ?pallet_unit)
      )
    :effect
      (and
        (pallet_unit_selected ?pallet_unit)
        (pallet_unit_has_layer_profile ?pallet_unit ?layer_stability_profile)
        (pallet_unit_has_pallet_profile ?pallet_unit ?pallet_stability_profile)
        (pallet_unit_variant_flag_a ?pallet_unit)
        (pallet_unit_variant_flag_b ?pallet_unit)
        (not
          (pallet_unit_available ?pallet_unit)
        )
      )
  )
  (:action lock_pallet_unit_layer_composition
    :parameters (?pallet_unit - pallet_unit ?picker_work_unit - picker_work_unit ?sku - sku)
    :precondition
      (and
        (pallet_unit_selected ?pallet_unit)
        (picker_work_unit_prepared ?picker_work_unit)
        (entity_reserved_sku ?picker_work_unit ?sku)
        (not
          (pallet_unit_layer_composed ?pallet_unit)
        )
      )
    :effect (pallet_unit_layer_composed ?pallet_unit)
  )
  (:action claim_layer_template_for_pallet_unit
    :parameters (?assembly_station - assembly_station ?layer_template - layer_template ?pallet_unit - pallet_unit)
    :precondition
      (and
        (entity_confirmed ?assembly_station)
        (assembly_station_assigned_pallet_unit ?assembly_station ?pallet_unit)
        (assembly_station_has_layer_template ?assembly_station ?layer_template)
        (layer_template_available ?layer_template)
        (pallet_unit_selected ?pallet_unit)
        (pallet_unit_layer_composed ?pallet_unit)
        (not
          (layer_template_claimed ?layer_template)
        )
      )
    :effect
      (and
        (layer_template_claimed ?layer_template)
        (layer_template_assigned_to_pallet ?layer_template ?pallet_unit)
        (not
          (layer_template_available ?layer_template)
        )
      )
  )
  (:action enable_station_for_template_attachment
    :parameters (?assembly_station - assembly_station ?layer_template - layer_template ?pallet_unit - pallet_unit ?sku - sku)
    :precondition
      (and
        (entity_confirmed ?assembly_station)
        (assembly_station_has_layer_template ?assembly_station ?layer_template)
        (layer_template_claimed ?layer_template)
        (layer_template_assigned_to_pallet ?layer_template ?pallet_unit)
        (entity_reserved_sku ?assembly_station ?sku)
        (not
          (pallet_unit_variant_flag_a ?pallet_unit)
        )
        (not
          (assembly_station_enabled ?assembly_station)
        )
      )
    :effect (assembly_station_enabled ?assembly_station)
  )
  (:action assign_packaging_configuration_to_station
    :parameters (?assembly_station - assembly_station ?packaging_configuration - packaging_configuration)
    :precondition
      (and
        (entity_confirmed ?assembly_station)
        (packaging_configuration_available ?packaging_configuration)
        (not
          (assembly_station_has_packaging_configuration ?assembly_station)
        )
      )
    :effect
      (and
        (assembly_station_has_packaging_configuration ?assembly_station)
