## Planning domain (PDDL)

(define (domain flood_affected_field_recovery)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_supertype - object asset_and_report_supertype - object site_and_route_supertype - object field_supertype - object affected_entity - field_supertype contingency_resource - resource_supertype assessment_kit - resource_supertype specialist_team - resource_supertype regulatory_permit - resource_supertype funding_allocation - resource_supertype salvage_option - resource_supertype decontamination_kit - resource_supertype laboratory_result - resource_supertype supply_item - asset_and_report_supertype inspection_report - asset_and_report_supertype insurance_document - asset_and_report_supertype inundation_zone - site_and_route_supertype access_route - site_and_route_supertype deployment_asset - site_and_route_supertype response_unit_supertype - affected_entity operation_supertype - affected_entity primary_response_unit - response_unit_supertype secondary_response_unit - response_unit_supertype recovery_operation - operation_supertype)
  (:predicates
    (affected_field_reported ?affected_field - affected_entity)
    (entity_assessed ?affected_field - affected_entity)
    (affected_field_triaged ?affected_field - affected_entity)
    (entity_reinstated ?affected_field - affected_entity)
    (entity_biosecurity_cleared ?affected_field - affected_entity)
    (salvage_option_attached_to_entity ?affected_field - affected_entity)
    (contingency_resource_available ?contingency_resource - contingency_resource)
    (resource_allocated_to_entity ?affected_field - affected_entity ?contingency_resource - contingency_resource)
    (assessment_kit_available ?assessment_kit - assessment_kit)
    (assessment_kit_assigned_to_entity ?affected_field - affected_entity ?assessment_kit - assessment_kit)
    (specialist_team_available ?specialist_team - specialist_team)
    (specialist_team_assigned_to_entity ?affected_field - affected_entity ?specialist_team - specialist_team)
    (supply_item_available ?supply_item - supply_item)
    (supply_assigned_to_primary_response_unit ?primary_response_unit - primary_response_unit ?supply_item - supply_item)
    (supply_assigned_to_secondary_response_unit ?secondary_response_unit - secondary_response_unit ?supply_item - supply_item)
    (response_unit_assigned_to_zone ?primary_response_unit - primary_response_unit ?inundation_zone - inundation_zone)
    (inundation_zone_inspected ?inundation_zone - inundation_zone)
    (inundation_zone_supplied ?inundation_zone - inundation_zone)
    (primary_response_unit_ready ?primary_response_unit - primary_response_unit)
    (response_unit_assigned_to_route ?secondary_response_unit - secondary_response_unit ?access_route - access_route)
    (access_route_verified ?access_route - access_route)
    (access_route_supplied ?access_route - access_route)
    (secondary_response_unit_ready ?secondary_response_unit - secondary_response_unit)
    (deployment_asset_available ?deployment_asset - deployment_asset)
    (deployment_asset_staged ?deployment_asset - deployment_asset)
    (deployment_asset_assigned_to_zone ?deployment_asset - deployment_asset ?inundation_zone - inundation_zone)
    (deployment_asset_assigned_to_route ?deployment_asset - deployment_asset ?access_route - access_route)
    (deployment_asset_configured_level_1 ?deployment_asset - deployment_asset)
    (deployment_asset_configured_level_2 ?deployment_asset - deployment_asset)
    (deployment_asset_deployment_confirmed ?deployment_asset - deployment_asset)
    (operation_assigned_primary_response_unit ?recovery_operation - recovery_operation ?primary_response_unit - primary_response_unit)
    (operation_assigned_secondary_response_unit ?recovery_operation - recovery_operation ?secondary_response_unit - secondary_response_unit)
    (operation_has_deployment_asset ?recovery_operation - recovery_operation ?deployment_asset - deployment_asset)
    (inspection_report_available ?inspection_report - inspection_report)
    (operation_has_inspection_report ?recovery_operation - recovery_operation ?inspection_report - inspection_report)
    (inspection_report_processed ?inspection_report - inspection_report)
    (inspection_report_linked_to_asset ?inspection_report - inspection_report ?deployment_asset - deployment_asset)
    (operation_inspection_verified ?recovery_operation - recovery_operation)
    (operation_inspection_cleared ?recovery_operation - recovery_operation)
    (operation_ready_for_closure ?recovery_operation - recovery_operation)
    (operation_permit_attached_flag ?recovery_operation - recovery_operation)
    (operation_permit_recorded ?recovery_operation - recovery_operation)
    (operation_ready_for_funding ?recovery_operation - recovery_operation)
    (operation_in_execution ?recovery_operation - recovery_operation)
    (insurance_document_available ?insurance_document - insurance_document)
    (operation_has_insurance_document ?recovery_operation - recovery_operation ?insurance_document - insurance_document)
    (operation_insurance_attached_flag ?recovery_operation - recovery_operation)
    (operation_insurance_reviewed ?recovery_operation - recovery_operation)
    (operation_insurance_approved ?recovery_operation - recovery_operation)
    (regulatory_permit_available ?regulatory_permit - regulatory_permit)
    (operation_has_regulatory_permit ?recovery_operation - recovery_operation ?regulatory_permit - regulatory_permit)
    (funding_allocation_available ?funding_allocation - funding_allocation)
    (operation_has_funding_allocation ?recovery_operation - recovery_operation ?funding_allocation - funding_allocation)
    (decontamination_kit_available ?decontamination_kit - decontamination_kit)
    (operation_has_decontamination_kit ?recovery_operation - recovery_operation ?decontamination_kit - decontamination_kit)
    (laboratory_result_available ?laboratory_result - laboratory_result)
    (operation_has_laboratory_result ?recovery_operation - recovery_operation ?laboratory_result - laboratory_result)
    (salvage_option_available ?salvage_option - salvage_option)
    (entity_has_salvage_option ?affected_field - affected_entity ?salvage_option - salvage_option)
    (primary_response_unit_prepared ?primary_response_unit - primary_response_unit)
    (secondary_response_unit_prepared ?secondary_response_unit - secondary_response_unit)
    (operation_finalized ?recovery_operation - recovery_operation)
  )
  (:action register_affected_field
    :parameters (?affected_field - affected_entity)
    :precondition
      (and
        (not
          (affected_field_reported ?affected_field)
        )
        (not
          (entity_reinstated ?affected_field)
        )
      )
    :effect (affected_field_reported ?affected_field)
  )
  (:action allocate_contingency_resource_to_entity
    :parameters (?affected_field - affected_entity ?contingency_resource - contingency_resource)
    :precondition
      (and
        (affected_field_reported ?affected_field)
        (not
          (affected_field_triaged ?affected_field)
        )
        (contingency_resource_available ?contingency_resource)
      )
    :effect
      (and
        (affected_field_triaged ?affected_field)
        (resource_allocated_to_entity ?affected_field ?contingency_resource)
        (not
          (contingency_resource_available ?contingency_resource)
        )
      )
  )
  (:action assign_assessment_kit_to_entity
    :parameters (?affected_field - affected_entity ?assessment_kit - assessment_kit)
    :precondition
      (and
        (affected_field_reported ?affected_field)
        (affected_field_triaged ?affected_field)
        (assessment_kit_available ?assessment_kit)
      )
    :effect
      (and
        (assessment_kit_assigned_to_entity ?affected_field ?assessment_kit)
        (not
          (assessment_kit_available ?assessment_kit)
        )
      )
  )
  (:action complete_entity_assessment
    :parameters (?affected_field - affected_entity ?assessment_kit - assessment_kit)
    :precondition
      (and
        (affected_field_reported ?affected_field)
        (affected_field_triaged ?affected_field)
        (assessment_kit_assigned_to_entity ?affected_field ?assessment_kit)
        (not
          (entity_assessed ?affected_field)
        )
      )
    :effect (entity_assessed ?affected_field)
  )
  (:action return_assessment_kit
    :parameters (?affected_field - affected_entity ?assessment_kit - assessment_kit)
    :precondition
      (and
        (assessment_kit_assigned_to_entity ?affected_field ?assessment_kit)
      )
    :effect
      (and
        (assessment_kit_available ?assessment_kit)
        (not
          (assessment_kit_assigned_to_entity ?affected_field ?assessment_kit)
        )
      )
  )
  (:action engage_specialist_team
    :parameters (?affected_field - affected_entity ?specialist_team - specialist_team)
    :precondition
      (and
        (entity_assessed ?affected_field)
        (specialist_team_available ?specialist_team)
      )
    :effect
      (and
        (specialist_team_assigned_to_entity ?affected_field ?specialist_team)
        (not
          (specialist_team_available ?specialist_team)
        )
      )
  )
  (:action release_specialist_team
    :parameters (?affected_field - affected_entity ?specialist_team - specialist_team)
    :precondition
      (and
        (specialist_team_assigned_to_entity ?affected_field ?specialist_team)
      )
    :effect
      (and
        (specialist_team_available ?specialist_team)
        (not
          (specialist_team_assigned_to_entity ?affected_field ?specialist_team)
        )
      )
  )
  (:action assign_decontamination_kit_to_operation
    :parameters (?recovery_operation - recovery_operation ?decontamination_kit - decontamination_kit)
    :precondition
      (and
        (entity_assessed ?recovery_operation)
        (decontamination_kit_available ?decontamination_kit)
      )
    :effect
      (and
        (operation_has_decontamination_kit ?recovery_operation ?decontamination_kit)
        (not
          (decontamination_kit_available ?decontamination_kit)
        )
      )
  )
  (:action return_decontamination_kit_to_pool
    :parameters (?recovery_operation - recovery_operation ?decontamination_kit - decontamination_kit)
    :precondition
      (and
        (operation_has_decontamination_kit ?recovery_operation ?decontamination_kit)
      )
    :effect
      (and
        (decontamination_kit_available ?decontamination_kit)
        (not
          (operation_has_decontamination_kit ?recovery_operation ?decontamination_kit)
        )
      )
  )
  (:action assign_laboratory_result_to_operation
    :parameters (?recovery_operation - recovery_operation ?laboratory_result - laboratory_result)
    :precondition
      (and
        (entity_assessed ?recovery_operation)
        (laboratory_result_available ?laboratory_result)
      )
    :effect
      (and
        (operation_has_laboratory_result ?recovery_operation ?laboratory_result)
        (not
          (laboratory_result_available ?laboratory_result)
        )
      )
  )
  (:action return_laboratory_result_to_pool
    :parameters (?recovery_operation - recovery_operation ?laboratory_result - laboratory_result)
    :precondition
      (and
        (operation_has_laboratory_result ?recovery_operation ?laboratory_result)
      )
    :effect
      (and
        (laboratory_result_available ?laboratory_result)
        (not
          (operation_has_laboratory_result ?recovery_operation ?laboratory_result)
        )
      )
  )
  (:action inspect_inundation_zone_by_primary_unit
    :parameters (?primary_response_unit - primary_response_unit ?inundation_zone - inundation_zone ?assessment_kit - assessment_kit)
    :precondition
      (and
        (entity_assessed ?primary_response_unit)
        (assessment_kit_assigned_to_entity ?primary_response_unit ?assessment_kit)
        (response_unit_assigned_to_zone ?primary_response_unit ?inundation_zone)
        (not
          (inundation_zone_inspected ?inundation_zone)
        )
        (not
          (inundation_zone_supplied ?inundation_zone)
        )
      )
    :effect (inundation_zone_inspected ?inundation_zone)
  )
  (:action prepare_primary_unit_with_team
    :parameters (?primary_response_unit - primary_response_unit ?inundation_zone - inundation_zone ?specialist_team - specialist_team)
    :precondition
      (and
        (entity_assessed ?primary_response_unit)
        (specialist_team_assigned_to_entity ?primary_response_unit ?specialist_team)
        (response_unit_assigned_to_zone ?primary_response_unit ?inundation_zone)
        (inundation_zone_inspected ?inundation_zone)
        (not
          (primary_response_unit_prepared ?primary_response_unit)
        )
      )
    :effect
      (and
        (primary_response_unit_prepared ?primary_response_unit)
        (primary_response_unit_ready ?primary_response_unit)
      )
  )
  (:action deploy_supply_to_primary_unit
    :parameters (?primary_response_unit - primary_response_unit ?inundation_zone - inundation_zone ?supply_item - supply_item)
    :precondition
      (and
        (entity_assessed ?primary_response_unit)
        (response_unit_assigned_to_zone ?primary_response_unit ?inundation_zone)
        (supply_item_available ?supply_item)
        (not
          (primary_response_unit_prepared ?primary_response_unit)
        )
      )
    :effect
      (and
        (inundation_zone_supplied ?inundation_zone)
        (primary_response_unit_prepared ?primary_response_unit)
        (supply_assigned_to_primary_response_unit ?primary_response_unit ?supply_item)
        (not
          (supply_item_available ?supply_item)
        )
      )
  )
  (:action finalize_primary_unit_inspection
    :parameters (?primary_response_unit - primary_response_unit ?inundation_zone - inundation_zone ?assessment_kit - assessment_kit ?supply_item - supply_item)
    :precondition
      (and
        (entity_assessed ?primary_response_unit)
        (assessment_kit_assigned_to_entity ?primary_response_unit ?assessment_kit)
        (response_unit_assigned_to_zone ?primary_response_unit ?inundation_zone)
        (inundation_zone_supplied ?inundation_zone)
        (supply_assigned_to_primary_response_unit ?primary_response_unit ?supply_item)
        (not
          (primary_response_unit_ready ?primary_response_unit)
        )
      )
    :effect
      (and
        (inundation_zone_inspected ?inundation_zone)
        (primary_response_unit_ready ?primary_response_unit)
        (supply_item_available ?supply_item)
        (not
          (supply_assigned_to_primary_response_unit ?primary_response_unit ?supply_item)
        )
      )
  )
  (:action inspect_inundation_zone_by_secondary_unit
    :parameters (?secondary_response_unit - secondary_response_unit ?access_route - access_route ?assessment_kit - assessment_kit)
    :precondition
      (and
        (entity_assessed ?secondary_response_unit)
        (assessment_kit_assigned_to_entity ?secondary_response_unit ?assessment_kit)
        (response_unit_assigned_to_route ?secondary_response_unit ?access_route)
        (not
          (access_route_verified ?access_route)
        )
        (not
          (access_route_supplied ?access_route)
        )
      )
    :effect (access_route_verified ?access_route)
  )
  (:action prepare_secondary_unit_with_team
    :parameters (?secondary_response_unit - secondary_response_unit ?access_route - access_route ?specialist_team - specialist_team)
    :precondition
      (and
        (entity_assessed ?secondary_response_unit)
        (specialist_team_assigned_to_entity ?secondary_response_unit ?specialist_team)
        (response_unit_assigned_to_route ?secondary_response_unit ?access_route)
        (access_route_verified ?access_route)
        (not
          (secondary_response_unit_prepared ?secondary_response_unit)
        )
      )
    :effect
      (and
        (secondary_response_unit_prepared ?secondary_response_unit)
        (secondary_response_unit_ready ?secondary_response_unit)
      )
  )
  (:action deploy_supply_to_secondary_unit
    :parameters (?secondary_response_unit - secondary_response_unit ?access_route - access_route ?supply_item - supply_item)
    :precondition
      (and
        (entity_assessed ?secondary_response_unit)
        (response_unit_assigned_to_route ?secondary_response_unit ?access_route)
        (supply_item_available ?supply_item)
        (not
          (secondary_response_unit_prepared ?secondary_response_unit)
        )
      )
    :effect
      (and
        (access_route_supplied ?access_route)
        (secondary_response_unit_prepared ?secondary_response_unit)
        (supply_assigned_to_secondary_response_unit ?secondary_response_unit ?supply_item)
        (not
          (supply_item_available ?supply_item)
        )
      )
  )
  (:action finalize_secondary_unit_inspection
    :parameters (?secondary_response_unit - secondary_response_unit ?access_route - access_route ?assessment_kit - assessment_kit ?supply_item - supply_item)
    :precondition
      (and
        (entity_assessed ?secondary_response_unit)
        (assessment_kit_assigned_to_entity ?secondary_response_unit ?assessment_kit)
        (response_unit_assigned_to_route ?secondary_response_unit ?access_route)
        (access_route_supplied ?access_route)
        (supply_assigned_to_secondary_response_unit ?secondary_response_unit ?supply_item)
        (not
          (secondary_response_unit_ready ?secondary_response_unit)
        )
      )
    :effect
      (and
        (access_route_verified ?access_route)
        (secondary_response_unit_ready ?secondary_response_unit)
        (supply_item_available ?supply_item)
        (not
          (supply_assigned_to_secondary_response_unit ?secondary_response_unit ?supply_item)
        )
      )
  )
  (:action stage_deployment_asset
    :parameters (?primary_response_unit - primary_response_unit ?secondary_response_unit - secondary_response_unit ?inundation_zone - inundation_zone ?access_route - access_route ?deployment_asset - deployment_asset)
    :precondition
      (and
        (primary_response_unit_prepared ?primary_response_unit)
        (secondary_response_unit_prepared ?secondary_response_unit)
        (response_unit_assigned_to_zone ?primary_response_unit ?inundation_zone)
        (response_unit_assigned_to_route ?secondary_response_unit ?access_route)
        (inundation_zone_inspected ?inundation_zone)
        (access_route_verified ?access_route)
        (primary_response_unit_ready ?primary_response_unit)
        (secondary_response_unit_ready ?secondary_response_unit)
        (deployment_asset_available ?deployment_asset)
      )
    :effect
      (and
        (deployment_asset_staged ?deployment_asset)
        (deployment_asset_assigned_to_zone ?deployment_asset ?inundation_zone)
        (deployment_asset_assigned_to_route ?deployment_asset ?access_route)
        (not
          (deployment_asset_available ?deployment_asset)
        )
      )
  )
  (:action stage_deployment_asset_with_config_1
    :parameters (?primary_response_unit - primary_response_unit ?secondary_response_unit - secondary_response_unit ?inundation_zone - inundation_zone ?access_route - access_route ?deployment_asset - deployment_asset)
    :precondition
      (and
        (primary_response_unit_prepared ?primary_response_unit)
        (secondary_response_unit_prepared ?secondary_response_unit)
        (response_unit_assigned_to_zone ?primary_response_unit ?inundation_zone)
        (response_unit_assigned_to_route ?secondary_response_unit ?access_route)
        (inundation_zone_supplied ?inundation_zone)
        (access_route_verified ?access_route)
        (not
          (primary_response_unit_ready ?primary_response_unit)
        )
        (secondary_response_unit_ready ?secondary_response_unit)
        (deployment_asset_available ?deployment_asset)
      )
    :effect
      (and
        (deployment_asset_staged ?deployment_asset)
        (deployment_asset_assigned_to_zone ?deployment_asset ?inundation_zone)
        (deployment_asset_assigned_to_route ?deployment_asset ?access_route)
        (deployment_asset_configured_level_1 ?deployment_asset)
        (not
          (deployment_asset_available ?deployment_asset)
        )
      )
  )
  (:action stage_deployment_asset_with_config_2
    :parameters (?primary_response_unit - primary_response_unit ?secondary_response_unit - secondary_response_unit ?inundation_zone - inundation_zone ?access_route - access_route ?deployment_asset - deployment_asset)
    :precondition
      (and
        (primary_response_unit_prepared ?primary_response_unit)
        (secondary_response_unit_prepared ?secondary_response_unit)
        (response_unit_assigned_to_zone ?primary_response_unit ?inundation_zone)
        (response_unit_assigned_to_route ?secondary_response_unit ?access_route)
        (inundation_zone_inspected ?inundation_zone)
        (access_route_supplied ?access_route)
        (primary_response_unit_ready ?primary_response_unit)
        (not
          (secondary_response_unit_ready ?secondary_response_unit)
        )
        (deployment_asset_available ?deployment_asset)
      )
    :effect
      (and
        (deployment_asset_staged ?deployment_asset)
        (deployment_asset_assigned_to_zone ?deployment_asset ?inundation_zone)
        (deployment_asset_assigned_to_route ?deployment_asset ?access_route)
        (deployment_asset_configured_level_2 ?deployment_asset)
        (not
          (deployment_asset_available ?deployment_asset)
        )
      )
  )
  (:action stage_deployment_asset_fully_configured
    :parameters (?primary_response_unit - primary_response_unit ?secondary_response_unit - secondary_response_unit ?inundation_zone - inundation_zone ?access_route - access_route ?deployment_asset - deployment_asset)
    :precondition
      (and
        (primary_response_unit_prepared ?primary_response_unit)
        (secondary_response_unit_prepared ?secondary_response_unit)
        (response_unit_assigned_to_zone ?primary_response_unit ?inundation_zone)
        (response_unit_assigned_to_route ?secondary_response_unit ?access_route)
        (inundation_zone_supplied ?inundation_zone)
        (access_route_supplied ?access_route)
        (not
          (primary_response_unit_ready ?primary_response_unit)
        )
        (not
          (secondary_response_unit_ready ?secondary_response_unit)
        )
        (deployment_asset_available ?deployment_asset)
      )
    :effect
      (and
        (deployment_asset_staged ?deployment_asset)
        (deployment_asset_assigned_to_zone ?deployment_asset ?inundation_zone)
        (deployment_asset_assigned_to_route ?deployment_asset ?access_route)
        (deployment_asset_configured_level_1 ?deployment_asset)
        (deployment_asset_configured_level_2 ?deployment_asset)
        (not
          (deployment_asset_available ?deployment_asset)
        )
      )
  )
  (:action confirm_deployment_asset_deployment
    :parameters (?deployment_asset - deployment_asset ?primary_response_unit - primary_response_unit ?assessment_kit - assessment_kit)
    :precondition
      (and
        (deployment_asset_staged ?deployment_asset)
        (primary_response_unit_prepared ?primary_response_unit)
        (assessment_kit_assigned_to_entity ?primary_response_unit ?assessment_kit)
        (not
          (deployment_asset_deployment_confirmed ?deployment_asset)
        )
      )
    :effect (deployment_asset_deployment_confirmed ?deployment_asset)
  )
  (:action process_inspection_report_for_operation
    :parameters (?recovery_operation - recovery_operation ?inspection_report - inspection_report ?deployment_asset - deployment_asset)
    :precondition
      (and
        (entity_assessed ?recovery_operation)
        (operation_has_deployment_asset ?recovery_operation ?deployment_asset)
        (operation_has_inspection_report ?recovery_operation ?inspection_report)
        (inspection_report_available ?inspection_report)
        (deployment_asset_staged ?deployment_asset)
        (deployment_asset_deployment_confirmed ?deployment_asset)
        (not
          (inspection_report_processed ?inspection_report)
        )
      )
    :effect
      (and
        (inspection_report_processed ?inspection_report)
        (inspection_report_linked_to_asset ?inspection_report ?deployment_asset)
        (not
          (inspection_report_available ?inspection_report)
        )
      )
  )
  (:action verify_operation_inspection
    :parameters (?recovery_operation - recovery_operation ?inspection_report - inspection_report ?deployment_asset - deployment_asset ?assessment_kit - assessment_kit)
    :precondition
      (and
        (entity_assessed ?recovery_operation)
        (operation_has_inspection_report ?recovery_operation ?inspection_report)
        (inspection_report_processed ?inspection_report)
        (inspection_report_linked_to_asset ?inspection_report ?deployment_asset)
        (assessment_kit_assigned_to_entity ?recovery_operation ?assessment_kit)
        (not
          (deployment_asset_configured_level_1 ?deployment_asset)
        )
        (not
          (operation_inspection_verified ?recovery_operation)
        )
      )
    :effect (operation_inspection_verified ?recovery_operation)
  )
  (:action attach_regulatory_permit_to_operation
    :parameters (?recovery_operation - recovery_operation ?regulatory_permit - regulatory_permit)
    :precondition
      (and
        (entity_assessed ?recovery_operation)
        (regulatory_permit_available ?regulatory_permit)
        (not
          (operation_permit_attached_flag ?recovery_operation)
        )
      )
    :effect
      (and
        (operation_permit_attached_flag ?recovery_operation)
        (operation_has_regulatory_permit ?recovery_operation ?regulatory_permit)
        (not
          (regulatory_permit_available ?regulatory_permit)
        )
      )
