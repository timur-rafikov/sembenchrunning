## Planning domain (PDDL)

(define (domain agri_emergency_reserve_release)
  (:requirements :strips :typing :negative-preconditions)
  (:types generic_object - object resource_category - generic_object consumable_category - generic_object transport_category - generic_object operational_unit - generic_object requesting_unit - operational_unit reserve_stock_item - resource_category commodity_type - resource_category operational_asset - resource_category authorization_token - resource_category technical_specialist - resource_category allocation_voucher - resource_category critical_component - resource_category regulatory_certificate - resource_category supply_package - consumable_category quality_test_resource - consumable_category priority_beneficiary - consumable_category primary_transport_resource - transport_category secondary_transport_resource - transport_category release_order - transport_category request_cluster - requesting_unit distribution_cluster - requesting_unit primary_beneficiary_unit - request_cluster secondary_beneficiary_unit - request_cluster distribution_center - distribution_cluster)

  (:predicates
    (requesting_unit_registered ?emergency_request - requesting_unit)
    (requesting_unit_validated ?emergency_request - requesting_unit)
    (requesting_unit_reserve_allocated ?emergency_request - requesting_unit)
    (requesting_unit_released ?emergency_request - requesting_unit)
    (requesting_unit_ready_for_dispatch ?emergency_request - requesting_unit)
    (allocation_confirmed ?emergency_request - requesting_unit)
    (reserve_stock_available ?reserve_stock_item - reserve_stock_item)
    (reserve_allocated_to_requesting_unit ?emergency_request - requesting_unit ?reserve_stock_item - reserve_stock_item)
    (commodity_available ?commodity_type - commodity_type)
    (commodity_reserved_for_requesting_unit ?emergency_request - requesting_unit ?commodity_type - commodity_type)
    (asset_available ?operational_asset - operational_asset)
    (asset_assigned_to_requesting_unit ?emergency_request - requesting_unit ?operational_asset - operational_asset)
    (supply_package_available ?supply_package - supply_package)
    (supply_package_assigned_to_primary_unit ?primary_beneficiary_unit - primary_beneficiary_unit ?supply_package - supply_package)
    (supply_package_assigned_to_secondary_unit ?secondary_beneficiary_unit - secondary_beneficiary_unit ?supply_package - supply_package)
    (primary_transport_association ?primary_beneficiary_unit - primary_beneficiary_unit ?primary_transport_resource - primary_transport_resource)
    (primary_transport_reserved ?primary_transport_resource - primary_transport_resource)
    (primary_transport_loaded ?primary_transport_resource - primary_transport_resource)
    (primary_unit_staged ?primary_beneficiary_unit - primary_beneficiary_unit)
    (secondary_transport_association ?secondary_beneficiary_unit - secondary_beneficiary_unit ?secondary_transport_resource - secondary_transport_resource)
    (secondary_transport_reserved ?secondary_transport_resource - secondary_transport_resource)
    (secondary_transport_loaded ?secondary_transport_resource - secondary_transport_resource)
    (secondary_unit_staged ?secondary_beneficiary_unit - secondary_beneficiary_unit)
    (release_order_open ?release_order - release_order)
    (release_order_confirmed ?release_order - release_order)
    (order_assigned_primary_transport ?release_order - release_order ?primary_transport_resource - primary_transport_resource)
    (order_assigned_secondary_transport ?release_order - release_order ?secondary_transport_resource - secondary_transport_resource)
    (order_primary_route_ready ?release_order - release_order)
    (order_secondary_route_ready ?release_order - release_order)
    (order_transport_verified ?release_order - release_order)
    (center_linked_primary_unit ?distribution_center - distribution_center ?primary_beneficiary_unit - primary_beneficiary_unit)
    (center_linked_secondary_unit ?distribution_center - distribution_center ?secondary_beneficiary_unit - secondary_beneficiary_unit)
    (center_has_release_order ?distribution_center - distribution_center ?release_order - release_order)
    (test_resource_available ?quality_test_resource - quality_test_resource)
    (center_has_test_resource ?distribution_center - distribution_center ?quality_test_resource - quality_test_resource)
    (test_resource_in_use ?quality_test_resource - quality_test_resource)
    (test_assigned_to_order ?quality_test_resource - quality_test_resource ?release_order - release_order)
    (center_testing_in_progress ?distribution_center - distribution_center)
    (center_test_passed ?distribution_center - distribution_center)
    (center_certificate_ready ?distribution_center - distribution_center)
    (center_authorized ?distribution_center - distribution_center)
    (center_quality_certified ?distribution_center - distribution_center)
    (center_approval_recorded ?distribution_center - distribution_center)
    (center_final_signoff ?distribution_center - distribution_center)
    (priority_beneficiary_available ?priority_beneficiary - priority_beneficiary)
    (center_assigned_priority_beneficiary ?distribution_center - distribution_center ?priority_beneficiary - priority_beneficiary)
    (center_has_priority_assignment ?distribution_center - distribution_center)
    (center_priority_processing_started ?distribution_center - distribution_center)
    (center_priority_processing_completed ?distribution_center - distribution_center)
    (authorization_token_available ?authorization_token - authorization_token)
    (authorization_assigned_to_center ?distribution_center - distribution_center ?authorization_token - authorization_token)
    (technical_specialist_available ?technical_specialist - technical_specialist)
    (specialist_assigned_to_center ?distribution_center - distribution_center ?technical_specialist - technical_specialist)
    (critical_component_available ?critical_component - critical_component)
    (critical_component_assigned_to_center ?distribution_center - distribution_center ?critical_component - critical_component)
    (regulatory_certificate_available ?regulatory_certificate_var - regulatory_certificate)
    (regulatory_certificate_assigned_to_center ?distribution_center - distribution_center ?regulatory_certificate_var - regulatory_certificate)
    (allocation_voucher_available ?allocation_voucher - allocation_voucher)
    (voucher_linked_to_requesting_unit ?emergency_request - requesting_unit ?allocation_voucher - allocation_voucher)
    (primary_unit_ready ?primary_beneficiary_unit - primary_beneficiary_unit)
    (secondary_unit_ready ?secondary_beneficiary_unit - secondary_beneficiary_unit)
    (center_dispatch_ready ?distribution_center - distribution_center)
  )
  (:action register_requesting_unit
    :parameters (?emergency_request - requesting_unit)
    :precondition
      (and
        (not
          (requesting_unit_registered ?emergency_request)
        )
        (not
          (requesting_unit_released ?emergency_request)
        )
      )
    :effect (requesting_unit_registered ?emergency_request)
  )
  (:action assign_reserve_stock_to_requesting_unit
    :parameters (?emergency_request - requesting_unit ?reserve_stock_item - reserve_stock_item)
    :precondition
      (and
        (requesting_unit_registered ?emergency_request)
        (not
          (requesting_unit_reserve_allocated ?emergency_request)
        )
        (reserve_stock_available ?reserve_stock_item)
      )
    :effect
      (and
        (requesting_unit_reserve_allocated ?emergency_request)
        (reserve_allocated_to_requesting_unit ?emergency_request ?reserve_stock_item)
        (not
          (reserve_stock_available ?reserve_stock_item)
        )
      )
  )
  (:action reserve_commodity_for_requesting_unit
    :parameters (?emergency_request - requesting_unit ?commodity_type - commodity_type)
    :precondition
      (and
        (requesting_unit_registered ?emergency_request)
        (requesting_unit_reserve_allocated ?emergency_request)
        (commodity_available ?commodity_type)
      )
    :effect
      (and
        (commodity_reserved_for_requesting_unit ?emergency_request ?commodity_type)
        (not
          (commodity_available ?commodity_type)
        )
      )
  )
  (:action validate_requesting_unit_with_commodity
    :parameters (?emergency_request - requesting_unit ?commodity_type - commodity_type)
    :precondition
      (and
        (requesting_unit_registered ?emergency_request)
        (requesting_unit_reserve_allocated ?emergency_request)
        (commodity_reserved_for_requesting_unit ?emergency_request ?commodity_type)
        (not
          (requesting_unit_validated ?emergency_request)
        )
      )
    :effect (requesting_unit_validated ?emergency_request)
  )
  (:action unreserve_commodity_for_requesting_unit
    :parameters (?emergency_request - requesting_unit ?commodity_type - commodity_type)
    :precondition
      (and
        (commodity_reserved_for_requesting_unit ?emergency_request ?commodity_type)
      )
    :effect
      (and
        (commodity_available ?commodity_type)
        (not
          (commodity_reserved_for_requesting_unit ?emergency_request ?commodity_type)
        )
      )
  )
  (:action assign_operational_asset_to_requesting_unit
    :parameters (?emergency_request - requesting_unit ?operational_asset - operational_asset)
    :precondition
      (and
        (requesting_unit_validated ?emergency_request)
        (asset_available ?operational_asset)
      )
    :effect
      (and
        (asset_assigned_to_requesting_unit ?emergency_request ?operational_asset)
        (not
          (asset_available ?operational_asset)
        )
      )
  )
  (:action release_operational_asset_from_requesting_unit
    :parameters (?emergency_request - requesting_unit ?operational_asset - operational_asset)
    :precondition
      (and
        (asset_assigned_to_requesting_unit ?emergency_request ?operational_asset)
      )
    :effect
      (and
        (asset_available ?operational_asset)
        (not
          (asset_assigned_to_requesting_unit ?emergency_request ?operational_asset)
        )
      )
  )
  (:action assign_critical_component_to_center
    :parameters (?distribution_center - distribution_center ?critical_component - critical_component)
    :precondition
      (and
        (requesting_unit_validated ?distribution_center)
        (critical_component_available ?critical_component)
      )
    :effect
      (and
        (critical_component_assigned_to_center ?distribution_center ?critical_component)
        (not
          (critical_component_available ?critical_component)
        )
      )
  )
  (:action release_critical_component_from_center
    :parameters (?distribution_center - distribution_center ?critical_component - critical_component)
    :precondition
      (and
        (critical_component_assigned_to_center ?distribution_center ?critical_component)
      )
    :effect
      (and
        (critical_component_available ?critical_component)
        (not
          (critical_component_assigned_to_center ?distribution_center ?critical_component)
        )
      )
  )
  (:action assign_regulatory_certificate_to_center
    :parameters (?distribution_center - distribution_center ?regulatory_certificate_var - regulatory_certificate)
    :precondition
      (and
        (requesting_unit_validated ?distribution_center)
        (regulatory_certificate_available ?regulatory_certificate_var)
      )
    :effect
      (and
        (regulatory_certificate_assigned_to_center ?distribution_center ?regulatory_certificate_var)
        (not
          (regulatory_certificate_available ?regulatory_certificate_var)
        )
      )
  )
  (:action release_regulatory_certificate_from_center
    :parameters (?distribution_center - distribution_center ?regulatory_certificate_var - regulatory_certificate)
    :precondition
      (and
        (regulatory_certificate_assigned_to_center ?distribution_center ?regulatory_certificate_var)
      )
    :effect
      (and
        (regulatory_certificate_available ?regulatory_certificate_var)
        (not
          (regulatory_certificate_assigned_to_center ?distribution_center ?regulatory_certificate_var)
        )
      )
  )
  (:action reserve_primary_transport_for_unit
    :parameters (?primary_beneficiary_unit - primary_beneficiary_unit ?primary_transport_resource - primary_transport_resource ?commodity_type - commodity_type)
    :precondition
      (and
        (requesting_unit_validated ?primary_beneficiary_unit)
        (commodity_reserved_for_requesting_unit ?primary_beneficiary_unit ?commodity_type)
        (primary_transport_association ?primary_beneficiary_unit ?primary_transport_resource)
        (not
          (primary_transport_reserved ?primary_transport_resource)
        )
        (not
          (primary_transport_loaded ?primary_transport_resource)
        )
      )
    :effect (primary_transport_reserved ?primary_transport_resource)
  )
  (:action apply_asset_and_mark_primary_prepared
    :parameters (?primary_beneficiary_unit - primary_beneficiary_unit ?primary_transport_resource - primary_transport_resource ?operational_asset - operational_asset)
    :precondition
      (and
        (requesting_unit_validated ?primary_beneficiary_unit)
        (asset_assigned_to_requesting_unit ?primary_beneficiary_unit ?operational_asset)
        (primary_transport_association ?primary_beneficiary_unit ?primary_transport_resource)
        (primary_transport_reserved ?primary_transport_resource)
        (not
          (primary_unit_ready ?primary_beneficiary_unit)
        )
      )
    :effect
      (and
        (primary_unit_ready ?primary_beneficiary_unit)
        (primary_unit_staged ?primary_beneficiary_unit)
      )
  )
  (:action stage_supply_package_for_primary_unit
    :parameters (?primary_beneficiary_unit - primary_beneficiary_unit ?primary_transport_resource - primary_transport_resource ?supply_package - supply_package)
    :precondition
      (and
        (requesting_unit_validated ?primary_beneficiary_unit)
        (primary_transport_association ?primary_beneficiary_unit ?primary_transport_resource)
        (supply_package_available ?supply_package)
        (not
          (primary_unit_ready ?primary_beneficiary_unit)
        )
      )
    :effect
      (and
        (primary_transport_loaded ?primary_transport_resource)
        (primary_unit_ready ?primary_beneficiary_unit)
        (supply_package_assigned_to_primary_unit ?primary_beneficiary_unit ?supply_package)
        (not
          (supply_package_available ?supply_package)
        )
      )
  )
  (:action finalize_primary_staging
    :parameters (?primary_beneficiary_unit - primary_beneficiary_unit ?primary_transport_resource - primary_transport_resource ?commodity_type - commodity_type ?supply_package - supply_package)
    :precondition
      (and
        (requesting_unit_validated ?primary_beneficiary_unit)
        (commodity_reserved_for_requesting_unit ?primary_beneficiary_unit ?commodity_type)
        (primary_transport_association ?primary_beneficiary_unit ?primary_transport_resource)
        (primary_transport_loaded ?primary_transport_resource)
        (supply_package_assigned_to_primary_unit ?primary_beneficiary_unit ?supply_package)
        (not
          (primary_unit_staged ?primary_beneficiary_unit)
        )
      )
    :effect
      (and
        (primary_transport_reserved ?primary_transport_resource)
        (primary_unit_staged ?primary_beneficiary_unit)
        (supply_package_available ?supply_package)
        (not
          (supply_package_assigned_to_primary_unit ?primary_beneficiary_unit ?supply_package)
        )
      )
  )
  (:action reserve_secondary_transport_for_unit
    :parameters (?secondary_beneficiary_unit - secondary_beneficiary_unit ?secondary_transport_resource - secondary_transport_resource ?commodity_type - commodity_type)
    :precondition
      (and
        (requesting_unit_validated ?secondary_beneficiary_unit)
        (commodity_reserved_for_requesting_unit ?secondary_beneficiary_unit ?commodity_type)
        (secondary_transport_association ?secondary_beneficiary_unit ?secondary_transport_resource)
        (not
          (secondary_transport_reserved ?secondary_transport_resource)
        )
        (not
          (secondary_transport_loaded ?secondary_transport_resource)
        )
      )
    :effect (secondary_transport_reserved ?secondary_transport_resource)
  )
  (:action apply_asset_and_mark_secondary_prepared
    :parameters (?secondary_beneficiary_unit - secondary_beneficiary_unit ?secondary_transport_resource - secondary_transport_resource ?operational_asset - operational_asset)
    :precondition
      (and
        (requesting_unit_validated ?secondary_beneficiary_unit)
        (asset_assigned_to_requesting_unit ?secondary_beneficiary_unit ?operational_asset)
        (secondary_transport_association ?secondary_beneficiary_unit ?secondary_transport_resource)
        (secondary_transport_reserved ?secondary_transport_resource)
        (not
          (secondary_unit_ready ?secondary_beneficiary_unit)
        )
      )
    :effect
      (and
        (secondary_unit_ready ?secondary_beneficiary_unit)
        (secondary_unit_staged ?secondary_beneficiary_unit)
      )
  )
  (:action stage_supply_package_for_secondary_unit
    :parameters (?secondary_beneficiary_unit - secondary_beneficiary_unit ?secondary_transport_resource - secondary_transport_resource ?supply_package - supply_package)
    :precondition
      (and
        (requesting_unit_validated ?secondary_beneficiary_unit)
        (secondary_transport_association ?secondary_beneficiary_unit ?secondary_transport_resource)
        (supply_package_available ?supply_package)
        (not
          (secondary_unit_ready ?secondary_beneficiary_unit)
        )
      )
    :effect
      (and
        (secondary_transport_loaded ?secondary_transport_resource)
        (secondary_unit_ready ?secondary_beneficiary_unit)
        (supply_package_assigned_to_secondary_unit ?secondary_beneficiary_unit ?supply_package)
        (not
          (supply_package_available ?supply_package)
        )
      )
  )
  (:action finalize_secondary_staging
    :parameters (?secondary_beneficiary_unit - secondary_beneficiary_unit ?secondary_transport_resource - secondary_transport_resource ?commodity_type - commodity_type ?supply_package - supply_package)
    :precondition
      (and
        (requesting_unit_validated ?secondary_beneficiary_unit)
        (commodity_reserved_for_requesting_unit ?secondary_beneficiary_unit ?commodity_type)
        (secondary_transport_association ?secondary_beneficiary_unit ?secondary_transport_resource)
        (secondary_transport_loaded ?secondary_transport_resource)
        (supply_package_assigned_to_secondary_unit ?secondary_beneficiary_unit ?supply_package)
        (not
          (secondary_unit_staged ?secondary_beneficiary_unit)
        )
      )
    :effect
      (and
        (secondary_transport_reserved ?secondary_transport_resource)
        (secondary_unit_staged ?secondary_beneficiary_unit)
        (supply_package_available ?supply_package)
        (not
          (supply_package_assigned_to_secondary_unit ?secondary_beneficiary_unit ?supply_package)
        )
      )
  )
  (:action consolidate_units_into_release_order
    :parameters (?primary_beneficiary_unit - primary_beneficiary_unit ?secondary_beneficiary_unit - secondary_beneficiary_unit ?primary_transport_resource - primary_transport_resource ?secondary_transport_resource - secondary_transport_resource ?release_order - release_order)
    :precondition
      (and
        (primary_unit_ready ?primary_beneficiary_unit)
        (secondary_unit_ready ?secondary_beneficiary_unit)
        (primary_transport_association ?primary_beneficiary_unit ?primary_transport_resource)
        (secondary_transport_association ?secondary_beneficiary_unit ?secondary_transport_resource)
        (primary_transport_reserved ?primary_transport_resource)
        (secondary_transport_reserved ?secondary_transport_resource)
        (primary_unit_staged ?primary_beneficiary_unit)
        (secondary_unit_staged ?secondary_beneficiary_unit)
        (release_order_open ?release_order)
      )
    :effect
      (and
        (release_order_confirmed ?release_order)
        (order_assigned_primary_transport ?release_order ?primary_transport_resource)
        (order_assigned_secondary_transport ?release_order ?secondary_transport_resource)
        (not
          (release_order_open ?release_order)
        )
      )
  )
  (:action consolidate_and_mark_primary_loaded
    :parameters (?primary_beneficiary_unit - primary_beneficiary_unit ?secondary_beneficiary_unit - secondary_beneficiary_unit ?primary_transport_resource - primary_transport_resource ?secondary_transport_resource - secondary_transport_resource ?release_order - release_order)
    :precondition
      (and
        (primary_unit_ready ?primary_beneficiary_unit)
        (secondary_unit_ready ?secondary_beneficiary_unit)
        (primary_transport_association ?primary_beneficiary_unit ?primary_transport_resource)
        (secondary_transport_association ?secondary_beneficiary_unit ?secondary_transport_resource)
        (primary_transport_loaded ?primary_transport_resource)
        (secondary_transport_reserved ?secondary_transport_resource)
        (not
          (primary_unit_staged ?primary_beneficiary_unit)
        )
        (secondary_unit_staged ?secondary_beneficiary_unit)
        (release_order_open ?release_order)
      )
    :effect
      (and
        (release_order_confirmed ?release_order)
        (order_assigned_primary_transport ?release_order ?primary_transport_resource)
        (order_assigned_secondary_transport ?release_order ?secondary_transport_resource)
        (order_primary_route_ready ?release_order)
        (not
          (release_order_open ?release_order)
        )
      )
  )
  (:action consolidate_and_mark_secondary_loaded
    :parameters (?primary_beneficiary_unit - primary_beneficiary_unit ?secondary_beneficiary_unit - secondary_beneficiary_unit ?primary_transport_resource - primary_transport_resource ?secondary_transport_resource - secondary_transport_resource ?release_order - release_order)
    :precondition
      (and
        (primary_unit_ready ?primary_beneficiary_unit)
        (secondary_unit_ready ?secondary_beneficiary_unit)
        (primary_transport_association ?primary_beneficiary_unit ?primary_transport_resource)
        (secondary_transport_association ?secondary_beneficiary_unit ?secondary_transport_resource)
        (primary_transport_reserved ?primary_transport_resource)
        (secondary_transport_loaded ?secondary_transport_resource)
        (primary_unit_staged ?primary_beneficiary_unit)
        (not
          (secondary_unit_staged ?secondary_beneficiary_unit)
        )
        (release_order_open ?release_order)
      )
    :effect
      (and
        (release_order_confirmed ?release_order)
        (order_assigned_primary_transport ?release_order ?primary_transport_resource)
        (order_assigned_secondary_transport ?release_order ?secondary_transport_resource)
        (order_secondary_route_ready ?release_order)
        (not
          (release_order_open ?release_order)
        )
      )
  )
  (:action consolidate_and_mark_both_loaded
    :parameters (?primary_beneficiary_unit - primary_beneficiary_unit ?secondary_beneficiary_unit - secondary_beneficiary_unit ?primary_transport_resource - primary_transport_resource ?secondary_transport_resource - secondary_transport_resource ?release_order - release_order)
    :precondition
      (and
        (primary_unit_ready ?primary_beneficiary_unit)
        (secondary_unit_ready ?secondary_beneficiary_unit)
        (primary_transport_association ?primary_beneficiary_unit ?primary_transport_resource)
        (secondary_transport_association ?secondary_beneficiary_unit ?secondary_transport_resource)
        (primary_transport_loaded ?primary_transport_resource)
        (secondary_transport_loaded ?secondary_transport_resource)
        (not
          (primary_unit_staged ?primary_beneficiary_unit)
        )
        (not
          (secondary_unit_staged ?secondary_beneficiary_unit)
        )
        (release_order_open ?release_order)
      )
    :effect
      (and
        (release_order_confirmed ?release_order)
        (order_assigned_primary_transport ?release_order ?primary_transport_resource)
        (order_assigned_secondary_transport ?release_order ?secondary_transport_resource)
        (order_primary_route_ready ?release_order)
        (order_secondary_route_ready ?release_order)
        (not
          (release_order_open ?release_order)
        )
      )
  )
  (:action verify_order_transport
    :parameters (?release_order - release_order ?primary_beneficiary_unit - primary_beneficiary_unit ?commodity_type - commodity_type)
    :precondition
      (and
        (release_order_confirmed ?release_order)
        (primary_unit_ready ?primary_beneficiary_unit)
        (commodity_reserved_for_requesting_unit ?primary_beneficiary_unit ?commodity_type)
        (not
          (order_transport_verified ?release_order)
        )
      )
    :effect (order_transport_verified ?release_order)
  )
  (:action allocate_test_resource_to_order
    :parameters (?distribution_center - distribution_center ?quality_test_resource - quality_test_resource ?release_order - release_order)
    :precondition
      (and
        (requesting_unit_validated ?distribution_center)
        (center_has_release_order ?distribution_center ?release_order)
        (center_has_test_resource ?distribution_center ?quality_test_resource)
        (test_resource_available ?quality_test_resource)
        (release_order_confirmed ?release_order)
        (order_transport_verified ?release_order)
        (not
          (test_resource_in_use ?quality_test_resource)
        )
      )
    :effect
      (and
        (test_resource_in_use ?quality_test_resource)
        (test_assigned_to_order ?quality_test_resource ?release_order)
        (not
          (test_resource_available ?quality_test_resource)
        )
      )
  )
  (:action start_quality_test_at_center
    :parameters (?distribution_center - distribution_center ?quality_test_resource - quality_test_resource ?release_order - release_order ?commodity_type - commodity_type)
    :precondition
      (and
        (requesting_unit_validated ?distribution_center)
        (center_has_test_resource ?distribution_center ?quality_test_resource)
        (test_resource_in_use ?quality_test_resource)
        (test_assigned_to_order ?quality_test_resource ?release_order)
        (commodity_reserved_for_requesting_unit ?distribution_center ?commodity_type)
        (not
          (order_primary_route_ready ?release_order)
        )
        (not
          (center_testing_in_progress ?distribution_center)
        )
      )
    :effect (center_testing_in_progress ?distribution_center)
  )
  (:action assign_authorization_to_center
    :parameters (?distribution_center - distribution_center ?authorization_token - authorization_token)
    :precondition
      (and
        (requesting_unit_validated ?distribution_center)
        (authorization_token_available ?authorization_token)
        (not
          (center_authorized ?distribution_center)
        )
      )
    :effect
      (and
        (center_authorized ?distribution_center)
        (authorization_assigned_to_center ?distribution_center ?authorization_token)
        (not
          (authorization_token_available ?authorization_token)
        )
      )
