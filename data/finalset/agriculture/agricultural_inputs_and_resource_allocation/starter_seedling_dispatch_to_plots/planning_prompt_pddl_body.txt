## Planning domain (PDDL)

(define (domain starter_seedling_dispatch)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_category - object temporal_slot - object geographic_unit - object dispatch_parent - object deployment_unit - dispatch_parent transport_unit - resource_category seedling_variety - resource_category operator - resource_category label - resource_category inspection_certificate - resource_category contingency_reserve - resource_category special_equipment - resource_category biosecurity_clearance - resource_category consumable_batch - temporal_slot container - temporal_slot site_permit - temporal_slot staging_slot - geographic_unit transport_slot - geographic_unit dispatch_pallet - geographic_unit plot_category - deployment_unit plot_subcategory - deployment_unit plot - plot_category subplot - plot_category seedling_batch - plot_subcategory)
  (:predicates
    (dispatch_order_created ?deployment_unit - deployment_unit)
    (dispatch_order_confirmed_for_unit ?deployment_unit - deployment_unit)
    (transport_assigned ?deployment_unit - deployment_unit)
    (deployment_completed ?deployment_unit - deployment_unit)
    (ready_for_deployment ?deployment_unit - deployment_unit)
    (dispatch_authorized ?deployment_unit - deployment_unit)
    (transport_unit_available ?transport_unit - transport_unit)
    (assigned_transport_to_unit ?deployment_unit - deployment_unit ?transport_unit - transport_unit)
    (variety_available ?seedling_variety - seedling_variety)
    (variety_reserved ?deployment_unit - deployment_unit ?seedling_variety - seedling_variety)
    (operator_available ?operator - operator)
    (operator_assigned_to_unit ?deployment_unit - deployment_unit ?operator - operator)
    (consumable_available ?consumable_batch - consumable_batch)
    (plot_consumable_allocated ?plot - plot ?consumable_batch - consumable_batch)
    (subplot_consumable_allocated ?subplot - subplot ?consumable_batch - consumable_batch)
    (plot_allocated_staging_slot ?plot - plot ?staging_slot - staging_slot)
    (staging_slot_reserved ?staging_slot - staging_slot)
    (staging_slot_loaded ?staging_slot - staging_slot)
    (plot_ready_for_pallet_assembly ?plot - plot)
    (subplot_allocated_transport_slot ?subplot - subplot ?transport_slot - transport_slot)
    (transport_slot_reserved ?transport_slot - transport_slot)
    (transport_slot_loaded ?transport_slot - transport_slot)
    (subplot_ready_for_pallet_assembly ?subplot - subplot)
    (dispatch_pallet_available ?dispatch_pallet - dispatch_pallet)
    (dispatch_pallet_prepared ?dispatch_pallet - dispatch_pallet)
    (dispatch_pallet_allocated_staging_slot ?dispatch_pallet - dispatch_pallet ?staging_slot - staging_slot)
    (dispatch_pallet_allocated_transport_slot ?dispatch_pallet - dispatch_pallet ?transport_slot - transport_slot)
    (dispatch_pallet_tag_staging_loaded ?dispatch_pallet - dispatch_pallet)
    (dispatch_pallet_tag_transport_loaded ?dispatch_pallet - dispatch_pallet)
    (dispatch_pallet_validated_for_plot ?dispatch_pallet - dispatch_pallet)
    (batch_assigned_to_plot ?seedling_batch - seedling_batch ?plot - plot)
    (batch_assigned_to_subplot ?seedling_batch - seedling_batch ?subplot - subplot)
    (batch_loaded_to_pallet ?seedling_batch - seedling_batch ?dispatch_pallet - dispatch_pallet)
    (container_available ?container - container)
    (batch_assigned_to_container ?seedling_batch - seedling_batch ?container - container)
    (container_sealed ?container - container)
    (container_loaded_to_pallet ?container - container ?dispatch_pallet - dispatch_pallet)
    (batch_packaging_ready ?seedling_batch - seedling_batch)
    (batch_equipment_attached ?seedling_batch - seedling_batch)
    (batch_inspection_performed ?seedling_batch - seedling_batch)
    (label_attached ?seedling_batch - seedling_batch)
    (label_verified ?seedling_batch - seedling_batch)
    (certificate_attached ?seedling_batch - seedling_batch)
    (batch_finalized_for_dispatch ?seedling_batch - seedling_batch)
    (site_permit_available ?site_permit - site_permit)
    (batch_has_site_permit ?seedling_batch - seedling_batch ?site_permit - site_permit)
    (batch_permit_recorded ?seedling_batch - seedling_batch)
    (batch_permit_verified ?seedling_batch - seedling_batch)
    (batch_permit_finalized ?seedling_batch - seedling_batch)
    (label_available ?label - label)
    (batch_assigned_label ?seedling_batch - seedling_batch ?label - label)
    (inspection_certificate_available ?inspection_certificate - inspection_certificate)
    (batch_assigned_certificate ?seedling_batch - seedling_batch ?inspection_certificate - inspection_certificate)
    (special_equipment_available ?special_equipment - special_equipment)
    (batch_assigned_special_equipment ?seedling_batch - seedling_batch ?special_equipment - special_equipment)
    (biosecurity_clearance_available ?biosecurity_clearance - biosecurity_clearance)
    (batch_assigned_biosecurity_clearance ?seedling_batch - seedling_batch ?biosecurity_clearance - biosecurity_clearance)
    (contingency_reserve_available ?contingency_reserve - contingency_reserve)
    (contingency_reserve_allocated ?deployment_unit - deployment_unit ?contingency_reserve - contingency_reserve)
    (plot_preparation_complete ?plot - plot)
    (subplot_preparation_complete ?subplot - subplot)
    (release_processed ?seedling_batch - seedling_batch)
  )
  (:action create_dispatch_order
    :parameters (?deployment_unit - deployment_unit)
    :precondition
      (and
        (not
          (dispatch_order_created ?deployment_unit)
        )
        (not
          (deployment_completed ?deployment_unit)
        )
      )
    :effect (dispatch_order_created ?deployment_unit)
  )
  (:action assign_transport_to_order
    :parameters (?deployment_unit - deployment_unit ?transport_unit - transport_unit)
    :precondition
      (and
        (dispatch_order_created ?deployment_unit)
        (not
          (transport_assigned ?deployment_unit)
        )
        (transport_unit_available ?transport_unit)
      )
    :effect
      (and
        (transport_assigned ?deployment_unit)
        (assigned_transport_to_unit ?deployment_unit ?transport_unit)
        (not
          (transport_unit_available ?transport_unit)
        )
      )
  )
  (:action reserve_variety_for_order
    :parameters (?deployment_unit - deployment_unit ?seedling_variety - seedling_variety)
    :precondition
      (and
        (dispatch_order_created ?deployment_unit)
        (transport_assigned ?deployment_unit)
        (variety_available ?seedling_variety)
      )
    :effect
      (and
        (variety_reserved ?deployment_unit ?seedling_variety)
        (not
          (variety_available ?seedling_variety)
        )
      )
  )
  (:action confirm_order_variety
    :parameters (?deployment_unit - deployment_unit ?seedling_variety - seedling_variety)
    :precondition
      (and
        (dispatch_order_created ?deployment_unit)
        (transport_assigned ?deployment_unit)
        (variety_reserved ?deployment_unit ?seedling_variety)
        (not
          (dispatch_order_confirmed_for_unit ?deployment_unit)
        )
      )
    :effect (dispatch_order_confirmed_for_unit ?deployment_unit)
  )
  (:action release_variety_reservation
    :parameters (?deployment_unit - deployment_unit ?seedling_variety - seedling_variety)
    :precondition
      (and
        (variety_reserved ?deployment_unit ?seedling_variety)
      )
    :effect
      (and
        (variety_available ?seedling_variety)
        (not
          (variety_reserved ?deployment_unit ?seedling_variety)
        )
      )
  )
  (:action assign_operator_to_order
    :parameters (?deployment_unit - deployment_unit ?operator - operator)
    :precondition
      (and
        (dispatch_order_confirmed_for_unit ?deployment_unit)
        (operator_available ?operator)
      )
    :effect
      (and
        (operator_assigned_to_unit ?deployment_unit ?operator)
        (not
          (operator_available ?operator)
        )
      )
  )
  (:action release_operator_from_order
    :parameters (?deployment_unit - deployment_unit ?operator - operator)
    :precondition
      (and
        (operator_assigned_to_unit ?deployment_unit ?operator)
      )
    :effect
      (and
        (operator_available ?operator)
        (not
          (operator_assigned_to_unit ?deployment_unit ?operator)
        )
      )
  )
  (:action assign_special_equipment_to_batch
    :parameters (?seedling_batch - seedling_batch ?special_equipment - special_equipment)
    :precondition
      (and
        (dispatch_order_confirmed_for_unit ?seedling_batch)
        (special_equipment_available ?special_equipment)
      )
    :effect
      (and
        (batch_assigned_special_equipment ?seedling_batch ?special_equipment)
        (not
          (special_equipment_available ?special_equipment)
        )
      )
  )
  (:action remove_special_equipment_from_batch
    :parameters (?seedling_batch - seedling_batch ?special_equipment - special_equipment)
    :precondition
      (and
        (batch_assigned_special_equipment ?seedling_batch ?special_equipment)
      )
    :effect
      (and
        (special_equipment_available ?special_equipment)
        (not
          (batch_assigned_special_equipment ?seedling_batch ?special_equipment)
        )
      )
  )
  (:action assign_biosecurity_clearance_to_batch
    :parameters (?seedling_batch - seedling_batch ?biosecurity_clearance - biosecurity_clearance)
    :precondition
      (and
        (dispatch_order_confirmed_for_unit ?seedling_batch)
        (biosecurity_clearance_available ?biosecurity_clearance)
      )
    :effect
      (and
        (batch_assigned_biosecurity_clearance ?seedling_batch ?biosecurity_clearance)
        (not
          (biosecurity_clearance_available ?biosecurity_clearance)
        )
      )
  )
  (:action remove_biosecurity_clearance_from_batch
    :parameters (?seedling_batch - seedling_batch ?biosecurity_clearance - biosecurity_clearance)
    :precondition
      (and
        (batch_assigned_biosecurity_clearance ?seedling_batch ?biosecurity_clearance)
      )
    :effect
      (and
        (biosecurity_clearance_available ?biosecurity_clearance)
        (not
          (batch_assigned_biosecurity_clearance ?seedling_batch ?biosecurity_clearance)
        )
      )
  )
  (:action reserve_staging_slot_for_plot
    :parameters (?plot - plot ?staging_slot - staging_slot ?seedling_variety - seedling_variety)
    :precondition
      (and
        (dispatch_order_confirmed_for_unit ?plot)
        (variety_reserved ?plot ?seedling_variety)
        (plot_allocated_staging_slot ?plot ?staging_slot)
        (not
          (staging_slot_reserved ?staging_slot)
        )
        (not
          (staging_slot_loaded ?staging_slot)
        )
      )
    :effect (staging_slot_reserved ?staging_slot)
  )
  (:action operator_prepare_staging_slot_for_plot
    :parameters (?plot - plot ?staging_slot - staging_slot ?operator - operator)
    :precondition
      (and
        (dispatch_order_confirmed_for_unit ?plot)
        (operator_assigned_to_unit ?plot ?operator)
        (plot_allocated_staging_slot ?plot ?staging_slot)
        (staging_slot_reserved ?staging_slot)
        (not
          (plot_preparation_complete ?plot)
        )
      )
    :effect
      (and
        (plot_preparation_complete ?plot)
        (plot_ready_for_pallet_assembly ?plot)
      )
  )
  (:action allocate_consumable_to_plot
    :parameters (?plot - plot ?staging_slot - staging_slot ?consumable_batch - consumable_batch)
    :precondition
      (and
        (dispatch_order_confirmed_for_unit ?plot)
        (plot_allocated_staging_slot ?plot ?staging_slot)
        (consumable_available ?consumable_batch)
        (not
          (plot_preparation_complete ?plot)
        )
      )
    :effect
      (and
        (staging_slot_loaded ?staging_slot)
        (plot_preparation_complete ?plot)
        (plot_consumable_allocated ?plot ?consumable_batch)
        (not
          (consumable_available ?consumable_batch)
        )
      )
  )
  (:action finalize_slot_preparation
    :parameters (?plot - plot ?staging_slot - staging_slot ?seedling_variety - seedling_variety ?consumable_batch - consumable_batch)
    :precondition
      (and
        (dispatch_order_confirmed_for_unit ?plot)
        (variety_reserved ?plot ?seedling_variety)
        (plot_allocated_staging_slot ?plot ?staging_slot)
        (staging_slot_loaded ?staging_slot)
        (plot_consumable_allocated ?plot ?consumable_batch)
        (not
          (plot_ready_for_pallet_assembly ?plot)
        )
      )
    :effect
      (and
        (staging_slot_reserved ?staging_slot)
        (plot_ready_for_pallet_assembly ?plot)
        (consumable_available ?consumable_batch)
        (not
          (plot_consumable_allocated ?plot ?consumable_batch)
        )
      )
  )
  (:action reserve_transport_slot_for_subplot
    :parameters (?subplot - subplot ?transport_slot - transport_slot ?seedling_variety - seedling_variety)
    :precondition
      (and
        (dispatch_order_confirmed_for_unit ?subplot)
        (variety_reserved ?subplot ?seedling_variety)
        (subplot_allocated_transport_slot ?subplot ?transport_slot)
        (not
          (transport_slot_reserved ?transport_slot)
        )
        (not
          (transport_slot_loaded ?transport_slot)
        )
      )
    :effect (transport_slot_reserved ?transport_slot)
  )
  (:action operator_prepare_transport_slot_for_subplot
    :parameters (?subplot - subplot ?transport_slot - transport_slot ?operator - operator)
    :precondition
      (and
        (dispatch_order_confirmed_for_unit ?subplot)
        (operator_assigned_to_unit ?subplot ?operator)
        (subplot_allocated_transport_slot ?subplot ?transport_slot)
        (transport_slot_reserved ?transport_slot)
        (not
          (subplot_preparation_complete ?subplot)
        )
      )
    :effect
      (and
        (subplot_preparation_complete ?subplot)
        (subplot_ready_for_pallet_assembly ?subplot)
      )
  )
  (:action allocate_consumable_to_subplot
    :parameters (?subplot - subplot ?transport_slot - transport_slot ?consumable_batch - consumable_batch)
    :precondition
      (and
        (dispatch_order_confirmed_for_unit ?subplot)
        (subplot_allocated_transport_slot ?subplot ?transport_slot)
        (consumable_available ?consumable_batch)
        (not
          (subplot_preparation_complete ?subplot)
        )
      )
    :effect
      (and
        (transport_slot_loaded ?transport_slot)
        (subplot_preparation_complete ?subplot)
        (subplot_consumable_allocated ?subplot ?consumable_batch)
        (not
          (consumable_available ?consumable_batch)
        )
      )
  )
  (:action finalize_subplot_preparation
    :parameters (?subplot - subplot ?transport_slot - transport_slot ?seedling_variety - seedling_variety ?consumable_batch - consumable_batch)
    :precondition
      (and
        (dispatch_order_confirmed_for_unit ?subplot)
        (variety_reserved ?subplot ?seedling_variety)
        (subplot_allocated_transport_slot ?subplot ?transport_slot)
        (transport_slot_loaded ?transport_slot)
        (subplot_consumable_allocated ?subplot ?consumable_batch)
        (not
          (subplot_ready_for_pallet_assembly ?subplot)
        )
      )
    :effect
      (and
        (transport_slot_reserved ?transport_slot)
        (subplot_ready_for_pallet_assembly ?subplot)
        (consumable_available ?consumable_batch)
        (not
          (subplot_consumable_allocated ?subplot ?consumable_batch)
        )
      )
  )
  (:action assemble_dispatch_pallet_from_reserved_slots
    :parameters (?plot - plot ?subplot - subplot ?staging_slot - staging_slot ?transport_slot - transport_slot ?dispatch_pallet - dispatch_pallet)
    :precondition
      (and
        (plot_preparation_complete ?plot)
        (subplot_preparation_complete ?subplot)
        (plot_allocated_staging_slot ?plot ?staging_slot)
        (subplot_allocated_transport_slot ?subplot ?transport_slot)
        (staging_slot_reserved ?staging_slot)
        (transport_slot_reserved ?transport_slot)
        (plot_ready_for_pallet_assembly ?plot)
        (subplot_ready_for_pallet_assembly ?subplot)
        (dispatch_pallet_available ?dispatch_pallet)
      )
    :effect
      (and
        (dispatch_pallet_prepared ?dispatch_pallet)
        (dispatch_pallet_allocated_staging_slot ?dispatch_pallet ?staging_slot)
        (dispatch_pallet_allocated_transport_slot ?dispatch_pallet ?transport_slot)
        (not
          (dispatch_pallet_available ?dispatch_pallet)
        )
      )
  )
  (:action assemble_dispatch_pallet_from_staging_loaded
    :parameters (?plot - plot ?subplot - subplot ?staging_slot - staging_slot ?transport_slot - transport_slot ?dispatch_pallet - dispatch_pallet)
    :precondition
      (and
        (plot_preparation_complete ?plot)
        (subplot_preparation_complete ?subplot)
        (plot_allocated_staging_slot ?plot ?staging_slot)
        (subplot_allocated_transport_slot ?subplot ?transport_slot)
        (staging_slot_loaded ?staging_slot)
        (transport_slot_reserved ?transport_slot)
        (not
          (plot_ready_for_pallet_assembly ?plot)
        )
        (subplot_ready_for_pallet_assembly ?subplot)
        (dispatch_pallet_available ?dispatch_pallet)
      )
    :effect
      (and
        (dispatch_pallet_prepared ?dispatch_pallet)
        (dispatch_pallet_allocated_staging_slot ?dispatch_pallet ?staging_slot)
        (dispatch_pallet_allocated_transport_slot ?dispatch_pallet ?transport_slot)
        (dispatch_pallet_tag_staging_loaded ?dispatch_pallet)
        (not
          (dispatch_pallet_available ?dispatch_pallet)
        )
      )
  )
  (:action assemble_dispatch_pallet_from_transport_loaded
    :parameters (?plot - plot ?subplot - subplot ?staging_slot - staging_slot ?transport_slot - transport_slot ?dispatch_pallet - dispatch_pallet)
    :precondition
      (and
        (plot_preparation_complete ?plot)
        (subplot_preparation_complete ?subplot)
        (plot_allocated_staging_slot ?plot ?staging_slot)
        (subplot_allocated_transport_slot ?subplot ?transport_slot)
        (staging_slot_reserved ?staging_slot)
        (transport_slot_loaded ?transport_slot)
        (plot_ready_for_pallet_assembly ?plot)
        (not
          (subplot_ready_for_pallet_assembly ?subplot)
        )
        (dispatch_pallet_available ?dispatch_pallet)
      )
    :effect
      (and
        (dispatch_pallet_prepared ?dispatch_pallet)
        (dispatch_pallet_allocated_staging_slot ?dispatch_pallet ?staging_slot)
        (dispatch_pallet_allocated_transport_slot ?dispatch_pallet ?transport_slot)
        (dispatch_pallet_tag_transport_loaded ?dispatch_pallet)
        (not
          (dispatch_pallet_available ?dispatch_pallet)
        )
      )
  )
  (:action assemble_dispatch_pallet_from_both_loaded
    :parameters (?plot - plot ?subplot - subplot ?staging_slot - staging_slot ?transport_slot - transport_slot ?dispatch_pallet - dispatch_pallet)
    :precondition
      (and
        (plot_preparation_complete ?plot)
        (subplot_preparation_complete ?subplot)
        (plot_allocated_staging_slot ?plot ?staging_slot)
        (subplot_allocated_transport_slot ?subplot ?transport_slot)
        (staging_slot_loaded ?staging_slot)
        (transport_slot_loaded ?transport_slot)
        (not
          (plot_ready_for_pallet_assembly ?plot)
        )
        (not
          (subplot_ready_for_pallet_assembly ?subplot)
        )
        (dispatch_pallet_available ?dispatch_pallet)
      )
    :effect
      (and
        (dispatch_pallet_prepared ?dispatch_pallet)
        (dispatch_pallet_allocated_staging_slot ?dispatch_pallet ?staging_slot)
        (dispatch_pallet_allocated_transport_slot ?dispatch_pallet ?transport_slot)
        (dispatch_pallet_tag_staging_loaded ?dispatch_pallet)
        (dispatch_pallet_tag_transport_loaded ?dispatch_pallet)
        (not
          (dispatch_pallet_available ?dispatch_pallet)
        )
      )
  )
  (:action validate_pallet_for_plot
    :parameters (?dispatch_pallet - dispatch_pallet ?plot - plot ?seedling_variety - seedling_variety)
    :precondition
      (and
        (dispatch_pallet_prepared ?dispatch_pallet)
        (plot_preparation_complete ?plot)
        (variety_reserved ?plot ?seedling_variety)
        (not
          (dispatch_pallet_validated_for_plot ?dispatch_pallet)
        )
      )
    :effect (dispatch_pallet_validated_for_plot ?dispatch_pallet)
  )
  (:action load_and_seal_container_into_pallet
    :parameters (?seedling_batch - seedling_batch ?container - container ?dispatch_pallet - dispatch_pallet)
    :precondition
      (and
        (dispatch_order_confirmed_for_unit ?seedling_batch)
        (batch_loaded_to_pallet ?seedling_batch ?dispatch_pallet)
        (batch_assigned_to_container ?seedling_batch ?container)
        (container_available ?container)
        (dispatch_pallet_prepared ?dispatch_pallet)
        (dispatch_pallet_validated_for_plot ?dispatch_pallet)
        (not
          (container_sealed ?container)
        )
      )
    :effect
      (and
        (container_sealed ?container)
        (container_loaded_to_pallet ?container ?dispatch_pallet)
        (not
          (container_available ?container)
        )
      )
  )
  (:action mark_batch_packaging_ready
    :parameters (?seedling_batch - seedling_batch ?container - container ?dispatch_pallet - dispatch_pallet ?seedling_variety - seedling_variety)
    :precondition
      (and
        (dispatch_order_confirmed_for_unit ?seedling_batch)
        (batch_assigned_to_container ?seedling_batch ?container)
        (container_sealed ?container)
        (container_loaded_to_pallet ?container ?dispatch_pallet)
        (variety_reserved ?seedling_batch ?seedling_variety)
        (not
          (dispatch_pallet_tag_staging_loaded ?dispatch_pallet)
        )
        (not
          (batch_packaging_ready ?seedling_batch)
        )
      )
    :effect (batch_packaging_ready ?seedling_batch)
  )
  (:action attach_label_to_batch
    :parameters (?seedling_batch - seedling_batch ?label - label)
    :precondition
      (and
        (dispatch_order_confirmed_for_unit ?seedling_batch)
        (label_available ?label)
        (not
          (label_attached ?seedling_batch)
        )
      )
    :effect
      (and
        (label_attached ?seedling_batch)
        (batch_assigned_label ?seedling_batch ?label)
        (not
          (label_available ?label)
        )
      )
  )
  (:action apply_label_and_mark_batch
    :parameters (?seedling_batch - seedling_batch ?container - container ?dispatch_pallet - dispatch_pallet ?seedling_variety - seedling_variety ?label - label)
    :precondition
      (and
        (dispatch_order_confirmed_for_unit ?seedling_batch)
        (batch_assigned_to_container ?seedling_batch ?container)
        (container_sealed ?container)
        (container_loaded_to_pallet ?container ?dispatch_pallet)
        (variety_reserved ?seedling_batch ?seedling_variety)
        (dispatch_pallet_tag_staging_loaded ?dispatch_pallet)
        (label_attached ?seedling_batch)
        (batch_assigned_label ?seedling_batch ?label)
        (not
          (batch_packaging_ready ?seedling_batch)
        )
      )
    :effect
      (and
        (batch_packaging_ready ?seedling_batch)
        (label_verified ?seedling_batch)
      )
  )
  (:action attach_equipment_to_batch
    :parameters (?seedling_batch - seedling_batch ?special_equipment - special_equipment ?operator - operator ?container - container ?dispatch_pallet - dispatch_pallet)
    :precondition
      (and
        (batch_packaging_ready ?seedling_batch)
        (batch_assigned_special_equipment ?seedling_batch ?special_equipment)
        (operator_assigned_to_unit ?seedling_batch ?operator)
        (batch_assigned_to_container ?seedling_batch ?container)
        (container_loaded_to_pallet ?container ?dispatch_pallet)
        (not
          (dispatch_pallet_tag_transport_loaded ?dispatch_pallet)
        )
        (not
          (batch_equipment_attached ?seedling_batch)
        )
      )
    :effect (batch_equipment_attached ?seedling_batch)
  )
  (:action attach_equipment_to_batch_when_pallet_tagged
    :parameters (?seedling_batch - seedling_batch ?special_equipment - special_equipment ?operator - operator ?container - container ?dispatch_pallet - dispatch_pallet)
    :precondition
      (and
        (batch_packaging_ready ?seedling_batch)
        (batch_assigned_special_equipment ?seedling_batch ?special_equipment)
        (operator_assigned_to_unit ?seedling_batch ?operator)
        (batch_assigned_to_container ?seedling_batch ?container)
        (container_loaded_to_pallet ?container ?dispatch_pallet)
