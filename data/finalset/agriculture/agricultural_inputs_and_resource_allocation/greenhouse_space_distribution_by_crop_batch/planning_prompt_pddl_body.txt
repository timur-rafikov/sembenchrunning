  (:action register_crop_batch
    :parameters (?crop_batch - crop_batch)
    :precondition
      (and
        (not
          (allocation_registered ?crop_batch)
        )
        (not
          (allocation_finalized ?crop_batch)
        )
      )
    :effect (allocation_registered ?crop_batch)
  )
  (:action reserve_space_slot_for_batch
    :parameters (?crop_batch - crop_batch ?space_slot - space_slot)
    :precondition
      (and
        (allocation_registered ?crop_batch)
        (not
          (allocation_slot_reserved ?crop_batch)
        )
        (space_slot_available ?space_slot)
      )
    :effect
      (and
        (allocation_slot_reserved ?crop_batch)
        (allocation_slot_binding ?crop_batch ?space_slot)
        (not
          (space_slot_available ?space_slot)
        )
      )
  )
  (:action assign_variety_to_batch
    :parameters (?crop_batch - crop_batch ?crop_variety - crop_variety)
    :precondition
      (and
        (allocation_registered ?crop_batch)
        (allocation_slot_reserved ?crop_batch)
        (variety_available ?crop_variety)
      )
    :effect
      (and
        (allocation_variety_binding ?crop_batch ?crop_variety)
        (not
          (variety_available ?crop_variety)
        )
      )
  )
  (:action confirm_variety_for_batch
    :parameters (?crop_batch - crop_batch ?crop_variety - crop_variety)
    :precondition
      (and
        (allocation_registered ?crop_batch)
        (allocation_slot_reserved ?crop_batch)
        (allocation_variety_binding ?crop_batch ?crop_variety)
        (not
          (allocation_confirmed ?crop_batch)
        )
      )
    :effect (allocation_confirmed ?crop_batch)
  )
  (:action rollback_variety_assignment_for_batch
    :parameters (?crop_batch - crop_batch ?crop_variety - crop_variety)
    :precondition
      (and
        (allocation_variety_binding ?crop_batch ?crop_variety)
      )
    :effect
      (and
        (variety_available ?crop_variety)
        (not
          (allocation_variety_binding ?crop_batch ?crop_variety)
        )
      )
  )
  (:action reserve_equipment_for_batch
    :parameters (?crop_batch - crop_batch ?equipment - equipment)
    :precondition
      (and
        (allocation_confirmed ?crop_batch)
        (equipment_available ?equipment)
      )
    :effect
      (and
        (allocation_equipment_binding ?crop_batch ?equipment)
        (not
          (equipment_available ?equipment)
        )
      )
  )
  (:action release_equipment_from_batch
    :parameters (?crop_batch - crop_batch ?equipment - equipment)
    :precondition
      (and
        (allocation_equipment_binding ?crop_batch ?equipment)
      )
    :effect
      (and
        (equipment_available ?equipment)
        (not
          (allocation_equipment_binding ?crop_batch ?equipment)
        )
      )
  )
  (:action provision_nutrient_to_facility
    :parameters (?greenhouse_facility - greenhouse_facility ?nutrient_solution - nutrient_solution)
    :precondition
      (and
        (allocation_confirmed ?greenhouse_facility)
        (nutrient_solution_available ?nutrient_solution)
      )
    :effect
      (and
        (facility_nutrient_binding ?greenhouse_facility ?nutrient_solution)
        (not
          (nutrient_solution_available ?nutrient_solution)
        )
      )
  )
  (:action revoke_nutrient_from_facility
    :parameters (?greenhouse_facility - greenhouse_facility ?nutrient_solution - nutrient_solution)

      (and
        (facility_nutrient_binding ?greenhouse_facility ?nutrient_solution)
      )

      (and
        (nutrient_solution_available ?nutrient_solution)

          (facility_nutrient_binding ?greenhouse_facility ?nutrient_solution)
        )
      )
  )

  (:action revoke_climate_buffer_from_facility
    :parameters (?greenhouse_facility - greenhouse_facility ?climate_buffer - climate_buffer)

      (and
        (facility_climate_buffer_binding ?greenhouse_facility ?climate_buffer)
      )
    :effect
      (and
        (climate_buffer_available ?climate_buffer)
        (not
          (facility_climate_buffer_binding ?greenhouse_facility ?climate_buffer)
        )
      )
  )
  (:action engage_irrigation_profile_for_bench
    :parameters (?bench_batch - bench_batch ?irrigation_profile - irrigation_profile ?crop_variety - crop_variety)
    :precondition
      (and
        (allocation_confirmed ?bench_batch)
        (allocation_variety_binding ?bench_batch ?crop_variety)
        (bench_irrigation_binding ?bench_batch ?irrigation_profile)
        (not
          (irrigation_profile_engaged ?irrigation_profile)
        )
        (not
          (irrigation_profile_prepared ?irrigation_profile)
        )
      )
    :effect (irrigation_profile_engaged ?irrigation_profile)
  )
  (:action prepare_bench_with_equipment
    :parameters (?bench_batch - bench_batch ?irrigation_profile - irrigation_profile ?equipment - equipment)
    :precondition
      (and
        (allocation_confirmed ?bench_batch)
        (allocation_equipment_binding ?bench_batch ?equipment)
        (bench_irrigation_binding ?bench_batch ?irrigation_profile)
        (irrigation_profile_engaged ?irrigation_profile)
        (not
          (bench_prepared ?bench_batch)
        )
      )
    :effect
      (and
        (bench_prepared ?bench_batch)
        (bench_ready_for_provisioning ?bench_batch)
      )
  )
  (:action prepare_bench_with_consumable
    :parameters (?bench_batch - bench_batch ?irrigation_profile - irrigation_profile ?consumable - consumable)
    :precondition
      (and
        (allocation_confirmed ?bench_batch)
        (bench_irrigation_binding ?bench_batch ?irrigation_profile)
        (consumable_available ?consumable)
        (not
          (bench_prepared ?bench_batch)
        )
      )
    :effect
      (and
        (irrigation_profile_prepared ?irrigation_profile)
        (bench_prepared ?bench_batch)
        (bench_consumable_binding ?bench_batch ?consumable)
        (not
          (consumable_available ?consumable)
        )
      )
  )
  (:action finalize_bench_preparation
    :parameters (?bench_batch - bench_batch ?irrigation_profile - irrigation_profile ?crop_variety - crop_variety ?consumable - consumable)
    :precondition
      (and
        (allocation_confirmed ?bench_batch)
        (allocation_variety_binding ?bench_batch ?crop_variety)
        (bench_irrigation_binding ?bench_batch ?irrigation_profile)
        (irrigation_profile_prepared ?irrigation_profile)
        (bench_consumable_binding ?bench_batch ?consumable)
        (not
          (bench_ready_for_provisioning ?bench_batch)
        )
      )
    :effect
      (and
        (irrigation_profile_engaged ?irrigation_profile)
        (bench_ready_for_provisioning ?bench_batch)
        (consumable_available ?consumable)
        (not
          (bench_consumable_binding ?bench_batch ?consumable)
        )
      )
  )

    :parameters (?rack_batch - rack_batch ?lighting_profile - lighting_profile ?crop_variety - crop_variety)

        (allocation_confirmed ?rack_batch)

          (lighting_profile_engaged ?lighting_profile)

  )

          (consumable_available ?consumable)
        )

  )
  (:action finalize_rack_preparation
    :parameters (?rack_batch - rack_batch ?lighting_profile - lighting_profile ?crop_variety - crop_variety ?consumable - consumable)
    :precondition
      (and
        (allocation_confirmed ?rack_batch)

        (lighting_profile_prepared ?lighting_profile)

        )

    :effect
      (and
        (lighting_profile_engaged ?lighting_profile)

          (rack_consumable_binding ?rack_batch ?consumable)
        )
      )
  )
  (:action provision_production_bay_with_engaged_profiles
    :parameters (?bench_batch - bench_batch ?rack_batch - rack_batch ?irrigation_profile - irrigation_profile ?lighting_profile - lighting_profile ?production_bay - production_bay)
    :precondition
      (and
        (bench_prepared ?bench_batch)
        (rack_prepared ?rack_batch)
        (bench_irrigation_binding ?bench_batch ?irrigation_profile)
        (rack_lighting_binding ?rack_batch ?lighting_profile)
        (irrigation_profile_engaged ?irrigation_profile)
        (lighting_profile_engaged ?lighting_profile)
        (bench_ready_for_provisioning ?bench_batch)
        (rack_ready_for_provisioning ?rack_batch)
        (production_bay_available ?production_bay)
      )
    :effect
      (and
        (production_bay_provisioned ?production_bay)
        (bay_irrigation_binding ?production_bay ?irrigation_profile)
        (bay_lighting_binding ?production_bay ?lighting_profile)
        (not
          (production_bay_available ?production_bay)
        )
      )
  )
  (:action provision_production_bay_with_prepared_irrigation
    :parameters (?bench_batch - bench_batch ?rack_batch - rack_batch ?irrigation_profile - irrigation_profile ?lighting_profile - lighting_profile ?production_bay - production_bay)
    :precondition
      (and
        (bench_prepared ?bench_batch)
        (rack_prepared ?rack_batch)
        (bench_irrigation_binding ?bench_batch ?irrigation_profile)
        (rack_lighting_binding ?rack_batch ?lighting_profile)
        (irrigation_profile_prepared ?irrigation_profile)
        (lighting_profile_engaged ?lighting_profile)
        (not
          (bench_ready_for_provisioning ?bench_batch)
        )
        (rack_ready_for_provisioning ?rack_batch)
        (production_bay_available ?production_bay)
      )
    :effect
      (and
        (production_bay_provisioned ?production_bay)
        (bay_irrigation_binding ?production_bay ?irrigation_profile)
        (bay_lighting_binding ?production_bay ?lighting_profile)
        (bay_irrigation_configured ?production_bay)
        (not
          (production_bay_available ?production_bay)
        )
      )
  )
  (:action provision_production_bay_with_prepared_lighting
    :parameters (?bench_batch - bench_batch ?rack_batch - rack_batch ?irrigation_profile - irrigation_profile ?lighting_profile - lighting_profile ?production_bay - production_bay)
    :precondition
      (and
        (bench_prepared ?bench_batch)
        (rack_prepared ?rack_batch)
        (bench_irrigation_binding ?bench_batch ?irrigation_profile)
        (rack_lighting_binding ?rack_batch ?lighting_profile)
        (irrigation_profile_engaged ?irrigation_profile)
        (lighting_profile_prepared ?lighting_profile)
        (bench_ready_for_provisioning ?bench_batch)
        (not
          (rack_ready_for_provisioning ?rack_batch)
        )
        (production_bay_available ?production_bay)
      )
    :effect
      (and
        (production_bay_provisioned ?production_bay)

        (bay_lighting_binding ?production_bay ?lighting_profile)
        (bay_lighting_configured ?production_bay)
        (not
          (production_bay_available ?production_bay)
        )
      )
  )
  (:action provision_production_bay_with_all_profiles_prepared
    :parameters (?bench_batch - bench_batch ?rack_batch - rack_batch ?irrigation_profile - irrigation_profile ?lighting_profile - lighting_profile ?production_bay - production_bay)
    :precondition
      (and
        (bench_prepared ?bench_batch)
        (rack_prepared ?rack_batch)
        (bench_irrigation_binding ?bench_batch ?irrigation_profile)
        (rack_lighting_binding ?rack_batch ?lighting_profile)

          (bench_ready_for_provisioning ?bench_batch)

          (rack_ready_for_provisioning ?rack_batch)
        )
        (production_bay_available ?production_bay)

    :effect
      (and
        (production_bay_provisioned ?production_bay)

        (bay_lighting_binding ?production_bay ?lighting_profile)
        (bay_irrigation_configured ?production_bay)
        (bay_lighting_configured ?production_bay)
        (not
          (production_bay_available ?production_bay)
        )
      )
  )
  (:action mark_bay_ready_for_substrate_loading
    :parameters (?production_bay - production_bay ?bench_batch - bench_batch ?crop_variety - crop_variety)
    :precondition
      (and
        (production_bay_provisioned ?production_bay)
        (bench_prepared ?bench_batch)
        (allocation_variety_binding ?bench_batch ?crop_variety)
        (not
          (bay_ready_for_substrate_loading ?production_bay)
        )
      )
    :effect (bay_ready_for_substrate_loading ?production_bay)
  )
  (:action load_substrate_into_bay
    :parameters (?greenhouse_facility - greenhouse_facility ?substrate_mix - substrate_mix ?production_bay - production_bay)
    :precondition
      (and
        (allocation_confirmed ?greenhouse_facility)
        (facility_bay_binding ?greenhouse_facility ?production_bay)
        (facility_substrate_binding ?greenhouse_facility ?substrate_mix)
        (substrate_mix_available ?substrate_mix)
        (production_bay_provisioned ?production_bay)
        (bay_ready_for_substrate_loading ?production_bay)
        (not
          (substrate_loaded ?substrate_mix)
        )
      )
    :effect
      (and
        (substrate_loaded ?substrate_mix)
        (substrate_bound_to_bay ?substrate_mix ?production_bay)
        (not
          (substrate_mix_available ?substrate_mix)
        )
      )
  )

      (and
        (allocation_confirmed ?greenhouse_facility)

        (allocation_variety_binding ?greenhouse_facility ?crop_variety)
        (not
          (bay_irrigation_configured ?production_bay)
        )
        (not
          (facility_substrate_assigned ?greenhouse_facility)
        )
      )
    :effect (facility_substrate_assigned ?greenhouse_facility)
  )

    :precondition
      (and
        (allocation_confirmed ?greenhouse_facility)
        (permit_available ?permit)

          (facility_permit_attached ?greenhouse_facility)

      )
    :effect

        (facility_permit_attached ?greenhouse_facility)

          (permit_available ?permit)
        )
      )
  )
  (:action apply_permit_and_advance_compliance_stage

    :precondition
      (and
        (allocation_confirmed ?greenhouse_facility)
        (facility_substrate_binding ?greenhouse_facility ?substrate_mix)
        (substrate_loaded ?substrate_mix)
        (substrate_bound_to_bay ?substrate_mix ?production_bay)
        (allocation_variety_binding ?greenhouse_facility ?crop_variety)
        (bay_irrigation_configured ?production_bay)
        (facility_permit_attached ?greenhouse_facility)
        (facility_permit_binding ?greenhouse_facility ?permit)
        (not
          (facility_substrate_assigned ?greenhouse_facility)
        )
      )
    :effect
      (and
        (facility_substrate_assigned ?greenhouse_facility)
        (facility_permit_stage_completed ?greenhouse_facility)
      )
  )
  (:action assign_nutrient_and_confirm_facility
    :parameters (?greenhouse_facility - greenhouse_facility ?nutrient_solution - nutrient_solution ?equipment - equipment ?substrate_mix - substrate_mix ?production_bay - production_bay)
    :precondition
      (and
        (facility_substrate_assigned ?greenhouse_facility)
        (facility_nutrient_binding ?greenhouse_facility ?nutrient_solution)
        (allocation_equipment_binding ?greenhouse_facility ?equipment)
        (facility_substrate_binding ?greenhouse_facility ?substrate_mix)
        (substrate_bound_to_bay ?substrate_mix ?production_bay)
        (not
          (bay_lighting_configured ?production_bay)
        )
        (not
          (facility_nutrient_assigned ?greenhouse_facility)
        )
      )
    :effect (facility_nutrient_assigned ?greenhouse_facility)
  )
  (:action reconfirm_nutrient_assignment_to_facility
    :parameters (?greenhouse_facility - greenhouse_facility ?nutrient_solution - nutrient_solution ?equipment - equipment ?substrate_mix - substrate_mix ?production_bay - production_bay)
    :precondition
      (and
        (facility_substrate_assigned ?greenhouse_facility)
        (facility_nutrient_binding ?greenhouse_facility ?nutrient_solution)
