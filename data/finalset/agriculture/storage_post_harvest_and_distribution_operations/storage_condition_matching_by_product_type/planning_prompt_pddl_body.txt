    (shipment_flag_channel_profile_present ?shipment_unit - shipment_unit)
    (shipment_staged_for_container_loading ?shipment_unit - shipment_unit)

  (:action register_product_intake
    :parameters (?product_unit - product_unit)
    :precondition
      (and
        (not
          (received_at_intake ?product_unit)
        )
        (not
          (release_scheduled ?product_unit)
        )
      )
    :effect (received_at_intake ?product_unit)
  )
  (:action assign_grading_station
    :parameters (?product_unit - product_unit ?grading_station - grading_station)
    :precondition
      (and
        (received_at_intake ?product_unit)
        (not
          (assigned_to_grading ?product_unit)
        )
        (grading_station_available ?grading_station)
      )
    :effect
      (and
        (assigned_to_grading ?product_unit)
        (assigned_to_grading_station ?product_unit ?grading_station)
        (not
          (grading_station_available ?grading_station)
        )
      )
  )
  (:action assign_quality_test_kit
    :parameters (?product_unit - product_unit ?quality_test_kit - quality_test_kit)
    :precondition
      (and
        (received_at_intake ?product_unit)
        (assigned_to_grading ?product_unit)
        (test_kit_available ?quality_test_kit)
      )
    :effect
      (and
        (allocated_test_kit_for_item ?product_unit ?quality_test_kit)
        (not
          (test_kit_available ?quality_test_kit)
        )
      )
  )
  (:action finalize_quality_test
    :parameters (?product_unit - product_unit ?quality_test_kit - quality_test_kit)
    :precondition
      (and
        (received_at_intake ?product_unit)
        (assigned_to_grading ?product_unit)
        (allocated_test_kit_for_item ?product_unit ?quality_test_kit)
        (not
          (quality_approved_item ?product_unit)
        )
      )
    :effect (quality_approved_item ?product_unit)
  )
  (:action release_quality_test_kit
    :parameters (?product_unit - product_unit ?quality_test_kit - quality_test_kit)
    :precondition
      (and
        (allocated_test_kit_for_item ?product_unit ?quality_test_kit)
      )
    :effect
      (and
        (test_kit_available ?quality_test_kit)
        (not
          (allocated_test_kit_for_item ?product_unit ?quality_test_kit)
        )
      )
  )
  (:action assign_conditioning_unit
    :parameters (?product_unit - product_unit ?conditioning_unit - conditioning_unit)

        (quality_approved_item ?product_unit)
        (conditioning_unit_available ?conditioning_unit)

          (conditioning_unit_available ?conditioning_unit)

      )

        (assigned_to_conditioning_unit ?product_unit ?conditioning_unit)
      )
    :effect
      (and
        (conditioning_unit_available ?conditioning_unit)
        (not
          (assigned_to_conditioning_unit ?product_unit ?conditioning_unit)
        )
      )
  )

        (container_assigned_processing_instruction ?storage_container ?processing_instruction)
      )
    :effect
      (and
        (processing_instruction_available ?processing_instruction)

    :parameters (?perishable_batch - perishable_batch ?temperature_profile - temperature_profile ?quality_test_kit - quality_test_kit)

        (allocated_test_kit_for_item ?perishable_batch ?quality_test_kit)
        (assigned_temperature_profile ?perishable_batch ?temperature_profile)
        (not
          (temperature_profile_candidate ?temperature_profile)
        )
        (not
          (temperature_profile_reserved ?temperature_profile)
        )

  )
  (:action confirm_temperature_profile_for_batch
    :parameters (?perishable_batch - perishable_batch ?temperature_profile - temperature_profile ?conditioning_unit - conditioning_unit)
    :precondition
      (and
        (quality_approved_item ?perishable_batch)
        (assigned_to_conditioning_unit ?perishable_batch ?conditioning_unit)
        (assigned_temperature_profile ?perishable_batch ?temperature_profile)
        (temperature_profile_candidate ?temperature_profile)
        (not
          (batch_ready_for_shipment ?perishable_batch)
        )

    :effect
      (and
        (batch_ready_for_shipment ?perishable_batch)

          (batch_ready_for_shipment ?perishable_batch)

    :effect
      (and
        (temperature_profile_reserved ?temperature_profile)
        (batch_ready_for_shipment ?perishable_batch)

          (label_attached_to_perishable_batch ?perishable_batch ?label_or_tag)
