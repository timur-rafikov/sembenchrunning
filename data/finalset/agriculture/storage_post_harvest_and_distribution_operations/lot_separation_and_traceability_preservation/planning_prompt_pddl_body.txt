## Planning domain (PDDL)

(define (domain agriculture_lot_separation_traceability)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_type - object label_or_tag_type - object logistics_unit_type - object core_lot_type - object lot - core_lot_type reception_slot - resource_type grading_station - resource_type conditioning_station - resource_type customer_order - resource_type carrier_service - resource_type release_document - resource_type temperature_profile - resource_type lab_test_result - resource_type traceability_tag - label_or_tag_type packaging_box - label_or_tag_type origin_certificate - label_or_tag_type storage_cell - logistics_unit_type distribution_channel - logistics_unit_type outbound_load - logistics_unit_type storage_view - lot deployment_view - lot storage_batch - storage_view picking_batch - storage_view pallet - deployment_view)
  (:predicates
    (received ?lot - lot)
    (grade_completed ?lot - lot)
    (staged ?lot - lot)
    (released_for_dispatch ?lot - lot)
    (ready_for_dispatch ?lot - lot)
    (release_document_attached ?lot - lot)
    (reception_slot_available ?reception_slot - reception_slot)
    (in_reception_slot ?lot - lot ?reception_slot - reception_slot)
    (grading_station_available ?grading_station - grading_station)
    (assigned_to_grading_station ?lot - lot ?grading_station - grading_station)
    (conditioning_station_available ?conditioning_station - conditioning_station)
    (assigned_to_conditioning_station ?lot - lot ?conditioning_station - conditioning_station)
    (traceability_tag_available ?traceability_tag - traceability_tag)
    (storage_batch_assigned_traceability_tag ?storage_batch - storage_batch ?traceability_tag - traceability_tag)
    (picking_batch_assigned_traceability_tag ?picking_batch - picking_batch ?traceability_tag - traceability_tag)
    (storage_batch_assigned_to_storage_cell ?storage_batch - storage_batch ?storage_cell - storage_cell)
    (storage_cell_reserved ?storage_cell - storage_cell)
    (storage_cell_tagged ?storage_cell - storage_cell)
    (storage_batch_available_for_outbound ?storage_batch - storage_batch)
    (picking_batch_assigned_to_distribution_channel ?picking_batch - picking_batch ?distribution_channel - distribution_channel)
    (distribution_channel_reserved ?distribution_channel - distribution_channel)
    (distribution_channel_tagged ?distribution_channel - distribution_channel)
    (picking_batch_available_for_outbound ?picking_batch - picking_batch)
    (outbound_load_available ?outbound_load - outbound_load)
    (outbound_load_reserved ?outbound_load - outbound_load)
    (outbound_load_linked_to_storage_cell ?outbound_load - outbound_load ?storage_cell - storage_cell)
    (outbound_load_linked_to_distribution_channel ?outbound_load - outbound_load ?distribution_channel - distribution_channel)
    (outbound_load_requires_temperature_profile ?outbound_load - outbound_load)
    (outbound_load_requires_origin_certificate ?outbound_load - outbound_load)
    (outbound_load_ready_for_packing ?outbound_load - outbound_load)
    (pallet_contains_storage_batch ?pallet - pallet ?storage_batch - storage_batch)
    (pallet_contains_picking_batch ?pallet - pallet ?picking_batch - picking_batch)
    (pallet_assigned_to_outbound_load ?pallet - pallet ?outbound_load - outbound_load)
    (packaging_box_available ?packaging_box - packaging_box)
    (pallet_has_packaging_box ?pallet - pallet ?packaging_box - packaging_box)
    (packaging_box_used ?packaging_box - packaging_box)
    (packaging_box_assigned_to_outbound_load ?packaging_box - packaging_box ?outbound_load - outbound_load)
    (pallet_packaging_initialized ?pallet - pallet)
    (pallet_packaging_confirmed ?pallet - pallet)
    (pallet_lab_result_linked ?pallet - pallet)
    (pallet_has_order_applied ?pallet - pallet)
    (pallet_order_confirmed ?pallet - pallet)
    (pallet_documentation_complete ?pallet - pallet)
    (pallet_final_inspection_passed ?pallet - pallet)
    (origin_certificate_available ?origin_certificate - origin_certificate)
    (pallet_assigned_origin_certificate ?pallet - pallet ?origin_certificate - origin_certificate)
    (pallet_origin_check_started ?pallet - pallet)
    (pallet_origin_validated ?pallet - pallet)
    (pallet_origin_verified ?pallet - pallet)
    (customer_order_available ?customer_order - customer_order)
    (pallet_linked_to_customer_order ?pallet - pallet ?customer_order - customer_order)
    (carrier_service_available ?carrier_service - carrier_service)
    (pallet_assigned_carrier_service ?pallet - pallet ?carrier_service - carrier_service)
    (temperature_profile_available ?temperature_profile - temperature_profile)
    (pallet_assigned_temperature_profile ?pallet - pallet ?temperature_profile - temperature_profile)
    (lab_test_result_available ?lab_test_result - lab_test_result)
    (pallet_assigned_lab_test_result ?pallet - pallet ?lab_test_result - lab_test_result)
    (release_document_available ?release_document - release_document)
    (linked_to_release_document ?lot - lot ?release_document - release_document)
    (storage_batch_placed ?storage_batch - storage_batch)
    (picking_batch_placed ?picking_batch - picking_batch)
    (pallet_ready_marker ?pallet - pallet)
  )
  (:action receive_and_register_lot
    :parameters (?lot - lot)
    :precondition
      (and
        (not
          (received ?lot)
        )
        (not
          (released_for_dispatch ?lot)
        )
      )
    :effect (received ?lot)
  )
  (:action assign_lot_to_reception_slot
    :parameters (?lot - lot ?reception_slot - reception_slot)
    :precondition
      (and
        (received ?lot)
        (not
          (staged ?lot)
        )
        (reception_slot_available ?reception_slot)
      )
    :effect
      (and
        (staged ?lot)
        (in_reception_slot ?lot ?reception_slot)
        (not
          (reception_slot_available ?reception_slot)
        )
      )
  )
  (:action assign_lot_to_grading_station
    :parameters (?lot - lot ?grading_station - grading_station)
    :precondition
      (and
        (received ?lot)
        (staged ?lot)
        (grading_station_available ?grading_station)
      )
    :effect
      (and
        (assigned_to_grading_station ?lot ?grading_station)
        (not
          (grading_station_available ?grading_station)
        )
      )
  )
  (:action complete_lot_grading
    :parameters (?lot - lot ?grading_station - grading_station)
    :precondition
      (and
        (received ?lot)
        (staged ?lot)
        (assigned_to_grading_station ?lot ?grading_station)
        (not
          (grade_completed ?lot)
        )
      )
    :effect (grade_completed ?lot)
  )
  (:action release_grading_station
    :parameters (?lot - lot ?grading_station - grading_station)
    :precondition
      (and
        (assigned_to_grading_station ?lot ?grading_station)
      )
    :effect
      (and
        (grading_station_available ?grading_station)
        (not
          (assigned_to_grading_station ?lot ?grading_station)
        )
      )
  )
  (:action assign_lot_to_conditioning_station
    :parameters (?lot - lot ?conditioning_station - conditioning_station)
    :precondition
      (and
        (grade_completed ?lot)
        (conditioning_station_available ?conditioning_station)
      )
    :effect
      (and
        (assigned_to_conditioning_station ?lot ?conditioning_station)
        (not
          (conditioning_station_available ?conditioning_station)
        )
      )
  )
  (:action release_conditioning_station
    :parameters (?lot - lot ?conditioning_station - conditioning_station)
    :precondition
      (and
        (assigned_to_conditioning_station ?lot ?conditioning_station)
      )
    :effect
      (and
        (conditioning_station_available ?conditioning_station)
        (not
          (assigned_to_conditioning_station ?lot ?conditioning_station)
        )
      )
  )
  (:action attach_temperature_profile_to_pallet
    :parameters (?pallet - pallet ?temperature_profile - temperature_profile)
    :precondition
      (and
        (grade_completed ?pallet)
        (temperature_profile_available ?temperature_profile)
      )
    :effect
      (and
        (pallet_assigned_temperature_profile ?pallet ?temperature_profile)
        (not
          (temperature_profile_available ?temperature_profile)
        )
      )
  )
  (:action detach_temperature_profile_from_pallet
    :parameters (?pallet - pallet ?temperature_profile - temperature_profile)
    :precondition
      (and
        (pallet_assigned_temperature_profile ?pallet ?temperature_profile)
      )
    :effect
      (and
        (temperature_profile_available ?temperature_profile)
        (not
          (pallet_assigned_temperature_profile ?pallet ?temperature_profile)
        )
      )
  )
  (:action attach_lab_test_result_to_pallet
    :parameters (?pallet - pallet ?lab_test_result - lab_test_result)
    :precondition
      (and
        (grade_completed ?pallet)
        (lab_test_result_available ?lab_test_result)
      )
    :effect
      (and
        (pallet_assigned_lab_test_result ?pallet ?lab_test_result)
        (not
          (lab_test_result_available ?lab_test_result)
        )
      )
  )
  (:action detach_lab_test_result_from_pallet
    :parameters (?pallet - pallet ?lab_test_result - lab_test_result)
    :precondition
      (and
        (pallet_assigned_lab_test_result ?pallet ?lab_test_result)
      )
    :effect
      (and
        (lab_test_result_available ?lab_test_result)
        (not
          (pallet_assigned_lab_test_result ?pallet ?lab_test_result)
        )
      )
  )
  (:action reserve_storage_cell_for_batch
    :parameters (?storage_batch - storage_batch ?storage_cell - storage_cell ?grading_station - grading_station)
    :precondition
      (and
        (grade_completed ?storage_batch)
        (assigned_to_grading_station ?storage_batch ?grading_station)
        (storage_batch_assigned_to_storage_cell ?storage_batch ?storage_cell)
        (not
          (storage_cell_reserved ?storage_cell)
        )
        (not
          (storage_cell_tagged ?storage_cell)
        )
      )
    :effect (storage_cell_reserved ?storage_cell)
  )
  (:action place_batch_into_storage_cell_and_mark
    :parameters (?storage_batch - storage_batch ?storage_cell - storage_cell ?conditioning_station - conditioning_station)
    :precondition
      (and
        (grade_completed ?storage_batch)
        (assigned_to_conditioning_station ?storage_batch ?conditioning_station)
        (storage_batch_assigned_to_storage_cell ?storage_batch ?storage_cell)
        (storage_cell_reserved ?storage_cell)
        (not
          (storage_batch_placed ?storage_batch)
        )
      )
    :effect
      (and
        (storage_batch_placed ?storage_batch)
        (storage_batch_available_for_outbound ?storage_batch)
      )
  )
  (:action assign_traceability_tag_to_storage_batch
    :parameters (?storage_batch - storage_batch ?storage_cell - storage_cell ?traceability_tag - traceability_tag)
    :precondition
      (and
        (grade_completed ?storage_batch)
        (storage_batch_assigned_to_storage_cell ?storage_batch ?storage_cell)
        (traceability_tag_available ?traceability_tag)
        (not
          (storage_batch_placed ?storage_batch)
        )
      )
    :effect
      (and
        (storage_cell_tagged ?storage_cell)
        (storage_batch_placed ?storage_batch)
        (storage_batch_assigned_traceability_tag ?storage_batch ?traceability_tag)
        (not
          (traceability_tag_available ?traceability_tag)
        )
      )
  )
  (:action confirm_tag_and_mark_storage_batch_available
    :parameters (?storage_batch - storage_batch ?storage_cell - storage_cell ?grading_station - grading_station ?traceability_tag - traceability_tag)
    :precondition
      (and
        (grade_completed ?storage_batch)
        (assigned_to_grading_station ?storage_batch ?grading_station)
        (storage_batch_assigned_to_storage_cell ?storage_batch ?storage_cell)
        (storage_cell_tagged ?storage_cell)
        (storage_batch_assigned_traceability_tag ?storage_batch ?traceability_tag)
        (not
          (storage_batch_available_for_outbound ?storage_batch)
        )
      )
    :effect
      (and
        (storage_cell_reserved ?storage_cell)
        (storage_batch_available_for_outbound ?storage_batch)
        (traceability_tag_available ?traceability_tag)
        (not
          (storage_batch_assigned_traceability_tag ?storage_batch ?traceability_tag)
        )
      )
  )
  (:action reserve_distribution_channel_for_picking_batch
    :parameters (?picking_batch - picking_batch ?distribution_channel - distribution_channel ?grading_station - grading_station)
    :precondition
      (and
        (grade_completed ?picking_batch)
        (assigned_to_grading_station ?picking_batch ?grading_station)
        (picking_batch_assigned_to_distribution_channel ?picking_batch ?distribution_channel)
        (not
          (distribution_channel_reserved ?distribution_channel)
        )
        (not
          (distribution_channel_tagged ?distribution_channel)
        )
      )
    :effect (distribution_channel_reserved ?distribution_channel)
  )
  (:action place_picking_batch_and_mark_available
    :parameters (?picking_batch - picking_batch ?distribution_channel - distribution_channel ?conditioning_station - conditioning_station)
    :precondition
      (and
        (grade_completed ?picking_batch)
        (assigned_to_conditioning_station ?picking_batch ?conditioning_station)
        (picking_batch_assigned_to_distribution_channel ?picking_batch ?distribution_channel)
        (distribution_channel_reserved ?distribution_channel)
        (not
          (picking_batch_placed ?picking_batch)
        )
      )
    :effect
      (and
        (picking_batch_placed ?picking_batch)
        (picking_batch_available_for_outbound ?picking_batch)
      )
  )
  (:action assign_traceability_tag_to_picking_batch
    :parameters (?picking_batch - picking_batch ?distribution_channel - distribution_channel ?traceability_tag - traceability_tag)
    :precondition
      (and
        (grade_completed ?picking_batch)
        (picking_batch_assigned_to_distribution_channel ?picking_batch ?distribution_channel)
        (traceability_tag_available ?traceability_tag)
        (not
          (picking_batch_placed ?picking_batch)
        )
      )
    :effect
      (and
        (distribution_channel_tagged ?distribution_channel)
        (picking_batch_placed ?picking_batch)
        (picking_batch_assigned_traceability_tag ?picking_batch ?traceability_tag)
        (not
          (traceability_tag_available ?traceability_tag)
        )
      )
  )
  (:action confirm_tag_and_mark_picking_batch_available
    :parameters (?picking_batch - picking_batch ?distribution_channel - distribution_channel ?grading_station - grading_station ?traceability_tag - traceability_tag)
    :precondition
      (and
        (grade_completed ?picking_batch)
        (assigned_to_grading_station ?picking_batch ?grading_station)
        (picking_batch_assigned_to_distribution_channel ?picking_batch ?distribution_channel)
        (distribution_channel_tagged ?distribution_channel)
        (picking_batch_assigned_traceability_tag ?picking_batch ?traceability_tag)
        (not
          (picking_batch_available_for_outbound ?picking_batch)
        )
      )
    :effect
      (and
        (distribution_channel_reserved ?distribution_channel)
        (picking_batch_available_for_outbound ?picking_batch)
        (traceability_tag_available ?traceability_tag)
        (not
          (picking_batch_assigned_traceability_tag ?picking_batch ?traceability_tag)
        )
      )
  )
  (:action initialize_outbound_load_reservation
    :parameters (?storage_batch - storage_batch ?picking_batch - picking_batch ?storage_cell - storage_cell ?distribution_channel - distribution_channel ?outbound_load - outbound_load)
    :precondition
      (and
        (storage_batch_placed ?storage_batch)
        (picking_batch_placed ?picking_batch)
        (storage_batch_assigned_to_storage_cell ?storage_batch ?storage_cell)
        (picking_batch_assigned_to_distribution_channel ?picking_batch ?distribution_channel)
        (storage_cell_reserved ?storage_cell)
        (distribution_channel_reserved ?distribution_channel)
        (storage_batch_available_for_outbound ?storage_batch)
        (picking_batch_available_for_outbound ?picking_batch)
        (outbound_load_available ?outbound_load)
      )
    :effect
      (and
        (outbound_load_reserved ?outbound_load)
        (outbound_load_linked_to_storage_cell ?outbound_load ?storage_cell)
        (outbound_load_linked_to_distribution_channel ?outbound_load ?distribution_channel)
        (not
          (outbound_load_available ?outbound_load)
        )
      )
  )
  (:action initialize_outbound_load_with_cell_tagged
    :parameters (?storage_batch - storage_batch ?picking_batch - picking_batch ?storage_cell - storage_cell ?distribution_channel - distribution_channel ?outbound_load - outbound_load)
    :precondition
      (and
        (storage_batch_placed ?storage_batch)
        (picking_batch_placed ?picking_batch)
        (storage_batch_assigned_to_storage_cell ?storage_batch ?storage_cell)
        (picking_batch_assigned_to_distribution_channel ?picking_batch ?distribution_channel)
        (storage_cell_tagged ?storage_cell)
        (distribution_channel_reserved ?distribution_channel)
        (not
          (storage_batch_available_for_outbound ?storage_batch)
        )
        (picking_batch_available_for_outbound ?picking_batch)
        (outbound_load_available ?outbound_load)
      )
    :effect
      (and
        (outbound_load_reserved ?outbound_load)
        (outbound_load_linked_to_storage_cell ?outbound_load ?storage_cell)
        (outbound_load_linked_to_distribution_channel ?outbound_load ?distribution_channel)
        (outbound_load_requires_temperature_profile ?outbound_load)
        (not
          (outbound_load_available ?outbound_load)
        )
      )
  )
  (:action initialize_outbound_load_with_cell_reserved
    :parameters (?storage_batch - storage_batch ?picking_batch - picking_batch ?storage_cell - storage_cell ?distribution_channel - distribution_channel ?outbound_load - outbound_load)
    :precondition
      (and
        (storage_batch_placed ?storage_batch)
        (picking_batch_placed ?picking_batch)
        (storage_batch_assigned_to_storage_cell ?storage_batch ?storage_cell)
        (picking_batch_assigned_to_distribution_channel ?picking_batch ?distribution_channel)
        (storage_cell_reserved ?storage_cell)
        (distribution_channel_tagged ?distribution_channel)
        (storage_batch_available_for_outbound ?storage_batch)
        (not
          (picking_batch_available_for_outbound ?picking_batch)
        )
        (outbound_load_available ?outbound_load)
      )
    :effect
      (and
        (outbound_load_reserved ?outbound_load)
        (outbound_load_linked_to_storage_cell ?outbound_load ?storage_cell)
        (outbound_load_linked_to_distribution_channel ?outbound_load ?distribution_channel)
        (outbound_load_requires_origin_certificate ?outbound_load)
        (not
          (outbound_load_available ?outbound_load)
        )
      )
  )
  (:action initialize_outbound_load_with_both_requirements
    :parameters (?storage_batch - storage_batch ?picking_batch - picking_batch ?storage_cell - storage_cell ?distribution_channel - distribution_channel ?outbound_load - outbound_load)
    :precondition
      (and
        (storage_batch_placed ?storage_batch)
        (picking_batch_placed ?picking_batch)
        (storage_batch_assigned_to_storage_cell ?storage_batch ?storage_cell)
        (picking_batch_assigned_to_distribution_channel ?picking_batch ?distribution_channel)
        (storage_cell_tagged ?storage_cell)
        (distribution_channel_tagged ?distribution_channel)
        (not
          (storage_batch_available_for_outbound ?storage_batch)
        )
        (not
          (picking_batch_available_for_outbound ?picking_batch)
        )
        (outbound_load_available ?outbound_load)
      )
    :effect
      (and
        (outbound_load_reserved ?outbound_load)
        (outbound_load_linked_to_storage_cell ?outbound_load ?storage_cell)
        (outbound_load_linked_to_distribution_channel ?outbound_load ?distribution_channel)
        (outbound_load_requires_temperature_profile ?outbound_load)
        (outbound_load_requires_origin_certificate ?outbound_load)
        (not
          (outbound_load_available ?outbound_load)
        )
      )
  )
  (:action confirm_outbound_load_ready_for_packing
    :parameters (?outbound_load - outbound_load ?storage_batch - storage_batch ?grading_station - grading_station)
    :precondition
      (and
        (outbound_load_reserved ?outbound_load)
        (storage_batch_placed ?storage_batch)
        (assigned_to_grading_station ?storage_batch ?grading_station)
        (not
          (outbound_load_ready_for_packing ?outbound_load)
        )
      )
    :effect (outbound_load_ready_for_packing ?outbound_load)
  )
  (:action assign_packaging_box_to_pallet_and_link_to_load
    :parameters (?pallet - pallet ?packaging_box - packaging_box ?outbound_load - outbound_load)
    :precondition
      (and
        (grade_completed ?pallet)
        (pallet_assigned_to_outbound_load ?pallet ?outbound_load)
        (pallet_has_packaging_box ?pallet ?packaging_box)
        (packaging_box_available ?packaging_box)
        (outbound_load_reserved ?outbound_load)
        (outbound_load_ready_for_packing ?outbound_load)
        (not
          (packaging_box_used ?packaging_box)
        )
      )
    :effect
      (and
        (packaging_box_used ?packaging_box)
        (packaging_box_assigned_to_outbound_load ?packaging_box ?outbound_load)
        (not
          (packaging_box_available ?packaging_box)
        )
      )
  )
  (:action initialize_pallet_packaging_process
    :parameters (?pallet - pallet ?packaging_box - packaging_box ?outbound_load - outbound_load ?grading_station - grading_station)
    :precondition
      (and
        (grade_completed ?pallet)
        (pallet_has_packaging_box ?pallet ?packaging_box)
        (packaging_box_used ?packaging_box)
        (packaging_box_assigned_to_outbound_load ?packaging_box ?outbound_load)
        (assigned_to_grading_station ?pallet ?grading_station)
        (not
          (outbound_load_requires_temperature_profile ?outbound_load)
        )
        (not
          (pallet_packaging_initialized ?pallet)
        )
      )
    :effect (pallet_packaging_initialized ?pallet)
  )
  (:action apply_customer_order_to_pallet
    :parameters (?pallet - pallet ?customer_order - customer_order)
    :precondition
      (and
        (grade_completed ?pallet)
        (customer_order_available ?customer_order)
        (not
          (pallet_has_order_applied ?pallet)
        )
      )
    :effect
      (and
        (pallet_has_order_applied ?pallet)
        (pallet_linked_to_customer_order ?pallet ?customer_order)
        (not
          (customer_order_available ?customer_order)
        )
      )
  )
  (:action confirm_packaging_and_order_binding
    :parameters (?pallet - pallet ?packaging_box - packaging_box ?outbound_load - outbound_load ?grading_station - grading_station ?customer_order - customer_order)
    :precondition
      (and
        (grade_completed ?pallet)
        (pallet_has_packaging_box ?pallet ?packaging_box)
        (packaging_box_used ?packaging_box)
        (packaging_box_assigned_to_outbound_load ?packaging_box ?outbound_load)
        (assigned_to_grading_station ?pallet ?grading_station)
        (outbound_load_requires_temperature_profile ?outbound_load)
        (pallet_has_order_applied ?pallet)
        (pallet_linked_to_customer_order ?pallet ?customer_order)
        (not
          (pallet_packaging_initialized ?pallet)
        )
      )
    :effect
      (and
        (pallet_packaging_initialized ?pallet)
        (pallet_order_confirmed ?pallet)
      )
  )
  (:action apply_temperature_profile_and_confirm_packaging_no_origin_requirement
    :parameters (?pallet - pallet ?temperature_profile - temperature_profile ?conditioning_station - conditioning_station ?packaging_box - packaging_box ?outbound_load - outbound_load)
    :precondition
      (and
        (pallet_packaging_initialized ?pallet)
        (pallet_assigned_temperature_profile ?pallet ?temperature_profile)
        (assigned_to_conditioning_station ?pallet ?conditioning_station)
        (pallet_has_packaging_box ?pallet ?packaging_box)
        (packaging_box_assigned_to_outbound_load ?packaging_box ?outbound_load)
        (not
          (outbound_load_requires_origin_certificate ?outbound_load)
        )
        (not
          (pallet_packaging_confirmed ?pallet)
        )
      )
    :effect (pallet_packaging_confirmed ?pallet)
  )
  (:action apply_temperature_profile_and_confirm_packaging_with_origin_requirement
    :parameters (?pallet - pallet ?temperature_profile - temperature_profile ?conditioning_station - conditioning_station ?packaging_box - packaging_box ?outbound_load - outbound_load)
    :precondition
      (and
