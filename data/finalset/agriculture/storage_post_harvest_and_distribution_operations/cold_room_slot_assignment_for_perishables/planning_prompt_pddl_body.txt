  (:action check_in_entity
    :parameters (?entity - batch_entity)
    :precondition
      (and
        (not
          (entity_checked_in ?entity)
        )
        (not
          (entity_final_channel_assignment ?entity)
        )
      )
    :effect (entity_checked_in ?entity)
  )
  (:action assign_entity_to_grading_station
    :parameters (?entity - batch_entity ?grading_station - grading_station)
    :precondition
      (and
        (entity_checked_in ?entity)
        (not
          (entity_assigned_for_grading ?entity)
        )
        (grading_station_available ?grading_station)
      )
    :effect
      (and
        (entity_assigned_for_grading ?entity)
        (entity_assigned_to_grading_station ?entity ?grading_station)
        (not
          (grading_station_available ?grading_station)
        )
      )
  )
  (:action assign_entity_to_grader
    :parameters (?entity - batch_entity ?grader - grader_equipment)
    :precondition
      (and
        (entity_checked_in ?entity)
        (entity_assigned_for_grading ?entity)
        (grader_available ?grader)
      )
    :effect
      (and
        (entity_assigned_to_grader ?entity ?grader)
        (not
          (grader_available ?grader)
        )
      )
  )
  (:action confirm_grading_completion
    :parameters (?entity - batch_entity ?grader - grader_equipment)
    :precondition
      (and
        (entity_checked_in ?entity)
        (entity_assigned_for_grading ?entity)
        (entity_assigned_to_grader ?entity ?grader)
        (not
          (entity_conditioned ?entity)
        )
      )
    :effect (entity_conditioned ?entity)
  )
  (:action release_grader
    :parameters (?entity - batch_entity ?grader - grader_equipment)
    :precondition
      (and
        (entity_assigned_to_grader ?entity ?grader)
      )
    :effect
      (and
        (grader_available ?grader)
        (not
          (entity_assigned_to_grader ?entity ?grader)
        )
      )
  )
  (:action assign_conditioning_unit_to_entity
    :parameters (?entity - batch_entity ?conditioning_unit - conditioning_unit)
    :precondition
      (and
        (entity_conditioned ?entity)
        (conditioning_unit_available ?conditioning_unit)
      )
    :effect
      (and
        (entity_assigned_to_conditioning_unit ?entity ?conditioning_unit)
        (not
          (conditioning_unit_available ?conditioning_unit)
        )
      )
  )
  (:action release_conditioning_unit
    :parameters (?entity - batch_entity ?conditioning_unit - conditioning_unit)

        (entity_assigned_to_conditioning_unit ?entity ?conditioning_unit)

        (conditioning_unit_available ?conditioning_unit)
        (not
          (entity_assigned_to_conditioning_unit ?entity ?conditioning_unit)
        )
      )
  )

  (:action release_temperature_profile_from_slot

        )
      )
  )
  (:action release_humidity_profile_from_slot
    :parameters (?storage_slot - storage_slot ?humidity_profile - humidity_profile)
    :precondition
      (and
        (slot_assigned_humidity_profile ?storage_slot ?humidity_profile)
      )
    :effect
      (and
        (humidity_profile_available ?humidity_profile)
        (not
          (slot_assigned_humidity_profile ?storage_slot ?humidity_profile)
        )
      )
  )
  (:action reserve_temperature_zone_for_pallet
    :parameters (?producer_pallet - producer_pallet ?temperature_zone - temperature_zone ?grader - grader_equipment)
    :precondition
      (and
        (entity_conditioned ?producer_pallet)
        (entity_assigned_to_grader ?producer_pallet ?grader)
        (placement_compatible_with_zone ?producer_pallet ?temperature_zone)
        (not
          (zone_reserved ?temperature_zone)
        )
        (not
          (zone_tagged ?temperature_zone)

  (:action confirm_zone_reservation_for_pallet
    :parameters (?producer_pallet - producer_pallet ?temperature_zone - temperature_zone ?conditioning_unit - conditioning_unit)
    :precondition
      (and
        (entity_conditioned ?producer_pallet)
        (entity_assigned_to_conditioning_unit ?producer_pallet ?conditioning_unit)
        (placement_compatible_with_zone ?producer_pallet ?temperature_zone)
        (zone_reserved ?temperature_zone)
        (not
          (pallet_preplacement_flag ?producer_pallet)
        )
      )
    :effect
      (and
        (pallet_preplacement_flag ?producer_pallet)
        (placement_confirmed ?producer_pallet)
      )
  )
  (:action tag_zone_and_assign_packaging_to_pallet
    :parameters (?producer_pallet - producer_pallet ?temperature_zone - temperature_zone ?packaging_material - packaging_material)
    :precondition
      (and
        (entity_conditioned ?producer_pallet)
        (placement_compatible_with_zone ?producer_pallet ?temperature_zone)
        (packaging_material_available ?packaging_material)
        (not
          (pallet_preplacement_flag ?producer_pallet)
        )
      )
    :effect
      (and
        (zone_tagged ?temperature_zone)
        (pallet_preplacement_flag ?producer_pallet)
        (producer_pallet_assigned_packaging ?producer_pallet ?packaging_material)
        (not
          (packaging_material_available ?packaging_material)
        )
      )
  )
  (:action finalize_zone_placement_for_pallet
    :parameters (?producer_pallet - producer_pallet ?temperature_zone - temperature_zone ?grader - grader_equipment ?packaging_material - packaging_material)
    :precondition
      (and
        (entity_conditioned ?producer_pallet)
        (entity_assigned_to_grader ?producer_pallet ?grader)
        (placement_compatible_with_zone ?producer_pallet ?temperature_zone)
        (zone_tagged ?temperature_zone)
        (producer_pallet_assigned_packaging ?producer_pallet ?packaging_material)
        (not
          (placement_confirmed ?producer_pallet)
        )
      )
    :effect
      (and
        (zone_reserved ?temperature_zone)
        (placement_confirmed ?producer_pallet)
        (packaging_material_available ?packaging_material)
        (not
          (producer_pallet_assigned_packaging ?producer_pallet ?packaging_material)
        )
      )
  )

        (placement_compatible_with_channel ?alternate_pallet ?distribution_channel)
        (packaging_material_available ?packaging_material)
        (not
          (alternate_pallet_preplacement_flag ?alternate_pallet)
        )

    :effect
      (and
        (channel_tagged ?distribution_channel)
        (alternate_pallet_preplacement_flag ?alternate_pallet)
        (alternate_pallet_assigned_packaging ?alternate_pallet ?packaging_material)
        (not
          (packaging_material_available ?packaging_material)
        )

  (:action finalize_channel_placement_for_alternate_pallet

          (channel_placement_confirmed ?alternate_pallet)
        )
      )
    :effect
      (and
        (channel_reserved ?distribution_channel)
        (channel_placement_confirmed ?alternate_pallet)

          (alternate_pallet_assigned_packaging ?alternate_pallet ?packaging_material)
        )
      )

  (:action assemble_outbound_container
    :parameters (?producer_pallet - producer_pallet ?alternate_pallet - alternate_pallet ?temperature_zone - temperature_zone ?distribution_channel - distribution_channel ?outbound_container - outbound_container)
    :precondition
      (and
        (pallet_preplacement_flag ?producer_pallet)
        (alternate_pallet_preplacement_flag ?alternate_pallet)

        (placement_compatible_with_channel ?alternate_pallet ?distribution_channel)
        (zone_reserved ?temperature_zone)
        (channel_reserved ?distribution_channel)
        (placement_confirmed ?producer_pallet)
        (channel_placement_confirmed ?alternate_pallet)
        (container_available ?outbound_container)
      )
    :effect
      (and
        (container_allocated ?outbound_container)
        (container_linked_to_zone ?outbound_container ?temperature_zone)
        (container_linked_to_channel ?outbound_container ?distribution_channel)
        (not
          (container_available ?outbound_container)
        )
      )
  )
  (:action assemble_and_apply_temperature_tag_to_container
    :parameters (?producer_pallet - producer_pallet ?alternate_pallet - alternate_pallet ?temperature_zone - temperature_zone ?distribution_channel - distribution_channel ?outbound_container - outbound_container)
    :precondition
      (and
        (pallet_preplacement_flag ?producer_pallet)
        (alternate_pallet_preplacement_flag ?alternate_pallet)
        (placement_compatible_with_zone ?producer_pallet ?temperature_zone)
        (placement_compatible_with_channel ?alternate_pallet ?distribution_channel)
        (zone_tagged ?temperature_zone)
        (channel_reserved ?distribution_channel)
        (not
          (placement_confirmed ?producer_pallet)
        )
        (channel_placement_confirmed ?alternate_pallet)
        (container_available ?outbound_container)
      )
    :effect
      (and
        (container_allocated ?outbound_container)
        (container_linked_to_zone ?outbound_container ?temperature_zone)
        (container_linked_to_channel ?outbound_container ?distribution_channel)
        (container_has_temperature_tag ?outbound_container)
        (not
          (container_available ?outbound_container)
        )
      )
  )
  (:action assemble_and_apply_humidity_tag_to_container
    :parameters (?producer_pallet - producer_pallet ?alternate_pallet - alternate_pallet ?temperature_zone - temperature_zone ?distribution_channel - distribution_channel ?outbound_container - outbound_container)
    :precondition
      (and
        (pallet_preplacement_flag ?producer_pallet)
        (alternate_pallet_preplacement_flag ?alternate_pallet)
        (placement_compatible_with_zone ?producer_pallet ?temperature_zone)
        (placement_compatible_with_channel ?alternate_pallet ?distribution_channel)
        (zone_reserved ?temperature_zone)
        (channel_tagged ?distribution_channel)
        (placement_confirmed ?producer_pallet)
        (not
          (channel_placement_confirmed ?alternate_pallet)
        )
        (container_available ?outbound_container)
      )
    :effect
      (and
        (container_allocated ?outbound_container)
        (container_linked_to_zone ?outbound_container ?temperature_zone)
        (container_linked_to_channel ?outbound_container ?distribution_channel)
        (container_has_humidity_tag ?outbound_container)
        (not
          (container_available ?outbound_container)
        )
      )
  )
  (:action assemble_container_with_environmental_tags
    :parameters (?producer_pallet - producer_pallet ?alternate_pallet - alternate_pallet ?temperature_zone - temperature_zone ?distribution_channel - distribution_channel ?outbound_container - outbound_container)
    :precondition
      (and
        (pallet_preplacement_flag ?producer_pallet)
        (alternate_pallet_preplacement_flag ?alternate_pallet)
        (placement_compatible_with_zone ?producer_pallet ?temperature_zone)
        (placement_compatible_with_channel ?alternate_pallet ?distribution_channel)
        (zone_tagged ?temperature_zone)
        (channel_tagged ?distribution_channel)
        (not
          (placement_confirmed ?producer_pallet)
        )
        (not
          (channel_placement_confirmed ?alternate_pallet)
        )
        (container_available ?outbound_container)
      )
    :effect
      (and
        (container_allocated ?outbound_container)
        (container_linked_to_zone ?outbound_container ?temperature_zone)
        (container_linked_to_channel ?outbound_container ?distribution_channel)
        (container_has_temperature_tag ?outbound_container)
        (container_has_humidity_tag ?outbound_container)
        (not
          (container_available ?outbound_container)
        )
      )
  )
  (:action stage_container_for_packing
    :parameters (?outbound_container - outbound_container ?producer_pallet - producer_pallet ?grader - grader_equipment)
    :precondition
      (and
        (container_allocated ?outbound_container)
        (pallet_preplacement_flag ?producer_pallet)
        (entity_assigned_to_grader ?producer_pallet ?grader)
        (not
          (container_staged ?outbound_container)
        )
      )

  )

          (tray_loaded ?tray)

      )
    :effect
      (and
        (tray_loaded ?tray)

        (not
          (tray_available ?tray)
        )
      )
  )
  (:action lock_slot_after_tray_loading

    :precondition
      (and
        (entity_conditioned ?storage_slot)
        (slot_has_tray ?storage_slot ?tray)
        (tray_loaded ?tray)
        (tray_loaded_into_container ?tray ?outbound_container)
        (entity_assigned_to_grader ?storage_slot ?grader)
        (not
          (container_has_temperature_tag ?outbound_container)
        )
        (not
          (slot_ready_for_profile_assignment ?storage_slot)
        )
      )
    :effect (slot_ready_for_profile_assignment ?storage_slot)
  )
  (:action apply_label_to_slot

      (and
        (entity_conditioned ?storage_slot)

        (not
          (label_attached ?storage_slot)

      )
    :effect
      (and
        (label_attached ?storage_slot)
        (slot_assigned_label ?storage_slot ?label)
        (not
          (label_available ?label)
        )
      )
  )
  (:action verify_label_and_prepare_slot_for_packing
    :parameters (?storage_slot - storage_slot ?tray - tray ?outbound_container - outbound_container ?grader - grader_equipment ?label - label)
    :precondition
      (and
        (entity_conditioned ?storage_slot)
        (slot_has_tray ?storage_slot ?tray)
        (tray_loaded ?tray)
        (tray_loaded_into_container ?tray ?outbound_container)
        (entity_assigned_to_grader ?storage_slot ?grader)
        (container_has_temperature_tag ?outbound_container)
        (label_attached ?storage_slot)
        (slot_assigned_label ?storage_slot ?label)
        (not
          (slot_ready_for_profile_assignment ?storage_slot)
        )
      )
    :effect
      (and
        (slot_ready_for_profile_assignment ?storage_slot)
        (label_verified ?storage_slot)
      )
  )
  (:action apply_temperature_profile_to_slot
    :parameters (?storage_slot - storage_slot ?temperature_profile - temperature_profile ?conditioning_unit - conditioning_unit ?tray - tray ?outbound_container - outbound_container)
    :precondition
      (and
        (slot_ready_for_profile_assignment ?storage_slot)
        (slot_assigned_temperature_profile ?storage_slot ?temperature_profile)
        (entity_assigned_to_conditioning_unit ?storage_slot ?conditioning_unit)
        (slot_has_tray ?storage_slot ?tray)
        (tray_loaded_into_container ?tray ?outbound_container)
        (not
          (container_has_humidity_tag ?outbound_container)
        )
        (not
          (slot_temperature_profile_applied ?storage_slot)
        )
      )
    :effect (slot_temperature_profile_applied ?storage_slot)
  )
  (:action attach_temperature_profile_to_slot
    :parameters (?storage_slot - storage_slot ?temperature_profile - temperature_profile ?conditioning_unit - conditioning_unit ?tray - tray ?outbound_container - outbound_container)
    :precondition
      (and
        (slot_ready_for_profile_assignment ?storage_slot)
        (slot_assigned_temperature_profile ?storage_slot ?temperature_profile)
        (entity_assigned_to_conditioning_unit ?storage_slot ?conditioning_unit)
        (slot_has_tray ?storage_slot ?tray)
        (tray_loaded_into_container ?tray ?outbound_container)
        (container_has_humidity_tag ?outbound_container)
        (not
          (slot_temperature_profile_applied ?storage_slot)
        )
      )
    :effect (slot_temperature_profile_applied ?storage_slot)
  )
  (:action confirm_humidity_profile_and_finalize_environment
    :parameters (?storage_slot - storage_slot ?humidity_profile - humidity_profile ?tray - tray ?outbound_container - outbound_container)
    :precondition
