  (:action identify_animal_for_booster_follow_up
    :parameters (?animal - animal)
    :precondition
      (and
        (not
          (marked_for_booster_follow_up ?animal)
        )
        (not
          (follow_up_scheduled ?animal)
        )
      )
    :effect (marked_for_booster_follow_up ?animal)
  )
  (:action assign_service_slot_to_animal
    :parameters (?animal - animal ?service_slot - service_slot)
    :precondition
      (and
        (marked_for_booster_follow_up ?animal)
        (not
          (has_slot_reservation ?animal)
        )
        (service_slot_available ?service_slot)
      )
    :effect
      (and
        (has_slot_reservation ?animal)
        (assigned_service_slot ?animal ?service_slot)
        (not
          (service_slot_available ?service_slot)
        )
      )
  )
