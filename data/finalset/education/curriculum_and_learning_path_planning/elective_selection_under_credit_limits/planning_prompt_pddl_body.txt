    (secondary_student_allocated_credit_token ?secondary_student - secondary_student ?credit_unit_token - credit_unit_token)

    (category_secondary_slot_marked ?elective_category_secondary - elective_category_secondary)
    (category_secondary_slot_locked ?elective_category_secondary - elective_category_secondary)
    (secondary_student_category_fulfilled ?secondary_student - secondary_student)
    (package_pending_assembly ?elective_package - elective_package)

    (package_linked_category_primary ?elective_package - elective_package ?elective_category_primary - elective_category_primary)

    (approval_finalized ?academic_plan - academic_plan)
    (endorsement_slot_available ?endorsement_slot - endorsement_slot)
    (plan_assigned_endorsement_slot ?academic_plan - academic_plan ?endorsement_slot - endorsement_slot)

  (:action initialize_plan_record
    :parameters (?academic_entity - academic_entity)
    :precondition
      (and
        (not
          (plan_initialized ?academic_entity)
        )
        (not
          (selection_finalized ?academic_entity)
        )
      )
    :effect (plan_initialized ?academic_entity)
  )
  (:action reserve_elective_offer
    :parameters (?academic_entity - academic_entity ?elective_offer - elective_offer)
    :precondition
      (and
        (plan_initialized ?academic_entity)
        (not
          (has_reserved_elective ?academic_entity)
        )
        (elective_offer_available ?elective_offer)
      )
    :effect
      (and
        (has_reserved_elective ?academic_entity)
        (reserved_offer_for ?academic_entity ?elective_offer)
        (not
          (elective_offer_available ?elective_offer)
        )
      )
  )
  (:action place_tentative_course_request
    :parameters (?academic_entity - academic_entity ?course_offering - course_offering)
    :precondition
      (and
        (plan_initialized ?academic_entity)
        (has_reserved_elective ?academic_entity)
        (course_offering_available ?course_offering)
      )
    :effect
      (and
        (requested_course_for ?academic_entity ?course_offering)
        (not
          (course_offering_available ?course_offering)
        )
      )
  )
  (:action mark_plan_ready_for_review
    :parameters (?academic_entity - academic_entity ?course_offering - course_offering)
    :precondition
      (and
        (plan_initialized ?academic_entity)
        (has_reserved_elective ?academic_entity)
        (requested_course_for ?academic_entity ?course_offering)
        (not
          (entity_ready_for_review ?academic_entity)
        )
      )
    :effect (entity_ready_for_review ?academic_entity)
  )
  (:action rollback_course_request
    :parameters (?academic_entity - academic_entity ?course_offering - course_offering)
    :precondition
      (and
        (requested_course_for ?academic_entity ?course_offering)
      )
    :effect
      (and
        (course_offering_available ?course_offering)
        (not
          (requested_course_for ?academic_entity ?course_offering)
        )
      )
  )
  (:action assign_advisor_to_plan
    :parameters (?academic_entity - academic_entity ?advisor - advisor)

        (entity_ready_for_review ?academic_entity)

    :effect
      (and
        (advisor_assigned_to ?academic_entity ?advisor)
        (not
          (advisor_available ?advisor)
        )
      )
  )
  (:action release_advisor_assignment
    :parameters (?academic_entity - academic_entity ?advisor - advisor)
    :precondition
      (and
        (advisor_assigned_to ?academic_entity ?advisor)
      )
    :effect
      (and
        (advisor_available ?advisor)
        (not
          (advisor_assigned_to ?academic_entity ?advisor)
        )
      )
  )
  (:action assign_administrative_resource

      (and
        (plan_assigned_faculty_approval_token ?academic_plan ?faculty_approval_token)
      )
    :effect
      (and
        (faculty_approval_token_available ?faculty_approval_token)

          (plan_assigned_faculty_approval_token ?academic_plan ?faculty_approval_token)
        )
      )
  )
  (:action open_primary_category_slot
    :parameters (?primary_student - primary_student ?elective_category_primary - elective_category_primary ?course_offering - course_offering)
    :precondition
      (and
        (entity_ready_for_review ?primary_student)
        (requested_course_for ?primary_student ?course_offering)
        (eligible_for_category_primary ?primary_student ?elective_category_primary)
        (not
          (category_primary_slot_marked ?elective_category_primary)
        )
        (not
          (category_primary_slot_locked ?elective_category_primary)
        )

    :effect (category_primary_slot_marked ?elective_category_primary)
  )
  (:action lock_primary_category_for_student
    :parameters (?primary_student - primary_student ?elective_category_primary - elective_category_primary ?advisor - advisor)
    :precondition
      (and
        (entity_ready_for_review ?primary_student)
        (advisor_assigned_to ?primary_student ?advisor)
        (eligible_for_category_primary ?primary_student ?elective_category_primary)
        (category_primary_slot_marked ?elective_category_primary)
        (not
          (primary_student_slot_processed ?primary_student)
        )
      )
    :effect
      (and
        (primary_student_slot_processed ?primary_student)
        (primary_student_category_fulfilled ?primary_student)
      )
  )
  (:action allocate_credit_token_primary_category
    :parameters (?primary_student - primary_student ?elective_category_primary - elective_category_primary ?credit_unit_token - credit_unit_token)

      (and
        (entity_ready_for_review ?primary_student)
        (eligible_for_category_primary ?primary_student ?elective_category_primary)
        (credit_token_available ?credit_unit_token)
        (not
          (primary_student_slot_processed ?primary_student)
        )
      )
    :effect
      (and
        (category_primary_slot_locked ?elective_category_primary)
        (primary_student_slot_processed ?primary_student)
        (primary_student_allocated_credit_token ?primary_student ?credit_unit_token)
        (not
          (credit_token_available ?credit_unit_token)
        )
      )
  )
  (:action confirm_primary_category_allocation

        (category_primary_slot_locked ?elective_category_primary)
        (primary_student_allocated_credit_token ?primary_student ?credit_unit_token)
        (not
          (primary_student_category_fulfilled ?primary_student)
        )
      )
    :effect
      (and
        (category_primary_slot_marked ?elective_category_primary)
        (primary_student_category_fulfilled ?primary_student)
        (credit_token_available ?credit_unit_token)
        (not
          (primary_student_allocated_credit_token ?primary_student ?credit_unit_token)
        )
      )
  )

    :precondition
      (and
        (entity_ready_for_review ?secondary_student)

        (eligible_for_category_secondary ?secondary_student ?elective_category_secondary)
        (category_secondary_slot_locked ?elective_category_secondary)
        (secondary_student_allocated_credit_token ?secondary_student ?credit_unit_token)
        (not
          (secondary_student_category_fulfilled ?secondary_student)

      (and
        (category_secondary_slot_marked ?elective_category_secondary)
        (secondary_student_category_fulfilled ?secondary_student)
        (credit_token_available ?credit_unit_token)
        (not
          (secondary_student_allocated_credit_token ?secondary_student ?credit_unit_token)
        )

  )
  (:action assemble_elective_package
    :parameters (?primary_student - primary_student ?secondary_student - secondary_student ?elective_category_primary - elective_category_primary ?elective_category_secondary - elective_category_secondary ?elective_package - elective_package)
    :precondition
      (and
        (primary_student_slot_processed ?primary_student)
        (secondary_student_slot_processed ?secondary_student)
        (eligible_for_category_primary ?primary_student ?elective_category_primary)
        (eligible_for_category_secondary ?secondary_student ?elective_category_secondary)
        (category_primary_slot_marked ?elective_category_primary)
        (category_secondary_slot_marked ?elective_category_secondary)
        (primary_student_category_fulfilled ?primary_student)
        (secondary_student_category_fulfilled ?secondary_student)
        (package_pending_assembly ?elective_package)
      )
    :effect
      (and
        (package_ready ?elective_package)
        (package_linked_category_primary ?elective_package ?elective_category_primary)
        (package_linked_category_secondary ?elective_package ?elective_category_secondary)
        (not
          (package_pending_assembly ?elective_package)
        )
      )
  )
  (:action assemble_package_with_primary_flag
    :parameters (?primary_student - primary_student ?secondary_student - secondary_student ?elective_category_primary - elective_category_primary ?elective_category_secondary - elective_category_secondary ?elective_package - elective_package)
    :precondition
      (and
        (primary_student_slot_processed ?primary_student)
        (secondary_student_slot_processed ?secondary_student)
        (eligible_for_category_primary ?primary_student ?elective_category_primary)
        (eligible_for_category_secondary ?secondary_student ?elective_category_secondary)
        (category_primary_slot_locked ?elective_category_primary)
        (category_secondary_slot_marked ?elective_category_secondary)
        (not
          (primary_student_category_fulfilled ?primary_student)
        )
        (secondary_student_category_fulfilled ?secondary_student)
        (package_pending_assembly ?elective_package)
      )
    :effect
      (and
        (package_ready ?elective_package)
        (package_linked_category_primary ?elective_package ?elective_category_primary)
        (package_linked_category_secondary ?elective_package ?elective_category_secondary)
        (package_has_primary_slot ?elective_package)
        (not
          (package_pending_assembly ?elective_package)
        )
      )
  )
  (:action assemble_package_with_secondary_flag
    :parameters (?primary_student - primary_student ?secondary_student - secondary_student ?elective_category_primary - elective_category_primary ?elective_category_secondary - elective_category_secondary ?elective_package - elective_package)
    :precondition
      (and
        (primary_student_slot_processed ?primary_student)
        (secondary_student_slot_processed ?secondary_student)
        (eligible_for_category_primary ?primary_student ?elective_category_primary)
        (eligible_for_category_secondary ?secondary_student ?elective_category_secondary)
        (category_primary_slot_marked ?elective_category_primary)
        (category_secondary_slot_locked ?elective_category_secondary)
        (primary_student_category_fulfilled ?primary_student)
        (not
          (secondary_student_category_fulfilled ?secondary_student)
        )
        (package_pending_assembly ?elective_package)
      )
    :effect
      (and
        (package_ready ?elective_package)

        (package_linked_category_secondary ?elective_package ?elective_category_secondary)
        (package_has_secondary_slot ?elective_package)
        (not
          (package_pending_assembly ?elective_package)
        )
      )
  )

    :parameters (?primary_student - primary_student ?secondary_student - secondary_student ?elective_category_primary - elective_category_primary ?elective_category_secondary - elective_category_secondary ?elective_package - elective_package)
    :precondition
      (and
        (primary_student_slot_processed ?primary_student)
        (secondary_student_slot_processed ?secondary_student)
        (eligible_for_category_primary ?primary_student ?elective_category_primary)
        (eligible_for_category_secondary ?secondary_student ?elective_category_secondary)
        (category_primary_slot_locked ?elective_category_primary)
        (category_secondary_slot_locked ?elective_category_secondary)
        (not
          (primary_student_category_fulfilled ?primary_student)
        )
        (not
          (secondary_student_category_fulfilled ?secondary_student)

        (package_pending_assembly ?elective_package)
      )
    :effect
      (and
        (package_ready ?elective_package)
        (package_linked_category_primary ?elective_package ?elective_category_primary)
        (package_linked_category_secondary ?elective_package ?elective_category_secondary)
        (package_has_primary_slot ?elective_package)
        (package_has_secondary_slot ?elective_package)
        (not
          (package_pending_assembly ?elective_package)
        )
      )
  )

    :parameters (?elective_package - elective_package ?primary_student - primary_student ?course_offering - course_offering)
    :precondition
      (and
        (package_ready ?elective_package)
        (primary_student_slot_processed ?primary_student)
        (requested_course_for ?primary_student ?course_offering)
        (not
          (package_committed ?elective_package)
        )
      )
    :effect (package_committed ?elective_package)
  )
  (:action satisfy_program_requirement
    :parameters (?academic_plan - academic_plan ?program_requirement - program_requirement ?elective_package - elective_package)
    :precondition
      (and
        (entity_ready_for_review ?academic_plan)
        (plan_includes_package ?academic_plan ?elective_package)
        (plan_has_requirement ?academic_plan ?program_requirement)
        (requirement_available ?program_requirement)
        (package_ready ?elective_package)

        (not
          (requirement_satisfied ?program_requirement)
        )
      )
    :effect
      (and
        (requirement_satisfied ?program_requirement)
        (requirement_satisfied_by_package ?program_requirement ?elective_package)
        (not
          (requirement_available ?program_requirement)
        )
      )
  )
  (:action verify_requirement_and_mark_plan

    :precondition
      (and
        (entity_ready_for_review ?academic_plan)
        (plan_has_requirement ?academic_plan ?program_requirement)
        (requirement_satisfied ?program_requirement)
        (requirement_satisfied_by_package ?program_requirement ?elective_package)
        (requested_course_for ?academic_plan ?course_offering)
        (not
          (package_has_primary_slot ?elective_package)
        )
        (not
          (plan_requirement_verified ?academic_plan)
        )
      )
    :effect (plan_requirement_verified ?academic_plan)
  )
  (:action reserve_endorsement_slot_for_plan
    :parameters (?academic_plan - academic_plan ?endorsement_slot - endorsement_slot)
    :precondition
      (and
        (entity_ready_for_review ?academic_plan)
        (endorsement_slot_available ?endorsement_slot)
        (not
          (endorsement_requested ?academic_plan)
        )
      )
    :effect
      (and
        (endorsement_requested ?academic_plan)
        (plan_assigned_endorsement_slot ?academic_plan ?endorsement_slot)
        (not
          (endorsement_slot_available ?endorsement_slot)
        )
      )
  )
  (:action finalize_endorsement_and_mark_plan
    :parameters (?academic_plan - academic_plan ?program_requirement - program_requirement ?elective_package - elective_package ?course_offering - course_offering ?endorsement_slot - endorsement_slot)
    :precondition
      (and
        (entity_ready_for_review ?academic_plan)
        (plan_has_requirement ?academic_plan ?program_requirement)
        (requirement_satisfied ?program_requirement)
        (requirement_satisfied_by_package ?program_requirement ?elective_package)
        (requested_course_for ?academic_plan ?course_offering)
        (package_has_primary_slot ?elective_package)
        (endorsement_requested ?academic_plan)
        (plan_assigned_endorsement_slot ?academic_plan ?endorsement_slot)
        (not
          (plan_requirement_verified ?academic_plan)
        )
      )
    :effect
      (and
        (plan_requirement_verified ?academic_plan)
