## Planning domain (PDDL)

(define (domain certification_requirement_alignment)
  (:requirements :strips :typing :negative-preconditions)
  (:types base_object - object requirement_family - base_object resource_family - base_object experience_family - base_object organisation_level - base_object curriculum_item - organisation_level certification_requirement - requirement_family course_offering - requirement_family instructor - requirement_family accreditation_condition - requirement_family resource_package - requirement_family academic_term - requirement_family optional_module - requirement_family research_project - requirement_family elective_option - resource_family assessment_instrument - resource_family project_requirement - resource_family competency_domain - experience_family practical_experience_type - experience_family credential_bundle - experience_family cohort_category - curriculum_item pathway_category - curriculum_item theory_profile - cohort_category practical_profile - cohort_category learning_path - pathway_category)

  (:predicates
    (curriculum_item_registered ?curriculum_item - curriculum_item)
    (curriculum_item_confirmed ?curriculum_item - curriculum_item)
    (requirement_assigned ?curriculum_item - curriculum_item)
    (certification_awarded ?curriculum_item - curriculum_item)
    (final_validation_ready ?curriculum_item - curriculum_item)
    (certification_eligibility_confirmed ?curriculum_item - curriculum_item)
    (available_certification_requirement ?certification_requirement - certification_requirement)
    (curriculum_item_assigned_requirement ?curriculum_item - curriculum_item ?certification_requirement - certification_requirement)
    (available_course_offering ?course_offering - course_offering)
    (curriculum_item_allocated_course ?curriculum_item - curriculum_item ?course_offering - course_offering)
    (available_instructor ?instructor - instructor)
    (curriculum_item_assigned_instructor ?curriculum_item - curriculum_item ?instructor - instructor)
    (available_elective_option ?elective_option - elective_option)
    (theory_profile_selected_elective ?theory_profile - theory_profile ?elective_option - elective_option)
    (practical_profile_selected_elective ?practical_profile - practical_profile ?elective_option - elective_option)
    (theory_profile_competency_mapping ?theory_profile - theory_profile ?competency_domain - competency_domain)
    (competency_domain_theory_ready ?competency_domain - competency_domain)
    (competency_domain_practice_ready ?competency_domain - competency_domain)
    (theory_profile_competency_verified ?theory_profile - theory_profile)
    (practical_profile_assigned_experience ?practical_profile - practical_profile ?practical_experience_type - practical_experience_type)
    (experience_type_reserved ?practical_experience_type - practical_experience_type)
    (experience_type_confirmed ?practical_experience_type - practical_experience_type)
    (practical_profile_competency_verified ?practical_profile - practical_profile)
    (available_credential_bundle ?credential_bundle - credential_bundle)
    (credential_bundle_created ?credential_bundle - credential_bundle)
    (credential_bundle_includes_competency_domain ?credential_bundle - credential_bundle ?competency_domain - competency_domain)
    (credential_bundle_includes_experience_type ?credential_bundle - credential_bundle ?practical_experience_type - practical_experience_type)
    (credential_bundle_specialization_slot_theory ?credential_bundle - credential_bundle)
    (credential_bundle_specialization_slot_practical ?credential_bundle - credential_bundle)
    (credential_bundle_locked ?credential_bundle - credential_bundle)
    (learning_path_associated_with_theory_profile ?learning_path - learning_path ?theory_profile - theory_profile)
    (learning_path_associated_with_practical_profile ?learning_path - learning_path ?practical_profile - practical_profile)
    (learning_path_contains_credential_bundle ?learning_path - learning_path ?credential_bundle - credential_bundle)
    (available_assessment_instrument ?assessment_instrument - assessment_instrument)
    (learning_path_assigned_assessment_instrument ?learning_path - learning_path ?assessment_instrument - assessment_instrument)
    (assessment_instrument_allocated ?assessment_instrument - assessment_instrument)
    (assessment_instrument_attached_to_credential_bundle ?assessment_instrument - assessment_instrument ?credential_bundle - credential_bundle)
    (learning_path_assessment_ready ?learning_path - learning_path)
    (learning_path_optional_module_confirmed ?learning_path - learning_path)
    (learning_path_assessments_validated ?learning_path - learning_path)
    (learning_path_has_accreditation_condition_assigned ?learning_path - learning_path)
    (learning_path_accreditation_accepted ?learning_path - learning_path)
    (learning_path_resource_package_attached ?learning_path - learning_path)
    (learning_path_ready_for_final_validation ?learning_path - learning_path)
    (available_project_requirement ?project_requirement - project_requirement)
    (learning_path_assigned_project_requirement ?learning_path - learning_path ?project_requirement - project_requirement)
    (learning_path_project_requirement_confirmed ?learning_path - learning_path)
    (learning_path_project_supervisor_allocated ?learning_path - learning_path)
    (learning_path_project_approved ?learning_path - learning_path)
    (available_accreditation_condition ?accreditation_condition - accreditation_condition)
    (learning_path_assigned_accreditation_condition ?learning_path - learning_path ?accreditation_condition - accreditation_condition)
    (available_resource_package ?resource_package - resource_package)
    (learning_path_assigned_resource_package ?learning_path - learning_path ?resource_package - resource_package)
    (available_optional_module ?optional_module - optional_module)
    (learning_path_assigned_optional_module ?learning_path - learning_path ?optional_module - optional_module)
    (available_research_project ?research_project - research_project)
    (learning_path_assigned_research_project ?learning_path - learning_path ?research_project - research_project)
    (available_academic_term ?academic_term - academic_term)
    (curriculum_item_scheduled_in_term ?curriculum_item - curriculum_item ?academic_term - academic_term)
    (theory_profile_competency_prepared ?theory_profile - theory_profile)
    (practical_profile_competency_prepared ?practical_profile - practical_profile)
    (learning_path_finalization_marker ?learning_path - learning_path)
  )
  (:action register_curriculum_item
    :parameters (?curriculum_item - curriculum_item)
    :precondition
      (and
        (not
          (curriculum_item_registered ?curriculum_item)
        )
        (not
          (certification_awarded ?curriculum_item)
        )
      )
    :effect (curriculum_item_registered ?curriculum_item)
  )
  (:action assign_certification_requirement_to_item
    :parameters (?curriculum_item - curriculum_item ?certification_requirement - certification_requirement)
    :precondition
      (and
        (curriculum_item_registered ?curriculum_item)
        (not
          (requirement_assigned ?curriculum_item)
        )
        (available_certification_requirement ?certification_requirement)
      )
    :effect
      (and
        (requirement_assigned ?curriculum_item)
        (curriculum_item_assigned_requirement ?curriculum_item ?certification_requirement)
        (not
          (available_certification_requirement ?certification_requirement)
        )
      )
  )
  (:action allocate_course_offering_to_item
    :parameters (?curriculum_item - curriculum_item ?course_offering - course_offering)
    :precondition
      (and
        (curriculum_item_registered ?curriculum_item)
        (requirement_assigned ?curriculum_item)
        (available_course_offering ?course_offering)
      )
    :effect
      (and
        (curriculum_item_allocated_course ?curriculum_item ?course_offering)
        (not
          (available_course_offering ?course_offering)
        )
      )
  )
  (:action confirm_course_allocation_for_item
    :parameters (?curriculum_item - curriculum_item ?course_offering - course_offering)
    :precondition
      (and
        (curriculum_item_registered ?curriculum_item)
        (requirement_assigned ?curriculum_item)
        (curriculum_item_allocated_course ?curriculum_item ?course_offering)
        (not
          (curriculum_item_confirmed ?curriculum_item)
        )
      )
    :effect (curriculum_item_confirmed ?curriculum_item)
  )
  (:action release_course_offering_from_item
    :parameters (?curriculum_item - curriculum_item ?course_offering - course_offering)
    :precondition
      (and
        (curriculum_item_allocated_course ?curriculum_item ?course_offering)
      )
    :effect
      (and
        (available_course_offering ?course_offering)
        (not
          (curriculum_item_allocated_course ?curriculum_item ?course_offering)
        )
      )
  )
  (:action assign_instructor_to_item
    :parameters (?curriculum_item - curriculum_item ?instructor - instructor)
    :precondition
      (and
        (curriculum_item_confirmed ?curriculum_item)
        (available_instructor ?instructor)
      )
    :effect
      (and
        (curriculum_item_assigned_instructor ?curriculum_item ?instructor)
        (not
          (available_instructor ?instructor)
        )
      )
  )
  (:action unassign_instructor_from_item
    :parameters (?curriculum_item - curriculum_item ?instructor - instructor)
    :precondition
      (and
        (curriculum_item_assigned_instructor ?curriculum_item ?instructor)
      )
    :effect
      (and
        (available_instructor ?instructor)
        (not
          (curriculum_item_assigned_instructor ?curriculum_item ?instructor)
        )
      )
  )
  (:action assign_optional_module_to_learning_path
    :parameters (?learning_path - learning_path ?optional_module - optional_module)
    :precondition
      (and
        (curriculum_item_confirmed ?learning_path)
        (available_optional_module ?optional_module)
      )
    :effect
      (and
        (learning_path_assigned_optional_module ?learning_path ?optional_module)
        (not
          (available_optional_module ?optional_module)
        )
      )
  )
  (:action release_optional_module_from_learning_path
    :parameters (?learning_path - learning_path ?optional_module - optional_module)
    :precondition
      (and
        (learning_path_assigned_optional_module ?learning_path ?optional_module)
      )
    :effect
      (and
        (available_optional_module ?optional_module)
        (not
          (learning_path_assigned_optional_module ?learning_path ?optional_module)
        )
      )
  )
  (:action assign_research_project_to_learning_path
    :parameters (?learning_path - learning_path ?research_project - research_project)
    :precondition
      (and
        (curriculum_item_confirmed ?learning_path)
        (available_research_project ?research_project)
      )
    :effect
      (and
        (learning_path_assigned_research_project ?learning_path ?research_project)
        (not
          (available_research_project ?research_project)
        )
      )
  )
  (:action unassign_research_project_from_learning_path
    :parameters (?learning_path - learning_path ?research_project - research_project)
    :precondition
      (and
        (learning_path_assigned_research_project ?learning_path ?research_project)
      )
    :effect
      (and
        (available_research_project ?research_project)
        (not
          (learning_path_assigned_research_project ?learning_path ?research_project)
        )
      )
  )
  (:action evaluate_competency_theory_for_profile
    :parameters (?theory_profile - theory_profile ?competency_domain - competency_domain ?course_offering - course_offering)
    :precondition
      (and
        (curriculum_item_confirmed ?theory_profile)
        (curriculum_item_allocated_course ?theory_profile ?course_offering)
        (theory_profile_competency_mapping ?theory_profile ?competency_domain)
        (not
          (competency_domain_theory_ready ?competency_domain)
        )
        (not
          (competency_domain_practice_ready ?competency_domain)
        )
      )
    :effect (competency_domain_theory_ready ?competency_domain)
  )
  (:action verify_profile_competency_with_instructor
    :parameters (?theory_profile - theory_profile ?competency_domain - competency_domain ?instructor - instructor)
    :precondition
      (and
        (curriculum_item_confirmed ?theory_profile)
        (curriculum_item_assigned_instructor ?theory_profile ?instructor)
        (theory_profile_competency_mapping ?theory_profile ?competency_domain)
        (competency_domain_theory_ready ?competency_domain)
        (not
          (theory_profile_competency_prepared ?theory_profile)
        )
      )
    :effect
      (and
        (theory_profile_competency_prepared ?theory_profile)
        (theory_profile_competency_verified ?theory_profile)
      )
  )
  (:action assign_elective_and_flag_competency_practice
    :parameters (?theory_profile - theory_profile ?competency_domain - competency_domain ?elective_option - elective_option)
    :precondition
      (and
        (curriculum_item_confirmed ?theory_profile)
        (theory_profile_competency_mapping ?theory_profile ?competency_domain)
        (available_elective_option ?elective_option)
        (not
          (theory_profile_competency_prepared ?theory_profile)
        )
      )
    :effect
      (and
        (competency_domain_practice_ready ?competency_domain)
        (theory_profile_competency_prepared ?theory_profile)
        (theory_profile_selected_elective ?theory_profile ?elective_option)
        (not
          (available_elective_option ?elective_option)
        )
      )
  )
  (:action finalize_competency_verification_for_profile
    :parameters (?theory_profile - theory_profile ?competency_domain - competency_domain ?course_offering - course_offering ?elective_option - elective_option)
    :precondition
      (and
        (curriculum_item_confirmed ?theory_profile)
        (curriculum_item_allocated_course ?theory_profile ?course_offering)
        (theory_profile_competency_mapping ?theory_profile ?competency_domain)
        (competency_domain_practice_ready ?competency_domain)
        (theory_profile_selected_elective ?theory_profile ?elective_option)
        (not
          (theory_profile_competency_verified ?theory_profile)
        )
      )
    :effect
      (and
        (competency_domain_theory_ready ?competency_domain)
        (theory_profile_competency_verified ?theory_profile)
        (available_elective_option ?elective_option)
        (not
          (theory_profile_selected_elective ?theory_profile ?elective_option)
        )
      )
  )
  (:action reserve_practical_experience_for_practical_profile
    :parameters (?practical_profile - practical_profile ?practical_experience_type - practical_experience_type ?course_offering - course_offering)
    :precondition
      (and
        (curriculum_item_confirmed ?practical_profile)
        (curriculum_item_allocated_course ?practical_profile ?course_offering)
        (practical_profile_assigned_experience ?practical_profile ?practical_experience_type)
        (not
          (experience_type_reserved ?practical_experience_type)
        )
        (not
          (experience_type_confirmed ?practical_experience_type)
        )
      )
    :effect (experience_type_reserved ?practical_experience_type)
  )
  (:action verify_practical_experience_with_instructor
    :parameters (?practical_profile - practical_profile ?practical_experience_type - practical_experience_type ?instructor - instructor)
    :precondition
      (and
        (curriculum_item_confirmed ?practical_profile)
        (curriculum_item_assigned_instructor ?practical_profile ?instructor)
        (practical_profile_assigned_experience ?practical_profile ?practical_experience_type)
        (experience_type_reserved ?practical_experience_type)
        (not
          (practical_profile_competency_prepared ?practical_profile)
        )
      )
    :effect
      (and
        (practical_profile_competency_prepared ?practical_profile)
        (practical_profile_competency_verified ?practical_profile)
      )
  )
  (:action assign_elective_and_flag_practical_experience
    :parameters (?practical_profile - practical_profile ?practical_experience_type - practical_experience_type ?elective_option - elective_option)
    :precondition
      (and
        (curriculum_item_confirmed ?practical_profile)
        (practical_profile_assigned_experience ?practical_profile ?practical_experience_type)
        (available_elective_option ?elective_option)
        (not
          (practical_profile_competency_prepared ?practical_profile)
        )
      )
    :effect
      (and
        (experience_type_confirmed ?practical_experience_type)
        (practical_profile_competency_prepared ?practical_profile)
        (practical_profile_selected_elective ?practical_profile ?elective_option)
        (not
          (available_elective_option ?elective_option)
        )
      )
  )
  (:action finalize_practical_experience_verification_for_practical_profile
    :parameters (?practical_profile - practical_profile ?practical_experience_type - practical_experience_type ?course_offering - course_offering ?elective_option - elective_option)
    :precondition
      (and
        (curriculum_item_confirmed ?practical_profile)
        (curriculum_item_allocated_course ?practical_profile ?course_offering)
        (practical_profile_assigned_experience ?practical_profile ?practical_experience_type)
        (experience_type_confirmed ?practical_experience_type)
        (practical_profile_selected_elective ?practical_profile ?elective_option)
        (not
          (practical_profile_competency_verified ?practical_profile)
        )
      )
    :effect
      (and
        (experience_type_reserved ?practical_experience_type)
        (practical_profile_competency_verified ?practical_profile)
        (available_elective_option ?elective_option)
        (not
          (practical_profile_selected_elective ?practical_profile ?elective_option)
        )
      )
  )
  (:action assemble_credential_bundle
    :parameters (?theory_profile - theory_profile ?practical_profile - practical_profile ?competency_domain - competency_domain ?practical_experience_type - practical_experience_type ?credential_bundle - credential_bundle)
    :precondition
      (and
        (theory_profile_competency_prepared ?theory_profile)
        (practical_profile_competency_prepared ?practical_profile)
        (theory_profile_competency_mapping ?theory_profile ?competency_domain)
        (practical_profile_assigned_experience ?practical_profile ?practical_experience_type)
        (competency_domain_theory_ready ?competency_domain)
        (experience_type_reserved ?practical_experience_type)
        (theory_profile_competency_verified ?theory_profile)
        (practical_profile_competency_verified ?practical_profile)
        (available_credential_bundle ?credential_bundle)
      )
    :effect
      (and
        (credential_bundle_created ?credential_bundle)
        (credential_bundle_includes_competency_domain ?credential_bundle ?competency_domain)
        (credential_bundle_includes_experience_type ?credential_bundle ?practical_experience_type)
        (not
          (available_credential_bundle ?credential_bundle)
        )
      )
  )
  (:action assemble_credential_bundle_with_slot_theory
    :parameters (?theory_profile - theory_profile ?practical_profile - practical_profile ?competency_domain - competency_domain ?practical_experience_type - practical_experience_type ?credential_bundle - credential_bundle)
    :precondition
      (and
        (theory_profile_competency_prepared ?theory_profile)
        (practical_profile_competency_prepared ?practical_profile)
        (theory_profile_competency_mapping ?theory_profile ?competency_domain)
        (practical_profile_assigned_experience ?practical_profile ?practical_experience_type)
        (competency_domain_practice_ready ?competency_domain)
        (experience_type_reserved ?practical_experience_type)
        (not
          (theory_profile_competency_verified ?theory_profile)
        )
        (practical_profile_competency_verified ?practical_profile)
        (available_credential_bundle ?credential_bundle)
      )
    :effect
      (and
        (credential_bundle_created ?credential_bundle)
        (credential_bundle_includes_competency_domain ?credential_bundle ?competency_domain)
        (credential_bundle_includes_experience_type ?credential_bundle ?practical_experience_type)
        (credential_bundle_specialization_slot_theory ?credential_bundle)
        (not
          (available_credential_bundle ?credential_bundle)
        )
      )
  )
  (:action assemble_credential_bundle_with_slot_practical
    :parameters (?theory_profile - theory_profile ?practical_profile - practical_profile ?competency_domain - competency_domain ?practical_experience_type - practical_experience_type ?credential_bundle - credential_bundle)
    :precondition
      (and
        (theory_profile_competency_prepared ?theory_profile)
        (practical_profile_competency_prepared ?practical_profile)
        (theory_profile_competency_mapping ?theory_profile ?competency_domain)
        (practical_profile_assigned_experience ?practical_profile ?practical_experience_type)
        (competency_domain_theory_ready ?competency_domain)
        (experience_type_confirmed ?practical_experience_type)
        (theory_profile_competency_verified ?theory_profile)
        (not
          (practical_profile_competency_verified ?practical_profile)
        )
        (available_credential_bundle ?credential_bundle)
      )
    :effect
      (and
        (credential_bundle_created ?credential_bundle)
        (credential_bundle_includes_competency_domain ?credential_bundle ?competency_domain)
        (credential_bundle_includes_experience_type ?credential_bundle ?practical_experience_type)
        (credential_bundle_specialization_slot_practical ?credential_bundle)
        (not
          (available_credential_bundle ?credential_bundle)
        )
      )
  )
  (:action assemble_credential_bundle_with_slots_theory_and_practical
    :parameters (?theory_profile - theory_profile ?practical_profile - practical_profile ?competency_domain - competency_domain ?practical_experience_type - practical_experience_type ?credential_bundle - credential_bundle)
    :precondition
      (and
        (theory_profile_competency_prepared ?theory_profile)
        (practical_profile_competency_prepared ?practical_profile)
        (theory_profile_competency_mapping ?theory_profile ?competency_domain)
        (practical_profile_assigned_experience ?practical_profile ?practical_experience_type)
        (competency_domain_practice_ready ?competency_domain)
        (experience_type_confirmed ?practical_experience_type)
        (not
          (theory_profile_competency_verified ?theory_profile)
        )
        (not
          (practical_profile_competency_verified ?practical_profile)
        )
        (available_credential_bundle ?credential_bundle)
      )
    :effect
      (and
        (credential_bundle_created ?credential_bundle)
        (credential_bundle_includes_competency_domain ?credential_bundle ?competency_domain)
        (credential_bundle_includes_experience_type ?credential_bundle ?practical_experience_type)
        (credential_bundle_specialization_slot_theory ?credential_bundle)
        (credential_bundle_specialization_slot_practical ?credential_bundle)
        (not
          (available_credential_bundle ?credential_bundle)
        )
      )
  )
  (:action lock_credential_bundle
    :parameters (?credential_bundle - credential_bundle ?theory_profile - theory_profile ?course_offering - course_offering)
    :precondition
      (and
        (credential_bundle_created ?credential_bundle)
        (theory_profile_competency_prepared ?theory_profile)
        (curriculum_item_allocated_course ?theory_profile ?course_offering)
        (not
          (credential_bundle_locked ?credential_bundle)
        )
      )
    :effect (credential_bundle_locked ?credential_bundle)
  )
  (:action attach_assessment_to_credential_bundle
    :parameters (?learning_path - learning_path ?assessment_instrument - assessment_instrument ?credential_bundle - credential_bundle)
    :precondition
      (and
        (curriculum_item_confirmed ?learning_path)
        (learning_path_contains_credential_bundle ?learning_path ?credential_bundle)
        (learning_path_assigned_assessment_instrument ?learning_path ?assessment_instrument)
        (available_assessment_instrument ?assessment_instrument)
        (credential_bundle_created ?credential_bundle)
        (credential_bundle_locked ?credential_bundle)
        (not
          (assessment_instrument_allocated ?assessment_instrument)
        )
      )
    :effect
      (and
        (assessment_instrument_allocated ?assessment_instrument)
        (assessment_instrument_attached_to_credential_bundle ?assessment_instrument ?credential_bundle)
        (not
          (available_assessment_instrument ?assessment_instrument)
        )
      )
  )
  (:action confirm_learning_path_assessment_readiness
    :parameters (?learning_path - learning_path ?assessment_instrument - assessment_instrument ?credential_bundle - credential_bundle ?course_offing - course_offering)
    :precondition
      (and
        (curriculum_item_confirmed ?learning_path)
        (learning_path_assigned_assessment_instrument ?learning_path ?assessment_instrument)
        (assessment_instrument_allocated ?assessment_instrument)
        (assessment_instrument_attached_to_credential_bundle ?assessment_instrument ?credential_bundle)
        (curriculum_item_allocated_course ?learning_path ?course_offing)
        (not
          (credential_bundle_specialization_slot_theory ?credential_bundle)
        )
        (not
          (learning_path_assessment_ready ?learning_path)
        )
      )
    :effect (learning_path_assessment_ready ?learning_path)
  )
  (:action assign_accreditation_condition_to_learning_path
    :parameters (?learning_path - learning_path ?accreditation_condition - accreditation_condition)
    :precondition
      (and
        (curriculum_item_confirmed ?learning_path)
        (available_accreditation_condition ?accreditation_condition)
        (not
          (learning_path_has_accreditation_condition_assigned ?learning_path)
        )
      )
    :effect
      (and
        (learning_path_has_accreditation_condition_assigned ?learning_path)
        (learning_path_assigned_accreditation_condition ?learning_path ?accreditation_condition)
        (not
