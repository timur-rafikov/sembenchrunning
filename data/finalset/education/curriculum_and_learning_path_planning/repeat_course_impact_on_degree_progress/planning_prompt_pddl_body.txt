    (course_open_for_registration ?course - course)
    (course_attempt_record ?learner_entity - learner_entity ?course - course)
    (section_open_for_registration ?course_section - course_section)
    (entity_section_enrollment ?learner_entity - learner_entity ?course_section - course_section)
    (instructor_available ?instructor - instructor_person)
    (instructor_assigned_to_entity ?learner_entity - learner_entity ?instructor - instructor_person)
    (prerequisite_material_available ?prerequisite_material - prerequisite_material)
    (undergrad_profile_prerequisite_record ?undergraduate_profile - undergraduate_profile ?prerequisite_material - prerequisite_material)
    (graduate_profile_prerequisite_record ?graduate_profile - graduate_profile ?prerequisite_material - prerequisite_material)
    (profile_term_session_assignment ?undergraduate_profile - undergraduate_profile ?term_session - term_session)

    (semester_session_confirmed ?semester_session - semester_session)
    (semester_session_prerequisite_assigned ?semester_session - semester_session)
    (graduate_prerequisite_satisfied ?graduate_profile - graduate_profile)
    (offering_pending ?course_offering - course_offering)
    (offering_active ?course_offering - course_offering)

    (offering_semester_session_link ?course_offering - course_offering ?semester_session - semester_session)
    (offering_has_term_component ?course_offering - course_offering)
    (offering_has_semester_component ?course_offering - course_offering)
    (offering_finalized ?course_offering - course_offering)
    (degree_program_has_undergrad_profile ?degree_program - degree_program ?undergraduate_profile - undergraduate_profile)
    (degree_program_has_graduate_profile ?degree_program - degree_program ?graduate_profile - graduate_profile)
    (degree_program_offering_link ?degree_program - degree_program ?course_offering - course_offering)

    (degree_program_assessment_ready ?degree_program - degree_program)
    (degree_program_milestone_qualified ?degree_program - degree_program)
    (capstone_attached_to_program ?degree_program - degree_program)
    (degree_program_advisory_acknowledged ?degree_program - degree_program)
    (enrollment_option_flow_enabled ?degree_program - degree_program)
    (degree_program_milestone_signed_off ?degree_program - degree_program)
    (advisory_note_available ?advisory_note - advisory_note)
    (degree_program_advisory_link ?degree_program - degree_program ?advisory_note - advisory_note)
    (advisory_gate_enabled ?degree_program - degree_program)
    (advisory_gate_acknowledged ?degree_program - degree_program)
    (advisory_gate_finalized ?degree_program - degree_program)

  (:action initialize_learner_record
    :parameters (?learner_entity - learner_entity)
    :precondition
      (and
        (not
          (learner_onboarded ?learner_entity)
        )
        (not
          (progression_flag_recorded ?learner_entity)
        )
      )
    :effect (learner_onboarded ?learner_entity)
  )
  (:action record_course_intent
    :parameters (?learner_entity - learner_entity ?course - course)
    :precondition
      (and
        (learner_onboarded ?learner_entity)
        (not
          (course_intent_declared ?learner_entity)
        )
        (course_open_for_registration ?course)
      )
    :effect
      (and
        (course_intent_declared ?learner_entity)
        (course_attempt_record ?learner_entity ?course)
        (not
          (course_open_for_registration ?course)
        )
      )
  )
  (:action assign_section_to_learner
    :parameters (?learner_entity - learner_entity ?course_section - course_section)
    :precondition
      (and
        (learner_onboarded ?learner_entity)
        (course_intent_declared ?learner_entity)
        (section_open_for_registration ?course_section)
      )
    :effect
      (and
        (entity_section_enrollment ?learner_entity ?course_section)
        (not
          (section_open_for_registration ?course_section)
        )
      )
  )
  (:action finalize_enrollment_record
    :parameters (?learner_entity - learner_entity ?course_section - course_section)

      (and
        (learner_onboarded ?learner_entity)
        (course_intent_declared ?learner_entity)
        (entity_section_enrollment ?learner_entity ?course_section)
        (not
          (entity_record_verified ?learner_entity)
        )
      )
    :effect (entity_record_verified ?learner_entity)
  )
  (:action withdraw_from_section
    :parameters (?learner_entity - learner_entity ?course_section - course_section)

      (and
        (entity_section_enrollment ?learner_entity ?course_section)
      )
    :effect
      (and
        (section_open_for_registration ?course_section)
        (not
          (entity_section_enrollment ?learner_entity ?course_section)
        )
      )
  )
  (:action assign_instructor_to_learner
    :parameters (?learner_entity - learner_entity ?instructor - instructor_person)
    :precondition
      (and
        (entity_record_verified ?learner_entity)
        (instructor_available ?instructor)
      )
    :effect
      (and
        (instructor_assigned_to_entity ?learner_entity ?instructor)
        (not
          (instructor_available ?instructor)
        )
      )
  )
  (:action unassign_instructor_from_learner
    :parameters (?learner_entity - learner_entity ?instructor - instructor_person)
    :precondition
      (and
        (instructor_assigned_to_entity ?learner_entity ?instructor)
      )
    :effect
      (and
        (instructor_available ?instructor)
        (not
          (instructor_assigned_to_entity ?learner_entity ?instructor)
        )
      )
  )
  (:action attach_assessment_component_to_program
    :parameters (?degree_program - degree_program ?assessment_component - assessment_component)

      (and
        (entity_record_verified ?degree_program)
        (assessment_component_available ?assessment_component)
      )

        (degree_program_assessment_component_link ?degree_program ?assessment_component)
        (not
          (assessment_component_available ?assessment_component)
        )
      )

    :parameters (?degree_program - degree_program ?assessment_component - assessment_component)

        )
      )
  )
  (:action attach_credential_to_program
    :parameters (?degree_program - degree_program ?credential - credential)

      (and

        (not
          (credential_available ?credential)
        )
      )
  )
  (:action detach_credential_from_program
    :parameters (?degree_program - degree_program ?credential - credential)
    :precondition
      (and
        (degree_program_credential_link ?degree_program ?credential)
      )
    :effect
      (and
        (credential_available ?credential)
        (not
          (degree_program_credential_link ?degree_program ?credential)
        )
      )
  )
  (:action confirm_term_session_for_profile
    :parameters (?undergraduate_profile - undergraduate_profile ?term_session - term_session ?course_section - course_section)

        (entity_record_verified ?undergraduate_profile)

  )
  (:action approve_instructor_and_mark_prerequisite
    :parameters (?undergraduate_profile - undergraduate_profile ?term_session - term_session ?instructor - instructor_person)
    :precondition
      (and
        (entity_record_verified ?undergraduate_profile)
        (instructor_assigned_to_entity ?undergraduate_profile ?instructor)
        (profile_term_session_assignment ?undergraduate_profile ?term_session)
        (term_session_confirmed ?term_session)
        (not
          (undergrad_profile_ready ?undergraduate_profile)
        )
      )
    :effect
      (and
        (undergrad_profile_ready ?undergraduate_profile)
        (undergrad_prerequisite_satisfied ?undergraduate_profile)
      )
  )
  (:action assign_prerequisite_material
    :parameters (?undergraduate_profile - undergraduate_profile ?term_session - term_session ?prerequisite_material - prerequisite_material)
    :precondition
      (and
        (entity_record_verified ?undergraduate_profile)
        (profile_term_session_assignment ?undergraduate_profile ?term_session)
        (prerequisite_material_available ?prerequisite_material)
        (not
          (undergrad_profile_ready ?undergraduate_profile)
        )
      )
    :effect
      (and
        (term_session_prerequisite_assigned ?term_session)
        (undergrad_profile_ready ?undergraduate_profile)
        (undergrad_profile_prerequisite_record ?undergraduate_profile ?prerequisite_material)
        (not
          (prerequisite_material_available ?prerequisite_material)
        )
      )
  )
  (:action satisfy_prerequisite_and_release_material
    :parameters (?undergraduate_profile - undergraduate_profile ?term_session - term_session ?course_section - course_section ?prerequisite_material - prerequisite_material)

      (and
        (entity_record_verified ?undergraduate_profile)
        (entity_section_enrollment ?undergraduate_profile ?course_section)

        (term_session_prerequisite_assigned ?term_session)
        (undergrad_profile_prerequisite_record ?undergraduate_profile ?prerequisite_material)
        (not
          (undergrad_prerequisite_satisfied ?undergraduate_profile)
        )
      )
    :effect
      (and
        (term_session_confirmed ?term_session)
        (undergrad_prerequisite_satisfied ?undergraduate_profile)
        (prerequisite_material_available ?prerequisite_material)
        (not
          (undergrad_profile_prerequisite_record ?undergraduate_profile ?prerequisite_material)
        )
      )
  )

    :parameters (?graduate_profile - graduate_profile ?semester_session - semester_session ?course_section - course_section)

          (prerequisite_material_available ?prerequisite_material)

      )
    :effect
      (and
        (offering_active ?course_offering)
        (offering_term_session_link ?course_offering ?term_session)

        (not
          (offering_pending ?course_offering)

  (:action assemble_and_activate_offering_with_term_component

        (not
          (offering_pending ?course_offering)
