## Planning domain (PDDL)

(define (domain attendance_window_compatible_schedule_planning)
  (:requirements :strips :typing :negative-preconditions)
  (:types actor_or_resource - object physical_resource - object temporal_resource - object registration_artifact - object registrable - registration_artifact seat_unit - actor_or_resource course - actor_or_resource approver - actor_or_resource department - actor_or_resource compliance_document - actor_or_resource attendance_deadline - actor_or_resource teaching_assignment_token - actor_or_resource authorization_artifact - actor_or_resource room_request - physical_resource venue - physical_resource external_approval_token - physical_resource time_slot - temporal_resource attendance_window - temporal_resource timetable_candidate - temporal_resource participant_role_group - registrable offering_group - registrable student - participant_role_group instructor - participant_role_group course_section - offering_group)
  (:predicates
    (registrable_created ?registrable - registrable)
    (registrable_validated ?registrable - registrable)
    (registrable_seat_hold_placed ?registrable - registrable)
    (registrable_enrollment_finalized ?registrable - registrable)
    (registrable_ready_for_enrollment ?registrable - registrable)
    (registrable_enrollment_committed ?registrable - registrable)
    (seat_available ?seat_unit - seat_unit)
    (registrable_holds_seat ?registrable - registrable ?seat_unit - seat_unit)
    (course_available ?course - course)
    (registrable_selected_course ?registrable - registrable ?course - course)
    (approver_available ?approver - approver)
    (registrable_assigned_approver ?registrable - registrable ?approver - approver)
    (room_request_available ?room_request - room_request)
    (student_has_room_request ?student - student ?room_request - room_request)
    (instructor_has_room_request ?instructor - instructor ?room_request - room_request)
    (student_available_time_slot ?student - student ?time_slot - time_slot)
    (time_slot_marked_by_student ?time_slot - time_slot)
    (time_slot_marked_by_room_request ?time_slot - time_slot)
    (student_availability_confirmed ?student - student)
    (instructor_has_attendance_window ?instructor - instructor ?attendance_window - attendance_window)
    (attendance_window_reserved_by_instructor ?attendance_window - attendance_window)
    (attendance_window_reserved_by_room_request ?attendance_window - attendance_window)
    (instructor_availability_confirmed ?instructor - instructor)
    (timetable_candidate_pending ?timetable_candidate - timetable_candidate)
    (timetable_candidate_configured ?timetable_candidate - timetable_candidate)
    (candidate_assigned_time_slot ?timetable_candidate - timetable_candidate ?time_slot - time_slot)
    (candidate_assigned_attendance_window ?timetable_candidate - timetable_candidate ?attendance_window - attendance_window)
    (candidate_requires_department_approval ?timetable_candidate - timetable_candidate)
    (candidate_requires_external_approval ?timetable_candidate - timetable_candidate)
    (timetable_candidate_ready_for_venue_assignment ?timetable_candidate - timetable_candidate)
    (section_has_student ?course_section - course_section ?student - student)
    (section_has_instructor ?course_section - course_section ?instructor - instructor)
    (section_links_candidate ?course_section - course_section ?timetable_candidate - timetable_candidate)
    (venue_available ?venue - venue)
    (section_has_venue ?course_section - course_section ?venue - venue)
    (venue_reserved ?venue - venue)
    (venue_assigned_to_candidate ?venue - venue ?timetable_candidate - timetable_candidate)
    (section_venue_assigned ?course_section - course_section)
    (section_instructor_assigned ?course_section - course_section)
    (section_compliance_passed ?course_section - course_section)
    (department_approval_attached ?course_section - course_section)
    (department_requirements_met ?course_section - course_section)
    (section_compliance_document_attached ?course_section - course_section)
    (section_validated ?course_section - course_section)
    (external_approval_token_available ?external_approval_token - external_approval_token)
    (section_has_external_approval_token ?course_section - course_section ?external_approval_token - external_approval_token)
    (external_approval_attached ?course_section - course_section)
    (authorization_requested ?course_section - course_section)
    (authorization_granted ?course_section - course_section)
    (department_token_available ?department - department)
    (section_attached_to_department ?course_section - course_section ?department - department)
    (compliance_document_available ?compliance_document - compliance_document)
    (section_attached_compliance_document ?course_section - course_section ?compliance_document - compliance_document)
    (teaching_assignment_token_available ?teaching_assignment_token - teaching_assignment_token)
    (section_assigned_teaching_token ?course_section - course_section ?teaching_assignment_token - teaching_assignment_token)
    (authorization_artifact_available ?authorization_artifact - authorization_artifact)
    (authorization_artifact_attached_to_section ?course_section - course_section ?authorization_artifact - authorization_artifact)
    (attendance_deadline_available ?attendance_deadline - attendance_deadline)
    (registrable_bound_to_attendance_deadline ?registrable - registrable ?attendance_deadline - attendance_deadline)
    (student_ready_for_candidate ?student - student)
    (instructor_ready_for_candidate ?instructor - instructor)
    (finalization_locked ?course_section - course_section)
  )
  (:action submit_registrable
    :parameters (?registrable - registrable)
    :precondition
      (and
        (not
          (registrable_created ?registrable)
        )
        (not
          (registrable_enrollment_finalized ?registrable)
        )
      )
    :effect (registrable_created ?registrable)
  )
  (:action place_registrable_seat_hold
    :parameters (?registrable - registrable ?seat_unit - seat_unit)
    :precondition
      (and
        (registrable_created ?registrable)
        (not
          (registrable_seat_hold_placed ?registrable)
        )
        (seat_available ?seat_unit)
      )
    :effect
      (and
        (registrable_seat_hold_placed ?registrable)
        (registrable_holds_seat ?registrable ?seat_unit)
        (not
          (seat_available ?seat_unit)
        )
      )
  )
  (:action select_course_for_registrable
    :parameters (?registrable - registrable ?course - course)
    :precondition
      (and
        (registrable_created ?registrable)
        (registrable_seat_hold_placed ?registrable)
        (course_available ?course)
      )
    :effect
      (and
        (registrable_selected_course ?registrable ?course)
        (not
          (course_available ?course)
        )
      )
  )
  (:action validate_registrable
    :parameters (?registrable - registrable ?course - course)
    :precondition
      (and
        (registrable_created ?registrable)
        (registrable_seat_hold_placed ?registrable)
        (registrable_selected_course ?registrable ?course)
        (not
          (registrable_validated ?registrable)
        )
      )
    :effect (registrable_validated ?registrable)
  )
  (:action deselect_course_for_registrable
    :parameters (?registrable - registrable ?course - course)
    :precondition
      (and
        (registrable_selected_course ?registrable ?course)
      )
    :effect
      (and
        (course_available ?course)
        (not
          (registrable_selected_course ?registrable ?course)
        )
      )
  )
  (:action assign_approver_to_registrable
    :parameters (?registrable - registrable ?approver - approver)
    :precondition
      (and
        (registrable_validated ?registrable)
        (approver_available ?approver)
      )
    :effect
      (and
        (registrable_assigned_approver ?registrable ?approver)
        (not
          (approver_available ?approver)
        )
      )
  )
  (:action unassign_approver_from_registrable
    :parameters (?registrable - registrable ?approver - approver)
    :precondition
      (and
        (registrable_assigned_approver ?registrable ?approver)
      )
    :effect
      (and
        (approver_available ?approver)
        (not
          (registrable_assigned_approver ?registrable ?approver)
        )
      )
  )
  (:action attach_teaching_assignment_token_to_section
    :parameters (?course_section - course_section ?teaching_assignment_token - teaching_assignment_token)
    :precondition
      (and
        (registrable_validated ?course_section)
        (teaching_assignment_token_available ?teaching_assignment_token)
      )
    :effect
      (and
        (section_assigned_teaching_token ?course_section ?teaching_assignment_token)
        (not
          (teaching_assignment_token_available ?teaching_assignment_token)
        )
      )
  )
  (:action detach_teaching_assignment_token_from_section
    :parameters (?course_section - course_section ?teaching_assignment_token - teaching_assignment_token)
    :precondition
      (and
        (section_assigned_teaching_token ?course_section ?teaching_assignment_token)
      )
    :effect
      (and
        (teaching_assignment_token_available ?teaching_assignment_token)
        (not
          (section_assigned_teaching_token ?course_section ?teaching_assignment_token)
        )
      )
  )
  (:action attach_authorization_artifact_to_section
    :parameters (?course_section - course_section ?authorization_artifact - authorization_artifact)
    :precondition
      (and
        (registrable_validated ?course_section)
        (authorization_artifact_available ?authorization_artifact)
      )
    :effect
      (and
        (authorization_artifact_attached_to_section ?course_section ?authorization_artifact)
        (not
          (authorization_artifact_available ?authorization_artifact)
        )
      )
  )
  (:action detach_authorization_artifact_from_section
    :parameters (?course_section - course_section ?authorization_artifact - authorization_artifact)
    :precondition
      (and
        (authorization_artifact_attached_to_section ?course_section ?authorization_artifact)
      )
    :effect
      (and
        (authorization_artifact_available ?authorization_artifact)
        (not
          (authorization_artifact_attached_to_section ?course_section ?authorization_artifact)
        )
      )
  )
  (:action student_reserve_time_slot
    :parameters (?student - student ?time_slot - time_slot ?course - course)
    :precondition
      (and
        (registrable_validated ?student)
        (registrable_selected_course ?student ?course)
        (student_available_time_slot ?student ?time_slot)
        (not
          (time_slot_marked_by_student ?time_slot)
        )
        (not
          (time_slot_marked_by_room_request ?time_slot)
        )
      )
    :effect (time_slot_marked_by_student ?time_slot)
  )
  (:action confirm_student_availability_with_approver
    :parameters (?student - student ?time_slot - time_slot ?approver - approver)
    :precondition
      (and
        (registrable_validated ?student)
        (registrable_assigned_approver ?student ?approver)
        (student_available_time_slot ?student ?time_slot)
        (time_slot_marked_by_student ?time_slot)
        (not
          (student_ready_for_candidate ?student)
        )
      )
    :effect
      (and
        (student_ready_for_candidate ?student)
        (student_availability_confirmed ?student)
      )
  )
  (:action reserve_time_slot_for_room_request
    :parameters (?student - student ?time_slot - time_slot ?room_request - room_request)
    :precondition
      (and
        (registrable_validated ?student)
        (student_available_time_slot ?student ?time_slot)
        (room_request_available ?room_request)
        (not
          (student_ready_for_candidate ?student)
        )
      )
    :effect
      (and
        (time_slot_marked_by_room_request ?time_slot)
        (student_ready_for_candidate ?student)
        (student_has_room_request ?student ?room_request)
        (not
          (room_request_available ?room_request)
        )
      )
  )
  (:action confirm_student_time_slot_assignment
    :parameters (?student - student ?time_slot - time_slot ?course - course ?room_request - room_request)

      (and
        (registrable_validated ?student)
        (registrable_selected_course ?student ?course)
        (student_available_time_slot ?student ?time_slot)
        (time_slot_marked_by_room_request ?time_slot)
        (student_has_room_request ?student ?room_request)
        (not
          (student_availability_confirmed ?student)
        )
      )
    :effect
      (and
        (time_slot_marked_by_student ?time_slot)
        (student_availability_confirmed ?student)
        (room_request_available ?room_request)
        (not
          (student_has_room_request ?student ?room_request)
        )
      )
  )
  (:action instructor_reserve_time_slot

  )
  (:action confirm_instructor_availability
    :parameters (?instructor - instructor ?attendance_window - attendance_window ?approver - approver)

  )
  (:action reserve_attendance_window_by_room_request

      (and
        (attendance_window_reserved_by_room_request ?attendance_window)

        (instructor_has_room_request ?instructor ?room_request)
        (not
          (room_request_available ?room_request)
        )
      )
  )
  (:action confirm_instructor_time_slot_assignment
    :parameters (?instructor - instructor ?attendance_window - attendance_window ?course - course ?room_request - room_request)

      (and
        (registrable_validated ?instructor)

          (instructor_has_room_request ?instructor ?room_request)

  (:action create_timetable_candidate
    :parameters (?student - student ?instructor - instructor ?time_slot - time_slot ?attendance_window - attendance_window ?timetable_candidate - timetable_candidate)

        (attendance_window_reserved_by_instructor ?attendance_window)
        (student_availability_confirmed ?student)
        (instructor_availability_confirmed ?instructor)
        (timetable_candidate_pending ?timetable_candidate)
      )
    :effect
      (and
        (timetable_candidate_configured ?timetable_candidate)
        (candidate_assigned_time_slot ?timetable_candidate ?time_slot)
        (candidate_assigned_attendance_window ?timetable_candidate ?attendance_window)
        (not
          (timetable_candidate_pending ?timetable_candidate)
        )
      )
  )
  (:action create_timetable_candidate_with_room_claim
    :parameters (?student - student ?instructor - instructor ?time_slot - time_slot ?attendance_window - attendance_window ?timetable_candidate - timetable_candidate)

          (timetable_candidate_pending ?timetable_candidate)

        (not
          (timetable_candidate_pending ?timetable_candidate)

          (timetable_candidate_pending ?timetable_candidate)
        )

        (not
          (section_instructor_assigned ?course_section)

  (:action assign_teaching_token_phase2
    :parameters (?course_section - course_section ?teaching_assignment_token - teaching_assignment_token ?approver - approver ?venue - venue ?timetable_candidate - timetable_candidate)
    :precondition
