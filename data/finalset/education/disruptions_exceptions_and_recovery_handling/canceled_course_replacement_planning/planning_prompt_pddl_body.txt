## Planning domain (PDDL)

(define (domain canceled_course_replacement_planning_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_root - object offering_root - object schedule_root - object case_subject_root - object academic_case - case_subject_root replacement_option - resource_root course_offering - resource_root approver - resource_root policy_exception_template - resource_root approval_token - resource_root student_consent_form - resource_root instructor_availability - resource_root department_signoff - resource_root supporting_document - offering_root course_section - offering_root external_authorization - offering_root primary_schedule_constraint - schedule_root secondary_schedule_constraint - schedule_root replacement_proposal - schedule_root requirement_root - academic_case case_root - academic_case affected_enrollment - requirement_root affected_requirement - requirement_root administrative_case_file - case_root)
  (:predicates
    (case_subject_initiated ?academic_case - academic_case)
    (case_subject_ready_for_processing ?academic_case - academic_case)
    (candidate_option_attached ?academic_case - academic_case)
    (replacement_enacted ?academic_case - academic_case)
    (records_updated ?academic_case - academic_case)
    (case_subject_commit_confirmed ?academic_case - academic_case)
    (replacement_option_available ?replacement_option - replacement_option)
    (case_subject_assigned_option ?academic_case - academic_case ?replacement_option - replacement_option)
    (offering_available ?course_offering - course_offering)
    (case_subject_assigned_offering ?academic_case - academic_case ?course_offering - course_offering)
    (approver_available ?approver - approver)
    (case_subject_assigned_approver ?academic_case - academic_case ?approver - approver)
    (supporting_document_available ?supporting_document - supporting_document)
    (enrollment_document_attached ?affected_enrollment - affected_enrollment ?supporting_document - supporting_document)
    (requirement_document_attached ?affected_requirement - affected_requirement ?supporting_document - supporting_document)
    (enrollment_primary_schedule_constraint ?affected_enrollment - affected_enrollment ?primary_schedule_constraint - primary_schedule_constraint)
    (primary_schedule_constraint_validated ?primary_schedule_constraint - primary_schedule_constraint)
    (primary_schedule_constraint_document_validated ?primary_schedule_constraint - primary_schedule_constraint)
    (enrollment_constraint_confirmed ?affected_enrollment - affected_enrollment)
    (requirement_secondary_schedule_constraint ?affected_requirement - affected_requirement ?secondary_schedule_constraint - secondary_schedule_constraint)
    (secondary_schedule_constraint_validated ?secondary_schedule_constraint - secondary_schedule_constraint)
    (secondary_schedule_constraint_document_validated ?secondary_schedule_constraint - secondary_schedule_constraint)
    (requirement_constraint_confirmed ?affected_requirement - affected_requirement)
    (proposal_draft ?replacement_proposal - replacement_proposal)
    (proposal_constructed ?replacement_proposal - replacement_proposal)
    (proposal_primary_schedule_constraint_link ?replacement_proposal - replacement_proposal ?primary_schedule_constraint - primary_schedule_constraint)
    (proposal_secondary_schedule_constraint_link ?replacement_proposal - replacement_proposal ?secondary_schedule_constraint - secondary_schedule_constraint)
    (proposal_primary_document_verified ?replacement_proposal - replacement_proposal)
    (proposal_secondary_document_verified ?replacement_proposal - replacement_proposal)
    (proposal_section_locked ?replacement_proposal - replacement_proposal)
    (file_associated_enrollment ?administrative_case_file - administrative_case_file ?affected_enrollment - affected_enrollment)
    (file_associated_requirement ?administrative_case_file - administrative_case_file ?affected_requirement - affected_requirement)
    (file_linked_proposal ?administrative_case_file - administrative_case_file ?replacement_proposal - replacement_proposal)
    (section_available ?course_section - course_section)
    (file_has_section ?administrative_case_file - administrative_case_file ?course_section - course_section)
    (section_reserved ?course_section - course_section)
    (section_assigned_to_proposal ?course_section - course_section ?replacement_proposal - replacement_proposal)
    (file_section_allocated ?administrative_case_file - administrative_case_file)
    (instructor_approval_received ?administrative_case_file - administrative_case_file)
    (authorization_stage_one_completed ?administrative_case_file - administrative_case_file)
    (file_has_policy_exception_template ?administrative_case_file - administrative_case_file)
    (file_policy_exception_applied ?administrative_case_file - administrative_case_file)
    (authorization_stage_two_completed ?administrative_case_file - administrative_case_file)
    (file_allocations_finalized ?administrative_case_file - administrative_case_file)
    (external_authorization_available ?external_authorization - external_authorization)
    (file_linked_external_authorization ?administrative_case_file - administrative_case_file ?external_authorization - external_authorization)
    (file_external_authorization_attached ?administrative_case_file - administrative_case_file)
    (file_followup_flag ?administrative_case_file - administrative_case_file)
    (file_department_signoff_recorded ?administrative_case_file - administrative_case_file)
    (policy_exception_template_available ?policy_exception_template - policy_exception_template)
    (file_linked_policy_exception_template ?administrative_case_file - administrative_case_file ?policy_exception_template - policy_exception_template)
    (approval_token_available ?approval_token - approval_token)
    (file_has_approval_token ?administrative_case_file - administrative_case_file ?approval_token - approval_token)
    (instructor_availability_available ?instructor_availability - instructor_availability)
    (file_linked_instructor_availability ?administrative_case_file - administrative_case_file ?instructor_availability - instructor_availability)
    (department_signoff_available ?department_signoff - department_signoff)
    (file_linked_department_signoff ?administrative_case_file - administrative_case_file ?department_signoff - department_signoff)
    (student_consent_form_available ?student_consent - student_consent_form)
    (case_subject_linked_student_consent_form ?academic_case - academic_case ?student_consent - student_consent_form)
    (enrollment_ready ?affected_enrollment - affected_enrollment)
    (requirement_ready ?affected_requirement - affected_requirement)
    (file_finalized ?administrative_case_file - administrative_case_file)
  )
  (:action initiate_academic_case
    :parameters (?academic_case - academic_case)
    :precondition
      (and
        (not
          (case_subject_initiated ?academic_case)
        )
        (not
          (replacement_enacted ?academic_case)
        )
      )
    :effect (case_subject_initiated ?academic_case)
  )
  (:action assign_replacement_option_to_case
    :parameters (?academic_case - academic_case ?replacement_option - replacement_option)
    :precondition
      (and
        (case_subject_initiated ?academic_case)
        (not
          (candidate_option_attached ?academic_case)
        )
        (replacement_option_available ?replacement_option)
      )
    :effect
      (and
        (candidate_option_attached ?academic_case)
        (case_subject_assigned_option ?academic_case ?replacement_option)
        (not
          (replacement_option_available ?replacement_option)
        )
      )
  )
  (:action attach_course_offering_to_case
    :parameters (?academic_case - academic_case ?course_offering - course_offering)
    :precondition
      (and
        (case_subject_initiated ?academic_case)
        (candidate_option_attached ?academic_case)
        (offering_available ?course_offering)
      )
    :effect
      (and
        (case_subject_assigned_offering ?academic_case ?course_offering)
        (not
          (offering_available ?course_offering)
        )
      )
  )
  (:action confirm_course_offering_for_case
    :parameters (?academic_case - academic_case ?course_offering - course_offering)
    :precondition
      (and
        (case_subject_initiated ?academic_case)
        (candidate_option_attached ?academic_case)
        (case_subject_assigned_offering ?academic_case ?course_offering)
        (not
          (case_subject_ready_for_processing ?academic_case)
        )
      )
    :effect (case_subject_ready_for_processing ?academic_case)
  )
  (:action release_course_offering_reservation
    :parameters (?academic_case - academic_case ?course_offering - course_offering)
    :precondition
      (and
        (case_subject_assigned_offering ?academic_case ?course_offering)
      )
    :effect
      (and
        (offering_available ?course_offering)
        (not
          (case_subject_assigned_offering ?academic_case ?course_offering)
        )
      )
  )
  (:action assign_approver_to_case
    :parameters (?academic_case - academic_case ?approver - approver)
    :precondition
      (and
        (case_subject_ready_for_processing ?academic_case)
        (approver_available ?approver)
      )
    :effect
      (and
        (case_subject_assigned_approver ?academic_case ?approver)
        (not
          (approver_available ?approver)
        )
      )
  )
  (:action unassign_approver_from_case
    :parameters (?academic_case - academic_case ?approver - approver)
    :precondition
      (and
        (case_subject_assigned_approver ?academic_case ?approver)
      )
    :effect
      (and
        (approver_available ?approver)
        (not
          (case_subject_assigned_approver ?academic_case ?approver)
        )
      )
  )
  (:action attach_instructor_availability_to_file
    :parameters (?administrative_case_file - administrative_case_file ?instructor_availability - instructor_availability)
    :precondition
      (and
        (case_subject_ready_for_processing ?administrative_case_file)
        (instructor_availability_available ?instructor_availability)
      )
    :effect
      (and
        (file_linked_instructor_availability ?administrative_case_file ?instructor_availability)
        (not
          (instructor_availability_available ?instructor_availability)
        )
      )
  )
  (:action detach_instructor_availability_from_file
    :parameters (?administrative_case_file - administrative_case_file ?instructor_availability - instructor_availability)
    :precondition
      (and
        (file_linked_instructor_availability ?administrative_case_file ?instructor_availability)
      )
    :effect
      (and
        (instructor_availability_available ?instructor_availability)
        (not
          (file_linked_instructor_availability ?administrative_case_file ?instructor_availability)
        )
      )
  )
  (:action attach_department_signoff_to_file
    :parameters (?administrative_case_file - administrative_case_file ?department_signoff - department_signoff)
    :precondition
      (and
        (case_subject_ready_for_processing ?administrative_case_file)
        (department_signoff_available ?department_signoff)
      )
    :effect
      (and
        (file_linked_department_signoff ?administrative_case_file ?department_signoff)
        (not
          (department_signoff_available ?department_signoff)
        )
      )
  )
  (:action detach_department_signoff_from_file
    :parameters (?administrative_case_file - administrative_case_file ?department_signoff - department_signoff)
    :precondition
      (and
        (file_linked_department_signoff ?administrative_case_file ?department_signoff)
      )
    :effect
      (and
        (department_signoff_available ?department_signoff)
        (not
          (file_linked_department_signoff ?administrative_case_file ?department_signoff)
        )
      )
  )
  (:action validate_primary_schedule_constraint_for_enrollment
    :parameters (?affected_enrollment - affected_enrollment ?primary_schedule_constraint - primary_schedule_constraint ?course_offering - course_offering)
    :precondition
      (and
        (case_subject_ready_for_processing ?affected_enrollment)
        (case_subject_assigned_offering ?affected_enrollment ?course_offering)
        (enrollment_primary_schedule_constraint ?affected_enrollment ?primary_schedule_constraint)
        (not
          (primary_schedule_constraint_validated ?primary_schedule_constraint)
        )
        (not
          (primary_schedule_constraint_document_validated ?primary_schedule_constraint)
        )
      )
    :effect (primary_schedule_constraint_validated ?primary_schedule_constraint)
  )
  (:action confirm_enrollment_with_approver_and_constraint
    :parameters (?affected_enrollment - affected_enrollment ?primary_schedule_constraint - primary_schedule_constraint ?approver - approver)
    :precondition
      (and
        (case_subject_ready_for_processing ?affected_enrollment)
        (case_subject_assigned_approver ?affected_enrollment ?approver)
        (enrollment_primary_schedule_constraint ?affected_enrollment ?primary_schedule_constraint)
        (primary_schedule_constraint_validated ?primary_schedule_constraint)
        (not
          (enrollment_ready ?affected_enrollment)
        )
      )
    :effect
      (and
        (enrollment_ready ?affected_enrollment)
        (enrollment_constraint_confirmed ?affected_enrollment)
      )
  )
  (:action reserve_primary_schedule_constraint_with_document
    :parameters (?affected_enrollment - affected_enrollment ?primary_schedule_constraint - primary_schedule_constraint ?supporting_document - supporting_document)
    :precondition
      (and
        (case_subject_ready_for_processing ?affected_enrollment)
        (enrollment_primary_schedule_constraint ?affected_enrollment ?primary_schedule_constraint)
        (supporting_document_available ?supporting_document)
        (not
          (enrollment_ready ?affected_enrollment)
        )
      )
    :effect
      (and
        (primary_schedule_constraint_document_validated ?primary_schedule_constraint)
        (enrollment_ready ?affected_enrollment)
        (enrollment_document_attached ?affected_enrollment ?supporting_document)
        (not
          (supporting_document_available ?supporting_document)
        )
      )
  )
  (:action reconcile_enrollment_constraint_with_document
    :parameters (?affected_enrollment - affected_enrollment ?primary_schedule_constraint - primary_schedule_constraint ?course_offering - course_offering ?supporting_document - supporting_document)
    :precondition
      (and
        (case_subject_ready_for_processing ?affected_enrollment)
        (case_subject_assigned_offering ?affected_enrollment ?course_offering)
        (enrollment_primary_schedule_constraint ?affected_enrollment ?primary_schedule_constraint)
        (primary_schedule_constraint_document_validated ?primary_schedule_constraint)
        (enrollment_document_attached ?affected_enrollment ?supporting_document)
        (not
          (enrollment_constraint_confirmed ?affected_enrollment)
        )
      )
    :effect
      (and
        (primary_schedule_constraint_validated ?primary_schedule_constraint)
        (enrollment_constraint_confirmed ?affected_enrollment)
        (supporting_document_available ?supporting_document)
        (not
          (enrollment_document_attached ?affected_enrollment ?supporting_document)
        )
      )
  )
  (:action validate_secondary_schedule_constraint_for_requirement
    :parameters (?affected_requirement - affected_requirement ?secondary_schedule_constraint - secondary_schedule_constraint ?course_offering - course_offering)
    :precondition
      (and
        (case_subject_ready_for_processing ?affected_requirement)
        (case_subject_assigned_offering ?affected_requirement ?course_offering)
        (requirement_secondary_schedule_constraint ?affected_requirement ?secondary_schedule_constraint)
        (not
          (secondary_schedule_constraint_validated ?secondary_schedule_constraint)
        )
        (not
          (secondary_schedule_constraint_document_validated ?secondary_schedule_constraint)
        )
      )
    :effect (secondary_schedule_constraint_validated ?secondary_schedule_constraint)
  )
  (:action confirm_requirement_with_approver_and_constraint
    :parameters (?affected_requirement - affected_requirement ?secondary_schedule_constraint - secondary_schedule_constraint ?approver - approver)
    :precondition
      (and
        (case_subject_ready_for_processing ?affected_requirement)
        (case_subject_assigned_approver ?affected_requirement ?approver)
        (requirement_secondary_schedule_constraint ?affected_requirement ?secondary_schedule_constraint)
        (secondary_schedule_constraint_validated ?secondary_schedule_constraint)
        (not
          (requirement_ready ?affected_requirement)
        )
      )
    :effect
      (and
        (requirement_ready ?affected_requirement)
        (requirement_constraint_confirmed ?affected_requirement)
      )
  )
  (:action reserve_secondary_schedule_constraint_with_document
    :parameters (?affected_requirement - affected_requirement ?secondary_schedule_constraint - secondary_schedule_constraint ?supporting_document - supporting_document)
    :precondition
      (and
        (case_subject_ready_for_processing ?affected_requirement)
        (requirement_secondary_schedule_constraint ?affected_requirement ?secondary_schedule_constraint)
        (supporting_document_available ?supporting_document)
        (not
          (requirement_ready ?affected_requirement)
        )
      )
    :effect
      (and
        (secondary_schedule_constraint_document_validated ?secondary_schedule_constraint)
        (requirement_ready ?affected_requirement)
        (requirement_document_attached ?affected_requirement ?supporting_document)
        (not
          (supporting_document_available ?supporting_document)
        )
      )
  )
  (:action reconcile_requirement_constraint_with_document
    :parameters (?affected_requirement - affected_requirement ?secondary_schedule_constraint - secondary_schedule_constraint ?course_offering - course_offering ?supporting_document - supporting_document)
    :precondition
      (and
        (case_subject_ready_for_processing ?affected_requirement)
        (case_subject_assigned_offering ?affected_requirement ?course_offering)
        (requirement_secondary_schedule_constraint ?affected_requirement ?secondary_schedule_constraint)
        (secondary_schedule_constraint_document_validated ?secondary_schedule_constraint)
        (requirement_document_attached ?affected_requirement ?supporting_document)
        (not
          (requirement_constraint_confirmed ?affected_requirement)
        )
      )
    :effect
      (and
        (secondary_schedule_constraint_validated ?secondary_schedule_constraint)
        (requirement_constraint_confirmed ?affected_requirement)
        (supporting_document_available ?supporting_document)
        (not
          (requirement_document_attached ?affected_requirement ?supporting_document)
        )
      )
  )
  (:action assemble_replacement_proposal
    :parameters (?affected_enrollment - affected_enrollment ?affected_requirement - affected_requirement ?primary_schedule_constraint - primary_schedule_constraint ?secondary_schedule_constraint - secondary_schedule_constraint ?replacement_proposal - replacement_proposal)
    :precondition
      (and
        (enrollment_ready ?affected_enrollment)
        (requirement_ready ?affected_requirement)
        (enrollment_primary_schedule_constraint ?affected_enrollment ?primary_schedule_constraint)
        (requirement_secondary_schedule_constraint ?affected_requirement ?secondary_schedule_constraint)
        (primary_schedule_constraint_validated ?primary_schedule_constraint)
        (secondary_schedule_constraint_validated ?secondary_schedule_constraint)
        (enrollment_constraint_confirmed ?affected_enrollment)
        (requirement_constraint_confirmed ?affected_requirement)
        (proposal_draft ?replacement_proposal)
      )
    :effect
      (and
        (proposal_constructed ?replacement_proposal)
        (proposal_primary_schedule_constraint_link ?replacement_proposal ?primary_schedule_constraint)
        (proposal_secondary_schedule_constraint_link ?replacement_proposal ?secondary_schedule_constraint)
        (not
          (proposal_draft ?replacement_proposal)
        )
      )
  )
  (:action assemble_replacement_proposal_primary_document_path
    :parameters (?affected_enrollment - affected_enrollment ?affected_requirement - affected_requirement ?primary_schedule_constraint - primary_schedule_constraint ?secondary_schedule_constraint - secondary_schedule_constraint ?replacement_proposal - replacement_proposal)
    :precondition
      (and
        (enrollment_ready ?affected_enrollment)
        (requirement_ready ?affected_requirement)
        (enrollment_primary_schedule_constraint ?affected_enrollment ?primary_schedule_constraint)
        (requirement_secondary_schedule_constraint ?affected_requirement ?secondary_schedule_constraint)
        (primary_schedule_constraint_document_validated ?primary_schedule_constraint)
        (secondary_schedule_constraint_validated ?secondary_schedule_constraint)
        (not
          (enrollment_constraint_confirmed ?affected_enrollment)
        )
        (requirement_constraint_confirmed ?affected_requirement)
        (proposal_draft ?replacement_proposal)
      )
    :effect
      (and
        (proposal_constructed ?replacement_proposal)
        (proposal_primary_schedule_constraint_link ?replacement_proposal ?primary_schedule_constraint)
        (proposal_secondary_schedule_constraint_link ?replacement_proposal ?secondary_schedule_constraint)
        (proposal_primary_document_verified ?replacement_proposal)
        (not
          (proposal_draft ?replacement_proposal)
        )
      )
  )
  (:action assemble_replacement_proposal_secondary_document_path
    :parameters (?affected_enrollment - affected_enrollment ?affected_requirement - affected_requirement ?primary_schedule_constraint - primary_schedule_constraint ?secondary_schedule_constraint - secondary_schedule_constraint ?replacement_proposal - replacement_proposal)
    :precondition
      (and
        (enrollment_ready ?affected_enrollment)
        (requirement_ready ?affected_requirement)
        (enrollment_primary_schedule_constraint ?affected_enrollment ?primary_schedule_constraint)
        (requirement_secondary_schedule_constraint ?affected_requirement ?secondary_schedule_constraint)
        (primary_schedule_constraint_validated ?primary_schedule_constraint)
        (secondary_schedule_constraint_document_validated ?secondary_schedule_constraint)
        (enrollment_constraint_confirmed ?affected_enrollment)
        (not
          (requirement_constraint_confirmed ?affected_requirement)
        )
        (proposal_draft ?replacement_proposal)
      )
    :effect
      (and
        (proposal_constructed ?replacement_proposal)
        (proposal_primary_schedule_constraint_link ?replacement_proposal ?primary_schedule_constraint)
        (proposal_secondary_schedule_constraint_link ?replacement_proposal ?secondary_schedule_constraint)
        (proposal_secondary_document_verified ?replacement_proposal)
        (not
          (proposal_draft ?replacement_proposal)
        )
      )
  )
  (:action assemble_replacement_proposal_both_document_paths
    :parameters (?affected_enrollment - affected_enrollment ?affected_requirement - affected_requirement ?primary_schedule_constraint - primary_schedule_constraint ?secondary_schedule_constraint - secondary_schedule_constraint ?replacement_proposal - replacement_proposal)
    :precondition
      (and
        (enrollment_ready ?affected_enrollment)
        (requirement_ready ?affected_requirement)
        (enrollment_primary_schedule_constraint ?affected_enrollment ?primary_schedule_constraint)
        (requirement_secondary_schedule_constraint ?affected_requirement ?secondary_schedule_constraint)
        (primary_schedule_constraint_document_validated ?primary_schedule_constraint)
        (secondary_schedule_constraint_document_validated ?secondary_schedule_constraint)
        (not
          (enrollment_constraint_confirmed ?affected_enrollment)
        )
        (not
          (requirement_constraint_confirmed ?affected_requirement)
        )
        (proposal_draft ?replacement_proposal)
      )
    :effect
      (and
        (proposal_constructed ?replacement_proposal)
        (proposal_primary_schedule_constraint_link ?replacement_proposal ?primary_schedule_constraint)
        (proposal_secondary_schedule_constraint_link ?replacement_proposal ?secondary_schedule_constraint)
        (proposal_primary_document_verified ?replacement_proposal)
        (proposal_secondary_document_verified ?replacement_proposal)
        (not
          (proposal_draft ?replacement_proposal)
        )
      )
  )
  (:action lock_proposal_section
    :parameters (?replacement_proposal - replacement_proposal ?affected_enrollment - affected_enrollment ?course_offering - course_offering)
    :precondition
      (and
        (proposal_constructed ?replacement_proposal)
        (enrollment_ready ?affected_enrollment)
        (case_subject_assigned_offering ?affected_enrollment ?course_offering)
        (not
          (proposal_section_locked ?replacement_proposal)
        )
      )
    :effect (proposal_section_locked ?replacement_proposal)
  )
  (:action reserve_course_section_for_file_proposal
    :parameters (?administrative_case_file - administrative_case_file ?course_section - course_section ?replacement_proposal - replacement_proposal)
    :precondition
      (and
        (case_subject_ready_for_processing ?administrative_case_file)
        (file_linked_proposal ?administrative_case_file ?replacement_proposal)
        (file_has_section ?administrative_case_file ?course_section)
        (section_available ?course_section)
        (proposal_constructed ?replacement_proposal)
        (proposal_section_locked ?replacement_proposal)
        (not
          (section_reserved ?course_section)
        )
      )
    :effect
      (and
        (section_reserved ?course_section)
        (section_assigned_to_proposal ?course_section ?replacement_proposal)
        (not
          (section_available ?course_section)
        )
      )
  )
  (:action allocate_section_to_file_proposal
    :parameters (?administrative_case_file - administrative_case_file ?course_section - course_section ?replacement_proposal - replacement_proposal ?course_offering - course_offering)
    :precondition
      (and
        (case_subject_ready_for_processing ?administrative_case_file)
        (file_has_section ?administrative_case_file ?course_section)
        (section_reserved ?course_section)
        (section_assigned_to_proposal ?course_section ?replacement_proposal)
        (case_subject_assigned_offering ?administrative_case_file ?course_offering)
        (not
          (proposal_primary_document_verified ?replacement_proposal)
        )
        (not
          (file_section_allocated ?administrative_case_file)
        )
      )
    :effect (file_section_allocated ?administrative_case_file)
  )
  (:action attach_policy_exception_template_to_file
    :parameters (?administrative_case_file - administrative_case_file ?policy_exception_template - policy_exception_template)
    :precondition
      (and
        (case_subject_ready_for_processing ?administrative_case_file)
        (policy_exception_template_available ?policy_exception_template)
        (not
          (file_has_policy_exception_template ?administrative_case_file)
        )
      )
