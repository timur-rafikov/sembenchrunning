    (case_eligibility_confirmed ?study_plan_case - study_plan_case)

    (proposal_ready_for_packaging ?resequencing_proposal - resequencing_proposal)
    (proposal_packaged ?resequencing_proposal - resequencing_proposal)
    (proposal_student_term_link ?resequencing_proposal - resequencing_proposal ?student_term_slot - student_term_slot)

    (proposal_includes_cohort_term_confirmation ?resequencing_proposal - resequencing_proposal)

    (admin_case_links_cohort ?administrative_unit_case - admin_case ?cohort_enrollment_record - cohort_enrollment_record)

  (:action create_study_plan_case
    :parameters (?study_plan_case - study_plan_case)
    :precondition
      (and
        (not
          (case_intake_registered ?study_plan_case)
        )
        (not
          (ready_for_resequencing ?study_plan_case)
        )
      )
    :effect (case_intake_registered ?study_plan_case)
  )
  (:action assign_officer_to_case
    :parameters (?study_plan_case - study_plan_case ?administrative_officer - administrative_officer)
    :precondition
      (and
        (case_intake_registered ?study_plan_case)
        (not
          (case_assigned ?study_plan_case)
        )
        (officer_available ?administrative_officer)
      )
    :effect
      (and
        (case_assigned ?study_plan_case)
        (case_assigned_to_officer ?study_plan_case ?administrative_officer)
        (not
          (officer_available ?administrative_officer)
        )
      )
  )
  (:action attach_course_component_to_case
    :parameters (?study_plan_case - study_plan_case ?course_component - course_component)
    :precondition
      (and
        (case_intake_registered ?study_plan_case)
        (case_assigned ?study_plan_case)
        (course_component_available ?course_component)
      )
    :effect
      (and
        (case_assigned_component ?study_plan_case ?course_component)
        (not
          (course_component_available ?course_component)
        )
      )
  )
  (:action confirm_case_eligibility
    :parameters (?study_plan_case - study_plan_case ?course_component - course_component)
    :precondition
      (and
        (case_intake_registered ?study_plan_case)
        (case_assigned ?study_plan_case)
        (case_assigned_component ?study_plan_case ?course_component)
        (not
          (case_eligibility_confirmed ?study_plan_case)
        )
      )
    :effect (case_eligibility_confirmed ?study_plan_case)
  )
  (:action unassign_course_component_from_case
    :parameters (?study_plan_case - study_plan_case ?course_component - course_component)
    :precondition
      (and
        (case_assigned_component ?study_plan_case ?course_component)
      )
    :effect
      (and
        (course_component_available ?course_component)
        (not
          (case_assigned_component ?study_plan_case ?course_component)
        )
      )
  )
  (:action assign_approver_to_case
    :parameters (?study_plan_case - study_plan_case ?approver_role - approver_role)
    :precondition
      (and
        (case_eligibility_confirmed ?study_plan_case)
        (approver_available ?approver_role)
      )
    :effect
      (and
        (case_approver_assigned ?study_plan_case ?approver_role)
        (not
          (approver_available ?approver_role)
        )
      )
  )
  (:action unassign_approver_from_case
    :parameters (?study_plan_case - study_plan_case ?approver_role - approver_role)
    :precondition
      (and
        (case_approver_assigned ?study_plan_case ?approver_role)
      )
    :effect
      (and
        (approver_available ?approver_role)
        (not
          (case_approver_assigned ?study_plan_case ?approver_role)
        )
      )
  )
  (:action attach_tuition_adjustment
    :parameters (?administrative_unit_case - admin_case ?tuition_adjustment_record - tuition_adjustment_record)

      (and
        (case_eligibility_confirmed ?administrative_unit_case)
        (tuition_adjustment_available ?tuition_adjustment_record)

          (tuition_adjustment_available ?tuition_adjustment_record)
        )

        (tuition_adjustment_available ?tuition_adjustment_record)
        (not
          (admin_case_has_tuition_adjustment ?administrative_unit_case ?tuition_adjustment_record)
        )
      )

  (:action attach_regulatory_notice
    :parameters (?administrative_unit_case - admin_case ?regulatory_notice - regulatory_notice)

        )

    :parameters (?administrative_unit_case - admin_case ?regulatory_notice - regulatory_notice)

        )

  (:action confirm_student_term_slot
    :parameters (?student_enrollment_record - student_enrollment_record ?student_term_slot - student_term_slot ?course_component - course_component)
    :precondition
      (and
        (case_eligibility_confirmed ?student_enrollment_record)
        (case_assigned_component ?student_enrollment_record ?course_component)
        (enrollment_term_slot_assigned ?student_enrollment_record ?student_term_slot)
        (not
          (student_term_slot_confirmed ?student_term_slot)
        )
        (not
          (student_term_slot_provisionally_reserved ?student_term_slot)
        )
      )
    :effect (student_term_slot_confirmed ?student_term_slot)
  )
  (:action reserve_enrollment_for_approval
    :parameters (?student_enrollment_record - student_enrollment_record ?student_term_slot - student_term_slot ?approver_role - approver_role)
    :precondition
      (and
        (case_eligibility_confirmed ?student_enrollment_record)
        (case_approver_assigned ?student_enrollment_record ?approver_role)
        (enrollment_term_slot_assigned ?student_enrollment_record ?student_term_slot)
        (student_term_slot_confirmed ?student_term_slot)
        (not
          (enrollment_assessed ?student_enrollment_record)
        )
      )
    :effect
      (and
        (enrollment_assessed ?student_enrollment_record)
        (enrollment_assessment_confirmed ?student_enrollment_record)
      )
  )
  (:action attach_support_document_and_reserve_term
    :parameters (?student_enrollment_record - student_enrollment_record ?student_term_slot - student_term_slot ?support_document - support_document)

        (case_eligibility_confirmed ?student_enrollment_record)

        (support_document_available ?support_document)
        (not
          (enrollment_assessed ?student_enrollment_record)
        )
      )
    :effect
      (and
        (student_term_slot_provisionally_reserved ?student_term_slot)
        (enrollment_assessed ?student_enrollment_record)
        (enrollment_support_document_linked ?student_enrollment_record ?support_document)
        (not
          (support_document_available ?support_document)
        )
      )
  )
  (:action finalize_student_term_reservation

        (student_term_slot_confirmed ?student_term_slot)

          (enrollment_support_document_linked ?student_enrollment_record ?support_document)

    :effect
      (and

    :precondition
      (and
