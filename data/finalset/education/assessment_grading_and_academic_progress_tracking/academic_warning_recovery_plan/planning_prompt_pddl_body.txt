## Planning domain (PDDL)

(define (domain academic_warning_recovery_plan_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types domain_entity - object resource_type_group - object grade_category - object case_root_type - object learner_case - case_root_type staff_resource - domain_entity assessment_slot - domain_entity advisor - domain_entity policy_clause - domain_entity authorization_credential - domain_entity time_slot - domain_entity remediation_activity - domain_entity external_clearance - domain_entity intervention_token - resource_type_group evidence_artifact - resource_type_group endorsement - resource_type_group assessment_component - grade_category exam_session - grade_category submission_package - grade_category program_case_subtype - learner_case cohort_case_subtype - learner_case student_enrollment - program_case_subtype section_enrollment - program_case_subtype recovery_plan - cohort_case_subtype)
  (:predicates
    (case_opened ?learner_case - learner_case)
    (case_eligibility_confirmed ?learner_case - learner_case)
    (case_has_assigned_staff ?learner_case - learner_case)
    (recovery_finalized ?learner_case - learner_case)
    (verification_completed ?learner_case - learner_case)
    (standing_update_authorized ?learner_case - learner_case)
    (staff_resource_available ?staff_resource - staff_resource)
    (assigned_staff_for_case ?learner_case - learner_case ?staff_resource - staff_resource)
    (assessment_slot_available ?assessment_instance - assessment_slot)
    (assessment_slot_reserved_for_case ?learner_case - learner_case ?assessment_instance - assessment_slot)
    (advisor_available ?advisor - advisor)
    (advisor_assigned_to_case ?learner_case - learner_case ?advisor - advisor)
    (intervention_token_available ?intervention_token - intervention_token)
    (enrollment_consumed_token ?student_enrollment - student_enrollment ?intervention_token - intervention_token)
    (section_consumed_token ?section_enrollment - section_enrollment ?intervention_token - intervention_token)
    (component_assigned_to_enrollment ?student_enrollment - student_enrollment ?assessment_component - assessment_component)
    (component_ready_flag ?assessment_component - assessment_component)
    (component_ready_via_intervention ?assessment_component - assessment_component)
    (enrollment_component_confirmed ?student_enrollment - student_enrollment)
    (section_has_exam_session ?section_enrollment - section_enrollment ?exam_session - exam_session)
    (exam_session_ready_flag ?exam_session - exam_session)
    (exam_session_ready_via_intervention ?exam_session - exam_session)
    (section_component_confirmed ?section_enrollment - section_enrollment)
    (submission_package_available ?submission_package - submission_package)
    (submission_package_composed ?submission_package - submission_package)
    (package_includes_component ?submission_package - submission_package ?assessment_component - assessment_component)
    (package_includes_exam_session ?submission_package - submission_package ?exam_session - exam_session)
    (package_requires_enrollment_confirmation ?submission_package - submission_package)
    (package_requires_section_confirmation ?submission_package - submission_package)
    (submission_package_executed ?submission_package - submission_package)
    (plan_linked_to_enrollment ?recovery_plan - recovery_plan ?student_enrollment - student_enrollment)
    (plan_linked_to_section ?recovery_plan - recovery_plan ?section_enrollment - section_enrollment)
    (plan_linked_to_submission_package ?recovery_plan - recovery_plan ?submission_package - submission_package)
    (evidence_artifact_available ?evidence_artifact - evidence_artifact)
    (plan_has_evidence_artifact ?recovery_plan - recovery_plan ?evidence_artifact - evidence_artifact)
    (artifact_verified ?evidence_artifact - evidence_artifact)
    (artifact_linked_to_package ?evidence_artifact - evidence_artifact ?submission_package - submission_package)
    (plan_initial_artifact_reviewed ?recovery_plan - recovery_plan)
    (plan_remediation_required ?recovery_plan - recovery_plan)
    (plan_artifact_review_passed ?recovery_plan - recovery_plan)
    (policy_attached_to_plan ?recovery_plan - recovery_plan)
    (policy_review_flag ?recovery_plan - recovery_plan)
    (plan_authorization_present ?recovery_plan - recovery_plan)
    (plan_ready_for_finalization ?recovery_plan - recovery_plan)
    (endorsement_available ?endorsement - endorsement)
    (plan_has_endorsement ?recovery_plan - recovery_plan ?endorsement - endorsement)
    (endorsement_attached_to_plan ?recovery_plan - recovery_plan)
    (advisor_signoff_recorded ?recovery_plan - recovery_plan)
    (external_clearance_recorded ?recovery_plan - recovery_plan)
    (policy_clause_available ?policy_clause - policy_clause)
    (policy_clause_attached_to_plan ?recovery_plan - recovery_plan ?policy_clause - policy_clause)
    (authorization_credential_available ?authorization_credential - authorization_credential)
    (plan_has_authorization_credential ?recovery_plan - recovery_plan ?authorization_credential - authorization_credential)
    (remediation_activity_available ?remediation_activity - remediation_activity)
    (plan_assigned_remediation_activity ?recovery_plan - recovery_plan ?remediation_activity - remediation_activity)
    (external_clearance_available ?external_clearance - external_clearance)
    (plan_assigned_external_clearance ?recovery_plan - recovery_plan ?external_clearance - external_clearance)
    (time_slot_available ?time_slot - time_slot)
    (case_scheduled_time_slot ?learner_case - learner_case ?time_slot - time_slot)
    (enrollment_milestone_flag ?student_enrollment - student_enrollment)
    (section_milestone_flag ?section_enrollment - section_enrollment)
    (plan_finalization_token ?recovery_plan - recovery_plan)
  )
  (:action open_learner_recovery_case
    :parameters (?learner_case - learner_case)
    :precondition
      (and
        (not
          (case_opened ?learner_case)
        )
        (not
          (recovery_finalized ?learner_case)
        )
      )
    :effect (case_opened ?learner_case)
  )
  (:action assign_staff_to_recovery_case
    :parameters (?learner_case - learner_case ?staff_resource - staff_resource)
    :precondition
      (and
        (case_opened ?learner_case)
        (not
          (case_has_assigned_staff ?learner_case)
        )
        (staff_resource_available ?staff_resource)
      )
    :effect
      (and
        (case_has_assigned_staff ?learner_case)
        (assigned_staff_for_case ?learner_case ?staff_resource)
        (not
          (staff_resource_available ?staff_resource)
        )
      )
  )
  (:action reserve_assessment_slot_for_case
    :parameters (?learner_case - learner_case ?assessment_instance - assessment_slot)
    :precondition
      (and
        (case_opened ?learner_case)
        (case_has_assigned_staff ?learner_case)
        (assessment_slot_available ?assessment_instance)
      )
    :effect
      (and
        (assessment_slot_reserved_for_case ?learner_case ?assessment_instance)
        (not
          (assessment_slot_available ?assessment_instance)
        )
      )
  )
  (:action confirm_case_eligibility
    :parameters (?learner_case - learner_case ?assessment_instance - assessment_slot)
    :precondition
      (and
        (case_opened ?learner_case)
        (case_has_assigned_staff ?learner_case)
        (assessment_slot_reserved_for_case ?learner_case ?assessment_instance)
        (not
          (case_eligibility_confirmed ?learner_case)
        )
      )
    :effect (case_eligibility_confirmed ?learner_case)
  )
  (:action release_assessment_slot_reservation_from_case
    :parameters (?learner_case - learner_case ?assessment_instance - assessment_slot)
    :precondition
      (and
        (assessment_slot_reserved_for_case ?learner_case ?assessment_instance)
      )
    :effect
      (and
        (assessment_slot_available ?assessment_instance)
        (not
          (assessment_slot_reserved_for_case ?learner_case ?assessment_instance)
        )
      )
  )
  (:action assign_advisor_to_case
    :parameters (?learner_case - learner_case ?advisor - advisor)
    :precondition
      (and
        (case_eligibility_confirmed ?learner_case)
        (advisor_available ?advisor)
      )
    :effect
      (and
        (advisor_assigned_to_case ?learner_case ?advisor)
        (not
          (advisor_available ?advisor)
        )
      )
  )
  (:action unassign_advisor_from_case
    :parameters (?learner_case - learner_case ?advisor - advisor)
    :precondition
      (and
        (advisor_assigned_to_case ?learner_case ?advisor)
      )
    :effect
      (and
        (advisor_available ?advisor)
        (not
          (advisor_assigned_to_case ?learner_case ?advisor)
        )
      )
  )
  (:action attach_remediation_activity_to_plan
    :parameters (?recovery_plan - recovery_plan ?remediation_activity - remediation_activity)
    :precondition
      (and
        (case_eligibility_confirmed ?recovery_plan)
        (remediation_activity_available ?remediation_activity)
      )
    :effect
      (and
        (plan_assigned_remediation_activity ?recovery_plan ?remediation_activity)
        (not
          (remediation_activity_available ?remediation_activity)
        )
      )
  )
  (:action detach_remediation_activity_from_plan
    :parameters (?recovery_plan - recovery_plan ?remediation_activity - remediation_activity)
    :precondition
      (and
        (plan_assigned_remediation_activity ?recovery_plan ?remediation_activity)
      )
    :effect
      (and
        (remediation_activity_available ?remediation_activity)
        (not
          (plan_assigned_remediation_activity ?recovery_plan ?remediation_activity)
        )
      )
  )
  (:action attach_external_clearance_to_plan
    :parameters (?recovery_plan - recovery_plan ?external_clearance - external_clearance)
    :precondition
      (and
        (case_eligibility_confirmed ?recovery_plan)
        (external_clearance_available ?external_clearance)
      )
    :effect
      (and
        (plan_assigned_external_clearance ?recovery_plan ?external_clearance)
        (not
          (external_clearance_available ?external_clearance)
        )
      )
  )
  (:action detach_external_clearance_from_plan
    :parameters (?recovery_plan - recovery_plan ?external_clearance - external_clearance)
    :precondition
      (and
        (plan_assigned_external_clearance ?recovery_plan ?external_clearance)
      )
    :effect
      (and
        (external_clearance_available ?external_clearance)
        (not
          (plan_assigned_external_clearance ?recovery_plan ?external_clearance)
        )
      )
  )
  (:action flag_enrollment_component_ready
    :parameters (?student_enrollment - student_enrollment ?assessment_component - assessment_component ?assessment_instance - assessment_slot)
    :precondition
      (and
        (case_eligibility_confirmed ?student_enrollment)
        (assessment_slot_reserved_for_case ?student_enrollment ?assessment_instance)
        (component_assigned_to_enrollment ?student_enrollment ?assessment_component)
        (not
          (component_ready_flag ?assessment_component)
        )
        (not
          (component_ready_via_intervention ?assessment_component)
        )
      )
    :effect (component_ready_flag ?assessment_component)
  )
  (:action confirm_enrollment_component_with_advisor
    :parameters (?student_enrollment - student_enrollment ?assessment_component - assessment_component ?advisor - advisor)
    :precondition
      (and
        (case_eligibility_confirmed ?student_enrollment)
        (advisor_assigned_to_case ?student_enrollment ?advisor)
        (component_assigned_to_enrollment ?student_enrollment ?assessment_component)
        (component_ready_flag ?assessment_component)
        (not
          (enrollment_milestone_flag ?student_enrollment)
        )
      )
    :effect
      (and
        (enrollment_milestone_flag ?student_enrollment)
        (enrollment_component_confirmed ?student_enrollment)
      )
  )
  (:action consume_intervention_token_for_enrollment_component
    :parameters (?student_enrollment - student_enrollment ?assessment_component - assessment_component ?intervention_token - intervention_token)
    :precondition
      (and
        (case_eligibility_confirmed ?student_enrollment)
        (component_assigned_to_enrollment ?student_enrollment ?assessment_component)
        (intervention_token_available ?intervention_token)
        (not
          (enrollment_milestone_flag ?student_enrollment)
        )
      )
    :effect
      (and
        (component_ready_via_intervention ?assessment_component)
        (enrollment_milestone_flag ?student_enrollment)
        (enrollment_consumed_token ?student_enrollment ?intervention_token)
        (not
          (intervention_token_available ?intervention_token)
        )
      )
  )
  (:action finalize_enrollment_component_activation
    :parameters (?student_enrollment - student_enrollment ?assessment_component - assessment_component ?assessment_instance - assessment_slot ?intervention_token - intervention_token)
    :precondition
      (and
        (case_eligibility_confirmed ?student_enrollment)
        (assessment_slot_reserved_for_case ?student_enrollment ?assessment_instance)
        (component_assigned_to_enrollment ?student_enrollment ?assessment_component)
        (component_ready_via_intervention ?assessment_component)
        (enrollment_consumed_token ?student_enrollment ?intervention_token)
        (not
          (enrollment_component_confirmed ?student_enrollment)
        )
      )
    :effect
      (and
        (component_ready_flag ?assessment_component)
        (enrollment_component_confirmed ?student_enrollment)
        (intervention_token_available ?intervention_token)
        (not
          (enrollment_consumed_token ?student_enrollment ?intervention_token)
        )
      )
  )
  (:action flag_section_component_ready
    :parameters (?section_enrollment - section_enrollment ?exam_session - exam_session ?assessment_instance - assessment_slot)
    :precondition
      (and
        (case_eligibility_confirmed ?section_enrollment)
        (assessment_slot_reserved_for_case ?section_enrollment ?assessment_instance)
        (section_has_exam_session ?section_enrollment ?exam_session)
        (not
          (exam_session_ready_flag ?exam_session)
        )
        (not
          (exam_session_ready_via_intervention ?exam_session)
        )
      )
    :effect (exam_session_ready_flag ?exam_session)
  )
  (:action confirm_section_component_with_advisor
    :parameters (?section_enrollment - section_enrollment ?exam_session - exam_session ?advisor - advisor)
    :precondition
      (and
        (case_eligibility_confirmed ?section_enrollment)
        (advisor_assigned_to_case ?section_enrollment ?advisor)
        (section_has_exam_session ?section_enrollment ?exam_session)
        (exam_session_ready_flag ?exam_session)
        (not
          (section_milestone_flag ?section_enrollment)
        )
      )
    :effect
      (and
        (section_milestone_flag ?section_enrollment)
        (section_component_confirmed ?section_enrollment)
      )
  )
  (:action consume_intervention_token_for_section_component
    :parameters (?section_enrollment - section_enrollment ?exam_session - exam_session ?intervention_token - intervention_token)
    :precondition
      (and
        (case_eligibility_confirmed ?section_enrollment)
        (section_has_exam_session ?section_enrollment ?exam_session)
        (intervention_token_available ?intervention_token)
        (not
          (section_milestone_flag ?section_enrollment)
        )
      )
    :effect
      (and
        (exam_session_ready_via_intervention ?exam_session)
        (section_milestone_flag ?section_enrollment)
        (section_consumed_token ?section_enrollment ?intervention_token)
        (not
          (intervention_token_available ?intervention_token)
        )
      )
  )
  (:action finalize_section_component_activation
    :parameters (?section_enrollment - section_enrollment ?exam_session - exam_session ?assessment_instance - assessment_slot ?intervention_token - intervention_token)
    :precondition
      (and
        (case_eligibility_confirmed ?section_enrollment)
        (assessment_slot_reserved_for_case ?section_enrollment ?assessment_instance)
        (section_has_exam_session ?section_enrollment ?exam_session)
        (exam_session_ready_via_intervention ?exam_session)
        (section_consumed_token ?section_enrollment ?intervention_token)
        (not
          (section_component_confirmed ?section_enrollment)
        )
      )
    :effect
      (and
        (exam_session_ready_flag ?exam_session)
        (section_component_confirmed ?section_enrollment)
        (intervention_token_available ?intervention_token)
        (not
          (section_consumed_token ?section_enrollment ?intervention_token)
        )
      )
  )
  (:action compose_submission_package
    :parameters (?student_enrollment - student_enrollment ?section_enrollment - section_enrollment ?assessment_component - assessment_component ?exam_session - exam_session ?submission_package - submission_package)
    :precondition
      (and
        (enrollment_milestone_flag ?student_enrollment)
        (section_milestone_flag ?section_enrollment)
        (component_assigned_to_enrollment ?student_enrollment ?assessment_component)
        (section_has_exam_session ?section_enrollment ?exam_session)
        (component_ready_flag ?assessment_component)
        (exam_session_ready_flag ?exam_session)
        (enrollment_component_confirmed ?student_enrollment)
        (section_component_confirmed ?section_enrollment)
        (submission_package_available ?submission_package)
      )
    :effect
      (and
        (submission_package_composed ?submission_package)
        (package_includes_component ?submission_package ?assessment_component)
        (package_includes_exam_session ?submission_package ?exam_session)
        (not
          (submission_package_available ?submission_package)
        )
      )
  )
  (:action compose_submission_package_requires_enrollment_confirmation
    :parameters (?student_enrollment - student_enrollment ?section_enrollment - section_enrollment ?assessment_component - assessment_component ?exam_session - exam_session ?submission_package - submission_package)
    :precondition
      (and
        (enrollment_milestone_flag ?student_enrollment)
        (section_milestone_flag ?section_enrollment)
        (component_assigned_to_enrollment ?student_enrollment ?assessment_component)
        (section_has_exam_session ?section_enrollment ?exam_session)
        (component_ready_via_intervention ?assessment_component)
        (exam_session_ready_flag ?exam_session)
        (not
          (enrollment_component_confirmed ?student_enrollment)
        )
        (section_component_confirmed ?section_enrollment)
        (submission_package_available ?submission_package)
      )
    :effect
      (and
        (submission_package_composed ?submission_package)
        (package_includes_component ?submission_package ?assessment_component)
        (package_includes_exam_session ?submission_package ?exam_session)
        (package_requires_enrollment_confirmation ?submission_package)
        (not
          (submission_package_available ?submission_package)
        )
      )
  )
  (:action compose_submission_package_requires_section_confirmation
    :parameters (?student_enrollment - student_enrollment ?section_enrollment - section_enrollment ?assessment_component - assessment_component ?exam_session - exam_session ?submission_package - submission_package)
    :precondition
      (and
        (enrollment_milestone_flag ?student_enrollment)
        (section_milestone_flag ?section_enrollment)
        (component_assigned_to_enrollment ?student_enrollment ?assessment_component)
        (section_has_exam_session ?section_enrollment ?exam_session)
        (component_ready_flag ?assessment_component)
        (exam_session_ready_via_intervention ?exam_session)
        (enrollment_component_confirmed ?student_enrollment)
        (not
          (section_component_confirmed ?section_enrollment)
        )
        (submission_package_available ?submission_package)
      )
    :effect
      (and
        (submission_package_composed ?submission_package)
        (package_includes_component ?submission_package ?assessment_component)
        (package_includes_exam_session ?submission_package ?exam_session)
        (package_requires_section_confirmation ?submission_package)
        (not
          (submission_package_available ?submission_package)
        )
      )
  )
  (:action compose_submission_package_requires_both_confirmations
    :parameters (?student_enrollment - student_enrollment ?section_enrollment - section_enrollment ?assessment_component - assessment_component ?exam_session - exam_session ?submission_package - submission_package)
    :precondition
      (and
        (enrollment_milestone_flag ?student_enrollment)
        (section_milestone_flag ?section_enrollment)
        (component_assigned_to_enrollment ?student_enrollment ?assessment_component)
        (section_has_exam_session ?section_enrollment ?exam_session)
        (component_ready_via_intervention ?assessment_component)
        (exam_session_ready_via_intervention ?exam_session)
        (not
          (enrollment_component_confirmed ?student_enrollment)
        )
        (not
          (section_component_confirmed ?section_enrollment)
        )
        (submission_package_available ?submission_package)
      )
    :effect
      (and
        (submission_package_composed ?submission_package)
        (package_includes_component ?submission_package ?assessment_component)
        (package_includes_exam_session ?submission_package ?exam_session)
        (package_requires_enrollment_confirmation ?submission_package)
        (package_requires_section_confirmation ?submission_package)
        (not
          (submission_package_available ?submission_package)
        )
      )
  )
  (:action record_submission_package_execution
    :parameters (?submission_package - submission_package ?student_enrollment - student_enrollment ?assessment_instance - assessment_slot)
    :precondition
      (and
        (submission_package_composed ?submission_package)
        (enrollment_milestone_flag ?student_enrollment)
        (assessment_slot_reserved_for_case ?student_enrollment ?assessment_instance)
        (not
          (submission_package_executed ?submission_package)
        )
      )
    :effect (submission_package_executed ?submission_package)
  )
  (:action verify_evidence_artifact_and_attach_to_package
    :parameters (?recovery_plan - recovery_plan ?evidence_artifact - evidence_artifact ?submission_package - submission_package)
    :precondition
      (and
        (case_eligibility_confirmed ?recovery_plan)
        (plan_linked_to_submission_package ?recovery_plan ?submission_package)
        (plan_has_evidence_artifact ?recovery_plan ?evidence_artifact)
        (evidence_artifact_available ?evidence_artifact)
        (submission_package_composed ?submission_package)
        (submission_package_executed ?submission_package)
        (not
          (artifact_verified ?evidence_artifact)
        )
      )
    :effect
      (and
        (artifact_verified ?evidence_artifact)
        (artifact_linked_to_package ?evidence_artifact ?submission_package)
        (not
          (evidence_artifact_available ?evidence_artifact)
        )
      )
  )
  (:action perform_initial_artifact_review
    :parameters (?recovery_plan - recovery_plan ?evidence_artifact - evidence_artifact ?submission_package - submission_package ?assessment_instance - assessment_slot)
    :precondition
      (and
        (case_eligibility_confirmed ?recovery_plan)
        (plan_has_evidence_artifact ?recovery_plan ?evidence_artifact)
        (artifact_verified ?evidence_artifact)
        (artifact_linked_to_package ?evidence_artifact ?submission_package)
        (assessment_slot_reserved_for_case ?recovery_plan ?assessment_instance)
        (not
          (package_requires_enrollment_confirmation ?submission_package)
        )
        (not
          (plan_initial_artifact_reviewed ?recovery_plan)
        )
      )
    :effect (plan_initial_artifact_reviewed ?recovery_plan)
  )
  (:action attach_policy_clause_to_plan
    :parameters (?recovery_plan - recovery_plan ?policy_clause - policy_clause)
    :precondition
      (and
        (case_eligibility_confirmed ?recovery_plan)
        (policy_clause_available ?policy_clause)
        (not
          (policy_attached_to_plan ?recovery_plan)
        )
      )
    :effect
      (and
        (policy_attached_to_plan ?recovery_plan)
        (policy_clause_attached_to_plan ?recovery_plan ?policy_clause)
        (not
          (policy_clause_available ?policy_clause)
        )
      )
  )
  (:action apply_policy_conditions_and_initiate_review
    :parameters (?recovery_plan - recovery_plan ?evidence_artifact - evidence_artifact ?submission_package - submission_package ?assessment_instance - assessment_slot ?policy_clause - policy_clause)
    :precondition
      (and
        (case_eligibility_confirmed ?recovery_plan)
        (plan_has_evidence_artifact ?recovery_plan ?evidence_artifact)
        (artifact_verified ?evidence_artifact)
        (artifact_linked_to_package ?evidence_artifact ?submission_package)
        (assessment_slot_reserved_for_case ?recovery_plan ?assessment_instance)
        (package_requires_enrollment_confirmation ?submission_package)
        (policy_attached_to_plan ?recovery_plan)
        (policy_clause_attached_to_plan ?recovery_plan ?policy_clause)
        (not
          (plan_initial_artifact_reviewed ?recovery_plan)
        )
      )
    :effect
      (and
        (plan_initial_artifact_reviewed ?recovery_plan)
