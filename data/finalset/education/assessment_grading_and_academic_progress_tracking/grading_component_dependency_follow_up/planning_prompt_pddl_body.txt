## Planning domain (PDDL)

(define (domain grading_component_dependency_follow_up_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types auxiliary_component_type - object auxiliary_scheduling_type - object auxiliary_participant_type - object assessed_entity_group - object assessable_item - assessed_entity_group assessment_slot - auxiliary_component_type assessment_event - auxiliary_component_type instructional_support - auxiliary_component_type approval_item - auxiliary_component_type adjustment_request - auxiliary_component_type academic_term_marker - auxiliary_component_type adjunct_grade_component - auxiliary_component_type administrative_clearance - auxiliary_component_type remediation_resource - auxiliary_scheduling_type assessment_evidence - auxiliary_scheduling_type policy_override - auxiliary_scheduling_type assessment_session_a - auxiliary_participant_type assessment_session_b - auxiliary_participant_type consolidated_grade_record - auxiliary_participant_type component_group - assessable_item enrollment_group - assessable_item prerequisite_component - component_group dependent_component - component_group student_enrollment_record - enrollment_group)
  (:predicates
    (submission_planned ?assessed_entity - assessable_item)
    (component_ready ?assessed_entity - assessable_item)
    (assignment_confirmed ?assessed_entity - assessable_item)
    (component_finalized ?assessed_entity - assessable_item)
    (progression_flag ?assessed_entity - assessable_item)
    (advancement_marker ?assessed_entity - assessable_item)
    (assessment_slot_available ?assessment_slot - assessment_slot)
    (entity_assigned_to_slot ?assessed_entity - assessable_item ?assessment_slot - assessment_slot)
    (assessment_event_available ?assessment_event - assessment_event)
    (scheduled_for_event ?assessed_entity - assessable_item ?assessment_event - assessment_event)
    (instructional_support_available ?instructional_support - instructional_support)
    (entity_assigned_instructional_support ?assessed_entity - assessable_item ?instructional_support - instructional_support)
    (remediation_resource_available ?remediation_resource - remediation_resource)
    (prerequisite_assigned_remediation ?prerequisite_component - prerequisite_component ?remediation_resource - remediation_resource)
    (dependent_assigned_remediation ?dependent_component - dependent_component ?remediation_resource - remediation_resource)
    (prerequisite_mapped_to_session_a ?prerequisite_component - prerequisite_component ?assessment_session_a - assessment_session_a)
    (assessment_session_a_ready ?assessment_session_a - assessment_session_a)
    (assessment_session_a_remediation_flag ?assessment_session_a - assessment_session_a)
    (prerequisite_processed ?prerequisite_component - prerequisite_component)
    (dependent_mapped_to_session_b ?dependent_component - dependent_component ?assessment_session_b - assessment_session_b)
    (assessment_session_b_ready ?assessment_session_b - assessment_session_b)
    (assessment_session_b_remediation_flag ?assessment_session_b - assessment_session_b)
    (dependent_ready_for_consolidation ?dependent_component - dependent_component)
    (grade_record_available ?consolidated_grade_record - consolidated_grade_record)
    (grade_record_produced ?consolidated_grade_record - consolidated_grade_record)
    (grade_record_associated_with_session_a ?consolidated_grade_record - consolidated_grade_record ?assessment_session_a - assessment_session_a)
    (grade_record_associated_with_session_b ?consolidated_grade_record - consolidated_grade_record ?assessment_session_b - assessment_session_b)
    (grade_record_session_a_flag ?consolidated_grade_record - consolidated_grade_record)
    (grade_record_session_b_flag ?consolidated_grade_record - consolidated_grade_record)
    (grade_record_ready_for_evidence ?consolidated_grade_record - consolidated_grade_record)
    (enrollment_has_prerequisite_component ?student_enrollment_record - student_enrollment_record ?prerequisite_component - prerequisite_component)
    (enrollment_has_dependent_component ?student_enrollment_record - student_enrollment_record ?dependent_component - dependent_component)
    (enrollment_linked_to_grade_record ?student_enrollment_record - student_enrollment_record ?consolidated_grade_record - consolidated_grade_record)
    (assessment_evidence_available ?assessment_evidence - assessment_evidence)
    (enrollment_has_evidence ?student_enrollment_record - student_enrollment_record ?assessment_evidence - assessment_evidence)
    (evidence_attached ?assessment_evidence - assessment_evidence)
    (evidence_associated_with_grade_record ?assessment_evidence - assessment_evidence ?consolidated_grade_record - consolidated_grade_record)
    (adjunct_component_validated ?student_enrollment_record - student_enrollment_record)
    (adjunct_component_ready ?student_enrollment_record - student_enrollment_record)
    (adjunct_checks_passed ?student_enrollment_record - student_enrollment_record)
    (approval_registered_on_record ?student_enrollment_record - student_enrollment_record)
    (approval_confirmed ?student_enrollment_record - student_enrollment_record)
    (adjustment_required_on_record ?student_enrollment_record - student_enrollment_record)
    (finalization_checks_passed ?student_enrollment_record - student_enrollment_record)
    (policy_override_available ?policy_override - policy_override)
    (enrollment_has_policy_override ?student_enrollment_record - student_enrollment_record ?policy_override - policy_override)
    (override_engaged_on_record ?student_enrollment_record - student_enrollment_record)
    (override_validated_on_record ?student_enrollment_record - student_enrollment_record)
    (override_confirmed_on_record ?student_enrollment_record - student_enrollment_record)
    (approval_item_available ?approval_item - approval_item)
    (enrollment_linked_to_approval_item ?student_enrollment_record - student_enrollment_record ?approval_item - approval_item)
    (adjustment_request_available ?adjustment_request - adjustment_request)
    (enrollment_linked_adjustment_request ?student_enrollment_record - student_enrollment_record ?adjustment_request - adjustment_request)
    (adjunct_component_available ?adjunct_grade_component - adjunct_grade_component)
    (enrollment_has_adjunct_component ?student_enrollment_record - student_enrollment_record ?adjunct_grade_component - adjunct_grade_component)
    (administrative_clearance_available ?administrative_clearance - administrative_clearance)
    (enrollment_has_administrative_clearance ?student_enrollment_record - student_enrollment_record ?administrative_clearance - administrative_clearance)
    (academic_term_marker_available ?academic_term_marker - academic_term_marker)
    (assessed_entity_linked_to_academic_term_marker ?assessed_entity - assessable_item ?academic_term_marker - academic_term_marker)
    (prerequisite_processing_flag ?prerequisite_component - prerequisite_component)
    (dependent_session_locked ?dependent_component - dependent_component)
    (finalization_mark ?student_enrollment_record - student_enrollment_record)
  )
  (:action plan_submission
    :parameters (?assessed_entity - assessable_item)
    :precondition
      (and
        (not
          (submission_planned ?assessed_entity)
        )
        (not
          (component_finalized ?assessed_entity)
        )
      )
    :effect (submission_planned ?assessed_entity)
  )
  (:action assign_assessment_slot_to_entity
    :parameters (?assessed_entity - assessable_item ?assessment_slot - assessment_slot)
    :precondition
      (and
        (submission_planned ?assessed_entity)
        (not
          (assignment_confirmed ?assessed_entity)
        )
        (assessment_slot_available ?assessment_slot)
      )
    :effect
      (and
        (assignment_confirmed ?assessed_entity)
        (entity_assigned_to_slot ?assessed_entity ?assessment_slot)
        (not
          (assessment_slot_available ?assessment_slot)
        )
      )
  )
  (:action schedule_assessment_event_for_entity
    :parameters (?assessed_entity - assessable_item ?assessment_event - assessment_event)
    :precondition
      (and
        (submission_planned ?assessed_entity)
        (assignment_confirmed ?assessed_entity)
        (assessment_event_available ?assessment_event)
      )
    :effect
      (and
        (scheduled_for_event ?assessed_entity ?assessment_event)
        (not
          (assessment_event_available ?assessment_event)
        )
      )
  )
  (:action confirm_schedule_and_mark_ready
    :parameters (?assessed_entity - assessable_item ?assessment_event - assessment_event)
    :precondition
      (and
        (submission_planned ?assessed_entity)
        (assignment_confirmed ?assessed_entity)
        (scheduled_for_event ?assessed_entity ?assessment_event)
        (not
          (component_ready ?assessed_entity)
        )
      )
    :effect (component_ready ?assessed_entity)
  )
  (:action unschedule_assessment_event_for_entity
    :parameters (?assessed_entity - assessable_item ?assessment_event - assessment_event)
    :precondition
      (and
        (scheduled_for_event ?assessed_entity ?assessment_event)
      )
    :effect
      (and
        (assessment_event_available ?assessment_event)
        (not
          (scheduled_for_event ?assessed_entity ?assessment_event)
        )
      )
  )
  (:action assign_instructional_support_to_entity
    :parameters (?assessed_entity - assessable_item ?instructional_support - instructional_support)
    :precondition
      (and
        (component_ready ?assessed_entity)
        (instructional_support_available ?instructional_support)
      )
    :effect
      (and
        (entity_assigned_instructional_support ?assessed_entity ?instructional_support)
        (not
          (instructional_support_available ?instructional_support)
        )
      )
  )
  (:action release_instructional_support_from_entity
    :parameters (?assessed_entity - assessable_item ?instructional_support - instructional_support)
    :precondition
      (and
        (entity_assigned_instructional_support ?assessed_entity ?instructional_support)
      )
    :effect
      (and
        (instructional_support_available ?instructional_support)
        (not
          (entity_assigned_instructional_support ?assessed_entity ?instructional_support)
        )
      )
  )
  (:action assign_adjunct_component_to_record
    :parameters (?student_enrollment_record - student_enrollment_record ?adjunct_grade_component - adjunct_grade_component)
    :precondition
      (and
        (component_ready ?student_enrollment_record)
        (adjunct_component_available ?adjunct_grade_component)
      )
    :effect
      (and
        (enrollment_has_adjunct_component ?student_enrollment_record ?adjunct_grade_component)
        (not
          (adjunct_component_available ?adjunct_grade_component)
        )
      )
  )
  (:action release_adjunct_component_from_record
    :parameters (?student_enrollment_record - student_enrollment_record ?adjunct_grade_component - adjunct_grade_component)
    :precondition
      (and
        (enrollment_has_adjunct_component ?student_enrollment_record ?adjunct_grade_component)
      )
    :effect
      (and
        (adjunct_component_available ?adjunct_grade_component)
        (not
          (enrollment_has_adjunct_component ?student_enrollment_record ?adjunct_grade_component)
        )
      )
  )
  (:action assign_administrative_clearance_to_record
    :parameters (?student_enrollment_record - student_enrollment_record ?administrative_clearance - administrative_clearance)
    :precondition
      (and
        (component_ready ?student_enrollment_record)
        (administrative_clearance_available ?administrative_clearance)
      )
    :effect
      (and
        (enrollment_has_administrative_clearance ?student_enrollment_record ?administrative_clearance)
        (not
          (administrative_clearance_available ?administrative_clearance)
        )
      )
  )
  (:action release_administrative_clearance_from_record
    :parameters (?student_enrollment_record - student_enrollment_record ?administrative_clearance - administrative_clearance)
    :precondition
      (and
        (enrollment_has_administrative_clearance ?student_enrollment_record ?administrative_clearance)
      )
    :effect
      (and
        (administrative_clearance_available ?administrative_clearance)
        (not
          (enrollment_has_administrative_clearance ?student_enrollment_record ?administrative_clearance)
        )
      )
  )
  (:action activate_assessment_session_a
    :parameters (?prerequisite_component - prerequisite_component ?assessment_session_a - assessment_session_a ?assessment_event - assessment_event)
    :precondition
      (and
        (component_ready ?prerequisite_component)
        (scheduled_for_event ?prerequisite_component ?assessment_event)
        (prerequisite_mapped_to_session_a ?prerequisite_component ?assessment_session_a)
        (not
          (assessment_session_a_ready ?assessment_session_a)
        )
        (not
          (assessment_session_a_remediation_flag ?assessment_session_a)
        )
      )
    :effect (assessment_session_a_ready ?assessment_session_a)
  )
  (:action process_prerequisite_with_instructional_support
    :parameters (?prerequisite_component - prerequisite_component ?assessment_session_a - assessment_session_a ?instructional_support - instructional_support)
    :precondition
      (and
        (component_ready ?prerequisite_component)
        (entity_assigned_instructional_support ?prerequisite_component ?instructional_support)
        (prerequisite_mapped_to_session_a ?prerequisite_component ?assessment_session_a)
        (assessment_session_a_ready ?assessment_session_a)
        (not
          (prerequisite_processing_flag ?prerequisite_component)
        )
      )
    :effect
      (and
        (prerequisite_processing_flag ?prerequisite_component)
        (prerequisite_processed ?prerequisite_component)
      )
  )
  (:action assign_remediation_and_lock_prerequisite
    :parameters (?prerequisite_component - prerequisite_component ?assessment_session_a - assessment_session_a ?remediation_resource - remediation_resource)
    :precondition
      (and
        (component_ready ?prerequisite_component)
        (prerequisite_mapped_to_session_a ?prerequisite_component ?assessment_session_a)
        (remediation_resource_available ?remediation_resource)
        (not
          (prerequisite_processing_flag ?prerequisite_component)
        )
      )
    :effect
      (and
        (assessment_session_a_remediation_flag ?assessment_session_a)
        (prerequisite_processing_flag ?prerequisite_component)
        (prerequisite_assigned_remediation ?prerequisite_component ?remediation_resource)
        (not
          (remediation_resource_available ?remediation_resource)
        )
      )
  )
  (:action complete_prerequisite_remediation_and_reactivate_session_a
    :parameters (?prerequisite_component - prerequisite_component ?assessment_session_a - assessment_session_a ?assessment_event - assessment_event ?remediation_resource - remediation_resource)
    :precondition
      (and
        (component_ready ?prerequisite_component)
        (scheduled_for_event ?prerequisite_component ?assessment_event)
        (prerequisite_mapped_to_session_a ?prerequisite_component ?assessment_session_a)
        (assessment_session_a_remediation_flag ?assessment_session_a)
        (prerequisite_assigned_remediation ?prerequisite_component ?remediation_resource)
        (not
          (prerequisite_processed ?prerequisite_component)
        )
      )
    :effect
      (and
        (assessment_session_a_ready ?assessment_session_a)
        (prerequisite_processed ?prerequisite_component)
        (remediation_resource_available ?remediation_resource)
        (not
          (prerequisite_assigned_remediation ?prerequisite_component ?remediation_resource)
        )
      )
  )
  (:action activate_assessment_session_b
    :parameters (?dependent_component - dependent_component ?assessment_session_b - assessment_session_b ?assessment_event - assessment_event)
    :precondition
      (and
        (component_ready ?dependent_component)
        (scheduled_for_event ?dependent_component ?assessment_event)
        (dependent_mapped_to_session_b ?dependent_component ?assessment_session_b)
        (not
          (assessment_session_b_ready ?assessment_session_b)
        )
        (not
          (assessment_session_b_remediation_flag ?assessment_session_b)
        )
      )
    :effect (assessment_session_b_ready ?assessment_session_b)
  )
  (:action process_dependent_with_instructional_support
    :parameters (?dependent_component - dependent_component ?assessment_session_b - assessment_session_b ?instructional_support - instructional_support)
    :precondition
      (and
        (component_ready ?dependent_component)
        (entity_assigned_instructional_support ?dependent_component ?instructional_support)
        (dependent_mapped_to_session_b ?dependent_component ?assessment_session_b)
        (assessment_session_b_ready ?assessment_session_b)
        (not
          (dependent_session_locked ?dependent_component)
        )
      )
    :effect
      (and
        (dependent_session_locked ?dependent_component)
        (dependent_ready_for_consolidation ?dependent_component)
      )
  )
  (:action assign_remediation_and_lock_dependent
    :parameters (?dependent_component - dependent_component ?assessment_session_b - assessment_session_b ?remediation_resource - remediation_resource)
    :precondition
      (and
        (component_ready ?dependent_component)
        (dependent_mapped_to_session_b ?dependent_component ?assessment_session_b)
        (remediation_resource_available ?remediation_resource)
        (not
          (dependent_session_locked ?dependent_component)
        )
      )
    :effect
      (and
        (assessment_session_b_remediation_flag ?assessment_session_b)
        (dependent_session_locked ?dependent_component)
        (dependent_assigned_remediation ?dependent_component ?remediation_resource)
        (not
          (remediation_resource_available ?remediation_resource)
        )
      )
  )
  (:action complete_dependent_remediation_and_reactivate_session_b
    :parameters (?dependent_component - dependent_component ?assessment_session_b - assessment_session_b ?assessment_event - assessment_event ?remediation_resource - remediation_resource)
    :precondition
      (and
        (component_ready ?dependent_component)
        (scheduled_for_event ?dependent_component ?assessment_event)
        (dependent_mapped_to_session_b ?dependent_component ?assessment_session_b)
        (assessment_session_b_remediation_flag ?assessment_session_b)
        (dependent_assigned_remediation ?dependent_component ?remediation_resource)
        (not
          (dependent_ready_for_consolidation ?dependent_component)
        )
      )
    :effect
      (and
        (assessment_session_b_ready ?assessment_session_b)
        (dependent_ready_for_consolidation ?dependent_component)
        (remediation_resource_available ?remediation_resource)
        (not
          (dependent_assigned_remediation ?dependent_component ?remediation_resource)
        )
      )
  )
  (:action consolidate_components_into_grade_record_standard
    :parameters (?prerequisite_component - prerequisite_component ?dependent_component - dependent_component ?assessment_session_a - assessment_session_a ?assessment_session_b - assessment_session_b ?consolidated_grade_record - consolidated_grade_record)
    :precondition
      (and
        (prerequisite_processing_flag ?prerequisite_component)
        (dependent_session_locked ?dependent_component)
        (prerequisite_mapped_to_session_a ?prerequisite_component ?assessment_session_a)
        (dependent_mapped_to_session_b ?dependent_component ?assessment_session_b)
        (assessment_session_a_ready ?assessment_session_a)
        (assessment_session_b_ready ?assessment_session_b)
        (prerequisite_processed ?prerequisite_component)
        (dependent_ready_for_consolidation ?dependent_component)
        (grade_record_available ?consolidated_grade_record)
      )
    :effect
      (and
        (grade_record_produced ?consolidated_grade_record)
        (grade_record_associated_with_session_a ?consolidated_grade_record ?assessment_session_a)
        (grade_record_associated_with_session_b ?consolidated_grade_record ?assessment_session_b)
        (not
          (grade_record_available ?consolidated_grade_record)
        )
      )
  )
  (:action consolidate_components_into_grade_record_with_session_a_flag
    :parameters (?prerequisite_component - prerequisite_component ?dependent_component - dependent_component ?assessment_session_a - assessment_session_a ?assessment_session_b - assessment_session_b ?consolidated_grade_record - consolidated_grade_record)
    :precondition
      (and
        (prerequisite_processing_flag ?prerequisite_component)
        (dependent_session_locked ?dependent_component)
        (prerequisite_mapped_to_session_a ?prerequisite_component ?assessment_session_a)
        (dependent_mapped_to_session_b ?dependent_component ?assessment_session_b)
        (assessment_session_a_remediation_flag ?assessment_session_a)
        (assessment_session_b_ready ?assessment_session_b)
        (not
          (prerequisite_processed ?prerequisite_component)
        )
        (dependent_ready_for_consolidation ?dependent_component)
        (grade_record_available ?consolidated_grade_record)
      )
    :effect
      (and
        (grade_record_produced ?consolidated_grade_record)
        (grade_record_associated_with_session_a ?consolidated_grade_record ?assessment_session_a)
        (grade_record_associated_with_session_b ?consolidated_grade_record ?assessment_session_b)
        (grade_record_session_a_flag ?consolidated_grade_record)
        (not
          (grade_record_available ?consolidated_grade_record)
        )
      )
  )
  (:action consolidate_components_into_grade_record_with_session_b_flag
    :parameters (?prerequisite_component - prerequisite_component ?dependent_component - dependent_component ?assessment_session_a - assessment_session_a ?assessment_session_b - assessment_session_b ?consolidated_grade_record - consolidated_grade_record)
    :precondition
      (and
        (prerequisite_processing_flag ?prerequisite_component)
        (dependent_session_locked ?dependent_component)
        (prerequisite_mapped_to_session_a ?prerequisite_component ?assessment_session_a)
        (dependent_mapped_to_session_b ?dependent_component ?assessment_session_b)
        (assessment_session_a_ready ?assessment_session_a)
        (assessment_session_b_remediation_flag ?assessment_session_b)
        (prerequisite_processed ?prerequisite_component)
        (not
          (dependent_ready_for_consolidation ?dependent_component)
        )
        (grade_record_available ?consolidated_grade_record)
      )
    :effect
      (and
        (grade_record_produced ?consolidated_grade_record)
        (grade_record_associated_with_session_a ?consolidated_grade_record ?assessment_session_a)
        (grade_record_associated_with_session_b ?consolidated_grade_record ?assessment_session_b)
        (grade_record_session_b_flag ?consolidated_grade_record)
        (not
          (grade_record_available ?consolidated_grade_record)
        )
      )
  )
  (:action consolidate_components_into_grade_record_with_both_session_flags
    :parameters (?prerequisite_component - prerequisite_component ?dependent_component - dependent_component ?assessment_session_a - assessment_session_a ?assessment_session_b - assessment_session_b ?consolidated_grade_record - consolidated_grade_record)
    :precondition
      (and
        (prerequisite_processing_flag ?prerequisite_component)
        (dependent_session_locked ?dependent_component)
        (prerequisite_mapped_to_session_a ?prerequisite_component ?assessment_session_a)
        (dependent_mapped_to_session_b ?dependent_component ?assessment_session_b)
        (assessment_session_a_remediation_flag ?assessment_session_a)
        (assessment_session_b_remediation_flag ?assessment_session_b)
        (not
          (prerequisite_processed ?prerequisite_component)
        )
        (not
          (dependent_ready_for_consolidation ?dependent_component)
        )
        (grade_record_available ?consolidated_grade_record)
      )
    :effect
      (and
        (grade_record_produced ?consolidated_grade_record)
        (grade_record_associated_with_session_a ?consolidated_grade_record ?assessment_session_a)
        (grade_record_associated_with_session_b ?consolidated_grade_record ?assessment_session_b)
        (grade_record_session_a_flag ?consolidated_grade_record)
        (grade_record_session_b_flag ?consolidated_grade_record)
        (not
          (grade_record_available ?consolidated_grade_record)
        )
      )
  )
  (:action mark_grade_record_ready_for_evidence_processing
    :parameters (?consolidated_grade_record - consolidated_grade_record ?prerequisite_component - prerequisite_component ?assessment_event - assessment_event)
    :precondition
      (and
        (grade_record_produced ?consolidated_grade_record)
        (prerequisite_processing_flag ?prerequisite_component)
        (scheduled_for_event ?prerequisite_component ?assessment_event)
        (not
          (grade_record_ready_for_evidence ?consolidated_grade_record)
        )
      )
    :effect (grade_record_ready_for_evidence ?consolidated_grade_record)
  )
  (:action attach_evidence_to_grade_record
    :parameters (?student_enrollment_record - student_enrollment_record ?assessment_evidence - assessment_evidence ?consolidated_grade_record - consolidated_grade_record)
    :precondition
      (and
        (component_ready ?student_enrollment_record)
        (enrollment_linked_to_grade_record ?student_enrollment_record ?consolidated_grade_record)
        (enrollment_has_evidence ?student_enrollment_record ?assessment_evidence)
        (assessment_evidence_available ?assessment_evidence)
        (grade_record_produced ?consolidated_grade_record)
        (grade_record_ready_for_evidence ?consolidated_grade_record)
        (not
          (evidence_attached ?assessment_evidence)
        )
      )
    :effect
      (and
        (evidence_attached ?assessment_evidence)
        (evidence_associated_with_grade_record ?assessment_evidence ?consolidated_grade_record)
        (not
          (assessment_evidence_available ?assessment_evidence)
        )
      )
  )
  (:action validate_evidence_and_update_student_record
    :parameters (?student_enrollment_record - student_enrollment_record ?assessment_evidence - assessment_evidence ?consolidated_grade_record - consolidated_grade_record ?assessment_event - assessment_event)
