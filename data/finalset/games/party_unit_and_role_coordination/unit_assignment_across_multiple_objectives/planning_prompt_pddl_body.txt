## Planning domain (PDDL)

(define (domain unit_assignment_across_multiple_objectives)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity - object unit_class - entity token_class - entity node_class - entity unit_pool - entity unit_candidate - unit_pool role_token - unit_class specialization - unit_class operator - unit_class tag_token - unit_class consumable - unit_class phase_marker - unit_class module_type - unit_class tactic_token - unit_class equipment - token_class module - token_class strategy_tag - token_class subnode_primary - node_class subnode_secondary - node_class asset_platform - node_class slot_class_primary - unit_candidate slot_class_secondary - unit_candidate objective_slot_primary - slot_class_primary objective_slot_secondary - slot_class_primary team - slot_class_secondary)

  (:predicates
    (deployment_unit_registered ?unit_candidate - unit_candidate)
    (deployment_unit_prepared ?unit_candidate - unit_candidate)
    (deployment_unit_role_claimed ?unit_candidate - unit_candidate)
    (deployment_unit_recycled ?unit_candidate - unit_candidate)
    (deployment_unit_activated ?unit_candidate - unit_candidate)
    (deployment_unit_completed_phase ?unit_candidate - unit_candidate)
    (role_token_available ?role_token - role_token)
    (deployment_unit_role_token_assigned ?unit_candidate - unit_candidate ?role_token - role_token)
    (specialization_available ?specialization - specialization)
    (deployment_unit_has_specialization ?unit_candidate - unit_candidate ?specialization - specialization)
    (operator_available ?operator - operator)
    (deployment_unit_bound_operator ?unit_candidate - unit_candidate ?operator - operator)
    (equipment_available ?equipment - equipment)
    (slot_has_equipment ?objective_slot_primary - objective_slot_primary ?equipment - equipment)
    (slot_secondary_has_equipment ?objective_slot_secondary - objective_slot_secondary ?equipment - equipment)
    (slot_has_subnode_primary ?objective_slot_primary - objective_slot_primary ?subnode_a - subnode_primary)
    (subnode_primary_covered ?subnode_a - subnode_primary)
    (subnode_primary_equipment_reserved ?subnode_a - subnode_primary)
    (slot_primary_ready ?objective_slot_primary - objective_slot_primary)
    (slot_has_subnode_secondary ?objective_slot_secondary - objective_slot_secondary ?subnode_b - subnode_secondary)
    (subnode_secondary_covered ?subnode_b - subnode_secondary)
    (subnode_secondary_equipment_reserved ?subnode_b - subnode_secondary)
    (slot_secondary_ready ?objective_slot_secondary - objective_slot_secondary)
    (asset_available ?asset_platform - asset_platform)
    (asset_ready ?asset_platform - asset_platform)
    (asset_covers_subnode_primary ?asset_platform - asset_platform ?subnode_a - subnode_primary)
    (asset_covers_subnode_secondary ?asset_platform - asset_platform ?subnode_b - subnode_secondary)
    (asset_marker_primary ?asset_platform - asset_platform)
    (asset_marker_secondary ?asset_platform - asset_platform)
    (asset_commissioned ?asset_platform - asset_platform)
    (team_assigned_primary_slot ?team - team ?objective_slot_primary - objective_slot_primary)
    (team_assigned_secondary_slot ?team - team ?objective_slot_secondary - objective_slot_secondary)
    (team_assigned_asset ?team - team ?asset_platform - asset_platform)
    (module_available ?module - module)
    (team_has_module ?team - team ?module - module)
    (module_installed ?module - module)
    (module_installed_on_asset ?module - module ?asset_platform - asset_platform)
    (team_integration_started ?team - team)
    (team_integration_stage_two ?team - team)
    (team_pre_activation_ready ?team - team)
    (team_tag_attached ?team - team)
    (team_tag_upgrade_applied ?team - team)
    (team_metadata_attached ?team - team)
    (team_final_checks_passed ?team - team)
    (strategy_tag_available ?strategy_tag - strategy_tag)
    (team_has_strategy_tag ?team - team ?strategy_tag - strategy_tag)
    (team_strategy_tag_activated ?team - team)
    (team_tag_ready_for_tactics ?team - team)
    (team_tactic_applied ?team - team)
    (tag_token_available ?tag_token - tag_token)
    (team_tag_assigned ?team - team ?tag_token - tag_token)
    (consumable_available ?consumable - consumable)
    (team_consumable_assigned ?team - team ?consumable - consumable)
    (module_type_available ?module_type - module_type)
    (team_module_type_assigned ?team - team ?module_type - module_type)
    (tactic_token_available ?tactic_token - tactic_token)
    (team_tactic_assigned ?team - team ?tactic_token - tactic_token)
    (phase_marker_available ?phase_marker - phase_marker)
    (deployment_unit_phase_marker_assigned ?unit_candidate - unit_candidate ?phase_marker - phase_marker)
    (slot_primary_initialized ?objective_slot_primary - objective_slot_primary)
    (slot_secondary_initialized ?objective_slot_secondary - objective_slot_secondary)
    (team_certified ?team - team)
  )
  (:action register_unit_candidate
    :parameters (?unit_candidate - unit_candidate)
    :precondition
      (and
        (not
          (deployment_unit_registered ?unit_candidate)
        )
        (not
          (deployment_unit_recycled ?unit_candidate)
        )
      )
    :effect (deployment_unit_registered ?unit_candidate)
  )
  (:action claim_role_token
    :parameters (?unit_candidate - unit_candidate ?role_token - role_token)
    :precondition
      (and
        (deployment_unit_registered ?unit_candidate)
        (not
          (deployment_unit_role_claimed ?unit_candidate)
        )
        (role_token_available ?role_token)
      )
    :effect
      (and
        (deployment_unit_role_claimed ?unit_candidate)
        (deployment_unit_role_token_assigned ?unit_candidate ?role_token)
        (not
          (role_token_available ?role_token)
        )
      )
  )
  (:action assign_specialization_to_unit
    :parameters (?unit_candidate - unit_candidate ?specialization - specialization)
    :precondition
      (and
        (deployment_unit_registered ?unit_candidate)
        (deployment_unit_role_claimed ?unit_candidate)
        (specialization_available ?specialization)
      )
    :effect
      (and
        (deployment_unit_has_specialization ?unit_candidate ?specialization)
        (not
          (specialization_available ?specialization)
        )
      )
  )
  (:action confirm_specialization_assignment
    :parameters (?unit_candidate - unit_candidate ?specialization - specialization)
    :precondition
      (and
        (deployment_unit_registered ?unit_candidate)
        (deployment_unit_role_claimed ?unit_candidate)
        (deployment_unit_has_specialization ?unit_candidate ?specialization)
        (not
          (deployment_unit_prepared ?unit_candidate)
        )
      )
    :effect (deployment_unit_prepared ?unit_candidate)
  )
  (:action unassign_specialization_from_unit
    :parameters (?unit_candidate - unit_candidate ?specialization - specialization)
    :precondition
      (and
        (deployment_unit_has_specialization ?unit_candidate ?specialization)
      )
    :effect
      (and
        (specialization_available ?specialization)
        (not
          (deployment_unit_has_specialization ?unit_candidate ?specialization)
        )
      )
  )
  (:action bind_operator_to_unit
    :parameters (?unit_candidate - unit_candidate ?operator - operator)
    :precondition
      (and
        (deployment_unit_prepared ?unit_candidate)
        (operator_available ?operator)
      )
    :effect
      (and
        (deployment_unit_bound_operator ?unit_candidate ?operator)
        (not
          (operator_available ?operator)
        )
      )
  )
  (:action unbind_operator_from_unit
    :parameters (?unit_candidate - unit_candidate ?operator - operator)
    :precondition
      (and
        (deployment_unit_bound_operator ?unit_candidate ?operator)
      )
    :effect
      (and
        (operator_available ?operator)
        (not
          (deployment_unit_bound_operator ?unit_candidate ?operator)
        )
      )
  )
  (:action install_module_type_on_team
    :parameters (?team - team ?module_type - module_type)
    :precondition
      (and
        (deployment_unit_prepared ?team)
        (module_type_available ?module_type)
      )
    :effect
      (and
        (team_module_type_assigned ?team ?module_type)
        (not
          (module_type_available ?module_type)
        )
      )
  )
  (:action uninstall_module_type_from_team
    :parameters (?team - team ?module_type - module_type)
    :precondition
      (and
        (team_module_type_assigned ?team ?module_type)
      )
    :effect
      (and
        (module_type_available ?module_type)
        (not
          (team_module_type_assigned ?team ?module_type)
        )
      )
  )
  (:action assign_tactic_to_team
    :parameters (?team - team ?tactic_token - tactic_token)
    :precondition
      (and
        (deployment_unit_prepared ?team)
        (tactic_token_available ?tactic_token)
      )
    :effect
      (and
        (team_tactic_assigned ?team ?tactic_token)
        (not
          (tactic_token_available ?tactic_token)
        )
      )
  )
  (:action remove_tactic_from_team
    :parameters (?team - team ?tactic_token - tactic_token)
    :precondition
      (and
        (team_tactic_assigned ?team ?tactic_token)
      )
    :effect
      (and
        (tactic_token_available ?tactic_token)
        (not
          (team_tactic_assigned ?team ?tactic_token)
        )
      )
  )
  (:action cover_subnode_primary_with_specialization
    :parameters (?objective_slot_primary - objective_slot_primary ?subnode_a - subnode_primary ?specialization - specialization)
    :precondition
      (and
        (deployment_unit_prepared ?objective_slot_primary)
        (deployment_unit_has_specialization ?objective_slot_primary ?specialization)
        (slot_has_subnode_primary ?objective_slot_primary ?subnode_a)
        (not
          (subnode_primary_covered ?subnode_a)
        )
        (not
          (subnode_primary_equipment_reserved ?subnode_a)
        )
      )
    :effect (subnode_primary_covered ?subnode_a)
  )
  (:action confirm_primary_slot_with_operator
    :parameters (?objective_slot_primary - objective_slot_primary ?subnode_a - subnode_primary ?operator - operator)
    :precondition
      (and
        (deployment_unit_prepared ?objective_slot_primary)
        (deployment_unit_bound_operator ?objective_slot_primary ?operator)
        (slot_has_subnode_primary ?objective_slot_primary ?subnode_a)
        (subnode_primary_covered ?subnode_a)
        (not
          (slot_primary_initialized ?objective_slot_primary)
        )
      )
    :effect
      (and
        (slot_primary_initialized ?objective_slot_primary)
        (slot_primary_ready ?objective_slot_primary)
      )
  )
  (:action attach_equipment_to_primary_slot
    :parameters (?objective_slot_primary - objective_slot_primary ?subnode_a - subnode_primary ?equipment - equipment)
    :precondition
      (and
        (deployment_unit_prepared ?objective_slot_primary)
        (slot_has_subnode_primary ?objective_slot_primary ?subnode_a)
        (equipment_available ?equipment)
        (not
          (slot_primary_initialized ?objective_slot_primary)
        )
      )
    :effect
      (and
        (subnode_primary_equipment_reserved ?subnode_a)
        (slot_primary_initialized ?objective_slot_primary)
        (slot_has_equipment ?objective_slot_primary ?equipment)
        (not
          (equipment_available ?equipment)
        )
      )
  )
  (:action finalize_primary_slot_coverage
    :parameters (?objective_slot_primary - objective_slot_primary ?subnode_a - subnode_primary ?specialization - specialization ?equipment - equipment)
    :precondition
      (and
        (deployment_unit_prepared ?objective_slot_primary)
        (deployment_unit_has_specialization ?objective_slot_primary ?specialization)
        (slot_has_subnode_primary ?objective_slot_primary ?subnode_a)
        (subnode_primary_equipment_reserved ?subnode_a)
        (slot_has_equipment ?objective_slot_primary ?equipment)
        (not
          (slot_primary_ready ?objective_slot_primary)
        )
      )
    :effect
      (and
        (subnode_primary_covered ?subnode_a)
        (slot_primary_ready ?objective_slot_primary)
        (equipment_available ?equipment)
        (not
          (slot_has_equipment ?objective_slot_primary ?equipment)
        )
      )
  )
  (:action cover_subnode_secondary_with_specialization
    :parameters (?objective_slot_secondary - objective_slot_secondary ?subnode_b - subnode_secondary ?specialization - specialization)
    :precondition
      (and
        (deployment_unit_prepared ?objective_slot_secondary)
        (deployment_unit_has_specialization ?objective_slot_secondary ?specialization)
        (slot_has_subnode_secondary ?objective_slot_secondary ?subnode_b)
        (not
          (subnode_secondary_covered ?subnode_b)
        )
        (not
          (subnode_secondary_equipment_reserved ?subnode_b)
        )
      )
    :effect (subnode_secondary_covered ?subnode_b)
  )
  (:action confirm_secondary_slot_with_operator
    :parameters (?objective_slot_secondary - objective_slot_secondary ?subnode_b - subnode_secondary ?operator - operator)
    :precondition
      (and
        (deployment_unit_prepared ?objective_slot_secondary)
        (deployment_unit_bound_operator ?objective_slot_secondary ?operator)
        (slot_has_subnode_secondary ?objective_slot_secondary ?subnode_b)
        (subnode_secondary_covered ?subnode_b)
        (not
          (slot_secondary_initialized ?objective_slot_secondary)
        )
      )
    :effect
      (and
        (slot_secondary_initialized ?objective_slot_secondary)
        (slot_secondary_ready ?objective_slot_secondary)
      )
  )
  (:action attach_equipment_to_secondary_slot
    :parameters (?objective_slot_secondary - objective_slot_secondary ?subnode_b - subnode_secondary ?equipment - equipment)
    :precondition
      (and
        (deployment_unit_prepared ?objective_slot_secondary)
        (slot_has_subnode_secondary ?objective_slot_secondary ?subnode_b)
        (equipment_available ?equipment)
        (not
          (slot_secondary_initialized ?objective_slot_secondary)
        )
      )
    :effect
      (and
        (subnode_secondary_equipment_reserved ?subnode_b)
        (slot_secondary_initialized ?objective_slot_secondary)
        (slot_secondary_has_equipment ?objective_slot_secondary ?equipment)
        (not
          (equipment_available ?equipment)
        )
      )
  )
  (:action finalize_secondary_slot_coverage
    :parameters (?objective_slot_secondary - objective_slot_secondary ?subnode_b - subnode_secondary ?specialization - specialization ?equipment - equipment)
    :precondition
      (and
        (deployment_unit_prepared ?objective_slot_secondary)
        (deployment_unit_has_specialization ?objective_slot_secondary ?specialization)
        (slot_has_subnode_secondary ?objective_slot_secondary ?subnode_b)
        (subnode_secondary_equipment_reserved ?subnode_b)
        (slot_secondary_has_equipment ?objective_slot_secondary ?equipment)
        (not
          (slot_secondary_ready ?objective_slot_secondary)
        )
      )
    :effect
      (and
        (subnode_secondary_covered ?subnode_b)
        (slot_secondary_ready ?objective_slot_secondary)
        (equipment_available ?equipment)
        (not
          (slot_secondary_has_equipment ?objective_slot_secondary ?equipment)
        )
      )
  )
  (:action assemble_asset
    :parameters (?objective_slot_primary - objective_slot_primary ?objective_slot_secondary - objective_slot_secondary ?subnode_a - subnode_primary ?subnode_b - subnode_secondary ?asset_platform - asset_platform)
    :precondition
      (and
        (slot_primary_initialized ?objective_slot_primary)
        (slot_secondary_initialized ?objective_slot_secondary)
        (slot_has_subnode_primary ?objective_slot_primary ?subnode_a)
        (slot_has_subnode_secondary ?objective_slot_secondary ?subnode_b)
        (subnode_primary_covered ?subnode_a)
        (subnode_secondary_covered ?subnode_b)
        (slot_primary_ready ?objective_slot_primary)
        (slot_secondary_ready ?objective_slot_secondary)
        (asset_available ?asset_platform)
      )
    :effect
      (and
        (asset_ready ?asset_platform)
        (asset_covers_subnode_primary ?asset_platform ?subnode_a)
        (asset_covers_subnode_secondary ?asset_platform ?subnode_b)
        (not
          (asset_available ?asset_platform)
        )
      )
  )
  (:action assemble_asset_set_marker_primary
    :parameters (?objective_slot_primary - objective_slot_primary ?objective_slot_secondary - objective_slot_secondary ?subnode_a - subnode_primary ?subnode_b - subnode_secondary ?asset_platform - asset_platform)
    :precondition
      (and
        (slot_primary_initialized ?objective_slot_primary)
        (slot_secondary_initialized ?objective_slot_secondary)
        (slot_has_subnode_primary ?objective_slot_primary ?subnode_a)
        (slot_has_subnode_secondary ?objective_slot_secondary ?subnode_b)
        (subnode_primary_equipment_reserved ?subnode_a)
        (subnode_secondary_covered ?subnode_b)
        (not
          (slot_primary_ready ?objective_slot_primary)
        )
        (slot_secondary_ready ?objective_slot_secondary)
        (asset_available ?asset_platform)
      )
    :effect
      (and
        (asset_ready ?asset_platform)
        (asset_covers_subnode_primary ?asset_platform ?subnode_a)
        (asset_covers_subnode_secondary ?asset_platform ?subnode_b)
        (asset_marker_primary ?asset_platform)
        (not
          (asset_available ?asset_platform)
        )
      )
  )
  (:action assemble_asset_set_marker_secondary
    :parameters (?objective_slot_primary - objective_slot_primary ?objective_slot_secondary - objective_slot_secondary ?subnode_a - subnode_primary ?subnode_b - subnode_secondary ?asset_platform - asset_platform)
    :precondition
      (and
        (slot_primary_initialized ?objective_slot_primary)
        (slot_secondary_initialized ?objective_slot_secondary)
        (slot_has_subnode_primary ?objective_slot_primary ?subnode_a)
        (slot_has_subnode_secondary ?objective_slot_secondary ?subnode_b)
        (subnode_primary_covered ?subnode_a)
        (subnode_secondary_equipment_reserved ?subnode_b)
        (slot_primary_ready ?objective_slot_primary)
        (not
          (slot_secondary_ready ?objective_slot_secondary)
        )
        (asset_available ?asset_platform)
      )
    :effect
      (and
        (asset_ready ?asset_platform)
        (asset_covers_subnode_primary ?asset_platform ?subnode_a)
        (asset_covers_subnode_secondary ?asset_platform ?subnode_b)
        (asset_marker_secondary ?asset_platform)
        (not
          (asset_available ?asset_platform)
        )
      )
  )
  (:action assemble_asset_with_both_markers
    :parameters (?objective_slot_primary - objective_slot_primary ?objective_slot_secondary - objective_slot_secondary ?subnode_a - subnode_primary ?subnode_b - subnode_secondary ?asset_platform - asset_platform)
    :precondition
      (and
        (slot_primary_initialized ?objective_slot_primary)
        (slot_secondary_initialized ?objective_slot_secondary)
        (slot_has_subnode_primary ?objective_slot_primary ?subnode_a)
        (slot_has_subnode_secondary ?objective_slot_secondary ?subnode_b)
        (subnode_primary_equipment_reserved ?subnode_a)
        (subnode_secondary_equipment_reserved ?subnode_b)
        (not
          (slot_primary_ready ?objective_slot_primary)
        )
        (not
          (slot_secondary_ready ?objective_slot_secondary)
        )
        (asset_available ?asset_platform)
      )
    :effect
      (and
        (asset_ready ?asset_platform)
        (asset_covers_subnode_primary ?asset_platform ?subnode_a)
        (asset_covers_subnode_secondary ?asset_platform ?subnode_b)
        (asset_marker_primary ?asset_platform)
        (asset_marker_secondary ?asset_platform)
        (not
          (asset_available ?asset_platform)
        )
      )
  )
  (:action commission_asset
    :parameters (?asset_platform - asset_platform ?objective_slot_primary - objective_slot_primary ?specialization - specialization)
    :precondition
      (and
        (asset_ready ?asset_platform)
        (slot_primary_initialized ?objective_slot_primary)
        (deployment_unit_has_specialization ?objective_slot_primary ?specialization)
        (not
          (asset_commissioned ?asset_platform)
        )
      )
    :effect (asset_commissioned ?asset_platform)
  )
  (:action integrate_module_into_asset
    :parameters (?team - team ?module - module ?asset_platform - asset_platform)
    :precondition
      (and
        (deployment_unit_prepared ?team)
        (team_assigned_asset ?team ?asset_platform)
        (team_has_module ?team ?module)
        (module_available ?module)
        (asset_ready ?asset_platform)
        (asset_commissioned ?asset_platform)
        (not
          (module_installed ?module)
        )
      )
    :effect
      (and
        (module_installed ?module)
        (module_installed_on_asset ?module ?asset_platform)
        (not
          (module_available ?module)
        )
      )
  )
  (:action finalize_module_integration
    :parameters (?team - team ?module - module ?asset_platform - asset_platform ?specialization - specialization)
    :precondition
      (and
        (deployment_unit_prepared ?team)
        (team_has_module ?team ?module)
        (module_installed ?module)
        (module_installed_on_asset ?module ?asset_platform)
        (deployment_unit_has_specialization ?team ?specialization)
        (not
          (asset_marker_primary ?asset_platform)
        )
        (not
          (team_integration_started ?team)
        )
      )
    :effect (team_integration_started ?team)
  )
  (:action apply_tag_token_to_team
    :parameters (?team - team ?tag_token - tag_token)
    :precondition
      (and
        (deployment_unit_prepared ?team)
        (tag_token_available ?tag_token)
        (not
          (team_tag_attached ?team)
        )
      )
    :effect
      (and
        (team_tag_attached ?team)
        (team_tag_assigned ?team ?tag_token)
        (not
          (tag_token_available ?tag_token)
        )
      )
  )
  (:action apply_tag_upgrade_to_team
    :parameters (?team - team ?module - module ?asset_platform - asset_platform ?specialization - specialization ?tag_token - tag_token)
    :precondition
      (and
        (deployment_unit_prepared ?team)
        (team_has_module ?team ?module)
        (module_installed ?module)
        (module_installed_on_asset ?module ?asset_platform)
        (deployment_unit_has_specialization ?team ?specialization)
        (asset_marker_primary ?asset_platform)
        (team_tag_attached ?team)
        (team_tag_assigned ?team ?tag_token)
        (not
          (team_integration_started ?team)
        )
      )
    :effect
      (and
        (team_integration_started ?team)
        (team_tag_upgrade_applied ?team)
      )
  )
  (:action apply_module_integration_to_team
    :parameters (?team - team ?module_type - module_type ?operator - operator ?module - module ?asset_platform - asset_platform)
    :precondition
      (and
        (team_integration_started ?team)
        (team_module_type_assigned ?team ?module_type)
        (deployment_unit_bound_operator ?team ?operator)
        (team_has_module ?team ?module)
        (module_installed_on_asset ?module ?asset_platform)
        (not
          (asset_marker_secondary ?asset_platform)
        )
        (not
          (team_integration_stage_two ?team)
        )
      )
    :effect (team_integration_stage_two ?team)
  )
  (:action complete_team_module_integration
    :parameters (?team - team ?module_type - module_type ?operator - operator ?module - module ?asset_platform - asset_platform)
    :precondition
      (and
        (team_integration_started ?team)
        (team_module_type_assigned ?team ?module_type)
        (deployment_unit_bound_operator ?team ?operator)
        (team_has_module ?team ?module)
        (module_installed_on_asset ?module ?asset_platform)
        (asset_marker_secondary ?asset_platform)
        (not
          (team_integration_stage_two ?team)
        )
      )
    :effect (team_integration_stage_two ?team)
  )
  (:action advance_team_to_pre_activation
    :parameters (?team - team ?tactic_token - tactic_token ?module - module ?asset_platform - asset_platform)
    :precondition
      (and
        (team_integration_stage_two ?team)
        (team_tactic_assigned ?team ?tactic_token)
        (team_has_module ?team ?module)
        (module_installed_on_asset ?module ?asset_platform)
        (not
          (asset_marker_primary ?asset_platform)
        )
        (not
          (asset_marker_secondary ?asset_platform)
        )
        (not
          (team_pre_activation_ready ?team)
        )
      )
    :effect (team_pre_activation_ready ?team)
  )
  (:action advance_team_and_attach_metadata
