## Planning domain (PDDL)

(define (domain defense_mission_wave_progression)
  (:requirements :strips :typing :negative-preconditions)
  (:types game_entity - object game_asset_type - game_entity item_category - game_entity location_category - game_entity mission_category - game_entity objective_node - mission_category spawn_group_template - game_asset_type enemy_unit_template - game_asset_type defender_structure_template - game_asset_type condition_token - game_asset_type milestone_token - game_asset_type consumable_resource - game_asset_type equipment_modifier - game_asset_type narrative_flag - game_asset_type support_item - item_category module_component - item_category affiliation_marker - item_category sector_slot - location_category sector_node - location_category reward_node - location_category wave_role_container - objective_node stage_container - objective_node attacker_wave_slot - wave_role_container ally_wave_slot - wave_role_container mission_instance - stage_container)

  (:predicates
    (mission_component_activated ?objective_node - objective_node)
    (mission_component_configured ?objective_node - objective_node)
    (mission_component_spawn_group_bound ?objective_node - objective_node)
    (mission_component_completed ?objective_node - objective_node)
    (mission_component_reward_claimed ?objective_node - objective_node)
    (mission_component_reward_granted ?objective_node - objective_node)
    (spawn_group_available ?spawn_group_template - spawn_group_template)
    (mission_component_spawn_group_assigned ?objective_node - objective_node ?spawn_group_template - spawn_group_template)
    (enemy_unit_available ?enemy_unit_template - enemy_unit_template)
    (mission_component_enemy_template_assigned ?objective_node - objective_node ?enemy_unit_template - enemy_unit_template)
    (structure_available ?defender_structure_template - defender_structure_template)
    (mission_component_structure_assigned ?objective_node - objective_node ?defender_structure_template - defender_structure_template)
    (support_item_available ?support_item - support_item)
    (attacker_wave_slot_has_support_item ?attacker_wave_slot - attacker_wave_slot ?support_item - support_item)
    (ally_wave_slot_has_support_item ?ally_wave_slot - ally_wave_slot ?support_item - support_item)
    (attacker_wave_slot_bound_to_sector_slot ?attacker_wave_slot - attacker_wave_slot ?sector_slot - sector_slot)
    (sector_attacker_flag ?sector_slot - sector_slot)
    (sector_support_flag ?sector_slot - sector_slot)
    (attacker_wave_slot_confirmed ?attacker_wave_slot - attacker_wave_slot)
    (ally_wave_slot_bound_to_sector_node ?ally_wave_slot - ally_wave_slot ?sector_node - sector_node)
    (sector_node_attacker_flag ?sector_node - sector_node)
    (sector_node_support_flag ?sector_node - sector_node)
    (ally_wave_slot_confirmed ?ally_wave_slot - ally_wave_slot)
    (reward_node_unpopulated ?reward_node - reward_node)
    (reward_node_populated ?reward_node - reward_node)
    (reward_node_linked_to_slot ?reward_node - reward_node ?sector_slot - sector_slot)
    (reward_node_linked_to_node ?reward_node - reward_node ?sector_node - sector_node)
    (reward_node_support_marker ?reward_node - reward_node)
    (reward_node_attack_marker ?reward_node - reward_node)
    (reward_node_validated ?reward_node - reward_node)
    (instance_has_attacker_slot ?mission_instance - mission_instance ?attacker_wave_slot - attacker_wave_slot)
    (instance_has_ally_slot ?mission_instance - mission_instance ?ally_wave_slot - ally_wave_slot)
    (instance_has_reward_node ?mission_instance - mission_instance ?reward_node - reward_node)
    (module_component_available ?module_component - module_component)
    (instance_has_module_component ?mission_instance - mission_instance ?module_component - module_component)
    (module_component_attached ?module_component - module_component)
    (module_component_linked_to_reward_node ?module_component - module_component ?reward_node - reward_node)
    (instance_stage_ready ?mission_instance - mission_instance)
    (instance_components_committed ?mission_instance - mission_instance)
    (instance_ready_for_finalization ?mission_instance - mission_instance)
    (instance_condition_claimed ?mission_instance - mission_instance)
    (instance_condition_stage2 ?mission_instance - mission_instance)
    (instance_milestone_applied ?mission_instance - mission_instance)
    (instance_checkpoint_applied ?mission_instance - mission_instance)
    (affiliation_marker_available ?affiliation_marker - affiliation_marker)
    (instance_has_affiliation_marker ?mission_instance - mission_instance ?affiliation_marker - affiliation_marker)
    (instance_affiliated ?mission_instance - mission_instance)
    (instance_condition_stage1 ?mission_instance - mission_instance)
    (instance_narrative_ready ?mission_instance - mission_instance)
    (condition_token_available ?condition_token - condition_token)
    (instance_has_condition_token ?mission_instance - mission_instance ?condition_token - condition_token)
    (milestone_token_available ?milestone_token - milestone_token)
    (instance_has_milestone_token ?mission_instance - mission_instance ?milestone_token - milestone_token)
    (equipment_modifier_available ?equipment_modifier - equipment_modifier)
    (instance_has_equipment_modifier ?mission_instance - mission_instance ?equipment_modifier - equipment_modifier)
    (narrative_flag_available ?narrative_flag - narrative_flag)
    (instance_has_narrative_flag ?mission_instance - mission_instance ?narrative_flag - narrative_flag)
    (consumable_resource_available ?consumable_resource - consumable_resource)
    (mission_component_has_consumable_resource ?objective_node - objective_node ?consumable_resource - consumable_resource)
    (attacker_wave_slot_progressed ?attacker_wave_slot - attacker_wave_slot)
    (ally_wave_slot_progressed ?ally_wave_slot - ally_wave_slot)
    (instance_stage_finalized ?mission_instance - mission_instance)
  )
  (:action activate_objective
    :parameters (?objective_node - objective_node)
    :precondition
      (and
        (not
          (mission_component_activated ?objective_node)
        )
        (not
          (mission_component_completed ?objective_node)
        )
      )
    :effect (mission_component_activated ?objective_node)
  )
  (:action assign_spawn_group_to_objective
    :parameters (?objective_node - objective_node ?spawn_group_template - spawn_group_template)
    :precondition
      (and
        (mission_component_activated ?objective_node)
        (not
          (mission_component_spawn_group_bound ?objective_node)
        )
        (spawn_group_available ?spawn_group_template)
      )
    :effect
      (and
        (mission_component_spawn_group_bound ?objective_node)
        (mission_component_spawn_group_assigned ?objective_node ?spawn_group_template)
        (not
          (spawn_group_available ?spawn_group_template)
        )
      )
  )
  (:action assign_enemy_template_to_objective
    :parameters (?objective_node - objective_node ?enemy_unit_template - enemy_unit_template)
    :precondition
      (and
        (mission_component_activated ?objective_node)
        (mission_component_spawn_group_bound ?objective_node)
        (enemy_unit_available ?enemy_unit_template)
      )
    :effect
      (and
        (mission_component_enemy_template_assigned ?objective_node ?enemy_unit_template)
        (not
          (enemy_unit_available ?enemy_unit_template)
        )
      )
  )
  (:action configure_objective
    :parameters (?objective_node - objective_node ?enemy_unit_template - enemy_unit_template)
    :precondition
      (and
        (mission_component_activated ?objective_node)
        (mission_component_spawn_group_bound ?objective_node)
        (mission_component_enemy_template_assigned ?objective_node ?enemy_unit_template)
        (not
          (mission_component_configured ?objective_node)
        )
      )
    :effect (mission_component_configured ?objective_node)
  )
  (:action release_enemy_template
    :parameters (?objective_node - objective_node ?enemy_unit_template - enemy_unit_template)
    :precondition
      (and
        (mission_component_enemy_template_assigned ?objective_node ?enemy_unit_template)
      )
    :effect
      (and
        (enemy_unit_available ?enemy_unit_template)
        (not
          (mission_component_enemy_template_assigned ?objective_node ?enemy_unit_template)
        )
      )
  )
  (:action assign_defender_to_objective
    :parameters (?objective_node - objective_node ?defender_structure_template - defender_structure_template)
    :precondition
      (and
        (mission_component_configured ?objective_node)
        (structure_available ?defender_structure_template)
      )
    :effect
      (and
        (mission_component_structure_assigned ?objective_node ?defender_structure_template)
        (not
          (structure_available ?defender_structure_template)
        )
      )
  )
  (:action release_defender_from_objective
    :parameters (?objective_node - objective_node ?defender_structure_template - defender_structure_template)
    :precondition
      (and
        (mission_component_structure_assigned ?objective_node ?defender_structure_template)
      )
    :effect
      (and
        (structure_available ?defender_structure_template)
        (not
          (mission_component_structure_assigned ?objective_node ?defender_structure_template)
        )
      )
  )
  (:action attach_equipment_modifier_to_instance
    :parameters (?mission_instance - mission_instance ?equipment_modifier - equipment_modifier)
    :precondition
      (and
        (mission_component_configured ?mission_instance)
        (equipment_modifier_available ?equipment_modifier)
      )
    :effect
      (and
        (instance_has_equipment_modifier ?mission_instance ?equipment_modifier)
        (not
          (equipment_modifier_available ?equipment_modifier)
        )
      )
  )
  (:action detach_equipment_modifier_from_instance
    :parameters (?mission_instance - mission_instance ?equipment_modifier - equipment_modifier)
    :precondition
      (and
        (instance_has_equipment_modifier ?mission_instance ?equipment_modifier)
      )
    :effect
      (and
        (equipment_modifier_available ?equipment_modifier)
        (not
          (instance_has_equipment_modifier ?mission_instance ?equipment_modifier)
        )
      )
  )
  (:action attach_narrative_flag_to_instance
    :parameters (?mission_instance - mission_instance ?narrative_flag - narrative_flag)
    :precondition
      (and
        (mission_component_configured ?mission_instance)
        (narrative_flag_available ?narrative_flag)
      )
    :effect
      (and
        (instance_has_narrative_flag ?mission_instance ?narrative_flag)
        (not
          (narrative_flag_available ?narrative_flag)
        )
      )
  )
  (:action detach_narrative_flag_from_instance
    :parameters (?mission_instance - mission_instance ?narrative_flag - narrative_flag)
    :precondition
      (and
        (instance_has_narrative_flag ?mission_instance ?narrative_flag)
      )
    :effect
      (and
        (narrative_flag_available ?narrative_flag)
        (not
          (instance_has_narrative_flag ?mission_instance ?narrative_flag)
        )
      )
  )
  (:action mark_sector_attacked
    :parameters (?attacker_wave_slot - attacker_wave_slot ?sector_slot - sector_slot ?enemy_unit_template - enemy_unit_template)
    :precondition
      (and
        (mission_component_configured ?attacker_wave_slot)
        (mission_component_enemy_template_assigned ?attacker_wave_slot ?enemy_unit_template)
        (attacker_wave_slot_bound_to_sector_slot ?attacker_wave_slot ?sector_slot)
        (not
          (sector_attacker_flag ?sector_slot)
        )
        (not
          (sector_support_flag ?sector_slot)
        )
      )
    :effect (sector_attacker_flag ?sector_slot)
  )
  (:action confirm_attacker_wave_progress
    :parameters (?attacker_wave_slot - attacker_wave_slot ?sector_slot - sector_slot ?defender_structure_template - defender_structure_template)
    :precondition
      (and
        (mission_component_configured ?attacker_wave_slot)
        (mission_component_structure_assigned ?attacker_wave_slot ?defender_structure_template)
        (attacker_wave_slot_bound_to_sector_slot ?attacker_wave_slot ?sector_slot)
        (sector_attacker_flag ?sector_slot)
        (not
          (attacker_wave_slot_progressed ?attacker_wave_slot)
        )
      )
    :effect
      (and
        (attacker_wave_slot_progressed ?attacker_wave_slot)
        (attacker_wave_slot_confirmed ?attacker_wave_slot)
      )
  )
  (:action apply_support_item_to_attacker_slot
    :parameters (?attacker_wave_slot - attacker_wave_slot ?sector_slot - sector_slot ?support_item - support_item)
    :precondition
      (and
        (mission_component_configured ?attacker_wave_slot)
        (attacker_wave_slot_bound_to_sector_slot ?attacker_wave_slot ?sector_slot)
        (support_item_available ?support_item)
        (not
          (attacker_wave_slot_progressed ?attacker_wave_slot)
        )
      )
    :effect
      (and
        (sector_support_flag ?sector_slot)
        (attacker_wave_slot_progressed ?attacker_wave_slot)
        (attacker_wave_slot_has_support_item ?attacker_wave_slot ?support_item)
        (not
          (support_item_available ?support_item)
        )
      )
  )
  (:action finalize_attacker_sector_progress
    :parameters (?attacker_wave_slot - attacker_wave_slot ?sector_slot - sector_slot ?enemy_unit_template - enemy_unit_template ?support_item - support_item)
    :precondition
      (and
        (mission_component_configured ?attacker_wave_slot)
        (mission_component_enemy_template_assigned ?attacker_wave_slot ?enemy_unit_template)
        (attacker_wave_slot_bound_to_sector_slot ?attacker_wave_slot ?sector_slot)
        (sector_support_flag ?sector_slot)
        (attacker_wave_slot_has_support_item ?attacker_wave_slot ?support_item)
        (not
          (attacker_wave_slot_confirmed ?attacker_wave_slot)
        )
      )
    :effect
      (and
        (sector_attacker_flag ?sector_slot)
        (attacker_wave_slot_confirmed ?attacker_wave_slot)
        (support_item_available ?support_item)
        (not
          (attacker_wave_slot_has_support_item ?attacker_wave_slot ?support_item)
        )
      )
  )
  (:action mark_sector_for_ally
    :parameters (?ally_wave_slot - ally_wave_slot ?sector_node - sector_node ?enemy_unit_template - enemy_unit_template)
    :precondition
      (and
        (mission_component_configured ?ally_wave_slot)
        (mission_component_enemy_template_assigned ?ally_wave_slot ?enemy_unit_template)
        (ally_wave_slot_bound_to_sector_node ?ally_wave_slot ?sector_node)
        (not
          (sector_node_attacker_flag ?sector_node)
        )
        (not
          (sector_node_support_flag ?sector_node)
        )
      )
    :effect (sector_node_attacker_flag ?sector_node)
  )
  (:action confirm_ally_wave_progress
    :parameters (?ally_wave_slot - ally_wave_slot ?sector_node - sector_node ?defender_structure_template - defender_structure_template)
    :precondition
      (and
        (mission_component_configured ?ally_wave_slot)
        (mission_component_structure_assigned ?ally_wave_slot ?defender_structure_template)
        (ally_wave_slot_bound_to_sector_node ?ally_wave_slot ?sector_node)
        (sector_node_attacker_flag ?sector_node)
        (not
          (ally_wave_slot_progressed ?ally_wave_slot)
        )
      )
    :effect
      (and
        (ally_wave_slot_progressed ?ally_wave_slot)
        (ally_wave_slot_confirmed ?ally_wave_slot)
      )
  )
  (:action apply_support_item_to_ally_slot
    :parameters (?ally_wave_slot - ally_wave_slot ?sector_node - sector_node ?support_item - support_item)
    :precondition
      (and
        (mission_component_configured ?ally_wave_slot)
        (ally_wave_slot_bound_to_sector_node ?ally_wave_slot ?sector_node)
        (support_item_available ?support_item)
        (not
          (ally_wave_slot_progressed ?ally_wave_slot)
        )
      )
    :effect
      (and
        (sector_node_support_flag ?sector_node)
        (ally_wave_slot_progressed ?ally_wave_slot)
        (ally_wave_slot_has_support_item ?ally_wave_slot ?support_item)
        (not
          (support_item_available ?support_item)
        )
      )
  )
  (:action finalize_ally_sector_progress
    :parameters (?ally_wave_slot - ally_wave_slot ?sector_node - sector_node ?enemy_unit_template - enemy_unit_template ?support_item - support_item)
    :precondition
      (and
        (mission_component_configured ?ally_wave_slot)
        (mission_component_enemy_template_assigned ?ally_wave_slot ?enemy_unit_template)
        (ally_wave_slot_bound_to_sector_node ?ally_wave_slot ?sector_node)
        (sector_node_support_flag ?sector_node)
        (ally_wave_slot_has_support_item ?ally_wave_slot ?support_item)
        (not
          (ally_wave_slot_confirmed ?ally_wave_slot)
        )
      )
    :effect
      (and
        (sector_node_attacker_flag ?sector_node)
        (ally_wave_slot_confirmed ?ally_wave_slot)
        (support_item_available ?support_item)
        (not
          (ally_wave_slot_has_support_item ?ally_wave_slot ?support_item)
        )
      )
  )
  (:action populate_reward_node
    :parameters (?attacker_wave_slot - attacker_wave_slot ?ally_wave_slot - ally_wave_slot ?sector_slot - sector_slot ?sector_node - sector_node ?reward_node - reward_node)
    :precondition
      (and
        (attacker_wave_slot_progressed ?attacker_wave_slot)
        (ally_wave_slot_progressed ?ally_wave_slot)
        (attacker_wave_slot_bound_to_sector_slot ?attacker_wave_slot ?sector_slot)
        (ally_wave_slot_bound_to_sector_node ?ally_wave_slot ?sector_node)
        (sector_attacker_flag ?sector_slot)
        (sector_node_attacker_flag ?sector_node)
        (attacker_wave_slot_confirmed ?attacker_wave_slot)
        (ally_wave_slot_confirmed ?ally_wave_slot)
        (reward_node_unpopulated ?reward_node)
      )
    :effect
      (and
        (reward_node_populated ?reward_node)
        (reward_node_linked_to_slot ?reward_node ?sector_slot)
        (reward_node_linked_to_node ?reward_node ?sector_node)
        (not
          (reward_node_unpopulated ?reward_node)
        )
      )
  )
  (:action populate_reward_node_with_support_marker
    :parameters (?attacker_wave_slot - attacker_wave_slot ?ally_wave_slot - ally_wave_slot ?sector_slot - sector_slot ?sector_node - sector_node ?reward_node - reward_node)
    :precondition
      (and
        (attacker_wave_slot_progressed ?attacker_wave_slot)
        (ally_wave_slot_progressed ?ally_wave_slot)
        (attacker_wave_slot_bound_to_sector_slot ?attacker_wave_slot ?sector_slot)
        (ally_wave_slot_bound_to_sector_node ?ally_wave_slot ?sector_node)
        (sector_support_flag ?sector_slot)
        (sector_node_attacker_flag ?sector_node)
        (not
          (attacker_wave_slot_confirmed ?attacker_wave_slot)
        )
        (ally_wave_slot_confirmed ?ally_wave_slot)
        (reward_node_unpopulated ?reward_node)
      )
    :effect
      (and
        (reward_node_populated ?reward_node)
        (reward_node_linked_to_slot ?reward_node ?sector_slot)
        (reward_node_linked_to_node ?reward_node ?sector_node)
        (reward_node_support_marker ?reward_node)
        (not
          (reward_node_unpopulated ?reward_node)
        )
      )
  )
  (:action populate_reward_node_with_attack_marker
    :parameters (?attacker_wave_slot - attacker_wave_slot ?ally_wave_slot - ally_wave_slot ?sector_slot - sector_slot ?sector_node - sector_node ?reward_node - reward_node)
    :precondition
      (and
        (attacker_wave_slot_progressed ?attacker_wave_slot)
        (ally_wave_slot_progressed ?ally_wave_slot)
        (attacker_wave_slot_bound_to_sector_slot ?attacker_wave_slot ?sector_slot)
        (ally_wave_slot_bound_to_sector_node ?ally_wave_slot ?sector_node)
        (sector_attacker_flag ?sector_slot)
        (sector_node_support_flag ?sector_node)
        (attacker_wave_slot_confirmed ?attacker_wave_slot)
        (not
          (ally_wave_slot_confirmed ?ally_wave_slot)
        )
        (reward_node_unpopulated ?reward_node)
      )
    :effect
      (and
        (reward_node_populated ?reward_node)
        (reward_node_linked_to_slot ?reward_node ?sector_slot)
        (reward_node_linked_to_node ?reward_node ?sector_node)
        (reward_node_attack_marker ?reward_node)
        (not
          (reward_node_unpopulated ?reward_node)
        )
      )
  )
  (:action populate_reward_node_with_both_markers
    :parameters (?attacker_wave_slot - attacker_wave_slot ?ally_wave_slot - ally_wave_slot ?sector_slot - sector_slot ?sector_node - sector_node ?reward_node - reward_node)
    :precondition
      (and
        (attacker_wave_slot_progressed ?attacker_wave_slot)
        (ally_wave_slot_progressed ?ally_wave_slot)
        (attacker_wave_slot_bound_to_sector_slot ?attacker_wave_slot ?sector_slot)
        (ally_wave_slot_bound_to_sector_node ?ally_wave_slot ?sector_node)
        (sector_support_flag ?sector_slot)
        (sector_node_support_flag ?sector_node)
        (not
          (attacker_wave_slot_confirmed ?attacker_wave_slot)
        )
        (not
          (ally_wave_slot_confirmed ?ally_wave_slot)
        )
        (reward_node_unpopulated ?reward_node)
      )
    :effect
      (and
        (reward_node_populated ?reward_node)
        (reward_node_linked_to_slot ?reward_node ?sector_slot)
        (reward_node_linked_to_node ?reward_node ?sector_node)
        (reward_node_support_marker ?reward_node)
        (reward_node_attack_marker ?reward_node)
        (not
          (reward_node_unpopulated ?reward_node)
        )
      )
  )
  (:action validate_reward_node
    :parameters (?reward_node - reward_node ?attacker_wave_slot - attacker_wave_slot ?enemy_unit_template - enemy_unit_template)
    :precondition
      (and
        (reward_node_populated ?reward_node)
        (attacker_wave_slot_progressed ?attacker_wave_slot)
        (mission_component_enemy_template_assigned ?attacker_wave_slot ?enemy_unit_template)
        (not
          (reward_node_validated ?reward_node)
        )
      )
    :effect (reward_node_validated ?reward_node)
  )
  (:action attach_module_to_reward_node
    :parameters (?mission_instance - mission_instance ?module_component - module_component ?reward_node - reward_node)
    :precondition
      (and
        (mission_component_configured ?mission_instance)
        (instance_has_reward_node ?mission_instance ?reward_node)
        (instance_has_module_component ?mission_instance ?module_component)
        (module_component_available ?module_component)
        (reward_node_populated ?reward_node)
        (reward_node_validated ?reward_node)
        (not
          (module_component_attached ?module_component)
        )
      )
    :effect
      (and
        (module_component_attached ?module_component)
        (module_component_linked_to_reward_node ?module_component ?reward_node)
        (not
          (module_component_available ?module_component)
        )
      )
  )
  (:action prepare_instance_stage
    :parameters (?mission_instance - mission_instance ?module_component - module_component ?reward_node - reward_node ?enemy_unit_template - enemy_unit_template)
    :precondition
      (and
        (mission_component_configured ?mission_instance)
        (instance_has_module_component ?mission_instance ?module_component)
        (module_component_attached ?module_component)
        (module_component_linked_to_reward_node ?module_component ?reward_node)
        (mission_component_enemy_template_assigned ?mission_instance ?enemy_unit_template)
        (not
          (reward_node_support_marker ?reward_node)
        )
        (not
          (instance_stage_ready ?mission_instance)
        )
      )
    :effect (instance_stage_ready ?mission_instance)
  )
  (:action assign_condition_token_to_instance
    :parameters (?mission_instance - mission_instance ?condition_token - condition_token)
    :precondition
      (and
        (mission_component_configured ?mission_instance)
        (condition_token_available ?condition_token)
        (not
          (instance_condition_claimed ?mission_instance)
        )
      )
    :effect
      (and
        (instance_condition_claimed ?mission_instance)
        (instance_has_condition_token ?mission_instance ?condition_token)
        (not
          (condition_token_available ?condition_token)
        )
      )
