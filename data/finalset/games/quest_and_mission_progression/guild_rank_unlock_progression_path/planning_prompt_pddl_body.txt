## Planning domain (PDDL)

(define (domain guild_rank_unlock_progression_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types gameplay_element_group - object item_group - object milestone_group - object top_progression_type - object rank_node - top_progression_type unlock_token - gameplay_element_group quest - gameplay_element_group agent - gameplay_element_group conditional_key - gameplay_element_group certification - gameplay_element_group badge - gameplay_element_group special_objective - gameplay_element_group endorsement - gameplay_element_group consumable_item - item_group resource_pack - item_group sponsor - item_group milestone_type_a - milestone_group milestone_type_b - milestone_group rank_seal - milestone_group branch_group - rank_node profile_group - rank_node progression_slot_a - branch_group progression_slot_b - branch_group rank_profile - profile_group)
  (:predicates
    (progression_entity_registered ?rank_node - rank_node)
    (progression_entity_verified ?rank_node - rank_node)
    (progression_entity_has_token ?rank_node - rank_node)
    (progression_entity_unlocked ?rank_node - rank_node)
    (progression_certified ?rank_node - rank_node)
    (progression_entity_endorsed ?rank_node - rank_node)
    (unlock_token_available ?unlock_token - unlock_token)
    (progression_entity_assigned_token ?rank_node - rank_node ?unlock_token - unlock_token)
    (quest_available ?quest - quest)
    (progression_entity_has_quest ?rank_node - rank_node ?quest - quest)
    (agent_available ?agent - agent)
    (progression_entity_assigned_agent ?rank_node - rank_node ?agent - agent)
    (consumable_available ?consumable_item - consumable_item)
    (slot_a_has_consumable ?progression_slot_a - progression_slot_a ?consumable_item - consumable_item)
    (slot_b_has_consumable ?progression_slot_b - progression_slot_b ?consumable_item - consumable_item)
    (slot_a_has_milestone_a ?progression_slot_a - progression_slot_a ?milestone_type_a - milestone_type_a)
    (milestone_type_a_ready ?milestone_type_a - milestone_type_a)
    (milestone_type_a_staged ?milestone_type_a - milestone_type_a)
    (slot_a_milestone_complete ?progression_slot_a - progression_slot_a)
    (slot_b_has_milestone_b ?progression_slot_b - progression_slot_b ?milestone_type_b - milestone_type_b)
    (milestone_type_b_ready ?milestone_type_b - milestone_type_b)
    (milestone_type_b_staged ?milestone_type_b - milestone_type_b)
    (slot_b_milestone_complete ?progression_slot_b - progression_slot_b)
    (rank_seal_available ?rank_seal - rank_seal)
    (rank_seal_created ?rank_seal - rank_seal)
    (rank_seal_includes_milestone_a ?rank_seal - rank_seal ?milestone_type_a - milestone_type_a)
    (rank_seal_includes_milestone_b ?rank_seal - rank_seal ?milestone_type_b - milestone_type_b)
    (rank_seal_includes_branch_a ?rank_seal - rank_seal)
    (rank_seal_includes_branch_b ?rank_seal - rank_seal)
    (rank_seal_validated ?rank_seal - rank_seal)
    (rank_profile_has_slot_a ?rank_profile - rank_profile ?progression_slot_a - progression_slot_a)
    (rank_profile_has_slot_b ?rank_profile - rank_profile ?progression_slot_b - progression_slot_b)
    (rank_profile_has_seal ?rank_profile - rank_profile ?rank_seal - rank_seal)
    (resource_pack_available ?resource_pack - resource_pack)
    (rank_profile_has_resource_pack ?rank_profile - rank_profile ?resource_pack - resource_pack)
    (resource_pack_consumed ?resource_pack - resource_pack)
    (resource_pack_bound_to_seal ?resource_pack - resource_pack ?rank_seal - rank_seal)
    (rank_profile_component_verified ?rank_profile - rank_profile)
    (rank_profile_processed ?rank_profile - rank_profile)
    (rank_profile_ready_for_finalization ?rank_profile - rank_profile)
    (rank_profile_has_key ?rank_profile - rank_profile)
    (rank_profile_key_applied ?rank_profile - rank_profile)
    (rank_profile_certification_ready ?rank_profile - rank_profile)
    (rank_profile_final_checks_passed ?rank_profile - rank_profile)
    (sponsor_available ?sponsor - sponsor)
    (rank_profile_has_sponsor ?rank_profile - rank_profile ?sponsor - sponsor)
    (rank_profile_sponsored ?rank_profile - rank_profile)
    (rank_profile_agent_confirmed ?rank_profile - rank_profile)
    (rank_profile_endorsed ?rank_profile - rank_profile)
    (conditional_key_available ?conditional_key - conditional_key)
    (rank_profile_attached_key ?rank_profile - rank_profile ?conditional_key - conditional_key)
    (certification_available ?certification - certification)
    (rank_profile_has_certification ?rank_profile - rank_profile ?certification - certification)
    (special_objective_available ?special_objective - special_objective)
    (rank_profile_has_special_objective ?rank_profile - rank_profile ?special_objective - special_objective)
    (endorsement_available ?endorsement - endorsement)
    (rank_profile_has_endorsement ?rank_profile - rank_profile ?endorsement - endorsement)
    (badge_available ?badge - badge)
    (progression_entity_has_badge ?rank_node - rank_node ?badge - badge)
    (progression_slot_a_ready ?progression_slot_a - progression_slot_a)
    (progression_slot_b_ready ?progression_slot_b - progression_slot_b)
    (rank_profile_finalized ?rank_profile - rank_profile)
  )
  (:action register_progression_entity
    :parameters (?rank_node - rank_node)
    :precondition
      (and
        (not
          (progression_entity_registered ?rank_node)
        )
        (not
          (progression_entity_unlocked ?rank_node)
        )
      )
    :effect (progression_entity_registered ?rank_node)
  )
  (:action assign_unlock_token_to_node
    :parameters (?rank_node - rank_node ?unlock_token - unlock_token)
    :precondition
      (and
        (progression_entity_registered ?rank_node)
        (not
          (progression_entity_has_token ?rank_node)
        )
        (unlock_token_available ?unlock_token)
      )
    :effect
      (and
        (progression_entity_has_token ?rank_node)
        (progression_entity_assigned_token ?rank_node ?unlock_token)
        (not
          (unlock_token_available ?unlock_token)
        )
      )
  )
  (:action attach_quest_to_node
    :parameters (?rank_node - rank_node ?quest - quest)
    :precondition
      (and
        (progression_entity_registered ?rank_node)
        (progression_entity_has_token ?rank_node)
        (quest_available ?quest)
      )
    :effect
      (and
        (progression_entity_has_quest ?rank_node ?quest)
        (not
          (quest_available ?quest)
        )
      )
  )
  (:action verify_node_with_quest
    :parameters (?rank_node - rank_node ?quest - quest)
    :precondition
      (and
        (progression_entity_registered ?rank_node)
        (progression_entity_has_token ?rank_node)
        (progression_entity_has_quest ?rank_node ?quest)
        (not
          (progression_entity_verified ?rank_node)
        )
      )
    :effect (progression_entity_verified ?rank_node)
  )
  (:action release_quest_from_node
    :parameters (?rank_node - rank_node ?quest - quest)
    :precondition
      (and
        (progression_entity_has_quest ?rank_node ?quest)
      )
    :effect
      (and
        (quest_available ?quest)
        (not
          (progression_entity_has_quest ?rank_node ?quest)
        )
      )
  )
  (:action assign_agent_to_node
    :parameters (?rank_node - rank_node ?agent - agent)
    :precondition
      (and
        (progression_entity_verified ?rank_node)
        (agent_available ?agent)
      )
    :effect
      (and
        (progression_entity_assigned_agent ?rank_node ?agent)
        (not
          (agent_available ?agent)
        )
      )
  )
  (:action release_agent_from_node
    :parameters (?rank_node - rank_node ?agent - agent)
    :precondition
      (and
        (progression_entity_assigned_agent ?rank_node ?agent)
      )
    :effect
      (and
        (agent_available ?agent)
        (not
          (progression_entity_assigned_agent ?rank_node ?agent)
        )
      )
  )
  (:action assign_special_objective_to_profile
    :parameters (?rank_profile - rank_profile ?special_objective - special_objective)
    :precondition
      (and
        (progression_entity_verified ?rank_profile)
        (special_objective_available ?special_objective)
      )
    :effect
      (and
        (rank_profile_has_special_objective ?rank_profile ?special_objective)
        (not
          (special_objective_available ?special_objective)
        )
      )
  )
  (:action release_special_objective_from_profile
    :parameters (?rank_profile - rank_profile ?special_objective - special_objective)
    :precondition
      (and
        (rank_profile_has_special_objective ?rank_profile ?special_objective)
      )
    :effect
      (and
        (special_objective_available ?special_objective)
        (not
          (rank_profile_has_special_objective ?rank_profile ?special_objective)
        )
      )
  )
  (:action assign_endorsement_to_profile
    :parameters (?rank_profile - rank_profile ?endorsement - endorsement)
    :precondition
      (and
        (progression_entity_verified ?rank_profile)
        (endorsement_available ?endorsement)
      )
    :effect
      (and
        (rank_profile_has_endorsement ?rank_profile ?endorsement)
        (not
          (endorsement_available ?endorsement)
        )
      )
  )
  (:action release_endorsement_from_profile
    :parameters (?rank_profile - rank_profile ?endorsement - endorsement)
    :precondition
      (and
        (rank_profile_has_endorsement ?rank_profile ?endorsement)
      )
    :effect
      (and
        (endorsement_available ?endorsement)
        (not
          (rank_profile_has_endorsement ?rank_profile ?endorsement)
        )
      )
  )
  (:action claim_milestone_type_a_for_slot
    :parameters (?progression_slot_a - progression_slot_a ?milestone_type_a - milestone_type_a ?quest - quest)
    :precondition
      (and
        (progression_entity_verified ?progression_slot_a)
        (progression_entity_has_quest ?progression_slot_a ?quest)
        (slot_a_has_milestone_a ?progression_slot_a ?milestone_type_a)
        (not
          (milestone_type_a_ready ?milestone_type_a)
        )
        (not
          (milestone_type_a_staged ?milestone_type_a)
        )
      )
    :effect (milestone_type_a_ready ?milestone_type_a)
  )
  (:action process_milestone_type_a_with_agent
    :parameters (?progression_slot_a - progression_slot_a ?milestone_type_a - milestone_type_a ?agent - agent)
    :precondition
      (and
        (progression_entity_verified ?progression_slot_a)
        (progression_entity_assigned_agent ?progression_slot_a ?agent)
        (slot_a_has_milestone_a ?progression_slot_a ?milestone_type_a)
        (milestone_type_a_ready ?milestone_type_a)
        (not
          (progression_slot_a_ready ?progression_slot_a)
        )
      )
    :effect
      (and
        (progression_slot_a_ready ?progression_slot_a)
        (slot_a_milestone_complete ?progression_slot_a)
      )
  )
  (:action apply_consumable_to_milestone_type_a
    :parameters (?progression_slot_a - progression_slot_a ?milestone_type_a - milestone_type_a ?consumable_item - consumable_item)
    :precondition
      (and
        (progression_entity_verified ?progression_slot_a)
        (slot_a_has_milestone_a ?progression_slot_a ?milestone_type_a)
        (consumable_available ?consumable_item)
        (not
          (progression_slot_a_ready ?progression_slot_a)
        )
      )
    :effect
      (and
        (milestone_type_a_staged ?milestone_type_a)
        (progression_slot_a_ready ?progression_slot_a)
        (slot_a_has_consumable ?progression_slot_a ?consumable_item)
        (not
          (consumable_available ?consumable_item)
        )
      )
  )
  (:action finalize_milestone_type_a_for_slot
    :parameters (?progression_slot_a - progression_slot_a ?milestone_type_a - milestone_type_a ?quest - quest ?consumable_item - consumable_item)
    :precondition
      (and
        (progression_entity_verified ?progression_slot_a)
        (progression_entity_has_quest ?progression_slot_a ?quest)
        (slot_a_has_milestone_a ?progression_slot_a ?milestone_type_a)
        (milestone_type_a_staged ?milestone_type_a)
        (slot_a_has_consumable ?progression_slot_a ?consumable_item)
        (not
          (slot_a_milestone_complete ?progression_slot_a)
        )
      )
    :effect
      (and
        (milestone_type_a_ready ?milestone_type_a)
        (slot_a_milestone_complete ?progression_slot_a)
        (consumable_available ?consumable_item)
        (not
          (slot_a_has_consumable ?progression_slot_a ?consumable_item)
        )
      )
  )
  (:action claim_milestone_type_b_for_slot
    :parameters (?progression_slot_b - progression_slot_b ?milestone_type_b - milestone_type_b ?quest - quest)
    :precondition
      (and
        (progression_entity_verified ?progression_slot_b)
        (progression_entity_has_quest ?progression_slot_b ?quest)
        (slot_b_has_milestone_b ?progression_slot_b ?milestone_type_b)
        (not
          (milestone_type_b_ready ?milestone_type_b)
        )
        (not
          (milestone_type_b_staged ?milestone_type_b)
        )
      )
    :effect (milestone_type_b_ready ?milestone_type_b)
  )
  (:action process_milestone_type_b_with_agent
    :parameters (?progression_slot_b - progression_slot_b ?milestone_type_b - milestone_type_b ?agent - agent)
    :precondition
      (and
        (progression_entity_verified ?progression_slot_b)
        (progression_entity_assigned_agent ?progression_slot_b ?agent)
        (slot_b_has_milestone_b ?progression_slot_b ?milestone_type_b)
        (milestone_type_b_ready ?milestone_type_b)
        (not
          (progression_slot_b_ready ?progression_slot_b)
        )
      )
    :effect
      (and
        (progression_slot_b_ready ?progression_slot_b)
        (slot_b_milestone_complete ?progression_slot_b)
      )
  )
  (:action apply_consumable_to_milestone_type_b
    :parameters (?progression_slot_b - progression_slot_b ?milestone_type_b - milestone_type_b ?consumable_item - consumable_item)
    :precondition
      (and
        (progression_entity_verified ?progression_slot_b)
        (slot_b_has_milestone_b ?progression_slot_b ?milestone_type_b)
        (consumable_available ?consumable_item)
        (not
          (progression_slot_b_ready ?progression_slot_b)
        )
      )
    :effect
      (and
        (milestone_type_b_staged ?milestone_type_b)
        (progression_slot_b_ready ?progression_slot_b)
        (slot_b_has_consumable ?progression_slot_b ?consumable_item)
        (not
          (consumable_available ?consumable_item)
        )
      )
  )
  (:action finalize_milestone_type_b_for_slot
    :parameters (?progression_slot_b - progression_slot_b ?milestone_type_b - milestone_type_b ?quest - quest ?consumable_item - consumable_item)
    :precondition
      (and
        (progression_entity_verified ?progression_slot_b)
        (progression_entity_has_quest ?progression_slot_b ?quest)
        (slot_b_has_milestone_b ?progression_slot_b ?milestone_type_b)
        (milestone_type_b_staged ?milestone_type_b)
        (slot_b_has_consumable ?progression_slot_b ?consumable_item)
        (not
          (slot_b_milestone_complete ?progression_slot_b)
        )
      )
    :effect
      (and
        (milestone_type_b_ready ?milestone_type_b)
        (slot_b_milestone_complete ?progression_slot_b)
        (consumable_available ?consumable_item)
        (not
          (slot_b_has_consumable ?progression_slot_b ?consumable_item)
        )
      )
  )
  (:action assemble_rank_seal_standard
    :parameters (?progression_slot_a - progression_slot_a ?progression_slot_b - progression_slot_b ?milestone_type_a - milestone_type_a ?milestone_type_b - milestone_type_b ?rank_seal - rank_seal)
    :precondition
      (and
        (progression_slot_a_ready ?progression_slot_a)
        (progression_slot_b_ready ?progression_slot_b)
        (slot_a_has_milestone_a ?progression_slot_a ?milestone_type_a)
        (slot_b_has_milestone_b ?progression_slot_b ?milestone_type_b)
        (milestone_type_a_ready ?milestone_type_a)
        (milestone_type_b_ready ?milestone_type_b)
        (slot_a_milestone_complete ?progression_slot_a)
        (slot_b_milestone_complete ?progression_slot_b)
        (rank_seal_available ?rank_seal)
      )
    :effect
      (and
        (rank_seal_created ?rank_seal)
        (rank_seal_includes_milestone_a ?rank_seal ?milestone_type_a)
        (rank_seal_includes_milestone_b ?rank_seal ?milestone_type_b)
        (not
          (rank_seal_available ?rank_seal)
        )
      )
  )
  (:action assemble_rank_seal_with_branch_a
    :parameters (?progression_slot_a - progression_slot_a ?progression_slot_b - progression_slot_b ?milestone_type_a - milestone_type_a ?milestone_type_b - milestone_type_b ?rank_seal - rank_seal)
    :precondition
      (and
        (progression_slot_a_ready ?progression_slot_a)
        (progression_slot_b_ready ?progression_slot_b)
        (slot_a_has_milestone_a ?progression_slot_a ?milestone_type_a)
        (slot_b_has_milestone_b ?progression_slot_b ?milestone_type_b)
        (milestone_type_a_staged ?milestone_type_a)
        (milestone_type_b_ready ?milestone_type_b)
        (not
          (slot_a_milestone_complete ?progression_slot_a)
        )
        (slot_b_milestone_complete ?progression_slot_b)
        (rank_seal_available ?rank_seal)
      )
    :effect
      (and
        (rank_seal_created ?rank_seal)
        (rank_seal_includes_milestone_a ?rank_seal ?milestone_type_a)
        (rank_seal_includes_milestone_b ?rank_seal ?milestone_type_b)
        (rank_seal_includes_branch_a ?rank_seal)
        (not
          (rank_seal_available ?rank_seal)
        )
      )
  )
  (:action assemble_rank_seal_with_branch_b
    :parameters (?progression_slot_a - progression_slot_a ?progression_slot_b - progression_slot_b ?milestone_type_a - milestone_type_a ?milestone_type_b - milestone_type_b ?rank_seal - rank_seal)
    :precondition
      (and
        (progression_slot_a_ready ?progression_slot_a)
        (progression_slot_b_ready ?progression_slot_b)
        (slot_a_has_milestone_a ?progression_slot_a ?milestone_type_a)
        (slot_b_has_milestone_b ?progression_slot_b ?milestone_type_b)
        (milestone_type_a_ready ?milestone_type_a)
        (milestone_type_b_staged ?milestone_type_b)
        (slot_a_milestone_complete ?progression_slot_a)
        (not
          (slot_b_milestone_complete ?progression_slot_b)
        )
        (rank_seal_available ?rank_seal)
      )
    :effect
      (and
        (rank_seal_created ?rank_seal)
        (rank_seal_includes_milestone_a ?rank_seal ?milestone_type_a)
        (rank_seal_includes_milestone_b ?rank_seal ?milestone_type_b)
        (rank_seal_includes_branch_b ?rank_seal)
        (not
          (rank_seal_available ?rank_seal)
        )
      )
  )
  (:action assemble_rank_seal_with_both_branches
    :parameters (?progression_slot_a - progression_slot_a ?progression_slot_b - progression_slot_b ?milestone_type_a - milestone_type_a ?milestone_type_b - milestone_type_b ?rank_seal - rank_seal)
    :precondition
      (and
        (progression_slot_a_ready ?progression_slot_a)
        (progression_slot_b_ready ?progression_slot_b)
        (slot_a_has_milestone_a ?progression_slot_a ?milestone_type_a)
        (slot_b_has_milestone_b ?progression_slot_b ?milestone_type_b)
        (milestone_type_a_staged ?milestone_type_a)
        (milestone_type_b_staged ?milestone_type_b)
        (not
          (slot_a_milestone_complete ?progression_slot_a)
        )
        (not
          (slot_b_milestone_complete ?progression_slot_b)
        )
        (rank_seal_available ?rank_seal)
      )
    :effect
      (and
        (rank_seal_created ?rank_seal)
        (rank_seal_includes_milestone_a ?rank_seal ?milestone_type_a)
        (rank_seal_includes_milestone_b ?rank_seal ?milestone_type_b)
        (rank_seal_includes_branch_a ?rank_seal)
        (rank_seal_includes_branch_b ?rank_seal)
        (not
          (rank_seal_available ?rank_seal)
        )
      )
  )
  (:action validate_rank_seal_with_slot_a
    :parameters (?rank_seal - rank_seal ?progression_slot_a - progression_slot_a ?quest - quest)
    :precondition
      (and
        (rank_seal_created ?rank_seal)
        (progression_slot_a_ready ?progression_slot_a)
        (progression_entity_has_quest ?progression_slot_a ?quest)
        (not
          (rank_seal_validated ?rank_seal)
        )
      )
    :effect (rank_seal_validated ?rank_seal)
  )
  (:action consume_resource_pack_attach_to_seal
    :parameters (?rank_profile - rank_profile ?resource_pack - resource_pack ?rank_seal - rank_seal)
    :precondition
      (and
        (progression_entity_verified ?rank_profile)
        (rank_profile_has_seal ?rank_profile ?rank_seal)
        (rank_profile_has_resource_pack ?rank_profile ?resource_pack)
        (resource_pack_available ?resource_pack)
        (rank_seal_created ?rank_seal)
        (rank_seal_validated ?rank_seal)
        (not
          (resource_pack_consumed ?resource_pack)
        )
      )
    :effect
      (and
        (resource_pack_consumed ?resource_pack)
        (resource_pack_bound_to_seal ?resource_pack ?rank_seal)
        (not
          (resource_pack_available ?resource_pack)
        )
      )
  )
  (:action finalize_resource_pack_processing_for_profile
    :parameters (?rank_profile - rank_profile ?resource_pack - resource_pack ?rank_seal - rank_seal ?quest - quest)
    :precondition
      (and
        (progression_entity_verified ?rank_profile)
        (rank_profile_has_resource_pack ?rank_profile ?resource_pack)
        (resource_pack_consumed ?resource_pack)
        (resource_pack_bound_to_seal ?resource_pack ?rank_seal)
        (progression_entity_has_quest ?rank_profile ?quest)
        (not
          (rank_seal_includes_branch_a ?rank_seal)
        )
        (not
          (rank_profile_component_verified ?rank_profile)
        )
      )
    :effect (rank_profile_component_verified ?rank_profile)
  )
  (:action attach_conditional_key_to_profile
    :parameters (?rank_profile - rank_profile ?conditional_key - conditional_key)
    :precondition
      (and
        (progression_entity_verified ?rank_profile)
        (conditional_key_available ?conditional_key)
        (not
          (rank_profile_has_key ?rank_profile)
        )
      )
    :effect
      (and
        (rank_profile_has_key ?rank_profile)
        (rank_profile_attached_key ?rank_profile ?conditional_key)
        (not
          (conditional_key_available ?conditional_key)
        )
      )
  )
  (:action process_key_and_resources_for_profile
    :parameters (?rank_profile - rank_profile ?resource_pack - resource_pack ?rank_seal - rank_seal ?quest - quest ?conditional_key - conditional_key)
    :precondition
      (and
        (progression_entity_verified ?rank_profile)
        (rank_profile_has_resource_pack ?rank_profile ?resource_pack)
