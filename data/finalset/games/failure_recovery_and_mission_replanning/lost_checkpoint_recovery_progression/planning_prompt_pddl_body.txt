## Planning domain (PDDL)

(define (domain lost_checkpoint_recovery)
  (:requirements :strips :typing :negative-preconditions)
  (:types meta_actor_group - object meta_item_group - object meta_asset_group - object entity_root - object participant - entity_root backup_resource - meta_actor_group loadout_option - meta_actor_group support_unit - meta_actor_group tactical_attachment - meta_actor_group strategic_modifier - meta_actor_group failure_marker - meta_actor_group equipment_module - meta_actor_group mission_modifier - meta_actor_group consumable_resource - meta_item_group progress_fragment - meta_item_group special_asset - meta_item_group route_segment - meta_asset_group objective_node - meta_asset_group checkpoint_bundle - meta_asset_group role_group - participant coordinator_group - participant primary_role - role_group secondary_role - role_group recovery_coordinator - coordinator_group)
  (:predicates
    (lost_checkpoint_detected ?participant - participant)
    (recovery_committed ?participant - participant)
    (backup_claimed ?participant - participant)
    (checkpoint_restored ?participant - participant)
    (role_ready ?participant - participant)
    (failure_marker_applied ?participant - participant)
    (backup_available ?backup_resource - backup_resource)
    (backup_assigned ?participant - participant ?backup_resource - backup_resource)
    (loadout_available ?loadout_option - loadout_option)
    (loadout_assigned ?participant - participant ?loadout_option - loadout_option)
    (support_available ?support_unit - support_unit)
    (support_attached ?participant - participant ?support_unit - support_unit)
    (consumable_available ?consumable_resource - consumable_resource)
    (primary_assigned_consumable ?primary_role - primary_role ?consumable_resource - consumable_resource)
    (secondary_assigned_consumable ?secondary_role - secondary_role ?consumable_resource - consumable_resource)
    (role_has_route_segment ?primary_role - primary_role ?route_segment - route_segment)
    (route_segment_probed ?route_segment - route_segment)
    (route_segment_tagged ?route_segment - route_segment)
    (primary_route_validated ?primary_role - primary_role)
    (role_has_objective_node ?secondary_role - secondary_role ?objective_node - objective_node)
    (objective_node_probed ?objective_node - objective_node)
    (objective_node_tagged ?objective_node - objective_node)
    (secondary_route_validated ?secondary_role - secondary_role)
    (bundle_available ?checkpoint_bundle - checkpoint_bundle)
    (bundle_initialized ?checkpoint_bundle - checkpoint_bundle)
    (bundle_includes_segment ?checkpoint_bundle - checkpoint_bundle ?route_segment - route_segment)
    (bundle_includes_objective ?checkpoint_bundle - checkpoint_bundle ?objective_node - objective_node)
    (bundle_flag_route_tagged ?checkpoint_bundle - checkpoint_bundle)
    (bundle_flag_objective_tagged ?checkpoint_bundle - checkpoint_bundle)
    (bundle_validated ?checkpoint_bundle - checkpoint_bundle)
    (coordinator_has_primary ?coordinator - recovery_coordinator ?primary_role - primary_role)
    (coordinator_has_secondary ?coordinator - recovery_coordinator ?secondary_role - secondary_role)
    (coordinator_has_bundle ?coordinator - recovery_coordinator ?checkpoint_bundle - checkpoint_bundle)
    (fragment_available ?progress_fragment - progress_fragment)
    (coordinator_has_fragment ?coordinator - recovery_coordinator ?progress_fragment - progress_fragment)
    (fragment_consumed ?progress_fragment - progress_fragment)
    (fragment_in_bundle ?progress_fragment - progress_fragment ?checkpoint_bundle - checkpoint_bundle)
    (coordinator_module_attached ?coordinator - recovery_coordinator)
    (coordinator_module_ready ?coordinator - recovery_coordinator)
    (modifier_sequence_started ?coordinator - recovery_coordinator)
    (attachment_engaged ?coordinator - recovery_coordinator)
    (attachment_confirmed ?coordinator - recovery_coordinator)
    (modifier_applied ?coordinator - recovery_coordinator)
    (coordinator_finalized ?coordinator - recovery_coordinator)
    (special_asset_available ?special_asset - special_asset)
    (coordinator_has_asset ?coordinator - recovery_coordinator ?special_asset - special_asset)
    (special_asset_locked ?coordinator - recovery_coordinator)
    (special_asset_phase1_engaged ?coordinator - recovery_coordinator)
    (special_asset_phase2_engaged ?coordinator - recovery_coordinator)
    (tactical_attachment_available ?tactical_attachment - tactical_attachment)
    (coordinator_has_attachment ?coordinator - recovery_coordinator ?tactical_attachment - tactical_attachment)
    (strategic_modifier_available ?strategic_modifier - strategic_modifier)
    (coordinator_has_strategic_modifier ?coordinator - recovery_coordinator ?strategic_modifier - strategic_modifier)
    (equipment_module_available ?equipment_module - equipment_module)
    (coordinator_has_equipment ?coordinator - recovery_coordinator ?equipment_module - equipment_module)
    (mission_modifier_available ?mission_modifier - mission_modifier)
    (coordinator_has_mission_modifier ?coordinator - recovery_coordinator ?mission_modifier - mission_modifier)
    (failure_marker_available ?failure_marker - failure_marker)
    (participant_failure_marker ?participant - participant ?failure_marker - failure_marker)
    (primary_probe_flag ?primary_role - primary_role)
    (secondary_probe_flag ?secondary_role - secondary_role)
    (coordinator_ready ?coordinator - recovery_coordinator)
  )
  (:action detect_lost_checkpoint
    :parameters (?participant - participant)
    :precondition
      (and
        (not
          (lost_checkpoint_detected ?participant)
        )
        (not
          (checkpoint_restored ?participant)
        )
      )
    :effect (lost_checkpoint_detected ?participant)
  )
  (:action assign_backup_resource
    :parameters (?participant - participant ?backup_resource - backup_resource)
    :precondition
      (and
        (lost_checkpoint_detected ?participant)
        (not
          (backup_claimed ?participant)
        )
        (backup_available ?backup_resource)
      )
    :effect
      (and
        (backup_claimed ?participant)
        (backup_assigned ?participant ?backup_resource)
        (not
          (backup_available ?backup_resource)
        )
      )
  )
  (:action propose_loadout_for_recovery
    :parameters (?participant - participant ?loadout_option - loadout_option)
    :precondition
      (and
        (lost_checkpoint_detected ?participant)
        (backup_claimed ?participant)
        (loadout_available ?loadout_option)
      )
    :effect
      (and
        (loadout_assigned ?participant ?loadout_option)
        (not
          (loadout_available ?loadout_option)
        )
      )
  )
  (:action commit_loadout_selection
    :parameters (?participant - participant ?loadout_option - loadout_option)
    :precondition
      (and
        (lost_checkpoint_detected ?participant)
        (backup_claimed ?participant)
        (loadout_assigned ?participant ?loadout_option)
        (not
          (recovery_committed ?participant)
        )
      )
    :effect (recovery_committed ?participant)
  )
  (:action release_loadout_option
    :parameters (?participant - participant ?loadout_option - loadout_option)
    :precondition
      (and
        (loadout_assigned ?participant ?loadout_option)
      )
    :effect
      (and
        (loadout_available ?loadout_option)
        (not
          (loadout_assigned ?participant ?loadout_option)
        )
      )
  )
  (:action attach_support_unit
    :parameters (?participant - participant ?support_unit - support_unit)
    :precondition
      (and
        (recovery_committed ?participant)
        (support_available ?support_unit)
      )
    :effect
      (and
        (support_attached ?participant ?support_unit)
        (not
          (support_available ?support_unit)
        )
      )
  )
  (:action detach_support_unit
    :parameters (?participant - participant ?support_unit - support_unit)
    :precondition
      (and
        (support_attached ?participant ?support_unit)
      )
    :effect
      (and
        (support_available ?support_unit)
        (not
          (support_attached ?participant ?support_unit)
        )
      )
  )
  (:action attach_equipment_module
    :parameters (?coordinator - recovery_coordinator ?equipment_module - equipment_module)
    :precondition
      (and
        (recovery_committed ?coordinator)
        (equipment_module_available ?equipment_module)
      )
    :effect
      (and
        (coordinator_has_equipment ?coordinator ?equipment_module)
        (not
          (equipment_module_available ?equipment_module)
        )
      )
  )
  (:action detach_equipment_module
    :parameters (?coordinator - recovery_coordinator ?equipment_module - equipment_module)
    :precondition
      (and
        (coordinator_has_equipment ?coordinator ?equipment_module)
      )
    :effect
      (and
        (equipment_module_available ?equipment_module)
        (not
          (coordinator_has_equipment ?coordinator ?equipment_module)
        )
      )
  )
  (:action attach_mission_modifier
    :parameters (?coordinator - recovery_coordinator ?mission_modifier - mission_modifier)
    :precondition
      (and
        (recovery_committed ?coordinator)
        (mission_modifier_available ?mission_modifier)
      )
    :effect
      (and
        (coordinator_has_mission_modifier ?coordinator ?mission_modifier)
        (not
          (mission_modifier_available ?mission_modifier)
        )
      )
  )
  (:action detach_mission_modifier
    :parameters (?coordinator - recovery_coordinator ?mission_modifier - mission_modifier)
    :precondition
      (and
        (coordinator_has_mission_modifier ?coordinator ?mission_modifier)
      )
    :effect
      (and
        (mission_modifier_available ?mission_modifier)
        (not
          (coordinator_has_mission_modifier ?coordinator ?mission_modifier)
        )
      )
  )
  (:action probe_route_segment_primary
    :parameters (?primary_role - primary_role ?route_segment - route_segment ?loadout_option - loadout_option)
    :precondition
      (and
        (recovery_committed ?primary_role)
        (loadout_assigned ?primary_role ?loadout_option)
        (role_has_route_segment ?primary_role ?route_segment)
        (not
          (route_segment_probed ?route_segment)
        )
        (not
          (route_segment_tagged ?route_segment)
        )
      )
    :effect (route_segment_probed ?route_segment)
  )
  (:action validate_primary_segment_traversal
    :parameters (?primary_role - primary_role ?route_segment - route_segment ?support_unit - support_unit)
    :precondition
      (and
        (recovery_committed ?primary_role)
        (support_attached ?primary_role ?support_unit)
        (role_has_route_segment ?primary_role ?route_segment)
        (route_segment_probed ?route_segment)
        (not
          (primary_probe_flag ?primary_role)
        )
      )
    :effect
      (and
        (primary_probe_flag ?primary_role)
        (primary_route_validated ?primary_role)
      )
  )
  (:action apply_consumable_to_segment_primary
    :parameters (?primary_role - primary_role ?route_segment - route_segment ?consumable_resource - consumable_resource)
    :precondition
      (and
        (recovery_committed ?primary_role)
        (role_has_route_segment ?primary_role ?route_segment)
        (consumable_available ?consumable_resource)
        (not
          (primary_probe_flag ?primary_role)
        )
      )
    :effect
      (and
        (route_segment_tagged ?route_segment)
        (primary_probe_flag ?primary_role)
        (primary_assigned_consumable ?primary_role ?consumable_resource)
        (not
          (consumable_available ?consumable_resource)
        )
      )
  )
  (:action finalize_primary_segment_traversal
    :parameters (?primary_role - primary_role ?route_segment - route_segment ?loadout_option - loadout_option ?consumable_resource - consumable_resource)
    :precondition
      (and
        (recovery_committed ?primary_role)
        (loadout_assigned ?primary_role ?loadout_option)
        (role_has_route_segment ?primary_role ?route_segment)
        (route_segment_tagged ?route_segment)
        (primary_assigned_consumable ?primary_role ?consumable_resource)
        (not
          (primary_route_validated ?primary_role)
        )
      )
    :effect
      (and
        (route_segment_probed ?route_segment)
        (primary_route_validated ?primary_role)
        (consumable_available ?consumable_resource)
        (not
          (primary_assigned_consumable ?primary_role ?consumable_resource)
        )
      )
  )
  (:action probe_route_segment_secondary
    :parameters (?secondary_role - secondary_role ?objective_node - objective_node ?loadout_option - loadout_option)
    :precondition
      (and
        (recovery_committed ?secondary_role)
        (loadout_assigned ?secondary_role ?loadout_option)
        (role_has_objective_node ?secondary_role ?objective_node)
        (not
          (objective_node_probed ?objective_node)
        )
        (not
          (objective_node_tagged ?objective_node)
        )
      )
    :effect (objective_node_probed ?objective_node)
  )
  (:action validate_secondary_segment_traversal
    :parameters (?secondary_role - secondary_role ?objective_node - objective_node ?support_unit - support_unit)
    :precondition
      (and
        (recovery_committed ?secondary_role)
        (support_attached ?secondary_role ?support_unit)
        (role_has_objective_node ?secondary_role ?objective_node)
        (objective_node_probed ?objective_node)
        (not
          (secondary_probe_flag ?secondary_role)
        )
      )
    :effect
      (and
        (secondary_probe_flag ?secondary_role)
        (secondary_route_validated ?secondary_role)
      )
  )
  (:action apply_consumable_to_segment_secondary
    :parameters (?secondary_role - secondary_role ?objective_node - objective_node ?consumable_resource - consumable_resource)
    :precondition
      (and
        (recovery_committed ?secondary_role)
        (role_has_objective_node ?secondary_role ?objective_node)
        (consumable_available ?consumable_resource)
        (not
          (secondary_probe_flag ?secondary_role)
        )
      )
    :effect
      (and
        (objective_node_tagged ?objective_node)
        (secondary_probe_flag ?secondary_role)
        (secondary_assigned_consumable ?secondary_role ?consumable_resource)
        (not
          (consumable_available ?consumable_resource)
        )
      )
  )
  (:action finalize_secondary_segment_traversal
    :parameters (?secondary_role - secondary_role ?objective_node - objective_node ?loadout_option - loadout_option ?consumable_resource - consumable_resource)
    :precondition
      (and
        (recovery_committed ?secondary_role)
        (loadout_assigned ?secondary_role ?loadout_option)
        (role_has_objective_node ?secondary_role ?objective_node)
        (objective_node_tagged ?objective_node)
        (secondary_assigned_consumable ?secondary_role ?consumable_resource)
        (not
          (secondary_route_validated ?secondary_role)
        )
      )
    :effect
      (and
        (objective_node_probed ?objective_node)
        (secondary_route_validated ?secondary_role)
        (consumable_available ?consumable_resource)
        (not
          (secondary_assigned_consumable ?secondary_role ?consumable_resource)
        )
      )
  )
  (:action assemble_checkpoint_bundle_standard
    :parameters (?primary_role - primary_role ?secondary_role - secondary_role ?route_segment - route_segment ?objective_node - objective_node ?checkpoint_bundle - checkpoint_bundle)
    :precondition
      (and
        (primary_probe_flag ?primary_role)
        (secondary_probe_flag ?secondary_role)
        (role_has_route_segment ?primary_role ?route_segment)
        (role_has_objective_node ?secondary_role ?objective_node)
        (route_segment_probed ?route_segment)
        (objective_node_probed ?objective_node)
        (primary_route_validated ?primary_role)
        (secondary_route_validated ?secondary_role)
        (bundle_available ?checkpoint_bundle)
      )
    :effect
      (and
        (bundle_initialized ?checkpoint_bundle)
        (bundle_includes_segment ?checkpoint_bundle ?route_segment)
        (bundle_includes_objective ?checkpoint_bundle ?objective_node)
        (not
          (bundle_available ?checkpoint_bundle)
        )
      )
  )
  (:action assemble_checkpoint_bundle_with_route_tag
    :parameters (?primary_role - primary_role ?secondary_role - secondary_role ?route_segment - route_segment ?objective_node - objective_node ?checkpoint_bundle - checkpoint_bundle)
    :precondition
      (and
        (primary_probe_flag ?primary_role)
        (secondary_probe_flag ?secondary_role)
        (role_has_route_segment ?primary_role ?route_segment)
        (role_has_objective_node ?secondary_role ?objective_node)
        (route_segment_tagged ?route_segment)
        (objective_node_probed ?objective_node)
        (not
          (primary_route_validated ?primary_role)
        )
        (secondary_route_validated ?secondary_role)
        (bundle_available ?checkpoint_bundle)
      )
    :effect
      (and
        (bundle_initialized ?checkpoint_bundle)
        (bundle_includes_segment ?checkpoint_bundle ?route_segment)
        (bundle_includes_objective ?checkpoint_bundle ?objective_node)
        (bundle_flag_route_tagged ?checkpoint_bundle)
        (not
          (bundle_available ?checkpoint_bundle)
        )
      )
  )
  (:action assemble_checkpoint_bundle_with_objective_tag
    :parameters (?primary_role - primary_role ?secondary_role - secondary_role ?route_segment - route_segment ?objective_node - objective_node ?checkpoint_bundle - checkpoint_bundle)
    :precondition
      (and
        (primary_probe_flag ?primary_role)
        (secondary_probe_flag ?secondary_role)
        (role_has_route_segment ?primary_role ?route_segment)
        (role_has_objective_node ?secondary_role ?objective_node)
        (route_segment_probed ?route_segment)
        (objective_node_tagged ?objective_node)
        (primary_route_validated ?primary_role)
        (not
          (secondary_route_validated ?secondary_role)
        )
        (bundle_available ?checkpoint_bundle)
      )
    :effect
      (and
        (bundle_initialized ?checkpoint_bundle)
        (bundle_includes_segment ?checkpoint_bundle ?route_segment)
        (bundle_includes_objective ?checkpoint_bundle ?objective_node)
        (bundle_flag_objective_tagged ?checkpoint_bundle)
        (not
          (bundle_available ?checkpoint_bundle)
        )
      )
  )
  (:action assemble_checkpoint_bundle_with_route_and_objective_tags
    :parameters (?primary_role - primary_role ?secondary_role - secondary_role ?route_segment - route_segment ?objective_node - objective_node ?checkpoint_bundle - checkpoint_bundle)
    :precondition
      (and
        (primary_probe_flag ?primary_role)
        (secondary_probe_flag ?secondary_role)
        (role_has_route_segment ?primary_role ?route_segment)
        (role_has_objective_node ?secondary_role ?objective_node)
        (route_segment_tagged ?route_segment)
        (objective_node_tagged ?objective_node)
        (not
          (primary_route_validated ?primary_role)
        )
        (not
          (secondary_route_validated ?secondary_role)
        )
        (bundle_available ?checkpoint_bundle)
      )
    :effect
      (and
        (bundle_initialized ?checkpoint_bundle)
        (bundle_includes_segment ?checkpoint_bundle ?route_segment)
        (bundle_includes_objective ?checkpoint_bundle ?objective_node)
        (bundle_flag_route_tagged ?checkpoint_bundle)
        (bundle_flag_objective_tagged ?checkpoint_bundle)
        (not
          (bundle_available ?checkpoint_bundle)
        )
      )
  )
  (:action validate_checkpoint_bundle
    :parameters (?checkpoint_bundle - checkpoint_bundle ?primary_role - primary_role ?loadout_option - loadout_option)
    :precondition
      (and
        (bundle_initialized ?checkpoint_bundle)
        (primary_probe_flag ?primary_role)
        (loadout_assigned ?primary_role ?loadout_option)
        (not
          (bundle_validated ?checkpoint_bundle)
        )
      )
    :effect (bundle_validated ?checkpoint_bundle)
  )
  (:action integrate_progress_fragment
    :parameters (?coordinator - recovery_coordinator ?progress_fragment - progress_fragment ?checkpoint_bundle - checkpoint_bundle)
    :precondition
      (and
        (recovery_committed ?coordinator)
        (coordinator_has_bundle ?coordinator ?checkpoint_bundle)
        (coordinator_has_fragment ?coordinator ?progress_fragment)
        (fragment_available ?progress_fragment)
        (bundle_initialized ?checkpoint_bundle)
        (bundle_validated ?checkpoint_bundle)
        (not
          (fragment_consumed ?progress_fragment)
        )
      )
    :effect
      (and
        (fragment_consumed ?progress_fragment)
        (fragment_in_bundle ?progress_fragment ?checkpoint_bundle)
        (not
          (fragment_available ?progress_fragment)
        )
      )
  )
  (:action finalize_fragment_integration
    :parameters (?coordinator - recovery_coordinator ?progress_fragment - progress_fragment ?checkpoint_bundle - checkpoint_bundle ?loadout_option - loadout_option)
    :precondition
      (and
        (recovery_committed ?coordinator)
        (coordinator_has_fragment ?coordinator ?progress_fragment)
        (fragment_consumed ?progress_fragment)
        (fragment_in_bundle ?progress_fragment ?checkpoint_bundle)
        (loadout_assigned ?coordinator ?loadout_option)
        (not
          (bundle_flag_route_tagged ?checkpoint_bundle)
        )
        (not
          (coordinator_module_attached ?coordinator)
        )
      )
    :effect (coordinator_module_attached ?coordinator)
  )
  (:action attach_tactical_attachment
    :parameters (?coordinator - recovery_coordinator ?tactical_attachment - tactical_attachment)
    :precondition
      (and
        (recovery_committed ?coordinator)
        (tactical_attachment_available ?tactical_attachment)
        (not
          (attachment_engaged ?coordinator)
        )
      )
    :effect
      (and
        (attachment_engaged ?coordinator)
        (coordinator_has_attachment ?coordinator ?tactical_attachment)
        (not
          (tactical_attachment_available ?tactical_attachment)
        )
      )
  )
  (:action integrate_attachment_and_prepare_coordinator
    :parameters (?coordinator - recovery_coordinator ?progress_fragment - progress_fragment ?checkpoint_bundle - checkpoint_bundle ?loadout_option - loadout_option ?tactical_attachment - tactical_attachment)
    :precondition
      (and
        (recovery_committed ?coordinator)
        (coordinator_has_fragment ?coordinator ?progress_fragment)
        (fragment_consumed ?progress_fragment)
        (fragment_in_bundle ?progress_fragment ?checkpoint_bundle)
        (loadout_assigned ?coordinator ?loadout_option)
        (bundle_flag_route_tagged ?checkpoint_bundle)
        (attachment_engaged ?coordinator)
        (coordinator_has_attachment ?coordinator ?tactical_attachment)
        (not
          (coordinator_module_attached ?coordinator)
        )
      )
    :effect
      (and
        (coordinator_module_attached ?coordinator)
        (attachment_confirmed ?coordinator)
      )
  )
  (:action apply_equipment_module_phase1
    :parameters (?coordinator - recovery_coordinator ?equipment_module - equipment_module ?support_unit - support_unit ?progress_fragment - progress_fragment ?checkpoint_bundle - checkpoint_bundle)
    :precondition
      (and
        (coordinator_module_attached ?coordinator)
        (coordinator_has_equipment ?coordinator ?equipment_module)
        (support_attached ?coordinator ?support_unit)
        (coordinator_has_fragment ?coordinator ?progress_fragment)
        (fragment_in_bundle ?progress_fragment ?checkpoint_bundle)
        (not
          (bundle_flag_objective_tagged ?checkpoint_bundle)
        )
        (not
          (coordinator_module_ready ?coordinator)
        )
      )
    :effect (coordinator_module_ready ?coordinator)
  )
  (:action apply_equipment_module_phase2
    :parameters (?coordinator - recovery_coordinator ?equipment_module - equipment_module ?support_unit - support_unit ?progress_fragment - progress_fragment ?checkpoint_bundle - checkpoint_bundle)
    :precondition
      (and
        (coordinator_module_attached ?coordinator)
        (coordinator_has_equipment ?coordinator ?equipment_module)
        (support_attached ?coordinator ?support_unit)
        (coordinator_has_fragment ?coordinator ?progress_fragment)
        (fragment_in_bundle ?progress_fragment ?checkpoint_bundle)
        (bundle_flag_objective_tagged ?checkpoint_bundle)
        (not
          (coordinator_module_ready ?coordinator)
        )
      )
    :effect (coordinator_module_ready ?coordinator)
  )
  (:action sequence_mission_modifier_step1
    :parameters (?coordinator - recovery_coordinator ?mission_modifier - mission_modifier ?progress_fragment - progress_fragment ?checkpoint_bundle - checkpoint_bundle)
    :precondition
      (and
        (coordinator_module_ready ?coordinator)
        (coordinator_has_mission_modifier ?coordinator ?mission_modifier)
        (coordinator_has_fragment ?coordinator ?progress_fragment)
        (fragment_in_bundle ?progress_fragment ?checkpoint_bundle)
        (not
          (bundle_flag_route_tagged ?checkpoint_bundle)
        )
        (not
          (bundle_flag_objective_tagged ?checkpoint_bundle)
        )
        (not
          (modifier_sequence_started ?coordinator)
        )
      )
    :effect (modifier_sequence_started ?coordinator)
  )
  (:action sequence_mission_modifier_step2
    :parameters (?coordinator - recovery_coordinator ?mission_modifier - mission_modifier ?progress_fragment - progress_fragment ?checkpoint_bundle - checkpoint_bundle)
    :precondition
      (and
        (coordinator_module_ready ?coordinator)
        (coordinator_has_mission_modifier ?coordinator ?mission_modifier)
        (coordinator_has_fragment ?coordinator ?progress_fragment)
        (fragment_in_bundle ?progress_fragment ?checkpoint_bundle)
        (bundle_flag_route_tagged ?checkpoint_bundle)
        (not
          (bundle_flag_objective_tagged ?checkpoint_bundle)
        )
        (not
          (modifier_sequence_started ?coordinator)
