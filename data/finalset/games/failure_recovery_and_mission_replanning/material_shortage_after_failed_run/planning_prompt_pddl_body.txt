## Planning domain (PDDL)

(define (domain material_shortage_recovery_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types asset_category - object cache_component_category - object category_support - object mission_container - object recoverable_unit - mission_container supply_pack - asset_category route_option - asset_category support_tool - asset_category optional_asset_a - asset_category consumable_type - asset_category resource_token - asset_category module_variant - asset_category key_component - asset_category material_unit - cache_component_category upgrade_module - cache_component_category blueprint_asset - cache_component_category scout_report - category_support support_report - category_support supply_cache - category_support member_slot_group_a - recoverable_unit member_slot_group_b - recoverable_unit scavenger_slot - member_slot_group_a responder_slot - member_slot_group_a squad_unit - member_slot_group_b)
  (:predicates
    (recoverable_marked_for_recovery ?mission_run - recoverable_unit)
    (ready_for_recovery ?mission_run - recoverable_unit)
    (recoverable_has_supply_pack ?mission_run - recoverable_unit)
    (ready_for_redeploy ?mission_run - recoverable_unit)
    (ready_for_restart ?mission_run - recoverable_unit)
    (recoverable_reenabled ?mission_run - recoverable_unit)
    (supply_pack_available ?supply_pack - supply_pack)
    (assigned_supply_pack ?mission_run - recoverable_unit ?supply_pack - supply_pack)
    (route_option_available ?route_option - route_option)
    (claimed_route ?mission_run - recoverable_unit ?route_option - route_option)
    (support_tool_available ?support_tool - support_tool)
    (attached_support_tool ?mission_run - recoverable_unit ?support_tool - support_tool)
    (material_unit_available ?material_unit - material_unit)
    (scavenger_collected_material ?scavenger_slot - scavenger_slot ?material_unit - material_unit)
    (responder_collected_material ?responder_slot - responder_slot ?material_unit - material_unit)
    (slot_has_scout_report ?scavenger_slot - scavenger_slot ?scout_report - scout_report)
    (scout_report_reserved ?scout_report - scout_report)
    (scout_report_verified ?scout_report - scout_report)
    (scavenger_collection_complete ?scavenger_slot - scavenger_slot)
    (responder_assigned_support_report ?responder_slot - responder_slot ?support_report - support_report)
    (support_report_reserved ?support_report - support_report)
    (support_report_verified ?support_report - support_report)
    (responder_collection_complete ?responder_slot - responder_slot)
    (supply_cache_available ?supply_cache - supply_cache)
    (supply_cache_active ?supply_cache - supply_cache)
    (cache_has_scout_report ?supply_cache - supply_cache ?scout_report - scout_report)
    (cache_has_support_report ?supply_cache - supply_cache ?support_report - support_report)
    (cache_variant_alpha ?supply_cache - supply_cache)
    (cache_variant_beta ?supply_cache - supply_cache)
    (cache_activated ?supply_cache - supply_cache)
    (squad_has_scavenger_slot ?squad_unit - squad_unit ?scavenger_slot - scavenger_slot)
    (squad_has_responder_slot ?squad_unit - squad_unit ?responder_slot - responder_slot)
    (squad_assigned_cache ?squad_unit - squad_unit ?supply_cache - supply_cache)
    (upgrade_module_available ?upgrade_module - upgrade_module)
    (squad_has_upgrade_module ?squad_unit - squad_unit ?upgrade_module - upgrade_module)
    (upgrade_module_consumed ?upgrade_module - upgrade_module)
    (module_attached_to_cache ?upgrade_module - upgrade_module ?supply_cache - supply_cache)
    (squad_module_slot_ready ?squad_unit - squad_unit)
    (squad_has_variant_installed ?squad_unit - squad_unit)
    (squad_ready_for_cache_activation ?squad_unit - squad_unit)
    (squad_has_optional_asset ?squad_unit - squad_unit)
    (squad_optional_asset_applied ?squad_unit - squad_unit)
    (squad_confirmed_for_activation ?squad_unit - squad_unit)
    (squad_ready_for_finalization ?squad_unit - squad_unit)
    (blueprint_asset_available ?blueprint_asset - blueprint_asset)
    (squad_has_blueprint_asset ?squad_unit - squad_unit ?blueprint_asset - blueprint_asset)
    (squad_blueprint_applied ?squad_unit - squad_unit)
    (squad_blueprint_ready ?squad_unit - squad_unit)
    (squad_blueprint_unlocked ?squad_unit - squad_unit)
    (optional_asset_available ?optional_asset_a - optional_asset_a)
    (squad_assigned_optional_asset ?squad_unit - squad_unit ?optional_asset_a - optional_asset_a)
    (consumable_available ?consumable_type - consumable_type)
    (squad_has_consumable ?squad_unit - squad_unit ?consumable_type - consumable_type)
    (module_variant_available ?module_variant - module_variant)
    (squad_has_module_variant ?squad_unit - squad_unit ?module_variant - module_variant)
    (key_component_available ?key_component - key_component)
    (squad_has_key_component ?squad_unit - squad_unit ?key_component - key_component)
    (resource_token_available ?resource_token - resource_token)
    (recoverable_has_resource_token ?mission_run - recoverable_unit ?resource_token - resource_token)
    (scavenger_slot_committed ?scavenger_slot - scavenger_slot)
    (responder_slot_committed ?responder_slot - responder_slot)
    (squad_ready_for_restart ?squad_unit - squad_unit)
  )
  (:action detect_and_mark_failed_recoverable
    :parameters (?mission_run - recoverable_unit)
    :precondition
      (and
        (not
          (recoverable_marked_for_recovery ?mission_run)
        )
        (not
          (ready_for_redeploy ?mission_run)
        )
      )
    :effect (recoverable_marked_for_recovery ?mission_run)
  )
  (:action assign_supply_pack_to_recoverable
    :parameters (?mission_run - recoverable_unit ?supply_pack - supply_pack)
    :precondition
      (and
        (recoverable_marked_for_recovery ?mission_run)
        (not
          (recoverable_has_supply_pack ?mission_run)
        )
        (supply_pack_available ?supply_pack)
      )
    :effect
      (and
        (recoverable_has_supply_pack ?mission_run)
        (assigned_supply_pack ?mission_run ?supply_pack)
        (not
          (supply_pack_available ?supply_pack)
        )
      )
  )
  (:action claim_route_for_recoverable
    :parameters (?mission_run - recoverable_unit ?route_option - route_option)
    :precondition
      (and
        (recoverable_marked_for_recovery ?mission_run)
        (recoverable_has_supply_pack ?mission_run)
        (route_option_available ?route_option)
      )
    :effect
      (and
        (claimed_route ?mission_run ?route_option)
        (not
          (route_option_available ?route_option)
        )
      )
  )
  (:action finalize_route_claim
    :parameters (?mission_run - recoverable_unit ?route_option - route_option)
    :precondition
      (and
        (recoverable_marked_for_recovery ?mission_run)
        (recoverable_has_supply_pack ?mission_run)
        (claimed_route ?mission_run ?route_option)
        (not
          (ready_for_recovery ?mission_run)
        )
      )
    :effect (ready_for_recovery ?mission_run)
  )
  (:action release_route_for_recoverable
    :parameters (?mission_run - recoverable_unit ?route_option - route_option)
    :precondition
      (and
        (claimed_route ?mission_run ?route_option)
      )
    :effect
      (and
        (route_option_available ?route_option)
        (not
          (claimed_route ?mission_run ?route_option)
        )
      )
  )
  (:action attach_support_tool_to_recoverable
    :parameters (?mission_run - recoverable_unit ?support_tool - support_tool)
    :precondition
      (and
        (ready_for_recovery ?mission_run)
        (support_tool_available ?support_tool)
      )
    :effect
      (and
        (attached_support_tool ?mission_run ?support_tool)
        (not
          (support_tool_available ?support_tool)
        )
      )
  )
  (:action detach_support_tool_from_recoverable
    :parameters (?mission_run - recoverable_unit ?support_tool - support_tool)
    :precondition
      (and
        (attached_support_tool ?mission_run ?support_tool)
      )
    :effect
      (and
        (support_tool_available ?support_tool)
        (not
          (attached_support_tool ?mission_run ?support_tool)
        )
      )
  )
  (:action install_module_variant_to_squad
    :parameters (?squad_unit - squad_unit ?module_variant - module_variant)
    :precondition
      (and
        (ready_for_recovery ?squad_unit)
        (module_variant_available ?module_variant)
      )
    :effect
      (and
        (squad_has_module_variant ?squad_unit ?module_variant)
        (not
          (module_variant_available ?module_variant)
        )
      )
  )
  (:action remove_module_variant_from_squad
    :parameters (?squad_unit - squad_unit ?module_variant - module_variant)
    :precondition
      (and
        (squad_has_module_variant ?squad_unit ?module_variant)
      )
    :effect
      (and
        (module_variant_available ?module_variant)
        (not
          (squad_has_module_variant ?squad_unit ?module_variant)
        )
      )
  )
  (:action install_key_component_to_squad
    :parameters (?squad_unit - squad_unit ?key_component - key_component)
    :precondition
      (and
        (ready_for_recovery ?squad_unit)
        (key_component_available ?key_component)
      )
    :effect
      (and
        (squad_has_key_component ?squad_unit ?key_component)
        (not
          (key_component_available ?key_component)
        )
      )
  )
  (:action remove_key_component_from_squad
    :parameters (?squad_unit - squad_unit ?key_component - key_component)
    :precondition
      (and
        (squad_has_key_component ?squad_unit ?key_component)
      )
    :effect
      (and
        (key_component_available ?key_component)
        (not
          (squad_has_key_component ?squad_unit ?key_component)
        )
      )
  )
  (:action reserve_scout_report_for_scavenger
    :parameters (?scavenger_slot - scavenger_slot ?scout_report - scout_report ?route_option - route_option)
    :precondition
      (and
        (ready_for_recovery ?scavenger_slot)
        (claimed_route ?scavenger_slot ?route_option)
        (slot_has_scout_report ?scavenger_slot ?scout_report)
        (not
          (scout_report_reserved ?scout_report)
        )
        (not
          (scout_report_verified ?scout_report)
        )
      )
    :effect (scout_report_reserved ?scout_report)
  )
  (:action scavenger_claim_collection_with_tool
    :parameters (?scavenger_slot - scavenger_slot ?scout_report - scout_report ?support_tool - support_tool)
    :precondition
      (and
        (ready_for_recovery ?scavenger_slot)
        (attached_support_tool ?scavenger_slot ?support_tool)
        (slot_has_scout_report ?scavenger_slot ?scout_report)
        (scout_report_reserved ?scout_report)
        (not
          (scavenger_slot_committed ?scavenger_slot)
        )
      )
    :effect
      (and
        (scavenger_slot_committed ?scavenger_slot)
        (scavenger_collection_complete ?scavenger_slot)
      )
  )
  (:action scavenger_collect_material
    :parameters (?scavenger_slot - scavenger_slot ?scout_report - scout_report ?material_unit - material_unit)
    :precondition
      (and
        (ready_for_recovery ?scavenger_slot)
        (slot_has_scout_report ?scavenger_slot ?scout_report)
        (material_unit_available ?material_unit)
        (not
          (scavenger_slot_committed ?scavenger_slot)
        )
      )
    :effect
      (and
        (scout_report_verified ?scout_report)
        (scavenger_slot_committed ?scavenger_slot)
        (scavenger_collected_material ?scavenger_slot ?material_unit)
        (not
          (material_unit_available ?material_unit)
        )
      )
  )
  (:action scavenger_finalize_collection
    :parameters (?scavenger_slot - scavenger_slot ?scout_report - scout_report ?route_option - route_option ?material_unit - material_unit)
    :precondition
      (and
        (ready_for_recovery ?scavenger_slot)
        (claimed_route ?scavenger_slot ?route_option)
        (slot_has_scout_report ?scavenger_slot ?scout_report)
        (scout_report_verified ?scout_report)
        (scavenger_collected_material ?scavenger_slot ?material_unit)
        (not
          (scavenger_collection_complete ?scavenger_slot)
        )
      )
    :effect
      (and
        (scout_report_reserved ?scout_report)
        (scavenger_collection_complete ?scavenger_slot)
        (material_unit_available ?material_unit)
        (not
          (scavenger_collected_material ?scavenger_slot ?material_unit)
        )
      )
  )
  (:action reserve_support_report_for_responder
    :parameters (?responder_slot - responder_slot ?support_report - support_report ?route_option - route_option)
    :precondition
      (and
        (ready_for_recovery ?responder_slot)
        (claimed_route ?responder_slot ?route_option)
        (responder_assigned_support_report ?responder_slot ?support_report)
        (not
          (support_report_reserved ?support_report)
        )
        (not
          (support_report_verified ?support_report)
        )
      )
    :effect (support_report_reserved ?support_report)
  )
  (:action responder_claim_collection_with_tool
    :parameters (?responder_slot - responder_slot ?support_report - support_report ?support_tool - support_tool)
    :precondition
      (and
        (ready_for_recovery ?responder_slot)
        (attached_support_tool ?responder_slot ?support_tool)
        (responder_assigned_support_report ?responder_slot ?support_report)
        (support_report_reserved ?support_report)
        (not
          (responder_slot_committed ?responder_slot)
        )
      )
    :effect
      (and
        (responder_slot_committed ?responder_slot)
        (responder_collection_complete ?responder_slot)
      )
  )
  (:action responder_collect_material
    :parameters (?responder_slot - responder_slot ?support_report - support_report ?material_unit - material_unit)
    :precondition
      (and
        (ready_for_recovery ?responder_slot)
        (responder_assigned_support_report ?responder_slot ?support_report)
        (material_unit_available ?material_unit)
        (not
          (responder_slot_committed ?responder_slot)
        )
      )
    :effect
      (and
        (support_report_verified ?support_report)
        (responder_slot_committed ?responder_slot)
        (responder_collected_material ?responder_slot ?material_unit)
        (not
          (material_unit_available ?material_unit)
        )
      )
  )
  (:action responder_finalize_collection
    :parameters (?responder_slot - responder_slot ?support_report - support_report ?route_option - route_option ?material_unit - material_unit)
    :precondition
      (and
        (ready_for_recovery ?responder_slot)
        (claimed_route ?responder_slot ?route_option)
        (responder_assigned_support_report ?responder_slot ?support_report)
        (support_report_verified ?support_report)
        (responder_collected_material ?responder_slot ?material_unit)
        (not
          (responder_collection_complete ?responder_slot)
        )
      )
    :effect
      (and
        (support_report_reserved ?support_report)
        (responder_collection_complete ?responder_slot)
        (material_unit_available ?material_unit)
        (not
          (responder_collected_material ?responder_slot ?material_unit)
        )
      )
  )
  (:action assemble_supply_cache_from_reports
    :parameters (?scavenger_slot - scavenger_slot ?responder_slot - responder_slot ?scout_report - scout_report ?support_report - support_report ?supply_cache - supply_cache)
    :precondition
      (and
        (scavenger_slot_committed ?scavenger_slot)
        (responder_slot_committed ?responder_slot)
        (slot_has_scout_report ?scavenger_slot ?scout_report)
        (responder_assigned_support_report ?responder_slot ?support_report)
        (scout_report_reserved ?scout_report)
        (support_report_reserved ?support_report)
        (scavenger_collection_complete ?scavenger_slot)
        (responder_collection_complete ?responder_slot)
        (supply_cache_available ?supply_cache)
      )
    :effect
      (and
        (supply_cache_active ?supply_cache)
        (cache_has_scout_report ?supply_cache ?scout_report)
        (cache_has_support_report ?supply_cache ?support_report)
        (not
          (supply_cache_available ?supply_cache)
        )
      )
  )
  (:action assemble_supply_cache_variant_alpha
    :parameters (?scavenger_slot - scavenger_slot ?responder_slot - responder_slot ?scout_report - scout_report ?support_report - support_report ?supply_cache - supply_cache)
    :precondition
      (and
        (scavenger_slot_committed ?scavenger_slot)
        (responder_slot_committed ?responder_slot)
        (slot_has_scout_report ?scavenger_slot ?scout_report)
        (responder_assigned_support_report ?responder_slot ?support_report)
        (scout_report_verified ?scout_report)
        (support_report_reserved ?support_report)
        (not
          (scavenger_collection_complete ?scavenger_slot)
        )
        (responder_collection_complete ?responder_slot)
        (supply_cache_available ?supply_cache)
      )
    :effect
      (and
        (supply_cache_active ?supply_cache)
        (cache_has_scout_report ?supply_cache ?scout_report)
        (cache_has_support_report ?supply_cache ?support_report)
        (cache_variant_alpha ?supply_cache)
        (not
          (supply_cache_available ?supply_cache)
        )
      )
  )
  (:action assemble_supply_cache_variant_beta
    :parameters (?scavenger_slot - scavenger_slot ?responder_slot - responder_slot ?scout_report - scout_report ?support_report - support_report ?supply_cache - supply_cache)
    :precondition
      (and
        (scavenger_slot_committed ?scavenger_slot)
        (responder_slot_committed ?responder_slot)
        (slot_has_scout_report ?scavenger_slot ?scout_report)
        (responder_assigned_support_report ?responder_slot ?support_report)
        (scout_report_reserved ?scout_report)
        (support_report_verified ?support_report)
        (scavenger_collection_complete ?scavenger_slot)
        (not
          (responder_collection_complete ?responder_slot)
        )
        (supply_cache_available ?supply_cache)
      )
    :effect
      (and
        (supply_cache_active ?supply_cache)
        (cache_has_scout_report ?supply_cache ?scout_report)
        (cache_has_support_report ?supply_cache ?support_report)
        (cache_variant_beta ?supply_cache)
        (not
          (supply_cache_available ?supply_cache)
        )
      )
  )
  (:action assemble_supply_cache_with_both_variants
    :parameters (?scavenger_slot - scavenger_slot ?responder_slot - responder_slot ?scout_report - scout_report ?support_report - support_report ?supply_cache - supply_cache)
    :precondition
      (and
        (scavenger_slot_committed ?scavenger_slot)
        (responder_slot_committed ?responder_slot)
        (slot_has_scout_report ?scavenger_slot ?scout_report)
        (responder_assigned_support_report ?responder_slot ?support_report)
        (scout_report_verified ?scout_report)
        (support_report_verified ?support_report)
        (not
          (scavenger_collection_complete ?scavenger_slot)
        )
        (not
          (responder_collection_complete ?responder_slot)
        )
        (supply_cache_available ?supply_cache)
      )
    :effect
      (and
        (supply_cache_active ?supply_cache)
        (cache_has_scout_report ?supply_cache ?scout_report)
        (cache_has_support_report ?supply_cache ?support_report)
        (cache_variant_alpha ?supply_cache)
        (cache_variant_beta ?supply_cache)
        (not
          (supply_cache_available ?supply_cache)
        )
      )
  )
  (:action activate_supply_cache
    :parameters (?supply_cache - supply_cache ?scavenger_slot - scavenger_slot ?route_option - route_option)
    :precondition
      (and
        (supply_cache_active ?supply_cache)
        (scavenger_slot_committed ?scavenger_slot)
        (claimed_route ?scavenger_slot ?route_option)
        (not
          (cache_activated ?supply_cache)
        )
      )
    :effect (cache_activated ?supply_cache)
  )
  (:action install_upgrade_module_into_cache
    :parameters (?squad_unit - squad_unit ?upgrade_module - upgrade_module ?supply_cache - supply_cache)
    :precondition
      (and
        (ready_for_recovery ?squad_unit)
        (squad_assigned_cache ?squad_unit ?supply_cache)
        (squad_has_upgrade_module ?squad_unit ?upgrade_module)
        (upgrade_module_available ?upgrade_module)
        (supply_cache_active ?supply_cache)
        (cache_activated ?supply_cache)
        (not
          (upgrade_module_consumed ?upgrade_module)
        )
      )
    :effect
      (and
        (upgrade_module_consumed ?upgrade_module)
        (module_attached_to_cache ?upgrade_module ?supply_cache)
        (not
          (upgrade_module_available ?upgrade_module)
        )
      )
  )
  (:action prepare_squad_for_cache_transfer
    :parameters (?squad_unit - squad_unit ?upgrade_module - upgrade_module ?supply_cache - supply_cache ?route_option - route_option)
    :precondition
      (and
        (ready_for_recovery ?squad_unit)
        (squad_has_upgrade_module ?squad_unit ?upgrade_module)
        (upgrade_module_consumed ?upgrade_module)
        (module_attached_to_cache ?upgrade_module ?supply_cache)
        (claimed_route ?squad_unit ?route_option)
        (not
          (cache_variant_alpha ?supply_cache)
        )
        (not
          (squad_module_slot_ready ?squad_unit)
        )
      )
    :effect (squad_module_slot_ready ?squad_unit)
  )
  (:action assign_optional_asset_to_squad
    :parameters (?squad_unit - squad_unit ?optional_asset_a - optional_asset_a)
    :precondition
      (and
        (ready_for_recovery ?squad_unit)
        (optional_asset_available ?optional_asset_a)
        (not
          (squad_has_optional_asset ?squad_unit)
        )
      )
    :effect
      (and
        (squad_has_optional_asset ?squad_unit)
        (squad_assigned_optional_asset ?squad_unit ?optional_asset_a)
        (not
          (optional_asset_available ?optional_asset_a)
        )
      )
  )
  (:action apply_optional_asset_and_stage_squad
    :parameters (?squad_unit - squad_unit ?upgrade_module - upgrade_module ?supply_cache - supply_cache ?route_option - route_option ?optional_asset_a - optional_asset_a)
    :precondition
      (and
        (ready_for_recovery ?squad_unit)
        (squad_has_upgrade_module ?squad_unit ?upgrade_module)
        (upgrade_module_consumed ?upgrade_module)
        (module_attached_to_cache ?upgrade_module ?supply_cache)
        (claimed_route ?squad_unit ?route_option)
        (cache_variant_alpha ?supply_cache)
        (squad_has_optional_asset ?squad_unit)
        (squad_assigned_optional_asset ?squad_unit ?optional_asset_a)
        (not
          (squad_module_slot_ready ?squad_unit)
        )
      )
    :effect
      (and
        (squad_module_slot_ready ?squad_unit)
        (squad_optional_asset_applied ?squad_unit)
      )
  )
  (:action install_variant_and_stage_squad_variant_alpha
    :parameters (?squad_unit - squad_unit ?module_variant - module_variant ?support_tool - support_tool ?upgrade_module - upgrade_module ?supply_cache - supply_cache)
    :precondition
      (and
        (squad_module_slot_ready ?squad_unit)
        (squad_has_module_variant ?squad_unit ?module_variant)
        (attached_support_tool ?squad_unit ?support_tool)
        (squad_has_upgrade_module ?squad_unit ?upgrade_module)
        (module_attached_to_cache ?upgrade_module ?supply_cache)
        (not
          (cache_variant_beta ?supply_cache)
        )
        (not
          (squad_has_variant_installed ?squad_unit)
        )
      )
