## Planning domain (PDDL)

(define (domain boss_failure_loadout_adjustment)
  (:requirements :strips :typing :negative-preconditions)
  (:types asset_category - object item_category - object bundle_category - object encounter_root - object encounter_entity - encounter_root loadout_template - asset_category weapon_option - asset_category support_unit - asset_category tactic_module - asset_category consumable_buff - asset_category recovery_resource - asset_category special_equipment - asset_category synergy_modifier - asset_category inventory_item - item_category attachment_module - item_category tactical_trait - item_category primary_equipment_slot - bundle_category secondary_equipment_slot - bundle_category loadout_bundle - bundle_category player_role_group - encounter_entity plan_group - encounter_entity attacker_slot - player_role_group support_slot - player_role_group operator - plan_group)
  (:predicates
    (encounter_active ?encounter_entity - encounter_entity)
    (entity_validated ?encounter_entity - encounter_entity)
    (loadout_template_selected ?encounter_entity - encounter_entity)
    (recovery_allocated ?encounter_entity - encounter_entity)
    (ready_for_redeployment ?encounter_entity - encounter_entity)
    (recovery_consumed ?encounter_entity - encounter_entity)
    (loadout_template_available ?loadout_template - loadout_template)
    (assigned_loadout_template ?encounter_entity - encounter_entity ?loadout_template - loadout_template)
    (weapon_available ?weapon_option - weapon_option)
    (bound_weapon ?encounter_entity - encounter_entity ?weapon_option - weapon_option)
    (support_unit_available ?support_unit - support_unit)
    (assigned_support_unit ?encounter_entity - encounter_entity ?support_unit - support_unit)
    (inventory_item_available ?inventory_item - inventory_item)
    (attacker_slot_item_assigned ?attacker_slot - attacker_slot ?inventory_item - inventory_item)
    (support_slot_item_assigned ?support_slot - support_slot ?inventory_item - inventory_item)
    (has_primary_slot ?attacker_slot - attacker_slot ?primary_equipment_slot - primary_equipment_slot)
    (primary_slot_staged ?primary_equipment_slot - primary_equipment_slot)
    (primary_slot_equipped ?primary_equipment_slot - primary_equipment_slot)
    (attacker_slot_committed ?attacker_slot - attacker_slot)
    (has_secondary_slot ?support_slot - support_slot ?secondary_equipment_slot - secondary_equipment_slot)
    (secondary_slot_staged ?secondary_equipment_slot - secondary_equipment_slot)
    (secondary_slot_equipped ?secondary_equipment_slot - secondary_equipment_slot)
    (support_slot_committed ?support_slot - support_slot)
    (bundle_template_available ?loadout_bundle - loadout_bundle)
    (bundle_committed ?loadout_bundle - loadout_bundle)
    (bundle_primary_assignment ?loadout_bundle - loadout_bundle ?primary_equipment_slot - primary_equipment_slot)
    (bundle_secondary_assignment ?loadout_bundle - loadout_bundle ?secondary_equipment_slot - secondary_equipment_slot)
    (bundle_primary_ready ?loadout_bundle - loadout_bundle)
    (bundle_secondary_ready ?loadout_bundle - loadout_bundle)
    (bundle_finalized ?loadout_bundle - loadout_bundle)
    (operator_has_attacker_slot ?mission_coordinator - operator ?attacker_slot - attacker_slot)
    (operator_has_support_slot ?mission_coordinator - operator ?support_slot - support_slot)
    (operator_has_bundle ?mission_coordinator - operator ?loadout_bundle - loadout_bundle)
    (attachment_module_available ?attachment_module - attachment_module)
    (operator_attachment_slot ?mission_coordinator - operator ?attachment_module - attachment_module)
    (module_committed ?attachment_module - attachment_module)
    (module_attached_to_bundle ?attachment_module - attachment_module ?loadout_bundle - loadout_bundle)
    (operator_module_applied ?mission_coordinator - operator)
    (operator_attachment_confirmed ?mission_coordinator - operator)
    (operator_synergy_validated ?mission_coordinator - operator)
    (operator_tactic_assigned ?mission_coordinator - operator)
    (operator_tactic_staged ?mission_coordinator - operator)
    (operator_trait_applied ?mission_coordinator - operator)
    (operator_activation_ready ?mission_coordinator - operator)
    (tactical_trait_available ?tactical_trait - tactical_trait)
    (operator_tactical_trait_assigned ?mission_coordinator - operator ?tactical_trait - tactical_trait)
    (operator_trait_active ?mission_coordinator - operator)
    (operator_trait_confirmed ?mission_coordinator - operator)
    (operator_trait_finalized ?mission_coordinator - operator)
    (tactic_module_available ?tactic_module - tactic_module)
    (operator_tactic_module_assigned ?mission_coordinator - operator ?tactic_module - tactic_module)
    (consumable_buff_available ?consumable_buff - consumable_buff)
    (operator_consumable_assigned ?mission_coordinator - operator ?consumable_buff - consumable_buff)
    (special_equipment_available ?special_equipment - special_equipment)
    (operator_special_equipment_assigned ?mission_coordinator - operator ?special_equipment - special_equipment)
    (synergy_modifier_available ?synergy_modifier - synergy_modifier)
    (operator_synergy_modifier_assigned ?mission_coordinator - operator ?synergy_modifier - synergy_modifier)
    (recovery_resource_available ?recovery_resource - recovery_resource)
    (entity_recovery_allocated ?encounter_entity - encounter_entity ?recovery_resource - recovery_resource)
    (attacker_slot_staged ?attacker_slot - attacker_slot)
    (support_slot_staged ?support_slot - support_slot)
    (operator_locked ?mission_coordinator - operator)
  )
  (:action activate_encounter
    :parameters (?encounter_entity - encounter_entity)
    :precondition
      (and
        (not
          (encounter_active ?encounter_entity)
        )
        (not
          (recovery_allocated ?encounter_entity)
        )
      )
    :effect (encounter_active ?encounter_entity)
  )
  (:action select_loadout_template
    :parameters (?encounter_entity - encounter_entity ?loadout_template - loadout_template)
    :precondition
      (and
        (encounter_active ?encounter_entity)
        (not
          (loadout_template_selected ?encounter_entity)
        )
        (loadout_template_available ?loadout_template)
      )
    :effect
      (and
        (loadout_template_selected ?encounter_entity)
        (assigned_loadout_template ?encounter_entity ?loadout_template)
        (not
          (loadout_template_available ?loadout_template)
        )
      )
  )
  (:action bind_weapon_to_entity
    :parameters (?encounter_entity - encounter_entity ?weapon_option - weapon_option)
    :precondition
      (and
        (encounter_active ?encounter_entity)
        (loadout_template_selected ?encounter_entity)
        (weapon_available ?weapon_option)
      )
    :effect
      (and
        (bound_weapon ?encounter_entity ?weapon_option)
        (not
          (weapon_available ?weapon_option)
        )
      )
  )
  (:action validate_weapon_binding
    :parameters (?encounter_entity - encounter_entity ?weapon_option - weapon_option)
    :precondition
      (and
        (encounter_active ?encounter_entity)
        (loadout_template_selected ?encounter_entity)
        (bound_weapon ?encounter_entity ?weapon_option)
        (not
          (entity_validated ?encounter_entity)
        )
      )
    :effect (entity_validated ?encounter_entity)
  )
  (:action release_weapon_binding
    :parameters (?encounter_entity - encounter_entity ?weapon_option - weapon_option)
    :precondition
      (and
        (bound_weapon ?encounter_entity ?weapon_option)
      )
    :effect
      (and
        (weapon_available ?weapon_option)
        (not
          (bound_weapon ?encounter_entity ?weapon_option)
        )
      )
  )
  (:action assign_support_unit_to_entity
    :parameters (?encounter_entity - encounter_entity ?support_unit - support_unit)
    :precondition
      (and
        (entity_validated ?encounter_entity)
        (support_unit_available ?support_unit)
      )
    :effect
      (and
        (assigned_support_unit ?encounter_entity ?support_unit)
        (not
          (support_unit_available ?support_unit)
        )
      )
  )
  (:action release_support_unit_from_entity
    :parameters (?encounter_entity - encounter_entity ?support_unit - support_unit)
    :precondition
      (and
        (assigned_support_unit ?encounter_entity ?support_unit)
      )
    :effect
      (and
        (support_unit_available ?support_unit)
        (not
          (assigned_support_unit ?encounter_entity ?support_unit)
        )
      )
  )
  (:action assign_special_equipment
    :parameters (?mission_coordinator - operator ?special_equipment - special_equipment)
    :precondition
      (and
        (entity_validated ?mission_coordinator)
        (special_equipment_available ?special_equipment)
      )
    :effect
      (and
        (operator_special_equipment_assigned ?mission_coordinator ?special_equipment)
        (not
          (special_equipment_available ?special_equipment)
        )
      )
  )
  (:action unassign_special_equipment
    :parameters (?mission_coordinator - operator ?special_equipment - special_equipment)
    :precondition
      (and
        (operator_special_equipment_assigned ?mission_coordinator ?special_equipment)
      )
    :effect
      (and
        (special_equipment_available ?special_equipment)
        (not
          (operator_special_equipment_assigned ?mission_coordinator ?special_equipment)
        )
      )
  )
  (:action assign_synergy_modifier
    :parameters (?mission_coordinator - operator ?synergy_modifier - synergy_modifier)
    :precondition
      (and
        (entity_validated ?mission_coordinator)
        (synergy_modifier_available ?synergy_modifier)
      )
    :effect
      (and
        (operator_synergy_modifier_assigned ?mission_coordinator ?synergy_modifier)
        (not
          (synergy_modifier_available ?synergy_modifier)
        )
      )
  )
  (:action unassign_synergy_modifier
    :parameters (?mission_coordinator - operator ?synergy_modifier - synergy_modifier)
    :precondition
      (and
        (operator_synergy_modifier_assigned ?mission_coordinator ?synergy_modifier)
      )
    :effect
      (and
        (synergy_modifier_available ?synergy_modifier)
        (not
          (operator_synergy_modifier_assigned ?mission_coordinator ?synergy_modifier)
        )
      )
  )
  (:action stage_primary_equipment_slot
    :parameters (?attacker_slot - attacker_slot ?primary_equipment_slot - primary_equipment_slot ?weapon_option - weapon_option)
    :precondition
      (and
        (entity_validated ?attacker_slot)
        (bound_weapon ?attacker_slot ?weapon_option)
        (has_primary_slot ?attacker_slot ?primary_equipment_slot)
        (not
          (primary_slot_staged ?primary_equipment_slot)
        )
        (not
          (primary_slot_equipped ?primary_equipment_slot)
        )
      )
    :effect (primary_slot_staged ?primary_equipment_slot)
  )
  (:action confirm_attacker_slot
    :parameters (?attacker_slot - attacker_slot ?primary_equipment_slot - primary_equipment_slot ?support_unit - support_unit)
    :precondition
      (and
        (entity_validated ?attacker_slot)
        (assigned_support_unit ?attacker_slot ?support_unit)
        (has_primary_slot ?attacker_slot ?primary_equipment_slot)
        (primary_slot_staged ?primary_equipment_slot)
        (not
          (attacker_slot_staged ?attacker_slot)
        )
      )
    :effect
      (and
        (attacker_slot_staged ?attacker_slot)
        (attacker_slot_committed ?attacker_slot)
      )
  )
  (:action attach_item_to_attacker_slot
    :parameters (?attacker_slot - attacker_slot ?primary_equipment_slot - primary_equipment_slot ?inventory_item - inventory_item)
    :precondition
      (and
        (entity_validated ?attacker_slot)
        (has_primary_slot ?attacker_slot ?primary_equipment_slot)
        (inventory_item_available ?inventory_item)
        (not
          (attacker_slot_staged ?attacker_slot)
        )
      )
    :effect
      (and
        (primary_slot_equipped ?primary_equipment_slot)
        (attacker_slot_staged ?attacker_slot)
        (attacker_slot_item_assigned ?attacker_slot ?inventory_item)
        (not
          (inventory_item_available ?inventory_item)
        )
      )
  )
  (:action commit_attacker_primary_setup
    :parameters (?attacker_slot - attacker_slot ?primary_equipment_slot - primary_equipment_slot ?weapon_option - weapon_option ?inventory_item - inventory_item)
    :precondition
      (and
        (entity_validated ?attacker_slot)
        (bound_weapon ?attacker_slot ?weapon_option)
        (has_primary_slot ?attacker_slot ?primary_equipment_slot)
        (primary_slot_equipped ?primary_equipment_slot)
        (attacker_slot_item_assigned ?attacker_slot ?inventory_item)
        (not
          (attacker_slot_committed ?attacker_slot)
        )
      )
    :effect
      (and
        (primary_slot_staged ?primary_equipment_slot)
        (attacker_slot_committed ?attacker_slot)
        (inventory_item_available ?inventory_item)
        (not
          (attacker_slot_item_assigned ?attacker_slot ?inventory_item)
        )
      )
  )
  (:action stage_secondary_equipment_slot
    :parameters (?support_slot - support_slot ?secondary_equipment_slot - secondary_equipment_slot ?weapon_option - weapon_option)
    :precondition
      (and
        (entity_validated ?support_slot)
        (bound_weapon ?support_slot ?weapon_option)
        (has_secondary_slot ?support_slot ?secondary_equipment_slot)
        (not
          (secondary_slot_staged ?secondary_equipment_slot)
        )
        (not
          (secondary_slot_equipped ?secondary_equipment_slot)
        )
      )
    :effect (secondary_slot_staged ?secondary_equipment_slot)
  )
  (:action confirm_support_slot
    :parameters (?support_slot - support_slot ?secondary_equipment_slot - secondary_equipment_slot ?support_unit - support_unit)
    :precondition
      (and
        (entity_validated ?support_slot)
        (assigned_support_unit ?support_slot ?support_unit)
        (has_secondary_slot ?support_slot ?secondary_equipment_slot)
        (secondary_slot_staged ?secondary_equipment_slot)
        (not
          (support_slot_staged ?support_slot)
        )
      )
    :effect
      (and
        (support_slot_staged ?support_slot)
        (support_slot_committed ?support_slot)
      )
  )
  (:action attach_item_to_support_slot
    :parameters (?support_slot - support_slot ?secondary_equipment_slot - secondary_equipment_slot ?inventory_item - inventory_item)
    :precondition
      (and
        (entity_validated ?support_slot)
        (has_secondary_slot ?support_slot ?secondary_equipment_slot)
        (inventory_item_available ?inventory_item)
        (not
          (support_slot_staged ?support_slot)
        )
      )
    :effect
      (and
        (secondary_slot_equipped ?secondary_equipment_slot)
        (support_slot_staged ?support_slot)
        (support_slot_item_assigned ?support_slot ?inventory_item)
        (not
          (inventory_item_available ?inventory_item)
        )
      )
  )
  (:action commit_support_secondary_setup
    :parameters (?support_slot - support_slot ?secondary_equipment_slot - secondary_equipment_slot ?weapon_option - weapon_option ?inventory_item - inventory_item)
    :precondition
      (and
        (entity_validated ?support_slot)
        (bound_weapon ?support_slot ?weapon_option)
        (has_secondary_slot ?support_slot ?secondary_equipment_slot)
        (secondary_slot_equipped ?secondary_equipment_slot)
        (support_slot_item_assigned ?support_slot ?inventory_item)
        (not
          (support_slot_committed ?support_slot)
        )
      )
    :effect
      (and
        (secondary_slot_staged ?secondary_equipment_slot)
        (support_slot_committed ?support_slot)
        (inventory_item_available ?inventory_item)
        (not
          (support_slot_item_assigned ?support_slot ?inventory_item)
        )
      )
  )
  (:action commit_loadout_bundle_basic
    :parameters (?attacker_slot - attacker_slot ?support_slot - support_slot ?primary_equipment_slot - primary_equipment_slot ?secondary_equipment_slot - secondary_equipment_slot ?loadout_bundle - loadout_bundle)
    :precondition
      (and
        (attacker_slot_staged ?attacker_slot)
        (support_slot_staged ?support_slot)
        (has_primary_slot ?attacker_slot ?primary_equipment_slot)
        (has_secondary_slot ?support_slot ?secondary_equipment_slot)
        (primary_slot_staged ?primary_equipment_slot)
        (secondary_slot_staged ?secondary_equipment_slot)
        (attacker_slot_committed ?attacker_slot)
        (support_slot_committed ?support_slot)
        (bundle_template_available ?loadout_bundle)
      )
    :effect
      (and
        (bundle_committed ?loadout_bundle)
        (bundle_primary_assignment ?loadout_bundle ?primary_equipment_slot)
        (bundle_secondary_assignment ?loadout_bundle ?secondary_equipment_slot)
        (not
          (bundle_template_available ?loadout_bundle)
        )
      )
  )
  (:action commit_loadout_bundle_primary_ready
    :parameters (?attacker_slot - attacker_slot ?support_slot - support_slot ?primary_equipment_slot - primary_equipment_slot ?secondary_equipment_slot - secondary_equipment_slot ?loadout_bundle - loadout_bundle)
    :precondition
      (and
        (attacker_slot_staged ?attacker_slot)
        (support_slot_staged ?support_slot)
        (has_primary_slot ?attacker_slot ?primary_equipment_slot)
        (has_secondary_slot ?support_slot ?secondary_equipment_slot)
        (primary_slot_equipped ?primary_equipment_slot)
        (secondary_slot_staged ?secondary_equipment_slot)
        (not
          (attacker_slot_committed ?attacker_slot)
        )
        (support_slot_committed ?support_slot)
        (bundle_template_available ?loadout_bundle)
      )
    :effect
      (and
        (bundle_committed ?loadout_bundle)
        (bundle_primary_assignment ?loadout_bundle ?primary_equipment_slot)
        (bundle_secondary_assignment ?loadout_bundle ?secondary_equipment_slot)
        (bundle_primary_ready ?loadout_bundle)
        (not
          (bundle_template_available ?loadout_bundle)
        )
      )
  )
  (:action commit_loadout_bundle_secondary_ready
    :parameters (?attacker_slot - attacker_slot ?support_slot - support_slot ?primary_equipment_slot - primary_equipment_slot ?secondary_equipment_slot - secondary_equipment_slot ?loadout_bundle - loadout_bundle)
    :precondition
      (and
        (attacker_slot_staged ?attacker_slot)
        (support_slot_staged ?support_slot)
        (has_primary_slot ?attacker_slot ?primary_equipment_slot)
        (has_secondary_slot ?support_slot ?secondary_equipment_slot)
        (primary_slot_staged ?primary_equipment_slot)
        (secondary_slot_equipped ?secondary_equipment_slot)
        (attacker_slot_committed ?attacker_slot)
        (not
          (support_slot_committed ?support_slot)
        )
        (bundle_template_available ?loadout_bundle)
      )
    :effect
      (and
        (bundle_committed ?loadout_bundle)
        (bundle_primary_assignment ?loadout_bundle ?primary_equipment_slot)
        (bundle_secondary_assignment ?loadout_bundle ?secondary_equipment_slot)
        (bundle_secondary_ready ?loadout_bundle)
        (not
          (bundle_template_available ?loadout_bundle)
        )
      )
  )
  (:action commit_loadout_bundle_fully_ready
    :parameters (?attacker_slot - attacker_slot ?support_slot - support_slot ?primary_equipment_slot - primary_equipment_slot ?secondary_equipment_slot - secondary_equipment_slot ?loadout_bundle - loadout_bundle)
    :precondition
      (and
        (attacker_slot_staged ?attacker_slot)
        (support_slot_staged ?support_slot)
        (has_primary_slot ?attacker_slot ?primary_equipment_slot)
        (has_secondary_slot ?support_slot ?secondary_equipment_slot)
        (primary_slot_equipped ?primary_equipment_slot)
        (secondary_slot_equipped ?secondary_equipment_slot)
        (not
          (attacker_slot_committed ?attacker_slot)
        )
        (not
          (support_slot_committed ?support_slot)
        )
        (bundle_template_available ?loadout_bundle)
      )
    :effect
      (and
        (bundle_committed ?loadout_bundle)
        (bundle_primary_assignment ?loadout_bundle ?primary_equipment_slot)
        (bundle_secondary_assignment ?loadout_bundle ?secondary_equipment_slot)
        (bundle_primary_ready ?loadout_bundle)
        (bundle_secondary_ready ?loadout_bundle)
        (not
          (bundle_template_available ?loadout_bundle)
        )
      )
  )
  (:action finalize_committed_bundle
    :parameters (?loadout_bundle - loadout_bundle ?attacker_slot - attacker_slot ?weapon_option - weapon_option)
    :precondition
      (and
        (bundle_committed ?loadout_bundle)
        (attacker_slot_staged ?attacker_slot)
        (bound_weapon ?attacker_slot ?weapon_option)
        (not
          (bundle_finalized ?loadout_bundle)
        )
      )
    :effect (bundle_finalized ?loadout_bundle)
  )
  (:action attach_module_to_bundle
    :parameters (?mission_coordinator - operator ?attachment_module - attachment_module ?loadout_bundle - loadout_bundle)
    :precondition
      (and
        (entity_validated ?mission_coordinator)
        (operator_has_bundle ?mission_coordinator ?loadout_bundle)
        (operator_attachment_slot ?mission_coordinator ?attachment_module)
        (attachment_module_available ?attachment_module)
        (bundle_committed ?loadout_bundle)
        (bundle_finalized ?loadout_bundle)
        (not
          (module_committed ?attachment_module)
        )
      )
    :effect
      (and
        (module_committed ?attachment_module)
        (module_attached_to_bundle ?attachment_module ?loadout_bundle)
        (not
          (attachment_module_available ?attachment_module)
        )
      )
  )
  (:action apply_module_to_operator
    :parameters (?mission_coordinator - operator ?attachment_module - attachment_module ?loadout_bundle - loadout_bundle ?weapon_option - weapon_option)
    :precondition
      (and
        (entity_validated ?mission_coordinator)
        (operator_attachment_slot ?mission_coordinator ?attachment_module)
        (module_committed ?attachment_module)
        (module_attached_to_bundle ?attachment_module ?loadout_bundle)
        (bound_weapon ?mission_coordinator ?weapon_option)
        (not
          (bundle_primary_ready ?loadout_bundle)
        )
        (not
          (operator_module_applied ?mission_coordinator)
        )
      )
    :effect (operator_module_applied ?mission_coordinator)
  )
  (:action assign_tactic_module_to_operator
    :parameters (?mission_coordinator - operator ?tactic_module - tactic_module)
    :precondition
      (and
        (entity_validated ?mission_coordinator)
        (tactic_module_available ?tactic_module)
        (not
          (operator_tactic_assigned ?mission_coordinator)
        )
      )
    :effect
      (and
        (operator_tactic_assigned ?mission_coordinator)
        (operator_tactic_module_assigned ?mission_coordinator ?tactic_module)
        (not
          (tactic_module_available ?tactic_module)
        )
      )
  )
  (:action attach_tactic_module_to_operator
    :parameters (?mission_coordinator - operator ?attachment_module - attachment_module ?loadout_bundle - loadout_bundle ?weapon_option - weapon_option ?tactic_module - tactic_module)
    :precondition
      (and
        (entity_validated ?mission_coordinator)
        (operator_attachment_slot ?mission_coordinator ?attachment_module)
        (module_committed ?attachment_module)
        (module_attached_to_bundle ?attachment_module ?loadout_bundle)
        (bound_weapon ?mission_coordinator ?weapon_option)
        (bundle_primary_ready ?loadout_bundle)
        (operator_tactic_assigned ?mission_coordinator)
        (operator_tactic_module_assigned ?mission_coordinator ?tactic_module)
        (not
          (operator_module_applied ?mission_coordinator)
        )
      )
    :effect
      (and
        (operator_module_applied ?mission_coordinator)
        (operator_tactic_staged ?mission_coordinator)
      )
  )
  (:action prepare_special_equipment
    :parameters (?mission_coordinator - operator ?special_equipment - special_equipment ?support_unit - support_unit ?attachment_module - attachment_module ?loadout_bundle - loadout_bundle)
    :precondition
      (and
        (operator_module_applied ?mission_coordinator)
        (operator_special_equipment_assigned ?mission_coordinator ?special_equipment)
        (assigned_support_unit ?mission_coordinator ?support_unit)
        (operator_attachment_slot ?mission_coordinator ?attachment_module)
        (module_attached_to_bundle ?attachment_module ?loadout_bundle)
        (not
          (bundle_secondary_ready ?loadout_bundle)
        )
        (not
          (operator_attachment_confirmed ?mission_coordinator)
        )
      )
    :effect (operator_attachment_confirmed ?mission_coordinator)
  )
  (:action prepare_special_equipment_variant
    :parameters (?mission_coordinator - operator ?special_equipment - special_equipment ?support_unit - support_unit ?attachment_module - attachment_module ?loadout_bundle - loadout_bundle)
    :precondition
      (and
        (operator_module_applied ?mission_coordinator)
        (operator_special_equipment_assigned ?mission_coordinator ?special_equipment)
        (assigned_support_unit ?mission_coordinator ?support_unit)
        (operator_attachment_slot ?mission_coordinator ?attachment_module)
