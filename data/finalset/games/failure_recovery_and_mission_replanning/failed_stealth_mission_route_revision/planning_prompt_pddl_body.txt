## Planning domain (PDDL)

(define (domain failed_stealth_mission_route_revision)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity_group - object item_group - object asset_group - object mission_context - object mission_node - mission_context path_option - entity_group intel_fragment - entity_group equipment_item - entity_group tactical_mod - entity_group special_ability - entity_group mission_resource - entity_group primary_attachment - entity_group secondary_attachment - entity_group consumable - item_group attachment - item_group mission_marker - item_group approach_vector_recon - asset_group approach_vector_assault - asset_group loadout - asset_group subobjective_group - mission_node subobjective_group_alt - mission_node recon_unit - subobjective_group assault_unit - subobjective_group planner_agent - subobjective_group_alt)
  (:predicates
    (entity_detected ?mission_node - mission_node)
    (entity_prepared ?mission_node - mission_node)
    (entity_path_claimed ?mission_node - mission_node)
    (entity_active ?mission_node - mission_node)
    (deployment_ready ?mission_node - mission_node)
    (entity_recovered ?mission_node - mission_node)
    (path_available ?path_option - path_option)
    (entity_assigned_path ?mission_node - mission_node ?path_option - path_option)
    (intel_available ?intel_fragment - intel_fragment)
    (entity_has_intel ?mission_node - mission_node ?intel_fragment - intel_fragment)
    (equipment_available ?equipment_item - equipment_item)
    (entity_equipment_attached ?mission_node - mission_node ?equipment_item - equipment_item)
    (consumable_available ?consumable - consumable)
    (recon_unit_has_consumable ?recon_unit - recon_unit ?consumable - consumable)
    (assault_unit_has_consumable ?assault_unit - assault_unit ?consumable - consumable)
    (approach_vector_recon_attached ?recon_unit - recon_unit ?approach_vector_recon - approach_vector_recon)
    (approach_vector_recon_confirmed ?approach_vector_recon - approach_vector_recon)
    (approach_vector_recon_exhausted ?approach_vector_recon - approach_vector_recon)
    (recon_unit_confirmed ?recon_unit - recon_unit)
    (approach_vector_assault_attached ?assault_unit - assault_unit ?approach_vector_assault - approach_vector_assault)
    (approach_vector_assault_confirmed ?approach_vector_assault - approach_vector_assault)
    (approach_vector_assault_exhausted ?approach_vector_assault - approach_vector_assault)
    (assault_unit_confirmed ?assault_unit - assault_unit)
    (loadout_available ?loadout - loadout)
    (loadout_built ?loadout - loadout)
    (loadout_has_approach_vector_recon ?loadout - loadout ?approach_vector_recon - approach_vector_recon)
    (loadout_has_approach_vector_assault ?loadout - loadout ?approach_vector_assault - approach_vector_assault)
    (loadout_recon_exhausted ?loadout - loadout)
    (loadout_assault_exhausted ?loadout - loadout)
    (loadout_finalized ?loadout - loadout)
    (agent_assigned_recon ?planner_agent - planner_agent ?recon_unit - recon_unit)
    (agent_assigned_assault ?planner_agent - planner_agent ?assault_unit - assault_unit)
    (agent_has_loadout ?planner_agent - planner_agent ?loadout - loadout)
    (attachment_available ?attachment - attachment)
    (agent_has_attachment ?planner_agent - planner_agent ?attachment - attachment)
    (attachment_installed ?attachment - attachment)
    (attachment_attached_to_loadout ?attachment - attachment ?loadout - loadout)
    (agent_attachment_staged ?planner_agent - planner_agent)
    (agent_attachment_locked ?planner_agent - planner_agent)
    (agent_configuration_finalized ?planner_agent - planner_agent)
    (agent_tactical_mod_active ?planner_agent - planner_agent)
    (agent_mod_staged ?planner_agent - planner_agent)
    (agent_supports_special_ability ?planner_agent - planner_agent)
    (agent_validated_for_execution ?planner_agent - planner_agent)
    (mission_marker_available ?mission_marker - mission_marker)
    (agent_assigned_marker ?planner_agent - planner_agent ?mission_marker - mission_marker)
    (agent_marker_claimed ?planner_agent - planner_agent)
    (agent_marker_prepared ?planner_agent - planner_agent)
    (agent_marker_confirmed ?planner_agent - planner_agent)
    (tactical_mod_available ?tactical_mod - tactical_mod)
    (agent_assigned_tactical_mod ?planner_agent - planner_agent ?tactical_mod - tactical_mod)
    (special_ability_available ?special_ability - special_ability)
    (agent_has_special_ability ?planner_agent - planner_agent ?special_ability - special_ability)
    (primary_attachment_available ?primary_attachment - primary_attachment)
    (agent_has_primary_attachment ?planner_agent - planner_agent ?primary_attachment - primary_attachment)
    (secondary_attachment_available ?secondary_attachment - secondary_attachment)
    (agent_has_secondary_attachment ?planner_agent - planner_agent ?secondary_attachment - secondary_attachment)
    (mission_resource_available ?mission_resource - mission_resource)
    (entity_has_resource ?mission_node - mission_node ?mission_resource - mission_resource)
    (recon_unit_ready ?recon_unit - recon_unit)
    (assault_unit_ready ?assault_unit - assault_unit)
    (agent_deployment_token ?planner_agent - planner_agent)
  )
  (:action detect_mission_node
    :parameters (?mission_node - mission_node)
    :precondition
      (and
        (not
          (entity_detected ?mission_node)
        )
        (not
          (entity_active ?mission_node)
        )
      )
    :effect (entity_detected ?mission_node)
  )
  (:action assign_path_option_to_node
    :parameters (?mission_node - mission_node ?path_option - path_option)
    :precondition
      (and
        (entity_detected ?mission_node)
        (not
          (entity_path_claimed ?mission_node)
        )
        (path_available ?path_option)
      )
    :effect
      (and
        (entity_path_claimed ?mission_node)
        (entity_assigned_path ?mission_node ?path_option)
        (not
          (path_available ?path_option)
        )
      )
  )
  (:action attach_intel_to_node
    :parameters (?mission_node - mission_node ?intel_fragment - intel_fragment)
    :precondition
      (and
        (entity_detected ?mission_node)
        (entity_path_claimed ?mission_node)
        (intel_available ?intel_fragment)
      )
    :effect
      (and
        (entity_has_intel ?mission_node ?intel_fragment)
        (not
          (intel_available ?intel_fragment)
        )
      )
  )
  (:action mark_node_prepared
    :parameters (?mission_node - mission_node ?intel_fragment - intel_fragment)
    :precondition
      (and
        (entity_detected ?mission_node)
        (entity_path_claimed ?mission_node)
        (entity_has_intel ?mission_node ?intel_fragment)
        (not
          (entity_prepared ?mission_node)
        )
      )
    :effect (entity_prepared ?mission_node)
  )
  (:action release_intel_from_node
    :parameters (?mission_node - mission_node ?intel_fragment - intel_fragment)
    :precondition
      (and
        (entity_has_intel ?mission_node ?intel_fragment)
      )
    :effect
      (and
        (intel_available ?intel_fragment)
        (not
          (entity_has_intel ?mission_node ?intel_fragment)
        )
      )
  )
  (:action attach_equipment_to_node
    :parameters (?mission_node - mission_node ?equipment_item - equipment_item)
    :precondition
      (and
        (entity_prepared ?mission_node)
        (equipment_available ?equipment_item)
      )
    :effect
      (and
        (entity_equipment_attached ?mission_node ?equipment_item)
        (not
          (equipment_available ?equipment_item)
        )
      )
  )
  (:action detach_equipment_from_node
    :parameters (?mission_node - mission_node ?equipment_item - equipment_item)
    :precondition
      (and
        (entity_equipment_attached ?mission_node ?equipment_item)
      )
    :effect
      (and
        (equipment_available ?equipment_item)
        (not
          (entity_equipment_attached ?mission_node ?equipment_item)
        )
      )
  )
  (:action assign_primary_attachment_to_agent
    :parameters (?planner_agent - planner_agent ?primary_attachment - primary_attachment)
    :precondition
      (and
        (entity_prepared ?planner_agent)
        (primary_attachment_available ?primary_attachment)
      )
    :effect
      (and
        (agent_has_primary_attachment ?planner_agent ?primary_attachment)
        (not
          (primary_attachment_available ?primary_attachment)
        )
      )
  )
  (:action unassign_primary_attachment_from_agent
    :parameters (?planner_agent - planner_agent ?primary_attachment - primary_attachment)
    :precondition
      (and
        (agent_has_primary_attachment ?planner_agent ?primary_attachment)
      )
    :effect
      (and
        (primary_attachment_available ?primary_attachment)
        (not
          (agent_has_primary_attachment ?planner_agent ?primary_attachment)
        )
      )
  )
  (:action assign_secondary_attachment_to_agent
    :parameters (?planner_agent - planner_agent ?secondary_attachment - secondary_attachment)
    :precondition
      (and
        (entity_prepared ?planner_agent)
        (secondary_attachment_available ?secondary_attachment)
      )
    :effect
      (and
        (agent_has_secondary_attachment ?planner_agent ?secondary_attachment)
        (not
          (secondary_attachment_available ?secondary_attachment)
        )
      )
  )
  (:action unassign_secondary_attachment_from_agent
    :parameters (?planner_agent - planner_agent ?secondary_attachment - secondary_attachment)
    :precondition
      (and
        (agent_has_secondary_attachment ?planner_agent ?secondary_attachment)
      )
    :effect
      (and
        (secondary_attachment_available ?secondary_attachment)
        (not
          (agent_has_secondary_attachment ?planner_agent ?secondary_attachment)
        )
      )
  )
  (:action validate_recon_approach_vector
    :parameters (?recon_unit - recon_unit ?approach_vector_recon - approach_vector_recon ?intel_fragment - intel_fragment)
    :precondition
      (and
        (entity_prepared ?recon_unit)
        (entity_has_intel ?recon_unit ?intel_fragment)
        (approach_vector_recon_attached ?recon_unit ?approach_vector_recon)
        (not
          (approach_vector_recon_confirmed ?approach_vector_recon)
        )
        (not
          (approach_vector_recon_exhausted ?approach_vector_recon)
        )
      )
    :effect (approach_vector_recon_confirmed ?approach_vector_recon)
  )
  (:action finalize_recon_vector_for_recon_unit
    :parameters (?recon_unit - recon_unit ?approach_vector_recon - approach_vector_recon ?equipment_item - equipment_item)
    :precondition
      (and
        (entity_prepared ?recon_unit)
        (entity_equipment_attached ?recon_unit ?equipment_item)
        (approach_vector_recon_attached ?recon_unit ?approach_vector_recon)
        (approach_vector_recon_confirmed ?approach_vector_recon)
        (not
          (recon_unit_ready ?recon_unit)
        )
      )
    :effect
      (and
        (recon_unit_ready ?recon_unit)
        (recon_unit_confirmed ?recon_unit)
      )
  )
  (:action assign_consumable_to_recon_unit
    :parameters (?recon_unit - recon_unit ?approach_vector_recon - approach_vector_recon ?consumable - consumable)
    :precondition
      (and
        (entity_prepared ?recon_unit)
        (approach_vector_recon_attached ?recon_unit ?approach_vector_recon)
        (consumable_available ?consumable)
        (not
          (recon_unit_ready ?recon_unit)
        )
      )
    :effect
      (and
        (approach_vector_recon_exhausted ?approach_vector_recon)
        (recon_unit_ready ?recon_unit)
        (recon_unit_has_consumable ?recon_unit ?consumable)
        (not
          (consumable_available ?consumable)
        )
      )
  )
  (:action confirm_recon_vector_after_validation
    :parameters (?recon_unit - recon_unit ?approach_vector_recon - approach_vector_recon ?intel_fragment - intel_fragment ?consumable - consumable)
    :precondition
      (and
        (entity_prepared ?recon_unit)
        (entity_has_intel ?recon_unit ?intel_fragment)
        (approach_vector_recon_attached ?recon_unit ?approach_vector_recon)
        (approach_vector_recon_exhausted ?approach_vector_recon)
        (recon_unit_has_consumable ?recon_unit ?consumable)
        (not
          (recon_unit_confirmed ?recon_unit)
        )
      )
    :effect
      (and
        (approach_vector_recon_confirmed ?approach_vector_recon)
        (recon_unit_confirmed ?recon_unit)
        (consumable_available ?consumable)
        (not
          (recon_unit_has_consumable ?recon_unit ?consumable)
        )
      )
  )
  (:action validate_assault_approach_vector
    :parameters (?assault_unit - assault_unit ?approach_vector_assault - approach_vector_assault ?intel_fragment - intel_fragment)
    :precondition
      (and
        (entity_prepared ?assault_unit)
        (entity_has_intel ?assault_unit ?intel_fragment)
        (approach_vector_assault_attached ?assault_unit ?approach_vector_assault)
        (not
          (approach_vector_assault_confirmed ?approach_vector_assault)
        )
        (not
          (approach_vector_assault_exhausted ?approach_vector_assault)
        )
      )
    :effect (approach_vector_assault_confirmed ?approach_vector_assault)
  )
  (:action finalize_assault_vector_with_equipment
    :parameters (?assault_unit - assault_unit ?approach_vector_assault - approach_vector_assault ?equipment_item - equipment_item)
    :precondition
      (and
        (entity_prepared ?assault_unit)
        (entity_equipment_attached ?assault_unit ?equipment_item)
        (approach_vector_assault_attached ?assault_unit ?approach_vector_assault)
        (approach_vector_assault_confirmed ?approach_vector_assault)
        (not
          (assault_unit_ready ?assault_unit)
        )
      )
    :effect
      (and
        (assault_unit_ready ?assault_unit)
        (assault_unit_confirmed ?assault_unit)
      )
  )
  (:action assign_consumable_to_assault_unit
    :parameters (?assault_unit - assault_unit ?approach_vector_assault - approach_vector_assault ?consumable - consumable)
    :precondition
      (and
        (entity_prepared ?assault_unit)
        (approach_vector_assault_attached ?assault_unit ?approach_vector_assault)
        (consumable_available ?consumable)
        (not
          (assault_unit_ready ?assault_unit)
        )
      )
    :effect
      (and
        (approach_vector_assault_exhausted ?approach_vector_assault)
        (assault_unit_ready ?assault_unit)
        (assault_unit_has_consumable ?assault_unit ?consumable)
        (not
          (consumable_available ?consumable)
        )
      )
  )
  (:action confirm_assault_vector_after_validation
    :parameters (?assault_unit - assault_unit ?approach_vector_assault - approach_vector_assault ?intel_fragment - intel_fragment ?consumable - consumable)
    :precondition
      (and
        (entity_prepared ?assault_unit)
        (entity_has_intel ?assault_unit ?intel_fragment)
        (approach_vector_assault_attached ?assault_unit ?approach_vector_assault)
        (approach_vector_assault_exhausted ?approach_vector_assault)
        (assault_unit_has_consumable ?assault_unit ?consumable)
        (not
          (assault_unit_confirmed ?assault_unit)
        )
      )
    :effect
      (and
        (approach_vector_assault_confirmed ?approach_vector_assault)
        (assault_unit_confirmed ?assault_unit)
        (consumable_available ?consumable)
        (not
          (assault_unit_has_consumable ?assault_unit ?consumable)
        )
      )
  )
  (:action assemble_loadout
    :parameters (?recon_unit - recon_unit ?assault_unit - assault_unit ?approach_vector_recon - approach_vector_recon ?approach_vector_assault - approach_vector_assault ?loadout - loadout)
    :precondition
      (and
        (recon_unit_ready ?recon_unit)
        (assault_unit_ready ?assault_unit)
        (approach_vector_recon_attached ?recon_unit ?approach_vector_recon)
        (approach_vector_assault_attached ?assault_unit ?approach_vector_assault)
        (approach_vector_recon_confirmed ?approach_vector_recon)
        (approach_vector_assault_confirmed ?approach_vector_assault)
        (recon_unit_confirmed ?recon_unit)
        (assault_unit_confirmed ?assault_unit)
        (loadout_available ?loadout)
      )
    :effect
      (and
        (loadout_built ?loadout)
        (loadout_has_approach_vector_recon ?loadout ?approach_vector_recon)
        (loadout_has_approach_vector_assault ?loadout ?approach_vector_assault)
        (not
          (loadout_available ?loadout)
        )
      )
  )
  (:action assemble_loadout_recon_exhausted_variant
    :parameters (?recon_unit - recon_unit ?assault_unit - assault_unit ?approach_vector_recon - approach_vector_recon ?approach_vector_assault - approach_vector_assault ?loadout - loadout)
    :precondition
      (and
        (recon_unit_ready ?recon_unit)
        (assault_unit_ready ?assault_unit)
        (approach_vector_recon_attached ?recon_unit ?approach_vector_recon)
        (approach_vector_assault_attached ?assault_unit ?approach_vector_assault)
        (approach_vector_recon_exhausted ?approach_vector_recon)
        (approach_vector_assault_confirmed ?approach_vector_assault)
        (not
          (recon_unit_confirmed ?recon_unit)
        )
        (assault_unit_confirmed ?assault_unit)
        (loadout_available ?loadout)
      )
    :effect
      (and
        (loadout_built ?loadout)
        (loadout_has_approach_vector_recon ?loadout ?approach_vector_recon)
        (loadout_has_approach_vector_assault ?loadout ?approach_vector_assault)
        (loadout_recon_exhausted ?loadout)
        (not
          (loadout_available ?loadout)
        )
      )
  )
  (:action assemble_loadout_assault_exhausted_variant
    :parameters (?recon_unit - recon_unit ?assault_unit - assault_unit ?approach_vector_recon - approach_vector_recon ?approach_vector_assault - approach_vector_assault ?loadout - loadout)
    :precondition
      (and
        (recon_unit_ready ?recon_unit)
        (assault_unit_ready ?assault_unit)
        (approach_vector_recon_attached ?recon_unit ?approach_vector_recon)
        (approach_vector_assault_attached ?assault_unit ?approach_vector_assault)
        (approach_vector_recon_confirmed ?approach_vector_recon)
        (approach_vector_assault_exhausted ?approach_vector_assault)
        (recon_unit_confirmed ?recon_unit)
        (not
          (assault_unit_confirmed ?assault_unit)
        )
        (loadout_available ?loadout)
      )
    :effect
      (and
        (loadout_built ?loadout)
        (loadout_has_approach_vector_recon ?loadout ?approach_vector_recon)
        (loadout_has_approach_vector_assault ?loadout ?approach_vector_assault)
        (loadout_assault_exhausted ?loadout)
        (not
          (loadout_available ?loadout)
        )
      )
  )
  (:action assemble_loadout_both_exhausted_variant
    :parameters (?recon_unit - recon_unit ?assault_unit - assault_unit ?approach_vector_recon - approach_vector_recon ?approach_vector_assault - approach_vector_assault ?loadout - loadout)
    :precondition
      (and
        (recon_unit_ready ?recon_unit)
        (assault_unit_ready ?assault_unit)
        (approach_vector_recon_attached ?recon_unit ?approach_vector_recon)
        (approach_vector_assault_attached ?assault_unit ?approach_vector_assault)
        (approach_vector_recon_exhausted ?approach_vector_recon)
        (approach_vector_assault_exhausted ?approach_vector_assault)
        (not
          (recon_unit_confirmed ?recon_unit)
        )
        (not
          (assault_unit_confirmed ?assault_unit)
        )
        (loadout_available ?loadout)
      )
    :effect
      (and
        (loadout_built ?loadout)
        (loadout_has_approach_vector_recon ?loadout ?approach_vector_recon)
        (loadout_has_approach_vector_assault ?loadout ?approach_vector_assault)
        (loadout_recon_exhausted ?loadout)
        (loadout_assault_exhausted ?loadout)
        (not
          (loadout_available ?loadout)
        )
      )
  )
  (:action finalize_loadout
    :parameters (?loadout - loadout ?recon_unit - recon_unit ?intel_fragment - intel_fragment)
    :precondition
      (and
        (loadout_built ?loadout)
        (recon_unit_ready ?recon_unit)
        (entity_has_intel ?recon_unit ?intel_fragment)
        (not
          (loadout_finalized ?loadout)
        )
      )
    :effect (loadout_finalized ?loadout)
  )
  (:action install_attachment_into_loadout
    :parameters (?planner_agent - planner_agent ?attachment - attachment ?loadout - loadout)
    :precondition
      (and
        (entity_prepared ?planner_agent)
        (agent_has_loadout ?planner_agent ?loadout)
        (agent_has_attachment ?planner_agent ?attachment)
        (attachment_available ?attachment)
        (loadout_built ?loadout)
        (loadout_finalized ?loadout)
        (not
          (attachment_installed ?attachment)
        )
      )
    :effect
      (and
        (attachment_installed ?attachment)
        (attachment_attached_to_loadout ?attachment ?loadout)
        (not
          (attachment_available ?attachment)
        )
      )
  )
  (:action stage_agent_for_attachment_integration
    :parameters (?planner_agent - planner_agent ?attachment - attachment ?loadout - loadout ?intel_fragment - intel_fragment)
    :precondition
      (and
        (entity_prepared ?planner_agent)
        (agent_has_attachment ?planner_agent ?attachment)
        (attachment_installed ?attachment)
        (attachment_attached_to_loadout ?attachment ?loadout)
        (entity_has_intel ?planner_agent ?intel_fragment)
        (not
          (loadout_recon_exhausted ?loadout)
        )
        (not
          (agent_attachment_staged ?planner_agent)
        )
      )
    :effect (agent_attachment_staged ?planner_agent)
  )
  (:action assign_tactical_mod_to_agent
    :parameters (?planner_agent - planner_agent ?tactical_mod - tactical_mod)
    :precondition
      (and
        (entity_prepared ?planner_agent)
        (tactical_mod_available ?tactical_mod)
        (not
          (agent_tactical_mod_active ?planner_agent)
        )
      )
    :effect
      (and
        (agent_tactical_mod_active ?planner_agent)
        (agent_assigned_tactical_mod ?planner_agent ?tactical_mod)
        (not
          (tactical_mod_available ?tactical_mod)
        )
      )
  )
  (:action stage_agent_tactical_mod_with_loadout
    :parameters (?planner_agent - planner_agent ?attachment - attachment ?loadout - loadout ?intel_fragment - intel_fragment ?tactical_mod - tactical_mod)
    :precondition
      (and
        (entity_prepared ?planner_agent)
        (agent_has_attachment ?planner_agent ?attachment)
        (attachment_installed ?attachment)
        (attachment_attached_to_loadout ?attachment ?loadout)
        (entity_has_intel ?planner_agent ?intel_fragment)
        (loadout_recon_exhausted ?loadout)
        (agent_tactical_mod_active ?planner_agent)
        (agent_assigned_tactical_mod ?planner_agent ?tactical_mod)
        (not
          (agent_attachment_staged ?planner_agent)
        )
      )
    :effect
      (and
        (agent_attachment_staged ?planner_agent)
        (agent_mod_staged ?planner_agent)
      )
  )
  (:action lock_agent_primary_attachment
    :parameters (?planner_agent - planner_agent ?primary_attachment - primary_attachment ?equipment_item - equipment_item ?attachment - attachment ?loadout - loadout)
    :precondition
      (and
        (agent_attachment_staged ?planner_agent)
        (agent_has_primary_attachment ?planner_agent ?primary_attachment)
        (entity_equipment_attached ?planner_agent ?equipment_item)
        (agent_has_attachment ?planner_agent ?attachment)
        (attachment_attached_to_loadout ?attachment ?loadout)
        (not
          (loadout_assault_exhausted ?loadout)
        )
        (not
          (agent_attachment_locked ?planner_agent)
        )
      )
    :effect (agent_attachment_locked ?planner_agent)
  )
  (:action lock_agent_primary_attachment_alt
    :parameters (?planner_agent - planner_agent ?primary_attachment - primary_attachment ?equipment_item - equipment_item ?attachment - attachment ?loadout - loadout)
    :precondition
