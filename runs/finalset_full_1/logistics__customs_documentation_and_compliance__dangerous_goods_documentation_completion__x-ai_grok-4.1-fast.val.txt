--- domain.pddl (as passed to VAL) ---
(define (domain dangerous_goods_documentation_completion_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types stakeholder_role_root - object document_component_root - object document_category_root - object business_entity_root - object consignment_record - business_entity_root broker_entity - stakeholder_role_root classification_agent - stakeholder_role_root compliance_reviewer - stakeholder_role_root special_condition - stakeholder_role_root carrier_constraint - stakeholder_role_root permit_document - stakeholder_role_root regulatory_endorsement - stakeholder_role_root certification - stakeholder_role_root supporting_document - document_component_root packaging_specification - document_component_root regulatory_reference - document_component_root hazard_assessment - document_category_root packing_authorization - document_category_root documentation_set - document_category_root export_grouping - consignment_record case_grouping - consignment_record export_unit_segment - export_grouping import_unit_segment - export_grouping compliance_case - case_grouping)
  (:predicates
    (record_registered ?consignment - consignment_record)
    (classification_finalized ?consignment - consignment_record)
    (broker_engaged ?consignment - consignment_record)
    (compliance_cleared ?consignment - consignment_record)
    (compliance_ready ?consignment - consignment_record)
    (permit_attached_to ?consignment - consignment_record)
    (broker_available ?broker - broker_entity)
    (record_has_broker ?consignment - consignment_record ?broker - broker_entity)
    (classification_agent_available ?classification_agent - classification_agent)
    (classified_by ?consignment - consignment_record ?classification_agent - classification_agent)
    (reviewer_available ?reviewer - compliance_reviewer)
    (record_has_reviewer ?consignment - consignment_record ?reviewer - compliance_reviewer)
    (supporting_document_available ?supporting_document - supporting_document)
    (export_grouping_has_document ?export_segment - export_unit_segment ?supporting_document - supporting_document)
    (import_grouping_has_document ?import_segment - import_unit_segment ?supporting_document - supporting_document)
    (export_has_hazard_assessment ?export_segment - export_unit_segment ?hazard_assessment - hazard_assessment)
    (hazard_assessment_confirmed ?hazard_assessment - hazard_assessment)
    (hazard_assessment_documented ?hazard_assessment - hazard_assessment)
    (export_segment_reviewed ?export_segment - export_unit_segment)
    (import_has_packing_authorization ?import_segment - import_unit_segment ?packing_authorization - packing_authorization)
    (packing_authorization_confirmed ?packing_authorization - packing_authorization)
    (packing_authorization_documented ?packing_authorization - packing_authorization)
    (import_segment_reviewed ?import_segment - import_unit_segment)
    (documentation_set_available ?documentation_set - documentation_set)
    (documentation_set_selected ?documentation_set - documentation_set)
    (documentation_set_linked_to_hazard_assessment ?documentation_set - documentation_set ?hazard_assessment - hazard_assessment)
    (documentation_set_linked_to_packing_authorization ?documentation_set - documentation_set ?packing_authorization - packing_authorization)
    (documentation_set_requires_endorsement ?documentation_set - documentation_set)
    (documentation_set_requires_certification ?documentation_set - documentation_set)
    (documentation_set_validated ?documentation_set - documentation_set)
    (case_has_export_segment ?compliance_case - compliance_case ?export_segment - export_unit_segment)
    (case_has_import_segment ?compliance_case - compliance_case ?import_segment - import_unit_segment)
    (case_has_documentation_set ?compliance_case - compliance_case ?documentation_set - documentation_set)
    (packaging_spec_available ?packaging_spec - packaging_specification)
    (case_has_packaging_specification ?compliance_case - compliance_case ?packaging_spec - packaging_specification)
    (packaging_spec_attached ?packaging_spec - packaging_specification)
    (packaging_spec_linked_to_documentation_set ?packaging_spec - packaging_specification ?documentation_set - documentation_set)
    (packaging_specification_approved ?compliance_case - compliance_case)
    (specialist_approval_recorded ?compliance_case - compliance_case)
    (certification_bound_to_case ?compliance_case - compliance_case)
    (special_condition_applied ?compliance_case - compliance_case)
    (labeling_integrated ?compliance_case - compliance_case)
    (case_ready_for_carrier_constraints ?compliance_case - compliance_case)
    (case_final_checks_complete ?compliance_case - compliance_case)
    (regulatory_reference_available ?regulatory_reference - regulatory_reference)
    (case_has_regulatory_reference ?compliance_case - compliance_case ?regulatory_reference - regulatory_reference)
    (regulatory_reference_acknowledged ?compliance_case - compliance_case)
    (regulatory_reference_validated ?compliance_case - compliance_case)
    (regulatory_reference_certified ?compliance_case - compliance_case)
    (special_condition_available ?special_condition - special_condition)
    (case_has_special_condition ?compliance_case - compliance_case ?special_condition - special_condition)
    (carrier_constraint_available ?carrier_constraint - carrier_constraint)
    (case_has_carrier_constraint ?compliance_case - compliance_case ?carrier_constraint - carrier_constraint)
    (regulatory_endorsement_available ?regulatory_endorsement - regulatory_endorsement)
    (case_has_regulatory_endorsement ?compliance_case - compliance_case ?regulatory_endorsement - regulatory_endorsement)
    (certification_available ?certification - certification)
    (case_has_certification ?compliance_case - compliance_case ?certification - certification)
    (permit_document_available ?permit_document - permit_document)
    (record_has_permit ?consignment - consignment_record ?permit_document - permit_document)
    (export_segment_ready ?export_segment - export_unit_segment)
    (import_segment_ready ?import_segment - import_unit_segment)
    (case_internal_checks_completed ?compliance_case - compliance_case)
  )
  (:action register_consignment
    :parameters (?consignment - consignment_record)
    :precondition
      (and
        (not
          (record_registered ?consignment)
        )
        (not
          (compliance_cleared ?consignment)
        )
      )
    :effect (record_registered ?consignment)
  )
  (:action assign_broker_to_consignment
    :parameters (?consignment - consignment_record ?broker - broker_entity)
    :precondition
      (and
        (record_registered ?consignment)
        (not
          (broker_engaged ?consignment)
        )
        (broker_available ?broker)
      )
    :effect
      (and
        (broker_engaged ?consignment)
        (record_has_broker ?consignment ?broker)
        (not
          (broker_available ?broker)
        )
      )
  )
  (:action request_classification
    :parameters (?consignment - consignment_record ?classification_agent - classification_agent)
    :precondition
      (and
        (record_registered ?consignment)
        (broker_engaged ?consignment)
        (classification_agent_available ?classification_agent)
      )
    :effect
      (and
        (classified_by ?consignment ?classification_agent)
        (not
          (classification_agent_available ?classification_agent)
        )
      )
  )
  (:action finalize_classification
    :parameters (?consignment - consignment_record ?classification_agent - classification_agent)
    :precondition
      (and
        (record_registered ?consignment)
        (broker_engaged ?consignment)
        (classified_by ?consignment ?classification_agent)
        (not
          (classification_finalized ?consignment)
        )
      )
    :effect (classification_finalized ?consignment)
  )
  (:action release_classification_agent
    :parameters (?consignment - consignment_record ?classification_agent - classification_agent)
    :precondition
      (and
        (classified_by ?consignment ?classification_agent)
      )
    :effect
      (and
        (classification_agent_available ?classification_agent)
        (not
          (classified_by ?consignment ?classification_agent)
        )
      )
  )
  (:action assign_reviewer_to_consignment
    :parameters (?consignment - consignment_record ?reviewer - compliance_reviewer)
    :precondition
      (and
        (classification_finalized ?consignment)
        (reviewer_available ?reviewer)
      )
    :effect
      (and
        (record_has_reviewer ?consignment ?reviewer)
        (not
          (reviewer_available ?reviewer)
        )
      )
  )
  (:action release_reviewer_from_consignment
    :parameters (?consignment - consignment_record ?reviewer - compliance_reviewer)
    :precondition
      (and
        (record_has_reviewer ?consignment ?reviewer)
      )
    :effect
      (and
        (reviewer_available ?reviewer)
        (not
          (record_has_reviewer ?consignment ?reviewer)
        )
      )
  )
  (:action attach_regulatory_endorsement
    :parameters (?compliance_case - compliance_case ?regulatory_endorsement - regulatory_endorsement)
    :precondition
      (and
        (classification_finalized ?compliance_case)
        (regulatory_endorsement_available ?regulatory_endorsement)
      )
    :effect
      (and
        (case_has_regulatory_endorsement ?compliance_case ?regulatory_endorsement)
        (not
          (regulatory_endorsement_available ?regulatory_endorsement)
        )
      )
  )
  (:action detach_regulatory_endorsement
    :parameters (?compliance_case - compliance_case ?regulatory_endorsement - regulatory_endorsement)
    :precondition
      (and
        (case_has_regulatory_endorsement ?compliance_case ?regulatory_endorsement)
      )
    :effect
      (and
        (regulatory_endorsement_available ?regulatory_endorsement)
        (not
          (case_has_regulatory_endorsement ?compliance_case ?regulatory_endorsement)
        )
      )
  )
  (:action attach_certification
    :parameters (?compliance_case - compliance_case ?certification - certification)
    :precondition
      (and
        (classification_finalized ?compliance_case)
        (certification_available ?certification)
      )
    :effect
      (and
        (case_has_certification ?compliance_case ?certification)
        (not
          (certification_available ?certification)
        )
      )
  )
  (:action detach_certification
    :parameters (?compliance_case - compliance_case ?certification - certification)
    :precondition
      (and
        (case_has_certification ?compliance_case ?certification)
      )
    :effect
      (and
        (certification_available ?certification)
        (not
          (case_has_certification ?compliance_case ?certification)
        )
      )
  )
  (:action evaluate_hazard_for_packing
    :parameters (?export_segment - export_unit_segment ?hazard_assessment - hazard_assessment ?classification_agent - classification_agent)
    :precondition
      (and
        (classification_finalized ?export_segment)
        (classified_by ?export_segment ?classification_agent)
        (export_has_hazard_assessment ?export_segment ?hazard_assessment)
        (not
          (hazard_assessment_confirmed ?hazard_assessment)
        )
        (not
          (hazard_assessment_documented ?hazard_assessment)
        )
      )
    :effect (hazard_assessment_confirmed ?hazard_assessment)
  )
  (:action approve_export_segment
    :parameters (?export_segment - export_unit_segment ?hazard_assessment - hazard_assessment ?reviewer - compliance_reviewer)
    :precondition
      (and
        (classification_finalized ?export_segment)
        (record_has_reviewer ?export_segment ?reviewer)
        (export_has_hazard_assessment ?export_segment ?hazard_assessment)
        (hazard_assessment_confirmed ?hazard_assessment)
        (not
          (export_segment_ready ?export_segment)
        )
      )
    :effect
      (and
        (export_segment_ready ?export_segment)
        (export_segment_reviewed ?export_segment)
      )
  )
  (:action attach_supporting_document_to_export
    :parameters (?export_segment - export_unit_segment ?hazard_assessment - hazard_assessment ?supporting_document - supporting_document)
    :precondition
      (and
        (classification_finalized ?export_segment)
        (export_has_hazard_assessment ?export_segment ?hazard_assessment)
        (supporting_document_available ?supporting_document)
        (not
          (export_segment_ready ?export_segment)
        )
      )
    :effect
      (and
        (hazard_assessment_documented ?hazard_assessment)
        (export_segment_ready ?export_segment)
        (export_grouping_has_document ?export_segment ?supporting_document)
        (not
          (supporting_document_available ?supporting_document)
        )
      )
  )
  (:action confirm_hazard_assessment_with_document
    :parameters (?export_segment - export_unit_segment ?hazard_assessment - hazard_assessment ?classification_agent - classification_agent ?supporting_document - supporting_document)
    :precondition
      (and
        (classification_finalized ?export_segment)
        (classified_by ?export_segment ?classification_agent)
        (export_has_hazard_assessment ?export_segment ?hazard_assessment)
        (hazard_assessment_documented ?hazard_assessment)
        (export_grouping_has_document ?export_segment ?supporting_document)
        (not
          (export_segment_reviewed ?export_segment)
        )
      )
    :effect
      (and
        (hazard_assessment_confirmed ?hazard_assessment)
        (export_segment_reviewed ?export_segment)
        (supporting_document_available ?supporting_document)
        (not
          (export_grouping_has_document ?export_segment ?supporting_document)
        )
      )
  )
  (:action evaluate_import_hazard_for_packing
    :parameters (?import_segment - import_unit_segment ?packing_authorization - packing_authorization ?classification_agent - classification_agent)
    :precondition
      (and
        (classification_finalized ?import_segment)
        (classified_by ?import_segment ?classification_agent)
        (import_has_packing_authorization ?import_segment ?packing_authorization)
        (not
          (packing_authorization_confirmed ?packing_authorization)
        )
        (not
          (packing_authorization_documented ?packing_authorization)
        )
      )
    :effect (packing_authorization_confirmed ?packing_authorization)
  )
  (:action approve_import_segment
    :parameters (?import_segment - import_unit_segment ?packing_authorization - packing_authorization ?reviewer - compliance_reviewer)
    :precondition
      (and
        (classification_finalized ?import_segment)
        (record_has_reviewer ?import_segment ?reviewer)
        (import_has_packing_authorization ?import_segment ?packing_authorization)
        (packing_authorization_confirmed ?packing_authorization)
        (not
          (import_segment_ready ?import_segment)
        )
      )
    :effect
      (and
        (import_segment_ready ?import_segment)
        (import_segment_reviewed ?import_segment)
      )
  )
  (:action attach_supporting_document_to_import
    :parameters (?import_segment - import_unit_segment ?packing_authorization - packing_authorization ?supporting_document - supporting_document)
    :precondition
      (and
        (classification_finalized ?import_segment)
        (import_has_packing_authorization ?import_segment ?packing_authorization)
        (supporting_document_available ?supporting_document)
        (not
          (import_segment_ready ?import_segment)
        )
      )
    :effect
      (and
        (packing_authorization_documented ?packing_authorization)
        (import_segment_ready ?import_segment)
        (import_grouping_has_document ?import_segment ?supporting_document)
        (not
          (supporting_document_available ?supporting_document)
        )
      )
  )
  (:action confirm_import_hazard_with_document
    :parameters (?import_segment - import_unit_segment ?packing_authorization - packing_authorization ?classification_agent - classification_agent ?supporting_document - supporting_document)
    :precondition
      (and
        (classification_finalized ?import_segment)
        (classified_by ?import_segment ?classification_agent)
        (import_has_packing_authorization ?import_segment ?packing_authorization)
        (packing_authorization_documented ?packing_authorization)
        (import_grouping_has_document ?import_segment ?supporting_document)
        (not
          (import_segment_reviewed ?import_segment)
        )
      )
    :effect
      (and
        (packing_authorization_confirmed ?packing_authorization)
        (import_segment_reviewed ?import_segment)
        (supporting_document_available ?supporting_document)
        (not
          (import_grouping_has_document ?import_segment ?supporting_document)
        )
      )
  )
  (:action assemble_documentation_set_standard
    :parameters (?export_segment - export_unit_segment ?import_segment - import_unit_segment ?hazard_assessment - hazard_assessment ?packing_authorization - packing_authorization ?documentation_set - documentation_set)
    :precondition
      (and
        (export_segment_ready ?export_segment)
        (import_segment_ready ?import_segment)
        (export_has_hazard_assessment ?export_segment ?hazard_assessment)
        (import_has_packing_authorization ?import_segment ?packing_authorization)
        (hazard_assessment_confirmed ?hazard_assessment)
        (packing_authorization_confirmed ?packing_authorization)
        (export_segment_reviewed ?export_segment)
        (import_segment_reviewed ?import_segment)
        (documentation_set_available ?documentation_set)
      )
    :effect
      (and
        (documentation_set_selected ?documentation_set)
        (documentation_set_linked_to_hazard_assessment ?documentation_set ?hazard_assessment)
        (documentation_set_linked_to_packing_authorization ?documentation_set ?packing_authorization)
        (not
          (documentation_set_available ?documentation_set)
        )
      )
  )
  (:action assemble_documentation_set_with_endorsement
    :parameters (?export_segment - export_unit_segment ?import_segment - import_unit_segment ?hazard_assessment - hazard_assessment ?packing_authorization - packing_authorization ?documentation_set - documentation_set)
    :precondition
      (and
        (export_segment_ready ?export_segment)
        (import_segment_ready ?import_segment)
        (export_has_hazard_assessment ?export_segment ?hazard_assessment)
        (import_has_packing_authorization ?import_segment ?packing_authorization)
        (hazard_assessment_documented ?hazard_assessment)
        (packing_authorization_confirmed ?packing_authorization)
        (not
          (export_segment_reviewed ?export_segment)
        )
        (import_segment_reviewed ?import_segment)
        (documentation_set_available ?documentation_set)
      )
    :effect
      (and
        (documentation_set_selected ?documentation_set)
        (documentation_set_linked_to_hazard_assessment ?documentation_set ?hazard_assessment)
        (documentation_set_linked_to_packing_authorization ?documentation_set ?packing_authorization)
        (documentation_set_requires_endorsement ?documentation_set)
        (not
          (documentation_set_available ?documentation_set)
        )
      )
  )
  (:action assemble_documentation_set_with_certification
    :parameters (?export_segment - export_unit_segment ?import_segment - import_unit_segment ?hazard_assessment - hazard_assessment ?packing_authorization - packing_authorization ?documentation_set - documentation_set)
    :precondition
      (and
        (export_segment_ready ?export_segment)
        (import_segment_ready ?import_segment)
        (export_has_hazard_assessment ?export_segment ?hazard_assessment)
        (import_has_packing_authorization ?import_segment ?packing_authorization)
        (hazard_assessment_confirmed ?hazard_assessment)
        (packing_authorization_documented ?packing_authorization)
        (export_segment_reviewed ?export_segment)
        (not
          (import_segment_reviewed ?import_segment)
        )
        (documentation_set_available ?documentation_set)
      )
    :effect
      (and
        (documentation_set_selected ?documentation_set)
        (documentation_set_linked_to_hazard_assessment ?documentation_set ?hazard_assessment)
        (documentation_set_linked_to_packing_authorization ?documentation_set ?packing_authorization)
        (documentation_set_requires_certification ?documentation_set)
        (not
          (documentation_set_available ?documentation_set)
        )
      )
  )
  (:action assemble_documentation_set_with_endorsement_and_certification
    :parameters (?export_segment - export_unit_segment ?import_segment - import_unit_segment ?hazard_assessment - hazard_assessment ?packing_authorization - packing_authorization ?documentation_set - documentation_set)
    :precondition
      (and
        (export_segment_ready ?export_segment)
        (import_segment_ready ?import_segment)
        (export_has_hazard_assessment ?export_segment ?hazard_assessment)
        (import_has_packing_authorization ?import_segment ?packing_authorization)
        (hazard_assessment_documented ?hazard_assessment)
        (packing_authorization_documented ?packing_authorization)
        (not
          (export_segment_reviewed ?export_segment)
        )
        (not
          (import_segment_reviewed ?import_segment)
        )
        (documentation_set_available ?documentation_set)
      )
    :effect
      (and
        (documentation_set_selected ?documentation_set)
        (documentation_set_linked_to_hazard_assessment ?documentation_set ?hazard_assessment)
        (documentation_set_linked_to_packing_authorization ?documentation_set ?packing_authorization)
        (documentation_set_requires_endorsement ?documentation_set)
        (documentation_set_requires_certification ?documentation_set)
        (not
          (documentation_set_available ?documentation_set)
        )
      )
  )
  (:action validate_documentation_set
    :parameters (?documentation_set - documentation_set ?export_segment - export_unit_segment ?classification_agent - classification_agent)
    :precondition
      (and
        (documentation_set_selected ?documentation_set)
        (export_segment_ready ?export_segment)
        (classified_by ?export_segment ?classification_agent)
        (not
          (documentation_set_validated ?documentation_set)
        )
      )
    :effect (documentation_set_validated ?documentation_set)
  )
  (:action attach_packaging_specification_to_case
    :parameters (?compliance_case - compliance_case ?packaging_spec - packaging_specification ?documentation_set - documentation_set)
    :precondition
      (and
        (classification_finalized ?compliance_case)
        (case_has_documentation_set ?compliance_case ?documentation_set)
        (case_has_packaging_specification ?compliance_case ?packaging_spec)
        (packaging_spec_available ?packaging_spec)
        (documentation_set_selected ?documentation_set)
        (documentation_set_validated ?documentation_set)
        (not
          (packaging_spec_attached ?packaging_spec)
        )
      )
    :effect
      (and
        (packaging_spec_attached ?packaging_spec)
        (packaging_spec_linked_to_documentation_set ?packaging_spec ?documentation_set)
        (not
          (packaging_spec_available ?packaging_spec)
        )
      )
  )
  (:action validate_packaging_specification_for_case
    :parameters (?compliance_case - compliance_case ?packaging_spec - packaging_specification ?documentation_set - documentation_set ?classification_agent - classification_agent)
    :precondition
      (and
        (classification_finalized ?compliance_case)
        (case_has_packaging_specification ?compliance_case ?packaging_spec)
        (packaging_spec_attached ?packaging_spec)
        (packaging_spec_linked_to_documentation_set ?packaging_spec ?documentation_set)
        (classified_by ?compliance_case ?classification_agent)
        (not
          (documentation_set_requires_endorsement ?documentation_set)
        )
        (not
          (packaging_specification_approved ?compliance_case)
        )
      )
    :effect (packaging_specification_approved ?compliance_case)
  )
  (:action apply_special_condition_to_case
    :parameters (?compliance_case - compliance_case ?special_condition - special_condition)
    :precondition
      (and
        (classification_finalized ?compliance_case)
        (special_condition_available ?special_condition)
        (not
          (special_condition_applied ?compliance_case)
        )
      )
    :effect
      (and
        (special_condition_applied ?compliance_case)
        (case_has_special_condition ?compliance_case ?special_condition)
        (not
          (special_condition_available ?special_condition)
        )
      )
  )
  (:action integrate_packaging_and_special_condition_into_case
    :parameters (?compliance_case - compliance_case ?packaging_spec - packaging_specification ?documentation_set - documentation_set ?classification_agent - classification_agent ?special_condition - special_condition)
    :precondition
      (and
        (classification_finalized ?compliance_case)
        (case_has_packaging_specification ?compliance_case ?packaging_spec)
        (packaging_spec_attached ?packaging_spec)
        (packaging_spec_linked_to_documentation_set ?packaging_spec ?documentation_set)
        (classified_by ?compliance_case ?classification_agent)
        (documentation_set_requires_endorsement ?documentation_set)
        (special_condition_applied ?compliance_case)
        (case_has_special_condition ?compliance_case ?special_condition)
        (not
          (packaging_specification_approved ?compliance_case)
        )
      )
    :effect
      (and
        (packaging_specification_approved ?compliance_case)
        (labeling_integrated ?compliance_case)
      )
  )
  (:action record_specialist_approval
    :parameters (?compliance_case - compliance_case ?regulatory_endorsement - regulatory_endorsement ?reviewer - compliance_reviewer ?packaging_spec - packaging_specification ?documentation_set - documentation_set)
    :precondition
      (and
        (packaging_specification_approved ?compliance_case)
        (case_has_regulatory_endorsement ?compliance_case ?regulatory_endorsement)
        (record_has_reviewer ?compliance_case ?reviewer)
        (case_has_packaging_specification ?compliance_case ?packaging_spec)
        (packaging_spec_linked_to_documentation_set ?packaging_spec ?documentation_set)
        (not
          (documentation_set_requires_certification ?documentation_set)
        )
        (not
          (specialist_approval_recorded ?compliance_case)
        )
      )
    :effect (specialist_approval_recorded ?compliance_case)
  )
  (:action record_specialist_approval_alternative
    :parameters (?compliance_case - compliance_case ?regulatory_endorsement - regulatory_endorsement ?reviewer - compliance_reviewer ?packaging_spec - packaging_specification ?documentation_set - documentation_set)
    :precondition
      (and
        (packaging_specification_approved ?compliance_case)
        (case_has_regulatory_endorsement ?compliance_case ?regulatory_endorsement)
        (record_has_reviewer ?compliance_case ?reviewer)
        (case_has_packaging_specification ?compliance_case ?packaging_spec)
        (packaging_spec_linked_to_documentation_set ?packaging_spec ?documentation_set)
