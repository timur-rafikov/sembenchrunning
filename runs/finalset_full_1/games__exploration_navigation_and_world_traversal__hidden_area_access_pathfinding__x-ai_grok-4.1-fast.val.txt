--- domain.pddl (as passed to VAL) ---
(define (domain hidden_area_access_pathfinding)
  (:requirements :strips :typing :negative-preconditions)
  (:types deployable_asset - object interactable_generic_b - object path_component - object location_root - object location - location_root access_token - deployable_asset movement_ability - deployable_asset traversal_tool - deployable_asset landmark - deployable_asset path_modifier - deployable_asset hazard_marker - deployable_asset enhancement_module - deployable_asset access_condition - deployable_asset consumable_resource - interactable_generic_b scout_sensor - interactable_generic_b trigger_device - interactable_generic_b route_segment_a - path_component route_segment_b - path_component assembled_path - path_component location_subtype - location location_subtype2 - location approach_primary - location_subtype approach_secondary - location_subtype hidden_area - location_subtype2)
  (:predicates
    (discovered ?location - location)
    (prepared ?location - location)
    (token_bound ?location - location)
    (unlocked ?location - location)
    (validated ?location - location)
    (access_ready ?location - location)
    (token_available ?access_token - access_token)
    (has_token ?location - location ?access_token - access_token)
    (ability_available ?movement_ability - movement_ability)
    (has_ability ?location - location ?movement_ability - movement_ability)
    (tool_available ?traversal_tool - traversal_tool)
    (has_tool ?location - location ?traversal_tool - traversal_tool)
    (consumable_available ?consumable - consumable_resource)
    (primary_consumable_attached ?approach_primary - approach_primary ?consumable - consumable_resource)
    (secondary_consumable_attached ?approach_secondary - approach_secondary ?consumable - consumable_resource)
    (approach_has_segment_a ?approach_primary - approach_primary ?segment_a - route_segment_a)
    (segment_a_verified ?segment_a - route_segment_a)
    (segment_a_consumable_deployed ?segment_a - route_segment_a)
    (approach_primary_cleared ?approach_primary - approach_primary)
    (approach_has_segment_b ?approach_secondary - approach_secondary ?segment_b - route_segment_b)
    (segment_b_verified ?segment_b - route_segment_b)
    (segment_b_consumable_deployed ?segment_b - route_segment_b)
    (approach_secondary_cleared ?approach_secondary - approach_secondary)
    (path_available ?assembled_path - assembled_path)
    (path_claimed ?assembled_path - assembled_path)
    (path_has_segment_a ?assembled_path - assembled_path ?segment_a - route_segment_a)
    (path_has_segment_b ?assembled_path - assembled_path ?segment_b - route_segment_b)
    (path_flag_segment_a_consumable ?assembled_path - assembled_path)
    (path_flag_segment_b_consumable ?assembled_path - assembled_path)
    (path_validated ?assembled_path - assembled_path)
    (hidden_area_has_primary_approach ?hidden_area - hidden_area ?approach_primary - approach_primary)
    (hidden_area_has_secondary_approach ?hidden_area - hidden_area ?approach_secondary - approach_secondary)
    (hidden_area_has_path ?hidden_area - hidden_area ?assembled_path - assembled_path)
    (sensor_available ?scout_sensor - scout_sensor)
    (hidden_area_has_sensor ?hidden_area - hidden_area ?scout_sensor - scout_sensor)
    (sensor_deployed ?scout_sensor - scout_sensor)
    (sensor_assigned_to_path ?scout_sensor - scout_sensor ?assembled_path - assembled_path)
    (sensors_validated ?hidden_area - hidden_area)
    (sensors_stage2_passed ?hidden_area - hidden_area)
    (validation_ready ?hidden_area - hidden_area)
    (landmark_paired ?hidden_area - hidden_area)
    (landmark_confirmation_flag ?hidden_area - hidden_area)
    (modifier_attached ?hidden_area - hidden_area)
    (final_checks_passed ?hidden_area - hidden_area)
    (trigger_available ?trigger_device - trigger_device)
    (hidden_area_has_trigger ?hidden_area - hidden_area ?trigger_device - trigger_device)
    (trigger_paired ?hidden_area - hidden_area)
    (trigger_activated ?hidden_area - hidden_area)
    (trigger_activation_confirmed ?hidden_area - hidden_area)
    (landmark_available ?landmark - landmark)
    (hidden_area_has_landmark ?hidden_area - hidden_area ?landmark - landmark)
    (modifier_available ?path_modifier - path_modifier)
    (hidden_area_has_modifier ?hidden_area - hidden_area ?path_modifier - path_modifier)
    (enhancement_module_available ?enhancement_module - enhancement_module)
    (hidden_area_has_enhancement ?hidden_area - hidden_area ?enhancement_module - enhancement_module)
    (access_condition_available ?access_condition - access_condition)
    (hidden_area_has_access_condition ?hidden_area - hidden_area ?access_condition - access_condition)
    (hazard_marker_available ?hazard_marker - hazard_marker)
    (has_hazard_marker ?location - location ?hazard_marker - hazard_marker)
    (primary_approach_ready ?approach_primary - approach_primary)
    (secondary_approach_ready ?approach_secondary - approach_secondary)
    (finalization_marker_set ?hidden_area - hidden_area)
  )
  (:action discover_location
    :parameters (?location - location)
    :precondition
      (and
        (not
          (discovered ?location)
        )
        (not
          (unlocked ?location)
        )
      )
    :effect (discovered ?location)
  )
  (:action assign_access_token
    :parameters (?location - location ?access_token - access_token)
    :precondition
      (and
        (discovered ?location)
        (not
          (token_bound ?location)
        )
        (token_available ?access_token)
      )
    :effect
      (and
        (token_bound ?location)
        (has_token ?location ?access_token)
        (not
          (token_available ?access_token)
        )
      )
  )
  (:action assign_movement_ability
    :parameters (?location - location ?movement_ability - movement_ability)
    :precondition
      (and
        (discovered ?location)
        (token_bound ?location)
        (ability_available ?movement_ability)
      )
    :effect
      (and
        (has_ability ?location ?movement_ability)
        (not
          (ability_available ?movement_ability)
        )
      )
  )
  (:action equip_location_with_ability
    :parameters (?location - location ?movement_ability - movement_ability)
    :precondition
      (and
        (discovered ?location)
        (token_bound ?location)
        (has_ability ?location ?movement_ability)
        (not
          (prepared ?location)
        )
      )
    :effect (prepared ?location)
  )
  (:action release_movement_ability
    :parameters (?location - location ?movement_ability - movement_ability)
    :precondition
      (and
        (has_ability ?location ?movement_ability)
      )
    :effect
      (and
        (ability_available ?movement_ability)
        (not
          (has_ability ?location ?movement_ability)
        )
      )
  )
  (:action assign_traversal_tool
    :parameters (?location - location ?traversal_tool - traversal_tool)
    :precondition
      (and
        (prepared ?location)
        (tool_available ?traversal_tool)
      )
    :effect
      (and
        (has_tool ?location ?traversal_tool)
        (not
          (tool_available ?traversal_tool)
        )
      )
  )
  (:action release_traversal_tool
    :parameters (?location - location ?traversal_tool - traversal_tool)
    :precondition
      (and
        (has_tool ?location ?traversal_tool)
      )
    :effect
      (and
        (tool_available ?traversal_tool)
        (not
          (has_tool ?location ?traversal_tool)
        )
      )
  )
  (:action install_enhancement_module
    :parameters (?hidden_area - hidden_area ?enhancement_module - enhancement_module)
    :precondition
      (and
        (prepared ?hidden_area)
        (enhancement_module_available ?enhancement_module)
      )
    :effect
      (and
        (hidden_area_has_enhancement ?hidden_area ?enhancement_module)
        (not
          (enhancement_module_available ?enhancement_module)
        )
      )
  )
  (:action uninstall_enhancement_module
    :parameters (?hidden_area - hidden_area ?enhancement_module - enhancement_module)
    :precondition
      (and
        (hidden_area_has_enhancement ?hidden_area ?enhancement_module)
      )
    :effect
      (and
        (enhancement_module_available ?enhancement_module)
        (not
          (hidden_area_has_enhancement ?hidden_area ?enhancement_module)
        )
      )
  )
  (:action assign_access_condition
    :parameters (?hidden_area - hidden_area ?access_condition - access_condition)
    :precondition
      (and
        (prepared ?hidden_area)
        (access_condition_available ?access_condition)
      )
    :effect
      (and
        (hidden_area_has_access_condition ?hidden_area ?access_condition)
        (not
          (access_condition_available ?access_condition)
        )
      )
  )
  (:action remove_access_condition
    :parameters (?hidden_area - hidden_area ?access_condition - access_condition)
    :precondition
      (and
        (hidden_area_has_access_condition ?hidden_area ?access_condition)
      )
    :effect
      (and
        (access_condition_available ?access_condition)
        (not
          (hidden_area_has_access_condition ?hidden_area ?access_condition)
        )
      )
  )
  (:action scan_route_segment_a
    :parameters (?approach_primary - approach_primary ?segment_a - route_segment_a ?movement_ability - movement_ability)
    :precondition
      (and
        (prepared ?approach_primary)
        (has_ability ?approach_primary ?movement_ability)
        (approach_has_segment_a ?approach_primary ?segment_a)
        (not
          (segment_a_verified ?segment_a)
        )
        (not
          (segment_a_consumable_deployed ?segment_a)
        )
      )
    :effect (segment_a_verified ?segment_a)
  )
  (:action clear_segment_a_with_tool
    :parameters (?approach_primary - approach_primary ?segment_a - route_segment_a ?traversal_tool - traversal_tool)
    :precondition
      (and
        (prepared ?approach_primary)
        (has_tool ?approach_primary ?traversal_tool)
        (approach_has_segment_a ?approach_primary ?segment_a)
        (segment_a_verified ?segment_a)
        (not
          (primary_approach_ready ?approach_primary)
        )
      )
    :effect
      (and
        (primary_approach_ready ?approach_primary)
        (approach_primary_cleared ?approach_primary)
      )
  )
  (:action apply_consumable_to_segment_a
    :parameters (?approach_primary - approach_primary ?segment_a - route_segment_a ?consumable - consumable_resource)
    :precondition
      (and
        (prepared ?approach_primary)
        (approach_has_segment_a ?approach_primary ?segment_a)
        (consumable_available ?consumable)
        (not
          (primary_approach_ready ?approach_primary)
        )
      )
    :effect
      (and
        (segment_a_consumable_deployed ?segment_a)
        (primary_approach_ready ?approach_primary)
        (primary_consumable_attached ?approach_primary ?consumable)
        (not
          (consumable_available ?consumable)
        )
      )
  )
  (:action resolve_segment_a_with_consumable
    :parameters (?approach_primary - approach_primary ?segment_a - route_segment_a ?movement_ability - movement_ability ?consumable - consumable_resource)
    :precondition
      (and
        (prepared ?approach_primary)
        (has_ability ?approach_primary ?movement_ability)
        (approach_has_segment_a ?approach_primary ?segment_a)
        (segment_a_consumable_deployed ?segment_a)
        (primary_consumable_attached ?approach_primary ?consumable)
        (not
          (approach_primary_cleared ?approach_primary)
        )
      )
    :effect
      (and
        (segment_a_verified ?segment_a)
        (approach_primary_cleared ?approach_primary)
        (consumable_available ?consumable)
        (not
          (primary_consumable_attached ?approach_primary ?consumable)
        )
      )
  )
  (:action scan_route_segment_b
    :parameters (?approach_secondary - approach_secondary ?segment_b - route_segment_b ?movement_ability - movement_ability)
    :precondition
      (and
        (prepared ?approach_secondary)
        (has_ability ?approach_secondary ?movement_ability)
        (approach_has_segment_b ?approach_secondary ?segment_b)
        (not
          (segment_b_verified ?segment_b)
        )
        (not
          (segment_b_consumable_deployed ?segment_b)
        )
      )
    :effect (segment_b_verified ?segment_b)
  )
  (:action clear_segment_b_with_tool
    :parameters (?approach_secondary - approach_secondary ?segment_b - route_segment_b ?traversal_tool - traversal_tool)
    :precondition
      (and
        (prepared ?approach_secondary)
        (has_tool ?approach_secondary ?traversal_tool)
        (approach_has_segment_b ?approach_secondary ?segment_b)
        (segment_b_verified ?segment_b)
        (not
          (secondary_approach_ready ?approach_secondary)
        )
      )
    :effect
      (and
        (secondary_approach_ready ?approach_secondary)
        (approach_secondary_cleared ?approach_secondary)
      )
  )
  (:action apply_consumable_to_segment_b
    :parameters (?approach_secondary - approach_secondary ?segment_b - route_segment_b ?consumable - consumable_resource)
    :precondition
      (and
        (prepared ?approach_secondary)
        (approach_has_segment_b ?approach_secondary ?segment_b)
        (consumable_available ?consumable)
        (not
          (secondary_approach_ready ?approach_secondary)
        )
      )
    :effect
      (and
        (segment_b_consumable_deployed ?segment_b)
        (secondary_approach_ready ?approach_secondary)
        (secondary_consumable_attached ?approach_secondary ?consumable)
        (not
          (consumable_available ?consumable)
        )
      )
  )
  (:action resolve_segment_b_with_consumable
    :parameters (?approach_secondary - approach_secondary ?segment_b - route_segment_b ?movement_ability - movement_ability ?consumable - consumable_resource)
    :precondition
      (and
        (prepared ?approach_secondary)
        (has_ability ?approach_secondary ?movement_ability)
        (approach_has_segment_b ?approach_secondary ?segment_b)
        (segment_b_consumable_deployed ?segment_b)
        (secondary_consumable_attached ?approach_secondary ?consumable)
        (not
          (approach_secondary_cleared ?approach_secondary)
        )
      )
    :effect
      (and
        (segment_b_verified ?segment_b)
        (approach_secondary_cleared ?approach_secondary)
        (consumable_available ?consumable)
        (not
          (secondary_consumable_attached ?approach_secondary ?consumable)
        )
      )
  )
  (:action assemble_path_from_segments
    :parameters (?approach_primary - approach_primary ?approach_secondary - approach_secondary ?segment_a - route_segment_a ?segment_b - route_segment_b ?assembled_path - assembled_path)
    :precondition
      (and
        (primary_approach_ready ?approach_primary)
        (secondary_approach_ready ?approach_secondary)
        (approach_has_segment_a ?approach_primary ?segment_a)
        (approach_has_segment_b ?approach_secondary ?segment_b)
        (segment_a_verified ?segment_a)
        (segment_b_verified ?segment_b)
        (approach_primary_cleared ?approach_primary)
        (approach_secondary_cleared ?approach_secondary)
        (path_available ?assembled_path)
      )
    :effect
      (and
        (path_claimed ?assembled_path)
        (path_has_segment_a ?assembled_path ?segment_a)
        (path_has_segment_b ?assembled_path ?segment_b)
        (not
          (path_available ?assembled_path)
        )
      )
  )
  (:action assemble_path_with_a_consumable
    :parameters (?approach_primary - approach_primary ?approach_secondary - approach_secondary ?segment_a - route_segment_a ?segment_b - route_segment_b ?assembled_path - assembled_path)
    :precondition
      (and
        (primary_approach_ready ?approach_primary)
        (secondary_approach_ready ?approach_secondary)
        (approach_has_segment_a ?approach_primary ?segment_a)
        (approach_has_segment_b ?approach_secondary ?segment_b)
        (segment_a_consumable_deployed ?segment_a)
        (segment_b_verified ?segment_b)
        (not
          (approach_primary_cleared ?approach_primary)
        )
        (approach_secondary_cleared ?approach_secondary)
        (path_available ?assembled_path)
      )
    :effect
      (and
        (path_claimed ?assembled_path)
        (path_has_segment_a ?assembled_path ?segment_a)
        (path_has_segment_b ?assembled_path ?segment_b)
        (path_flag_segment_a_consumable ?assembled_path)
        (not
          (path_available ?assembled_path)
        )
      )
  )
  (:action assemble_path_with_b_consumable
    :parameters (?approach_primary - approach_primary ?approach_secondary - approach_secondary ?segment_a - route_segment_a ?segment_b - route_segment_b ?assembled_path - assembled_path)
    :precondition
      (and
        (primary_approach_ready ?approach_primary)
        (secondary_approach_ready ?approach_secondary)
        (approach_has_segment_a ?approach_primary ?segment_a)
        (approach_has_segment_b ?approach_secondary ?segment_b)
        (segment_a_verified ?segment_a)
        (segment_b_consumable_deployed ?segment_b)
        (approach_primary_cleared ?approach_primary)
        (not
          (approach_secondary_cleared ?approach_secondary)
        )
        (path_available ?assembled_path)
      )
    :effect
      (and
        (path_claimed ?assembled_path)
        (path_has_segment_a ?assembled_path ?segment_a)
        (path_has_segment_b ?assembled_path ?segment_b)
        (path_flag_segment_b_consumable ?assembled_path)
        (not
          (path_available ?assembled_path)
        )
      )
  )
  (:action assemble_path_with_both_consumables
    :parameters (?approach_primary - approach_primary ?approach_secondary - approach_secondary ?segment_a - route_segment_a ?segment_b - route_segment_b ?assembled_path - assembled_path)
    :precondition
      (and
        (primary_approach_ready ?approach_primary)
        (secondary_approach_ready ?approach_secondary)
        (approach_has_segment_a ?approach_primary ?segment_a)
        (approach_has_segment_b ?approach_secondary ?segment_b)
        (segment_a_consumable_deployed ?segment_a)
        (segment_b_consumable_deployed ?segment_b)
        (not
          (approach_primary_cleared ?approach_primary)
        )
        (not
          (approach_secondary_cleared ?approach_secondary)
        )
        (path_available ?assembled_path)
      )
    :effect
      (and
        (path_claimed ?assembled_path)
        (path_has_segment_a ?assembled_path ?segment_a)
        (path_has_segment_b ?assembled_path ?segment_b)
        (path_flag_segment_a_consumable ?assembled_path)
        (path_flag_segment_b_consumable ?assembled_path)
        (not
          (path_available ?assembled_path)
        )
      )
  )
  (:action validate_path
    :parameters (?assembled_path - assembled_path ?approach_primary - approach_primary ?movement_ability - movement_ability)
    :precondition
      (and
        (path_claimed ?assembled_path)
        (primary_approach_ready ?approach_primary)
        (has_ability ?approach_primary ?movement_ability)
        (not
          (path_validated ?assembled_path)
        )
      )
    :effect (path_validated ?assembled_path)
  )
  (:action deploy_sensor_to_path
    :parameters (?hidden_area - hidden_area ?scout_sensor - scout_sensor ?assembled_path - assembled_path)
    :precondition
      (and
        (prepared ?hidden_area)
        (hidden_area_has_path ?hidden_area ?assembled_path)
        (hidden_area_has_sensor ?hidden_area ?scout_sensor)
        (sensor_available ?scout_sensor)
        (path_claimed ?assembled_path)
        (path_validated ?assembled_path)
        (not
          (sensor_deployed ?scout_sensor)
        )
      )
    :effect
      (and
        (sensor_deployed ?scout_sensor)
        (sensor_assigned_to_path ?scout_sensor ?assembled_path)
        (not
          (sensor_available ?scout_sensor)
        )
      )
  )
  (:action confirm_sensor_on_hidden_area
    :parameters (?hidden_area - hidden_area ?scout_sensor - scout_sensor ?assembled_path - assembled_path ?movement_ability - movement_ability)
    :precondition
      (and
        (prepared ?hidden_area)
        (hidden_area_has_sensor ?hidden_area ?scout_sensor)
        (sensor_deployed ?scout_sensor)
        (sensor_assigned_to_path ?scout_sensor ?assembled_path)
        (has_ability ?hidden_area ?movement_ability)
        (not
          (path_flag_segment_a_consumable ?assembled_path)
        )
        (not
          (sensors_validated ?hidden_area)
        )
      )
    :effect (sensors_validated ?hidden_area)
  )
  (:action pair_landmark_with_hidden_area
    :parameters (?hidden_area - hidden_area ?landmark - landmark)
    :precondition
      (and
        (prepared ?hidden_area)
        (landmark_available ?landmark)
        (not
          (landmark_paired ?hidden_area)
        )
      )
    :effect
      (and
        (landmark_paired ?hidden_area)
        (hidden_area_has_landmark ?hidden_area ?landmark)
        (not
          (landmark_available ?landmark)
        )
      )
  )
  (:action confirm_hidden_area_with_landmark
    :parameters (?hidden_area - hidden_area ?scout_sensor - scout_sensor ?assembled_path - assembled_path ?movement_ability - movement_ability ?landmark - landmark)
    :precondition
      (and
        (prepared ?hidden_area)
        (hidden_area_has_sensor ?hidden_area ?scout_sensor)
        (sensor_deployed ?scout_sensor)
        (sensor_assigned_to_path ?scout_sensor ?assembled_path)
        (has_ability ?hidden_area ?movement_ability)
        (path_flag_segment_a_consumable ?assembled_path)
        (landmark_paired ?hidden_area)
        (hidden_area_has_landmark ?hidden_area ?landmark)
        (not
          (sensors_validated ?hidden_area)
        )
      )
    :effect
      (and
        (sensors_validated ?hidden_area)
        (landmark_confirmation_flag ?hidden_area)
      )
  )
  (:action install_enhancement_module_stage1
    :parameters (?hidden_area - hidden_area ?enhancement_module - enhancement_module ?traversal_tool - traversal_tool ?scout_sensor - scout_sensor ?assembled_path - assembled_path)
    :precondition
      (and
        (sensors_validated ?hidden_area)
        (hidden_area_has_enhancement ?hidden_area ?enhancement_module)
        (has_tool ?hidden_area ?traversal_tool)
        (hidden_area_has_sensor ?hidden_area ?scout_sensor)
        (sensor_assigned_to_path ?scout_sensor ?assembled_path)
        (not
          (path_flag_segment_b_consumable ?assembled_path)
        )
        (not
          (sensors_stage2_passed ?hidden_area)
        )
      )
    :effect (sensors_stage2_passed ?hidden_area)
  )
  (:action install_enhancement_module_stage2
    :parameters (?hidden_area - hidden_area ?enhancement_module - enhancement_module ?traversal_tool - traversal_tool ?scout_sensor - scout_sensor ?assembled_path - assembled_path)
    :precondition
      (and
        (sensors_validated ?hidden_area)
        (hidden_area_has_enhancement ?hidden_area ?enhancement_module)
        (has_tool ?hidden_area ?traversal_tool)
        (hidden_area_has_sensor ?hidden_area ?scout_sensor)
        (sensor_assigned_to_path ?scout_sensor ?assembled_path)
        (path_flag_segment_b_consumable ?assembled_path)
        (not
          (sensors_stage2_passed ?hidden_area)
        )
      )
    :effect (sensors_stage2_passed ?hidden_area)
  )
  (:action prepare_hidden_area_for_final_validation
    :parameters (?hidden_area - hidden_area ?access_condition - access_condition ?scout_sensor - scout_sensor ?assembled_path - assembled_path)
    :precondition
      (and
        (sensors_stage2_passed ?hidden_area)
        (hidden_area_has_access_condition ?hidden_area ?access_condition)
        (hidden_area_has_sensor ?hidden_area ?scout_sensor)
        (sensor_assigned_to_path ?scout_sensor ?assembled_path)
        (not
          (path_flag_segment_a_consumable ?assembled_path)
        )
        (not
          (path_flag_segment_b_consumable ?assembled_path)
        )
        (not
          (validation_ready ?hidden_area)
        )
      )
    :effect (validation_ready ?hidden_area)
  )
  (:action prepare_hidden_area_for_final_validation_with_a_flag
    :parameters (?hidden_area - hidden_area ?access_condition - access_condition ?scout_sensor - scout_sensor ?assembled_path - assembled_path)
    :precondition
      (and
        (sensors_stage2_passed ?hidden_area)
        (hidden_area_has_access_condition ?hidden_area ?access_condition)
        (hidden_area_has_sensor ?hidden_area ?scout_sensor)
        (sensor_assigned_to_path ?scout_sensor ?assembled_path)
        (path_flag_segment_a_consumable ?assembled_path)
        (not
          (path_flag_segment_b_consumable ?assembled_path)
        )
        (not
          (validation_ready ?hidden_area)
        )
      )
