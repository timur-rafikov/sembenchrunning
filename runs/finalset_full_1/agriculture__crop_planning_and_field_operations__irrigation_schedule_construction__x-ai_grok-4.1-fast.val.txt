--- domain.pddl (as passed to VAL) ---
(define (domain irrigation_schedule_construction)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_family - object asset_or_resource - object system_component - object farm_entity_family - object managed_field_unit - farm_entity_family water_source - resource_family schedule_slot - resource_family crew_team - resource_family irrigation_mode - resource_family operation_template - resource_family permit_record - resource_family equipment_model - resource_family regulatory_constraint - resource_family application_material - asset_or_resource sensor_node - asset_or_resource irrigation_strategy - asset_or_resource irrigation_zone - system_component zone_controller - system_component irrigation_job - system_component field_unit_group - managed_field_unit schedule_group - managed_field_unit primary_field_block - field_unit_group secondary_field_block - field_unit_group irrigation_schedule - schedule_group)
  (:predicates
    (managed_unit_registered ?managed_field_unit - managed_field_unit)
    (managed_unit_scheduled ?managed_field_unit - managed_field_unit)
    (managed_unit_has_assigned_water_source ?managed_field_unit - managed_field_unit)
    (irrigation_application_finalized ?managed_field_unit - managed_field_unit)
    (irrigation_executed ?managed_field_unit - managed_field_unit)
    (application_recorded ?managed_field_unit - managed_field_unit)
    (water_source_available ?water_source - water_source)
    (managed_unit_assigned_water_source ?managed_field_unit - managed_field_unit ?water_source - water_source)
    (schedule_slot_available ?schedule_slot - schedule_slot)
    (managed_unit_assigned_slot ?managed_field_unit - managed_field_unit ?schedule_slot - schedule_slot)
    (crew_team_available ?crew_team - crew_team)
    (managed_unit_assigned_crew ?managed_field_unit - managed_field_unit ?crew_team - crew_team)
    (material_available ?application_material - application_material)
    (primary_block_assigned_material ?primary_field_block - primary_field_block ?application_material - application_material)
    (secondary_block_assigned_material ?secondary_field_block - secondary_field_block ?application_material - application_material)
    (primary_block_has_zone ?primary_field_block - primary_field_block ?irrigation_zone - irrigation_zone)
    (irrigation_zone_ready ?irrigation_zone - irrigation_zone)
    (irrigation_zone_material_loaded ?irrigation_zone - irrigation_zone)
    (primary_block_ready ?primary_field_block - primary_field_block)
    (secondary_block_assigned_controller ?secondary_field_block - secondary_field_block ?zone_controller - zone_controller)
    (zone_controller_ready ?zone_controller - zone_controller)
    (zone_controller_material_loaded ?zone_controller - zone_controller)
    (secondary_block_ready ?secondary_field_block - secondary_field_block)
    (irrigation_job_draft ?irrigation_job - irrigation_job)
    (irrigation_job_queued ?irrigation_job - irrigation_job)
    (irrigation_job_assigned_zone ?irrigation_job - irrigation_job ?irrigation_zone - irrigation_zone)
    (irrigation_job_assigned_controller ?irrigation_job - irrigation_job ?zone_controller - zone_controller)
    (irrigation_job_route_primary ?irrigation_job - irrigation_job)
    (irrigation_job_route_secondary ?irrigation_job - irrigation_job)
    (irrigation_job_ready_for_sensor_binding ?irrigation_job - irrigation_job)
    (schedule_assigned_primary_block ?irrigation_schedule - irrigation_schedule ?primary_field_block - primary_field_block)
    (schedule_assigned_secondary_block ?irrigation_schedule - irrigation_schedule ?secondary_field_block - secondary_field_block)
    (schedule_has_job ?irrigation_schedule - irrigation_schedule ?irrigation_job - irrigation_job)
    (sensor_available ?sensor_node - sensor_node)
    (schedule_has_sensor ?irrigation_schedule - irrigation_schedule ?sensor_node - sensor_node)
    (sensor_enrolled ?sensor_node - sensor_node)
    (sensor_bound_to_job ?sensor_node - sensor_node ?irrigation_job - irrigation_job)
    (schedule_sensors_verified ?irrigation_schedule - irrigation_schedule)
    (schedule_validated ?irrigation_schedule - irrigation_schedule)
    (schedule_preflight_passed ?irrigation_schedule - irrigation_schedule)
    (schedule_has_irrigation_mode ?irrigation_schedule - irrigation_schedule)
    (schedule_mode_activated ?irrigation_schedule - irrigation_schedule)
    (schedule_metadata_attached ?irrigation_schedule - irrigation_schedule)
    (schedule_authorized_for_execution ?irrigation_schedule - irrigation_schedule)
    (irrigation_strategy_available ?irrigation_strategy - irrigation_strategy)
    (schedule_assigned_strategy ?irrigation_schedule - irrigation_schedule ?irrigation_strategy - irrigation_strategy)
    (schedule_has_strategy_assigned ?irrigation_schedule - irrigation_schedule)
    (schedule_strategy_confirmed ?irrigation_schedule - irrigation_schedule)
    (schedule_constraints_bound ?irrigation_schedule - irrigation_schedule)
    (irrigation_mode_available ?irrigation_mode - irrigation_mode)
    (schedule_assigned_irrigation_mode ?irrigation_schedule - irrigation_schedule ?irrigation_mode - irrigation_mode)
    (operation_template_available ?operation_template - operation_template)
    (schedule_assigned_operation_template ?irrigation_schedule - irrigation_schedule ?operation_template - operation_template)
    (equipment_model_available ?equipment_model - equipment_model)
    (schedule_assigned_equipment_model ?irrigation_schedule - irrigation_schedule ?equipment_model - equipment_model)
    (regulatory_constraint_available ?regulatory_constraint - regulatory_constraint)
    (schedule_assigned_regulatory_constraint ?irrigation_schedule - irrigation_schedule ?regulatory_constraint - regulatory_constraint)
    (permit_record_available ?permit_record - permit_record)
    (managed_unit_assigned_permit_record ?managed_field_unit - managed_field_unit ?permit_record - permit_record)
    (primary_block_prepared ?primary_field_block - primary_field_block)
    (secondary_block_prepared ?secondary_field_block - secondary_field_block)
    (schedule_finalized ?irrigation_schedule - irrigation_schedule)
  )
  (:action register_managed_field_unit
    :parameters (?managed_field_unit - managed_field_unit)
    :precondition
      (and
        (not
          (managed_unit_registered ?managed_field_unit)
        )
        (not
          (irrigation_application_finalized ?managed_field_unit)
        )
      )
    :effect (managed_unit_registered ?managed_field_unit)
  )
  (:action assign_water_source_to_field
    :parameters (?managed_field_unit - managed_field_unit ?water_source - water_source)
    :precondition
      (and
        (managed_unit_registered ?managed_field_unit)
        (not
          (managed_unit_has_assigned_water_source ?managed_field_unit)
        )
        (water_source_available ?water_source)
      )
    :effect
      (and
        (managed_unit_has_assigned_water_source ?managed_field_unit)
        (managed_unit_assigned_water_source ?managed_field_unit ?water_source)
        (not
          (water_source_available ?water_source)
        )
      )
  )
  (:action reserve_schedule_slot_for_field
    :parameters (?managed_field_unit - managed_field_unit ?schedule_slot - schedule_slot)
    :precondition
      (and
        (managed_unit_registered ?managed_field_unit)
        (managed_unit_has_assigned_water_source ?managed_field_unit)
        (schedule_slot_available ?schedule_slot)
      )
    :effect
      (and
        (managed_unit_assigned_slot ?managed_field_unit ?schedule_slot)
        (not
          (schedule_slot_available ?schedule_slot)
        )
      )
  )
  (:action confirm_field_schedule_slot
    :parameters (?managed_field_unit - managed_field_unit ?schedule_slot - schedule_slot)
    :precondition
      (and
        (managed_unit_registered ?managed_field_unit)
        (managed_unit_has_assigned_water_source ?managed_field_unit)
        (managed_unit_assigned_slot ?managed_field_unit ?schedule_slot)
        (not
          (managed_unit_scheduled ?managed_field_unit)
        )
      )
    :effect (managed_unit_scheduled ?managed_field_unit)
  )
  (:action release_field_schedule_slot
    :parameters (?managed_field_unit - managed_field_unit ?schedule_slot - schedule_slot)
    :precondition
      (and
        (managed_unit_assigned_slot ?managed_field_unit ?schedule_slot)
      )
    :effect
      (and
        (schedule_slot_available ?schedule_slot)
        (not
          (managed_unit_assigned_slot ?managed_field_unit ?schedule_slot)
        )
      )
  )
  (:action reserve_crew_team_for_field
    :parameters (?managed_field_unit - managed_field_unit ?crew_team - crew_team)
    :precondition
      (and
        (managed_unit_scheduled ?managed_field_unit)
        (crew_team_available ?crew_team)
      )
    :effect
      (and
        (managed_unit_assigned_crew ?managed_field_unit ?crew_team)
        (not
          (crew_team_available ?crew_team)
        )
      )
  )
  (:action release_crew_team_from_field
    :parameters (?managed_field_unit - managed_field_unit ?crew_team - crew_team)
    :precondition
      (and
        (managed_unit_assigned_crew ?managed_field_unit ?crew_team)
      )
    :effect
      (and
        (crew_team_available ?crew_team)
        (not
          (managed_unit_assigned_crew ?managed_field_unit ?crew_team)
        )
      )
  )
  (:action reserve_equipment_model_for_schedule
    :parameters (?irrigation_schedule - irrigation_schedule ?equipment_model - equipment_model)
    :precondition
      (and
        (managed_unit_scheduled ?irrigation_schedule)
        (equipment_model_available ?equipment_model)
      )
    :effect
      (and
        (schedule_assigned_equipment_model ?irrigation_schedule ?equipment_model)
        (not
          (equipment_model_available ?equipment_model)
        )
      )
  )
  (:action release_equipment_model_from_schedule
    :parameters (?irrigation_schedule - irrigation_schedule ?equipment_model - equipment_model)
    :precondition
      (and
        (schedule_assigned_equipment_model ?irrigation_schedule ?equipment_model)
      )
    :effect
      (and
        (equipment_model_available ?equipment_model)
        (not
          (schedule_assigned_equipment_model ?irrigation_schedule ?equipment_model)
        )
      )
  )
  (:action reserve_regulatory_constraint_for_schedule
    :parameters (?irrigation_schedule - irrigation_schedule ?regulatory_constraint - regulatory_constraint)
    :precondition
      (and
        (managed_unit_scheduled ?irrigation_schedule)
        (regulatory_constraint_available ?regulatory_constraint)
      )
    :effect
      (and
        (schedule_assigned_regulatory_constraint ?irrigation_schedule ?regulatory_constraint)
        (not
          (regulatory_constraint_available ?regulatory_constraint)
        )
      )
  )
  (:action release_regulatory_constraint_from_schedule
    :parameters (?irrigation_schedule - irrigation_schedule ?regulatory_constraint - regulatory_constraint)
    :precondition
      (and
        (schedule_assigned_regulatory_constraint ?irrigation_schedule ?regulatory_constraint)
      )
    :effect
      (and
        (regulatory_constraint_available ?regulatory_constraint)
        (not
          (schedule_assigned_regulatory_constraint ?irrigation_schedule ?regulatory_constraint)
        )
      )
  )
  (:action mark_zone_ready_for_primary_block
    :parameters (?primary_field_block - primary_field_block ?irrigation_zone - irrigation_zone ?schedule_slot - schedule_slot)
    :precondition
      (and
        (managed_unit_scheduled ?primary_field_block)
        (managed_unit_assigned_slot ?primary_field_block ?schedule_slot)
        (primary_block_has_zone ?primary_field_block ?irrigation_zone)
        (not
          (irrigation_zone_ready ?irrigation_zone)
        )
        (not
          (irrigation_zone_material_loaded ?irrigation_zone)
        )
      )
    :effect (irrigation_zone_ready ?irrigation_zone)
  )
  (:action assign_crew_to_primary_block_zone
    :parameters (?primary_field_block - primary_field_block ?irrigation_zone - irrigation_zone ?crew_team - crew_team)
    :precondition
      (and
        (managed_unit_scheduled ?primary_field_block)
        (managed_unit_assigned_crew ?primary_field_block ?crew_team)
        (primary_block_has_zone ?primary_field_block ?irrigation_zone)
        (irrigation_zone_ready ?irrigation_zone)
        (not
          (primary_block_prepared ?primary_field_block)
        )
      )
    :effect
      (and
        (primary_block_prepared ?primary_field_block)
        (primary_block_ready ?primary_field_block)
      )
  )
  (:action load_material_for_primary_block_zone
    :parameters (?primary_field_block - primary_field_block ?irrigation_zone - irrigation_zone ?application_material - application_material)
    :precondition
      (and
        (managed_unit_scheduled ?primary_field_block)
        (primary_block_has_zone ?primary_field_block ?irrigation_zone)
        (material_available ?application_material)
        (not
          (primary_block_prepared ?primary_field_block)
        )
      )
    :effect
      (and
        (irrigation_zone_material_loaded ?irrigation_zone)
        (primary_block_prepared ?primary_field_block)
        (primary_block_assigned_material ?primary_field_block ?application_material)
        (not
          (material_available ?application_material)
        )
      )
  )
  (:action apply_material_to_primary_block_zone
    :parameters (?primary_field_block - primary_field_block ?irrigation_zone - irrigation_zone ?schedule_slot - schedule_slot ?application_material - application_material)
    :precondition
      (and
        (managed_unit_scheduled ?primary_field_block)
        (managed_unit_assigned_slot ?primary_field_block ?schedule_slot)
        (primary_block_has_zone ?primary_field_block ?irrigation_zone)
        (irrigation_zone_material_loaded ?irrigation_zone)
        (primary_block_assigned_material ?primary_field_block ?application_material)
        (not
          (primary_block_ready ?primary_field_block)
        )
      )
    :effect
      (and
        (irrigation_zone_ready ?irrigation_zone)
        (primary_block_ready ?primary_field_block)
        (material_available ?application_material)
        (not
          (primary_block_assigned_material ?primary_field_block ?application_material)
        )
      )
  )
  (:action mark_controller_ready_for_secondary_block
    :parameters (?secondary_field_block - secondary_field_block ?zone_controller - zone_controller ?schedule_slot - schedule_slot)
    :precondition
      (and
        (managed_unit_scheduled ?secondary_field_block)
        (managed_unit_assigned_slot ?secondary_field_block ?schedule_slot)
        (secondary_block_assigned_controller ?secondary_field_block ?zone_controller)
        (not
          (zone_controller_ready ?zone_controller)
        )
        (not
          (zone_controller_material_loaded ?zone_controller)
        )
      )
    :effect (zone_controller_ready ?zone_controller)
  )
  (:action assign_crew_to_secondary_block_controller
    :parameters (?secondary_field_block - secondary_field_block ?zone_controller - zone_controller ?crew_team - crew_team)
    :precondition
      (and
        (managed_unit_scheduled ?secondary_field_block)
        (managed_unit_assigned_crew ?secondary_field_block ?crew_team)
        (secondary_block_assigned_controller ?secondary_field_block ?zone_controller)
        (zone_controller_ready ?zone_controller)
        (not
          (secondary_block_prepared ?secondary_field_block)
        )
      )
    :effect
      (and
        (secondary_block_prepared ?secondary_field_block)
        (secondary_block_ready ?secondary_field_block)
      )
  )
  (:action load_material_for_secondary_block_controller
    :parameters (?secondary_field_block - secondary_field_block ?zone_controller - zone_controller ?application_material - application_material)
    :precondition
      (and
        (managed_unit_scheduled ?secondary_field_block)
        (secondary_block_assigned_controller ?secondary_field_block ?zone_controller)
        (material_available ?application_material)
        (not
          (secondary_block_prepared ?secondary_field_block)
        )
      )
    :effect
      (and
        (zone_controller_material_loaded ?zone_controller)
        (secondary_block_prepared ?secondary_field_block)
        (secondary_block_assigned_material ?secondary_field_block ?application_material)
        (not
          (material_available ?application_material)
        )
      )
  )
  (:action apply_material_to_secondary_block_controller
    :parameters (?secondary_field_block - secondary_field_block ?zone_controller - zone_controller ?schedule_slot - schedule_slot ?application_material - application_material)
    :precondition
      (and
        (managed_unit_scheduled ?secondary_field_block)
        (managed_unit_assigned_slot ?secondary_field_block ?schedule_slot)
        (secondary_block_assigned_controller ?secondary_field_block ?zone_controller)
        (zone_controller_material_loaded ?zone_controller)
        (secondary_block_assigned_material ?secondary_field_block ?application_material)
        (not
          (secondary_block_ready ?secondary_field_block)
        )
      )
    :effect
      (and
        (zone_controller_ready ?zone_controller)
        (secondary_block_ready ?secondary_field_block)
        (material_available ?application_material)
        (not
          (secondary_block_assigned_material ?secondary_field_block ?application_material)
        )
      )
  )
  (:action assemble_irrigation_job
    :parameters (?primary_field_block - primary_field_block ?secondary_field_block - secondary_field_block ?irrigation_zone - irrigation_zone ?zone_controller - zone_controller ?irrigation_job - irrigation_job)
    :precondition
      (and
        (primary_block_prepared ?primary_field_block)
        (secondary_block_prepared ?secondary_field_block)
        (primary_block_has_zone ?primary_field_block ?irrigation_zone)
        (secondary_block_assigned_controller ?secondary_field_block ?zone_controller)
        (irrigation_zone_ready ?irrigation_zone)
        (zone_controller_ready ?zone_controller)
        (primary_block_ready ?primary_field_block)
        (secondary_block_ready ?secondary_field_block)
        (irrigation_job_draft ?irrigation_job)
      )
    :effect
      (and
        (irrigation_job_queued ?irrigation_job)
        (irrigation_job_assigned_zone ?irrigation_job ?irrigation_zone)
        (irrigation_job_assigned_controller ?irrigation_job ?zone_controller)
        (not
          (irrigation_job_draft ?irrigation_job)
        )
      )
  )
  (:action assemble_irrigation_job_primary_route
    :parameters (?primary_field_block - primary_field_block ?secondary_field_block - secondary_field_block ?irrigation_zone - irrigation_zone ?zone_controller - zone_controller ?irrigation_job - irrigation_job)
    :precondition
      (and
        (primary_block_prepared ?primary_field_block)
        (secondary_block_prepared ?secondary_field_block)
        (primary_block_has_zone ?primary_field_block ?irrigation_zone)
        (secondary_block_assigned_controller ?secondary_field_block ?zone_controller)
        (irrigation_zone_material_loaded ?irrigation_zone)
        (zone_controller_ready ?zone_controller)
        (not
          (primary_block_ready ?primary_field_block)
        )
        (secondary_block_ready ?secondary_field_block)
        (irrigation_job_draft ?irrigation_job)
      )
    :effect
      (and
        (irrigation_job_queued ?irrigation_job)
        (irrigation_job_assigned_zone ?irrigation_job ?irrigation_zone)
        (irrigation_job_assigned_controller ?irrigation_job ?zone_controller)
        (irrigation_job_route_primary ?irrigation_job)
        (not
          (irrigation_job_draft ?irrigation_job)
        )
      )
  )
  (:action assemble_irrigation_job_secondary_route
    :parameters (?primary_field_block - primary_field_block ?secondary_field_block - secondary_field_block ?irrigation_zone - irrigation_zone ?zone_controller - zone_controller ?irrigation_job - irrigation_job)
    :precondition
      (and
        (primary_block_prepared ?primary_field_block)
        (secondary_block_prepared ?secondary_field_block)
        (primary_block_has_zone ?primary_field_block ?irrigation_zone)
        (secondary_block_assigned_controller ?secondary_field_block ?zone_controller)
        (irrigation_zone_ready ?irrigation_zone)
        (zone_controller_material_loaded ?zone_controller)
        (primary_block_ready ?primary_field_block)
        (not
          (secondary_block_ready ?secondary_field_block)
        )
        (irrigation_job_draft ?irrigation_job)
      )
    :effect
      (and
        (irrigation_job_queued ?irrigation_job)
        (irrigation_job_assigned_zone ?irrigation_job ?irrigation_zone)
        (irrigation_job_assigned_controller ?irrigation_job ?zone_controller)
        (irrigation_job_route_secondary ?irrigation_job)
        (not
          (irrigation_job_draft ?irrigation_job)
        )
      )
  )
  (:action assemble_irrigation_job_both_routes
    :parameters (?primary_field_block - primary_field_block ?secondary_field_block - secondary_field_block ?irrigation_zone - irrigation_zone ?zone_controller - zone_controller ?irrigation_job - irrigation_job)
    :precondition
      (and
        (primary_block_prepared ?primary_field_block)
        (secondary_block_prepared ?secondary_field_block)
        (primary_block_has_zone ?primary_field_block ?irrigation_zone)
        (secondary_block_assigned_controller ?secondary_field_block ?zone_controller)
        (irrigation_zone_material_loaded ?irrigation_zone)
        (zone_controller_material_loaded ?zone_controller)
        (not
          (primary_block_ready ?primary_field_block)
        )
        (not
          (secondary_block_ready ?secondary_field_block)
        )
        (irrigation_job_draft ?irrigation_job)
      )
    :effect
      (and
        (irrigation_job_queued ?irrigation_job)
        (irrigation_job_assigned_zone ?irrigation_job ?irrigation_zone)
        (irrigation_job_assigned_controller ?irrigation_job ?zone_controller)
        (irrigation_job_route_primary ?irrigation_job)
        (irrigation_job_route_secondary ?irrigation_job)
        (not
          (irrigation_job_draft ?irrigation_job)
        )
      )
  )
  (:action mark_job_ready_for_sensor_binding
    :parameters (?irrigation_job - irrigation_job ?primary_field_block - primary_field_block ?schedule_slot - schedule_slot)
    :precondition
      (and
        (irrigation_job_queued ?irrigation_job)
        (primary_block_prepared ?primary_field_block)
        (managed_unit_assigned_slot ?primary_field_block ?schedule_slot)
        (not
          (irrigation_job_ready_for_sensor_binding ?irrigation_job)
        )
      )
    :effect (irrigation_job_ready_for_sensor_binding ?irrigation_job)
  )
  (:action enroll_sensor_node_to_job
    :parameters (?irrigation_schedule - irrigation_schedule ?sensor_node - sensor_node ?irrigation_job - irrigation_job)
    :precondition
      (and
        (managed_unit_scheduled ?irrigation_schedule)
        (schedule_has_job ?irrigation_schedule ?irrigation_job)
        (schedule_has_sensor ?irrigation_schedule ?sensor_node)
        (sensor_available ?sensor_node)
        (irrigation_job_queued ?irrigation_job)
        (irrigation_job_ready_for_sensor_binding ?irrigation_job)
        (not
          (sensor_enrolled ?sensor_node)
        )
      )
    :effect
      (and
        (sensor_enrolled ?sensor_node)
        (sensor_bound_to_job ?sensor_node ?irrigation_job)
        (not
          (sensor_available ?sensor_node)
        )
      )
  )
  (:action verify_sensor_binding_for_schedule
    :parameters (?irrigation_schedule - irrigation_schedule ?sensor_node - sensor_node ?irrigation_job - irrigation_job ?schedule_slot - schedule_slot)
    :precondition
      (and
        (managed_unit_scheduled ?irrigation_schedule)
        (schedule_has_sensor ?irrigation_schedule ?sensor_node)
        (sensor_enrolled ?sensor_node)
        (sensor_bound_to_job ?sensor_node ?irrigation_job)
        (managed_unit_assigned_slot ?irrigation_schedule ?schedule_slot)
        (not
          (irrigation_job_route_primary ?irrigation_job)
        )
        (not
          (schedule_sensors_verified ?irrigation_schedule)
        )
      )
    :effect (schedule_sensors_verified ?irrigation_schedule)
  )
  (:action assign_irrigation_mode_to_schedule
    :parameters (?irrigation_schedule - irrigation_schedule ?irrigation_mode - irrigation_mode)
    :precondition
      (and
        (managed_unit_scheduled ?irrigation_schedule)
        (irrigation_mode_available ?irrigation_mode)
        (not
          (schedule_has_irrigation_mode ?irrigation_schedule)
        )
      )
    :effect
      (and
        (schedule_has_irrigation_mode ?irrigation_schedule)
        (schedule_assigned_irrigation_mode ?irrigation_schedule ?irrigation_mode)
        (not
          (irrigation_mode_available ?irrigation_mode)
        )
      )
  )
  (:action apply_irrigation_mode_and_enrich_schedule
    :parameters (?irrigation_schedule - irrigation_schedule ?sensor_node - sensor_node ?irrigation_job - irrigation_job ?schedule_slot - schedule_slot ?irrigation_mode - irrigation_mode)
    :precondition
      (and
        (managed_unit_scheduled ?irrigation_schedule)
        (schedule_has_sensor ?irrigation_schedule ?sensor_node)
        (sensor_enrolled ?sensor_node)
        (sensor_bound_to_job ?sensor_node ?irrigation_job)
