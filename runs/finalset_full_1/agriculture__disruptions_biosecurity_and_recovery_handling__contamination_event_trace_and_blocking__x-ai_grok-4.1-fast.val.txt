(define (domain agriculture_contamination_trace_and_blocking)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_category - object material_category - object logistics_category - object site_type - object agricultural_unit - site_type inspection_team - resource_category sample - resource_category field_officer - resource_category authorization_permit - resource_category logistics_slot - resource_category disposal_facility - resource_category decon_equipment - resource_category clearance_protocol - resource_category decon_kit - material_category test_kit - material_category external_consultant - material_category contamination_zone - logistics_category trace_endpoint - logistics_category remediation_package - logistics_category site_subtype - agricultural_unit site_subtype2 - agricultural_unit producer_node - site_subtype distributor_node - site_subtype response_center - site_subtype2)
  (:predicates
    (contamination_reported ?agricultural_unit - agricultural_unit)
    (contamination_confirmed ?agricultural_unit - agricultural_unit)
    (inspection_assigned ?agricultural_unit - agricultural_unit)
    (unit_cleared ?agricultural_unit - agricultural_unit)
    (eligible_for_clearance ?agricultural_unit - agricultural_unit)
    (clearance_authorized ?agricultural_unit - agricultural_unit)
    (inspection_team_available ?inspection_team - inspection_team)
    (team_assigned ?agricultural_unit - agricultural_unit ?inspection_team - inspection_team)
    (sample_available ?sample - sample)
    (sample_linked ?agricultural_unit - agricultural_unit ?sample - sample)
    (field_officer_available ?field_officer - field_officer)
    (field_officer_assigned ?agricultural_unit - agricultural_unit ?field_officer - field_officer)
    (decon_kit_available ?decon_kit - decon_kit)
    (decon_kit_assigned_to_producer ?producer_node - producer_node ?decon_kit - decon_kit)
    (decon_kit_assigned_to_distributor ?distributor_node - distributor_node ?decon_kit - decon_kit)
    (producer_zone_association ?producer_node - producer_node ?contamination_zone - contamination_zone)
    (zone_primary ?contamination_zone - contamination_zone)
    (zone_secondary ?contamination_zone - contamination_zone)
    (producer_verified ?producer_node - producer_node)
    (distributor_endpoint_link ?distributor_node - distributor_node ?trace_endpoint - trace_endpoint)
    (endpoint_primary ?trace_endpoint - trace_endpoint)
    (endpoint_secondary ?trace_endpoint - trace_endpoint)
    (distributor_verified ?distributor_node - distributor_node)
    (remediation_package_available ?remediation_package - remediation_package)
    (remediation_package_reserved ?remediation_package - remediation_package)
    (remediation_package_assigned_to_zone ?remediation_package - remediation_package ?contamination_zone - contamination_zone)
    (remediation_package_assigned_to_endpoint ?remediation_package - remediation_package ?trace_endpoint - trace_endpoint)
    (remediation_package_ready_stage_1 ?remediation_package - remediation_package)
    (remediation_package_ready_stage_2 ?remediation_package - remediation_package)
    (remediation_package_deployment_executed ?remediation_package - remediation_package)
    (center_producer_link ?response_center - response_center ?producer_node - producer_node)
    (center_distributor_link ?response_center - response_center ?distributor_node - distributor_node)
    (center_remediation_package_link ?response_center - response_center ?remediation_package - remediation_package)
    (testkit_available ?test_kit - test_kit)
    (center_has_testkit ?response_center - response_center ?test_kit - test_kit)
    (testkit_consumed ?test_kit - test_kit)
    (testkit_assigned_to_remediation_package ?test_kit - test_kit ?remediation_package - remediation_package)
    (center_lab_results_recorded ?response_center - response_center)
    (center_review_pending ?response_center - response_center)
    (center_review_complete ?response_center - response_center)
    (center_permit_assigned ?response_center - response_center)
    (center_authorization_flag ?response_center - response_center)
    (center_additional_validation ?response_center - response_center)
    (center_deployment_ready ?response_center - response_center)
    (consultant_available ?external_consultant - external_consultant)
    (center_consultant_link ?response_center - response_center ?external_consultant - external_consultant)
    (consultant_acknowledged ?response_center - response_center)
    (officer_review_recorded ?response_center - response_center)
    (protocol_review_recorded ?response_center - response_center)
    (permit_available ?authorization_permit - authorization_permit)
    (center_permit_link ?response_center - response_center ?authorization_permit - authorization_permit)
    (logistics_slot_available ?logistics_slot - logistics_slot)
    (center_logistics_slot_assigned ?response_center - response_center ?logistics_slot - logistics_slot)
    (decon_equipment_available ?decon_equipment - decon_equipment)
    (center_decon_equipment_assigned ?response_center - response_center ?decon_equipment - decon_equipment)
    (clearance_protocol_available ?clearance_protocol - clearance_protocol)
    (center_clearance_protocol_assigned ?response_center - response_center ?clearance_protocol - clearance_protocol)
    (disposal_facility_available ?disposal_facility - disposal_facility)
    (unit_assigned_disposal_facility ?agricultural_unit - agricultural_unit ?disposal_facility - disposal_facility)
    (producer_prepared ?producer_node - producer_node)
    (distributor_prepared ?distributor_node - distributor_node)
    (center_case_closed ?response_center - response_center)
  )
  (:action report_suspected_contamination
    :parameters (?agricultural_unit - agricultural_unit)
    :precondition
      (and
        (not
          (contamination_reported ?agricultural_unit)
        )
        (not
          (unit_cleared ?agricultural_unit)
        )
      )
    :effect (contamination_reported ?agricultural_unit)
  )
  (:action dispatch_inspection_team
    :parameters (?agricultural_unit - agricultural_unit ?inspection_team - inspection_team)
    :precondition
      (and
        (contamination_reported ?agricultural_unit)
        (not
          (inspection_assigned ?agricultural_unit)
        )
        (inspection_team_available ?inspection_team)
      )
    :effect
      (and
        (inspection_assigned ?agricultural_unit)
        (team_assigned ?agricultural_unit ?inspection_team)
        (not
          (inspection_team_available ?inspection_team)
        )
      )
  )
  (:action associate_sample_with_unit
    :parameters (?agricultural_unit - agricultural_unit ?sample - sample)
    :precondition
      (and
        (contamination_reported ?agricultural_unit)
        (inspection_assigned ?agricultural_unit)
        (sample_available ?sample)
      )
    :effect
      (and
        (sample_linked ?agricultural_unit ?sample)
        (not
          (sample_available ?sample)
        )
      )
  )
  (:action confirm_contamination
    :parameters (?agricultural_unit - agricultural_unit ?sample - sample)
    :precondition
      (and
        (contamination_reported ?agricultural_unit)
        (inspection_assigned ?agricultural_unit)
        (sample_linked ?agricultural_unit ?sample)
        (not
          (contamination_confirmed ?agricultural_unit)
        )
      )
    :effect (contamination_confirmed ?agricultural_unit)
  )
  (:action release_sample_to_pool
    :parameters (?agricultural_unit - agricultural_unit ?sample - sample)
    :precondition
      (and
        (sample_linked ?agricultural_unit ?sample)
      )
    :effect
      (and
        (sample_available ?sample)
        (not
          (sample_linked ?agricultural_unit ?sample)
        )
      )
  )
  (:action assign_field_officer_to_unit
    :parameters (?agricultural_unit - agricultural_unit ?field_officer - field_officer)
    :precondition
      (and
        (contamination_confirmed ?agricultural_unit)
        (field_officer_available ?field_officer)
      )
    :effect
      (and
        (field_officer_assigned ?agricultural_unit ?field_officer)
        (not
          (field_officer_available ?field_officer)
        )
      )
  )
  (:action release_field_officer_from_unit
    :parameters (?agricultural_unit - agricultural_unit ?field_officer - field_officer)

      (and
        (field_officer_assigned ?agricultural_unit ?field_officer)
      )
    :effect
      (and
        (field_officer_available ?field_officer)
        (not
          (field_officer_assigned ?agricultural_unit ?field_officer)
        )
      )
  )
  (:action assign_decon_equipment_to_center
    :parameters (?response_center - response_center ?decon_equipment - decon_equipment)

  (:action assign_clearance_protocol_to_center
    :parameters (?response_center - response_center ?clearance_protocol - clearance_protocol)

      (and

        )
      )

  (:action release_clearance_protocol_from_center
    :parameters (?response_center - response_center ?clearance_protocol - clearance_protocol)
    :precondition
      (and
        (center_clearance_protocol_assigned ?response_center ?clearance_protocol)
      )
    :effect
      (and
        (clearance_protocol_available ?clearance_protocol)
        (not
          (center_clearance_protocol_assigned ?response_center ?clearance_protocol)
        )
      )
  )
  (:action classify_zone_primary
    :parameters (?producer_node - producer_node ?contamination_zone - contamination_zone ?sample - sample)
    :precondition
      (and
        (contamination_confirmed ?producer_node)
        (sample_linked ?producer_node ?sample)
        (producer_zone_association ?producer_node ?contamination_zone)
        (not
          (zone_primary ?contamination_zone)
        )
        (not
          (zone_secondary ?contamination_zone)
        )
      )
    :effect (zone_primary ?contamination_zone)
  )
  (:action verify_producer_for_remediation
    :parameters (?producer_node - producer_node ?contamination_zone - contamination_zone ?field_officer - field_officer)
    :precondition
      (and
        (contamination_confirmed ?producer_node)
        (field_officer_assigned ?producer_node ?field_officer)
        (producer_zone_association ?producer_node ?contamination_zone)
        (zone_primary ?contamination_zone)
        (not
          (producer_prepared ?producer_node)
        )
      )
    :effect
      (and
        (producer_prepared ?producer_node)
        (producer_verified ?producer_node)
      )
  )
  (:action deploy_decon_kit_to_producer
    :parameters (?producer_node - producer_node ?contamination_zone - contamination_zone ?decon_kit - decon_kit)
    :precondition
      (and
        (contamination_confirmed ?producer_node)
        (producer_zone_association ?producer_node ?contamination_zone)
        (decon_kit_available ?decon_kit)
        (not
          (producer_prepared ?producer_node)
        )
      )
    :effect
      (and
        (zone_secondary ?contamination_zone)
        (producer_prepared ?producer_node)
        (decon_kit_assigned_to_producer ?producer_node ?decon_kit)
        (not
          (decon_kit_available ?decon_kit)
        )
      )
  )
  (:action finalize_producer_zone_status
    :parameters (?producer_node - producer_node ?contamination_zone - contamination_zone ?sample - sample ?decon_kit - decon_kit)

        (decon_kit_assigned_to_producer ?producer_node ?decon_kit)
        (not
          (producer_verified ?producer_node)
        )
      )
    :effect
      (and
        (zone_primary ?contamination_zone)
        (producer_verified ?producer_node)
        (decon_kit_available ?decon_kit)
        (not
          (decon_kit_assigned_to_producer ?producer_node ?decon_kit)

        (decon_kit_assigned_to_distributor ?distributor_node ?decon_kit)
        (not
          (decon_kit_available ?decon_kit)
        )
      )
  )
  (:action finalize_distributor_zone_status
    :parameters (?distributor_node - distributor_node ?trace_endpoint - trace_endpoint ?sample - sample ?decon_kit - decon_kit)
    :precondition
      (and
        (contamination_confirmed ?distributor_node)

          (distributor_verified ?distributor_node)

    :effect

        (endpoint_primary ?trace_endpoint)

  (:action assemble_remediation_package_primary
    :parameters (?producer_node - producer_node ?distributor_node - distributor_node ?contamination_zone - contamination_zone ?trace_endpoint - trace_endpoint ?remediation_package - remediation_package)
    :precondition
      (and
        (producer_prepared ?producer_node)
        (distributor_prepared ?distributor_node)
        (producer_zone_association ?producer_node ?contamination_zone)
        (distributor_endpoint_link ?distributor_node ?trace_endpoint)
        (zone_primary ?contamination_zone)
        (endpoint_primary ?trace_endpoint)
        (producer_verified ?producer_node)
        (distributor_verified ?distributor_node)
        (remediation_package_available ?remediation_package)
      )
    :effect
      (and
        (remediation_package_reserved ?remediation_package)
        (remediation_package_assigned_to_zone ?remediation_package ?contamination_zone)
        (remediation_package_assigned_to_endpoint ?remediation_package ?trace_endpoint)
        (not
          (remediation_package_available ?remediation_package)
        )
      )
  )
  (:action assemble_remediation_package_secondary_zone
    :parameters (?producer_node - producer_node ?distributor_node - distributor_node ?contamination_zone - contamination_zone ?trace_endpoint - trace_endpoint ?remediation_package - remediation_package)
    :precondition
      (and
        (producer_prepared ?producer_node)
        (distributor_prepared ?distributor_node)
        (producer_zone_association ?producer_node ?contamination_zone)
        (distributor_endpoint_link ?distributor_node ?trace_endpoint)
        (zone_secondary ?contamination_zone)
        (endpoint_primary ?trace_endpoint)
        (not
          (producer_verified ?producer_node)

        (distributor_verified ?distributor_node)
        (remediation_package_available ?remediation_package)
      )
    :effect
      (and
        (remediation_package_reserved ?remediation_package)
        (remediation_package_assigned_to_zone ?remediation_package ?contamination_zone)
        (remediation_package_assigned_to_endpoint ?remediation_package ?trace_endpoint)

          (remediation_package_available ?remediation_package)

        (distributor_endpoint_link ?distributor_node ?trace_endpoint)

        (endpoint_secondary ?trace_endpoint)
        (producer_verified ?producer_node)
        (not
          (distributor_verified ?distributor_node)

  (:action assemble_remediation_package_full_secondary
    :parameters (?producer_node - producer_node ?distributor_node - distributor_node ?contamination_zone - contamination_zone ?trace_endpoint - trace_endpoint ?remediation_package - remediation_package)
    :precondition
      (and
        (producer_prepared ?producer_node)
        (distributor_prepared ?distributor_node)
        (producer_zone_association ?producer_node ?contamination_zone)

        (remediation_package_ready_stage_2 ?remediation_package)
        (not
          (remediation_package_available ?remediation_package)

  (:action mark_package_ready_for_execution
    :parameters (?remediation_package - remediation_package ?producer_node - producer_node ?sample - sample)

      (and
        (remediation_package_reserved ?remediation_package)
        (producer_prepared ?producer_node)

          (remediation_package_deployment_executed ?remediation_package)
        )
      )
    :effect (remediation_package_deployment_executed ?remediation_package)
  )
  (:action consume_testkit_and_assign

          (testkit_available ?test_kit)
        )
      )
  )

        )
        (not
          (center_lab_results_recorded ?response_center)

        (testkit_consumed ?test_kit)
        (testkit_assigned_to_remediation_package ?test_kit ?remediation_package)

        (remediation_package_ready_stage_1 ?remediation_package)
        (center_permit_assigned ?response_center)
        (center_permit_link ?response_center ?authorization_permit)
        (not
          (center_lab_results_recorded ?response_center)
        )

    :effect
      (and
        (center_lab_results_recorded ?response_center)
        (center_authorization_flag ?response_center)
      )
  )
  (:action prepare_center_for_consultant_review
    :parameters (?response_center - response_center ?decon_equipment - decon_equipment ?field_officer - field_officer ?test_kit - test_kit ?remediation_package - remediation_package)
    :precondition
      (and

        (center_has_testkit ?response_center ?test_kit)

        (not
          (remediation_package_ready_stage_2 ?remediation_package)

          (center_review_pending ?response_center)

  (:action prepare_center_for_consultant_review_variant
    :parameters (?response_center - response_center ?decon_equipment - decon_equipment ?field_officer - field_officer ?test_kit - test_kit ?remediation_package - remediation_package)
