    (declaration_permit_record_available ?declaration_permit_record - declaration_permit_record)
    (export_declaration_permit_record_attached ?export_declaration_var - export_declaration ?declaration_permit_record - declaration_permit_record)
    (import_declaration_permit_record_attached ?import_declaration_var - import_declaration ?declaration_permit_record - declaration_permit_record)
    (export_declaration_origin_jurisdiction ?export_declaration_var - export_declaration ?origin_jurisdiction - origin_jurisdiction)
    (origin_requirement_marked ?origin_jurisdiction - origin_jurisdiction)
    (origin_permit_attached ?origin_jurisdiction - origin_jurisdiction)
    (export_declaration_requirements_satisfied ?export_declaration_var - export_declaration)
    (import_declaration_destination_jurisdiction ?import_declaration_var - import_declaration ?destination_jurisdiction - destination_jurisdiction)

    (submission_has_destination_jurisdiction ?clearance_submission - clearance_submission ?destination_jurisdiction - destination_jurisdiction)
    (submission_origin_permit_required ?clearance_submission - clearance_submission)
    (submission_destination_permit_required ?clearance_submission - clearance_submission)
    (submission_ready ?clearance_submission - clearance_submission)
    (ticket_linked_to_export_declaration ?escalation_ticket - escalation_ticket ?export_declaration_var - export_declaration)
    (ticket_linked_to_import_declaration ?escalation_ticket - escalation_ticket ?import_declaration_var - import_declaration)
    (ticket_linked_to_submission ?escalation_ticket - escalation_ticket ?clearance_submission - clearance_submission)
    (checklist_available ?compliance_checklist - compliance_checklist)
    (ticket_has_checklist ?escalation_ticket - escalation_ticket ?compliance_checklist - compliance_checklist)
    (checklist_attached ?compliance_checklist - compliance_checklist)

    (attachment_verified ?escalation_ticket - escalation_ticket)
    (ticket_prioritized_for_escalation ?escalation_ticket - escalation_ticket)
    (escalation_executed ?escalation_ticket - escalation_ticket)
    (external_contact_available ?external_regulatory_contact - external_regulatory_contact)
    (external_contact_linked ?escalation_ticket - escalation_ticket ?external_regulatory_contact - external_regulatory_contact)
    (external_contact_engaged ?escalation_ticket - escalation_ticket)
    (external_query_sent ?escalation_ticket - escalation_ticket)
    (external_response_received ?escalation_ticket - escalation_ticket)
    (attachment_type_available ?evidence_attachment_type - evidence_attachment_type)
    (ticket_has_attachment_type ?escalation_ticket - escalation_ticket ?evidence_attachment_type - evidence_attachment_type)

    (ticket_permit_license_record_assigned ?escalation_ticket - escalation_ticket ?ticket_permit_license_record - ticket_permit_license_record)

  (:action register_clearance_case
    :parameters (?clearance_case - clearance_case)
    :precondition
      (and
        (not
          (case_registered ?clearance_case)
        )
        (not
          (clearance_finalized ?clearance_case)
        )
      )
    :effect (case_registered ?clearance_case)
  )
  (:action allocate_service_provider_to_case
    :parameters (?clearance_case - clearance_case ?clearance_service_provider - clearance_service_provider)
    :precondition
      (and
        (case_registered ?clearance_case)
        (not
          (service_provider_assigned ?clearance_case)
        )
        (service_provider_available ?clearance_service_provider)
      )
    :effect
      (and
        (service_provider_assigned ?clearance_case)
        (service_provider_assigned_to_case ?clearance_case ?clearance_service_provider)
        (not
          (service_provider_available ?clearance_service_provider)
        )
      )
  )
  (:action link_supporting_document_to_case
    :parameters (?clearance_case - clearance_case ?supporting_document_var - supporting_document)
    :precondition
      (and
        (case_registered ?clearance_case)
        (service_provider_assigned ?clearance_case)
        (document_available ?supporting_document_var)
      )
    :effect
      (and
        (case_document_linked ?clearance_case ?supporting_document_var)
        (not
          (document_available ?supporting_document_var)
        )
      )
  )
  (:action confirm_document_classification
    :parameters (?clearance_case - clearance_case ?supporting_document_var - supporting_document)
    :precondition
      (and
        (case_registered ?clearance_case)
        (service_provider_assigned ?clearance_case)
        (case_document_linked ?clearance_case ?supporting_document_var)
        (not
          (classification_completed ?clearance_case)
        )
      )
    :effect (classification_completed ?clearance_case)
  )
  (:action unlink_supporting_document_from_case
    :parameters (?clearance_case - clearance_case ?supporting_document_var - supporting_document)
    :precondition
      (and
        (case_document_linked ?clearance_case ?supporting_document_var)
      )
    :effect
      (and
        (document_available ?supporting_document_var)
        (not
          (case_document_linked ?clearance_case ?supporting_document_var)
        )
      )
  )
  (:action assign_reviewer
    :parameters (?clearance_case - clearance_case ?reviewer - reviewer)
    :precondition
      (and
        (classification_completed ?clearance_case)
        (reviewer_available ?reviewer)
      )
    :effect
      (and
        (reviewer_assigned ?clearance_case ?reviewer)
        (not
          (reviewer_available ?reviewer)
        )
      )
  )
  (:action unassign_reviewer
    :parameters (?clearance_case - clearance_case ?reviewer - reviewer)
    :precondition
      (and
        (reviewer_assigned ?clearance_case ?reviewer)
      )
    :effect
      (and
        (reviewer_available ?reviewer)
        (not
          (reviewer_assigned ?clearance_case ?reviewer)
        )
      )
  )
  (:action attach_permit_license_to_ticket
    :parameters (?escalation_ticket - escalation_ticket ?ticket_permit_license_record - ticket_permit_license_record)
    :precondition
      (and
        (classification_completed ?escalation_ticket)
        (ticket_permit_license_record_available ?ticket_permit_license_record)
      )
    :effect
      (and
        (ticket_permit_license_record_assigned ?escalation_ticket ?ticket_permit_license_record)
        (not
          (ticket_permit_license_record_available ?ticket_permit_license_record)
        )
      )
  )
  (:action detach_permit_license_from_ticket
    :parameters (?escalation_ticket - escalation_ticket ?ticket_permit_license_record - ticket_permit_license_record)
    :precondition
      (and
        (ticket_permit_license_record_assigned ?escalation_ticket ?ticket_permit_license_record)
      )
    :effect
      (and
        (ticket_permit_license_record_available ?ticket_permit_license_record)
        (not
          (ticket_permit_license_record_assigned ?escalation_ticket ?ticket_permit_license_record)
        )
      )
  )
  (:action attach_regulatory_query_template_to_ticket
    :parameters (?escalation_ticket - escalation_ticket ?regulatory_query_template - regulatory_query_template)
    :precondition
      (and
        (classification_completed ?escalation_ticket)
        (query_template_available ?regulatory_query_template)
      )
    :effect
      (and
        (ticket_has_regulatory_query_template ?escalation_ticket ?regulatory_query_template)
        (not
          (query_template_available ?regulatory_query_template)
        )
      )
  )
  (:action detach_regulatory_query_template_from_ticket
    :parameters (?escalation_ticket - escalation_ticket ?regulatory_query_template - regulatory_query_template)
    :precondition
      (and
        (ticket_has_regulatory_query_template ?escalation_ticket ?regulatory_query_template)
      )
    :effect
      (and
        (query_template_available ?regulatory_query_template)
        (not
          (ticket_has_regulatory_query_template ?escalation_ticket ?regulatory_query_template)
        )
      )
  )
  (:action initiate_origin_requirement_check
    :parameters (?export_declaration_var - export_declaration ?origin_jurisdiction - origin_jurisdiction ?supporting_document_var - supporting_document)
    :precondition
      (and
        (classification_completed ?export_declaration_var)
        (case_document_linked ?export_declaration_var ?supporting_document_var)
        (export_declaration_origin_jurisdiction ?export_declaration_var ?origin_jurisdiction)
        (not
          (origin_requirement_marked ?origin_jurisdiction)
        )
        (not
          (origin_permit_attached ?origin_jurisdiction)
        )
      )
    :effect (origin_requirement_marked ?origin_jurisdiction)
  )
  (:action assign_reviewer_for_origin_requirement
    :parameters (?export_declaration_var - export_declaration ?origin_jurisdiction - origin_jurisdiction ?reviewer - reviewer)
    :precondition
      (and
        (classification_completed ?export_declaration_var)
        (reviewer_assigned ?export_declaration_var ?reviewer)
        (export_declaration_origin_jurisdiction ?export_declaration_var ?origin_jurisdiction)
        (origin_requirement_marked ?origin_jurisdiction)
        (not
          (export_declaration_ready ?export_declaration_var)
        )
      )
    :effect
      (and
        (export_declaration_ready ?export_declaration_var)
        (export_declaration_requirements_satisfied ?export_declaration_var)
      )
  )
  (:action request_permit_for_export_declaration
    :parameters (?export_declaration_var - export_declaration ?origin_jurisdiction - origin_jurisdiction ?declaration_permit_record - declaration_permit_record)
    :precondition
      (and
        (classification_completed ?export_declaration_var)
        (export_declaration_origin_jurisdiction ?export_declaration_var ?origin_jurisdiction)
        (declaration_permit_record_available ?declaration_permit_record)
        (not
          (export_declaration_ready ?export_declaration_var)
        )
      )
    :effect
      (and
        (origin_permit_attached ?origin_jurisdiction)
        (export_declaration_ready ?export_declaration_var)
        (export_declaration_permit_record_attached ?export_declaration_var ?declaration_permit_record)
        (not
          (declaration_permit_record_available ?declaration_permit_record)
        )
      )
  )
  (:action fulfill_origin_requirement
    :parameters (?export_declaration_var - export_declaration ?origin_jurisdiction - origin_jurisdiction ?supporting_document_var - supporting_document ?declaration_permit_record - declaration_permit_record)
    :precondition
      (and
        (classification_completed ?export_declaration_var)
        (case_document_linked ?export_declaration_var ?supporting_document_var)
        (export_declaration_origin_jurisdiction ?export_declaration_var ?origin_jurisdiction)
        (origin_permit_attached ?origin_jurisdiction)
        (export_declaration_permit_record_attached ?export_declaration_var ?declaration_permit_record)
        (not
          (export_declaration_requirements_satisfied ?export_declaration_var)
        )
      )
    :effect
      (and
        (origin_requirement_marked ?origin_jurisdiction)
        (export_declaration_requirements_satisfied ?export_declaration_var)
        (declaration_permit_record_available ?declaration_permit_record)
        (not
          (export_declaration_permit_record_attached ?export_declaration_var ?declaration_permit_record)
        )
      )
  )
  (:action initiate_destination_requirement_check
    :parameters (?import_declaration_var - import_declaration ?destination_jurisdiction - destination_jurisdiction ?supporting_document_var - supporting_document)
    :precondition
      (and
        (classification_completed ?import_declaration_var)
        (case_document_linked ?import_declaration_var ?supporting_document_var)
        (import_declaration_destination_jurisdiction ?import_declaration_var ?destination_jurisdiction)
        (not
          (destination_requirement_marked ?destination_jurisdiction)
        )
        (not
          (destination_permit_attached ?destination_jurisdiction)
        )
      )
    :effect (destination_requirement_marked ?destination_jurisdiction)
  )
  (:action assign_reviewer_for_destination_requirement
    :parameters (?import_declaration_var - import_declaration ?destination_jurisdiction - destination_jurisdiction ?reviewer - reviewer)
    :precondition
      (and
        (classification_completed ?import_declaration_var)
        (reviewer_assigned ?import_declaration_var ?reviewer)
        (import_declaration_destination_jurisdiction ?import_declaration_var ?destination_jurisdiction)
        (destination_requirement_marked ?destination_jurisdiction)
        (not
          (import_declaration_ready ?import_declaration_var)
        )
      )
    :effect
      (and
        (import_declaration_ready ?import_declaration_var)
        (import_declaration_requirements_satisfied ?import_declaration_var)
      )
  )
  (:action request_permit_for_import_declaration
    :parameters (?import_declaration_var - import_declaration ?destination_jurisdiction - destination_jurisdiction ?declaration_permit_record - declaration_permit_record)
    :precondition
      (and
        (classification_completed ?import_declaration_var)
        (import_declaration_destination_jurisdiction ?import_declaration_var ?destination_jurisdiction)
        (declaration_permit_record_available ?declaration_permit_record)
        (not
          (import_declaration_ready ?import_declaration_var)
        )
      )
    :effect
      (and
        (destination_permit_attached ?destination_jurisdiction)
        (import_declaration_ready ?import_declaration_var)
        (import_declaration_permit_record_attached ?import_declaration_var ?declaration_permit_record)
        (not
          (declaration_permit_record_available ?declaration_permit_record)
        )
      )
  )
  (:action fulfill_destination_requirement
    :parameters (?import_declaration_var - import_declaration ?destination_jurisdiction - destination_jurisdiction ?supporting_document_var - supporting_document ?declaration_permit_record - declaration_permit_record)
    :precondition
      (and
        (classification_completed ?import_declaration_var)
        (case_document_linked ?import_declaration_var ?supporting_document_var)
        (import_declaration_destination_jurisdiction ?import_declaration_var ?destination_jurisdiction)
        (destination_permit_attached ?destination_jurisdiction)
        (import_declaration_permit_record_attached ?import_declaration_var ?declaration_permit_record)
        (not
          (import_declaration_requirements_satisfied ?import_declaration_var)
        )
      )
    :effect
      (and
        (destination_requirement_marked ?destination_jurisdiction)
        (import_declaration_requirements_satisfied ?import_declaration_var)
        (declaration_permit_record_available ?declaration_permit_record)
        (not
          (import_declaration_permit_record_attached ?import_declaration_var ?declaration_permit_record)
        )
      )
  )
  (:action assemble_clearance_submission
    :parameters (?export_declaration_var - export_declaration ?import_declaration_var - import_declaration ?origin_jurisdiction - origin_jurisdiction ?destination_jurisdiction - destination_jurisdiction ?clearance_submission - clearance_submission)
    :precondition
      (and
        (export_declaration_ready ?export_declaration_var)
        (import_declaration_ready ?import_declaration_var)
        (export_declaration_origin_jurisdiction ?export_declaration_var ?origin_jurisdiction)
        (import_declaration_destination_jurisdiction ?import_declaration_var ?destination_jurisdiction)
        (origin_requirement_marked ?origin_jurisdiction)
        (destination_requirement_marked ?destination_jurisdiction)
        (export_declaration_requirements_satisfied ?export_declaration_var)
        (import_declaration_requirements_satisfied ?import_declaration_var)
        (submission_available ?clearance_submission)
      )
    :effect
      (and
        (submission_assembled ?clearance_submission)
        (submission_has_origin_jurisdiction ?clearance_submission ?origin_jurisdiction)
        (submission_has_destination_jurisdiction ?clearance_submission ?destination_jurisdiction)
        (not
          (submission_available ?clearance_submission)
        )
      )
  )
  (:action assemble_submission_with_origin_permit
    :parameters (?export_declaration_var - export_declaration ?import_declaration_var - import_declaration ?origin_jurisdiction - origin_jurisdiction ?destination_jurisdiction - destination_jurisdiction ?clearance_submission - clearance_submission)
    :precondition
      (and
        (export_declaration_ready ?export_declaration_var)
        (import_declaration_ready ?import_declaration_var)
        (export_declaration_origin_jurisdiction ?export_declaration_var ?origin_jurisdiction)
        (import_declaration_destination_jurisdiction ?import_declaration_var ?destination_jurisdiction)
        (origin_permit_attached ?origin_jurisdiction)
        (destination_requirement_marked ?destination_jurisdiction)
        (not
          (export_declaration_requirements_satisfied ?export_declaration_var)
        )
        (import_declaration_requirements_satisfied ?import_declaration_var)
        (submission_available ?clearance_submission)
      )
    :effect
      (and
        (submission_assembled ?clearance_submission)
        (submission_has_origin_jurisdiction ?clearance_submission ?origin_jurisdiction)
        (submission_has_destination_jurisdiction ?clearance_submission ?destination_jurisdiction)
        (submission_origin_permit_required ?clearance_submission)
        (not
          (submission_available ?clearance_submission)
        )
      )
  )
  (:action assemble_submission_with_destination_permit
    :parameters (?export_declaration_var - export_declaration ?import_declaration_var - import_declaration ?origin_jurisdiction - origin_jurisdiction ?destination_jurisdiction - destination_jurisdiction ?clearance_submission - clearance_submission)
    :precondition
      (and
        (export_declaration_ready ?export_declaration_var)
        (import_declaration_ready ?import_declaration_var)
        (export_declaration_origin_jurisdiction ?export_declaration_var ?origin_jurisdiction)
        (import_declaration_destination_jurisdiction ?import_declaration_var ?destination_jurisdiction)
        (origin_requirement_marked ?origin_jurisdiction)
        (destination_permit_attached ?destination_jurisdiction)
        (export_declaration_requirements_satisfied ?export_declaration_var)
        (not
          (import_declaration_requirements_satisfied ?import_declaration_var)
        )
        (submission_available ?clearance_submission)
      )
    :effect
      (and
        (submission_assembled ?clearance_submission)
        (submission_has_origin_jurisdiction ?clearance_submission ?origin_jurisdiction)
        (submission_has_destination_jurisdiction ?clearance_submission ?destination_jurisdiction)
        (submission_destination_permit_required ?clearance_submission)
        (not
          (submission_available ?clearance_submission)
        )
      )
  )
  (:action assemble_submission_with_both_permits
    :parameters (?export_declaration_var - export_declaration ?import_declaration_var - import_declaration ?origin_jurisdiction - origin_jurisdiction ?destination_jurisdiction - destination_jurisdiction ?clearance_submission - clearance_submission)
    :precondition
      (and
        (export_declaration_ready ?export_declaration_var)
        (import_declaration_ready ?import_declaration_var)
        (export_declaration_origin_jurisdiction ?export_declaration_var ?origin_jurisdiction)
        (import_declaration_destination_jurisdiction ?import_declaration_var ?destination_jurisdiction)
        (origin_permit_attached ?origin_jurisdiction)
        (destination_permit_attached ?destination_jurisdiction)
        (not
          (export_declaration_requirements_satisfied ?export_declaration_var)
        )
        (not
          (import_declaration_requirements_satisfied ?import_declaration_var)
        )

      )

      (and
        (submission_assembled ?clearance_submission)
        (submission_has_origin_jurisdiction ?clearance_submission ?origin_jurisdiction)
        (submission_has_destination_jurisdiction ?clearance_submission ?destination_jurisdiction)
        (submission_origin_permit_required ?clearance_submission)
        (submission_destination_permit_required ?clearance_submission)
        (not
          (submission_available ?clearance_submission)
        )
      )
  )
  (:action mark_submission_ready_for_filing
    :parameters (?clearance_submission - clearance_submission ?export_declaration_var - export_declaration ?supporting_document_var - supporting_document)
    :precondition
      (and
        (submission_assembled ?clearance_submission)
        (export_declaration_ready ?export_declaration_var)
        (case_document_linked ?export_declaration_var ?supporting_document_var)
        (not
          (submission_ready ?clearance_submission)
        )
      )
    :effect (submission_ready ?clearance_submission)
  )
  (:action attach_checklist_to_ticket_and_submission
    :parameters (?escalation_ticket - escalation_ticket ?compliance_checklist - compliance_checklist ?clearance_submission - clearance_submission)
    :precondition
      (and
        (classification_completed ?escalation_ticket)
        (ticket_linked_to_submission ?escalation_ticket ?clearance_submission)
        (ticket_has_checklist ?escalation_ticket ?compliance_checklist)
        (checklist_available ?compliance_checklist)
        (submission_assembled ?clearance_submission)
        (submission_ready ?clearance_submission)
        (not
          (checklist_attached ?compliance_checklist)
        )
      )
    :effect
      (and
        (checklist_attached ?compliance_checklist)
        (checklist_associated_with_submission ?compliance_checklist ?clearance_submission)
        (not
          (checklist_available ?compliance_checklist)
        )
      )
  )
  (:action start_ticket_assessment
    :parameters (?escalation_ticket - escalation_ticket ?compliance_checklist - compliance_checklist ?clearance_submission - clearance_submission ?supporting_document_var - supporting_document)
    :precondition
      (and
        (classification_completed ?escalation_ticket)
        (ticket_has_checklist ?escalation_ticket ?compliance_checklist)
        (checklist_attached ?compliance_checklist)
        (checklist_associated_with_submission ?compliance_checklist ?clearance_submission)
        (case_document_linked ?escalation_ticket ?supporting_document_var)
        (not
          (submission_origin_permit_required ?clearance_submission)
        )
        (not
          (ticket_assessment_ready ?escalation_ticket)
        )
      )
    :effect (ticket_assessment_ready ?escalation_ticket)
  )
  (:action assign_attachment_type_to_ticket
    :parameters (?escalation_ticket - escalation_ticket ?evidence_attachment_type - evidence_attachment_type)
    :precondition
      (and
        (classification_completed ?escalation_ticket)
