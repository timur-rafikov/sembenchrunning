--- domain.pddl (as passed to VAL) ---
(define (domain wave_release_timing_for_fulfillment)
  (:requirements :strips :typing :negative-preconditions)
  (:types fulfillment_resource - object logistics_unit - object consolidation_group - object fulfillment_entity - object fulfillment_unit - fulfillment_entity inventory_location - fulfillment_resource item_sku - fulfillment_resource work_station - fulfillment_resource carrier_option - fulfillment_resource handling_instruction - fulfillment_resource delivery_window - fulfillment_resource packing_station - fulfillment_resource carrier_service - fulfillment_resource inventory_unit - logistics_unit package - logistics_unit routing_profile - logistics_unit pick_zone - consolidation_group ship_zone - consolidation_group shipment - consolidation_group sales_order_root - fulfillment_unit wave_root - fulfillment_unit sales_order - sales_order_root sales_order_segment - sales_order_root wave - wave_root)
  (:predicates
    (release_candidate ?fulfillment_unit - fulfillment_unit)
    (entity_sourcing_confirmed ?fulfillment_unit - fulfillment_unit)
    (inventory_sourced ?fulfillment_unit - fulfillment_unit)
    (released_for_dispatch ?fulfillment_unit - fulfillment_unit)
    (entity_ready_for_delivery_assignment ?fulfillment_unit - fulfillment_unit)
    (entity_delivery_window_assigned ?fulfillment_unit - fulfillment_unit)
    (location_available ?inventory_location - inventory_location)
    (entity_sourced_from_location ?fulfillment_unit - fulfillment_unit ?inventory_location - inventory_location)
    (sku_available ?item_sku - item_sku)
    (entity_assigned_sku ?fulfillment_unit - fulfillment_unit ?item_sku - item_sku)
    (workstation_available ?work_station - work_station)
    (entity_assigned_workstation ?fulfillment_unit - fulfillment_unit ?work_station - work_station)
    (inventory_unit_available ?inventory_unit - inventory_unit)
    (inventory_unit_assigned_to_order ?sales_order - sales_order ?inventory_unit - inventory_unit)
    (inventory_unit_assigned_to_segment ?sales_order_segment - sales_order_segment ?inventory_unit - inventory_unit)
    (order_assigned_pick_zone ?sales_order - sales_order ?pick_zone - pick_zone)
    (pick_zone_staged ?pick_zone - pick_zone)
    (pick_zone_inventory_reserved ?pick_zone - pick_zone)
    (picking_completed ?sales_order - sales_order)
    (assigned_ship_zone ?sales_order_segment - sales_order_segment ?ship_zone - ship_zone)
    (ship_zone_staged ?ship_zone - ship_zone)
    (ship_zone_inventory_reserved ?ship_zone - ship_zone)
    (segment_ready_for_shipment ?sales_order_segment - sales_order_segment)
    (shipment_available ?shipment - shipment)
    (shipment_staged ?shipment - shipment)
    (shipment_assigned_pick_zone ?shipment - shipment ?pick_zone - pick_zone)
    (shipment_assigned_ship_zone ?shipment - shipment ?ship_zone - ship_zone)
    (requires_packaging_stage_1 ?shipment - shipment)
    (requires_packaging_stage_2 ?shipment - shipment)
    (packaging_completed ?shipment - shipment)
    (wave_contains_order ?wave - wave ?sales_order - sales_order)
    (wave_contains_order_segment ?wave - wave ?sales_order_segment - sales_order_segment)
    (wave_contains_shipment ?wave - wave ?shipment - shipment)
    (package_available ?package - package)
    (wave_assigned_package ?wave - wave ?package - package)
    (package_allocated ?package - package)
    (package_assigned_to_shipment ?package - package ?shipment - shipment)
    (wave_package_ready ?wave - wave)
    (wave_packing_in_progress ?wave - wave)
    (wave_packing_completed ?wave - wave)
    (carrier_option_attached ?wave - wave)
    (carrier_option_confirmed ?wave - wave)
    (wave_ready_for_package_assignment ?wave - wave)
    (wave_packing_finalized ?wave - wave)
    (routing_profile_available ?routing_profile - routing_profile)
    (wave_assigned_routing_profile ?wave - wave ?routing_profile - routing_profile)
    (wave_routing_confirmed ?wave - wave)
    (wave_ready_for_carrier_assignment ?wave - wave)
    (carrier_service_attached ?wave - wave)
    (carrier_option_available ?carrier_option - carrier_option)
    (wave_assigned_carrier_option ?wave - wave ?carrier_option - carrier_option)
    (handling_instruction_available ?handling_instruction - handling_instruction)
    (wave_assigned_handling_instruction ?wave - wave ?handling_instruction - handling_instruction)
    (packing_station_available ?packing_station - packing_station)
    (wave_assigned_packing_station ?wave - wave ?packing_station - packing_station)
    (carrier_service_available ?carrier_service - carrier_service)
    (wave_assigned_carrier_service ?wave - wave ?carrier_service - carrier_service)
    (delivery_window_available ?delivery_window - delivery_window)
    (entity_assigned_delivery_window ?fulfillment_unit - fulfillment_unit ?delivery_window - delivery_window)
    (order_ready_for_consolidation ?sales_order - sales_order)
    (segment_ready_for_consolidation ?sales_order_segment - sales_order_segment)
    (wave_finalized ?wave - wave)
  )
  (:action mark_fulfillment_unit_as_release_candidate
    :parameters (?fulfillment_unit - fulfillment_unit)
    :precondition
      (and
        (not
          (release_candidate ?fulfillment_unit)
        )
        (not
          (released_for_dispatch ?fulfillment_unit)
        )
      )
    :effect (release_candidate ?fulfillment_unit)
  )
  (:action assign_inventory_location_to_fulfillment_unit
    :parameters (?fulfillment_unit - fulfillment_unit ?inventory_location - inventory_location)
    :precondition
      (and
        (release_candidate ?fulfillment_unit)
        (not
          (inventory_sourced ?fulfillment_unit)
        )
        (location_available ?inventory_location)
      )
    :effect
      (and
        (inventory_sourced ?fulfillment_unit)
        (entity_sourced_from_location ?fulfillment_unit ?inventory_location)
        (not
          (location_available ?inventory_location)
        )
      )
  )
  (:action allocate_sku_to_fulfillment_unit
    :parameters (?fulfillment_unit - fulfillment_unit ?item_sku - item_sku)
    :precondition
      (and
        (release_candidate ?fulfillment_unit)
        (inventory_sourced ?fulfillment_unit)
        (sku_available ?item_sku)
      )
    :effect
      (and
        (entity_assigned_sku ?fulfillment_unit ?item_sku)
        (not
          (sku_available ?item_sku)
        )
      )
  )
  (:action finalize_sourcing_for_fulfillment_unit
    :parameters (?fulfillment_unit - fulfillment_unit ?item_sku - item_sku)
    :precondition
      (and
        (release_candidate ?fulfillment_unit)
        (inventory_sourced ?fulfillment_unit)
        (entity_assigned_sku ?fulfillment_unit ?item_sku)
        (not
          (entity_sourcing_confirmed ?fulfillment_unit)
        )
      )
    :effect (entity_sourcing_confirmed ?fulfillment_unit)
  )
  (:action release_sku_from_fulfillment_unit
    :parameters (?fulfillment_unit - fulfillment_unit ?item_sku - item_sku)
    :precondition
      (and
        (entity_assigned_sku ?fulfillment_unit ?item_sku)
      )
    :effect
      (and
        (sku_available ?item_sku)
        (not
          (entity_assigned_sku ?fulfillment_unit ?item_sku)
        )
      )
  )
  (:action assign_workstation_to_fulfillment_unit
    :parameters (?fulfillment_unit - fulfillment_unit ?work_station - work_station)
    :precondition
      (and
        (entity_sourcing_confirmed ?fulfillment_unit)
        (workstation_available ?work_station)
      )
    :effect
      (and
        (entity_assigned_workstation ?fulfillment_unit ?work_station)
        (not
          (workstation_available ?work_station)
        )
      )
  )
  (:action release_workstation_from_fulfillment_unit
    :parameters (?fulfillment_unit - fulfillment_unit ?work_station - work_station)
    :precondition
      (and
        (entity_assigned_workstation ?fulfillment_unit ?work_station)
      )
    :effect
      (and
        (workstation_available ?work_station)
        (not
          (entity_assigned_workstation ?fulfillment_unit ?work_station)
        )
      )
  )
  (:action assign_packing_station_to_wave
    :parameters (?wave - wave ?packing_station - packing_station)
    :precondition
      (and
        (entity_sourcing_confirmed ?wave)
        (packing_station_available ?packing_station)
      )
    :effect
      (and
        (wave_assigned_packing_station ?wave ?packing_station)
        (not
          (packing_station_available ?packing_station)
        )
      )
  )
  (:action release_packing_station_from_wave
    :parameters (?wave - wave ?packing_station - packing_station)
    :precondition
      (and
        (wave_assigned_packing_station ?wave ?packing_station)
      )
    :effect
      (and
        (packing_station_available ?packing_station)
        (not
          (wave_assigned_packing_station ?wave ?packing_station)
        )
      )
  )
  (:action assign_carrier_service_to_wave
    :parameters (?wave - wave ?carrier_service - carrier_service)
    :precondition
      (and
        (entity_sourcing_confirmed ?wave)
        (carrier_service_available ?carrier_service)
      )
    :effect
      (and
        (wave_assigned_carrier_service ?wave ?carrier_service)
        (not
          (carrier_service_available ?carrier_service)
        )
      )
  )
  (:action release_carrier_service_from_wave
    :parameters (?wave - wave ?carrier_service - carrier_service)
    :precondition
      (and
        (wave_assigned_carrier_service ?wave ?carrier_service)
      )
    :effect
      (and
        (carrier_service_available ?carrier_service)
        (not
          (wave_assigned_carrier_service ?wave ?carrier_service)
        )
      )
  )
  (:action stage_pick_zone_for_order
    :parameters (?sales_order - sales_order ?pick_zone - pick_zone ?item_sku - item_sku)
    :precondition
      (and
        (entity_sourcing_confirmed ?sales_order)
        (entity_assigned_sku ?sales_order ?item_sku)
        (order_assigned_pick_zone ?sales_order ?pick_zone)
        (not
          (pick_zone_staged ?pick_zone)
        )
        (not
          (pick_zone_inventory_reserved ?pick_zone)
        )
      )
    :effect (pick_zone_staged ?pick_zone)
  )
  (:action confirm_pick_assignment_for_order
    :parameters (?sales_order - sales_order ?pick_zone - pick_zone ?work_station - work_station)
    :precondition
      (and
        (entity_sourcing_confirmed ?sales_order)
        (entity_assigned_workstation ?sales_order ?work_station)
        (order_assigned_pick_zone ?sales_order ?pick_zone)
        (pick_zone_staged ?pick_zone)
        (not
          (order_ready_for_consolidation ?sales_order)
        )
      )
    :effect
      (and
        (order_ready_for_consolidation ?sales_order)
        (picking_completed ?sales_order)
      )
  )
  (:action reserve_inventory_unit_for_order
    :parameters (?sales_order - sales_order ?pick_zone - pick_zone ?inventory_unit - inventory_unit)
    :precondition
      (and
        (entity_sourcing_confirmed ?sales_order)
        (order_assigned_pick_zone ?sales_order ?pick_zone)
        (inventory_unit_available ?inventory_unit)
        (not
          (order_ready_for_consolidation ?sales_order)
        )
      )
    :effect
      (and
        (pick_zone_inventory_reserved ?pick_zone)
        (order_ready_for_consolidation ?sales_order)
        (inventory_unit_assigned_to_order ?sales_order ?inventory_unit)
        (not
          (inventory_unit_available ?inventory_unit)
        )
      )
  )
  (:action finalize_pick_allocation
    :parameters (?sales_order - sales_order ?pick_zone - pick_zone ?item_sku - item_sku ?inventory_unit - inventory_unit)
    :precondition
      (and
        (entity_sourcing_confirmed ?sales_order)
        (entity_assigned_sku ?sales_order ?item_sku)
        (order_assigned_pick_zone ?sales_order ?pick_zone)
        (pick_zone_inventory_reserved ?pick_zone)
        (inventory_unit_assigned_to_order ?sales_order ?inventory_unit)
        (not
          (picking_completed ?sales_order)
        )
      )
    :effect
      (and
        (pick_zone_staged ?pick_zone)
        (picking_completed ?sales_order)
        (inventory_unit_available ?inventory_unit)
        (not
          (inventory_unit_assigned_to_order ?sales_order ?inventory_unit)
        )
      )
  )
  (:action stage_ship_zone_for_segment
    :parameters (?sales_order_segment - sales_order_segment ?ship_zone - ship_zone ?item_sku - item_sku)
    :precondition
      (and
        (entity_sourcing_confirmed ?sales_order_segment)
        (entity_assigned_sku ?sales_order_segment ?item_sku)
        (assigned_ship_zone ?sales_order_segment ?ship_zone)
        (not
          (ship_zone_staged ?ship_zone)
        )
        (not
          (ship_zone_inventory_reserved ?ship_zone)
        )
      )
    :effect (ship_zone_staged ?ship_zone)
  )
  (:action confirm_ship_assignment_for_segment
    :parameters (?sales_order_segment - sales_order_segment ?ship_zone - ship_zone ?work_station - work_station)
    :precondition
      (and
        (entity_sourcing_confirmed ?sales_order_segment)
        (entity_assigned_workstation ?sales_order_segment ?work_station)
        (assigned_ship_zone ?sales_order_segment ?ship_zone)
        (ship_zone_staged ?ship_zone)
        (not
          (segment_ready_for_consolidation ?sales_order_segment)
        )
      )
    :effect
      (and
        (segment_ready_for_consolidation ?sales_order_segment)
        (segment_ready_for_shipment ?sales_order_segment)
      )
  )
  (:action reserve_inventory_unit_for_segment
    :parameters (?sales_order_segment - sales_order_segment ?ship_zone - ship_zone ?inventory_unit - inventory_unit)
    :precondition
      (and
        (entity_sourcing_confirmed ?sales_order_segment)
        (assigned_ship_zone ?sales_order_segment ?ship_zone)
        (inventory_unit_available ?inventory_unit)
        (not
          (segment_ready_for_consolidation ?sales_order_segment)
        )
      )
    :effect
      (and
        (ship_zone_inventory_reserved ?ship_zone)
        (segment_ready_for_consolidation ?sales_order_segment)
        (inventory_unit_assigned_to_segment ?sales_order_segment ?inventory_unit)
        (not
          (inventory_unit_available ?inventory_unit)
        )
      )
  )
  (:action finalize_ship_allocation_for_segment
    :parameters (?sales_order_segment - sales_order_segment ?ship_zone - ship_zone ?item_sku - item_sku ?inventory_unit - inventory_unit)
    :precondition
      (and
        (entity_sourcing_confirmed ?sales_order_segment)
        (entity_assigned_sku ?sales_order_segment ?item_sku)
        (assigned_ship_zone ?sales_order_segment ?ship_zone)
        (ship_zone_inventory_reserved ?ship_zone)
        (inventory_unit_assigned_to_segment ?sales_order_segment ?inventory_unit)
        (not
          (segment_ready_for_shipment ?sales_order_segment)
        )
      )
    :effect
      (and
        (ship_zone_staged ?ship_zone)
        (segment_ready_for_shipment ?sales_order_segment)
        (inventory_unit_available ?inventory_unit)
        (not
          (inventory_unit_assigned_to_segment ?sales_order_segment ?inventory_unit)
        )
      )
  )
  (:action form_shipment_from_zones
    :parameters (?sales_order - sales_order ?sales_order_segment - sales_order_segment ?pick_zone - pick_zone ?ship_zone - ship_zone ?shipment - shipment)
    :precondition
      (and
        (order_ready_for_consolidation ?sales_order)
        (segment_ready_for_consolidation ?sales_order_segment)
        (order_assigned_pick_zone ?sales_order ?pick_zone)
        (assigned_ship_zone ?sales_order_segment ?ship_zone)
        (pick_zone_staged ?pick_zone)
        (ship_zone_staged ?ship_zone)
        (picking_completed ?sales_order)
        (segment_ready_for_shipment ?sales_order_segment)
        (shipment_available ?shipment)
      )
    :effect
      (and
        (shipment_staged ?shipment)
        (shipment_assigned_pick_zone ?shipment ?pick_zone)
        (shipment_assigned_ship_zone ?shipment ?ship_zone)
        (not
          (shipment_available ?shipment)
        )
      )
  )
  (:action form_shipment_with_pack_stage_1
    :parameters (?sales_order - sales_order ?sales_order_segment - sales_order_segment ?pick_zone - pick_zone ?ship_zone - ship_zone ?shipment - shipment)
    :precondition
      (and
        (order_ready_for_consolidation ?sales_order)
        (segment_ready_for_consolidation ?sales_order_segment)
        (order_assigned_pick_zone ?sales_order ?pick_zone)
        (assigned_ship_zone ?sales_order_segment ?ship_zone)
        (pick_zone_inventory_reserved ?pick_zone)
        (ship_zone_staged ?ship_zone)
        (not
          (picking_completed ?sales_order)
        )
        (segment_ready_for_shipment ?sales_order_segment)
        (shipment_available ?shipment)
      )
    :effect
      (and
        (shipment_staged ?shipment)
        (shipment_assigned_pick_zone ?shipment ?pick_zone)
        (shipment_assigned_ship_zone ?shipment ?ship_zone)
        (requires_packaging_stage_1 ?shipment)
        (not
          (shipment_available ?shipment)
        )
      )
  )
  (:action form_shipment_with_pack_stage_2
    :parameters (?sales_order - sales_order ?sales_order_segment - sales_order_segment ?pick_zone - pick_zone ?ship_zone - ship_zone ?shipment - shipment)
    :precondition
      (and
        (order_ready_for_consolidation ?sales_order)
        (segment_ready_for_consolidation ?sales_order_segment)
        (order_assigned_pick_zone ?sales_order ?pick_zone)
        (assigned_ship_zone ?sales_order_segment ?ship_zone)
        (pick_zone_staged ?pick_zone)
        (ship_zone_inventory_reserved ?ship_zone)
        (picking_completed ?sales_order)
        (not
          (segment_ready_for_shipment ?sales_order_segment)
        )
        (shipment_available ?shipment)
      )
    :effect
      (and
        (shipment_staged ?shipment)
        (shipment_assigned_pick_zone ?shipment ?pick_zone)
        (shipment_assigned_ship_zone ?shipment ?ship_zone)
        (requires_packaging_stage_2 ?shipment)
        (not
          (shipment_available ?shipment)
        )
      )
  )
  (:action form_shipment_with_both_pack_stages
    :parameters (?sales_order - sales_order ?sales_order_segment - sales_order_segment ?pick_zone - pick_zone ?ship_zone - ship_zone ?shipment - shipment)
    :precondition
      (and
        (order_ready_for_consolidation ?sales_order)
        (segment_ready_for_consolidation ?sales_order_segment)
        (order_assigned_pick_zone ?sales_order ?pick_zone)
        (assigned_ship_zone ?sales_order_segment ?ship_zone)
        (pick_zone_inventory_reserved ?pick_zone)
        (ship_zone_inventory_reserved ?ship_zone)
        (not
          (picking_completed ?sales_order)
        )
        (not
          (segment_ready_for_shipment ?sales_order_segment)
        )
        (shipment_available ?shipment)
      )
    :effect
      (and
        (shipment_staged ?shipment)
        (shipment_assigned_pick_zone ?shipment ?pick_zone)
        (shipment_assigned_ship_zone ?shipment ?ship_zone)
        (requires_packaging_stage_1 ?shipment)
        (requires_packaging_stage_2 ?shipment)
        (not
          (shipment_available ?shipment)
        )
      )
  )
  (:action mark_shipment_as_packed
    :parameters (?shipment - shipment ?sales_order - sales_order ?item_sku - item_sku)
    :precondition
      (and
        (shipment_staged ?shipment)
        (order_ready_for_consolidation ?sales_order)
        (entity_assigned_sku ?sales_order ?item_sku)
        (not
          (packaging_completed ?shipment)
        )
      )
    :effect (packaging_completed ?shipment)
  )
  (:action allocate_package_to_shipment
    :parameters (?wave - wave ?package - package ?shipment - shipment)
    :precondition
      (and
        (entity_sourcing_confirmed ?wave)
        (wave_contains_shipment ?wave ?shipment)
        (wave_assigned_package ?wave ?package)
        (package_available ?package)
        (shipment_staged ?shipment)
        (packaging_completed ?shipment)
        (not
          (package_allocated ?package)
        )
      )
    :effect
      (and
        (package_allocated ?package)
        (package_assigned_to_shipment ?package ?shipment)
        (not
          (package_available ?package)
        )
      )
  )
  (:action prepare_wave_for_packaging
    :parameters (?wave - wave ?package - package ?shipment - shipment ?item_sku - item_sku)
    :precondition
      (and
        (entity_sourcing_confirmed ?wave)
        (wave_assigned_package ?wave ?package)
        (package_allocated ?package)
        (package_assigned_to_shipment ?package ?shipment)
        (entity_assigned_sku ?wave ?item_sku)
        (not
          (requires_packaging_stage_1 ?shipment)
        )
        (not
          (wave_package_ready ?wave)
        )
      )
    :effect (wave_package_ready ?wave)
  )
  (:action assign_carrier_option_to_wave
    :parameters (?wave - wave ?carrier_option - carrier_option)
    :precondition
      (and
        (entity_sourcing_confirmed ?wave)
        (carrier_option_available ?carrier_option)
        (not
          (carrier_option_attached ?wave)
        )
      )
    :effect
      (and
        (carrier_option_attached ?wave)
        (wave_assigned_carrier_option ?wave ?carrier_option)
        (not
          (carrier_option_available ?carrier_option)
        )
      )
  )
  (:action confirm_carrier_option_and_prepare_wave
    :parameters (?wave - wave ?package - package ?shipment - shipment ?item_sku - item_sku ?carrier_option - carrier_option)
    :precondition
      (and
        (entity_sourcing_confirmed ?wave)
        (wave_assigned_package ?wave ?package)
        (package_allocated ?package)
        (package_assigned_to_shipment ?package ?shipment)
        (entity_assigned_sku ?wave ?item_sku)
        (requires_packaging_stage_1 ?shipment)
        (carrier_option_attached ?wave)
        (wave_assigned_carrier_option ?wave ?carrier_option)
        (not
          (wave_package_ready ?wave)
        )
      )
    :effect
      (and
        (wave_package_ready ?wave)
        (carrier_option_confirmed ?wave)
      )
  )
  (:action initiate_wave_packing_variant_a
    :parameters (?wave - wave ?packing_station - packing_station ?work_station - work_station ?package - package ?shipment - shipment)
    :precondition
      (and
        (wave_package_ready ?wave)
        (wave_assigned_packing_station ?wave ?packing_station)
        (entity_assigned_workstation ?wave ?work_station)
        (wave_assigned_package ?wave ?package)
        (package_assigned_to_shipment ?package ?shipment)
        (not
          (requires_packaging_stage_2 ?shipment)
        )
        (not
          (wave_packing_in_progress ?wave)
        )
      )
    :effect (wave_packing_in_progress ?wave)
  )
  (:action initiate_wave_packing_variant_b
    :parameters (?wave - wave ?packing_station - packing_station ?work_station - work_station ?package - package ?shipment - shipment)
    :precondition
      (and
        (wave_package_ready ?wave)
        (wave_assigned_packing_station ?wave ?packing_station)
        (entity_assigned_workstation ?wave ?work_station)
        (wave_assigned_package ?wave ?package)
        (package_assigned_to_shipment ?package ?shipment)
        (requires_packaging_stage_2 ?shipment)
        (not
          (wave_packing_in_progress ?wave)
        )
      )
    :effect (wave_packing_in_progress ?wave)
  )
  (:action complete_wave_packing_stage
    :parameters (?wave - wave ?carrier_service - carrier_service ?package - package ?shipment - shipment)
    :precondition
      (and
        (wave_packing_in_progress ?wave)
        (wave_assigned_carrier_service ?wave ?carrier_service)
        (wave_assigned_package ?wave ?package)
        (package_assigned_to_shipment ?package ?shipment)
        (not
          (requires_packaging_stage_1 ?shipment)
        )
        (not
          (requires_packaging_stage_2 ?shipment)
        )
        (not
          (wave_packing_completed ?wave)
        )
      )
    :effect (wave_packing_completed ?wave)
