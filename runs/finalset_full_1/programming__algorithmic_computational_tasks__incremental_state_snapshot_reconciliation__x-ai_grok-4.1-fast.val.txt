--- domain.pddl (as passed to VAL) ---
(define (domain incremental_state_snapshot_reconciliation)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_token - object data_fragment_type - object version_identifier - object global_context - object entity_key - global_context allocation_slot - resource_token state_delta - resource_token binding_handle - resource_token configuration_flag - resource_token optional_modifier - resource_token commit_marker - resource_token resource_attachment - resource_token policy_attachment - resource_token auxiliary_token - data_fragment_type validation_artifact - data_fragment_type external_constraint - data_fragment_type candidate_version_a - version_identifier candidate_version_b - version_identifier merged_candidate - version_identifier source_a_container - entity_key source_b_container - entity_key source_a_record - source_a_container source_b_record - source_a_container reconciliation_unit - source_b_container)
  (:predicates
    (entity_discovered ?entity_key - entity_key)
    (entity_ready ?entity_key - entity_key)
    (entity_allocated ?entity_key - entity_key)
    (entity_reconciled ?entity_key - entity_key)
    (unit_finalization_token ?entity_key - entity_key)
    (entity_marked_for_commit ?entity_key - entity_key)
    (allocation_slot_available ?allocation_slot - allocation_slot)
    (entity_assigned_slot ?entity_key - entity_key ?allocation_slot - allocation_slot)
    (state_delta_available ?state_delta - state_delta)
    (entity_attached_state_delta ?entity_key - entity_key ?state_delta - state_delta)
    (binding_handle_available ?binding_handle - binding_handle)
    (entity_bound_handle ?entity_key - entity_key ?binding_handle - binding_handle)
    (auxiliary_token_available ?auxiliary_token - auxiliary_token)
    (source_a_attached_aux_token ?source_a_record - source_a_record ?auxiliary_token - auxiliary_token)
    (source_b_attached_aux_token ?source_b_record - source_b_record ?auxiliary_token - auxiliary_token)
    (source_a_has_candidate_version ?source_a_record - source_a_record ?candidate_version_a - candidate_version_a)
    (candidate_version_a_primary ?candidate_version_a - candidate_version_a)
    (candidate_version_a_secondary ?candidate_version_a - candidate_version_a)
    (source_a_confirmed ?source_a_record - source_a_record)
    (source_b_has_candidate_version ?source_b_record - source_b_record ?candidate_version_b - candidate_version_b)
    (candidate_version_b_primary ?candidate_version_b - candidate_version_b)
    (candidate_version_b_secondary ?candidate_version_b - candidate_version_b)
    (source_b_confirmed ?source_b_record - source_b_record)
    (merged_candidate_available ?merged_snapshot_candidate - merged_candidate)
    (merged_candidate_reserved ?merged_snapshot_candidate - merged_candidate)
    (merged_includes_candidate_a ?merged_snapshot_candidate - merged_candidate ?candidate_version_a - candidate_version_a)
    (merged_includes_candidate_b ?merged_snapshot_candidate - merged_candidate ?candidate_version_b - candidate_version_b)
    (merged_candidate_partial_pass ?merged_snapshot_candidate - merged_candidate)
    (merged_candidate_secondary_pass ?merged_snapshot_candidate - merged_candidate)
    (merged_candidate_final_pass ?merged_snapshot_candidate - merged_candidate)
    (reconciliation_unit_bound_source_a ?reconciliation_unit - reconciliation_unit ?source_a_record - source_a_record)
    (reconciliation_unit_bound_source_b ?reconciliation_unit - reconciliation_unit ?source_b_record - source_b_record)
    (reconciliation_unit_has_merged_candidate ?reconciliation_unit - reconciliation_unit ?merged_snapshot_candidate - merged_candidate)
    (validation_artifact_available ?validation_artifact - validation_artifact)
    (reconciliation_unit_has_validation_artifact ?reconciliation_unit - reconciliation_unit ?validation_artifact - validation_artifact)
    (validation_artifact_consumed ?validation_artifact - validation_artifact)
    (validation_artifact_for_merged_candidate ?validation_artifact - validation_artifact ?merged_snapshot_candidate - merged_candidate)
    (reconciliation_unit_configuration_applied ?reconciliation_unit - reconciliation_unit)
    (reconciliation_unit_resource_attached ?reconciliation_unit - reconciliation_unit)
    (reconciliation_unit_config_validated ?reconciliation_unit - reconciliation_unit)
    (reconciliation_unit_has_configuration_flag ?reconciliation_unit - reconciliation_unit)
    (reconciliation_unit_configuration_verified ?reconciliation_unit - reconciliation_unit)
    (reconciliation_unit_additional_config_applied ?reconciliation_unit - reconciliation_unit)
    (reconciliation_unit_final_checks_passed ?reconciliation_unit - reconciliation_unit)
    (external_constraint_available ?external_constraint - external_constraint)
    (reconciliation_unit_external_constraint ?reconciliation_unit - reconciliation_unit ?external_constraint - external_constraint)
    (reconciliation_unit_constraint_bound ?reconciliation_unit - reconciliation_unit)
    (reconciliation_unit_constraint_applied ?reconciliation_unit - reconciliation_unit)
    (reconciliation_unit_policy_attached ?reconciliation_unit - reconciliation_unit)
    (configuration_flag_available ?configuration_flag - configuration_flag)
    (reconciliation_unit_configuration_flag ?reconciliation_unit - reconciliation_unit ?configuration_flag - configuration_flag)
    (optional_modifier_available ?optional_modifier - optional_modifier)
    (reconciliation_unit_optional_modifier ?reconciliation_unit - reconciliation_unit ?optional_modifier - optional_modifier)
    (resource_attachment_available ?resource_attachment - resource_attachment)
    (reconciliation_unit_resource_attachment ?reconciliation_unit - reconciliation_unit ?resource_attachment - resource_attachment)
    (policy_attachment_available ?policy_attachment - policy_attachment)
    (reconciliation_unit_policy_attachment ?reconciliation_unit - reconciliation_unit ?policy_attachment - policy_attachment)
    (commit_marker_available ?commit_marker - commit_marker)
    (entity_has_commit_marker ?entity_key - entity_key ?commit_marker - commit_marker)
    (source_a_prepared ?source_a_record - source_a_record)
    (source_b_prepared ?source_b_record - source_b_record)
    (reconciliation_unit_finalized ?reconciliation_unit - reconciliation_unit)
  )
  (:action register_entity_discovery
    :parameters (?entity_key - entity_key)
    :precondition
      (and
        (not
          (entity_discovered ?entity_key)
        )
        (not
          (entity_reconciled ?entity_key)
        )
      )
    :effect (entity_discovered ?entity_key)
  )
  (:action assign_allocation_slot
    :parameters (?entity_key - entity_key ?allocation_slot - allocation_slot)
    :precondition
      (and
        (entity_discovered ?entity_key)
        (not
          (entity_allocated ?entity_key)
        )
        (allocation_slot_available ?allocation_slot)
      )
    :effect
      (and
        (entity_allocated ?entity_key)
        (entity_assigned_slot ?entity_key ?allocation_slot)
        (not
          (allocation_slot_available ?allocation_slot)
        )
      )
  )
  (:action attach_state_delta
    :parameters (?entity_key - entity_key ?state_delta - state_delta)
    :precondition
      (and
        (entity_discovered ?entity_key)
        (entity_allocated ?entity_key)
        (state_delta_available ?state_delta)
      )
    :effect
      (and
        (entity_attached_state_delta ?entity_key ?state_delta)
        (not
          (state_delta_available ?state_delta)
        )
      )
  )
  (:action advance_entity_to_ready
    :parameters (?entity_key - entity_key ?state_delta - state_delta)
    :precondition
      (and
        (entity_discovered ?entity_key)
        (entity_allocated ?entity_key)
        (entity_attached_state_delta ?entity_key ?state_delta)
        (not
          (entity_ready ?entity_key)
        )
      )
    :effect (entity_ready ?entity_key)
  )
  (:action release_state_delta
    :parameters (?entity_key - entity_key ?state_delta - state_delta)
    :precondition
      (and
        (entity_attached_state_delta ?entity_key ?state_delta)
      )
    :effect
      (and
        (state_delta_available ?state_delta)
        (not
          (entity_attached_state_delta ?entity_key ?state_delta)
        )
      )
  )
  (:action claim_binding_handle_for_entity
    :parameters (?entity_key - entity_key ?binding_handle - binding_handle)
    :precondition
      (and
        (entity_ready ?entity_key)
        (binding_handle_available ?binding_handle)
      )
    :effect
      (and
        (entity_bound_handle ?entity_key ?binding_handle)
        (not
          (binding_handle_available ?binding_handle)
        )
      )
  )
  (:action release_binding_handle_from_entity
    :parameters (?entity_key - entity_key ?binding_handle - binding_handle)
    :precondition
      (and
        (entity_bound_handle ?entity_key ?binding_handle)
      )
    :effect
      (and
        (binding_handle_available ?binding_handle)
        (not
          (entity_bound_handle ?entity_key ?binding_handle)
        )
      )
  )
  (:action attach_resource_to_reconciliation_unit
    :parameters (?reconciliation_unit - reconciliation_unit ?resource_attachment - resource_attachment)
    :precondition
      (and
        (entity_ready ?reconciliation_unit)
        (resource_attachment_available ?resource_attachment)
      )
    :effect
      (and
        (reconciliation_unit_resource_attachment ?reconciliation_unit ?resource_attachment)
        (not
          (resource_attachment_available ?resource_attachment)
        )
      )
  )
  (:action detach_resource_from_reconciliation_unit
    :parameters (?reconciliation_unit - reconciliation_unit ?resource_attachment - resource_attachment)
    :precondition
      (and
        (reconciliation_unit_resource_attachment ?reconciliation_unit ?resource_attachment)
      )
    :effect
      (and
        (resource_attachment_available ?resource_attachment)
        (not
          (reconciliation_unit_resource_attachment ?reconciliation_unit ?resource_attachment)
        )
      )
  )
  (:action attach_policy_to_reconciliation_unit
    :parameters (?reconciliation_unit - reconciliation_unit ?policy_attachment - policy_attachment)
    :precondition
      (and
        (entity_ready ?reconciliation_unit)
        (policy_attachment_available ?policy_attachment)
      )
    :effect
      (and
        (reconciliation_unit_policy_attachment ?reconciliation_unit ?policy_attachment)
        (not
          (policy_attachment_available ?policy_attachment)
        )
      )
  )
  (:action detach_policy_from_reconciliation_unit
    :parameters (?reconciliation_unit - reconciliation_unit ?policy_attachment - policy_attachment)
    :precondition
      (and
        (reconciliation_unit_policy_attachment ?reconciliation_unit ?policy_attachment)
      )
    :effect
      (and
        (policy_attachment_available ?policy_attachment)
        (not
          (reconciliation_unit_policy_attachment ?reconciliation_unit ?policy_attachment)
        )
      )
  )
  (:action mark_candidate_version_a_primary
    :parameters (?source_a_record - source_a_record ?candidate_version_a - candidate_version_a ?state_delta - state_delta)
    :precondition
      (and
        (entity_ready ?source_a_record)
        (entity_attached_state_delta ?source_a_record ?state_delta)
        (source_a_has_candidate_version ?source_a_record ?candidate_version_a)
        (not
          (candidate_version_a_primary ?candidate_version_a)
        )
        (not
          (candidate_version_a_secondary ?candidate_version_a)
        )
      )
    :effect (candidate_version_a_primary ?candidate_version_a)
  )
  (:action confirm_source_a_candidate
    :parameters (?source_a_record - source_a_record ?candidate_version_a - candidate_version_a ?binding_handle - binding_handle)
    :precondition
      (and
        (entity_ready ?source_a_record)
        (entity_bound_handle ?source_a_record ?binding_handle)
        (source_a_has_candidate_version ?source_a_record ?candidate_version_a)
        (candidate_version_a_primary ?candidate_version_a)
        (not
          (source_a_prepared ?source_a_record)
        )
      )
    :effect
      (and
        (source_a_prepared ?source_a_record)
        (source_a_confirmed ?source_a_record)
      )
  )
  (:action apply_auxiliary_token_to_source_a
    :parameters (?source_a_record - source_a_record ?candidate_version_a - candidate_version_a ?auxiliary_token - auxiliary_token)
    :precondition
      (and
        (entity_ready ?source_a_record)
        (source_a_has_candidate_version ?source_a_record ?candidate_version_a)
        (auxiliary_token_available ?auxiliary_token)
        (not
          (source_a_prepared ?source_a_record)
        )
      )
    :effect
      (and
        (candidate_version_a_secondary ?candidate_version_a)
        (source_a_prepared ?source_a_record)
        (source_a_attached_aux_token ?source_a_record ?auxiliary_token)
        (not
          (auxiliary_token_available ?auxiliary_token)
        )
      )
  )
  (:action promote_source_a_candidate_with_aux_token
    :parameters (?source_a_record - source_a_record ?candidate_version_a - candidate_version_a ?state_delta - state_delta ?auxiliary_token - auxiliary_token)
    :precondition
      (and
        (entity_ready ?source_a_record)
        (entity_attached_state_delta ?source_a_record ?state_delta)
        (source_a_has_candidate_version ?source_a_record ?candidate_version_a)
        (candidate_version_a_secondary ?candidate_version_a)
        (source_a_attached_aux_token ?source_a_record ?auxiliary_token)
        (not
          (source_a_confirmed ?source_a_record)
        )
      )
    :effect
      (and
        (candidate_version_a_primary ?candidate_version_a)
        (source_a_confirmed ?source_a_record)
        (auxiliary_token_available ?auxiliary_token)
        (not
          (source_a_attached_aux_token ?source_a_record ?auxiliary_token)
        )
      )
  )
  (:action mark_candidate_version_b_primary
    :parameters (?source_b_record - source_b_record ?candidate_version_b - candidate_version_b ?state_delta - state_delta)
    :precondition
      (and
        (entity_ready ?source_b_record)
        (entity_attached_state_delta ?source_b_record ?state_delta)
        (source_b_has_candidate_version ?source_b_record ?candidate_version_b)
        (not
          (candidate_version_b_primary ?candidate_version_b)
        )
        (not
          (candidate_version_b_secondary ?candidate_version_b)
        )
      )
    :effect (candidate_version_b_primary ?candidate_version_b)
  )
  (:action confirm_source_b_candidate
    :parameters (?source_b_record - source_b_record ?candidate_version_b - candidate_version_b ?binding_handle - binding_handle)
    :precondition
      (and
        (entity_ready ?source_b_record)
        (entity_bound_handle ?source_b_record ?binding_handle)
        (source_b_has_candidate_version ?source_b_record ?candidate_version_b)
        (candidate_version_b_primary ?candidate_version_b)
        (not
          (source_b_prepared ?source_b_record)
        )
      )
    :effect
      (and
        (source_b_prepared ?source_b_record)
        (source_b_confirmed ?source_b_record)
      )
  )
  (:action apply_auxiliary_token_to_source_b
    :parameters (?source_b_record - source_b_record ?candidate_version_b - candidate_version_b ?auxiliary_token - auxiliary_token)
    :precondition
      (and
        (entity_ready ?source_b_record)
        (source_b_has_candidate_version ?source_b_record ?candidate_version_b)
        (auxiliary_token_available ?auxiliary_token)
        (not
          (source_b_prepared ?source_b_record)
        )
      )
    :effect
      (and
        (candidate_version_b_secondary ?candidate_version_b)
        (source_b_prepared ?source_b_record)
        (source_b_attached_aux_token ?source_b_record ?auxiliary_token)
        (not
          (auxiliary_token_available ?auxiliary_token)
        )
      )
  )
  (:action promote_source_b_candidate_with_aux_token
    :parameters (?source_b_record - source_b_record ?candidate_version_b - candidate_version_b ?state_delta - state_delta ?auxiliary_token - auxiliary_token)
    :precondition
      (and
        (entity_ready ?source_b_record)
        (entity_attached_state_delta ?source_b_record ?state_delta)
        (source_b_has_candidate_version ?source_b_record ?candidate_version_b)
        (candidate_version_b_secondary ?candidate_version_b)
        (source_b_attached_aux_token ?source_b_record ?auxiliary_token)
        (not
          (source_b_confirmed ?source_b_record)
        )
      )
    :effect
      (and
        (candidate_version_b_primary ?candidate_version_b)
        (source_b_confirmed ?source_b_record)
        (auxiliary_token_available ?auxiliary_token)
        (not
          (source_b_attached_aux_token ?source_b_record ?auxiliary_token)
        )
      )
  )
  (:action assemble_and_reserve_merged_candidate
    :parameters (?source_a_record - source_a_record ?source_b_record - source_b_record ?candidate_version_a - candidate_version_a ?candidate_version_b - candidate_version_b ?merged_snapshot_candidate - merged_candidate)
    :precondition
      (and
        (source_a_prepared ?source_a_record)
        (source_b_prepared ?source_b_record)
        (source_a_has_candidate_version ?source_a_record ?candidate_version_a)
        (source_b_has_candidate_version ?source_b_record ?candidate_version_b)
        (candidate_version_a_primary ?candidate_version_a)
        (candidate_version_b_primary ?candidate_version_b)
        (source_a_confirmed ?source_a_record)
        (source_b_confirmed ?source_b_record)
        (merged_candidate_available ?merged_snapshot_candidate)
      )
    :effect
      (and
        (merged_candidate_reserved ?merged_snapshot_candidate)
        (merged_includes_candidate_a ?merged_snapshot_candidate ?candidate_version_a)
        (merged_includes_candidate_b ?merged_snapshot_candidate ?candidate_version_b)
        (not
          (merged_candidate_available ?merged_snapshot_candidate)
        )
      )
  )
  (:action assemble_and_reserve_merged_candidate_with_partial_validation
    :parameters (?source_a_record - source_a_record ?source_b_record - source_b_record ?candidate_version_a - candidate_version_a ?candidate_version_b - candidate_version_b ?merged_snapshot_candidate - merged_candidate)
    :precondition
      (and
        (source_a_prepared ?source_a_record)
        (source_b_prepared ?source_b_record)
        (source_a_has_candidate_version ?source_a_record ?candidate_version_a)
        (source_b_has_candidate_version ?source_b_record ?candidate_version_b)
        (candidate_version_a_secondary ?candidate_version_a)
        (candidate_version_b_primary ?candidate_version_b)
        (not
          (source_a_confirmed ?source_a_record)
        )
        (source_b_confirmed ?source_b_record)
        (merged_candidate_available ?merged_snapshot_candidate)
      )
    :effect
      (and
        (merged_candidate_reserved ?merged_snapshot_candidate)
        (merged_includes_candidate_a ?merged_snapshot_candidate ?candidate_version_a)
        (merged_includes_candidate_b ?merged_snapshot_candidate ?candidate_version_b)
        (merged_candidate_partial_pass ?merged_snapshot_candidate)
        (not
          (merged_candidate_available ?merged_snapshot_candidate)
        )
      )
  )
  (:action assemble_and_reserve_merged_candidate_with_secondary_validation
    :parameters (?source_a_record - source_a_record ?source_b_record - source_b_record ?candidate_version_a - candidate_version_a ?candidate_version_b - candidate_version_b ?merged_snapshot_candidate - merged_candidate)
    :precondition
      (and
        (source_a_prepared ?source_a_record)
        (source_b_prepared ?source_b_record)
        (source_a_has_candidate_version ?source_a_record ?candidate_version_a)
        (source_b_has_candidate_version ?source_b_record ?candidate_version_b)
        (candidate_version_a_primary ?candidate_version_a)
        (candidate_version_b_secondary ?candidate_version_b)
        (source_a_confirmed ?source_a_record)
        (not
          (source_b_confirmed ?source_b_record)
        )
        (merged_candidate_available ?merged_snapshot_candidate)
      )
    :effect
      (and
        (merged_candidate_reserved ?merged_snapshot_candidate)
        (merged_includes_candidate_a ?merged_snapshot_candidate ?candidate_version_a)
        (merged_includes_candidate_b ?merged_snapshot_candidate ?candidate_version_b)
        (merged_candidate_secondary_pass ?merged_snapshot_candidate)
        (not
          (merged_candidate_available ?merged_snapshot_candidate)
        )
      )
  )
  (:action assemble_and_reserve_merged_candidate_with_both_validations
    :parameters (?source_a_record - source_a_record ?source_b_record - source_b_record ?candidate_version_a - candidate_version_a ?candidate_version_b - candidate_version_b ?merged_snapshot_candidate - merged_candidate)
    :precondition
      (and
        (source_a_prepared ?source_a_record)
        (source_b_prepared ?source_b_record)
        (source_a_has_candidate_version ?source_a_record ?candidate_version_a)
        (source_b_has_candidate_version ?source_b_record ?candidate_version_b)
        (candidate_version_a_secondary ?candidate_version_a)
        (candidate_version_b_secondary ?candidate_version_b)
        (not
          (source_a_confirmed ?source_a_record)
        )
        (not
          (source_b_confirmed ?source_b_record)
        )
        (merged_candidate_available ?merged_snapshot_candidate)
      )
    :effect
      (and
        (merged_candidate_reserved ?merged_snapshot_candidate)
        (merged_includes_candidate_a ?merged_snapshot_candidate ?candidate_version_a)
        (merged_includes_candidate_b ?merged_snapshot_candidate ?candidate_version_b)
        (merged_candidate_partial_pass ?merged_snapshot_candidate)
        (merged_candidate_secondary_pass ?merged_snapshot_candidate)
        (not
          (merged_candidate_available ?merged_snapshot_candidate)
        )
      )
  )
  (:action finalize_merged_candidate_pass
    :parameters (?merged_snapshot_candidate - merged_candidate ?source_a_record - source_a_record ?state_delta - state_delta)
    :precondition
      (and
        (merged_candidate_reserved ?merged_snapshot_candidate)
        (source_a_prepared ?source_a_record)
        (entity_attached_state_delta ?source_a_record ?state_delta)
        (not
          (merged_candidate_final_pass ?merged_snapshot_candidate)
        )
      )
    :effect (merged_candidate_final_pass ?merged_snapshot_candidate)
  )
  (:action consume_validation_artifact_for_merged_candidate
    :parameters (?reconciliation_unit - reconciliation_unit ?validation_artifact - validation_artifact ?merged_snapshot_candidate - merged_candidate)
    :precondition
      (and
        (entity_ready ?reconciliation_unit)
        (reconciliation_unit_has_merged_candidate ?reconciliation_unit ?merged_snapshot_candidate)
        (reconciliation_unit_has_validation_artifact ?reconciliation_unit ?validation_artifact)
        (validation_artifact_available ?validation_artifact)
        (merged_candidate_reserved ?merged_snapshot_candidate)
        (merged_candidate_final_pass ?merged_snapshot_candidate)
        (not
          (validation_artifact_consumed ?validation_artifact)
        )
      )
    :effect
      (and
        (validation_artifact_consumed ?validation_artifact)
        (validation_artifact_for_merged_candidate ?validation_artifact ?merged_snapshot_candidate)
        (not
          (validation_artifact_available ?validation_artifact)
        )
      )
  )
  (:action enable_unit_resource_attachment
    :parameters (?reconciliation_unit - reconciliation_unit ?validation_artifact - validation_artifact ?merged_snapshot_candidate - merged_candidate ?state_delta - state_delta)
    :precondition
      (and
        (entity_ready ?reconciliation_unit)
        (reconciliation_unit_has_validation_artifact ?reconciliation_unit ?validation_artifact)
        (validation_artifact_consumed ?validation_artifact)
        (validation_artifact_for_merged_candidate ?validation_artifact ?merged_snapshot_candidate)
        (entity_attached_state_delta ?reconciliation_unit ?state_delta)
        (not
          (merged_candidate_partial_pass ?merged_snapshot_candidate)
        )
        (not
          (reconciliation_unit_configuration_applied ?reconciliation_unit)
        )
      )
    :effect (reconciliation_unit_configuration_applied ?reconciliation_unit)
  )
  (:action bind_configuration_flag_to_unit
    :parameters (?reconciliation_unit - reconciliation_unit ?configuration_flag - configuration_flag)
    :precondition
      (and
        (entity_ready ?reconciliation_unit)
        (configuration_flag_available ?configuration_flag)
        (not
          (reconciliation_unit_has_configuration_flag ?reconciliation_unit)
        )
      )
    :effect
      (and
        (reconciliation_unit_has_configuration_flag ?reconciliation_unit)
        (reconciliation_unit_configuration_flag ?reconciliation_unit ?configuration_flag)
        (not
          (configuration_flag_available ?configuration_flag)
        )
      )
