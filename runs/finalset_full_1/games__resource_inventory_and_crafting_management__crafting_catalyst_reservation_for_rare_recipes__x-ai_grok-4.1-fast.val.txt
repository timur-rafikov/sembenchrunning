--- domain.pddl (as passed to VAL) ---
(define (domain crafting_catalyst_reservation)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_category - object material_category - object component_category - object objective_category - object craft_objective - objective_category catalyst_unit - resource_category consumable_unit - resource_category maintenance_tool - resource_category optional_modifier_slot - resource_category consumable_modifier - resource_category preserved_reserve - resource_category equipment_upgrade - resource_category advanced_upgrade - resource_category material_token - material_category material_batch - material_category special_attachment - material_category material_variant_a - component_category material_variant_b - component_category intermediate_component_slot - component_category preparation_category - craft_objective preparation_subcategory - craft_objective primary_preparation_slot - preparation_category secondary_preparation_slot - preparation_category rare_recipe_instance - preparation_subcategory)
  (:predicates
    (objective_selected ?craft_objective - craft_objective)
    (objective_prepared ?craft_objective - craft_objective)
    (catalyst_reserved_for_objective ?craft_objective - craft_objective)
    (completion_locked ?craft_objective - craft_objective)
    (finalization_ready ?craft_objective - craft_objective)
    (consumable_preserved ?craft_objective - craft_objective)
    (catalyst_available ?catalyst_unit - catalyst_unit)
    (catalyst_assigned ?craft_objective - craft_objective ?catalyst_unit - catalyst_unit)
    (consumable_available ?consumable_unit - consumable_unit)
    (consumable_assigned ?craft_objective - craft_objective ?consumable_unit - consumable_unit)
    (tool_available ?maintenance_tool - maintenance_tool)
    (tool_assigned ?craft_objective - craft_objective ?maintenance_tool - maintenance_tool)
    (material_token_available ?material_token - material_token)
    (material_claimed_primary ?primary_preparation_slot - primary_preparation_slot ?material_token - material_token)
    (material_claimed_secondary ?secondary_preparation_slot - secondary_preparation_slot ?material_token - material_token)
    (preparation_slot_variant_a ?primary_preparation_slot - primary_preparation_slot ?material_variant_a - material_variant_a)
    (variant_a_claimed_by_consumable ?material_variant_a - material_variant_a)
    (variant_a_claimed_by_token ?material_variant_a - material_variant_a)
    (primary_branch_ready ?primary_preparation_slot - primary_preparation_slot)
    (preparation_slot_variant_b ?secondary_preparation_slot - secondary_preparation_slot ?material_variant_b - material_variant_b)
    (variant_b_claimed_by_consumable ?material_variant_b - material_variant_b)
    (variant_b_claimed_by_token ?material_variant_b - material_variant_b)
    (secondary_branch_ready ?secondary_preparation_slot - secondary_preparation_slot)
    (intermediate_slot_available ?intermediate_component_slot - intermediate_component_slot)
    (intermediate_component_present ?intermediate_component_slot - intermediate_component_slot)
    (intermediate_contains_variant_a ?intermediate_component_slot - intermediate_component_slot ?material_variant_a - material_variant_a)
    (intermediate_contains_variant_b ?intermediate_component_slot - intermediate_component_slot ?material_variant_b - material_variant_b)
    (intermediate_priority_variant_a ?intermediate_component_slot - intermediate_component_slot)
    (intermediate_priority_variant_b ?intermediate_component_slot - intermediate_component_slot)
    (intermediate_activated ?intermediate_component_slot - intermediate_component_slot)
    (recipe_primary_slot_link ?rare_recipe_instance - rare_recipe_instance ?primary_preparation_slot - primary_preparation_slot)
    (recipe_secondary_slot_link ?rare_recipe_instance - rare_recipe_instance ?secondary_preparation_slot - secondary_preparation_slot)
    (recipe_intermediate_slot_link ?rare_recipe_instance - rare_recipe_instance ?intermediate_component_slot - intermediate_component_slot)
    (material_batch_available ?material_batch - material_batch)
    (material_batch_assigned ?rare_recipe_instance - rare_recipe_instance ?material_batch - material_batch)
    (material_batch_prepared ?material_batch - material_batch)
    (material_batch_assigned_to_intermediate ?material_batch - material_batch ?intermediate_component_slot - intermediate_component_slot)
    (modification_stage_unlocked ?rare_recipe_instance - rare_recipe_instance)
    (modifiers_attached ?rare_recipe_instance - rare_recipe_instance)
    (modification_configuration_complete ?rare_recipe_instance - rare_recipe_instance)
    (optional_modifier_attached ?rare_recipe_instance - rare_recipe_instance)
    (optional_modifier_registered ?rare_recipe_instance - rare_recipe_instance)
    (modifier_slot_active ?rare_recipe_instance - rare_recipe_instance)
    (quality_check_passed ?rare_recipe_instance - rare_recipe_instance)
    (attachment_available ?special_attachment - special_attachment)
    (attachment_assigned ?rare_recipe_instance - rare_recipe_instance ?special_attachment - special_attachment)
    (attachment_staged ?rare_recipe_instance - rare_recipe_instance)
    (attachment_tool_applied ?rare_recipe_instance - rare_recipe_instance)
    (attachment_validation_complete ?rare_recipe_instance - rare_recipe_instance)
    (optional_modifier_slot_available ?optional_modifier_slot - optional_modifier_slot)
    (optional_modifier_slot_assigned ?rare_recipe_instance - rare_recipe_instance ?optional_modifier_slot - optional_modifier_slot)
    (consumable_modifier_available ?consumable_modifier - consumable_modifier)
    (consumable_modifier_assigned ?rare_recipe_instance - rare_recipe_instance ?consumable_modifier - consumable_modifier)
    (equipment_upgrade_available ?equipment_upgrade - equipment_upgrade)
    (equipment_upgrade_assigned ?rare_recipe_instance - rare_recipe_instance ?equipment_upgrade - equipment_upgrade)
    (advanced_upgrade_available ?advanced_upgrade - advanced_upgrade)
    (advanced_upgrade_assigned ?rare_recipe_instance - rare_recipe_instance ?advanced_upgrade - advanced_upgrade)
    (preserved_reserve_available ?preserved_consumable_reserve - preserved_reserve)
    (preserved_reserve_assigned ?craft_objective - craft_objective ?preserved_consumable_reserve - preserved_reserve)
    (primary_variant_locked ?primary_preparation_slot - primary_preparation_slot)
    (secondary_variant_locked ?secondary_preparation_slot - secondary_preparation_slot)
    (completion_marked ?rare_recipe_instance - rare_recipe_instance)
  )
  (:action select_craft_objective
    :parameters (?craft_objective - craft_objective)
    :precondition
      (and
        (not
          (objective_selected ?craft_objective)
        )
        (not
          (completion_locked ?craft_objective)
        )
      )
    :effect (objective_selected ?craft_objective)
  )
  (:action assign_catalyst_to_objective
    :parameters (?craft_objective - craft_objective ?catalyst_unit - catalyst_unit)
    :precondition
      (and
        (objective_selected ?craft_objective)
        (not
          (catalyst_reserved_for_objective ?craft_objective)
        )
        (catalyst_available ?catalyst_unit)
      )
    :effect
      (and
        (catalyst_reserved_for_objective ?craft_objective)
        (catalyst_assigned ?craft_objective ?catalyst_unit)
        (not
          (catalyst_available ?catalyst_unit)
        )
      )
  )
  (:action assign_consumable_to_objective
    :parameters (?craft_objective - craft_objective ?consumable_unit - consumable_unit)
    :precondition
      (and
        (objective_selected ?craft_objective)
        (catalyst_reserved_for_objective ?craft_objective)
        (consumable_available ?consumable_unit)
      )
    :effect
      (and
        (consumable_assigned ?craft_objective ?consumable_unit)
        (not
          (consumable_available ?consumable_unit)
        )
      )
  )
  (:action mark_objective_prepared
    :parameters (?craft_objective - craft_objective ?consumable_unit - consumable_unit)
    :precondition
      (and
        (objective_selected ?craft_objective)
        (catalyst_reserved_for_objective ?craft_objective)
        (consumable_assigned ?craft_objective ?consumable_unit)
        (not
          (objective_prepared ?craft_objective)
        )
      )
    :effect (objective_prepared ?craft_objective)
  )
  (:action release_consumable_from_objective
    :parameters (?craft_objective - craft_objective ?consumable_unit - consumable_unit)
    :precondition
      (and
        (consumable_assigned ?craft_objective ?consumable_unit)
      )
    :effect
      (and
        (consumable_available ?consumable_unit)
        (not
          (consumable_assigned ?craft_objective ?consumable_unit)
        )
      )
  )
  (:action checkout_tool
    :parameters (?craft_objective - craft_objective ?maintenance_tool - maintenance_tool)
    :precondition
      (and
        (objective_prepared ?craft_objective)
        (tool_available ?maintenance_tool)
      )
    :effect
      (and
        (tool_assigned ?craft_objective ?maintenance_tool)
        (not
          (tool_available ?maintenance_tool)
        )
      )
  )
  (:action return_tool
    :parameters (?craft_objective - craft_objective ?maintenance_tool - maintenance_tool)
    :precondition
      (and
        (tool_assigned ?craft_objective ?maintenance_tool)
      )
    :effect
      (and
        (tool_available ?maintenance_tool)
        (not
          (tool_assigned ?craft_objective ?maintenance_tool)
        )
      )
  )
  (:action attach_equipment_upgrade
    :parameters (?rare_recipe_instance - rare_recipe_instance ?equipment_upgrade - equipment_upgrade)
    :precondition
      (and
        (objective_prepared ?rare_recipe_instance)
        (equipment_upgrade_available ?equipment_upgrade)
      )
    :effect
      (and
        (equipment_upgrade_assigned ?rare_recipe_instance ?equipment_upgrade)
        (not
          (equipment_upgrade_available ?equipment_upgrade)
        )
      )
  )
  (:action detach_equipment_upgrade
    :parameters (?rare_recipe_instance - rare_recipe_instance ?equipment_upgrade - equipment_upgrade)
    :precondition
      (and
        (equipment_upgrade_assigned ?rare_recipe_instance ?equipment_upgrade)
      )
    :effect
      (and
        (equipment_upgrade_available ?equipment_upgrade)
        (not
          (equipment_upgrade_assigned ?rare_recipe_instance ?equipment_upgrade)
        )
      )
  )
  (:action attach_advanced_upgrade
    :parameters (?rare_recipe_instance - rare_recipe_instance ?advanced_upgrade - advanced_upgrade)
    :precondition
      (and
        (objective_prepared ?rare_recipe_instance)
        (advanced_upgrade_available ?advanced_upgrade)
      )
    :effect
      (and
        (advanced_upgrade_assigned ?rare_recipe_instance ?advanced_upgrade)
        (not
          (advanced_upgrade_available ?advanced_upgrade)
        )
      )
  )
  (:action detach_advanced_upgrade
    :parameters (?rare_recipe_instance - rare_recipe_instance ?advanced_upgrade - advanced_upgrade)
    :precondition
      (and
        (advanced_upgrade_assigned ?rare_recipe_instance ?advanced_upgrade)
      )
    :effect
      (and
        (advanced_upgrade_available ?advanced_upgrade)
        (not
          (advanced_upgrade_assigned ?rare_recipe_instance ?advanced_upgrade)
        )
      )
  )
  (:action stage_primary_variant
    :parameters (?primary_preparation_slot - primary_preparation_slot ?material_variant_a - material_variant_a ?consumable_unit - consumable_unit)
    :precondition
      (and
        (objective_prepared ?primary_preparation_slot)
        (consumable_assigned ?primary_preparation_slot ?consumable_unit)
        (preparation_slot_variant_a ?primary_preparation_slot ?material_variant_a)
        (not
          (variant_a_claimed_by_consumable ?material_variant_a)
        )
        (not
          (variant_a_claimed_by_token ?material_variant_a)
        )
      )
    :effect (variant_a_claimed_by_consumable ?material_variant_a)
  )
  (:action claim_primary_variant_with_tool
    :parameters (?primary_preparation_slot - primary_preparation_slot ?material_variant_a - material_variant_a ?maintenance_tool - maintenance_tool)
    :precondition
      (and
        (objective_prepared ?primary_preparation_slot)
        (tool_assigned ?primary_preparation_slot ?maintenance_tool)
        (preparation_slot_variant_a ?primary_preparation_slot ?material_variant_a)
        (variant_a_claimed_by_consumable ?material_variant_a)
        (not
          (primary_variant_locked ?primary_preparation_slot)
        )
      )
    :effect
      (and
        (primary_variant_locked ?primary_preparation_slot)
        (primary_branch_ready ?primary_preparation_slot)
      )
  )
  (:action claim_primary_variant_with_material_token
    :parameters (?primary_preparation_slot - primary_preparation_slot ?material_variant_a - material_variant_a ?material_token - material_token)
    :precondition
      (and
        (objective_prepared ?primary_preparation_slot)
        (preparation_slot_variant_a ?primary_preparation_slot ?material_variant_a)
        (material_token_available ?material_token)
        (not
          (primary_variant_locked ?primary_preparation_slot)
        )
      )
    :effect
      (and
        (variant_a_claimed_by_token ?material_variant_a)
        (primary_variant_locked ?primary_preparation_slot)
        (material_claimed_primary ?primary_preparation_slot ?material_token)
        (not
          (material_token_available ?material_token)
        )
      )
  )
  (:action finalize_primary_variant_from_token
    :parameters (?primary_preparation_slot - primary_preparation_slot ?material_variant_a - material_variant_a ?consumable_unit - consumable_unit ?material_token - material_token)
    :precondition
      (and
        (objective_prepared ?primary_preparation_slot)
        (consumable_assigned ?primary_preparation_slot ?consumable_unit)
        (preparation_slot_variant_a ?primary_preparation_slot ?material_variant_a)
        (variant_a_claimed_by_token ?material_variant_a)
        (material_claimed_primary ?primary_preparation_slot ?material_token)
        (not
          (primary_branch_ready ?primary_preparation_slot)
        )
      )
    :effect
      (and
        (variant_a_claimed_by_consumable ?material_variant_a)
        (primary_branch_ready ?primary_preparation_slot)
        (material_token_available ?material_token)
        (not
          (material_claimed_primary ?primary_preparation_slot ?material_token)
        )
      )
  )
  (:action stage_secondary_variant
    :parameters (?secondary_preparation_slot - secondary_preparation_slot ?material_variant_b - material_variant_b ?consumable_unit - consumable_unit)
    :precondition
      (and
        (objective_prepared ?secondary_preparation_slot)
        (consumable_assigned ?secondary_preparation_slot ?consumable_unit)
        (preparation_slot_variant_b ?secondary_preparation_slot ?material_variant_b)
        (not
          (variant_b_claimed_by_consumable ?material_variant_b)
        )
        (not
          (variant_b_claimed_by_token ?material_variant_b)
        )
      )
    :effect (variant_b_claimed_by_consumable ?material_variant_b)
  )
  (:action claim_secondary_variant_with_tool
    :parameters (?secondary_preparation_slot - secondary_preparation_slot ?material_variant_b - material_variant_b ?maintenance_tool - maintenance_tool)
    :precondition
      (and
        (objective_prepared ?secondary_preparation_slot)
        (tool_assigned ?secondary_preparation_slot ?maintenance_tool)
        (preparation_slot_variant_b ?secondary_preparation_slot ?material_variant_b)
        (variant_b_claimed_by_consumable ?material_variant_b)
        (not
          (secondary_variant_locked ?secondary_preparation_slot)
        )
      )
    :effect
      (and
        (secondary_variant_locked ?secondary_preparation_slot)
        (secondary_branch_ready ?secondary_preparation_slot)
      )
  )
  (:action claim_secondary_variant_with_material_token
    :parameters (?secondary_preparation_slot - secondary_preparation_slot ?material_variant_b - material_variant_b ?material_token - material_token)
    :precondition
      (and
        (objective_prepared ?secondary_preparation_slot)
        (preparation_slot_variant_b ?secondary_preparation_slot ?material_variant_b)
        (material_token_available ?material_token)
        (not
          (secondary_variant_locked ?secondary_preparation_slot)
        )
      )
    :effect
      (and
        (variant_b_claimed_by_token ?material_variant_b)
        (secondary_variant_locked ?secondary_preparation_slot)
        (material_claimed_secondary ?secondary_preparation_slot ?material_token)
        (not
          (material_token_available ?material_token)
        )
      )
  )
  (:action finalize_secondary_variant_from_token
    :parameters (?secondary_preparation_slot - secondary_preparation_slot ?material_variant_b - material_variant_b ?consumable_unit - consumable_unit ?material_token - material_token)
    :precondition
      (and
        (objective_prepared ?secondary_preparation_slot)
        (consumable_assigned ?secondary_preparation_slot ?consumable_unit)
        (preparation_slot_variant_b ?secondary_preparation_slot ?material_variant_b)
        (variant_b_claimed_by_token ?material_variant_b)
        (material_claimed_secondary ?secondary_preparation_slot ?material_token)
        (not
          (secondary_branch_ready ?secondary_preparation_slot)
        )
      )
    :effect
      (and
        (variant_b_claimed_by_consumable ?material_variant_b)
        (secondary_branch_ready ?secondary_preparation_slot)
        (material_token_available ?material_token)
        (not
          (material_claimed_secondary ?secondary_preparation_slot ?material_token)
        )
      )
  )
  (:action assemble_intermediate_component
    :parameters (?primary_preparation_slot - primary_preparation_slot ?secondary_preparation_slot - secondary_preparation_slot ?material_variant_a - material_variant_a ?material_variant_b - material_variant_b ?intermediate_component_slot - intermediate_component_slot)
    :precondition
      (and
        (primary_variant_locked ?primary_preparation_slot)
        (secondary_variant_locked ?secondary_preparation_slot)
        (preparation_slot_variant_a ?primary_preparation_slot ?material_variant_a)
        (preparation_slot_variant_b ?secondary_preparation_slot ?material_variant_b)
        (variant_a_claimed_by_consumable ?material_variant_a)
        (variant_b_claimed_by_consumable ?material_variant_b)
        (primary_branch_ready ?primary_preparation_slot)
        (secondary_branch_ready ?secondary_preparation_slot)
        (intermediate_slot_available ?intermediate_component_slot)
      )
    :effect
      (and
        (intermediate_component_present ?intermediate_component_slot)
        (intermediate_contains_variant_a ?intermediate_component_slot ?material_variant_a)
        (intermediate_contains_variant_b ?intermediate_component_slot ?material_variant_b)
        (not
          (intermediate_slot_available ?intermediate_component_slot)
        )
      )
  )
  (:action assemble_intermediate_with_primary_priority
    :parameters (?primary_preparation_slot - primary_preparation_slot ?secondary_preparation_slot - secondary_preparation_slot ?material_variant_a - material_variant_a ?material_variant_b - material_variant_b ?intermediate_component_slot - intermediate_component_slot)
    :precondition
      (and
        (primary_variant_locked ?primary_preparation_slot)
        (secondary_variant_locked ?secondary_preparation_slot)
        (preparation_slot_variant_a ?primary_preparation_slot ?material_variant_a)
        (preparation_slot_variant_b ?secondary_preparation_slot ?material_variant_b)
        (variant_a_claimed_by_token ?material_variant_a)
        (variant_b_claimed_by_consumable ?material_variant_b)
        (not
          (primary_branch_ready ?primary_preparation_slot)
        )
        (secondary_branch_ready ?secondary_preparation_slot)
        (intermediate_slot_available ?intermediate_component_slot)
      )
    :effect
      (and
        (intermediate_component_present ?intermediate_component_slot)
        (intermediate_contains_variant_a ?intermediate_component_slot ?material_variant_a)
        (intermediate_contains_variant_b ?intermediate_component_slot ?material_variant_b)
        (intermediate_priority_variant_a ?intermediate_component_slot)
        (not
          (intermediate_slot_available ?intermediate_component_slot)
        )
      )
  )
  (:action assemble_intermediate_with_secondary_priority
    :parameters (?primary_preparation_slot - primary_preparation_slot ?secondary_preparation_slot - secondary_preparation_slot ?material_variant_a - material_variant_a ?material_variant_b - material_variant_b ?intermediate_component_slot - intermediate_component_slot)
    :precondition
      (and
        (primary_variant_locked ?primary_preparation_slot)
        (secondary_variant_locked ?secondary_preparation_slot)
        (preparation_slot_variant_a ?primary_preparation_slot ?material_variant_a)
        (preparation_slot_variant_b ?secondary_preparation_slot ?material_variant_b)
        (variant_a_claimed_by_consumable ?material_variant_a)
        (variant_b_claimed_by_token ?material_variant_b)
        (primary_branch_ready ?primary_preparation_slot)
        (not
          (secondary_branch_ready ?secondary_preparation_slot)
        )
        (intermediate_slot_available ?intermediate_component_slot)
      )
    :effect
      (and
        (intermediate_component_present ?intermediate_component_slot)
        (intermediate_contains_variant_a ?intermediate_component_slot ?material_variant_a)
        (intermediate_contains_variant_b ?intermediate_component_slot ?material_variant_b)
        (intermediate_priority_variant_b ?intermediate_component_slot)
        (not
          (intermediate_slot_available ?intermediate_component_slot)
        )
      )
  )
  (:action assemble_intermediate_with_both_priorities
    :parameters (?primary_preparation_slot - primary_preparation_slot ?secondary_preparation_slot - secondary_preparation_slot ?material_variant_a - material_variant_a ?material_variant_b - material_variant_b ?intermediate_component_slot - intermediate_component_slot)
    :precondition
      (and
        (primary_variant_locked ?primary_preparation_slot)
        (secondary_variant_locked ?secondary_preparation_slot)
        (preparation_slot_variant_a ?primary_preparation_slot ?material_variant_a)
        (preparation_slot_variant_b ?secondary_preparation_slot ?material_variant_b)
        (variant_a_claimed_by_token ?material_variant_a)
        (variant_b_claimed_by_token ?material_variant_b)
        (not
          (primary_branch_ready ?primary_preparation_slot)
        )
        (not
          (secondary_branch_ready ?secondary_preparation_slot)
        )
        (intermediate_slot_available ?intermediate_component_slot)
      )
    :effect
      (and
        (intermediate_component_present ?intermediate_component_slot)
        (intermediate_contains_variant_a ?intermediate_component_slot ?material_variant_a)
        (intermediate_contains_variant_b ?intermediate_component_slot ?material_variant_b)
        (intermediate_priority_variant_a ?intermediate_component_slot)
        (intermediate_priority_variant_b ?intermediate_component_slot)
        (not
          (intermediate_slot_available ?intermediate_component_slot)
        )
      )
  )
  (:action activate_intermediate_component
    :parameters (?intermediate_component_slot - intermediate_component_slot ?primary_preparation_slot - primary_preparation_slot ?consumable_unit - consumable_unit)
    :precondition
      (and
        (intermediate_component_present ?intermediate_component_slot)
        (primary_variant_locked ?primary_preparation_slot)
        (consumable_assigned ?primary_preparation_slot ?consumable_unit)
        (not
          (intermediate_activated ?intermediate_component_slot)
        )
      )
    :effect (intermediate_activated ?intermediate_component_slot)
  )
  (:action reserve_material_batch_for_recipe
    :parameters (?rare_recipe_instance - rare_recipe_instance ?material_batch - material_batch ?intermediate_component_slot - intermediate_component_slot)
    :precondition
      (and
        (objective_prepared ?rare_recipe_instance)
        (recipe_intermediate_slot_link ?rare_recipe_instance ?intermediate_component_slot)
        (material_batch_assigned ?rare_recipe_instance ?material_batch)
        (material_batch_available ?material_batch)
        (intermediate_component_present ?intermediate_component_slot)
        (intermediate_activated ?intermediate_component_slot)
        (not
          (material_batch_prepared ?material_batch)
        )
      )
    :effect
      (and
        (material_batch_prepared ?material_batch)
        (material_batch_assigned_to_intermediate ?material_batch ?intermediate_component_slot)
        (not
          (material_batch_available ?material_batch)
        )
      )
  )
  (:action unlock_modification_stage_for_recipe
    :parameters (?rare_recipe_instance - rare_recipe_instance ?material_batch - material_batch ?intermediate_component_slot - intermediate_component_slot ?consumable_unit - consumable_unit)
    :precondition
      (and
        (objective_prepared ?rare_recipe_instance)
        (material_batch_assigned ?rare_recipe_instance ?material_batch)
        (material_batch_prepared ?material_batch)
        (material_batch_assigned_to_intermediate ?material_batch ?intermediate_component_slot)
        (consumable_assigned ?rare_recipe_instance ?consumable_unit)
        (not
          (intermediate_priority_variant_a ?intermediate_component_slot)
        )
        (not
          (modification_stage_unlocked ?rare_recipe_instance)
        )
      )
    :effect (modification_stage_unlocked ?rare_recipe_instance)
  )
  (:action attach_optional_modifier_slot
    :parameters (?rare_recipe_instance - rare_recipe_instance ?optional_modifier_slot - optional_modifier_slot)
    :precondition
      (and
        (objective_prepared ?rare_recipe_instance)
        (optional_modifier_slot_available ?optional_modifier_slot)
        (not
          (optional_modifier_attached ?rare_recipe_instance)
        )
      )
    :effect
