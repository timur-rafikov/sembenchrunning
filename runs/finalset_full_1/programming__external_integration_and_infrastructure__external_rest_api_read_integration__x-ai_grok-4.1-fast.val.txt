  (:action register_endpoint
    :parameters (?endpoint - endpoint)
    :precondition
      (and
        (not
          (registered ?endpoint)
        )
        (not
          (integration_committed ?endpoint)
        )
      )
    :effect (registered ?endpoint)
  )
  (:action claim_connector_for_endpoint
    :parameters (?endpoint - endpoint ?connector - connector)
    :precondition
      (and
        (registered ?endpoint)
        (not
          (connector_allocated ?endpoint)
        )
        (connector_available ?connector)
      )
    :effect
      (and
        (connector_allocated ?endpoint)
        (connector_bound ?endpoint ?connector)
        (not
          (connector_available ?connector)
        )
      )
  )
  (:action bind_request_template_to_endpoint
    :parameters (?endpoint - endpoint ?request_template - api_request_template)

    :effect
      (and
        (bound_request_template ?endpoint ?request_template)
        (not
          (request_template_available ?request_template)
        )
      )
  )
  (:action finalize_endpoint_configuration
    :parameters (?endpoint - endpoint ?request_template - api_request_template)
    :precondition
      (and
        (registered ?endpoint)
        (connector_allocated ?endpoint)
        (bound_request_template ?endpoint ?request_template)

          (ready ?endpoint)

    :effect (ready ?endpoint)
  )
  (:action release_request_template_from_endpoint

        (request_template_available ?request_template)

          (bound_request_template ?endpoint ?request_template)
        )
      )
  )

        (credential_bound ?endpoint ?credential)

          (credential_available ?credential)
        )
