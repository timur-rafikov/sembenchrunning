    (fragment_confirmed ?fragment - fragment)

    (fragment_linked_activity ?fragment - fragment ?activity - activity)
    (provider_available ?service_provider - service_provider)
    (fragment_assigned_provider ?fragment - fragment ?service_provider - service_provider)
    (luggage_category_available ?luggage_category - luggage_category)
    (day_luggage_category_allocated ?day_segment - day_segment ?luggage_category - luggage_category)
    (time_segment_luggage_category_allocated ?time_segment - time_segment ?luggage_category - luggage_category)
    (day_has_storage_window ?day_segment - day_segment ?storage_time_window - storage_time_window)
    (storage_window_claimed ?storage_time_window - storage_time_window)
    (storage_window_held ?storage_time_window - storage_time_window)
    (day_segment_ready_for_reservation ?day_segment - day_segment)
    (time_segment_activity_window_linked ?time_segment - time_segment ?activity_time_window - activity_time_window)
    (activity_window_claimed ?activity_time_window - activity_time_window)
    (activity_window_held ?activity_time_window - activity_time_window)

    (reservation_record_available ?storage_reservation_record - storage_reservation_record)
    (reservation_record_allocated ?storage_reservation_record - storage_reservation_record)
    (reservation_assigned_storage_window ?storage_reservation_record - storage_reservation_record ?storage_time_window - storage_time_window)
    (reservation_assigned_activity_window ?storage_reservation_record - storage_reservation_record ?activity_time_window - activity_time_window)
    (reservation_option_requires_amenities ?storage_reservation_record - storage_reservation_record)
    (reservation_option_requires_staff_assignment ?storage_reservation_record - storage_reservation_record)
    (reservation_ready_for_luggage_allocation ?storage_reservation_record - storage_reservation_record)
    (plan_day_segment_link ?itinerary_plan - itinerary_plan ?day_segment - day_segment)
    (plan_time_segment_link ?itinerary_plan - itinerary_plan ?time_segment - time_segment)
    (plan_reservation_record_link ?itinerary_plan - itinerary_plan ?storage_reservation_record - storage_reservation_record)
    (luggage_unit_available ?luggage_unit - luggage_unit)
    (plan_has_luggage_unit ?itinerary_plan - itinerary_plan ?luggage_unit - luggage_unit)
    (luggage_unit_allocated ?luggage_unit - luggage_unit)
    (luggage_unit_assigned_to_reservation ?luggage_unit - luggage_unit ?storage_reservation_record - storage_reservation_record)
    (plan_units_locked ?itinerary_plan - itinerary_plan)
    (plan_payment_bound ?itinerary_plan - itinerary_plan)
    (plan_staffing_prepared ?itinerary_plan - itinerary_plan)
    (plan_has_amenity ?itinerary_plan - itinerary_plan)
    (plan_amenity_confirmed ?itinerary_plan - itinerary_plan)
    (plan_payment_confirmed ?itinerary_plan - itinerary_plan)
    (plan_units_ready_for_handling ?itinerary_plan - itinerary_plan)
    (voucher_available ?discount_voucher - discount_voucher)
    (plan_linked_voucher ?itinerary_plan - itinerary_plan ?discount_voucher - discount_voucher)
    (voucher_applied_to_plan ?itinerary_plan - itinerary_plan)
    (voucher_redeemed_for_plan ?itinerary_plan - itinerary_plan)

    (payment_option_available ?payment_option - payment_option)
    (plan_payment_attached ?itinerary_plan - itinerary_plan ?payment_option - payment_option)
    (staff_assignment_available ?staff_assignment - staff_assignment)
    (plan_staff_assigned ?itinerary_plan - itinerary_plan ?staff_assignment - staff_assignment)

    (time_segment_allocated ?time_segment - time_segment)
    (plan_finalized ?itinerary_plan - itinerary_plan)

  (:action create_traveler_request
    :parameters (?fragment - fragment)
    :precondition
      (and
        (not
          (request_initiated ?fragment)
        )
        (not
          (fragment_allocation_confirmed ?fragment)
        )
      )
    :effect (request_initiated ?fragment)
  )
  (:action select_storage_option_for_profile
    :parameters (?fragment - fragment ?storage_service_option - storage_service_option)
    :precondition
      (and
        (request_initiated ?fragment)
        (not
          (storage_option_selected ?fragment)
        )
        (storage_service_available ?storage_service_option)
      )
    :effect
      (and
        (storage_option_selected ?fragment)
        (fragment_selected_storage_option ?fragment ?storage_service_option)
        (not
          (storage_service_available ?storage_service_option)
        )
      )
  )
  (:action link_activity_to_profile
    :parameters (?fragment - fragment ?activity - activity)
    :precondition
      (and
        (request_initiated ?fragment)
        (storage_option_selected ?fragment)
        (activity_available ?activity)
      )
    :effect
      (and
        (fragment_linked_activity ?fragment ?activity)
        (not
          (activity_available ?activity)
        )
      )
  )
  (:action confirm_linked_activity
    :parameters (?fragment - fragment ?activity - activity)
    :precondition
      (and
        (request_initiated ?fragment)
        (storage_option_selected ?fragment)
        (fragment_linked_activity ?fragment ?activity)
        (not
          (fragment_confirmed ?fragment)
        )
      )
    :effect (fragment_confirmed ?fragment)
  )
  (:action unlink_activity_from_profile
    :parameters (?fragment - fragment ?activity - activity)
    :precondition
      (and
        (fragment_linked_activity ?fragment ?activity)
      )
    :effect
      (and
        (activity_available ?activity)
        (not
          (fragment_linked_activity ?fragment ?activity)
        )
      )
  )
  (:action bind_provider_to_profile
    :parameters (?fragment - fragment ?service_provider - service_provider)
    :precondition
      (and
        (fragment_confirmed ?fragment)
        (provider_available ?service_provider)
      )
    :effect
      (and
        (fragment_assigned_provider ?fragment ?service_provider)
        (not
          (provider_available ?service_provider)
        )
      )
  )
  (:action unbind_provider_from_profile
    :parameters (?fragment - fragment ?service_provider - service_provider)
    :precondition
      (and
        (fragment_assigned_provider ?fragment ?service_provider)
      )
    :effect
      (and
        (provider_available ?service_provider)
        (not
          (fragment_assigned_provider ?fragment ?service_provider)
        )
      )
  )
  (:action attach_payment_to_itinerary
    :parameters (?itinerary_plan - itinerary_plan ?payment_option - payment_option)
    :precondition
      (and
        (fragment_confirmed ?itinerary_plan)
        (payment_option_available ?payment_option)
      )
    :effect
      (and
        (plan_payment_attached ?itinerary_plan ?payment_option)
        (not
          (payment_option_available ?payment_option)
        )
      )
  )
  (:action detach_payment_from_itinerary
    :parameters (?itinerary_plan - itinerary_plan ?payment_option - payment_option)
    :precondition
      (and
        (plan_payment_attached ?itinerary_plan ?payment_option)
      )
    :effect
      (and
        (payment_option_available ?payment_option)
        (not
          (plan_payment_attached ?itinerary_plan ?payment_option)
        )
      )
  )
  (:action assign_staff_to_itinerary
    :parameters (?itinerary_plan - itinerary_plan ?staff_assignment - staff_assignment)
    :precondition
      (and
        (fragment_confirmed ?itinerary_plan)
        (staff_assignment_available ?staff_assignment)
      )
    :effect
      (and
        (plan_staff_assigned ?itinerary_plan ?staff_assignment)
        (not
          (staff_assignment_available ?staff_assignment)
        )
      )
  )
  (:action unassign_staff_from_itinerary
    :parameters (?itinerary_plan - itinerary_plan ?staff_assignment - staff_assignment)
    :precondition
      (and
        (plan_staff_assigned ?itinerary_plan ?staff_assignment)
      )
    :effect
      (and
        (staff_assignment_available ?staff_assignment)
        (not
          (plan_staff_assigned ?itinerary_plan ?staff_assignment)
        )
      )
  )
  (:action claim_storage_window
    :parameters (?day_segment - day_segment ?storage_time_window - storage_time_window ?activity - activity)
    :precondition
      (and
        (fragment_confirmed ?day_segment)
        (fragment_linked_activity ?day_segment ?activity)
        (day_has_storage_window ?day_segment ?storage_time_window)
        (not
          (storage_window_claimed ?storage_time_window)
        )
        (not
          (storage_window_held ?storage_time_window)
        )
      )
    :effect (storage_window_claimed ?storage_time_window)
  )
  (:action hold_storage_window_for_day
    :parameters (?day_segment - day_segment ?storage_time_window - storage_time_window ?service_provider - service_provider)
    :precondition
      (and
        (fragment_confirmed ?day_segment)
        (fragment_assigned_provider ?day_segment ?service_provider)
        (day_has_storage_window ?day_segment ?storage_time_window)
        (storage_window_claimed ?storage_time_window)
        (not
          (day_segment_allocated ?day_segment)
        )
      )
    :effect
      (and
        (day_segment_allocated ?day_segment)
        (day_segment_ready_for_reservation ?day_segment)
      )
  )
  (:action allocate_luggage_category_to_day
    :parameters (?day_segment - day_segment ?storage_time_window - storage_time_window ?luggage_category - luggage_category)
    :precondition
      (and
        (fragment_confirmed ?day_segment)
        (day_has_storage_window ?day_segment ?storage_time_window)
        (luggage_category_available ?luggage_category)
        (not
          (day_segment_allocated ?day_segment)
        )
      )
    :effect
      (and
        (storage_window_held ?storage_time_window)
        (day_segment_allocated ?day_segment)
        (day_luggage_category_allocated ?day_segment ?luggage_category)
        (not
          (luggage_category_available ?luggage_category)
        )
      )
  )
  (:action pair_activity_with_storage_for_day
    :parameters (?day_segment - day_segment ?storage_time_window - storage_time_window ?activity - activity ?luggage_category - luggage_category)
    :precondition
      (and
        (fragment_confirmed ?day_segment)
        (fragment_linked_activity ?day_segment ?activity)
        (day_has_storage_window ?day_segment ?storage_time_window)
        (storage_window_held ?storage_time_window)
        (day_luggage_category_allocated ?day_segment ?luggage_category)
        (not
          (day_segment_ready_for_reservation ?day_segment)
        )
      )
    :effect
      (and
        (storage_window_claimed ?storage_time_window)
        (day_segment_ready_for_reservation ?day_segment)
        (luggage_category_available ?luggage_category)
        (not
          (day_luggage_category_allocated ?day_segment ?luggage_category)
        )
      )
  )
  (:action claim_activity_window
    :parameters (?time_segment - time_segment ?activity_time_window - activity_time_window ?activity - activity)
    :precondition
      (and
        (fragment_confirmed ?time_segment)
        (fragment_linked_activity ?time_segment ?activity)
        (time_segment_activity_window_linked ?time_segment ?activity_time_window)
        (not
          (activity_window_claimed ?activity_time_window)
        )
        (not
          (activity_window_held ?activity_time_window)
        )
      )
    :effect (activity_window_claimed ?activity_time_window)
  )
  (:action hold_activity_window_for_time_segment
    :parameters (?time_segment - time_segment ?activity_time_window - activity_time_window ?service_provider - service_provider)
    :precondition
      (and
        (fragment_confirmed ?time_segment)

        (activity_window_claimed ?activity_time_window)

          (time_segment_allocated ?time_segment)
        )
      )
    :effect
      (and
        (time_segment_allocated ?time_segment)
        (time_segment_ready_for_reservation ?time_segment)
      )
  )
  (:action allocate_luggage_category_to_time_segment
    :parameters (?time_segment - time_segment ?activity_time_window - activity_time_window ?luggage_category - luggage_category)
    :precondition
      (and
        (fragment_confirmed ?time_segment)
        (time_segment_activity_window_linked ?time_segment ?activity_time_window)
        (luggage_category_available ?luggage_category)
        (not
          (time_segment_allocated ?time_segment)

      (and
        (activity_window_held ?activity_time_window)

        (not
          (luggage_category_available ?luggage_category)
        )

  (:action pair_activity_with_activity_window_for_time_segment
    :parameters (?time_segment - time_segment ?activity_time_window - activity_time_window ?activity - activity ?luggage_category - luggage_category)
    :precondition
      (and
        (fragment_confirmed ?time_segment)
        (fragment_linked_activity ?time_segment ?activity)
        (time_segment_activity_window_linked ?time_segment ?activity_time_window)
        (activity_window_held ?activity_time_window)
        (time_segment_luggage_category_allocated ?time_segment ?luggage_category)
        (not
          (time_segment_ready_for_reservation ?time_segment)
        )
      )
    :effect
      (and
        (activity_window_claimed ?activity_time_window)
        (time_segment_ready_for_reservation ?time_segment)
        (luggage_category_available ?luggage_category)
        (not
          (time_segment_luggage_category_allocated ?time_segment ?luggage_category)
        )
      )
  )
  (:action assemble_reservation_core
    :parameters (?day_segment - day_segment ?time_segment - time_segment ?storage_time_window - storage_time_window ?activity_time_window - activity_time_window ?storage_reservation_record - storage_reservation_record)
    :precondition
      (and
        (day_segment_allocated ?day_segment)
        (time_segment_allocated ?time_segment)
        (day_has_storage_window ?day_segment ?storage_time_window)
        (time_segment_activity_window_linked ?time_segment ?activity_time_window)
        (storage_window_claimed ?storage_time_window)
        (activity_window_claimed ?activity_time_window)
        (day_segment_ready_for_reservation ?day_segment)
        (time_segment_ready_for_reservation ?time_segment)
        (reservation_record_available ?storage_reservation_record)
      )
    :effect
      (and
        (reservation_record_allocated ?storage_reservation_record)
        (reservation_assigned_storage_window ?storage_reservation_record ?storage_time_window)
        (reservation_assigned_activity_window ?storage_reservation_record ?activity_time_window)
        (not
          (reservation_record_available ?storage_reservation_record)
        )
      )
  )
  (:action assemble_reservation_with_hold
    :parameters (?day_segment - day_segment ?time_segment - time_segment ?storage_time_window - storage_time_window ?activity_time_window - activity_time_window ?storage_reservation_record - storage_reservation_record)
    :precondition
      (and
        (day_segment_allocated ?day_segment)
        (time_segment_allocated ?time_segment)
        (day_has_storage_window ?day_segment ?storage_time_window)
        (time_segment_activity_window_linked ?time_segment ?activity_time_window)
        (storage_window_held ?storage_time_window)
        (activity_window_claimed ?activity_time_window)
        (not
          (day_segment_ready_for_reservation ?day_segment)
        )
        (time_segment_ready_for_reservation ?time_segment)
        (reservation_record_available ?storage_reservation_record)
      )
    :effect
      (and
        (reservation_record_allocated ?storage_reservation_record)
        (reservation_assigned_storage_window ?storage_reservation_record ?storage_time_window)
        (reservation_assigned_activity_window ?storage_reservation_record ?activity_time_window)
        (reservation_option_requires_amenities ?storage_reservation_record)
        (not
          (reservation_record_available ?storage_reservation_record)
        )
      )
  )
  (:action assemble_reservation_with_alternate_hold

    :precondition
      (and
        (day_segment_allocated ?day_segment)
        (time_segment_allocated ?time_segment)
        (day_has_storage_window ?day_segment ?storage_time_window)
        (time_segment_activity_window_linked ?time_segment ?activity_time_window)
        (storage_window_claimed ?storage_time_window)
        (activity_window_held ?activity_time_window)
        (day_segment_ready_for_reservation ?day_segment)
        (not
          (time_segment_ready_for_reservation ?time_segment)
        )
        (reservation_record_available ?storage_reservation_record)
      )
    :effect
      (and
        (reservation_record_allocated ?storage_reservation_record)
        (reservation_assigned_storage_window ?storage_reservation_record ?storage_time_window)
        (reservation_assigned_activity_window ?storage_reservation_record ?activity_time_window)
        (reservation_option_requires_staff_assignment ?storage_reservation_record)
        (not
          (reservation_record_available ?storage_reservation_record)
        )
      )
  )
  (:action assemble_reservation_with_combined_holds
    :parameters (?day_segment - day_segment ?time_segment - time_segment ?storage_time_window - storage_time_window ?activity_time_window - activity_time_window ?storage_reservation_record - storage_reservation_record)
    :precondition
      (and
        (day_segment_allocated ?day_segment)
        (time_segment_allocated ?time_segment)
        (day_has_storage_window ?day_segment ?storage_time_window)
        (time_segment_activity_window_linked ?time_segment ?activity_time_window)
        (storage_window_held ?storage_time_window)
        (activity_window_held ?activity_time_window)
        (not
          (day_segment_ready_for_reservation ?day_segment)
        )
        (not
          (time_segment_ready_for_reservation ?time_segment)
        )
        (reservation_record_available ?storage_reservation_record)
      )
    :effect
      (and
        (reservation_record_allocated ?storage_reservation_record)
        (reservation_assigned_storage_window ?storage_reservation_record ?storage_time_window)
        (reservation_assigned_activity_window ?storage_reservation_record ?activity_time_window)
        (reservation_option_requires_amenities ?storage_reservation_record)
        (reservation_option_requires_staff_assignment ?storage_reservation_record)
        (not
          (reservation_record_available ?storage_reservation_record)
        )
      )
  )

    :parameters (?storage_reservation_record - storage_reservation_record ?day_segment - day_segment ?activity - activity)
    :precondition
      (and
        (reservation_record_allocated ?storage_reservation_record)
        (day_segment_allocated ?day_segment)
        (fragment_linked_activity ?day_segment ?activity)
        (not
          (reservation_ready_for_luggage_allocation ?storage_reservation_record)
        )
      )
    :effect (reservation_ready_for_luggage_allocation ?storage_reservation_record)
  )

    :parameters (?itinerary_plan - itinerary_plan ?luggage_unit - luggage_unit ?storage_reservation_record - storage_reservation_record)
    :precondition
      (and
        (fragment_confirmed ?itinerary_plan)
        (plan_reservation_record_link ?itinerary_plan ?storage_reservation_record)
        (plan_has_luggage_unit ?itinerary_plan ?luggage_unit)
        (luggage_unit_available ?luggage_unit)
        (reservation_record_allocated ?storage_reservation_record)
        (reservation_ready_for_luggage_allocation ?storage_reservation_record)
        (not
          (luggage_unit_allocated ?luggage_unit)
        )
      )
    :effect
      (and
        (luggage_unit_allocated ?luggage_unit)
        (luggage_unit_assigned_to_reservation ?luggage_unit ?storage_reservation_record)
        (not
          (luggage_unit_available ?luggage_unit)
        )
      )
  )
  (:action finalize_luggage_unit_assignment
    :parameters (?itinerary_plan - itinerary_plan ?luggage_unit - luggage_unit ?storage_reservation_record - storage_reservation_record ?activity - activity)
    :precondition
      (and
        (fragment_confirmed ?itinerary_plan)
        (plan_has_luggage_unit ?itinerary_plan ?luggage_unit)
        (luggage_unit_allocated ?luggage_unit)
        (luggage_unit_assigned_to_reservation ?luggage_unit ?storage_reservation_record)
        (fragment_linked_activity ?itinerary_plan ?activity)
        (not
          (reservation_option_requires_amenities ?storage_reservation_record)
        )
        (not
          (plan_units_locked ?itinerary_plan)
        )
      )
    :effect (plan_units_locked ?itinerary_plan)
  )
  (:action attach_amenity
    :parameters (?itinerary_plan - itinerary_plan ?amenity_option - amenity_option)
    :precondition
      (and
        (fragment_confirmed ?itinerary_plan)
        (amenity_available ?amenity_option)
        (not
          (plan_has_amenity ?itinerary_plan)
        )
      )
    :effect
      (and
        (plan_has_amenity ?itinerary_plan)
        (plan_amenity_attached ?itinerary_plan ?amenity_option)
        (not
          (amenity_available ?amenity_option)
        )
      )
  )
  (:action confirm_amenity_and_lock
    :parameters (?itinerary_plan - itinerary_plan ?luggage_unit - luggage_unit ?storage_reservation_record - storage_reservation_record ?activity - activity ?amenity_option - amenity_option)
    :precondition
      (and
        (fragment_confirmed ?itinerary_plan)
        (plan_has_luggage_unit ?itinerary_plan ?luggage_unit)
        (luggage_unit_allocated ?luggage_unit)
        (luggage_unit_assigned_to_reservation ?luggage_unit ?storage_reservation_record)
        (fragment_linked_activity ?itinerary_plan ?activity)
        (reservation_option_requires_amenities ?storage_reservation_record)
        (plan_has_amenity ?itinerary_plan)
        (plan_amenity_attached ?itinerary_plan ?amenity_option)
        (not
          (plan_units_locked ?itinerary_plan)
        )
      )
    :effect
      (and
        (plan_units_locked ?itinerary_plan)
        (plan_amenity_confirmed ?itinerary_plan)
      )
  )
  (:action apply_payment_and_flag_itinerary
    :parameters (?itinerary_plan - itinerary_plan ?payment_option - payment_option ?service_provider - service_provider ?luggage_unit - luggage_unit ?storage_reservation_record - storage_reservation_record)
    :precondition
      (and
        (plan_units_locked ?itinerary_plan)
        (plan_payment_attached ?itinerary_plan ?payment_option)
