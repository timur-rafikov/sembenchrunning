--- domain.pddl (as passed to VAL) ---
(define (domain constraint_satisfaction_validation)
  (:requirements :strips :typing :negative-preconditions)
  (:types domain_category - object component_category - object attribute_category - object problem_element_domain - object variable - problem_element_domain resource_slot - domain_category domain_value - domain_category operator - domain_category attribute_x - domain_category attribute_y - domain_category commit_marker - domain_category feature_token - domain_category variant_token - domain_category transient_token - component_category component_instance - component_category named_qualifier - component_category component_role_a - attribute_category component_role_b - attribute_category composite_candidate - attribute_category variable_partition - variable solver_component_group - variable primary_variable - variable_partition secondary_variable - variable_partition solver_component - solver_component_group)
  (:predicates
    (entity_active ?variable - variable)
    (entity_confirmed ?variable - variable)
    (entity_has_slot ?variable - variable)
    (entity_committed ?variable - variable)
    (candidate_accepted ?variable - variable)
    (entity_ready_for_commit ?variable - variable)
    (resource_slot_available ?resource_slot - resource_slot)
    (entity_bound_to_slot ?variable - variable ?resource_slot - resource_slot)
    (domain_value_available ?domain_value - domain_value)
    (entity_bound_to_value ?variable - variable ?domain_value - domain_value)
    (operator_available ?operator - operator)
    (variable_bound_to_operator ?variable - variable ?operator - operator)
    (transient_token_available ?transient_token - transient_token)
    (primary_bound_transient_token ?primary_variable - primary_variable ?transient_token - transient_token)
    (secondary_bound_transient_token ?secondary_variable - secondary_variable ?transient_token - transient_token)
    (primary_compatible_with_role_a ?primary_variable - primary_variable ?component_role_a - component_role_a)
    (role_a_proposed ?component_role_a - component_role_a)
    (role_a_reserved ?component_role_a - component_role_a)
    (primary_role_confirmed ?primary_variable - primary_variable)
    (secondary_compatible_with_role_b ?secondary_variable - secondary_variable ?component_role_b - component_role_b)
    (role_b_proposed ?component_role_b - component_role_b)
    (role_b_reserved ?component_role_b - component_role_b)
    (secondary_role_confirmed ?secondary_variable - secondary_variable)
    (composite_candidate_available ?composite_candidate - composite_candidate)
    (composite_candidate_constructed ?composite_candidate - composite_candidate)
    (composite_contains_role_a ?composite_candidate - composite_candidate ?component_role_a - component_role_a)
    (composite_contains_role_b ?composite_candidate - composite_candidate ?component_role_b - component_role_b)
    (candidate_role_a_validated ?composite_candidate - composite_candidate)
    (candidate_role_b_validated ?composite_candidate - composite_candidate)
    (candidate_checked ?composite_candidate - composite_candidate)
    (component_associated_primary_variable ?solver_component - solver_component ?primary_variable - primary_variable)
    (component_associated_secondary_variable ?solver_component - solver_component ?secondary_variable - secondary_variable)
    (component_associated_candidate ?solver_component - solver_component ?composite_candidate - composite_candidate)
    (component_instance_available ?component_instance - component_instance)
    (component_has_instance ?solver_component - solver_component ?component_instance - component_instance)
    (component_instance_installed ?component_instance - component_instance)
    (instance_installed_in_candidate ?component_instance - component_instance ?composite_candidate - composite_candidate)
    (component_instance_bound ?solver_component - solver_component)
    (component_ready_for_feature_binding ?solver_component - solver_component)
    (component_evaluated ?solver_component - solver_component)
    (component_has_qualifier ?solver_component - solver_component)
    (component_qualifier_attached ?solver_component - solver_component)
    (component_scored ?solver_component - solver_component)
    (component_approved ?solver_component - solver_component)
    (qualifier_available ?named_qualifier - named_qualifier)
    (component_associated_qualifier ?solver_component - solver_component ?named_qualifier - named_qualifier)
    (component_qualifier_reserved ?solver_component - solver_component)
    (component_qualifier_locked ?solver_component - solver_component)
    (component_qualifier_finalized ?solver_component - solver_component)
    (attribute_x_available ?attribute_x - attribute_x)
    (component_has_attribute_x ?solver_component - solver_component ?attribute_x - attribute_x)
    (attribute_y_available ?attribute_y - attribute_y)
    (component_has_attribute_y ?solver_component - solver_component ?attribute_y - attribute_y)
    (feature_token_available ?feature_token - feature_token)
    (component_has_feature_token ?solver_component - solver_component ?feature_token - feature_token)
    (variant_token_available ?variant_token - variant_token)
    (component_has_variant ?solver_component - solver_component ?variant_token - variant_token)
    (commit_marker_available ?commit_marker - commit_marker)
    (entity_bound_commit_marker ?variable - variable ?commit_marker - commit_marker)
    (primary_ready_for_assembly ?primary_variable - primary_variable)
    (secondary_ready_for_assembly ?secondary_variable - secondary_variable)
    (component_finalized ?solver_component - solver_component)
  )
  (:action activate_variable
    :parameters (?variable - variable)
    :precondition
      (and
        (not
          (entity_active ?variable)
        )
        (not
          (entity_committed ?variable)
        )
      )
    :effect (entity_active ?variable)
  )
  (:action assign_slot_to_variable
    :parameters (?variable - variable ?resource_slot - resource_slot)
    :precondition
      (and
        (entity_active ?variable)
        (not
          (entity_has_slot ?variable)
        )
        (resource_slot_available ?resource_slot)
      )
    :effect
      (and
        (entity_has_slot ?variable)
        (entity_bound_to_slot ?variable ?resource_slot)
        (not
          (resource_slot_available ?resource_slot)
        )
      )
  )
  (:action assign_value_candidate
    :parameters (?variable - variable ?domain_value - domain_value)
    :precondition
      (and
        (entity_active ?variable)
        (entity_has_slot ?variable)
        (domain_value_available ?domain_value)
      )
    :effect
      (and
        (entity_bound_to_value ?variable ?domain_value)
        (not
          (domain_value_available ?domain_value)
        )
      )
  )
  (:action confirm_variable_binding
    :parameters (?variable - variable ?domain_value - domain_value)
    :precondition
      (and
        (entity_active ?variable)
        (entity_has_slot ?variable)
        (entity_bound_to_value ?variable ?domain_value)
        (not
          (entity_confirmed ?variable)
        )
      )
    :effect (entity_confirmed ?variable)
  )
  (:action release_value_candidate
    :parameters (?variable - variable ?domain_value - domain_value)
    :precondition
      (and
        (entity_bound_to_value ?variable ?domain_value)
      )
    :effect
      (and
        (domain_value_available ?domain_value)
        (not
          (entity_bound_to_value ?variable ?domain_value)
        )
      )
  )
  (:action bind_operator_to_variable
    :parameters (?variable - variable ?operator - operator)
    :precondition
      (and
        (entity_confirmed ?variable)
        (operator_available ?operator)
      )
    :effect
      (and
        (variable_bound_to_operator ?variable ?operator)
        (not
          (operator_available ?operator)
        )
      )
  )
  (:action release_operator_from_variable
    :parameters (?variable - variable ?operator - operator)
    :precondition
      (and
        (variable_bound_to_operator ?variable ?operator)
      )
    :effect
      (and
        (operator_available ?operator)
        (not
          (variable_bound_to_operator ?variable ?operator)
        )
      )
  )
  (:action assign_feature_to_component
    :parameters (?solver_component - solver_component ?feature_token - feature_token)
    :precondition
      (and
        (entity_confirmed ?solver_component)
        (feature_token_available ?feature_token)
      )
    :effect
      (and
        (component_has_feature_token ?solver_component ?feature_token)
        (not
          (feature_token_available ?feature_token)
        )
      )
  )
  (:action release_feature_from_component
    :parameters (?solver_component - solver_component ?feature_token - feature_token)
    :precondition
      (and
        (component_has_feature_token ?solver_component ?feature_token)
      )
    :effect
      (and
        (feature_token_available ?feature_token)
        (not
          (component_has_feature_token ?solver_component ?feature_token)
        )
      )
  )
  (:action assign_variant_to_component
    :parameters (?solver_component - solver_component ?variant_token - variant_token)
    :precondition
      (and
        (entity_confirmed ?solver_component)
        (variant_token_available ?variant_token)
      )
    :effect
      (and
        (component_has_variant ?solver_component ?variant_token)
        (not
          (variant_token_available ?variant_token)
        )
      )
  )
  (:action release_variant_from_component
    :parameters (?solver_component - solver_component ?variant_token - variant_token)
    :precondition
      (and
        (component_has_variant ?solver_component ?variant_token)
      )
    :effect
      (and
        (variant_token_available ?variant_token)
        (not
          (component_has_variant ?solver_component ?variant_token)
        )
      )
  )
  (:action propose_role_a
    :parameters (?primary_variable - primary_variable ?component_role_a - component_role_a ?domain_value - domain_value)
    :precondition
      (and
        (entity_confirmed ?primary_variable)
        (entity_bound_to_value ?primary_variable ?domain_value)
        (primary_compatible_with_role_a ?primary_variable ?component_role_a)
        (not
          (role_a_proposed ?component_role_a)
        )
        (not
          (role_a_reserved ?component_role_a)
        )
      )
    :effect (role_a_proposed ?component_role_a)
  )
  (:action confirm_role_a_with_operator
    :parameters (?primary_variable - primary_variable ?component_role_a - component_role_a ?operator - operator)
    :precondition
      (and
        (entity_confirmed ?primary_variable)
        (variable_bound_to_operator ?primary_variable ?operator)
        (primary_compatible_with_role_a ?primary_variable ?component_role_a)
        (role_a_proposed ?component_role_a)
        (not
          (primary_ready_for_assembly ?primary_variable)
        )
      )
    :effect
      (and
        (primary_ready_for_assembly ?primary_variable)
        (primary_role_confirmed ?primary_variable)
      )
  )
  (:action reserve_role_a_with_token
    :parameters (?primary_variable - primary_variable ?component_role_a - component_role_a ?transient_token - transient_token)
    :precondition
      (and
        (entity_confirmed ?primary_variable)
        (primary_compatible_with_role_a ?primary_variable ?component_role_a)
        (transient_token_available ?transient_token)
        (not
          (primary_ready_for_assembly ?primary_variable)
        )
      )
    :effect
      (and
        (role_a_reserved ?component_role_a)
        (primary_ready_for_assembly ?primary_variable)
        (primary_bound_transient_token ?primary_variable ?transient_token)
        (not
          (transient_token_available ?transient_token)
        )
      )
  )
  (:action finalize_role_a_token_assignment
    :parameters (?primary_variable - primary_variable ?component_role_a - component_role_a ?domain_value - domain_value ?transient_token - transient_token)
    :precondition
      (and
        (entity_confirmed ?primary_variable)
        (entity_bound_to_value ?primary_variable ?domain_value)
        (primary_compatible_with_role_a ?primary_variable ?component_role_a)
        (role_a_reserved ?component_role_a)
        (primary_bound_transient_token ?primary_variable ?transient_token)
        (not
          (primary_role_confirmed ?primary_variable)
        )
      )
    :effect
      (and
        (role_a_proposed ?component_role_a)
        (primary_role_confirmed ?primary_variable)
        (transient_token_available ?transient_token)
        (not
          (primary_bound_transient_token ?primary_variable ?transient_token)
        )
      )
  )
  (:action propose_role_b
    :parameters (?secondary_variable - secondary_variable ?component_role_b - component_role_b ?domain_value - domain_value)
    :precondition
      (and
        (entity_confirmed ?secondary_variable)
        (entity_bound_to_value ?secondary_variable ?domain_value)
        (secondary_compatible_with_role_b ?secondary_variable ?component_role_b)
        (not
          (role_b_proposed ?component_role_b)
        )
        (not
          (role_b_reserved ?component_role_b)
        )
      )
    :effect (role_b_proposed ?component_role_b)
  )
  (:action confirm_role_b_with_operator
    :parameters (?secondary_variable - secondary_variable ?component_role_b - component_role_b ?operator - operator)
    :precondition
      (and
        (entity_confirmed ?secondary_variable)
        (variable_bound_to_operator ?secondary_variable ?operator)
        (secondary_compatible_with_role_b ?secondary_variable ?component_role_b)
        (role_b_proposed ?component_role_b)
        (not
          (secondary_ready_for_assembly ?secondary_variable)
        )
      )
    :effect
      (and
        (secondary_ready_for_assembly ?secondary_variable)
        (secondary_role_confirmed ?secondary_variable)
      )
  )
  (:action reserve_role_b_with_token
    :parameters (?secondary_variable - secondary_variable ?component_role_b - component_role_b ?transient_token - transient_token)
    :precondition
      (and
        (entity_confirmed ?secondary_variable)
        (secondary_compatible_with_role_b ?secondary_variable ?component_role_b)
        (transient_token_available ?transient_token)
        (not
          (secondary_ready_for_assembly ?secondary_variable)
        )
      )
    :effect
      (and
        (role_b_reserved ?component_role_b)
        (secondary_ready_for_assembly ?secondary_variable)
        (secondary_bound_transient_token ?secondary_variable ?transient_token)
        (not
          (transient_token_available ?transient_token)
        )
      )
  )
  (:action finalize_role_b_token_assignment
    :parameters (?secondary_variable - secondary_variable ?component_role_b - component_role_b ?domain_value - domain_value ?transient_token - transient_token)
    :precondition
      (and
        (entity_confirmed ?secondary_variable)
        (entity_bound_to_value ?secondary_variable ?domain_value)
        (secondary_compatible_with_role_b ?secondary_variable ?component_role_b)
        (role_b_reserved ?component_role_b)
        (secondary_bound_transient_token ?secondary_variable ?transient_token)
        (not
          (secondary_role_confirmed ?secondary_variable)
        )
      )
    :effect
      (and
        (role_b_proposed ?component_role_b)
        (secondary_role_confirmed ?secondary_variable)
        (transient_token_available ?transient_token)
        (not
          (secondary_bound_transient_token ?secondary_variable ?transient_token)
        )
      )
  )
  (:action assemble_composite_candidate
    :parameters (?primary_variable - primary_variable ?secondary_variable - secondary_variable ?component_role_a - component_role_a ?component_role_b - component_role_b ?composite_candidate - composite_candidate)
    :precondition
      (and
        (primary_ready_for_assembly ?primary_variable)
        (secondary_ready_for_assembly ?secondary_variable)
        (primary_compatible_with_role_a ?primary_variable ?component_role_a)
        (secondary_compatible_with_role_b ?secondary_variable ?component_role_b)
        (role_a_proposed ?component_role_a)
        (role_b_proposed ?component_role_b)
        (primary_role_confirmed ?primary_variable)
        (secondary_role_confirmed ?secondary_variable)
        (composite_candidate_available ?composite_candidate)
      )
    :effect
      (and
        (composite_candidate_constructed ?composite_candidate)
        (composite_contains_role_a ?composite_candidate ?component_role_a)
        (composite_contains_role_b ?composite_candidate ?component_role_b)
        (not
          (composite_candidate_available ?composite_candidate)
        )
      )
  )
  (:action assemble_composite_candidate_with_role_a_token
    :parameters (?primary_variable - primary_variable ?secondary_variable - secondary_variable ?component_role_a - component_role_a ?component_role_b - component_role_b ?composite_candidate - composite_candidate)
    :precondition
      (and
        (primary_ready_for_assembly ?primary_variable)
        (secondary_ready_for_assembly ?secondary_variable)
        (primary_compatible_with_role_a ?primary_variable ?component_role_a)
        (secondary_compatible_with_role_b ?secondary_variable ?component_role_b)
        (role_a_reserved ?component_role_a)
        (role_b_proposed ?component_role_b)
        (not
          (primary_role_confirmed ?primary_variable)
        )
        (secondary_role_confirmed ?secondary_variable)
        (composite_candidate_available ?composite_candidate)
      )
    :effect
      (and
        (composite_candidate_constructed ?composite_candidate)
        (composite_contains_role_a ?composite_candidate ?component_role_a)
        (composite_contains_role_b ?composite_candidate ?component_role_b)
        (candidate_role_a_validated ?composite_candidate)
        (not
          (composite_candidate_available ?composite_candidate)
        )
      )
  )
  (:action assemble_composite_candidate_with_role_b_token
    :parameters (?primary_variable - primary_variable ?secondary_variable - secondary_variable ?component_role_a - component_role_a ?component_role_b - component_role_b ?composite_candidate - composite_candidate)
    :precondition
      (and
        (primary_ready_for_assembly ?primary_variable)
        (secondary_ready_for_assembly ?secondary_variable)
        (primary_compatible_with_role_a ?primary_variable ?component_role_a)
        (secondary_compatible_with_role_b ?secondary_variable ?component_role_b)
        (role_a_proposed ?component_role_a)
        (role_b_reserved ?component_role_b)
        (primary_role_confirmed ?primary_variable)
        (not
          (secondary_role_confirmed ?secondary_variable)
        )
        (composite_candidate_available ?composite_candidate)
      )
    :effect
      (and
        (composite_candidate_constructed ?composite_candidate)
        (composite_contains_role_a ?composite_candidate ?component_role_a)
        (composite_contains_role_b ?composite_candidate ?component_role_b)
        (candidate_role_b_validated ?composite_candidate)
        (not
          (composite_candidate_available ?composite_candidate)
        )
      )
  )
  (:action assemble_composite_candidate_with_both_tokens
    :parameters (?primary_variable - primary_variable ?secondary_variable - secondary_variable ?component_role_a - component_role_a ?component_role_b - component_role_b ?composite_candidate - composite_candidate)
    :precondition
      (and
        (primary_ready_for_assembly ?primary_variable)
        (secondary_ready_for_assembly ?secondary_variable)
        (primary_compatible_with_role_a ?primary_variable ?component_role_a)
        (secondary_compatible_with_role_b ?secondary_variable ?component_role_b)
        (role_a_reserved ?component_role_a)
        (role_b_reserved ?component_role_b)
        (not
          (primary_role_confirmed ?primary_variable)
        )
        (not
          (secondary_role_confirmed ?secondary_variable)
        )
        (composite_candidate_available ?composite_candidate)
      )
    :effect
      (and
        (composite_candidate_constructed ?composite_candidate)
        (composite_contains_role_a ?composite_candidate ?component_role_a)
        (composite_contains_role_b ?composite_candidate ?component_role_b)
        (candidate_role_a_validated ?composite_candidate)
        (candidate_role_b_validated ?composite_candidate)
        (not
          (composite_candidate_available ?composite_candidate)
        )
      )
  )
  (:action validate_composite_candidate
    :parameters (?composite_candidate - composite_candidate ?primary_variable - primary_variable ?domain_value - domain_value)
    :precondition
      (and
        (composite_candidate_constructed ?composite_candidate)
        (primary_ready_for_assembly ?primary_variable)
        (entity_bound_to_value ?primary_variable ?domain_value)
        (not
          (candidate_checked ?composite_candidate)
        )
      )
    :effect (candidate_checked ?composite_candidate)
  )
  (:action install_component_instance
    :parameters (?solver_component - solver_component ?component_instance - component_instance ?composite_candidate - composite_candidate)
    :precondition
      (and
        (entity_confirmed ?solver_component)
        (component_associated_candidate ?solver_component ?composite_candidate)
        (component_has_instance ?solver_component ?component_instance)
        (component_instance_available ?component_instance)
        (composite_candidate_constructed ?composite_candidate)
        (candidate_checked ?composite_candidate)
        (not
          (component_instance_installed ?component_instance)
        )
      )
    :effect
      (and
        (component_instance_installed ?component_instance)
        (instance_installed_in_candidate ?component_instance ?composite_candidate)
        (not
          (component_instance_available ?component_instance)
        )
      )
  )
  (:action bind_component_instance_phase1
    :parameters (?solver_component - solver_component ?component_instance - component_instance ?composite_candidate - composite_candidate ?domain_value - domain_value)
    :precondition
      (and
        (entity_confirmed ?solver_component)
        (component_has_instance ?solver_component ?component_instance)
        (component_instance_installed ?component_instance)
        (instance_installed_in_candidate ?component_instance ?composite_candidate)
        (entity_bound_to_value ?solver_component ?domain_value)
        (not
          (candidate_role_a_validated ?composite_candidate)
        )
        (not
          (component_instance_bound ?solver_component)
        )
      )
    :effect (component_instance_bound ?solver_component)
  )
  (:action attach_attribute_x
    :parameters (?solver_component - solver_component ?attribute_x - attribute_x)
    :precondition
      (and
        (entity_confirmed ?solver_component)
        (attribute_x_available ?attribute_x)
        (not
          (component_has_qualifier ?solver_component)
        )
      )
    :effect
      (and
        (component_has_qualifier ?solver_component)
        (component_has_attribute_x ?solver_component ?attribute_x)
        (not
          (attribute_x_available ?attribute_x)
        )
      )
  )
  (:action bind_component_and_attach_attribute
    :parameters (?solver_component - solver_component ?component_instance - component_instance ?composite_candidate - composite_candidate ?domain_value - domain_value ?attribute_x - attribute_x)
    :precondition
      (and
        (entity_confirmed ?solver_component)
        (component_has_instance ?solver_component ?component_instance)
        (component_instance_installed ?component_instance)
        (instance_installed_in_candidate ?component_instance ?composite_candidate)
        (entity_bound_to_value ?solver_component ?domain_value)
        (candidate_role_a_validated ?composite_candidate)
        (component_has_qualifier ?solver_component)
        (component_has_attribute_x ?solver_component ?attribute_x)
        (not
          (component_instance_bound ?solver_component)
        )
      )
    :effect
      (and
        (component_instance_bound ?solver_component)
        (component_qualifier_attached ?solver_component)
      )
  )
  (:action apply_feature_token_to_component_variant1
    :parameters (?solver_component - solver_component ?feature_token - feature_token ?operator - operator ?component_instance - component_instance ?composite_candidate - composite_candidate)
    :precondition
      (and
        (component_instance_bound ?solver_component)
        (component_has_feature_token ?solver_component ?feature_token)
        (variable_bound_to_operator ?solver_component ?operator)
        (component_has_instance ?solver_component ?component_instance)
        (instance_installed_in_candidate ?component_instance ?composite_candidate)
        (not
          (candidate_role_b_validated ?composite_candidate)
        )
        (not
          (component_ready_for_feature_binding ?solver_component)
        )
      )
    :effect (component_ready_for_feature_binding ?solver_component)
  )
  (:action apply_feature_token_to_component_variant2
    :parameters (?solver_component - solver_component ?feature_token - feature_token ?operator - operator ?component_instance - component_instance ?composite_candidate - composite_candidate)
    :precondition
      (and
        (component_instance_bound ?solver_component)
        (component_has_feature_token ?solver_component ?feature_token)
        (variable_bound_to_operator ?solver_component ?operator)
        (component_has_instance ?solver_component ?component_instance)
        (instance_installed_in_candidate ?component_instance ?composite_candidate)
        (candidate_role_b_validated ?composite_candidate)
        (not
          (component_ready_for_feature_binding ?solver_component)
        )
      )
    :effect (component_ready_for_feature_binding ?solver_component)
  )
  (:action apply_variant_to_component_option1
    :parameters (?solver_component - solver_component ?variant_token - variant_token ?component_instance - component_instance ?composite_candidate - composite_candidate)
    :precondition
      (and
        (component_ready_for_feature_binding ?solver_component)
