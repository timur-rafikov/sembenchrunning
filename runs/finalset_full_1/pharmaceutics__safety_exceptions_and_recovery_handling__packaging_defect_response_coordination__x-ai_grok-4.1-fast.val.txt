--- domain.pddl (as passed to VAL) ---
(define (domain pharm_packaging_defect_response_coordination)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity - object resource_class - entity role_class - entity organization_role - entity incident_category - entity incident_case - incident_category response_team - resource_class investigator - resource_class quality_representative - resource_class expertise_area - resource_class regulatory_authority - resource_class notification_channel - resource_class mitigation_option - resource_class external_expert - resource_class logistics_asset - role_class sample - role_class stakeholder - role_class defect_mode - organization_role supply_node - organization_role recall_action - organization_role case_subgroup_a - incident_case case_subgroup_b - incident_case manufacturing_site - case_subgroup_a distribution_site - case_subgroup_a product_line - case_subgroup_b)

  (:predicates
    (entity_reported ?incident_case - incident_case)
    (entity_triaged ?incident_case - incident_case)
    (entity_has_assigned_team ?incident_case - incident_case)
    (entity_remediation_executed ?incident_case - incident_case)
    (entity_remediation_authorized ?incident_case - incident_case)
    (entity_remediation_committed ?incident_case - incident_case)
    (response_team_available ?response_team - response_team)
    (entity_assigned_response_team ?incident_case - incident_case ?response_team - response_team)
    (investigator_available ?investigator - investigator)
    (entity_assigned_investigator ?incident_case - incident_case ?investigator - investigator)
    (quality_representative_available ?quality_representative - quality_representative)
    (entity_assigned_quality_representative ?incident_case - incident_case ?quality_representative - quality_representative)
    (logistics_asset_available ?logistics_asset - logistics_asset)
    (site_assigned_logistics_asset ?manufacturing_site - manufacturing_site ?logistics_asset - logistics_asset)
    (distribution_assigned_logistics_asset ?distribution_site - distribution_site ?logistics_asset - logistics_asset)
    (site_has_defect_mode ?manufacturing_site - manufacturing_site ?defect_mode - defect_mode)
    (defect_mode_flagged ?defect_mode - defect_mode)
    (defect_mode_sample_collected ?defect_mode - defect_mode)
    (site_assessment_complete ?manufacturing_site - manufacturing_site)
    (distribution_has_supply_node ?distribution_site - distribution_site ?supply_node - supply_node)
    (supply_node_flagged ?supply_node - supply_node)
    (supply_node_evidence_collected ?supply_node - supply_node)
    (distribution_assessment_complete ?distribution_site - distribution_site)
    (recall_template_available ?recall_action - recall_action)
    (recall_initiated ?recall_action - recall_action)
    (recall_targets_defect_mode ?recall_action - recall_action ?defect_mode - defect_mode)
    (recall_targets_supply_node ?recall_action - recall_action ?supply_node - supply_node)
    (recall_needs_evidence_review ?recall_action - recall_action)
    (recall_needs_expert_review ?recall_action - recall_action)
    (recall_execution_recorded ?recall_action - recall_action)
    (product_line_manufactured_at ?product_line - product_line ?manufacturing_site - manufacturing_site)
    (product_line_distributed_to ?product_line - product_line ?distribution_site - distribution_site)
    (product_line_linked_to_recall ?product_line - product_line ?recall_action - recall_action)
    (sample_available ?sample - sample)
    (product_line_has_sample ?product_line - product_line ?sample - sample)
    (sample_in_custody ?sample - sample)
    (sample_linked_to_recall ?sample - sample ?recall_action - recall_action)
    (product_line_evidence_reviewed ?product_line - product_line)
    (product_line_testing_started ?product_line - product_line)
    (product_line_expert_review_completed ?product_line - product_line)
    (product_line_expertise_assigned ?product_line - product_line)
    (product_line_expert_review_progressed ?product_line - product_line)
    (product_line_documentation_ready ?product_line - product_line)
    (product_line_ready_for_finalization ?product_line - product_line)
    (stakeholder_available ?stakeholder - stakeholder)
    (product_line_linked_stakeholder ?product_line - product_line ?stakeholder - stakeholder)
    (stakeholder_engaged ?product_line - product_line)
    (quality_approval_requested ?product_line - product_line)
    (quality_approval_granted ?product_line - product_line)
    (expertise_area_available ?expertise_area - expertise_area)
    (product_line_linked_expertise_area ?product_line - product_line ?expertise_area - expertise_area)
    (regulatory_authority_available ?regulatory_authority - regulatory_authority)
    (product_line_linked_regulatory_authority ?product_line - product_line ?regulatory_authority - regulatory_authority)
    (mitigation_option_available ?mitigation_option - mitigation_option)
    (product_line_has_mitigation_option ?product_line - product_line ?mitigation_option - mitigation_option)
    (external_expert_available ?external_expert - external_expert)
    (product_line_linked_external_expert ?product_line - product_line ?external_expert - external_expert)
    (notification_channel_available ?notification_channel - notification_channel)
    (entity_linked_notification_channel ?incident_case - incident_case ?notification_channel - notification_channel)
    (manufacturing_site_quarantined ?manufacturing_site - manufacturing_site)
    (distribution_site_quarantined ?distribution_site - distribution_site)
    (product_line_finalized ?product_line - product_line)
  )
  (:action record_incident_case
    :parameters (?incident_case - incident_case)
    :precondition
      (and
        (not
          (entity_reported ?incident_case)
        )
        (not
          (entity_remediation_executed ?incident_case)
        )
      )
    :effect (entity_reported ?incident_case)
  )
  (:action assign_response_team_to_case
    :parameters (?incident_case - incident_case ?response_team - response_team)
    :precondition
      (and
        (entity_reported ?incident_case)
        (not
          (entity_has_assigned_team ?incident_case)
        )
        (response_team_available ?response_team)
      )
    :effect
      (and
        (entity_has_assigned_team ?incident_case)
        (entity_assigned_response_team ?incident_case ?response_team)
        (not
          (response_team_available ?response_team)
        )
      )
  )
  (:action assign_investigator_to_case
    :parameters (?incident_case - incident_case ?investigator - investigator)
    :precondition
      (and
        (entity_reported ?incident_case)
        (entity_has_assigned_team ?incident_case)
        (investigator_available ?investigator)
      )
    :effect
      (and
        (entity_assigned_investigator ?incident_case ?investigator)
        (not
          (investigator_available ?investigator)
        )
      )
  )
  (:action confirm_case_triage
    :parameters (?incident_case - incident_case ?investigator - investigator)
    :precondition
      (and
        (entity_reported ?incident_case)
        (entity_has_assigned_team ?incident_case)
        (entity_assigned_investigator ?incident_case ?investigator)
        (not
          (entity_triaged ?incident_case)
        )
      )
    :effect (entity_triaged ?incident_case)
  )
  (:action release_investigator_from_case
    :parameters (?incident_case - incident_case ?investigator - investigator)
    :precondition
      (and
        (entity_assigned_investigator ?incident_case ?investigator)
      )
    :effect
      (and
        (investigator_available ?investigator)
        (not
          (entity_assigned_investigator ?incident_case ?investigator)
        )
      )
  )
  (:action assign_quality_representative_to_case
    :parameters (?incident_case - incident_case ?quality_representative - quality_representative)
    :precondition
      (and
        (entity_triaged ?incident_case)
        (quality_representative_available ?quality_representative)
      )
    :effect
      (and
        (entity_assigned_quality_representative ?incident_case ?quality_representative)
        (not
          (quality_representative_available ?quality_representative)
        )
      )
  )
  (:action release_quality_representative_from_case
    :parameters (?incident_case - incident_case ?quality_representative - quality_representative)
    :precondition
      (and
        (entity_assigned_quality_representative ?incident_case ?quality_representative)
      )
    :effect
      (and
        (quality_representative_available ?quality_representative)
        (not
          (entity_assigned_quality_representative ?incident_case ?quality_representative)
        )
      )
  )
  (:action allocate_mitigation_option_to_product_line
    :parameters (?product_line - product_line ?mitigation_option - mitigation_option)
    :precondition
      (and
        (entity_triaged ?product_line)
        (mitigation_option_available ?mitigation_option)
      )
    :effect
      (and
        (product_line_has_mitigation_option ?product_line ?mitigation_option)
        (not
          (mitigation_option_available ?mitigation_option)
        )
      )
  )
  (:action deallocate_mitigation_option_from_product_line
    :parameters (?product_line - product_line ?mitigation_option - mitigation_option)
    :precondition
      (and
        (product_line_has_mitigation_option ?product_line ?mitigation_option)
      )
    :effect
      (and
        (mitigation_option_available ?mitigation_option)
        (not
          (product_line_has_mitigation_option ?product_line ?mitigation_option)
        )
      )
  )
  (:action assign_external_expert_to_product_line
    :parameters (?product_line - product_line ?external_expert - external_expert)
    :precondition
      (and
        (entity_triaged ?product_line)
        (external_expert_available ?external_expert)
      )
    :effect
      (and
        (product_line_linked_external_expert ?product_line ?external_expert)
        (not
          (external_expert_available ?external_expert)
        )
      )
  )
  (:action release_external_expert_from_product_line
    :parameters (?product_line - product_line ?external_expert - external_expert)
    :precondition
      (and
        (product_line_linked_external_expert ?product_line ?external_expert)
      )
    :effect
      (and
        (external_expert_available ?external_expert)
        (not
          (product_line_linked_external_expert ?product_line ?external_expert)
        )
      )
  )
  (:action evaluate_defect_mode_at_manufacturing_site
    :parameters (?manufacturing_site - manufacturing_site ?defect_mode - defect_mode ?investigator - investigator)
    :precondition
      (and
        (entity_triaged ?manufacturing_site)
        (entity_assigned_investigator ?manufacturing_site ?investigator)
        (site_has_defect_mode ?manufacturing_site ?defect_mode)
        (not
          (defect_mode_flagged ?defect_mode)
        )
        (not
          (defect_mode_sample_collected ?defect_mode)
        )
      )
    :effect (defect_mode_flagged ?defect_mode)
  )
  (:action initiate_site_containment_with_quality
    :parameters (?manufacturing_site - manufacturing_site ?defect_mode - defect_mode ?quality_representative - quality_representative)
    :precondition
      (and
        (entity_triaged ?manufacturing_site)
        (entity_assigned_quality_representative ?manufacturing_site ?quality_representative)
        (site_has_defect_mode ?manufacturing_site ?defect_mode)
        (defect_mode_flagged ?defect_mode)
        (not
          (manufacturing_site_quarantined ?manufacturing_site)
        )
      )
    :effect
      (and
        (manufacturing_site_quarantined ?manufacturing_site)
        (site_assessment_complete ?manufacturing_site)
      )
  )
  (:action allocate_logistics_asset_for_site_containment
    :parameters (?manufacturing_site - manufacturing_site ?defect_mode - defect_mode ?logistics_asset - logistics_asset)
    :precondition
      (and
        (entity_triaged ?manufacturing_site)
        (site_has_defect_mode ?manufacturing_site ?defect_mode)
        (logistics_asset_available ?logistics_asset)
        (not
          (manufacturing_site_quarantined ?manufacturing_site)
        )
      )
    :effect
      (and
        (defect_mode_sample_collected ?defect_mode)
        (manufacturing_site_quarantined ?manufacturing_site)
        (site_assigned_logistics_asset ?manufacturing_site ?logistics_asset)
        (not
          (logistics_asset_available ?logistics_asset)
        )
      )
  )
  (:action confirm_site_sample_and_assessment
    :parameters (?manufacturing_site - manufacturing_site ?defect_mode - defect_mode ?investigator - investigator ?logistics_asset - logistics_asset)
    :precondition
      (and
        (entity_triaged ?manufacturing_site)
        (entity_assigned_investigator ?manufacturing_site ?investigator)
        (site_has_defect_mode ?manufacturing_site ?defect_mode)
        (defect_mode_sample_collected ?defect_mode)
        (site_assigned_logistics_asset ?manufacturing_site ?logistics_asset)
        (not
          (site_assessment_complete ?manufacturing_site)
        )
      )
    :effect
      (and
        (defect_mode_flagged ?defect_mode)
        (site_assessment_complete ?manufacturing_site)
        (logistics_asset_available ?logistics_asset)
        (not
          (site_assigned_logistics_asset ?manufacturing_site ?logistics_asset)
        )
      )
  )
  (:action evaluate_defect_mode_at_distribution_site
    :parameters (?distribution_site - distribution_site ?supply_node - supply_node ?investigator - investigator)
    :precondition
      (and
        (entity_triaged ?distribution_site)
        (entity_assigned_investigator ?distribution_site ?investigator)
        (distribution_has_supply_node ?distribution_site ?supply_node)
        (not
          (supply_node_flagged ?supply_node)
        )
        (not
          (supply_node_evidence_collected ?supply_node)
        )
      )
    :effect (supply_node_flagged ?supply_node)
  )
  (:action initiate_distribution_containment_with_quality
    :parameters (?distribution_site - distribution_site ?supply_node - supply_node ?quality_representative - quality_representative)
    :precondition
      (and
        (entity_triaged ?distribution_site)
        (entity_assigned_quality_representative ?distribution_site ?quality_representative)
        (distribution_has_supply_node ?distribution_site ?supply_node)
        (supply_node_flagged ?supply_node)
        (not
          (distribution_site_quarantined ?distribution_site)
        )
      )
    :effect
      (and
        (distribution_site_quarantined ?distribution_site)
        (distribution_assessment_complete ?distribution_site)
      )
  )
  (:action allocate_logistics_asset_for_distribution_containment
    :parameters (?distribution_site - distribution_site ?supply_node - supply_node ?logistics_asset - logistics_asset)
    :precondition
      (and
        (entity_triaged ?distribution_site)
        (distribution_has_supply_node ?distribution_site ?supply_node)
        (logistics_asset_available ?logistics_asset)
        (not
          (distribution_site_quarantined ?distribution_site)
        )
      )
    :effect
      (and
        (supply_node_evidence_collected ?supply_node)
        (distribution_site_quarantined ?distribution_site)
        (distribution_assigned_logistics_asset ?distribution_site ?logistics_asset)
        (not
          (logistics_asset_available ?logistics_asset)
        )
      )
  )
  (:action confirm_distribution_sample_and_assessment
    :parameters (?distribution_site - distribution_site ?supply_node - supply_node ?investigator - investigator ?logistics_asset - logistics_asset)
    :precondition
      (and
        (entity_triaged ?distribution_site)
        (entity_assigned_investigator ?distribution_site ?investigator)
        (distribution_has_supply_node ?distribution_site ?supply_node)
        (supply_node_evidence_collected ?supply_node)
        (distribution_assigned_logistics_asset ?distribution_site ?logistics_asset)
        (not
          (distribution_assessment_complete ?distribution_site)
        )
      )
    :effect
      (and
        (supply_node_flagged ?supply_node)
        (distribution_assessment_complete ?distribution_site)
        (logistics_asset_available ?logistics_asset)
        (not
          (distribution_assigned_logistics_asset ?distribution_site ?logistics_asset)
        )
      )
  )
  (:action initiate_recall_action
    :parameters (?manufacturing_site - manufacturing_site ?distribution_site - distribution_site ?defect_mode - defect_mode ?supply_node - supply_node ?recall_action - recall_action)
    :precondition
      (and
        (manufacturing_site_quarantined ?manufacturing_site)
        (distribution_site_quarantined ?distribution_site)
        (site_has_defect_mode ?manufacturing_site ?defect_mode)
        (distribution_has_supply_node ?distribution_site ?supply_node)
        (defect_mode_flagged ?defect_mode)
        (supply_node_flagged ?supply_node)
        (site_assessment_complete ?manufacturing_site)
        (distribution_assessment_complete ?distribution_site)
        (recall_template_available ?recall_action)
      )
    :effect
      (and
        (recall_initiated ?recall_action)
        (recall_targets_defect_mode ?recall_action ?defect_mode)
        (recall_targets_supply_node ?recall_action ?supply_node)
        (not
          (recall_template_available ?recall_action)
        )
      )
  )
  (:action escalate_recall_stage_1
    :parameters (?manufacturing_site - manufacturing_site ?distribution_site - distribution_site ?defect_mode - defect_mode ?supply_node - supply_node ?recall_action - recall_action)
    :precondition
      (and
        (manufacturing_site_quarantined ?manufacturing_site)
        (distribution_site_quarantined ?distribution_site)
        (site_has_defect_mode ?manufacturing_site ?defect_mode)
        (distribution_has_supply_node ?distribution_site ?supply_node)
        (defect_mode_sample_collected ?defect_mode)
        (supply_node_flagged ?supply_node)
        (not
          (site_assessment_complete ?manufacturing_site)
        )
        (distribution_assessment_complete ?distribution_site)
        (recall_template_available ?recall_action)
      )
    :effect
      (and
        (recall_initiated ?recall_action)
        (recall_targets_defect_mode ?recall_action ?defect_mode)
        (recall_targets_supply_node ?recall_action ?supply_node)
        (recall_needs_evidence_review ?recall_action)
        (not
          (recall_template_available ?recall_action)
        )
      )
  )
  (:action escalate_recall_stage_2
    :parameters (?manufacturing_site - manufacturing_site ?distribution_site - distribution_site ?defect_mode - defect_mode ?supply_node - supply_node ?recall_action - recall_action)
    :precondition
      (and
        (manufacturing_site_quarantined ?manufacturing_site)
        (distribution_site_quarantined ?distribution_site)
        (site_has_defect_mode ?manufacturing_site ?defect_mode)
        (distribution_has_supply_node ?distribution_site ?supply_node)
        (defect_mode_flagged ?defect_mode)
        (supply_node_evidence_collected ?supply_node)
        (site_assessment_complete ?manufacturing_site)
        (not
          (distribution_assessment_complete ?distribution_site)
        )
        (recall_template_available ?recall_action)
      )
    :effect
      (and
        (recall_initiated ?recall_action)
        (recall_targets_defect_mode ?recall_action ?defect_mode)
        (recall_targets_supply_node ?recall_action ?supply_node)
        (recall_needs_expert_review ?recall_action)
        (not
          (recall_template_available ?recall_action)
        )
      )
  )
  (:action escalate_recall_stage_3
    :parameters (?manufacturing_site - manufacturing_site ?distribution_site - distribution_site ?defect_mode - defect_mode ?supply_node - supply_node ?recall_action - recall_action)
    :precondition
      (and
        (manufacturing_site_quarantined ?manufacturing_site)
        (distribution_site_quarantined ?distribution_site)
        (site_has_defect_mode ?manufacturing_site ?defect_mode)
        (distribution_has_supply_node ?distribution_site ?supply_node)
        (defect_mode_sample_collected ?defect_mode)
        (supply_node_evidence_collected ?supply_node)
        (not
          (site_assessment_complete ?manufacturing_site)
        )
        (not
          (distribution_assessment_complete ?distribution_site)
        )
        (recall_template_available ?recall_action)
      )
    :effect
      (and
        (recall_initiated ?recall_action)
        (recall_targets_defect_mode ?recall_action ?defect_mode)
        (recall_targets_supply_node ?recall_action ?supply_node)
        (recall_needs_evidence_review ?recall_action)
        (recall_needs_expert_review ?recall_action)
        (not
          (recall_template_available ?recall_action)
        )
      )
  )
  (:action authorize_recall_execution
    :parameters (?recall_action - recall_action ?manufacturing_site - manufacturing_site ?investigator - investigator)
    :precondition
      (and
        (recall_initiated ?recall_action)
        (manufacturing_site_quarantined ?manufacturing_site)
        (entity_assigned_investigator ?manufacturing_site ?investigator)
        (not
          (recall_execution_recorded ?recall_action)
        )
      )
    :effect (recall_execution_recorded ?recall_action)
  )
  (:action submit_sample_to_lab
    :parameters (?product_line - product_line ?sample - sample ?recall_action - recall_action)
    :precondition
      (and
        (entity_triaged ?product_line)
        (product_line_linked_to_recall ?product_line ?recall_action)
        (product_line_has_sample ?product_line ?sample)
        (sample_available ?sample)
        (recall_initiated ?recall_action)
        (recall_execution_recorded ?recall_action)
        (not
          (sample_in_custody ?sample)
        )
      )
    :effect
      (and
        (sample_in_custody ?sample)
        (sample_linked_to_recall ?sample ?recall_action)
        (not
          (sample_available ?sample)
        )
      )
  )
  (:action confirm_lab_receipt_and_schedule_testing
    :parameters (?product_line - product_line ?sample - sample ?recall_action - recall_action ?investigator - investigator)
    :precondition
      (and
        (entity_triaged ?product_line)
        (product_line_has_sample ?product_line ?sample)
        (sample_in_custody ?sample)
        (sample_linked_to_recall ?sample ?recall_action)
        (entity_assigned_investigator ?product_line ?investigator)
        (not
          (recall_needs_evidence_review ?recall_action)
        )
        (not
          (product_line_evidence_reviewed ?product_line)
        )
      )
    :effect (product_line_evidence_reviewed ?product_line)
  )
  (:action assign_expertise_area_to_product_line
    :parameters (?product_line - product_line ?expertise_area - expertise_area)
    :precondition
      (and
        (entity_triaged ?product_line)
        (expertise_area_available ?expertise_area)
        (not
          (product_line_expertise_assigned ?product_line)
        )
      )
    :effect
      (and
        (product_line_expertise_assigned ?product_line)
        (product_line_linked_expertise_area ?product_line ?expertise_area)
        (not
          (expertise_area_available ?expertise_area)
        )
      )
  )
  (:action start_expert_review_with_sample
    :parameters (?product_line - product_line ?sample - sample ?recall_action - recall_action ?investigator - investigator ?expertise_area - expertise_area)
    :precondition
      (and
        (entity_triaged ?product_line)
        (product_line_has_sample ?product_line ?sample)
        (sample_in_custody ?sample)
        (sample_linked_to_recall ?sample ?recall_action)
        (entity_assigned_investigator ?product_line ?investigator)
        (recall_needs_evidence_review ?recall_action)
        (product_line_expertise_assigned ?product_line)
        (product_line_linked_expertise_area ?product_line ?expertise_area)
        (not
          (product_line_evidence_reviewed ?product_line)
        )
      )
    :effect
      (and
        (product_line_evidence_reviewed ?product_line)
        (product_line_expert_review_progressed ?product_line)
      )
  )
  (:action start_mitigation_option_analysis
    :parameters (?product_line - product_line ?mitigation_option - mitigation_option ?quality_representative - quality_representative ?sample - sample ?recall_action - recall_action)
    :precondition
      (and
        (product_line_evidence_reviewed ?product_line)
        (product_line_has_mitigation_option ?product_line ?mitigation_option)
        (entity_assigned_quality_representative ?product_line ?quality_representative)
        (product_line_has_sample ?product_line ?sample)
        (sample_linked_to_recall ?sample ?recall_action)
        (not
          (recall_needs_expert_review ?recall_action)
        )
        (not
          (product_line_testing_started ?product_line)
        )
      )
    :effect (product_line_testing_started ?product_line)
  )
  (:action continue_mitigation_option_analysis
    :parameters (?product_line - product_line ?mitigation_option - mitigation_option ?quality_representative - quality_representative ?sample - sample ?recall_action - recall_action)
    :precondition
      (and
        (product_line_evidence_reviewed ?product_line)
