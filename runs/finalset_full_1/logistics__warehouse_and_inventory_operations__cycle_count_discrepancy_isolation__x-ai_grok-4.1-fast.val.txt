--- domain.pddl (as passed to VAL) ---
(define (domain warehouse_cycle_count_discrepancy_isolation)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_group - object equipment_group - object asset_group - object case_container - object cycle_count_task - case_container cycle_counter - resource_group scanner_device - resource_group material_handler - resource_group photo_evidence - resource_group audit_checklist - resource_group discrepancy_code - resource_group correction_code - resource_group audit_reference - resource_group count_tag - equipment_group sku_item - equipment_group supervisor_approver - equipment_group shelf_slot - asset_group pick_slot - asset_group adjustment_record - asset_group location_group - cycle_count_task sub_location_group - cycle_count_task storage_bin - location_group associated_bin - location_group discrepancy_case - sub_location_group)
  (:predicates
    (task_created ?cycle_count_task - cycle_count_task)
    (verification_completed ?cycle_count_task - cycle_count_task)
    (task_counter_assigned ?cycle_count_task - cycle_count_task)
    (reconciliation_applied_entity ?cycle_count_task - cycle_count_task)
    (marked_for_reconciliation_entity ?cycle_count_task - cycle_count_task)
    (adjustment_authorized_entity ?cycle_count_task - cycle_count_task)
    (counter_available ?cycle_counter - cycle_counter)
    (assigned_counter ?cycle_count_task - cycle_count_task ?cycle_counter - cycle_counter)
    (scanner_available ?scanner_device - scanner_device)
    (assigned_scanner ?cycle_count_task - cycle_count_task ?scanner_device - scanner_device)
    (handler_available ?material_handler - material_handler)
    (assigned_handler ?cycle_count_task - cycle_count_task ?material_handler - material_handler)
    (tag_available ?count_tag - count_tag)
    (tag_attached_to_bin ?storage_bin - storage_bin ?count_tag - count_tag)
    (tag_attached_to_associated_bin ?associated_bin - associated_bin ?count_tag - count_tag)
    (bin_slot_linked ?storage_bin - storage_bin ?shelf_slot - shelf_slot)
    (slot_locked ?shelf_slot - shelf_slot)
    (slot_tagged ?shelf_slot - shelf_slot)
    (bin_evidence_recorded ?storage_bin - storage_bin)
    (associated_bin_slot_linked ?associated_bin - associated_bin ?pick_slot - pick_slot)
    (pick_slot_locked ?pick_slot - pick_slot)
    (pick_slot_tagged ?pick_slot - pick_slot)
    (associated_bin_evidence_recorded ?associated_bin - associated_bin)
    (adjustment_record_available ?adjustment_record - adjustment_record)
    (adjustment_record_active ?adjustment_record - adjustment_record)
    (adjustment_record_slot_link ?adjustment_record - adjustment_record ?shelf_slot - shelf_slot)
    (adjustment_record_pick_link ?adjustment_record - adjustment_record ?pick_slot - pick_slot)
    (record_requires_slot_tag_verification ?adjustment_record - adjustment_record)
    (record_requires_pick_tag_verification ?adjustment_record - adjustment_record)
    (record_staged_for_sku_association ?adjustment_record - adjustment_record)
    (case_bin_linked ?discrepancy_case - discrepancy_case ?storage_bin - storage_bin)
    (case_associated_bin_linked ?discrepancy_case - discrepancy_case ?associated_bin - associated_bin)
    (case_adjustment_record_linked ?discrepancy_case - discrepancy_case ?adjustment_record - adjustment_record)
    (sku_available ?sku_item - sku_item)
    (case_sku_linked ?discrepancy_case - discrepancy_case ?sku_item - sku_item)
    (sku_verified ?sku_item - sku_item)
    (sku_linked_to_adjustment_record ?sku_item - sku_item ?adjustment_record - adjustment_record)
    (case_sku_verified ?discrepancy_case - discrepancy_case)
    (case_evidence_collection_complete ?discrepancy_case - discrepancy_case)
    (investigation_in_progress ?discrepancy_case - discrepancy_case)
    (case_photo_attached ?discrepancy_case - discrepancy_case)
    (case_prepared_for_investigation ?discrepancy_case - discrepancy_case)
    (checklist_attached ?discrepancy_case - discrepancy_case)
    (investigator_check_complete ?discrepancy_case - discrepancy_case)
    (approver_available ?supervisor_approver - supervisor_approver)
    (case_approver_linked ?discrepancy_case - discrepancy_case ?supervisor_approver - supervisor_approver)
    (case_approved ?discrepancy_case - discrepancy_case)
    (supervisor_reviewed ?discrepancy_case - discrepancy_case)
    (supervisor_finalized ?discrepancy_case - discrepancy_case)
    (photo_available ?photo_evidence - photo_evidence)
    (case_linked_photo ?discrepancy_case - discrepancy_case ?photo_evidence - photo_evidence)
    (checklist_available ?audit_checklist - audit_checklist)
    (case_linked_checklist ?discrepancy_case - discrepancy_case ?audit_checklist - audit_checklist)
    (correction_code_available ?correction_code - correction_code)
    (case_has_correction_code ?discrepancy_case - discrepancy_case ?correction_code - correction_code)
    (audit_reference_available ?audit_reference - audit_reference)
    (case_has_audit_reference ?discrepancy_case - discrepancy_case ?audit_reference - audit_reference)
    (discrepancy_code_available ?discrepancy_code - discrepancy_code)
    (discrepancy_code_linked ?cycle_count_task - cycle_count_task ?discrepancy_code - discrepancy_code)
    (bin_isolation_marked ?storage_bin - storage_bin)
    (associated_bin_isolation_marked ?associated_bin - associated_bin)
    (case_ready_for_reconciliation ?discrepancy_case - discrepancy_case)
  )
  (:action initiate_cycle_count_task
    :parameters (?cycle_count_task - cycle_count_task)
    :precondition
      (and
        (not
          (task_created ?cycle_count_task)
        )
        (not
          (reconciliation_applied_entity ?cycle_count_task)
        )
      )
    :effect (task_created ?cycle_count_task)
  )
  (:action assign_counter_to_task
    :parameters (?cycle_count_task - cycle_count_task ?cycle_counter - cycle_counter)
    :precondition
      (and
        (task_created ?cycle_count_task)
        (not
          (task_counter_assigned ?cycle_count_task)
        )
        (counter_available ?cycle_counter)
      )
    :effect
      (and
        (task_counter_assigned ?cycle_count_task)
        (assigned_counter ?cycle_count_task ?cycle_counter)
        (not
          (counter_available ?cycle_counter)
        )
      )
  )
  (:action assign_scanner_to_task
    :parameters (?cycle_count_task - cycle_count_task ?scanner_device - scanner_device)
    :precondition
      (and
        (task_created ?cycle_count_task)
        (task_counter_assigned ?cycle_count_task)
        (scanner_available ?scanner_device)
      )
    :effect
      (and
        (assigned_scanner ?cycle_count_task ?scanner_device)
        (not
          (scanner_available ?scanner_device)
        )
      )
  )
  (:action finalize_task_scan
    :parameters (?cycle_count_task - cycle_count_task ?scanner_device - scanner_device)
    :precondition
      (and
        (task_created ?cycle_count_task)
        (task_counter_assigned ?cycle_count_task)
        (assigned_scanner ?cycle_count_task ?scanner_device)
        (not
          (verification_completed ?cycle_count_task)
        )
      )
    :effect (verification_completed ?cycle_count_task)
  )
  (:action release_scanner_from_task
    :parameters (?cycle_count_task - cycle_count_task ?scanner_device - scanner_device)
    :precondition
      (and
        (assigned_scanner ?cycle_count_task ?scanner_device)
      )
    :effect
      (and
        (scanner_available ?scanner_device)
        (not
          (assigned_scanner ?cycle_count_task ?scanner_device)
        )
      )
  )
  (:action assign_handler_to_task
    :parameters (?cycle_count_task - cycle_count_task ?material_handler - material_handler)
    :precondition
      (and
        (verification_completed ?cycle_count_task)
        (handler_available ?material_handler)
      )
    :effect
      (and
        (assigned_handler ?cycle_count_task ?material_handler)
        (not
          (handler_available ?material_handler)
        )
      )
  )
  (:action release_handler_from_task
    :parameters (?cycle_count_task - cycle_count_task ?material_handler - material_handler)
    :precondition
      (and
        (assigned_handler ?cycle_count_task ?material_handler)
      )
    :effect
      (and
        (handler_available ?material_handler)
        (not
          (assigned_handler ?cycle_count_task ?material_handler)
        )
      )
  )
  (:action attach_correction_code_to_case
    :parameters (?discrepancy_case - discrepancy_case ?correction_code - correction_code)
    :precondition
      (and
        (verification_completed ?discrepancy_case)
        (correction_code_available ?correction_code)
      )
    :effect
      (and
        (case_has_correction_code ?discrepancy_case ?correction_code)
        (not
          (correction_code_available ?correction_code)
        )
      )
  )
  (:action detach_correction_code_from_case
    :parameters (?discrepancy_case - discrepancy_case ?correction_code - correction_code)
    :precondition
      (and
        (case_has_correction_code ?discrepancy_case ?correction_code)
      )
    :effect
      (and
        (correction_code_available ?correction_code)
        (not
          (case_has_correction_code ?discrepancy_case ?correction_code)
        )
      )
  )
  (:action attach_audit_reference_to_case
    :parameters (?discrepancy_case - discrepancy_case ?audit_reference - audit_reference)
    :precondition
      (and
        (verification_completed ?discrepancy_case)
        (audit_reference_available ?audit_reference)
      )
    :effect
      (and
        (case_has_audit_reference ?discrepancy_case ?audit_reference)
        (not
          (audit_reference_available ?audit_reference)
        )
      )
  )
  (:action detach_audit_reference_from_case
    :parameters (?discrepancy_case - discrepancy_case ?audit_reference - audit_reference)
    :precondition
      (and
        (case_has_audit_reference ?discrepancy_case ?audit_reference)
      )
    :effect
      (and
        (audit_reference_available ?audit_reference)
        (not
          (case_has_audit_reference ?discrepancy_case ?audit_reference)
        )
      )
  )
  (:action isolate_shelf_slot_for_bin
    :parameters (?storage_bin - storage_bin ?shelf_slot - shelf_slot ?scanner_device - scanner_device)
    :precondition
      (and
        (verification_completed ?storage_bin)
        (assigned_scanner ?storage_bin ?scanner_device)
        (bin_slot_linked ?storage_bin ?shelf_slot)
        (not
          (slot_locked ?shelf_slot)
        )
        (not
          (slot_tagged ?shelf_slot)
        )
      )
    :effect (slot_locked ?shelf_slot)
  )
  (:action collect_bin_evidence
    :parameters (?storage_bin - storage_bin ?shelf_slot - shelf_slot ?material_handler - material_handler)
    :precondition
      (and
        (verification_completed ?storage_bin)
        (assigned_handler ?storage_bin ?material_handler)
        (bin_slot_linked ?storage_bin ?shelf_slot)
        (slot_locked ?shelf_slot)
        (not
          (bin_isolation_marked ?storage_bin)
        )
      )
    :effect
      (and
        (bin_isolation_marked ?storage_bin)
        (bin_evidence_recorded ?storage_bin)
      )
  )
  (:action tag_slot_for_investigation
    :parameters (?storage_bin - storage_bin ?shelf_slot - shelf_slot ?count_tag - count_tag)
    :precondition
      (and
        (verification_completed ?storage_bin)
        (bin_slot_linked ?storage_bin ?shelf_slot)
        (tag_available ?count_tag)
        (not
          (bin_isolation_marked ?storage_bin)
        )
      )
    :effect
      (and
        (slot_tagged ?shelf_slot)
        (bin_isolation_marked ?storage_bin)
        (tag_attached_to_bin ?storage_bin ?count_tag)
        (not
          (tag_available ?count_tag)
        )
      )
  )
  (:action confirm_tag_and_lock_slot
    :parameters (?storage_bin - storage_bin ?shelf_slot - shelf_slot ?scanner_device - scanner_device ?count_tag - count_tag)
    :precondition
      (and
        (verification_completed ?storage_bin)
        (assigned_scanner ?storage_bin ?scanner_device)
        (bin_slot_linked ?storage_bin ?shelf_slot)
        (slot_tagged ?shelf_slot)
        (tag_attached_to_bin ?storage_bin ?count_tag)
        (not
          (bin_evidence_recorded ?storage_bin)
        )
      )
    :effect
      (and
        (slot_locked ?shelf_slot)
        (bin_evidence_recorded ?storage_bin)
        (tag_available ?count_tag)
        (not
          (tag_attached_to_bin ?storage_bin ?count_tag)
        )
      )
  )
  (:action isolate_pick_slot_for_associated_bin
    :parameters (?associated_bin - associated_bin ?pick_slot - pick_slot ?scanner_device - scanner_device)
    :precondition
      (and
        (verification_completed ?associated_bin)
        (assigned_scanner ?associated_bin ?scanner_device)
        (associated_bin_slot_linked ?associated_bin ?pick_slot)
        (not
          (pick_slot_locked ?pick_slot)
        )
        (not
          (pick_slot_tagged ?pick_slot)
        )
      )
    :effect (pick_slot_locked ?pick_slot)
  )
  (:action collect_associated_bin_evidence
    :parameters (?associated_bin - associated_bin ?pick_slot - pick_slot ?material_handler - material_handler)
    :precondition
      (and
        (verification_completed ?associated_bin)
        (assigned_handler ?associated_bin ?material_handler)
        (associated_bin_slot_linked ?associated_bin ?pick_slot)
        (pick_slot_locked ?pick_slot)
        (not
          (associated_bin_isolation_marked ?associated_bin)
        )
      )
    :effect
      (and
        (associated_bin_isolation_marked ?associated_bin)
        (associated_bin_evidence_recorded ?associated_bin)
      )
  )
  (:action tag_pick_slot_for_investigation
    :parameters (?associated_bin - associated_bin ?pick_slot - pick_slot ?count_tag - count_tag)
    :precondition
      (and
        (verification_completed ?associated_bin)
        (associated_bin_slot_linked ?associated_bin ?pick_slot)
        (tag_available ?count_tag)
        (not
          (associated_bin_isolation_marked ?associated_bin)
        )
      )
    :effect
      (and
        (pick_slot_tagged ?pick_slot)
        (associated_bin_isolation_marked ?associated_bin)
        (tag_attached_to_associated_bin ?associated_bin ?count_tag)
        (not
          (tag_available ?count_tag)
        )
      )
  )
  (:action confirm_pick_tag_and_lock
    :parameters (?associated_bin - associated_bin ?pick_slot - pick_slot ?scanner_device - scanner_device ?count_tag - count_tag)
    :precondition
      (and
        (verification_completed ?associated_bin)
        (assigned_scanner ?associated_bin ?scanner_device)
        (associated_bin_slot_linked ?associated_bin ?pick_slot)
        (pick_slot_tagged ?pick_slot)
        (tag_attached_to_associated_bin ?associated_bin ?count_tag)
        (not
          (associated_bin_evidence_recorded ?associated_bin)
        )
      )
    :effect
      (and
        (pick_slot_locked ?pick_slot)
        (associated_bin_evidence_recorded ?associated_bin)
        (tag_available ?count_tag)
        (not
          (tag_attached_to_associated_bin ?associated_bin ?count_tag)
        )
      )
  )
  (:action generate_adjustment_record
    :parameters (?storage_bin - storage_bin ?associated_bin - associated_bin ?shelf_slot - shelf_slot ?pick_slot - pick_slot ?adjustment_record - adjustment_record)
    :precondition
      (and
        (bin_isolation_marked ?storage_bin)
        (associated_bin_isolation_marked ?associated_bin)
        (bin_slot_linked ?storage_bin ?shelf_slot)
        (associated_bin_slot_linked ?associated_bin ?pick_slot)
        (slot_locked ?shelf_slot)
        (pick_slot_locked ?pick_slot)
        (bin_evidence_recorded ?storage_bin)
        (associated_bin_evidence_recorded ?associated_bin)
        (adjustment_record_available ?adjustment_record)
      )
    :effect
      (and
        (adjustment_record_active ?adjustment_record)
        (adjustment_record_slot_link ?adjustment_record ?shelf_slot)
        (adjustment_record_pick_link ?adjustment_record ?pick_slot)
        (not
          (adjustment_record_available ?adjustment_record)
        )
      )
  )
  (:action generate_adjustment_record_from_tagged_slot
    :parameters (?storage_bin - storage_bin ?associated_bin - associated_bin ?shelf_slot - shelf_slot ?pick_slot - pick_slot ?adjustment_record - adjustment_record)
    :precondition
      (and
        (bin_isolation_marked ?storage_bin)
        (associated_bin_isolation_marked ?associated_bin)
        (bin_slot_linked ?storage_bin ?shelf_slot)
        (associated_bin_slot_linked ?associated_bin ?pick_slot)
        (slot_tagged ?shelf_slot)
        (pick_slot_locked ?pick_slot)
        (not
          (bin_evidence_recorded ?storage_bin)
        )
        (associated_bin_evidence_recorded ?associated_bin)
        (adjustment_record_available ?adjustment_record)
      )
    :effect
      (and
        (adjustment_record_active ?adjustment_record)
        (adjustment_record_slot_link ?adjustment_record ?shelf_slot)
        (adjustment_record_pick_link ?adjustment_record ?pick_slot)
        (record_requires_slot_tag_verification ?adjustment_record)
        (not
          (adjustment_record_available ?adjustment_record)
        )
      )
  )
  (:action generate_adjustment_record_from_pick_tag
    :parameters (?storage_bin - storage_bin ?associated_bin - associated_bin ?shelf_slot - shelf_slot ?pick_slot - pick_slot ?adjustment_record - adjustment_record)
    :precondition
      (and
        (bin_isolation_marked ?storage_bin)
        (associated_bin_isolation_marked ?associated_bin)
        (bin_slot_linked ?storage_bin ?shelf_slot)
        (associated_bin_slot_linked ?associated_bin ?pick_slot)
        (slot_locked ?shelf_slot)
        (pick_slot_tagged ?pick_slot)
        (bin_evidence_recorded ?storage_bin)
        (not
          (associated_bin_evidence_recorded ?associated_bin)
        )
        (adjustment_record_available ?adjustment_record)
      )
    :effect
      (and
        (adjustment_record_active ?adjustment_record)
        (adjustment_record_slot_link ?adjustment_record ?shelf_slot)
        (adjustment_record_pick_link ?adjustment_record ?pick_slot)
        (record_requires_pick_tag_verification ?adjustment_record)
        (not
          (adjustment_record_available ?adjustment_record)
        )
      )
  )
  (:action generate_adjustment_record_from_both_tags
    :parameters (?storage_bin - storage_bin ?associated_bin - associated_bin ?shelf_slot - shelf_slot ?pick_slot - pick_slot ?adjustment_record - adjustment_record)
    :precondition
      (and
        (bin_isolation_marked ?storage_bin)
        (associated_bin_isolation_marked ?associated_bin)
        (bin_slot_linked ?storage_bin ?shelf_slot)
        (associated_bin_slot_linked ?associated_bin ?pick_slot)
        (slot_tagged ?shelf_slot)
        (pick_slot_tagged ?pick_slot)
        (not
          (bin_evidence_recorded ?storage_bin)
        )
        (not
          (associated_bin_evidence_recorded ?associated_bin)
        )
        (adjustment_record_available ?adjustment_record)
      )
    :effect
      (and
        (adjustment_record_active ?adjustment_record)
        (adjustment_record_slot_link ?adjustment_record ?shelf_slot)
        (adjustment_record_pick_link ?adjustment_record ?pick_slot)
        (record_requires_slot_tag_verification ?adjustment_record)
        (record_requires_pick_tag_verification ?adjustment_record)
        (not
          (adjustment_record_available ?adjustment_record)
        )
      )
  )
  (:action stage_adjustment_record_for_sku_association
    :parameters (?adjustment_record - adjustment_record ?storage_bin - storage_bin ?scanner_device - scanner_device)
    :precondition
      (and
        (adjustment_record_active ?adjustment_record)
        (bin_isolation_marked ?storage_bin)
        (assigned_scanner ?storage_bin ?scanner_device)
        (not
          (record_staged_for_sku_association ?adjustment_record)
        )
      )
    :effect (record_staged_for_sku_association ?adjustment_record)
  )
  (:action associate_sku_with_adjustment_record
    :parameters (?discrepancy_case - discrepancy_case ?sku_item - sku_item ?adjustment_record - adjustment_record)
    :precondition
      (and
        (verification_completed ?discrepancy_case)
        (case_adjustment_record_linked ?discrepancy_case ?adjustment_record)
        (case_sku_linked ?discrepancy_case ?sku_item)
        (sku_available ?sku_item)
        (adjustment_record_active ?adjustment_record)
        (record_staged_for_sku_association ?adjustment_record)
        (not
          (sku_verified ?sku_item)
        )
      )
    :effect
      (and
        (sku_verified ?sku_item)
        (sku_linked_to_adjustment_record ?sku_item ?adjustment_record)
        (not
          (sku_available ?sku_item)
        )
      )
  )
  (:action verify_sku_on_case
    :parameters (?discrepancy_case - discrepancy_case ?sku_item - sku_item ?adjustment_record - adjustment_record ?scanner_device - scanner_device)
    :precondition
      (and
        (verification_completed ?discrepancy_case)
        (case_sku_linked ?discrepancy_case ?sku_item)
        (sku_verified ?sku_item)
        (sku_linked_to_adjustment_record ?sku_item ?adjustment_record)
        (assigned_scanner ?discrepancy_case ?scanner_device)
        (not
          (record_requires_slot_tag_verification ?adjustment_record)
        )
        (not
          (case_sku_verified ?discrepancy_case)
        )
      )
    :effect (case_sku_verified ?discrepancy_case)
  )
  (:action attach_photo_evidence_to_case
    :parameters (?discrepancy_case - discrepancy_case ?photo_evidence - photo_evidence)
    :precondition
      (and
        (verification_completed ?discrepancy_case)
        (photo_available ?photo_evidence)
        (not
          (case_photo_attached ?discrepancy_case)
        )
      )
    :effect
      (and
        (case_photo_attached ?discrepancy_case)
        (case_linked_photo ?discrepancy_case ?photo_evidence)
        (not
          (photo_available ?photo_evidence)
        )
      )
  )
  (:action prepare_discrepancy_case
    :parameters (?discrepancy_case - discrepancy_case ?sku_item - sku_item ?adjustment_record - adjustment_record ?scanner_device - scanner_device ?photo_evidence - photo_evidence)
    :precondition
      (and
        (verification_completed ?discrepancy_case)
        (case_sku_linked ?discrepancy_case ?sku_item)
        (sku_verified ?sku_item)
        (sku_linked_to_adjustment_record ?sku_item ?adjustment_record)
        (assigned_scanner ?discrepancy_case ?scanner_device)
        (record_requires_slot_tag_verification ?adjustment_record)
        (case_photo_attached ?discrepancy_case)
        (case_linked_photo ?discrepancy_case ?photo_evidence)
        (not
          (case_sku_verified ?discrepancy_case)
        )
      )
    :effect
      (and
        (case_sku_verified ?discrepancy_case)
        (case_prepared_for_investigation ?discrepancy_case)
      )
  )
  (:action finalize_case_evidence_step1
    :parameters (?discrepancy_case - discrepancy_case ?correction_code - correction_code ?material_handler - material_handler ?sku_item - sku_item ?adjustment_record - adjustment_record)
    :precondition
      (and
        (case_sku_verified ?discrepancy_case)
        (case_has_correction_code ?discrepancy_case ?correction_code)
        (assigned_handler ?discrepancy_case ?material_handler)
        (case_sku_linked ?discrepancy_case ?sku_item)
        (sku_linked_to_adjustment_record ?sku_item ?adjustment_record)
        (not
          (record_requires_pick_tag_verification ?adjustment_record)
        )
        (not
          (case_evidence_collection_complete ?discrepancy_case)
        )
      )
    :effect (case_evidence_collection_complete ?discrepancy_case)
  )
  (:action finalize_case_evidence_step2
    :parameters (?discrepancy_case - discrepancy_case ?correction_code - correction_code ?material_handler - material_handler ?sku_item - sku_item ?adjustment_record - adjustment_record)
    :precondition
      (and
        (case_sku_verified ?discrepancy_case)
        (case_has_correction_code ?discrepancy_case ?correction_code)
        (assigned_handler ?discrepancy_case ?material_handler)
        (case_sku_linked ?discrepancy_case ?sku_item)
        (sku_linked_to_adjustment_record ?sku_item ?adjustment_record)
