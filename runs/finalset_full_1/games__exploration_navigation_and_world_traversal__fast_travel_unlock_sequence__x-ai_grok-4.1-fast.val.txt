--- domain.pddl (as passed to VAL) ---
(define (domain fast_travel_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types actor_or_item_category - object bindable_asset_category - object path_and_anchor_category - object site_category - object map_location - site_category registration_device - actor_or_item_category survey_device - actor_or_item_category attaching_agent - actor_or_item_category special_key - actor_or_item_category upgrade_module - actor_or_item_category signature_token - actor_or_item_category activation_component - actor_or_item_category relay_module - actor_or_item_category consumable_token - bindable_asset_category map_tile - bindable_asset_category location_stamp - bindable_asset_category approach_path_a - path_and_anchor_category approach_path_b - path_and_anchor_category anchor_candidate - path_and_anchor_category entry_subgroup - map_location hub_subgroup - map_location region_entry_a - entry_subgroup region_entry_b - entry_subgroup fast_travel_hub - hub_subgroup)
  (:predicates
    (location_discovered ?map_location - map_location)
    (location_verified ?map_location - map_location)
    (location_registered ?map_location - map_location)
    (fast_travel_unlocked ?map_location - map_location)
    (location_certified ?map_location - map_location)
    (player_signature_marked ?map_location - map_location)
    (registration_device_available ?registration_device - registration_device)
    (registered_with_device ?map_location - map_location ?registration_device - registration_device)
    (survey_device_available ?survey_device - survey_device)
    (surveyed_with_device ?map_location - map_location ?survey_device - survey_device)
    (attaching_agent_available ?attaching_agent - attaching_agent)
    (agent_attached_to_location ?map_location - map_location ?attaching_agent - attaching_agent)
    (consumable_token_available ?consumable_token - consumable_token)
    (entry_a_token_applied ?region_entry_a - region_entry_a ?consumable_token - consumable_token)
    (entry_b_token_applied ?region_entry_b - region_entry_b ?consumable_token - consumable_token)
    (entry_a_approach_path ?region_entry_a - region_entry_a ?approach_path_a - approach_path_a)
    (approach_path_a_cleared ?approach_path_a - approach_path_a)
    (approach_path_a_flagged ?approach_path_a - approach_path_a)
    (entry_a_cleared ?region_entry_a - region_entry_a)
    (entry_b_approach_path ?region_entry_b - region_entry_b ?approach_path_b - approach_path_b)
    (approach_path_b_cleared ?approach_path_b - approach_path_b)
    (approach_path_b_flagged ?approach_path_b - approach_path_b)
    (entry_b_cleared ?region_entry_b - region_entry_b)
    (anchor_candidate_available ?anchor_candidate - anchor_candidate)
    (anchor_candidate_assembled ?anchor_candidate - anchor_candidate)
    (anchor_candidate_link_path_a ?anchor_candidate - anchor_candidate ?approach_path_a - approach_path_a)
    (anchor_candidate_link_path_b ?anchor_candidate - anchor_candidate ?approach_path_b - approach_path_b)
    (anchor_candidate_path_a_flagged ?anchor_candidate - anchor_candidate)
    (anchor_candidate_path_b_flagged ?anchor_candidate - anchor_candidate)
    (anchor_candidate_verified ?anchor_candidate - anchor_candidate)
    (hub_link_entry_a ?fast_travel_hub - fast_travel_hub ?region_entry_a - region_entry_a)
    (hub_link_entry_b ?fast_travel_hub - fast_travel_hub ?region_entry_b - region_entry_b)
    (hub_has_candidate ?fast_travel_hub - fast_travel_hub ?anchor_candidate - anchor_candidate)
    (map_tile_available ?map_tile - map_tile)
    (hub_tile_slot ?fast_travel_hub - fast_travel_hub ?map_tile - map_tile)
    (tile_bound ?map_tile - map_tile)
    (tile_bound_to_anchor_candidate ?map_tile - map_tile ?anchor_candidate - anchor_candidate)
    (hub_ready_for_activation ?fast_travel_hub - fast_travel_hub)
    (hub_activation_component_installed ?fast_travel_hub - fast_travel_hub)
    (hub_checks_passed ?fast_travel_hub - fast_travel_hub)
    (hub_key_applied ?fast_travel_hub - fast_travel_hub)
    (hub_key_pair_confirmed ?fast_travel_hub - fast_travel_hub)
    (hub_upgrade_ready ?fast_travel_hub - fast_travel_hub)
    (hub_final_stage_ready ?fast_travel_hub - fast_travel_hub)
    (location_stamp_available ?location_stamp - location_stamp)
    (hub_has_stamp ?fast_travel_hub - fast_travel_hub ?location_stamp - location_stamp)
    (hub_stamp_applied ?fast_travel_hub - fast_travel_hub)
    (hub_stamp_verified ?fast_travel_hub - fast_travel_hub)
    (hub_stamp_certified ?fast_travel_hub - fast_travel_hub)
    (special_key_available ?special_key - special_key)
    (hub_key_link ?fast_travel_hub - fast_travel_hub ?special_key - special_key)
    (upgrade_module_available ?upgrade_module - upgrade_module)
    (hub_upgrade_installed ?fast_travel_hub - fast_travel_hub ?upgrade_module - upgrade_module)
    (activation_component_available ?activation_component - activation_component)
    (hub_activation_component_attached ?fast_travel_hub - fast_travel_hub ?activation_component - activation_component)
    (relay_module_available ?relay_module - relay_module)
    (hub_relay_attached ?fast_travel_hub - fast_travel_hub ?relay_module - relay_module)
    (signature_token_available ?signature_token - signature_token)
    (signature_attached_to_location ?map_location - map_location ?signature_token - signature_token)
    (entry_a_ready ?region_entry_a - region_entry_a)
    (entry_b_ready ?region_entry_b - region_entry_b)
    (hub_activation_persisted ?fast_travel_hub - fast_travel_hub)
  )
  (:action discover_map_location
    :parameters (?map_location - map_location)
    :precondition
      (and
        (not
          (location_discovered ?map_location)
        )
        (not
          (fast_travel_unlocked ?map_location)
        )
      )
    :effect (location_discovered ?map_location)
  )
  (:action register_location_with_device
    :parameters (?map_location - map_location ?registration_device - registration_device)
    :precondition
      (and
        (location_discovered ?map_location)
        (not
          (location_registered ?map_location)
        )
        (registration_device_available ?registration_device)
      )
    :effect
      (and
        (location_registered ?map_location)
        (registered_with_device ?map_location ?registration_device)
        (not
          (registration_device_available ?registration_device)
        )
      )
  )
  (:action survey_location_with_device
    :parameters (?map_location - map_location ?survey_device - survey_device)
    :precondition
      (and
        (location_discovered ?map_location)
        (location_registered ?map_location)
        (survey_device_available ?survey_device)
      )
    :effect
      (and
        (surveyed_with_device ?map_location ?survey_device)
        (not
          (survey_device_available ?survey_device)
        )
      )
  )
  (:action verify_location
    :parameters (?map_location - map_location ?survey_device - survey_device)
    :precondition
      (and
        (location_discovered ?map_location)
        (location_registered ?map_location)
        (surveyed_with_device ?map_location ?survey_device)
        (not
          (location_verified ?map_location)
        )
      )
    :effect (location_verified ?map_location)
  )
  (:action return_survey_device
    :parameters (?map_location - map_location ?survey_device - survey_device)
    :precondition
      (and
        (surveyed_with_device ?map_location ?survey_device)
      )
    :effect
      (and
        (survey_device_available ?survey_device)
        (not
          (surveyed_with_device ?map_location ?survey_device)
        )
      )
  )
  (:action attach_agent_to_location
    :parameters (?map_location - map_location ?attaching_agent - attaching_agent)
    :precondition
      (and
        (location_verified ?map_location)
        (attaching_agent_available ?attaching_agent)
      )
    :effect
      (and
        (agent_attached_to_location ?map_location ?attaching_agent)
        (not
          (attaching_agent_available ?attaching_agent)
        )
      )
  )
  (:action detach_agent_from_location
    :parameters (?map_location - map_location ?attaching_agent - attaching_agent)
    :precondition
      (and
        (agent_attached_to_location ?map_location ?attaching_agent)
      )
    :effect
      (and
        (attaching_agent_available ?attaching_agent)
        (not
          (agent_attached_to_location ?map_location ?attaching_agent)
        )
      )
  )
  (:action attach_activation_component_to_hub
    :parameters (?fast_travel_hub - fast_travel_hub ?activation_component - activation_component)
    :precondition
      (and
        (location_verified ?fast_travel_hub)
        (activation_component_available ?activation_component)
      )
    :effect
      (and
        (hub_activation_component_attached ?fast_travel_hub ?activation_component)
        (not
          (activation_component_available ?activation_component)
        )
      )
  )
  (:action detach_activation_component_from_hub
    :parameters (?fast_travel_hub - fast_travel_hub ?activation_component - activation_component)
    :precondition
      (and
        (hub_activation_component_attached ?fast_travel_hub ?activation_component)
      )
    :effect
      (and
        (activation_component_available ?activation_component)
        (not
          (hub_activation_component_attached ?fast_travel_hub ?activation_component)
        )
      )
  )
  (:action attach_relay_module_to_hub
    :parameters (?fast_travel_hub - fast_travel_hub ?relay_module - relay_module)
    :precondition
      (and
        (location_verified ?fast_travel_hub)
        (relay_module_available ?relay_module)
      )
    :effect
      (and
        (hub_relay_attached ?fast_travel_hub ?relay_module)
        (not
          (relay_module_available ?relay_module)
        )
      )
  )
  (:action detach_relay_module_from_hub
    :parameters (?fast_travel_hub - fast_travel_hub ?relay_module - relay_module)
    :precondition
      (and
        (hub_relay_attached ?fast_travel_hub ?relay_module)
      )
    :effect
      (and
        (relay_module_available ?relay_module)
        (not
          (hub_relay_attached ?fast_travel_hub ?relay_module)
        )
      )
  )
  (:action verify_and_clear_approach_path_a
    :parameters (?region_entry_a - region_entry_a ?approach_path_a - approach_path_a ?survey_device - survey_device)
    :precondition
      (and
        (location_verified ?region_entry_a)
        (surveyed_with_device ?region_entry_a ?survey_device)
        (entry_a_approach_path ?region_entry_a ?approach_path_a)
        (not
          (approach_path_a_cleared ?approach_path_a)
        )
        (not
          (approach_path_a_flagged ?approach_path_a)
        )
      )
    :effect (approach_path_a_cleared ?approach_path_a)
  )
  (:action bind_agent_and_mark_entry_a
    :parameters (?region_entry_a - region_entry_a ?approach_path_a - approach_path_a ?attaching_agent - attaching_agent)
    :precondition
      (and
        (location_verified ?region_entry_a)
        (agent_attached_to_location ?region_entry_a ?attaching_agent)
        (entry_a_approach_path ?region_entry_a ?approach_path_a)
        (approach_path_a_cleared ?approach_path_a)
        (not
          (entry_a_ready ?region_entry_a)
        )
      )
    :effect
      (and
        (entry_a_ready ?region_entry_a)
        (entry_a_cleared ?region_entry_a)
      )
  )
  (:action apply_consumable_to_entry_a
    :parameters (?region_entry_a - region_entry_a ?approach_path_a - approach_path_a ?consumable_token - consumable_token)
    :precondition
      (and
        (location_verified ?region_entry_a)
        (entry_a_approach_path ?region_entry_a ?approach_path_a)
        (consumable_token_available ?consumable_token)
        (not
          (entry_a_ready ?region_entry_a)
        )
      )
    :effect
      (and
        (approach_path_a_flagged ?approach_path_a)
        (entry_a_ready ?region_entry_a)
        (entry_a_token_applied ?region_entry_a ?consumable_token)
        (not
          (consumable_token_available ?consumable_token)
        )
      )
  )
  (:action finalize_entry_a_clearance
    :parameters (?region_entry_a - region_entry_a ?approach_path_a - approach_path_a ?survey_device - survey_device ?consumable_token - consumable_token)
    :precondition
      (and
        (location_verified ?region_entry_a)
        (surveyed_with_device ?region_entry_a ?survey_device)
        (entry_a_approach_path ?region_entry_a ?approach_path_a)
        (approach_path_a_flagged ?approach_path_a)
        (entry_a_token_applied ?region_entry_a ?consumable_token)
        (not
          (entry_a_cleared ?region_entry_a)
        )
      )
    :effect
      (and
        (approach_path_a_cleared ?approach_path_a)
        (entry_a_cleared ?region_entry_a)
        (consumable_token_available ?consumable_token)
        (not
          (entry_a_token_applied ?region_entry_a ?consumable_token)
        )
      )
  )
  (:action verify_and_clear_approach_path_b
    :parameters (?region_entry_b - region_entry_b ?approach_path_b - approach_path_b ?survey_device - survey_device)
    :precondition
      (and
        (location_verified ?region_entry_b)
        (surveyed_with_device ?region_entry_b ?survey_device)
        (entry_b_approach_path ?region_entry_b ?approach_path_b)
        (not
          (approach_path_b_cleared ?approach_path_b)
        )
        (not
          (approach_path_b_flagged ?approach_path_b)
        )
      )
    :effect (approach_path_b_cleared ?approach_path_b)
  )
  (:action bind_agent_and_mark_entry_b
    :parameters (?region_entry_b - region_entry_b ?approach_path_b - approach_path_b ?attaching_agent - attaching_agent)
    :precondition
      (and
        (location_verified ?region_entry_b)
        (agent_attached_to_location ?region_entry_b ?attaching_agent)
        (entry_b_approach_path ?region_entry_b ?approach_path_b)
        (approach_path_b_cleared ?approach_path_b)
        (not
          (entry_b_ready ?region_entry_b)
        )
      )
    :effect
      (and
        (entry_b_ready ?region_entry_b)
        (entry_b_cleared ?region_entry_b)
      )
  )
  (:action apply_consumable_to_entry_b
    :parameters (?region_entry_b - region_entry_b ?approach_path_b - approach_path_b ?consumable_token - consumable_token)
    :precondition
      (and
        (location_verified ?region_entry_b)
        (entry_b_approach_path ?region_entry_b ?approach_path_b)
        (consumable_token_available ?consumable_token)
        (not
          (entry_b_ready ?region_entry_b)
        )
      )
    :effect
      (and
        (approach_path_b_flagged ?approach_path_b)
        (entry_b_ready ?region_entry_b)
        (entry_b_token_applied ?region_entry_b ?consumable_token)
        (not
          (consumable_token_available ?consumable_token)
        )
      )
  )
  (:action finalize_entry_b_clearance
    :parameters (?region_entry_b - region_entry_b ?approach_path_b - approach_path_b ?survey_device - survey_device ?consumable_token - consumable_token)
    :precondition
      (and
        (location_verified ?region_entry_b)
        (surveyed_with_device ?region_entry_b ?survey_device)
        (entry_b_approach_path ?region_entry_b ?approach_path_b)
        (approach_path_b_flagged ?approach_path_b)
        (entry_b_token_applied ?region_entry_b ?consumable_token)
        (not
          (entry_b_cleared ?region_entry_b)
        )
      )
    :effect
      (and
        (approach_path_b_cleared ?approach_path_b)
        (entry_b_cleared ?region_entry_b)
        (consumable_token_available ?consumable_token)
        (not
          (entry_b_token_applied ?region_entry_b ?consumable_token)
        )
      )
  )
  (:action assemble_anchor_candidate_standard
    :parameters (?region_entry_a - region_entry_a ?region_entry_b - region_entry_b ?approach_path_a - approach_path_a ?approach_path_b - approach_path_b ?anchor_candidate - anchor_candidate)
    :precondition
      (and
        (entry_a_ready ?region_entry_a)
        (entry_b_ready ?region_entry_b)
        (entry_a_approach_path ?region_entry_a ?approach_path_a)
        (entry_b_approach_path ?region_entry_b ?approach_path_b)
        (approach_path_a_cleared ?approach_path_a)
        (approach_path_b_cleared ?approach_path_b)
        (entry_a_cleared ?region_entry_a)
        (entry_b_cleared ?region_entry_b)
        (anchor_candidate_available ?anchor_candidate)
      )
    :effect
      (and
        (anchor_candidate_assembled ?anchor_candidate)
        (anchor_candidate_link_path_a ?anchor_candidate ?approach_path_a)
        (anchor_candidate_link_path_b ?anchor_candidate ?approach_path_b)
        (not
          (anchor_candidate_available ?anchor_candidate)
        )
      )
  )
  (:action assemble_anchor_candidate_with_path_a_flag
    :parameters (?region_entry_a - region_entry_a ?region_entry_b - region_entry_b ?approach_path_a - approach_path_a ?approach_path_b - approach_path_b ?anchor_candidate - anchor_candidate)
    :precondition
      (and
        (entry_a_ready ?region_entry_a)
        (entry_b_ready ?region_entry_b)
        (entry_a_approach_path ?region_entry_a ?approach_path_a)
        (entry_b_approach_path ?region_entry_b ?approach_path_b)
        (approach_path_a_flagged ?approach_path_a)
        (approach_path_b_cleared ?approach_path_b)
        (not
          (entry_a_cleared ?region_entry_a)
        )
        (entry_b_cleared ?region_entry_b)
        (anchor_candidate_available ?anchor_candidate)
      )
    :effect
      (and
        (anchor_candidate_assembled ?anchor_candidate)
        (anchor_candidate_link_path_a ?anchor_candidate ?approach_path_a)
        (anchor_candidate_link_path_b ?anchor_candidate ?approach_path_b)
        (anchor_candidate_path_a_flagged ?anchor_candidate)
        (not
          (anchor_candidate_available ?anchor_candidate)
        )
      )
  )
  (:action assemble_anchor_candidate_with_path_b_flag
    :parameters (?region_entry_a - region_entry_a ?region_entry_b - region_entry_b ?approach_path_a - approach_path_a ?approach_path_b - approach_path_b ?anchor_candidate - anchor_candidate)
    :precondition
      (and
        (entry_a_ready ?region_entry_a)
        (entry_b_ready ?region_entry_b)
        (entry_a_approach_path ?region_entry_a ?approach_path_a)
        (entry_b_approach_path ?region_entry_b ?approach_path_b)
        (approach_path_a_cleared ?approach_path_a)
        (approach_path_b_flagged ?approach_path_b)
        (entry_a_cleared ?region_entry_a)
        (not
          (entry_b_cleared ?region_entry_b)
        )
        (anchor_candidate_available ?anchor_candidate)
      )
    :effect
      (and
        (anchor_candidate_assembled ?anchor_candidate)
        (anchor_candidate_link_path_a ?anchor_candidate ?approach_path_a)
        (anchor_candidate_link_path_b ?anchor_candidate ?approach_path_b)
        (anchor_candidate_path_b_flagged ?anchor_candidate)
        (not
          (anchor_candidate_available ?anchor_candidate)
        )
      )
  )
  (:action assemble_anchor_candidate_with_both_flags
    :parameters (?region_entry_a - region_entry_a ?region_entry_b - region_entry_b ?approach_path_a - approach_path_a ?approach_path_b - approach_path_b ?anchor_candidate - anchor_candidate)
    :precondition
      (and
        (entry_a_ready ?region_entry_a)
        (entry_b_ready ?region_entry_b)
        (entry_a_approach_path ?region_entry_a ?approach_path_a)
        (entry_b_approach_path ?region_entry_b ?approach_path_b)
        (approach_path_a_flagged ?approach_path_a)
        (approach_path_b_flagged ?approach_path_b)
        (not
          (entry_a_cleared ?region_entry_a)
        )
        (not
          (entry_b_cleared ?region_entry_b)
        )
        (anchor_candidate_available ?anchor_candidate)
      )
    :effect
      (and
        (anchor_candidate_assembled ?anchor_candidate)
        (anchor_candidate_link_path_a ?anchor_candidate ?approach_path_a)
        (anchor_candidate_link_path_b ?anchor_candidate ?approach_path_b)
        (anchor_candidate_path_a_flagged ?anchor_candidate)
        (anchor_candidate_path_b_flagged ?anchor_candidate)
        (not
          (anchor_candidate_available ?anchor_candidate)
        )
      )
  )
  (:action validate_anchor_candidate_with_survey
    :parameters (?anchor_candidate - anchor_candidate ?region_entry_a - region_entry_a ?survey_device - survey_device)
    :precondition
      (and
        (anchor_candidate_assembled ?anchor_candidate)
        (entry_a_ready ?region_entry_a)
        (surveyed_with_device ?region_entry_a ?survey_device)
        (not
          (anchor_candidate_verified ?anchor_candidate)
        )
      )
    :effect (anchor_candidate_verified ?anchor_candidate)
  )
  (:action bind_anchor_candidate_to_tile
    :parameters (?fast_travel_hub - fast_travel_hub ?map_tile - map_tile ?anchor_candidate - anchor_candidate)
    :precondition
      (and
        (location_verified ?fast_travel_hub)
        (hub_has_candidate ?fast_travel_hub ?anchor_candidate)
        (hub_tile_slot ?fast_travel_hub ?map_tile)
        (map_tile_available ?map_tile)
        (anchor_candidate_assembled ?anchor_candidate)
        (anchor_candidate_verified ?anchor_candidate)
        (not
          (tile_bound ?map_tile)
        )
      )
    :effect
      (and
        (tile_bound ?map_tile)
        (tile_bound_to_anchor_candidate ?map_tile ?anchor_candidate)
        (not
          (map_tile_available ?map_tile)
        )
      )
  )
  (:action prepare_hub_for_activation
    :parameters (?fast_travel_hub - fast_travel_hub ?map_tile - map_tile ?anchor_candidate - anchor_candidate ?survey_device - survey_device)
    :precondition
      (and
        (location_verified ?fast_travel_hub)
        (hub_tile_slot ?fast_travel_hub ?map_tile)
        (tile_bound ?map_tile)
        (tile_bound_to_anchor_candidate ?map_tile ?anchor_candidate)
        (surveyed_with_device ?fast_travel_hub ?survey_device)
        (not
          (anchor_candidate_path_a_flagged ?anchor_candidate)
        )
        (not
          (hub_ready_for_activation ?fast_travel_hub)
        )
      )
    :effect (hub_ready_for_activation ?fast_travel_hub)
  )
  (:action apply_special_key_to_hub
    :parameters (?fast_travel_hub - fast_travel_hub ?special_key - special_key)
    :precondition
      (and
        (location_verified ?fast_travel_hub)
        (special_key_available ?special_key)
        (not
          (hub_key_applied ?fast_travel_hub)
        )
      )
    :effect
      (and
        (hub_key_applied ?fast_travel_hub)
        (hub_key_link ?fast_travel_hub ?special_key)
        (not
          (special_key_available ?special_key)
        )
      )
  )
  (:action progress_hub_binding_with_key_and_tile
    :parameters (?fast_travel_hub - fast_travel_hub ?map_tile - map_tile ?anchor_candidate - anchor_candidate ?survey_device - survey_device ?special_key - special_key)
    :precondition
      (and
        (location_verified ?fast_travel_hub)
