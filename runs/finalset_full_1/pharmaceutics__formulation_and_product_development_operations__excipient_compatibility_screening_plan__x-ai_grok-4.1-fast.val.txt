--- domain.pddl (as passed to VAL) ---
(define (domain excipient_compatibility_screening)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity - object resource_category - entity analytical_category - entity process_category - entity candidate_root - entity candidate - candidate_root excipient_supplier - resource_category analytical_method - resource_category process_step - resource_category critical_quality_attribute - resource_category resource_allocation_record - resource_category evidence_artifact - resource_category material_property_test - resource_category stress_condition - resource_category sample_material - analytical_category analytical_batch - analytical_category regulatory_submission_input - analytical_category compatibility_flag_a - process_category compatibility_flag_b - process_category stability_or_scaleup_plan - process_category excipient_candidate_family - candidate formulation_study_family - candidate excipient_candidate_a - excipient_candidate_family excipient_candidate_b - excipient_candidate_family formulation_study - formulation_study_family)

  (:predicates
    (api_registered ?api_candidate - candidate)
    (entity_analysis_result_recorded ?api_candidate - candidate)
    (supplier_allocated ?api_candidate - candidate)
    (entity_compatibility_qualified ?api_candidate - candidate)
    (entity_ready_for_final_qualification ?api_candidate - candidate)
    (entity_final_review_completed ?api_candidate - candidate)
    (supplier_available ?excipient_supplier - excipient_supplier)
    (assigned_supplier ?api_candidate - candidate ?excipient_supplier - excipient_supplier)
    (method_available ?analytical_method - analytical_method)
    (entity_assigned_method ?api_candidate - candidate ?analytical_method - analytical_method)
    (process_step_available ?process_step - process_step)
    (assigned_process_step ?api_candidate - candidate ?process_step - process_step)
    (sample_material_available ?sample_material - sample_material)
    (sample_assigned_to_excipient_a ?excipient_candidate_a - excipient_candidate_a ?sample_material - sample_material)
    (sample_assigned_to_excipient_b ?excipient_candidate_b - excipient_candidate_b ?sample_material - sample_material)
    (compatibility_flag_a_applicable ?excipient_candidate_a - excipient_candidate_a ?compatibility_flag_a - compatibility_flag_a)
    (compatibility_flag_a_positive ?compatibility_flag_a - compatibility_flag_a)
    (compatibility_flag_a_negative ?compatibility_flag_a - compatibility_flag_a)
    (excipient_a_test_confirmed ?excipient_candidate_a - excipient_candidate_a)
    (compatibility_flag_b_applicable ?excipient_candidate_b - excipient_candidate_b ?compatibility_flag_b - compatibility_flag_b)
    (compatibility_flag_b_positive ?compatibility_flag_b - compatibility_flag_b)
    (compatibility_flag_b_negative ?compatibility_flag_b - compatibility_flag_b)
    (excipient_b_test_confirmed ?excipient_candidate_b - excipient_candidate_b)
    (stability_plan_available ?stability_or_scaleup_plan - stability_or_scaleup_plan)
    (stability_plan_committed ?stability_or_scaleup_plan - stability_or_scaleup_plan)
    (plan_includes_flag_a ?stability_or_scaleup_plan - stability_or_scaleup_plan ?compatibility_flag_a - compatibility_flag_a)
    (plan_includes_flag_b ?stability_or_scaleup_plan - stability_or_scaleup_plan ?compatibility_flag_b - compatibility_flag_b)
    (plan_requires_cqa_investigation ?stability_or_scaleup_plan - stability_or_scaleup_plan)
    (plan_requires_additional_material_test ?stability_or_scaleup_plan - stability_or_scaleup_plan)
    (plan_analysis_scheduled ?stability_or_scaleup_plan - stability_or_scaleup_plan)
    (study_has_excipient_a ?formulation_study - formulation_study ?excipient_candidate_a - excipient_candidate_a)
    (study_has_excipient_b ?formulation_study - formulation_study ?excipient_candidate_b - excipient_candidate_b)
    (study_linked_to_plan ?formulation_study - formulation_study ?stability_or_scaleup_plan - stability_or_scaleup_plan)
    (analytical_batch_available ?analytical_batch - analytical_batch)
    (study_assigned_analytical_batch ?formulation_study - formulation_study ?analytical_batch - analytical_batch)
    (analytical_batch_reserved ?analytical_batch - analytical_batch)
    (analytical_batch_linked_to_plan ?analytical_batch - analytical_batch ?stability_or_scaleup_plan - stability_or_scaleup_plan)
    (study_characterization_allocated ?formulation_study - formulation_study)
    (material_test_allocated ?formulation_study - formulation_study)
    (material_test_completed ?formulation_study - formulation_study)
    (study_cqa_reserved ?formulation_study - formulation_study)
    (study_cqa_reviewed ?formulation_study - formulation_study)
    (study_evidence_ready ?formulation_study - formulation_study)
    (study_characterization_confirmed ?formulation_study - formulation_study)
    (regulatory_input_available ?regulatory_submission_input - regulatory_submission_input)
    (study_link_regulatory_input ?formulation_study - formulation_study ?regulatory_submission_input - regulatory_submission_input)
    (regulatory_input_acknowledged ?formulation_study - formulation_study)
    (regulatory_review_started ?formulation_study - formulation_study)
    (regulatory_review_completed ?formulation_study - formulation_study)
    (cqa_available ?critical_quality_attribute - critical_quality_attribute)
    (study_has_cqa ?formulation_study - formulation_study ?critical_quality_attribute - critical_quality_attribute)
    (resource_allocation_available ?resource_allocation_record - resource_allocation_record)
    (study_linked_resource_allocation ?formulation_study - formulation_study ?resource_allocation_record - resource_allocation_record)
    (material_property_test_available ?material_property_test - material_property_test)
    (study_has_material_test ?formulation_study - formulation_study ?material_property_test - material_property_test)
    (stress_condition_available ?stress_condition - stress_condition)
    (study_has_stress_condition ?formulation_study - formulation_study ?stress_condition - stress_condition)
    (evidence_artifact_available ?evidence_artifact - evidence_artifact)
    (candidate_linked_evidence ?api_candidate - candidate ?evidence_artifact - evidence_artifact)
    (excipient_a_ready_for_plan ?excipient_candidate_a - excipient_candidate_a)
    (excipient_b_ready_for_plan ?excipient_candidate_b - excipient_candidate_b)
    (study_finalized ?formulation_study - formulation_study)
  )
  (:action register_api_candidate
    :parameters (?api_candidate - candidate)
    :precondition
      (and
        (not
          (api_registered ?api_candidate)
        )
        (not
          (entity_compatibility_qualified ?api_candidate)
        )
      )
    :effect (api_registered ?api_candidate)
  )
  (:action associate_supplier_with_candidate
    :parameters (?api_candidate - candidate ?excipient_supplier - excipient_supplier)
    :precondition
      (and
        (api_registered ?api_candidate)
        (not
          (supplier_allocated ?api_candidate)
        )
        (supplier_available ?excipient_supplier)
      )
    :effect
      (and
        (supplier_allocated ?api_candidate)
        (assigned_supplier ?api_candidate ?excipient_supplier)
        (not
          (supplier_available ?excipient_supplier)
        )
      )
  )
  (:action allocate_analytical_method_to_candidate
    :parameters (?api_candidate - candidate ?analytical_method - analytical_method)
    :precondition
      (and
        (api_registered ?api_candidate)
        (supplier_allocated ?api_candidate)
        (method_available ?analytical_method)
      )
    :effect
      (and
        (entity_assigned_method ?api_candidate ?analytical_method)
        (not
          (method_available ?analytical_method)
        )
      )
  )
  (:action confirm_candidate_analysis_result
    :parameters (?api_candidate - candidate ?analytical_method - analytical_method)
    :precondition
      (and
        (api_registered ?api_candidate)
        (supplier_allocated ?api_candidate)
        (entity_assigned_method ?api_candidate ?analytical_method)
        (not
          (entity_analysis_result_recorded ?api_candidate)
        )
      )
    :effect (entity_analysis_result_recorded ?api_candidate)
  )
  (:action release_analytical_method_from_candidate
    :parameters (?api_candidate - candidate ?analytical_method - analytical_method)
    :precondition
      (and
        (entity_assigned_method ?api_candidate ?analytical_method)
      )
    :effect
      (and
        (method_available ?analytical_method)
        (not
          (entity_assigned_method ?api_candidate ?analytical_method)
        )
      )
  )
  (:action assign_process_step_to_candidate
    :parameters (?api_candidate - candidate ?process_step - process_step)
    :precondition
      (and
        (entity_analysis_result_recorded ?api_candidate)
        (process_step_available ?process_step)
      )
    :effect
      (and
        (assigned_process_step ?api_candidate ?process_step)
        (not
          (process_step_available ?process_step)
        )
      )
  )
  (:action release_process_step_from_candidate
    :parameters (?api_candidate - candidate ?process_step - process_step)
    :precondition
      (and
        (assigned_process_step ?api_candidate ?process_step)
      )
    :effect
      (and
        (process_step_available ?process_step)
        (not
          (assigned_process_step ?api_candidate ?process_step)
        )
      )
  )
  (:action assign_material_test_to_study
    :parameters (?formulation_study - formulation_study ?material_property_test - material_property_test)
    :precondition
      (and
        (entity_analysis_result_recorded ?formulation_study)
        (material_property_test_available ?material_property_test)
      )
    :effect
      (and
        (study_has_material_test ?formulation_study ?material_property_test)
        (not
          (material_property_test_available ?material_property_test)
        )
      )
  )
  (:action release_material_test_from_study
    :parameters (?formulation_study - formulation_study ?material_property_test - material_property_test)
    :precondition
      (and
        (study_has_material_test ?formulation_study ?material_property_test)
      )
    :effect
      (and
        (material_property_test_available ?material_property_test)
        (not
          (study_has_material_test ?formulation_study ?material_property_test)
        )
      )
  )
  (:action assign_stress_condition_to_study
    :parameters (?formulation_study - formulation_study ?stress_condition - stress_condition)
    :precondition
      (and
        (entity_analysis_result_recorded ?formulation_study)
        (stress_condition_available ?stress_condition)
      )
    :effect
      (and
        (study_has_stress_condition ?formulation_study ?stress_condition)
        (not
          (stress_condition_available ?stress_condition)
        )
      )
  )
  (:action release_stress_condition_from_study
    :parameters (?formulation_study - formulation_study ?stress_condition - stress_condition)
    :precondition
      (and
        (study_has_stress_condition ?formulation_study ?stress_condition)
      )
    :effect
      (and
        (stress_condition_available ?stress_condition)
        (not
          (study_has_stress_condition ?formulation_study ?stress_condition)
        )
      )
  )
  (:action record_flag_a_positive
    :parameters (?excipient_candidate_a - excipient_candidate_a ?compatibility_flag_a - compatibility_flag_a ?analytical_method - analytical_method)
    :precondition
      (and
        (entity_analysis_result_recorded ?excipient_candidate_a)
        (entity_assigned_method ?excipient_candidate_a ?analytical_method)
        (compatibility_flag_a_applicable ?excipient_candidate_a ?compatibility_flag_a)
        (not
          (compatibility_flag_a_positive ?compatibility_flag_a)
        )
        (not
          (compatibility_flag_a_negative ?compatibility_flag_a)
        )
      )
    :effect (compatibility_flag_a_positive ?compatibility_flag_a)
  )
  (:action record_excipient_a_positive_outcome
    :parameters (?excipient_candidate_a - excipient_candidate_a ?compatibility_flag_a - compatibility_flag_a ?process_step - process_step)
    :precondition
      (and
        (entity_analysis_result_recorded ?excipient_candidate_a)
        (assigned_process_step ?excipient_candidate_a ?process_step)
        (compatibility_flag_a_applicable ?excipient_candidate_a ?compatibility_flag_a)
        (compatibility_flag_a_positive ?compatibility_flag_a)
        (not
          (excipient_a_ready_for_plan ?excipient_candidate_a)
        )
      )
    :effect
      (and
        (excipient_a_ready_for_plan ?excipient_candidate_a)
        (excipient_a_test_confirmed ?excipient_candidate_a)
      )
  )
  (:action record_flag_a_negative
    :parameters (?excipient_candidate_a - excipient_candidate_a ?compatibility_flag_a - compatibility_flag_a ?sample_material - sample_material)
    :precondition
      (and
        (entity_analysis_result_recorded ?excipient_candidate_a)
        (compatibility_flag_a_applicable ?excipient_candidate_a ?compatibility_flag_a)
        (sample_material_available ?sample_material)
        (not
          (excipient_a_ready_for_plan ?excipient_candidate_a)
        )
      )
    :effect
      (and
        (compatibility_flag_a_negative ?compatibility_flag_a)
        (excipient_a_ready_for_plan ?excipient_candidate_a)
        (sample_assigned_to_excipient_a ?excipient_candidate_a ?sample_material)
        (not
          (sample_material_available ?sample_material)
        )
      )
  )
  (:action reconcile_flag_a_results_for_excipient_a
    :parameters (?excipient_candidate_a - excipient_candidate_a ?compatibility_flag_a - compatibility_flag_a ?analytical_method - analytical_method ?sample_material - sample_material)
    :precondition
      (and
        (entity_analysis_result_recorded ?excipient_candidate_a)
        (entity_assigned_method ?excipient_candidate_a ?analytical_method)
        (compatibility_flag_a_applicable ?excipient_candidate_a ?compatibility_flag_a)
        (compatibility_flag_a_negative ?compatibility_flag_a)
        (sample_assigned_to_excipient_a ?excipient_candidate_a ?sample_material)
        (not
          (excipient_a_test_confirmed ?excipient_candidate_a)
        )
      )
    :effect
      (and
        (compatibility_flag_a_positive ?compatibility_flag_a)
        (excipient_a_test_confirmed ?excipient_candidate_a)
        (sample_material_available ?sample_material)
        (not
          (sample_assigned_to_excipient_a ?excipient_candidate_a ?sample_material)
        )
      )

  (:action record_excipient_b_positive_outcome
    :parameters (?excipient_candidate_b - excipient_candidate_b ?compatibility_flag_b - compatibility_flag_b ?process_step - process_step)
    :precondition
      (and
        (entity_analysis_result_recorded ?excipient_candidate_b)

        (compatibility_flag_b_applicable ?excipient_candidate_b ?compatibility_flag_b)
        (compatibility_flag_b_positive ?compatibility_flag_b)

          (excipient_b_ready_for_plan ?excipient_candidate_b)

        (excipient_b_test_confirmed ?excipient_candidate_b)

    :parameters (?excipient_candidate_b - excipient_candidate_b ?compatibility_flag_b - compatibility_flag_b ?sample_material - sample_material)
    :precondition

          (excipient_b_ready_for_plan ?excipient_candidate_b)

    :effect
      (and
        (compatibility_flag_b_negative ?compatibility_flag_b)
        (excipient_b_ready_for_plan ?excipient_candidate_b)

  (:action reconcile_flag_b_results_for_excipient_b
    :parameters (?excipient_candidate_b - excipient_candidate_b ?compatibility_flag_b - compatibility_flag_b ?analytical_method - analytical_method ?sample_material - sample_material)
    :precondition
      (and
        (entity_analysis_result_recorded ?excipient_candidate_b)
        (entity_assigned_method ?excipient_candidate_b ?analytical_method)
        (compatibility_flag_b_applicable ?excipient_candidate_b ?compatibility_flag_b)
        (compatibility_flag_b_negative ?compatibility_flag_b)
        (sample_assigned_to_excipient_b ?excipient_candidate_b ?sample_material)

          (excipient_b_test_confirmed ?excipient_candidate_b)

    :effect
      (and
        (compatibility_flag_b_positive ?compatibility_flag_b)
        (excipient_b_test_confirmed ?excipient_candidate_b)
        (sample_material_available ?sample_material)

          (sample_assigned_to_excipient_b ?excipient_candidate_b ?sample_material)

  (:action assemble_stability_plan_standard
    :parameters (?excipient_candidate_a - excipient_candidate_a ?excipient_candidate_b - excipient_candidate_b ?compatibility_flag_a - compatibility_flag_a ?compatibility_flag_b - compatibility_flag_b ?stability_or_scaleup_plan - stability_or_scaleup_plan)
    :precondition
      (and
        (excipient_a_ready_for_plan ?excipient_candidate_a)
        (excipient_b_ready_for_plan ?excipient_candidate_b)
        (compatibility_flag_a_applicable ?excipient_candidate_a ?compatibility_flag_a)
        (compatibility_flag_b_applicable ?excipient_candidate_b ?compatibility_flag_b)
        (compatibility_flag_a_positive ?compatibility_flag_a)
        (compatibility_flag_b_positive ?compatibility_flag_b)
        (excipient_a_test_confirmed ?excipient_candidate_a)
        (excipient_b_test_confirmed ?excipient_candidate_b)
        (stability_plan_available ?stability_or_scaleup_plan)
      )
    :effect
      (and
        (stability_plan_committed ?stability_or_scaleup_plan)
        (plan_includes_flag_a ?stability_or_scaleup_plan ?compatibility_flag_a)
        (plan_includes_flag_b ?stability_or_scaleup_plan ?compatibility_flag_b)
        (not
          (stability_plan_available ?stability_or_scaleup_plan)
        )
      )
  )
  (:action assemble_stability_plan_with_cqa_investigation
    :parameters (?excipient_candidate_a - excipient_candidate_a ?excipient_candidate_b - excipient_candidate_b ?compatibility_flag_a - compatibility_flag_a ?compatibility_flag_b - compatibility_flag_b ?stability_or_scaleup_plan - stability_or_scaleup_plan)
    :precondition
      (and
        (excipient_a_ready_for_plan ?excipient_candidate_a)
        (excipient_b_ready_for_plan ?excipient_candidate_b)
        (compatibility_flag_a_applicable ?excipient_candidate_a ?compatibility_flag_a)
        (compatibility_flag_b_applicable ?excipient_candidate_b ?compatibility_flag_b)
        (compatibility_flag_a_negative ?compatibility_flag_a)
        (compatibility_flag_b_positive ?compatibility_flag_b)
        (not
          (excipient_a_test_confirmed ?excipient_candidate_a)
        )
        (excipient_b_test_confirmed ?excipient_candidate_b)
        (stability_plan_available ?stability_or_scaleup_plan)
      )
    :effect
      (and
        (stability_plan_committed ?stability_or_scaleup_plan)
        (plan_includes_flag_a ?stability_or_scaleup_plan ?compatibility_flag_a)
        (plan_includes_flag_b ?stability_or_scaleup_plan ?compatibility_flag_b)
        (plan_requires_cqa_investigation ?stability_or_scaleup_plan)
        (not
          (stability_plan_available ?stability_or_scaleup_plan)

  (:action assemble_stability_plan_with_material_test_requirement

    :precondition
      (and
        (excipient_a_ready_for_plan ?excipient_candidate_a)

    :parameters (?stability_or_scaleup_plan - stability_or_scaleup_plan ?excipient_candidate_a - excipient_candidate_a ?analytical_method - analytical_method)
    :precondition
      (and
        (stability_plan_committed ?stability_or_scaleup_plan)
        (excipient_a_ready_for_plan ?excipient_candidate_a)
        (entity_assigned_method ?excipient_candidate_a ?analytical_method)
        (not
          (plan_analysis_scheduled ?stability_or_scaleup_plan)
        )
      )
    :effect (plan_analysis_scheduled ?stability_or_scaleup_plan)
  )
  (:action reserve_analytical_batch_for_study
    :parameters (?formulation_study - formulation_study ?analytical_batch - analytical_batch ?stability_or_scaleup_plan - stability_or_scaleup_plan)
    :precondition
