(define (domain commuting_constraint_aware_timetable_design_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types academic_administration_object - object administrative_entity - academic_administration_object resource_category - academic_administration_object temporal_category - academic_administration_object participant_category - academic_administration_object enrollment_case - participant_category commute_option - administrative_entity meeting_time - administrative_entity academic_officer - administrative_entity curriculum_attribute - administrative_entity delivery_format - administrative_entity commute_token - administrative_entity instructor - administrative_entity safety_certificate - administrative_entity mobility_accommodation - resource_category classroom - resource_category special_permission - resource_category origin_commute_slot - temporal_category destination_commute_slot - temporal_category timetable_proposal - temporal_category enrollment_subcase - enrollment_case section_subcase - enrollment_case primary_student_profile - enrollment_subcase secondary_student_profile - enrollment_subcase course_section - section_subcase)

  (:predicates
    (enrollment_opened ?enrollment_case - enrollment_case)
    (eligibility_confirmed ?enrollment_case - enrollment_case)
    (commute_selection_recorded ?enrollment_case - enrollment_case)
    (enrollment_closed ?enrollment_case - enrollment_case)
    (administrative_clearance ?enrollment_case - enrollment_case)
    (commute_token_verified ?enrollment_case - enrollment_case)
    (commute_option_available ?commute_option - commute_option)

    (meeting_time_available ?meeting_time - meeting_time)
    (meeting_time_assigned ?enrollment_case - enrollment_case ?meeting_time - meeting_time)
    (academic_officer_available ?academic_officer - academic_officer)
    (academic_officer_assigned ?enrollment_case - enrollment_case ?academic_officer - academic_officer)
    (accommodation_available ?mobility_accommodation - mobility_accommodation)
    (primary_accommodation_assigned ?primary_student - primary_student_profile ?mobility_accommodation - mobility_accommodation)
    (secondary_accommodation_assigned ?secondary_student - secondary_student_profile ?mobility_accommodation - mobility_accommodation)
    (primary_origin_slot_linked ?primary_student - primary_student_profile ?origin_commute_slot - origin_commute_slot)
    (origin_slot_selected ?origin_commute_slot - origin_commute_slot)
    (origin_slot_accommodation_flagged ?origin_commute_slot - origin_commute_slot)
    (primary_commute_verified ?primary_student - primary_student_profile)
    (secondary_destination_slot_linked ?secondary_student - secondary_student_profile ?destination_commute_slot - destination_commute_slot)
    (destination_slot_selected ?destination_commute_slot - destination_commute_slot)
    (destination_slot_accommodation_flagged ?destination_commute_slot - destination_commute_slot)
    (secondary_commute_verified ?secondary_student - secondary_student_profile)
    (proposal_available ?timetable_proposal - timetable_proposal)
    (proposal_compiled ?timetable_proposal - timetable_proposal)
    (proposal_origin_slot_linked ?timetable_proposal - timetable_proposal ?origin_commute_slot - origin_commute_slot)
    (proposal_destination_slot_linked ?timetable_proposal - timetable_proposal ?destination_commute_slot - destination_commute_slot)
    (proposal_origin_accommodation_flag ?timetable_proposal - timetable_proposal)
    (proposal_destination_accommodation_flag ?timetable_proposal - timetable_proposal)
    (proposal_feasibility_confirmed ?timetable_proposal - timetable_proposal)
    (section_primary_student_linked ?course_section - course_section ?primary_student - primary_student_profile)
    (section_secondary_student_linked ?course_section - course_section ?secondary_student - secondary_student_profile)
    (section_proposal_linked ?course_section - course_section ?timetable_proposal - timetable_proposal)
    (classroom_available ?classroom - classroom)
    (section_classroom_assigned ?course_section - course_section ?classroom - classroom)
    (classroom_reserved ?classroom - classroom)
    (proposal_classroom_linked ?classroom - classroom ?timetable_proposal - timetable_proposal)
    (section_review_ready ?course_section - course_section)
    (section_document_reviewed ?course_section - course_section)
    (section_internal_reviewed ?course_section - course_section)
    (section_curriculum_tagged ?course_section - course_section)
    (section_schedule_validated ?course_section - course_section)
    (section_resource_ready ?course_section - course_section)
    (section_completion_verified ?course_section - course_section)
    (special_permission_available ?special_permission - special_permission)
    (section_permission_linked ?course_section - course_section ?special_permission - special_permission)
    (special_permission_attached ?course_section - course_section)
    (special_permission_validated ?course_section - course_section)
    (special_permission_confirmed ?course_section - course_section)
    (curriculum_attribute_available ?curriculum_attribute - curriculum_attribute)
    (section_curriculum_attribute_linked ?course_section - course_section ?curriculum_attribute - curriculum_attribute)
    (delivery_format_available ?delivery_format - delivery_format)
    (section_delivery_format_linked ?course_section - course_section ?delivery_format - delivery_format)
    (instructor_available ?instructor - instructor)
    (section_instructor_linked ?course_section - course_section ?instructor - instructor)
    (safety_certificate_available ?safety_certificate - safety_certificate)
    (section_safety_certificate_linked ?course_section - course_section ?safety_certificate - safety_certificate)
    (commute_token_available ?commute_token - commute_token)
    (case_commute_token_linked ?enrollment_case - enrollment_case ?commute_token - commute_token)
    (primary_commute_locked ?primary_student - primary_student_profile)
    (secondary_commute_locked ?secondary_student - secondary_student_profile)
    (section_final_approved ?course_section - course_section)
  )
  (:action open_enrollment_case
    :parameters (?enrollment_case - enrollment_case)
    :precondition
      (and
        (not
          (enrollment_opened ?enrollment_case)
        )
        (not
          (enrollment_closed ?enrollment_case)
        )
      )
    :effect (enrollment_opened ?enrollment_case)
  )
  (:action reserve_commute_option
    :parameters (?enrollment_case - enrollment_case ?commute_option - commute_option)
    :precondition
      (and
        (enrollment_opened ?enrollment_case)
        (not
          (commute_selection_recorded ?enrollment_case)
        )
        (commute_option_available ?commute_option)
      )
    :effect
      (and
        (commute_selection_recorded ?enrollment_case)
        (has_commute_option ?enrollment_case ?commute_option)
        (not
          (commute_option_available ?commute_option)
        )
      )
  )
  (:action assign_meeting_time
    :parameters (?enrollment_case - enrollment_case ?meeting_time - meeting_time)
    :precondition
      (and
        (enrollment_opened ?enrollment_case)
        (commute_selection_recorded ?enrollment_case)
        (meeting_time_available ?meeting_time)
      )
    :effect
      (and
        (meeting_time_assigned ?enrollment_case ?meeting_time)
        (not
          (meeting_time_available ?meeting_time)
        )
      )
  )
  (:action confirm_enrollment_eligibility
    :parameters (?enrollment_case - enrollment_case ?meeting_time - meeting_time)
    :precondition
      (and
        (enrollment_opened ?enrollment_case)
        (commute_selection_recorded ?enrollment_case)
        (meeting_time_assigned ?enrollment_case ?meeting_time)
        (not
          (eligibility_confirmed ?enrollment_case)
        )
      )
    :effect (eligibility_confirmed ?enrollment_case)
  )
  (:action release_meeting_time
    :parameters (?enrollment_case - enrollment_case ?meeting_time - meeting_time)
    :precondition
      (and
        (meeting_time_assigned ?enrollment_case ?meeting_time)
      )
    :effect
      (and
        (meeting_time_available ?meeting_time)
        (not
          (meeting_time_assigned ?enrollment_case ?meeting_time)
        )
      )
  )
  (:action assign_academic_officer
    :parameters (?enrollment_case - enrollment_case ?academic_officer - academic_officer)
    :precondition
      (and
        (eligibility_confirmed ?enrollment_case)
        (academic_officer_available ?academic_officer)
      )
    :effect
      (and
        (academic_officer_assigned ?enrollment_case ?academic_officer)
        (not
          (academic_officer_available ?academic_officer)
        )
      )
  )
  (:action release_academic_officer
    :parameters (?enrollment_case - enrollment_case ?academic_officer - academic_officer)
    :precondition
      (and
        (academic_officer_assigned ?enrollment_case ?academic_officer)
      )
    :effect
      (and
        (academic_officer_available ?academic_officer)
        (not
          (academic_officer_assigned ?enrollment_case ?academic_officer)
        )
      )
  )
  (:action assign_instructor
    :parameters (?course_section - course_section ?instructor - instructor)
    :precondition
      (and
        (eligibility_confirmed ?course_section)
        (instructor_available ?instructor)
      )
    :effect
      (and
        (section_instructor_linked ?course_section ?instructor)
        (not
          (instructor_available ?instructor)
        )
      )
  )
  (:action release_instructor
    :parameters (?course_section - course_section ?instructor - instructor)
    :precondition
      (and
        (section_instructor_linked ?course_section ?instructor)
      )
    :effect
      (and
        (instructor_available ?instructor)
        (not
          (section_instructor_linked ?course_section ?instructor)
        )
      )
  )
  (:action assign_safety_certificate
    :parameters (?course_section - course_section ?safety_certificate - safety_certificate)
    :precondition
      (and
        (eligibility_confirmed ?course_section)
        (safety_certificate_available ?safety_certificate)
      )
    :effect
      (and
        (section_safety_certificate_linked ?course_section ?safety_certificate)
        (not
          (safety_certificate_available ?safety_certificate)
        )
      )
  )
  (:action release_safety_certificate
    :parameters (?course_section - course_section ?safety_certificate - safety_certificate)
    :precondition
      (and
        (section_safety_certificate_linked ?course_section ?safety_certificate)
      )
    :effect
      (and
        (safety_certificate_available ?safety_certificate)
        (not
          (section_safety_certificate_linked ?course_section ?safety_certificate)
        )
      )
  )
  (:action confirm_origin_commute_slot
    :parameters (?primary_student - primary_student_profile ?origin_commute_slot - origin_commute_slot ?meeting_time - meeting_time)
    :precondition
      (and
        (eligibility_confirmed ?primary_student)
        (meeting_time_assigned ?primary_student ?meeting_time)
        (primary_origin_slot_linked ?primary_student ?origin_commute_slot)
        (not
          (origin_slot_selected ?origin_commute_slot)
        )
        (not
          (origin_slot_accommodation_flagged ?origin_commute_slot)
        )
      )
    :effect (origin_slot_selected ?origin_commute_slot)
  )
  (:action lock_primary_commute
    :parameters (?primary_student - primary_student_profile ?origin_commute_slot - origin_commute_slot ?academic_officer - academic_officer)
    :precondition
      (and
        (eligibility_confirmed ?primary_student)
        (academic_officer_assigned ?primary_student ?academic_officer)
        (primary_origin_slot_linked ?primary_student ?origin_commute_slot)
        (origin_slot_selected ?origin_commute_slot)
        (not
          (primary_commute_locked ?primary_student)
        )
      )
    :effect
      (and
        (primary_commute_locked ?primary_student)
        (primary_commute_verified ?primary_student)
      )
  )
  (:action allocate_primary_accommodation
    :parameters (?primary_student - primary_student_profile ?origin_commute_slot - origin_commute_slot ?mobility_accommodation - mobility_accommodation)
    :precondition
      (and
        (eligibility_confirmed ?primary_student)
        (primary_origin_slot_linked ?primary_student ?origin_commute_slot)
        (accommodation_available ?mobility_accommodation)
        (not
          (primary_commute_locked ?primary_student)
        )
      )
    :effect
      (and
        (origin_slot_accommodation_flagged ?origin_commute_slot)
        (primary_commute_locked ?primary_student)
        (primary_accommodation_assigned ?primary_student ?mobility_accommodation)
        (not
          (accommodation_available ?mobility_accommodation)
        )
      )
  )
  (:action finalize_primary_commute_plan
    :parameters (?primary_student - primary_student_profile ?origin_commute_slot - origin_commute_slot ?meeting_time - meeting_time ?mobility_accommodation - mobility_accommodation)
    :precondition
      (and
        (eligibility_confirmed ?primary_student)
        (meeting_time_assigned ?primary_student ?meeting_time)
        (primary_origin_slot_linked ?primary_student ?origin_commute_slot)
        (origin_slot_accommodation_flagged ?origin_commute_slot)
        (primary_accommodation_assigned ?primary_student ?mobility_accommodation)
        (not
          (primary_commute_verified ?primary_student)
        )
      )
    :effect
      (and
        (origin_slot_selected ?origin_commute_slot)
        (primary_commute_verified ?primary_student)
        (accommodation_available ?mobility_accommodation)
        (not
          (primary_accommodation_assigned ?primary_student ?mobility_accommodation)
        )
      )
  )
  (:action confirm_destination_commute_slot
    :parameters (?secondary_student - secondary_student_profile ?destination_commute_slot - destination_commute_slot ?meeting_time - meeting_time)
    :precondition
      (and
        (eligibility_confirmed ?secondary_student)
        (meeting_time_assigned ?secondary_student ?meeting_time)
        (secondary_destination_slot_linked ?secondary_student ?destination_commute_slot)
        (not
          (destination_slot_selected ?destination_commute_slot)
        )
        (not
          (destination_slot_accommodation_flagged ?destination_commute_slot)
        )
      )
    :effect (destination_slot_selected ?destination_commute_slot)
  )
  (:action lock_secondary_commute
    :parameters (?secondary_student - secondary_student_profile ?destination_commute_slot - destination_commute_slot ?academic_officer - academic_officer)
    :precondition
      (and
        (eligibility_confirmed ?secondary_student)
        (academic_officer_assigned ?secondary_student ?academic_officer)
        (secondary_destination_slot_linked ?secondary_student ?destination_commute_slot)
        (destination_slot_selected ?destination_commute_slot)
        (not
          (secondary_commute_locked ?secondary_student)
        )
      )
    :effect
      (and
        (secondary_commute_locked ?secondary_student)
        (secondary_commute_verified ?secondary_student)
      )
  )
  (:action allocate_secondary_accommodation
    :parameters (?secondary_student - secondary_student_profile ?destination_commute_slot - destination_commute_slot ?mobility_accommodation - mobility_accommodation)
    :precondition
      (and
        (eligibility_confirmed ?secondary_student)
        (secondary_destination_slot_linked ?secondary_student ?destination_commute_slot)
        (accommodation_available ?mobility_accommodation)
        (not
          (secondary_commute_locked ?secondary_student)
        )
      )
    :effect
      (and
        (destination_slot_accommodation_flagged ?destination_commute_slot)
        (secondary_commute_locked ?secondary_student)
        (secondary_accommodation_assigned ?secondary_student ?mobility_accommodation)
        (not
          (accommodation_available ?mobility_accommodation)
        )
      )
  )
  (:action finalize_secondary_commute_plan
    :parameters (?secondary_student - secondary_student_profile ?destination_commute_slot - destination_commute_slot ?meeting_time - meeting_time ?mobility_accommodation - mobility_accommodation)
    :precondition
      (and
        (eligibility_confirmed ?secondary_student)
        (meeting_time_assigned ?secondary_student ?meeting_time)
        (secondary_destination_slot_linked ?secondary_student ?destination_commute_slot)
        (destination_slot_accommodation_flagged ?destination_commute_slot)
        (secondary_accommodation_assigned ?secondary_student ?mobility_accommodation)
        (not
          (secondary_commute_verified ?secondary_student)
        )
      )
    :effect
      (and
        (destination_slot_selected ?destination_commute_slot)
        (secondary_commute_verified ?secondary_student)
        (accommodation_available ?mobility_accommodation)
        (not
          (secondary_accommodation_assigned ?secondary_student ?mobility_accommodation)
        )
      )
  )
  (:action compose_timetable_proposal
    :parameters (?primary_student - primary_student_profile ?secondary_student - secondary_student_profile ?origin_commute_slot - origin_commute_slot ?destination_commute_slot - destination_commute_slot ?timetable_proposal - timetable_proposal)
    :precondition
      (and
        (primary_commute_locked ?primary_student)
        (secondary_commute_locked ?secondary_student)
        (primary_origin_slot_linked ?primary_student ?origin_commute_slot)
        (secondary_destination_slot_linked ?secondary_student ?destination_commute_slot)
        (origin_slot_selected ?origin_commute_slot)
        (destination_slot_selected ?destination_commute_slot)
        (primary_commute_verified ?primary_student)
        (secondary_commute_verified ?secondary_student)
        (proposal_available ?timetable_proposal)
      )
    :effect
      (and
        (proposal_compiled ?timetable_proposal)
        (proposal_origin_slot_linked ?timetable_proposal ?origin_commute_slot)
        (proposal_destination_slot_linked ?timetable_proposal ?destination_commute_slot)
        (not
          (proposal_available ?timetable_proposal)
        )
      )
  )
  (:action compose_origin_accommodation_proposal
    :parameters (?primary_student - primary_student_profile ?secondary_student - secondary_student_profile ?origin_commute_slot - origin_commute_slot ?destination_commute_slot - destination_commute_slot ?timetable_proposal - timetable_proposal)
    :precondition
      (and
        (primary_commute_locked ?primary_student)
        (secondary_commute_locked ?secondary_student)
        (primary_origin_slot_linked ?primary_student ?origin_commute_slot)
        (secondary_destination_slot_linked ?secondary_student ?destination_commute_slot)
        (origin_slot_accommodation_flagged ?origin_commute_slot)
        (destination_slot_selected ?destination_commute_slot)
        (not
          (primary_commute_verified ?primary_student)
        )
        (secondary_commute_verified ?secondary_student)
        (proposal_available ?timetable_proposal)
      )
    :effect
      (and
        (proposal_compiled ?timetable_proposal)
        (proposal_origin_slot_linked ?timetable_proposal ?origin_commute_slot)
        (proposal_destination_slot_linked ?timetable_proposal ?destination_commute_slot)
        (proposal_origin_accommodation_flag ?timetable_proposal)
        (not
          (proposal_available ?timetable_proposal)
        )
      )
  )
  (:action compose_destination_accommodation_proposal
    :parameters (?primary_student - primary_student_profile ?secondary_student - secondary_student_profile ?origin_commute_slot - origin_commute_slot ?destination_commute_slot - destination_commute_slot ?timetable_proposal - timetable_proposal)
    :precondition
      (and
        (primary_commute_locked ?primary_student)
        (secondary_commute_locked ?secondary_student)
        (primary_origin_slot_linked ?primary_student ?origin_commute_slot)
        (secondary_destination_slot_linked ?secondary_student ?destination_commute_slot)
        (origin_slot_selected ?origin_commute_slot)
        (destination_slot_accommodation_flagged ?destination_commute_slot)
        (primary_commute_verified ?primary_student)
        (not
          (secondary_commute_verified ?secondary_student)
        )
        (proposal_available ?timetable_proposal)
      )
    :effect
      (and
        (proposal_compiled ?timetable_proposal)
        (proposal_origin_slot_linked ?timetable_proposal ?origin_commute_slot)
        (proposal_destination_slot_linked ?timetable_proposal ?destination_commute_slot)
        (proposal_destination_accommodation_flag ?timetable_proposal)
        (not
          (proposal_available ?timetable_proposal)
        )
      )
  )
  (:action compose_full_accommodation_proposal
    :parameters (?primary_student - primary_student_profile ?secondary_student - secondary_student_profile ?origin_commute_slot - origin_commute_slot ?destination_commute_slot - destination_commute_slot ?timetable_proposal - timetable_proposal)
    :precondition
      (and
        (primary_commute_locked ?primary_student)
        (secondary_commute_locked ?secondary_student)
        (primary_origin_slot_linked ?primary_student ?origin_commute_slot)
        (secondary_destination_slot_linked ?secondary_student ?destination_commute_slot)
        (origin_slot_accommodation_flagged ?origin_commute_slot)
        (destination_slot_accommodation_flagged ?destination_commute_slot)
        (not
          (primary_commute_verified ?primary_student)
        )
        (not
          (secondary_commute_verified ?secondary_student)
        )
        (proposal_available ?timetable_proposal)
      )
    :effect
      (and
        (proposal_compiled ?timetable_proposal)
        (proposal_origin_slot_linked ?timetable_proposal ?origin_commute_slot)
        (proposal_destination_slot_linked ?timetable_proposal ?destination_commute_slot)
        (proposal_origin_accommodation_flag ?timetable_proposal)
        (proposal_destination_accommodation_flag ?timetable_proposal)
        (not
          (proposal_available ?timetable_proposal)
        )
      )
  )
  (:action validate_timetable_feasibility
    :parameters (?timetable_proposal - timetable_proposal ?primary_student - primary_student_profile ?meeting_time - meeting_time)
    :precondition
      (and
        (proposal_compiled ?timetable_proposal)
        (primary_commute_locked ?primary_student)
        (meeting_time_assigned ?primary_student ?meeting_time)
        (not
          (proposal_feasibility_confirmed ?timetable_proposal)
        )
      )
    :effect (proposal_feasibility_confirmed ?timetable_proposal)
  )
  (:action reserve_classroom_for_proposal
    :parameters (?course_section - course_section ?classroom - classroom ?timetable_proposal - timetable_proposal)
    :precondition
      (and
        (eligibility_confirmed ?course_section)
        (section_proposal_linked ?course_section ?timetable_proposal)
        (section_classroom_assigned ?course_section ?classroom)
        (classroom_available ?classroom)
        (proposal_compiled ?timetable_proposal)
        (proposal_feasibility_confirmed ?timetable_proposal)
        (not
          (classroom_reserved ?classroom)
        )
      )
    :effect
      (and
        (classroom_reserved ?classroom)
        (proposal_classroom_linked ?classroom ?timetable_proposal)
        (not
          (classroom_available ?classroom)
        )
      )
  )
  (:action certify_section_room_assignment
    :parameters (?course_section - course_section ?classroom - classroom ?timetable_proposal - timetable_proposal ?meeting_time - meeting_time)
    :precondition
      (and
        (eligibility_confirmed ?course_section)
        (section_classroom_assigned ?course_section ?classroom)
        (classroom_reserved ?classroom)
        (proposal_classroom_linked ?classroom ?timetable_proposal)
        (meeting_time_assigned ?course_section ?meeting_time)
        (not
          (proposal_origin_accommodation_flag ?timetable_proposal)
        )
        (not
          (section_review_ready ?course_section)
        )
      )
    :effect (section_review_ready ?course_section)
  )
  (:action attach_curriculum_attribute
    :parameters (?course_section - course_section ?curriculum_attribute - curriculum_attribute)
    :precondition
      (and
        (eligibility_confirmed ?course_section)
        (curriculum_attribute_available ?curriculum_attribute)
        (not
          (section_curriculum_tagged ?course_section)
        )
      )
    :effect
      (and
        (section_curriculum_tagged ?course_section)
        (section_curriculum_attribute_linked ?course_section ?curriculum_attribute)
        (not
          (curriculum_attribute_available ?curriculum_attribute)
        )
      )
  )
  (:action confirm_curriculum_alignment
    :parameters (?course_section - course_section ?classroom - classroom ?timetable_proposal - timetable_proposal ?meeting_time - meeting_time ?curriculum_attribute - curriculum_attribute)
    :precondition
      (and
        (eligibility_confirmed ?course_section)
        (section_classroom_assigned ?course_section ?classroom)
        (classroom_reserved ?classroom)
        (proposal_classroom_linked ?classroom ?timetable_proposal)
        (meeting_time_assigned ?course_section ?meeting_time)
        (proposal_origin_accommodation_flag ?timetable_proposal)
        (section_curriculum_tagged ?course_section)
        (section_curriculum_attribute_linked ?course_section ?curriculum_attribute)
        (not
          (section_review_ready ?course_section)
        )
      )
    :effect
      (and
        (section_review_ready ?course_section)
        (section_schedule_validated ?course_section)
