  (:action create_case_for_batch
    :parameters (?product_batch_record - product_batch_record)
    :precondition
      (and
        (not
          (case_record_created ?product_batch_record)
        )
        (not
          (release_applied ?product_batch_record)
        )
      )
    :effect (case_record_created ?product_batch_record)
  )
  (:action assign_investigator_to_case
    :parameters (?product_batch_record - product_batch_record ?investigator_resource - investigator_resource)
    :precondition
      (and
        (case_record_created ?product_batch_record)
        (not
          (investigation_assigned ?product_batch_record)
        )
        (investigator_available ?investigator_resource)
      )
    :effect
      (and
        (investigation_assigned ?product_batch_record)
        (investigator_assigned ?product_batch_record ?investigator_resource)
        (not
          (investigator_available ?investigator_resource)
        )
      )
  )
  (:action schedule_laboratory_test
    :parameters (?product_batch_record - product_batch_record ?laboratory_test - laboratory_test)
    :precondition
      (and
        (case_record_created ?product_batch_record)
        (investigation_assigned ?product_batch_record)
        (laboratory_test_available ?laboratory_test)
      )
    :effect
      (and
        (test_scheduled ?product_batch_record ?laboratory_test)
        (not
          (laboratory_test_available ?laboratory_test)
        )
      )
  )
  (:action record_test_result_and_mark_case_ready
    :parameters (?product_batch_record - product_batch_record ?laboratory_test - laboratory_test)
    :precondition
      (and
        (case_record_created ?product_batch_record)
        (investigation_assigned ?product_batch_record)
        (test_scheduled ?product_batch_record ?laboratory_test)
        (not
          (ready_for_review ?product_batch_record)
        )
      )
    :effect (ready_for_review ?product_batch_record)
  )
  (:action release_laboratory_test_resource
    :parameters (?product_batch_record - product_batch_record ?laboratory_test - laboratory_test)
    :precondition
      (and
        (test_scheduled ?product_batch_record ?laboratory_test)
      )
    :effect
      (and
        (laboratory_test_available ?laboratory_test)
        (not
          (test_scheduled ?product_batch_record ?laboratory_test)
        )
      )
  )
  (:action assign_quality_representative_to_case
    :parameters (?product_batch_record - product_batch_record ?quality_representative - quality_representative)
    :precondition
      (and
        (ready_for_review ?product_batch_record)
        (qa_representative_available ?quality_representative)
      )
    :effect
      (and
        (qa_assigned_to_batch ?product_batch_record ?quality_representative)
        (not
          (qa_representative_available ?quality_representative)
        )
      )
  )
  (:action unassign_quality_representative_from_case
    :parameters (?product_batch_record - product_batch_record ?quality_representative - quality_representative)
    :precondition
      (and
        (qa_assigned_to_batch ?product_batch_record ?quality_representative)
      )
    :effect
      (and
        (qa_representative_available ?quality_representative)
        (not
          (qa_assigned_to_batch ?product_batch_record ?quality_representative)
        )
      )
  )
  (:action assign_technical_specialist_to_decision
    :parameters (?decision_record - decision_record ?technical_specialist - technical_specialist)
    :precondition
      (and
        (ready_for_review ?decision_record)
        (technical_specialist_available ?technical_specialist)
      )
    :effect
      (and
        (technical_specialist_assigned_to_decision ?decision_record ?technical_specialist)
        (not
          (technical_specialist_available ?technical_specialist)
        )
      )
  )
  (:action release_technical_specialist_from_decision
    :parameters (?decision_record - decision_record ?technical_specialist - technical_specialist)

      (and
        (technical_specialist_assigned_to_decision ?decision_record ?technical_specialist)
      )
    :effect
      (and
        (technical_specialist_available ?technical_specialist)
        (not
          (technical_specialist_assigned_to_decision ?decision_record ?technical_specialist)
        )
      )
  )
  (:action assign_executive_approver_to_decision
    :parameters (?decision_record - decision_record ?executive_approver - executive_approver)
    :precondition
      (and
        (ready_for_review ?decision_record)
        (executive_approver_available ?executive_approver)
      )
    :effect
      (and
        (executive_approval_assigned_to_decision ?decision_record ?executive_approver)
        (not
          (executive_approver_available ?executive_approver)
        )
      )
  )
  (:action release_executive_approver_from_decision
    :parameters (?decision_record - decision_record ?executive_approver - executive_approver)
    :precondition
      (and
        (executive_approval_assigned_to_decision ?decision_record ?executive_approver)
      )
    :effect
      (and
        (executive_approver_available ?executive_approver)
        (not
          (executive_approval_assigned_to_decision ?decision_record ?executive_approver)
        )
      )
  )
  (:action activate_contamination_indicator
    :parameters (?manufacturing_batch_unit - manufacturing_batch_unit ?contamination_indicator - contamination_indicator ?laboratory_test - laboratory_test)
    :precondition
      (and
        (ready_for_review ?manufacturing_batch_unit)
        (test_scheduled ?manufacturing_batch_unit ?laboratory_test)
        (contamination_indicator_tagged_on_mfg_unit ?manufacturing_batch_unit ?contamination_indicator)
        (not
          (contamination_indicator_activated ?contamination_indicator)
        )
        (not
          (contamination_indicator_mitigation_present ?contamination_indicator)
        )
      )
    :effect (contamination_indicator_activated ?contamination_indicator)
  )
  (:action confirm_mfg_unit_containment
    :parameters (?manufacturing_batch_unit - manufacturing_batch_unit ?contamination_indicator - contamination_indicator ?quality_representative - quality_representative)
    :precondition
      (and
        (ready_for_review ?manufacturing_batch_unit)
        (qa_assigned_to_batch ?manufacturing_batch_unit ?quality_representative)
        (contamination_indicator_tagged_on_mfg_unit ?manufacturing_batch_unit ?contamination_indicator)
        (contamination_indicator_activated ?contamination_indicator)
        (not
          (mfg_unit_ready_for_dossier ?manufacturing_batch_unit)
        )
      )
    :effect
      (and
        (mfg_unit_ready_for_dossier ?manufacturing_batch_unit)
        (mfg_unit_contained ?manufacturing_batch_unit)
      )
  )
  (:action attach_mitigation_to_mfg_unit
    :parameters (?manufacturing_batch_unit - manufacturing_batch_unit ?contamination_indicator - contamination_indicator ?mitigation_resource - mitigation_resource)
    :precondition
      (and
        (ready_for_review ?manufacturing_batch_unit)
        (contamination_indicator_tagged_on_mfg_unit ?manufacturing_batch_unit ?contamination_indicator)
        (mitigation_resource_available ?mitigation_resource)
        (not
          (mfg_unit_ready_for_dossier ?manufacturing_batch_unit)
        )
      )
    :effect
      (and
        (contamination_indicator_mitigation_present ?contamination_indicator)
        (mfg_unit_ready_for_dossier ?manufacturing_batch_unit)
        (mitigation_assigned_to_mfg_unit ?manufacturing_batch_unit ?mitigation_resource)
        (not
          (mitigation_resource_available ?mitigation_resource)
        )
      )
  )
  (:action apply_mitigation_and_update_mfg_unit
    :parameters (?manufacturing_batch_unit - manufacturing_batch_unit ?contamination_indicator - contamination_indicator ?laboratory_test - laboratory_test ?mitigation_resource - mitigation_resource)
    :precondition
      (and
        (ready_for_review ?manufacturing_batch_unit)
        (test_scheduled ?manufacturing_batch_unit ?laboratory_test)
        (contamination_indicator_tagged_on_mfg_unit ?manufacturing_batch_unit ?contamination_indicator)
        (contamination_indicator_mitigation_present ?contamination_indicator)
        (mitigation_assigned_to_mfg_unit ?manufacturing_batch_unit ?mitigation_resource)
        (not
          (mfg_unit_contained ?manufacturing_batch_unit)
        )
      )
    :effect
      (and
        (contamination_indicator_activated ?contamination_indicator)
        (mfg_unit_contained ?manufacturing_batch_unit)
        (mitigation_resource_available ?mitigation_resource)
        (not
          (mitigation_assigned_to_mfg_unit ?manufacturing_batch_unit ?mitigation_resource)
        )
      )
  )
  (:action mark_supply_node_indicator_active
    :parameters (?distribution_batch_unit - distribution_batch_unit ?supply_node_record - supply_node_record ?laboratory_test - laboratory_test)
    :precondition
      (and
        (ready_for_review ?distribution_batch_unit)
        (test_scheduled ?distribution_batch_unit ?laboratory_test)
        (distribution_unit_linked_to_supply_node ?distribution_batch_unit ?supply_node_record)
        (not
          (supply_node_indicator_activated ?supply_node_record)
        )
        (not
          (supply_node_mitigation_present ?supply_node_record)
        )
      )
    :effect (supply_node_indicator_activated ?supply_node_record)
  )
  (:action confirm_distribution_unit_containment
    :parameters (?distribution_batch_unit - distribution_batch_unit ?supply_node_record - supply_node_record ?quality_representative - quality_representative)
    :precondition
      (and
        (ready_for_review ?distribution_batch_unit)
        (qa_assigned_to_batch ?distribution_batch_unit ?quality_representative)
        (distribution_unit_linked_to_supply_node ?distribution_batch_unit ?supply_node_record)
        (supply_node_indicator_activated ?supply_node_record)
        (not
          (distribution_unit_ready_for_dossier ?distribution_batch_unit)
        )
      )
    :effect
      (and
        (distribution_unit_ready_for_dossier ?distribution_batch_unit)
        (distribution_unit_contained ?distribution_batch_unit)
      )
  )
  (:action attach_mitigation_to_distribution_unit
    :parameters (?distribution_batch_unit - distribution_batch_unit ?supply_node_record - supply_node_record ?mitigation_resource - mitigation_resource)
    :precondition
      (and
        (ready_for_review ?distribution_batch_unit)
        (distribution_unit_linked_to_supply_node ?distribution_batch_unit ?supply_node_record)
        (mitigation_resource_available ?mitigation_resource)
        (not
          (distribution_unit_ready_for_dossier ?distribution_batch_unit)
        )
      )
    :effect
      (and
        (supply_node_mitigation_present ?supply_node_record)
        (distribution_unit_ready_for_dossier ?distribution_batch_unit)
        (mitigation_assigned_to_distribution_unit ?distribution_batch_unit ?mitigation_resource)
        (not
          (mitigation_resource_available ?mitigation_resource)
        )
      )
  )
  (:action apply_mitigation_and_update_distribution_unit
    :parameters (?distribution_batch_unit - distribution_batch_unit ?supply_node_record - supply_node_record ?laboratory_test - laboratory_test ?mitigation_resource - mitigation_resource)
    :precondition
      (and
        (ready_for_review ?distribution_batch_unit)
        (test_scheduled ?distribution_batch_unit ?laboratory_test)
        (distribution_unit_linked_to_supply_node ?distribution_batch_unit ?supply_node_record)
        (supply_node_mitigation_present ?supply_node_record)
        (mitigation_assigned_to_distribution_unit ?distribution_batch_unit ?mitigation_resource)
        (not
          (distribution_unit_contained ?distribution_batch_unit)
        )
      )
    :effect
      (and
        (supply_node_indicator_activated ?supply_node_record)
        (distribution_unit_contained ?distribution_batch_unit)
        (mitigation_resource_available ?mitigation_resource)
        (not
          (mitigation_assigned_to_distribution_unit ?distribution_batch_unit ?mitigation_resource)
        )
      )
  )
  (:action create_case_dossier_for_evidence
    :parameters (?manufacturing_batch_unit - manufacturing_batch_unit ?distribution_batch_unit - distribution_batch_unit ?contamination_indicator - contamination_indicator ?supply_node_record - supply_node_record ?case_dossier - case_dossier)
    :precondition
      (and
        (mfg_unit_ready_for_dossier ?manufacturing_batch_unit)
        (distribution_unit_ready_for_dossier ?distribution_batch_unit)
        (contamination_indicator_tagged_on_mfg_unit ?manufacturing_batch_unit ?contamination_indicator)
        (distribution_unit_linked_to_supply_node ?distribution_batch_unit ?supply_node_record)
        (contamination_indicator_activated ?contamination_indicator)
        (supply_node_indicator_activated ?supply_node_record)
        (mfg_unit_contained ?manufacturing_batch_unit)
        (distribution_unit_contained ?distribution_batch_unit)
        (dossier_available ?case_dossier)
      )
    :effect
      (and
        (dossier_created ?case_dossier)
        (dossier_contains_contamination_indicator ?case_dossier ?contamination_indicator)
        (dossier_contains_supply_node ?case_dossier ?supply_node_record)
        (not
          (dossier_available ?case_dossier)
        )
      )
  )
  (:action create_case_dossier_with_mitigation
    :parameters (?manufacturing_batch_unit - manufacturing_batch_unit ?distribution_batch_unit - distribution_batch_unit ?contamination_indicator - contamination_indicator ?supply_node_record - supply_node_record ?case_dossier - case_dossier)
    :precondition
      (and
        (mfg_unit_ready_for_dossier ?manufacturing_batch_unit)
        (distribution_unit_ready_for_dossier ?distribution_batch_unit)
        (contamination_indicator_tagged_on_mfg_unit ?manufacturing_batch_unit ?contamination_indicator)

        (contamination_indicator_mitigation_present ?contamination_indicator)

        (not
          (mfg_unit_contained ?manufacturing_batch_unit)
        )
        (distribution_unit_contained ?distribution_batch_unit)
        (dossier_available ?case_dossier)
      )
    :effect
      (and
        (dossier_created ?case_dossier)
        (dossier_contains_contamination_indicator ?case_dossier ?contamination_indicator)
        (dossier_contains_supply_node ?case_dossier ?supply_node_record)
        (dossier_contains_manufacturing_evidence ?case_dossier)
        (not
          (dossier_available ?case_dossier)
        )
      )
  )
  (:action create_case_dossier_with_supply_evidence
    :parameters (?manufacturing_batch_unit - manufacturing_batch_unit ?distribution_batch_unit - distribution_batch_unit ?contamination_indicator - contamination_indicator ?supply_node_record - supply_node_record ?case_dossier - case_dossier)
    :precondition
      (and
        (mfg_unit_ready_for_dossier ?manufacturing_batch_unit)
        (distribution_unit_ready_for_dossier ?distribution_batch_unit)

        (not
          (distribution_unit_contained ?distribution_batch_unit)

        (dossier_available ?case_dossier)

    :effect
      (and
        (dossier_created ?case_dossier)

        (dossier_contains_supply_node ?case_dossier ?supply_node_record)

    :precondition
      (and
        (mfg_unit_ready_for_dossier ?manufacturing_batch_unit)

        (not
          (distribution_unit_contained ?distribution_batch_unit)

        (dossier_contains_supply_node ?case_dossier ?supply_node_record)

        (dossier_contains_distribution_evidence ?case_dossier)
        (not
          (dossier_available ?case_dossier)
        )
      )
  )
  (:action validate_case_dossier
    :parameters (?case_dossier - case_dossier ?manufacturing_batch_unit - manufacturing_batch_unit ?laboratory_test - laboratory_test)
    :precondition
      (and
        (dossier_created ?case_dossier)
        (mfg_unit_ready_for_dossier ?manufacturing_batch_unit)
        (test_scheduled ?manufacturing_batch_unit ?laboratory_test)
        (not
          (dossier_validated ?case_dossier)
        )
      )
    :effect (dossier_validated ?case_dossier)
  )
  (:action accept_sample_into_dossier
    :parameters (?decision_record - decision_record ?sample_record - sample_record ?case_dossier - case_dossier)
    :precondition
      (and
        (ready_for_review ?decision_record)
        (decision_links_to_dossier ?decision_record ?case_dossier)
        (decision_links_to_sample ?decision_record ?sample_record)
        (sample_record_available ?sample_record)
        (dossier_created ?case_dossier)
        (dossier_validated ?case_dossier)
        (not
          (sample_processed ?sample_record)
        )
      )
    :effect
      (and
        (sample_processed ?sample_record)
        (sample_linked_to_dossier ?sample_record ?case_dossier)
        (not
          (sample_record_available ?sample_record)
        )
      )
  )
  (:action process_sample_and_mark_technical_review_ready
    :parameters (?decision_record - decision_record ?sample_record - sample_record ?case_dossier - case_dossier ?laboratory_test - laboratory_test)
    :precondition
      (and
        (ready_for_review ?decision_record)
        (decision_links_to_sample ?decision_record ?sample_record)
        (sample_processed ?sample_record)
        (sample_linked_to_dossier ?sample_record ?case_dossier)
        (test_scheduled ?decision_record ?laboratory_test)
        (not
          (dossier_contains_manufacturing_evidence ?case_dossier)
