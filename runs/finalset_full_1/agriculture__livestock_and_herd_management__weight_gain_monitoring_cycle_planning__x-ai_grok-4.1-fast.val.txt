  (:action enroll_animal
    :parameters (?animal - animal)
    :precondition
      (and
        (not
          (animal_enrolled ?animal)
        )
        (not
          (entity_processing_finalized ?animal)
        )
      )
    :effect (animal_enrolled ?animal)
  )
  (:action allocate_handling_equipment_to_animal
    :parameters (?animal - animal ?handling_equipment - handling_equipment)
    :precondition
      (and
        (animal_enrolled ?animal)
        (not
          (animal_equipment_assigned ?animal)
        )
        (equipment_available ?handling_equipment)
      )
    :effect
      (and
        (animal_equipment_assigned ?animal)
        (equipment_assigned_to_entity ?animal ?handling_equipment)
        (not
          (equipment_available ?handling_equipment)
        )
      )
  )
  (:action assign_technician_to_animal
    :parameters (?animal - animal ?technician - technician)
    :precondition
      (and
        (animal_enrolled ?animal)
        (animal_equipment_assigned ?animal)
        (technician_available ?technician)
      )
    :effect
      (and
        (technician_assigned_to_entity ?animal ?technician)
        (not
          (technician_available ?technician)
        )
      )
  )
  (:action mark_animal_ready_for_weighing
    :parameters (?animal - animal ?technician - technician)
    :precondition
      (and
        (animal_enrolled ?animal)
        (animal_equipment_assigned ?animal)
        (technician_assigned_to_entity ?animal ?technician)
        (not
          (entity_ready_for_weighing ?animal)
        )
      )
    :effect (entity_ready_for_weighing ?animal)
  )
  (:action release_technician
    :parameters (?animal - animal ?technician - technician)
    :precondition
      (and
        (technician_assigned_to_entity ?animal ?technician)
      )
    :effect
      (and
        (technician_available ?technician)
        (not
          (technician_assigned_to_entity ?animal ?technician)
        )
      )
  )
  (:action assign_stockperson_to_animal
    :parameters (?animal - animal ?stockperson - stockperson)
    :precondition
      (and
        (entity_ready_for_weighing ?animal)
        (stockperson_available ?stockperson)
      )
    :effect
      (and
        (stockperson_assigned_to_entity ?animal ?stockperson)
        (not
          (stockperson_available ?stockperson)
        )
      )
  )
  (:action release_stockperson_from_animal
    :parameters (?animal - animal ?stockperson - stockperson)
    :precondition
      (and
        (stockperson_assigned_to_entity ?animal ?stockperson)
      )
    :effect

        (stockperson_available ?stockperson)
        (not
          (stockperson_assigned_to_entity ?animal ?stockperson)
        )
      )

        (lab_test_kit_assigned_to_cycle ?monitoring_cycle ?lab_test_kit)

  (:action confirm_group_preparation
    :parameters (?source_herd_group - source_herd_group ?weigh_station - weigh_station ?stockperson - stockperson)

      (and

        (stockperson_assigned_to_entity ?source_herd_group ?stockperson)
        (group_assigned_weigh_station ?source_herd_group ?weigh_station)
        (weigh_station_primary_ready ?weigh_station)
        (not
          (source_group_ready ?source_herd_group)
        )
      )
    :effect
      (and
        (source_group_ready ?source_herd_group)
        (group_prepared_for_weighing ?source_herd_group)
      )
  )
  (:action allocate_feed_ration_and_prepare_group
    :parameters (?source_herd_group - source_herd_group ?weigh_station - weigh_station ?feed_ration - feed_ration)

          (feed_ration_assigned_source_group ?source_herd_group ?feed_ration)

    :parameters (?comparison_herd_group - comparison_herd_group ?secondary_weigh_station - secondary_weigh_station ?feed_ration - feed_ration)
