  (:action register_participant
    :parameters (?integration_participant - integration_participant)
    :precondition
      (and
        (not
          (participant_registered ?integration_participant)
        )
        (not
          (integration_activated ?integration_participant)
        )
      )
    :effect (participant_registered ?integration_participant)
  )
  (:action bind_participant_to_endpoint
    :parameters (?integration_participant - integration_participant ?search_endpoint - search_endpoint)
    :precondition
      (and
        (participant_registered ?integration_participant)
        (not
          (participant_endpoint_configured ?integration_participant)
        )
        (endpoint_available ?search_endpoint)
      )
    :effect
      (and
        (participant_endpoint_configured ?integration_participant)
        (participant_endpoint_binding ?integration_participant ?search_endpoint)
        (not
          (endpoint_available ?search_endpoint)
        )
      )
  )
  (:action attach_schema_to_participant
    :parameters (?integration_participant - integration_participant ?schema_definition - schema_definition)
    :precondition
      (and
        (participant_registered ?integration_participant)

    :effect
      (and
        (participant_schema_binding ?integration_participant ?schema_definition)
        (not
          (schema_available ?schema_definition)
        )
      )
  )

    :parameters (?integration_participant - integration_participant ?schema_definition - schema_definition)

          (participant_ready ?integration_participant)

  )
  (:action detach_schema_from_participant
    :parameters (?integration_participant - integration_participant ?schema_definition - schema_definition)

      (and
        (participant_schema_binding ?integration_participant ?schema_definition)
      )
    :effect

        (schema_available ?schema_definition)
        (not
          (participant_schema_binding ?integration_participant ?schema_definition)
        )
      )
  )

    :parameters (?integration_participant - integration_participant ?operator_credential - operator_credential)

        (participant_ready ?integration_participant)

        (participant_credential_binding ?integration_participant ?operator_credential)

          (credential_available ?operator_credential)

        (pipeline_compute_attached ?pipeline_instance ?compute_resource)

          (pipeline_build_attached ?pipeline_instance ?build_artifact)

      )

    :parameters (?indexing_worker - indexing_worker ?source_shard - source_shard ?schema_definition - schema_definition)

        (participant_ready ?indexing_worker)
        (participant_schema_binding ?indexing_worker ?schema_definition)

        (worker_assigned_source_shard ?indexing_worker ?source_shard)
        (source_shard_ready ?source_shard)

          (indexing_worker_ready ?indexing_worker)

        (indexing_worker_ready ?indexing_worker)

        (worker_assigned_source_shard ?indexing_worker ?source_shard)

          (indexing_worker_ready ?indexing_worker)
        )
      )
    :effect
      (and
        (source_shard_reserved ?source_shard)
        (indexing_worker_ready ?indexing_worker)
        (indexing_worker_batch_assignment ?indexing_worker ?document_batch)
        (not
          (batch_available ?document_batch)
        )
      )
  )
  (:action finalize_batch_assignment_for_indexer
    :parameters (?indexing_worker - indexing_worker ?source_shard - source_shard ?schema_definition - schema_definition ?document_batch - document_batch)

          (indexing_worker_shard_claimed ?indexing_worker)

          (query_worker_ready ?query_worker)
        )
      )
    :effect
      (and
        (target_shard_reserved ?target_shard)
        (query_worker_ready ?query_worker)
        (query_worker_batch_assignment ?query_worker ?document_batch)
        (not
          (batch_available ?document_batch)
        )

  )
  (:action finalize_batch_assignment_for_query_worker
    :parameters (?query_worker - query_worker ?target_shard - target_shard ?schema_definition - schema_definition ?document_batch - document_batch)
    :precondition
      (and
        (participant_ready ?query_worker)

        (worker_assigned_target_shard ?query_worker ?target_shard)

  (:action provision_external_index
    :parameters (?indexing_worker - indexing_worker ?query_worker - query_worker ?source_shard - source_shard ?target_shard - target_shard ?external_index - external_index)
    :precondition
      (and
        (indexing_worker_ready ?indexing_worker)
        (query_worker_ready ?query_worker)
        (worker_assigned_source_shard ?indexing_worker ?source_shard)
        (worker_assigned_target_shard ?query_worker ?target_shard)
        (source_shard_ready ?source_shard)
        (target_shard_ready ?target_shard)
        (indexing_worker_shard_claimed ?indexing_worker)
        (query_worker_shard_claimed ?query_worker)
        (index_needs_provisioning ?external_index)
      )
    :effect
      (and
        (index_provisioned ?external_index)
        (index_assigned_source_shard ?external_index ?source_shard)
        (index_assigned_target_shard ?external_index ?target_shard)
        (not
          (index_needs_provisioning ?external_index)
        )
      )
  )
  (:action provision_external_index_with_source_allocation
    :parameters (?indexing_worker - indexing_worker ?query_worker - query_worker ?source_shard - source_shard ?target_shard - target_shard ?external_index - external_index)
    :precondition
      (and
        (indexing_worker_ready ?indexing_worker)
        (query_worker_ready ?query_worker)
        (worker_assigned_source_shard ?indexing_worker ?source_shard)
        (worker_assigned_target_shard ?query_worker ?target_shard)
        (source_shard_reserved ?source_shard)
        (target_shard_ready ?target_shard)

          (indexing_worker_shard_claimed ?indexing_worker)
        )
        (query_worker_shard_claimed ?query_worker)
        (index_needs_provisioning ?external_index)
      )
    :effect
      (and
        (index_provisioned ?external_index)
        (index_assigned_source_shard ?external_index ?source_shard)
        (index_assigned_target_shard ?external_index ?target_shard)
        (index_allocation_flag_a ?external_index)
        (not
          (index_needs_provisioning ?external_index)
        )
      )
  )
  (:action provision_external_index_with_target_allocation
    :parameters (?indexing_worker - indexing_worker ?query_worker - query_worker ?source_shard - source_shard ?target_shard - target_shard ?external_index - external_index)
    :precondition
      (and
        (indexing_worker_ready ?indexing_worker)
        (query_worker_ready ?query_worker)
        (worker_assigned_source_shard ?indexing_worker ?source_shard)

        (target_shard_reserved ?target_shard)

          (query_worker_shard_claimed ?query_worker)

        (index_needs_provisioning ?external_index)
      )
    :effect
      (and
        (index_provisioned ?external_index)
        (index_assigned_source_shard ?external_index ?source_shard)
        (index_assigned_target_shard ?external_index ?target_shard)
        (index_allocation_flag_b ?external_index)
        (not
          (index_needs_provisioning ?external_index)
        )
      )

    :precondition
      (and
        (indexing_worker_ready ?indexing_worker)

          (index_needs_provisioning ?external_index)

          (document_available ?document)

  )

        (participant_schema_binding ?pipeline_instance ?schema_definition)

          (index_allocation_flag_a ?external_index)
        )
        (not
          (pipeline_stage_ready ?pipeline_instance)
        )
      )
    :effect (pipeline_stage_ready ?pipeline_instance)
  )

      (and

        (pipeline_has_document ?pipeline_instance ?document)

        (participant_schema_binding ?pipeline_instance ?schema_definition)
        (index_allocation_flag_a ?external_index)
        (pipeline_plugin_staged ?pipeline_instance)
        (pipeline_plugin_attached ?pipeline_instance ?index_plugin)
        (not
          (pipeline_stage_ready ?pipeline_instance)

      (and
        (pipeline_stage_ready ?pipeline_instance)
        (pipeline_plugin_promoted ?pipeline_instance)

    :parameters (?pipeline_instance - pipeline_instance ?compute_resource - compute_resource ?operator_credential - operator_credential ?document - document ?external_index - external_index)
    :precondition
      (and
        (pipeline_stage_ready ?pipeline_instance)
        (pipeline_compute_attached ?pipeline_instance ?compute_resource)
        (participant_credential_binding ?pipeline_instance ?operator_credential)
        (pipeline_has_document ?pipeline_instance ?document)
        (document_index_assignment ?document ?external_index)
        (not
          (index_allocation_flag_b ?external_index)
        )
        (not
          (pipeline_compute_allocated ?pipeline_instance)
        )
      )
    :effect (pipeline_compute_allocated ?pipeline_instance)
  )
  (:action resume_pipeline_compute_allocation
    :parameters (?pipeline_instance - pipeline_instance ?compute_resource - compute_resource ?operator_credential - operator_credential ?document - document ?external_index - external_index)
    :precondition
      (and
        (pipeline_stage_ready ?pipeline_instance)
        (pipeline_compute_attached ?pipeline_instance ?compute_resource)
        (participant_credential_binding ?pipeline_instance ?operator_credential)
        (pipeline_has_document ?pipeline_instance ?document)
        (document_index_assignment ?document ?external_index)
        (index_allocation_flag_b ?external_index)
        (not
          (pipeline_compute_allocated ?pipeline_instance)
        )
      )
    :effect (pipeline_compute_allocated ?pipeline_instance)
  )
  (:action attach_build_artifact_and_prepare_pipeline
    :parameters (?pipeline_instance - pipeline_instance ?build_artifact - build_artifact ?document - document ?external_index - external_index)
    :precondition
      (and
