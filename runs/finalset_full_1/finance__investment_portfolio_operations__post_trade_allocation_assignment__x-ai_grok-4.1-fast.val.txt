--- domain.pddl (as passed to VAL) ---
(define (domain post_trade_allocation_assignment_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types allocation_component - object allocation_attribute - object business_artifact - object allocatable_root - object allocatable_entity - allocatable_root allocation_method - allocation_component security - allocation_component allocator - allocation_component approval_policy - allocation_component execution_profile - allocation_component allocation_profile - allocation_component compliance_check - allocation_component custody_instruction - allocation_component allocation_preference - allocation_attribute lot - allocation_attribute reason_code - allocation_attribute account_rule - business_artifact model_rule - business_artifact allocation_ticket - business_artifact account_type - allocatable_entity model_type - allocatable_entity client_account - account_type model_portfolio - account_type trade_instruction - model_type)
  (:predicates
    (allocation_candidate_registered ?allocatable_entity - allocatable_entity)
    (allocation_initialized ?allocatable_entity - allocatable_entity)
    (allocation_method_attached ?allocatable_entity - allocatable_entity)
    (allocation_committed ?allocatable_entity - allocatable_entity)
    (ready_for_assignment ?allocatable_entity - allocatable_entity)
    (allocation_profile_attached ?allocatable_entity - allocatable_entity)
    (allocation_method_available ?allocation_method - allocation_method)
    (allocation_uses_method ?allocatable_entity - allocatable_entity ?allocation_method - allocation_method)
    (security_available ?security - security)
    (entity_has_security ?allocatable_entity - allocatable_entity ?security - security)
    (allocator_available ?allocator - allocator)
    (entity_has_allocator ?allocatable_entity - allocatable_entity ?allocator - allocator)
    (allocation_preference_available ?allocation_preference - allocation_preference)
    (account_has_preference ?client_account - client_account ?allocation_preference - allocation_preference)
    (model_has_preference ?model_portfolio - model_portfolio ?allocation_preference - allocation_preference)
    (account_rule_applicable ?client_account - client_account ?account_rule - account_rule)
    (account_rule_flagged ?account_rule - account_rule)
    (account_rule_preference_applied ?account_rule - account_rule)
    (account_rule_confirmed ?client_account - client_account)
    (model_rule_applicable ?model_portfolio - model_portfolio ?model_rule - model_rule)
    (model_rule_flagged ?model_rule - model_rule)
    (model_rule_preference_applied ?model_rule - model_rule)
    (model_rule_confirmed ?model_portfolio - model_portfolio)
    (ticket_unpopulated ?allocation_ticket - allocation_ticket)
    (ticket_populated ?allocation_ticket - allocation_ticket)
    (ticket_linked_account_rule ?allocation_ticket - allocation_ticket ?account_rule - account_rule)
    (ticket_linked_model_rule ?allocation_ticket - allocation_ticket ?model_rule - model_rule)
    (ticket_account_pref_flag ?allocation_ticket - allocation_ticket)
    (ticket_model_pref_flag ?allocation_ticket - allocation_ticket)
    (ticket_ready_for_lot_assignment ?allocation_ticket - allocation_ticket)
    (instruction_targets_account ?trade_instruction - trade_instruction ?client_account - client_account)
    (instruction_targets_model ?trade_instruction - trade_instruction ?model_portfolio - model_portfolio)
    (instruction_linked_ticket ?trade_instruction - trade_instruction ?allocation_ticket - allocation_ticket)
    (lot_available ?lot - lot)
    (instruction_has_lot ?trade_instruction - trade_instruction ?lot - lot)
    (lot_locked ?lot - lot)
    (lot_assigned_to_ticket ?lot - lot ?allocation_ticket - allocation_ticket)
    (instruction_lot_match_confirmed ?trade_instruction - trade_instruction)
    (instruction_ready_for_validation ?trade_instruction - trade_instruction)
    (instruction_validation_performed ?trade_instruction - trade_instruction)
    (approval_policy_attached ?trade_instruction - trade_instruction)
    (approval_policy_validated ?trade_instruction - trade_instruction)
    (compliance_checks_passed ?trade_instruction - trade_instruction)
    (final_checks_completed ?trade_instruction - trade_instruction)
    (reason_code_available ?reason_code - reason_code)
    (instruction_has_reason_code ?trade_instruction - trade_instruction ?reason_code - reason_code)
    (instruction_reason_code_applied ?trade_instruction - trade_instruction)
    (approval_requested ?trade_instruction - trade_instruction)
    (instruction_approved ?trade_instruction - trade_instruction)
    (approval_policy_available ?approval_policy - approval_policy)
    (instruction_linked_approval_policy ?trade_instruction - trade_instruction ?approval_policy - approval_policy)
    (execution_profile_available ?execution_profile - execution_profile)
    (instruction_linked_execution_profile ?trade_instruction - trade_instruction ?execution_profile - execution_profile)
    (compliance_check_available ?compliance_check - compliance_check)
    (instruction_has_compliance_check ?trade_instruction - trade_instruction ?compliance_check - compliance_check)
    (custody_instruction_available ?custody_instruction - custody_instruction)
    (instruction_has_custody_instruction ?trade_instruction - trade_instruction ?custody_instruction - custody_instruction)
    (allocation_profile_available ?allocation_profile - allocation_profile)
    (entity_has_allocation_profile ?allocatable_entity - allocatable_entity ?allocation_profile - allocation_profile)
    (account_eligibility_flag ?client_account - client_account)
    (model_eligibility_flag ?model_portfolio - model_portfolio)
    (instruction_finalized ?trade_instruction - trade_instruction)
  )
  (:action create_allocation_candidate
    :parameters (?allocatable_entity - allocatable_entity)
    :precondition
      (and
        (not
          (allocation_candidate_registered ?allocatable_entity)
        )
        (not
          (allocation_committed ?allocatable_entity)
        )
      )
    :effect (allocation_candidate_registered ?allocatable_entity)
  )
  (:action attach_allocation_method
    :parameters (?allocatable_entity - allocatable_entity ?allocation_method - allocation_method)
    :precondition
      (and
        (allocation_candidate_registered ?allocatable_entity)
        (not
          (allocation_method_attached ?allocatable_entity)
        )
        (allocation_method_available ?allocation_method)
      )
    :effect
      (and
        (allocation_method_attached ?allocatable_entity)
        (allocation_uses_method ?allocatable_entity ?allocation_method)
        (not
          (allocation_method_available ?allocation_method)
        )
      )
  )
  (:action link_security_to_allocatable
    :parameters (?allocatable_entity - allocatable_entity ?security - security)
    :precondition
      (and
        (allocation_candidate_registered ?allocatable_entity)
        (allocation_method_attached ?allocatable_entity)
        (security_available ?security)
      )
    :effect
      (and
        (entity_has_security ?allocatable_entity ?security)
        (not
          (security_available ?security)
        )
      )
  )
  (:action initialize_allocation_candidate
    :parameters (?allocatable_entity - allocatable_entity ?security - security)
    :precondition
      (and
        (allocation_candidate_registered ?allocatable_entity)
        (allocation_method_attached ?allocatable_entity)
        (entity_has_security ?allocatable_entity ?security)
        (not
          (allocation_initialized ?allocatable_entity)
        )
      )
    :effect (allocation_initialized ?allocatable_entity)
  )
  (:action release_security_from_allocatable
    :parameters (?allocatable_entity - allocatable_entity ?security - security)
    :precondition
      (and
        (entity_has_security ?allocatable_entity ?security)
      )
    :effect
      (and
        (security_available ?security)
        (not
          (entity_has_security ?allocatable_entity ?security)
        )
      )
  )
  (:action assign_allocator
    :parameters (?allocatable_entity - allocatable_entity ?allocator - allocator)
    :precondition
      (and
        (allocation_initialized ?allocatable_entity)
        (allocator_available ?allocator)
      )
    :effect
      (and
        (entity_has_allocator ?allocatable_entity ?allocator)
        (not
          (allocator_available ?allocator)
        )
      )
  )
  (:action release_allocator
    :parameters (?allocatable_entity - allocatable_entity ?allocator - allocator)
    :precondition
      (and
        (entity_has_allocator ?allocatable_entity ?allocator)
      )
    :effect
      (and
        (allocator_available ?allocator)
        (not
          (entity_has_allocator ?allocatable_entity ?allocator)
        )
      )
  )
  (:action attach_compliance_check_to_instruction
    :parameters (?trade_instruction - trade_instruction ?compliance_check - compliance_check)
    :precondition
      (and
        (allocation_initialized ?trade_instruction)
        (compliance_check_available ?compliance_check)
      )
    :effect
      (and
        (instruction_has_compliance_check ?trade_instruction ?compliance_check)
        (not
          (compliance_check_available ?compliance_check)
        )
      )
  )
  (:action detach_compliance_check_from_instruction
    :parameters (?trade_instruction - trade_instruction ?compliance_check - compliance_check)
    :precondition
      (and
        (instruction_has_compliance_check ?trade_instruction ?compliance_check)
      )
    :effect
      (and
        (compliance_check_available ?compliance_check)
        (not
          (instruction_has_compliance_check ?trade_instruction ?compliance_check)
        )
      )
  )
  (:action attach_custody_instruction_to_instruction
    :parameters (?trade_instruction - trade_instruction ?custody_instruction - custody_instruction)
    :precondition
      (and
        (allocation_initialized ?trade_instruction)
        (custody_instruction_available ?custody_instruction)
      )
    :effect
      (and
        (instruction_has_custody_instruction ?trade_instruction ?custody_instruction)
        (not
          (custody_instruction_available ?custody_instruction)
        )
      )
  )
  (:action detach_custody_instruction_from_instruction
    :parameters (?trade_instruction - trade_instruction ?custody_instruction - custody_instruction)
    :precondition
      (and
        (instruction_has_custody_instruction ?trade_instruction ?custody_instruction)
      )
    :effect
      (and
        (custody_instruction_available ?custody_instruction)
        (not
          (instruction_has_custody_instruction ?trade_instruction ?custody_instruction)
        )
      )
  )
  (:action evaluate_account_rule_for_security
    :parameters (?client_account - client_account ?account_rule - account_rule ?security - security)
    :precondition
      (and
        (allocation_initialized ?client_account)
        (entity_has_security ?client_account ?security)
        (account_rule_applicable ?client_account ?account_rule)
        (not
          (account_rule_flagged ?account_rule)
        )
        (not
          (account_rule_preference_applied ?account_rule)
        )
      )
    :effect (account_rule_flagged ?account_rule)
  )
  (:action confirm_account_eligibility_with_allocator
    :parameters (?client_account - client_account ?account_rule - account_rule ?allocator - allocator)
    :precondition
      (and
        (allocation_initialized ?client_account)
        (entity_has_allocator ?client_account ?allocator)
        (account_rule_applicable ?client_account ?account_rule)
        (account_rule_flagged ?account_rule)
        (not
          (account_eligibility_flag ?client_account)
        )
      )
    :effect
      (and
        (account_eligibility_flag ?client_account)
        (account_rule_confirmed ?client_account)
      )
  )
  (:action apply_allocation_preference_to_account
    :parameters (?client_account - client_account ?account_rule - account_rule ?allocation_preference - allocation_preference)
    :precondition
      (and
        (allocation_initialized ?client_account)
        (account_rule_applicable ?client_account ?account_rule)
        (allocation_preference_available ?allocation_preference)
        (not
          (account_eligibility_flag ?client_account)
        )
      )
    :effect
      (and
        (account_rule_preference_applied ?account_rule)
        (account_eligibility_flag ?client_account)
        (account_has_preference ?client_account ?allocation_preference)
        (not
          (allocation_preference_available ?allocation_preference)
        )
      )
  )
  (:action finalize_account_preference_assignment
    :parameters (?client_account - client_account ?account_rule - account_rule ?security - security ?allocation_preference - allocation_preference)
    :precondition
      (and
        (allocation_initialized ?client_account)
        (entity_has_security ?client_account ?security)
        (account_rule_applicable ?client_account ?account_rule)
        (account_rule_preference_applied ?account_rule)
        (account_has_preference ?client_account ?allocation_preference)
        (not
          (account_rule_confirmed ?client_account)
        )
      )
    :effect
      (and
        (account_rule_flagged ?account_rule)
        (account_rule_confirmed ?client_account)
        (allocation_preference_available ?allocation_preference)
        (not
          (account_has_preference ?client_account ?allocation_preference)
        )
      )
  )
  (:action evaluate_model_rule_for_security
    :parameters (?model_portfolio - model_portfolio ?model_rule - model_rule ?security - security)
    :precondition
      (and
        (allocation_initialized ?model_portfolio)
        (entity_has_security ?model_portfolio ?security)
        (model_rule_applicable ?model_portfolio ?model_rule)
        (not
          (model_rule_flagged ?model_rule)
        )
        (not
          (model_rule_preference_applied ?model_rule)
        )
      )
    :effect (model_rule_flagged ?model_rule)
  )
  (:action confirm_model_eligibility_with_allocator
    :parameters (?model_portfolio - model_portfolio ?model_rule - model_rule ?allocator - allocator)
    :precondition
      (and
        (allocation_initialized ?model_portfolio)
        (entity_has_allocator ?model_portfolio ?allocator)
        (model_rule_applicable ?model_portfolio ?model_rule)
        (model_rule_flagged ?model_rule)
        (not
          (model_eligibility_flag ?model_portfolio)
        )
      )
    :effect
      (and
        (model_eligibility_flag ?model_portfolio)
        (model_rule_confirmed ?model_portfolio)
      )
  )
  (:action apply_allocation_preference_to_model
    :parameters (?model_portfolio - model_portfolio ?model_rule - model_rule ?allocation_preference - allocation_preference)
    :precondition
      (and
        (allocation_initialized ?model_portfolio)
        (model_rule_applicable ?model_portfolio ?model_rule)
        (allocation_preference_available ?allocation_preference)
        (not
          (model_eligibility_flag ?model_portfolio)
        )
      )
    :effect
      (and
        (model_rule_preference_applied ?model_rule)
        (model_eligibility_flag ?model_portfolio)
        (model_has_preference ?model_portfolio ?allocation_preference)
        (not
          (allocation_preference_available ?allocation_preference)
        )
      )
  )
  (:action finalize_model_preference_assignment
    :parameters (?model_portfolio - model_portfolio ?model_rule - model_rule ?security - security ?allocation_preference - allocation_preference)
    :precondition
      (and
        (allocation_initialized ?model_portfolio)
        (entity_has_security ?model_portfolio ?security)
        (model_rule_applicable ?model_portfolio ?model_rule)
        (model_rule_preference_applied ?model_rule)
        (model_has_preference ?model_portfolio ?allocation_preference)
        (not
          (model_rule_confirmed ?model_portfolio)
        )
      )
    :effect
      (and
        (model_rule_flagged ?model_rule)
        (model_rule_confirmed ?model_portfolio)
        (allocation_preference_available ?allocation_preference)
        (not
          (model_has_preference ?model_portfolio ?allocation_preference)
        )
      )
  )
  (:action build_allocation_ticket
    :parameters (?client_account - client_account ?model_portfolio - model_portfolio ?account_rule - account_rule ?model_rule - model_rule ?allocation_ticket - allocation_ticket)
    :precondition
      (and
        (account_eligibility_flag ?client_account)
        (model_eligibility_flag ?model_portfolio)
        (account_rule_applicable ?client_account ?account_rule)
        (model_rule_applicable ?model_portfolio ?model_rule)
        (account_rule_flagged ?account_rule)
        (model_rule_flagged ?model_rule)
        (account_rule_confirmed ?client_account)
        (model_rule_confirmed ?model_portfolio)
        (ticket_unpopulated ?allocation_ticket)
      )
    :effect
      (and
        (ticket_populated ?allocation_ticket)
        (ticket_linked_account_rule ?allocation_ticket ?account_rule)
        (ticket_linked_model_rule ?allocation_ticket ?model_rule)
        (not
          (ticket_unpopulated ?allocation_ticket)
        )
      )
  )
  (:action build_allocation_ticket_with_account_preference
    :parameters (?client_account - client_account ?model_portfolio - model_portfolio ?account_rule - account_rule ?model_rule - model_rule ?allocation_ticket - allocation_ticket)
    :precondition
      (and
        (account_eligibility_flag ?client_account)
        (model_eligibility_flag ?model_portfolio)
        (account_rule_applicable ?client_account ?account_rule)
        (model_rule_applicable ?model_portfolio ?model_rule)
        (account_rule_preference_applied ?account_rule)
        (model_rule_flagged ?model_rule)
        (not
          (account_rule_confirmed ?client_account)
        )
        (model_rule_confirmed ?model_portfolio)
        (ticket_unpopulated ?allocation_ticket)
      )
    :effect
      (and
        (ticket_populated ?allocation_ticket)
        (ticket_linked_account_rule ?allocation_ticket ?account_rule)
        (ticket_linked_model_rule ?allocation_ticket ?model_rule)
        (ticket_account_pref_flag ?allocation_ticket)
        (not
          (ticket_unpopulated ?allocation_ticket)
        )
      )
  )
  (:action build_allocation_ticket_with_model_preference
    :parameters (?client_account - client_account ?model_portfolio - model_portfolio ?account_rule - account_rule ?model_rule - model_rule ?allocation_ticket - allocation_ticket)
    :precondition
      (and
        (account_eligibility_flag ?client_account)
        (model_eligibility_flag ?model_portfolio)
        (account_rule_applicable ?client_account ?account_rule)
        (model_rule_applicable ?model_portfolio ?model_rule)
        (account_rule_flagged ?account_rule)
        (model_rule_preference_applied ?model_rule)
        (account_rule_confirmed ?client_account)
        (not
          (model_rule_confirmed ?model_portfolio)
        )
        (ticket_unpopulated ?allocation_ticket)
      )
    :effect
      (and
        (ticket_populated ?allocation_ticket)
        (ticket_linked_account_rule ?allocation_ticket ?account_rule)
        (ticket_linked_model_rule ?allocation_ticket ?model_rule)
        (ticket_model_pref_flag ?allocation_ticket)
        (not
          (ticket_unpopulated ?allocation_ticket)
        )
      )
  )
  (:action build_allocation_ticket_with_both_preferences
    :parameters (?client_account - client_account ?model_portfolio - model_portfolio ?account_rule - account_rule ?model_rule - model_rule ?allocation_ticket - allocation_ticket)
    :precondition
      (and
        (account_eligibility_flag ?client_account)
        (model_eligibility_flag ?model_portfolio)
        (account_rule_applicable ?client_account ?account_rule)
        (model_rule_applicable ?model_portfolio ?model_rule)
        (account_rule_preference_applied ?account_rule)
        (model_rule_preference_applied ?model_rule)
        (not
          (account_rule_confirmed ?client_account)
        )
        (not
          (model_rule_confirmed ?model_portfolio)
        )
        (ticket_unpopulated ?allocation_ticket)
      )
    :effect
      (and
        (ticket_populated ?allocation_ticket)
        (ticket_linked_account_rule ?allocation_ticket ?account_rule)
        (ticket_linked_model_rule ?allocation_ticket ?model_rule)
        (ticket_account_pref_flag ?allocation_ticket)
        (ticket_model_pref_flag ?allocation_ticket)
        (not
          (ticket_unpopulated ?allocation_ticket)
        )
      )
  )
  (:action prepare_ticket_for_lot_assignment
    :parameters (?allocation_ticket - allocation_ticket ?client_account - client_account ?security - security)
    :precondition
      (and
        (ticket_populated ?allocation_ticket)
        (account_eligibility_flag ?client_account)
        (entity_has_security ?client_account ?security)
        (not
          (ticket_ready_for_lot_assignment ?allocation_ticket)
        )
      )
    :effect (ticket_ready_for_lot_assignment ?allocation_ticket)
  )
  (:action lock_lot_and_assign_to_ticket
    :parameters (?trade_instruction - trade_instruction ?lot - lot ?allocation_ticket - allocation_ticket)
    :precondition
      (and
        (allocation_initialized ?trade_instruction)
        (instruction_linked_ticket ?trade_instruction ?allocation_ticket)
        (instruction_has_lot ?trade_instruction ?lot)
        (lot_available ?lot)
        (ticket_populated ?allocation_ticket)
        (ticket_ready_for_lot_assignment ?allocation_ticket)
        (not
          (lot_locked ?lot)
        )
      )
    :effect
      (and
        (lot_locked ?lot)
        (lot_assigned_to_ticket ?lot ?allocation_ticket)
        (not
          (lot_available ?lot)
        )
      )
  )
  (:action confirm_instruction_lot_match
    :parameters (?trade_instruction - trade_instruction ?lot - lot ?allocation_ticket - allocation_ticket ?security - security)
    :precondition
      (and
        (allocation_initialized ?trade_instruction)
        (instruction_has_lot ?trade_instruction ?lot)
        (lot_locked ?lot)
        (lot_assigned_to_ticket ?lot ?allocation_ticket)
        (entity_has_security ?trade_instruction ?security)
        (not
          (ticket_account_pref_flag ?allocation_ticket)
        )
        (not
          (instruction_lot_match_confirmed ?trade_instruction)
        )
      )
    :effect (instruction_lot_match_confirmed ?trade_instruction)
  )
  (:action assign_approval_policy_to_instruction
    :parameters (?trade_instruction - trade_instruction ?approval_policy - approval_policy)
    :precondition
      (and
        (allocation_initialized ?trade_instruction)
        (approval_policy_available ?approval_policy)
        (not
          (approval_policy_attached ?trade_instruction)
        )
      )
    :effect
      (and
        (approval_policy_attached ?trade_instruction)
        (instruction_linked_approval_policy ?trade_instruction ?approval_policy)
        (not
          (approval_policy_available ?approval_policy)
        )
      )
  )
  (:action apply_approval_policy_to_instruction
    :parameters (?trade_instruction - trade_instruction ?lot - lot ?allocation_ticket - allocation_ticket ?security - security ?approval_policy - approval_policy)
    :precondition
      (and
        (allocation_initialized ?trade_instruction)
        (instruction_has_lot ?trade_instruction ?lot)
        (lot_locked ?lot)
        (lot_assigned_to_ticket ?lot ?allocation_ticket)
        (entity_has_security ?trade_instruction ?security)
        (ticket_account_pref_flag ?allocation_ticket)
        (approval_policy_attached ?trade_instruction)
        (instruction_linked_approval_policy ?trade_instruction ?approval_policy)
        (not
          (instruction_lot_match_confirmed ?trade_instruction)
        )
      )
    :effect
      (and
        (instruction_lot_match_confirmed ?trade_instruction)
        (approval_policy_validated ?trade_instruction)
      )
  )
  (:action mark_instruction_ready_for_approval_with_compliance
    :parameters (?trade_instruction - trade_instruction ?compliance_check - compliance_check ?allocator - allocator ?lot - lot ?allocation_ticket - allocation_ticket)
    :precondition
      (and
        (instruction_lot_match_confirmed ?trade_instruction)
        (instruction_has_compliance_check ?trade_instruction ?compliance_check)
        (entity_has_allocator ?trade_instruction ?allocator)
        (instruction_has_lot ?trade_instruction ?lot)
        (lot_assigned_to_ticket ?lot ?allocation_ticket)
        (not
          (ticket_model_pref_flag ?allocation_ticket)
        )
        (not
          (instruction_ready_for_validation ?trade_instruction)
        )
      )
    :effect (instruction_ready_for_validation ?trade_instruction)
  )
  (:action mark_instruction_ready_for_approval_post_ticket
    :parameters (?trade_instruction - trade_instruction ?compliance_check - compliance_check ?allocator - allocator ?lot - lot ?allocation_ticket - allocation_ticket)
    :precondition
      (and
        (instruction_lot_match_confirmed ?trade_instruction)
        (instruction_has_compliance_check ?trade_instruction ?compliance_check)
        (entity_has_allocator ?trade_instruction ?allocator)
        (instruction_has_lot ?trade_instruction ?lot)
        (lot_assigned_to_ticket ?lot ?allocation_ticket)
        (ticket_model_pref_flag ?allocation_ticket)
        (not
          (instruction_ready_for_validation ?trade_instruction)
        )
      )
    :effect (instruction_ready_for_validation ?trade_instruction)
  )
  (:action execute_custody_and_compliance_checks
    :parameters (?trade_instruction - trade_instruction ?custody_instruction - custody_instruction ?lot - lot ?allocation_ticket - allocation_ticket)
    :precondition
      (and
        (instruction_ready_for_validation ?trade_instruction)
        (instruction_has_custody_instruction ?trade_instruction ?custody_instruction)
        (instruction_has_lot ?trade_instruction ?lot)
        (lot_assigned_to_ticket ?lot ?allocation_ticket)
        (not
          (ticket_account_pref_flag ?allocation_ticket)
        )
        (not
          (ticket_model_pref_flag ?allocation_ticket)
        )
        (not
          (instruction_validation_performed ?trade_instruction)
        )
      )
    :effect (instruction_validation_performed ?trade_instruction)
  )
  (:action execute_custody_and_compliance_checks_with_account_pref
    :parameters (?trade_instruction - trade_instruction ?custody_instruction - custody_instruction ?lot - lot ?allocation_ticket - allocation_ticket)
