  (:requirements :strips :typing :negative-preconditions)
  (:types response_asset_type - object supply_type - object infrastructure_type - object zone_type - object quarantine_zone - zone_type response_team - response_asset_type diagnostic_test_kit - response_asset_type specialized_equipment - response_asset_type regulatory_permit - response_asset_type logistics_asset - response_asset_type clearance_document - response_asset_type trained_personnel - response_asset_type external_assistance - response_asset_type consumable_supply - supply_type resource_batch - supply_type stakeholder_agency - supply_type boundary_segment - infrastructure_type access_point - infrastructure_type recovery_plan - infrastructure_type subsite_type - quarantine_zone management_unit_type - quarantine_zone infected_subsite - subsite_type exposed_subsite - subsite_type incident_management_unit - management_unit_type)
  (:predicates
    (zone_identified ?quarantine_zone - quarantine_zone)
    (control_measures_active_for_entity ?quarantine_zone - quarantine_zone)
    (team_assigned ?quarantine_zone - quarantine_zone)
    (operational_reopened ?quarantine_zone - quarantine_zone)
    (certified_for_reopening ?quarantine_zone - quarantine_zone)
    (clearance_requested ?quarantine_zone - quarantine_zone)
    (asset_available ?response_team - response_team)
    (team_assigned_to_zone ?quarantine_zone - quarantine_zone ?response_team - response_team)
    (diagnostic_kit_available ?diagnostic_test_kit - diagnostic_test_kit)
    (diagnostic_test_association ?quarantine_zone - quarantine_zone ?diagnostic_test_kit - diagnostic_test_kit)
    (equipment_available ?specialized_equipment - specialized_equipment)
    (equipment_assigned ?quarantine_zone - quarantine_zone ?specialized_equipment - specialized_equipment)
    (consumable_available ?consumable_supply - consumable_supply)
    (infected_subsite_supply_assigned ?infected_subsite - infected_subsite ?consumable_supply - consumable_supply)
    (exposed_subsite_supply_assigned ?exposed_subsite - exposed_subsite ?consumable_supply - consumable_supply)
    (infected_subsite_boundary_link ?infected_subsite - infected_subsite ?boundary_segment - boundary_segment)
    (boundary_segment_ready ?boundary_segment - boundary_segment)
    (boundary_segment_treated ?boundary_segment - boundary_segment)
    (infected_subsite_ready ?infected_subsite - infected_subsite)
    (exposed_subsite_access_link ?exposed_subsite - exposed_subsite ?access_point - access_point)
    (access_point_ready ?access_point - access_point)
    (access_point_treated ?access_point - access_point)
    (exposed_subsite_ready ?exposed_subsite - exposed_subsite)
    (recovery_plan_available ?recovery_plan - recovery_plan)
    (recovery_plan_initialized ?recovery_plan - recovery_plan)
    (recovery_plan_boundary_link ?recovery_plan - recovery_plan ?boundary_segment - boundary_segment)
    (recovery_plan_access_link ?recovery_plan - recovery_plan ?access_point - access_point)
    (recovery_plan_variant_a ?recovery_plan - recovery_plan)
    (recovery_plan_variant_b ?recovery_plan - recovery_plan)
    (recovery_plan_stage_activated ?recovery_plan - recovery_plan)
    (management_unit_assigned_to_infected_subsite ?incident_management_unit - incident_management_unit ?infected_subsite - infected_subsite)
    (management_unit_assigned_to_exposed_subsite ?incident_management_unit - incident_management_unit ?exposed_subsite - exposed_subsite)
    (management_unit_has_recovery_plan ?incident_management_unit - incident_management_unit ?recovery_plan - recovery_plan)
    (resource_batch_available ?resource_batch - resource_batch)
    (management_unit_resource_allocated ?incident_management_unit - incident_management_unit ?resource_batch - resource_batch)
    (resource_batch_deployed ?resource_batch - resource_batch)
    (resource_batch_linked_to_plan ?resource_batch - resource_batch ?recovery_plan - recovery_plan)
    (management_unit_prepared ?incident_management_unit - incident_management_unit)
    (management_unit_execution_ready ?incident_management_unit - incident_management_unit)
    (management_unit_ready_for_certification ?incident_management_unit - incident_management_unit)
    (management_unit_permit_attached ?incident_management_unit - incident_management_unit)
    (management_unit_review_stage_1_completed ?incident_management_unit - incident_management_unit)
    (management_unit_permit_verified ?incident_management_unit - incident_management_unit)
    (management_unit_final_check_completed ?incident_management_unit - incident_management_unit)
    (stakeholder_agency_available ?stakeholder_agency - stakeholder_agency)
    (management_unit_stakeholder_engaged ?incident_management_unit - incident_management_unit ?stakeholder_agency - stakeholder_agency)
    (management_unit_stakeholder_endorsed ?incident_management_unit - incident_management_unit)
    (management_unit_permit_initiated ?incident_management_unit - incident_management_unit)
    (management_unit_permit_confirmed ?incident_management_unit - incident_management_unit)
    (regulatory_permit_available ?regulatory_permit - regulatory_permit)
    (management_unit_permit_linked ?incident_management_unit - incident_management_unit ?regulatory_permit - regulatory_permit)
    (logistics_asset_available ?logistics_asset - logistics_asset)
    (management_unit_logistics_assigned ?incident_management_unit - incident_management_unit ?logistics_asset - logistics_asset)
    (trained_personnel_available ?trained_personnel - trained_personnel)
    (management_unit_personnel_assigned ?incident_management_unit - incident_management_unit ?trained_personnel - trained_personnel)
    (external_assistance_available ?external_assistance - external_assistance)
    (management_unit_external_assistance_assigned ?incident_management_unit - incident_management_unit ?external_assistance - external_assistance)
    (clearance_document_available ?clearance_document - clearance_document)
    (clearance_document_link ?quarantine_zone - quarantine_zone ?clearance_document - clearance_document)
    (infected_subsite_assessment_complete ?infected_subsite - infected_subsite)
    (exposed_subsite_assessment_complete ?exposed_subsite - exposed_subsite)
    (management_unit_certification_recorded ?incident_management_unit - incident_management_unit)
  )
  (:action identify_quarantine_zone
    :parameters (?quarantine_zone - quarantine_zone)
    :precondition
      (and
        (not
          (zone_identified ?quarantine_zone)
        )
        (not
          (operational_reopened ?quarantine_zone)
        )
      )
    :effect (zone_identified ?quarantine_zone)
  )
  (:action assign_response_team_to_quarantine_zone
    :parameters (?quarantine_zone - quarantine_zone ?response_team - response_team)
    :precondition
      (and
        (zone_identified ?quarantine_zone)
        (not
          (team_assigned ?quarantine_zone)
        )
        (asset_available ?response_team)
      )
    :effect
      (and
        (team_assigned ?quarantine_zone)
        (team_assigned_to_zone ?quarantine_zone ?response_team)
        (not
          (asset_available ?response_team)
        )
      )
  )
  (:action associate_diagnostic_test_with_zone
    :parameters (?quarantine_zone - quarantine_zone ?diagnostic_test_kit - diagnostic_test_kit)
    :precondition
      (and
        (zone_identified ?quarantine_zone)
        (team_assigned ?quarantine_zone)
        (diagnostic_kit_available ?diagnostic_test_kit)
      )
    :effect
      (and
        (diagnostic_test_association ?quarantine_zone ?diagnostic_test_kit)
        (not
          (diagnostic_kit_available ?diagnostic_test_kit)
        )
      )
  )
  (:action activate_control_measures
    :parameters (?quarantine_zone - quarantine_zone ?diagnostic_test_kit - diagnostic_test_kit)
    :precondition
      (and
        (zone_identified ?quarantine_zone)
        (team_assigned ?quarantine_zone)
        (diagnostic_test_association ?quarantine_zone ?diagnostic_test_kit)
        (not
          (control_measures_active_for_entity ?quarantine_zone)
        )
      )
    :effect (control_measures_active_for_entity ?quarantine_zone)
  )
  (:action release_diagnostic_test_kit_from_zone
    :parameters (?quarantine_zone - quarantine_zone ?diagnostic_test_kit - diagnostic_test_kit)
    :precondition
      (and
        (diagnostic_test_association ?quarantine_zone ?diagnostic_test_kit)
      )
    :effect
      (and
        (diagnostic_kit_available ?diagnostic_test_kit)
        (not
          (diagnostic_test_association ?quarantine_zone ?diagnostic_test_kit)
        )
      )
  )
  (:action assign_equipment_to_quarantine_zone
    :parameters (?quarantine_zone - quarantine_zone ?specialized_equipment - specialized_equipment)
    :precondition
      (and
        (control_measures_active_for_entity ?quarantine_zone)
        (equipment_available ?specialized_equipment)
      )
    :effect
      (and
        (equipment_assigned ?quarantine_zone ?specialized_equipment)
        (not
          (equipment_available ?specialized_equipment)
        )
      )
  )
  (:action release_equipment_from_quarantine_zone
    :parameters (?quarantine_zone - quarantine_zone ?specialized_equipment - specialized_equipment)
    :precondition
      (and
        (equipment_assigned ?quarantine_zone ?specialized_equipment)
      )
    :effect
      (and
        (equipment_available ?specialized_equipment)
        (not
          (equipment_assigned ?quarantine_zone ?specialized_equipment)
        )
      )
  )
  (:action assign_trained_personnel_to_management_unit
    :parameters (?incident_management_unit - incident_management_unit ?trained_personnel - trained_personnel)
    :precondition
      (and
        (control_measures_active_for_entity ?incident_management_unit)
        (trained_personnel_available ?trained_personnel)
      )
    :effect
      (and
        (management_unit_personnel_assigned ?incident_management_unit ?trained_personnel)
        (not
          (trained_personnel_available ?trained_personnel)
        )
      )
  )
  (:action release_trained_personnel_from_management_unit
    :parameters (?incident_management_unit - incident_management_unit ?trained_personnel - trained_personnel)
    :precondition
      (and
        (management_unit_personnel_assigned ?incident_management_unit ?trained_personnel)
      )
    :effect
      (and
        (trained_personnel_available ?trained_personnel)
        (not
          (management_unit_personnel_assigned ?incident_management_unit ?trained_personnel)
        )
      )
  )
  (:action assign_external_assistance_to_management_unit
    :parameters (?incident_management_unit - incident_management_unit ?external_assistance - external_assistance)
    :precondition
      (and
        (control_measures_active_for_entity ?incident_management_unit)
        (external_assistance_available ?external_assistance)
      )
    :effect
      (and
        (management_unit_external_assistance_assigned ?incident_management_unit ?external_assistance)
        (not
          (external_assistance_available ?external_assistance)
        )
      )
  )
  (:action release_external_assistance_from_management_unit
    :parameters (?incident_management_unit - incident_management_unit ?external_assistance - external_assistance)
    :precondition
      (and
        (management_unit_external_assistance_assigned ?incident_management_unit ?external_assistance)
      )
    :effect
      (and
        (external_assistance_available ?external_assistance)
        (not
          (management_unit_external_assistance_assigned ?incident_management_unit ?external_assistance)
        )
      )
  )
  (:action inspect_and_mark_boundary_segment
    :parameters (?infected_subsite - infected_subsite ?boundary_segment - boundary_segment ?diagnostic_test_kit - diagnostic_test_kit)
    :precondition
      (and
        (control_measures_active_for_entity ?infected_subsite)
        (diagnostic_test_association ?infected_subsite ?diagnostic_test_kit)
        (infected_subsite_boundary_link ?infected_subsite ?boundary_segment)
        (not
          (boundary_segment_ready ?boundary_segment)
        )
        (not
          (boundary_segment_treated ?boundary_segment)
        )
      )
    :effect (boundary_segment_ready ?boundary_segment)
  )
  (:action apply_equipment_to_infected_subsite
    :parameters (?infected_subsite - infected_subsite ?boundary_segment - boundary_segment ?specialized_equipment - specialized_equipment)
    :precondition
      (and
        (control_measures_active_for_entity ?infected_subsite)
        (equipment_assigned ?infected_subsite ?specialized_equipment)
        (infected_subsite_boundary_link ?infected_subsite ?boundary_segment)
        (boundary_segment_ready ?boundary_segment)
        (not
          (infected_subsite_assessment_complete ?infected_subsite)
        )
      )
    :effect
      (and
        (infected_subsite_assessment_complete ?infected_subsite)
        (infected_subsite_ready ?infected_subsite)
      )
  )
  (:action deploy_consumable_to_infected_subsite
    :parameters (?infected_subsite - infected_subsite ?boundary_segment - boundary_segment ?consumable_supply - consumable_supply)
    :precondition
      (and
        (control_measures_active_for_entity ?infected_subsite)
        (infected_subsite_boundary_link ?infected_subsite ?boundary_segment)
        (consumable_available ?consumable_supply)
        (not
          (infected_subsite_assessment_complete ?infected_subsite)
        )
      )
    :effect
      (and
        (boundary_segment_treated ?boundary_segment)
        (infected_subsite_assessment_complete ?infected_subsite)
        (infected_subsite_supply_assigned ?infected_subsite ?consumable_supply)
        (not
          (consumable_available ?consumable_supply)
        )
      )
  )
  (:action treat_infected_subsite_and_release_supply
    :parameters (?infected_subsite - infected_subsite ?boundary_segment - boundary_segment ?diagnostic_test_kit - diagnostic_test_kit ?consumable_supply - consumable_supply)
    :precondition
      (and
        (control_measures_active_for_entity ?infected_subsite)
        (diagnostic_test_association ?infected_subsite ?diagnostic_test_kit)
        (infected_subsite_boundary_link ?infected_subsite ?boundary_segment)
        (boundary_segment_treated ?boundary_segment)
        (infected_subsite_supply_assigned ?infected_subsite ?consumable_supply)
        (not
          (infected_subsite_ready ?infected_subsite)
        )
      )
    :effect
      (and
        (boundary_segment_ready ?boundary_segment)
        (infected_subsite_ready ?infected_subsite)
        (consumable_available ?consumable_supply)
        (not
          (infected_subsite_supply_assigned ?infected_subsite ?consumable_supply)
        )
      )
  )
  (:action inspect_and_mark_access_point
    :parameters (?exposed_subsite - exposed_subsite ?access_point - access_point ?diagnostic_test_kit - diagnostic_test_kit)

          (access_point_ready ?access_point)

          (access_point_treated ?access_point)

      )
    :effect (access_point_ready ?access_point)
  )
  (:action apply_equipment_to_exposed_subsite
    :parameters (?exposed_subsite - exposed_subsite ?access_point - access_point ?specialized_equipment - specialized_equipment)
    :precondition
      (and
        (control_measures_active_for_entity ?exposed_subsite)

        (exposed_subsite_access_link ?exposed_subsite ?access_point)
        (access_point_ready ?access_point)
        (not
          (exposed_subsite_assessment_complete ?exposed_subsite)
        )
      )
    :effect
      (and
        (exposed_subsite_assessment_complete ?exposed_subsite)
        (exposed_subsite_ready ?exposed_subsite)
      )
  )
  (:action deploy_consumable_to_exposed_subsite
    :parameters (?exposed_subsite - exposed_subsite ?access_point - access_point ?consumable_supply - consumable_supply)
    :precondition
      (and
        (control_measures_active_for_entity ?exposed_subsite)
        (exposed_subsite_access_link ?exposed_subsite ?access_point)
        (consumable_available ?consumable_supply)
        (not
          (exposed_subsite_assessment_complete ?exposed_subsite)
        )
      )
    :effect
      (and
        (access_point_treated ?access_point)
        (exposed_subsite_assessment_complete ?exposed_subsite)
        (exposed_subsite_supply_assigned ?exposed_subsite ?consumable_supply)
        (not
          (consumable_available ?consumable_supply)
        )
      )
  )
  (:action treat_exposed_subsite_and_release_supply
    :parameters (?exposed_subsite - exposed_subsite ?access_point - access_point ?diagnostic_test_kit - diagnostic_test_kit ?consumable_supply - consumable_supply)
    :precondition
      (and
        (control_measures_active_for_entity ?exposed_subsite)
        (diagnostic_test_association ?exposed_subsite ?diagnostic_test_kit)
        (exposed_subsite_access_link ?exposed_subsite ?access_point)
        (access_point_treated ?access_point)
        (exposed_subsite_supply_assigned ?exposed_subsite ?consumable_supply)
        (not
          (exposed_subsite_ready ?exposed_subsite)
        )
      )
    :effect
      (and
        (access_point_ready ?access_point)
        (exposed_subsite_ready ?exposed_subsite)
        (consumable_available ?consumable_supply)
        (not
          (exposed_subsite_supply_assigned ?exposed_subsite ?consumable_supply)
        )
      )
  )
  (:action create_recovery_plan_basic
    :parameters (?infected_subsite - infected_subsite ?exposed_subsite - exposed_subsite ?boundary_segment - boundary_segment ?access_point - access_point ?recovery_plan - recovery_plan)
    :precondition
      (and
        (infected_subsite_assessment_complete ?infected_subsite)
        (exposed_subsite_assessment_complete ?exposed_subsite)
        (infected_subsite_boundary_link ?infected_subsite ?boundary_segment)
        (exposed_subsite_access_link ?exposed_subsite ?access_point)
        (boundary_segment_ready ?boundary_segment)
        (access_point_ready ?access_point)
        (infected_subsite_ready ?infected_subsite)
        (exposed_subsite_ready ?exposed_subsite)
        (recovery_plan_available ?recovery_plan)
      )
    :effect
      (and
        (recovery_plan_initialized ?recovery_plan)
        (recovery_plan_boundary_link ?recovery_plan ?boundary_segment)
        (recovery_plan_access_link ?recovery_plan ?access_point)
        (not
          (recovery_plan_available ?recovery_plan)
        )
      )
  )
  (:action create_recovery_plan_variant_a
    :parameters (?infected_subsite - infected_subsite ?exposed_subsite - exposed_subsite ?boundary_segment - boundary_segment ?access_point - access_point ?recovery_plan - recovery_plan)
    :precondition
      (and
        (infected_subsite_assessment_complete ?infected_subsite)
        (exposed_subsite_assessment_complete ?exposed_subsite)
        (infected_subsite_boundary_link ?infected_subsite ?boundary_segment)
        (exposed_subsite_access_link ?exposed_subsite ?access_point)
        (boundary_segment_treated ?boundary_segment)
        (access_point_ready ?access_point)
        (not
          (infected_subsite_ready ?infected_subsite)
        )
        (exposed_subsite_ready ?exposed_subsite)
        (recovery_plan_available ?recovery_plan)
      )
    :effect
      (and
        (recovery_plan_initialized ?recovery_plan)
        (recovery_plan_boundary_link ?recovery_plan ?boundary_segment)
        (recovery_plan_access_link ?recovery_plan ?access_point)
        (recovery_plan_variant_a ?recovery_plan)
        (not
          (recovery_plan_available ?recovery_plan)
        )
      )
  )
  (:action create_recovery_plan_variant_b
    :parameters (?infected_subsite - infected_subsite ?exposed_subsite - exposed_subsite ?boundary_segment - boundary_segment ?access_point - access_point ?recovery_plan - recovery_plan)
    :precondition
      (and
        (infected_subsite_assessment_complete ?infected_subsite)
        (exposed_subsite_assessment_complete ?exposed_subsite)
        (infected_subsite_boundary_link ?infected_subsite ?boundary_segment)
        (exposed_subsite_access_link ?exposed_subsite ?access_point)
        (boundary_segment_ready ?boundary_segment)
        (access_point_treated ?access_point)
        (infected_subsite_ready ?infected_subsite)
        (not
          (exposed_subsite_ready ?exposed_subsite)
        )
        (recovery_plan_available ?recovery_plan)
      )
    :effect
      (and
        (recovery_plan_initialized ?recovery_plan)
        (recovery_plan_boundary_link ?recovery_plan ?boundary_segment)
        (recovery_plan_access_link ?recovery_plan ?access_point)
        (recovery_plan_variant_b ?recovery_plan)
        (not
          (recovery_plan_available ?recovery_plan)
        )
      )
  )
  (:action create_recovery_plan_variant_combined
    :parameters (?infected_subsite - infected_subsite ?exposed_subsite - exposed_subsite ?boundary_segment - boundary_segment ?access_point - access_point ?recovery_plan - recovery_plan)
    :precondition
      (and
        (infected_subsite_assessment_complete ?infected_subsite)
        (exposed_subsite_assessment_complete ?exposed_subsite)
        (infected_subsite_boundary_link ?infected_subsite ?boundary_segment)
        (exposed_subsite_access_link ?exposed_subsite ?access_point)
        (boundary_segment_treated ?boundary_segment)
        (access_point_treated ?access_point)
        (not
          (infected_subsite_ready ?infected_subsite)

        (not
          (exposed_subsite_ready ?exposed_subsite)

      (and
        (recovery_plan_initialized ?recovery_plan)
        (recovery_plan_boundary_link ?recovery_plan ?boundary_segment)
        (recovery_plan_access_link ?recovery_plan ?access_point)
        (recovery_plan_variant_a ?recovery_plan)
        (recovery_plan_variant_b ?recovery_plan)
        (not
          (recovery_plan_available ?recovery_plan)
        )
      )
  )
  (:action activate_recovery_plan_stage
    :parameters (?recovery_plan - recovery_plan ?infected_subsite - infected_subsite ?diagnostic_test_kit - diagnostic_test_kit)
    :precondition
      (and
        (recovery_plan_initialized ?recovery_plan)
        (infected_subsite_assessment_complete ?infected_subsite)
        (diagnostic_test_association ?infected_subsite ?diagnostic_test_kit)
        (not
          (recovery_plan_stage_activated ?recovery_plan)
        )
      )
    :effect (recovery_plan_stage_activated ?recovery_plan)
  )
  (:action allocate_resource_batch_to_plan
    :parameters (?incident_management_unit - incident_management_unit ?resource_batch - resource_batch ?recovery_plan - recovery_plan)
    :precondition
      (and
        (control_measures_active_for_entity ?incident_management_unit)
        (management_unit_has_recovery_plan ?incident_management_unit ?recovery_plan)
        (management_unit_resource_allocated ?incident_management_unit ?resource_batch)
        (resource_batch_available ?resource_batch)
        (recovery_plan_initialized ?recovery_plan)
        (recovery_plan_stage_activated ?recovery_plan)
        (not
          (resource_batch_deployed ?resource_batch)
        )
      )
    :effect
      (and
        (resource_batch_deployed ?resource_batch)
        (resource_batch_linked_to_plan ?resource_batch ?recovery_plan)
        (not
          (resource_batch_available ?resource_batch)
        )
      )
  )
  (:action prepare_management_unit_for_plan_execution
    :parameters (?incident_management_unit - incident_management_unit ?resource_batch - resource_batch ?recovery_plan - recovery_plan ?diagnostic_test_kit - diagnostic_test_kit)
    :precondition
      (and
        (control_measures_active_for_entity ?incident_management_unit)
        (management_unit_resource_allocated ?incident_management_unit ?resource_batch)
        (resource_batch_deployed ?resource_batch)
        (resource_batch_linked_to_plan ?resource_batch ?recovery_plan)
        (diagnostic_test_association ?incident_management_unit ?diagnostic_test_kit)
        (not
          (recovery_plan_variant_a ?recovery_plan)
        )
        (not
          (management_unit_prepared ?incident_management_unit)
        )
      )
    :effect (management_unit_prepared ?incident_management_unit)
  )
  (:action attach_regulatory_permit_to_management_unit
    :parameters (?incident_management_unit - incident_management_unit ?regulatory_permit - regulatory_permit)
    :precondition
      (and
        (control_measures_active_for_entity ?incident_management_unit)
        (regulatory_permit_available ?regulatory_permit)
        (not
          (management_unit_permit_attached ?incident_management_unit)
        )
      )
    :effect
      (and
        (management_unit_permit_attached ?incident_management_unit)
        (management_unit_permit_linked ?incident_management_unit ?regulatory_permit)
        (not
          (regulatory_permit_available ?regulatory_permit)
        )
      )
  )
  (:action integrate_permit_into_management_review
    :parameters (?incident_management_unit - incident_management_unit ?resource_batch - resource_batch ?recovery_plan - recovery_plan ?diagnostic_test_kit - diagnostic_test_kit ?regulatory_permit - regulatory_permit)
