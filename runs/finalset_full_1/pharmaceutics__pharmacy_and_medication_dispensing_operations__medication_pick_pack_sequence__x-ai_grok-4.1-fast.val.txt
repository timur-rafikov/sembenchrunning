--- domain.pddl (as passed to VAL) ---
(define (domain medication_pick_pack_sequence)
  (:requirements :strips :typing :negative-preconditions)
  (:types pharmacy_resource - object supporting_resource - object location_or_container - object prescription_record_root - object prescription_order - prescription_record_root stock_unit - pharmacy_resource medication_product - pharmacy_resource staff - pharmacy_resource packaging_material - pharmacy_resource instruction_template - pharmacy_resource patient_profile - pharmacy_resource clinical_check - pharmacy_resource regulatory_document - pharmacy_resource accessory_item - supporting_resource label_template - supporting_resource prior_auth_document - supporting_resource pick_bin - location_or_container pack_station - location_or_container package - location_or_container dispensing_task - prescription_order dispense_job_container - prescription_order pick_task - dispensing_task pack_task - dispensing_task dispense_job - dispense_job_container)
  (:predicates
    (received ?prescription_order - prescription_order)
    (verified ?prescription_order - prescription_order)
    (has_allocated_stock ?prescription_order - prescription_order)
    (ready_for_dispensing ?prescription_order - prescription_order)
    (final_quality_check_passed ?prescription_order - prescription_order)
    (patient_instructions_attached ?prescription_order - prescription_order)
    (stock_unit_available ?stock_unit - stock_unit)
    (has_allocated_stock_unit ?prescription_order - prescription_order ?stock_unit - stock_unit)
    (medication_product_available ?medication_product - medication_product)
    (has_medication_product ?prescription_order - prescription_order ?medication_product - medication_product)
    (staff_available ?staff - staff)
    (has_assigned_staff ?prescription_order - prescription_order ?staff - staff)
    (accessory_item_available ?accessory_item - accessory_item)
    (pick_task_assigned_accessory ?pick_task - pick_task ?accessory_item - accessory_item)
    (pack_task_assigned_accessory ?pack_task - pack_task ?accessory_item - accessory_item)
    (pick_task_assigned_bin ?pick_task - pick_task ?pick_bin - pick_bin)
    (pick_bin_claimed ?pick_bin - pick_bin)
    (pick_bin_marked_for_accessory ?pick_bin - pick_bin)
    (pick_task_completed ?pick_task - pick_task)
    (pack_task_assigned_station ?pack_task - pack_task ?pack_station - pack_station)
    (pack_station_claimed ?pack_station - pack_station)
    (pack_station_marked_for_accessory ?pack_station - pack_station)
    (pack_task_completed ?pack_task - pack_task)
    (package_available ?package - package)
    (package_allocated ?package - package)
    (package_assigned_pick_bin ?package - package ?pick_bin - pick_bin)
    (package_assigned_pack_station ?package - package ?pack_station - pack_station)
    (package_requires_label_template ?package - package)
    (package_requires_packaging_material ?package - package)
    (package_contents_verified ?package - package)
    (dispense_job_has_pick_task ?dispense_job - dispense_job ?pick_task - pick_task)
    (dispense_job_has_pack_task ?dispense_job - dispense_job ?pack_task - pack_task)
    (dispense_job_assigned_package ?dispense_job - dispense_job ?package - package)
    (label_template_available ?label_template - label_template)
    (dispense_job_assigned_label_template ?dispense_job - dispense_job ?label_template - label_template)
    (label_template_bound ?label_template - label_template)
    (label_template_applied_to_package ?label_template - label_template ?package - package)
    (dispense_job_ready_for_finalization ?dispense_job - dispense_job)
    (materials_attached_to_job ?dispense_job - dispense_job)
    (dispense_job_documentation_compiled ?dispense_job - dispense_job)
    (packaging_material_allocated ?dispense_job - dispense_job)
    (dispense_job_packaging_confirmed ?dispense_job - dispense_job)
    (dispense_job_ready_for_instruction_binding ?dispense_job - dispense_job)
    (final_checks_completed ?dispense_job - dispense_job)
    (prior_auth_document_available ?prior_auth_document - prior_auth_document)
    (dispense_job_attached_prior_auth ?dispense_job - dispense_job ?prior_auth_document - prior_auth_document)
    (prior_auth_verified ?dispense_job - dispense_job)
    (prior_authorization_applied ?dispense_job - dispense_job)
    (prior_authorization_finalized ?dispense_job - dispense_job)
    (packaging_material_available ?packaging_material - packaging_material)
    (dispense_job_assigned_packaging_material ?dispense_job - dispense_job ?packaging_material - packaging_material)
    (instruction_template_available ?instruction_template - instruction_template)
    (dispense_job_assigned_instruction_template ?dispense_job - dispense_job ?instruction_template - instruction_template)
    (clinical_check_available ?clinical_check - clinical_check)
    (dispense_job_assigned_clinical_check ?dispense_job - dispense_job ?clinical_check - clinical_check)
    (regulatory_document_available ?regulatory_document - regulatory_document)
    (dispense_job_assigned_regulatory_document ?dispense_job - dispense_job ?regulatory_document - regulatory_document)
    (patient_profile_available ?patient_profile - patient_profile)
    (has_assigned_patient_profile ?prescription_order - prescription_order ?patient_profile - patient_profile)
    (pick_task_claimed ?pick_task - pick_task)
    (pack_task_claimed ?pack_task - pack_task)
    (dispense_job_locked_for_release ?dispense_job - dispense_job)
  )
  (:action register_order
    :parameters (?prescription_order - prescription_order)
    :precondition
      (and
        (not
          (received ?prescription_order)
        )
        (not
          (ready_for_dispensing ?prescription_order)
        )
      )
    :effect (received ?prescription_order)
  )
  (:action assign_stock_unit_to_order
    :parameters (?prescription_order - prescription_order ?stock_unit - stock_unit)
    :precondition
      (and
        (received ?prescription_order)
        (not
          (has_allocated_stock ?prescription_order)
        )
        (stock_unit_available ?stock_unit)
      )
    :effect
      (and
        (has_allocated_stock ?prescription_order)
        (has_allocated_stock_unit ?prescription_order ?stock_unit)
        (not
          (stock_unit_available ?stock_unit)
        )
      )
  )
  (:action associate_medication_product_with_order
    :parameters (?prescription_order - prescription_order ?medication_product - medication_product)
    :precondition
      (and
        (received ?prescription_order)
        (has_allocated_stock ?prescription_order)
        (medication_product_available ?medication_product)
      )
    :effect
      (and
        (has_medication_product ?prescription_order ?medication_product)
        (not
          (medication_product_available ?medication_product)
        )
      )
  )
  (:action verify_order
    :parameters (?prescription_order - prescription_order ?medication_product - medication_product)
    :precondition
      (and
        (received ?prescription_order)
        (has_allocated_stock ?prescription_order)
        (has_medication_product ?prescription_order ?medication_product)
        (not
          (verified ?prescription_order)
        )
      )
    :effect (verified ?prescription_order)
  )
  (:action release_medication_product_to_inventory
    :parameters (?prescription_order - prescription_order ?medication_product - medication_product)
    :precondition
      (and
        (has_medication_product ?prescription_order ?medication_product)
      )
    :effect
      (and
        (medication_product_available ?medication_product)
        (not
          (has_medication_product ?prescription_order ?medication_product)
        )
      )
  )
  (:action assign_staff_to_order
    :parameters (?prescription_order - prescription_order ?staff - staff)
    :precondition
      (and
        (verified ?prescription_order)
        (staff_available ?staff)
      )
    :effect
      (and
        (has_assigned_staff ?prescription_order ?staff)
        (not
          (staff_available ?staff)
        )
      )
  )
  (:action unassign_staff_from_order
    :parameters (?prescription_order - prescription_order ?staff - staff)
    :precondition
      (and
        (has_assigned_staff ?prescription_order ?staff)
      )
    :effect
      (and
        (staff_available ?staff)
        (not
          (has_assigned_staff ?prescription_order ?staff)
        )
      )
  )
  (:action assign_clinical_check_to_dispense_job
    :parameters (?dispense_job - dispense_job ?clinical_check - clinical_check)
    :precondition
      (and
        (verified ?dispense_job)
        (clinical_check_available ?clinical_check)
      )
    :effect
      (and
        (dispense_job_assigned_clinical_check ?dispense_job ?clinical_check)
        (not
          (clinical_check_available ?clinical_check)
        )
      )
  )
  (:action unassign_clinical_check_from_dispense_job
    :parameters (?dispense_job - dispense_job ?clinical_check - clinical_check)
    :precondition
      (and
        (dispense_job_assigned_clinical_check ?dispense_job ?clinical_check)
      )
    :effect
      (and
        (clinical_check_available ?clinical_check)
        (not
          (dispense_job_assigned_clinical_check ?dispense_job ?clinical_check)
        )
      )
  )
  (:action assign_regulatory_document_to_dispense_job
    :parameters (?dispense_job - dispense_job ?regulatory_document - regulatory_document)
    :precondition
      (and
        (verified ?dispense_job)
        (regulatory_document_available ?regulatory_document)
      )
    :effect
      (and
        (dispense_job_assigned_regulatory_document ?dispense_job ?regulatory_document)
        (not
          (regulatory_document_available ?regulatory_document)
        )
      )
  )
  (:action unassign_regulatory_document_from_dispense_job
    :parameters (?dispense_job - dispense_job ?regulatory_document - regulatory_document)
    :precondition
      (and
        (dispense_job_assigned_regulatory_document ?dispense_job ?regulatory_document)
      )
    :effect
      (and
        (regulatory_document_available ?regulatory_document)
        (not
          (dispense_job_assigned_regulatory_document ?dispense_job ?regulatory_document)
        )
      )
  )
  (:action claim_pick_bin_for_task
    :parameters (?pick_task - pick_task ?pick_bin - pick_bin ?medication_product - medication_product)
    :precondition
      (and
        (verified ?pick_task)
        (has_medication_product ?pick_task ?medication_product)
        (pick_task_assigned_bin ?pick_task ?pick_bin)
        (not
          (pick_bin_claimed ?pick_bin)
        )
        (not
          (pick_bin_marked_for_accessory ?pick_bin)
        )
      )
    :effect (pick_bin_claimed ?pick_bin)
  )
  (:action confirm_pick_task_completion
    :parameters (?pick_task - pick_task ?pick_bin - pick_bin ?staff - staff)
    :precondition
      (and
        (verified ?pick_task)
        (has_assigned_staff ?pick_task ?staff)
        (pick_task_assigned_bin ?pick_task ?pick_bin)
        (pick_bin_claimed ?pick_bin)
        (not
          (pick_task_claimed ?pick_task)
        )
      )
    :effect
      (and
        (pick_task_claimed ?pick_task)
        (pick_task_completed ?pick_task)
      )
  )
  (:action attach_accessory_to_pick_task
    :parameters (?pick_task - pick_task ?pick_bin - pick_bin ?accessory_item - accessory_item)
    :precondition
      (and
        (verified ?pick_task)
        (pick_task_assigned_bin ?pick_task ?pick_bin)
        (accessory_item_available ?accessory_item)
        (not
          (pick_task_claimed ?pick_task)
        )
      )
    :effect
      (and
        (pick_bin_marked_for_accessory ?pick_bin)
        (pick_task_claimed ?pick_task)
        (pick_task_assigned_accessory ?pick_task ?accessory_item)
        (not
          (accessory_item_available ?accessory_item)
        )
      )
  )
  (:action finalize_pick_with_accessory
    :parameters (?pick_task - pick_task ?pick_bin - pick_bin ?medication_product - medication_product ?accessory_item - accessory_item)
    :precondition
      (and
        (verified ?pick_task)
        (has_medication_product ?pick_task ?medication_product)
        (pick_task_assigned_bin ?pick_task ?pick_bin)
        (pick_bin_marked_for_accessory ?pick_bin)
        (pick_task_assigned_accessory ?pick_task ?accessory_item)
        (not
          (pick_task_completed ?pick_task)
        )
      )
    :effect
      (and
        (pick_bin_claimed ?pick_bin)
        (pick_task_completed ?pick_task)
        (accessory_item_available ?accessory_item)
        (not
          (pick_task_assigned_accessory ?pick_task ?accessory_item)
        )
      )
  )
  (:action claim_pack_station_for_task
    :parameters (?pack_task - pack_task ?pack_station - pack_station ?medication_product - medication_product)
    :precondition
      (and
        (verified ?pack_task)
        (has_medication_product ?pack_task ?medication_product)
        (pack_task_assigned_station ?pack_task ?pack_station)
        (not
          (pack_station_claimed ?pack_station)
        )
        (not
          (pack_station_marked_for_accessory ?pack_station)
        )
      )
    :effect (pack_station_claimed ?pack_station)
  )
  (:action complete_pack_task_by_staff
    :parameters (?pack_task - pack_task ?pack_station - pack_station ?staff - staff)
    :precondition
      (and
        (verified ?pack_task)
        (has_assigned_staff ?pack_task ?staff)
        (pack_task_assigned_station ?pack_task ?pack_station)
        (pack_station_claimed ?pack_station)
        (not
          (pack_task_claimed ?pack_task)
        )
      )
    :effect
      (and
        (pack_task_claimed ?pack_task)
        (pack_task_completed ?pack_task)
      )
  )
  (:action attach_accessory_to_pack_task
    :parameters (?pack_task - pack_task ?pack_station - pack_station ?accessory_item - accessory_item)
    :precondition
      (and
        (verified ?pack_task)
        (pack_task_assigned_station ?pack_task ?pack_station)
        (accessory_item_available ?accessory_item)
        (not
          (pack_task_claimed ?pack_task)
        )
      )
    :effect
      (and
        (pack_station_marked_for_accessory ?pack_station)
        (pack_task_claimed ?pack_task)
        (pack_task_assigned_accessory ?pack_task ?accessory_item)
        (not
          (accessory_item_available ?accessory_item)
        )
      )
  )
  (:action finalize_pack_with_accessory
    :parameters (?pack_task - pack_task ?pack_station - pack_station ?medication_product - medication_product ?accessory_item - accessory_item)
    :precondition
      (and
        (verified ?pack_task)
        (has_medication_product ?pack_task ?medication_product)
        (pack_task_assigned_station ?pack_task ?pack_station)
        (pack_station_marked_for_accessory ?pack_station)
        (pack_task_assigned_accessory ?pack_task ?accessory_item)
        (not
          (pack_task_completed ?pack_task)
        )
      )
    :effect
      (and
        (pack_station_claimed ?pack_station)
        (pack_task_completed ?pack_task)
        (accessory_item_available ?accessory_item)
        (not
          (pack_task_assigned_accessory ?pack_task ?accessory_item)
        )
      )
  )
  (:action allocate_package_for_pick_pack
    :parameters (?pick_task - pick_task ?pack_task - pack_task ?pick_bin - pick_bin ?pack_station - pack_station ?package - package)
    :precondition
      (and
        (pick_task_claimed ?pick_task)
        (pack_task_claimed ?pack_task)
        (pick_task_assigned_bin ?pick_task ?pick_bin)
        (pack_task_assigned_station ?pack_task ?pack_station)
        (pick_bin_claimed ?pick_bin)
        (pack_station_claimed ?pack_station)
        (pick_task_completed ?pick_task)
        (pack_task_completed ?pack_task)
        (package_available ?package)
      )
    :effect
      (and
        (package_allocated ?package)
        (package_assigned_pick_bin ?package ?pick_bin)
        (package_assigned_pack_station ?package ?pack_station)
        (not
          (package_available ?package)
        )
      )
  )
  (:action allocate_package_and_flag_for_label_template
    :parameters (?pick_task - pick_task ?pack_task - pack_task ?pick_bin - pick_bin ?pack_station - pack_station ?package - package)
    :precondition
      (and
        (pick_task_claimed ?pick_task)
        (pack_task_claimed ?pack_task)
        (pick_task_assigned_bin ?pick_task ?pick_bin)
        (pack_task_assigned_station ?pack_task ?pack_station)
        (pick_bin_marked_for_accessory ?pick_bin)
        (pack_station_claimed ?pack_station)
        (not
          (pick_task_completed ?pick_task)
        )
        (pack_task_completed ?pack_task)
        (package_available ?package)
      )
    :effect
      (and
        (package_allocated ?package)
        (package_assigned_pick_bin ?package ?pick_bin)
        (package_assigned_pack_station ?package ?pack_station)
        (package_requires_label_template ?package)
        (not
          (package_available ?package)
        )
      )
  )
  (:action allocate_package_and_flag_packaging_materials
    :parameters (?pick_task - pick_task ?pack_task - pack_task ?pick_bin - pick_bin ?pack_station - pack_station ?package - package)
    :precondition
      (and
        (pick_task_claimed ?pick_task)
        (pack_task_claimed ?pack_task)
        (pick_task_assigned_bin ?pick_task ?pick_bin)
        (pack_task_assigned_station ?pack_task ?pack_station)
        (pick_bin_claimed ?pick_bin)
        (pack_station_marked_for_accessory ?pack_station)
        (pick_task_completed ?pick_task)
        (not
          (pack_task_completed ?pack_task)
        )
        (package_available ?package)
      )
    :effect
      (and
        (package_allocated ?package)
        (package_assigned_pick_bin ?package ?pick_bin)
        (package_assigned_pack_station ?package ?pack_station)
        (package_requires_packaging_material ?package)
        (not
          (package_available ?package)
        )
      )
  )
  (:action allocate_package_and_flag_label_and_materials
    :parameters (?pick_task - pick_task ?pack_task - pack_task ?pick_bin - pick_bin ?pack_station - pack_station ?package - package)
    :precondition
      (and
        (pick_task_claimed ?pick_task)
        (pack_task_claimed ?pack_task)
        (pick_task_assigned_bin ?pick_task ?pick_bin)
        (pack_task_assigned_station ?pack_task ?pack_station)
        (pick_bin_marked_for_accessory ?pick_bin)
        (pack_station_marked_for_accessory ?pack_station)
        (not
          (pick_task_completed ?pick_task)
        )
        (not
          (pack_task_completed ?pack_task)
        )
        (package_available ?package)
      )
    :effect
      (and
        (package_allocated ?package)
        (package_assigned_pick_bin ?package ?pick_bin)
        (package_assigned_pack_station ?package ?pack_station)
        (package_requires_label_template ?package)
        (package_requires_packaging_material ?package)
        (not
          (package_available ?package)
        )
      )
  )
  (:action verify_package_contents
    :parameters (?package - package ?pick_task - pick_task ?medication_product - medication_product)
    :precondition
      (and
        (package_allocated ?package)
        (pick_task_claimed ?pick_task)
        (has_medication_product ?pick_task ?medication_product)
        (not
          (package_contents_verified ?package)
        )
      )
    :effect (package_contents_verified ?package)
  )
  (:action bind_label_template_to_package_for_job
    :parameters (?dispense_job - dispense_job ?label_template - label_template ?package - package)
    :precondition
      (and
        (verified ?dispense_job)
        (dispense_job_assigned_package ?dispense_job ?package)
        (dispense_job_assigned_label_template ?dispense_job ?label_template)
        (label_template_available ?label_template)
        (package_allocated ?package)
        (package_contents_verified ?package)
        (not
          (label_template_bound ?label_template)
        )
      )
    :effect
      (and
        (label_template_bound ?label_template)
        (label_template_applied_to_package ?label_template ?package)
        (not
          (label_template_available ?label_template)
        )
      )
  )
  (:action mark_dispense_job_ready_for_finalization
    :parameters (?dispense_job - dispense_job ?label_template - label_template ?package - package ?medication_product - medication_product)
    :precondition
      (and
        (verified ?dispense_job)
        (dispense_job_assigned_label_template ?dispense_job ?label_template)
        (label_template_bound ?label_template)
        (label_template_applied_to_package ?label_template ?package)
        (has_medication_product ?dispense_job ?medication_product)
        (not
          (package_requires_label_template ?package)
        )
        (not
          (dispense_job_ready_for_finalization ?dispense_job)
        )
      )
    :effect (dispense_job_ready_for_finalization ?dispense_job)
  )
  (:action assign_packaging_material_to_job
    :parameters (?dispense_job - dispense_job ?packaging_material - packaging_material)
    :precondition
      (and
        (verified ?dispense_job)
        (packaging_material_available ?packaging_material)
        (not
          (packaging_material_allocated ?dispense_job)
        )
      )
    :effect
      (and
        (packaging_material_allocated ?dispense_job)
        (dispense_job_assigned_packaging_material ?dispense_job ?packaging_material)
        (not
          (packaging_material_available ?packaging_material)
        )
      )
  )
  (:action apply_packaging_material_to_job
    :parameters (?dispense_job - dispense_job ?label_template - label_template ?package - package ?medication_product - medication_product ?packaging_material - packaging_material)
    :precondition
      (and
        (verified ?dispense_job)
        (dispense_job_assigned_label_template ?dispense_job ?label_template)
        (label_template_bound ?label_template)
        (label_template_applied_to_package ?label_template ?package)
        (has_medication_product ?dispense_job ?medication_product)
        (package_requires_label_template ?package)
        (packaging_material_allocated ?dispense_job)
        (dispense_job_assigned_packaging_material ?dispense_job ?packaging_material)
        (not
          (dispense_job_ready_for_finalization ?dispense_job)
        )
      )
    :effect
      (and
        (dispense_job_ready_for_finalization ?dispense_job)
        (dispense_job_packaging_confirmed ?dispense_job)
      )
  )
  (:action attach_clinical_check_and_mark_materials
    :parameters (?dispense_job - dispense_job ?clinical_check - clinical_check ?staff - staff ?label_template - label_template ?package - package)
    :precondition
      (and
        (dispense_job_ready_for_finalization ?dispense_job)
        (dispense_job_assigned_clinical_check ?dispense_job ?clinical_check)
        (has_assigned_staff ?dispense_job ?staff)
        (dispense_job_assigned_label_template ?dispense_job ?label_template)
        (label_template_applied_to_package ?label_template ?package)
        (not
          (package_requires_packaging_material ?package)
        )
        (not
          (materials_attached_to_job ?dispense_job)
        )
      )
    :effect (materials_attached_to_job ?dispense_job)
  )
  (:action attach_clinical_check_with_material_flag
    :parameters (?dispense_job - dispense_job ?clinical_check - clinical_check ?staff - staff ?label_template - label_template ?package - package)
    :precondition
      (and
        (dispense_job_ready_for_finalization ?dispense_job)
        (dispense_job_assigned_clinical_check ?dispense_job ?clinical_check)
        (has_assigned_staff ?dispense_job ?staff)
        (dispense_job_assigned_label_template ?dispense_job ?label_template)
        (label_template_applied_to_package ?label_template ?package)
        (package_requires_packaging_material ?package)
        (not
          (materials_attached_to_job ?dispense_job)
        )
      )
    :effect (materials_attached_to_job ?dispense_job)
  )
  (:action attach_regulatory_document_to_job_basic
    :parameters (?dispense_job - dispense_job ?regulatory_document - regulatory_document ?label_template - label_template ?package - package)
    :precondition
      (and
        (materials_attached_to_job ?dispense_job)
        (dispense_job_assigned_regulatory_document ?dispense_job ?regulatory_document)
        (dispense_job_assigned_label_template ?dispense_job ?label_template)
        (label_template_applied_to_package ?label_template ?package)
        (not
          (package_requires_label_template ?package)
        )
        (not
          (package_requires_packaging_material ?package)
        )
        (not
          (dispense_job_documentation_compiled ?dispense_job)
        )
      )
