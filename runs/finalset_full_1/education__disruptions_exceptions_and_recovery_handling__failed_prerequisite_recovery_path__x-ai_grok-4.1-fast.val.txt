--- domain.pddl (as passed to VAL) ---
(define (domain failed_prerequisite_recovery)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity - object case_resource - entity supporting_item - entity process_artifact - entity exception_case - entity failed_prerequisite_case - exception_case recovery_option - case_resource remedial_course_offering - case_resource faculty_reviewer - case_resource authorization_token - case_resource administrative_resource_quota - case_resource supporting_document - case_resource external_credit_record - case_resource justification_document - case_resource support_service - supporting_item evidence_artifact - supporting_item policy_reference - supporting_item prerequisite_requirement - process_artifact term_session - process_artifact recovery_submission_document - process_artifact case_subject - failed_prerequisite_case case_participant_role - failed_prerequisite_case student_record - case_subject study_plan_instance - case_subject case_handler - case_participant_role)

  (:predicates
    (recovery_subject_registered ?failed_prerequisite_case - failed_prerequisite_case)
    (recovery_case_validated ?failed_prerequisite_case - failed_prerequisite_case)
    (recovery_subject_recovery_option_selected ?failed_prerequisite_case - failed_prerequisite_case)
    (recovery_subject_resolved ?failed_prerequisite_case - failed_prerequisite_case)
    (recovery_subject_ready_for_finalization ?failed_prerequisite_case - failed_prerequisite_case)
    (recovery_subject_outcome_approved ?failed_prerequisite_case - failed_prerequisite_case)
    (recovery_option_available ?recovery_option - recovery_option)
    (recovery_subject_assigned_recovery_option ?failed_prerequisite_case - failed_prerequisite_case ?recovery_option - recovery_option)
    (remedial_offering_available ?remedial_course_offering - remedial_course_offering)
    (recovery_subject_assigned_remedial_offering ?failed_prerequisite_case - failed_prerequisite_case ?remedial_course_offering - remedial_course_offering)
    (faculty_reviewer_available ?faculty_reviewer - faculty_reviewer)
    (recovery_subject_assigned_reviewer ?failed_prerequisite_case - failed_prerequisite_case ?faculty_reviewer - faculty_reviewer)
    (support_service_available ?support_service - support_service)
    (student_assigned_support_service ?student_record - student_record ?support_service - support_service)
    (study_plan_assigned_support_service ?study_plan_instance - study_plan_instance ?support_service - support_service)
    (student_has_prerequisite_requirement ?student_record - student_record ?prerequisite_requirement - prerequisite_requirement)
    (requirement_flagged_for_review ?prerequisite_requirement - prerequisite_requirement)
    (requirement_support_allocated ?prerequisite_requirement - prerequisite_requirement)
    (student_requirement_verified ?student_record - student_record)
    (study_plan_scheduled_in_term ?study_plan_instance - study_plan_instance ?term_session - term_session)
    (term_session_confirmed ?term_session - term_session)
    (term_session_provisionally_scheduled ?term_session - term_session)
    (study_plan_session_confirmed ?study_plan_instance - study_plan_instance)
    (submission_document_draft ?recovery_submission_document - recovery_submission_document)
    (submission_document_staged ?recovery_submission_document - recovery_submission_document)
    (submission_document_contains_requirement ?recovery_submission_document - recovery_submission_document ?prerequisite_requirement - prerequisite_requirement)
    (submission_document_contains_session ?recovery_submission_document - recovery_submission_document ?term_session - term_session)
    (submission_document_requires_authorization ?recovery_submission_document - recovery_submission_document)
    (submission_document_requires_justification ?recovery_submission_document - recovery_submission_document)
    (submission_document_ready_for_evidence ?recovery_submission_document - recovery_submission_document)
    (handler_assigned_student_record ?case_handler - case_handler ?student_record - student_record)
    (handler_assigned_study_plan ?case_handler - case_handler ?study_plan_instance - study_plan_instance)
    (handler_has_recovery_submission_document ?case_handler - case_handler ?recovery_submission_document - recovery_submission_document)
    (evidence_artifact_available ?evidence_artifact - evidence_artifact)
    (handler_has_evidence_artifact ?case_handler - case_handler ?evidence_artifact - evidence_artifact)
    (evidence_artifact_verified ?evidence_artifact - evidence_artifact)
    (evidence_attached_to_document ?evidence_artifact - evidence_artifact ?recovery_submission_document - recovery_submission_document)
    (handler_completed_evidence_verification ?case_handler - case_handler)
    (handler_external_credit_assessment_started ?case_handler - case_handler)
    (handler_justification_review_started ?case_handler - case_handler)
    (handler_has_policy_attachment ?case_handler - case_handler)
    (handler_policy_application_completed ?case_handler - case_handler)
    (handler_ready_for_final_review ?case_handler - case_handler)
    (handler_final_review_completed ?case_handler - case_handler)
    (policy_reference_available ?policy_reference - policy_reference)
    (handler_has_policy_reference ?case_handler - case_handler ?policy_reference - policy_reference)
    (handler_has_selected_policy ?case_handler - case_handler)
    (handler_has_confirmed_policy ?case_handler - case_handler)
    (handler_has_finalized_policy ?case_handler - case_handler)
    (authorization_token_available ?authorization_token - authorization_token)
    (handler_has_authorization_token ?case_handler - case_handler ?authorization_token - authorization_token)
    (administrative_quota_available ?administrative_resource_quota - administrative_resource_quota)
    (handler_reserved_administrative_quota ?case_handler - case_handler ?administrative_resource_quota - administrative_resource_quota)
    (external_credit_available ?external_credit_record - external_credit_record)
    (handler_has_external_credit ?case_handler - case_handler ?external_credit_record - external_credit_record)
    (justification_document_available ?justification_document - justification_document)
    (handler_has_justification_document ?case_handler - case_handler ?justification_document - justification_document)
    (supporting_document_available ?supporting_document - supporting_document)
    (recovery_subject_attached_supporting_document ?failed_prerequisite_case - failed_prerequisite_case ?supporting_document - supporting_document)
    (student_submission_ready ?student_record - student_record)
    (study_plan_submission_ready ?study_plan_instance - study_plan_instance)
    (handler_signed_off ?case_handler - case_handler)
  )
  (:action register_failed_prerequisite_case
    :parameters (?failed_prerequisite_case - failed_prerequisite_case)
    :precondition
      (and
        (not
          (recovery_subject_registered ?failed_prerequisite_case)
        )
        (not
          (recovery_subject_resolved ?failed_prerequisite_case)
        )
      )
    :effect (recovery_subject_registered ?failed_prerequisite_case)
  )
  (:action assign_recovery_option_to_case
    :parameters (?failed_prerequisite_case - failed_prerequisite_case ?recovery_option - recovery_option)
    :precondition
      (and
        (recovery_subject_registered ?failed_prerequisite_case)
        (not
          (recovery_subject_recovery_option_selected ?failed_prerequisite_case)
        )
        (recovery_option_available ?recovery_option)
      )
    :effect
      (and
        (recovery_subject_recovery_option_selected ?failed_prerequisite_case)
        (recovery_subject_assigned_recovery_option ?failed_prerequisite_case ?recovery_option)
        (not
          (recovery_option_available ?recovery_option)
        )
      )
  )
  (:action assign_remedial_offering_to_case
    :parameters (?failed_prerequisite_case - failed_prerequisite_case ?remedial_course_offering - remedial_course_offering)
    :precondition
      (and
        (recovery_subject_registered ?failed_prerequisite_case)
        (recovery_subject_recovery_option_selected ?failed_prerequisite_case)
        (remedial_offering_available ?remedial_course_offering)
      )
    :effect
      (and
        (recovery_subject_assigned_remedial_offering ?failed_prerequisite_case ?remedial_course_offering)
        (not
          (remedial_offering_available ?remedial_course_offering)
        )
      )
  )
  (:action confirm_remedial_completion_for_case
    :parameters (?failed_prerequisite_case - failed_prerequisite_case ?remedial_course_offering - remedial_course_offering)
    :precondition
      (and
        (recovery_subject_registered ?failed_prerequisite_case)
        (recovery_subject_recovery_option_selected ?failed_prerequisite_case)
        (recovery_subject_assigned_remedial_offering ?failed_prerequisite_case ?remedial_course_offering)
        (not
          (recovery_case_validated ?failed_prerequisite_case)
        )
      )
    :effect (recovery_case_validated ?failed_prerequisite_case)
  )
  (:action withdraw_remedial_offering_assignment
    :parameters (?failed_prerequisite_case - failed_prerequisite_case ?remedial_course_offering - remedial_course_offering)
    :precondition
      (and
        (recovery_subject_assigned_remedial_offering ?failed_prerequisite_case ?remedial_course_offering)
      )
    :effect
      (and
        (remedial_offering_available ?remedial_course_offering)
        (not
          (recovery_subject_assigned_remedial_offering ?failed_prerequisite_case ?remedial_course_offering)
        )
      )
  )
  (:action assign_faculty_reviewer_to_case
    :parameters (?failed_prerequisite_case - failed_prerequisite_case ?faculty_reviewer - faculty_reviewer)
    :precondition
      (and
        (recovery_case_validated ?failed_prerequisite_case)
        (faculty_reviewer_available ?faculty_reviewer)
      )
    :effect
      (and
        (recovery_subject_assigned_reviewer ?failed_prerequisite_case ?faculty_reviewer)
        (not
          (faculty_reviewer_available ?faculty_reviewer)
        )
      )
  )
  (:action release_faculty_reviewer_from_case
    :parameters (?failed_prerequisite_case - failed_prerequisite_case ?faculty_reviewer - faculty_reviewer)
    :precondition
      (and
        (recovery_subject_assigned_reviewer ?failed_prerequisite_case ?faculty_reviewer)
      )
    :effect
      (and
        (faculty_reviewer_available ?faculty_reviewer)
        (not
          (recovery_subject_assigned_reviewer ?failed_prerequisite_case ?faculty_reviewer)
        )
      )
  )
  (:action attach_external_credit_to_handler
    :parameters (?case_handler - case_handler ?external_credit_record - external_credit_record)
    :precondition
      (and
        (recovery_case_validated ?case_handler)
        (external_credit_available ?external_credit_record)
      )
    :effect
      (and
        (handler_has_external_credit ?case_handler ?external_credit_record)
        (not
          (external_credit_available ?external_credit_record)
        )
      )
  )
  (:action detach_external_credit_from_handler
    :parameters (?case_handler - case_handler ?external_credit_record - external_credit_record)
    :precondition
      (and
        (handler_has_external_credit ?case_handler ?external_credit_record)
      )
    :effect
      (and
        (external_credit_available ?external_credit_record)
        (not
          (handler_has_external_credit ?case_handler ?external_credit_record)
        )
      )
  )
  (:action attach_justification_to_handler
    :parameters (?case_handler - case_handler ?justification_document - justification_document)
    :precondition
      (and
        (recovery_case_validated ?case_handler)
        (justification_document_available ?justification_document)
      )
    :effect
      (and
        (handler_has_justification_document ?case_handler ?justification_document)
        (not
          (justification_document_available ?justification_document)
        )
      )
  )
  (:action detach_justification_from_handler
    :parameters (?case_handler - case_handler ?justification_document - justification_document)
    :precondition
      (and
        (handler_has_justification_document ?case_handler ?justification_document)
      )
    :effect
      (and
        (justification_document_available ?justification_document)
        (not
          (handler_has_justification_document ?case_handler ?justification_document)
        )
      )
  )
  (:action flag_prerequisite_for_review
    :parameters (?student_record - student_record ?prerequisite_requirement - prerequisite_requirement ?remedial_course_offering - remedial_course_offering)
    :precondition
      (and
        (recovery_case_validated ?student_record)
        (recovery_subject_assigned_remedial_offering ?student_record ?remedial_course_offering)
        (student_has_prerequisite_requirement ?student_record ?prerequisite_requirement)
        (not
          (requirement_flagged_for_review ?prerequisite_requirement)
        )
        (not
          (requirement_support_allocated ?prerequisite_requirement)
        )
      )
    :effect (requirement_flagged_for_review ?prerequisite_requirement)
  )
  (:action assign_reviewer_and_verify_requirement
    :parameters (?student_record - student_record ?prerequisite_requirement - prerequisite_requirement ?faculty_reviewer - faculty_reviewer)
    :precondition
      (and
        (recovery_case_validated ?student_record)
        (recovery_subject_assigned_reviewer ?student_record ?faculty_reviewer)
        (student_has_prerequisite_requirement ?student_record ?prerequisite_requirement)
        (requirement_flagged_for_review ?prerequisite_requirement)
        (not
          (student_submission_ready ?student_record)
        )
      )
    :effect
      (and
        (student_submission_ready ?student_record)
        (student_requirement_verified ?student_record)
      )
  )
  (:action assign_support_service_to_student_requirement
    :parameters (?student_record - student_record ?prerequisite_requirement - prerequisite_requirement ?support_service - support_service)
    :precondition
      (and
        (recovery_case_validated ?student_record)
        (student_has_prerequisite_requirement ?student_record ?prerequisite_requirement)
        (support_service_available ?support_service)
        (not
          (student_submission_ready ?student_record)
        )
      )
    :effect
      (and
        (requirement_support_allocated ?prerequisite_requirement)
        (student_submission_ready ?student_record)
        (student_assigned_support_service ?student_record ?support_service)
        (not
          (support_service_available ?support_service)
        )
      )
  )
  (:action finalize_support_service_assignment
    :parameters (?student_record - student_record ?prerequisite_requirement - prerequisite_requirement ?remedial_course_offering - remedial_course_offering ?support_service - support_service)
    :precondition
      (and
        (recovery_case_validated ?student_record)
        (recovery_subject_assigned_remedial_offering ?student_record ?remedial_course_offering)
        (student_has_prerequisite_requirement ?student_record ?prerequisite_requirement)
        (requirement_support_allocated ?prerequisite_requirement)
        (student_assigned_support_service ?student_record ?support_service)
        (not
          (student_requirement_verified ?student_record)
        )
      )
    :effect
      (and
        (requirement_flagged_for_review ?prerequisite_requirement)
        (student_requirement_verified ?student_record)
        (support_service_available ?support_service)
        (not
          (student_assigned_support_service ?student_record ?support_service)
        )
      )
  )
  (:action reserve_term_session_for_study_plan
    :parameters (?study_plan_instance - study_plan_instance ?term_session - term_session ?remedial_course_offering - remedial_course_offering)
    :precondition
      (and
        (recovery_case_validated ?study_plan_instance)
        (recovery_subject_assigned_remedial_offering ?study_plan_instance ?remedial_course_offering)
        (study_plan_scheduled_in_term ?study_plan_instance ?term_session)
        (not
          (term_session_confirmed ?term_session)
        )
        (not
          (term_session_provisionally_scheduled ?term_session)
        )
      )
    :effect (term_session_confirmed ?term_session)
  )
  (:action assign_reviewer_and_confirm_study_plan_session
    :parameters (?study_plan_instance - study_plan_instance ?term_session - term_session ?faculty_reviewer - faculty_reviewer)
    :precondition
      (and
        (recovery_case_validated ?study_plan_instance)
        (recovery_subject_assigned_reviewer ?study_plan_instance ?faculty_reviewer)
        (study_plan_scheduled_in_term ?study_plan_instance ?term_session)
        (term_session_confirmed ?term_session)
        (not
          (study_plan_submission_ready ?study_plan_instance)
        )
      )
    :effect
      (and
        (study_plan_submission_ready ?study_plan_instance)
        (study_plan_session_confirmed ?study_plan_instance)
      )
  )
  (:action assign_support_service_to_study_plan
    :parameters (?study_plan_instance - study_plan_instance ?term_session - term_session ?support_service - support_service)
    :precondition
      (and
        (recovery_case_validated ?study_plan_instance)
        (study_plan_scheduled_in_term ?study_plan_instance ?term_session)
        (support_service_available ?support_service)
        (not
          (study_plan_submission_ready ?study_plan_instance)
        )
      )
    :effect
      (and
        (term_session_provisionally_scheduled ?term_session)
        (study_plan_submission_ready ?study_plan_instance)
        (study_plan_assigned_support_service ?study_plan_instance ?support_service)
        (not
          (support_service_available ?support_service)
        )
      )
  )
  (:action finalize_support_service_assignment_for_study_plan
    :parameters (?study_plan_instance - study_plan_instance ?term_session - term_session ?remedial_course_offering - remedial_course_offering ?support_service - support_service)
    :precondition
      (and
        (recovery_case_validated ?study_plan_instance)
        (recovery_subject_assigned_remedial_offering ?study_plan_instance ?remedial_course_offering)
        (study_plan_scheduled_in_term ?study_plan_instance ?term_session)
        (term_session_provisionally_scheduled ?term_session)
        (study_plan_assigned_support_service ?study_plan_instance ?support_service)
        (not
          (study_plan_session_confirmed ?study_plan_instance)
        )
      )
    :effect
      (and
        (term_session_confirmed ?term_session)
        (study_plan_session_confirmed ?study_plan_instance)
        (support_service_available ?support_service)
        (not
          (study_plan_assigned_support_service ?study_plan_instance ?support_service)
        )
      )
  )
  (:action assemble_recovery_submission_document
    :parameters (?student_record - student_record ?study_plan_instance - study_plan_instance ?prerequisite_requirement - prerequisite_requirement ?term_session - term_session ?recovery_submission_document - recovery_submission_document)
    :precondition
      (and
        (student_submission_ready ?student_record)
        (study_plan_submission_ready ?study_plan_instance)
        (student_has_prerequisite_requirement ?student_record ?prerequisite_requirement)
        (study_plan_scheduled_in_term ?study_plan_instance ?term_session)
        (requirement_flagged_for_review ?prerequisite_requirement)
        (term_session_confirmed ?term_session)
        (student_requirement_verified ?student_record)
        (study_plan_session_confirmed ?study_plan_instance)
        (submission_document_draft ?recovery_submission_document)
      )
    :effect
      (and
        (submission_document_staged ?recovery_submission_document)
        (submission_document_contains_requirement ?recovery_submission_document ?prerequisite_requirement)
        (submission_document_contains_session ?recovery_submission_document ?term_session)
        (not
          (submission_document_draft ?recovery_submission_document)
        )
      )
  )
  (:action assemble_submission_document_with_support_requirement
    :parameters (?student_record - student_record ?study_plan_instance - study_plan_instance ?prerequisite_requirement - prerequisite_requirement ?term_session - term_session ?recovery_submission_document - recovery_submission_document)
    :precondition
      (and
        (student_submission_ready ?student_record)
        (study_plan_submission_ready ?study_plan_instance)
        (student_has_prerequisite_requirement ?student_record ?prerequisite_requirement)
        (study_plan_scheduled_in_term ?study_plan_instance ?term_session)
        (requirement_support_allocated ?prerequisite_requirement)
        (term_session_confirmed ?term_session)
        (not
          (student_requirement_verified ?student_record)
        )
        (study_plan_session_confirmed ?study_plan_instance)
        (submission_document_draft ?recovery_submission_document)
      )
    :effect
      (and
        (submission_document_staged ?recovery_submission_document)
        (submission_document_contains_requirement ?recovery_submission_document ?prerequisite_requirement)
        (submission_document_contains_session ?recovery_submission_document ?term_session)
        (submission_document_requires_authorization ?recovery_submission_document)
        (not
          (submission_document_draft ?recovery_submission_document)
        )
      )
  )
  (:action assemble_submission_document_with_alternate_session
    :parameters (?student_record - student_record ?study_plan_instance - study_plan_instance ?prerequisite_requirement - prerequisite_requirement ?term_session - term_session ?recovery_submission_document - recovery_submission_document)
    :precondition
      (and
        (student_submission_ready ?student_record)
        (study_plan_submission_ready ?study_plan_instance)
        (student_has_prerequisite_requirement ?student_record ?prerequisite_requirement)
        (study_plan_scheduled_in_term ?study_plan_instance ?term_session)
        (requirement_flagged_for_review ?prerequisite_requirement)
        (term_session_provisionally_scheduled ?term_session)
        (student_requirement_verified ?student_record)
        (not
          (study_plan_session_confirmed ?study_plan_instance)
        )
        (submission_document_draft ?recovery_submission_document)
      )
    :effect
      (and
        (submission_document_staged ?recovery_submission_document)
        (submission_document_contains_requirement ?recovery_submission_document ?prerequisite_requirement)
        (submission_document_contains_session ?recovery_submission_document ?term_session)
        (submission_document_requires_justification ?recovery_submission_document)
        (not
          (submission_document_draft ?recovery_submission_document)
        )
      )
  )
  (:action assemble_submission_document_with_support_and_alternate_session
    :parameters (?student_record - student_record ?study_plan_instance - study_plan_instance ?prerequisite_requirement - prerequisite_requirement ?term_session - term_session ?recovery_submission_document - recovery_submission_document)
    :precondition
      (and
        (student_submission_ready ?student_record)
        (study_plan_submission_ready ?study_plan_instance)
        (student_has_prerequisite_requirement ?student_record ?prerequisite_requirement)
        (study_plan_scheduled_in_term ?study_plan_instance ?term_session)
        (requirement_support_allocated ?prerequisite_requirement)
        (term_session_provisionally_scheduled ?term_session)
        (not
          (student_requirement_verified ?student_record)
        )
        (not
          (study_plan_session_confirmed ?study_plan_instance)
        )
        (submission_document_draft ?recovery_submission_document)
      )
    :effect
      (and
        (submission_document_staged ?recovery_submission_document)
        (submission_document_contains_requirement ?recovery_submission_document ?prerequisite_requirement)
        (submission_document_contains_session ?recovery_submission_document ?term_session)
        (submission_document_requires_authorization ?recovery_submission_document)
        (submission_document_requires_justification ?recovery_submission_document)
        (not
          (submission_document_draft ?recovery_submission_document)
        )
      )
  )
  (:action finalize_submission_document
    :parameters (?recovery_submission_document - recovery_submission_document ?student_record - student_record ?remedial_course_offering - remedial_course_offering)
    :precondition
      (and
        (submission_document_staged ?recovery_submission_document)
        (student_submission_ready ?student_record)
        (recovery_subject_assigned_remedial_offering ?student_record ?remedial_course_offering)
        (not
          (submission_document_ready_for_evidence ?recovery_submission_document)
        )
      )
    :effect (submission_document_ready_for_evidence ?recovery_submission_document)
  )
  (:action ingest_and_attach_evidence_artifact
    :parameters (?case_handler - case_handler ?evidence_artifact - evidence_artifact ?recovery_submission_document - recovery_submission_document)
    :precondition
      (and
        (recovery_case_validated ?case_handler)
        (handler_has_recovery_submission_document ?case_handler ?recovery_submission_document)
        (handler_has_evidence_artifact ?case_handler ?evidence_artifact)
        (evidence_artifact_available ?evidence_artifact)
        (submission_document_staged ?recovery_submission_document)
        (submission_document_ready_for_evidence ?recovery_submission_document)
        (not
          (evidence_artifact_verified ?evidence_artifact)
        )
      )
    :effect
      (and
        (evidence_artifact_verified ?evidence_artifact)
        (evidence_attached_to_document ?evidence_artifact ?recovery_submission_document)
        (not
          (evidence_artifact_available ?evidence_artifact)
        )
      )
  )
  (:action verify_attached_evidence_artifact
    :parameters (?case_handler - case_handler ?evidence_artifact - evidence_artifact ?recovery_submission_document - recovery_submission_document ?remedial_course_offering - remedial_course_offering)
