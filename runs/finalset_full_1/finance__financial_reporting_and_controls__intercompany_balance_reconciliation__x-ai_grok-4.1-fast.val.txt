--- domain.pddl (as passed to VAL) ---
(define (domain intercompany_balance_reconciliation)
  (:requirements :strips :typing :negative-preconditions)
  (:types domain_entity - object operational_artifact - domain_entity document_group - domain_entity key_group - domain_entity reconciliation_subject_root - domain_entity reconciliation_subject - reconciliation_subject_root match_token - operational_artifact transaction_extract - operational_artifact reviewer_role - operational_artifact signatory - operational_artifact checklist - operational_artifact investigation_record - operational_artifact control_procedure - operational_artifact audit_request - operational_artifact supporting_evidence - document_group document_set - document_group audit_note - document_group reconciliation_key_a - key_group reconciliation_key_b - key_group reconciliation_package - key_group ledger_account_group - reconciliation_subject workstream_group - reconciliation_subject ledger_account_a - ledger_account_group ledger_account_b - ledger_account_group reconciliation_workstream - workstream_group)

  (:predicates
    (recon_subject_initialized ?reconciliation_subject - reconciliation_subject)
    (recon_subject_ready_for_review ?reconciliation_subject - reconciliation_subject)
    (match_token_assigned ?reconciliation_subject - reconciliation_subject)
    (recon_subject_closed ?reconciliation_subject - reconciliation_subject)
    (recon_subject_certified ?reconciliation_subject - reconciliation_subject)
    (recon_subject_posting_authorized ?reconciliation_subject - reconciliation_subject)
    (match_token_available ?match_token - match_token)
    (allocated_match_token ?reconciliation_subject - reconciliation_subject ?match_token - match_token)
    (transaction_extract_available ?transaction_extract - transaction_extract)
    (extract_attached_to_subject ?reconciliation_subject - reconciliation_subject ?transaction_extract - transaction_extract)
    (reviewer_available ?reviewer_role - reviewer_role)
    (assigned_reviewer ?reconciliation_subject - reconciliation_subject ?reviewer_role - reviewer_role)
    (supporting_evidence_available ?supporting_evidence - supporting_evidence)
    (ledger_account_a_evidence_attached ?ledger_account_a - ledger_account_a ?supporting_evidence - supporting_evidence)
    (ledger_account_b_evidence_attached ?ledger_account_b - ledger_account_b ?supporting_evidence - supporting_evidence)
    (ledger_account_a_has_key ?ledger_account_a - ledger_account_a ?reconciliation_key_a - reconciliation_key_a)
    (reconciliation_key_a_ready ?reconciliation_key_a - reconciliation_key_a)
    (reconciliation_key_a_evidence_attached ?reconciliation_key_a - reconciliation_key_a)
    (ledger_account_a_validated ?ledger_account_a - ledger_account_a)
    (ledger_account_b_has_key ?ledger_account_b - ledger_account_b ?reconciliation_key_b - reconciliation_key_b)
    (reconciliation_key_b_ready ?reconciliation_key_b - reconciliation_key_b)
    (reconciliation_key_b_evidence_attached ?reconciliation_key_b - reconciliation_key_b)
    (ledger_account_b_validated ?ledger_account_b - ledger_account_b)
    (reconciliation_package_available ?reconciliation_package - reconciliation_package)
    (reconciliation_package_created ?reconciliation_package - reconciliation_package)
    (package_contains_key_a ?reconciliation_package - reconciliation_package ?reconciliation_key_a - reconciliation_key_a)
    (package_contains_key_b ?reconciliation_package - reconciliation_package ?reconciliation_key_b - reconciliation_key_b)
    (package_side_a_documents_attached ?reconciliation_package - reconciliation_package)
    (package_side_b_documents_attached ?reconciliation_package - reconciliation_package)
    (package_ready_for_evidence_attachment ?reconciliation_package - reconciliation_package)
    (workstream_links_ledger_a ?reconciliation_workstream - reconciliation_workstream ?ledger_account_a - ledger_account_a)
    (workstream_links_ledger_b ?reconciliation_workstream - reconciliation_workstream ?ledger_account_b - ledger_account_b)
    (workstream_contains_package ?reconciliation_workstream - reconciliation_workstream ?reconciliation_package - reconciliation_package)
    (document_set_available ?document_set - document_set)
    (workstream_has_document_set ?reconciliation_workstream - reconciliation_workstream ?document_set - document_set)
    (document_set_locked ?document_set - document_set)
    (document_set_linked_to_package ?document_set - document_set ?reconciliation_package - reconciliation_package)
    (workstream_documents_locked ?reconciliation_workstream - reconciliation_workstream)
    (workstream_documents_verified ?reconciliation_workstream - reconciliation_workstream)
    (workstream_ready_for_approval ?reconciliation_workstream - reconciliation_workstream)
    (workstream_signatory_assigned ?reconciliation_workstream - reconciliation_workstream)
    (workstream_signatory_confirmed ?reconciliation_workstream - reconciliation_workstream)
    (workstream_signatory_verified ?reconciliation_workstream - reconciliation_workstream)
    (workstream_controls_completed ?reconciliation_workstream - reconciliation_workstream)
    (audit_note_available ?audit_note - audit_note)
    (workstream_has_audit_note ?reconciliation_workstream - reconciliation_workstream ?audit_note - audit_note)
    (workstream_audit_requested ?reconciliation_workstream - reconciliation_workstream)
    (workstream_audit_assigned ?reconciliation_workstream - reconciliation_workstream)
    (workstream_audit_completed ?reconciliation_workstream - reconciliation_workstream)
    (signatory_available ?signatory - signatory)
    (workstream_assigned_signatory ?reconciliation_workstream - reconciliation_workstream ?signatory - signatory)
    (checklist_available ?checklist - checklist)
    (workstream_has_checklist ?reconciliation_workstream - reconciliation_workstream ?checklist - checklist)
    (control_procedure_available ?control_procedure - control_procedure)
    (workstream_has_control_procedure ?reconciliation_workstream - reconciliation_workstream ?control_procedure - control_procedure)
    (audit_request_available ?audit_request - audit_request)
    (workstream_has_audit_request ?reconciliation_workstream - reconciliation_workstream ?audit_request - audit_request)
    (investigation_record_available ?investigation_record - investigation_record)
    (recon_subject_has_investigation_record ?reconciliation_subject - reconciliation_subject ?investigation_record - investigation_record)
    (ledger_account_a_review_completed ?ledger_account_a - ledger_account_a)
    (ledger_account_b_review_completed ?ledger_account_b - ledger_account_b)
    (workstream_finalized ?reconciliation_workstream - reconciliation_workstream)
  )
  (:action initialize_reconciliation_subject
    :parameters (?reconciliation_subject - reconciliation_subject)
    :precondition
      (and
        (not
          (recon_subject_initialized ?reconciliation_subject)
        )
        (not
          (recon_subject_closed ?reconciliation_subject)
        )
      )
    :effect (recon_subject_initialized ?reconciliation_subject)
  )
  (:action allocate_match_token
    :parameters (?reconciliation_subject - reconciliation_subject ?match_token - match_token)
    :precondition
      (and
        (recon_subject_initialized ?reconciliation_subject)
        (not
          (match_token_assigned ?reconciliation_subject)
        )
        (match_token_available ?match_token)
      )
    :effect
      (and
        (match_token_assigned ?reconciliation_subject)
        (allocated_match_token ?reconciliation_subject ?match_token)
        (not
          (match_token_available ?match_token)
        )
      )
  )
  (:action ingest_transaction_extract
    :parameters (?reconciliation_subject - reconciliation_subject ?transaction_extract - transaction_extract)
    :precondition
      (and
        (recon_subject_initialized ?reconciliation_subject)
        (match_token_assigned ?reconciliation_subject)
        (transaction_extract_available ?transaction_extract)
      )
    :effect
      (and
        (extract_attached_to_subject ?reconciliation_subject ?transaction_extract)
        (not
          (transaction_extract_available ?transaction_extract)
        )
      )
  )
  (:action mark_subject_ready_for_review
    :parameters (?reconciliation_subject - reconciliation_subject ?transaction_extract - transaction_extract)
    :precondition
      (and
        (recon_subject_initialized ?reconciliation_subject)
        (match_token_assigned ?reconciliation_subject)
        (extract_attached_to_subject ?reconciliation_subject ?transaction_extract)
        (not
          (recon_subject_ready_for_review ?reconciliation_subject)
        )
      )
    :effect (recon_subject_ready_for_review ?reconciliation_subject)
  )
  (:action release_transaction_extract
    :parameters (?reconciliation_subject - reconciliation_subject ?transaction_extract - transaction_extract)
    :precondition
      (and
        (extract_attached_to_subject ?reconciliation_subject ?transaction_extract)
      )
    :effect
      (and
        (transaction_extract_available ?transaction_extract)
        (not
          (extract_attached_to_subject ?reconciliation_subject ?transaction_extract)
        )
      )
  )
  (:action assign_reviewer_to_subject
    :parameters (?reconciliation_subject - reconciliation_subject ?reviewer_role - reviewer_role)
    :precondition
      (and
        (recon_subject_ready_for_review ?reconciliation_subject)
        (reviewer_available ?reviewer_role)
      )
    :effect
      (and
        (assigned_reviewer ?reconciliation_subject ?reviewer_role)
        (not
          (reviewer_available ?reviewer_role)
        )
      )
  )
  (:action unassign_reviewer_from_subject
    :parameters (?reconciliation_subject - reconciliation_subject ?reviewer_role - reviewer_role)
    :precondition
      (and
        (assigned_reviewer ?reconciliation_subject ?reviewer_role)
      )
    :effect
      (and
        (reviewer_available ?reviewer_role)
        (not
          (assigned_reviewer ?reconciliation_subject ?reviewer_role)
        )
      )
  )
  (:action apply_control_procedure_to_workstream
    :parameters (?reconciliation_workstream - reconciliation_workstream ?control_procedure - control_procedure)
    :precondition
      (and
        (recon_subject_ready_for_review ?reconciliation_workstream)
        (control_procedure_available ?control_procedure)
      )
    :effect
      (and
        (workstream_has_control_procedure ?reconciliation_workstream ?control_procedure)
        (not
          (control_procedure_available ?control_procedure)
        )
      )
  )
  (:action remove_control_procedure_from_workstream
    :parameters (?reconciliation_workstream - reconciliation_workstream ?control_procedure - control_procedure)
    :precondition
      (and
        (workstream_has_control_procedure ?reconciliation_workstream ?control_procedure)
      )
    :effect
      (and
        (control_procedure_available ?control_procedure)
        (not
          (workstream_has_control_procedure ?reconciliation_workstream ?control_procedure)
        )
      )
  )
  (:action assign_audit_request_to_workstream
    :parameters (?reconciliation_workstream - reconciliation_workstream ?audit_request - audit_request)
    :precondition
      (and
        (recon_subject_ready_for_review ?reconciliation_workstream)
        (audit_request_available ?audit_request)
      )
    :effect
      (and
        (workstream_has_audit_request ?reconciliation_workstream ?audit_request)
        (not
          (audit_request_available ?audit_request)
        )
      )
  )
  (:action remove_audit_request_from_workstream
    :parameters (?reconciliation_workstream - reconciliation_workstream ?audit_request - audit_request)
    :precondition
      (and
        (workstream_has_audit_request ?reconciliation_workstream ?audit_request)
      )
    :effect
      (and
        (audit_request_available ?audit_request)
        (not
          (workstream_has_audit_request ?reconciliation_workstream ?audit_request)
        )
      )
  )
  (:action flag_recon_key_a_ready
    :parameters (?ledger_account_a - ledger_account_a ?reconciliation_key_a - reconciliation_key_a ?transaction_extract - transaction_extract)
    :precondition
      (and
        (recon_subject_ready_for_review ?ledger_account_a)
        (extract_attached_to_subject ?ledger_account_a ?transaction_extract)
        (ledger_account_a_has_key ?ledger_account_a ?reconciliation_key_a)
        (not
          (reconciliation_key_a_ready ?reconciliation_key_a)
        )
        (not
          (reconciliation_key_a_evidence_attached ?reconciliation_key_a)
        )
      )
    :effect (reconciliation_key_a_ready ?reconciliation_key_a)
  )
  (:action confirm_ledger_a_validation
    :parameters (?ledger_account_a - ledger_account_a ?reconciliation_key_a - reconciliation_key_a ?reviewer_role - reviewer_role)
    :precondition
      (and
        (recon_subject_ready_for_review ?ledger_account_a)
        (assigned_reviewer ?ledger_account_a ?reviewer_role)
        (ledger_account_a_has_key ?ledger_account_a ?reconciliation_key_a)
        (reconciliation_key_a_ready ?reconciliation_key_a)
        (not
          (ledger_account_a_review_completed ?ledger_account_a)
        )
      )
    :effect
      (and
        (ledger_account_a_review_completed ?ledger_account_a)
        (ledger_account_a_validated ?ledger_account_a)
      )
  )
  (:action attach_supporting_evidence_to_ledger_a
    :parameters (?ledger_account_a - ledger_account_a ?reconciliation_key_a - reconciliation_key_a ?supporting_evidence - supporting_evidence)
    :precondition
      (and
        (recon_subject_ready_for_review ?ledger_account_a)
        (ledger_account_a_has_key ?ledger_account_a ?reconciliation_key_a)
        (supporting_evidence_available ?supporting_evidence)
        (not
          (ledger_account_a_review_completed ?ledger_account_a)
        )
      )
    :effect
      (and
        (reconciliation_key_a_evidence_attached ?reconciliation_key_a)
        (ledger_account_a_review_completed ?ledger_account_a)
        (ledger_account_a_evidence_attached ?ledger_account_a ?supporting_evidence)
        (not
          (supporting_evidence_available ?supporting_evidence)
        )
      )
  )
  (:action reconcile_ledger_a_with_evidence
    :parameters (?ledger_account_a - ledger_account_a ?reconciliation_key_a - reconciliation_key_a ?transaction_extract - transaction_extract ?supporting_evidence - supporting_evidence)
    :precondition
      (and
        (recon_subject_ready_for_review ?ledger_account_a)
        (extract_attached_to_subject ?ledger_account_a ?transaction_extract)
        (ledger_account_a_has_key ?ledger_account_a ?reconciliation_key_a)
        (reconciliation_key_a_evidence_attached ?reconciliation_key_a)
        (ledger_account_a_evidence_attached ?ledger_account_a ?supporting_evidence)
        (not
          (ledger_account_a_validated ?ledger_account_a)
        )
      )
    :effect
      (and
        (reconciliation_key_a_ready ?reconciliation_key_a)
        (ledger_account_a_validated ?ledger_account_a)
        (supporting_evidence_available ?supporting_evidence)
        (not
          (ledger_account_a_evidence_attached ?ledger_account_a ?supporting_evidence)
        )
      )
  )
  (:action flag_recon_key_b_ready
    :parameters (?ledger_account_b - ledger_account_b ?reconciliation_key_b - reconciliation_key_b ?transaction_extract - transaction_extract)
    :precondition
      (and
        (recon_subject_ready_for_review ?ledger_account_b)
        (extract_attached_to_subject ?ledger_account_b ?transaction_extract)
        (ledger_account_b_has_key ?ledger_account_b ?reconciliation_key_b)
        (not
          (reconciliation_key_b_ready ?reconciliation_key_b)
        )
        (not
          (reconciliation_key_b_evidence_attached ?reconciliation_key_b)
        )
      )
    :effect (reconciliation_key_b_ready ?reconciliation_key_b)
  )
  (:action confirm_ledger_b_validation
    :parameters (?ledger_account_b - ledger_account_b ?reconciliation_key_b - reconciliation_key_b ?reviewer_role - reviewer_role)
    :precondition
      (and
        (recon_subject_ready_for_review ?ledger_account_b)
        (assigned_reviewer ?ledger_account_b ?reviewer_role)
        (ledger_account_b_has_key ?ledger_account_b ?reconciliation_key_b)
        (reconciliation_key_b_ready ?reconciliation_key_b)
        (not
          (ledger_account_b_review_completed ?ledger_account_b)
        )
      )
    :effect
      (and
        (ledger_account_b_review_completed ?ledger_account_b)
        (ledger_account_b_validated ?ledger_account_b)
      )
  )
  (:action attach_supporting_evidence_to_ledger_b
    :parameters (?ledger_account_b - ledger_account_b ?reconciliation_key_b - reconciliation_key_b ?supporting_evidence - supporting_evidence)
    :precondition
      (and
        (recon_subject_ready_for_review ?ledger_account_b)
        (ledger_account_b_has_key ?ledger_account_b ?reconciliation_key_b)
        (supporting_evidence_available ?supporting_evidence)
        (not
          (ledger_account_b_review_completed ?ledger_account_b)
        )
      )
    :effect
      (and
        (reconciliation_key_b_evidence_attached ?reconciliation_key_b)
        (ledger_account_b_review_completed ?ledger_account_b)
        (ledger_account_b_evidence_attached ?ledger_account_b ?supporting_evidence)
        (not
          (supporting_evidence_available ?supporting_evidence)
        )
      )
  )
  (:action reconcile_ledger_b_with_evidence
    :parameters (?ledger_account_b - ledger_account_b ?reconciliation_key_b - reconciliation_key_b ?transaction_extract - transaction_extract ?supporting_evidence - supporting_evidence)
    :precondition
      (and
        (recon_subject_ready_for_review ?ledger_account_b)
        (extract_attached_to_subject ?ledger_account_b ?transaction_extract)
        (ledger_account_b_has_key ?ledger_account_b ?reconciliation_key_b)
        (reconciliation_key_b_evidence_attached ?reconciliation_key_b)
        (ledger_account_b_evidence_attached ?ledger_account_b ?supporting_evidence)
        (not
          (ledger_account_b_validated ?ledger_account_b)
        )
      )
    :effect
      (and
        (reconciliation_key_b_ready ?reconciliation_key_b)
        (ledger_account_b_validated ?ledger_account_b)
        (supporting_evidence_available ?supporting_evidence)
        (not
          (ledger_account_b_evidence_attached ?ledger_account_b ?supporting_evidence)
        )
      )
  )
  (:action assemble_reconciliation_package_from_keys
    :parameters (?ledger_account_a - ledger_account_a ?ledger_account_b - ledger_account_b ?reconciliation_key_a - reconciliation_key_a ?reconciliation_key_b - reconciliation_key_b ?reconciliation_package - reconciliation_package)
    :precondition
      (and
        (ledger_account_a_review_completed ?ledger_account_a)
        (ledger_account_b_review_completed ?ledger_account_b)
        (ledger_account_a_has_key ?ledger_account_a ?reconciliation_key_a)
        (ledger_account_b_has_key ?ledger_account_b ?reconciliation_key_b)
        (reconciliation_key_a_ready ?reconciliation_key_a)
        (reconciliation_key_b_ready ?reconciliation_key_b)
        (ledger_account_a_validated ?ledger_account_a)
        (ledger_account_b_validated ?ledger_account_b)
        (reconciliation_package_available ?reconciliation_package)
      )
    :effect
      (and
        (reconciliation_package_created ?reconciliation_package)
        (package_contains_key_a ?reconciliation_package ?reconciliation_key_a)
        (package_contains_key_b ?reconciliation_package ?reconciliation_key_b)
        (not
          (reconciliation_package_available ?reconciliation_package)
        )
      )
  )
  (:action assemble_reconciliation_package_with_side_a_docs
    :parameters (?ledger_account_a - ledger_account_a ?ledger_account_b - ledger_account_b ?reconciliation_key_a - reconciliation_key_a ?reconciliation_key_b - reconciliation_key_b ?reconciliation_package - reconciliation_package)
    :precondition
      (and
        (ledger_account_a_review_completed ?ledger_account_a)
        (ledger_account_b_review_completed ?ledger_account_b)
        (ledger_account_a_has_key ?ledger_account_a ?reconciliation_key_a)
        (ledger_account_b_has_key ?ledger_account_b ?reconciliation_key_b)
        (reconciliation_key_a_evidence_attached ?reconciliation_key_a)
        (reconciliation_key_b_ready ?reconciliation_key_b)
        (not
          (ledger_account_a_validated ?ledger_account_a)
        )
        (ledger_account_b_validated ?ledger_account_b)
        (reconciliation_package_available ?reconciliation_package)
      )
    :effect
      (and
        (reconciliation_package_created ?reconciliation_package)
        (package_contains_key_a ?reconciliation_package ?reconciliation_key_a)
        (package_contains_key_b ?reconciliation_package ?reconciliation_key_b)
        (package_side_a_documents_attached ?reconciliation_package)
        (not
          (reconciliation_package_available ?reconciliation_package)
        )
      )
  )
  (:action assemble_reconciliation_package_with_side_b_docs
    :parameters (?ledger_account_a - ledger_account_a ?ledger_account_b - ledger_account_b ?reconciliation_key_a - reconciliation_key_a ?reconciliation_key_b - reconciliation_key_b ?reconciliation_package - reconciliation_package)
    :precondition
      (and
        (ledger_account_a_review_completed ?ledger_account_a)
        (ledger_account_b_review_completed ?ledger_account_b)
        (ledger_account_a_has_key ?ledger_account_a ?reconciliation_key_a)
        (ledger_account_b_has_key ?ledger_account_b ?reconciliation_key_b)
        (reconciliation_key_a_ready ?reconciliation_key_a)
        (reconciliation_key_b_evidence_attached ?reconciliation_key_b)
        (ledger_account_a_validated ?ledger_account_a)
        (not
          (ledger_account_b_validated ?ledger_account_b)
        )
        (reconciliation_package_available ?reconciliation_package)
      )
    :effect
      (and
        (reconciliation_package_created ?reconciliation_package)
        (package_contains_key_a ?reconciliation_package ?reconciliation_key_a)
        (package_contains_key_b ?reconciliation_package ?reconciliation_key_b)
        (package_side_b_documents_attached ?reconciliation_package)
        (not
          (reconciliation_package_available ?reconciliation_package)
        )
      )
  )
  (:action assemble_reconciliation_package_with_both_docs
    :parameters (?ledger_account_a - ledger_account_a ?ledger_account_b - ledger_account_b ?reconciliation_key_a - reconciliation_key_a ?reconciliation_key_b - reconciliation_key_b ?reconciliation_package - reconciliation_package)
    :precondition
      (and
        (ledger_account_a_review_completed ?ledger_account_a)
        (ledger_account_b_review_completed ?ledger_account_b)
        (ledger_account_a_has_key ?ledger_account_a ?reconciliation_key_a)
        (ledger_account_b_has_key ?ledger_account_b ?reconciliation_key_b)
        (reconciliation_key_a_evidence_attached ?reconciliation_key_a)
        (reconciliation_key_b_evidence_attached ?reconciliation_key_b)
        (not
          (ledger_account_a_validated ?ledger_account_a)
        )
        (not
          (ledger_account_b_validated ?ledger_account_b)
        )
        (reconciliation_package_available ?reconciliation_package)
      )
    :effect
      (and
        (reconciliation_package_created ?reconciliation_package)
        (package_contains_key_a ?reconciliation_package ?reconciliation_key_a)
        (package_contains_key_b ?reconciliation_package ?reconciliation_key_b)
        (package_side_a_documents_attached ?reconciliation_package)
        (package_side_b_documents_attached ?reconciliation_package)
        (not
          (reconciliation_package_available ?reconciliation_package)
        )
      )
  )
  (:action mark_package_ready_for_evidence_attachment
    :parameters (?reconciliation_package - reconciliation_package ?ledger_account_a - ledger_account_a ?transaction_extract - transaction_extract)
    :precondition
      (and
        (reconciliation_package_created ?reconciliation_package)
        (ledger_account_a_review_completed ?ledger_account_a)
        (extract_attached_to_subject ?ledger_account_a ?transaction_extract)
        (not
          (package_ready_for_evidence_attachment ?reconciliation_package)
        )
      )
    :effect (package_ready_for_evidence_attachment ?reconciliation_package)
  )
  (:action attach_document_set_to_package
    :parameters (?reconciliation_workstream - reconciliation_workstream ?document_set - document_set ?reconciliation_package - reconciliation_package)
    :precondition
      (and
        (recon_subject_ready_for_review ?reconciliation_workstream)
        (workstream_contains_package ?reconciliation_workstream ?reconciliation_package)
        (workstream_has_document_set ?reconciliation_workstream ?document_set)
        (document_set_available ?document_set)
        (reconciliation_package_created ?reconciliation_package)
        (package_ready_for_evidence_attachment ?reconciliation_package)
        (not
          (document_set_locked ?document_set)
        )
      )
