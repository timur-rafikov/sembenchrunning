--- domain.pddl (as passed to VAL) ---
(define (domain carrier_capacity_shipment_assignment_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_type - object item_type - object domain_location_type - object fulfillment_entity_root - object fulfillment_entity - fulfillment_entity_root carrier_resource - resource_type order_line_item - resource_type warehouse_terminal - resource_type routing_attribute - resource_type special_handling_requirement - resource_type service_option - resource_type equipment_type - resource_type operational_attribute - resource_type packaging_unit - item_type piece_package - item_type contract_offer - item_type origin_time_slot - domain_location_type destination_time_slot - domain_location_type formed_shipment - domain_location_type flow_unit_category - fulfillment_entity consolidation_category - fulfillment_entity origin_fulfillment_unit - flow_unit_category destination_fulfillment_unit - flow_unit_category consolidation_shipment - consolidation_category)
  (:predicates
    (released_for_planning ?fulfillment_entity - fulfillment_entity)
    (validated_for_planning ?fulfillment_entity - fulfillment_entity)
    (entity_carrier_reserved ?fulfillment_entity - fulfillment_entity)
    (assignment_finalized ?fulfillment_entity - fulfillment_entity)
    (operationally_confirmed ?fulfillment_entity - fulfillment_entity)
    (assignment_token_consumed ?fulfillment_entity - fulfillment_entity)
    (carrier_available ?carrier_resource - carrier_resource)
    (assigned_to_carrier ?fulfillment_entity - fulfillment_entity ?carrier_resource - carrier_resource)
    (item_available ?order_line_item - order_line_item)
    (line_item_allocated ?fulfillment_entity - fulfillment_entity ?order_line_item - order_line_item)
    (terminal_available ?warehouse_terminal - warehouse_terminal)
    (assigned_terminal ?fulfillment_entity - fulfillment_entity ?warehouse_terminal - warehouse_terminal)
    (packaging_available ?packaging_unit - packaging_unit)
    (origin_has_packaging ?origin_fulfillment_unit - origin_fulfillment_unit ?packaging_unit - packaging_unit)
    (destination_has_packaging ?destination_fulfillment_unit - destination_fulfillment_unit ?packaging_unit - packaging_unit)
    (compatible_with_origin_slot ?origin_fulfillment_unit - origin_fulfillment_unit ?origin_time_slot - origin_time_slot)
    (origin_slot_on_hold ?origin_time_slot - origin_time_slot)
    (origin_slot_confirmed ?origin_time_slot - origin_time_slot)
    (origin_slot_confirmed_for_unit ?origin_fulfillment_unit - origin_fulfillment_unit)
    (compatible_with_destination_slot ?destination_fulfillment_unit - destination_fulfillment_unit ?destination_time_slot - destination_time_slot)
    (destination_slot_on_hold ?destination_time_slot - destination_time_slot)
    (destination_slot_confirmed ?destination_time_slot - destination_time_slot)
    (destination_slot_confirmed_for_unit ?destination_fulfillment_unit - destination_fulfillment_unit)
    (formed_shipment_available ?formed_shipment - formed_shipment)
    (formed_shipment_active ?formed_shipment - formed_shipment)
    (shipment_has_origin_slot ?formed_shipment - formed_shipment ?origin_time_slot - origin_time_slot)
    (shipment_has_destination_slot ?formed_shipment - formed_shipment ?destination_time_slot - destination_time_slot)
    (formed_shipment_origin_confirmed ?formed_shipment - formed_shipment)
    (formed_shipment_destination_confirmed ?formed_shipment - formed_shipment)
    (formed_shipment_manifest_ready ?formed_shipment - formed_shipment)
    (consolidation_contains_origin ?consolidation_shipment - consolidation_shipment ?origin_fulfillment_unit - origin_fulfillment_unit)
    (consolidation_contains_destination ?consolidation_shipment - consolidation_shipment ?destination_fulfillment_unit - destination_fulfillment_unit)
    (consolidation_linked_to_formed_shipment ?consolidation_shipment - consolidation_shipment ?formed_shipment - formed_shipment)
    (package_available ?piece_package - piece_package)
    (consolidation_has_package ?consolidation_shipment - consolidation_shipment ?piece_package - piece_package)
    (package_manifested ?piece_package - piece_package)
    (package_assigned_to_shipment ?piece_package - piece_package ?formed_shipment - formed_shipment)
    (consolidation_prepared ?consolidation_shipment - consolidation_shipment)
    (attributes_prepared ?consolidation_shipment - consolidation_shipment)
    (attributes_confirmed ?consolidation_shipment - consolidation_shipment)
    (routing_attribute_attached ?consolidation_shipment - consolidation_shipment)
    (routing_attribute_enforced ?consolidation_shipment - consolidation_shipment)
    (awaiting_special_handling_attachment ?consolidation_shipment - consolidation_shipment)
    (final_readiness_verified ?consolidation_shipment - consolidation_shipment)
    (contract_offer_available ?contract_offer - contract_offer)
    (consolidation_has_contract_offer ?consolidation_shipment - consolidation_shipment ?contract_offer - contract_offer)
    (contract_offer_applied ?consolidation_shipment - consolidation_shipment)
    (contract_acknowledged ?consolidation_shipment - consolidation_shipment)
    (contract_requirements_confirmed ?consolidation_shipment - consolidation_shipment)
    (routing_attribute_available ?routing_attribute - routing_attribute)
    (consolidation_has_routing_attribute ?consolidation_shipment - consolidation_shipment ?routing_attribute - routing_attribute)
    (special_handling_available ?special_handling_requirement - special_handling_requirement)
    (consolidation_has_special_handling ?consolidation_shipment - consolidation_shipment ?special_handling_requirement - special_handling_requirement)
    (equipment_type_available ?equipment_type - equipment_type)
    (equipment_assigned_to_consolidation ?consolidation_shipment - consolidation_shipment ?equipment_type - equipment_type)
    (operational_attribute_available ?operational_attribute - operational_attribute)
    (consolidation_has_operational_attribute ?consolidation_shipment - consolidation_shipment ?operational_attribute - operational_attribute)
    (service_option_available ?service_option - service_option)
    (entity_assigned_service_option ?fulfillment_entity - fulfillment_entity ?service_option - service_option)
    (origin_unit_ready ?origin_fulfillment_unit - origin_fulfillment_unit)
    (destination_unit_ready ?destination_fulfillment_unit - destination_fulfillment_unit)
    (finalization_recorded ?consolidation_shipment - consolidation_shipment)
  )
  (:action release_fulfillment_entity
    :parameters (?fulfillment_entity - fulfillment_entity)
    :precondition
      (and
        (not
          (released_for_planning ?fulfillment_entity)
        )
        (not
          (assignment_finalized ?fulfillment_entity)
        )
      )
    :effect (released_for_planning ?fulfillment_entity)
  )
  (:action reserve_carrier_for_entity
    :parameters (?fulfillment_entity - fulfillment_entity ?carrier_resource - carrier_resource)
    :precondition
      (and
        (released_for_planning ?fulfillment_entity)
        (not
          (entity_carrier_reserved ?fulfillment_entity)
        )
        (carrier_available ?carrier_resource)
      )
    :effect
      (and
        (entity_carrier_reserved ?fulfillment_entity)
        (assigned_to_carrier ?fulfillment_entity ?carrier_resource)
        (not
          (carrier_available ?carrier_resource)
        )
      )
  )
  (:action allocate_line_item_to_entity
    :parameters (?fulfillment_entity - fulfillment_entity ?order_line_item - order_line_item)
    :precondition
      (and
        (released_for_planning ?fulfillment_entity)
        (entity_carrier_reserved ?fulfillment_entity)
        (item_available ?order_line_item)
      )
    :effect
      (and
        (line_item_allocated ?fulfillment_entity ?order_line_item)
        (not
          (item_available ?order_line_item)
        )
      )
  )
  (:action confirm_item_allocation
    :parameters (?fulfillment_entity - fulfillment_entity ?order_line_item - order_line_item)
    :precondition
      (and
        (released_for_planning ?fulfillment_entity)
        (entity_carrier_reserved ?fulfillment_entity)
        (line_item_allocated ?fulfillment_entity ?order_line_item)
        (not
          (validated_for_planning ?fulfillment_entity)
        )
      )
    :effect (validated_for_planning ?fulfillment_entity)
  )
  (:action rollback_item_allocation
    :parameters (?fulfillment_entity - fulfillment_entity ?order_line_item - order_line_item)
    :precondition
      (and
        (line_item_allocated ?fulfillment_entity ?order_line_item)
      )
    :effect
      (and
        (item_available ?order_line_item)
        (not
          (line_item_allocated ?fulfillment_entity ?order_line_item)
        )
      )
  )
  (:action assign_terminal_to_entity
    :parameters (?fulfillment_entity - fulfillment_entity ?warehouse_terminal - warehouse_terminal)
    :precondition
      (and
        (validated_for_planning ?fulfillment_entity)
        (terminal_available ?warehouse_terminal)
      )
    :effect
      (and
        (assigned_terminal ?fulfillment_entity ?warehouse_terminal)
        (not
          (terminal_available ?warehouse_terminal)
        )
      )
  )
  (:action release_terminal_assignment
    :parameters (?fulfillment_entity - fulfillment_entity ?warehouse_terminal - warehouse_terminal)
    :precondition
      (and
        (assigned_terminal ?fulfillment_entity ?warehouse_terminal)
      )
    :effect
      (and
        (terminal_available ?warehouse_terminal)
        (not
          (assigned_terminal ?fulfillment_entity ?warehouse_terminal)
        )
      )
  )
  (:action assign_equipment_to_consolidation
    :parameters (?consolidation_shipment - consolidation_shipment ?equipment_type - equipment_type)
    :precondition
      (and
        (validated_for_planning ?consolidation_shipment)
        (equipment_type_available ?equipment_type)
      )
    :effect
      (and
        (equipment_assigned_to_consolidation ?consolidation_shipment ?equipment_type)
        (not
          (equipment_type_available ?equipment_type)
        )
      )
  )
  (:action release_equipment_from_consolidation
    :parameters (?consolidation_shipment - consolidation_shipment ?equipment_type - equipment_type)
    :precondition
      (and
        (equipment_assigned_to_consolidation ?consolidation_shipment ?equipment_type)
      )
    :effect
      (and
        (equipment_type_available ?equipment_type)
        (not
          (equipment_assigned_to_consolidation ?consolidation_shipment ?equipment_type)
        )
      )
  )
  (:action assign_operational_attribute_to_consolidation
    :parameters (?consolidation_shipment - consolidation_shipment ?operational_attribute - operational_attribute)
    :precondition
      (and
        (validated_for_planning ?consolidation_shipment)
        (operational_attribute_available ?operational_attribute)
      )
    :effect
      (and
        (consolidation_has_operational_attribute ?consolidation_shipment ?operational_attribute)
        (not
          (operational_attribute_available ?operational_attribute)
        )
      )
  )
  (:action release_operational_attribute_from_consolidation
    :parameters (?consolidation_shipment - consolidation_shipment ?operational_attribute - operational_attribute)
    :precondition
      (and
        (consolidation_has_operational_attribute ?consolidation_shipment ?operational_attribute)
      )
    :effect
      (and
        (operational_attribute_available ?operational_attribute)
        (not
          (consolidation_has_operational_attribute ?consolidation_shipment ?operational_attribute)
        )
      )
  )
  (:action hold_origin_time_slot
    :parameters (?origin_fulfillment_unit - origin_fulfillment_unit ?origin_time_slot - origin_time_slot ?order_line_item - order_line_item)
    :precondition
      (and
        (validated_for_planning ?origin_fulfillment_unit)
        (line_item_allocated ?origin_fulfillment_unit ?order_line_item)
        (compatible_with_origin_slot ?origin_fulfillment_unit ?origin_time_slot)
        (not
          (origin_slot_on_hold ?origin_time_slot)
        )
        (not
          (origin_slot_confirmed ?origin_time_slot)
        )
      )
    :effect (origin_slot_on_hold ?origin_time_slot)
  )
  (:action confirm_origin_slot_for_unit
    :parameters (?origin_fulfillment_unit - origin_fulfillment_unit ?origin_time_slot - origin_time_slot ?warehouse_terminal - warehouse_terminal)
    :precondition
      (and
        (validated_for_planning ?origin_fulfillment_unit)
        (assigned_terminal ?origin_fulfillment_unit ?warehouse_terminal)
        (compatible_with_origin_slot ?origin_fulfillment_unit ?origin_time_slot)
        (origin_slot_on_hold ?origin_time_slot)
        (not
          (origin_unit_ready ?origin_fulfillment_unit)
        )
      )
    :effect
      (and
        (origin_unit_ready ?origin_fulfillment_unit)
        (origin_slot_confirmed_for_unit ?origin_fulfillment_unit)
      )
  )
  (:action reserve_origin_slot_and_assign_packaging
    :parameters (?origin_fulfillment_unit - origin_fulfillment_unit ?origin_time_slot - origin_time_slot ?packaging_unit - packaging_unit)
    :precondition
      (and
        (validated_for_planning ?origin_fulfillment_unit)
        (compatible_with_origin_slot ?origin_fulfillment_unit ?origin_time_slot)
        (packaging_available ?packaging_unit)
        (not
          (origin_unit_ready ?origin_fulfillment_unit)
        )
      )
    :effect
      (and
        (origin_slot_confirmed ?origin_time_slot)
        (origin_unit_ready ?origin_fulfillment_unit)
        (origin_has_packaging ?origin_fulfillment_unit ?packaging_unit)
        (not
          (packaging_available ?packaging_unit)
        )
      )
  )
  (:action finalize_origin_slot_reservation
    :parameters (?origin_fulfillment_unit - origin_fulfillment_unit ?origin_time_slot - origin_time_slot ?order_line_item - order_line_item ?packaging_unit - packaging_unit)
    :precondition
      (and
        (validated_for_planning ?origin_fulfillment_unit)
        (line_item_allocated ?origin_fulfillment_unit ?order_line_item)
        (compatible_with_origin_slot ?origin_fulfillment_unit ?origin_time_slot)
        (origin_slot_confirmed ?origin_time_slot)
        (origin_has_packaging ?origin_fulfillment_unit ?packaging_unit)
        (not
          (origin_slot_confirmed_for_unit ?origin_fulfillment_unit)
        )
      )
    :effect
      (and
        (origin_slot_on_hold ?origin_time_slot)
        (origin_slot_confirmed_for_unit ?origin_fulfillment_unit)
        (packaging_available ?packaging_unit)
        (not
          (origin_has_packaging ?origin_fulfillment_unit ?packaging_unit)
        )
      )
  )
  (:action hold_destination_time_slot
    :parameters (?destination_fulfillment_unit - destination_fulfillment_unit ?destination_time_slot - destination_time_slot ?order_line_item - order_line_item)
    :precondition
      (and
        (validated_for_planning ?destination_fulfillment_unit)
        (line_item_allocated ?destination_fulfillment_unit ?order_line_item)
        (compatible_with_destination_slot ?destination_fulfillment_unit ?destination_time_slot)
        (not
          (destination_slot_on_hold ?destination_time_slot)
        )
        (not
          (destination_slot_confirmed ?destination_time_slot)
        )
      )
    :effect (destination_slot_on_hold ?destination_time_slot)
  )
  (:action confirm_destination_slot_for_unit
    :parameters (?destination_fulfillment_unit - destination_fulfillment_unit ?destination_time_slot - destination_time_slot ?warehouse_terminal - warehouse_terminal)
    :precondition
      (and
        (validated_for_planning ?destination_fulfillment_unit)
        (assigned_terminal ?destination_fulfillment_unit ?warehouse_terminal)
        (compatible_with_destination_slot ?destination_fulfillment_unit ?destination_time_slot)
        (destination_slot_on_hold ?destination_time_slot)
        (not
          (destination_unit_ready ?destination_fulfillment_unit)
        )
      )
    :effect
      (and
        (destination_unit_ready ?destination_fulfillment_unit)
        (destination_slot_confirmed_for_unit ?destination_fulfillment_unit)
      )
  )
  (:action reserve_destination_slot_and_assign_packaging
    :parameters (?destination_fulfillment_unit - destination_fulfillment_unit ?destination_time_slot - destination_time_slot ?packaging_unit - packaging_unit)
    :precondition
      (and
        (validated_for_planning ?destination_fulfillment_unit)
        (compatible_with_destination_slot ?destination_fulfillment_unit ?destination_time_slot)
        (packaging_available ?packaging_unit)
        (not
          (destination_unit_ready ?destination_fulfillment_unit)
        )
      )
    :effect
      (and
        (destination_slot_confirmed ?destination_time_slot)
        (destination_unit_ready ?destination_fulfillment_unit)
        (destination_has_packaging ?destination_fulfillment_unit ?packaging_unit)
        (not
          (packaging_available ?packaging_unit)
        )
      )
  )
  (:action finalize_destination_slot_reservation
    :parameters (?destination_fulfillment_unit - destination_fulfillment_unit ?destination_time_slot - destination_time_slot ?order_line_item - order_line_item ?packaging_unit - packaging_unit)
    :precondition
      (and
        (validated_for_planning ?destination_fulfillment_unit)
        (line_item_allocated ?destination_fulfillment_unit ?order_line_item)
        (compatible_with_destination_slot ?destination_fulfillment_unit ?destination_time_slot)
        (destination_slot_confirmed ?destination_time_slot)
        (destination_has_packaging ?destination_fulfillment_unit ?packaging_unit)
        (not
          (destination_slot_confirmed_for_unit ?destination_fulfillment_unit)
        )
      )
    :effect
      (and
        (destination_slot_on_hold ?destination_time_slot)
        (destination_slot_confirmed_for_unit ?destination_fulfillment_unit)
        (packaging_available ?packaging_unit)
        (not
          (destination_has_packaging ?destination_fulfillment_unit ?packaging_unit)
        )
      )
  )
  (:action assemble_formed_shipment
    :parameters (?origin_fulfillment_unit - origin_fulfillment_unit ?destination_fulfillment_unit - destination_fulfillment_unit ?origin_time_slot - origin_time_slot ?destination_time_slot - destination_time_slot ?formed_shipment - formed_shipment)
    :precondition
      (and
        (origin_unit_ready ?origin_fulfillment_unit)
        (destination_unit_ready ?destination_fulfillment_unit)
        (compatible_with_origin_slot ?origin_fulfillment_unit ?origin_time_slot)
        (compatible_with_destination_slot ?destination_fulfillment_unit ?destination_time_slot)
        (origin_slot_on_hold ?origin_time_slot)
        (destination_slot_on_hold ?destination_time_slot)
        (origin_slot_confirmed_for_unit ?origin_fulfillment_unit)
        (destination_slot_confirmed_for_unit ?destination_fulfillment_unit)
        (formed_shipment_available ?formed_shipment)
      )
    :effect
      (and
        (formed_shipment_active ?formed_shipment)
        (shipment_has_origin_slot ?formed_shipment ?origin_time_slot)
        (shipment_has_destination_slot ?formed_shipment ?destination_time_slot)
        (not
          (formed_shipment_available ?formed_shipment)
        )
      )
  )
  (:action assemble_formed_shipment_origin_confirmed
    :parameters (?origin_fulfillment_unit - origin_fulfillment_unit ?destination_fulfillment_unit - destination_fulfillment_unit ?origin_time_slot - origin_time_slot ?destination_time_slot - destination_time_slot ?formed_shipment - formed_shipment)
    :precondition
      (and
        (origin_unit_ready ?origin_fulfillment_unit)
        (destination_unit_ready ?destination_fulfillment_unit)
        (compatible_with_origin_slot ?origin_fulfillment_unit ?origin_time_slot)
        (compatible_with_destination_slot ?destination_fulfillment_unit ?destination_time_slot)
        (origin_slot_confirmed ?origin_time_slot)
        (destination_slot_on_hold ?destination_time_slot)
        (not
          (origin_slot_confirmed_for_unit ?origin_fulfillment_unit)
        )
        (destination_slot_confirmed_for_unit ?destination_fulfillment_unit)
        (formed_shipment_available ?formed_shipment)
      )
    :effect
      (and
        (formed_shipment_active ?formed_shipment)
        (shipment_has_origin_slot ?formed_shipment ?origin_time_slot)
        (shipment_has_destination_slot ?formed_shipment ?destination_time_slot)
        (formed_shipment_origin_confirmed ?formed_shipment)
        (not
          (formed_shipment_available ?formed_shipment)
        )
      )
  )
  (:action assemble_formed_shipment_destination_confirmed
    :parameters (?origin_fulfillment_unit - origin_fulfillment_unit ?destination_fulfillment_unit - destination_fulfillment_unit ?origin_time_slot - origin_time_slot ?destination_time_slot - destination_time_slot ?formed_shipment - formed_shipment)
    :precondition
      (and
        (origin_unit_ready ?origin_fulfillment_unit)
        (destination_unit_ready ?destination_fulfillment_unit)
        (compatible_with_origin_slot ?origin_fulfillment_unit ?origin_time_slot)
        (compatible_with_destination_slot ?destination_fulfillment_unit ?destination_time_slot)
        (origin_slot_on_hold ?origin_time_slot)
        (destination_slot_confirmed ?destination_time_slot)
        (origin_slot_confirmed_for_unit ?origin_fulfillment_unit)
        (not
          (destination_slot_confirmed_for_unit ?destination_fulfillment_unit)
        )
        (formed_shipment_available ?formed_shipment)
      )
    :effect
      (and
        (formed_shipment_active ?formed_shipment)
        (shipment_has_origin_slot ?formed_shipment ?origin_time_slot)
        (shipment_has_destination_slot ?formed_shipment ?destination_time_slot)
        (formed_shipment_destination_confirmed ?formed_shipment)
        (not
          (formed_shipment_available ?formed_shipment)
        )
      )
  )
  (:action assemble_formed_shipment_both_confirmed
    :parameters (?origin_fulfillment_unit - origin_fulfillment_unit ?destination_fulfillment_unit - destination_fulfillment_unit ?origin_time_slot - origin_time_slot ?destination_time_slot - destination_time_slot ?formed_shipment - formed_shipment)
    :precondition
      (and
        (origin_unit_ready ?origin_fulfillment_unit)
        (destination_unit_ready ?destination_fulfillment_unit)
        (compatible_with_origin_slot ?origin_fulfillment_unit ?origin_time_slot)
        (compatible_with_destination_slot ?destination_fulfillment_unit ?destination_time_slot)
        (origin_slot_confirmed ?origin_time_slot)
        (destination_slot_confirmed ?destination_time_slot)
        (not
          (origin_slot_confirmed_for_unit ?origin_fulfillment_unit)
        )
        (not
          (destination_slot_confirmed_for_unit ?destination_fulfillment_unit)
        )
        (formed_shipment_available ?formed_shipment)
      )
    :effect
      (and
        (formed_shipment_active ?formed_shipment)
        (shipment_has_origin_slot ?formed_shipment ?origin_time_slot)
        (shipment_has_destination_slot ?formed_shipment ?destination_time_slot)
        (formed_shipment_origin_confirmed ?formed_shipment)
        (formed_shipment_destination_confirmed ?formed_shipment)
        (not
          (formed_shipment_available ?formed_shipment)
        )
      )
  )
  (:action mark_shipment_manifest_ready
    :parameters (?formed_shipment - formed_shipment ?origin_fulfillment_unit - origin_fulfillment_unit ?order_line_item - order_line_item)
    :precondition
      (and
        (formed_shipment_active ?formed_shipment)
        (origin_unit_ready ?origin_fulfillment_unit)
        (line_item_allocated ?origin_fulfillment_unit ?order_line_item)
        (not
          (formed_shipment_manifest_ready ?formed_shipment)
        )
      )
    :effect (formed_shipment_manifest_ready ?formed_shipment)
  )
  (:action bind_package_to_shipment
    :parameters (?consolidation_shipment - consolidation_shipment ?piece_package - piece_package ?formed_shipment - formed_shipment)
    :precondition
      (and
        (validated_for_planning ?consolidation_shipment)
        (consolidation_linked_to_formed_shipment ?consolidation_shipment ?formed_shipment)
        (consolidation_has_package ?consolidation_shipment ?piece_package)
        (package_available ?piece_package)
        (formed_shipment_active ?formed_shipment)
        (formed_shipment_manifest_ready ?formed_shipment)
        (not
          (package_manifested ?piece_package)
        )
      )
    :effect
      (and
        (package_manifested ?piece_package)
        (package_assigned_to_shipment ?piece_package ?formed_shipment)
        (not
          (package_available ?piece_package)
        )
      )
  )
  (:action prepare_consolidation_for_manifesting
    :parameters (?consolidation_shipment - consolidation_shipment ?piece_package - piece_package ?formed_shipment - formed_shipment ?order_line_item - order_line_item)
    :precondition
      (and
        (validated_for_planning ?consolidation_shipment)
        (consolidation_has_package ?consolidation_shipment ?piece_package)
        (package_manifested ?piece_package)
        (package_assigned_to_shipment ?piece_package ?formed_shipment)
        (line_item_allocated ?consolidation_shipment ?order_line_item)
        (not
          (formed_shipment_origin_confirmed ?formed_shipment)
        )
        (not
          (consolidation_prepared ?consolidation_shipment)
        )
      )
    :effect (consolidation_prepared ?consolidation_shipment)
  )
  (:action attach_routing_attribute_to_consolidation
    :parameters (?consolidation_shipment - consolidation_shipment ?routing_attribute - routing_attribute)
    :precondition
      (and
        (validated_for_planning ?consolidation_shipment)
