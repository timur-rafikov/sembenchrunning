  (:action register_strategy_proposal
    :parameters (?strategy_proposal - strategy_proposal)
    :precondition
      (and
        (not
          (entity_registered ?strategy_proposal)
        )
        (not
          (approved_for_release_for_entity ?strategy_proposal)
        )
      )
    :effect (entity_registered ?strategy_proposal)
  )
  (:action assign_masking_agent_to_strategy
    :parameters (?strategy_proposal - strategy_proposal ?masking_agent - masking_agent)
    :precondition
      (and
        (entity_registered ?strategy_proposal)
        (not
          (entity_masking_assignment_confirmed ?strategy_proposal)
        )
        (masking_agent_available ?masking_agent)
      )
    :effect
      (and
        (entity_masking_assignment_confirmed ?strategy_proposal)
        (assigned_masking_agent_to_entity ?strategy_proposal ?masking_agent)
        (not
          (masking_agent_available ?masking_agent)
        )
      )
  )
  (:action schedule_sensory_test_for_strategy
    :parameters (?strategy_proposal - strategy_proposal ?sensory_test - sensory_test)
    :precondition
      (and
        (entity_registered ?strategy_proposal)
        (entity_masking_assignment_confirmed ?strategy_proposal)
        (sensory_test_available ?sensory_test)
      )
    :effect
      (and
        (assigned_sensory_test_to_entity ?strategy_proposal ?sensory_test)
        (not
          (sensory_test_available ?sensory_test)
        )
      )
  )
  (:action record_sensory_test_result
    :parameters (?strategy_proposal - strategy_proposal ?sensory_test - sensory_test)

        (entity_registered ?strategy_proposal)

          (entity_evidence_linked ?strategy_proposal)

  )
  (:action complete_sensory_test
    :parameters (?strategy_proposal - strategy_proposal ?sensory_test - sensory_test)

      (and
        (assigned_sensory_test_to_entity ?strategy_proposal ?sensory_test)
      )
    :effect
      (and
        (sensory_test_available ?sensory_test)
        (not
          (assigned_sensory_test_to_entity ?strategy_proposal ?sensory_test)
        )
      )
  )
