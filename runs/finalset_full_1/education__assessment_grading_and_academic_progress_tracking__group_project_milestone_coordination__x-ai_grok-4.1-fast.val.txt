--- domain.pddl (as passed to VAL) ---
(define (domain education_group_project_milestone_coordination)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_category - object artifact_category - object checkpoint_category - object academic_entity_type - object participant - academic_entity_type scheduling_slot - resource_category milestone_deliverable - resource_category evaluator - resource_category rubric_item - resource_category grade_component - resource_category recovery_option - resource_category assessment_feedback - resource_category administrative_override - resource_category resource_token - artifact_category supporting_document - artifact_category external_approval - artifact_category lead_checkpoint - checkpoint_category contrib_checkpoint - checkpoint_category submission_package - checkpoint_category member_profile - participant group_affiliation - participant lead_member - member_profile contributor_member - member_profile project_group - group_affiliation)
  (:predicates
    (submission_plan_created ?participant - participant)
    (submission_validated ?participant - participant)
    (entity_scheduled ?participant - participant)
    (grade_finalized ?participant - participant)
    (eligible_for_grading ?participant - participant)
    (grade_recorded ?participant - participant)
    (scheduling_slot_available ?scheduling_slot - scheduling_slot)
    (entity_assigned_slot ?participant - participant ?scheduling_slot - scheduling_slot)
    (deliverable_available ?milestone_deliverable - milestone_deliverable)
    (entity_submitted_deliverable ?participant - participant ?milestone_deliverable - milestone_deliverable)
    (evaluator_available ?evaluator - evaluator)
    (evaluator_assigned_to_entity ?participant - participant ?evaluator - evaluator)
    (resource_token_available ?resource_token - resource_token)
    (lead_has_resource_token ?lead_member - lead_member ?resource_token - resource_token)
    (contributor_has_resource_token ?contributor_member - contributor_member ?resource_token - resource_token)
    (lead_assigned_checkpoint ?lead_member - lead_member ?lead_checkpoint - lead_checkpoint)
    (lead_checkpoint_activated ?lead_checkpoint - lead_checkpoint)
    (lead_checkpoint_reserved ?lead_checkpoint - lead_checkpoint)
    (lead_checkpoint_confirmation_recorded ?lead_member - lead_member)
    (contributor_assigned_checkpoint ?contributor_member - contributor_member ?contrib_checkpoint - contrib_checkpoint)
    (contrib_checkpoint_activated ?contrib_checkpoint - contrib_checkpoint)
    (contrib_checkpoint_reserved ?contrib_checkpoint - contrib_checkpoint)
    (contrib_checkpoint_confirmation_recorded ?contributor_member - contributor_member)
    (submission_package_open ?submission_package - submission_package)
    (submission_package_registered ?submission_package - submission_package)
    (package_contains_lead_checkpoint ?submission_package - submission_package ?lead_checkpoint - lead_checkpoint)
    (package_contains_contrib_checkpoint ?submission_package - submission_package ?contrib_checkpoint - contrib_checkpoint)
    (package_flag_lead_checkpoint_secured ?submission_package - submission_package)
    (package_flag_contrib_checkpoint_secured ?submission_package - submission_package)
    (submission_package_processed ?submission_package - submission_package)
    (group_has_lead_member ?project_group - project_group ?lead_member - lead_member)
    (group_has_contributor_member ?project_group - project_group ?contributor_member - contributor_member)
    (group_has_submission_package ?project_group - project_group ?submission_package - submission_package)
    (supporting_document_available ?supporting_document - supporting_document)
    (group_has_supporting_document ?project_group - project_group ?supporting_document - supporting_document)
    (supporting_document_attached ?supporting_document - supporting_document)
    (document_linked_to_package ?supporting_document - supporting_document ?submission_package - submission_package)
    (group_document_processing_started ?project_group - project_group)
    (group_documents_processing_verified ?project_group - project_group)
    (group_documentation_verified ?project_group - project_group)
    (group_rubric_assigned ?project_group - project_group)
    (group_rubric_confirmed ?project_group - project_group)
    (group_rubric_present ?project_group - project_group)
    (group_finalization_approved ?project_group - project_group)
    (external_approval_available ?external_approval - external_approval)
    (group_has_external_approval ?project_group - project_group ?external_approval - external_approval)
    (group_external_approval_attached ?project_group - project_group)
    (group_external_approval_acknowledged ?project_group - project_group)
    (group_external_approval_processed ?project_group - project_group)
    (rubric_item_available ?rubric_item - rubric_item)
    (group_has_rubric_item ?project_group - project_group ?rubric_item - rubric_item)
    (grade_component_available ?grade_component - grade_component)
    (group_has_grade_component ?project_group - project_group ?grade_component - grade_component)
    (assessment_feedback_available ?assessment_feedback - assessment_feedback)
    (group_has_assessment_feedback ?project_group - project_group ?assessment_feedback - assessment_feedback)
    (administrative_override_available ?administrative_override - administrative_override)
    (group_has_administrative_override ?project_group - project_group ?administrative_override - administrative_override)
    (recovery_option_available ?recovery_option - recovery_option)
    (entity_has_recovery_option ?participant - participant ?recovery_option - recovery_option)
    (lead_ready ?lead_member - lead_member)
    (contrib_ready ?contributor_member - contributor_member)
    (group_eligible_for_grading ?project_group - project_group)
  )
  (:action create_submission_plan
    :parameters (?participant - participant)
    :precondition
      (and
        (not
          (submission_plan_created ?participant)
        )
        (not
          (grade_finalized ?participant)
        )
      )
    :effect (submission_plan_created ?participant)
  )
  (:action allocate_scheduling_slot_to_participant
    :parameters (?participant - participant ?scheduling_slot - scheduling_slot)
    :precondition
      (and
        (submission_plan_created ?participant)
        (not
          (entity_scheduled ?participant)
        )
        (scheduling_slot_available ?scheduling_slot)
      )
    :effect
      (and
        (entity_scheduled ?participant)
        (entity_assigned_slot ?participant ?scheduling_slot)
        (not
          (scheduling_slot_available ?scheduling_slot)
        )
      )
  )
  (:action submit_milestone_deliverable
    :parameters (?participant - participant ?milestone_deliverable - milestone_deliverable)
    :precondition
      (and
        (submission_plan_created ?participant)
        (entity_scheduled ?participant)
        (deliverable_available ?milestone_deliverable)
      )
    :effect
      (and
        (entity_submitted_deliverable ?participant ?milestone_deliverable)
        (not
          (deliverable_available ?milestone_deliverable)
        )
      )
  )
  (:action validate_milestone_submission
    :parameters (?participant - participant ?milestone_deliverable - milestone_deliverable)
    :precondition
      (and
        (submission_plan_created ?participant)
        (entity_scheduled ?participant)
        (entity_submitted_deliverable ?participant ?milestone_deliverable)
        (not
          (submission_validated ?participant)
        )
      )
    :effect (submission_validated ?participant)
  )
  (:action retract_milestone_submission
    :parameters (?participant - participant ?milestone_deliverable - milestone_deliverable)
    :precondition
      (and
        (entity_submitted_deliverable ?participant ?milestone_deliverable)
      )
    :effect
      (and
        (deliverable_available ?milestone_deliverable)
        (not
          (entity_submitted_deliverable ?participant ?milestone_deliverable)
        )
      )
  )
  (:action assign_evaluator_to_participant
    :parameters (?participant - participant ?evaluator - evaluator)
    :precondition
      (and
        (submission_validated ?participant)
        (evaluator_available ?evaluator)
      )
    :effect
      (and
        (evaluator_assigned_to_entity ?participant ?evaluator)
        (not
          (evaluator_available ?evaluator)
        )
      )
  )
  (:action withdraw_evaluator_from_participant
    :parameters (?participant - participant ?evaluator - evaluator)
    :precondition
      (and
        (evaluator_assigned_to_entity ?participant ?evaluator)
      )
    :effect
      (and
        (evaluator_available ?evaluator)
        (not
          (evaluator_assigned_to_entity ?participant ?evaluator)
        )
      )
  )
  (:action attach_feedback_to_group
    :parameters (?project_group - project_group ?assessment_feedback - assessment_feedback)
    :precondition
      (and
        (submission_validated ?project_group)
        (assessment_feedback_available ?assessment_feedback)
      )
    :effect
      (and
        (group_has_assessment_feedback ?project_group ?assessment_feedback)
        (not
          (assessment_feedback_available ?assessment_feedback)
        )
      )
  )
  (:action detach_feedback_from_group
    :parameters (?project_group - project_group ?assessment_feedback - assessment_feedback)
    :precondition
      (and
        (group_has_assessment_feedback ?project_group ?assessment_feedback)
      )
    :effect
      (and
        (assessment_feedback_available ?assessment_feedback)
        (not
          (group_has_assessment_feedback ?project_group ?assessment_feedback)
        )
      )
  )
  (:action apply_administrative_override_to_group
    :parameters (?project_group - project_group ?administrative_override - administrative_override)
    :precondition
      (and
        (submission_validated ?project_group)
        (administrative_override_available ?administrative_override)
      )
    :effect
      (and
        (group_has_administrative_override ?project_group ?administrative_override)
        (not
          (administrative_override_available ?administrative_override)
        )
      )
  )
  (:action revert_administrative_override_on_group
    :parameters (?project_group - project_group ?administrative_override - administrative_override)
    :precondition
      (and
        (group_has_administrative_override ?project_group ?administrative_override)
      )
    :effect
      (and
        (administrative_override_available ?administrative_override)
        (not
          (group_has_administrative_override ?project_group ?administrative_override)
        )
      )
  )
  (:action activate_lead_checkpoint
    :parameters (?lead_member - lead_member ?lead_checkpoint - lead_checkpoint ?milestone_deliverable - milestone_deliverable)
    :precondition
      (and
        (submission_validated ?lead_member)
        (entity_submitted_deliverable ?lead_member ?milestone_deliverable)
        (lead_assigned_checkpoint ?lead_member ?lead_checkpoint)
        (not
          (lead_checkpoint_activated ?lead_checkpoint)
        )
        (not
          (lead_checkpoint_reserved ?lead_checkpoint)
        )
      )
    :effect (lead_checkpoint_activated ?lead_checkpoint)
  )
  (:action confirm_lead_checkpoint_with_evaluator
    :parameters (?lead_member - lead_member ?lead_checkpoint - lead_checkpoint ?evaluator - evaluator)
    :precondition
      (and
        (submission_validated ?lead_member)
        (evaluator_assigned_to_entity ?lead_member ?evaluator)
        (lead_assigned_checkpoint ?lead_member ?lead_checkpoint)
        (lead_checkpoint_activated ?lead_checkpoint)
        (not
          (lead_ready ?lead_member)
        )
      )
    :effect
      (and
        (lead_ready ?lead_member)
        (lead_checkpoint_confirmation_recorded ?lead_member)
      )
  )
  (:action reserve_lead_checkpoint_with_token
    :parameters (?lead_member - lead_member ?lead_checkpoint - lead_checkpoint ?resource_token - resource_token)
    :precondition
      (and
        (submission_validated ?lead_member)
        (lead_assigned_checkpoint ?lead_member ?lead_checkpoint)
        (resource_token_available ?resource_token)
        (not
          (lead_ready ?lead_member)
        )
      )
    :effect
      (and
        (lead_checkpoint_reserved ?lead_checkpoint)
        (lead_ready ?lead_member)
        (lead_has_resource_token ?lead_member ?resource_token)
        (not
          (resource_token_available ?resource_token)
        )
      )
  )
  (:action finalize_lead_checkpoint
    :parameters (?lead_member - lead_member ?lead_checkpoint - lead_checkpoint ?milestone_deliverable - milestone_deliverable ?resource_token - resource_token)
    :precondition
      (and
        (submission_validated ?lead_member)
        (entity_submitted_deliverable ?lead_member ?milestone_deliverable)
        (lead_assigned_checkpoint ?lead_member ?lead_checkpoint)
        (lead_checkpoint_reserved ?lead_checkpoint)
        (lead_has_resource_token ?lead_member ?resource_token)
        (not
          (lead_checkpoint_confirmation_recorded ?lead_member)
        )
      )
    :effect
      (and
        (lead_checkpoint_activated ?lead_checkpoint)
        (lead_checkpoint_confirmation_recorded ?lead_member)
        (resource_token_available ?resource_token)
        (not
          (lead_has_resource_token ?lead_member ?resource_token)
        )
      )
  )
  (:action activate_contrib_checkpoint
    :parameters (?contributor_member - contributor_member ?contrib_checkpoint - contrib_checkpoint ?milestone_deliverable - milestone_deliverable)
    :precondition
      (and
        (submission_validated ?contributor_member)
        (entity_submitted_deliverable ?contributor_member ?milestone_deliverable)
        (contributor_assigned_checkpoint ?contributor_member ?contrib_checkpoint)
        (not
          (contrib_checkpoint_activated ?contrib_checkpoint)
        )
        (not
          (contrib_checkpoint_reserved ?contrib_checkpoint)
        )
      )
    :effect (contrib_checkpoint_activated ?contrib_checkpoint)
  )
  (:action confirm_contrib_checkpoint_with_evaluator
    :parameters (?contributor_member - contributor_member ?contrib_checkpoint - contrib_checkpoint ?evaluator - evaluator)
    :precondition
      (and
        (submission_validated ?contributor_member)
        (evaluator_assigned_to_entity ?contributor_member ?evaluator)
        (contributor_assigned_checkpoint ?contributor_member ?contrib_checkpoint)
        (contrib_checkpoint_activated ?contrib_checkpoint)
        (not
          (contrib_ready ?contributor_member)
        )
      )
    :effect
      (and
        (contrib_ready ?contributor_member)
        (contrib_checkpoint_confirmation_recorded ?contributor_member)
      )
  )
  (:action reserve_contrib_checkpoint_with_token
    :parameters (?contributor_member - contributor_member ?contrib_checkpoint - contrib_checkpoint ?resource_token - resource_token)
    :precondition
      (and
        (submission_validated ?contributor_member)
        (contributor_assigned_checkpoint ?contributor_member ?contrib_checkpoint)
        (resource_token_available ?resource_token)
        (not
          (contrib_ready ?contributor_member)
        )
      )
    :effect
      (and
        (contrib_checkpoint_reserved ?contrib_checkpoint)
        (contrib_ready ?contributor_member)
        (contributor_has_resource_token ?contributor_member ?resource_token)
        (not
          (resource_token_available ?resource_token)
        )
      )
  )
  (:action finalize_contrib_checkpoint
    :parameters (?contributor_member - contributor_member ?contrib_checkpoint - contrib_checkpoint ?milestone_deliverable - milestone_deliverable ?resource_token - resource_token)
    :precondition
      (and
        (submission_validated ?contributor_member)
        (entity_submitted_deliverable ?contributor_member ?milestone_deliverable)
        (contributor_assigned_checkpoint ?contributor_member ?contrib_checkpoint)
        (contrib_checkpoint_reserved ?contrib_checkpoint)
        (contributor_has_resource_token ?contributor_member ?resource_token)
        (not
          (contrib_checkpoint_confirmation_recorded ?contributor_member)
        )
      )
    :effect
      (and
        (contrib_checkpoint_activated ?contrib_checkpoint)
        (contrib_checkpoint_confirmation_recorded ?contributor_member)
        (resource_token_available ?resource_token)
        (not
          (contributor_has_resource_token ?contributor_member ?resource_token)
        )
      )
  )
  (:action assemble_submission_package_variant_a
    :parameters (?lead_member - lead_member ?contributor_member - contributor_member ?lead_checkpoint - lead_checkpoint ?contrib_checkpoint - contrib_checkpoint ?submission_package - submission_package)
    :precondition
      (and
        (lead_ready ?lead_member)
        (contrib_ready ?contributor_member)
        (lead_assigned_checkpoint ?lead_member ?lead_checkpoint)
        (contributor_assigned_checkpoint ?contributor_member ?contrib_checkpoint)
        (lead_checkpoint_activated ?lead_checkpoint)
        (contrib_checkpoint_activated ?contrib_checkpoint)
        (lead_checkpoint_confirmation_recorded ?lead_member)
        (contrib_checkpoint_confirmation_recorded ?contributor_member)
        (submission_package_open ?submission_package)
      )
    :effect
      (and
        (submission_package_registered ?submission_package)
        (package_contains_lead_checkpoint ?submission_package ?lead_checkpoint)
        (package_contains_contrib_checkpoint ?submission_package ?contrib_checkpoint)
        (not
          (submission_package_open ?submission_package)
        )
      )
  )
  (:action assemble_submission_package_variant_b
    :parameters (?lead_member - lead_member ?contributor_member - contributor_member ?lead_checkpoint - lead_checkpoint ?contrib_checkpoint - contrib_checkpoint ?submission_package - submission_package)
    :precondition
      (and
        (lead_ready ?lead_member)
        (contrib_ready ?contributor_member)
        (lead_assigned_checkpoint ?lead_member ?lead_checkpoint)
        (contributor_assigned_checkpoint ?contributor_member ?contrib_checkpoint)
        (lead_checkpoint_reserved ?lead_checkpoint)
        (contrib_checkpoint_activated ?contrib_checkpoint)
        (not
          (lead_checkpoint_confirmation_recorded ?lead_member)
        )
        (contrib_checkpoint_confirmation_recorded ?contributor_member)
        (submission_package_open ?submission_package)
      )
    :effect
      (and
        (submission_package_registered ?submission_package)
        (package_contains_lead_checkpoint ?submission_package ?lead_checkpoint)
        (package_contains_contrib_checkpoint ?submission_package ?contrib_checkpoint)
        (package_flag_lead_checkpoint_secured ?submission_package)
        (not
          (submission_package_open ?submission_package)
        )
      )
  )
  (:action assemble_submission_package_variant_c
    :parameters (?lead_member - lead_member ?contributor_member - contributor_member ?lead_checkpoint - lead_checkpoint ?contrib_checkpoint - contrib_checkpoint ?submission_package - submission_package)
    :precondition
      (and
        (lead_ready ?lead_member)
        (contrib_ready ?contributor_member)
        (lead_assigned_checkpoint ?lead_member ?lead_checkpoint)
        (contributor_assigned_checkpoint ?contributor_member ?contrib_checkpoint)
        (lead_checkpoint_activated ?lead_checkpoint)
        (contrib_checkpoint_reserved ?contrib_checkpoint)
        (lead_checkpoint_confirmation_recorded ?lead_member)
        (not
          (contrib_checkpoint_confirmation_recorded ?contributor_member)
        )
        (submission_package_open ?submission_package)
      )
    :effect
      (and
        (submission_package_registered ?submission_package)
        (package_contains_lead_checkpoint ?submission_package ?lead_checkpoint)
        (package_contains_contrib_checkpoint ?submission_package ?contrib_checkpoint)
        (package_flag_contrib_checkpoint_secured ?submission_package)
        (not
          (submission_package_open ?submission_package)
        )
      )
  )
  (:action assemble_submission_package_variant_d
    :parameters (?lead_member - lead_member ?contributor_member - contributor_member ?lead_checkpoint - lead_checkpoint ?contrib_checkpoint - contrib_checkpoint ?submission_package - submission_package)
    :precondition
      (and
        (lead_ready ?lead_member)
        (contrib_ready ?contributor_member)
        (lead_assigned_checkpoint ?lead_member ?lead_checkpoint)
        (contributor_assigned_checkpoint ?contributor_member ?contrib_checkpoint)
        (lead_checkpoint_reserved ?lead_checkpoint)
        (contrib_checkpoint_reserved ?contrib_checkpoint)
        (not
          (lead_checkpoint_confirmation_recorded ?lead_member)
        )
        (not
          (contrib_checkpoint_confirmation_recorded ?contributor_member)
        )
        (submission_package_open ?submission_package)
      )
    :effect
      (and
        (submission_package_registered ?submission_package)
        (package_contains_lead_checkpoint ?submission_package ?lead_checkpoint)
        (package_contains_contrib_checkpoint ?submission_package ?contrib_checkpoint)
        (package_flag_lead_checkpoint_secured ?submission_package)
        (package_flag_contrib_checkpoint_secured ?submission_package)
        (not
          (submission_package_open ?submission_package)
        )
      )
  )
  (:action flag_submission_package_for_document_processing
    :parameters (?submission_package - submission_package ?lead_member - lead_member ?milestone_deliverable - milestone_deliverable)
    :precondition
      (and
        (submission_package_registered ?submission_package)
        (lead_ready ?lead_member)
        (entity_submitted_deliverable ?lead_member ?milestone_deliverable)
        (not
          (submission_package_processed ?submission_package)
        )
      )
    :effect (submission_package_processed ?submission_package)
  )
  (:action attach_supporting_document_to_group_and_package
    :parameters (?project_group - project_group ?supporting_document - supporting_document ?submission_package - submission_package)
    :precondition
      (and
        (submission_validated ?project_group)
        (group_has_submission_package ?project_group ?submission_package)
        (group_has_supporting_document ?project_group ?supporting_document)
        (supporting_document_available ?supporting_document)
        (submission_package_registered ?submission_package)
        (submission_package_processed ?submission_package)
        (not
          (supporting_document_attached ?supporting_document)
        )
      )
    :effect
      (and
        (supporting_document_attached ?supporting_document)
        (document_linked_to_package ?supporting_document ?submission_package)
        (not
          (supporting_document_available ?supporting_document)
        )
      )
  )
  (:action process_supporting_document_for_group
    :parameters (?project_group - project_group ?supporting_document - supporting_document ?submission_package - submission_package ?milestone_deliverable - milestone_deliverable)
    :precondition
      (and
        (submission_validated ?project_group)
        (group_has_supporting_document ?project_group ?supporting_document)
        (supporting_document_attached ?supporting_document)
        (document_linked_to_package ?supporting_document ?submission_package)
        (entity_submitted_deliverable ?project_group ?milestone_deliverable)
        (not
          (package_flag_lead_checkpoint_secured ?submission_package)
        )
        (not
          (group_document_processing_started ?project_group)
        )
      )
    :effect (group_document_processing_started ?project_group)
  )
  (:action assign_rubric_item_to_group
    :parameters (?project_group - project_group ?rubric_item - rubric_item)
    :precondition
      (and
        (submission_validated ?project_group)
        (rubric_item_available ?rubric_item)
        (not
          (group_rubric_assigned ?project_group)
        )
      )
    :effect
      (and
        (group_rubric_assigned ?project_group)
        (group_has_rubric_item ?project_group ?rubric_item)
        (not
          (rubric_item_available ?rubric_item)
        )
      )
  )
  (:action confirm_rubric_and_prepare_group
    :parameters (?project_group - project_group ?supporting_document - supporting_document ?submission_package - submission_package ?milestone_deliverable - milestone_deliverable ?rubric_item - rubric_item)
    :precondition
      (and
        (submission_validated ?project_group)
        (group_has_supporting_document ?project_group ?supporting_document)
        (supporting_document_attached ?supporting_document)
        (document_linked_to_package ?supporting_document ?submission_package)
        (entity_submitted_deliverable ?project_group ?milestone_deliverable)
        (package_flag_lead_checkpoint_secured ?submission_package)
        (group_rubric_assigned ?project_group)
        (group_has_rubric_item ?project_group ?rubric_item)
        (not
          (group_document_processing_started ?project_group)
        )
      )
    :effect
      (and
        (group_document_processing_started ?project_group)
        (group_rubric_confirmed ?project_group)
      )
  )
  (:action apply_feedback_to_group_and_mark_verified
    :parameters (?project_group - project_group ?assessment_feedback - assessment_feedback ?evaluator - evaluator ?supporting_document - supporting_document ?submission_package - submission_package)
    :precondition
      (and
        (group_document_processing_started ?project_group)
        (group_has_assessment_feedback ?project_group ?assessment_feedback)
        (evaluator_assigned_to_entity ?project_group ?evaluator)
        (group_has_supporting_document ?project_group ?supporting_document)
        (document_linked_to_package ?supporting_document ?submission_package)
        (not
          (package_flag_contrib_checkpoint_secured ?submission_package)
        )
        (not
          (group_documents_processing_verified ?project_group)
        )
      )
    :effect (group_documents_processing_verified ?project_group)
  )
  (:action apply_feedback_to_group_variant
    :parameters (?project_group - project_group ?assessment_feedback - assessment_feedback ?evaluator - evaluator ?supporting_document - supporting_document ?submission_package - submission_package)
    :precondition
      (and
        (group_document_processing_started ?project_group)
