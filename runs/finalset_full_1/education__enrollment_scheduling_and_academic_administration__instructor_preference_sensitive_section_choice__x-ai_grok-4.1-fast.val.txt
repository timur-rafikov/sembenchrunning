--- domain.pddl (as passed to VAL) ---
(define (domain education_enrollment_instructor_preference_scheduling)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity - object entity_group - entity preference_role_group - entity time_option_group - entity application_group - entity registration_case - application_group instructor - entity_group time_slot - entity_group course_administrator - entity_group section_attribute - entity_group pedagogical_constraint - entity_group priority_token - entity_group resource_request - entity_group accommodation_request - entity_group student_preference - preference_role_group classroom - preference_role_group instructor_qualification - preference_role_group lecture_time_option - time_option_group discussion_time_option - time_option_group section_schedule_proposal - time_option_group lecture_section_type - registration_case discussion_section_type - registration_case lecture_section - lecture_section_type discussion_section - lecture_section_type course_offering - discussion_section_type)

  (:predicates
    (registration_case_open ?registration_case - registration_case)
    (ready_for_assignment ?registration_case - registration_case)
    (instructor_selected_for_case ?registration_case - registration_case)
    (assignment_confirmed ?registration_case - registration_case)
    (ready_for_final_approval ?registration_case - registration_case)
    (priority_allocated ?registration_case - registration_case)
    (instructor_available ?instructor - instructor)
    (has_instructor_claim ?registration_case - registration_case ?instructor - instructor)
    (time_slot_available ?time_slot - time_slot)
    (has_time_assignment ?registration_case - registration_case ?time_slot - time_slot)
    (administrator_available ?administrator - course_administrator)
    (has_administrator_claim ?registration_case - registration_case ?administrator - course_administrator)
    (student_preference_available ?student_preference - student_preference)
    (lecture_section_student_preference ?lecture_section - lecture_section ?student_preference - student_preference)
    (discussion_section_student_preference ?discussion_section - discussion_section ?student_preference - student_preference)
    (lecture_section_time_option ?lecture_section - lecture_section ?lecture_time_option - lecture_time_option)
    (lecture_time_option_confirmed ?lecture_time_option - lecture_time_option)
    (lecture_time_option_tentative ?lecture_time_option - lecture_time_option)
    (lecture_section_ready ?lecture_section - lecture_section)
    (discussion_section_time_option ?discussion_section - discussion_section ?discussion_time_option - discussion_time_option)
    (discussion_time_option_confirmed ?discussion_time_option - discussion_time_option)
    (discussion_time_option_tentative ?discussion_time_option - discussion_time_option)
    (discussion_section_ready ?discussion_section - discussion_section)
    (proposal_unallocated ?proposal - section_schedule_proposal)
    (proposal_reserved ?proposal - section_schedule_proposal)
    (proposal_lecture_time_option ?proposal - section_schedule_proposal ?lecture_time_option - lecture_time_option)
    (proposal_discussion_time_option ?proposal - section_schedule_proposal ?discussion_time_option - discussion_time_option)
    (proposal_requires_lecture_room_check ?proposal - section_schedule_proposal)
    (proposal_requires_discussion_room_check ?proposal - section_schedule_proposal)
    (proposal_time_confirmed ?proposal - section_schedule_proposal)
    (offering_has_lecture_section ?course_offering - course_offering ?lecture_section - lecture_section)
    (offering_has_discussion_section ?course_offering - course_offering ?discussion_section - discussion_section)
    (offering_has_proposal ?course_offering - course_offering ?proposal - section_schedule_proposal)
    (classroom_available ?classroom - classroom)
    (offering_classroom_compatibility ?course_offering - course_offering ?classroom - classroom)
    (classroom_reserved ?classroom - classroom)
    (classroom_reserved_for_proposal ?classroom - classroom ?proposal - section_schedule_proposal)
    (offering_room_allocated ?course_offering - course_offering)
    (offering_resources_reserved ?course_offering - course_offering)
    (offering_ready_for_finalization ?course_offering - course_offering)
    (offering_attribute_claimed ?course_offering - course_offering)
    (offering_attribute_confirmed ?course_offering - course_offering)
    (offering_has_pedagogical_constraint ?course_offering - course_offering)
    (offering_validated_for_confirmation ?course_offering - course_offering)
    (instructor_qualification_available ?instructor_qualification - instructor_qualification)
    (offering_instructor_qualification_required ?course_offering - course_offering ?instructor_qualification - instructor_qualification)
    (offering_qualification_claimed ?course_offering - course_offering)
    (offering_qualification_allocated ?course_offering - course_offering)
    (offering_qualification_approved ?course_offering - course_offering)
    (section_attribute_available ?section_attribute - section_attribute)
    (offering_has_attribute ?course_offering - course_offering ?section_attribute - section_attribute)
    (pedagogical_constraint_available ?pedagogical_constraint - pedagogical_constraint)
    (offering_pedagogical_constraint_attached ?course_offering - course_offering ?pedagogical_constraint - pedagogical_constraint)
    (resource_request_available ?resource_request - resource_request)
    (offering_has_resource_request ?course_offering - course_offering ?resource_request - resource_request)
    (accommodation_request_available ?accommodation_request - accommodation_request)
    (offering_has_accommodation_request ?course_offering - course_offering ?accommodation_request - accommodation_request)
    (priority_token_available ?priority_token - priority_token)
    (has_priority_token ?registration_case - registration_case ?priority_token - priority_token)
    (lecture_section_locked ?lecture_section - lecture_section)
    (discussion_section_locked ?discussion_section - discussion_section)
    (offering_finalized ?course_offering - course_offering)
  )
  (:action open_registration_case
    :parameters (?registration_case - registration_case)
    :precondition
      (and
        (not
          (registration_case_open ?registration_case)
        )
        (not
          (assignment_confirmed ?registration_case)
        )
      )
    :effect (registration_case_open ?registration_case)
  )
  (:action claim_instructor_for_case
    :parameters (?registration_case - registration_case ?instructor - instructor)
    :precondition
      (and
        (registration_case_open ?registration_case)
        (not
          (instructor_selected_for_case ?registration_case)
        )
        (instructor_available ?instructor)
      )
    :effect
      (and
        (instructor_selected_for_case ?registration_case)
        (has_instructor_claim ?registration_case ?instructor)
        (not
          (instructor_available ?instructor)
        )
      )
  )
  (:action select_time_option_for_case
    :parameters (?registration_case - registration_case ?time_slot - time_slot)
    :precondition
      (and
        (registration_case_open ?registration_case)
        (instructor_selected_for_case ?registration_case)
        (time_slot_available ?time_slot)
      )
    :effect
      (and
        (has_time_assignment ?registration_case ?time_slot)
        (not
          (time_slot_available ?time_slot)
        )
      )
  )
  (:action submit_case_for_administrative_review
    :parameters (?registration_case - registration_case ?time_slot - time_slot)
    :precondition
      (and
        (registration_case_open ?registration_case)
        (instructor_selected_for_case ?registration_case)
        (has_time_assignment ?registration_case ?time_slot)
        (not
          (ready_for_assignment ?registration_case)
        )
      )
    :effect (ready_for_assignment ?registration_case)
  )
  (:action release_time_selection_for_case
    :parameters (?registration_case - registration_case ?time_slot - time_slot)
    :precondition
      (and
        (has_time_assignment ?registration_case ?time_slot)
      )
    :effect
      (and
        (time_slot_available ?time_slot)
        (not
          (has_time_assignment ?registration_case ?time_slot)
        )
      )
  )
  (:action assign_administrator_to_case
    :parameters (?registration_case - registration_case ?administrator - course_administrator)
    :precondition
      (and
        (ready_for_assignment ?registration_case)
        (administrator_available ?administrator)
      )
    :effect
      (and
        (has_administrator_claim ?registration_case ?administrator)
        (not
          (administrator_available ?administrator)
        )
      )
  )
  (:action release_administrator_from_case
    :parameters (?registration_case - registration_case ?administrator - course_administrator)
    :precondition
      (and
        (has_administrator_claim ?registration_case ?administrator)
      )
    :effect
      (and
        (administrator_available ?administrator)
        (not
          (has_administrator_claim ?registration_case ?administrator)
        )
      )
  )
  (:action attach_resource_request_to_offering
    :parameters (?course_offering - course_offering ?resource_request - resource_request)
    :precondition
      (and
        (ready_for_assignment ?course_offering)
        (resource_request_available ?resource_request)
      )
    :effect
      (and
        (offering_has_resource_request ?course_offering ?resource_request)
        (not
          (resource_request_available ?resource_request)
        )
      )
  )
  (:action release_resource_request_from_offering
    :parameters (?course_offering - course_offering ?resource_request - resource_request)
    :precondition
      (and
        (offering_has_resource_request ?course_offering ?resource_request)
      )
    :effect
      (and
        (resource_request_available ?resource_request)
        (not
          (offering_has_resource_request ?course_offering ?resource_request)
        )
      )
  )
  (:action attach_accommodation_request_to_offering
    :parameters (?course_offering - course_offering ?accommodation_request - accommodation_request)
    :precondition
      (and
        (ready_for_assignment ?course_offering)
        (accommodation_request_available ?accommodation_request)
      )
    :effect
      (and
        (offering_has_accommodation_request ?course_offering ?accommodation_request)
        (not
          (accommodation_request_available ?accommodation_request)
        )
      )
  )
  (:action release_accommodation_request_from_offering
    :parameters (?course_offering - course_offering ?accommodation_request - accommodation_request)
    :precondition
      (and
        (offering_has_accommodation_request ?course_offering ?accommodation_request)
      )
    :effect
      (and
        (accommodation_request_available ?accommodation_request)
        (not
          (offering_has_accommodation_request ?course_offering ?accommodation_request)
        )
      )
  )
  (:action confirm_lecture_time_option_for_section
    :parameters (?lecture_section - lecture_section ?lecture_time_option - lecture_time_option ?time_slot - time_slot)
    :precondition
      (and
        (ready_for_assignment ?lecture_section)
        (has_time_assignment ?lecture_section ?time_slot)
        (lecture_section_time_option ?lecture_section ?lecture_time_option)
        (not
          (lecture_time_option_confirmed ?lecture_time_option)
        )
        (not
          (lecture_time_option_tentative ?lecture_time_option)
        )
      )
    :effect (lecture_time_option_confirmed ?lecture_time_option)
  )
  (:action reserve_lecture_section_for_proposal
    :parameters (?lecture_section - lecture_section ?lecture_time_option - lecture_time_option ?administrator - course_administrator)
    :precondition
      (and
        (ready_for_assignment ?lecture_section)
        (has_administrator_claim ?lecture_section ?administrator)
        (lecture_section_time_option ?lecture_section ?lecture_time_option)
        (lecture_time_option_confirmed ?lecture_time_option)
        (not
          (lecture_section_locked ?lecture_section)
        )
      )
    :effect
      (and
        (lecture_section_locked ?lecture_section)
        (lecture_section_ready ?lecture_section)
      )
  )
  (:action apply_student_preference_to_lecture_section
    :parameters (?lecture_section - lecture_section ?lecture_time_option - lecture_time_option ?student_preference - student_preference)
    :precondition
      (and
        (ready_for_assignment ?lecture_section)
        (lecture_section_time_option ?lecture_section ?lecture_time_option)
        (student_preference_available ?student_preference)
        (not
          (lecture_section_locked ?lecture_section)
        )
      )
    :effect
      (and
        (lecture_time_option_tentative ?lecture_time_option)
        (lecture_section_locked ?lecture_section)
        (lecture_section_student_preference ?lecture_section ?student_preference)
        (not
          (student_preference_available ?student_preference)
        )
      )
  )
  (:action confirm_student_preference_on_lecture_section
    :parameters (?lecture_section - lecture_section ?lecture_time_option - lecture_time_option ?time_slot - time_slot ?student_preference - student_preference)
    :precondition
      (and
        (ready_for_assignment ?lecture_section)
        (has_time_assignment ?lecture_section ?time_slot)
        (lecture_section_time_option ?lecture_section ?lecture_time_option)
        (lecture_time_option_tentative ?lecture_time_option)
        (lecture_section_student_preference ?lecture_section ?student_preference)
        (not
          (lecture_section_ready ?lecture_section)
        )
      )
    :effect
      (and
        (lecture_time_option_confirmed ?lecture_time_option)
        (lecture_section_ready ?lecture_section)
        (student_preference_available ?student_preference)
        (not
          (lecture_section_student_preference ?lecture_section ?student_preference)
        )
      )
  )
  (:action confirm_discussion_time_option_for_section
    :parameters (?discussion_section - discussion_section ?discussion_time_option - discussion_time_option ?time_slot - time_slot)
    :precondition
      (and
        (ready_for_assignment ?discussion_section)
        (has_time_assignment ?discussion_section ?time_slot)
        (discussion_section_time_option ?discussion_section ?discussion_time_option)
        (not
          (discussion_time_option_confirmed ?discussion_time_option)
        )
        (not
          (discussion_time_option_tentative ?discussion_time_option)
        )
      )
    :effect (discussion_time_option_confirmed ?discussion_time_option)
  )
  (:action reserve_discussion_section_for_proposal
    :parameters (?discussion_section - discussion_section ?discussion_time_option - discussion_time_option ?administrator - course_administrator)
    :precondition
      (and
        (ready_for_assignment ?discussion_section)
        (has_administrator_claim ?discussion_section ?administrator)
        (discussion_section_time_option ?discussion_section ?discussion_time_option)
        (discussion_time_option_confirmed ?discussion_time_option)
        (not
          (discussion_section_locked ?discussion_section)
        )
      )
    :effect
      (and
        (discussion_section_locked ?discussion_section)
        (discussion_section_ready ?discussion_section)
      )
  )
  (:action apply_student_preference_to_discussion_section
    :parameters (?discussion_section - discussion_section ?discussion_time_option - discussion_time_option ?student_preference - student_preference)
    :precondition
      (and
        (ready_for_assignment ?discussion_section)
        (discussion_section_time_option ?discussion_section ?discussion_time_option)
        (student_preference_available ?student_preference)
        (not
          (discussion_section_locked ?discussion_section)
        )
      )
    :effect
      (and
        (discussion_time_option_tentative ?discussion_time_option)
        (discussion_section_locked ?discussion_section)
        (discussion_section_student_preference ?discussion_section ?student_preference)
        (not
          (student_preference_available ?student_preference)
        )
      )
  )
  (:action confirm_student_preference_on_discussion_section
    :parameters (?discussion_section - discussion_section ?discussion_time_option - discussion_time_option ?time_slot - time_slot ?student_preference - student_preference)
    :precondition
      (and
        (ready_for_assignment ?discussion_section)
        (has_time_assignment ?discussion_section ?time_slot)
        (discussion_section_time_option ?discussion_section ?discussion_time_option)
        (discussion_time_option_tentative ?discussion_time_option)
        (discussion_section_student_preference ?discussion_section ?student_preference)
        (not
          (discussion_section_ready ?discussion_section)
        )
      )
    :effect
      (and
        (discussion_time_option_confirmed ?discussion_time_option)
        (discussion_section_ready ?discussion_section)
        (student_preference_available ?student_preference)
        (not
          (discussion_section_student_preference ?discussion_section ?student_preference)
        )
      )
  )
  (:action create_schedule_proposal
    :parameters (?lecture_section - lecture_section ?discussion_section - discussion_section ?lecture_time_option - lecture_time_option ?discussion_time_option - discussion_time_option ?proposal - section_schedule_proposal)
    :precondition
      (and
        (lecture_section_locked ?lecture_section)
        (discussion_section_locked ?discussion_section)
        (lecture_section_time_option ?lecture_section ?lecture_time_option)
        (discussion_section_time_option ?discussion_section ?discussion_time_option)
        (lecture_time_option_confirmed ?lecture_time_option)
        (discussion_time_option_confirmed ?discussion_time_option)
        (lecture_section_ready ?lecture_section)
        (discussion_section_ready ?discussion_section)
        (proposal_unallocated ?proposal)
      )
    :effect
      (and
        (proposal_reserved ?proposal)
        (proposal_lecture_time_option ?proposal ?lecture_time_option)
        (proposal_discussion_time_option ?proposal ?discussion_time_option)
        (not
          (proposal_unallocated ?proposal)
        )
      )
  )
  (:action create_schedule_proposal_with_lecture_confirmation
    :parameters (?lecture_section - lecture_section ?discussion_section - discussion_section ?lecture_time_option - lecture_time_option ?discussion_time_option - discussion_time_option ?proposal - section_schedule_proposal)
    :precondition
      (and
        (lecture_section_locked ?lecture_section)
        (discussion_section_locked ?discussion_section)
        (lecture_section_time_option ?lecture_section ?lecture_time_option)
        (discussion_section_time_option ?discussion_section ?discussion_time_option)
        (lecture_time_option_tentative ?lecture_time_option)
        (discussion_time_option_confirmed ?discussion_time_option)
        (not
          (lecture_section_ready ?lecture_section)
        )
        (discussion_section_ready ?discussion_section)
        (proposal_unallocated ?proposal)
      )
    :effect
      (and
        (proposal_reserved ?proposal)
        (proposal_lecture_time_option ?proposal ?lecture_time_option)
        (proposal_discussion_time_option ?proposal ?discussion_time_option)
        (proposal_requires_lecture_room_check ?proposal)
        (not
          (proposal_unallocated ?proposal)
        )
      )
  )
  (:action create_schedule_proposal_with_discussion_confirmation
    :parameters (?lecture_section - lecture_section ?discussion_section - discussion_section ?lecture_time_option - lecture_time_option ?discussion_time_option - discussion_time_option ?proposal - section_schedule_proposal)
    :precondition
      (and
        (lecture_section_locked ?lecture_section)
        (discussion_section_locked ?discussion_section)
        (lecture_section_time_option ?lecture_section ?lecture_time_option)
        (discussion_section_time_option ?discussion_section ?discussion_time_option)
        (lecture_time_option_confirmed ?lecture_time_option)
        (discussion_time_option_tentative ?discussion_time_option)
        (lecture_section_ready ?lecture_section)
        (not
          (discussion_section_ready ?discussion_section)
        )
        (proposal_unallocated ?proposal)
      )
    :effect
      (and
        (proposal_reserved ?proposal)
        (proposal_lecture_time_option ?proposal ?lecture_time_option)
        (proposal_discussion_time_option ?proposal ?discussion_time_option)
        (proposal_requires_discussion_room_check ?proposal)
        (not
          (proposal_unallocated ?proposal)
        )
      )
  )
  (:action create_schedule_proposal_with_both_confirmations
    :parameters (?lecture_section - lecture_section ?discussion_section - discussion_section ?lecture_time_option - lecture_time_option ?discussion_time_option - discussion_time_option ?proposal - section_schedule_proposal)
    :precondition
      (and
        (lecture_section_locked ?lecture_section)
        (discussion_section_locked ?discussion_section)
        (lecture_section_time_option ?lecture_section ?lecture_time_option)
        (discussion_section_time_option ?discussion_section ?discussion_time_option)
        (lecture_time_option_tentative ?lecture_time_option)
        (discussion_time_option_tentative ?discussion_time_option)
        (not
          (lecture_section_ready ?lecture_section)
        )
        (not
          (discussion_section_ready ?discussion_section)
        )
        (proposal_unallocated ?proposal)
      )
    :effect
      (and
        (proposal_reserved ?proposal)
        (proposal_lecture_time_option ?proposal ?lecture_time_option)
        (proposal_discussion_time_option ?proposal ?discussion_time_option)
        (proposal_requires_lecture_room_check ?proposal)
        (proposal_requires_discussion_room_check ?proposal)
        (not
          (proposal_unallocated ?proposal)
        )
      )
  )
  (:action confirm_proposal_time_assignment
    :parameters (?proposal - section_schedule_proposal ?lecture_section - lecture_section ?time_slot - time_slot)
    :precondition
      (and
        (proposal_reserved ?proposal)
        (lecture_section_locked ?lecture_section)
        (has_time_assignment ?lecture_section ?time_slot)
        (not
          (proposal_time_confirmed ?proposal)
        )
      )
    :effect (proposal_time_confirmed ?proposal)
  )
  (:action reserve_classroom_for_proposal
    :parameters (?course_offering - course_offering ?classroom - classroom ?proposal - section_schedule_proposal)
    :precondition
      (and
        (ready_for_assignment ?course_offering)
        (offering_has_proposal ?course_offering ?proposal)
        (offering_classroom_compatibility ?course_offering ?classroom)
        (classroom_available ?classroom)
        (proposal_reserved ?proposal)
        (proposal_time_confirmed ?proposal)
        (not
          (classroom_reserved ?classroom)
        )
      )
    :effect
      (and
        (classroom_reserved ?classroom)
        (classroom_reserved_for_proposal ?classroom ?proposal)
        (not
          (classroom_available ?classroom)
        )
      )
  )
  (:action allocate_classroom_to_offering
    :parameters (?course_offering - course_offering ?classroom - classroom ?proposal - section_schedule_proposal ?time_slot - time_slot)
    :precondition
      (and
        (ready_for_assignment ?course_offering)
        (offering_classroom_compatibility ?course_offering ?classroom)
        (classroom_reserved ?classroom)
        (classroom_reserved_for_proposal ?classroom ?proposal)
        (has_time_assignment ?course_offering ?time_slot)
        (not
          (proposal_requires_lecture_room_check ?proposal)
        )
        (not
          (offering_room_allocated ?course_offering)
        )
      )
    :effect (offering_room_allocated ?course_offering)
  )
  (:action assign_section_attribute_to_offering
    :parameters (?course_offering - course_offering ?section_attribute - section_attribute)
    :precondition
      (and
        (ready_for_assignment ?course_offering)
        (section_attribute_available ?section_attribute)
        (not
          (offering_attribute_claimed ?course_offering)
        )
      )
    :effect
      (and
        (offering_attribute_claimed ?course_offering)
        (offering_has_attribute ?course_offering ?section_attribute)
        (not
          (section_attribute_available ?section_attribute)
        )
      )
  )
  (:action apply_attribute_and_confirm_offering_allocation
    :parameters (?course_offering - course_offering ?classroom - classroom ?proposal - section_schedule_proposal ?time_slot - time_slot ?section_attribute - section_attribute)
    :precondition
      (and
        (ready_for_assignment ?course_offering)
        (offering_classroom_compatibility ?course_offering ?classroom)
        (classroom_reserved ?classroom)
        (classroom_reserved_for_proposal ?classroom ?proposal)
        (has_time_assignment ?course_offering ?time_slot)
        (proposal_requires_lecture_room_check ?proposal)
        (offering_attribute_claimed ?course_offering)
        (offering_has_attribute ?course_offering ?section_attribute)
        (not
          (offering_room_allocated ?course_offering)
        )
      )
    :effect
      (and
        (offering_room_allocated ?course_offering)
        (offering_attribute_confirmed ?course_offering)
      )
  )
  (:action allocate_resources_and_mark_offering
    :parameters (?course_offering - course_offering ?resource_request - resource_request ?administrator - course_administrator ?classroom - classroom ?proposal - section_schedule_proposal)
    :precondition
      (and
        (offering_room_allocated ?course_offering)
        (offering_has_resource_request ?course_offering ?resource_request)
        (has_administrator_claim ?course_offering ?administrator)
        (offering_classroom_compatibility ?course_offering ?classroom)
        (classroom_reserved_for_proposal ?classroom ?proposal)
        (not
          (proposal_requires_discussion_room_check ?proposal)
