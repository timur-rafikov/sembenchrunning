--- domain.pddl (as passed to VAL) ---
(define (domain finance_account_certification_workflow)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_token_group - object document_type - object metadata_type - object certification_root_type - object account_certification - certification_root_type approver_availability_token - resource_token_group evidence_document - resource_token_group approver_user - resource_token_group compliance_element - resource_token_group disclosure_checklist_item - resource_token_group signoff_credential - resource_token_group special_approval - resource_token_group audit_trail_reference - resource_token_group exception_record - document_type workpaper - document_type external_reviewer_credential - document_type control_test - metadata_type review_test - metadata_type certification_report - metadata_type certification_role_container - account_certification certification_package_container - account_certification preparer_assignment - certification_role_container reviewer_assignment - certification_role_container certification_package - certification_package_container)
  (:predicates
    (certification_entity_initiated ?certification_entity - account_certification)
    (certification_entity_engaged ?certification_entity - account_certification)
    (certification_entity_approver_allocated ?certification_entity - account_certification)
    (certification_entity_closed ?certification_entity - account_certification)
    (certification_entity_signoff_ready ?certification_entity - account_certification)
    (certification_entity_signed_off ?certification_entity - account_certification)
    (approver_token_available ?approver_token - approver_availability_token)
    (certification_entity_has_approver_token ?certification_entity - account_certification ?approver_token - approver_availability_token)
    (evidence_available ?evidence_document - evidence_document)
    (certification_entity_has_evidence ?certification_entity - account_certification ?evidence_document - evidence_document)
    (approver_user_available ?approver_user - approver_user)
    (certification_entity_assigned_approver ?certification_entity - account_certification ?approver_user - approver_user)
    (exception_available ?exception_record - exception_record)
    (preparer_exception_assigned ?preparer_assignment - preparer_assignment ?exception_record - exception_record)
    (reviewer_exception_assigned ?reviewer_assignment - reviewer_assignment ?exception_record - exception_record)
    (preparer_has_control_test ?preparer_assignment - preparer_assignment ?control_test - control_test)
    (control_test_passed ?control_test - control_test)
    (control_test_failed ?control_test - control_test)
    (preparer_validation_recorded ?preparer_assignment - preparer_assignment)
    (reviewer_has_review_test ?reviewer_assignment - reviewer_assignment ?review_test - review_test)
    (review_test_passed ?review_test - review_test)
    (review_test_failed ?review_test - review_test)
    (reviewer_validation_recorded ?reviewer_assignment - reviewer_assignment)
    (report_template_available ?certification_report - certification_report)
    (report_assembled ?certification_report - certification_report)
    (report_includes_control_test ?certification_report - certification_report ?control_test - control_test)
    (report_includes_review_test ?certification_report - certification_report ?review_test - review_test)
    (report_preparer_exception_flag ?certification_report - certification_report)
    (report_reviewer_exception_flag ?certification_report - certification_report)
    (report_preparer_validation_complete ?certification_report - certification_report)
    (package_includes_preparer_assignment ?certification_package - certification_package ?preparer_assignment - preparer_assignment)
    (package_includes_reviewer_assignment ?certification_package - certification_package ?reviewer_assignment - reviewer_assignment)
    (package_includes_report ?certification_package - certification_package ?certification_report - certification_report)
    (workpaper_available ?workpaper - workpaper)
    (package_contains_workpaper ?certification_package - certification_package ?workpaper - workpaper)
    (workpaper_attached ?workpaper - workpaper)
    (workpaper_linked_report ?workpaper - workpaper ?certification_report - certification_report)
    (package_compilation_started ?certification_package - certification_package)
    (package_compiled ?certification_package - certification_package)
    (package_under_external_review ?certification_package - certification_package)
    (package_has_compliance_element ?certification_package - certification_package)
    (package_compliance_element_validated ?certification_package - certification_package)
    (package_checklist_completed ?certification_package - certification_package)
    (package_ready_for_signoff ?certification_package - certification_package)
    (external_reviewer_credential_available ?external_reviewer_credential - external_reviewer_credential)
    (package_assigned_external_reviewer ?certification_package - certification_package ?external_reviewer_credential - external_reviewer_credential)
    (control_element_approval_requested ?certification_package - certification_package)
    (control_element_approval_recorded ?certification_package - certification_package)
    (control_element_approval_confirmed ?certification_package - certification_package)
    (compliance_element_available ?compliance_element - compliance_element)
    (package_includes_compliance_element ?certification_package - certification_package ?compliance_element - compliance_element)
    (disclosure_checklist_item_available ?disclosure_checklist_item - disclosure_checklist_item)
    (package_includes_disclosure_item ?certification_package - certification_package ?disclosure_checklist_item - disclosure_checklist_item)
    (special_approval_available ?special_approval - special_approval)
    (package_has_special_approval ?certification_package - certification_package ?special_approval - special_approval)
    (audit_trail_reference_available ?audit_trail_reference - audit_trail_reference)
    (package_includes_audit_trail_reference ?certification_package - certification_package ?audit_trail_reference - audit_trail_reference)
    (signoff_credential_available ?signoff_credential - signoff_credential)
    (certification_entity_signed_with_credential ?certification_entity - account_certification ?signoff_credential - signoff_credential)
    (preparer_tests_completed ?preparer_assignment - preparer_assignment)
    (reviewer_tests_completed ?reviewer_assignment - reviewer_assignment)
    (signoff_registered ?certification_package - certification_package)
  )
  (:action create_account_certification
    :parameters (?certification_entity - account_certification)
    :precondition
      (and
        (not
          (certification_entity_initiated ?certification_entity)
        )
        (not
          (certification_entity_closed ?certification_entity)
        )
      )
    :effect (certification_entity_initiated ?certification_entity)
  )
  (:action assign_approver_token
    :parameters (?certification_entity - account_certification ?approver_token - approver_availability_token)
    :precondition
      (and
        (certification_entity_initiated ?certification_entity)
        (not
          (certification_entity_approver_allocated ?certification_entity)
        )
        (approver_token_available ?approver_token)
      )
    :effect
      (and
        (certification_entity_approver_allocated ?certification_entity)
        (certification_entity_has_approver_token ?certification_entity ?approver_token)
        (not
          (approver_token_available ?approver_token)
        )
      )
  )
  (:action attach_evidence_document
    :parameters (?certification_entity - account_certification ?evidence_document - evidence_document)
    :precondition
      (and
        (certification_entity_initiated ?certification_entity)
        (certification_entity_approver_allocated ?certification_entity)
        (evidence_available ?evidence_document)
      )
    :effect
      (and
        (certification_entity_has_evidence ?certification_entity ?evidence_document)
        (not
          (evidence_available ?evidence_document)
        )
      )
  )
  (:action activate_preparer_work
    :parameters (?certification_entity - account_certification ?evidence_document - evidence_document)
    :precondition
      (and
        (certification_entity_initiated ?certification_entity)
        (certification_entity_approver_allocated ?certification_entity)
        (certification_entity_has_evidence ?certification_entity ?evidence_document)
        (not
          (certification_entity_engaged ?certification_entity)
        )
      )
    :effect (certification_entity_engaged ?certification_entity)
  )
  (:action detach_evidence_document
    :parameters (?certification_entity - account_certification ?evidence_document - evidence_document)
    :precondition
      (and
        (certification_entity_has_evidence ?certification_entity ?evidence_document)
      )
    :effect
      (and
        (evidence_available ?evidence_document)
        (not
          (certification_entity_has_evidence ?certification_entity ?evidence_document)
        )
      )
  )
  (:action assign_approver_user
    :parameters (?certification_entity - account_certification ?approver_user - approver_user)
    :precondition
      (and
        (certification_entity_engaged ?certification_entity)
        (approver_user_available ?approver_user)
      )
    :effect
      (and
        (certification_entity_assigned_approver ?certification_entity ?approver_user)
        (not
          (approver_user_available ?approver_user)
        )
      )
  )
  (:action unassign_approver_user
    :parameters (?certification_entity - account_certification ?approver_user - approver_user)
    :precondition
      (and
        (certification_entity_assigned_approver ?certification_entity ?approver_user)
      )
    :effect
      (and
        (approver_user_available ?approver_user)
        (not
          (certification_entity_assigned_approver ?certification_entity ?approver_user)
        )
      )
  )
  (:action request_special_approval
    :parameters (?certification_package - certification_package ?special_approval - special_approval)
    :precondition
      (and
        (certification_entity_engaged ?certification_package)
        (special_approval_available ?special_approval)
      )
    :effect
      (and
        (package_has_special_approval ?certification_package ?special_approval)
        (not
          (special_approval_available ?special_approval)
        )
      )
  )
  (:action revoke_special_approval
    :parameters (?certification_package - certification_package ?special_approval - special_approval)
    :precondition
      (and
        (package_has_special_approval ?certification_package ?special_approval)
      )
    :effect
      (and
        (special_approval_available ?special_approval)
        (not
          (package_has_special_approval ?certification_package ?special_approval)
        )
      )
  )
  (:action record_external_review_reference
    :parameters (?certification_package - certification_package ?audit_trail_reference - audit_trail_reference)
    :precondition
      (and
        (certification_entity_engaged ?certification_package)
        (audit_trail_reference_available ?audit_trail_reference)
      )
    :effect
      (and
        (package_includes_audit_trail_reference ?certification_package ?audit_trail_reference)
        (not
          (audit_trail_reference_available ?audit_trail_reference)
        )
      )
  )
  (:action remove_external_review_reference
    :parameters (?certification_package - certification_package ?audit_trail_reference - audit_trail_reference)
    :precondition
      (and
        (package_includes_audit_trail_reference ?certification_package ?audit_trail_reference)
      )
    :effect
      (and
        (audit_trail_reference_available ?audit_trail_reference)
        (not
          (package_includes_audit_trail_reference ?certification_package ?audit_trail_reference)
        )
      )
  )
  (:action execute_preparer_control_test
    :parameters (?preparer_assignment - preparer_assignment ?control_test - control_test ?evidence_document - evidence_document)
    :precondition
      (and
        (certification_entity_engaged ?preparer_assignment)
        (certification_entity_has_evidence ?preparer_assignment ?evidence_document)
        (preparer_has_control_test ?preparer_assignment ?control_test)
        (not
          (control_test_passed ?control_test)
        )
        (not
          (control_test_failed ?control_test)
        )
      )
    :effect (control_test_passed ?control_test)
  )
  (:action validate_preparer_test
    :parameters (?preparer_assignment - preparer_assignment ?control_test - control_test ?approver_user - approver_user)
    :precondition
      (and
        (certification_entity_engaged ?preparer_assignment)
        (certification_entity_assigned_approver ?preparer_assignment ?approver_user)
        (preparer_has_control_test ?preparer_assignment ?control_test)
        (control_test_passed ?control_test)
        (not
          (preparer_tests_completed ?preparer_assignment)
        )
      )
    :effect
      (and
        (preparer_tests_completed ?preparer_assignment)
        (preparer_validation_recorded ?preparer_assignment)
      )
  )
  (:action raise_exception_for_preparer_test
    :parameters (?preparer_assignment - preparer_assignment ?control_test - control_test ?exception_record - exception_record)
    :precondition
      (and
        (certification_entity_engaged ?preparer_assignment)
        (preparer_has_control_test ?preparer_assignment ?control_test)
        (exception_available ?exception_record)
        (not
          (preparer_tests_completed ?preparer_assignment)
        )
      )
    :effect
      (and
        (control_test_failed ?control_test)
        (preparer_tests_completed ?preparer_assignment)
        (preparer_exception_assigned ?preparer_assignment ?exception_record)
        (not
          (exception_available ?exception_record)
        )
      )
  )
  (:action remediate_preparer_exception
    :parameters (?preparer_assignment - preparer_assignment ?control_test - control_test ?evidence_document - evidence_document ?exception_record - exception_record)
    :precondition
      (and
        (certification_entity_engaged ?preparer_assignment)
        (certification_entity_has_evidence ?preparer_assignment ?evidence_document)
        (preparer_has_control_test ?preparer_assignment ?control_test)
        (control_test_failed ?control_test)
        (preparer_exception_assigned ?preparer_assignment ?exception_record)
        (not
          (preparer_validation_recorded ?preparer_assignment)
        )
      )
    :effect
      (and
        (control_test_passed ?control_test)
        (preparer_validation_recorded ?preparer_assignment)
        (exception_available ?exception_record)
        (not
          (preparer_exception_assigned ?preparer_assignment ?exception_record)
        )
      )
  )
  (:action execute_reviewer_control_test
    :parameters (?reviewer_assignment - reviewer_assignment ?review_test - review_test ?evidence_document - evidence_document)
    :precondition
      (and
        (certification_entity_engaged ?reviewer_assignment)
        (certification_entity_has_evidence ?reviewer_assignment ?evidence_document)
        (reviewer_has_review_test ?reviewer_assignment ?review_test)
        (not
          (review_test_passed ?review_test)
        )
        (not
          (review_test_failed ?review_test)
        )
      )
    :effect (review_test_passed ?review_test)
  )
  (:action validate_reviewer_test
    :parameters (?reviewer_assignment - reviewer_assignment ?review_test - review_test ?approver_user - approver_user)
    :precondition
      (and
        (certification_entity_engaged ?reviewer_assignment)
        (certification_entity_assigned_approver ?reviewer_assignment ?approver_user)
        (reviewer_has_review_test ?reviewer_assignment ?review_test)
        (review_test_passed ?review_test)
        (not
          (reviewer_tests_completed ?reviewer_assignment)
        )
      )
    :effect
      (and
        (reviewer_tests_completed ?reviewer_assignment)
        (reviewer_validation_recorded ?reviewer_assignment)
      )
  )
  (:action raise_exception_for_reviewer_test
    :parameters (?reviewer_assignment - reviewer_assignment ?review_test - review_test ?exception_record - exception_record)
    :precondition
      (and
        (certification_entity_engaged ?reviewer_assignment)
        (reviewer_has_review_test ?reviewer_assignment ?review_test)
        (exception_available ?exception_record)
        (not
          (reviewer_tests_completed ?reviewer_assignment)
        )
      )
    :effect
      (and
        (review_test_failed ?review_test)
        (reviewer_tests_completed ?reviewer_assignment)
        (reviewer_exception_assigned ?reviewer_assignment ?exception_record)
        (not
          (exception_available ?exception_record)
        )
      )
  )
  (:action remediate_reviewer_exception
    :parameters (?reviewer_assignment - reviewer_assignment ?review_test - review_test ?evidence_document - evidence_document ?exception_record - exception_record)
    :precondition
      (and
        (certification_entity_engaged ?reviewer_assignment)
        (certification_entity_has_evidence ?reviewer_assignment ?evidence_document)
        (reviewer_has_review_test ?reviewer_assignment ?review_test)
        (review_test_failed ?review_test)
        (reviewer_exception_assigned ?reviewer_assignment ?exception_record)
        (not
          (reviewer_validation_recorded ?reviewer_assignment)
        )
      )
    :effect
      (and
        (review_test_passed ?review_test)
        (reviewer_validation_recorded ?reviewer_assignment)
        (exception_available ?exception_record)
        (not
          (reviewer_exception_assigned ?reviewer_assignment ?exception_record)
        )
      )
  )
  (:action assemble_certification_report
    :parameters (?preparer_assignment - preparer_assignment ?reviewer_assignment - reviewer_assignment ?control_test - control_test ?review_test - review_test ?certification_report - certification_report)
    :precondition
      (and
        (preparer_tests_completed ?preparer_assignment)
        (reviewer_tests_completed ?reviewer_assignment)
        (preparer_has_control_test ?preparer_assignment ?control_test)
        (reviewer_has_review_test ?reviewer_assignment ?review_test)
        (control_test_passed ?control_test)
        (review_test_passed ?review_test)
        (preparer_validation_recorded ?preparer_assignment)
        (reviewer_validation_recorded ?reviewer_assignment)
        (report_template_available ?certification_report)
      )
    :effect
      (and
        (report_assembled ?certification_report)
        (report_includes_control_test ?certification_report ?control_test)
        (report_includes_review_test ?certification_report ?review_test)
        (not
          (report_template_available ?certification_report)
        )
      )
  )
  (:action assemble_report_with_preparer_issue
    :parameters (?preparer_assignment - preparer_assignment ?reviewer_assignment - reviewer_assignment ?control_test - control_test ?review_test - review_test ?certification_report - certification_report)
    :precondition
      (and
        (preparer_tests_completed ?preparer_assignment)
        (reviewer_tests_completed ?reviewer_assignment)
        (preparer_has_control_test ?preparer_assignment ?control_test)
        (reviewer_has_review_test ?reviewer_assignment ?review_test)
        (control_test_failed ?control_test)
        (review_test_passed ?review_test)
        (not
          (preparer_validation_recorded ?preparer_assignment)
        )
        (reviewer_validation_recorded ?reviewer_assignment)
        (report_template_available ?certification_report)
      )
    :effect
      (and
        (report_assembled ?certification_report)
        (report_includes_control_test ?certification_report ?control_test)
        (report_includes_review_test ?certification_report ?review_test)
        (report_preparer_exception_flag ?certification_report)
        (not
          (report_template_available ?certification_report)
        )
      )
  )
  (:action assemble_report_with_reviewer_issue
    :parameters (?preparer_assignment - preparer_assignment ?reviewer_assignment - reviewer_assignment ?control_test - control_test ?review_test - review_test ?certification_report - certification_report)
    :precondition
      (and
        (preparer_tests_completed ?preparer_assignment)
        (reviewer_tests_completed ?reviewer_assignment)
        (preparer_has_control_test ?preparer_assignment ?control_test)
        (reviewer_has_review_test ?reviewer_assignment ?review_test)
        (control_test_passed ?control_test)
        (review_test_failed ?review_test)
        (preparer_validation_recorded ?preparer_assignment)
        (not
          (reviewer_validation_recorded ?reviewer_assignment)
        )
        (report_template_available ?certification_report)
      )
    :effect
      (and
        (report_assembled ?certification_report)
        (report_includes_control_test ?certification_report ?control_test)
        (report_includes_review_test ?certification_report ?review_test)
        (report_reviewer_exception_flag ?certification_report)
        (not
          (report_template_available ?certification_report)
        )
      )
  )
  (:action assemble_report_with_both_issues
    :parameters (?preparer_assignment - preparer_assignment ?reviewer_assignment - reviewer_assignment ?control_test - control_test ?review_test - review_test ?certification_report - certification_report)
    :precondition
      (and
        (preparer_tests_completed ?preparer_assignment)
        (reviewer_tests_completed ?reviewer_assignment)
        (preparer_has_control_test ?preparer_assignment ?control_test)
        (reviewer_has_review_test ?reviewer_assignment ?review_test)
        (control_test_failed ?control_test)
        (review_test_failed ?review_test)
        (not
          (preparer_validation_recorded ?preparer_assignment)
        )
        (not
          (reviewer_validation_recorded ?reviewer_assignment)
        )
        (report_template_available ?certification_report)
      )
    :effect
      (and
        (report_assembled ?certification_report)
        (report_includes_control_test ?certification_report ?control_test)
        (report_includes_review_test ?certification_report ?review_test)
        (report_preparer_exception_flag ?certification_report)
        (report_reviewer_exception_flag ?certification_report)
        (not
          (report_template_available ?certification_report)
        )
      )
  )
  (:action finalize_report_preparer_validation
    :parameters (?certification_report - certification_report ?preparer_assignment - preparer_assignment ?evidence_document - evidence_document)
    :precondition
      (and
        (report_assembled ?certification_report)
        (preparer_tests_completed ?preparer_assignment)
        (certification_entity_has_evidence ?preparer_assignment ?evidence_document)
        (not
          (report_preparer_validation_complete ?certification_report)
        )
      )
    :effect (report_preparer_validation_complete ?certification_report)
  )
  (:action attach_workpaper_to_package
    :parameters (?certification_package - certification_package ?workpaper - workpaper ?certification_report - certification_report)
    :precondition
      (and
        (certification_entity_engaged ?certification_package)
        (package_includes_report ?certification_package ?certification_report)
        (package_contains_workpaper ?certification_package ?workpaper)
        (workpaper_available ?workpaper)
        (report_assembled ?certification_report)
        (report_preparer_validation_complete ?certification_report)
        (not
          (workpaper_attached ?workpaper)
        )
      )
    :effect
      (and
        (workpaper_attached ?workpaper)
        (workpaper_linked_report ?workpaper ?certification_report)
        (not
          (workpaper_available ?workpaper)
        )
      )
  )
  (:action validate_workpaper
    :parameters (?certification_package - certification_package ?workpaper - workpaper ?certification_report - certification_report ?evidence_document - evidence_document)
    :precondition
      (and
        (certification_entity_engaged ?certification_package)
        (package_contains_workpaper ?certification_package ?workpaper)
        (workpaper_attached ?workpaper)
        (workpaper_linked_report ?workpaper ?certification_report)
        (certification_entity_has_evidence ?certification_package ?evidence_document)
        (not
          (report_preparer_exception_flag ?certification_report)
        )
        (not
          (package_compilation_started ?certification_package)
        )
      )
    :effect (package_compilation_started ?certification_package)
  )
  (:action attach_compliance_element_to_package
    :parameters (?certification_package - certification_package ?compliance_element - compliance_element)
    :precondition
      (and
        (certification_entity_engaged ?certification_package)
        (compliance_element_available ?compliance_element)
        (not
          (package_has_compliance_element ?certification_package)
        )
      )
    :effect
      (and
        (package_has_compliance_element ?certification_package)
        (package_includes_compliance_element ?certification_package ?compliance_element)
        (not
          (compliance_element_available ?compliance_element)
        )
      )
  )
  (:action complete_compliance_element_validation
    :parameters (?certification_package - certification_package ?workpaper - workpaper ?certification_report - certification_report ?evidence_document - evidence_document ?compliance_element - compliance_element)
    :precondition
      (and
        (certification_entity_engaged ?certification_package)
        (package_contains_workpaper ?certification_package ?workpaper)
        (workpaper_attached ?workpaper)
        (workpaper_linked_report ?workpaper ?certification_report)
        (certification_entity_has_evidence ?certification_package ?evidence_document)
        (report_preparer_exception_flag ?certification_report)
        (package_has_compliance_element ?certification_package)
        (package_includes_compliance_element ?certification_package ?compliance_element)
        (not
          (package_compilation_started ?certification_package)
        )
      )
    :effect
      (and
        (package_compilation_started ?certification_package)
        (package_compliance_element_validated ?certification_package)
      )
  )
  (:action apply_special_approval_and_compile_package
    :parameters (?certification_package - certification_package ?special_approval - special_approval ?approver_user - approver_user ?workpaper - workpaper ?certification_report - certification_report)
    :precondition
      (and
        (package_compilation_started ?certification_package)
        (package_has_special_approval ?certification_package ?special_approval)
        (certification_entity_assigned_approver ?certification_package ?approver_user)
        (package_contains_workpaper ?certification_package ?workpaper)
