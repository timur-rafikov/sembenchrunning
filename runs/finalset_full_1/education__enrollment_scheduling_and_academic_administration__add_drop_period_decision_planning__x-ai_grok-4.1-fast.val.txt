  (:requirements :strips :typing :negative-preconditions)
  (:types institutional_entity - object temporal_entity - object physical_entity - object add_drop_case_root - object case_entity - add_drop_case_root section_instance - institutional_entity student_identity - institutional_entity approver_account - institutional_entity departmental_approval_slot - institutional_entity priority_code - institutional_entity justification_document - institutional_entity qualification_token - institutional_entity external_approval_token - institutional_entity supporting_document - temporal_entity special_resource - temporal_entity departmental_signature - temporal_entity student_timeslot_candidate - physical_entity instructor_timeslot_candidate - physical_entity classroom_resource - physical_entity personnel_group - case_entity organizational_unit - case_entity student_record - personnel_group instructor_record - personnel_group course_offering - organizational_unit)
  (:predicates
    (case_recorded ?course_registration_request - case_entity)
    (case_validated ?course_registration_request - case_entity)
    (case_provisionally_assigned ?course_registration_request - case_entity)
    (case_enrollment_finalized ?course_registration_request - case_entity)
    (case_finalization_ready ?course_registration_request - case_entity)
    (case_authorized ?course_registration_request - case_entity)
    (section_available ?section_instance - section_instance)
    (case_assigned_section ?course_registration_request - case_entity ?section_instance - section_instance)

    (case_linked_student ?course_registration_request - case_entity ?student_identity - student_identity)
    (approver_available ?approver_account - approver_account)
    (case_assigned_approver ?course_registration_request - case_entity ?approver_account - approver_account)
    (supporting_document_available ?supporting_document - supporting_document)
    (student_record_document_attached ?student_record - student_record ?supporting_document - supporting_document)
    (instructor_record_document_attached ?instructor_record - instructor_record ?supporting_document - supporting_document)
    (student_timeslot_candidate_linked ?student_record - student_record ?student_timeslot_candidate - student_timeslot_candidate)
    (student_timeslot_confirmed ?student_timeslot_candidate - student_timeslot_candidate)
    (student_timeslot_reserved ?student_timeslot_candidate - student_timeslot_candidate)
    (student_timeslot_committed ?student_record - student_record)
    (instructor_timeslot_candidate_linked ?instructor_record - instructor_record ?instructor_timeslot_candidate - instructor_timeslot_candidate)
    (instructor_timeslot_confirmed ?instructor_timeslot_candidate - instructor_timeslot_candidate)
    (instructor_timeslot_reserved ?instructor_timeslot_candidate - instructor_timeslot_candidate)
    (instructor_timeslot_committed ?instructor_record - instructor_record)
    (room_available ?classroom_resource - classroom_resource)
    (room_provisionally_reserved ?classroom_resource - classroom_resource)
    (room_assigned_student_timeslot ?classroom_resource - classroom_resource ?student_timeslot_candidate - student_timeslot_candidate)
    (room_assigned_instructor_timeslot ?classroom_resource - classroom_resource ?instructor_timeslot_candidate - instructor_timeslot_candidate)
    (room_secondary_confirmation ?classroom_resource - classroom_resource)
    (room_alternative_confirmation ?classroom_resource - classroom_resource)
    (room_allocation_confirmed ?classroom_resource - classroom_resource)
    (offering_linked_student_record ?course_offering - course_offering ?student_record - student_record)
    (offering_linked_instructor_record ?course_offering - course_offering ?instructor_record - instructor_record)
    (offering_assigned_room ?course_offering - course_offering ?classroom_resource - classroom_resource)
    (special_resource_available ?special_resource - special_resource)
    (offering_requires_special_resource ?course_offering - course_offering ?special_resource - special_resource)
    (special_resource_reserved ?special_resource - special_resource)

    (offering_departmental_slot_reserved ?course_offering - course_offering)
    (offering_departmental_approval_attached ?course_offering - course_offering)
    (offering_priority_confirmed ?course_offering - course_offering)
    (offering_schedule_confirmed ?course_offering - course_offering)
    (departmental_signature_available ?departmental_signature - departmental_signature)
    (offering_linked_departmental_signature ?course_offering - course_offering ?departmental_signature - departmental_signature)
    (offering_departmental_signature_attached ?course_offering - course_offering)
    (offering_signature_accepted ?course_offering - course_offering)
    (offering_external_approval_attached ?course_offering - course_offering)
    (departmental_approval_slot_available ?departmental_approval_slot - departmental_approval_slot)
    (offering_assigned_departmental_slot ?course_offering - course_offering ?departmental_approval_slot - departmental_approval_slot)
    (priority_code_available ?priority_code - priority_code)
    (offering_assigned_priority_code ?course_offering - course_offering ?priority_code - priority_code)
    (qualification_token_available ?qualification_token - qualification_token)
    (offering_assigned_qualification_token ?course_offering - course_offering ?qualification_token - qualification_token)
    (external_approval_token_available ?external_approval_token - external_approval_token)
    (offering_assigned_external_approval_token ?course_offering - course_offering ?external_approval_token - external_approval_token)

    (case_attached_justification ?course_registration_request - case_entity ?justification_document - justification_document)

    (instructor_ready_for_proposal ?instructor_record - instructor_record)
    (offering_commit_recorded ?course_offering - course_offering)
  )
  (:action submit_add_drop_request
    :parameters (?course_registration_request - case_entity)
    :precondition
      (and
        (not
          (case_recorded ?course_registration_request)
        )
        (not
          (case_enrollment_finalized ?course_registration_request)
        )
      )
    :effect (case_recorded ?course_registration_request)
  )
  (:action provisionally_assign_section_to_request
    :parameters (?course_registration_request - case_entity ?section_instance - section_instance)
    :precondition
      (and
        (case_recorded ?course_registration_request)
        (not
          (case_provisionally_assigned ?course_registration_request)
        )
        (section_available ?section_instance)
      )
    :effect
      (and
        (case_provisionally_assigned ?course_registration_request)
        (case_assigned_section ?course_registration_request ?section_instance)
        (not
          (section_available ?section_instance)
        )
      )
  )
  (:action link_request_to_student_identity
    :parameters (?course_registration_request - case_entity ?student_identity - student_identity)
    :precondition
      (and
        (case_recorded ?course_registration_request)
        (case_provisionally_assigned ?course_registration_request)
        (student_availability_flag ?student_identity)
      )
    :effect
      (and
        (case_linked_student ?course_registration_request ?student_identity)
        (not
          (student_availability_flag ?student_identity)
        )
      )
  )
  (:action validate_request
    :parameters (?course_registration_request - case_entity ?student_identity - student_identity)
    :precondition
      (and
        (case_recorded ?course_registration_request)
        (case_provisionally_assigned ?course_registration_request)
        (case_linked_student ?course_registration_request ?student_identity)
        (not
          (case_validated ?course_registration_request)
        )
      )
    :effect (case_validated ?course_registration_request)
  )
  (:action release_student_identity_link
    :parameters (?course_registration_request - case_entity ?student_identity - student_identity)
    :precondition
      (and
        (case_linked_student ?course_registration_request ?student_identity)
      )
    :effect
      (and
        (student_availability_flag ?student_identity)
        (not
          (case_linked_student ?course_registration_request ?student_identity)
        )
      )
  )
  (:action assign_approver_to_request
    :parameters (?course_registration_request - case_entity ?approver_account - approver_account)
    :precondition
      (and
        (case_validated ?course_registration_request)
        (approver_available ?approver_account)
      )
    :effect
      (and
        (case_assigned_approver ?course_registration_request ?approver_account)
        (not
          (approver_available ?approver_account)
        )
      )
  )
  (:action unassign_approver_from_request
    :parameters (?course_registration_request - case_entity ?approver_account - approver_account)
    :precondition
      (and
        (case_assigned_approver ?course_registration_request ?approver_account)
      )
    :effect
      (and
        (approver_available ?approver_account)
        (not
          (case_assigned_approver ?course_registration_request ?approver_account)
        )
      )
  )
  (:action assign_qualification_to_offering
    :parameters (?course_offering - course_offering ?qualification_token - qualification_token)
    :precondition
      (and
        (case_validated ?course_offering)
        (qualification_token_available ?qualification_token)
      )
    :effect
      (and
        (offering_assigned_qualification_token ?course_offering ?qualification_token)
        (not
          (qualification_token_available ?qualification_token)
        )
      )
  )
  (:action release_qualification_from_offering
    :parameters (?course_offering - course_offering ?qualification_token - qualification_token)
    :precondition
      (and
        (offering_assigned_qualification_token ?course_offering ?qualification_token)
      )
    :effect
      (and
        (qualification_token_available ?qualification_token)
        (not
          (offering_assigned_qualification_token ?course_offering ?qualification_token)
        )
      )
  )
  (:action assign_external_approval_to_offering
    :parameters (?course_offering - course_offering ?external_approval_token - external_approval_token)
    :precondition
      (and
        (case_validated ?course_offering)
        (external_approval_token_available ?external_approval_token)
      )
    :effect
      (and
        (offering_assigned_external_approval_token ?course_offering ?external_approval_token)
        (not
          (external_approval_token_available ?external_approval_token)
        )
      )
  )
  (:action release_external_approval_from_offering
    :parameters (?course_offering - course_offering ?external_approval_token - external_approval_token)
    :precondition
      (and
        (offering_assigned_external_approval_token ?course_offering ?external_approval_token)
      )
    :effect
      (and
        (external_approval_token_available ?external_approval_token)
        (not
          (offering_assigned_external_approval_token ?course_offering ?external_approval_token)
        )
      )
  )
  (:action select_student_timeslot
    :parameters (?student_record - student_record ?student_timeslot_candidate - student_timeslot_candidate ?student_identity - student_identity)
    :precondition
      (and
        (case_validated ?student_record)
        (case_linked_student ?student_record ?student_identity)
        (student_timeslot_candidate_linked ?student_record ?student_timeslot_candidate)
        (not
          (student_timeslot_confirmed ?student_timeslot_candidate)
        )
        (not
          (student_timeslot_reserved ?student_timeslot_candidate)
        )
      )
    :effect (student_timeslot_confirmed ?student_timeslot_candidate)
  )
  (:action confirm_student_timeslot_and_lock_record
    :parameters (?student_record - student_record ?student_timeslot_candidate - student_timeslot_candidate ?approver_account - approver_account)
    :precondition
      (and
        (case_validated ?student_record)
        (case_assigned_approver ?student_record ?approver_account)
        (student_timeslot_candidate_linked ?student_record ?student_timeslot_candidate)
        (student_timeslot_confirmed ?student_timeslot_candidate)
        (not
          (student_ready_for_proposal ?student_record)
        )
      )
    :effect
      (and
        (student_ready_for_proposal ?student_record)
        (student_timeslot_committed ?student_record)
      )
  )
  (:action reserve_student_timeslot_and_attach_document
    :parameters (?student_record - student_record ?student_timeslot_candidate - student_timeslot_candidate ?supporting_document - supporting_document)
    :precondition
      (and
        (case_validated ?student_record)
        (student_timeslot_candidate_linked ?student_record ?student_timeslot_candidate)
        (supporting_document_available ?supporting_document)
        (not
          (student_ready_for_proposal ?student_record)
        )
      )
    :effect
      (and
        (student_timeslot_reserved ?student_timeslot_candidate)
        (student_ready_for_proposal ?student_record)
        (student_record_document_attached ?student_record ?supporting_document)
        (not
          (supporting_document_available ?supporting_document)
        )
      )
  )
  (:action finalize_student_timeslot_and_restore_document
    :parameters (?student_record - student_record ?student_timeslot_candidate - student_timeslot_candidate ?student_identity - student_identity ?supporting_document - supporting_document)
    :precondition
      (and
        (case_validated ?student_record)
        (case_linked_student ?student_record ?student_identity)
        (student_timeslot_candidate_linked ?student_record ?student_timeslot_candidate)
        (student_timeslot_reserved ?student_timeslot_candidate)
        (student_record_document_attached ?student_record ?supporting_document)
        (not
          (student_timeslot_committed ?student_record)
        )
      )
    :effect
      (and
        (student_timeslot_confirmed ?student_timeslot_candidate)
        (student_timeslot_committed ?student_record)
        (supporting_document_available ?supporting_document)
        (not
          (student_record_document_attached ?student_record ?supporting_document)
        )
      )
  )
  (:action select_instructor_timeslot
    :parameters (?instructor_record - instructor_record ?instructor_timeslot_candidate - instructor_timeslot_candidate ?student_identity - student_identity)

      (and

  )
  (:action confirm_instructor_timeslot_and_lock_instructor
    :parameters (?instructor_record - instructor_record ?instructor_timeslot_candidate - instructor_timeslot_candidate ?approver_account - approver_account)
    :precondition
      (and
        (case_validated ?instructor_record)
        (case_assigned_approver ?instructor_record ?approver_account)
        (instructor_timeslot_candidate_linked ?instructor_record ?instructor_timeslot_candidate)
        (instructor_timeslot_confirmed ?instructor_timeslot_candidate)
        (not
          (instructor_ready_for_proposal ?instructor_record)
        )
      )
    :effect
      (and
        (instructor_ready_for_proposal ?instructor_record)
        (instructor_timeslot_committed ?instructor_record)
      )
  )
  (:action reserve_instructor_timeslot_and_attach_document
    :parameters (?instructor_record - instructor_record ?instructor_timeslot_candidate - instructor_timeslot_candidate ?supporting_document - supporting_document)
    :precondition
      (and
        (case_validated ?instructor_record)
        (instructor_timeslot_candidate_linked ?instructor_record ?instructor_timeslot_candidate)
        (supporting_document_available ?supporting_document)
        (not
          (instructor_ready_for_proposal ?instructor_record)
        )
      )
    :effect
      (and
        (instructor_timeslot_reserved ?instructor_timeslot_candidate)
        (instructor_ready_for_proposal ?instructor_record)
        (instructor_record_document_attached ?instructor_record ?supporting_document)
        (not
          (supporting_document_available ?supporting_document)
        )
      )
  )
  (:action finalize_instructor_timeslot_and_restore_document
    :parameters (?instructor_record - instructor_record ?instructor_timeslot_candidate - instructor_timeslot_candidate ?student_identity - student_identity ?supporting_document - supporting_document)
    :precondition
      (and
        (case_validated ?instructor_record)
        (case_linked_student ?instructor_record ?student_identity)
        (instructor_timeslot_candidate_linked ?instructor_record ?instructor_timeslot_candidate)
        (instructor_timeslot_reserved ?instructor_timeslot_candidate)
        (instructor_record_document_attached ?instructor_record ?supporting_document)
        (not
          (instructor_timeslot_committed ?instructor_record)
        )
      )
    :effect
      (and
        (instructor_timeslot_confirmed ?instructor_timeslot_candidate)
        (instructor_timeslot_committed ?instructor_record)
        (supporting_document_available ?supporting_document)
        (not
          (instructor_record_document_attached ?instructor_record ?supporting_document)
        )
      )
  )
  (:action propose_room_assignment
    :parameters (?student_record - student_record ?instructor_record - instructor_record ?student_timeslot_candidate - student_timeslot_candidate ?instructor_timeslot_candidate - instructor_timeslot_candidate ?classroom_resource - classroom_resource)
    :precondition
      (and
        (student_ready_for_proposal ?student_record)
        (instructor_ready_for_proposal ?instructor_record)
        (student_timeslot_candidate_linked ?student_record ?student_timeslot_candidate)
        (instructor_timeslot_candidate_linked ?instructor_record ?instructor_timeslot_candidate)
        (student_timeslot_confirmed ?student_timeslot_candidate)
        (instructor_timeslot_confirmed ?instructor_timeslot_candidate)
        (student_timeslot_committed ?student_record)
        (instructor_timeslot_committed ?instructor_record)
        (room_available ?classroom_resource)
      )
    :effect
      (and
        (room_provisionally_reserved ?classroom_resource)
        (room_assigned_student_timeslot ?classroom_resource ?student_timeslot_candidate)
        (room_assigned_instructor_timeslot ?classroom_resource ?instructor_timeslot_candidate)
        (not
          (room_available ?classroom_resource)
        )
      )
  )
  (:action propose_room_assignment_with_secondary_confirmation

    :precondition
      (and
        (student_ready_for_proposal ?student_record)
        (instructor_ready_for_proposal ?instructor_record)
        (student_timeslot_candidate_linked ?student_record ?student_timeslot_candidate)
        (instructor_timeslot_candidate_linked ?instructor_record ?instructor_timeslot_candidate)
        (student_timeslot_reserved ?student_timeslot_candidate)
        (instructor_timeslot_confirmed ?instructor_timeslot_candidate)
        (not
          (student_timeslot_committed ?student_record)
        )
        (instructor_timeslot_committed ?instructor_record)
        (room_available ?classroom_resource)
      )
    :effect
      (and
        (room_provisionally_reserved ?classroom_resource)
        (room_assigned_student_timeslot ?classroom_resource ?student_timeslot_candidate)
        (room_assigned_instructor_timeslot ?classroom_resource ?instructor_timeslot_candidate)
        (room_secondary_confirmation ?classroom_resource)
        (not
          (room_available ?classroom_resource)
        )
      )
  )
  (:action propose_room_assignment_with_alternative_confirmation
    :parameters (?student_record - student_record ?instructor_record - instructor_record ?student_timeslot_candidate - student_timeslot_candidate ?instructor_timeslot_candidate - instructor_timeslot_candidate ?classroom_resource - classroom_resource)
    :precondition
      (and
        (student_ready_for_proposal ?student_record)
        (instructor_ready_for_proposal ?instructor_record)
        (student_timeslot_candidate_linked ?student_record ?student_timeslot_candidate)
        (instructor_timeslot_candidate_linked ?instructor_record ?instructor_timeslot_candidate)
        (student_timeslot_confirmed ?student_timeslot_candidate)
        (instructor_timeslot_reserved ?instructor_timeslot_candidate)
        (student_timeslot_committed ?student_record)
        (not
          (instructor_timeslot_committed ?instructor_record)
        )
        (room_available ?classroom_resource)
      )
    :effect
      (and
        (room_provisionally_reserved ?classroom_resource)
        (room_assigned_student_timeslot ?classroom_resource ?student_timeslot_candidate)
        (room_assigned_instructor_timeslot ?classroom_resource ?instructor_timeslot_candidate)
        (room_alternative_confirmation ?classroom_resource)
        (not
          (room_available ?classroom_resource)
        )
      )
  )
  (:action propose_room_assignment_with_full_confirmations
    :parameters (?student_record - student_record ?instructor_record - instructor_record ?student_timeslot_candidate - student_timeslot_candidate ?instructor_timeslot_candidate - instructor_timeslot_candidate ?classroom_resource - classroom_resource)
    :precondition
      (and
        (student_ready_for_proposal ?student_record)
        (instructor_ready_for_proposal ?instructor_record)
        (student_timeslot_candidate_linked ?student_record ?student_timeslot_candidate)
        (instructor_timeslot_candidate_linked ?instructor_record ?instructor_timeslot_candidate)

        (instructor_timeslot_reserved ?instructor_timeslot_candidate)

          (student_timeslot_committed ?student_record)
        )
        (not
          (instructor_timeslot_committed ?instructor_record)
        )
        (room_available ?classroom_resource)
      )
    :effect
      (and
        (room_provisionally_reserved ?classroom_resource)
        (room_assigned_student_timeslot ?classroom_resource ?student_timeslot_candidate)
        (room_assigned_instructor_timeslot ?classroom_resource ?instructor_timeslot_candidate)
        (room_secondary_confirmation ?classroom_resource)
        (room_alternative_confirmation ?classroom_resource)
        (not
          (room_available ?classroom_resource)
        )
      )
  )
  (:action confirm_room_allocation
    :parameters (?classroom_resource - classroom_resource ?student_record - student_record ?student_identity - student_identity)
    :precondition
      (and
        (room_provisionally_reserved ?classroom_resource)
        (student_ready_for_proposal ?student_record)
        (case_linked_student ?student_record ?student_identity)
        (not
          (room_allocation_confirmed ?classroom_resource)
        )
      )
    :effect (room_allocation_confirmed ?classroom_resource)
  )
  (:action reserve_special_resource_for_offering
    :parameters (?course_offering - course_offering ?special_resource - special_resource ?classroom_resource - classroom_resource)
    :precondition
      (and
        (case_validated ?course_offering)
        (offering_assigned_room ?course_offering ?classroom_resource)
        (offering_requires_special_resource ?course_offering ?special_resource)
        (special_resource_available ?special_resource)
        (room_provisionally_reserved ?classroom_resource)
        (room_allocation_confirmed ?classroom_resource)
        (not
          (special_resource_reserved ?special_resource)
        )
      )
    :effect
      (and
        (special_resource_reserved ?special_resource)
        (special_resource_assigned_room ?special_resource ?classroom_resource)
        (not
          (special_resource_available ?special_resource)
        )
      )
  )
  (:action allocate_offering_resources
    :parameters (?course_offering - course_offering ?special_resource - special_resource ?classroom_resource - classroom_resource ?student_identity - student_identity)
    :precondition
      (and
        (case_validated ?course_offering)
        (offering_requires_special_resource ?course_offering ?special_resource)
        (special_resource_reserved ?special_resource)
        (special_resource_assigned_room ?special_resource ?classroom_resource)
        (case_linked_student ?course_offering ?student_identity)
        (not
          (room_secondary_confirmation ?classroom_resource)
        )
        (not
          (offering_resources_allocated ?course_offering)
        )
      )
    :effect (offering_resources_allocated ?course_offering)
  )
  (:action assign_departmental_slot_to_offering
    :parameters (?course_offering - course_offering ?departmental_approval_slot - departmental_approval_slot)
    :precondition
      (and
        (case_validated ?course_offering)
        (departmental_approval_slot_available ?departmental_approval_slot)
        (not
          (offering_departmental_slot_reserved ?course_offering)
        )
      )
    :effect
      (and
        (offering_departmental_slot_reserved ?course_offering)
        (offering_assigned_departmental_slot ?course_offering ?departmental_approval_slot)
        (not
          (departmental_approval_slot_available ?departmental_approval_slot)
        )
      )
