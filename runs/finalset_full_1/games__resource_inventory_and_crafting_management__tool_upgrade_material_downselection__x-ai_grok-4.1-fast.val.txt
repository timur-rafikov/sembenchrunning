--- domain.pddl (as passed to VAL) ---
(define (domain tool_upgrade_material_downselection)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_token - object material_token - object attachment_token - object equipment_root - object upgradeable_equipment - equipment_root raw_material - resource_token subcomponent - resource_token attachment_piece - resource_token optional_modifier - resource_token slot_modifier - resource_token preservative_item - resource_token attachment_type_a - resource_token attachment_type_c - resource_token consumable_bundle - material_token upgrade_blueprint - material_token attachment_type_b - material_token material_category - attachment_token material_variant - attachment_token upgrade_module - attachment_token tool_group_primary - upgradeable_equipment tool_group_auxiliary - upgradeable_equipment primary_tool - tool_group_primary secondary_tool - tool_group_primary auxiliary_tool - tool_group_auxiliary)
  (:predicates
    (equipment_is_candidate ?upgradeable_equipment - upgradeable_equipment)
    (equipment_selection_confirmed ?upgradeable_equipment - upgradeable_equipment)
    (equipment_material_proposed ?upgradeable_equipment - upgradeable_equipment)
    (equipment_upgrade_applied ?upgradeable_equipment - upgradeable_equipment)
    (equipment_ready_for_preservative ?upgradeable_equipment - upgradeable_equipment)
    (equipment_preservative_applied ?upgradeable_equipment - upgradeable_equipment)
    (raw_material_available ?raw_material - raw_material)
    (raw_material_reserved_for_equipment ?upgradeable_equipment - upgradeable_equipment ?raw_material - raw_material)
    (subcomponent_available ?subcomponent - subcomponent)
    (subcomponent_reserved_for_equipment ?upgradeable_equipment - upgradeable_equipment ?subcomponent - subcomponent)
    (attachment_piece_available ?attachment_piece - attachment_piece)
    (attachment_reserved_on_equipment ?upgradeable_equipment - upgradeable_equipment ?attachment_piece - attachment_piece)
    (consumable_bundle_available ?consumable_bundle - consumable_bundle)
    (consumable_allocated_to_primary_tool ?primary_tool - primary_tool ?consumable_bundle - consumable_bundle)
    (consumable_allocated_to_secondary_tool ?secondary_tool - secondary_tool ?consumable_bundle - consumable_bundle)
    (primary_tool_material_category ?primary_tool - primary_tool ?material_category - material_category)
    (material_category_marked ?material_category - material_category)
    (material_category_assigned_to_consumable ?material_category - material_category)
    (primary_tool_category_confirmed ?primary_tool - primary_tool)
    (secondary_tool_material_variant ?secondary_tool - secondary_tool ?material_variant - material_variant)
    (material_variant_marked ?material_variant - material_variant)
    (material_variant_assigned_to_consumable ?material_variant - material_variant)
    (secondary_tool_category_confirmed ?secondary_tool - secondary_tool)
    (upgrade_module_unprepared ?upgrade_module - upgrade_module)
    (upgrade_module_prepared ?upgrade_module - upgrade_module)
    (upgrade_module_uses_material_category ?upgrade_module - upgrade_module ?material_category - material_category)
    (upgrade_module_uses_material_variant ?upgrade_module - upgrade_module ?material_variant - material_variant)
    (upgrade_module_option_flag_1 ?upgrade_module - upgrade_module)
    (upgrade_module_option_flag_2 ?upgrade_module - upgrade_module)
    (upgrade_module_finalized ?upgrade_module - upgrade_module)
    (aux_tool_compatible_with_primary_tool ?auxiliary_tool - auxiliary_tool ?primary_tool - primary_tool)
    (aux_tool_compatible_with_secondary_tool ?auxiliary_tool - auxiliary_tool ?secondary_tool - secondary_tool)
    (aux_tool_compatible_with_module ?auxiliary_tool - auxiliary_tool ?upgrade_module - upgrade_module)
    (upgrade_blueprint_available ?upgrade_blueprint - upgrade_blueprint)
    (aux_tool_bound_to_blueprint ?auxiliary_tool - auxiliary_tool ?upgrade_blueprint - upgrade_blueprint)
    (upgrade_blueprint_activated ?upgrade_blueprint - upgrade_blueprint)
    (blueprint_bound_to_module ?upgrade_blueprint - upgrade_blueprint ?upgrade_module - upgrade_module)
    (aux_tool_prepared_with_blueprint ?auxiliary_tool - auxiliary_tool)
    (auxiliary_tool_attachment_type_b_reserved_state ?auxiliary_tool - auxiliary_tool)
    (aux_tool_attachment_verified ?auxiliary_tool - auxiliary_tool)
    (aux_tool_optional_modifier_reserved ?auxiliary_tool - auxiliary_tool)
    (aux_tool_optional_modifier_applied ?auxiliary_tool - auxiliary_tool)
    (aux_tool_slot_modifier_applied ?auxiliary_tool - auxiliary_tool)
    (aux_tool_slot_activated ?auxiliary_tool - auxiliary_tool)
    (attachment_type_b_available ?attachment_type_b - attachment_type_b)
    (auxiliary_tool_attachment_type_b_reserved_for ?auxiliary_tool - auxiliary_tool ?attachment_type_b - attachment_type_b)
    (auxiliary_tool_attachment_type_b_reserved_marker ?auxiliary_tool - auxiliary_tool)
    (auxiliary_tool_attachment_type_b_staged ?auxiliary_tool - auxiliary_tool)
    (aux_tool_attachment_type_c_confirmed ?auxiliary_tool - auxiliary_tool)
    (optional_modifier_available ?optional_modifier - optional_modifier)
    (aux_tool_optional_modifier_bound ?auxiliary_tool - auxiliary_tool ?optional_modifier - optional_modifier)
    (slot_modifier_available ?slot_modifier - slot_modifier)
    (aux_tool_slot_modifier_bound ?auxiliary_tool - auxiliary_tool ?slot_modifier - slot_modifier)
    (attachment_type_a_available ?attachment_type_a - attachment_type_a)
    (aux_tool_attachment_type_a_reserved ?auxiliary_tool - auxiliary_tool ?attachment_type_a - attachment_type_a)
    (attachment_type_c_available ?attachment_type_c - attachment_type_c)
    (aux_tool_attachment_type_c_reserved ?auxiliary_tool - auxiliary_tool ?attachment_type_c - attachment_type_c)
    (preservative_item_available ?preservative_item - preservative_item)
    (equipment_preservative_reserved ?upgradeable_equipment - upgradeable_equipment ?preservative_item - preservative_item)
    (primary_tool_local_ready ?primary_tool - primary_tool)
    (secondary_tool_local_ready ?secondary_tool - secondary_tool)
    (aux_tool_installation_finalized ?auxiliary_tool - auxiliary_tool)
  )
  (:action mark_equipment_as_candidate
    :parameters (?upgradeable_equipment - upgradeable_equipment)
    :precondition
      (and
        (not
          (equipment_is_candidate ?upgradeable_equipment)
        )
        (not
          (equipment_upgrade_applied ?upgradeable_equipment)
        )
      )
    :effect (equipment_is_candidate ?upgradeable_equipment)
  )
  (:action propose_raw_material_for_equipment
    :parameters (?upgradeable_equipment - upgradeable_equipment ?raw_material - raw_material)
    :precondition
      (and
        (equipment_is_candidate ?upgradeable_equipment)
        (not
          (equipment_material_proposed ?upgradeable_equipment)
        )
        (raw_material_available ?raw_material)
      )
    :effect
      (and
        (equipment_material_proposed ?upgradeable_equipment)
        (raw_material_reserved_for_equipment ?upgradeable_equipment ?raw_material)
        (not
          (raw_material_available ?raw_material)
        )
      )
  )
  (:action reserve_subcomponent_for_equipment
    :parameters (?upgradeable_equipment - upgradeable_equipment ?subcomponent - subcomponent)
    :precondition
      (and
        (equipment_is_candidate ?upgradeable_equipment)
        (equipment_material_proposed ?upgradeable_equipment)
        (subcomponent_available ?subcomponent)
      )
    :effect
      (and
        (subcomponent_reserved_for_equipment ?upgradeable_equipment ?subcomponent)
        (not
          (subcomponent_available ?subcomponent)
        )
      )
  )
  (:action confirm_equipment_selection
    :parameters (?upgradeable_equipment - upgradeable_equipment ?subcomponent - subcomponent)
    :precondition
      (and
        (equipment_is_candidate ?upgradeable_equipment)
        (equipment_material_proposed ?upgradeable_equipment)
        (subcomponent_reserved_for_equipment ?upgradeable_equipment ?subcomponent)
        (not
          (equipment_selection_confirmed ?upgradeable_equipment)
        )
      )
    :effect (equipment_selection_confirmed ?upgradeable_equipment)
  )
  (:action release_subcomponent_from_equipment
    :parameters (?upgradeable_equipment - upgradeable_equipment ?subcomponent - subcomponent)
    :precondition
      (and
        (subcomponent_reserved_for_equipment ?upgradeable_equipment ?subcomponent)
      )
    :effect
      (and
        (subcomponent_available ?subcomponent)
        (not
          (subcomponent_reserved_for_equipment ?upgradeable_equipment ?subcomponent)
        )
      )
  )
  (:action reserve_attachment_piece_for_equipment
    :parameters (?upgradeable_equipment - upgradeable_equipment ?attachment_piece - attachment_piece)
    :precondition
      (and
        (equipment_selection_confirmed ?upgradeable_equipment)
        (attachment_piece_available ?attachment_piece)
      )
    :effect
      (and
        (attachment_reserved_on_equipment ?upgradeable_equipment ?attachment_piece)
        (not
          (attachment_piece_available ?attachment_piece)
        )
      )
  )
  (:action release_attachment_from_equipment
    :parameters (?upgradeable_equipment - upgradeable_equipment ?attachment_piece - attachment_piece)
    :precondition
      (and
        (attachment_reserved_on_equipment ?upgradeable_equipment ?attachment_piece)
      )
    :effect
      (and
        (attachment_piece_available ?attachment_piece)
        (not
          (attachment_reserved_on_equipment ?upgradeable_equipment ?attachment_piece)
        )
      )
  )
  (:action reserve_attachment_type_a_for_aux_tool
    :parameters (?auxiliary_tool - auxiliary_tool ?attachment_type_a - attachment_type_a)
    :precondition
      (and
        (equipment_selection_confirmed ?auxiliary_tool)
        (attachment_type_a_available ?attachment_type_a)
      )
    :effect
      (and
        (aux_tool_attachment_type_a_reserved ?auxiliary_tool ?attachment_type_a)
        (not
          (attachment_type_a_available ?attachment_type_a)
        )
      )
  )
  (:action release_attachment_type_a_from_aux_tool
    :parameters (?auxiliary_tool - auxiliary_tool ?attachment_type_a - attachment_type_a)
    :precondition
      (and
        (aux_tool_attachment_type_a_reserved ?auxiliary_tool ?attachment_type_a)
      )
    :effect
      (and
        (attachment_type_a_available ?attachment_type_a)
        (not
          (aux_tool_attachment_type_a_reserved ?auxiliary_tool ?attachment_type_a)
        )
      )
  )
  (:action reserve_attachment_type_c_for_aux_tool
    :parameters (?auxiliary_tool - auxiliary_tool ?attachment_type_c - attachment_type_c)
    :precondition
      (and
        (equipment_selection_confirmed ?auxiliary_tool)
        (attachment_type_c_available ?attachment_type_c)
      )
    :effect
      (and
        (aux_tool_attachment_type_c_reserved ?auxiliary_tool ?attachment_type_c)
        (not
          (attachment_type_c_available ?attachment_type_c)
        )
      )
  )
  (:action release_attachment_type_c_from_aux_tool
    :parameters (?auxiliary_tool - auxiliary_tool ?attachment_type_c - attachment_type_c)
    :precondition
      (and
        (aux_tool_attachment_type_c_reserved ?auxiliary_tool ?attachment_type_c)
      )
    :effect
      (and
        (attachment_type_c_available ?attachment_type_c)
        (not
          (aux_tool_attachment_type_c_reserved ?auxiliary_tool ?attachment_type_c)
        )
      )
  )
  (:action mark_material_category_for_primary_tool
    :parameters (?primary_tool - primary_tool ?material_category - material_category ?subcomponent - subcomponent)
    :precondition
      (and
        (equipment_selection_confirmed ?primary_tool)
        (subcomponent_reserved_for_equipment ?primary_tool ?subcomponent)
        (primary_tool_material_category ?primary_tool ?material_category)
        (not
          (material_category_marked ?material_category)
        )
        (not
          (material_category_assigned_to_consumable ?material_category)
        )
      )
    :effect (material_category_marked ?material_category)
  )
  (:action allocate_material_category_with_attachment_for_primary_tool
    :parameters (?primary_tool - primary_tool ?material_category - material_category ?attachment_piece - attachment_piece)
    :precondition
      (and
        (equipment_selection_confirmed ?primary_tool)
        (attachment_reserved_on_equipment ?primary_tool ?attachment_piece)
        (primary_tool_material_category ?primary_tool ?material_category)
        (material_category_marked ?material_category)
        (not
          (primary_tool_local_ready ?primary_tool)
        )
      )
    :effect
      (and
        (primary_tool_local_ready ?primary_tool)
        (primary_tool_category_confirmed ?primary_tool)
      )
  )
  (:action reserve_consumable_for_primary_tool_category
    :parameters (?primary_tool - primary_tool ?material_category - material_category ?consumable_bundle - consumable_bundle)
    :precondition
      (and
        (equipment_selection_confirmed ?primary_tool)
        (primary_tool_material_category ?primary_tool ?material_category)
        (consumable_bundle_available ?consumable_bundle)
        (not
          (primary_tool_local_ready ?primary_tool)
        )
      )
    :effect
      (and
        (material_category_assigned_to_consumable ?material_category)
        (primary_tool_local_ready ?primary_tool)
        (consumable_allocated_to_primary_tool ?primary_tool ?consumable_bundle)
        (not
          (consumable_bundle_available ?consumable_bundle)
        )
      )
  )
  (:action commit_consumable_and_mark_primary_tool_category
    :parameters (?primary_tool - primary_tool ?material_category - material_category ?subcomponent - subcomponent ?consumable_bundle - consumable_bundle)
    :precondition
      (and
        (equipment_selection_confirmed ?primary_tool)
        (subcomponent_reserved_for_equipment ?primary_tool ?subcomponent)
        (primary_tool_material_category ?primary_tool ?material_category)
        (material_category_assigned_to_consumable ?material_category)
        (consumable_allocated_to_primary_tool ?primary_tool ?consumable_bundle)
        (not
          (primary_tool_category_confirmed ?primary_tool)
        )
      )
    :effect
      (and
        (material_category_marked ?material_category)
        (primary_tool_category_confirmed ?primary_tool)
        (consumable_bundle_available ?consumable_bundle)
        (not
          (consumable_allocated_to_primary_tool ?primary_tool ?consumable_bundle)
        )
      )
  )
  (:action mark_material_variant_for_secondary_tool
    :parameters (?secondary_tool - secondary_tool ?material_variant - material_variant ?subcomponent - subcomponent)
    :precondition
      (and
        (equipment_selection_confirmed ?secondary_tool)
        (subcomponent_reserved_for_equipment ?secondary_tool ?subcomponent)
        (secondary_tool_material_variant ?secondary_tool ?material_variant)
        (not
          (material_variant_marked ?material_variant)
        )
        (not
          (material_variant_assigned_to_consumable ?material_variant)
        )
      )
    :effect (material_variant_marked ?material_variant)
  )
  (:action allocate_material_variant_with_attachment_for_secondary_tool
    :parameters (?secondary_tool - secondary_tool ?material_variant - material_variant ?attachment_piece - attachment_piece)
    :precondition
      (and
        (equipment_selection_confirmed ?secondary_tool)
        (attachment_reserved_on_equipment ?secondary_tool ?attachment_piece)
        (secondary_tool_material_variant ?secondary_tool ?material_variant)
        (material_variant_marked ?material_variant)
        (not
          (secondary_tool_local_ready ?secondary_tool)
        )
      )
    :effect
      (and
        (secondary_tool_local_ready ?secondary_tool)
        (secondary_tool_category_confirmed ?secondary_tool)
      )
  )
  (:action reserve_consumable_for_secondary_tool_variant
    :parameters (?secondary_tool - secondary_tool ?material_variant - material_variant ?consumable_bundle - consumable_bundle)
    :precondition
      (and
        (equipment_selection_confirmed ?secondary_tool)
        (secondary_tool_material_variant ?secondary_tool ?material_variant)
        (consumable_bundle_available ?consumable_bundle)
        (not
          (secondary_tool_local_ready ?secondary_tool)
        )
      )
    :effect
      (and
        (material_variant_assigned_to_consumable ?material_variant)
        (secondary_tool_local_ready ?secondary_tool)
        (consumable_allocated_to_secondary_tool ?secondary_tool ?consumable_bundle)
        (not
          (consumable_bundle_available ?consumable_bundle)
        )
      )
  )
  (:action commit_consumable_and_mark_secondary_tool_variant
    :parameters (?secondary_tool - secondary_tool ?material_variant - material_variant ?subcomponent - subcomponent ?consumable_bundle - consumable_bundle)
    :precondition
      (and
        (equipment_selection_confirmed ?secondary_tool)
        (subcomponent_reserved_for_equipment ?secondary_tool ?subcomponent)
        (secondary_tool_material_variant ?secondary_tool ?material_variant)
        (material_variant_assigned_to_consumable ?material_variant)
        (consumable_allocated_to_secondary_tool ?secondary_tool ?consumable_bundle)
        (not
          (secondary_tool_category_confirmed ?secondary_tool)
        )
      )
    :effect
      (and
        (material_variant_marked ?material_variant)
        (secondary_tool_category_confirmed ?secondary_tool)
        (consumable_bundle_available ?consumable_bundle)
        (not
          (consumable_allocated_to_secondary_tool ?secondary_tool ?consumable_bundle)
        )
      )
  )
  (:action craft_upgrade_module_standard
    :parameters (?primary_tool - primary_tool ?secondary_tool - secondary_tool ?material_category - material_category ?material_variant - material_variant ?upgrade_module - upgrade_module)
    :precondition
      (and
        (primary_tool_local_ready ?primary_tool)
        (secondary_tool_local_ready ?secondary_tool)
        (primary_tool_material_category ?primary_tool ?material_category)
        (secondary_tool_material_variant ?secondary_tool ?material_variant)
        (material_category_marked ?material_category)
        (material_variant_marked ?material_variant)
        (primary_tool_category_confirmed ?primary_tool)
        (secondary_tool_category_confirmed ?secondary_tool)
        (upgrade_module_unprepared ?upgrade_module)
      )
    :effect
      (and
        (upgrade_module_prepared ?upgrade_module)
        (upgrade_module_uses_material_category ?upgrade_module ?material_category)
        (upgrade_module_uses_material_variant ?upgrade_module ?material_variant)
        (not
          (upgrade_module_unprepared ?upgrade_module)
        )
      )
  )
  (:action craft_upgrade_module_with_option_1
    :parameters (?primary_tool - primary_tool ?secondary_tool - secondary_tool ?material_category - material_category ?material_variant - material_variant ?upgrade_module - upgrade_module)
    :precondition
      (and
        (primary_tool_local_ready ?primary_tool)
        (secondary_tool_local_ready ?secondary_tool)
        (primary_tool_material_category ?primary_tool ?material_category)
        (secondary_tool_material_variant ?secondary_tool ?material_variant)
        (material_category_assigned_to_consumable ?material_category)
        (material_variant_marked ?material_variant)
        (not
          (primary_tool_category_confirmed ?primary_tool)
        )
        (secondary_tool_category_confirmed ?secondary_tool)
        (upgrade_module_unprepared ?upgrade_module)
      )
    :effect
      (and
        (upgrade_module_prepared ?upgrade_module)
        (upgrade_module_uses_material_category ?upgrade_module ?material_category)
        (upgrade_module_uses_material_variant ?upgrade_module ?material_variant)
        (upgrade_module_option_flag_1 ?upgrade_module)
        (not
          (upgrade_module_unprepared ?upgrade_module)
        )
      )
  )
  (:action craft_upgrade_module_with_option_2
    :parameters (?primary_tool - primary_tool ?secondary_tool - secondary_tool ?material_category - material_category ?material_variant - material_variant ?upgrade_module - upgrade_module)
    :precondition
      (and
        (primary_tool_local_ready ?primary_tool)
        (secondary_tool_local_ready ?secondary_tool)
        (primary_tool_material_category ?primary_tool ?material_category)
        (secondary_tool_material_variant ?secondary_tool ?material_variant)
        (material_category_marked ?material_category)
        (material_variant_assigned_to_consumable ?material_variant)
        (primary_tool_category_confirmed ?primary_tool)
        (not
          (secondary_tool_category_confirmed ?secondary_tool)
        )
        (upgrade_module_unprepared ?upgrade_module)
      )
    :effect
      (and
        (upgrade_module_prepared ?upgrade_module)
        (upgrade_module_uses_material_category ?upgrade_module ?material_category)
        (upgrade_module_uses_material_variant ?upgrade_module ?material_variant)
        (upgrade_module_option_flag_2 ?upgrade_module)
        (not
          (upgrade_module_unprepared ?upgrade_module)
        )
      )
  )
  (:action craft_upgrade_module_with_options_1_and_2
    :parameters (?primary_tool - primary_tool ?secondary_tool - secondary_tool ?material_category - material_category ?material_variant - material_variant ?upgrade_module - upgrade_module)
    :precondition
      (and
        (primary_tool_local_ready ?primary_tool)
        (secondary_tool_local_ready ?secondary_tool)
        (primary_tool_material_category ?primary_tool ?material_category)
        (secondary_tool_material_variant ?secondary_tool ?material_variant)
        (material_category_assigned_to_consumable ?material_category)
        (material_variant_assigned_to_consumable ?material_variant)
        (not
          (primary_tool_category_confirmed ?primary_tool)
        )
        (not
          (secondary_tool_category_confirmed ?secondary_tool)
        )
        (upgrade_module_unprepared ?upgrade_module)
      )
    :effect
      (and
        (upgrade_module_prepared ?upgrade_module)
        (upgrade_module_uses_material_category ?upgrade_module ?material_category)
        (upgrade_module_uses_material_variant ?upgrade_module ?material_variant)
        (upgrade_module_option_flag_1 ?upgrade_module)
        (upgrade_module_option_flag_2 ?upgrade_module)
        (not
          (upgrade_module_unprepared ?upgrade_module)
        )
      )
  )
  (:action finalize_upgrade_module
    :parameters (?upgrade_module - upgrade_module ?primary_tool - primary_tool ?subcomponent - subcomponent)
    :precondition
      (and
        (upgrade_module_prepared ?upgrade_module)
        (primary_tool_local_ready ?primary_tool)
        (subcomponent_reserved_for_equipment ?primary_tool ?subcomponent)
        (not
          (upgrade_module_finalized ?upgrade_module)
        )
      )
    :effect (upgrade_module_finalized ?upgrade_module)
  )
  (:action bind_blueprint_to_module_for_aux_tool
    :parameters (?auxiliary_tool - auxiliary_tool ?upgrade_blueprint - upgrade_blueprint ?upgrade_module - upgrade_module)
    :precondition
      (and
        (equipment_selection_confirmed ?auxiliary_tool)
        (aux_tool_compatible_with_module ?auxiliary_tool ?upgrade_module)
        (aux_tool_bound_to_blueprint ?auxiliary_tool ?upgrade_blueprint)
        (upgrade_blueprint_available ?upgrade_blueprint)
        (upgrade_module_prepared ?upgrade_module)
        (upgrade_module_finalized ?upgrade_module)
        (not
          (upgrade_blueprint_activated ?upgrade_blueprint)
        )
      )
    :effect
      (and
        (upgrade_blueprint_activated ?upgrade_blueprint)
        (blueprint_bound_to_module ?upgrade_blueprint ?upgrade_module)
        (not
          (upgrade_blueprint_available ?upgrade_blueprint)
        )
      )
  )
  (:action prepare_aux_tool_with_blueprint_and_module
    :parameters (?auxiliary_tool - auxiliary_tool ?upgrade_blueprint - upgrade_blueprint ?upgrade_module - upgrade_module ?subcomponent - subcomponent)
    :precondition
      (and
        (equipment_selection_confirmed ?auxiliary_tool)
        (aux_tool_bound_to_blueprint ?auxiliary_tool ?upgrade_blueprint)
        (upgrade_blueprint_activated ?upgrade_blueprint)
        (blueprint_bound_to_module ?upgrade_blueprint ?upgrade_module)
        (subcomponent_reserved_for_equipment ?auxiliary_tool ?subcomponent)
        (not
          (upgrade_module_option_flag_1 ?upgrade_module)
        )
        (not
          (aux_tool_prepared_with_blueprint ?auxiliary_tool)
        )
      )
    :effect (aux_tool_prepared_with_blueprint ?auxiliary_tool)
  )
  (:action reserve_optional_modifier_for_aux_tool
    :parameters (?auxiliary_tool - auxiliary_tool ?optional_modifier - optional_modifier)
    :precondition
      (and
        (equipment_selection_confirmed ?auxiliary_tool)
        (optional_modifier_available ?optional_modifier)
        (not
          (aux_tool_optional_modifier_reserved ?auxiliary_tool)
        )
      )
    :effect
      (and
        (aux_tool_optional_modifier_reserved ?auxiliary_tool)
        (aux_tool_optional_modifier_bound ?auxiliary_tool ?optional_modifier)
        (not
          (optional_modifier_available ?optional_modifier)
        )
      )
  )
  (:action apply_optional_modifier_to_aux_tool_preparation
    :parameters (?auxiliary_tool - auxiliary_tool ?upgrade_blueprint - upgrade_blueprint ?upgrade_module - upgrade_module ?subcomponent - subcomponent ?optional_modifier - optional_modifier)
    :precondition
      (and
        (equipment_selection_confirmed ?auxiliary_tool)
        (aux_tool_bound_to_blueprint ?auxiliary_tool ?upgrade_blueprint)
        (upgrade_blueprint_activated ?upgrade_blueprint)
        (blueprint_bound_to_module ?upgrade_blueprint ?upgrade_module)
        (subcomponent_reserved_for_equipment ?auxiliary_tool ?subcomponent)
        (upgrade_module_option_flag_1 ?upgrade_module)
        (aux_tool_optional_modifier_reserved ?auxiliary_tool)
        (aux_tool_optional_modifier_bound ?auxiliary_tool ?optional_modifier)
        (not
          (aux_tool_prepared_with_blueprint ?auxiliary_tool)
        )
      )
    :effect
      (and
        (aux_tool_prepared_with_blueprint ?auxiliary_tool)
        (aux_tool_optional_modifier_applied ?auxiliary_tool)
      )
