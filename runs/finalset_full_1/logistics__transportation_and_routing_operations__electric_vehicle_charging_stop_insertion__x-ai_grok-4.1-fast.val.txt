  (:action initialize_route_element
    :parameters (?route_element - route_element)
    :precondition
      (and
        (not
          (route_element_initialized ?route_element)
        )
        (not
          (route_element_inserted ?route_element)
        )
      )
    :effect (route_element_initialized ?route_element)
  )
  (:action provisionally_bind_route_element_to_station
    :parameters (?route_element - route_element ?charger_station - charger_station)
    :precondition
      (and
        (route_element_initialized ?route_element)
        (not
          (route_element_provisionally_bound ?route_element)
        )
        (charger_station_available ?charger_station)
      )
    :effect
      (and
        (route_element_provisionally_bound ?route_element)
        (route_element_assigned_charger ?route_element ?charger_station)
        (not
          (charger_station_available ?charger_station)
        )
      )
  )
  (:action assign_connector_requirement_to_route_element
    :parameters (?route_element - route_element ?connector_requirement - connector_requirement)
    :precondition
      (and
        (route_element_initialized ?route_element)
        (route_element_provisionally_bound ?route_element)
        (connector_requirement_available ?connector_requirement)
      )
    :effect
      (and
        (route_element_has_connector_requirement ?route_element ?connector_requirement)
        (not
          (connector_requirement_available ?connector_requirement)
        )
      )
  )
  (:action mark_route_element_planned
    :parameters (?route_element - route_element ?connector_requirement - connector_requirement)
    :precondition
      (and
        (route_element_initialized ?route_element)
        (route_element_provisionally_bound ?route_element)
        (route_element_has_connector_requirement ?route_element ?connector_requirement)
        (not
          (route_element_planned ?route_element)
        )
      )
    :effect (route_element_planned ?route_element)
  )
  (:action unassign_connector_requirement_from_route_element
    :parameters (?route_element - route_element ?connector_requirement - connector_requirement)
    :precondition
      (and
        (route_element_has_connector_requirement ?route_element ?connector_requirement)
      )
    :effect
      (and
        (connector_requirement_available ?connector_requirement)
        (not
          (route_element_has_connector_requirement ?route_element ?connector_requirement)
        )
      )
  )
  (:action reserve_connector_for_route_element
    :parameters (?route_element - route_element ?charger_connector - charger_connector)
    :precondition
      (and
        (route_element_planned ?route_element)
        (charger_connector_available ?charger_connector)
      )
    :effect
      (and
        (route_element_reserved_charger_connector ?route_element ?charger_connector)
        (not
          (charger_connector_available ?charger_connector)
        )
      )
  )
  (:action release_connector_reservation
    :parameters (?route_element - route_element ?charger_connector - charger_connector)
    :precondition
      (and
        (route_element_reserved_charger_connector ?route_element ?charger_connector)
      )
    :effect
      (and
        (charger_connector_available ?charger_connector)
        (not
          (route_element_reserved_charger_connector ?route_element ?charger_connector)
        )
      )
  )
  (:action allocate_charger_model_to_vehicle_route
    :parameters (?vehicle_route - vehicle_route ?charger_model - charger_model)
    :precondition
      (and
        (route_element_planned ?vehicle_route)
        (charger_model_available ?charger_model)
      )

        (vehicle_route_allocated_charger_model ?vehicle_route ?charger_model)

      )

    :parameters (?vehicle_route - vehicle_route ?charger_model - charger_model)

        (vehicle_route_allocated_charger_model ?vehicle_route ?charger_model)

        (charger_model_available ?charger_model)

          (vehicle_route_allocated_charger_model ?vehicle_route ?charger_model)
        )
      )

  (:action allocate_session_token_to_vehicle_route
    :parameters (?vehicle_route - vehicle_route ?session_token - session_token)

      (and
        (route_element_planned ?vehicle_route)

      )
    :effect

        )

  )

    :parameters (?vehicle_route - vehicle_route ?session_token - session_token)
    :precondition
      (and
        (session_token_bound_to_vehicle_route ?vehicle_route ?session_token)
      )
    :effect
      (and
        (session_token_available ?session_token)
        (not
          (session_token_bound_to_vehicle_route ?vehicle_route ?session_token)
        )
      )
  )
  (:action activate_candidate_departure_charger
    :parameters (?departure_leg - departure_leg ?candidate_charger_departure - candidate_charger_departure ?connector_requirement - connector_requirement)
    :precondition
      (and
        (route_element_planned ?departure_leg)
        (route_element_has_connector_requirement ?departure_leg ?connector_requirement)
        (leg_mapped_to_candidate_departure_charger ?departure_leg ?candidate_charger_departure)
        (not
          (candidate_departure_charger_active ?candidate_charger_departure)
        )
        (not
          (candidate_departure_charger_tentative ?candidate_charger_departure)
        )
      )
    :effect (candidate_departure_charger_active ?candidate_charger_departure)
  )
  (:action confirm_departure_leg_feasibility_with_reserved_connector
    :parameters (?departure_leg - departure_leg ?candidate_charger_departure - candidate_charger_departure ?charger_connector - charger_connector)
    :precondition
      (and
        (route_element_planned ?departure_leg)
        (route_element_reserved_charger_connector ?departure_leg ?charger_connector)
        (leg_mapped_to_candidate_departure_charger ?departure_leg ?candidate_charger_departure)
        (candidate_departure_charger_active ?candidate_charger_departure)
        (not
          (departure_leg_processed ?departure_leg)
        )
      )
    :effect
      (and
        (departure_leg_processed ?departure_leg)
        (departure_leg_feasible ?departure_leg)
      )
  )
  (:action allocate_adapter_to_departure_leg
    :parameters (?departure_leg - departure_leg ?candidate_charger_departure - candidate_charger_departure ?adapter_unit - adapter_unit)
    :precondition

        (route_element_planned ?departure_leg)

        (not
          (departure_leg_processed ?departure_leg)
        )
      )
    :effect
      (and
        (candidate_departure_charger_tentative ?candidate_charger_departure)
        (departure_leg_processed ?departure_leg)
        (adapter_allocated_to_departure_leg ?departure_leg ?adapter_unit)
        (not
          (adapter_unit_available ?adapter_unit)
        )
      )
  )
  (:action activate_departure_charger_from_tentative

          (departure_leg_feasible ?departure_leg)

      (and
        (candidate_departure_charger_active ?candidate_charger_departure)

  (:action confirm_arrival_leg_feasibility_with_reserved_connector
    :parameters (?arrival_leg - arrival_leg ?candidate_charger_arrival - candidate_charger_arrival ?charger_connector - charger_connector)
    :precondition
      (and

  (:action activate_arrival_charger_from_tentative
    :parameters (?arrival_leg - arrival_leg ?candidate_charger_arrival - candidate_charger_arrival ?connector_requirement - connector_requirement ?adapter_unit - adapter_unit)
    :precondition
      (and
        (route_element_planned ?arrival_leg)
        (route_element_has_connector_requirement ?arrival_leg ?connector_requirement)
        (leg_mapped_to_candidate_arrival_charger ?arrival_leg ?candidate_charger_arrival)
        (candidate_arrival_charger_tentative ?candidate_charger_arrival)
        (adapter_allocated_to_arrival_leg ?arrival_leg ?adapter_unit)
        (not
          (arrival_leg_feasible ?arrival_leg)

    :effect
      (and
        (candidate_arrival_charger_active ?candidate_charger_arrival)

          (adapter_allocated_to_arrival_leg ?arrival_leg ?adapter_unit)

    :precondition
      (and
        (departure_leg_processed ?departure_leg)
        (arrival_leg_processed ?arrival_leg)
        (leg_mapped_to_candidate_departure_charger ?departure_leg ?candidate_charger_departure)
        (leg_mapped_to_candidate_arrival_charger ?arrival_leg ?candidate_charger_arrival)

        (charging_stop_candidate_available ?charging_stop_candidate)

    :effect
      (and
        (charging_stop_candidate_composed ?charging_stop_candidate)
        (stop_candidate_selected_departure_charger ?charging_stop_candidate ?candidate_charger_departure)

        (arrival_leg_feasible ?arrival_leg)
        (charging_stop_candidate_available ?charging_stop_candidate)

    :effect
      (and
        (charging_stop_candidate_composed ?charging_stop_candidate)
        (stop_candidate_selected_departure_charger ?charging_stop_candidate ?candidate_charger_departure)

          (charging_stop_candidate_available ?charging_stop_candidate)

          (arrival_leg_feasible ?arrival_leg)

          (charging_stop_candidate_available ?charging_stop_candidate)

    :parameters (?charging_stop_candidate - charging_stop_candidate ?departure_leg - departure_leg ?connector_requirement - connector_requirement)
    :precondition
      (and
        (charging_stop_candidate_composed ?charging_stop_candidate)
        (departure_leg_processed ?departure_leg)
        (route_element_has_connector_requirement ?departure_leg ?connector_requirement)
        (not
          (stop_candidate_scheduled ?charging_stop_candidate)
        )
      )
    :effect (stop_candidate_scheduled ?charging_stop_candidate)
  )
