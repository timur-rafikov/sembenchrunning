  (:action open_delinquency_case
    :parameters (?credit_account_case - credit_account_case)
    :precondition
      (and
        (not
          (delinquency_initiated ?credit_account_case)
        )
        (not
          (entity_returned_to_pool ?credit_account_case)
        )
      )
    :effect (delinquency_initiated ?credit_account_case)
  )
  (:action assign_collection_agent
    :parameters (?credit_account_case - credit_account_case ?collection_agent - collection_agent)
    :precondition
      (and
        (delinquency_initiated ?credit_account_case)
        (not
          (agent_assigned ?credit_account_case)
        )
        (agent_available ?collection_agent)
      )
    :effect
      (and
        (agent_assigned ?credit_account_case)
        (entity_collection_assignment ?credit_account_case ?collection_agent)
        (not
          (agent_available ?collection_agent)
        )
      )
  )
  (:action attach_notice_template
    :parameters (?credit_account_case - credit_account_case ?notice_template - notice_template)
    :precondition
      (and
        (delinquency_initiated ?credit_account_case)
        (agent_assigned ?credit_account_case)
        (notice_template_available ?notice_template)
      )
    :effect
      (and
        (entity_notice_attached ?credit_account_case ?notice_template)
        (not
          (notice_template_available ?notice_template)
        )
      )
  )
  (:action mark_notice_sent
    :parameters (?credit_account_case - credit_account_case ?notice_template - notice_template)

        (delinquency_initiated ?credit_account_case)

        (entity_notice_attached ?credit_account_case ?notice_template)

          (entity_collection_eligible ?credit_account_case)

    :effect (entity_collection_eligible ?credit_account_case)
  )
  (:action retract_notice_template
    :parameters (?credit_account_case - credit_account_case ?notice_template - notice_template)
    :precondition
      (and
        (entity_notice_attached ?credit_account_case ?notice_template)
      )
    :effect
      (and
        (notice_template_available ?notice_template)
        (not
          (entity_notice_attached ?credit_account_case ?notice_template)
        )
      )
  )
  (:action attach_remediation_action
    :parameters (?credit_account_case - credit_account_case ?remediation_action - remediation_action)

        (remediation_attached ?credit_account_case ?remediation_action)

          (remediation_action_available ?remediation_action)

      (and
        (remediation_attached ?credit_account_case ?remediation_action)

        (remediation_action_available ?remediation_action)

          (remediation_attached ?credit_account_case ?remediation_action)
        )
      )
  )

        (entity_collection_eligible ?primary_obligor)

        (entity_collection_eligible ?primary_obligor)

        (primary_bucket_flagged ?primary_delinquency_bucket)
        (not
          (obligor_package_ready ?primary_obligor)

    :effect
      (and
        (obligor_package_ready ?primary_obligor)
        (primary_obligor_bucket_ready ?primary_obligor)
