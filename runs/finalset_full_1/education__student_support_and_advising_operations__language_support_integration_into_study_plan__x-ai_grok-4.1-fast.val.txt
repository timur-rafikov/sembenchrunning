--- domain.pddl (as passed to VAL) ---
(define (domain language_support_integration)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_type_category - object document_or_resource - object plan_and_profile_category - object institution_entity - object advising_subject - institution_entity language_specialist_slot - resource_type_category language_assessment - resource_type_category interpreter_tutor_slot - resource_type_category policy_form - resource_type_category approval_token - resource_type_category accommodation_record - resource_type_category support_intervention - resource_type_category specialist_consultation - resource_type_category support_service_offering - document_or_resource assessment_report - document_or_resource external_referral_destination - document_or_resource language_need_profile - plan_and_profile_category course_accommodation_profile - plan_and_profile_category study_plan_document - plan_and_profile_category person_role_group - advising_subject case_file_group - advising_subject student - person_role_group course_instructor - person_role_group case_file - case_file_group)
  (:predicates
    (intake_opened ?advising_case - advising_subject)
    (advising_case_assessed ?advising_case - advising_subject)
    (specialist_referred ?advising_case - advising_subject)
    (plan_integrated ?advising_case - advising_subject)
    (case_activated ?advising_case - advising_subject)
    (advising_case_accommodation_recorded ?advising_case - advising_subject)
    (specialist_slot_available ?language_specialist_slot - language_specialist_slot)
    (case_reserved_specialist_slot ?advising_case - advising_subject ?language_specialist_slot - language_specialist_slot)
    (assessment_slot_available ?language_assessment - language_assessment)
    (case_scheduled_assessment ?advising_case - advising_subject ?language_assessment - language_assessment)
    (interpreter_tutor_slot_available ?interpreter_tutor_slot - interpreter_tutor_slot)
    (case_reserved_interpreter_slot ?advising_case - advising_subject ?interpreter_tutor_slot - interpreter_tutor_slot)
    (support_service_available ?support_service - support_service_offering)
    (student_assigned_support_service ?student - student ?support_service - support_service_offering)
    (instructor_assigned_support_service ?instructor - course_instructor ?support_service - support_service_offering)
    (student_has_language_need_profile ?student - student ?language_need_profile - language_need_profile)
    (need_profile_flagged ?language_need_profile - language_need_profile)
    (need_profile_intervention_allocated ?language_need_profile - language_need_profile)
    (student_ready_for_plan ?student - student)
    (instructor_has_accommodation_profile ?instructor - course_instructor ?accommodation_profile - course_accommodation_profile)
    (accommodation_profile_marked ?accommodation_profile - course_accommodation_profile)
    (accommodation_profile_service_reserved ?accommodation_profile - course_accommodation_profile)
    (instructor_ready_for_plan ?instructor - course_instructor)
    (study_plan_template_available ?study_plan - study_plan_document)
    (study_plan_document_prepared ?study_plan - study_plan_document)
    (study_plan_linked_need_profile ?study_plan - study_plan_document ?language_need_profile - language_need_profile)
    (study_plan_linked_accommodation_profile ?study_plan - study_plan_document ?accommodation_profile - course_accommodation_profile)
    (study_plan_flag_student_review ?study_plan - study_plan_document)
    (study_plan_flag_instructor_review ?study_plan - study_plan_document)
    (study_plan_finalized ?study_plan - study_plan_document)
    (case_file_links_student ?case_file - case_file ?student - student)
    (case_file_links_instructor ?case_file - case_file ?instructor - course_instructor)
    (case_file_linked_study_plan ?case_file - case_file ?study_plan - study_plan_document)
    (assessment_report_available ?assessment_report - assessment_report)
    (case_file_has_assessment_report ?case_file - case_file ?assessment_report - assessment_report)
    (assessment_report_attached ?assessment_report - assessment_report)
    (assessment_report_linked_to_study_plan ?assessment_report - assessment_report ?study_plan - study_plan_document)
    (case_file_assessment_report_validated ?case_file - case_file)
    (case_file_assessment_report_reviewed ?case_file - case_file)
    (case_file_assessment_report_approved_for_activation ?case_file - case_file)
    (policy_form_attached_to_case_file ?case_file - case_file)
    (policy_form_validated ?case_file - case_file)
    (case_file_ready_for_approval ?case_file - case_file)
    (case_file_review_completed ?case_file - case_file)
    (referral_destination_available ?referral_destination - external_referral_destination)
    (case_file_linked_referral_destination ?case_file - case_file ?referral_destination - external_referral_destination)
    (external_referral_attached ?case_file - case_file)
    (referral_follow_up_recorded ?case_file - case_file)
    (referral_consultation_scheduled ?case_file - case_file)
    (policy_form_available ?policy_form - policy_form)
    (case_file_linked_policy_form ?case_file - case_file ?policy_form - policy_form)
    (approval_token_available ?approval_token - approval_token)
    (case_file_has_approval_token ?case_file - case_file ?approval_token - approval_token)
    (support_intervention_available ?support_intervention - support_intervention)
    (case_file_has_support_intervention ?case_file - case_file ?support_intervention - support_intervention)
    (specialist_consultation_available ?specialist_consultation - specialist_consultation)
    (case_file_has_specialist_consultation ?case_file - case_file ?specialist_consultation - specialist_consultation)
    (accommodation_record_available ?accommodation_record - accommodation_record)
    (case_linked_accommodation_record ?advising_case - advising_subject ?accommodation_record - accommodation_record)
    (student_support_ready ?student - student)
    (instructor_ready ?instructor - course_instructor)
    (case_file_finalized ?case_file - case_file)
  )
  (:action create_advising_intake
    :parameters (?advising_case - advising_subject)
    :precondition
      (and
        (not
          (intake_opened ?advising_case)
        )
        (not
          (plan_integrated ?advising_case)
        )
      )
    :effect (intake_opened ?advising_case)
  )
  (:action refer_to_language_specialist
    :parameters (?advising_case - advising_subject ?language_specialist_slot - language_specialist_slot)
    :precondition
      (and
        (intake_opened ?advising_case)
        (not
          (specialist_referred ?advising_case)
        )
        (specialist_slot_available ?language_specialist_slot)
      )
    :effect
      (and
        (specialist_referred ?advising_case)
        (case_reserved_specialist_slot ?advising_case ?language_specialist_slot)
        (not
          (specialist_slot_available ?language_specialist_slot)
        )
      )
  )
  (:action schedule_language_assessment
    :parameters (?advising_case - advising_subject ?language_assessment - language_assessment)
    :precondition
      (and
        (intake_opened ?advising_case)
        (specialist_referred ?advising_case)
        (assessment_slot_available ?language_assessment)
      )
    :effect
      (and
        (case_scheduled_assessment ?advising_case ?language_assessment)
        (not
          (assessment_slot_available ?language_assessment)
        )
      )
  )
  (:action complete_language_assessment
    :parameters (?advising_case - advising_subject ?language_assessment - language_assessment)
    :precondition
      (and
        (intake_opened ?advising_case)
        (specialist_referred ?advising_case)
        (case_scheduled_assessment ?advising_case ?language_assessment)
        (not
          (advising_case_assessed ?advising_case)
        )
      )
    :effect (advising_case_assessed ?advising_case)
  )
  (:action cancel_language_assessment
    :parameters (?advising_case - advising_subject ?language_assessment - language_assessment)
    :precondition
      (and
        (case_scheduled_assessment ?advising_case ?language_assessment)
      )
    :effect
      (and
        (assessment_slot_available ?language_assessment)
        (not
          (case_scheduled_assessment ?advising_case ?language_assessment)
        )
      )
  )
  (:action reserve_interpreter_tutor_slot
    :parameters (?advising_case - advising_subject ?interpreter_tutor_slot - interpreter_tutor_slot)
    :precondition
      (and
        (advising_case_assessed ?advising_case)
        (interpreter_tutor_slot_available ?interpreter_tutor_slot)
      )
    :effect
      (and
        (case_reserved_interpreter_slot ?advising_case ?interpreter_tutor_slot)
        (not
          (interpreter_tutor_slot_available ?interpreter_tutor_slot)
        )
      )
  )
  (:action release_interpreter_tutor_slot
    :parameters (?advising_case - advising_subject ?interpreter_tutor_slot - interpreter_tutor_slot)
    :precondition
      (and
        (case_reserved_interpreter_slot ?advising_case ?interpreter_tutor_slot)
      )
    :effect
      (and
        (interpreter_tutor_slot_available ?interpreter_tutor_slot)
        (not
          (case_reserved_interpreter_slot ?advising_case ?interpreter_tutor_slot)
        )
      )
  )
  (:action assign_support_intervention_to_case_file
    :parameters (?case_file - case_file ?support_intervention - support_intervention)
    :precondition
      (and
        (advising_case_assessed ?case_file)
        (support_intervention_available ?support_intervention)
      )
    :effect
      (and
        (case_file_has_support_intervention ?case_file ?support_intervention)
        (not
          (support_intervention_available ?support_intervention)
        )
      )
  )
  (:action release_support_intervention_from_case_file
    :parameters (?case_file - case_file ?support_intervention - support_intervention)
    :precondition
      (and
        (case_file_has_support_intervention ?case_file ?support_intervention)
      )
    :effect
      (and
        (support_intervention_available ?support_intervention)
        (not
          (case_file_has_support_intervention ?case_file ?support_intervention)
        )
      )
  )
  (:action assign_specialist_consultation_to_case_file
    :parameters (?case_file - case_file ?specialist_consultation - specialist_consultation)
    :precondition
      (and
        (advising_case_assessed ?case_file)
        (specialist_consultation_available ?specialist_consultation)
      )
    :effect
      (and
        (case_file_has_specialist_consultation ?case_file ?specialist_consultation)
        (not
          (specialist_consultation_available ?specialist_consultation)
        )
      )
  )
  (:action release_specialist_consultation_from_case_file
    :parameters (?case_file - case_file ?specialist_consultation - specialist_consultation)
    :precondition
      (and
        (case_file_has_specialist_consultation ?case_file ?specialist_consultation)
      )
    :effect
      (and
        (specialist_consultation_available ?specialist_consultation)
        (not
          (case_file_has_specialist_consultation ?case_file ?specialist_consultation)
        )
      )
  )
  (:action flag_language_need_profile_for_review
    :parameters (?student - student ?language_need_profile - language_need_profile ?language_assessment - language_assessment)
    :precondition
      (and
        (advising_case_assessed ?student)
        (case_scheduled_assessment ?student ?language_assessment)
        (student_has_language_need_profile ?student ?language_need_profile)
        (not
          (need_profile_flagged ?language_need_profile)
        )
        (not
          (need_profile_intervention_allocated ?language_need_profile)
        )
      )
    :effect (need_profile_flagged ?language_need_profile)
  )
  (:action assign_interpreter_and_triage_student
    :parameters (?student - student ?language_need_profile - language_need_profile ?interpreter_tutor_slot - interpreter_tutor_slot)
    :precondition
      (and
        (advising_case_assessed ?student)
        (case_reserved_interpreter_slot ?student ?interpreter_tutor_slot)
        (student_has_language_need_profile ?student ?language_need_profile)
        (need_profile_flagged ?language_need_profile)
        (not
          (student_support_ready ?student)
        )
      )
    :effect
      (and
        (student_support_ready ?student)
        (student_ready_for_plan ?student)
      )
  )
  (:action assign_support_service_to_student
    :parameters (?student - student ?language_need_profile - language_need_profile ?support_service - support_service_offering)
    :precondition
      (and
        (advising_case_assessed ?student)
        (student_has_language_need_profile ?student ?language_need_profile)
        (support_service_available ?support_service)
        (not
          (student_support_ready ?student)
        )
      )
    :effect
      (and
        (need_profile_intervention_allocated ?language_need_profile)
        (student_support_ready ?student)
        (student_assigned_support_service ?student ?support_service)
        (not
          (support_service_available ?support_service)
        )
      )
  )
  (:action finalize_support_service_assignment_for_student
    :parameters (?student - student ?language_need_profile - language_need_profile ?language_assessment - language_assessment ?support_service - support_service_offering)
    :precondition
      (and
        (advising_case_assessed ?student)
        (case_scheduled_assessment ?student ?language_assessment)
        (student_has_language_need_profile ?student ?language_need_profile)
        (need_profile_intervention_allocated ?language_need_profile)
        (student_assigned_support_service ?student ?support_service)
        (not
          (student_ready_for_plan ?student)
        )
      )
    :effect
      (and
        (need_profile_flagged ?language_need_profile)
        (student_ready_for_plan ?student)
        (support_service_available ?support_service)
        (not
          (student_assigned_support_service ?student ?support_service)
        )
      )
  )
  (:action flag_accommodation_profile_for_review
    :parameters (?instructor - course_instructor ?accommodation_profile - course_accommodation_profile ?language_assessment - language_assessment)
    :precondition
      (and
        (advising_case_assessed ?instructor)
        (case_scheduled_assessment ?instructor ?language_assessment)
        (instructor_has_accommodation_profile ?instructor ?accommodation_profile)
        (not
          (accommodation_profile_marked ?accommodation_profile)
        )
        (not
          (accommodation_profile_service_reserved ?accommodation_profile)
        )
      )
    :effect (accommodation_profile_marked ?accommodation_profile)
  )
  (:action instructor_confirm_accommodation_profile
    :parameters (?instructor - course_instructor ?accommodation_profile - course_accommodation_profile ?interpreter_tutor_slot - interpreter_tutor_slot)
    :precondition
      (and
        (advising_case_assessed ?instructor)
        (case_reserved_interpreter_slot ?instructor ?interpreter_tutor_slot)
        (instructor_has_accommodation_profile ?instructor ?accommodation_profile)
        (accommodation_profile_marked ?accommodation_profile)
        (not
          (instructor_ready ?instructor)
        )
      )
    :effect
      (and
        (instructor_ready ?instructor)
        (instructor_ready_for_plan ?instructor)
      )
  )
  (:action assign_support_service_to_instructor
    :parameters (?instructor - course_instructor ?accommodation_profile - course_accommodation_profile ?support_service - support_service_offering)
    :precondition
      (and
        (advising_case_assessed ?instructor)
        (instructor_has_accommodation_profile ?instructor ?accommodation_profile)
        (support_service_available ?support_service)
        (not
          (instructor_ready ?instructor)
        )
      )
    :effect
      (and
        (accommodation_profile_service_reserved ?accommodation_profile)
        (instructor_ready ?instructor)
        (instructor_assigned_support_service ?instructor ?support_service)
        (not
          (support_service_available ?support_service)
        )
      )
  )
  (:action finalize_support_service_assignment_for_instructor
    :parameters (?instructor - course_instructor ?accommodation_profile - course_accommodation_profile ?language_assessment - language_assessment ?support_service - support_service_offering)
    :precondition
      (and
        (advising_case_assessed ?instructor)
        (case_scheduled_assessment ?instructor ?language_assessment)
        (instructor_has_accommodation_profile ?instructor ?accommodation_profile)
        (accommodation_profile_service_reserved ?accommodation_profile)
        (instructor_assigned_support_service ?instructor ?support_service)
        (not
          (instructor_ready_for_plan ?instructor)
        )
      )
    :effect
      (and
        (accommodation_profile_marked ?accommodation_profile)
        (instructor_ready_for_plan ?instructor)
        (support_service_available ?support_service)
        (not
          (instructor_assigned_support_service ?instructor ?support_service)
        )
      )
  )
  (:action assemble_study_plan
    :parameters (?student - student ?instructor - course_instructor ?language_need_profile - language_need_profile ?accommodation_profile - course_accommodation_profile ?study_plan - study_plan_document)
    :precondition
      (and
        (student_support_ready ?student)
        (instructor_ready ?instructor)
        (student_has_language_need_profile ?student ?language_need_profile)
        (instructor_has_accommodation_profile ?instructor ?accommodation_profile)
        (need_profile_flagged ?language_need_profile)
        (accommodation_profile_marked ?accommodation_profile)
        (student_ready_for_plan ?student)
        (instructor_ready_for_plan ?instructor)
        (study_plan_template_available ?study_plan)
      )
    :effect
      (and
        (study_plan_document_prepared ?study_plan)
        (study_plan_linked_need_profile ?study_plan ?language_need_profile)
        (study_plan_linked_accommodation_profile ?study_plan ?accommodation_profile)
        (not
          (study_plan_template_available ?study_plan)
        )
      )
  )
  (:action assemble_study_plan_mark_for_student_review
    :parameters (?student - student ?instructor - course_instructor ?language_need_profile - language_need_profile ?accommodation_profile - course_accommodation_profile ?study_plan - study_plan_document)
    :precondition
      (and
        (student_support_ready ?student)
        (instructor_ready ?instructor)
        (student_has_language_need_profile ?student ?language_need_profile)
        (instructor_has_accommodation_profile ?instructor ?accommodation_profile)
        (need_profile_intervention_allocated ?language_need_profile)
        (accommodation_profile_marked ?accommodation_profile)
        (not
          (student_ready_for_plan ?student)
        )
        (instructor_ready_for_plan ?instructor)
        (study_plan_template_available ?study_plan)
      )
    :effect
      (and
        (study_plan_document_prepared ?study_plan)
        (study_plan_linked_need_profile ?study_plan ?language_need_profile)
        (study_plan_linked_accommodation_profile ?study_plan ?accommodation_profile)
        (study_plan_flag_student_review ?study_plan)
        (not
          (study_plan_template_available ?study_plan)
        )
      )
  )
  (:action assemble_study_plan_mark_for_instructor_review
    :parameters (?student - student ?instructor - course_instructor ?language_need_profile - language_need_profile ?accommodation_profile - course_accommodation_profile ?study_plan - study_plan_document)
    :precondition
      (and
        (student_support_ready ?student)
        (instructor_ready ?instructor)
        (student_has_language_need_profile ?student ?language_need_profile)
        (instructor_has_accommodation_profile ?instructor ?accommodation_profile)
        (need_profile_flagged ?language_need_profile)
        (accommodation_profile_service_reserved ?accommodation_profile)
        (student_ready_for_plan ?student)
        (not
          (instructor_ready_for_plan ?instructor)
        )
        (study_plan_template_available ?study_plan)
      )
    :effect
      (and
        (study_plan_document_prepared ?study_plan)
        (study_plan_linked_need_profile ?study_plan ?language_need_profile)
        (study_plan_linked_accommodation_profile ?study_plan ?accommodation_profile)
        (study_plan_flag_instructor_review ?study_plan)
        (not
          (study_plan_template_available ?study_plan)
        )
      )
  )
  (:action assemble_study_plan_mark_for_both_reviews
    :parameters (?student - student ?instructor - course_instructor ?language_need_profile - language_need_profile ?accommodation_profile - course_accommodation_profile ?study_plan - study_plan_document)
    :precondition
      (and
        (student_support_ready ?student)
        (instructor_ready ?instructor)
        (student_has_language_need_profile ?student ?language_need_profile)
        (instructor_has_accommodation_profile ?instructor ?accommodation_profile)
        (need_profile_intervention_allocated ?language_need_profile)
        (accommodation_profile_service_reserved ?accommodation_profile)
        (not
          (student_ready_for_plan ?student)
        )
        (not
          (instructor_ready_for_plan ?instructor)
        )
        (study_plan_template_available ?study_plan)
      )
    :effect
      (and
        (study_plan_document_prepared ?study_plan)
        (study_plan_linked_need_profile ?study_plan ?language_need_profile)
        (study_plan_linked_accommodation_profile ?study_plan ?accommodation_profile)
        (study_plan_flag_student_review ?study_plan)
        (study_plan_flag_instructor_review ?study_plan)
        (not
          (study_plan_template_available ?study_plan)
        )
      )
  )
  (:action finalize_study_plan_document
    :parameters (?study_plan - study_plan_document ?student - student ?language_assessment - language_assessment)
    :precondition
      (and
        (study_plan_document_prepared ?study_plan)
        (student_support_ready ?student)
        (case_scheduled_assessment ?student ?language_assessment)
        (not
          (study_plan_finalized ?study_plan)
        )
      )
    :effect (study_plan_finalized ?study_plan)
  )
  (:action attach_assessment_report_to_case_and_plan
    :parameters (?case_file - case_file ?assessment_report - assessment_report ?study_plan - study_plan_document)
    :precondition
      (and
        (advising_case_assessed ?case_file)
        (case_file_linked_study_plan ?case_file ?study_plan)
        (case_file_has_assessment_report ?case_file ?assessment_report)
        (assessment_report_available ?assessment_report)
        (study_plan_document_prepared ?study_plan)
        (study_plan_finalized ?study_plan)
        (not
          (assessment_report_attached ?assessment_report)
        )
      )
    :effect
      (and
        (assessment_report_attached ?assessment_report)
        (assessment_report_linked_to_study_plan ?assessment_report ?study_plan)
        (not
          (assessment_report_available ?assessment_report)
        )
      )
  )
  (:action validate_assessment_report_for_case_file
    :parameters (?case_file - case_file ?assessment_report - assessment_report ?study_plan - study_plan_document ?language_assessment - language_assessment)
    :precondition
      (and
        (advising_case_assessed ?case_file)
        (case_file_has_assessment_report ?case_file ?assessment_report)
        (assessment_report_attached ?assessment_report)
        (assessment_report_linked_to_study_plan ?assessment_report ?study_plan)
        (case_scheduled_assessment ?case_file ?language_assessment)
        (not
          (study_plan_flag_student_review ?study_plan)
        )
        (not
          (case_file_assessment_report_validated ?case_file)
        )
      )
    :effect (case_file_assessment_report_validated ?case_file)
  )
  (:action attach_policy_form_to_case_file
    :parameters (?case_file - case_file ?policy_form - policy_form)
    :precondition
      (and
        (advising_case_assessed ?case_file)
        (policy_form_available ?policy_form)
        (not
          (policy_form_attached_to_case_file ?case_file)
        )
      )
    :effect
      (and
        (policy_form_attached_to_case_file ?case_file)
        (case_file_linked_policy_form ?case_file ?policy_form)
        (not
          (policy_form_available ?policy_form)
        )
      )
  )
  (:action validate_policy_and_report_and_progress
    :parameters (?case_file - case_file ?assessment_report - assessment_report ?study_plan - study_plan_document ?language_assessment - language_assessment ?policy_form - policy_form)
    :precondition
      (and
        (advising_case_assessed ?case_file)
        (case_file_has_assessment_report ?case_file ?assessment_report)
        (assessment_report_attached ?assessment_report)
        (assessment_report_linked_to_study_plan ?assessment_report ?study_plan)
