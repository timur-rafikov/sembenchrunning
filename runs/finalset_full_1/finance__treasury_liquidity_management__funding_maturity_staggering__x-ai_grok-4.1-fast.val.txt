    (option_reserved ?funding_request - funding_request)

    (internal_buffer_assigned ?internal_request - internal_request ?liquidity_buffer_profile - liquidity_buffer_profile)
    (external_buffer_assigned ?external_request - external_request ?liquidity_buffer_profile - liquidity_buffer_profile)
    (internal_slot_assigned ?internal_request - internal_request ?internal_maturity_slot - internal_maturity_slot)

    (internal_slot_buffered ?internal_maturity_slot - internal_maturity_slot)
    (internal_request_ready ?internal_request - internal_request)
    (external_slot_assigned ?external_request - external_request ?external_maturity_slot - external_maturity_slot)
    (external_slot_ready ?external_maturity_slot - external_maturity_slot)
    (external_slot_buffered ?external_maturity_slot - external_maturity_slot)

    (instrument_external_slot_assigned ?funding_instrument - funding_instrument ?external_maturity_slot - external_maturity_slot)
    (allocation_depth_1 ?funding_instrument - funding_instrument)
    (allocation_depth_2 ?funding_instrument - funding_instrument)

    (plan_includes_instrument ?maturity_stagger_plan - maturity_stagger_plan ?funding_instrument - funding_instrument)

    (plan_settlement_profile_assigned ?maturity_stagger_plan - maturity_stagger_plan ?settlement_profile - settlement_profile)

  (:action initialize_funding_request
    :parameters (?funding_request - funding_request)
    :precondition
      (and
        (not
          (request_initialized ?funding_request)
        )
        (not
          (request_executed_for_subject ?funding_request)
        )
      )
    :effect (request_initialized ?funding_request)
  )
  (:action assign_funding_option
    :parameters (?funding_request - funding_request ?funding_option - funding_option)
    :precondition
      (and
        (request_initialized ?funding_request)
        (not
          (option_reserved ?funding_request)
        )
        (funding_option_available ?funding_option)
      )
    :effect
      (and
        (option_reserved ?funding_request)
        (request_assigned_option_for_subject ?funding_request ?funding_option)
        (not
          (funding_option_available ?funding_option)
        )
      )
  )
  (:action assign_tenor_profile_to_request
    :parameters (?funding_request - funding_request ?tenor_profile - tenor_profile)
    :precondition
      (and
        (request_initialized ?funding_request)
        (option_reserved ?funding_request)
        (tenor_profile_available ?tenor_profile)
      )
    :effect
      (and
        (request_tenor_assignment_for_subject ?funding_request ?tenor_profile)
        (not
          (tenor_profile_available ?tenor_profile)
        )
      )
  )
  (:action confirm_tenor_assignment
    :parameters (?funding_request - funding_request ?tenor_profile - tenor_profile)
    :precondition
      (and
        (request_initialized ?funding_request)
        (option_reserved ?funding_request)
        (request_tenor_assignment_for_subject ?funding_request ?tenor_profile)
        (not
          (tenor_confirmed_for_subject ?funding_request)
        )
      )
    :effect (tenor_confirmed_for_subject ?funding_request)
  )
  (:action revert_tenor_assignment
    :parameters (?funding_request - funding_request ?tenor_profile - tenor_profile)
    :precondition
      (and
        (request_tenor_assignment_for_subject ?funding_request ?tenor_profile)
      )
    :effect
      (and
        (tenor_profile_available ?tenor_profile)
        (not
          (request_tenor_assignment_for_subject ?funding_request ?tenor_profile)
        )
      )
  )
  (:action assign_execution_desk
    :parameters (?funding_request - funding_request ?execution_desk - execution_desk)
    :precondition
      (and
        (tenor_confirmed_for_subject ?funding_request)
        (execution_desk_available ?execution_desk)
      )
    :effect
      (and
        (request_assigned_desk_for_subject ?funding_request ?execution_desk)
        (not
          (execution_desk_available ?execution_desk)
        )
      )
  )
  (:action release_execution_desk
    :parameters (?funding_request - funding_request ?execution_desk - execution_desk)
    :precondition
      (and
        (request_assigned_desk_for_subject ?funding_request ?execution_desk)
      )
    :effect
      (and
        (execution_desk_available ?execution_desk)
        (not
          (request_assigned_desk_for_subject ?funding_request ?execution_desk)
        )
      )
  )
  (:action assign_issuance_method_to_plan
    :parameters (?maturity_stagger_plan - maturity_stagger_plan ?issuance_method - issuance_method)
    :precondition
      (and
        (tenor_confirmed_for_subject ?maturity_stagger_plan)
        (issuance_method_available ?issuance_method)
      )
    :effect
      (and
        (plan_issuance_method_linked ?maturity_stagger_plan ?issuance_method)
        (not
          (issuance_method_available ?issuance_method)
        )
      )
  )
  (:action retract_issuance_method_from_plan
    :parameters (?maturity_stagger_plan - maturity_stagger_plan ?issuance_method - issuance_method)
    :precondition
      (and
        (plan_issuance_method_linked ?maturity_stagger_plan ?issuance_method)
      )
    :effect
      (and
        (issuance_method_available ?issuance_method)
        (not
          (plan_issuance_method_linked ?maturity_stagger_plan ?issuance_method)
        )
      )
  )
  (:action attach_compliance_document_to_plan
    :parameters (?maturity_stagger_plan - maturity_stagger_plan ?compliance_document - compliance_document)
    :precondition
      (and
        (tenor_confirmed_for_subject ?maturity_stagger_plan)
        (compliance_document_available ?compliance_document)
      )
    :effect
      (and
        (plan_compliance_document_assigned ?maturity_stagger_plan ?compliance_document)
        (not
          (compliance_document_available ?compliance_document)
        )
      )
  )
  (:action detach_compliance_document_from_plan
    :parameters (?maturity_stagger_plan - maturity_stagger_plan ?compliance_document - compliance_document)
    :precondition
      (and
        (plan_compliance_document_assigned ?maturity_stagger_plan ?compliance_document)
      )
    :effect
      (and
        (compliance_document_available ?compliance_document)
        (not
          (plan_compliance_document_assigned ?maturity_stagger_plan ?compliance_document)
        )
      )
  )
  (:action prepare_internal_slot
    :parameters (?internal_request - internal_request ?internal_maturity_slot - internal_maturity_slot ?tenor_profile - tenor_profile)
    :precondition
      (and
        (tenor_confirmed_for_subject ?internal_request)
        (request_tenor_assignment_for_subject ?internal_request ?tenor_profile)
        (internal_slot_assigned ?internal_request ?internal_maturity_slot)
        (not
          (internal_slot_ready ?internal_maturity_slot)
        )
        (not
          (internal_slot_buffered ?internal_maturity_slot)
        )
      )
    :effect (internal_slot_ready ?internal_maturity_slot)
  )
  (:action activate_internal_slot
    :parameters (?internal_request - internal_request ?internal_maturity_slot - internal_maturity_slot ?execution_desk - execution_desk)
    :precondition
      (and
        (tenor_confirmed_for_subject ?internal_request)
        (request_assigned_desk_for_subject ?internal_request ?execution_desk)
        (internal_slot_assigned ?internal_request ?internal_maturity_slot)
        (internal_slot_ready ?internal_maturity_slot)
        (not
          (internal_buffer_engaged ?internal_request)
        )
      )
    :effect
      (and
        (internal_buffer_engaged ?internal_request)
        (internal_request_ready ?internal_request)
      )
  )
  (:action apply_internal_buffer
    :parameters (?internal_request - internal_request ?internal_maturity_slot - internal_maturity_slot ?liquidity_buffer_profile - liquidity_buffer_profile)
    :precondition
      (and
        (tenor_confirmed_for_subject ?internal_request)
        (internal_slot_assigned ?internal_request ?internal_maturity_slot)
        (buffer_profile_available ?liquidity_buffer_profile)
        (not
          (internal_buffer_engaged ?internal_request)
        )
      )
    :effect
      (and
        (internal_slot_buffered ?internal_maturity_slot)
        (internal_buffer_engaged ?internal_request)
        (internal_buffer_assigned ?internal_request ?liquidity_buffer_profile)
        (not
          (buffer_profile_available ?liquidity_buffer_profile)
        )
      )
  )
  (:action finalize_internal_slot_preparation
    :parameters (?internal_request - internal_request ?internal_maturity_slot - internal_maturity_slot ?tenor_profile - tenor_profile ?liquidity_buffer_profile - liquidity_buffer_profile)
    :precondition
      (and
        (tenor_confirmed_for_subject ?internal_request)
        (request_tenor_assignment_for_subject ?internal_request ?tenor_profile)
        (internal_slot_assigned ?internal_request ?internal_maturity_slot)
        (internal_slot_buffered ?internal_maturity_slot)
        (internal_buffer_assigned ?internal_request ?liquidity_buffer_profile)
        (not
          (internal_request_ready ?internal_request)
        )
      )
    :effect
      (and
        (internal_slot_ready ?internal_maturity_slot)
        (internal_request_ready ?internal_request)
        (buffer_profile_available ?liquidity_buffer_profile)
        (not
          (internal_buffer_assigned ?internal_request ?liquidity_buffer_profile)
        )
      )
  )
  (:action prepare_external_slot
    :parameters (?external_request - external_request ?external_maturity_slot - external_maturity_slot ?tenor_profile - tenor_profile)
    :precondition
      (and
        (tenor_confirmed_for_subject ?external_request)
        (request_tenor_assignment_for_subject ?external_request ?tenor_profile)

          (external_slot_ready ?external_maturity_slot)

    :effect (external_slot_ready ?external_maturity_slot)
  )
  (:action activate_external_slot
    :parameters (?external_request - external_request ?external_maturity_slot - external_maturity_slot ?execution_desk - execution_desk)
    :precondition
      (and
        (tenor_confirmed_for_subject ?external_request)
        (request_assigned_desk_for_subject ?external_request ?execution_desk)
        (external_slot_assigned ?external_request ?external_maturity_slot)
        (external_slot_ready ?external_maturity_slot)
        (not
          (external_buffer_engaged ?external_request)
        )
      )
    :effect
      (and
        (external_buffer_engaged ?external_request)
        (external_request_ready ?external_request)
      )
  )
  (:action apply_external_buffer
    :parameters (?external_request - external_request ?external_maturity_slot - external_maturity_slot ?liquidity_buffer_profile - liquidity_buffer_profile)
    :precondition
      (and
        (tenor_confirmed_for_subject ?external_request)
        (external_slot_assigned ?external_request ?external_maturity_slot)
        (buffer_profile_available ?liquidity_buffer_profile)
        (not
          (external_buffer_engaged ?external_request)
        )
      )
    :effect
      (and
        (external_slot_buffered ?external_maturity_slot)
        (external_buffer_engaged ?external_request)
        (external_buffer_assigned ?external_request ?liquidity_buffer_profile)
        (not
          (buffer_profile_available ?liquidity_buffer_profile)
        )
      )
  )
  (:action finalize_external_slot_preparation
    :parameters (?external_request - external_request ?external_maturity_slot - external_maturity_slot ?tenor_profile - tenor_profile ?liquidity_buffer_profile - liquidity_buffer_profile)
    :precondition
      (and
        (tenor_confirmed_for_subject ?external_request)
        (request_tenor_assignment_for_subject ?external_request ?tenor_profile)
        (external_slot_assigned ?external_request ?external_maturity_slot)
        (external_slot_buffered ?external_maturity_slot)
        (external_buffer_assigned ?external_request ?liquidity_buffer_profile)
        (not
          (external_request_ready ?external_request)
        )
      )
    :effect
      (and
        (external_slot_ready ?external_maturity_slot)
        (external_request_ready ?external_request)
        (buffer_profile_available ?liquidity_buffer_profile)
        (not
          (external_buffer_assigned ?external_request ?liquidity_buffer_profile)
        )
      )
  )
  (:action create_and_tag_funding_instrument
    :parameters (?internal_request - internal_request ?external_request - external_request ?internal_maturity_slot - internal_maturity_slot ?external_maturity_slot - external_maturity_slot ?funding_instrument - funding_instrument)
    :precondition
      (and
        (internal_buffer_engaged ?internal_request)
        (external_buffer_engaged ?external_request)
        (internal_slot_assigned ?internal_request ?internal_maturity_slot)
        (external_slot_assigned ?external_request ?external_maturity_slot)
        (internal_slot_ready ?internal_maturity_slot)
        (external_slot_ready ?external_maturity_slot)
        (internal_request_ready ?internal_request)
        (external_request_ready ?external_request)
        (instrument_available ?funding_instrument)
      )
    :effect
      (and
        (instrument_allocated ?funding_instrument)
        (instrument_internal_slot_assigned ?funding_instrument ?internal_maturity_slot)
        (instrument_external_slot_assigned ?funding_instrument ?external_maturity_slot)
        (not
          (instrument_available ?funding_instrument)
        )
      )
  )
  (:action create_and_tag_funding_instrument_with_allocation_depth_1
    :parameters (?internal_request - internal_request ?external_request - external_request ?internal_maturity_slot - internal_maturity_slot ?external_maturity_slot - external_maturity_slot ?funding_instrument - funding_instrument)
    :precondition
      (and
        (internal_buffer_engaged ?internal_request)
        (external_buffer_engaged ?external_request)
        (internal_slot_assigned ?internal_request ?internal_maturity_slot)
        (external_slot_assigned ?external_request ?external_maturity_slot)
        (internal_slot_buffered ?internal_maturity_slot)
        (external_slot_ready ?external_maturity_slot)
        (not
          (internal_request_ready ?internal_request)
        )
        (external_request_ready ?external_request)
        (instrument_available ?funding_instrument)
      )
    :effect
      (and
        (instrument_allocated ?funding_instrument)
        (instrument_internal_slot_assigned ?funding_instrument ?internal_maturity_slot)
        (instrument_external_slot_assigned ?funding_instrument ?external_maturity_slot)
        (allocation_depth_1 ?funding_instrument)
        (not
          (instrument_available ?funding_instrument)
        )
      )
  )
  (:action create_and_tag_funding_instrument_with_allocation_depth_2
    :parameters (?internal_request - internal_request ?external_request - external_request ?internal_maturity_slot - internal_maturity_slot ?external_maturity_slot - external_maturity_slot ?funding_instrument - funding_instrument)
    :precondition
      (and
        (internal_buffer_engaged ?internal_request)
        (external_buffer_engaged ?external_request)
        (internal_slot_assigned ?internal_request ?internal_maturity_slot)
        (external_slot_assigned ?external_request ?external_maturity_slot)
        (internal_slot_ready ?internal_maturity_slot)
        (external_slot_buffered ?external_maturity_slot)
        (internal_request_ready ?internal_request)
        (not
          (external_request_ready ?external_request)
        )
        (instrument_available ?funding_instrument)
      )
    :effect
      (and
        (instrument_allocated ?funding_instrument)
        (instrument_internal_slot_assigned ?funding_instrument ?internal_maturity_slot)
        (instrument_external_slot_assigned ?funding_instrument ?external_maturity_slot)
        (allocation_depth_2 ?funding_instrument)
        (not
          (instrument_available ?funding_instrument)
        )
      )
  )
  (:action create_and_tag_funding_instrument_with_allocation_depths_1_2
    :parameters (?internal_request - internal_request ?external_request - external_request ?internal_maturity_slot - internal_maturity_slot ?external_maturity_slot - external_maturity_slot ?funding_instrument - funding_instrument)
    :precondition
      (and
        (internal_buffer_engaged ?internal_request)
        (external_buffer_engaged ?external_request)
        (internal_slot_assigned ?internal_request ?internal_maturity_slot)
        (external_slot_assigned ?external_request ?external_maturity_slot)
        (internal_slot_buffered ?internal_maturity_slot)
        (external_slot_buffered ?external_maturity_slot)
        (not
          (internal_request_ready ?internal_request)
        )
        (not
          (external_request_ready ?external_request)
        )
        (instrument_available ?funding_instrument)
      )
    :effect
      (and
        (instrument_allocated ?funding_instrument)
        (instrument_internal_slot_assigned ?funding_instrument ?internal_maturity_slot)
        (instrument_external_slot_assigned ?funding_instrument ?external_maturity_slot)
        (allocation_depth_1 ?funding_instrument)
        (allocation_depth_2 ?funding_instrument)
        (not
          (instrument_available ?funding_instrument)
        )
      )
  )
  (:action lock_funding_instrument
    :parameters (?funding_instrument - funding_instrument ?internal_request - internal_request ?tenor_profile - tenor_profile)
    :precondition
      (and
        (instrument_allocated ?funding_instrument)
        (internal_buffer_engaged ?internal_request)
        (request_tenor_assignment_for_subject ?internal_request ?tenor_profile)
        (not
          (instrument_locked ?funding_instrument)
        )
      )
    :effect (instrument_locked ?funding_instrument)
  )
  (:action reserve_settlement_profile_for_instrument
    :parameters (?maturity_stagger_plan - maturity_stagger_plan ?settlement_profile - settlement_profile ?funding_instrument - funding_instrument)
    :precondition
      (and
        (tenor_confirmed_for_subject ?maturity_stagger_plan)
        (plan_includes_instrument ?maturity_stagger_plan ?funding_instrument)
        (plan_settlement_profile_assigned ?maturity_stagger_plan ?settlement_profile)
        (settlement_profile_available ?settlement_profile)
        (instrument_allocated ?funding_instrument)
        (instrument_locked ?funding_instrument)
        (not
          (settlement_profile_reserved ?settlement_profile)
        )
      )
    :effect
      (and
        (settlement_profile_reserved ?settlement_profile)
        (settlement_profile_linked_instrument ?settlement_profile ?funding_instrument)
        (not
          (settlement_profile_available ?settlement_profile)
        )
      )
  )
  (:action link_settlement_profile_and_mark_plan
    :parameters (?maturity_stagger_plan - maturity_stagger_plan ?settlement_profile - settlement_profile ?funding_instrument - funding_instrument ?tenor_profile - tenor_profile)
    :precondition
      (and
        (tenor_confirmed_for_subject ?maturity_stagger_plan)
        (plan_settlement_profile_assigned ?maturity_stagger_plan ?settlement_profile)
        (settlement_profile_reserved ?settlement_profile)
        (settlement_profile_linked_instrument ?settlement_profile ?funding_instrument)
        (request_tenor_assignment_for_subject ?maturity_stagger_plan ?tenor_profile)
        (not
          (allocation_depth_1 ?funding_instrument)
        )
        (not
          (plan_instrument_settlement_linked ?maturity_stagger_plan)
        )
      )
    :effect (plan_instrument_settlement_linked ?maturity_stagger_plan)
  )
  (:action apply_regulatory_flag_to_plan
    :parameters (?maturity_stagger_plan - maturity_stagger_plan ?regulatory_flag - regulatory_flag)
    :precondition
      (and
        (tenor_confirmed_for_subject ?maturity_stagger_plan)
        (regulatory_flag_available ?regulatory_flag)
        (not
          (regulatory_flag_applied ?maturity_stagger_plan)
        )
      )
    :effect
      (and
        (regulatory_flag_applied ?maturity_stagger_plan)
        (plan_regulatory_flag_assigned ?maturity_stagger_plan ?regulatory_flag)
        (not
          (regulatory_flag_available ?regulatory_flag)
        )
      )
  )
  (:action apply_regulatory_flag_and_progress_plan
    :parameters (?maturity_stagger_plan - maturity_stagger_plan ?settlement_profile - settlement_profile ?funding_instrument - funding_instrument ?tenor_profile - tenor_profile ?regulatory_flag - regulatory_flag)
    :precondition
