    (work_entity_assigned_crew ?work_item - work_item ?crew_unit - crew_unit)
    (agro_input_available ?agro_input_unit - agro_input_unit)
    (task_reserved_input ?field_task_primary - field_task_primary ?agro_input_unit - agro_input_unit)
    (secondary_task_reserved_input ?field_task_secondary - field_task_secondary ?agro_input_unit - agro_input_unit)
    (task_has_primary_slot ?field_task_primary - field_task_primary ?field_time_slot_primary - field_time_slot_primary)
    (slot_primary_ready ?field_time_slot_primary - field_time_slot_primary)
    (slot_primary_reserved ?field_time_slot_primary - field_time_slot_primary)
    (task_primary_ready_for_order ?field_task_primary - field_task_primary)

    (work_order_available ?work_order - work_order)
    (work_order_assembled ?work_order - work_order)
    (work_order_assigned_primary_slot ?work_order - work_order ?field_time_slot_primary - field_time_slot_primary)
    (work_order_assigned_secondary_slot ?work_order - work_order ?field_time_slot_secondary - field_time_slot_secondary)
    (work_order_requires_certificate ?work_order - work_order)
    (work_order_requires_additional_inspection ?work_order - work_order)
    (work_order_activated ?work_order - work_order)
    (plan_has_primary_task ?field_operation_plan - field_operation_plan ?field_task_primary - field_task_primary)
    (plan_has_secondary_task ?field_operation_plan - field_operation_plan ?field_task_secondary - field_task_secondary)
    (plan_linked_work_order ?field_operation_plan - field_operation_plan ?work_order - work_order)
    (material_batch_available ?material_batch - material_batch)
    (plan_has_material_batch ?field_operation_plan - field_operation_plan ?material_batch - material_batch)
    (material_reserved ?material_batch - material_batch)
    (material_allocated_to_work_order ?material_batch - material_batch ?work_order - work_order)

    (plan_prepped_for_treatment ?field_operation_plan - field_operation_plan)

    (plan_certificate_attached ?field_operation_plan - field_operation_plan)

    (plan_logistics_allocated ?field_operation_plan - field_operation_plan)
    (plan_execution_staged ?field_operation_plan - field_operation_plan)
    (quality_check_available ?quality_check - quality_check)
    (plan_linked_quality_check ?field_operation_plan - field_operation_plan ?quality_check - quality_check)
    (plan_inspection_recorded ?field_operation_plan - field_operation_plan)
    (plan_inspection_subtask_triggered ?field_operation_plan - field_operation_plan)
    (plan_inspection_subtask_completed ?field_operation_plan - field_operation_plan)
    (compliance_certificate_available ?compliance_certificate - compliance_certificate)
    (plan_attached_certificate ?field_operation_plan - field_operation_plan ?compliance_certificate - compliance_certificate)
    (logistical_resource_available ?logistical_resource - logistical_resource)
    (plan_assigned_logistics ?field_operation_plan - field_operation_plan ?logistical_resource - logistical_resource)
    (treatment_type_available ?treatment_type - treatment_type)
    (plan_assigned_treatment ?field_operation_plan - field_operation_plan ?treatment_type - treatment_type)
    (regulatory_approval_available ?regulatory_approval - regulatory_approval)
    (plan_has_regulatory_approval ?field_operation_plan - field_operation_plan ?regulatory_approval - regulatory_approval)
    (forecast_block_available ?forecast_block - forecast_block)

    (secondary_task_ready_flag ?field_task_secondary - field_task_secondary)
    (plan_finalized ?field_operation_plan - field_operation_plan)
  )
  (:action create_work_item
    :parameters (?work_item - work_item)
    :precondition
      (and
        (not
          (work_entity_created ?work_item)
        )
        (not
          (work_entity_closed ?work_item)
        )
      )
    :effect (work_entity_created ?work_item)
  )
  (:action allocate_equipment_to_work_item
    :parameters (?work_item - work_item ?equipment_unit - equipment_unit)
    :precondition
      (and
        (work_entity_created ?work_item)
        (not
          (work_entity_has_equipment ?work_item)
        )
        (equipment_available ?equipment_unit)
      )
    :effect
      (and
        (work_entity_has_equipment ?work_item)
        (work_entity_assigned_equipment ?work_item ?equipment_unit)
        (not
          (equipment_available ?equipment_unit)
        )
      )
  )
  (:action assign_weather_window_to_work_item
    :parameters (?work_item - work_item ?weather_window - weather_window)
    :precondition
      (and
        (work_entity_created ?work_item)
        (work_entity_has_equipment ?work_item)
        (weather_window_available ?weather_window)
      )
    :effect
      (and
        (work_entity_assigned_weather_window ?work_item ?weather_window)
        (not
          (weather_window_available ?weather_window)
        )
      )
  )
  (:action confirm_work_item_schedule
    :parameters (?work_item - work_item ?weather_window - weather_window)
    :precondition
      (and
        (work_entity_created ?work_item)
        (work_entity_has_equipment ?work_item)
        (work_entity_assigned_weather_window ?work_item ?weather_window)
        (not
          (work_entity_planned ?work_item)
        )
      )
    :effect (work_entity_planned ?work_item)
  )
  (:action release_weather_window_from_work_item
    :parameters (?work_item - work_item ?weather_window - weather_window)
    :precondition
      (and
        (work_entity_assigned_weather_window ?work_item ?weather_window)
      )
    :effect
      (and
        (weather_window_available ?weather_window)
        (not
          (work_entity_assigned_weather_window ?work_item ?weather_window)
        )
      )
  )
  (:action assign_crew_to_work_item
    :parameters (?work_item - work_item ?crew_unit - crew_unit)
    :precondition
      (and
        (work_entity_planned ?work_item)
        (crew_available ?crew_unit)
      )
    :effect
      (and
        (work_entity_assigned_crew ?work_item ?crew_unit)
        (not
          (crew_available ?crew_unit)
        )
      )
  )
  (:action release_crew_from_work_item
    :parameters (?work_item - work_item ?crew_unit - crew_unit)
    :precondition
      (and
        (work_entity_assigned_crew ?work_item ?crew_unit)
      )
    :effect
      (and
        (crew_available ?crew_unit)
        (not
          (work_entity_assigned_crew ?work_item ?crew_unit)
        )
      )
  )
  (:action attach_treatment_to_plan
    :parameters (?field_operation_plan - field_operation_plan ?treatment_type - treatment_type)
    :precondition
      (and
        (work_entity_planned ?field_operation_plan)
        (treatment_type_available ?treatment_type)
      )
    :effect
      (and
        (plan_assigned_treatment ?field_operation_plan ?treatment_type)
        (not
          (treatment_type_available ?treatment_type)
        )
      )
  )
  (:action detach_treatment_from_plan
    :parameters (?field_operation_plan - field_operation_plan ?treatment_type - treatment_type)
    :precondition
      (and
        (plan_assigned_treatment ?field_operation_plan ?treatment_type)
      )
    :effect
      (and
        (treatment_type_available ?treatment_type)
        (not
          (plan_assigned_treatment ?field_operation_plan ?treatment_type)
        )
      )
  )
  (:action attach_regulatory_approval_to_plan
    :parameters (?field_operation_plan - field_operation_plan ?regulatory_approval - regulatory_approval)
    :precondition
      (and
        (work_entity_planned ?field_operation_plan)
        (regulatory_approval_available ?regulatory_approval)
      )
    :effect
      (and
        (plan_has_regulatory_approval ?field_operation_plan ?regulatory_approval)
        (not
          (regulatory_approval_available ?regulatory_approval)
        )
      )
  )
  (:action detach_regulatory_approval_from_plan
    :parameters (?field_operation_plan - field_operation_plan ?regulatory_approval - regulatory_approval)
    :precondition
      (and
        (plan_has_regulatory_approval ?field_operation_plan ?regulatory_approval)
      )
    :effect
      (and
        (regulatory_approval_available ?regulatory_approval)
        (not
          (plan_has_regulatory_approval ?field_operation_plan ?regulatory_approval)
        )
      )
  )
  (:action mark_time_slot_primary_ready_for_task
    :parameters (?field_task_primary - field_task_primary ?field_time_slot_primary - field_time_slot_primary ?weather_window - weather_window)
    :precondition
      (and
        (work_entity_planned ?field_task_primary)
        (work_entity_assigned_weather_window ?field_task_primary ?weather_window)
        (task_has_primary_slot ?field_task_primary ?field_time_slot_primary)
        (not
          (slot_primary_ready ?field_time_slot_primary)
        )
        (not
          (slot_primary_reserved ?field_time_slot_primary)
        )
      )
    :effect (slot_primary_ready ?field_time_slot_primary)
  )
  (:action assign_crew_and_flag_primary_task_for_order
    :parameters (?field_task_primary - field_task_primary ?field_time_slot_primary - field_time_slot_primary ?crew_unit - crew_unit)
    :precondition
      (and
        (work_entity_planned ?field_task_primary)
        (work_entity_assigned_crew ?field_task_primary ?crew_unit)
        (task_has_primary_slot ?field_task_primary ?field_time_slot_primary)
        (slot_primary_ready ?field_time_slot_primary)
        (not
          (primary_task_ready_flag ?field_task_primary)
        )
      )
    :effect
      (and
        (primary_task_ready_flag ?field_task_primary)
        (task_primary_ready_for_order ?field_task_primary)
      )
  )
  (:action allocate_input_and_reserve_slot_primary
    :parameters (?field_task_primary - field_task_primary ?field_time_slot_primary - field_time_slot_primary ?agro_input_unit - agro_input_unit)
    :precondition
      (and
        (work_entity_planned ?field_task_primary)
        (task_has_primary_slot ?field_task_primary ?field_time_slot_primary)
        (agro_input_available ?agro_input_unit)
        (not
          (primary_task_ready_flag ?field_task_primary)
        )
      )
    :effect
      (and
        (slot_primary_reserved ?field_time_slot_primary)
        (primary_task_ready_flag ?field_task_primary)
        (task_reserved_input ?field_task_primary ?agro_input_unit)
        (not
          (agro_input_available ?agro_input_unit)
        )
      )
  )
  (:action finalize_input_allocation_and_toggle_slot_primary
    :parameters (?field_task_primary - field_task_primary ?field_time_slot_primary - field_time_slot_primary ?weather_window - weather_window ?agro_input_unit - agro_input_unit)
    :precondition
      (and
        (work_entity_planned ?field_task_primary)
        (work_entity_assigned_weather_window ?field_task_primary ?weather_window)
        (task_has_primary_slot ?field_task_primary ?field_time_slot_primary)
        (slot_primary_reserved ?field_time_slot_primary)
        (task_reserved_input ?field_task_primary ?agro_input_unit)
        (not
          (task_primary_ready_for_order ?field_task_primary)
        )
      )
    :effect
      (and
        (slot_primary_ready ?field_time_slot_primary)
        (task_primary_ready_for_order ?field_task_primary)
        (agro_input_available ?agro_input_unit)
        (not
          (task_reserved_input ?field_task_primary ?agro_input_unit)
        )
      )
  )
  (:action mark_time_slot_secondary_ready_for_task
    :parameters (?field_task_secondary - field_task_secondary ?field_time_slot_secondary - field_time_slot_secondary ?weather_window - weather_window)
    :precondition
      (and
        (work_entity_planned ?field_task_secondary)
        (work_entity_assigned_weather_window ?field_task_secondary ?weather_window)
        (task_has_secondary_slot ?field_task_secondary ?field_time_slot_secondary)
        (not
          (slot_secondary_ready ?field_time_slot_secondary)
        )
        (not
          (slot_secondary_reserved ?field_time_slot_secondary)
        )
      )
    :effect (slot_secondary_ready ?field_time_slot_secondary)
  )
  (:action assign_crew_and_flag_secondary_task_for_order
    :parameters (?field_task_secondary - field_task_secondary ?field_time_slot_secondary - field_time_slot_secondary ?crew_unit - crew_unit)
    :precondition
      (and
        (work_entity_planned ?field_task_secondary)
        (work_entity_assigned_crew ?field_task_secondary ?crew_unit)
        (task_has_secondary_slot ?field_task_secondary ?field_time_slot_secondary)
        (slot_secondary_ready ?field_time_slot_secondary)
        (not
          (secondary_task_ready_flag ?field_task_secondary)
        )
      )
    :effect
      (and
        (secondary_task_ready_flag ?field_task_secondary)
        (task_secondary_ready_for_order ?field_task_secondary)
      )
  )
  (:action allocate_input_and_reserve_slot_secondary
    :parameters (?field_task_secondary - field_task_secondary ?field_time_slot_secondary - field_time_slot_secondary ?agro_input_unit - agro_input_unit)
    :precondition
      (and
        (work_entity_planned ?field_task_secondary)
        (task_has_secondary_slot ?field_task_secondary ?field_time_slot_secondary)
        (agro_input_available ?agro_input_unit)
        (not
          (secondary_task_ready_flag ?field_task_secondary)
        )
      )
    :effect
      (and
        (slot_secondary_reserved ?field_time_slot_secondary)
        (secondary_task_ready_flag ?field_task_secondary)
        (secondary_task_reserved_input ?field_task_secondary ?agro_input_unit)
        (not
          (agro_input_available ?agro_input_unit)
        )
      )
  )
  (:action finalize_input_allocation_and_toggle_slot_secondary
    :parameters (?field_task_secondary - field_task_secondary ?field_time_slot_secondary - field_time_slot_secondary ?weather_window - weather_window ?agro_input_unit - agro_input_unit)
    :precondition
      (and
        (work_entity_planned ?field_task_secondary)
        (work_entity_assigned_weather_window ?field_task_secondary ?weather_window)
        (task_has_secondary_slot ?field_task_secondary ?field_time_slot_secondary)
        (slot_secondary_reserved ?field_time_slot_secondary)
        (secondary_task_reserved_input ?field_task_secondary ?agro_input_unit)
        (not
          (task_secondary_ready_for_order ?field_task_secondary)
        )
      )
    :effect
      (and
        (slot_secondary_ready ?field_time_slot_secondary)
        (task_secondary_ready_for_order ?field_task_secondary)
        (agro_input_available ?agro_input_unit)
        (not
          (secondary_task_reserved_input ?field_task_secondary ?agro_input_unit)
        )
      )
  )
  (:action assemble_work_order_from_ready_slots
    :parameters (?field_task_primary - field_task_primary ?field_task_secondary - field_task_secondary ?field_time_slot_primary - field_time_slot_primary ?field_time_slot_secondary - field_time_slot_secondary ?work_order - work_order)
    :precondition
      (and
        (primary_task_ready_flag ?field_task_primary)
        (secondary_task_ready_flag ?field_task_secondary)
        (task_has_primary_slot ?field_task_primary ?field_time_slot_primary)
        (task_has_secondary_slot ?field_task_secondary ?field_time_slot_secondary)
        (slot_primary_ready ?field_time_slot_primary)
        (slot_secondary_ready ?field_time_slot_secondary)
        (task_primary_ready_for_order ?field_task_primary)
        (task_secondary_ready_for_order ?field_task_secondary)
        (work_order_available ?work_order)
      )
    :effect
      (and
        (work_order_assembled ?work_order)
        (work_order_assigned_primary_slot ?work_order ?field_time_slot_primary)
        (work_order_assigned_secondary_slot ?work_order ?field_time_slot_secondary)
        (not
          (work_order_available ?work_order)
        )
      )
  )
  (:action assemble_work_order_with_reserved_primary_slot
    :parameters (?field_task_primary - field_task_primary ?field_task_secondary - field_task_secondary ?field_time_slot_primary - field_time_slot_primary ?field_time_slot_secondary - field_time_slot_secondary ?work_order - work_order)
    :precondition
      (and
        (primary_task_ready_flag ?field_task_primary)
        (secondary_task_ready_flag ?field_task_secondary)
        (task_has_primary_slot ?field_task_primary ?field_time_slot_primary)
        (task_has_secondary_slot ?field_task_secondary ?field_time_slot_secondary)
        (slot_primary_reserved ?field_time_slot_primary)
        (slot_secondary_ready ?field_time_slot_secondary)
        (not
          (task_primary_ready_for_order ?field_task_primary)
        )
        (task_secondary_ready_for_order ?field_task_secondary)
        (work_order_available ?work_order)
      )
    :effect
      (and
        (work_order_assembled ?work_order)
        (work_order_assigned_primary_slot ?work_order ?field_time_slot_primary)
        (work_order_assigned_secondary_slot ?work_order ?field_time_slot_secondary)
        (work_order_requires_certificate ?work_order)
        (not
          (work_order_available ?work_order)
        )
      )
  )
  (:action assemble_work_order_with_reserved_secondary_slot
    :parameters (?field_task_primary - field_task_primary ?field_task_secondary - field_task_secondary ?field_time_slot_primary - field_time_slot_primary ?field_time_slot_secondary - field_time_slot_secondary ?work_order - work_order)
    :precondition
      (and
        (primary_task_ready_flag ?field_task_primary)
        (secondary_task_ready_flag ?field_task_secondary)
        (task_has_primary_slot ?field_task_primary ?field_time_slot_primary)
        (task_has_secondary_slot ?field_task_secondary ?field_time_slot_secondary)
        (slot_primary_ready ?field_time_slot_primary)
        (slot_secondary_reserved ?field_time_slot_secondary)
        (task_primary_ready_for_order ?field_task_primary)
        (not
          (task_secondary_ready_for_order ?field_task_secondary)
        )
        (work_order_available ?work_order)
      )
    :effect
      (and
        (work_order_assembled ?work_order)
        (work_order_assigned_primary_slot ?work_order ?field_time_slot_primary)
        (work_order_assigned_secondary_slot ?work_order ?field_time_slot_secondary)
        (work_order_requires_additional_inspection ?work_order)
        (not
          (work_order_available ?work_order)
        )
      )
  )
  (:action assemble_work_order_with_both_reserved_slots
    :parameters (?field_task_primary - field_task_primary ?field_task_secondary - field_task_secondary ?field_time_slot_primary - field_time_slot_primary ?field_time_slot_secondary - field_time_slot_secondary ?work_order - work_order)
    :precondition
      (and
        (primary_task_ready_flag ?field_task_primary)
        (secondary_task_ready_flag ?field_task_secondary)
        (task_has_primary_slot ?field_task_primary ?field_time_slot_primary)
        (task_has_secondary_slot ?field_task_secondary ?field_time_slot_secondary)
        (slot_primary_reserved ?field_time_slot_primary)
        (slot_secondary_reserved ?field_time_slot_secondary)
        (not
          (task_primary_ready_for_order ?field_task_primary)
        )
        (not
          (task_secondary_ready_for_order ?field_task_secondary)
        )
        (work_order_available ?work_order)
      )
    :effect
      (and
        (work_order_assembled ?work_order)
        (work_order_assigned_primary_slot ?work_order ?field_time_slot_primary)
        (work_order_assigned_secondary_slot ?work_order ?field_time_slot_secondary)
        (work_order_requires_certificate ?work_order)
        (work_order_requires_additional_inspection ?work_order)
        (not
          (work_order_available ?work_order)
        )
      )
  )
  (:action activate_work_order_for_execution
    :parameters (?work_order - work_order ?field_task_primary - field_task_primary ?weather_window - weather_window)
    :precondition
      (and
        (work_order_assembled ?work_order)
        (primary_task_ready_flag ?field_task_primary)
        (work_entity_assigned_weather_window ?field_task_primary ?weather_window)
        (not
          (work_order_activated ?work_order)
        )
      )
    :effect (work_order_activated ?work_order)
  )
  (:action reserve_material_batch_and_allocate_to_work_order
    :parameters (?field_operation_plan - field_operation_plan ?material_batch - material_batch ?work_order - work_order)
    :precondition
      (and
        (work_entity_planned ?field_operation_plan)
        (plan_linked_work_order ?field_operation_plan ?work_order)
        (plan_has_material_batch ?field_operation_plan ?material_batch)
        (material_batch_available ?material_batch)
        (work_order_assembled ?work_order)
        (work_order_activated ?work_order)
        (not
          (material_reserved ?material_batch)
        )
      )
    :effect
      (and
        (material_reserved ?material_batch)
        (material_allocated_to_work_order ?material_batch ?work_order)
        (not
          (material_batch_available ?material_batch)
        )
      )
  )
  (:action confirm_material_allocation_for_plan
    :parameters (?field_operation_plan - field_operation_plan ?material_batch - material_batch ?work_order - work_order ?weather_window - weather_window)
    :precondition
      (and
        (work_entity_planned ?field_operation_plan)
        (plan_has_material_batch ?field_operation_plan ?material_batch)
        (material_reserved ?material_batch)
        (material_allocated_to_work_order ?material_batch ?work_order)
        (work_entity_assigned_weather_window ?field_operation_plan ?weather_window)
        (not
          (work_order_requires_certificate ?work_order)
        )
        (not
          (plan_material_confirmed ?field_operation_plan)
        )
      )
    :effect (plan_material_confirmed ?field_operation_plan)
  )
  (:action attach_compliance_certificate_to_plan
    :parameters (?field_operation_plan - field_operation_plan ?compliance_certificate - compliance_certificate)
    :precondition
      (and
        (work_entity_planned ?field_operation_plan)
        (compliance_certificate_available ?compliance_certificate)
        (not
          (plan_certificate_attached ?field_operation_plan)
        )
      )
