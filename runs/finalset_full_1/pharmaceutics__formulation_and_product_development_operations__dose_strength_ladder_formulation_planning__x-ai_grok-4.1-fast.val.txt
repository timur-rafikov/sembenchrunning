--- domain.pddl (as passed to VAL) ---
(define (domain pharmaceutics_dose_strength_ladder_planning_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_supertype - object strength_node_supertype - object component_supertype - object development_entity_supertype - object development_entity - development_entity_supertype raw_material_batch - resource_supertype analytical_resource - resource_supertype equipment_or_expertise - resource_supertype packaging_option - resource_supertype regulatory_document - resource_supertype evidence_item - resource_supertype manufacturing_process - resource_supertype critical_quality_attribute - resource_supertype excipient_material - strength_node_supertype stability_protocol - strength_node_supertype risk_item - strength_node_supertype strength_profile_low - component_supertype strength_profile_high - component_supertype formulation_prototype - component_supertype dose_level_supertype - development_entity formulation_campaign_supertype - development_entity dose_level - dose_level_supertype dose_level_variant - dose_level_supertype formulation_campaign - formulation_campaign_supertype)
  (:predicates
    (development_entity_registered ?development_entity - development_entity)
    (development_entity_validated ?development_entity - development_entity)
    (development_entity_raw_material_assigned ?development_entity - development_entity)
    (scale_up_ready ?development_entity - development_entity)
    (evidence_package_assembled ?development_entity - development_entity)
    (evidence_validated ?development_entity - development_entity)
    (raw_material_available ?raw_material_batch - raw_material_batch)
    (development_entity_assigned_raw_material ?development_entity - development_entity ?raw_material_batch - raw_material_batch)
    (analytical_resource_available ?analytical_resource - analytical_resource)
    (development_entity_assigned_analytical_resource ?development_entity - development_entity ?analytical_resource - analytical_resource)
    (equipment_available ?equipment_or_expertise - equipment_or_expertise)
    (development_entity_assigned_equipment ?development_entity - development_entity ?equipment_or_expertise - equipment_or_expertise)
    (excipient_available ?excipient_material - excipient_material)
    (dose_excipient_assigned ?dose_level - dose_level ?excipient_material - excipient_material)
    (dose_variant_excipient_assigned ?dose_level_variant - dose_level_variant ?excipient_material - excipient_material)
    (dose_low_profile_linked ?dose_level - dose_level ?strength_profile_low - strength_profile_low)
    (low_profile_precheck_passed ?strength_profile_low - strength_profile_low)
    (low_profile_compatibility_initiated ?strength_profile_low - strength_profile_low)
    (dose_level_process_ready ?dose_level - dose_level)
    (variant_high_profile_linked ?dose_level_variant - dose_level_variant ?strength_profile_high - strength_profile_high)
    (high_profile_precheck_passed ?strength_profile_high - strength_profile_high)
    (high_profile_compatibility_initiated ?strength_profile_high - strength_profile_high)
    (dose_variant_process_ready ?dose_level_variant - dose_level_variant)
    (prototype_slot_unallocated ?formulation_prototype - formulation_prototype)
    (prototype_registered ?formulation_prototype - formulation_prototype)
    (prototype_linked_low_profile ?formulation_prototype - formulation_prototype ?strength_profile_low - strength_profile_low)
    (prototype_linked_high_profile ?formulation_prototype - formulation_prototype ?strength_profile_high - strength_profile_high)
    (prototype_ready_for_scaleup_low ?formulation_prototype - formulation_prototype)
    (prototype_ready_for_scaleup_high ?formulation_prototype - formulation_prototype)
    (prototype_quality_tested ?formulation_prototype - formulation_prototype)
    (campaign_includes_dose_level ?formulation_campaign - formulation_campaign ?dose_level - dose_level)
    (campaign_includes_dose_variant ?formulation_campaign - formulation_campaign ?dose_level_variant - dose_level_variant)
    (campaign_includes_prototype ?formulation_campaign - formulation_campaign ?formulation_prototype - formulation_prototype)
    (stability_protocol_available ?stability_protocol - stability_protocol)
    (campaign_has_stability_protocol ?formulation_campaign - formulation_campaign ?stability_protocol - stability_protocol)
    (stability_protocol_assigned ?stability_protocol - stability_protocol)
    (stability_protocol_linked_to_prototype ?stability_protocol - stability_protocol ?formulation_prototype - formulation_prototype)
    (campaign_process_mapped ?formulation_campaign - formulation_campaign)
    (campaign_process_confirmed ?formulation_campaign - formulation_campaign)
    (campaign_stability_evidence_aggregated ?formulation_campaign - formulation_campaign)
    (campaign_packaging_reserved ?formulation_campaign - formulation_campaign)
    (campaign_packaging_confirmed ?formulation_campaign - formulation_campaign)
    (campaign_regulatory_ready ?formulation_campaign - formulation_campaign)
    (campaign_process_validation_completed ?formulation_campaign - formulation_campaign)
    (risk_item_available ?risk_item - risk_item)
    (campaign_has_risk_item ?formulation_campaign - formulation_campaign ?risk_item - risk_item)
    (campaign_risk_acknowledged ?formulation_campaign - formulation_campaign)
    (campaign_risk_mitigation_assigned ?formulation_campaign - formulation_campaign)
    (campaign_risk_mitigation_completed ?formulation_campaign - formulation_campaign)
    (packaging_option_available ?packaging_option - packaging_option)
    (campaign_packaging_assigned ?formulation_campaign - formulation_campaign ?packaging_option - packaging_option)
    (regulatory_document_available ?regulatory_document - regulatory_document)
    (campaign_regulatory_document_assigned ?formulation_campaign - formulation_campaign ?regulatory_document - regulatory_document)
    (manufacturing_process_available ?manufacturing_process - manufacturing_process)
    (campaign_manufacturing_process_assigned ?formulation_campaign - formulation_campaign ?manufacturing_process - manufacturing_process)
    (critical_quality_attribute_available ?critical_quality_attribute - critical_quality_attribute)
    (campaign_cqa_assigned ?formulation_campaign - formulation_campaign ?critical_quality_attribute - critical_quality_attribute)
    (evidence_item_available ?evidence_item - evidence_item)
    (entity_has_evidence_item ?development_entity - development_entity ?evidence_item - evidence_item)
    (dose_level_ready_for_prototype ?dose_level - dose_level)
    (dose_variant_ready_for_prototype ?dose_level_variant - dose_level_variant)
    (campaign_finalized ?formulation_campaign - formulation_campaign)
  )
  (:action register_development_entity
    :parameters (?development_entity - development_entity)
    :precondition
      (and
        (not
          (development_entity_registered ?development_entity)
        )
        (not
          (scale_up_ready ?development_entity)
        )
      )
    :effect (development_entity_registered ?development_entity)
  )
  (:action assign_raw_material_to_development_entity
    :parameters (?development_entity - development_entity ?raw_material_batch - raw_material_batch)
    :precondition
      (and
        (development_entity_registered ?development_entity)
        (not
          (development_entity_raw_material_assigned ?development_entity)
        )
        (raw_material_available ?raw_material_batch)
      )
    :effect
      (and
        (development_entity_raw_material_assigned ?development_entity)
        (development_entity_assigned_raw_material ?development_entity ?raw_material_batch)
        (not
          (raw_material_available ?raw_material_batch)
        )
      )
  )
  (:action reserve_analytical_resource_for_development_entity
    :parameters (?development_entity - development_entity ?analytical_resource - analytical_resource)
    :precondition
      (and
        (development_entity_registered ?development_entity)
        (development_entity_raw_material_assigned ?development_entity)
        (analytical_resource_available ?analytical_resource)
      )
    :effect
      (and
        (development_entity_assigned_analytical_resource ?development_entity ?analytical_resource)
        (not
          (analytical_resource_available ?analytical_resource)
        )
      )
  )
  (:action validate_development_entity_analytically
    :parameters (?development_entity - development_entity ?analytical_resource - analytical_resource)
    :precondition
      (and
        (development_entity_registered ?development_entity)
        (development_entity_raw_material_assigned ?development_entity)
        (development_entity_assigned_analytical_resource ?development_entity ?analytical_resource)
        (not
          (development_entity_validated ?development_entity)
        )
      )
    :effect (development_entity_validated ?development_entity)
  )
  (:action release_analytical_resource_from_development_entity
    :parameters (?development_entity - development_entity ?analytical_resource - analytical_resource)
    :precondition
      (and
        (development_entity_assigned_analytical_resource ?development_entity ?analytical_resource)
      )
    :effect
      (and
        (analytical_resource_available ?analytical_resource)
        (not
          (development_entity_assigned_analytical_resource ?development_entity ?analytical_resource)
        )
      )
  )
  (:action reserve_equipment_for_development_entity
    :parameters (?development_entity - development_entity ?equipment_or_expertise - equipment_or_expertise)
    :precondition
      (and
        (development_entity_validated ?development_entity)
        (equipment_available ?equipment_or_expertise)
      )
    :effect
      (and
        (development_entity_assigned_equipment ?development_entity ?equipment_or_expertise)
        (not
          (equipment_available ?equipment_or_expertise)
        )
      )
  )
  (:action release_equipment_from_development_entity
    :parameters (?development_entity - development_entity ?equipment_or_expertise - equipment_or_expertise)
    :precondition
      (and
        (development_entity_assigned_equipment ?development_entity ?equipment_or_expertise)
      )
    :effect
      (and
        (equipment_available ?equipment_or_expertise)
        (not
          (development_entity_assigned_equipment ?development_entity ?equipment_or_expertise)
        )
      )
  )
  (:action assign_manufacturing_process_to_campaign
    :parameters (?formulation_campaign - formulation_campaign ?manufacturing_process - manufacturing_process)
    :precondition
      (and
        (development_entity_validated ?formulation_campaign)
        (manufacturing_process_available ?manufacturing_process)
      )
    :effect
      (and
        (campaign_manufacturing_process_assigned ?formulation_campaign ?manufacturing_process)
        (not
          (manufacturing_process_available ?manufacturing_process)
        )
      )
  )
  (:action release_manufacturing_process_from_campaign
    :parameters (?formulation_campaign - formulation_campaign ?manufacturing_process - manufacturing_process)
    :precondition
      (and
        (campaign_manufacturing_process_assigned ?formulation_campaign ?manufacturing_process)
      )
    :effect
      (and
        (manufacturing_process_available ?manufacturing_process)
        (not
          (campaign_manufacturing_process_assigned ?formulation_campaign ?manufacturing_process)
        )
      )
  )
  (:action assign_cqa_to_campaign
    :parameters (?formulation_campaign - formulation_campaign ?critical_quality_attribute - critical_quality_attribute)
    :precondition
      (and
        (development_entity_validated ?formulation_campaign)
        (critical_quality_attribute_available ?critical_quality_attribute)
      )
    :effect
      (and
        (campaign_cqa_assigned ?formulation_campaign ?critical_quality_attribute)
        (not
          (critical_quality_attribute_available ?critical_quality_attribute)
        )
      )
  )
  (:action release_cqa_from_campaign
    :parameters (?formulation_campaign - formulation_campaign ?critical_quality_attribute - critical_quality_attribute)
    :precondition
      (and
        (campaign_cqa_assigned ?formulation_campaign ?critical_quality_attribute)
      )
    :effect
      (and
        (critical_quality_attribute_available ?critical_quality_attribute)
        (not
          (campaign_cqa_assigned ?formulation_campaign ?critical_quality_attribute)
        )
      )
  )
  (:action perform_low_profile_precheck
    :parameters (?dose_level - dose_level ?strength_profile_low - strength_profile_low ?analytical_resource - analytical_resource)
    :precondition
      (and
        (development_entity_validated ?dose_level)
        (development_entity_assigned_analytical_resource ?dose_level ?analytical_resource)
        (dose_low_profile_linked ?dose_level ?strength_profile_low)
        (not
          (low_profile_precheck_passed ?strength_profile_low)
        )
        (not
          (low_profile_compatibility_initiated ?strength_profile_low)
        )
      )
    :effect (low_profile_precheck_passed ?strength_profile_low)
  )
  (:action execute_low_profile_equipment_evaluation
    :parameters (?dose_level - dose_level ?strength_profile_low - strength_profile_low ?equipment_or_expertise - equipment_or_expertise)
    :precondition
      (and
        (development_entity_validated ?dose_level)
        (development_entity_assigned_equipment ?dose_level ?equipment_or_expertise)
        (dose_low_profile_linked ?dose_level ?strength_profile_low)
        (low_profile_precheck_passed ?strength_profile_low)
        (not
          (dose_level_ready_for_prototype ?dose_level)
        )
      )
    :effect
      (and
        (dose_level_ready_for_prototype ?dose_level)
        (dose_level_process_ready ?dose_level)
      )
  )
  (:action initiate_excipient_compatibility_test
    :parameters (?dose_level - dose_level ?strength_profile_low - strength_profile_low ?excipient_material - excipient_material)
    :precondition
      (and
        (development_entity_validated ?dose_level)
        (dose_low_profile_linked ?dose_level ?strength_profile_low)
        (excipient_available ?excipient_material)
        (not
          (dose_level_ready_for_prototype ?dose_level)
        )
      )
    :effect
      (and
        (low_profile_compatibility_initiated ?strength_profile_low)
        (dose_level_ready_for_prototype ?dose_level)
        (dose_excipient_assigned ?dose_level ?excipient_material)
        (not
          (excipient_available ?excipient_material)
        )
      )
  )
  (:action finalize_excipient_compatibility_for_dose
    :parameters (?dose_level - dose_level ?strength_profile_low - strength_profile_low ?analytical_resource - analytical_resource ?excipient_material - excipient_material)
    :precondition
      (and
        (development_entity_validated ?dose_level)
        (development_entity_assigned_analytical_resource ?dose_level ?analytical_resource)
        (dose_low_profile_linked ?dose_level ?strength_profile_low)
        (low_profile_compatibility_initiated ?strength_profile_low)
        (dose_excipient_assigned ?dose_level ?excipient_material)
        (not
          (dose_level_process_ready ?dose_level)
        )
      )
    :effect
      (and
        (low_profile_precheck_passed ?strength_profile_low)
        (dose_level_process_ready ?dose_level)
        (excipient_available ?excipient_material)
        (not
          (dose_excipient_assigned ?dose_level ?excipient_material)
        )
      )
  )
  (:action perform_high_profile_precheck
    :parameters (?dose_level_variant - dose_level_variant ?strength_profile_high - strength_profile_high ?analytical_resource - analytical_resource)
    :precondition
      (and
        (development_entity_validated ?dose_level_variant)
        (development_entity_assigned_analytical_resource ?dose_level_variant ?analytical_resource)
        (variant_high_profile_linked ?dose_level_variant ?strength_profile_high)
        (not
          (high_profile_precheck_passed ?strength_profile_high)
        )
        (not
          (high_profile_compatibility_initiated ?strength_profile_high)
        )
      )
    :effect (high_profile_precheck_passed ?strength_profile_high)
  )
  (:action execute_high_profile_equipment_evaluation
    :parameters (?dose_level_variant - dose_level_variant ?strength_profile_high - strength_profile_high ?equipment_or_expertise - equipment_or_expertise)
    :precondition
      (and
        (development_entity_validated ?dose_level_variant)
        (development_entity_assigned_equipment ?dose_level_variant ?equipment_or_expertise)
        (variant_high_profile_linked ?dose_level_variant ?strength_profile_high)
        (high_profile_precheck_passed ?strength_profile_high)
        (not
          (dose_variant_ready_for_prototype ?dose_level_variant)
        )
      )
    :effect
      (and
        (dose_variant_ready_for_prototype ?dose_level_variant)
        (dose_variant_process_ready ?dose_level_variant)
      )
  )
  (:action initiate_variant_excipient_compatibility_test
    :parameters (?dose_level_variant - dose_level_variant ?strength_profile_high - strength_profile_high ?excipient_material - excipient_material)
    :precondition
      (and
        (development_entity_validated ?dose_level_variant)
        (variant_high_profile_linked ?dose_level_variant ?strength_profile_high)
        (excipient_available ?excipient_material)
        (not
          (dose_variant_ready_for_prototype ?dose_level_variant)
        )
      )
    :effect
      (and
        (high_profile_compatibility_initiated ?strength_profile_high)
        (dose_variant_ready_for_prototype ?dose_level_variant)
        (dose_variant_excipient_assigned ?dose_level_variant ?excipient_material)
        (not
          (excipient_available ?excipient_material)
        )
      )
  )
  (:action finalize_variant_excipient_compatibility_for_dose_variant
    :parameters (?dose_level_variant - dose_level_variant ?strength_profile_high - strength_profile_high ?analytical_resource - analytical_resource ?excipient_material - excipient_material)
    :precondition
      (and
        (development_entity_validated ?dose_level_variant)
        (development_entity_assigned_analytical_resource ?dose_level_variant ?analytical_resource)
        (variant_high_profile_linked ?dose_level_variant ?strength_profile_high)
        (high_profile_compatibility_initiated ?strength_profile_high)
        (dose_variant_excipient_assigned ?dose_level_variant ?excipient_material)
        (not
          (dose_variant_process_ready ?dose_level_variant)
        )
      )
    :effect
      (and
        (high_profile_precheck_passed ?strength_profile_high)
        (dose_variant_process_ready ?dose_level_variant)
        (excipient_available ?excipient_material)
        (not
          (dose_variant_excipient_assigned ?dose_level_variant ?excipient_material)
        )
      )
  )
  (:action assemble_formulation_prototype
    :parameters (?dose_level - dose_level ?dose_level_variant - dose_level_variant ?strength_profile_low - strength_profile_low ?strength_profile_high - strength_profile_high ?formulation_prototype - formulation_prototype)
    :precondition
      (and
        (dose_level_ready_for_prototype ?dose_level)
        (dose_variant_ready_for_prototype ?dose_level_variant)
        (dose_low_profile_linked ?dose_level ?strength_profile_low)
        (variant_high_profile_linked ?dose_level_variant ?strength_profile_high)
        (low_profile_precheck_passed ?strength_profile_low)
        (high_profile_precheck_passed ?strength_profile_high)
        (dose_level_process_ready ?dose_level)
        (dose_variant_process_ready ?dose_level_variant)
        (prototype_slot_unallocated ?formulation_prototype)
      )
    :effect
      (and
        (prototype_registered ?formulation_prototype)
        (prototype_linked_low_profile ?formulation_prototype ?strength_profile_low)
        (prototype_linked_high_profile ?formulation_prototype ?strength_profile_high)
        (not
          (prototype_slot_unallocated ?formulation_prototype)
        )
      )
  )
  (:action initialize_prototype_for_low_profile
    :parameters (?dose_level - dose_level ?dose_level_variant - dose_level_variant ?strength_profile_low - strength_profile_low ?strength_profile_high - strength_profile_high ?formulation_prototype - formulation_prototype)
    :precondition
      (and
        (dose_level_ready_for_prototype ?dose_level)
        (dose_variant_ready_for_prototype ?dose_level_variant)
        (dose_low_profile_linked ?dose_level ?strength_profile_low)
        (variant_high_profile_linked ?dose_level_variant ?strength_profile_high)
        (low_profile_compatibility_initiated ?strength_profile_low)
        (high_profile_precheck_passed ?strength_profile_high)
        (not
          (dose_level_process_ready ?dose_level)
        )
        (dose_variant_process_ready ?dose_level_variant)
        (prototype_slot_unallocated ?formulation_prototype)
      )
    :effect
      (and
        (prototype_registered ?formulation_prototype)
        (prototype_linked_low_profile ?formulation_prototype ?strength_profile_low)
        (prototype_linked_high_profile ?formulation_prototype ?strength_profile_high)
        (prototype_ready_for_scaleup_low ?formulation_prototype)
        (not
          (prototype_slot_unallocated ?formulation_prototype)
        )
      )
  )
  (:action initialize_prototype_for_high_profile
    :parameters (?dose_level - dose_level ?dose_level_variant - dose_level_variant ?strength_profile_low - strength_profile_low ?strength_profile_high - strength_profile_high ?formulation_prototype - formulation_prototype)
    :precondition
      (and
        (dose_level_ready_for_prototype ?dose_level)
        (dose_variant_ready_for_prototype ?dose_level_variant)
        (dose_low_profile_linked ?dose_level ?strength_profile_low)
        (variant_high_profile_linked ?dose_level_variant ?strength_profile_high)
        (low_profile_precheck_passed ?strength_profile_low)
        (high_profile_compatibility_initiated ?strength_profile_high)
        (dose_level_process_ready ?dose_level)
        (not
          (dose_variant_process_ready ?dose_level_variant)
        )
        (prototype_slot_unallocated ?formulation_prototype)
      )
    :effect
      (and
        (prototype_registered ?formulation_prototype)
        (prototype_linked_low_profile ?formulation_prototype ?strength_profile_low)
        (prototype_linked_high_profile ?formulation_prototype ?strength_profile_high)
        (prototype_ready_for_scaleup_high ?formulation_prototype)
        (not
          (prototype_slot_unallocated ?formulation_prototype)
        )
      )
  )
  (:action initialize_prototype_for_both_profiles
    :parameters (?dose_level - dose_level ?dose_level_variant - dose_level_variant ?strength_profile_low - strength_profile_low ?strength_profile_high - strength_profile_high ?formulation_prototype - formulation_prototype)
    :precondition
      (and
        (dose_level_ready_for_prototype ?dose_level)
        (dose_variant_ready_for_prototype ?dose_level_variant)
        (dose_low_profile_linked ?dose_level ?strength_profile_low)
        (variant_high_profile_linked ?dose_level_variant ?strength_profile_high)
        (low_profile_compatibility_initiated ?strength_profile_low)
        (high_profile_compatibility_initiated ?strength_profile_high)
        (not
          (dose_level_process_ready ?dose_level)
        )
        (not
          (dose_variant_process_ready ?dose_level_variant)
        )
        (prototype_slot_unallocated ?formulation_prototype)
      )
    :effect
      (and
        (prototype_registered ?formulation_prototype)
        (prototype_linked_low_profile ?formulation_prototype ?strength_profile_low)
        (prototype_linked_high_profile ?formulation_prototype ?strength_profile_high)
        (prototype_ready_for_scaleup_low ?formulation_prototype)
        (prototype_ready_for_scaleup_high ?formulation_prototype)
        (not
          (prototype_slot_unallocated ?formulation_prototype)
        )
      )
  )
  (:action record_prototype_quality_test
    :parameters (?formulation_prototype - formulation_prototype ?dose_level - dose_level ?analytical_resource - analytical_resource)
    :precondition
      (and
        (prototype_registered ?formulation_prototype)
        (dose_level_ready_for_prototype ?dose_level)
        (development_entity_assigned_analytical_resource ?dose_level ?analytical_resource)
        (not
          (prototype_quality_tested ?formulation_prototype)
        )
      )
    :effect (prototype_quality_tested ?formulation_prototype)
  )
  (:action assign_stability_protocol_to_campaign
    :parameters (?formulation_campaign - formulation_campaign ?stability_protocol - stability_protocol ?formulation_prototype - formulation_prototype)
    :precondition
      (and
        (development_entity_validated ?formulation_campaign)
        (campaign_includes_prototype ?formulation_campaign ?formulation_prototype)
        (campaign_has_stability_protocol ?formulation_campaign ?stability_protocol)
        (stability_protocol_available ?stability_protocol)
        (prototype_registered ?formulation_prototype)
        (prototype_quality_tested ?formulation_prototype)
        (not
          (stability_protocol_assigned ?stability_protocol)
        )
      )
    :effect
      (and
        (stability_protocol_assigned ?stability_protocol)
        (stability_protocol_linked_to_prototype ?stability_protocol ?formulation_prototype)
        (not
          (stability_protocol_available ?stability_protocol)
        )
      )
  )
  (:action confirm_stability_assignment_and_flag_campaign
    :parameters (?formulation_campaign - formulation_campaign ?stability_protocol - stability_protocol ?formulation_prototype - formulation_prototype ?analytical_resource - analytical_resource)
    :precondition
      (and
        (development_entity_validated ?formulation_campaign)
        (campaign_has_stability_protocol ?formulation_campaign ?stability_protocol)
        (stability_protocol_assigned ?stability_protocol)
        (stability_protocol_linked_to_prototype ?stability_protocol ?formulation_prototype)
        (development_entity_assigned_analytical_resource ?formulation_campaign ?analytical_resource)
        (not
          (prototype_ready_for_scaleup_low ?formulation_prototype)
        )
        (not
          (campaign_process_mapped ?formulation_campaign)
        )
      )
    :effect (campaign_process_mapped ?formulation_campaign)
  )
  (:action reserve_packaging_for_campaign
    :parameters (?formulation_campaign - formulation_campaign ?packaging_option - packaging_option)
    :precondition
      (and
        (development_entity_validated ?formulation_campaign)
        (packaging_option_available ?packaging_option)
        (not
          (campaign_packaging_reserved ?formulation_campaign)
        )
      )
    :effect
      (and
        (campaign_packaging_reserved ?formulation_campaign)
        (campaign_packaging_assigned ?formulation_campaign ?packaging_option)
        (not
          (packaging_option_available ?packaging_option)
        )
      )
  )
  (:action assign_packaging_and_confirm_campaign
    :parameters (?formulation_campaign - formulation_campaign ?stability_protocol - stability_protocol ?formulation_prototype - formulation_prototype ?analytical_resource - analytical_resource ?packaging_option - packaging_option)
