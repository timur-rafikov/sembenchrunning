--- domain.pddl (as passed to VAL) ---
(define (domain raid_mechanic_role_rotation)
  (:requirements :strips :typing :negative-preconditions)
  (:types base_object - object role_or_resource_root - base_object marker_root - base_object position_root - base_object raid_entity_root - base_object raid_entity - raid_entity_root role_token - role_or_resource_root mechanic_marker - role_or_resource_root support_device - role_or_resource_root modifier_trigger - role_or_resource_root synergy_buff - role_or_resource_root priority_marker - role_or_resource_root aux_tool - role_or_resource_root phase_flag - role_or_resource_root consumable_item - marker_root objective_check - marker_root tactic - marker_root position_node_a - position_root position_node_b - position_root rotation_window - position_root slot_group - raid_entity alt_slot_group - raid_entity subteam_a_member - slot_group subteam_b_member - slot_group player - alt_slot_group)

  (:predicates
    (slot_prepared ?raid_entity - raid_entity)
    (slot_ready ?raid_entity - raid_entity)
    (role_claimed ?raid_entity - raid_entity)
    (assignment_committed ?raid_entity - raid_entity)
    (ready_for_execution ?raid_entity - raid_entity)
    (execution_started ?raid_entity - raid_entity)
    (role_token_available ?role_token - role_token)
    (assigned_role_token ?raid_entity - raid_entity ?role_token - role_token)
    (mechanic_marker_available ?mechanic_marker - mechanic_marker)
    (mechanic_marker_bound ?raid_entity - raid_entity ?mechanic_marker - mechanic_marker)
    (support_device_available ?support_device - support_device)
    (support_device_equipped ?raid_entity - raid_entity ?support_device - support_device)
    (consumable_available ?consumable_item - consumable_item)
    (subteam_a_consumable_assigned ?subteam_a_member - subteam_a_member ?consumable_item - consumable_item)
    (subteam_b_consumable_assigned ?subteam_b_member - subteam_b_member ?consumable_item - consumable_item)
    (subteam_a_at_position ?subteam_a_member - subteam_a_member ?position_node_a - position_node_a)
    (position_node_a_captured ?position_node_a - position_node_a)
    (position_node_a_marked ?position_node_a - position_node_a)
    (subteam_a_ready_for_window ?subteam_a_member - subteam_a_member)
    (subteam_b_at_position ?subteam_b_member - subteam_b_member ?position_node_b - position_node_b)
    (position_node_b_captured ?position_node_b - position_node_b)
    (position_node_b_marked ?position_node_b - position_node_b)
    (subteam_b_ready_for_window ?subteam_b_member - subteam_b_member)
    (rotation_window_available ?rotation_window - rotation_window)
    (rotation_window_configured ?rotation_window - rotation_window)
    (rotation_window_has_node_a ?rotation_window - rotation_window ?position_node_a - position_node_a)
    (rotation_window_has_node_b ?rotation_window - rotation_window ?position_node_b - position_node_b)
    (rotation_window_flag_a ?rotation_window - rotation_window)
    (rotation_window_flag_b ?rotation_window - rotation_window)
    (rotation_window_activated ?rotation_window - rotation_window)
    (player_linked_to_subteam_a ?player - player ?subteam_a_member - subteam_a_member)
    (player_linked_to_subteam_b ?player - player ?subteam_b_member - subteam_b_member)
    (player_assigned_to_window ?player - player ?rotation_window - rotation_window)
    (objective_check_available ?objective_check - objective_check)
    (player_has_objective_check ?player - player ?objective_check - objective_check)
    (objective_check_satisfied ?objective_check - objective_check)
    (objective_check_linked_to_window ?objective_check - objective_check ?rotation_window - rotation_window)
    (player_preparation_stage_one ?player - player)
    (player_preparation_stage_two ?player - player)
    (player_phase_acknowledged ?player - player)
    (player_modifier_prepared ?player - player)
    (player_modifier_stage ?player - player)
    (player_additional_ready_flag ?player - player)
    (player_joint_preparation_complete ?player - player)
    (tactic_available ?tactic - tactic)
    (player_has_tactic ?player - player ?tactic - tactic)
    (player_tactic_locked ?player - player)
    (player_tactic_confirmed ?player - player)
    (player_tactic_active ?player - player)
    (modifier_trigger_available ?modifier_trigger - modifier_trigger)
    (player_has_modifier ?player - player ?modifier_trigger - modifier_trigger)
    (synergy_buff_available ?synergy_buff - synergy_buff)
    (player_has_synergy_buff ?player - player ?synergy_buff - synergy_buff)
    (aux_tool_available ?aux_tool - aux_tool)
    (player_has_aux_tool ?player - player ?aux_tool - aux_tool)
    (phase_flag_available ?phase_flag - phase_flag)
    (player_phase_flag_assigned ?player - player ?phase_flag - phase_flag)
    (priority_marker_available ?priority_marker - priority_marker)
    (assigned_priority_marker ?raid_entity - raid_entity ?priority_marker - priority_marker)
    (subteam_a_member_engaged ?subteam_a_member - subteam_a_member)
    (subteam_b_member_engaged ?subteam_b_member - subteam_b_member)
    (player_finalized ?player - player)
  )
  (:action prepare_slot
    :parameters (?raid_entity - raid_entity)
    :precondition
      (and
        (not
          (slot_prepared ?raid_entity)
        )
        (not
          (assignment_committed ?raid_entity)
        )
      )
    :effect (slot_prepared ?raid_entity)
  )
  (:action claim_role_token
    :parameters (?raid_entity - raid_entity ?role_token - role_token)
    :precondition
      (and
        (slot_prepared ?raid_entity)
        (not
          (role_claimed ?raid_entity)
        )
        (role_token_available ?role_token)
      )
    :effect
      (and
        (role_claimed ?raid_entity)
        (assigned_role_token ?raid_entity ?role_token)
        (not
          (role_token_available ?role_token)
        )
      )
  )
  (:action attach_mechanic_marker
    :parameters (?raid_entity - raid_entity ?mechanic_marker - mechanic_marker)
    :precondition
      (and
        (slot_prepared ?raid_entity)
        (role_claimed ?raid_entity)
        (mechanic_marker_available ?mechanic_marker)
      )
    :effect
      (and
        (mechanic_marker_bound ?raid_entity ?mechanic_marker)
        (not
          (mechanic_marker_available ?mechanic_marker)
        )
      )
  )
  (:action confirm_slot_ready
    :parameters (?raid_entity - raid_entity ?mechanic_marker - mechanic_marker)
    :precondition
      (and
        (slot_prepared ?raid_entity)
        (role_claimed ?raid_entity)
        (mechanic_marker_bound ?raid_entity ?mechanic_marker)
        (not
          (slot_ready ?raid_entity)
        )
      )
    :effect (slot_ready ?raid_entity)
  )
  (:action release_mechanic_marker
    :parameters (?raid_entity - raid_entity ?mechanic_marker - mechanic_marker)
    :precondition
      (and
        (mechanic_marker_bound ?raid_entity ?mechanic_marker)
      )
    :effect
      (and
        (mechanic_marker_available ?mechanic_marker)
        (not
          (mechanic_marker_bound ?raid_entity ?mechanic_marker)
        )
      )
  )
  (:action equip_support_device
    :parameters (?raid_entity - raid_entity ?support_device - support_device)
    :precondition
      (and
        (slot_ready ?raid_entity)
        (support_device_available ?support_device)
      )
    :effect
      (and
        (support_device_equipped ?raid_entity ?support_device)
        (not
          (support_device_available ?support_device)
        )
      )
  )
  (:action unequip_support_device
    :parameters (?raid_entity - raid_entity ?support_device - support_device)
    :precondition
      (and
        (support_device_equipped ?raid_entity ?support_device)
      )
    :effect
      (and
        (support_device_available ?support_device)
        (not
          (support_device_equipped ?raid_entity ?support_device)
        )
      )
  )
  (:action equip_aux_tool
    :parameters (?player - player ?aux_tool - aux_tool)
    :precondition
      (and
        (slot_ready ?player)
        (aux_tool_available ?aux_tool)
      )
    :effect
      (and
        (player_has_aux_tool ?player ?aux_tool)
        (not
          (aux_tool_available ?aux_tool)
        )
      )
  )
  (:action unequip_aux_tool
    :parameters (?player - player ?aux_tool - aux_tool)
    :precondition
      (and
        (player_has_aux_tool ?player ?aux_tool)
      )
    :effect
      (and
        (aux_tool_available ?aux_tool)
        (not
          (player_has_aux_tool ?player ?aux_tool)
        )
      )
  )
  (:action attach_phase_flag
    :parameters (?player - player ?phase_flag - phase_flag)
    :precondition
      (and
        (slot_ready ?player)
        (phase_flag_available ?phase_flag)
      )
    :effect
      (and
        (player_phase_flag_assigned ?player ?phase_flag)
        (not
          (phase_flag_available ?phase_flag)
        )
      )
  )
  (:action remove_phase_flag
    :parameters (?player - player ?phase_flag - phase_flag)
    :precondition
      (and
        (player_phase_flag_assigned ?player ?phase_flag)
      )
    :effect
      (and
        (phase_flag_available ?phase_flag)
        (not
          (player_phase_flag_assigned ?player ?phase_flag)
        )
      )
  )
  (:action capture_position_node_a
    :parameters (?subteam_a_member - subteam_a_member ?position_node_a - position_node_a ?mechanic_marker - mechanic_marker)
    :precondition
      (and
        (slot_ready ?subteam_a_member)
        (mechanic_marker_bound ?subteam_a_member ?mechanic_marker)
        (subteam_a_at_position ?subteam_a_member ?position_node_a)
        (not
          (position_node_a_captured ?position_node_a)
        )
        (not
          (position_node_a_marked ?position_node_a)
        )
      )
    :effect (position_node_a_captured ?position_node_a)
  )
  (:action confirm_subteam_a_local_rotation
    :parameters (?subteam_a_member - subteam_a_member ?position_node_a - position_node_a ?support_device - support_device)
    :precondition
      (and
        (slot_ready ?subteam_a_member)
        (support_device_equipped ?subteam_a_member ?support_device)
        (subteam_a_at_position ?subteam_a_member ?position_node_a)
        (position_node_a_captured ?position_node_a)
        (not
          (subteam_a_member_engaged ?subteam_a_member)
        )
      )
    :effect
      (and
        (subteam_a_member_engaged ?subteam_a_member)
        (subteam_a_ready_for_window ?subteam_a_member)
      )
  )
  (:action apply_consumable_to_node_a
    :parameters (?subteam_a_member - subteam_a_member ?position_node_a - position_node_a ?consumable_item - consumable_item)
    :precondition
      (and
        (slot_ready ?subteam_a_member)
        (subteam_a_at_position ?subteam_a_member ?position_node_a)
        (consumable_available ?consumable_item)
        (not
          (subteam_a_member_engaged ?subteam_a_member)
        )
      )
    :effect
      (and
        (position_node_a_marked ?position_node_a)
        (subteam_a_member_engaged ?subteam_a_member)
        (subteam_a_consumable_assigned ?subteam_a_member ?consumable_item)
        (not
          (consumable_available ?consumable_item)
        )
      )
  )
  (:action finalize_consumable_capture_a
    :parameters (?subteam_a_member - subteam_a_member ?position_node_a - position_node_a ?mechanic_marker - mechanic_marker ?consumable_item - consumable_item)
    :precondition
      (and
        (slot_ready ?subteam_a_member)
        (mechanic_marker_bound ?subteam_a_member ?mechanic_marker)
        (subteam_a_at_position ?subteam_a_member ?position_node_a)
        (position_node_a_marked ?position_node_a)
        (subteam_a_consumable_assigned ?subteam_a_member ?consumable_item)
        (not
          (subteam_a_ready_for_window ?subteam_a_member)
        )
      )
    :effect
      (and
        (position_node_a_captured ?position_node_a)
        (subteam_a_ready_for_window ?subteam_a_member)
        (consumable_available ?consumable_item)
        (not
          (subteam_a_consumable_assigned ?subteam_a_member ?consumable_item)
        )
      )
  )
  (:action capture_position_node_b
    :parameters (?subteam_b_member - subteam_b_member ?position_node_b - position_node_b ?mechanic_marker - mechanic_marker)
    :precondition
      (and
        (slot_ready ?subteam_b_member)
        (mechanic_marker_bound ?subteam_b_member ?mechanic_marker)
        (subteam_b_at_position ?subteam_b_member ?position_node_b)
        (not
          (position_node_b_captured ?position_node_b)
        )
        (not
          (position_node_b_marked ?position_node_b)
        )
      )
    :effect (position_node_b_captured ?position_node_b)
  )
  (:action confirm_subteam_b_local_rotation
    :parameters (?subteam_b_member - subteam_b_member ?position_node_b - position_node_b ?support_device - support_device)
    :precondition
      (and
        (slot_ready ?subteam_b_member)
        (support_device_equipped ?subteam_b_member ?support_device)
        (subteam_b_at_position ?subteam_b_member ?position_node_b)
        (position_node_b_captured ?position_node_b)
        (not
          (subteam_b_member_engaged ?subteam_b_member)
        )
      )
    :effect
      (and
        (subteam_b_member_engaged ?subteam_b_member)
        (subteam_b_ready_for_window ?subteam_b_member)
      )
  )
  (:action apply_consumable_to_node_b
    :parameters (?subteam_b_member - subteam_b_member ?position_node_b - position_node_b ?consumable_item - consumable_item)
    :precondition
      (and
        (slot_ready ?subteam_b_member)
        (subteam_b_at_position ?subteam_b_member ?position_node_b)
        (consumable_available ?consumable_item)
        (not
          (subteam_b_member_engaged ?subteam_b_member)
        )
      )
    :effect
      (and
        (position_node_b_marked ?position_node_b)
        (subteam_b_member_engaged ?subteam_b_member)
        (subteam_b_consumable_assigned ?subteam_b_member ?consumable_item)
        (not
          (consumable_available ?consumable_item)
        )
      )
  )
  (:action finalize_consumable_capture_b
    :parameters (?subteam_b_member - subteam_b_member ?position_node_b - position_node_b ?mechanic_marker - mechanic_marker ?consumable_item - consumable_item)
    :precondition
      (and
        (slot_ready ?subteam_b_member)
        (mechanic_marker_bound ?subteam_b_member ?mechanic_marker)
        (subteam_b_at_position ?subteam_b_member ?position_node_b)
        (position_node_b_marked ?position_node_b)
        (subteam_b_consumable_assigned ?subteam_b_member ?consumable_item)
        (not
          (subteam_b_ready_for_window ?subteam_b_member)
        )
      )
    :effect
      (and
        (position_node_b_captured ?position_node_b)
        (subteam_b_ready_for_window ?subteam_b_member)
        (consumable_available ?consumable_item)
        (not
          (subteam_b_consumable_assigned ?subteam_b_member ?consumable_item)
        )
      )
  )
  (:action configure_rotation_window_default
    :parameters (?subteam_a_member - subteam_a_member ?subteam_b_member - subteam_b_member ?position_node_a - position_node_a ?position_node_b - position_node_b ?rotation_window - rotation_window)
    :precondition
      (and
        (subteam_a_member_engaged ?subteam_a_member)
        (subteam_b_member_engaged ?subteam_b_member)
        (subteam_a_at_position ?subteam_a_member ?position_node_a)
        (subteam_b_at_position ?subteam_b_member ?position_node_b)
        (position_node_a_captured ?position_node_a)
        (position_node_b_captured ?position_node_b)
        (subteam_a_ready_for_window ?subteam_a_member)
        (subteam_b_ready_for_window ?subteam_b_member)
        (rotation_window_available ?rotation_window)
      )
    :effect
      (and
        (rotation_window_configured ?rotation_window)
        (rotation_window_has_node_a ?rotation_window ?position_node_a)
        (rotation_window_has_node_b ?rotation_window ?position_node_b)
        (not
          (rotation_window_available ?rotation_window)
        )
      )
  )
  (:action configure_rotation_window_with_flag_a
    :parameters (?subteam_a_member - subteam_a_member ?subteam_b_member - subteam_b_member ?position_node_a - position_node_a ?position_node_b - position_node_b ?rotation_window - rotation_window)
    :precondition
      (and
        (subteam_a_member_engaged ?subteam_a_member)
        (subteam_b_member_engaged ?subteam_b_member)
        (subteam_a_at_position ?subteam_a_member ?position_node_a)
        (subteam_b_at_position ?subteam_b_member ?position_node_b)
        (position_node_a_marked ?position_node_a)
        (position_node_b_captured ?position_node_b)
        (not
          (subteam_a_ready_for_window ?subteam_a_member)
        )
        (subteam_b_ready_for_window ?subteam_b_member)
        (rotation_window_available ?rotation_window)
      )
    :effect
      (and
        (rotation_window_configured ?rotation_window)
        (rotation_window_has_node_a ?rotation_window ?position_node_a)
        (rotation_window_has_node_b ?rotation_window ?position_node_b)
        (rotation_window_flag_a ?rotation_window)
        (not
          (rotation_window_available ?rotation_window)
        )
      )
  )
  (:action configure_rotation_window_with_flag_b
    :parameters (?subteam_a_member - subteam_a_member ?subteam_b_member - subteam_b_member ?position_node_a - position_node_a ?position_node_b - position_node_b ?rotation_window - rotation_window)
    :precondition
      (and
        (subteam_a_member_engaged ?subteam_a_member)
        (subteam_b_member_engaged ?subteam_b_member)
        (subteam_a_at_position ?subteam_a_member ?position_node_a)
        (subteam_b_at_position ?subteam_b_member ?position_node_b)
        (position_node_a_captured ?position_node_a)
        (position_node_b_marked ?position_node_b)
        (subteam_a_ready_for_window ?subteam_a_member)
        (not
          (subteam_b_ready_for_window ?subteam_b_member)
        )
        (rotation_window_available ?rotation_window)
      )
    :effect
      (and
        (rotation_window_configured ?rotation_window)
        (rotation_window_has_node_a ?rotation_window ?position_node_a)
        (rotation_window_has_node_b ?rotation_window ?position_node_b)
        (rotation_window_flag_b ?rotation_window)
        (not
          (rotation_window_available ?rotation_window)
        )
      )
  )
  (:action configure_rotation_window_with_flags_a_b
    :parameters (?subteam_a_member - subteam_a_member ?subteam_b_member - subteam_b_member ?position_node_a - position_node_a ?position_node_b - position_node_b ?rotation_window - rotation_window)
    :precondition
      (and
        (subteam_a_member_engaged ?subteam_a_member)
        (subteam_b_member_engaged ?subteam_b_member)
        (subteam_a_at_position ?subteam_a_member ?position_node_a)
        (subteam_b_at_position ?subteam_b_member ?position_node_b)
        (position_node_a_marked ?position_node_a)
        (position_node_b_marked ?position_node_b)
        (not
          (subteam_a_ready_for_window ?subteam_a_member)
        )
        (not
          (subteam_b_ready_for_window ?subteam_b_member)
        )
        (rotation_window_available ?rotation_window)
      )
    :effect
      (and
        (rotation_window_configured ?rotation_window)
        (rotation_window_has_node_a ?rotation_window ?position_node_a)
        (rotation_window_has_node_b ?rotation_window ?position_node_b)
        (rotation_window_flag_a ?rotation_window)
        (rotation_window_flag_b ?rotation_window)
        (not
          (rotation_window_available ?rotation_window)
        )
      )
  )
  (:action activate_rotation_window
    :parameters (?rotation_window - rotation_window ?subteam_a_member - subteam_a_member ?mechanic_marker - mechanic_marker)
    :precondition
      (and
        (rotation_window_configured ?rotation_window)
        (subteam_a_member_engaged ?subteam_a_member)
        (mechanic_marker_bound ?subteam_a_member ?mechanic_marker)
        (not
          (rotation_window_activated ?rotation_window)
        )
      )
    :effect (rotation_window_activated ?rotation_window)
  )
  (:action activate_objective_check
    :parameters (?player - player ?objective_check - objective_check ?rotation_window - rotation_window)
    :precondition
      (and
        (slot_ready ?player)
        (player_assigned_to_window ?player ?rotation_window)
        (player_has_objective_check ?player ?objective_check)
        (objective_check_available ?objective_check)
        (rotation_window_configured ?rotation_window)
        (rotation_window_activated ?rotation_window)
        (not
          (objective_check_satisfied ?objective_check)
        )
      )
    :effect
      (and
        (objective_check_satisfied ?objective_check)
        (objective_check_linked_to_window ?objective_check ?rotation_window)
        (not
          (objective_check_available ?objective_check)
        )
      )
  )
  (:action start_player_preparation_stage
    :parameters (?player - player ?objective_check - objective_check ?rotation_window - rotation_window ?mechanic_marker - mechanic_marker)
    :precondition
      (and
        (slot_ready ?player)
        (player_has_objective_check ?player ?objective_check)
        (objective_check_satisfied ?objective_check)
        (objective_check_linked_to_window ?objective_check ?rotation_window)
        (mechanic_marker_bound ?player ?mechanic_marker)
        (not
          (rotation_window_flag_a ?rotation_window)
        )
        (not
          (player_preparation_stage_one ?player)
        )
      )
    :effect (player_preparation_stage_one ?player)
  )
  (:action assign_modifier_trigger_to_player
    :parameters (?player - player ?modifier_trigger - modifier_trigger)
    :precondition
      (and
        (slot_ready ?player)
        (modifier_trigger_available ?modifier_trigger)
        (not
          (player_modifier_prepared ?player)
        )
      )
    :effect
      (and
        (player_modifier_prepared ?player)
        (player_has_modifier ?player ?modifier_trigger)
        (not
          (modifier_trigger_available ?modifier_trigger)
        )
      )
  )
  (:action apply_modifier_trigger_and_mark_player
    :parameters (?player - player ?objective_check - objective_check ?rotation_window - rotation_window ?mechanic_marker - mechanic_marker ?modifier_trigger - modifier_trigger)
    :precondition
      (and
        (slot_ready ?player)
        (player_has_objective_check ?player ?objective_check)
        (objective_check_satisfied ?objective_check)
        (objective_check_linked_to_window ?objective_check ?rotation_window)
        (mechanic_marker_bound ?player ?mechanic_marker)
        (rotation_window_flag_a ?rotation_window)
        (player_modifier_prepared ?player)
        (player_has_modifier ?player ?modifier_trigger)
        (not
          (player_preparation_stage_one ?player)
        )
      )
    :effect
      (and
        (player_preparation_stage_one ?player)
        (player_modifier_stage ?player)
      )
  )
  (:action prepare_player_with_tool_option1
    :parameters (?player - player ?aux_tool - aux_tool ?support_device - support_device ?objective_check - objective_check ?rotation_window - rotation_window)
    :precondition
      (and
        (player_preparation_stage_one ?player)
