--- domain.pddl (as passed to VAL) ---
(define (domain revive_priority_coordination)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_type - object role_type - object position_type - object player_type - object downed_token - player_type revive_item - resource_type healer_resource - resource_type support_assist - resource_type tactic_marker - resource_type buff_token - resource_type timing_token - resource_type priority_resource - resource_type specialist_token - resource_type utility_item - role_type consumable_pack - role_type custom_marker - role_type anchor_point_a - position_type anchor_point_b - position_type revival_site - position_type member_subtype_a - downed_token member_subtype_b - downed_token subforce_member_a - member_subtype_a subforce_member_b - member_subtype_a coordinator_agent - member_subtype_b)
  (:predicates
    (downed_tagged ?downed_token - downed_token)
    (ready_for_revive ?downed_token - downed_token)
    (entity_item_reserved ?downed_token - downed_token)
    (member_revived ?downed_token - downed_token)
    (commit_signal ?downed_token - downed_token)
    (revive_committed ?downed_token - downed_token)
    (revive_item_available ?revive_item - revive_item)
    (entity_assigned_revive_item ?downed_token - downed_token ?revive_item - revive_item)
    (healer_slot_available ?healer_resource - healer_resource)
    (entity_assigned_healer ?downed_token - downed_token ?healer_resource - healer_resource)
    (support_assist_available ?support_assist - support_assist)
    (entity_assigned_support_assist ?downed_token - downed_token ?support_assist - support_assist)
    (utility_item_available ?utility_item - utility_item)
    (member_a_assigned_utility ?subforce_member_a - subforce_member_a ?utility_item - utility_item)
    (member_b_assigned_utility ?subforce_member_b - subforce_member_b ?utility_item - utility_item)
    (member_a_anchor_assigned ?subforce_member_a - subforce_member_a ?anchor_point_a - anchor_point_a)
    (anchor_a_ready ?anchor_point_a - anchor_point_a)
    (anchor_a_prepared ?anchor_point_a - anchor_point_a)
    (member_a_committed ?subforce_member_a - subforce_member_a)
    (member_b_anchor_assigned ?subforce_member_b - subforce_member_b ?anchor_point_b - anchor_point_b)
    (anchor_b_ready ?anchor_point_b - anchor_point_b)
    (anchor_b_prepared ?anchor_point_b - anchor_point_b)
    (member_b_committed ?subforce_member_b - subforce_member_b)
    (revival_site_available ?revival_site - revival_site)
    (revival_site_reserved ?revival_site - revival_site)
    (revival_site_anchor_a_attached ?revival_site - revival_site ?anchor_point_a - anchor_point_a)
    (revival_site_anchor_b_attached ?revival_site - revival_site ?anchor_point_b - anchor_point_b)
    (revival_site_stage_a ?revival_site - revival_site)
    (revival_site_stage_b ?revival_site - revival_site)
    (revival_site_execution_committed ?revival_site - revival_site)
    (coordinator_member_link_a ?coordinator_agent - coordinator_agent ?subforce_member_a - subforce_member_a)
    (coordinator_member_link_b ?coordinator_agent - coordinator_agent ?subforce_member_b - subforce_member_b)
    (coordinator_attached_site ?coordinator_agent - coordinator_agent ?revival_site - revival_site)
    (consumable_pack_available ?consumable_pack - consumable_pack)
    (coordinator_has_pack ?coordinator_agent - coordinator_agent ?consumable_pack - consumable_pack)
    (consumable_pack_staged ?consumable_pack - consumable_pack)
    (consumable_pack_attached_to_site ?consumable_pack - consumable_pack ?revival_site - revival_site)
    (coordinator_pack_prepared ?coordinator_agent - coordinator_agent)
    (coordinator_stage_two_ready ?coordinator_agent - coordinator_agent)
    (coordinator_ready_for_commit ?coordinator_agent - coordinator_agent)
    (coordinator_has_tactic_marker ?coordinator_agent - coordinator_agent)
    (coordinator_marker_staged ?coordinator_agent - coordinator_agent)
    (coordinator_marker_confirmed ?coordinator_agent - coordinator_agent)
    (coordinator_commit_ready ?coordinator_agent - coordinator_agent)
    (custom_marker_available ?custom_marker - custom_marker)
    (coordinator_has_custom_marker ?coordinator_agent - coordinator_agent ?custom_marker - custom_marker)
    (coordinator_custom_marker_active ?coordinator_agent - coordinator_agent)
    (coordinator_marker_stage_two ?coordinator_agent - coordinator_agent)
    (coordinator_marker_stage_three ?coordinator_agent - coordinator_agent)
    (tactic_marker_available ?tactic_marker - tactic_marker)
    (coordinator_assigned_tactic_marker ?coordinator_agent - coordinator_agent ?tactic_marker - tactic_marker)
    (buff_token_available ?buff_token - buff_token)
    (coordinator_assigned_buff_token ?coordinator_agent - coordinator_agent ?buff_token - buff_token)
    (priority_resource_available ?priority_resource - priority_resource)
    (coordinator_assigned_priority_resource ?coordinator_agent - coordinator_agent ?priority_resource - priority_resource)
    (specialist_token_available ?specialist_token - specialist_token)
    (coordinator_assigned_specialist ?coordinator_agent - coordinator_agent ?specialist_token - specialist_token)
    (timing_token_available ?timing_token - timing_token)
    (entity_assigned_timing_token ?downed_token - downed_token ?timing_token - timing_token)
    (subforce_a_ready ?subforce_member_a - subforce_member_a)
    (subforce_b_ready ?subforce_member_b - subforce_member_b)
    (coordinator_committed ?coordinator_agent - coordinator_agent)
  )
  (:action tag_downed_candidate
    :parameters (?downed_token - downed_token)
    :precondition
      (and
        (not
          (downed_tagged ?downed_token)
        )
        (not
          (member_revived ?downed_token)
        )
      )
    :effect (downed_tagged ?downed_token)
  )
  (:action assign_revive_item
    :parameters (?downed_token - downed_token ?revive_item - revive_item)
    :precondition
      (and
        (downed_tagged ?downed_token)
        (not
          (entity_item_reserved ?downed_token)
        )
        (revive_item_available ?revive_item)
      )
    :effect
      (and
        (entity_item_reserved ?downed_token)
        (entity_assigned_revive_item ?downed_token ?revive_item)
        (not
          (revive_item_available ?revive_item)
        )
      )
  )
  (:action assign_healer_slot
    :parameters (?downed_token - downed_token ?healer_resource - healer_resource)
    :precondition
      (and
        (downed_tagged ?downed_token)
        (entity_item_reserved ?downed_token)
        (healer_slot_available ?healer_resource)
      )
    :effect
      (and
        (entity_assigned_healer ?downed_token ?healer_resource)
        (not
          (healer_slot_available ?healer_resource)
        )
      )
  )
  (:action confirm_candidate_ready
    :parameters (?downed_token - downed_token ?healer_resource - healer_resource)
    :precondition
      (and
        (downed_tagged ?downed_token)
        (entity_item_reserved ?downed_token)
        (entity_assigned_healer ?downed_token ?healer_resource)
        (not
          (ready_for_revive ?downed_token)
        )
      )
    :effect (ready_for_revive ?downed_token)
  )
  (:action release_healer_slot
    :parameters (?downed_token - downed_token ?healer_resource - healer_resource)
    :precondition
      (and
        (entity_assigned_healer ?downed_token ?healer_resource)
      )
    :effect
      (and
        (healer_slot_available ?healer_resource)
        (not
          (entity_assigned_healer ?downed_token ?healer_resource)
        )
      )
  )
  (:action assign_support_assist
    :parameters (?downed_token - downed_token ?support_assist - support_assist)
    :precondition
      (and
        (ready_for_revive ?downed_token)
        (support_assist_available ?support_assist)
      )
    :effect
      (and
        (entity_assigned_support_assist ?downed_token ?support_assist)
        (not
          (support_assist_available ?support_assist)
        )
      )
  )
  (:action release_support_assist
    :parameters (?downed_token - downed_token ?support_assist - support_assist)
    :precondition
      (and
        (entity_assigned_support_assist ?downed_token ?support_assist)
      )
    :effect
      (and
        (support_assist_available ?support_assist)
        (not
          (entity_assigned_support_assist ?downed_token ?support_assist)
        )
      )
  )
  (:action coordinator_attach_priority_resource
    :parameters (?coordinator_agent - coordinator_agent ?priority_resource - priority_resource)
    :precondition
      (and
        (ready_for_revive ?coordinator_agent)
        (priority_resource_available ?priority_resource)
      )
    :effect
      (and
        (coordinator_assigned_priority_resource ?coordinator_agent ?priority_resource)
        (not
          (priority_resource_available ?priority_resource)
        )
      )
  )
  (:action coordinator_detach_priority_resource
    :parameters (?coordinator_agent - coordinator_agent ?priority_resource - priority_resource)
    :precondition
      (and
        (coordinator_assigned_priority_resource ?coordinator_agent ?priority_resource)
      )
    :effect
      (and
        (priority_resource_available ?priority_resource)
        (not
          (coordinator_assigned_priority_resource ?coordinator_agent ?priority_resource)
        )
      )
  )
  (:action coordinator_attach_specialist
    :parameters (?coordinator_agent - coordinator_agent ?specialist_token - specialist_token)
    :precondition
      (and
        (ready_for_revive ?coordinator_agent)
        (specialist_token_available ?specialist_token)
      )
    :effect
      (and
        (coordinator_assigned_specialist ?coordinator_agent ?specialist_token)
        (not
          (specialist_token_available ?specialist_token)
        )
      )
  )
  (:action coordinator_detach_specialist
    :parameters (?coordinator_agent - coordinator_agent ?specialist_token - specialist_token)
    :precondition
      (and
        (coordinator_assigned_specialist ?coordinator_agent ?specialist_token)
      )
    :effect
      (and
        (specialist_token_available ?specialist_token)
        (not
          (coordinator_assigned_specialist ?coordinator_agent ?specialist_token)
        )
      )
  )
  (:action prepare_anchor_a
    :parameters (?subforce_member_a - subforce_member_a ?anchor_point_a - anchor_point_a ?healer_resource - healer_resource)
    :precondition
      (and
        (ready_for_revive ?subforce_member_a)
        (entity_assigned_healer ?subforce_member_a ?healer_resource)
        (member_a_anchor_assigned ?subforce_member_a ?anchor_point_a)
        (not
          (anchor_a_ready ?anchor_point_a)
        )
        (not
          (anchor_a_prepared ?anchor_point_a)
        )
      )
    :effect (anchor_a_ready ?anchor_point_a)
  )
  (:action confirm_anchor_a_and_mark_member
    :parameters (?subforce_member_a - subforce_member_a ?anchor_point_a - anchor_point_a ?support_assist - support_assist)
    :precondition
      (and
        (ready_for_revive ?subforce_member_a)
        (entity_assigned_support_assist ?subforce_member_a ?support_assist)
        (member_a_anchor_assigned ?subforce_member_a ?anchor_point_a)
        (anchor_a_ready ?anchor_point_a)
        (not
          (subforce_a_ready ?subforce_member_a)
        )
      )
    :effect
      (and
        (subforce_a_ready ?subforce_member_a)
        (member_a_committed ?subforce_member_a)
      )
  )
  (:action use_utility_item_on_member_a
    :parameters (?subforce_member_a - subforce_member_a ?anchor_point_a - anchor_point_a ?utility_item - utility_item)
    :precondition
      (and
        (ready_for_revive ?subforce_member_a)
        (member_a_anchor_assigned ?subforce_member_a ?anchor_point_a)
        (utility_item_available ?utility_item)
        (not
          (subforce_a_ready ?subforce_member_a)
        )
      )
    :effect
      (and
        (anchor_a_prepared ?anchor_point_a)
        (subforce_a_ready ?subforce_member_a)
        (member_a_assigned_utility ?subforce_member_a ?utility_item)
        (not
          (utility_item_available ?utility_item)
        )
      )
  )
  (:action finalize_anchor_a
    :parameters (?subforce_member_a - subforce_member_a ?anchor_point_a - anchor_point_a ?healer_resource - healer_resource ?utility_item - utility_item)
    :precondition
      (and
        (ready_for_revive ?subforce_member_a)
        (entity_assigned_healer ?subforce_member_a ?healer_resource)
        (member_a_anchor_assigned ?subforce_member_a ?anchor_point_a)
        (anchor_a_prepared ?anchor_point_a)
        (member_a_assigned_utility ?subforce_member_a ?utility_item)
        (not
          (member_a_committed ?subforce_member_a)
        )
      )
    :effect
      (and
        (anchor_a_ready ?anchor_point_a)
        (member_a_committed ?subforce_member_a)
        (utility_item_available ?utility_item)
        (not
          (member_a_assigned_utility ?subforce_member_a ?utility_item)
        )
      )
  )
  (:action prepare_anchor_b
    :parameters (?subforce_member_b - subforce_member_b ?anchor_point_b - anchor_point_b ?healer_resource - healer_resource)
    :precondition
      (and
        (ready_for_revive ?subforce_member_b)
        (entity_assigned_healer ?subforce_member_b ?healer_resource)
        (member_b_anchor_assigned ?subforce_member_b ?anchor_point_b)
        (not
          (anchor_b_ready ?anchor_point_b)
        )
        (not
          (anchor_b_prepared ?anchor_point_b)
        )
      )
    :effect (anchor_b_ready ?anchor_point_b)
  )
  (:action confirm_anchor_b_and_mark_member
    :parameters (?subforce_member_b - subforce_member_b ?anchor_point_b - anchor_point_b ?support_assist - support_assist)
    :precondition
      (and
        (ready_for_revive ?subforce_member_b)
        (entity_assigned_support_assist ?subforce_member_b ?support_assist)
        (member_b_anchor_assigned ?subforce_member_b ?anchor_point_b)
        (anchor_b_ready ?anchor_point_b)
        (not
          (subforce_b_ready ?subforce_member_b)
        )
      )
    :effect
      (and
        (subforce_b_ready ?subforce_member_b)
        (member_b_committed ?subforce_member_b)
      )
  )
  (:action use_utility_item_on_member_b
    :parameters (?subforce_member_b - subforce_member_b ?anchor_point_b - anchor_point_b ?utility_item - utility_item)
    :precondition
      (and
        (ready_for_revive ?subforce_member_b)
        (member_b_anchor_assigned ?subforce_member_b ?anchor_point_b)
        (utility_item_available ?utility_item)
        (not
          (subforce_b_ready ?subforce_member_b)
        )
      )
    :effect
      (and
        (anchor_b_prepared ?anchor_point_b)
        (subforce_b_ready ?subforce_member_b)
        (member_b_assigned_utility ?subforce_member_b ?utility_item)
        (not
          (utility_item_available ?utility_item)
        )
      )
  )
  (:action finalize_anchor_b
    :parameters (?subforce_member_b - subforce_member_b ?anchor_point_b - anchor_point_b ?healer_resource - healer_resource ?utility_item - utility_item)
    :precondition
      (and
        (ready_for_revive ?subforce_member_b)
        (entity_assigned_healer ?subforce_member_b ?healer_resource)
        (member_b_anchor_assigned ?subforce_member_b ?anchor_point_b)
        (anchor_b_prepared ?anchor_point_b)
        (member_b_assigned_utility ?subforce_member_b ?utility_item)
        (not
          (member_b_committed ?subforce_member_b)
        )
      )
    :effect
      (and
        (anchor_b_ready ?anchor_point_b)
        (member_b_committed ?subforce_member_b)
        (utility_item_available ?utility_item)
        (not
          (member_b_assigned_utility ?subforce_member_b ?utility_item)
        )
      )
  )
  (:action reserve_revival_site
    :parameters (?subforce_member_a - subforce_member_a ?subforce_member_b - subforce_member_b ?anchor_point_a - anchor_point_a ?anchor_point_b - anchor_point_b ?revival_site - revival_site)
    :precondition
      (and
        (subforce_a_ready ?subforce_member_a)
        (subforce_b_ready ?subforce_member_b)
        (member_a_anchor_assigned ?subforce_member_a ?anchor_point_a)
        (member_b_anchor_assigned ?subforce_member_b ?anchor_point_b)
        (anchor_a_ready ?anchor_point_a)
        (anchor_b_ready ?anchor_point_b)
        (member_a_committed ?subforce_member_a)
        (member_b_committed ?subforce_member_b)
        (revival_site_available ?revival_site)
      )
    :effect
      (and
        (revival_site_reserved ?revival_site)
        (revival_site_anchor_a_attached ?revival_site ?anchor_point_a)
        (revival_site_anchor_b_attached ?revival_site ?anchor_point_b)
        (not
          (revival_site_available ?revival_site)
        )
      )
  )
  (:action reserve_revival_site_stage_a
    :parameters (?subforce_member_a - subforce_member_a ?subforce_member_b - subforce_member_b ?anchor_point_a - anchor_point_a ?anchor_point_b - anchor_point_b ?revival_site - revival_site)
    :precondition
      (and
        (subforce_a_ready ?subforce_member_a)
        (subforce_b_ready ?subforce_member_b)
        (member_a_anchor_assigned ?subforce_member_a ?anchor_point_a)
        (member_b_anchor_assigned ?subforce_member_b ?anchor_point_b)
        (anchor_a_prepared ?anchor_point_a)
        (anchor_b_ready ?anchor_point_b)
        (not
          (member_a_committed ?subforce_member_a)
        )
        (member_b_committed ?subforce_member_b)
        (revival_site_available ?revival_site)
      )
    :effect
      (and
        (revival_site_reserved ?revival_site)
        (revival_site_anchor_a_attached ?revival_site ?anchor_point_a)
        (revival_site_anchor_b_attached ?revival_site ?anchor_point_b)
        (revival_site_stage_a ?revival_site)
        (not
          (revival_site_available ?revival_site)
        )
      )
  )
  (:action reserve_revival_site_stage_b
    :parameters (?subforce_member_a - subforce_member_a ?subforce_member_b - subforce_member_b ?anchor_point_a - anchor_point_a ?anchor_point_b - anchor_point_b ?revival_site - revival_site)
    :precondition
      (and
        (subforce_a_ready ?subforce_member_a)
        (subforce_b_ready ?subforce_member_b)
        (member_a_anchor_assigned ?subforce_member_a ?anchor_point_a)
        (member_b_anchor_assigned ?subforce_member_b ?anchor_point_b)
        (anchor_a_ready ?anchor_point_a)
        (anchor_b_prepared ?anchor_point_b)
        (member_a_committed ?subforce_member_a)
        (not
          (member_b_committed ?subforce_member_b)
        )
        (revival_site_available ?revival_site)
      )
    :effect
      (and
        (revival_site_reserved ?revival_site)
        (revival_site_anchor_a_attached ?revival_site ?anchor_point_a)
        (revival_site_anchor_b_attached ?revival_site ?anchor_point_b)
        (revival_site_stage_b ?revival_site)
        (not
          (revival_site_available ?revival_site)
        )
      )
  )
  (:action reserve_revival_site_full
    :parameters (?subforce_member_a - subforce_member_a ?subforce_member_b - subforce_member_b ?anchor_point_a - anchor_point_a ?anchor_point_b - anchor_point_b ?revival_site - revival_site)
    :precondition
      (and
        (subforce_a_ready ?subforce_member_a)
        (subforce_b_ready ?subforce_member_b)
        (member_a_anchor_assigned ?subforce_member_a ?anchor_point_a)
        (member_b_anchor_assigned ?subforce_member_b ?anchor_point_b)
        (anchor_a_prepared ?anchor_point_a)
        (anchor_b_prepared ?anchor_point_b)
        (not
          (member_a_committed ?subforce_member_a)
        )
        (not
          (member_b_committed ?subforce_member_b)
        )
        (revival_site_available ?revival_site)
      )
    :effect
      (and
        (revival_site_reserved ?revival_site)
        (revival_site_anchor_a_attached ?revival_site ?anchor_point_a)
        (revival_site_anchor_b_attached ?revival_site ?anchor_point_b)
        (revival_site_stage_a ?revival_site)
        (revival_site_stage_b ?revival_site)
        (not
          (revival_site_available ?revival_site)
        )
      )
  )
  (:action lock_revival_site_execution
    :parameters (?revival_site - revival_site ?subforce_member_a - subforce_member_a ?healer_resource - healer_resource)
    :precondition
      (and
        (revival_site_reserved ?revival_site)
        (subforce_a_ready ?subforce_member_a)
        (entity_assigned_healer ?subforce_member_a ?healer_resource)
        (not
          (revival_site_execution_committed ?revival_site)
        )
      )
    :effect (revival_site_execution_committed ?revival_site)
  )
  (:action stage_consumable_pack_on_site
    :parameters (?coordinator_agent - coordinator_agent ?consumable_pack - consumable_pack ?revival_site - revival_site)
    :precondition
      (and
        (ready_for_revive ?coordinator_agent)
        (coordinator_attached_site ?coordinator_agent ?revival_site)
        (coordinator_has_pack ?coordinator_agent ?consumable_pack)
        (consumable_pack_available ?consumable_pack)
        (revival_site_reserved ?revival_site)
        (revival_site_execution_committed ?revival_site)
        (not
          (consumable_pack_staged ?consumable_pack)
        )
      )
    :effect
      (and
        (consumable_pack_staged ?consumable_pack)
        (consumable_pack_attached_to_site ?consumable_pack ?revival_site)
        (not
          (consumable_pack_available ?consumable_pack)
        )
      )
  )
  (:action coordinator_prepare_with_pack
    :parameters (?coordinator_agent - coordinator_agent ?consumable_pack - consumable_pack ?revival_site - revival_site ?healer_resource - healer_resource)
    :precondition
      (and
        (ready_for_revive ?coordinator_agent)
        (coordinator_has_pack ?coordinator_agent ?consumable_pack)
        (consumable_pack_staged ?consumable_pack)
        (consumable_pack_attached_to_site ?consumable_pack ?revival_site)
        (entity_assigned_healer ?coordinator_agent ?healer_resource)
        (not
          (revival_site_stage_a ?revival_site)
        )
        (not
          (coordinator_pack_prepared ?coordinator_agent)
        )
      )
    :effect (coordinator_pack_prepared ?coordinator_agent)
  )
  (:action assign_tactic_marker_to_coordinator
    :parameters (?coordinator_agent - coordinator_agent ?tactic_marker - tactic_marker)
    :precondition
      (and
        (ready_for_revive ?coordinator_agent)
        (tactic_marker_available ?tactic_marker)
        (not
          (coordinator_has_tactic_marker ?coordinator_agent)
        )
      )
    :effect
      (and
        (coordinator_has_tactic_marker ?coordinator_agent)
        (coordinator_assigned_tactic_marker ?coordinator_agent ?tactic_marker)
        (not
          (tactic_marker_available ?tactic_marker)
        )
      )
  )
  (:action coordinator_stage_marker_with_pack
    :parameters (?coordinator_agent - coordinator_agent ?consumable_pack - consumable_pack ?revival_site - revival_site ?healer_resource - healer_resource ?tactic_marker - tactic_marker)
    :precondition
      (and
        (ready_for_revive ?coordinator_agent)
