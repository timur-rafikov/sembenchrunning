  (:action register_animal
    :parameters (?livestock_individual - subject)
    :precondition
      (and
        (not
          (subject_registered ?livestock_individual)
        )
        (not
          (shelter_protection_assigned ?livestock_individual)
        )
      )
    :effect (subject_registered ?livestock_individual)
  )
  (:action stage_animal_in_holding_slot
    :parameters (?livestock_individual - subject ?holding_slot - holding_slot)
    :precondition
      (and
        (subject_registered ?livestock_individual)
        (not
          (subject_staged_for_processing ?livestock_individual)
        )
        (holding_slot_available ?holding_slot)
      )
    :effect
      (and
        (subject_staged_for_processing ?livestock_individual)
        (subject_allocated_to_holding_slot ?livestock_individual ?holding_slot)
        (not
          (holding_slot_available ?holding_slot)
        )
      )
  )
  (:action assign_handler_to_animal
    :parameters (?livestock_individual - subject ?handler_personnel - handler_personnel)
    :precondition
      (and
        (subject_registered ?livestock_individual)
        (subject_staged_for_processing ?livestock_individual)
        (handler_available ?handler_personnel)
      )
    :effect
      (and
        (subject_assigned_to_handler ?livestock_individual ?handler_personnel)
        (not
          (handler_available ?handler_personnel)
        )
      )
  )
  (:action finalize_animal_preparation
    :parameters (?livestock_individual - subject ?handler_personnel - handler_personnel)
    :precondition
      (and
        (subject_registered ?livestock_individual)
        (subject_staged_for_processing ?livestock_individual)
        (subject_assigned_to_handler ?livestock_individual ?handler_personnel)
        (not
          (subject_ready_for_admission ?livestock_individual)
        )
      )
    :effect (subject_ready_for_admission ?livestock_individual)
  )
  (:action release_handler_from_animal
    :parameters (?livestock_individual - subject ?handler_personnel - handler_personnel)

        (subject_assigned_to_handler ?livestock_individual ?handler_personnel)
      )
    :effect
      (and
        (handler_available ?handler_personnel)
        (not
          (subject_assigned_to_handler ?livestock_individual ?handler_personnel)
        )
      )
  )
