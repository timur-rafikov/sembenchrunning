  (:action select_booking_option
    :parameters (?booking_entity - booking_entity)
    :precondition
      (and
        (not
          (booking_option_selected ?booking_entity)
        )
        (not
          (record_aligned ?booking_entity)
        )
      )
    :effect (booking_option_selected ?booking_entity)
  )
  (:action assign_channel_identifier_to_booking
    :parameters (?booking_entity - booking_entity ?channel_identifier - channel_identifier)
    :precondition
      (and
        (booking_option_selected ?booking_entity)
        (not
          (channel_identifier_assigned ?booking_entity)
        )
        (channel_identifier_available ?channel_identifier)
      )
    :effect
      (and
        (channel_identifier_assigned ?booking_entity)
        (booking_has_channel_identifier ?booking_entity ?channel_identifier)
        (not
          (channel_identifier_available ?channel_identifier)
        )
      )
  )
  (:action request_supplier_confirmation
    :parameters (?booking_entity - booking_entity ?supplier_confirmation_token - supplier_token)

      (and
        (booking_option_selected ?booking_entity)
        (channel_identifier_assigned ?booking_entity)
        (supplier_token_available ?supplier_confirmation_token)
      )
    :effect
      (and
        (booking_holds_supplier_token ?booking_entity ?supplier_confirmation_token)
        (not
          (supplier_token_available ?supplier_confirmation_token)
        )
      )
  )

    :parameters (?booking_entity - booking_entity ?supplier_confirmation_token - supplier_token)

        (booking_option_selected ?booking_entity)

        (booking_holds_supplier_token ?booking_entity ?supplier_confirmation_token)

          (booking_committed ?booking_entity)
        )
      )
    :effect (booking_committed ?booking_entity)
  )
  (:action release_supplier_confirmation_token
    :parameters (?booking_entity - booking_entity ?supplier_confirmation_token - supplier_token)

      (and
        (booking_holds_supplier_token ?booking_entity ?supplier_confirmation_token)
      )
    :effect
      (and
        (supplier_token_available ?supplier_confirmation_token)

          (booking_holds_supplier_token ?booking_entity ?supplier_confirmation_token)
        )
      )
  )
