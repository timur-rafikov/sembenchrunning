--- domain.pddl (as passed to VAL) ---
(define (domain duration_target_realignment)
  (:requirements :strips :typing :negative-preconditions)
  (:types execution_domain - object instrument_domain - object trade_leg_domain - object case_category - object realignment_case - case_category execution_desk - execution_domain analytics_signal - execution_domain approval_unit - execution_domain broker_lp - execution_domain allocation_template - execution_domain post_trade_assignment_record - execution_domain execution_parameter - execution_domain counterparty_constraint - execution_domain instrument_candidate - instrument_domain settlement_account - instrument_domain compliance_rule - instrument_domain sell_leg - trade_leg_domain buy_leg - trade_leg_domain order_ticket - trade_leg_domain portfolio_role_group - realignment_case portfolio_segment_group - realignment_case source_portfolio - portfolio_role_group target_portfolio - portfolio_role_group rebalancing_plan - portfolio_segment_group)
  (:predicates
    (initialized_for_entity ?realignment_case - realignment_case)
    (requires_approval ?realignment_case - realignment_case)
    (execution_desk_assigned ?realignment_case - realignment_case)
    (completed_for_entity ?realignment_case - realignment_case)
    (execution_confirmed_for_entity ?realignment_case - realignment_case)
    (post_trade_assignment_started ?realignment_case - realignment_case)
    (execution_desk_available ?execution_desk - execution_desk)
    (assigned_to_execution_desk ?realignment_case - realignment_case ?execution_desk - execution_desk)
    (analytics_signal_available ?analytics_signal - analytics_signal)
    (linked_to_signal ?realignment_case - realignment_case ?analytics_signal - analytics_signal)
    (approval_unit_available ?approval_unit - approval_unit)
    (linked_to_approval_unit ?realignment_case - realignment_case ?approval_unit - approval_unit)
    (instrument_candidate_available ?instrument_candidate - instrument_candidate)
    (candidate_reserved_for_source ?source_portfolio - source_portfolio ?instrument_candidate - instrument_candidate)
    (candidate_reserved_for_target ?target_portfolio - target_portfolio ?instrument_candidate - instrument_candidate)
    (source_portfolio_has_sell_leg ?source_portfolio - source_portfolio ?sell_leg - sell_leg)
    (sell_leg_selected ?sell_leg - sell_leg)
    (sell_leg_candidate_assigned ?sell_leg - sell_leg)
    (source_portfolio_ready ?source_portfolio - source_portfolio)
    (target_portfolio_has_buy_leg ?target_portfolio - target_portfolio ?buy_leg - buy_leg)
    (buy_leg_selected ?buy_leg - buy_leg)
    (buy_leg_candidate_assigned ?buy_leg - buy_leg)
    (target_portfolio_ready ?target_portfolio - target_portfolio)
    (order_ticket_available ?order_ticket - order_ticket)
    (order_ticket_assembled ?order_ticket - order_ticket)
    (ticket_has_sell_leg ?order_ticket - order_ticket ?sell_leg - sell_leg)
    (ticket_has_buy_leg ?order_ticket - order_ticket ?buy_leg - buy_leg)
    (ticket_sell_leg_candidate_assigned ?order_ticket - order_ticket)
    (ticket_buy_leg_candidate_assigned ?order_ticket - order_ticket)
    (order_ticket_settlement_ready ?order_ticket - order_ticket)
    (plan_links_to_source_portfolio ?rebalancing_plan - rebalancing_plan ?source_portfolio - source_portfolio)
    (plan_links_to_target_portfolio ?rebalancing_plan - rebalancing_plan ?target_portfolio - target_portfolio)
    (plan_links_to_order_ticket ?rebalancing_plan - rebalancing_plan ?order_ticket - order_ticket)
    (settlement_account_available ?settlement_account - settlement_account)
    (plan_links_to_settlement_account ?rebalancing_plan - rebalancing_plan ?settlement_account - settlement_account)
    (settlement_account_assigned ?settlement_account - settlement_account)
    (settlement_account_assigned_to_ticket ?settlement_account - settlement_account ?order_ticket - order_ticket)
    (plan_prepared_for_broker_assignment ?rebalancing_plan - rebalancing_plan)
    (plan_broker_assignment_done ?rebalancing_plan - rebalancing_plan)
    (plan_pre_trade_ready ?rebalancing_plan - rebalancing_plan)
    (broker_assigned_to_plan ?rebalancing_plan - rebalancing_plan)
    (broker_assignment_verified ?rebalancing_plan - rebalancing_plan)
    (allocation_template_requested ?rebalancing_plan - rebalancing_plan)
    (plan_batched ?rebalancing_plan - rebalancing_plan)
    (compliance_rule_available ?compliance_rule - compliance_rule)
    (plan_linked_to_compliance_rule ?rebalancing_plan - rebalancing_plan ?compliance_rule - compliance_rule)
    (compliance_check_requested ?rebalancing_plan - rebalancing_plan)
    (compliance_check_in_progress ?rebalancing_plan - rebalancing_plan)
    (compliance_approved ?rebalancing_plan - rebalancing_plan)
    (broker_available ?broker_lp - broker_lp)
    (plan_linked_to_broker ?rebalancing_plan - rebalancing_plan ?broker_lp - broker_lp)
    (allocation_template_available ?allocation_template - allocation_template)
    (plan_has_allocation_template ?rebalancing_plan - rebalancing_plan ?allocation_template - allocation_template)
    (execution_parameter_available ?execution_parameter - execution_parameter)
    (plan_has_execution_parameter ?rebalancing_plan - rebalancing_plan ?execution_parameter - execution_parameter)
    (counterparty_constraint_available ?counterparty_constraint - counterparty_constraint)
    (plan_has_counterparty_constraint ?rebalancing_plan - rebalancing_plan ?counterparty_constraint - counterparty_constraint)
    (post_trade_assignment_record_available ?post_trade_assignment_record - post_trade_assignment_record)
    (linked_to_post_trade_record ?realignment_case - realignment_case ?post_trade_assignment_record - post_trade_assignment_record)
    (source_portfolio_prepared ?source_portfolio - source_portfolio)
    (target_portfolio_prepared ?target_portfolio - target_portfolio)
    (plan_finalized_for_execution ?rebalancing_plan - rebalancing_plan)
  )
  (:action register_realignment_case
    :parameters (?realignment_case - realignment_case)
    :precondition
      (and
        (not
          (initialized_for_entity ?realignment_case)
        )
        (not
          (completed_for_entity ?realignment_case)
        )
      )
    :effect (initialized_for_entity ?realignment_case)
  )
  (:action assign_execution_desk_to_case
    :parameters (?realignment_case - realignment_case ?execution_desk - execution_desk)
    :precondition
      (and
        (initialized_for_entity ?realignment_case)
        (not
          (execution_desk_assigned ?realignment_case)
        )
        (execution_desk_available ?execution_desk)
      )
    :effect
      (and
        (execution_desk_assigned ?realignment_case)
        (assigned_to_execution_desk ?realignment_case ?execution_desk)
        (not
          (execution_desk_available ?execution_desk)
        )
      )
  )
  (:action attach_analytics_signal_to_case
    :parameters (?realignment_case - realignment_case ?analytics_signal - analytics_signal)
    :precondition
      (and
        (initialized_for_entity ?realignment_case)
        (execution_desk_assigned ?realignment_case)
        (analytics_signal_available ?analytics_signal)
      )
    :effect
      (and
        (linked_to_signal ?realignment_case ?analytics_signal)
        (not
          (analytics_signal_available ?analytics_signal)
        )
      )
  )
  (:action mark_case_approval_required
    :parameters (?realignment_case - realignment_case ?analytics_signal - analytics_signal)
    :precondition
      (and
        (initialized_for_entity ?realignment_case)
        (execution_desk_assigned ?realignment_case)
        (linked_to_signal ?realignment_case ?analytics_signal)
        (not
          (requires_approval ?realignment_case)
        )
      )
    :effect (requires_approval ?realignment_case)
  )
  (:action release_analytics_signal_from_case
    :parameters (?realignment_case - realignment_case ?analytics_signal - analytics_signal)
    :precondition
      (and
        (linked_to_signal ?realignment_case ?analytics_signal)
      )
    :effect
      (and
        (analytics_signal_available ?analytics_signal)
        (not
          (linked_to_signal ?realignment_case ?analytics_signal)
        )
      )
  )
  (:action assign_approval_unit_to_case
    :parameters (?realignment_case - realignment_case ?approval_unit - approval_unit)
    :precondition
      (and
        (requires_approval ?realignment_case)
        (approval_unit_available ?approval_unit)
      )
    :effect
      (and
        (linked_to_approval_unit ?realignment_case ?approval_unit)
        (not
          (approval_unit_available ?approval_unit)
        )
      )
  )
  (:action release_approval_unit_from_case
    :parameters (?realignment_case - realignment_case ?approval_unit - approval_unit)
    :precondition
      (and
        (linked_to_approval_unit ?realignment_case ?approval_unit)
      )
    :effect
      (and
        (approval_unit_available ?approval_unit)
        (not
          (linked_to_approval_unit ?realignment_case ?approval_unit)
        )
      )
  )
  (:action attach_execution_parameter_to_plan
    :parameters (?rebalancing_plan - rebalancing_plan ?execution_parameter - execution_parameter)
    :precondition
      (and
        (requires_approval ?rebalancing_plan)
        (execution_parameter_available ?execution_parameter)
      )
    :effect
      (and
        (plan_has_execution_parameter ?rebalancing_plan ?execution_parameter)
        (not
          (execution_parameter_available ?execution_parameter)
        )
      )
  )
  (:action detach_execution_parameter_from_plan
    :parameters (?rebalancing_plan - rebalancing_plan ?execution_parameter - execution_parameter)
    :precondition
      (and
        (plan_has_execution_parameter ?rebalancing_plan ?execution_parameter)
      )
    :effect
      (and
        (execution_parameter_available ?execution_parameter)
        (not
          (plan_has_execution_parameter ?rebalancing_plan ?execution_parameter)
        )
      )
  )
  (:action attach_counterparty_constraint_to_plan
    :parameters (?rebalancing_plan - rebalancing_plan ?counterparty_constraint - counterparty_constraint)
    :precondition
      (and
        (requires_approval ?rebalancing_plan)
        (counterparty_constraint_available ?counterparty_constraint)
      )
    :effect
      (and
        (plan_has_counterparty_constraint ?rebalancing_plan ?counterparty_constraint)
        (not
          (counterparty_constraint_available ?counterparty_constraint)
        )
      )
  )
  (:action detach_counterparty_constraint_from_plan
    :parameters (?rebalancing_plan - rebalancing_plan ?counterparty_constraint - counterparty_constraint)
    :precondition
      (and
        (plan_has_counterparty_constraint ?rebalancing_plan ?counterparty_constraint)
      )
    :effect
      (and
        (counterparty_constraint_available ?counterparty_constraint)
        (not
          (plan_has_counterparty_constraint ?rebalancing_plan ?counterparty_constraint)
        )
      )
  )
  (:action select_sell_leg_for_source_portfolio
    :parameters (?source_portfolio - source_portfolio ?sell_leg - sell_leg ?analytics_signal - analytics_signal)
    :precondition
      (and
        (requires_approval ?source_portfolio)
        (linked_to_signal ?source_portfolio ?analytics_signal)
        (source_portfolio_has_sell_leg ?source_portfolio ?sell_leg)
        (not
          (sell_leg_selected ?sell_leg)
        )
        (not
          (sell_leg_candidate_assigned ?sell_leg)
        )
      )
    :effect (sell_leg_selected ?sell_leg)
  )
  (:action approve_and_finalize_source_leg
    :parameters (?source_portfolio - source_portfolio ?sell_leg - sell_leg ?approval_unit - approval_unit)
    :precondition
      (and
        (requires_approval ?source_portfolio)
        (linked_to_approval_unit ?source_portfolio ?approval_unit)
        (source_portfolio_has_sell_leg ?source_portfolio ?sell_leg)
        (sell_leg_selected ?sell_leg)
        (not
          (source_portfolio_prepared ?source_portfolio)
        )
      )
    :effect
      (and
        (source_portfolio_prepared ?source_portfolio)
        (source_portfolio_ready ?source_portfolio)
      )
  )
  (:action assign_candidate_to_sell_leg
    :parameters (?source_portfolio - source_portfolio ?sell_leg - sell_leg ?instrument_candidate - instrument_candidate)
    :precondition
      (and
        (requires_approval ?source_portfolio)
        (source_portfolio_has_sell_leg ?source_portfolio ?sell_leg)
        (instrument_candidate_available ?instrument_candidate)
        (not
          (source_portfolio_prepared ?source_portfolio)
        )
      )
    :effect
      (and
        (sell_leg_candidate_assigned ?sell_leg)
        (source_portfolio_prepared ?source_portfolio)
        (candidate_reserved_for_source ?source_portfolio ?instrument_candidate)
        (not
          (instrument_candidate_available ?instrument_candidate)
        )
      )
  )
  (:action finalize_sell_leg_candidate
    :parameters (?source_portfolio - source_portfolio ?sell_leg - sell_leg ?analytics_signal - analytics_signal ?instrument_candidate - instrument_candidate)
    :precondition
      (and
        (requires_approval ?source_portfolio)
        (linked_to_signal ?source_portfolio ?analytics_signal)
        (source_portfolio_has_sell_leg ?source_portfolio ?sell_leg)
        (sell_leg_candidate_assigned ?sell_leg)
        (candidate_reserved_for_source ?source_portfolio ?instrument_candidate)
        (not
          (source_portfolio_ready ?source_portfolio)
        )
      )
    :effect
      (and
        (sell_leg_selected ?sell_leg)
        (source_portfolio_ready ?source_portfolio)
        (instrument_candidate_available ?instrument_candidate)
        (not
          (candidate_reserved_for_source ?source_portfolio ?instrument_candidate)
        )
      )
  )
  (:action select_buy_leg_for_target_portfolio
    :parameters (?target_portfolio - target_portfolio ?buy_leg - buy_leg ?analytics_signal - analytics_signal)
    :precondition
      (and
        (requires_approval ?target_portfolio)
        (linked_to_signal ?target_portfolio ?analytics_signal)
        (target_portfolio_has_buy_leg ?target_portfolio ?buy_leg)
        (not
          (buy_leg_selected ?buy_leg)
        )
        (not
          (buy_leg_candidate_assigned ?buy_leg)
        )
      )
    :effect (buy_leg_selected ?buy_leg)
  )
  (:action approve_and_finalize_target_leg
    :parameters (?target_portfolio - target_portfolio ?buy_leg - buy_leg ?approval_unit - approval_unit)
    :precondition
      (and
        (requires_approval ?target_portfolio)
        (linked_to_approval_unit ?target_portfolio ?approval_unit)
        (target_portfolio_has_buy_leg ?target_portfolio ?buy_leg)
        (buy_leg_selected ?buy_leg)
        (not
          (target_portfolio_prepared ?target_portfolio)
        )
      )
    :effect
      (and
        (target_portfolio_prepared ?target_portfolio)
        (target_portfolio_ready ?target_portfolio)
      )
  )
  (:action assign_candidate_to_buy_leg
    :parameters (?target_portfolio - target_portfolio ?buy_leg - buy_leg ?instrument_candidate - instrument_candidate)
    :precondition
      (and
        (requires_approval ?target_portfolio)
        (target_portfolio_has_buy_leg ?target_portfolio ?buy_leg)
        (instrument_candidate_available ?instrument_candidate)
        (not
          (target_portfolio_prepared ?target_portfolio)
        )
      )
    :effect
      (and
        (buy_leg_candidate_assigned ?buy_leg)
        (target_portfolio_prepared ?target_portfolio)
        (candidate_reserved_for_target ?target_portfolio ?instrument_candidate)
        (not
          (instrument_candidate_available ?instrument_candidate)
        )
      )
  )
  (:action finalize_buy_leg_candidate
    :parameters (?target_portfolio - target_portfolio ?buy_leg - buy_leg ?analytics_signal - analytics_signal ?instrument_candidate - instrument_candidate)
    :precondition
      (and
        (requires_approval ?target_portfolio)
        (linked_to_signal ?target_portfolio ?analytics_signal)
        (target_portfolio_has_buy_leg ?target_portfolio ?buy_leg)
        (buy_leg_candidate_assigned ?buy_leg)
        (candidate_reserved_for_target ?target_portfolio ?instrument_candidate)
        (not
          (target_portfolio_ready ?target_portfolio)
        )
      )
    :effect
      (and
        (buy_leg_selected ?buy_leg)
        (target_portfolio_ready ?target_portfolio)
        (instrument_candidate_available ?instrument_candidate)
        (not
          (candidate_reserved_for_target ?target_portfolio ?instrument_candidate)
        )
      )
  )
  (:action assemble_order_ticket
    :parameters (?source_portfolio - source_portfolio ?target_portfolio - target_portfolio ?sell_leg - sell_leg ?buy_leg - buy_leg ?order_ticket - order_ticket)
    :precondition
      (and
        (source_portfolio_prepared ?source_portfolio)
        (target_portfolio_prepared ?target_portfolio)
        (source_portfolio_has_sell_leg ?source_portfolio ?sell_leg)
        (target_portfolio_has_buy_leg ?target_portfolio ?buy_leg)
        (sell_leg_selected ?sell_leg)
        (buy_leg_selected ?buy_leg)
        (source_portfolio_ready ?source_portfolio)
        (target_portfolio_ready ?target_portfolio)
        (order_ticket_available ?order_ticket)
      )
    :effect
      (and
        (order_ticket_assembled ?order_ticket)
        (ticket_has_sell_leg ?order_ticket ?sell_leg)
        (ticket_has_buy_leg ?order_ticket ?buy_leg)
        (not
          (order_ticket_available ?order_ticket)
        )
      )
  )
  (:action assemble_order_ticket_with_sell_candidate_assigned
    :parameters (?source_portfolio - source_portfolio ?target_portfolio - target_portfolio ?sell_leg - sell_leg ?buy_leg - buy_leg ?order_ticket - order_ticket)
    :precondition
      (and
        (source_portfolio_prepared ?source_portfolio)
        (target_portfolio_prepared ?target_portfolio)
        (source_portfolio_has_sell_leg ?source_portfolio ?sell_leg)
        (target_portfolio_has_buy_leg ?target_portfolio ?buy_leg)
        (sell_leg_candidate_assigned ?sell_leg)
        (buy_leg_selected ?buy_leg)
        (not
          (source_portfolio_ready ?source_portfolio)
        )
        (target_portfolio_ready ?target_portfolio)
        (order_ticket_available ?order_ticket)
      )
    :effect
      (and
        (order_ticket_assembled ?order_ticket)
        (ticket_has_sell_leg ?order_ticket ?sell_leg)
        (ticket_has_buy_leg ?order_ticket ?buy_leg)
        (ticket_sell_leg_candidate_assigned ?order_ticket)
        (not
          (order_ticket_available ?order_ticket)
        )
      )
  )
  (:action assemble_order_ticket_with_buy_candidate_assigned
    :parameters (?source_portfolio - source_portfolio ?target_portfolio - target_portfolio ?sell_leg - sell_leg ?buy_leg - buy_leg ?order_ticket - order_ticket)
    :precondition
      (and
        (source_portfolio_prepared ?source_portfolio)
        (target_portfolio_prepared ?target_portfolio)
        (source_portfolio_has_sell_leg ?source_portfolio ?sell_leg)
        (target_portfolio_has_buy_leg ?target_portfolio ?buy_leg)
        (sell_leg_selected ?sell_leg)
        (buy_leg_candidate_assigned ?buy_leg)
        (source_portfolio_ready ?source_portfolio)
        (not
          (target_portfolio_ready ?target_portfolio)
        )
        (order_ticket_available ?order_ticket)
      )
    :effect
      (and
        (order_ticket_assembled ?order_ticket)
        (ticket_has_sell_leg ?order_ticket ?sell_leg)
        (ticket_has_buy_leg ?order_ticket ?buy_leg)
        (ticket_buy_leg_candidate_assigned ?order_ticket)
        (not
          (order_ticket_available ?order_ticket)
        )
      )
  )
  (:action assemble_order_ticket_with_both_candidates_assigned
    :parameters (?source_portfolio - source_portfolio ?target_portfolio - target_portfolio ?sell_leg - sell_leg ?buy_leg - buy_leg ?order_ticket - order_ticket)
    :precondition
      (and
        (source_portfolio_prepared ?source_portfolio)
        (target_portfolio_prepared ?target_portfolio)
        (source_portfolio_has_sell_leg ?source_portfolio ?sell_leg)
        (target_portfolio_has_buy_leg ?target_portfolio ?buy_leg)
        (sell_leg_candidate_assigned ?sell_leg)
        (buy_leg_candidate_assigned ?buy_leg)
        (not
          (source_portfolio_ready ?source_portfolio)
        )
        (not
          (target_portfolio_ready ?target_portfolio)
        )
        (order_ticket_available ?order_ticket)
      )
    :effect
      (and
        (order_ticket_assembled ?order_ticket)
        (ticket_has_sell_leg ?order_ticket ?sell_leg)
        (ticket_has_buy_leg ?order_ticket ?buy_leg)
        (ticket_sell_leg_candidate_assigned ?order_ticket)
        (ticket_buy_leg_candidate_assigned ?order_ticket)
        (not
          (order_ticket_available ?order_ticket)
        )
      )
  )
  (:action mark_order_ticket_settlement_ready
    :parameters (?order_ticket - order_ticket ?source_portfolio - source_portfolio ?analytics_signal - analytics_signal)
    :precondition
      (and
        (order_ticket_assembled ?order_ticket)
        (source_portfolio_prepared ?source_portfolio)
        (linked_to_signal ?source_portfolio ?analytics_signal)
        (not
          (order_ticket_settlement_ready ?order_ticket)
        )
      )
    :effect (order_ticket_settlement_ready ?order_ticket)
  )
  (:action assign_settlement_account_to_plan
    :parameters (?rebalancing_plan - rebalancing_plan ?settlement_account - settlement_account ?order_ticket - order_ticket)
    :precondition
      (and
        (requires_approval ?rebalancing_plan)
        (plan_links_to_order_ticket ?rebalancing_plan ?order_ticket)
        (plan_links_to_settlement_account ?rebalancing_plan ?settlement_account)
        (settlement_account_available ?settlement_account)
        (order_ticket_assembled ?order_ticket)
        (order_ticket_settlement_ready ?order_ticket)
        (not
          (settlement_account_assigned ?settlement_account)
        )
      )
    :effect
      (and
        (settlement_account_assigned ?settlement_account)
        (settlement_account_assigned_to_ticket ?settlement_account ?order_ticket)
        (not
          (settlement_account_available ?settlement_account)
        )
      )
  )
  (:action validate_settlement_and_prepare_plan
    :parameters (?rebalancing_plan - rebalancing_plan ?settlement_account - settlement_account ?order_ticket - order_ticket ?analytics_signal - analytics_signal)
    :precondition
      (and
        (requires_approval ?rebalancing_plan)
        (plan_links_to_settlement_account ?rebalancing_plan ?settlement_account)
        (settlement_account_assigned ?settlement_account)
        (settlement_account_assigned_to_ticket ?settlement_account ?order_ticket)
        (linked_to_signal ?rebalancing_plan ?analytics_signal)
        (not
          (ticket_sell_leg_candidate_assigned ?order_ticket)
        )
        (not
          (plan_prepared_for_broker_assignment ?rebalancing_plan)
        )
      )
    :effect (plan_prepared_for_broker_assignment ?rebalancing_plan)
  )
  (:action assign_broker_to_plan
    :parameters (?rebalancing_plan - rebalancing_plan ?broker_lp - broker_lp)
    :precondition
      (and
        (requires_approval ?rebalancing_plan)
        (broker_available ?broker_lp)
        (not
          (broker_assigned_to_plan ?rebalancing_plan)
        )
      )
    :effect
      (and
        (broker_assigned_to_plan ?rebalancing_plan)
        (plan_linked_to_broker ?rebalancing_plan ?broker_lp)
        (not
          (broker_available ?broker_lp)
        )
      )
  )
  (:action attach_broker_and_prepare_plan
    :parameters (?rebalancing_plan - rebalancing_plan ?settlement_account - settlement_account ?order_ticket - order_ticket ?analytics_signal - analytics_signal ?broker_lp - broker_lp)
    :precondition
      (and
        (requires_approval ?rebalancing_plan)
        (plan_links_to_settlement_account ?rebalancing_plan ?settlement_account)
        (settlement_account_assigned ?settlement_account)
        (settlement_account_assigned_to_ticket ?settlement_account ?order_ticket)
        (linked_to_signal ?rebalancing_plan ?analytics_signal)
        (ticket_sell_leg_candidate_assigned ?order_ticket)
        (broker_assigned_to_plan ?rebalancing_plan)
        (plan_linked_to_broker ?rebalancing_plan ?broker_lp)
        (not
          (plan_prepared_for_broker_assignment ?rebalancing_plan)
        )
      )
    :effect
      (and
        (plan_prepared_for_broker_assignment ?rebalancing_plan)
        (broker_assignment_verified ?rebalancing_plan)
      )
