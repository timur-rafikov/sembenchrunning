--- domain.pddl (as passed to VAL) ---
(define (domain multi_step_internal_workflow_execution_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types object_entity - object workflow_component - object_entity resource_category - object_entity asset_category - object_entity aggregate_root - object_entity workflow_instance - aggregate_root resource_token - workflow_component operation_type - workflow_component capability - workflow_component policy_rule - workflow_component metadata_record - workflow_component audit_token - workflow_component execution_parameter - workflow_component notification_template - workflow_component artifact - resource_category attachment_record - resource_category approval_reference - resource_category signal_channel_a - asset_category signal_channel_b - asset_category output_bundle - asset_category participant_group_a - workflow_instance participant_group_b - workflow_instance primary_participant - participant_group_a secondary_participant - participant_group_a orchestrator - participant_group_b)

  (:predicates
    (workflow_instance_open ?workflow_instance - workflow_instance)
    (context_enabled ?workflow_instance - workflow_instance)
    (resource_allocated ?workflow_instance - workflow_instance)
    (finalized ?workflow_instance - workflow_instance)
    (completion_signaled ?workflow_instance - workflow_instance)
    (audit_stage_open ?workflow_instance - workflow_instance)
    (resource_available ?resource_token - resource_token)
    (workflow_resource_binding ?workflow_instance - workflow_instance ?resource_token - resource_token)
    (operation_type_available ?operation_type - operation_type)
    (workflow_operation_binding ?workflow_instance - workflow_instance ?operation_type - operation_type)
    (capability_available ?capability - capability)
    (workflow_capability_binding ?workflow_instance - workflow_instance ?capability - capability)
    (artifact_available ?artifact - artifact)
    (primary_artifact_binding ?primary_participant - primary_participant ?artifact - artifact)
    (secondary_artifact_binding ?secondary_participant - secondary_participant ?artifact - artifact)
    (primary_channel_binding ?primary_participant - primary_participant ?signal_channel_a - signal_channel_a)
    (signal_a_emitted ?signal_channel_a - signal_channel_a)
    (signal_a_buffered ?signal_channel_a - signal_channel_a)
    (primary_participant_committed ?primary_participant - primary_participant)
    (secondary_channel_binding ?secondary_participant - secondary_participant ?signal_channel_b - signal_channel_b)
    (signal_b_emitted ?signal_channel_b - signal_channel_b)
    (signal_b_buffered ?signal_channel_b - signal_channel_b)
    (secondary_participant_committed ?secondary_participant - secondary_participant)
    (output_bundle_available ?output_bundle - output_bundle)
    (output_bundle_prepared ?output_bundle - output_bundle)
    (bundle_signal_a_link ?output_bundle - output_bundle ?signal_channel_a - signal_channel_a)
    (bundle_signal_b_link ?output_bundle - output_bundle ?signal_channel_b - signal_channel_b)
    (bundle_marker_a ?output_bundle - output_bundle)
    (bundle_marker_b ?output_bundle - output_bundle)
    (bundle_verified ?output_bundle - output_bundle)
    (orchestrator_primary_link ?orchestrator - orchestrator ?primary_participant - primary_participant)
    (orchestrator_secondary_link ?orchestrator - orchestrator ?secondary_participant - secondary_participant)
    (orchestrator_output_link ?orchestrator - orchestrator ?output_bundle - output_bundle)
    (attachment_available ?attachment_record - attachment_record)
    (orchestrator_attachment_link ?orchestrator - orchestrator ?attachment_record - attachment_record)
    (attachment_bound ?attachment_record - attachment_record)
    (attachment_bundle_link ?attachment_record - attachment_record ?output_bundle - output_bundle)
    (orchestrator_review_stage_a ?orchestrator - orchestrator)
    (orchestrator_review_stage_b ?orchestrator - orchestrator)
    (orchestrator_review_stage_c ?orchestrator - orchestrator)
    (policy_bound ?orchestrator - orchestrator)
    (orchestrator_approval_ready ?orchestrator - orchestrator)
    (orchestrator_enrichment_ready ?orchestrator - orchestrator)
    (orchestrator_validation_passed ?orchestrator - orchestrator)
    (approval_reference_available ?approval_reference - approval_reference)
    (orchestrator_approval_reference_link ?orchestrator - orchestrator ?approval_reference - approval_reference)
    (orchestrator_review_started ?orchestrator - orchestrator)
    (orchestrator_review_confirmed ?orchestrator - orchestrator)
    (orchestrator_review_closed ?orchestrator - orchestrator)
    (policy_available ?policy_rule - policy_rule)
    (orchestrator_policy_link ?orchestrator - orchestrator ?policy_rule - policy_rule)
    (metadata_available ?metadata_record - metadata_record)
    (orchestrator_metadata_link ?orchestrator - orchestrator ?metadata_record - metadata_record)
    (execution_parameter_available ?execution_parameter - execution_parameter)
    (orchestrator_parameter_link ?orchestrator - orchestrator ?execution_parameter - execution_parameter)
    (notification_template_available ?notification_template - notification_template)
    (orchestrator_notification_link ?orchestrator - orchestrator ?notification_template - notification_template)
    (audit_token_available ?audit_token - audit_token)
    (workflow_audit_link ?workflow_instance - workflow_instance ?audit_token - audit_token)
    (primary_participant_primed ?primary_participant - primary_participant)
    (secondary_participant_primed ?secondary_participant - secondary_participant)
    (orchestrator_finalized ?orchestrator - orchestrator)
  )
  (:action initialize_workflow_instance
    :parameters (?workflow_instance - workflow_instance)
    :precondition
      (and
        (not
          (workflow_instance_open ?workflow_instance)
        )
        (not
          (finalized ?workflow_instance)
        )
      )
    :effect (workflow_instance_open ?workflow_instance)
  )
  (:action allocate_workflow_resource
    :parameters (?workflow_instance - workflow_instance ?resource_token - resource_token)
    :precondition
      (and
        (workflow_instance_open ?workflow_instance)
        (not
          (resource_allocated ?workflow_instance)
        )
        (resource_available ?resource_token)
      )
    :effect
      (and
        (resource_allocated ?workflow_instance)
        (workflow_resource_binding ?workflow_instance ?resource_token)
        (not
          (resource_available ?resource_token)
        )
      )
  )
  (:action bind_operation_type_to_workflow
    :parameters (?workflow_instance - workflow_instance ?operation_type - operation_type)
    :precondition
      (and
        (workflow_instance_open ?workflow_instance)
        (resource_allocated ?workflow_instance)
        (operation_type_available ?operation_type)
      )
    :effect
      (and
        (workflow_operation_binding ?workflow_instance ?operation_type)
        (not
          (operation_type_available ?operation_type)
        )
      )
  )
  (:action confirm_workflow_operation_binding
    :parameters (?workflow_instance - workflow_instance ?operation_type - operation_type)
    :precondition
      (and
        (workflow_instance_open ?workflow_instance)
        (resource_allocated ?workflow_instance)
        (workflow_operation_binding ?workflow_instance ?operation_type)
        (not
          (context_enabled ?workflow_instance)
        )
      )
    :effect (context_enabled ?workflow_instance)
  )
  (:action release_operation_type_binding
    :parameters (?workflow_instance - workflow_instance ?operation_type - operation_type)
    :precondition
      (and
        (workflow_operation_binding ?workflow_instance ?operation_type)
      )
    :effect
      (and
        (operation_type_available ?operation_type)
        (not
          (workflow_operation_binding ?workflow_instance ?operation_type)
        )
      )
  )
  (:action bind_capability_to_workflow
    :parameters (?workflow_instance - workflow_instance ?capability - capability)
    :precondition
      (and
        (context_enabled ?workflow_instance)
        (capability_available ?capability)
      )
    :effect
      (and
        (workflow_capability_binding ?workflow_instance ?capability)
        (not
          (capability_available ?capability)
        )
      )
  )
  (:action release_capability_binding
    :parameters (?workflow_instance - workflow_instance ?capability - capability)
    :precondition
      (and
        (workflow_capability_binding ?workflow_instance ?capability)
      )
    :effect
      (and
        (capability_available ?capability)
        (not
          (workflow_capability_binding ?workflow_instance ?capability)
        )
      )
  )
  (:action bind_execution_parameter_to_orchestrator
    :parameters (?orchestrator - orchestrator ?execution_parameter - execution_parameter)
    :precondition
      (and
        (context_enabled ?orchestrator)
        (execution_parameter_available ?execution_parameter)
      )
    :effect
      (and
        (orchestrator_parameter_link ?orchestrator ?execution_parameter)
        (not
          (execution_parameter_available ?execution_parameter)
        )
      )
  )
  (:action unbind_execution_parameter_from_orchestrator
    :parameters (?orchestrator - orchestrator ?execution_parameter - execution_parameter)
    :precondition
      (and
        (orchestrator_parameter_link ?orchestrator ?execution_parameter)
      )
    :effect
      (and
        (execution_parameter_available ?execution_parameter)
        (not
          (orchestrator_parameter_link ?orchestrator ?execution_parameter)
        )
      )
  )
  (:action bind_notification_template_to_orchestrator
    :parameters (?orchestrator - orchestrator ?notification_template - notification_template)
    :precondition
      (and
        (context_enabled ?orchestrator)
        (notification_template_available ?notification_template)
      )
    :effect
      (and
        (orchestrator_notification_link ?orchestrator ?notification_template)
        (not
          (notification_template_available ?notification_template)
        )
      )
  )
  (:action unbind_notification_template_from_orchestrator
    :parameters (?orchestrator - orchestrator ?notification_template - notification_template)
    :precondition
      (and
        (orchestrator_notification_link ?orchestrator ?notification_template)
      )
    :effect
      (and
        (notification_template_available ?notification_template)
        (not
          (orchestrator_notification_link ?orchestrator ?notification_template)
        )
      )
  )
  (:action emit_primary_signal_a
    :parameters (?primary_participant - primary_participant ?signal_channel_a - signal_channel_a ?operation_type - operation_type)
    :precondition
      (and
        (context_enabled ?primary_participant)
        (workflow_operation_binding ?primary_participant ?operation_type)
        (primary_channel_binding ?primary_participant ?signal_channel_a)
        (not
          (signal_a_emitted ?signal_channel_a)
        )
        (not
          (signal_a_buffered ?signal_channel_a)
        )
      )
    :effect (signal_a_emitted ?signal_channel_a)
  )
  (:action prime_primary_participant
    :parameters (?primary_participant - primary_participant ?signal_channel_a - signal_channel_a ?capability - capability)
    :precondition
      (and
        (context_enabled ?primary_participant)
        (workflow_capability_binding ?primary_participant ?capability)
        (primary_channel_binding ?primary_participant ?signal_channel_a)
        (signal_a_emitted ?signal_channel_a)
        (not
          (primary_participant_primed ?primary_participant)
        )
      )
    :effect
      (and
        (primary_participant_primed ?primary_participant)
        (primary_participant_committed ?primary_participant)
      )
  )
  (:action attach_primary_artifact
    :parameters (?primary_participant - primary_participant ?signal_channel_a - signal_channel_a ?artifact - artifact)
    :precondition
      (and
        (context_enabled ?primary_participant)
        (primary_channel_binding ?primary_participant ?signal_channel_a)
        (artifact_available ?artifact)
        (not
          (primary_participant_primed ?primary_participant)
        )
      )
    :effect
      (and
        (signal_a_buffered ?signal_channel_a)
        (primary_participant_primed ?primary_participant)
        (primary_artifact_binding ?primary_participant ?artifact)
        (not
          (artifact_available ?artifact)
        )
      )
  )
  (:action restore_primary_artifact
    :parameters (?primary_participant - primary_participant ?signal_channel_a - signal_channel_a ?operation_type - operation_type ?artifact - artifact)
    :precondition
      (and
        (context_enabled ?primary_participant)
        (workflow_operation_binding ?primary_participant ?operation_type)
        (primary_channel_binding ?primary_participant ?signal_channel_a)
        (signal_a_buffered ?signal_channel_a)
        (primary_artifact_binding ?primary_participant ?artifact)
        (not
          (primary_participant_committed ?primary_participant)
        )
      )
    :effect
      (and
        (signal_a_emitted ?signal_channel_a)
        (primary_participant_committed ?primary_participant)
        (artifact_available ?artifact)
        (not
          (primary_artifact_binding ?primary_participant ?artifact)
        )
      )
  )
  (:action emit_secondary_signal_b
    :parameters (?secondary_participant - secondary_participant ?signal_channel_b - signal_channel_b ?operation_type - operation_type)
    :precondition
      (and
        (context_enabled ?secondary_participant)
        (workflow_operation_binding ?secondary_participant ?operation_type)
        (secondary_channel_binding ?secondary_participant ?signal_channel_b)
        (not
          (signal_b_emitted ?signal_channel_b)
        )
        (not
          (signal_b_buffered ?signal_channel_b)
        )
      )
    :effect (signal_b_emitted ?signal_channel_b)
  )
  (:action prime_secondary_participant
    :parameters (?secondary_participant - secondary_participant ?signal_channel_b - signal_channel_b ?capability - capability)
    :precondition
      (and
        (context_enabled ?secondary_participant)
        (workflow_capability_binding ?secondary_participant ?capability)
        (secondary_channel_binding ?secondary_participant ?signal_channel_b)
        (signal_b_emitted ?signal_channel_b)
        (not
          (secondary_participant_primed ?secondary_participant)
        )
      )
    :effect
      (and
        (secondary_participant_primed ?secondary_participant)
        (secondary_participant_committed ?secondary_participant)
      )
  )
  (:action attach_secondary_artifact
    :parameters (?secondary_participant - secondary_participant ?signal_channel_b - signal_channel_b ?artifact - artifact)
    :precondition
      (and
        (context_enabled ?secondary_participant)
        (secondary_channel_binding ?secondary_participant ?signal_channel_b)
        (artifact_available ?artifact)
        (not
          (secondary_participant_primed ?secondary_participant)
        )
      )
    :effect
      (and
        (signal_b_buffered ?signal_channel_b)
        (secondary_participant_primed ?secondary_participant)
        (secondary_artifact_binding ?secondary_participant ?artifact)
        (not
          (artifact_available ?artifact)
        )
      )
  )
  (:action restore_secondary_artifact
    :parameters (?secondary_participant - secondary_participant ?signal_channel_b - signal_channel_b ?operation_type - operation_type ?artifact - artifact)
    :precondition
      (and
        (context_enabled ?secondary_participant)
        (workflow_operation_binding ?secondary_participant ?operation_type)
        (secondary_channel_binding ?secondary_participant ?signal_channel_b)
        (signal_b_buffered ?signal_channel_b)
        (secondary_artifact_binding ?secondary_participant ?artifact)
        (not
          (secondary_participant_committed ?secondary_participant)
        )
      )
    :effect
      (and
        (signal_b_emitted ?signal_channel_b)
        (secondary_participant_committed ?secondary_participant)
        (artifact_available ?artifact)
        (not
          (secondary_artifact_binding ?secondary_participant ?artifact)
        )
      )
  )
  (:action consolidate_output_bundle_variant_a
    :parameters (?primary_participant - primary_participant ?secondary_participant - secondary_participant ?signal_channel_a - signal_channel_a ?signal_channel_b - signal_channel_b ?output_bundle - output_bundle)
    :precondition
      (and
        (primary_participant_primed ?primary_participant)
        (secondary_participant_primed ?secondary_participant)
        (primary_channel_binding ?primary_participant ?signal_channel_a)
        (secondary_channel_binding ?secondary_participant ?signal_channel_b)
        (signal_a_emitted ?signal_channel_a)
        (signal_b_emitted ?signal_channel_b)
        (primary_participant_committed ?primary_participant)
        (secondary_participant_committed ?secondary_participant)
        (output_bundle_available ?output_bundle)
      )
    :effect
      (and
        (output_bundle_prepared ?output_bundle)
        (bundle_signal_a_link ?output_bundle ?signal_channel_a)
        (bundle_signal_b_link ?output_bundle ?signal_channel_b)
        (not
          (output_bundle_available ?output_bundle)
        )
      )
  )
  (:action consolidate_output_bundle_variant_b
    :parameters (?primary_participant - primary_participant ?secondary_participant - secondary_participant ?signal_channel_a - signal_channel_a ?signal_channel_b - signal_channel_b ?output_bundle - output_bundle)
    :precondition
      (and
        (primary_participant_primed ?primary_participant)
        (secondary_participant_primed ?secondary_participant)
        (primary_channel_binding ?primary_participant ?signal_channel_a)
        (secondary_channel_binding ?secondary_participant ?signal_channel_b)
        (signal_a_buffered ?signal_channel_a)
        (signal_b_emitted ?signal_channel_b)
        (not
          (primary_participant_committed ?primary_participant)
        )
        (secondary_participant_committed ?secondary_participant)
        (output_bundle_available ?output_bundle)
      )
    :effect
      (and
        (output_bundle_prepared ?output_bundle)
        (bundle_signal_a_link ?output_bundle ?signal_channel_a)
        (bundle_signal_b_link ?output_bundle ?signal_channel_b)
        (bundle_marker_a ?output_bundle)
        (not
          (output_bundle_available ?output_bundle)
        )
      )
  )
  (:action consolidate_output_bundle_variant_c
    :parameters (?primary_participant - primary_participant ?secondary_participant - secondary_participant ?signal_channel_a - signal_channel_a ?signal_channel_b - signal_channel_b ?output_bundle - output_bundle)
    :precondition
      (and
        (primary_participant_primed ?primary_participant)
        (secondary_participant_primed ?secondary_participant)
        (primary_channel_binding ?primary_participant ?signal_channel_a)
        (secondary_channel_binding ?secondary_participant ?signal_channel_b)
        (signal_a_emitted ?signal_channel_a)
        (signal_b_buffered ?signal_channel_b)
        (primary_participant_committed ?primary_participant)
        (not
          (secondary_participant_committed ?secondary_participant)
        )
        (output_bundle_available ?output_bundle)
      )
    :effect
      (and
        (output_bundle_prepared ?output_bundle)
        (bundle_signal_a_link ?output_bundle ?signal_channel_a)
        (bundle_signal_b_link ?output_bundle ?signal_channel_b)
        (bundle_marker_b ?output_bundle)
        (not
          (output_bundle_available ?output_bundle)
        )
      )
  )
  (:action consolidate_output_bundle_variant_d
    :parameters (?primary_participant - primary_participant ?secondary_participant - secondary_participant ?signal_channel_a - signal_channel_a ?signal_channel_b - signal_channel_b ?output_bundle - output_bundle)
    :precondition
      (and
        (primary_participant_primed ?primary_participant)
        (secondary_participant_primed ?secondary_participant)
        (primary_channel_binding ?primary_participant ?signal_channel_a)
        (secondary_channel_binding ?secondary_participant ?signal_channel_b)
        (signal_a_buffered ?signal_channel_a)
        (signal_b_buffered ?signal_channel_b)
        (not
          (primary_participant_committed ?primary_participant)
        )
        (not
          (secondary_participant_committed ?secondary_participant)
        )
        (output_bundle_available ?output_bundle)
      )
    :effect
      (and
        (output_bundle_prepared ?output_bundle)
        (bundle_signal_a_link ?output_bundle ?signal_channel_a)
        (bundle_signal_b_link ?output_bundle ?signal_channel_b)
        (bundle_marker_a ?output_bundle)
        (bundle_marker_b ?output_bundle)
        (not
          (output_bundle_available ?output_bundle)
        )
      )
  )
  (:action verify_output_bundle
    :parameters (?output_bundle - output_bundle ?primary_participant - primary_participant ?operation_type - operation_type)
    :precondition
      (and
        (output_bundle_prepared ?output_bundle)
        (primary_participant_primed ?primary_participant)
        (workflow_operation_binding ?primary_participant ?operation_type)
        (not
          (bundle_verified ?output_bundle)
        )
      )
    :effect (bundle_verified ?output_bundle)
  )
  (:action bind_attachment_to_orchestrator
    :parameters (?orchestrator - orchestrator ?attachment_record - attachment_record ?output_bundle - output_bundle)
    :precondition
      (and
        (context_enabled ?orchestrator)
        (orchestrator_output_link ?orchestrator ?output_bundle)
        (orchestrator_attachment_link ?orchestrator ?attachment_record)
        (attachment_available ?attachment_record)
        (output_bundle_prepared ?output_bundle)
        (bundle_verified ?output_bundle)
        (not
          (attachment_bound ?attachment_record)
        )
      )
    :effect
      (and
        (attachment_bound ?attachment_record)
        (attachment_bundle_link ?attachment_record ?output_bundle)
        (not
          (attachment_available ?attachment_record)
        )
      )
  )
  (:action certify_attachment_binding
    :parameters (?orchestrator - orchestrator ?attachment_record - attachment_record ?output_bundle - output_bundle ?operation_type - operation_type)
    :precondition
      (and
        (context_enabled ?orchestrator)
        (orchestrator_attachment_link ?orchestrator ?attachment_record)
        (attachment_bound ?attachment_record)
        (attachment_bundle_link ?attachment_record ?output_bundle)
        (workflow_operation_binding ?orchestrator ?operation_type)
        (not
          (bundle_marker_a ?output_bundle)
        )
        (not
          (orchestrator_review_stage_a ?orchestrator)
        )
      )
    :effect (orchestrator_review_stage_a ?orchestrator)
  )
  (:action bind_policy_to_orchestrator
    :parameters (?orchestrator - orchestrator ?policy_rule - policy_rule)
    :precondition
      (and
        (context_enabled ?orchestrator)
        (policy_available ?policy_rule)
        (not
          (policy_bound ?orchestrator)
        )
      )
    :effect
      (and
        (policy_bound ?orchestrator)
        (orchestrator_policy_link ?orchestrator ?policy_rule)
        (not
          (policy_available ?policy_rule)
        )
      )
  )
  (:action certify_policy_binding
    :parameters (?orchestrator - orchestrator ?attachment_record - attachment_record ?output_bundle - output_bundle ?operation_type - operation_type ?policy_rule - policy_rule)
    :precondition
      (and
        (context_enabled ?orchestrator)
        (orchestrator_attachment_link ?orchestrator ?attachment_record)
        (attachment_bound ?attachment_record)
        (attachment_bundle_link ?attachment_record ?output_bundle)
        (workflow_operation_binding ?orchestrator ?operation_type)
        (bundle_marker_a ?output_bundle)
        (policy_bound ?orchestrator)
        (orchestrator_policy_link ?orchestrator ?policy_rule)
        (not
          (orchestrator_review_stage_a ?orchestrator)
        )
      )
    :effect
      (and
        (orchestrator_review_stage_a ?orchestrator)
        (orchestrator_approval_ready ?orchestrator)
      )
  )
  (:action promote_orchestrator_checkpoint_a
    :parameters (?orchestrator - orchestrator ?execution_parameter - execution_parameter ?capability - capability ?attachment_record - attachment_record ?output_bundle - output_bundle)
    :precondition
      (and
        (orchestrator_review_stage_a ?orchestrator)
        (orchestrator_parameter_link ?orchestrator ?execution_parameter)
        (workflow_capability_binding ?orchestrator ?capability)
        (orchestrator_attachment_link ?orchestrator ?attachment_record)
        (attachment_bundle_link ?attachment_record ?output_bundle)
        (not
          (bundle_marker_b ?output_bundle)
        )
        (not
          (orchestrator_review_stage_b ?orchestrator)
        )
      )
    :effect (orchestrator_review_stage_b ?orchestrator)
  )
  (:action promote_orchestrator_checkpoint_b
    :parameters (?orchestrator - orchestrator ?execution_parameter - execution_parameter ?capability - capability ?attachment_record - attachment_record ?output_bundle - output_bundle)
    :precondition
      (and
        (orchestrator_review_stage_a ?orchestrator)
        (orchestrator_parameter_link ?orchestrator ?execution_parameter)
        (workflow_capability_binding ?orchestrator ?capability)
        (orchestrator_attachment_link ?orchestrator ?attachment_record)
        (attachment_bundle_link ?attachment_record ?output_bundle)
        (bundle_marker_b ?output_bundle)
        (not
          (orchestrator_review_stage_b ?orchestrator)
        )
      )
    :effect (orchestrator_review_stage_b ?orchestrator)
  )
  (:action promote_orchestrator_checkpoint_c
    :parameters (?orchestrator - orchestrator ?notification_template - notification_template ?attachment_record - attachment_record ?output_bundle - output_bundle)
    :precondition
      (and
        (orchestrator_review_stage_b ?orchestrator)
        (orchestrator_notification_link ?orchestrator ?notification_template)
        (orchestrator_attachment_link ?orchestrator ?attachment_record)
