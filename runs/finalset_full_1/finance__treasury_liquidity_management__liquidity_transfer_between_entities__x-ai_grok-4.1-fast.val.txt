  (:action create_liquidity_request_context
    :parameters (?legal_entity - domain_entity)
    :precondition
      (and
        (not
          (entity_liquidity_context_created ?legal_entity)
        )
        (not
          (entity_settlement_marked ?legal_entity)
        )
      )
    :effect (entity_liquidity_context_created ?legal_entity)
  )
  (:action assign_payment_channel_to_entity
    :parameters (?legal_entity - domain_entity ?payment_channel - payment_channel)
    :precondition
      (and
        (entity_liquidity_context_created ?legal_entity)
        (not
          (entity_request_registered ?legal_entity)
        )
        (payment_channel_available ?payment_channel)
      )
    :effect
      (and
        (entity_request_registered ?legal_entity)
        (entity_payment_channel_assigned ?legal_entity ?payment_channel)
        (not
          (payment_channel_available ?payment_channel)
        )
      )
  )
  (:action associate_liquidity_request_with_entity
    :parameters (?legal_entity - domain_entity ?liquidity_request - liquidity_request)
    :precondition
      (and
        (entity_liquidity_context_created ?legal_entity)
        (entity_request_registered ?legal_entity)
        (liquidity_request_available ?liquidity_request)
      )
    :effect
      (and
        (entity_request_association ?legal_entity ?liquidity_request)
        (not
          (liquidity_request_available ?liquidity_request)
        )
      )
  )
  (:action validate_liquidity_request
    :parameters (?legal_entity - domain_entity ?liquidity_request - liquidity_request)
    :precondition
      (and
        (entity_liquidity_context_created ?legal_entity)
        (entity_request_registered ?legal_entity)
        (entity_request_association ?legal_entity ?liquidity_request)
        (not
          (entity_request_validated ?legal_entity)
        )
      )
    :effect (entity_request_validated ?legal_entity)
  )
  (:action deregister_liquidity_request
    :parameters (?legal_entity - domain_entity ?liquidity_request - liquidity_request)
    :precondition
      (and
        (entity_request_association ?legal_entity ?liquidity_request)
      )
    :effect
      (and
        (liquidity_request_available ?liquidity_request)
        (not
          (entity_request_association ?legal_entity ?liquidity_request)
        )
      )
  )
  (:action assign_treasury_approver
    :parameters (?legal_entity - domain_entity ?treasury_approver - treasury_approver)
    :precondition
      (and
        (entity_request_validated ?legal_entity)
        (treasury_approver_available ?treasury_approver)
      )
    :effect
      (and
        (entity_treasury_approver_assigned ?legal_entity ?treasury_approver)
        (not
          (treasury_approver_available ?treasury_approver)
        )
      )
  )
  (:action unassign_treasury_approver
    :parameters (?legal_entity - domain_entity ?treasury_approver - treasury_approver)
    :precondition
      (and
        (entity_treasury_approver_assigned ?legal_entity ?treasury_approver)
      )
    :effect
      (and
        (treasury_approver_available ?treasury_approver)
        (not
          (entity_treasury_approver_assigned ?legal_entity ?treasury_approver)
        )
      )
  )
  (:action attach_approval_credential
    :parameters (?transfer_instruction - transfer_instruction ?approval_credential - approval_credential)
    :precondition
      (and
        (entity_request_validated ?transfer_instruction)
        (approval_credential_available ?approval_credential)
      )
    :effect
      (and
        (instruction_has_approval_credential ?transfer_instruction ?approval_credential)
        (not
          (approval_credential_available ?approval_credential)
        )
      )
  )
  (:action detach_approval_credential
    :parameters (?transfer_instruction - transfer_instruction ?approval_credential - approval_credential)
    :precondition
      (and
        (instruction_has_approval_credential ?transfer_instruction ?approval_credential)
      )
    :effect
      (and
        (approval_credential_available ?approval_credential)
        (not
          (instruction_has_approval_credential ?transfer_instruction ?approval_credential)
        )
      )
  )
  (:action attach_execution_signature
    :parameters (?transfer_instruction - transfer_instruction ?execution_signature - execution_signature)
    :precondition
      (and
        (entity_request_validated ?transfer_instruction)
        (execution_signature_available ?execution_signature)
      )
    :effect
      (and
        (instruction_has_execution_signature ?transfer_instruction ?execution_signature)
        (not
          (execution_signature_available ?execution_signature)
        )
      )
  )
  (:action detach_execution_signature
    :parameters (?transfer_instruction - transfer_instruction ?execution_signature - execution_signature)
    :precondition
      (and
        (instruction_has_execution_signature ?transfer_instruction ?execution_signature)
      )
    :effect
      (and
        (execution_signature_available ?execution_signature)
        (not
          (instruction_has_execution_signature ?transfer_instruction ?execution_signature)
        )
      )
  )
  (:action reserve_source_buffer
    :parameters (?source_account - source_account ?liquidity_buffer - liquidity_buffer ?liquidity_request - liquidity_request)
    :precondition
      (and
        (entity_request_validated ?source_account)
        (entity_request_association ?source_account ?liquidity_request)
        (source_buffer_linked ?source_account ?liquidity_buffer)
        (not
          (liquidity_buffer_reserved ?liquidity_buffer)
        )
        (not
          (liquidity_buffer_committed ?liquidity_buffer)
        )
      )
    :effect (liquidity_buffer_reserved ?liquidity_buffer)
  )
  (:action confirm_source_account_readiness
    :parameters (?source_account - source_account ?liquidity_buffer - liquidity_buffer ?treasury_approver - treasury_approver)
    :precondition
      (and
        (entity_request_validated ?source_account)
        (entity_treasury_approver_assigned ?source_account ?treasury_approver)
        (source_buffer_linked ?source_account ?liquidity_buffer)
        (liquidity_buffer_reserved ?liquidity_buffer)
        (not
          (source_account_marked ?source_account)
        )
      )
    :effect
      (and
        (source_account_marked ?source_account)
        (source_account_ready ?source_account)
      )
  )
  (:action allocate_funding_instrument_to_source
    :parameters (?source_account - source_account ?liquidity_buffer - liquidity_buffer ?funding_instrument - funding_instrument)
    :precondition
      (and
        (entity_request_validated ?source_account)
        (source_buffer_linked ?source_account ?liquidity_buffer)
        (funding_instrument_available ?funding_instrument)
        (not
          (source_account_marked ?source_account)
        )
      )
    :effect
      (and
        (liquidity_buffer_committed ?liquidity_buffer)
        (source_account_marked ?source_account)
        (source_assigned_instrument ?source_account ?funding_instrument)
        (not
          (funding_instrument_available ?funding_instrument)
        )
      )
  )
  (:action finalize_source_instrument_allocation
    :parameters (?source_account - source_account ?liquidity_buffer - liquidity_buffer ?liquidity_request - liquidity_request ?funding_instrument - funding_instrument)
    :precondition
      (and
        (entity_request_validated ?source_account)
        (entity_request_association ?source_account ?liquidity_request)
        (source_buffer_linked ?source_account ?liquidity_buffer)
        (liquidity_buffer_committed ?liquidity_buffer)
        (source_assigned_instrument ?source_account ?funding_instrument)
        (not
          (source_account_ready ?source_account)
        )
      )
    :effect
      (and
        (liquidity_buffer_reserved ?liquidity_buffer)
        (source_account_ready ?source_account)
        (funding_instrument_available ?funding_instrument)
        (not
          (source_assigned_instrument ?source_account ?funding_instrument)
        )
      )
  )
  (:action reserve_target_buffer
    :parameters (?destination_account - destination_account ?target_buffer - target_buffer ?liquidity_request - liquidity_request)
    :precondition
      (and
        (entity_request_validated ?destination_account)
        (entity_request_association ?destination_account ?liquidity_request)
        (destination_buffer_linked ?destination_account ?target_buffer)

          (target_buffer_reserved ?target_buffer)

        (destination_account_marked ?destination_account)

      )
  )
  (:action allocate_funding_instrument_to_destination
    :parameters (?destination_account - destination_account ?target_buffer - target_buffer ?funding_instrument - funding_instrument)
    :precondition
      (and
        (entity_request_validated ?destination_account)
        (destination_buffer_linked ?destination_account ?target_buffer)
        (funding_instrument_available ?funding_instrument)
        (not
          (destination_account_marked ?destination_account)
        )
      )
    :effect
      (and
        (target_buffer_committed ?target_buffer)
        (destination_account_marked ?destination_account)
        (destination_assigned_instrument ?destination_account ?funding_instrument)
        (not
          (funding_instrument_available ?funding_instrument)
        )
      )
  )
  (:action finalize_destination_instrument_allocation
    :parameters (?destination_account - destination_account ?target_buffer - target_buffer ?liquidity_request - liquidity_request ?funding_instrument - funding_instrument)
    :precondition
      (and
        (entity_request_validated ?destination_account)
        (entity_request_association ?destination_account ?liquidity_request)
        (destination_buffer_linked ?destination_account ?target_buffer)
        (target_buffer_committed ?target_buffer)
        (destination_assigned_instrument ?destination_account ?funding_instrument)
        (not
          (destination_account_ready ?destination_account)
        )
      )
    :effect
      (and
        (target_buffer_reserved ?target_buffer)
        (destination_account_ready ?destination_account)
        (funding_instrument_available ?funding_instrument)
        (not
          (destination_assigned_instrument ?destination_account ?funding_instrument)
        )
      )
  )
  (:action assemble_settlement_batch_standard
    :parameters (?source_account - source_account ?destination_account - destination_account ?liquidity_buffer - liquidity_buffer ?target_buffer - target_buffer ?settlement_batch - settlement_batch)
    :precondition
      (and
        (source_account_marked ?source_account)
        (destination_account_marked ?destination_account)
        (source_buffer_linked ?source_account ?liquidity_buffer)
        (destination_buffer_linked ?destination_account ?target_buffer)
        (liquidity_buffer_reserved ?liquidity_buffer)
        (target_buffer_reserved ?target_buffer)
        (source_account_ready ?source_account)
        (destination_account_ready ?destination_account)
        (settlement_batch_available ?settlement_batch)
      )
    :effect
      (and
        (settlement_batch_active ?settlement_batch)
        (batch_includes_source_buffer ?settlement_batch ?liquidity_buffer)
        (batch_includes_target_buffer ?settlement_batch ?target_buffer)
        (not
          (settlement_batch_available ?settlement_batch)
        )
      )
  )
  (:action assemble_settlement_batch_with_confirmation_a
    :parameters (?source_account - source_account ?destination_account - destination_account ?liquidity_buffer - liquidity_buffer ?target_buffer - target_buffer ?settlement_batch - settlement_batch)
    :precondition
      (and
        (source_account_marked ?source_account)
        (destination_account_marked ?destination_account)
        (source_buffer_linked ?source_account ?liquidity_buffer)
        (destination_buffer_linked ?destination_account ?target_buffer)
        (liquidity_buffer_committed ?liquidity_buffer)
        (target_buffer_reserved ?target_buffer)
        (not
          (source_account_ready ?source_account)
        )
        (destination_account_ready ?destination_account)
        (settlement_batch_available ?settlement_batch)
      )
    :effect
      (and
        (settlement_batch_active ?settlement_batch)
        (batch_includes_source_buffer ?settlement_batch ?liquidity_buffer)
        (batch_includes_target_buffer ?settlement_batch ?target_buffer)
        (batch_processing_flag_a ?settlement_batch)
        (not
          (settlement_batch_available ?settlement_batch)
        )
      )
  )
  (:action assemble_settlement_batch_with_confirmation_b
    :parameters (?source_account - source_account ?destination_account - destination_account ?liquidity_buffer - liquidity_buffer ?target_buffer - target_buffer ?settlement_batch - settlement_batch)
    :precondition
      (and
        (source_account_marked ?source_account)
        (destination_account_marked ?destination_account)
        (source_buffer_linked ?source_account ?liquidity_buffer)
        (destination_buffer_linked ?destination_account ?target_buffer)
        (liquidity_buffer_reserved ?liquidity_buffer)
        (target_buffer_committed ?target_buffer)
        (source_account_ready ?source_account)
        (not
          (destination_account_ready ?destination_account)
        )
        (settlement_batch_available ?settlement_batch)
      )
    :effect
      (and
        (settlement_batch_active ?settlement_batch)
        (batch_includes_source_buffer ?settlement_batch ?liquidity_buffer)
        (batch_includes_target_buffer ?settlement_batch ?target_buffer)
        (batch_processing_flag_b ?settlement_batch)
        (not
          (settlement_batch_available ?settlement_batch)
        )
      )
  )
  (:action assemble_settlement_batch_with_confirmation_a_and_b
    :parameters (?source_account - source_account ?destination_account - destination_account ?liquidity_buffer - liquidity_buffer ?target_buffer - target_buffer ?settlement_batch - settlement_batch)
    :precondition
      (and
        (source_account_marked ?source_account)
        (destination_account_marked ?destination_account)
        (source_buffer_linked ?source_account ?liquidity_buffer)
        (destination_buffer_linked ?destination_account ?target_buffer)
        (liquidity_buffer_committed ?liquidity_buffer)
        (target_buffer_committed ?target_buffer)
        (not
          (source_account_ready ?source_account)
        )
        (not
          (destination_account_ready ?destination_account)
        )
        (settlement_batch_available ?settlement_batch)
      )
    :effect
      (and
        (settlement_batch_active ?settlement_batch)
        (batch_includes_source_buffer ?settlement_batch ?liquidity_buffer)
        (batch_includes_target_buffer ?settlement_batch ?target_buffer)
        (batch_processing_flag_a ?settlement_batch)
        (batch_processing_flag_b ?settlement_batch)
        (not
          (settlement_batch_available ?settlement_batch)
        )
      )
  )
  (:action confirm_settlement_batch
    :parameters (?settlement_batch - settlement_batch ?source_account - source_account ?liquidity_request - liquidity_request)
    :precondition
      (and
        (settlement_batch_active ?settlement_batch)
        (source_account_marked ?source_account)
        (entity_request_association ?source_account ?liquidity_request)
        (not
          (batch_confirmed ?settlement_batch)
        )
      )
    :effect (batch_confirmed ?settlement_batch)
  )
  (:action ingest_payment_file_for_instruction
    :parameters (?transfer_instruction - transfer_instruction ?payment_file - payment_file ?settlement_batch - settlement_batch)
    :precondition
      (and
        (entity_request_validated ?transfer_instruction)
        (instruction_in_settlement_batch ?transfer_instruction ?settlement_batch)
        (instruction_has_payment_file ?transfer_instruction ?payment_file)
        (payment_file_available ?payment_file)
        (settlement_batch_active ?settlement_batch)
        (batch_confirmed ?settlement_batch)
        (not
          (payment_file_locked ?payment_file)
        )
      )
    :effect
      (and
        (payment_file_locked ?payment_file)
        (payment_file_assigned_to_batch ?payment_file ?settlement_batch)
        (not
          (payment_file_available ?payment_file)
        )
      )
  )
  (:action validate_instruction_file
    :parameters (?transfer_instruction - transfer_instruction ?payment_file - payment_file ?settlement_batch - settlement_batch ?liquidity_request - liquidity_request)
    :precondition
      (and
        (entity_request_validated ?transfer_instruction)
        (instruction_has_payment_file ?transfer_instruction ?payment_file)
        (payment_file_locked ?payment_file)
        (payment_file_assigned_to_batch ?payment_file ?settlement_batch)
        (entity_request_association ?transfer_instruction ?liquidity_request)
        (not
          (batch_processing_flag_a ?settlement_batch)
        )
        (not
          (instruction_file_validated ?transfer_instruction)
        )
      )
    :effect (instruction_file_validated ?transfer_instruction)
  )
  (:action assign_compliance_rule_to_instruction
    :parameters (?transfer_instruction - transfer_instruction ?compliance_rule - compliance_rule)
    :precondition
      (and
        (entity_request_validated ?transfer_instruction)
        (compliance_rule_available ?compliance_rule)
        (not
          (compliance_rule_assigned ?transfer_instruction)
        )
      )
    :effect
      (and
        (compliance_rule_assigned ?transfer_instruction)
        (instruction_compliance_rule_link ?transfer_instruction ?compliance_rule)
        (not
          (compliance_rule_available ?compliance_rule)
        )
      )
  )
  (:action evaluate_compliance_rule_for_instruction
    :parameters (?transfer_instruction - transfer_instruction ?payment_file - payment_file ?settlement_batch - settlement_batch ?liquidity_request - liquidity_request ?compliance_rule - compliance_rule)
    :precondition
      (and
        (entity_request_validated ?transfer_instruction)
        (instruction_has_payment_file ?transfer_instruction ?payment_file)
        (payment_file_locked ?payment_file)
        (payment_file_assigned_to_batch ?payment_file ?settlement_batch)
        (entity_request_association ?transfer_instruction ?liquidity_request)
        (batch_processing_flag_a ?settlement_batch)
        (compliance_rule_assigned ?transfer_instruction)
        (instruction_compliance_rule_link ?transfer_instruction ?compliance_rule)
        (not
          (instruction_file_validated ?transfer_instruction)
        )
      )
    :effect
      (and
        (instruction_file_validated ?transfer_instruction)
        (compliance_rule_check_passed ?transfer_instruction)
      )
  )
  (:action apply_approval_credential_and_mark
    :parameters (?transfer_instruction - transfer_instruction ?approval_credential - approval_credential ?treasury_approver - treasury_approver ?payment_file - payment_file ?settlement_batch - settlement_batch)
    :precondition
      (and
        (instruction_file_validated ?transfer_instruction)
        (instruction_has_approval_credential ?transfer_instruction ?approval_credential)
        (entity_treasury_approver_assigned ?transfer_instruction ?treasury_approver)
        (instruction_has_payment_file ?transfer_instruction ?payment_file)
        (payment_file_assigned_to_batch ?payment_file ?settlement_batch)
        (not
          (batch_processing_flag_b ?settlement_batch)
        )
        (not
          (instruction_approval_attached ?transfer_instruction)
        )
      )
    :effect (instruction_approval_attached ?transfer_instruction)
  )
  (:action apply_approval_credential_for_flagged_batch
    :parameters (?transfer_instruction - transfer_instruction ?approval_credential - approval_credential ?treasury_approver - treasury_approver ?payment_file - payment_file ?settlement_batch - settlement_batch)
    :precondition
      (and
        (instruction_file_validated ?transfer_instruction)
        (instruction_has_approval_credential ?transfer_instruction ?approval_credential)
        (entity_treasury_approver_assigned ?transfer_instruction ?treasury_approver)
        (instruction_has_payment_file ?transfer_instruction ?payment_file)
        (payment_file_assigned_to_batch ?payment_file ?settlement_batch)
        (batch_processing_flag_b ?settlement_batch)
        (not
          (instruction_approval_attached ?transfer_instruction)
        )
      )
    :effect (instruction_approval_attached ?transfer_instruction)
  )
  (:action apply_execution_signature_standard
    :parameters (?transfer_instruction - transfer_instruction ?execution_signature - execution_signature ?payment_file - payment_file ?settlement_batch - settlement_batch)
