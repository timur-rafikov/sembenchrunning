  (:action register_consignment_case
    :parameters (?consignment_record - consignment_record)
    :precondition
      (and
        (not
          (case_registered ?consignment_record)
        )
        (not
          (item_compliance_cleared ?consignment_record)
        )
      )
    :effect (case_registered ?consignment_record)
  )
  (:action assign_classification_resource_to_consignment
    :parameters (?consignment_record - consignment_record ?classification_resource - classification_resource)
    :precondition
      (and
        (case_registered ?consignment_record)
        (not
          (classification_assigned ?consignment_record)
        )
        (classification_resource_available ?classification_resource)
      )
    :effect
      (and
        (classification_assigned ?consignment_record)
        (item_assigned_classification_resource ?consignment_record ?classification_resource)
        (not
          (classification_resource_available ?classification_resource)
        )
      )
  )
  (:action propose_classification_candidate_for_consignment
    :parameters (?consignment_record - consignment_record ?commodity_classification_candidate - commodity_classification_candidate)
    :precondition
      (and
        (case_registered ?consignment_record)
        (classification_assigned ?consignment_record)
        (classification_candidate_available ?commodity_classification_candidate)
      )
    :effect
      (and
        (item_has_classification_candidate ?consignment_record ?commodity_classification_candidate)
        (not
          (classification_candidate_available ?commodity_classification_candidate)
        )
      )
  )
  (:action confirm_classification_candidate_for_consignment
    :parameters (?consignment_record - consignment_record ?commodity_classification_candidate - commodity_classification_candidate)

      (and
        (case_registered ?consignment_record)
        (classification_assigned ?consignment_record)
        (item_has_classification_candidate ?consignment_record ?commodity_classification_candidate)
        (not
          (classification_confirmed_for_item ?consignment_record)
        )

    :effect (classification_confirmed_for_item ?consignment_record)
  )
  (:action revert_classification_candidate_assignment
    :parameters (?consignment_record - consignment_record ?commodity_classification_candidate - commodity_classification_candidate)
