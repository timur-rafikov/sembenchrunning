  (:action register_payment
    :parameters (?payment_entity - payment_entity)
    :precondition
      (and
        (not
          (entity_registered ?payment_entity)
        )
        (not
          (settled ?payment_entity)
        )
      )
    :effect (entity_registered ?payment_entity)
  )
  (:action assign_settlement_channel_to_payment
    :parameters (?payment_entity - payment_entity ?settlement_channel - settlement_channel)
    :precondition
      (and
        (entity_registered ?payment_entity)
        (not
          (channel_allocated ?payment_entity)
        )
        (channel_available ?settlement_channel)
      )
    :effect
      (and
        (channel_allocated ?payment_entity)
        (allocated_to_settlement_channel ?payment_entity ?settlement_channel)
        (not
          (channel_available ?settlement_channel)
        )
      )
  )
  (:action assign_validation_slot_to_payment
    :parameters (?payment_entity - payment_entity ?validation_slot - validation_slot)
    :precondition
      (and
        (entity_registered ?payment_entity)
        (channel_allocated ?payment_entity)
        (validation_slot_available ?validation_slot)
      )
    :effect
      (and
        (assigned_validation_slot ?payment_entity ?validation_slot)
        (not
          (validation_slot_available ?validation_slot)
        )
      )
  )
  (:action mark_payment_released
    :parameters (?payment_entity - payment_entity ?validation_slot - validation_slot)
    :precondition
      (and
        (entity_registered ?payment_entity)
        (channel_allocated ?payment_entity)
        (assigned_validation_slot ?payment_entity ?validation_slot)
        (not
          (entity_released ?payment_entity)
        )
      )
    :effect (entity_released ?payment_entity)
  )
  (:action release_validation_slot
    :parameters (?payment_entity - payment_entity ?validation_slot - validation_slot)
    :precondition
      (and
        (assigned_validation_slot ?payment_entity ?validation_slot)
      )
    :effect
      (and
        (validation_slot_available ?validation_slot)
        (not
          (assigned_validation_slot ?payment_entity ?validation_slot)
        )
      )
  )
  (:action assign_processing_queue_to_payment
    :parameters (?payment_entity - payment_entity ?processing_queue - processing_queue)
    :precondition
      (and
        (entity_released ?payment_entity)
        (processing_queue_available ?processing_queue)
      )
    :effect
      (and
        (assigned_processing_queue ?payment_entity ?processing_queue)
        (not
          (processing_queue_available ?processing_queue)
        )
      )
  )
  (:action release_processing_queue_from_payment
    :parameters (?payment_entity - payment_entity ?processing_queue - processing_queue)
    :precondition
      (and
        (assigned_processing_queue ?payment_entity ?processing_queue)
      )
    :effect
      (and
        (processing_queue_available ?processing_queue)
        (not
          (assigned_processing_queue ?payment_entity ?processing_queue)
        )
      )
  )
  (:action allocate_external_resource_to_job
    :parameters (?processing_job - processing_job ?external_resource_token - external_resource_token)

      (and
        (entity_released ?processing_job)
        (external_resource_token_available ?external_resource_token)
      )
    :effect
      (and
        (processing_job_has_external_resource_token ?processing_job ?external_resource_token)

          (external_resource_token_available ?external_resource_token)
        )
      )
  )
  (:action release_external_resource_from_job
    :parameters (?processing_job - processing_job ?external_resource_token - external_resource_token)

      (and
        (processing_job_has_external_resource_token ?processing_job ?external_resource_token)
      )
    :effect
      (and
        (external_resource_token_available ?external_resource_token)
        (not
          (processing_job_has_external_resource_token ?processing_job ?external_resource_token)
        )
      )
  )

    :parameters (?processing_job - processing_job ?third_party_ack - third_party_ack)

        (third_party_ack_available ?third_party_ack)

          (third_party_ack_available ?third_party_ack)
        )
      )
  )
  (:action release_third_party_ack_from_job
    :parameters (?processing_job - processing_job ?third_party_ack - third_party_ack)
    :precondition
      (and
        (processing_job_third_party_acknowledged ?processing_job ?third_party_ack)
      )
    :effect
      (and
        (third_party_ack_available ?third_party_ack)
        (not
          (processing_job_third_party_acknowledged ?processing_job ?third_party_ack)
        )
      )
  )
  (:action reserve_payer_capacity_unit
    :parameters (?payer_batch - payer_batch ?rail_capacity_unit_source - rail_capacity_unit_source ?validation_slot - validation_slot)
    :precondition
      (and
        (entity_released ?payer_batch)
        (assigned_validation_slot ?payer_batch ?validation_slot)
        (payer_capacity_unit_mapping ?payer_batch ?rail_capacity_unit_source)
        (not
          (source_capacity_reserved ?rail_capacity_unit_source)
        )
        (not
          (source_capacity_on_hold ?rail_capacity_unit_source)
        )
      )
    :effect (source_capacity_reserved ?rail_capacity_unit_source)
  )
  (:action confirm_payer_capacity_reservation
    :parameters (?payer_batch - payer_batch ?rail_capacity_unit_source - rail_capacity_unit_source ?processing_queue - processing_queue)
    :precondition
      (and
        (entity_released ?payer_batch)
        (assigned_processing_queue ?payer_batch ?processing_queue)
        (payer_capacity_unit_mapping ?payer_batch ?rail_capacity_unit_source)
        (source_capacity_reserved ?rail_capacity_unit_source)
        (not
          (payer_capacity_allocated_flag ?payer_batch)
        )
      )
    :effect
      (and
        (payer_capacity_allocated_flag ?payer_batch)
        (payer_batch_ready ?payer_batch)
      )
  )
  (:action reserve_payer_split_token
    :parameters (?payer_batch - payer_batch ?rail_capacity_unit_source - rail_capacity_unit_source ?split_token - split_token)
    :precondition
      (and
        (entity_released ?payer_batch)
        (payer_capacity_unit_mapping ?payer_batch ?rail_capacity_unit_source)
        (split_token_available ?split_token)
        (not
          (payer_capacity_allocated_flag ?payer_batch)
        )
      )
    :effect
      (and
        (source_capacity_on_hold ?rail_capacity_unit_source)
        (payer_capacity_allocated_flag ?payer_batch)
        (split_token_assigned_to_payer ?payer_batch ?split_token)
        (not
          (split_token_available ?split_token)
        )
      )
  )
  (:action finalize_payer_capacity_allocation
    :parameters (?payer_batch - payer_batch ?rail_capacity_unit_source - rail_capacity_unit_source ?validation_slot - validation_slot ?split_token - split_token)

        (split_token_assigned_to_payer ?payer_batch ?split_token)
        (not
          (payer_batch_ready ?payer_batch)
        )
      )
    :effect
      (and
        (source_capacity_reserved ?rail_capacity_unit_source)
        (payer_batch_ready ?payer_batch)
        (split_token_available ?split_token)
        (not
          (split_token_assigned_to_payer ?payer_batch ?split_token)
        )
      )
  )

  (:action reserve_payee_split_token

    :parameters (?payee_batch - payee_batch ?rail_capacity_unit_destination - rail_capacity_unit_destination ?validation_slot - validation_slot ?split_token - split_token)

          (payee_batch_ready ?payee_batch)

  (:action create_split_batch

    :precondition
      (and

        (payee_capacity_unit_mapping ?payee_batch ?rail_capacity_unit_destination)

        (payee_batch_ready ?payee_batch)

      )
    :effect
      (and
        (split_batch_created ?split_batch)
        (split_batch_assigned_source_capacity_unit ?split_batch ?rail_capacity_unit_source)

          (split_batch_slot_available ?split_batch)

    :effect
      (and
        (split_batch_created ?split_batch)

        (not
          (split_batch_slot_available ?split_batch)
        )
      )

          (split_batch_slot_available ?split_batch)

          (split_batch_staged ?split_batch)

      )

          (instruction_file_available ?instruction_file)

        (entity_released ?processing_job)

          (split_batch_source_ready ?split_batch)

          (processing_job_ready_for_transmission ?processing_job)

          (priority_attached ?processing_job)

        (priority_attached ?processing_job)

          (priority_marker_available ?priority_marker)

      )

  (:action apply_priority_and_prepare_job
    :parameters (?processing_job - processing_job ?instruction_file - instruction_file ?split_batch - split_batch ?validation_slot - validation_slot ?priority_marker - priority_marker)
    :precondition
      (and
        (entity_released ?processing_job)
        (processing_job_has_instruction_file ?processing_job ?instruction_file)
        (instruction_file_consumed ?instruction_file)
        (instruction_file_attached_to_split_batch ?instruction_file ?split_batch)
        (assigned_validation_slot ?processing_job ?validation_slot)
        (split_batch_source_ready ?split_batch)
        (priority_attached ?processing_job)
        (processing_job_has_priority_marker ?processing_job ?priority_marker)
        (not
          (processing_job_ready_for_transmission ?processing_job)
        )
      )
