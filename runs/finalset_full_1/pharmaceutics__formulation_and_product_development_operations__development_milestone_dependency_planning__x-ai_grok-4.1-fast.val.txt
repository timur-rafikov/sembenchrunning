--- domain.pddl (as passed to VAL) ---
(define (domain pharmaceutics_development_milestone_planning)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_supertype - object variant_supertype - object plan_package_supertype - object development_item_supertype - object development_milestone - development_item_supertype assessment_resource_slot - resource_supertype analytical_assay - resource_supertype specialist_reviewer - resource_supertype component_spec - resource_supertype capability_resource - resource_supertype regulatory_requirement - resource_supertype assay_specification - resource_supertype external_consultant - resource_supertype sample_batch - variant_supertype stability_protocol - variant_supertype regulatory_submission - variant_supertype formulation_variant - plan_package_supertype process_variant - plan_package_supertype manufacturing_plan - plan_package_supertype formulation_milestone_category - development_milestone candidate_category - development_milestone formulation_workstream_milestone - formulation_milestone_category process_workstream_milestone - formulation_milestone_category product_candidate - candidate_category)
  (:predicates
    (milestone_created ?development_milestone - development_milestone)
    (development_item_assay_verified ?development_milestone - development_milestone)
    (milestone_assessment_allocated ?development_milestone - development_milestone)
    (development_item_completed ?development_milestone - development_milestone)
    (workstream_signoff ?development_milestone - development_milestone)
    (regulatory_requirement_applied ?development_milestone - development_milestone)
    (assessment_slot_available ?assessment_resource_slot - assessment_resource_slot)
    (development_item_assigned_resource_slot ?development_milestone - development_milestone ?assessment_resource_slot - assessment_resource_slot)
    (assay_available ?analytical_assay - analytical_assay)
    (development_item_assigned_assay ?development_milestone - development_milestone ?analytical_assay - analytical_assay)
    (reviewer_available ?specialist_reviewer - specialist_reviewer)
    (development_item_assigned_reviewer ?development_milestone - development_milestone ?specialist_reviewer - specialist_reviewer)
    (sample_available ?sample_batch - sample_batch)
    (formulation_milestone_bound_sample ?formulation_workstream_milestone - formulation_workstream_milestone ?sample_batch - sample_batch)
    (process_milestone_bound_sample ?process_workstream_milestone - process_workstream_milestone ?sample_batch - sample_batch)
    (formulation_milestone_variant_association ?formulation_workstream_milestone - formulation_workstream_milestone ?formulation_variant - formulation_variant)
    (formulation_variant_validated ?formulation_variant - formulation_variant)
    (formulation_variant_has_sample ?formulation_variant - formulation_variant)
    (formulation_milestone_ready ?formulation_workstream_milestone - formulation_workstream_milestone)
    (process_milestone_variant_association ?process_workstream_milestone - process_workstream_milestone ?process_variant - process_variant)
    (process_variant_validated ?process_variant - process_variant)
    (process_variant_has_sample ?process_variant - process_variant)
    (process_milestone_ready ?process_workstream_milestone - process_workstream_milestone)
    (manufacturing_plan_available ?manufacturing_plan - manufacturing_plan)
    (manufacturing_plan_generated ?manufacturing_plan - manufacturing_plan)
    (plan_includes_formulation_variant ?manufacturing_plan - manufacturing_plan ?formulation_variant - formulation_variant)
    (plan_includes_process_variant ?manufacturing_plan - manufacturing_plan ?process_variant - process_variant)
    (plan_formulation_confirmed ?manufacturing_plan - manufacturing_plan)
    (plan_process_confirmed ?manufacturing_plan - manufacturing_plan)
    (plan_scaleup_verified ?manufacturing_plan - manufacturing_plan)
    (candidate_has_formulation_milestone ?product_candidate - product_candidate ?formulation_workstream_milestone - formulation_workstream_milestone)
    (candidate_has_process_milestone ?product_candidate - product_candidate ?process_workstream_milestone - process_workstream_milestone)
    (candidate_associated_with_plan ?product_candidate - product_candidate ?manufacturing_plan - manufacturing_plan)
    (stability_protocol_available ?stability_protocol - stability_protocol)
    (candidate_bound_stability_protocol ?product_candidate - product_candidate ?stability_protocol - stability_protocol)
    (stability_protocol_committed ?stability_protocol - stability_protocol)
    (stability_protocol_attached_to_plan ?stability_protocol - stability_protocol ?manufacturing_plan - manufacturing_plan)
    (candidate_stability_and_assay_integrated ?product_candidate - product_candidate)
    (candidate_capability_reviewed ?product_candidate - product_candidate)
    (candidate_analytical_review_ready ?product_candidate - product_candidate)
    (candidate_has_component_spec ?product_candidate - product_candidate)
    (candidate_component_spec_confirmed ?product_candidate - product_candidate)
    (candidate_capability_confirmed ?product_candidate - product_candidate)
    (candidate_final_internal_review_passed ?product_candidate - product_candidate)
    (regulatory_submission_available ?regulatory_submission - regulatory_submission)
    (candidate_has_regulatory_submission_document ?product_candidate - product_candidate ?regulatory_submission - regulatory_submission)
    (candidate_submission_attached ?product_candidate - product_candidate)
    (candidate_submission_review_in_progress ?product_candidate - product_candidate)
    (candidate_submission_review_completed ?product_candidate - product_candidate)
    (component_spec_available ?component_spec - component_spec)
    (candidate_bound_component_spec ?product_candidate - product_candidate ?component_spec - component_spec)
    (capability_resource_available ?capability_resource - capability_resource)
    (candidate_bound_capability_resource ?product_candidate - product_candidate ?capability_resource - capability_resource)
    (assay_specification_available ?assay_specification - assay_specification)
    (candidate_bound_assay_specification ?product_candidate - product_candidate ?assay_specification - assay_specification)
    (external_consultant_available ?external_consultant - external_consultant)
    (candidate_engaged_external_consultant ?product_candidate - product_candidate ?external_consultant - external_consultant)
    (regulatory_requirement_available ?regulatory_requirement - regulatory_requirement)
    (development_item_has_regulatory_requirement ?development_milestone - development_milestone ?regulatory_requirement - regulatory_requirement)
    (formulation_milestone_ready_for_integration ?formulation_workstream_milestone - formulation_workstream_milestone)
    (process_milestone_ready_for_integration ?process_workstream_milestone - process_workstream_milestone)
    (candidate_finalized ?product_candidate - product_candidate)
  )
  (:action create_development_milestone
    :parameters (?development_milestone - development_milestone)
    :precondition
      (and
        (not
          (milestone_created ?development_milestone)
        )
        (not
          (development_item_completed ?development_milestone)
        )
      )
    :effect (milestone_created ?development_milestone)
  )
  (:action assign_assessment_slot_to_milestone
    :parameters (?development_milestone - development_milestone ?assessment_resource_slot - assessment_resource_slot)
    :precondition
      (and
        (milestone_created ?development_milestone)
        (not
          (milestone_assessment_allocated ?development_milestone)
        )
        (assessment_slot_available ?assessment_resource_slot)
      )
    :effect
      (and
        (milestone_assessment_allocated ?development_milestone)
        (development_item_assigned_resource_slot ?development_milestone ?assessment_resource_slot)
        (not
          (assessment_slot_available ?assessment_resource_slot)
        )
      )
  )
  (:action assign_assay_to_milestone
    :parameters (?development_milestone - development_milestone ?analytical_assay - analytical_assay)
    :precondition
      (and
        (milestone_created ?development_milestone)
        (milestone_assessment_allocated ?development_milestone)
        (assay_available ?analytical_assay)
      )
    :effect
      (and
        (development_item_assigned_assay ?development_milestone ?analytical_assay)
        (not
          (assay_available ?analytical_assay)
        )
      )
  )
  (:action confirm_milestone_assay_verification
    :parameters (?development_milestone - development_milestone ?analytical_assay - analytical_assay)
    :precondition
      (and
        (milestone_created ?development_milestone)
        (milestone_assessment_allocated ?development_milestone)
        (development_item_assigned_assay ?development_milestone ?analytical_assay)
        (not
          (development_item_assay_verified ?development_milestone)
        )
      )
    :effect (development_item_assay_verified ?development_milestone)
  )
  (:action release_assay_from_milestone
    :parameters (?development_milestone - development_milestone ?analytical_assay - analytical_assay)
    :precondition
      (and
        (development_item_assigned_assay ?development_milestone ?analytical_assay)
      )
    :effect
      (and
        (assay_available ?analytical_assay)
        (not
          (development_item_assigned_assay ?development_milestone ?analytical_assay)
        )
      )
  )
  (:action assign_specialist_reviewer_to_milestone
    :parameters (?development_milestone - development_milestone ?specialist_reviewer - specialist_reviewer)
    :precondition
      (and
        (development_item_assay_verified ?development_milestone)
        (reviewer_available ?specialist_reviewer)
      )
    :effect
      (and
        (development_item_assigned_reviewer ?development_milestone ?specialist_reviewer)
        (not
          (reviewer_available ?specialist_reviewer)
        )
      )
  )
  (:action release_specialist_reviewer_from_milestone
    :parameters (?development_milestone - development_milestone ?specialist_reviewer - specialist_reviewer)
    :precondition
      (and
        (development_item_assigned_reviewer ?development_milestone ?specialist_reviewer)
      )
    :effect
      (and
        (reviewer_available ?specialist_reviewer)
        (not
          (development_item_assigned_reviewer ?development_milestone ?specialist_reviewer)
        )
      )
  )
  (:action bind_assay_specification_to_candidate
    :parameters (?product_candidate - product_candidate ?assay_specification - assay_specification)
    :precondition
      (and
        (development_item_assay_verified ?product_candidate)
        (assay_specification_available ?assay_specification)
      )
    :effect
      (and
        (candidate_bound_assay_specification ?product_candidate ?assay_specification)
        (not
          (assay_specification_available ?assay_specification)
        )
      )
  )
  (:action release_assay_specification_from_candidate
    :parameters (?product_candidate - product_candidate ?assay_specification - assay_specification)
    :precondition
      (and
        (candidate_bound_assay_specification ?product_candidate ?assay_specification)
      )
    :effect
      (and
        (assay_specification_available ?assay_specification)
        (not
          (candidate_bound_assay_specification ?product_candidate ?assay_specification)
        )
      )
  )
  (:action engage_external_consultant_for_candidate
    :parameters (?product_candidate - product_candidate ?external_consultant - external_consultant)
    :precondition
      (and
        (development_item_assay_verified ?product_candidate)
        (external_consultant_available ?external_consultant)
      )
    :effect
      (and
        (candidate_engaged_external_consultant ?product_candidate ?external_consultant)
        (not
          (external_consultant_available ?external_consultant)
        )
      )
  )
  (:action disengage_external_consultant_from_candidate
    :parameters (?product_candidate - product_candidate ?external_consultant - external_consultant)
    :precondition
      (and
        (candidate_engaged_external_consultant ?product_candidate ?external_consultant)
      )
    :effect
      (and
        (external_consultant_available ?external_consultant)
        (not
          (candidate_engaged_external_consultant ?product_candidate ?external_consultant)
        )
      )
  )
  (:action validate_formulation_variant
    :parameters (?formulation_workstream_milestone - formulation_workstream_milestone ?formulation_variant - formulation_variant ?analytical_assay - analytical_assay)
    :precondition
      (and
        (development_item_assay_verified ?formulation_workstream_milestone)
        (development_item_assigned_assay ?formulation_workstream_milestone ?analytical_assay)
        (formulation_milestone_variant_association ?formulation_workstream_milestone ?formulation_variant)
        (not
          (formulation_variant_validated ?formulation_variant)
        )
        (not
          (formulation_variant_has_sample ?formulation_variant)
        )
      )
    :effect (formulation_variant_validated ?formulation_variant)
  )
  (:action approve_formulation_variant_for_integration
    :parameters (?formulation_workstream_milestone - formulation_workstream_milestone ?formulation_variant - formulation_variant ?specialist_reviewer - specialist_reviewer)
    :precondition
      (and
        (development_item_assay_verified ?formulation_workstream_milestone)
        (development_item_assigned_reviewer ?formulation_workstream_milestone ?specialist_reviewer)
        (formulation_milestone_variant_association ?formulation_workstream_milestone ?formulation_variant)
        (formulation_variant_validated ?formulation_variant)
        (not
          (formulation_milestone_ready_for_integration ?formulation_workstream_milestone)
        )
      )
    :effect
      (and
        (formulation_milestone_ready_for_integration ?formulation_workstream_milestone)
        (formulation_milestone_ready ?formulation_workstream_milestone)
      )
  )
  (:action attach_sample_to_formulation_milestone
    :parameters (?formulation_workstream_milestone - formulation_workstream_milestone ?formulation_variant - formulation_variant ?sample_batch - sample_batch)
    :precondition
      (and
        (development_item_assay_verified ?formulation_workstream_milestone)
        (formulation_milestone_variant_association ?formulation_workstream_milestone ?formulation_variant)
        (sample_available ?sample_batch)
        (not
          (formulation_milestone_ready_for_integration ?formulation_workstream_milestone)
        )
      )
    :effect
      (and
        (formulation_variant_has_sample ?formulation_variant)
        (formulation_milestone_ready_for_integration ?formulation_workstream_milestone)
        (formulation_milestone_bound_sample ?formulation_workstream_milestone ?sample_batch)
        (not
          (sample_available ?sample_batch)
        )
      )
  )
  (:action evaluate_formulation_variant_with_assay_and_sample
    :parameters (?formulation_workstream_milestone - formulation_workstream_milestone ?formulation_variant - formulation_variant ?analytical_assay - analytical_assay ?sample_batch - sample_batch)
    :precondition
      (and
        (development_item_assay_verified ?formulation_workstream_milestone)
        (development_item_assigned_assay ?formulation_workstream_milestone ?analytical_assay)
        (formulation_milestone_variant_association ?formulation_workstream_milestone ?formulation_variant)
        (formulation_variant_has_sample ?formulation_variant)
        (formulation_milestone_bound_sample ?formulation_workstream_milestone ?sample_batch)
        (not
          (formulation_milestone_ready ?formulation_workstream_milestone)
        )
      )
    :effect
      (and
        (formulation_variant_validated ?formulation_variant)
        (formulation_milestone_ready ?formulation_workstream_milestone)
        (sample_available ?sample_batch)
        (not
          (formulation_milestone_bound_sample ?formulation_workstream_milestone ?sample_batch)
        )
      )
  )
  (:action validate_process_variant
    :parameters (?process_workstream_milestone - process_workstream_milestone ?process_variant - process_variant ?analytical_assay - analytical_assay)
    :precondition
      (and
        (development_item_assay_verified ?process_workstream_milestone)
        (development_item_assigned_assay ?process_workstream_milestone ?analytical_assay)
        (process_milestone_variant_association ?process_workstream_milestone ?process_variant)
        (not
          (process_variant_validated ?process_variant)
        )
        (not
          (process_variant_has_sample ?process_variant)
        )
      )
    :effect (process_variant_validated ?process_variant)
  )
  (:action approve_process_variant_for_integration
    :parameters (?process_workstream_milestone - process_workstream_milestone ?process_variant - process_variant ?specialist_reviewer - specialist_reviewer)
    :precondition
      (and
        (development_item_assay_verified ?process_workstream_milestone)
        (development_item_assigned_reviewer ?process_workstream_milestone ?specialist_reviewer)
        (process_milestone_variant_association ?process_workstream_milestone ?process_variant)
        (process_variant_validated ?process_variant)
        (not
          (process_milestone_ready_for_integration ?process_workstream_milestone)
        )
      )
    :effect
      (and
        (process_milestone_ready_for_integration ?process_workstream_milestone)
        (process_milestone_ready ?process_workstream_milestone)
      )
  )
  (:action attach_sample_to_process_milestone
    :parameters (?process_workstream_milestone - process_workstream_milestone ?process_variant - process_variant ?sample_batch - sample_batch)
    :precondition
      (and
        (development_item_assay_verified ?process_workstream_milestone)
        (process_milestone_variant_association ?process_workstream_milestone ?process_variant)
        (sample_available ?sample_batch)
        (not
          (process_milestone_ready_for_integration ?process_workstream_milestone)
        )
      )
    :effect
      (and
        (process_variant_has_sample ?process_variant)
        (process_milestone_ready_for_integration ?process_workstream_milestone)
        (process_milestone_bound_sample ?process_workstream_milestone ?sample_batch)
        (not
          (sample_available ?sample_batch)
        )
      )
  )
  (:action evaluate_process_variant_with_assay_and_sample
    :parameters (?process_workstream_milestone - process_workstream_milestone ?process_variant - process_variant ?analytical_assay - analytical_assay ?sample_batch - sample_batch)
    :precondition
      (and
        (development_item_assay_verified ?process_workstream_milestone)
        (development_item_assigned_assay ?process_workstream_milestone ?analytical_assay)
        (process_milestone_variant_association ?process_workstream_milestone ?process_variant)
        (process_variant_has_sample ?process_variant)
        (process_milestone_bound_sample ?process_workstream_milestone ?sample_batch)
        (not
          (process_milestone_ready ?process_workstream_milestone)
        )
      )
    :effect
      (and
        (process_variant_validated ?process_variant)
        (process_milestone_ready ?process_workstream_milestone)
        (sample_available ?sample_batch)
        (not
          (process_milestone_bound_sample ?process_workstream_milestone ?sample_batch)
        )
      )
  )
  (:action generate_manufacturing_plan
    :parameters (?formulation_workstream_milestone - formulation_workstream_milestone ?process_workstream_milestone - process_workstream_milestone ?formulation_variant - formulation_variant ?process_variant - process_variant ?manufacturing_plan - manufacturing_plan)
    :precondition
      (and
        (formulation_milestone_ready_for_integration ?formulation_workstream_milestone)
        (process_milestone_ready_for_integration ?process_workstream_milestone)
        (formulation_milestone_variant_association ?formulation_workstream_milestone ?formulation_variant)
        (process_milestone_variant_association ?process_workstream_milestone ?process_variant)
        (formulation_variant_validated ?formulation_variant)
        (process_variant_validated ?process_variant)
        (formulation_milestone_ready ?formulation_workstream_milestone)
        (process_milestone_ready ?process_workstream_milestone)
        (manufacturing_plan_available ?manufacturing_plan)
      )
    :effect
      (and
        (manufacturing_plan_generated ?manufacturing_plan)
        (plan_includes_formulation_variant ?manufacturing_plan ?formulation_variant)
        (plan_includes_process_variant ?manufacturing_plan ?process_variant)
        (not
          (manufacturing_plan_available ?manufacturing_plan)
        )
      )
  )
  (:action generate_manufacturing_plan_with_formulation_confirmation
    :parameters (?formulation_workstream_milestone - formulation_workstream_milestone ?process_workstream_milestone - process_workstream_milestone ?formulation_variant - formulation_variant ?process_variant - process_variant ?manufacturing_plan - manufacturing_plan)
    :precondition
      (and
        (formulation_milestone_ready_for_integration ?formulation_workstream_milestone)
        (process_milestone_ready_for_integration ?process_workstream_milestone)
        (formulation_milestone_variant_association ?formulation_workstream_milestone ?formulation_variant)
        (process_milestone_variant_association ?process_workstream_milestone ?process_variant)
        (formulation_variant_has_sample ?formulation_variant)
        (process_variant_validated ?process_variant)
        (not
          (formulation_milestone_ready ?formulation_workstream_milestone)
        )
        (process_milestone_ready ?process_workstream_milestone)
        (manufacturing_plan_available ?manufacturing_plan)
      )
    :effect
      (and
        (manufacturing_plan_generated ?manufacturing_plan)
        (plan_includes_formulation_variant ?manufacturing_plan ?formulation_variant)
        (plan_includes_process_variant ?manufacturing_plan ?process_variant)
        (plan_formulation_confirmed ?manufacturing_plan)
        (not
          (manufacturing_plan_available ?manufacturing_plan)
        )
      )
  )
  (:action generate_manufacturing_plan_with_process_confirmation
    :parameters (?formulation_workstream_milestone - formulation_workstream_milestone ?process_workstream_milestone - process_workstream_milestone ?formulation_variant - formulation_variant ?process_variant - process_variant ?manufacturing_plan - manufacturing_plan)
    :precondition
      (and
        (formulation_milestone_ready_for_integration ?formulation_workstream_milestone)
        (process_milestone_ready_for_integration ?process_workstream_milestone)
        (formulation_milestone_variant_association ?formulation_workstream_milestone ?formulation_variant)
        (process_milestone_variant_association ?process_workstream_milestone ?process_variant)
        (formulation_variant_validated ?formulation_variant)
        (process_variant_has_sample ?process_variant)
        (formulation_milestone_ready ?formulation_workstream_milestone)
        (not
          (process_milestone_ready ?process_workstream_milestone)
        )
        (manufacturing_plan_available ?manufacturing_plan)
      )
    :effect
      (and
        (manufacturing_plan_generated ?manufacturing_plan)
        (plan_includes_formulation_variant ?manufacturing_plan ?formulation_variant)
        (plan_includes_process_variant ?manufacturing_plan ?process_variant)
        (plan_process_confirmed ?manufacturing_plan)
        (not
          (manufacturing_plan_available ?manufacturing_plan)
        )
      )
  )
  (:action generate_manufacturing_plan_with_full_confirmations
    :parameters (?formulation_workstream_milestone - formulation_workstream_milestone ?process_workstream_milestone - process_workstream_milestone ?formulation_variant - formulation_variant ?process_variant - process_variant ?manufacturing_plan - manufacturing_plan)
    :precondition
      (and
        (formulation_milestone_ready_for_integration ?formulation_workstream_milestone)
        (process_milestone_ready_for_integration ?process_workstream_milestone)
        (formulation_milestone_variant_association ?formulation_workstream_milestone ?formulation_variant)
        (process_milestone_variant_association ?process_workstream_milestone ?process_variant)
        (formulation_variant_has_sample ?formulation_variant)
        (process_variant_has_sample ?process_variant)
        (not
          (formulation_milestone_ready ?formulation_workstream_milestone)
        )
        (not
          (process_milestone_ready ?process_workstream_milestone)
        )
        (manufacturing_plan_available ?manufacturing_plan)
      )
    :effect
      (and
        (manufacturing_plan_generated ?manufacturing_plan)
        (plan_includes_formulation_variant ?manufacturing_plan ?formulation_variant)
        (plan_includes_process_variant ?manufacturing_plan ?process_variant)
        (plan_formulation_confirmed ?manufacturing_plan)
        (plan_process_confirmed ?manufacturing_plan)
        (not
          (manufacturing_plan_available ?manufacturing_plan)
        )
      )
  )
  (:action verify_manufacturing_plan_scaleup
    :parameters (?manufacturing_plan - manufacturing_plan ?formulation_workstream_milestone - formulation_workstream_milestone ?analytical_assay - analytical_assay)
    :precondition
      (and
        (manufacturing_plan_generated ?manufacturing_plan)
        (formulation_milestone_ready_for_integration ?formulation_workstream_milestone)
        (development_item_assigned_assay ?formulation_workstream_milestone ?analytical_assay)
        (not
          (plan_scaleup_verified ?manufacturing_plan)
        )
      )
    :effect (plan_scaleup_verified ?manufacturing_plan)
  )
  (:action commit_stability_protocol_to_plan
    :parameters (?product_candidate - product_candidate ?stability_protocol - stability_protocol ?manufacturing_plan - manufacturing_plan)
    :precondition
      (and
        (development_item_assay_verified ?product_candidate)
        (candidate_associated_with_plan ?product_candidate ?manufacturing_plan)
        (candidate_bound_stability_protocol ?product_candidate ?stability_protocol)
        (stability_protocol_available ?stability_protocol)
        (manufacturing_plan_generated ?manufacturing_plan)
        (plan_scaleup_verified ?manufacturing_plan)
        (not
          (stability_protocol_committed ?stability_protocol)
        )
      )
    :effect
      (and
        (stability_protocol_committed ?stability_protocol)
        (stability_protocol_attached_to_plan ?stability_protocol ?manufacturing_plan)
        (not
          (stability_protocol_available ?stability_protocol)
        )
      )
  )
  (:action integrate_stability_and_assay_for_candidate
    :parameters (?product_candidate - product_candidate ?stability_protocol - stability_protocol ?manufacturing_plan - manufacturing_plan ?analytical_assay - analytical_assay)
    :precondition
      (and
        (development_item_assay_verified ?product_candidate)
        (candidate_bound_stability_protocol ?product_candidate ?stability_protocol)
        (stability_protocol_committed ?stability_protocol)
        (stability_protocol_attached_to_plan ?stability_protocol ?manufacturing_plan)
        (development_item_assigned_assay ?product_candidate ?analytical_assay)
        (not
          (plan_formulation_confirmed ?manufacturing_plan)
