  (:action release_order
    :parameters (?order - customer_order)
    :precondition
      (and
        (not
          (entity_released ?order)
        )
        (not
          (packaging_finalized ?order)
        )
      )
    :effect (entity_released ?order)
  )
  (:action allocate_packaging_type_to_order
    :parameters (?order - customer_order ?packaging_type - packaging_type)
    :precondition
      (and
        (entity_released ?order)
        (not
          (entity_allocated ?order)
        )
        (packaging_type_available ?packaging_type)
      )
    :effect
      (and
        (entity_allocated ?order)
        (entity_assigned_packaging_type ?order ?packaging_type)
        (not
          (packaging_type_available ?packaging_type)
        )
      )
  )
  (:action reserve_sku_for_order
    :parameters (?order - customer_order ?sku - item_sku)
    :precondition
      (and
        (entity_released ?order)
        (entity_allocated ?order)
        (sku_available ?sku)
      )
    :effect
      (and
        (entity_has_sku_reserved ?order ?sku)
        (not
          (sku_available ?sku)
        )
      )
  )
  (:action confirm_sku_allocation_for_order
    :parameters (?order - customer_order ?sku - item_sku)
    :precondition
      (and
        (entity_released ?order)
        (entity_allocated ?order)
        (entity_has_sku_reserved ?order ?sku)
        (not
          (entity_ready_for_packing ?order)
        )
      )
    :effect (entity_ready_for_packing ?order)
  )
  (:action release_sku_reservation_from_order
    :parameters (?order - customer_order ?sku - item_sku)
    :precondition
      (and
        (entity_has_sku_reserved ?order ?sku)
      )
    :effect
      (and
        (sku_available ?sku)
        (not
          (entity_has_sku_reserved ?order ?sku)
        )
      )
  )
  (:action assign_packing_station_to_order
    :parameters (?order - customer_order ?packing_station - packing_station)

      (and
        (entity_ready_for_packing ?order)
        (packing_station_available ?packing_station)
      )
    :effect
      (and
        (entity_assigned_packing_station ?order ?packing_station)
        (not
          (packing_station_available ?packing_station)
        )
      )

        (entity_assigned_packing_station ?order ?packing_station)

        (template_populated ?pack_template)
