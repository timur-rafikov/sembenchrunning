  (:action create_close_work_item
    :parameters (?close_task - work_item)
    :precondition
      (and
        (not
          (work_item_created ?close_task)
        )
        (not
          (closed_for_period ?close_task)
        )
      )
    :effect (work_item_created ?close_task)
  )

    :parameters (?close_task - work_item ?resource - resource)
    :precondition

        (work_item_created ?close_task)

      )

        (work_item_assigned ?close_task)
        (work_item_has_resource ?close_task ?resource)
        (not
          (resource_available ?resource)
        )
      )
  )
