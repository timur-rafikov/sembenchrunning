--- domain.pddl (as passed to VAL) ---
(define (domain veterinary_intervention_priority_ordering)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity - object resource - entity material - entity classification - entity animal - entity animal_patient - animal treatment_station - resource diagnostic_kit - resource procedure_equipment - resource regulatory_document - resource consumable_unit - resource schedule_slot - resource specialist_resource - resource diagnostic_device - resource medication_batch - material sample_record - material owner_authorization - material clinical_signature - classification triage_category - classification intervention_bundle - classification animal_age_group - animal_patient case_file_group - animal_patient adult_animal - animal_age_group juvenile_animal - animal_age_group case_file - case_file_group)

  (:predicates
    (animal_registered ?animal_patient - animal_patient)
    (subject_ready_for_intervention ?animal_patient - animal_patient)
    (station_allocated ?animal_patient - animal_patient)
    (discharged_entity ?animal_patient - animal_patient)
    (treatment_performed_on_subject ?animal_patient - animal_patient)
    (release_ready ?animal_patient - animal_patient)
    (station_available ?treatment_station - treatment_station)
    (assigned_to_station ?animal_patient - animal_patient ?treatment_station - treatment_station)
    (diagnostic_kit_available ?diagnostic_kit - diagnostic_kit)
    (diagnostic_kit_assigned ?animal_patient - animal_patient ?diagnostic_kit - diagnostic_kit)
    (equipment_available ?procedure_equipment - procedure_equipment)
    (equipment_allocated ?animal_patient - animal_patient ?procedure_equipment - procedure_equipment)
    (medication_batch_available ?medication_batch - medication_batch)
    (medication_allocated_to_adult ?adult_animal - adult_animal ?medication_batch - medication_batch)
    (medication_allocated_to_juvenile ?juvenile_animal - juvenile_animal ?medication_batch - medication_batch)
    (adult_has_clinical_signature ?adult_animal - adult_animal ?clinical_signature - clinical_signature)
    (clinical_signature_flag_primary ?clinical_signature - clinical_signature)
    (clinical_signature_flag_secondary ?clinical_signature - clinical_signature)
    (adult_ready_for_bundle ?adult_animal - adult_animal)
    (juvenile_has_triage_category ?juvenile_animal - juvenile_animal ?triage_category - triage_category)
    (triage_category_assigned ?triage_category - triage_category)
    (triage_category_secondary_assigned ?triage_category - triage_category)
    (juvenile_ready_for_bundle ?juvenile_animal - juvenile_animal)
    (intervention_bundle_available ?intervention_bundle - intervention_bundle)
    (intervention_bundle_ready ?intervention_bundle - intervention_bundle)
    (bundle_includes_clinical_signature ?intervention_bundle - intervention_bundle ?clinical_signature - clinical_signature)
    (bundle_includes_triage_category ?intervention_bundle - intervention_bundle ?triage_category - triage_category)
    (bundle_flag_consents_required ?intervention_bundle - intervention_bundle)
    (bundle_flag_specialist_required ?intervention_bundle - intervention_bundle)
    (bundle_locked_for_execution ?intervention_bundle - intervention_bundle)
    (case_linked_to_adult ?case_file - case_file ?adult_animal - adult_animal)
    (case_linked_to_juvenile ?case_file - case_file ?juvenile_animal - juvenile_animal)
    (case_linked_to_bundle ?case_file - case_file ?intervention_bundle - intervention_bundle)
    (sample_record_available ?sample_record - sample_record)
    (case_has_sample ?case_file - case_file ?sample_record - sample_record)
    (sample_taken ?sample_record - sample_record)
    (sample_linked_to_bundle ?sample_record - sample_record ?intervention_bundle - intervention_bundle)
    (case_ready_for_specialist_allocation ?case_file - case_file)
    (specialist_resource_allocated ?case_file - case_file)
    (specialist_resource_ready ?case_file - case_file)
    (regulatory_document_attached ?case_file - case_file)
    (specialist_document_verified ?case_file - case_file)
    (specialist_checks_completed ?case_file - case_file)
    (procedure_executed_on_case ?case_file - case_file)
    (owner_authorization_available ?owner_authorization - owner_authorization)
    (case_has_owner_authorization ?case_file - case_file ?owner_authorization - owner_authorization)
    (owner_authorization_attached ?case_file - case_file)
    (equipment_cleared ?case_file - case_file)
    (equipment_verified ?case_file - case_file)
    (regulatory_document_available ?regulatory_document - regulatory_document)
    (case_has_regulatory_document ?case_file - case_file ?regulatory_document - regulatory_document)
    (consumable_unit_available ?consumable_unit - consumable_unit)
    (case_allocated_consumable_unit ?case_file - case_file ?consumable_unit - consumable_unit)
    (specialist_resource_available ?specialist_resource - specialist_resource)
    (allocated_specialist_resource ?case_file - case_file ?specialist_resource - specialist_resource)
    (diagnostic_device_available ?diagnostic_device - diagnostic_device)
    (assigned_diagnostic_device ?case_file - case_file ?diagnostic_device - diagnostic_device)
    (schedule_slot_available ?schedule_slot - schedule_slot)
    (assigned_schedule_slot ?animal_patient - animal_patient ?schedule_slot - schedule_slot)
    (adult_flagged_for_bundle ?adult_animal - adult_animal)
    (juvenile_flagged_for_bundle ?juvenile_animal - juvenile_animal)
    (case_file_finalized ?case_file - case_file)
  )
  (:action register_animal
    :parameters (?animal_patient - animal_patient)
    :precondition
      (and
        (not
          (animal_registered ?animal_patient)
        )
        (not
          (discharged_entity ?animal_patient)
        )
      )
    :effect (animal_registered ?animal_patient)
  )
  (:action assign_treatment_station
    :parameters (?animal_patient - animal_patient ?treatment_station - treatment_station)
    :precondition
      (and
        (animal_registered ?animal_patient)
        (not
          (station_allocated ?animal_patient)
        )
        (station_available ?treatment_station)
      )
    :effect
      (and
        (station_allocated ?animal_patient)
        (assigned_to_station ?animal_patient ?treatment_station)
        (not
          (station_available ?treatment_station)
        )
      )
  )
  (:action assign_diagnostic_kit
    :parameters (?animal_patient - animal_patient ?diagnostic_kit - diagnostic_kit)
    :precondition
      (and
        (animal_registered ?animal_patient)
        (station_allocated ?animal_patient)
        (diagnostic_kit_available ?diagnostic_kit)
      )
    :effect
      (and
        (diagnostic_kit_assigned ?animal_patient ?diagnostic_kit)
        (not
          (diagnostic_kit_available ?diagnostic_kit)
        )
      )
  )
  (:action finalize_diagnostics
    :parameters (?animal_patient - animal_patient ?diagnostic_kit - diagnostic_kit)
    :precondition
      (and
        (animal_registered ?animal_patient)
        (station_allocated ?animal_patient)
        (diagnostic_kit_assigned ?animal_patient ?diagnostic_kit)
        (not
          (subject_ready_for_intervention ?animal_patient)
        )
      )
    :effect (subject_ready_for_intervention ?animal_patient)
  )
  (:action release_diagnostic_kit
    :parameters (?animal_patient - animal_patient ?diagnostic_kit - diagnostic_kit)
    :precondition
      (and
        (diagnostic_kit_assigned ?animal_patient ?diagnostic_kit)
      )
    :effect
      (and
        (diagnostic_kit_available ?diagnostic_kit)
        (not
          (diagnostic_kit_assigned ?animal_patient ?diagnostic_kit)
        )
      )
  )
  (:action allocate_procedure_equipment
    :parameters (?animal_patient - animal_patient ?procedure_equipment - procedure_equipment)
    :precondition
      (and
        (subject_ready_for_intervention ?animal_patient)
        (equipment_available ?procedure_equipment)
      )
    :effect
      (and
        (equipment_allocated ?animal_patient ?procedure_equipment)
        (not
          (equipment_available ?procedure_equipment)
        )
      )
  )
  (:action release_procedure_equipment
    :parameters (?animal_patient - animal_patient ?procedure_equipment - procedure_equipment)
    :precondition
      (and
        (equipment_allocated ?animal_patient ?procedure_equipment)
      )
    :effect
      (and
        (equipment_available ?procedure_equipment)
        (not
          (equipment_allocated ?animal_patient ?procedure_equipment)
        )
      )
  )
  (:action assign_specialist_resource
    :parameters (?case_file - case_file ?specialist_resource - specialist_resource)
    :precondition
      (and
        (subject_ready_for_intervention ?case_file)
        (specialist_resource_available ?specialist_resource)
      )
    :effect
      (and
        (allocated_specialist_resource ?case_file ?specialist_resource)
        (not
          (specialist_resource_available ?specialist_resource)
        )
      )
  )
  (:action release_specialist_resource
    :parameters (?case_file - case_file ?specialist_resource - specialist_resource)
    :precondition
      (and
        (allocated_specialist_resource ?case_file ?specialist_resource)
      )
    :effect
      (and
        (specialist_resource_available ?specialist_resource)
        (not
          (allocated_specialist_resource ?case_file ?specialist_resource)
        )
      )
  )
  (:action assign_diagnostic_device
    :parameters (?case_file - case_file ?diagnostic_device - diagnostic_device)
    :precondition
      (and
        (subject_ready_for_intervention ?case_file)
        (diagnostic_device_available ?diagnostic_device)
      )
    :effect
      (and
        (assigned_diagnostic_device ?case_file ?diagnostic_device)
        (not
          (diagnostic_device_available ?diagnostic_device)
        )
      )
  )
  (:action release_diagnostic_device
    :parameters (?case_file - case_file ?diagnostic_device - diagnostic_device)
    :precondition
      (and
        (assigned_diagnostic_device ?case_file ?diagnostic_device)
      )
    :effect
      (and
        (diagnostic_device_available ?diagnostic_device)
        (not
          (assigned_diagnostic_device ?case_file ?diagnostic_device)
        )
      )
  )
  (:action activate_clinical_signature
    :parameters (?adult_animal - adult_animal ?clinical_signature - clinical_signature ?diagnostic_kit - diagnostic_kit)
    :precondition
      (and
        (subject_ready_for_intervention ?adult_animal)
        (diagnostic_kit_assigned ?adult_animal ?diagnostic_kit)
        (adult_has_clinical_signature ?adult_animal ?clinical_signature)
        (not
          (clinical_signature_flag_primary ?clinical_signature)
        )
        (not
          (clinical_signature_flag_secondary ?clinical_signature)
        )
      )
    :effect (clinical_signature_flag_primary ?clinical_signature)
  )
  (:action confirm_equipment_and_flag_adult
    :parameters (?adult_animal - adult_animal ?clinical_signature - clinical_signature ?procedure_equipment - procedure_equipment)
    :precondition
      (and
        (subject_ready_for_intervention ?adult_animal)
        (equipment_allocated ?adult_animal ?procedure_equipment)
        (adult_has_clinical_signature ?adult_animal ?clinical_signature)
        (clinical_signature_flag_primary ?clinical_signature)
        (not
          (adult_flagged_for_bundle ?adult_animal)
        )
      )
    :effect
      (and
        (adult_flagged_for_bundle ?adult_animal)
        (adult_ready_for_bundle ?adult_animal)
      )
  )
  (:action allocate_medication_to_adult
    :parameters (?adult_animal - adult_animal ?clinical_signature - clinical_signature ?medication_batch - medication_batch)
    :precondition
      (and
        (subject_ready_for_intervention ?adult_animal)
        (adult_has_clinical_signature ?adult_animal ?clinical_signature)
        (medication_batch_available ?medication_batch)
        (not
          (adult_flagged_for_bundle ?adult_animal)
        )
      )
    :effect
      (and
        (clinical_signature_flag_secondary ?clinical_signature)
        (adult_flagged_for_bundle ?adult_animal)
        (medication_allocated_to_adult ?adult_animal ?medication_batch)
        (not
          (medication_batch_available ?medication_batch)
        )
      )
  )
  (:action finalize_adult_medication_allocation
    :parameters (?adult_animal - adult_animal ?clinical_signature - clinical_signature ?diagnostic_kit - diagnostic_kit ?medication_batch - medication_batch)
    :precondition
      (and
        (subject_ready_for_intervention ?adult_animal)
        (diagnostic_kit_assigned ?adult_animal ?diagnostic_kit)
        (adult_has_clinical_signature ?adult_animal ?clinical_signature)
        (clinical_signature_flag_secondary ?clinical_signature)
        (medication_allocated_to_adult ?adult_animal ?medication_batch)
        (not
          (adult_ready_for_bundle ?adult_animal)
        )
      )
    :effect
      (and
        (clinical_signature_flag_primary ?clinical_signature)
        (adult_ready_for_bundle ?adult_animal)
        (medication_batch_available ?medication_batch)
        (not
          (medication_allocated_to_adult ?adult_animal ?medication_batch)
        )
      )
  )
  (:action assign_triage_category_to_juvenile
    :parameters (?juvenile_animal - juvenile_animal ?triage_category - triage_category ?diagnostic_kit - diagnostic_kit)
    :precondition
      (and
        (subject_ready_for_intervention ?juvenile_animal)
        (diagnostic_kit_assigned ?juvenile_animal ?diagnostic_kit)
        (juvenile_has_triage_category ?juvenile_animal ?triage_category)
        (not
          (triage_category_assigned ?triage_category)
        )
        (not
          (triage_category_secondary_assigned ?triage_category)
        )
      )
    :effect (triage_category_assigned ?triage_category)
  )
  (:action confirm_equipment_for_juvenile
    :parameters (?juvenile_animal - juvenile_animal ?triage_category - triage_category ?procedure_equipment - procedure_equipment)
    :precondition
      (and
        (subject_ready_for_intervention ?juvenile_animal)
        (equipment_allocated ?juvenile_animal ?procedure_equipment)
        (juvenile_has_triage_category ?juvenile_animal ?triage_category)
        (triage_category_assigned ?triage_category)
        (not
          (juvenile_flagged_for_bundle ?juvenile_animal)
        )
      )
    :effect
      (and
        (juvenile_flagged_for_bundle ?juvenile_animal)
        (juvenile_ready_for_bundle ?juvenile_animal)
      )
  )
  (:action allocate_medication_to_juvenile
    :parameters (?juvenile_animal - juvenile_animal ?triage_category - triage_category ?medication_batch - medication_batch)
    :precondition
      (and
        (subject_ready_for_intervention ?juvenile_animal)
        (juvenile_has_triage_category ?juvenile_animal ?triage_category)
        (medication_batch_available ?medication_batch)
        (not
          (juvenile_flagged_for_bundle ?juvenile_animal)
        )
      )
    :effect
      (and
        (triage_category_secondary_assigned ?triage_category)
        (juvenile_flagged_for_bundle ?juvenile_animal)
        (medication_allocated_to_juvenile ?juvenile_animal ?medication_batch)
        (not
          (medication_batch_available ?medication_batch)
        )
      )
  )
  (:action finalize_juvenile_medication_allocation
    :parameters (?juvenile_animal - juvenile_animal ?triage_category - triage_category ?diagnostic_kit - diagnostic_kit ?medication_batch - medication_batch)
    :precondition
      (and
        (subject_ready_for_intervention ?juvenile_animal)
        (diagnostic_kit_assigned ?juvenile_animal ?diagnostic_kit)
        (juvenile_has_triage_category ?juvenile_animal ?triage_category)
        (triage_category_secondary_assigned ?triage_category)
        (medication_allocated_to_juvenile ?juvenile_animal ?medication_batch)
        (not
          (juvenile_ready_for_bundle ?juvenile_animal)
        )
      )
    :effect
      (and
        (triage_category_assigned ?triage_category)
        (juvenile_ready_for_bundle ?juvenile_animal)
        (medication_batch_available ?medication_batch)
        (not
          (medication_allocated_to_juvenile ?juvenile_animal ?medication_batch)
        )
      )
  )
  (:action assemble_intervention_bundle
    :parameters (?adult_animal - adult_animal ?juvenile_animal - juvenile_animal ?clinical_signature - clinical_signature ?triage_category - triage_category ?intervention_bundle - intervention_bundle)
    :precondition
      (and
        (adult_flagged_for_bundle ?adult_animal)
        (juvenile_flagged_for_bundle ?juvenile_animal)
        (adult_has_clinical_signature ?adult_animal ?clinical_signature)
        (juvenile_has_triage_category ?juvenile_animal ?triage_category)
        (clinical_signature_flag_primary ?clinical_signature)
        (triage_category_assigned ?triage_category)
        (adult_ready_for_bundle ?adult_animal)
        (juvenile_ready_for_bundle ?juvenile_animal)
        (intervention_bundle_available ?intervention_bundle)
      )
    :effect
      (and
        (intervention_bundle_ready ?intervention_bundle)
        (bundle_includes_clinical_signature ?intervention_bundle ?clinical_signature)
        (bundle_includes_triage_category ?intervention_bundle ?triage_category)
        (not
          (intervention_bundle_available ?intervention_bundle)
        )
      )
  )
  (:action assemble_bundle_with_secondary_signature
    :parameters (?adult_animal - adult_animal ?juvenile_animal - juvenile_animal ?clinical_signature - clinical_signature ?triage_category - triage_category ?intervention_bundle - intervention_bundle)
    :precondition
      (and
        (adult_flagged_for_bundle ?adult_animal)
        (juvenile_flagged_for_bundle ?juvenile_animal)
        (adult_has_clinical_signature ?adult_animal ?clinical_signature)
        (juvenile_has_triage_category ?juvenile_animal ?triage_category)
        (clinical_signature_flag_secondary ?clinical_signature)
        (triage_category_assigned ?triage_category)
        (not
          (adult_ready_for_bundle ?adult_animal)
        )
        (juvenile_ready_for_bundle ?juvenile_animal)
        (intervention_bundle_available ?intervention_bundle)
      )
    :effect
      (and
        (intervention_bundle_ready ?intervention_bundle)
        (bundle_includes_clinical_signature ?intervention_bundle ?clinical_signature)
        (bundle_includes_triage_category ?intervention_bundle ?triage_category)
        (bundle_flag_consents_required ?intervention_bundle)
        (not
          (intervention_bundle_available ?intervention_bundle)
        )
      )
  )
  (:action assemble_bundle_with_specialist_flag
    :parameters (?adult_animal - adult_animal ?juvenile_animal - juvenile_animal ?clinical_signature - clinical_signature ?triage_category - triage_category ?intervention_bundle - intervention_bundle)
    :precondition
      (and
        (adult_flagged_for_bundle ?adult_animal)
        (juvenile_flagged_for_bundle ?juvenile_animal)
        (adult_has_clinical_signature ?adult_animal ?clinical_signature)
        (juvenile_has_triage_category ?juvenile_animal ?triage_category)
        (clinical_signature_flag_primary ?clinical_signature)
        (triage_category_secondary_assigned ?triage_category)
        (adult_ready_for_bundle ?adult_animal)
        (not
          (juvenile_ready_for_bundle ?juvenile_animal)
        )
        (intervention_bundle_available ?intervention_bundle)
      )
    :effect
      (and
        (intervention_bundle_ready ?intervention_bundle)
        (bundle_includes_clinical_signature ?intervention_bundle ?clinical_signature)
        (bundle_includes_triage_category ?intervention_bundle ?triage_category)
        (bundle_flag_specialist_required ?intervention_bundle)
        (not
          (intervention_bundle_available ?intervention_bundle)
        )
      )
  )
  (:action assemble_bundle_with_both_flags
    :parameters (?adult_animal - adult_animal ?juvenile_animal - juvenile_animal ?clinical_signature - clinical_signature ?triage_category - triage_category ?intervention_bundle - intervention_bundle)
    :precondition
      (and
        (adult_flagged_for_bundle ?adult_animal)
        (juvenile_flagged_for_bundle ?juvenile_animal)
        (adult_has_clinical_signature ?adult_animal ?clinical_signature)
        (juvenile_has_triage_category ?juvenile_animal ?triage_category)
        (clinical_signature_flag_secondary ?clinical_signature)
        (triage_category_secondary_assigned ?triage_category)
        (not
          (adult_ready_for_bundle ?adult_animal)
        )
        (not
          (juvenile_ready_for_bundle ?juvenile_animal)
        )
        (intervention_bundle_available ?intervention_bundle)
      )
    :effect
      (and
        (intervention_bundle_ready ?intervention_bundle)
        (bundle_includes_clinical_signature ?intervention_bundle ?clinical_signature)
        (bundle_includes_triage_category ?intervention_bundle ?triage_category)
        (bundle_flag_consents_required ?intervention_bundle)
        (bundle_flag_specialist_required ?intervention_bundle)
        (not
          (intervention_bundle_available ?intervention_bundle)
        )
      )
  )
  (:action lock_bundle_for_execution
    :parameters (?intervention_bundle - intervention_bundle ?adult_animal - adult_animal ?diagnostic_kit - diagnostic_kit)
    :precondition
      (and
        (intervention_bundle_ready ?intervention_bundle)
        (adult_flagged_for_bundle ?adult_animal)
        (diagnostic_kit_assigned ?adult_animal ?diagnostic_kit)
        (not
          (bundle_locked_for_execution ?intervention_bundle)
        )
      )
    :effect (bundle_locked_for_execution ?intervention_bundle)
  )
  (:action reserve_and_link_sample_to_bundle
    :parameters (?case_file - case_file ?sample_record - sample_record ?intervention_bundle - intervention_bundle)
    :precondition
      (and
        (subject_ready_for_intervention ?case_file)
        (case_linked_to_bundle ?case_file ?intervention_bundle)
        (case_has_sample ?case_file ?sample_record)
        (sample_record_available ?sample_record)
        (intervention_bundle_ready ?intervention_bundle)
        (bundle_locked_for_execution ?intervention_bundle)
        (not
          (sample_taken ?sample_record)
        )
      )
    :effect
      (and
        (sample_taken ?sample_record)
        (sample_linked_to_bundle ?sample_record ?intervention_bundle)
        (not
          (sample_record_available ?sample_record)
        )
      )
  )
  (:action prepare_case_for_specialist_allocation
    :parameters (?case_file - case_file ?sample_record - sample_record ?intervention_bundle - intervention_bundle ?diagnostic_kit - diagnostic_kit)
    :precondition
      (and
        (subject_ready_for_intervention ?case_file)
        (case_has_sample ?case_file ?sample_record)
        (sample_taken ?sample_record)
        (sample_linked_to_bundle ?sample_record ?intervention_bundle)
        (diagnostic_kit_assigned ?case_file ?diagnostic_kit)
        (not
          (bundle_flag_consents_required ?intervention_bundle)
        )
        (not
          (case_ready_for_specialist_allocation ?case_file)
        )
      )
    :effect (case_ready_for_specialist_allocation ?case_file)
  )
  (:action attach_regulatory_document_to_case
    :parameters (?case_file - case_file ?regulatory_document - regulatory_document)
    :precondition
      (and
        (subject_ready_for_intervention ?case_file)
        (regulatory_document_available ?regulatory_document)
        (not
          (regulatory_document_attached ?case_file)
        )
      )
    :effect
      (and
        (regulatory_document_attached ?case_file)
        (case_has_regulatory_document ?case_file ?regulatory_document)
        (not
          (regulatory_document_available ?regulatory_document)
        )
      )
  )
  (:action verify_regulatory_and_ready_case
    :parameters (?case_file - case_file ?sample_record - sample_record ?intervention_bundle - intervention_bundle ?diagnostic_kit - diagnostic_kit ?regulatory_document - regulatory_document)
    :precondition
      (and
        (subject_ready_for_intervention ?case_file)
        (case_has_sample ?case_file ?sample_record)
        (sample_taken ?sample_record)
        (sample_linked_to_bundle ?sample_record ?intervention_bundle)
        (diagnostic_kit_assigned ?case_file ?diagnostic_kit)
        (bundle_flag_consents_required ?intervention_bundle)
        (regulatory_document_attached ?case_file)
        (case_has_regulatory_document ?case_file ?regulatory_document)
        (not
          (case_ready_for_specialist_allocation ?case_file)
        )
      )
    :effect
      (and
        (case_ready_for_specialist_allocation ?case_file)
        (specialist_document_verified ?case_file)
      )
  )
  (:action assign_specialist_and_confirm
    :parameters (?case_file - case_file ?specialist_resource - specialist_resource ?procedure_equipment - procedure_equipment ?sample_record - sample_record ?intervention_bundle - intervention_bundle)
    :precondition
      (and
        (case_ready_for_specialist_allocation ?case_file)
        (allocated_specialist_resource ?case_file ?specialist_resource)
        (equipment_allocated ?case_file ?procedure_equipment)
        (case_has_sample ?case_file ?sample_record)
        (sample_linked_to_bundle ?sample_record ?intervention_bundle)
        (not
          (bundle_flag_specialist_required ?intervention_bundle)
        )
        (not
          (specialist_resource_allocated ?case_file)
        )
      )
    :effect (specialist_resource_allocated ?case_file)
  )
  (:action assign_specialist_and_confirm_variant
    :parameters (?case_file - case_file ?specialist_resource - specialist_resource ?procedure_equipment - procedure_equipment ?sample_record - sample_record ?intervention_bundle - intervention_bundle)
    :precondition
      (and
        (case_ready_for_specialist_allocation ?case_file)
        (allocated_specialist_resource ?case_file ?specialist_resource)
        (equipment_allocated ?case_file ?procedure_equipment)
        (case_has_sample ?case_file ?sample_record)
        (sample_linked_to_bundle ?sample_record ?intervention_bundle)
        (bundle_flag_specialist_required ?intervention_bundle)
        (not
          (specialist_resource_allocated ?case_file)
        )
      )
    :effect (specialist_resource_allocated ?case_file)
