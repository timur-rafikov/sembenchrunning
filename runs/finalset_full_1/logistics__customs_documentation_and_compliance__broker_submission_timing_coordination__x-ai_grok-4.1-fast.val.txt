    (export_permit_attached ?export_consignment - export_consignment ?permit_record - permit_record)
    (import_permit_attached ?import_consignment - import_consignment ?permit_record - permit_record)
    (has_origin_movement_segment ?export_consignment - export_consignment ?origin_movement_segment - origin_movement_segment)
    (origin_segment_verified ?origin_movement_segment - origin_movement_segment)
    (origin_segment_requires_permit ?origin_movement_segment - origin_movement_segment)

    (submission_has_destination_segment ?submission_record - submission_record ?destination_movement_segment - destination_movement_segment)
    (submission_origin_segment_verified ?submission_record - submission_record)
    (submission_destination_segment_verified ?submission_record - submission_record)

    (broker_file_for_import_consignment ?broker_file - broker_file ?import_consignment - import_consignment)

    (supporting_document_available ?supporting_document - supporting_document)

  (:action register_consignment_case
    :parameters (?consignment_case - consignment_case)
    :precondition
      (and
        (not
          (case_registered ?consignment_case)
        )
        (not
          (submission_recorded ?consignment_case)
        )
      )
    :effect (case_registered ?consignment_case)
  )
  (:action assign_broker_resource_to_case
    :parameters (?consignment_case - consignment_case ?broker_resource - broker_resource)
    :precondition
      (and
        (case_registered ?consignment_case)
        (not
          (broker_assigned ?consignment_case)
        )
        (broker_resource_available ?broker_resource)
      )
    :effect
      (and
        (broker_assigned ?consignment_case)
        (assigned_broker_resource ?consignment_case ?broker_resource)
        (not
          (broker_resource_available ?broker_resource)
        )
      )
  )
  (:action assign_classification_service_to_case
    :parameters (?consignment_case - consignment_case ?classification_service - classification_service)
    :precondition
      (and
        (case_registered ?consignment_case)
        (broker_assigned ?consignment_case)
        (classification_service_available ?classification_service)
      )
    :effect
      (and
        (classification_assigned_to_case ?consignment_case ?classification_service)
        (not
          (classification_service_available ?classification_service)
        )
      )
  )
  (:action finalize_classification_for_case
    :parameters (?consignment_case - consignment_case ?classification_service - classification_service)
    :precondition
      (and
        (case_registered ?consignment_case)
        (broker_assigned ?consignment_case)
        (classification_assigned_to_case ?consignment_case ?classification_service)
        (not
          (classification_completed ?consignment_case)
        )
      )
    :effect (classification_completed ?consignment_case)
  )
  (:action unassign_classification_service_from_case
    :parameters (?consignment_case - consignment_case ?classification_service - classification_service)
    :precondition
      (and
        (classification_assigned_to_case ?consignment_case ?classification_service)
      )
    :effect
      (and
        (classification_service_available ?classification_service)
        (not
          (classification_assigned_to_case ?consignment_case ?classification_service)
        )
      )
  )
  (:action assign_compliance_officer_to_case
    :parameters (?consignment_case - consignment_case ?compliance_officer - compliance_officer)
    :precondition
      (and
        (classification_completed ?consignment_case)
        (compliance_officer_available ?compliance_officer)
      )
    :effect
      (and
        (compliance_officer_assigned ?consignment_case ?compliance_officer)
        (not
          (compliance_officer_available ?compliance_officer)
        )
      )
  )
  (:action release_compliance_officer_from_case
    :parameters (?consignment_case - consignment_case ?compliance_officer - compliance_officer)
    :precondition
      (and
        (compliance_officer_assigned ?consignment_case ?compliance_officer)
      )
    :effect
      (and
        (compliance_officer_available ?compliance_officer)
        (not
          (compliance_officer_assigned ?consignment_case ?compliance_officer)
        )
      )
  )
  (:action attach_commercial_document_to_broker_file
    :parameters (?broker_file - broker_file ?commercial_document - commercial_document)
    :precondition
      (and
        (classification_completed ?broker_file)
        (commercial_document_available ?commercial_document)
      )
    :effect
      (and
        (broker_file_commercial_document_attached ?broker_file ?commercial_document)

      )

    :parameters (?broker_file - broker_file ?commercial_document - commercial_document)

      (and
        (broker_file_commercial_document_attached ?broker_file ?commercial_document)

    :effect
      (and
        (commercial_document_available ?commercial_document)
        (not
          (broker_file_commercial_document_attached ?broker_file ?commercial_document)
        )
      )
  )
  (:action link_inspection_authority_to_broker_file
    :parameters (?broker_file - broker_file ?inspection_authority - inspection_authority)
    :precondition
      (and
        (classification_completed ?broker_file)
        (inspection_authority_available ?inspection_authority)
      )
    :effect
      (and
        (broker_file_inspection_authority_linked ?broker_file ?inspection_authority)
        (not
          (inspection_authority_available ?inspection_authority)
        )
      )
  )
  (:action unlink_inspection_authority_from_broker_file
    :parameters (?broker_file - broker_file ?inspection_authority - inspection_authority)
    :precondition
      (and
        (broker_file_inspection_authority_linked ?broker_file ?inspection_authority)
      )
    :effect
      (and
        (inspection_authority_available ?inspection_authority)
        (not
          (broker_file_inspection_authority_linked ?broker_file ?inspection_authority)
        )
      )
  )
  (:action mark_origin_movement_segment_verified
    :parameters (?export_consignment - export_consignment ?origin_movement_segment - origin_movement_segment ?classification_service - classification_service)
    :precondition
      (and
        (classification_completed ?export_consignment)
        (classification_assigned_to_case ?export_consignment ?classification_service)
        (has_origin_movement_segment ?export_consignment ?origin_movement_segment)
        (not
          (origin_segment_verified ?origin_movement_segment)
        )
        (not
          (origin_segment_requires_permit ?origin_movement_segment)
        )
      )
    :effect (origin_segment_verified ?origin_movement_segment)
  )
  (:action complete_origin_verification_with_officer
    :parameters (?export_consignment - export_consignment ?origin_movement_segment - origin_movement_segment ?compliance_officer - compliance_officer)
    :precondition
      (and
        (classification_completed ?export_consignment)
        (compliance_officer_assigned ?export_consignment ?compliance_officer)
        (has_origin_movement_segment ?export_consignment ?origin_movement_segment)
        (origin_segment_verified ?origin_movement_segment)
        (not
          (origin_eligibility_flag ?export_consignment)
        )
      )
    :effect
      (and
        (origin_eligibility_flag ?export_consignment)
        (origin_verification_complete ?export_consignment)
      )
  )
  (:action attach_permit_record_to_origin_segment
    :parameters (?export_consignment - export_consignment ?origin_movement_segment - origin_movement_segment ?permit_record - permit_record)

      (and
        (classification_completed ?export_consignment)
        (has_origin_movement_segment ?export_consignment ?origin_movement_segment)
        (permit_record_available ?permit_record)
        (not
          (origin_eligibility_flag ?export_consignment)
        )
      )
    :effect
      (and
        (origin_segment_requires_permit ?origin_movement_segment)
        (origin_eligibility_flag ?export_consignment)
        (export_permit_attached ?export_consignment ?permit_record)
        (not
          (permit_record_available ?permit_record)
        )
      )
  )
  (:action reconcile_origin_permit_and_verification

      (and
        (classification_completed ?export_consignment)
        (classification_assigned_to_case ?export_consignment ?classification_service)
        (has_origin_movement_segment ?export_consignment ?origin_movement_segment)
        (origin_segment_requires_permit ?origin_movement_segment)
        (export_permit_attached ?export_consignment ?permit_record)
        (not
          (origin_verification_complete ?export_consignment)
        )
      )
    :effect
      (and
        (origin_segment_verified ?origin_movement_segment)
        (origin_verification_complete ?export_consignment)
        (permit_record_available ?permit_record)
        (not
          (export_permit_attached ?export_consignment ?permit_record)
        )
      )
  )

      (and
        (classification_completed ?import_consignment)

  (:action attach_permit_record_to_destination_segment
    :parameters (?import_consignment - import_consignment ?destination_movement_segment - destination_movement_segment ?permit_record - permit_record)

      (and

    :parameters (?import_consignment - import_consignment ?destination_movement_segment - destination_movement_segment ?classification_service - classification_service ?permit_record - permit_record)
    :precondition
      (and
        (classification_completed ?import_consignment)
        (classification_assigned_to_case ?import_consignment ?classification_service)
        (has_destination_movement_segment ?import_consignment ?destination_movement_segment)
        (destination_segment_requires_permit ?destination_movement_segment)
        (import_permit_attached ?import_consignment ?permit_record)
        (not
          (destination_verification_complete ?import_consignment)

    :effect
      (and
        (destination_segment_verified ?destination_movement_segment)
        (destination_verification_complete ?import_consignment)

        (not
          (import_permit_attached ?import_consignment ?permit_record)

  (:action reserve_submission_record_with_segments
    :parameters (?export_consignment - export_consignment ?import_consignment - import_consignment ?origin_movement_segment - origin_movement_segment ?destination_movement_segment - destination_movement_segment ?submission_record - submission_record)
    :precondition
      (and
        (origin_eligibility_flag ?export_consignment)
        (destination_eligibility_flag ?import_consignment)
        (has_origin_movement_segment ?export_consignment ?origin_movement_segment)
        (has_destination_movement_segment ?import_consignment ?destination_movement_segment)
        (origin_segment_verified ?origin_movement_segment)
        (destination_segment_verified ?destination_movement_segment)
        (origin_verification_complete ?export_consignment)
        (destination_verification_complete ?import_consignment)
        (submission_record_available ?submission_record)
      )
    :effect
      (and
        (submission_record_reserved ?submission_record)
        (submission_has_origin_segment ?submission_record ?origin_movement_segment)

        (not
          (submission_record_available ?submission_record)
        )
      )
  )
  (:action reserve_submission_record_with_origin_verified

        (has_destination_movement_segment ?import_consignment ?destination_movement_segment)
        (origin_segment_requires_permit ?origin_movement_segment)
        (destination_segment_verified ?destination_movement_segment)
        (not
          (origin_verification_complete ?export_consignment)

        (destination_verification_complete ?import_consignment)
        (submission_record_available ?submission_record)
      )
    :effect
      (and
        (submission_record_reserved ?submission_record)
        (submission_has_origin_segment ?submission_record ?origin_movement_segment)
        (submission_has_destination_segment ?submission_record ?destination_movement_segment)
        (submission_origin_segment_verified ?submission_record)
        (not
          (submission_record_available ?submission_record)
        )
      )
  )
  (:action reserve_submission_record_with_destination_verified

    :precondition
      (and
        (origin_eligibility_flag ?export_consignment)
        (destination_eligibility_flag ?import_consignment)
        (has_origin_movement_segment ?export_consignment ?origin_movement_segment)
        (has_destination_movement_segment ?import_consignment ?destination_movement_segment)

          (destination_verification_complete ?import_consignment)

        (submission_record_available ?submission_record)

    :effect
      (and
        (submission_record_reserved ?submission_record)
        (submission_has_origin_segment ?submission_record ?origin_movement_segment)
        (submission_has_destination_segment ?submission_record ?destination_movement_segment)
        (submission_destination_segment_verified ?submission_record)
        (not
          (submission_record_available ?submission_record)
        )
      )
  )

        (submission_destination_segment_verified ?submission_record)
        (not
          (submission_record_available ?submission_record)

      )
  )

  )

      (and
        (supporting_document_attached ?supporting_document)
        (supporting_document_linked_to_submission ?supporting_document ?submission_record)

          (supporting_document_available ?supporting_document)
        )
      )
  )

          (submission_origin_segment_verified ?submission_record)

          (broker_file_documents_verified ?broker_file)

  (:action attach_regulatory_reference_to_broker_file
    :parameters (?broker_file - broker_file ?regulatory_reference - regulatory_reference)
    :precondition

        (not
          (broker_file_has_regulatory_reference ?broker_file)
        )

    :effect
      (and
        (broker_file_has_regulatory_reference ?broker_file)
        (broker_file_regulatory_reference_linked ?broker_file ?regulatory_reference)
        (not
          (regulatory_reference_available ?regulatory_reference)
        )
      )
  )
  (:action verify_supporting_document_and_regulatory_reference
    :parameters (?broker_file - broker_file ?supporting_document - supporting_document ?submission_record - submission_record ?classification_service - classification_service ?regulatory_reference - regulatory_reference)
    :precondition
      (and
        (classification_completed ?broker_file)
        (broker_file_has_supporting_document ?broker_file ?supporting_document)
