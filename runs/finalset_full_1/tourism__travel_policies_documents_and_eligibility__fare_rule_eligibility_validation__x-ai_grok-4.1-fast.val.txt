--- domain.pddl (as passed to VAL) ---
(define (domain fare_rule_eligibility_validation_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity_base - object resource_category - entity_base requirement_category - entity_base classification_category - entity_base case_root - entity_base case_entity - case_root document_instance - resource_category required_document_type - resource_category verifying_authority - resource_category fare_class - resource_category concession_profile - resource_category certificate_type - resource_category endorsement_type - resource_category policy_exception - resource_category supporting_evidence - requirement_category document_template - requirement_category issuing_agency - requirement_category jurisdiction - classification_category jurisdiction_requirement_profile - classification_category fare_option - classification_category passenger_case_group - case_entity rule_case_group - case_entity primary_itinerary - passenger_case_group companion_itinerary - passenger_case_group fare_rule_case - rule_case_group)

  (:predicates
    (case_registered ?case_entity - case_entity)
    (case_documents_verified ?case_entity - case_entity)
    (case_document_attached ?case_entity - case_entity)
    (eligibility_confirmed ?case_entity - case_entity)
    (provisional_eligibility ?case_entity - case_entity)
    (certificate_consumed ?case_entity - case_entity)
    (document_available ?document_instance - document_instance)
    (case_has_document ?case_entity - case_entity ?document_instance - document_instance)
    (document_type_available ?document_type_requirement - required_document_type)
    (case_matches_document_type_requirement ?case_entity - case_entity ?document_type_requirement - required_document_type)
    (authority_available ?verifying_authority - verifying_authority)
    (case_assigned_authority ?case_entity - case_entity ?verifying_authority - verifying_authority)
    (evidence_available ?supporting_evidence - supporting_evidence)
    (itinerary_evidence_link ?primary_itinerary - primary_itinerary ?supporting_evidence - supporting_evidence)
    (companion_evidence_link ?companion_itinerary - companion_itinerary ?supporting_evidence - supporting_evidence)
    (itinerary_jurisdiction_link ?primary_itinerary - primary_itinerary ?jurisdiction - jurisdiction)
    (jurisdiction_document_match_flag ?jurisdiction - jurisdiction)
    (jurisdiction_evidence_flag ?jurisdiction - jurisdiction)
    (primary_itinerary_requirement_satisfied ?primary_itinerary - primary_itinerary)
    (companion_jurisdiction_profile_link ?companion_itinerary - companion_itinerary ?jurisdiction_requirement_profile - jurisdiction_requirement_profile)
    (jurisdiction_profile_primary_satisfied ?jurisdiction_requirement_profile - jurisdiction_requirement_profile)
    (jurisdiction_profile_evidence_satisfied ?jurisdiction_requirement_profile - jurisdiction_requirement_profile)
    (companion_itinerary_requirement_satisfied ?companion_itinerary - companion_itinerary)
    (fare_option_available ?fare_option - fare_option)
    (fare_candidate_selected ?fare_option - fare_option)
    (fare_option_jurisdiction_link ?fare_option - fare_option ?jurisdiction - jurisdiction)
    (fare_option_profile_link ?fare_option - fare_option ?jurisdiction_requirement_profile - jurisdiction_requirement_profile)
    (fare_option_primary_profile_restriction ?fare_option - fare_option)
    (fare_option_companion_profile_restriction ?fare_option - fare_option)
    (fare_option_ready_for_template_matching ?fare_option - fare_option)
    (fare_rule_matches_primary_itinerary ?fare_rule_case - fare_rule_case ?primary_itinerary - primary_itinerary)
    (fare_rule_matches_companion_itinerary ?fare_rule_case - fare_rule_case ?companion_itinerary - companion_itinerary)
    (fare_rule_associated_fare_option ?fare_rule_case - fare_rule_case ?fare_option - fare_option)
    (document_template_available ?document_template - document_template)
    (fare_rule_template_match ?fare_rule_case - fare_rule_case ?document_template - document_template)
    (document_template_consumed ?document_template - document_template)
    (document_template_matches_fare_option ?document_template - document_template ?fare_option - fare_option)
    (fare_rule_template_matching_complete ?fare_rule_case - fare_rule_case)
    (fare_rule_endorsement_assigned ?fare_rule_case - fare_rule_case)
    (fare_rule_scoring_complete ?fare_rule_case - fare_rule_case)
    (fare_rule_has_assigned_fare_class ?fare_rule_case - fare_rule_case)
    (fare_rule_template_and_class_linked ?fare_rule_case - fare_rule_case)
    (fare_rule_concession_attached ?fare_rule_case - fare_rule_case)
    (fare_rule_scored ?fare_rule_case - fare_rule_case)
    (issuing_agency_available ?issuing_agency - issuing_agency)
    (fare_rule_assigned_issuing_agency ?fare_rule_case - fare_rule_case ?issuing_agency - issuing_agency)
    (fare_rule_alternative_agency_accepted ?fare_rule_case - fare_rule_case)
    (fare_rule_alternative_authority_assigned ?fare_rule_case - fare_rule_case)
    (fare_rule_alternative_path_ready ?fare_rule_case - fare_rule_case)
    (fare_class_available ?fare_class - fare_class)
    (fare_rule_assigned_fare_class ?fare_rule_case - fare_rule_case ?fare_class - fare_class)
    (concession_profile_available ?concession_profile - concession_profile)
    (fare_rule_assigned_concession_profile ?fare_rule_case - fare_rule_case ?concession_profile - concession_profile)
    (endorsement_type_available ?endorsement_type - endorsement_type)
    (fare_rule_assigned_endorsement ?fare_rule_case - fare_rule_case ?endorsement_type - endorsement_type)
    (policy_exception_available ?policy_exception - policy_exception)
    (fare_rule_assigned_policy_exception ?fare_rule_case - fare_rule_case ?policy_exception - policy_exception)
    (certificate_available ?certificate_type - certificate_type)
    (case_certificate_link ?case_entity - case_entity ?certificate_type - certificate_type)
    (primary_itinerary_checks_completed ?primary_itinerary - primary_itinerary)
    (companion_itinerary_checks_completed ?companion_itinerary - companion_itinerary)
    (policy_decision_recorded ?fare_rule_case - fare_rule_case)
  )
  (:action register_validation_case
    :parameters (?case_entity - case_entity)
    :precondition
      (and
        (not
          (case_registered ?case_entity)
        )
        (not
          (eligibility_confirmed ?case_entity)
        )
      )
    :effect (case_registered ?case_entity)
  )
  (:action attach_document_to_case
    :parameters (?case_entity - case_entity ?document_instance - document_instance)
    :precondition
      (and
        (case_registered ?case_entity)
        (not
          (case_document_attached ?case_entity)
        )
        (document_available ?document_instance)
      )
    :effect
      (and
        (case_document_attached ?case_entity)
        (case_has_document ?case_entity ?document_instance)
        (not
          (document_available ?document_instance)
        )
      )
  )
  (:action match_case_with_doc_requirement
    :parameters (?case_entity - case_entity ?document_type_requirement - required_document_type)
    :precondition
      (and
        (case_registered ?case_entity)
        (case_document_attached ?case_entity)
        (document_type_available ?document_type_requirement)
      )
    :effect
      (and
        (case_matches_document_type_requirement ?case_entity ?document_type_requirement)
        (not
          (document_type_available ?document_type_requirement)
        )
      )
  )
  (:action confirm_document_verification
    :parameters (?case_entity - case_entity ?document_type_requirement - required_document_type)
    :precondition
      (and
        (case_registered ?case_entity)
        (case_document_attached ?case_entity)
        (case_matches_document_type_requirement ?case_entity ?document_type_requirement)
        (not
          (case_documents_verified ?case_entity)
        )
      )
    :effect (case_documents_verified ?case_entity)
  )
  (:action unmatch_case_doc_requirement
    :parameters (?case_entity - case_entity ?document_type_requirement - required_document_type)
    :precondition
      (and
        (case_matches_document_type_requirement ?case_entity ?document_type_requirement)
      )
    :effect
      (and
        (document_type_available ?document_type_requirement)
        (not
          (case_matches_document_type_requirement ?case_entity ?document_type_requirement)
        )
      )
  )
  (:action assign_verifying_authority_to_case
    :parameters (?case_entity - case_entity ?verifying_authority - verifying_authority)
    :precondition
      (and
        (case_documents_verified ?case_entity)
        (authority_available ?verifying_authority)
      )
    :effect
      (and
        (case_assigned_authority ?case_entity ?verifying_authority)
        (not
          (authority_available ?verifying_authority)
        )
      )
  )
  (:action unassign_verifying_authority_from_case
    :parameters (?case_entity - case_entity ?verifying_authority - verifying_authority)
    :precondition
      (and
        (case_assigned_authority ?case_entity ?verifying_authority)
      )
    :effect
      (and
        (authority_available ?verifying_authority)
        (not
          (case_assigned_authority ?case_entity ?verifying_authority)
        )
      )
  )
  (:action assign_endorsement_to_fare_rule_case
    :parameters (?fare_rule_case - fare_rule_case ?endorsement_type - endorsement_type)
    :precondition
      (and
        (case_documents_verified ?fare_rule_case)
        (endorsement_type_available ?endorsement_type)
      )
    :effect
      (and
        (fare_rule_assigned_endorsement ?fare_rule_case ?endorsement_type)
        (not
          (endorsement_type_available ?endorsement_type)
        )
      )
  )
  (:action unassign_endorsement_from_fare_rule_case
    :parameters (?fare_rule_case - fare_rule_case ?endorsement_type - endorsement_type)
    :precondition
      (and
        (fare_rule_assigned_endorsement ?fare_rule_case ?endorsement_type)
      )
    :effect
      (and
        (endorsement_type_available ?endorsement_type)
        (not
          (fare_rule_assigned_endorsement ?fare_rule_case ?endorsement_type)
        )
      )
  )
  (:action attach_policy_exception_to_fare_rule_case
    :parameters (?fare_rule_case - fare_rule_case ?policy_exception - policy_exception)
    :precondition
      (and
        (case_documents_verified ?fare_rule_case)
        (policy_exception_available ?policy_exception)
      )
    :effect
      (and
        (fare_rule_assigned_policy_exception ?fare_rule_case ?policy_exception)
        (not
          (policy_exception_available ?policy_exception)
        )
      )
  )
  (:action detach_policy_exception_from_fare_rule_case
    :parameters (?fare_rule_case - fare_rule_case ?policy_exception - policy_exception)
    :precondition
      (and
        (fare_rule_assigned_policy_exception ?fare_rule_case ?policy_exception)
      )
    :effect
      (and
        (policy_exception_available ?policy_exception)
        (not
          (fare_rule_assigned_policy_exception ?fare_rule_case ?policy_exception)
        )
      )
  )
  (:action satisfy_jurisdiction_primary_requirement
    :parameters (?primary_itinerary - primary_itinerary ?jurisdiction - jurisdiction ?document_type_requirement - required_document_type)
    :precondition
      (and
        (case_documents_verified ?primary_itinerary)
        (case_matches_document_type_requirement ?primary_itinerary ?document_type_requirement)
        (itinerary_jurisdiction_link ?primary_itinerary ?jurisdiction)
        (not
          (jurisdiction_document_match_flag ?jurisdiction)
        )
        (not
          (jurisdiction_evidence_flag ?jurisdiction)
        )
      )
    :effect (jurisdiction_document_match_flag ?jurisdiction)
  )
  (:action apply_authority_verification_to_primary_itinerary
    :parameters (?primary_itinerary - primary_itinerary ?jurisdiction - jurisdiction ?verifying_authority - verifying_authority)
    :precondition
      (and
        (case_documents_verified ?primary_itinerary)
        (case_assigned_authority ?primary_itinerary ?verifying_authority)
        (itinerary_jurisdiction_link ?primary_itinerary ?jurisdiction)
        (jurisdiction_document_match_flag ?jurisdiction)
        (not
          (primary_itinerary_checks_completed ?primary_itinerary)
        )
      )
    :effect
      (and
        (primary_itinerary_checks_completed ?primary_itinerary)
        (primary_itinerary_requirement_satisfied ?primary_itinerary)
      )
  )
  (:action apply_supporting_evidence_to_primary_itinerary
    :parameters (?primary_itinerary - primary_itinerary ?jurisdiction - jurisdiction ?supporting_evidence - supporting_evidence)
    :precondition
      (and
        (case_documents_verified ?primary_itinerary)
        (itinerary_jurisdiction_link ?primary_itinerary ?jurisdiction)
        (evidence_available ?supporting_evidence)
        (not
          (primary_itinerary_checks_completed ?primary_itinerary)
        )
      )
    :effect
      (and
        (jurisdiction_evidence_flag ?jurisdiction)
        (primary_itinerary_checks_completed ?primary_itinerary)
        (itinerary_evidence_link ?primary_itinerary ?supporting_evidence)
        (not
          (evidence_available ?supporting_evidence)
        )
      )
  )
  (:action finalize_primary_itinerary_requirement_satisfaction
    :parameters (?primary_itinerary - primary_itinerary ?jurisdiction - jurisdiction ?document_type_requirement - required_document_type ?supporting_evidence - supporting_evidence)
    :precondition
      (and
        (case_documents_verified ?primary_itinerary)
        (case_matches_document_type_requirement ?primary_itinerary ?document_type_requirement)
        (itinerary_jurisdiction_link ?primary_itinerary ?jurisdiction)
        (jurisdiction_evidence_flag ?jurisdiction)
        (itinerary_evidence_link ?primary_itinerary ?supporting_evidence)
        (not
          (primary_itinerary_requirement_satisfied ?primary_itinerary)
        )
      )
    :effect
      (and
        (jurisdiction_document_match_flag ?jurisdiction)
        (primary_itinerary_requirement_satisfied ?primary_itinerary)
        (evidence_available ?supporting_evidence)
        (not
          (itinerary_evidence_link ?primary_itinerary ?supporting_evidence)
        )
      )
  )
  (:action satisfy_companion_jurisdiction_primary_requirement
    :parameters (?companion_itinerary - companion_itinerary ?jurisdiction_requirement_profile - jurisdiction_requirement_profile ?document_type_requirement - required_document_type)
    :precondition
      (and
        (case_documents_verified ?companion_itinerary)
        (case_matches_document_type_requirement ?companion_itinerary ?document_type_requirement)
        (companion_jurisdiction_profile_link ?companion_itinerary ?jurisdiction_requirement_profile)
        (not
          (jurisdiction_profile_primary_satisfied ?jurisdiction_requirement_profile)
        )
        (not
          (jurisdiction_profile_evidence_satisfied ?jurisdiction_requirement_profile)
        )
      )
    :effect (jurisdiction_profile_primary_satisfied ?jurisdiction_requirement_profile)
  )
  (:action apply_authority_verification_to_companion_itinerary
    :parameters (?companion_itinerary - companion_itinerary ?jurisdiction_requirement_profile - jurisdiction_requirement_profile ?verifying_authority - verifying_authority)
    :precondition
      (and
        (case_documents_verified ?companion_itinerary)
        (case_assigned_authority ?companion_itinerary ?verifying_authority)
        (companion_jurisdiction_profile_link ?companion_itinerary ?jurisdiction_requirement_profile)
        (jurisdiction_profile_primary_satisfied ?jurisdiction_requirement_profile)
        (not
          (companion_itinerary_checks_completed ?companion_itinerary)
        )
      )
    :effect
      (and
        (companion_itinerary_checks_completed ?companion_itinerary)
        (companion_itinerary_requirement_satisfied ?companion_itinerary)
      )
  )
  (:action apply_supporting_evidence_to_companion_itinerary
    :parameters (?companion_itinerary - companion_itinerary ?jurisdiction_requirement_profile - jurisdiction_requirement_profile ?supporting_evidence - supporting_evidence)
    :precondition
      (and
        (case_documents_verified ?companion_itinerary)
        (companion_jurisdiction_profile_link ?companion_itinerary ?jurisdiction_requirement_profile)
        (evidence_available ?supporting_evidence)
        (not
          (companion_itinerary_checks_completed ?companion_itinerary)
        )
      )
    :effect
      (and
        (jurisdiction_profile_evidence_satisfied ?jurisdiction_requirement_profile)
        (companion_itinerary_checks_completed ?companion_itinerary)
        (companion_evidence_link ?companion_itinerary ?supporting_evidence)
        (not
          (evidence_available ?supporting_evidence)
        )
      )
  )
  (:action finalize_companion_itinerary_requirement_satisfaction
    :parameters (?companion_itinerary - companion_itinerary ?jurisdiction_requirement_profile - jurisdiction_requirement_profile ?document_type_requirement - required_document_type ?supporting_evidence - supporting_evidence)
    :precondition
      (and
        (case_documents_verified ?companion_itinerary)
        (case_matches_document_type_requirement ?companion_itinerary ?document_type_requirement)
        (companion_jurisdiction_profile_link ?companion_itinerary ?jurisdiction_requirement_profile)
        (jurisdiction_profile_evidence_satisfied ?jurisdiction_requirement_profile)
        (companion_evidence_link ?companion_itinerary ?supporting_evidence)
        (not
          (companion_itinerary_requirement_satisfied ?companion_itinerary)
        )
      )
    :effect
      (and
        (jurisdiction_profile_primary_satisfied ?jurisdiction_requirement_profile)
        (companion_itinerary_requirement_satisfied ?companion_itinerary)
        (evidence_available ?supporting_evidence)
        (not
          (companion_evidence_link ?companion_itinerary ?supporting_evidence)
        )
      )
  )
  (:action assemble_fare_candidate
    :parameters (?primary_itinerary - primary_itinerary ?companion_itinerary - companion_itinerary ?jurisdiction - jurisdiction ?jurisdiction_requirement_profile - jurisdiction_requirement_profile ?fare_option - fare_option)
    :precondition
      (and
        (primary_itinerary_checks_completed ?primary_itinerary)
        (companion_itinerary_checks_completed ?companion_itinerary)
        (itinerary_jurisdiction_link ?primary_itinerary ?jurisdiction)
        (companion_jurisdiction_profile_link ?companion_itinerary ?jurisdiction_requirement_profile)
        (jurisdiction_document_match_flag ?jurisdiction)
        (jurisdiction_profile_primary_satisfied ?jurisdiction_requirement_profile)
        (primary_itinerary_requirement_satisfied ?primary_itinerary)
        (companion_itinerary_requirement_satisfied ?companion_itinerary)
        (fare_option_available ?fare_option)
      )
    :effect
      (and
        (fare_candidate_selected ?fare_option)
        (fare_option_jurisdiction_link ?fare_option ?jurisdiction)
        (fare_option_profile_link ?fare_option ?jurisdiction_requirement_profile)
        (not
          (fare_option_available ?fare_option)
        )
      )
  )
  (:action assemble_fare_candidate_with_primary_evidence
    :parameters (?primary_itinerary - primary_itinerary ?companion_itinerary - companion_itinerary ?jurisdiction - jurisdiction ?jurisdiction_requirement_profile - jurisdiction_requirement_profile ?fare_option - fare_option)
    :precondition
      (and
        (primary_itinerary_checks_completed ?primary_itinerary)
        (companion_itinerary_checks_completed ?companion_itinerary)
        (itinerary_jurisdiction_link ?primary_itinerary ?jurisdiction)
        (companion_jurisdiction_profile_link ?companion_itinerary ?jurisdiction_requirement_profile)
        (jurisdiction_evidence_flag ?jurisdiction)
        (jurisdiction_profile_primary_satisfied ?jurisdiction_requirement_profile)
        (not
          (primary_itinerary_requirement_satisfied ?primary_itinerary)
        )
        (companion_itinerary_requirement_satisfied ?companion_itinerary)
        (fare_option_available ?fare_option)
      )
    :effect
      (and
        (fare_candidate_selected ?fare_option)
        (fare_option_jurisdiction_link ?fare_option ?jurisdiction)
        (fare_option_profile_link ?fare_option ?jurisdiction_requirement_profile)
        (fare_option_primary_profile_restriction ?fare_option)
        (not
          (fare_option_available ?fare_option)
        )
      )
  )
  (:action assemble_fare_candidate_with_companion_evidence
    :parameters (?primary_itinerary - primary_itinerary ?companion_itinerary - companion_itinerary ?jurisdiction - jurisdiction ?jurisdiction_requirement_profile - jurisdiction_requirement_profile ?fare_option - fare_option)
    :precondition
      (and
        (primary_itinerary_checks_completed ?primary_itinerary)
        (companion_itinerary_checks_completed ?companion_itinerary)
        (itinerary_jurisdiction_link ?primary_itinerary ?jurisdiction)
        (companion_jurisdiction_profile_link ?companion_itinerary ?jurisdiction_requirement_profile)
        (jurisdiction_document_match_flag ?jurisdiction)
        (jurisdiction_profile_evidence_satisfied ?jurisdiction_requirement_profile)
        (primary_itinerary_requirement_satisfied ?primary_itinerary)
        (not
          (companion_itinerary_requirement_satisfied ?companion_itinerary)
        )
        (fare_option_available ?fare_option)
      )
    :effect
      (and
        (fare_candidate_selected ?fare_option)
        (fare_option_jurisdiction_link ?fare_option ?jurisdiction)
        (fare_option_profile_link ?fare_option ?jurisdiction_requirement_profile)
        (fare_option_companion_profile_restriction ?fare_option)
        (not
          (fare_option_available ?fare_option)
        )
      )
  )
  (:action assemble_fare_candidate_with_both_evidences
    :parameters (?primary_itinerary - primary_itinerary ?companion_itinerary - companion_itinerary ?jurisdiction - jurisdiction ?jurisdiction_requirement_profile - jurisdiction_requirement_profile ?fare_option - fare_option)
    :precondition
      (and
        (primary_itinerary_checks_completed ?primary_itinerary)
        (companion_itinerary_checks_completed ?companion_itinerary)
        (itinerary_jurisdiction_link ?primary_itinerary ?jurisdiction)
        (companion_jurisdiction_profile_link ?companion_itinerary ?jurisdiction_requirement_profile)
        (jurisdiction_evidence_flag ?jurisdiction)
        (jurisdiction_profile_evidence_satisfied ?jurisdiction_requirement_profile)
        (not
          (primary_itinerary_requirement_satisfied ?primary_itinerary)
        )
        (not
          (companion_itinerary_requirement_satisfied ?companion_itinerary)
        )
        (fare_option_available ?fare_option)
      )
    :effect
      (and
        (fare_candidate_selected ?fare_option)
        (fare_option_jurisdiction_link ?fare_option ?jurisdiction)
        (fare_option_profile_link ?fare_option ?jurisdiction_requirement_profile)
        (fare_option_primary_profile_restriction ?fare_option)
        (fare_option_companion_profile_restriction ?fare_option)
        (not
          (fare_option_available ?fare_option)
        )
      )
  )
  (:action prepare_fare_option_for_template_matching
    :parameters (?fare_option - fare_option ?primary_itinerary - primary_itinerary ?document_type_requirement - required_document_type)
    :precondition
      (and
        (fare_candidate_selected ?fare_option)
        (primary_itinerary_checks_completed ?primary_itinerary)
        (case_matches_document_type_requirement ?primary_itinerary ?document_type_requirement)
        (not
          (fare_option_ready_for_template_matching ?fare_option)
        )
      )
    :effect (fare_option_ready_for_template_matching ?fare_option)
  )
  (:action apply_document_template_to_fare_rule
    :parameters (?fare_rule_case - fare_rule_case ?document_template - document_template ?fare_option - fare_option)
    :precondition
      (and
        (case_documents_verified ?fare_rule_case)
        (fare_rule_associated_fare_option ?fare_rule_case ?fare_option)
        (fare_rule_template_match ?fare_rule_case ?document_template)
        (document_template_available ?document_template)
        (fare_candidate_selected ?fare_option)
        (fare_option_ready_for_template_matching ?fare_option)
        (not
          (document_template_consumed ?document_template)
        )
      )
    :effect
      (and
        (document_template_consumed ?document_template)
        (document_template_matches_fare_option ?document_template ?fare_option)
        (not
          (document_template_available ?document_template)
        )
      )
  )
  (:action finalize_fare_rule_template_matching
    :parameters (?fare_rule_case - fare_rule_case ?document_template - document_template ?fare_option - fare_option ?document_type_requirement - required_document_type)
    :precondition
      (and
        (case_documents_verified ?fare_rule_case)
        (fare_rule_template_match ?fare_rule_case ?document_template)
        (document_template_consumed ?document_template)
        (document_template_matches_fare_option ?document_template ?fare_option)
        (case_matches_document_type_requirement ?fare_rule_case ?document_type_requirement)
        (not
          (fare_option_primary_profile_restriction ?fare_option)
        )
        (not
          (fare_rule_template_matching_complete ?fare_rule_case)
        )
      )
    :effect (fare_rule_template_matching_complete ?fare_rule_case)
  )
  (:action assign_fare_class_to_fare_rule
    :parameters (?fare_rule_case - fare_rule_case ?fare_class - fare_class)
    :precondition
      (and
        (case_documents_verified ?fare_rule_case)
