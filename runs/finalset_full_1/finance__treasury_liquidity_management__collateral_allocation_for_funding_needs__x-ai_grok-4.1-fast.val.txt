--- domain.pddl (as passed to VAL) ---
(define (domain treasury_collateral_allocation)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_group - object asset_metadata_group - object bucket_group - object account_base - object account - account_base collateral_pool - resource_group eligibility_check - resource_group custodian - resource_group allocation_rule - resource_group valuation_profile - resource_group settlement_type - resource_group buffer_profile - resource_group counterparty_approval - resource_group collateral_asset - asset_metadata_group settlement_lot - asset_metadata_group regulatory_constraint - asset_metadata_group funding_bucket - bucket_group liquidity_bucket - bucket_group allocation_candidate - bucket_group account_role_group - account portfolio_group - account funding_account - account_role_group receiving_account - account_role_group portfolio - portfolio_group)
  (:predicates
    (entity_funding_need_registered ?account - account)
    (entity_eligibility_validated ?account - account)
    (entity_collateral_pool_attached ?account - account)
    (entity_funding_need_fulfilled ?account - account)
    (entity_ready_for_posting ?account - account)
    (entity_posting_committed ?account - account)
    (collateral_pool_available ?collateral_pool - collateral_pool)
    (entity_assigned_collateral_pool ?account - account ?collateral_pool - collateral_pool)
    (eligibility_snapshot_available ?eligibility_snapshot - eligibility_check)
    (entity_has_eligibility_snapshot ?account - account ?eligibility_snapshot - eligibility_check)
    (custodian_available ?custodian - custodian)
    (entity_custodian_hold ?account - account ?custodian - custodian)
    (collateral_asset_available ?collateral_asset - collateral_asset)
    (funding_account_reserved_asset ?funding_account - funding_account ?collateral_asset - collateral_asset)
    (receiving_account_reserved_asset ?receiving_account - receiving_account ?collateral_asset - collateral_asset)
    (funding_account_bucket ?funding_account - funding_account ?funding_bucket - funding_bucket)
    (funding_bucket_tagged_source ?funding_bucket - funding_bucket)
    (funding_bucket_tagged_candidate ?funding_bucket - funding_bucket)
    (funding_account_marked_source ?funding_account - funding_account)
    (receiving_account_liquidity_bucket ?receiving_account - receiving_account ?liquidity_bucket - liquidity_bucket)
    (liquidity_bucket_tagged_source ?liquidity_bucket - liquidity_bucket)
    (liquidity_bucket_tagged_candidate ?liquidity_bucket - liquidity_bucket)
    (receiving_account_marked_destination ?receiving_account - receiving_account)
    (allocation_candidate_available ?allocation_candidate - allocation_candidate)
    (allocation_candidate_locked ?allocation_candidate - allocation_candidate)
    (candidate_funding_bucket_link ?allocation_candidate - allocation_candidate ?funding_bucket - funding_bucket)
    (candidate_liquidity_bucket_link ?allocation_candidate - allocation_candidate ?liquidity_bucket - liquidity_bucket)
    (candidate_funding_matched ?allocation_candidate - allocation_candidate)
    (candidate_liquidity_matched ?allocation_candidate - allocation_candidate)
    (candidate_ready_for_settlement ?allocation_candidate - allocation_candidate)
    (portfolio_funding_account_link ?portfolio - portfolio ?funding_account - funding_account)
    (portfolio_receiving_account_link ?portfolio - portfolio ?receiving_account - receiving_account)
    (portfolio_candidate_link ?portfolio - portfolio ?allocation_candidate - allocation_candidate)
    (settlement_lot_available ?settlement_lot - settlement_lot)
    (portfolio_settlement_lot_link ?portfolio - portfolio ?settlement_lot - settlement_lot)
    (settlement_lot_reserved ?settlement_lot - settlement_lot)
    (settlement_lot_assigned_to_candidate ?settlement_lot - settlement_lot ?allocation_candidate - allocation_candidate)
    (portfolio_reservation_marked ?portfolio - portfolio)
    (portfolio_reservation_confirmed ?portfolio - portfolio)
    (portfolio_ready_for_allocation ?portfolio - portfolio)
    (portfolio_has_allocation_rule ?portfolio - portfolio)
    (portfolio_allocation_rule_confirmed ?portfolio - portfolio)
    (portfolio_valuation_applied ?portfolio - portfolio)
    (portfolio_checks_passed ?portfolio - portfolio)
    (regulatory_constraint_available ?regulatory_constraint - regulatory_constraint)
    (portfolio_regulatory_constraint_link ?portfolio - portfolio ?regulatory_constraint - regulatory_constraint)
    (portfolio_regulatory_check_applied ?portfolio - portfolio)
    (portfolio_regulatory_rule_passed ?portfolio - portfolio)
    (portfolio_regulatory_approval_done ?portfolio - portfolio)
    (allocation_rule_available ?allocation_rule - allocation_rule)
    (portfolio_allocation_rule_link ?portfolio - portfolio ?allocation_rule - allocation_rule)
    (valuation_profile_available ?valuation_profile - valuation_profile)
    (portfolio_valuation_profile_link ?portfolio - portfolio ?valuation_profile - valuation_profile)
    (buffer_profile_available ?buffer_profile - buffer_profile)
    (portfolio_buffer_profile_link ?portfolio - portfolio ?buffer_profile - buffer_profile)
    (counterparty_approval_available ?counterparty_approval - counterparty_approval)
    (portfolio_counterparty_approval_link ?portfolio - portfolio ?counterparty_approval - counterparty_approval)
    (settlement_type_available ?settlement_type - settlement_type)
    (entity_settlement_type_selected ?account - account ?settlement_type - settlement_type)
    (funding_account_ready ?funding_account - funding_account)
    (receiving_account_ready ?receiving_account - receiving_account)
    (portfolio_finalized ?portfolio - portfolio)
  )
  (:action register_funding_need
    :parameters (?account - account)
    :precondition
      (and
        (not
          (entity_funding_need_registered ?account)
        )
        (not
          (entity_funding_need_fulfilled ?account)
        )
      )
    :effect (entity_funding_need_registered ?account)
  )
  (:action assign_collateral_pool_to_account
    :parameters (?account - account ?collateral_pool - collateral_pool)
    :precondition
      (and
        (entity_funding_need_registered ?account)
        (not
          (entity_collateral_pool_attached ?account)
        )
        (collateral_pool_available ?collateral_pool)
      )
    :effect
      (and
        (entity_collateral_pool_attached ?account)
        (entity_assigned_collateral_pool ?account ?collateral_pool)
        (not
          (collateral_pool_available ?collateral_pool)
        )
      )
  )
  (:action attach_eligibility_snapshot_to_account
    :parameters (?account - account ?eligibility_snapshot - eligibility_check)
    :precondition
      (and
        (entity_funding_need_registered ?account)
        (entity_collateral_pool_attached ?account)
        (eligibility_snapshot_available ?eligibility_snapshot)
      )
    :effect
      (and
        (entity_has_eligibility_snapshot ?account ?eligibility_snapshot)
        (not
          (eligibility_snapshot_available ?eligibility_snapshot)
        )
      )
  )
  (:action confirm_account_eligibility
    :parameters (?account - account ?eligibility_snapshot - eligibility_check)
    :precondition
      (and
        (entity_funding_need_registered ?account)
        (entity_collateral_pool_attached ?account)
        (entity_has_eligibility_snapshot ?account ?eligibility_snapshot)
        (not
          (entity_eligibility_validated ?account)
        )
      )
    :effect (entity_eligibility_validated ?account)
  )
  (:action release_eligibility_snapshot_from_account
    :parameters (?account - account ?eligibility_snapshot - eligibility_check)
    :precondition
      (and
        (entity_has_eligibility_snapshot ?account ?eligibility_snapshot)
      )
    :effect
      (and
        (eligibility_snapshot_available ?eligibility_snapshot)
        (not
          (entity_has_eligibility_snapshot ?account ?eligibility_snapshot)
        )
      )
  )
  (:action request_custodian_hold_for_account
    :parameters (?account - account ?custodian - custodian)
    :precondition
      (and
        (entity_eligibility_validated ?account)
        (custodian_available ?custodian)
      )
    :effect
      (and
        (entity_custodian_hold ?account ?custodian)
        (not
          (custodian_available ?custodian)
        )
      )
  )
  (:action release_custodian_hold_for_account
    :parameters (?account - account ?custodian - custodian)
    :precondition
      (and
        (entity_custodian_hold ?account ?custodian)
      )
    :effect
      (and
        (custodian_available ?custodian)
        (not
          (entity_custodian_hold ?account ?custodian)
        )
      )
  )
  (:action assign_buffer_profile_to_portfolio
    :parameters (?portfolio - portfolio ?buffer_profile - buffer_profile)
    :precondition
      (and
        (entity_eligibility_validated ?portfolio)
        (buffer_profile_available ?buffer_profile)
      )
    :effect
      (and
        (portfolio_buffer_profile_link ?portfolio ?buffer_profile)
        (not
          (buffer_profile_available ?buffer_profile)
        )
      )
  )
  (:action remove_buffer_profile_from_portfolio
    :parameters (?portfolio - portfolio ?buffer_profile - buffer_profile)
    :precondition
      (and
        (portfolio_buffer_profile_link ?portfolio ?buffer_profile)
      )
    :effect
      (and
        (buffer_profile_available ?buffer_profile)
        (not
          (portfolio_buffer_profile_link ?portfolio ?buffer_profile)
        )
      )
  )
  (:action assign_counterparty_approval_to_portfolio
    :parameters (?portfolio - portfolio ?counterparty_approval - counterparty_approval)
    :precondition
      (and
        (entity_eligibility_validated ?portfolio)
        (counterparty_approval_available ?counterparty_approval)
      )
    :effect
      (and
        (portfolio_counterparty_approval_link ?portfolio ?counterparty_approval)
        (not
          (counterparty_approval_available ?counterparty_approval)
        )
      )
  )
  (:action remove_counterparty_approval_from_portfolio
    :parameters (?portfolio - portfolio ?counterparty_approval - counterparty_approval)
    :precondition
      (and
        (portfolio_counterparty_approval_link ?portfolio ?counterparty_approval)
      )
    :effect
      (and
        (counterparty_approval_available ?counterparty_approval)
        (not
          (portfolio_counterparty_approval_link ?portfolio ?counterparty_approval)
        )
      )
  )
  (:action tag_funding_bucket_for_source
    :parameters (?funding_account - funding_account ?funding_bucket - funding_bucket ?eligibility_snapshot - eligibility_check)
    :precondition
      (and
        (entity_eligibility_validated ?funding_account)
        (entity_has_eligibility_snapshot ?funding_account ?eligibility_snapshot)
        (funding_account_bucket ?funding_account ?funding_bucket)
        (not
          (funding_bucket_tagged_source ?funding_bucket)
        )
        (not
          (funding_bucket_tagged_candidate ?funding_bucket)
        )
      )
    :effect (funding_bucket_tagged_source ?funding_bucket)
  )
  (:action confirm_funding_source_with_custodian
    :parameters (?funding_account - funding_account ?funding_bucket - funding_bucket ?custodian - custodian)
    :precondition
      (and
        (entity_eligibility_validated ?funding_account)
        (entity_custodian_hold ?funding_account ?custodian)
        (funding_account_bucket ?funding_account ?funding_bucket)
        (funding_bucket_tagged_source ?funding_bucket)
        (not
          (funding_account_ready ?funding_account)
        )
      )
    :effect
      (and
        (funding_account_ready ?funding_account)
        (funding_account_marked_source ?funding_account)
      )
  )
  (:action reserve_asset_for_funding_account
    :parameters (?funding_account - funding_account ?funding_bucket - funding_bucket ?collateral_asset - collateral_asset)
    :precondition
      (and
        (entity_eligibility_validated ?funding_account)
        (funding_account_bucket ?funding_account ?funding_bucket)
        (collateral_asset_available ?collateral_asset)
        (not
          (funding_account_ready ?funding_account)
        )
      )
    :effect
      (and
        (funding_bucket_tagged_candidate ?funding_bucket)
        (funding_account_ready ?funding_account)
        (funding_account_reserved_asset ?funding_account ?collateral_asset)
        (not
          (collateral_asset_available ?collateral_asset)
        )
      )
  )
  (:action finalize_funding_asset_selection
    :parameters (?funding_account - funding_account ?funding_bucket - funding_bucket ?eligibility_snapshot - eligibility_check ?collateral_asset - collateral_asset)
    :precondition
      (and
        (entity_eligibility_validated ?funding_account)
        (entity_has_eligibility_snapshot ?funding_account ?eligibility_snapshot)
        (funding_account_bucket ?funding_account ?funding_bucket)
        (funding_bucket_tagged_candidate ?funding_bucket)
        (funding_account_reserved_asset ?funding_account ?collateral_asset)
        (not
          (funding_account_marked_source ?funding_account)
        )
      )
    :effect
      (and
        (funding_bucket_tagged_source ?funding_bucket)
        (funding_account_marked_source ?funding_account)
        (collateral_asset_available ?collateral_asset)
        (not
          (funding_account_reserved_asset ?funding_account ?collateral_asset)
        )
      )
  )
  (:action tag_liquidity_bucket_for_destination
    :parameters (?receiving_account - receiving_account ?liquidity_bucket - liquidity_bucket ?eligibility_snapshot - eligibility_check)
    :precondition
      (and
        (entity_eligibility_validated ?receiving_account)
        (entity_has_eligibility_snapshot ?receiving_account ?eligibility_snapshot)
        (receiving_account_liquidity_bucket ?receiving_account ?liquidity_bucket)
        (not
          (liquidity_bucket_tagged_source ?liquidity_bucket)
        )
        (not
          (liquidity_bucket_tagged_candidate ?liquidity_bucket)
        )
      )
    :effect (liquidity_bucket_tagged_source ?liquidity_bucket)
  )
  (:action confirm_receiving_source_with_custodian
    :parameters (?receiving_account - receiving_account ?liquidity_bucket - liquidity_bucket ?custodian - custodian)
    :precondition
      (and
        (entity_eligibility_validated ?receiving_account)
        (entity_custodian_hold ?receiving_account ?custodian)
        (receiving_account_liquidity_bucket ?receiving_account ?liquidity_bucket)
        (liquidity_bucket_tagged_source ?liquidity_bucket)
        (not
          (receiving_account_ready ?receiving_account)
        )
      )
    :effect
      (and
        (receiving_account_ready ?receiving_account)
        (receiving_account_marked_destination ?receiving_account)
      )
  )
  (:action reserve_asset_for_receiving_account
    :parameters (?receiving_account - receiving_account ?liquidity_bucket - liquidity_bucket ?collateral_asset - collateral_asset)
    :precondition
      (and
        (entity_eligibility_validated ?receiving_account)
        (receiving_account_liquidity_bucket ?receiving_account ?liquidity_bucket)
        (collateral_asset_available ?collateral_asset)
        (not
          (receiving_account_ready ?receiving_account)
        )
      )
    :effect
      (and
        (liquidity_bucket_tagged_candidate ?liquidity_bucket)
        (receiving_account_ready ?receiving_account)
        (receiving_account_reserved_asset ?receiving_account ?collateral_asset)
        (not
          (collateral_asset_available ?collateral_asset)
        )
      )
  )
  (:action finalize_receiving_asset_selection
    :parameters (?receiving_account - receiving_account ?liquidity_bucket - liquidity_bucket ?eligibility_snapshot - eligibility_check ?collateral_asset - collateral_asset)
    :precondition
      (and
        (entity_eligibility_validated ?receiving_account)
        (entity_has_eligibility_snapshot ?receiving_account ?eligibility_snapshot)
        (receiving_account_liquidity_bucket ?receiving_account ?liquidity_bucket)
        (liquidity_bucket_tagged_candidate ?liquidity_bucket)
        (receiving_account_reserved_asset ?receiving_account ?collateral_asset)
        (not
          (receiving_account_marked_destination ?receiving_account)
        )
      )
    :effect
      (and
        (liquidity_bucket_tagged_source ?liquidity_bucket)
        (receiving_account_marked_destination ?receiving_account)
        (collateral_asset_available ?collateral_asset)
        (not
          (receiving_account_reserved_asset ?receiving_account ?collateral_asset)
        )
      )
  )
  (:action create_allocation_candidate_standard
    :parameters (?funding_account - funding_account ?receiving_account - receiving_account ?funding_bucket - funding_bucket ?liquidity_bucket - liquidity_bucket ?allocation_candidate - allocation_candidate)
    :precondition
      (and
        (funding_account_ready ?funding_account)
        (receiving_account_ready ?receiving_account)
        (funding_account_bucket ?funding_account ?funding_bucket)
        (receiving_account_liquidity_bucket ?receiving_account ?liquidity_bucket)
        (funding_bucket_tagged_source ?funding_bucket)
        (liquidity_bucket_tagged_source ?liquidity_bucket)
        (funding_account_marked_source ?funding_account)
        (receiving_account_marked_destination ?receiving_account)
        (allocation_candidate_available ?allocation_candidate)
      )
    :effect
      (and
        (allocation_candidate_locked ?allocation_candidate)
        (candidate_funding_bucket_link ?allocation_candidate ?funding_bucket)
        (candidate_liquidity_bucket_link ?allocation_candidate ?liquidity_bucket)
        (not
          (allocation_candidate_available ?allocation_candidate)
        )
      )
  )
  (:action create_allocation_candidate_with_funding_tag
    :parameters (?funding_account - funding_account ?receiving_account - receiving_account ?funding_bucket - funding_bucket ?liquidity_bucket - liquidity_bucket ?allocation_candidate - allocation_candidate)
    :precondition
      (and
        (funding_account_ready ?funding_account)
        (receiving_account_ready ?receiving_account)
        (funding_account_bucket ?funding_account ?funding_bucket)
        (receiving_account_liquidity_bucket ?receiving_account ?liquidity_bucket)
        (funding_bucket_tagged_candidate ?funding_bucket)
        (liquidity_bucket_tagged_source ?liquidity_bucket)
        (not
          (funding_account_marked_source ?funding_account)
        )
        (receiving_account_marked_destination ?receiving_account)
        (allocation_candidate_available ?allocation_candidate)
      )
    :effect
      (and
        (allocation_candidate_locked ?allocation_candidate)
        (candidate_funding_bucket_link ?allocation_candidate ?funding_bucket)
        (candidate_liquidity_bucket_link ?allocation_candidate ?liquidity_bucket)
        (candidate_funding_matched ?allocation_candidate)
        (not
          (allocation_candidate_available ?allocation_candidate)
        )
      )
  )
  (:action create_allocation_candidate_with_liquidity_tag
    :parameters (?funding_account - funding_account ?receiving_account - receiving_account ?funding_bucket - funding_bucket ?liquidity_bucket - liquidity_bucket ?allocation_candidate - allocation_candidate)
    :precondition
      (and
        (funding_account_ready ?funding_account)
        (receiving_account_ready ?receiving_account)
        (funding_account_bucket ?funding_account ?funding_bucket)
        (receiving_account_liquidity_bucket ?receiving_account ?liquidity_bucket)
        (funding_bucket_tagged_source ?funding_bucket)
        (liquidity_bucket_tagged_candidate ?liquidity_bucket)
        (funding_account_marked_source ?funding_account)
        (not
          (receiving_account_marked_destination ?receiving_account)
        )
        (allocation_candidate_available ?allocation_candidate)
      )
    :effect
      (and
        (allocation_candidate_locked ?allocation_candidate)
        (candidate_funding_bucket_link ?allocation_candidate ?funding_bucket)
        (candidate_liquidity_bucket_link ?allocation_candidate ?liquidity_bucket)
        (candidate_liquidity_matched ?allocation_candidate)
        (not
          (allocation_candidate_available ?allocation_candidate)
        )
      )
  )
  (:action create_allocation_candidate_full_match
    :parameters (?funding_account - funding_account ?receiving_account - receiving_account ?funding_bucket - funding_bucket ?liquidity_bucket - liquidity_bucket ?allocation_candidate - allocation_candidate)
    :precondition
      (and
        (funding_account_ready ?funding_account)
        (receiving_account_ready ?receiving_account)
        (funding_account_bucket ?funding_account ?funding_bucket)
        (receiving_account_liquidity_bucket ?receiving_account ?liquidity_bucket)
        (funding_bucket_tagged_candidate ?funding_bucket)
        (liquidity_bucket_tagged_candidate ?liquidity_bucket)
        (not
          (funding_account_marked_source ?funding_account)
        )
        (not
          (receiving_account_marked_destination ?receiving_account)
        )
        (allocation_candidate_available ?allocation_candidate)
      )
    :effect
      (and
        (allocation_candidate_locked ?allocation_candidate)
        (candidate_funding_bucket_link ?allocation_candidate ?funding_bucket)
        (candidate_liquidity_bucket_link ?allocation_candidate ?liquidity_bucket)
        (candidate_funding_matched ?allocation_candidate)
        (candidate_liquidity_matched ?allocation_candidate)
        (not
          (allocation_candidate_available ?allocation_candidate)
        )
      )
  )
  (:action mark_candidate_ready_for_settlement
    :parameters (?allocation_candidate - allocation_candidate ?funding_account - funding_account ?eligibility_snapshot - eligibility_check)
    :precondition
      (and
        (allocation_candidate_locked ?allocation_candidate)
        (funding_account_ready ?funding_account)
        (entity_has_eligibility_snapshot ?funding_account ?eligibility_snapshot)
        (not
          (candidate_ready_for_settlement ?allocation_candidate)
        )
      )
    :effect (candidate_ready_for_settlement ?allocation_candidate)
  )
  (:action reserve_settlement_lot_for_candidate
    :parameters (?portfolio - portfolio ?settlement_lot - settlement_lot ?allocation_candidate - allocation_candidate)
    :precondition
      (and
        (entity_eligibility_validated ?portfolio)
        (portfolio_candidate_link ?portfolio ?allocation_candidate)
        (portfolio_settlement_lot_link ?portfolio ?settlement_lot)
        (settlement_lot_available ?settlement_lot)
        (allocation_candidate_locked ?allocation_candidate)
        (candidate_ready_for_settlement ?allocation_candidate)
        (not
          (settlement_lot_reserved ?settlement_lot)
        )
      )
    :effect
      (and
        (settlement_lot_reserved ?settlement_lot)
        (settlement_lot_assigned_to_candidate ?settlement_lot ?allocation_candidate)
        (not
          (settlement_lot_available ?settlement_lot)
        )
      )
  )
  (:action lock_portfolio_assets_for_candidate
    :parameters (?portfolio - portfolio ?settlement_lot - settlement_lot ?allocation_candidate - allocation_candidate ?eligibility_snapshot - eligibility_check)
    :precondition
      (and
        (entity_eligibility_validated ?portfolio)
        (portfolio_settlement_lot_link ?portfolio ?settlement_lot)
        (settlement_lot_reserved ?settlement_lot)
        (settlement_lot_assigned_to_candidate ?settlement_lot ?allocation_candidate)
        (entity_has_eligibility_snapshot ?portfolio ?eligibility_snapshot)
        (not
          (candidate_funding_matched ?allocation_candidate)
        )
        (not
          (portfolio_reservation_marked ?portfolio)
        )
      )
    :effect (portfolio_reservation_marked ?portfolio)
  )
  (:action apply_allocation_rule_to_portfolio
    :parameters (?portfolio - portfolio ?allocation_rule - allocation_rule)
    :precondition
      (and
        (entity_eligibility_validated ?portfolio)
        (allocation_rule_available ?allocation_rule)
        (not
          (portfolio_has_allocation_rule ?portfolio)
        )
      )
    :effect
      (and
        (portfolio_has_allocation_rule ?portfolio)
        (portfolio_allocation_rule_link ?portfolio ?allocation_rule)
        (not
          (allocation_rule_available ?allocation_rule)
        )
      )
  )
  (:action assign_allocation_rule_and_reserve_assets
    :parameters (?portfolio - portfolio ?settlement_lot - settlement_lot ?allocation_candidate - allocation_candidate ?eligibility_snapshot - eligibility_check ?allocation_rule - allocation_rule)
    :precondition
      (and
        (entity_eligibility_validated ?portfolio)
        (portfolio_settlement_lot_link ?portfolio ?settlement_lot)
        (settlement_lot_reserved ?settlement_lot)
        (settlement_lot_assigned_to_candidate ?settlement_lot ?allocation_candidate)
        (entity_has_eligibility_snapshot ?portfolio ?eligibility_snapshot)
        (candidate_funding_matched ?allocation_candidate)
        (portfolio_has_allocation_rule ?portfolio)
        (portfolio_allocation_rule_link ?portfolio ?allocation_rule)
        (not
          (portfolio_reservation_marked ?portfolio)
        )
      )
    :effect
      (and
        (portfolio_reservation_marked ?portfolio)
        (portfolio_allocation_rule_confirmed ?portfolio)
      )
  )
  (:action apply_buffer_profile_and_confirm_reservation
    :parameters (?portfolio - portfolio ?buffer_profile - buffer_profile ?custodian - custodian ?settlement_lot - settlement_lot ?allocation_candidate - allocation_candidate)
    :precondition
      (and
        (portfolio_reservation_marked ?portfolio)
        (portfolio_buffer_profile_link ?portfolio ?buffer_profile)
