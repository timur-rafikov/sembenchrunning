--- domain.pddl (as passed to VAL) ---
(define (domain multi_stage_mission_checkpoint_progression_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_family - object checkpoint_family - object record_family - object mission_element_root - object mission_element - mission_element_root activation_token - resource_family objective_type - resource_family interaction_node - resource_family variant_flag - resource_family milestone_modifier - resource_family stage_seal - resource_family modifier_a - resource_family modifier_c - resource_family consumable_resource - checkpoint_family optional_artifact - checkpoint_family modifier_b - checkpoint_family primary_checkpoint - record_family secondary_checkpoint - record_family checkpoint_record - record_family stage_subtype_a - mission_element stage_subtype_b - mission_element primary_stage - stage_subtype_a secondary_stage - stage_subtype_a mission_definition - stage_subtype_b)
  (:predicates
    (definition_element_revealed ?mission_element - mission_element)
    (definition_element_ready ?mission_element - mission_element)
    (definition_element_token_assigned ?mission_element - mission_element)
    (definition_element_completed ?mission_element - mission_element)
    (definition_element_finalized ?mission_element - mission_element)
    (definition_element_stage_sealed ?mission_element - mission_element)
    (activation_token_available ?activation_token - activation_token)
    (definition_element_token_binding ?mission_element - mission_element ?activation_token - activation_token)
    (objective_type_available ?objective_type - objective_type)
    (definition_element_has_objective ?mission_element - mission_element ?objective_type - objective_type)
    (interaction_node_available ?interaction_node - interaction_node)
    (definition_element_has_interaction ?mission_element - mission_element ?interaction_node - interaction_node)
    (consumable_available ?consumable_resource - consumable_resource)
    (primary_stage_has_consumable ?primary_stage - primary_stage ?consumable_resource - consumable_resource)
    (secondary_stage_has_consumable ?secondary_stage - secondary_stage ?consumable_resource - consumable_resource)
    (primary_stage_has_checkpoint ?primary_stage - primary_stage ?primary_checkpoint - primary_checkpoint)
    (primary_checkpoint_ready ?primary_checkpoint - primary_checkpoint)
    (primary_checkpoint_consumable_attached ?primary_checkpoint - primary_checkpoint)
    (primary_stage_ready ?primary_stage - primary_stage)
    (secondary_stage_has_checkpoint ?secondary_stage - secondary_stage ?secondary_checkpoint - secondary_checkpoint)
    (secondary_checkpoint_ready ?secondary_checkpoint - secondary_checkpoint)
    (secondary_checkpoint_consumable_attached ?secondary_checkpoint - secondary_checkpoint)
    (secondary_stage_ready ?secondary_stage - secondary_stage)
    (checkpoint_record_available ?checkpoint_record - checkpoint_record)
    (checkpoint_record_active ?checkpoint_record - checkpoint_record)
    (record_primary_checkpoint ?checkpoint_record - checkpoint_record ?primary_checkpoint - primary_checkpoint)
    (record_secondary_checkpoint ?checkpoint_record - checkpoint_record ?secondary_checkpoint - secondary_checkpoint)
    (record_variant_flag_a ?checkpoint_record - checkpoint_record)
    (record_variant_flag_b ?checkpoint_record - checkpoint_record)
    (checkpoint_record_committed ?checkpoint_record - checkpoint_record)
    (definition_element_has_primary_stage ?mission_definition - mission_definition ?primary_stage - primary_stage)
    (definition_element_has_secondary_stage ?mission_definition - mission_definition ?secondary_stage - secondary_stage)
    (definition_element_has_checkpoint_record ?mission_definition - mission_definition ?checkpoint_record - checkpoint_record)
    (artifact_available ?optional_artifact - optional_artifact)
    (definition_element_has_artifact ?mission_definition - mission_definition ?optional_artifact - optional_artifact)
    (artifact_attached ?optional_artifact - optional_artifact)
    (artifact_assigned_to_record ?optional_artifact - optional_artifact ?checkpoint_record - checkpoint_record)
    (definition_element_modifier_phase1 ?mission_definition - mission_definition)
    (definition_element_modifier_phase2 ?mission_definition - mission_definition)
    (definition_element_modifier_phase3 ?mission_definition - mission_definition)
    (definition_element_variant_selected ?mission_definition - mission_definition)
    (definition_element_variant_confirmed ?mission_definition - mission_definition)
    (definition_element_modifier_b_applied ?mission_definition - mission_definition)
    (definition_element_ready_for_finalization ?mission_definition - mission_definition)
    (modifier_b_available ?modifier_b - modifier_b)
    (definition_element_has_modifier_b ?mission_definition - mission_definition ?modifier_b - modifier_b)
    (definition_element_subtask_unlocked ?mission_definition - mission_definition)
    (definition_element_subtask_interaction_assigned ?mission_definition - mission_definition)
    (definition_element_subtask_completed ?mission_definition - mission_definition)
    (variant_flag_available ?variant_flag - variant_flag)
    (definition_element_has_variant_flag ?mission_definition - mission_definition ?variant_flag - variant_flag)
    (milestone_modifier_available ?milestone_modifier - milestone_modifier)
    (definition_element_has_milestone_modifier ?mission_definition - mission_definition ?milestone_modifier - milestone_modifier)
    (modifier_a_available ?modifier_a - modifier_a)
    (definition_element_has_modifier_a ?mission_definition - mission_definition ?modifier_a - modifier_a)
    (modifier_c_available ?modifier_c - modifier_c)
    (definition_element_has_modifier_c ?mission_definition - mission_definition ?modifier_c - modifier_c)
    (stage_seal_available ?stage_seal - stage_seal)
    (definition_element_has_stage_seal ?mission_element - mission_element ?stage_seal - stage_seal)
    (primary_stage_committed ?primary_stage - primary_stage)
    (secondary_stage_committed ?secondary_stage - secondary_stage)
    (definition_element_rewards_bound ?mission_definition - mission_definition)
  )
  (:action reveal_mission_element
    :parameters (?mission_element - mission_element)
    :precondition
      (and
        (not
          (definition_element_revealed ?mission_element)
        )
        (not
          (definition_element_completed ?mission_element)
        )
      )
    :effect (definition_element_revealed ?mission_element)
  )
  (:action assign_activation_token_to_element
    :parameters (?mission_element - mission_element ?activation_token - activation_token)
    :precondition
      (and
        (definition_element_revealed ?mission_element)
        (not
          (definition_element_token_assigned ?mission_element)
        )
        (activation_token_available ?activation_token)
      )
    :effect
      (and
        (definition_element_token_assigned ?mission_element)
        (definition_element_token_binding ?mission_element ?activation_token)
        (not
          (activation_token_available ?activation_token)
        )
      )
  )
  (:action assign_objective_to_element
    :parameters (?mission_element - mission_element ?objective_type - objective_type)
    :precondition
      (and
        (definition_element_revealed ?mission_element)
        (definition_element_token_assigned ?mission_element)
        (objective_type_available ?objective_type)
      )
    :effect
      (and
        (definition_element_has_objective ?mission_element ?objective_type)
        (not
          (objective_type_available ?objective_type)
        )
      )
  )
  (:action complete_objective_on_element
    :parameters (?mission_element - mission_element ?objective_type - objective_type)
    :precondition
      (and
        (definition_element_revealed ?mission_element)
        (definition_element_token_assigned ?mission_element)
        (definition_element_has_objective ?mission_element ?objective_type)
        (not
          (definition_element_ready ?mission_element)
        )
      )
    :effect (definition_element_ready ?mission_element)
  )
  (:action reset_objective_on_element
    :parameters (?mission_element - mission_element ?objective_type - objective_type)
    :precondition
      (and
        (definition_element_has_objective ?mission_element ?objective_type)
      )
    :effect
      (and
        (objective_type_available ?objective_type)
        (not
          (definition_element_has_objective ?mission_element ?objective_type)
        )
      )
  )
  (:action assign_interaction_to_element
    :parameters (?mission_element - mission_element ?interaction_node - interaction_node)
    :precondition
      (and
        (definition_element_ready ?mission_element)
        (interaction_node_available ?interaction_node)
      )
    :effect
      (and
        (definition_element_has_interaction ?mission_element ?interaction_node)
        (not
          (interaction_node_available ?interaction_node)
        )
      )
  )
  (:action release_interaction_from_element
    :parameters (?mission_element - mission_element ?interaction_node - interaction_node)
    :precondition
      (and
        (definition_element_has_interaction ?mission_element ?interaction_node)
      )
    :effect
      (and
        (interaction_node_available ?interaction_node)
        (not
          (definition_element_has_interaction ?mission_element ?interaction_node)
        )
      )
  )
  (:action assign_modifier_a_to_definition
    :parameters (?mission_definition - mission_definition ?modifier_a - modifier_a)
    :precondition
      (and
        (definition_element_ready ?mission_definition)
        (modifier_a_available ?modifier_a)
      )
    :effect
      (and
        (definition_element_has_modifier_a ?mission_definition ?modifier_a)
        (not
          (modifier_a_available ?modifier_a)
        )
      )
  )
  (:action revoke_modifier_a_from_definition
    :parameters (?mission_definition - mission_definition ?modifier_a - modifier_a)
    :precondition
      (and
        (definition_element_has_modifier_a ?mission_definition ?modifier_a)
      )
    :effect
      (and
        (modifier_a_available ?modifier_a)
        (not
          (definition_element_has_modifier_a ?mission_definition ?modifier_a)
        )
      )
  )
  (:action assign_modifier_c_to_definition
    :parameters (?mission_definition - mission_definition ?modifier_c - modifier_c)
    :precondition
      (and
        (definition_element_ready ?mission_definition)
        (modifier_c_available ?modifier_c)
      )
    :effect
      (and
        (definition_element_has_modifier_c ?mission_definition ?modifier_c)
        (not
          (modifier_c_available ?modifier_c)
        )
      )
  )
  (:action revoke_modifier_c_from_definition
    :parameters (?mission_definition - mission_definition ?modifier_c - modifier_c)
    :precondition
      (and
        (definition_element_has_modifier_c ?mission_definition ?modifier_c)
      )
    :effect
      (and
        (modifier_c_available ?modifier_c)
        (not
          (definition_element_has_modifier_c ?mission_definition ?modifier_c)
        )
      )
  )
  (:action mark_primary_checkpoint_ready
    :parameters (?primary_stage - primary_stage ?primary_checkpoint - primary_checkpoint ?objective_type - objective_type)
    :precondition
      (and
        (definition_element_ready ?primary_stage)
        (definition_element_has_objective ?primary_stage ?objective_type)
        (primary_stage_has_checkpoint ?primary_stage ?primary_checkpoint)
        (not
          (primary_checkpoint_ready ?primary_checkpoint)
        )
        (not
          (primary_checkpoint_consumable_attached ?primary_checkpoint)
        )
      )
    :effect (primary_checkpoint_ready ?primary_checkpoint)
  )
  (:action commit_primary_stage_with_interaction
    :parameters (?primary_stage - primary_stage ?primary_checkpoint - primary_checkpoint ?interaction_node - interaction_node)
    :precondition
      (and
        (definition_element_ready ?primary_stage)
        (definition_element_has_interaction ?primary_stage ?interaction_node)
        (primary_stage_has_checkpoint ?primary_stage ?primary_checkpoint)
        (primary_checkpoint_ready ?primary_checkpoint)
        (not
          (primary_stage_committed ?primary_stage)
        )
      )
    :effect
      (and
        (primary_stage_committed ?primary_stage)
        (primary_stage_ready ?primary_stage)
      )
  )
  (:action apply_artifact_to_primary_stage
    :parameters (?primary_stage - primary_stage ?primary_checkpoint - primary_checkpoint ?consumable_resource - consumable_resource)
    :precondition
      (and
        (definition_element_ready ?primary_stage)
        (primary_stage_has_checkpoint ?primary_stage ?primary_checkpoint)
        (consumable_available ?consumable_resource)
        (not
          (primary_stage_committed ?primary_stage)
        )
      )
    :effect
      (and
        (primary_checkpoint_consumable_attached ?primary_checkpoint)
        (primary_stage_committed ?primary_stage)
        (primary_stage_has_consumable ?primary_stage ?consumable_resource)
        (not
          (consumable_available ?consumable_resource)
        )
      )
  )
  (:action resolve_primary_checkpoint
    :parameters (?primary_stage - primary_stage ?primary_checkpoint - primary_checkpoint ?objective_type - objective_type ?consumable_resource - consumable_resource)
    :precondition
      (and
        (definition_element_ready ?primary_stage)
        (definition_element_has_objective ?primary_stage ?objective_type)
        (primary_stage_has_checkpoint ?primary_stage ?primary_checkpoint)
        (primary_checkpoint_consumable_attached ?primary_checkpoint)
        (primary_stage_has_consumable ?primary_stage ?consumable_resource)
        (not
          (primary_stage_ready ?primary_stage)
        )
      )
    :effect
      (and
        (primary_checkpoint_ready ?primary_checkpoint)
        (primary_stage_ready ?primary_stage)
        (consumable_available ?consumable_resource)
        (not
          (primary_stage_has_consumable ?primary_stage ?consumable_resource)
        )
      )
  )
  (:action mark_secondary_checkpoint_ready
    :parameters (?secondary_stage - secondary_stage ?secondary_checkpoint - secondary_checkpoint ?objective_type - objective_type)
    :precondition
      (and
        (definition_element_ready ?secondary_stage)
        (definition_element_has_objective ?secondary_stage ?objective_type)
        (secondary_stage_has_checkpoint ?secondary_stage ?secondary_checkpoint)
        (not
          (secondary_checkpoint_ready ?secondary_checkpoint)
        )
        (not
          (secondary_checkpoint_consumable_attached ?secondary_checkpoint)
        )
      )
    :effect (secondary_checkpoint_ready ?secondary_checkpoint)
  )
  (:action commit_secondary_stage_with_interaction
    :parameters (?secondary_stage - secondary_stage ?secondary_checkpoint - secondary_checkpoint ?interaction_node - interaction_node)
    :precondition
      (and
        (definition_element_ready ?secondary_stage)
        (definition_element_has_interaction ?secondary_stage ?interaction_node)
        (secondary_stage_has_checkpoint ?secondary_stage ?secondary_checkpoint)
        (secondary_checkpoint_ready ?secondary_checkpoint)
        (not
          (secondary_stage_committed ?secondary_stage)
        )
      )
    :effect
      (and
        (secondary_stage_committed ?secondary_stage)
        (secondary_stage_ready ?secondary_stage)
      )
  )
  (:action apply_artifact_to_secondary_stage
    :parameters (?secondary_stage - secondary_stage ?secondary_checkpoint - secondary_checkpoint ?consumable_resource - consumable_resource)
    :precondition
      (and
        (definition_element_ready ?secondary_stage)
        (secondary_stage_has_checkpoint ?secondary_stage ?secondary_checkpoint)
        (consumable_available ?consumable_resource)
        (not
          (secondary_stage_committed ?secondary_stage)
        )
      )
    :effect
      (and
        (secondary_checkpoint_consumable_attached ?secondary_checkpoint)
        (secondary_stage_committed ?secondary_stage)
        (secondary_stage_has_consumable ?secondary_stage ?consumable_resource)
        (not
          (consumable_available ?consumable_resource)
        )
      )
  )
  (:action resolve_secondary_checkpoint
    :parameters (?secondary_stage - secondary_stage ?secondary_checkpoint - secondary_checkpoint ?objective_type - objective_type ?consumable_resource - consumable_resource)
    :precondition
      (and
        (definition_element_ready ?secondary_stage)
        (definition_element_has_objective ?secondary_stage ?objective_type)
        (secondary_stage_has_checkpoint ?secondary_stage ?secondary_checkpoint)
        (secondary_checkpoint_consumable_attached ?secondary_checkpoint)
        (secondary_stage_has_consumable ?secondary_stage ?consumable_resource)
        (not
          (secondary_stage_ready ?secondary_stage)
        )
      )
    :effect
      (and
        (secondary_checkpoint_ready ?secondary_checkpoint)
        (secondary_stage_ready ?secondary_stage)
        (consumable_available ?consumable_resource)
        (not
          (secondary_stage_has_consumable ?secondary_stage ?consumable_resource)
        )
      )
  )
  (:action consolidate_checkpoints_to_record
    :parameters (?primary_stage - primary_stage ?secondary_stage - secondary_stage ?primary_checkpoint - primary_checkpoint ?secondary_checkpoint - secondary_checkpoint ?checkpoint_record - checkpoint_record)
    :precondition
      (and
        (primary_stage_committed ?primary_stage)
        (secondary_stage_committed ?secondary_stage)
        (primary_stage_has_checkpoint ?primary_stage ?primary_checkpoint)
        (secondary_stage_has_checkpoint ?secondary_stage ?secondary_checkpoint)
        (primary_checkpoint_ready ?primary_checkpoint)
        (secondary_checkpoint_ready ?secondary_checkpoint)
        (primary_stage_ready ?primary_stage)
        (secondary_stage_ready ?secondary_stage)
        (checkpoint_record_available ?checkpoint_record)
      )
    :effect
      (and
        (checkpoint_record_active ?checkpoint_record)
        (record_primary_checkpoint ?checkpoint_record ?primary_checkpoint)
        (record_secondary_checkpoint ?checkpoint_record ?secondary_checkpoint)
        (not
          (checkpoint_record_available ?checkpoint_record)
        )
      )
  )
  (:action consolidate_checkpoints_to_record_variant_a
    :parameters (?primary_stage - primary_stage ?secondary_stage - secondary_stage ?primary_checkpoint - primary_checkpoint ?secondary_checkpoint - secondary_checkpoint ?checkpoint_record - checkpoint_record)
    :precondition
      (and
        (primary_stage_committed ?primary_stage)
        (secondary_stage_committed ?secondary_stage)
        (primary_stage_has_checkpoint ?primary_stage ?primary_checkpoint)
        (secondary_stage_has_checkpoint ?secondary_stage ?secondary_checkpoint)
        (primary_checkpoint_consumable_attached ?primary_checkpoint)
        (secondary_checkpoint_ready ?secondary_checkpoint)
        (not
          (primary_stage_ready ?primary_stage)
        )
        (secondary_stage_ready ?secondary_stage)
        (checkpoint_record_available ?checkpoint_record)
      )
    :effect
      (and
        (checkpoint_record_active ?checkpoint_record)
        (record_primary_checkpoint ?checkpoint_record ?primary_checkpoint)
        (record_secondary_checkpoint ?checkpoint_record ?secondary_checkpoint)
        (record_variant_flag_a ?checkpoint_record)
        (not
          (checkpoint_record_available ?checkpoint_record)
        )
      )
  )
  (:action consolidate_checkpoints_to_record_variant_b
    :parameters (?primary_stage - primary_stage ?secondary_stage - secondary_stage ?primary_checkpoint - primary_checkpoint ?secondary_checkpoint - secondary_checkpoint ?checkpoint_record - checkpoint_record)
    :precondition
      (and
        (primary_stage_committed ?primary_stage)
        (secondary_stage_committed ?secondary_stage)
        (primary_stage_has_checkpoint ?primary_stage ?primary_checkpoint)
        (secondary_stage_has_checkpoint ?secondary_stage ?secondary_checkpoint)
        (primary_checkpoint_ready ?primary_checkpoint)
        (secondary_checkpoint_consumable_attached ?secondary_checkpoint)
        (primary_stage_ready ?primary_stage)
        (not
          (secondary_stage_ready ?secondary_stage)
        )
        (checkpoint_record_available ?checkpoint_record)
      )
    :effect
      (and
        (checkpoint_record_active ?checkpoint_record)
        (record_primary_checkpoint ?checkpoint_record ?primary_checkpoint)
        (record_secondary_checkpoint ?checkpoint_record ?secondary_checkpoint)
        (record_variant_flag_b ?checkpoint_record)
        (not
          (checkpoint_record_available ?checkpoint_record)
        )
      )
  )
  (:action consolidate_checkpoints_to_record_both_variants
    :parameters (?primary_stage - primary_stage ?secondary_stage - secondary_stage ?primary_checkpoint - primary_checkpoint ?secondary_checkpoint - secondary_checkpoint ?checkpoint_record - checkpoint_record)
    :precondition
      (and
        (primary_stage_committed ?primary_stage)
        (secondary_stage_committed ?secondary_stage)
        (primary_stage_has_checkpoint ?primary_stage ?primary_checkpoint)
        (secondary_stage_has_checkpoint ?secondary_stage ?secondary_checkpoint)
        (primary_checkpoint_consumable_attached ?primary_checkpoint)
        (secondary_checkpoint_consumable_attached ?secondary_checkpoint)
        (not
          (primary_stage_ready ?primary_stage)
        )
        (not
          (secondary_stage_ready ?secondary_stage)
        )
        (checkpoint_record_available ?checkpoint_record)
      )
    :effect
      (and
        (checkpoint_record_active ?checkpoint_record)
        (record_primary_checkpoint ?checkpoint_record ?primary_checkpoint)
        (record_secondary_checkpoint ?checkpoint_record ?secondary_checkpoint)
        (record_variant_flag_a ?checkpoint_record)
        (record_variant_flag_b ?checkpoint_record)
        (not
          (checkpoint_record_available ?checkpoint_record)
        )
      )
  )
  (:action commit_checkpoint_record
    :parameters (?checkpoint_record - checkpoint_record ?primary_stage - primary_stage ?objective_type - objective_type)
    :precondition
      (and
        (checkpoint_record_active ?checkpoint_record)
        (primary_stage_committed ?primary_stage)
        (definition_element_has_objective ?primary_stage ?objective_type)
        (not
          (checkpoint_record_committed ?checkpoint_record)
        )
      )
    :effect (checkpoint_record_committed ?checkpoint_record)
  )
  (:action attach_artifact_to_record
    :parameters (?mission_definition - mission_definition ?optional_artifact - optional_artifact ?checkpoint_record - checkpoint_record)
    :precondition
      (and
        (definition_element_ready ?mission_definition)
        (definition_element_has_checkpoint_record ?mission_definition ?checkpoint_record)
        (definition_element_has_artifact ?mission_definition ?optional_artifact)
        (artifact_available ?optional_artifact)
        (checkpoint_record_active ?checkpoint_record)
        (checkpoint_record_committed ?checkpoint_record)
        (not
          (artifact_attached ?optional_artifact)
        )
      )
    :effect
      (and
        (artifact_attached ?optional_artifact)
        (artifact_assigned_to_record ?optional_artifact ?checkpoint_record)
        (not
          (artifact_available ?optional_artifact)
        )
      )
  )
  (:action activate_definition_modifier_phase1
    :parameters (?mission_definition - mission_definition ?optional_artifact - optional_artifact ?checkpoint_record - checkpoint_record ?objective_type - objective_type)
    :precondition
      (and
        (definition_element_ready ?mission_definition)
        (definition_element_has_artifact ?mission_definition ?optional_artifact)
        (artifact_attached ?optional_artifact)
        (artifact_assigned_to_record ?optional_artifact ?checkpoint_record)
        (definition_element_has_objective ?mission_definition ?objective_type)
        (not
          (record_variant_flag_a ?checkpoint_record)
        )
        (not
          (definition_element_modifier_phase1 ?mission_definition)
        )
      )
    :effect (definition_element_modifier_phase1 ?mission_definition)
  )
  (:action select_definition_variant
    :parameters (?mission_definition - mission_definition ?variant_flag - variant_flag)
    :precondition
      (and
        (definition_element_ready ?mission_definition)
        (variant_flag_available ?variant_flag)
        (not
          (definition_element_variant_selected ?mission_definition)
        )
      )
    :effect
      (and
        (definition_element_variant_selected ?mission_definition)
        (definition_element_has_variant_flag ?mission_definition ?variant_flag)
        (not
          (variant_flag_available ?variant_flag)
        )
      )
  )
  (:action confirm_definition_variant
    :parameters (?mission_definition - mission_definition ?optional_artifact - optional_artifact ?checkpoint_record - checkpoint_record ?objective_type - objective_type ?variant_flag - variant_flag)
    :precondition
      (and
        (definition_element_ready ?mission_definition)
        (definition_element_has_artifact ?mission_definition ?optional_artifact)
        (artifact_attached ?optional_artifact)
        (artifact_assigned_to_record ?optional_artifact ?checkpoint_record)
        (definition_element_has_objective ?mission_definition ?objective_type)
        (record_variant_flag_a ?checkpoint_record)
        (definition_element_variant_selected ?mission_definition)
        (definition_element_has_variant_flag ?mission_definition ?variant_flag)
        (not
          (definition_element_modifier_phase1 ?mission_definition)
        )
      )
    :effect
      (and
        (definition_element_modifier_phase1 ?mission_definition)
        (definition_element_variant_confirmed ?mission_definition)
      )
  )
  (:action apply_modifier_a_to_definition_phase1
    :parameters (?mission_definition - mission_definition ?modifier_a - modifier_a ?interaction_node - interaction_node ?optional_artifact - optional_artifact ?checkpoint_record - checkpoint_record)
    :precondition
      (and
        (definition_element_modifier_phase1 ?mission_definition)
        (definition_element_has_modifier_a ?mission_definition ?modifier_a)
        (definition_element_has_interaction ?mission_definition ?interaction_node)
        (definition_element_has_artifact ?mission_definition ?optional_artifact)
        (artifact_assigned_to_record ?optional_artifact ?checkpoint_record)
        (not
          (record_variant_flag_b ?checkpoint_record)
        )
        (not
          (definition_element_modifier_phase2 ?mission_definition)
        )
      )
    :effect (definition_element_modifier_phase2 ?mission_definition)
  )
  (:action apply_modifier_a_to_definition_phase2
    :parameters (?mission_definition - mission_definition ?modifier_a - modifier_a ?interaction_node - interaction_node ?optional_artifact - optional_artifact ?checkpoint_record - checkpoint_record)
    :precondition
      (and
        (definition_element_modifier_phase1 ?mission_definition)
