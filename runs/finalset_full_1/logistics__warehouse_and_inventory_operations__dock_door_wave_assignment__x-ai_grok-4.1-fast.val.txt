--- domain.pddl (as passed to VAL) ---
(define (domain dock_door_wave_assignment)
  (:requirements :strips :typing :negative-preconditions)
  (:types physical_resource_supertype - object location_resource_supertype - object staging_zone_supertype - object administrative_group - object dock_wave - administrative_group dock_door - physical_resource_supertype inbound_load - physical_resource_supertype handler - physical_resource_supertype equipment_unit - physical_resource_supertype packaging_spec - physical_resource_supertype pick_face - physical_resource_supertype storage_profile - physical_resource_supertype inventory_tag - physical_resource_supertype pick_unit - location_resource_supertype inventory_slot - location_resource_supertype routing_instruction - location_resource_supertype inbound_staging_area - staging_zone_supertype outbound_staging_area - staging_zone_supertype load_container - staging_zone_supertype receiving_subarea - dock_wave picking_subarea - dock_wave receiving_area - receiving_subarea picking_area - receiving_subarea load_task - picking_subarea)
  (:predicates
    (dock_wave_created ?dock_wave - dock_wave)
    (entity_ready_for_assignment ?dock_wave - dock_wave)
    (dock_wave_door_reserved ?dock_wave - dock_wave)
    (assignment_finalized ?dock_wave - dock_wave)
    (ready_for_inventory_tagging ?dock_wave - dock_wave)
    (inventory_tag_applied_to_entity ?dock_wave - dock_wave)
    (dock_door_available ?dock_door - dock_door)
    (entity_assigned_to_dock_door ?dock_wave - dock_wave ?dock_door - dock_door)
    (inbound_load_available ?inbound_load - inbound_load)
    (inbound_load_attached_to_entity ?dock_wave - dock_wave ?inbound_load - inbound_load)
    (handler_available ?handler - handler)
    (handler_assigned_to_entity ?dock_wave - dock_wave ?handler - handler)
    (pick_unit_available ?pick_unit - pick_unit)
    (receiving_area_pick_unit_assigned ?receiving_area - receiving_area ?pick_unit - pick_unit)
    (picking_area_pick_unit_assigned ?picking_area - picking_area ?pick_unit - pick_unit)
    (receiving_area_linked_inbound_staging ?receiving_area - receiving_area ?inbound_staging_area - inbound_staging_area)
    (inbound_staging_ready ?inbound_staging_area - inbound_staging_area)
    (inbound_staging_occupied ?inbound_staging_area - inbound_staging_area)
    (receiving_area_processing_completed ?receiving_area - receiving_area)
    (picking_area_linked_outbound_staging ?picking_area - picking_area ?outbound_staging_area - outbound_staging_area)
    (outbound_staging_ready ?outbound_staging_area - outbound_staging_area)
    (outbound_staging_occupied ?outbound_staging_area - outbound_staging_area)
    (picking_area_processing_completed ?picking_area - picking_area)
    (load_container_available ?load_container - load_container)
    (load_container_initialized ?load_container - load_container)
    (load_container_assigned_inbound_staging ?load_container - load_container ?inbound_staging_area - inbound_staging_area)
    (load_container_assigned_outbound_staging ?load_container - load_container ?outbound_staging_area - outbound_staging_area)
    (load_container_inbound_ready_flag ?load_container - load_container)
    (load_container_outbound_ready_flag ?load_container - load_container)
    (load_container_receiving_check_completed ?load_container - load_container)
    (load_task_assigned_receiving_area ?load_task - load_task ?receiving_area - receiving_area)
    (load_task_assigned_picking_area ?load_task - load_task ?picking_area - picking_area)
    (load_task_assigned_container ?load_task - load_task ?load_container - load_container)
    (inventory_slot_available ?inventory_slot - inventory_slot)
    (load_task_assigned_inventory_slot ?load_task - load_task ?inventory_slot - inventory_slot)
    (inventory_slot_reserved ?inventory_slot - inventory_slot)
    (inventory_slot_assigned_to_container ?inventory_slot - inventory_slot ?load_container - load_container)
    (load_task_prevalidation_completed ?load_task - load_task)
    (load_task_containerization_ready ?load_task - load_task)
    (load_task_final_checks_completed ?load_task - load_task)
    (load_task_equipment_allocated ?load_task - load_task)
    (load_task_equipment_check_passed ?load_task - load_task)
    (load_task_packaging_attached ?load_task - load_task)
    (load_task_assignment_locked ?load_task - load_task)
    (routing_instruction_available ?routing_instruction - routing_instruction)
    (load_task_assigned_routing_instruction ?load_task - load_task ?routing_instruction - routing_instruction)
    (load_task_routing_engaged ?load_task - load_task)
    (load_task_routing_acknowledged ?load_task - load_task)
    (load_task_routing_finalized ?load_task - load_task)
    (equipment_unit_available ?equipment_unit - equipment_unit)
    (load_task_assigned_equipment_unit ?load_task - load_task ?equipment_unit - equipment_unit)
    (packaging_spec_available ?packaging_spec - packaging_spec)
    (load_task_assigned_packaging_spec ?load_task - load_task ?packaging_spec - packaging_spec)
    (storage_profile_available ?storage_profile - storage_profile)
    (load_task_assigned_storage_profile ?load_task - load_task ?storage_profile - storage_profile)
    (inventory_tag_available ?inventory_tag - inventory_tag)
    (load_task_assigned_inventory_tag ?load_task - load_task ?inventory_tag - inventory_tag)
    (pick_face_available ?pick_face - pick_face)
    (entity_assigned_pick_face ?dock_wave - dock_wave ?pick_face - pick_face)
    (receiving_area_ready_for_load_build ?receiving_area - receiving_area)
    (picking_area_ready_for_load_build ?picking_area - picking_area)
    (load_task_ready_for_inventory_tagging ?load_task - load_task)
  )
  (:action create_dock_wave
    :parameters (?dock_wave - dock_wave)
    :precondition
      (and
        (not
          (dock_wave_created ?dock_wave)
        )
        (not
          (assignment_finalized ?dock_wave)
        )
      )
    :effect (dock_wave_created ?dock_wave)
  )
  (:action reserve_dock_door_for_wave
    :parameters (?dock_wave - dock_wave ?dock_door - dock_door)
    :precondition
      (and
        (dock_wave_created ?dock_wave)
        (not
          (dock_wave_door_reserved ?dock_wave)
        )
        (dock_door_available ?dock_door)
      )
    :effect
      (and
        (dock_wave_door_reserved ?dock_wave)
        (entity_assigned_to_dock_door ?dock_wave ?dock_door)
        (not
          (dock_door_available ?dock_door)
        )
      )
  )
  (:action attach_inbound_load_to_wave
    :parameters (?dock_wave - dock_wave ?inbound_load - inbound_load)
    :precondition
      (and
        (dock_wave_created ?dock_wave)
        (dock_wave_door_reserved ?dock_wave)
        (inbound_load_available ?inbound_load)
      )
    :effect
      (and
        (inbound_load_attached_to_entity ?dock_wave ?inbound_load)
        (not
          (inbound_load_available ?inbound_load)
        )
      )
  )
  (:action acknowledge_wave_load_attachment
    :parameters (?dock_wave - dock_wave ?inbound_load - inbound_load)
    :precondition
      (and
        (dock_wave_created ?dock_wave)
        (dock_wave_door_reserved ?dock_wave)
        (inbound_load_attached_to_entity ?dock_wave ?inbound_load)
        (not
          (entity_ready_for_assignment ?dock_wave)
        )
      )
    :effect (entity_ready_for_assignment ?dock_wave)
  )
  (:action detach_inbound_load_from_wave
    :parameters (?dock_wave - dock_wave ?inbound_load - inbound_load)
    :precondition
      (and
        (inbound_load_attached_to_entity ?dock_wave ?inbound_load)
      )
    :effect
      (and
        (inbound_load_available ?inbound_load)
        (not
          (inbound_load_attached_to_entity ?dock_wave ?inbound_load)
        )
      )
  )
  (:action assign_handler_to_wave
    :parameters (?dock_wave - dock_wave ?handler - handler)
    :precondition
      (and
        (entity_ready_for_assignment ?dock_wave)
        (handler_available ?handler)
      )
    :effect
      (and
        (handler_assigned_to_entity ?dock_wave ?handler)
        (not
          (handler_available ?handler)
        )
      )
  )
  (:action release_handler_from_wave
    :parameters (?dock_wave - dock_wave ?handler - handler)
    :precondition
      (and
        (handler_assigned_to_entity ?dock_wave ?handler)
      )
    :effect
      (and
        (handler_available ?handler)
        (not
          (handler_assigned_to_entity ?dock_wave ?handler)
        )
      )
  )
  (:action assign_storage_profile_to_task
    :parameters (?load_task - load_task ?storage_profile - storage_profile)
    :precondition
      (and
        (entity_ready_for_assignment ?load_task)
        (storage_profile_available ?storage_profile)
      )
    :effect
      (and
        (load_task_assigned_storage_profile ?load_task ?storage_profile)
        (not
          (storage_profile_available ?storage_profile)
        )
      )
  )
  (:action release_storage_profile_from_task
    :parameters (?load_task - load_task ?storage_profile - storage_profile)
    :precondition
      (and
        (load_task_assigned_storage_profile ?load_task ?storage_profile)
      )
    :effect
      (and
        (storage_profile_available ?storage_profile)
        (not
          (load_task_assigned_storage_profile ?load_task ?storage_profile)
        )
      )
  )
  (:action assign_inventory_tag_to_task
    :parameters (?load_task - load_task ?inventory_tag - inventory_tag)
    :precondition
      (and
        (entity_ready_for_assignment ?load_task)
        (inventory_tag_available ?inventory_tag)
      )
    :effect
      (and
        (load_task_assigned_inventory_tag ?load_task ?inventory_tag)
        (not
          (inventory_tag_available ?inventory_tag)
        )
      )
  )
  (:action release_inventory_tag_from_task
    :parameters (?load_task - load_task ?inventory_tag - inventory_tag)
    :precondition
      (and
        (load_task_assigned_inventory_tag ?load_task ?inventory_tag)
      )
    :effect
      (and
        (inventory_tag_available ?inventory_tag)
        (not
          (load_task_assigned_inventory_tag ?load_task ?inventory_tag)
        )
      )
  )
  (:action reserve_inbound_staging_for_putaway
    :parameters (?receiving_area - receiving_area ?inbound_staging_area - inbound_staging_area ?inbound_load - inbound_load)
    :precondition
      (and
        (entity_ready_for_assignment ?receiving_area)
        (inbound_load_attached_to_entity ?receiving_area ?inbound_load)
        (receiving_area_linked_inbound_staging ?receiving_area ?inbound_staging_area)
        (not
          (inbound_staging_ready ?inbound_staging_area)
        )
        (not
          (inbound_staging_occupied ?inbound_staging_area)
        )
      )
    :effect (inbound_staging_ready ?inbound_staging_area)
  )
  (:action process_inbound_at_receiving_area
    :parameters (?receiving_area - receiving_area ?inbound_staging_area - inbound_staging_area ?handler - handler)
    :precondition
      (and
        (entity_ready_for_assignment ?receiving_area)
        (handler_assigned_to_entity ?receiving_area ?handler)
        (receiving_area_linked_inbound_staging ?receiving_area ?inbound_staging_area)
        (inbound_staging_ready ?inbound_staging_area)
        (not
          (receiving_area_ready_for_load_build ?receiving_area)
        )
      )
    :effect
      (and
        (receiving_area_ready_for_load_build ?receiving_area)
        (receiving_area_processing_completed ?receiving_area)
      )
  )
  (:action assign_pick_unit_to_receiving_area
    :parameters (?receiving_area - receiving_area ?inbound_staging_area - inbound_staging_area ?pick_unit - pick_unit)
    :precondition
      (and
        (entity_ready_for_assignment ?receiving_area)
        (receiving_area_linked_inbound_staging ?receiving_area ?inbound_staging_area)
        (pick_unit_available ?pick_unit)
        (not
          (receiving_area_ready_for_load_build ?receiving_area)
        )
      )
    :effect
      (and
        (inbound_staging_occupied ?inbound_staging_area)
        (receiving_area_ready_for_load_build ?receiving_area)
        (receiving_area_pick_unit_assigned ?receiving_area ?pick_unit)
        (not
          (pick_unit_available ?pick_unit)
        )
      )
  )
  (:action complete_receiving_pick_unit_processing
    :parameters (?receiving_area - receiving_area ?inbound_staging_area - inbound_staging_area ?inbound_load - inbound_load ?pick_unit - pick_unit)
    :precondition
      (and
        (entity_ready_for_assignment ?receiving_area)
        (inbound_load_attached_to_entity ?receiving_area ?inbound_load)
        (receiving_area_linked_inbound_staging ?receiving_area ?inbound_staging_area)
        (inbound_staging_occupied ?inbound_staging_area)
        (receiving_area_pick_unit_assigned ?receiving_area ?pick_unit)
        (not
          (receiving_area_processing_completed ?receiving_area)
        )
      )
    :effect
      (and
        (inbound_staging_ready ?inbound_staging_area)
        (receiving_area_processing_completed ?receiving_area)
        (pick_unit_available ?pick_unit)
        (not
          (receiving_area_pick_unit_assigned ?receiving_area ?pick_unit)
        )
      )
  )
  (:action reserve_outbound_staging_for_picking
    :parameters (?picking_area - picking_area ?outbound_staging_area - outbound_staging_area ?inbound_load - inbound_load)
    :precondition
      (and
        (entity_ready_for_assignment ?picking_area)
        (inbound_load_attached_to_entity ?picking_area ?inbound_load)
        (picking_area_linked_outbound_staging ?picking_area ?outbound_staging_area)
        (not
          (outbound_staging_ready ?outbound_staging_area)
        )
        (not
          (outbound_staging_occupied ?outbound_staging_area)
        )
      )
    :effect (outbound_staging_ready ?outbound_staging_area)
  )
  (:action assign_handler_to_picking_area
    :parameters (?picking_area - picking_area ?outbound_staging_area - outbound_staging_area ?handler - handler)
    :precondition
      (and
        (entity_ready_for_assignment ?picking_area)
        (handler_assigned_to_entity ?picking_area ?handler)
        (picking_area_linked_outbound_staging ?picking_area ?outbound_staging_area)
        (outbound_staging_ready ?outbound_staging_area)
        (not
          (picking_area_ready_for_load_build ?picking_area)
        )
      )
    :effect
      (and
        (picking_area_ready_for_load_build ?picking_area)
        (picking_area_processing_completed ?picking_area)
      )
  )
  (:action assign_pick_unit_to_picking_area
    :parameters (?picking_area - picking_area ?outbound_staging_area - outbound_staging_area ?pick_unit - pick_unit)
    :precondition
      (and
        (entity_ready_for_assignment ?picking_area)
        (picking_area_linked_outbound_staging ?picking_area ?outbound_staging_area)
        (pick_unit_available ?pick_unit)
        (not
          (picking_area_ready_for_load_build ?picking_area)
        )
      )
    :effect
      (and
        (outbound_staging_occupied ?outbound_staging_area)
        (picking_area_ready_for_load_build ?picking_area)
        (picking_area_pick_unit_assigned ?picking_area ?pick_unit)
        (not
          (pick_unit_available ?pick_unit)
        )
      )
  )
  (:action complete_picking_pick_unit_processing
    :parameters (?picking_area - picking_area ?outbound_staging_area - outbound_staging_area ?inbound_load - inbound_load ?pick_unit - pick_unit)
    :precondition
      (and
        (entity_ready_for_assignment ?picking_area)
        (inbound_load_attached_to_entity ?picking_area ?inbound_load)
        (picking_area_linked_outbound_staging ?picking_area ?outbound_staging_area)
        (outbound_staging_occupied ?outbound_staging_area)
        (picking_area_pick_unit_assigned ?picking_area ?pick_unit)
        (not
          (picking_area_processing_completed ?picking_area)
        )
      )
    :effect
      (and
        (outbound_staging_ready ?outbound_staging_area)
        (picking_area_processing_completed ?picking_area)
        (pick_unit_available ?pick_unit)
        (not
          (picking_area_pick_unit_assigned ?picking_area ?pick_unit)
        )
      )
  )
  (:action initialize_load_container_and_assign_staging
    :parameters (?receiving_area - receiving_area ?picking_area - picking_area ?inbound_staging_area - inbound_staging_area ?outbound_staging_area - outbound_staging_area ?load_container - load_container)
    :precondition
      (and
        (receiving_area_ready_for_load_build ?receiving_area)
        (picking_area_ready_for_load_build ?picking_area)
        (receiving_area_linked_inbound_staging ?receiving_area ?inbound_staging_area)
        (picking_area_linked_outbound_staging ?picking_area ?outbound_staging_area)
        (inbound_staging_ready ?inbound_staging_area)
        (outbound_staging_ready ?outbound_staging_area)
        (receiving_area_processing_completed ?receiving_area)
        (picking_area_processing_completed ?picking_area)
        (load_container_available ?load_container)
      )
    :effect
      (and
        (load_container_initialized ?load_container)
        (load_container_assigned_inbound_staging ?load_container ?inbound_staging_area)
        (load_container_assigned_outbound_staging ?load_container ?outbound_staging_area)
        (not
          (load_container_available ?load_container)
        )
      )
  )
  (:action initialize_container_with_inbound_occupancy
    :parameters (?receiving_area - receiving_area ?picking_area - picking_area ?inbound_staging_area - inbound_staging_area ?outbound_staging_area - outbound_staging_area ?load_container - load_container)
    :precondition
      (and
        (receiving_area_ready_for_load_build ?receiving_area)
        (picking_area_ready_for_load_build ?picking_area)
        (receiving_area_linked_inbound_staging ?receiving_area ?inbound_staging_area)
        (picking_area_linked_outbound_staging ?picking_area ?outbound_staging_area)
        (inbound_staging_occupied ?inbound_staging_area)
        (outbound_staging_ready ?outbound_staging_area)
        (not
          (receiving_area_processing_completed ?receiving_area)
        )
        (picking_area_processing_completed ?picking_area)
        (load_container_available ?load_container)
      )
    :effect
      (and
        (load_container_initialized ?load_container)
        (load_container_assigned_inbound_staging ?load_container ?inbound_staging_area)
        (load_container_assigned_outbound_staging ?load_container ?outbound_staging_area)
        (load_container_inbound_ready_flag ?load_container)
        (not
          (load_container_available ?load_container)
        )
      )
  )
  (:action initialize_container_with_outbound_occupancy
    :parameters (?receiving_area - receiving_area ?picking_area - picking_area ?inbound_staging_area - inbound_staging_area ?outbound_staging_area - outbound_staging_area ?load_container - load_container)
    :precondition
      (and
        (receiving_area_ready_for_load_build ?receiving_area)
        (picking_area_ready_for_load_build ?picking_area)
        (receiving_area_linked_inbound_staging ?receiving_area ?inbound_staging_area)
        (picking_area_linked_outbound_staging ?picking_area ?outbound_staging_area)
        (inbound_staging_ready ?inbound_staging_area)
        (outbound_staging_occupied ?outbound_staging_area)
        (receiving_area_processing_completed ?receiving_area)
        (not
          (picking_area_processing_completed ?picking_area)
        )
        (load_container_available ?load_container)
      )
    :effect
      (and
        (load_container_initialized ?load_container)
        (load_container_assigned_inbound_staging ?load_container ?inbound_staging_area)
        (load_container_assigned_outbound_staging ?load_container ?outbound_staging_area)
        (load_container_outbound_ready_flag ?load_container)
        (not
          (load_container_available ?load_container)
        )
      )
  )
  (:action initialize_container_with_both_occupancies
    :parameters (?receiving_area - receiving_area ?picking_area - picking_area ?inbound_staging_area - inbound_staging_area ?outbound_staging_area - outbound_staging_area ?load_container - load_container)
    :precondition
      (and
        (receiving_area_ready_for_load_build ?receiving_area)
        (picking_area_ready_for_load_build ?picking_area)
        (receiving_area_linked_inbound_staging ?receiving_area ?inbound_staging_area)
        (picking_area_linked_outbound_staging ?picking_area ?outbound_staging_area)
        (inbound_staging_occupied ?inbound_staging_area)
        (outbound_staging_occupied ?outbound_staging_area)
        (not
          (receiving_area_processing_completed ?receiving_area)
        )
        (not
          (picking_area_processing_completed ?picking_area)
        )
        (load_container_available ?load_container)
      )
    :effect
      (and
        (load_container_initialized ?load_container)
        (load_container_assigned_inbound_staging ?load_container ?inbound_staging_area)
        (load_container_assigned_outbound_staging ?load_container ?outbound_staging_area)
        (load_container_inbound_ready_flag ?load_container)
        (load_container_outbound_ready_flag ?load_container)
        (not
          (load_container_available ?load_container)
        )
      )
  )
  (:action complete_container_receiving_check
    :parameters (?load_container - load_container ?receiving_area - receiving_area ?inbound_load - inbound_load)
    :precondition
      (and
        (load_container_initialized ?load_container)
        (receiving_area_ready_for_load_build ?receiving_area)
        (inbound_load_attached_to_entity ?receiving_area ?inbound_load)
        (not
          (load_container_receiving_check_completed ?load_container)
        )
      )
    :effect (load_container_receiving_check_completed ?load_container)
  )
  (:action reserve_inventory_slot_and_bind_to_container
    :parameters (?load_task - load_task ?inventory_slot - inventory_slot ?load_container - load_container)
    :precondition
      (and
        (entity_ready_for_assignment ?load_task)
        (load_task_assigned_container ?load_task ?load_container)
        (load_task_assigned_inventory_slot ?load_task ?inventory_slot)
        (inventory_slot_available ?inventory_slot)
        (load_container_initialized ?load_container)
        (load_container_receiving_check_completed ?load_container)
        (not
          (inventory_slot_reserved ?inventory_slot)
        )
      )
    :effect
      (and
        (inventory_slot_reserved ?inventory_slot)
        (inventory_slot_assigned_to_container ?inventory_slot ?load_container)
        (not
          (inventory_slot_available ?inventory_slot)
        )
      )
  )
  (:action validate_slot_binding_and_mark_task_prepared
    :parameters (?load_task - load_task ?inventory_slot - inventory_slot ?load_container - load_container ?inbound_load - inbound_load)
    :precondition
      (and
        (entity_ready_for_assignment ?load_task)
        (load_task_assigned_inventory_slot ?load_task ?inventory_slot)
        (inventory_slot_reserved ?inventory_slot)
        (inventory_slot_assigned_to_container ?inventory_slot ?load_container)
        (inbound_load_attached_to_entity ?load_task ?inbound_load)
        (not
          (load_container_inbound_ready_flag ?load_container)
        )
        (not
          (load_task_prevalidation_completed ?load_task)
        )
      )
    :effect (load_task_prevalidation_completed ?load_task)
  )
  (:action allocate_equipment_to_load_task
    :parameters (?load_task - load_task ?equipment_unit - equipment_unit)
    :precondition
      (and
        (entity_ready_for_assignment ?load_task)
        (equipment_unit_available ?equipment_unit)
        (not
