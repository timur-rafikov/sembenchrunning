--- domain.pddl (as passed to VAL) ---
(define (domain rule_based_filtering_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types component - object auxiliary_component - object key_component - object entity_root - object filterable_item - entity_root rule_token - component matcher - component handler - component preference_tag - component modifier - component validation_token - component capability - component extension - component aux_resource - auxiliary_component attribute_group - auxiliary_component annotation - auxiliary_component selection_key_primary - key_component selection_key_secondary - key_component output_slot - key_component source_item_category - filterable_item candidate_item_category - filterable_item source_item_primary - source_item_category source_item_secondary - source_item_category filter_candidate - candidate_item_category)
  (:predicates
    (item_discovered ?filterable_item - filterable_item)
    (filterable_item_ready ?filterable_item - filterable_item)
    (rule_assigned ?filterable_item - filterable_item)
    (accepted ?filterable_item - filterable_item)
    (marked_for_finalization ?filterable_item - filterable_item)
    (validated ?filterable_item - filterable_item)
    (rule_token_available ?rule_token - rule_token)
    (has_rule_token ?filterable_item - filterable_item ?rule_token - rule_token)
    (matcher_available ?matcher - matcher)
    (matched_with ?filterable_item - filterable_item ?matcher - matcher)
    (handler_available ?handler - handler)
    (handler_bound ?filterable_item - filterable_item ?handler - handler)
    (aux_resource_available ?aux_resource - aux_resource)
    (aux_assigned_to_source_primary ?source_primary - source_item_primary ?aux_resource - aux_resource)
    (aux_assigned_to_source_secondary ?source_secondary - source_item_secondary ?aux_resource - aux_resource)
    (has_selection_key_primary ?source_primary - source_item_primary ?selection_key_primary - selection_key_primary)
    (key_primary_prepared ?selection_key_primary - selection_key_primary)
    (key_primary_reserved ?selection_key_primary - selection_key_primary)
    (source_primary_ready ?source_primary - source_item_primary)
    (has_selection_key_secondary ?source_secondary - source_item_secondary ?selection_key_secondary - selection_key_secondary)
    (key_secondary_prepared ?selection_key_secondary - selection_key_secondary)
    (key_secondary_reserved ?selection_key_secondary - selection_key_secondary)
    (source_secondary_ready ?source_secondary - source_item_secondary)
    (output_slot_available ?output_slot - output_slot)
    (output_slot_proposed ?output_slot - output_slot)
    (output_uses_key_primary ?output_slot - output_slot ?selection_key_primary - selection_key_primary)
    (output_uses_key_secondary ?output_slot - output_slot ?selection_key_secondary - selection_key_secondary)
    (slot_provisional_primary ?output_slot - output_slot)
    (slot_provisional_secondary ?output_slot - output_slot)
    (slot_locked ?output_slot - output_slot)
    (candidate_associated_with_source_primary ?candidate - filter_candidate ?source_primary - source_item_primary)
    (candidate_associated_with_source_secondary ?candidate - filter_candidate ?source_secondary - source_item_secondary)
    (candidate_assigned_to_slot ?candidate - filter_candidate ?output_slot - output_slot)
    (attribute_group_available ?attribute_group - attribute_group)
    (candidate_has_attribute_group ?candidate - filter_candidate ?attribute_group - attribute_group)
    (attribute_group_committed ?attribute_group - attribute_group)
    (attribute_group_allocated_to_slot ?attribute_group - attribute_group ?output_slot - output_slot)
    (candidate_attribute_evaluated ?candidate - filter_candidate)
    (candidate_capability_confirmed ?candidate - filter_candidate)
    (candidate_prefinal ?candidate - filter_candidate)
    (candidate_preference_flag ?candidate - filter_candidate)
    (candidate_preference_applied ?candidate - filter_candidate)
    (candidate_augmented ?candidate - filter_candidate)
    (candidate_ready_for_finalization ?candidate - filter_candidate)
    (annotation_available ?annotation - annotation)
    (candidate_has_annotation ?candidate - filter_candidate ?annotation - annotation)
    (annotation_selected_for_candidate ?candidate - filter_candidate)
    (annotation_applied ?candidate - filter_candidate)
    (annotation_validated ?candidate - filter_candidate)
    (preference_tag_available ?preference_tag - preference_tag)
    (candidate_has_preference_tag ?candidate - filter_candidate ?preference_tag - preference_tag)
    (modifier_available ?modifier - modifier)
    (candidate_has_modifier ?candidate - filter_candidate ?modifier - modifier)
    (capability_available ?capability - capability)
    (candidate_has_capability ?candidate - filter_candidate ?capability - capability)
    (extension_available ?extension - extension)
    (candidate_has_extension ?candidate - filter_candidate ?extension - extension)
    (validation_token_available ?validation_token - validation_token)
    (validation_assigned ?filterable_item - filterable_item ?validation_token - validation_token)
    (source_primary_prepared ?source_primary - source_item_primary)
    (source_secondary_prepared ?source_secondary - source_item_secondary)
    (finalization_token_acquired ?candidate - filter_candidate)
  )
  (:action discover_item
    :parameters (?filterable_item - filterable_item)
    :precondition
      (and
        (not
          (item_discovered ?filterable_item)
        )
        (not
          (accepted ?filterable_item)
        )
      )
    :effect (item_discovered ?filterable_item)
  )
  (:action assign_rule_token_to_item
    :parameters (?filterable_item - filterable_item ?rule_token - rule_token)
    :precondition
      (and
        (item_discovered ?filterable_item)
        (not
          (rule_assigned ?filterable_item)
        )
        (rule_token_available ?rule_token)
      )
    :effect
      (and
        (rule_assigned ?filterable_item)
        (has_rule_token ?filterable_item ?rule_token)
        (not
          (rule_token_available ?rule_token)
        )
      )
  )
  (:action apply_matcher_to_item
    :parameters (?filterable_item - filterable_item ?matcher - matcher)
    :precondition
      (and
        (item_discovered ?filterable_item)
        (rule_assigned ?filterable_item)
        (matcher_available ?matcher)
      )
    :effect
      (and
        (matched_with ?filterable_item ?matcher)
        (not
          (matcher_available ?matcher)
        )
      )
  )
  (:action activate_item
    :parameters (?filterable_item - filterable_item ?matcher - matcher)
    :precondition
      (and
        (item_discovered ?filterable_item)
        (rule_assigned ?filterable_item)
        (matched_with ?filterable_item ?matcher)
        (not
          (filterable_item_ready ?filterable_item)
        )
      )
    :effect (filterable_item_ready ?filterable_item)
  )
  (:action restore_matcher
    :parameters (?filterable_item - filterable_item ?matcher - matcher)
    :precondition
      (and
        (matched_with ?filterable_item ?matcher)
      )
    :effect
      (and
        (matcher_available ?matcher)
        (not
          (matched_with ?filterable_item ?matcher)
        )
      )
  )
  (:action bind_handler_to_item
    :parameters (?filterable_item - filterable_item ?handler - handler)
    :precondition
      (and
        (filterable_item_ready ?filterable_item)
        (handler_available ?handler)
      )
    :effect
      (and
        (handler_bound ?filterable_item ?handler)
        (not
          (handler_available ?handler)
        )
      )
  )
  (:action unbind_handler_from_item
    :parameters (?filterable_item - filterable_item ?handler - handler)
    :precondition
      (and
        (handler_bound ?filterable_item ?handler)
      )
    :effect
      (and
        (handler_available ?handler)
        (not
          (handler_bound ?filterable_item ?handler)
        )
      )
  )
  (:action allocate_capability_to_candidate
    :parameters (?candidate - filter_candidate ?capability - capability)
    :precondition
      (and
        (filterable_item_ready ?candidate)
        (capability_available ?capability)
      )
    :effect
      (and
        (candidate_has_capability ?candidate ?capability)
        (not
          (capability_available ?capability)
        )
      )
  )
  (:action release_capability_from_candidate
    :parameters (?candidate - filter_candidate ?capability - capability)
    :precondition
      (and
        (candidate_has_capability ?candidate ?capability)
      )
    :effect
      (and
        (capability_available ?capability)
        (not
          (candidate_has_capability ?candidate ?capability)
        )
      )
  )
  (:action allocate_extension_to_candidate
    :parameters (?candidate - filter_candidate ?extension - extension)
    :precondition
      (and
        (filterable_item_ready ?candidate)
        (extension_available ?extension)
      )
    :effect
      (and
        (candidate_has_extension ?candidate ?extension)
        (not
          (extension_available ?extension)
        )
      )
  )
  (:action release_extension_from_candidate
    :parameters (?candidate - filter_candidate ?extension - extension)
    :precondition
      (and
        (candidate_has_extension ?candidate ?extension)
      )
    :effect
      (and
        (extension_available ?extension)
        (not
          (candidate_has_extension ?candidate ?extension)
        )
      )
  )
  (:action detect_selection_key_primary
    :parameters (?source_primary - source_item_primary ?selection_key_primary - selection_key_primary ?matcher - matcher)
    :precondition
      (and
        (filterable_item_ready ?source_primary)
        (matched_with ?source_primary ?matcher)
        (has_selection_key_primary ?source_primary ?selection_key_primary)
        (not
          (key_primary_prepared ?selection_key_primary)
        )
        (not
          (key_primary_reserved ?selection_key_primary)
        )
      )
    :effect (key_primary_prepared ?selection_key_primary)
  )
  (:action confirm_key_with_handler_primary
    :parameters (?source_primary - source_item_primary ?selection_key_primary - selection_key_primary ?handler - handler)
    :precondition
      (and
        (filterable_item_ready ?source_primary)
        (handler_bound ?source_primary ?handler)
        (has_selection_key_primary ?source_primary ?selection_key_primary)
        (key_primary_prepared ?selection_key_primary)
        (not
          (source_primary_prepared ?source_primary)
        )
      )
    :effect
      (and
        (source_primary_prepared ?source_primary)
        (source_primary_ready ?source_primary)
      )
  )
  (:action assign_aux_resource_to_source_primary
    :parameters (?source_primary - source_item_primary ?selection_key_primary - selection_key_primary ?aux_resource - aux_resource)
    :precondition
      (and
        (filterable_item_ready ?source_primary)
        (has_selection_key_primary ?source_primary ?selection_key_primary)
        (aux_resource_available ?aux_resource)
        (not
          (source_primary_prepared ?source_primary)
        )
      )
    :effect
      (and
        (key_primary_reserved ?selection_key_primary)
        (source_primary_prepared ?source_primary)
        (aux_assigned_to_source_primary ?source_primary ?aux_resource)
        (not
          (aux_resource_available ?aux_resource)
        )
      )
  )
  (:action finalize_key_and_aux_for_primary
    :parameters (?source_primary - source_item_primary ?selection_key_primary - selection_key_primary ?matcher - matcher ?aux_resource - aux_resource)
    :precondition
      (and
        (filterable_item_ready ?source_primary)
        (matched_with ?source_primary ?matcher)
        (has_selection_key_primary ?source_primary ?selection_key_primary)
        (key_primary_reserved ?selection_key_primary)
        (aux_assigned_to_source_primary ?source_primary ?aux_resource)
        (not
          (source_primary_ready ?source_primary)
        )
      )
    :effect
      (and
        (key_primary_prepared ?selection_key_primary)
        (source_primary_ready ?source_primary)
        (aux_resource_available ?aux_resource)
        (not
          (aux_assigned_to_source_primary ?source_primary ?aux_resource)
        )
      )
  )
  (:action detect_selection_key_secondary
    :parameters (?source_secondary - source_item_secondary ?selection_key_secondary - selection_key_secondary ?matcher - matcher)
    :precondition
      (and
        (filterable_item_ready ?source_secondary)
        (matched_with ?source_secondary ?matcher)
        (has_selection_key_secondary ?source_secondary ?selection_key_secondary)
        (not
          (key_secondary_prepared ?selection_key_secondary)
        )
        (not
          (key_secondary_reserved ?selection_key_secondary)
        )
      )
    :effect (key_secondary_prepared ?selection_key_secondary)
  )
  (:action confirm_key_with_handler_secondary
    :parameters (?source_secondary - source_item_secondary ?selection_key_secondary - selection_key_secondary ?handler - handler)
    :precondition
      (and
        (filterable_item_ready ?source_secondary)
        (handler_bound ?source_secondary ?handler)
        (has_selection_key_secondary ?source_secondary ?selection_key_secondary)
        (key_secondary_prepared ?selection_key_secondary)
        (not
          (source_secondary_prepared ?source_secondary)
        )
      )
    :effect
      (and
        (source_secondary_prepared ?source_secondary)
        (source_secondary_ready ?source_secondary)
      )
  )
  (:action assign_aux_resource_to_source_secondary
    :parameters (?source_secondary - source_item_secondary ?selection_key_secondary - selection_key_secondary ?aux_resource - aux_resource)
    :precondition
      (and
        (filterable_item_ready ?source_secondary)
        (has_selection_key_secondary ?source_secondary ?selection_key_secondary)
        (aux_resource_available ?aux_resource)
        (not
          (source_secondary_prepared ?source_secondary)
        )
      )
    :effect
      (and
        (key_secondary_reserved ?selection_key_secondary)
        (source_secondary_prepared ?source_secondary)
        (aux_assigned_to_source_secondary ?source_secondary ?aux_resource)
        (not
          (aux_resource_available ?aux_resource)
        )
      )
  )
  (:action finalize_key_and_aux_for_secondary
    :parameters (?source_secondary - source_item_secondary ?selection_key_secondary - selection_key_secondary ?matcher - matcher ?aux_resource - aux_resource)
    :precondition
      (and
        (filterable_item_ready ?source_secondary)
        (matched_with ?source_secondary ?matcher)
        (has_selection_key_secondary ?source_secondary ?selection_key_secondary)
        (key_secondary_reserved ?selection_key_secondary)
        (aux_assigned_to_source_secondary ?source_secondary ?aux_resource)
        (not
          (source_secondary_ready ?source_secondary)
        )
      )
    :effect
      (and
        (key_secondary_prepared ?selection_key_secondary)
        (source_secondary_ready ?source_secondary)
        (aux_resource_available ?aux_resource)
        (not
          (aux_assigned_to_source_secondary ?source_secondary ?aux_resource)
        )
      )
  )
  (:action assemble_output_slot
    :parameters (?source_primary - source_item_primary ?source_secondary - source_item_secondary ?selection_key_primary - selection_key_primary ?selection_key_secondary - selection_key_secondary ?output_slot - output_slot)
    :precondition
      (and
        (source_primary_prepared ?source_primary)
        (source_secondary_prepared ?source_secondary)
        (has_selection_key_primary ?source_primary ?selection_key_primary)
        (has_selection_key_secondary ?source_secondary ?selection_key_secondary)
        (key_primary_prepared ?selection_key_primary)
        (key_secondary_prepared ?selection_key_secondary)
        (source_primary_ready ?source_primary)
        (source_secondary_ready ?source_secondary)
        (output_slot_available ?output_slot)
      )
    :effect
      (and
        (output_slot_proposed ?output_slot)
        (output_uses_key_primary ?output_slot ?selection_key_primary)
        (output_uses_key_secondary ?output_slot ?selection_key_secondary)
        (not
          (output_slot_available ?output_slot)
        )
      )
  )
  (:action assemble_output_slot_with_flag1
    :parameters (?source_primary - source_item_primary ?source_secondary - source_item_secondary ?selection_key_primary - selection_key_primary ?selection_key_secondary - selection_key_secondary ?output_slot - output_slot)
    :precondition
      (and
        (source_primary_prepared ?source_primary)
        (source_secondary_prepared ?source_secondary)
        (has_selection_key_primary ?source_primary ?selection_key_primary)
        (has_selection_key_secondary ?source_secondary ?selection_key_secondary)
        (key_primary_reserved ?selection_key_primary)
        (key_secondary_prepared ?selection_key_secondary)
        (not
          (source_primary_ready ?source_primary)
        )
        (source_secondary_ready ?source_secondary)
        (output_slot_available ?output_slot)
      )
    :effect
      (and
        (output_slot_proposed ?output_slot)
        (output_uses_key_primary ?output_slot ?selection_key_primary)
        (output_uses_key_secondary ?output_slot ?selection_key_secondary)
        (slot_provisional_primary ?output_slot)
        (not
          (output_slot_available ?output_slot)
        )
      )
  )
  (:action assemble_output_slot_with_flag2
    :parameters (?source_primary - source_item_primary ?source_secondary - source_item_secondary ?selection_key_primary - selection_key_primary ?selection_key_secondary - selection_key_secondary ?output_slot - output_slot)
    :precondition
      (and
        (source_primary_prepared ?source_primary)
        (source_secondary_prepared ?source_secondary)
        (has_selection_key_primary ?source_primary ?selection_key_primary)
        (has_selection_key_secondary ?source_secondary ?selection_key_secondary)
        (key_primary_prepared ?selection_key_primary)
        (key_secondary_reserved ?selection_key_secondary)
        (source_primary_ready ?source_primary)
        (not
          (source_secondary_ready ?source_secondary)
        )
        (output_slot_available ?output_slot)
      )
    :effect
      (and
        (output_slot_proposed ?output_slot)
        (output_uses_key_primary ?output_slot ?selection_key_primary)
        (output_uses_key_secondary ?output_slot ?selection_key_secondary)
        (slot_provisional_secondary ?output_slot)
        (not
          (output_slot_available ?output_slot)
        )
      )
  )
  (:action assemble_output_slot_with_both_flags
    :parameters (?source_primary - source_item_primary ?source_secondary - source_item_secondary ?selection_key_primary - selection_key_primary ?selection_key_secondary - selection_key_secondary ?output_slot - output_slot)
    :precondition
      (and
        (source_primary_prepared ?source_primary)
        (source_secondary_prepared ?source_secondary)
        (has_selection_key_primary ?source_primary ?selection_key_primary)
        (has_selection_key_secondary ?source_secondary ?selection_key_secondary)
        (key_primary_reserved ?selection_key_primary)
        (key_secondary_reserved ?selection_key_secondary)
        (not
          (source_primary_ready ?source_primary)
        )
        (not
          (source_secondary_ready ?source_secondary)
        )
        (output_slot_available ?output_slot)
      )
    :effect
      (and
        (output_slot_proposed ?output_slot)
        (output_uses_key_primary ?output_slot ?selection_key_primary)
        (output_uses_key_secondary ?output_slot ?selection_key_secondary)
        (slot_provisional_primary ?output_slot)
        (slot_provisional_secondary ?output_slot)
        (not
          (output_slot_available ?output_slot)
        )
      )
  )
  (:action lock_output_slot
    :parameters (?output_slot - output_slot ?source_primary - source_item_primary ?matcher - matcher)
    :precondition
      (and
        (output_slot_proposed ?output_slot)
        (source_primary_prepared ?source_primary)
        (matched_with ?source_primary ?matcher)
        (not
          (slot_locked ?output_slot)
        )
      )
    :effect (slot_locked ?output_slot)
  )
  (:action commit_attribute_group_to_slot
    :parameters (?candidate - filter_candidate ?attribute_group - attribute_group ?output_slot - output_slot)
    :precondition
      (and
        (filterable_item_ready ?candidate)
        (candidate_assigned_to_slot ?candidate ?output_slot)
        (candidate_has_attribute_group ?candidate ?attribute_group)
        (attribute_group_available ?attribute_group)
        (output_slot_proposed ?output_slot)
        (slot_locked ?output_slot)
        (not
          (attribute_group_committed ?attribute_group)
        )
      )
    :effect
      (and
        (attribute_group_committed ?attribute_group)
        (attribute_group_allocated_to_slot ?attribute_group ?output_slot)
        (not
          (attribute_group_available ?attribute_group)
        )
      )
  )
  (:action evaluate_candidate_attributes
    :parameters (?candidate - filter_candidate ?attribute_group - attribute_group ?output_slot - output_slot ?matcher - matcher)
    :precondition
      (and
        (filterable_item_ready ?candidate)
        (candidate_has_attribute_group ?candidate ?attribute_group)
        (attribute_group_committed ?attribute_group)
        (attribute_group_allocated_to_slot ?attribute_group ?output_slot)
        (matched_with ?candidate ?matcher)
        (not
          (slot_provisional_primary ?output_slot)
        )
        (not
          (candidate_attribute_evaluated ?candidate)
        )
      )
    :effect (candidate_attribute_evaluated ?candidate)
  )
  (:action attach_preference_tag
    :parameters (?candidate - filter_candidate ?preference_tag - preference_tag)
    :precondition
      (and
        (filterable_item_ready ?candidate)
        (preference_tag_available ?preference_tag)
        (not
          (candidate_preference_flag ?candidate)
        )
      )
    :effect
      (and
        (candidate_preference_flag ?candidate)
        (candidate_has_preference_tag ?candidate ?preference_tag)
        (not
          (preference_tag_available ?preference_tag)
        )
      )
  )
  (:action apply_preference_and_augment_candidate
    :parameters (?candidate - filter_candidate ?attribute_group - attribute_group ?output_slot - output_slot ?matcher - matcher ?preference_tag - preference_tag)
    :precondition
      (and
        (filterable_item_ready ?candidate)
        (candidate_has_attribute_group ?candidate ?attribute_group)
        (attribute_group_committed ?attribute_group)
        (attribute_group_allocated_to_slot ?attribute_group ?output_slot)
        (matched_with ?candidate ?matcher)
        (slot_provisional_primary ?output_slot)
        (candidate_preference_flag ?candidate)
        (candidate_has_preference_tag ?candidate ?preference_tag)
        (not
          (candidate_attribute_evaluated ?candidate)
        )
      )
    :effect
      (and
        (candidate_attribute_evaluated ?candidate)
        (candidate_preference_applied ?candidate)
      )
  )
  (:action confirm_candidate_capability_stage1
    :parameters (?candidate - filter_candidate ?capability - capability ?handler - handler ?attribute_group - attribute_group ?output_slot - output_slot)
    :precondition
      (and
        (candidate_attribute_evaluated ?candidate)
        (candidate_has_capability ?candidate ?capability)
        (handler_bound ?candidate ?handler)
        (candidate_has_attribute_group ?candidate ?attribute_group)
        (attribute_group_allocated_to_slot ?attribute_group ?output_slot)
        (not
          (slot_provisional_secondary ?output_slot)
        )
        (not
          (candidate_capability_confirmed ?candidate)
        )
      )
    :effect (candidate_capability_confirmed ?candidate)
  )
  (:action confirm_candidate_capability_stage2
    :parameters (?candidate - filter_candidate ?capability - capability ?handler - handler ?attribute_group - attribute_group ?output_slot - output_slot)
    :precondition
      (and
        (candidate_attribute_evaluated ?candidate)
        (candidate_has_capability ?candidate ?capability)
        (handler_bound ?candidate ?handler)
        (candidate_has_attribute_group ?candidate ?attribute_group)
        (attribute_group_allocated_to_slot ?attribute_group ?output_slot)
        (slot_provisional_secondary ?output_slot)
        (not
          (candidate_capability_confirmed ?candidate)
        )
      )
    :effect (candidate_capability_confirmed ?candidate)
  )
  (:action evaluate_candidate_with_extension_stage1
    :parameters (?candidate - filter_candidate ?extension - extension ?attribute_group - attribute_group ?output_slot - output_slot)
    :precondition
      (and
        (candidate_capability_confirmed ?candidate)
        (candidate_has_extension ?candidate ?extension)
        (candidate_has_attribute_group ?candidate ?attribute_group)
        (attribute_group_allocated_to_slot ?attribute_group ?output_slot)
        (not
          (slot_provisional_primary ?output_slot)
        )
        (not
          (slot_provisional_secondary ?output_slot)
        )
        (not
          (candidate_prefinal ?candidate)
        )
      )
    :effect (candidate_prefinal ?candidate)
  )
  (:action evaluate_candidate_with_extension_stage2
    :parameters (?candidate - filter_candidate ?extension - extension ?attribute_group - attribute_group ?output_slot - output_slot)
    :precondition
      (and
        (candidate_capability_confirmed ?candidate)
        (candidate_has_extension ?candidate ?extension)
        (candidate_has_attribute_group ?candidate ?attribute_group)
        (attribute_group_allocated_to_slot ?attribute_group ?output_slot)
        (slot_provisional_primary ?output_slot)
        (not
          (slot_provisional_secondary ?output_slot)
        )
