    :parameters (?customs_hold_case - customs_hold_case)
    :precondition
      (and
        (not
          (subject_logged ?customs_hold_case)
        )
        (not
          (subject_closed_or_released ?customs_hold_case)
        )
      )
    :effect (subject_logged ?customs_hold_case)
  )
  (:action assign_release_channel_to_case
    :parameters (?customs_hold_case - customs_hold_case ?release_channel - release_channel)
    :precondition
      (and
        (subject_logged ?customs_hold_case)
        (not
          (release_action_initiated ?customs_hold_case)
        )
        (release_channel_available ?release_channel)
      )
    :effect
      (and
        (release_action_initiated ?customs_hold_case)
        (subject_assigned_channel ?customs_hold_case ?release_channel)
        (not
          (release_channel_available ?release_channel)
        )
      )
  )
