  (:action register_participant
    :parameters (?case_participant - case_participant)
    :precondition
      (and
        (not
          (participant_registered ?case_participant)
        )
        (not
          (participant_activated ?case_participant)
        )
      )
    :effect (participant_registered ?case_participant)
  )
  (:action assign_role_profile_to_participant
    :parameters (?case_participant - case_participant ?role_profile - role_profile)
    :precondition
      (and
        (participant_registered ?case_participant)
        (not
          (role_assigned ?case_participant)
        )
        (role_profile_available ?role_profile)
      )
    :effect
      (and
        (role_assigned ?case_participant)
        (assigned_role_profile ?case_participant ?role_profile)
        (not
          (role_profile_available ?role_profile)
        )
      )
  )
  (:action schedule_assessment_slot
    :parameters (?case_participant - case_participant ?assessment_slot - assessment_slot)
    :precondition
      (and
        (participant_registered ?case_participant)
        (role_assigned ?case_participant)
        (assessment_slot_available ?assessment_slot)
      )
    :effect
      (and
        (participant_session_scheduled ?case_participant ?assessment_slot)
        (not
          (assessment_slot_available ?assessment_slot)
        )
      )
  )
  (:action complete_assessment_slot
    :parameters (?case_participant - case_participant ?assessment_slot - assessment_slot)
    :precondition
      (and
        (participant_registered ?case_participant)
        (role_assigned ?case_participant)
        (participant_session_scheduled ?case_participant ?assessment_slot)
        (not
          (intake_completed ?case_participant)
        )
      )
    :effect (intake_completed ?case_participant)
  )
  (:action cancel_assessment_slot
    :parameters (?case_participant - case_participant ?assessment_slot - assessment_slot)
    :precondition
      (and
        (participant_session_scheduled ?case_participant ?assessment_slot)
      )
    :effect
      (and
        (assessment_slot_available ?assessment_slot)
        (not
          (participant_session_scheduled ?case_participant ?assessment_slot)
        )
      )
  )
  (:action assign_support_staff_to_participant
    :parameters (?case_participant - case_participant ?support_staff - support_staff)
    :precondition
      (and
        (intake_completed ?case_participant)
        (support_staff_available ?support_staff)
      )
    :effect
      (and
        (support_staff_assigned ?case_participant ?support_staff)
        (not
          (support_staff_available ?support_staff)
        )
      )
  )
  (:action release_support_staff_from_participant
    :parameters (?case_participant - case_participant ?support_staff - support_staff)

      (and
        (support_staff_assigned ?case_participant ?support_staff)
      )
    :effect
      (and
        (support_staff_available ?support_staff)
        (not
          (support_staff_assigned ?case_participant ?support_staff)
        )
      )
  )
  (:action assign_accommodation_option_to_coordinator

        (intake_completed ?case_coordinator)

      (and
        (coordinator_assigned_accommodation ?case_coordinator ?accommodation_option)
      )

      (and

          (coordinator_assigned_accommodation ?case_coordinator ?accommodation_option)
        )
      )

  (:action assign_external_partner_to_coordinator
    :parameters (?case_coordinator - case_coordinator ?external_referral_partner - external_referral_partner)

      (and

      (and
        (coordinator_assigned_external_partner ?case_coordinator ?external_referral_partner)
      )
    :effect
      (and
        (external_partner_available ?external_referral_partner)

      (and
        (intake_completed ?mentee)
        (participant_session_scheduled ?mentee ?assessment_slot)
        (mentee_has_need_area ?mentee ?mentoring_need_area)
        (not
          (need_assessed ?mentoring_need_area)
        )
        (not
          (need_referred ?mentoring_need_area)

    :effect (need_assessed ?mentoring_need_area)

  (:action assign_staff_for_mentee_need
    :parameters (?mentee - mentee ?mentoring_need_area - mentoring_need_area ?support_staff - support_staff)

      (and

        (support_staff_assigned ?mentee ?support_staff)
        (mentee_has_need_area ?mentee ?mentoring_need_area)
        (need_assessed ?mentoring_need_area)
        (not
          (mentee_intervention_flag ?mentee)

      (and
        (mentee_intervention_flag ?mentee)

          (mentee_intervention_flag ?mentee)

        (need_referred ?mentoring_need_area)
