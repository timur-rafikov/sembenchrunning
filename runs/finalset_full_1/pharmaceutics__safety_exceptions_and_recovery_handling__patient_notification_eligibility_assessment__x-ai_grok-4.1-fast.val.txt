--- domain.pddl (as passed to VAL) ---
(define (domain pharmaceutics_patient_notification_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types organizational_entity - object resource - object asset - object case_category - object domain_entity - case_category notification_channel - organizational_entity product_lot - organizational_entity subject_matter_expert - organizational_entity regulatory_body - organizational_entity message_template - organizational_entity eligibility_criterion - organizational_entity escalation_approval - organizational_entity public_health_authorization - organizational_entity mitigation_option - resource recipient_list - resource legal_clearance_document - resource distribution_channel - asset dispensing_site - asset notification_bundle - asset site_group - domain_entity team_group - domain_entity origin_site - site_group receiving_site - site_group response_coordinator - team_group)
  (:predicates
    (safety_event_recorded ?safety_event - domain_entity)
    (entity_assessment_confirmed ?safety_event - domain_entity)
    (entity_notification_channel_allocated ?safety_event - domain_entity)
    (eligibility_propagated ?safety_event - domain_entity)
    (entity_internal_approval_granted ?safety_event - domain_entity)
    (entity_eligibility_determined ?safety_event - domain_entity)
    (notification_channel_available ?notification_channel - notification_channel)
    (entity_notification_channel_linked ?safety_event - domain_entity ?notification_channel - notification_channel)
    (product_lot_available ?product_lot - product_lot)
    (entity_product_lot_linked ?safety_event - domain_entity ?product_lot - product_lot)
    (subject_matter_expert_available ?subject_matter_expert - subject_matter_expert)
    (entity_sme_assigned ?safety_event - domain_entity ?subject_matter_expert - subject_matter_expert)
    (mitigation_option_available ?mitigation_option - mitigation_option)
    (origin_site_mitigation_assigned ?origin_site - origin_site ?mitigation_option - mitigation_option)
    (receiving_site_mitigation_assigned ?receiving_site - receiving_site ?mitigation_option - mitigation_option)
    (origin_site_channel_linked ?origin_site - origin_site ?distribution_channel - distribution_channel)
    (distribution_channel_confirmed ?distribution_channel - distribution_channel)
    (distribution_channel_suspected ?distribution_channel - distribution_channel)
    (origin_site_confirmed_exposure ?origin_site - origin_site)
    (receiving_site_dispensing_link ?receiving_site - receiving_site ?dispensing_site - dispensing_site)
    (dispensing_site_confirmed_exposure ?dispensing_site - dispensing_site)
    (dispensing_site_suspected_exposure ?dispensing_site - dispensing_site)
    (receiving_site_confirmed_exposure ?receiving_site - receiving_site)
    (notification_bundle_available ?notification_bundle - notification_bundle)
    (notification_bundle_registered ?notification_bundle - notification_bundle)
    (notification_bundle_channel_linked ?notification_bundle - notification_bundle ?distribution_channel - distribution_channel)
    (notification_bundle_dispensing_site_linked ?notification_bundle - notification_bundle ?dispensing_site - dispensing_site)
    (bundle_channel_followup_required ?notification_bundle - notification_bundle)
    (bundle_site_followup_required ?notification_bundle - notification_bundle)
    (bundle_ready_for_recipient_assignment ?notification_bundle - notification_bundle)
    (coordinator_assigned_origin_site ?response_coordinator - response_coordinator ?origin_site - origin_site)
    (coordinator_assigned_receiving_site ?response_coordinator - response_coordinator ?receiving_site - receiving_site)
    (coordinator_assigned_notification_bundle ?response_coordinator - response_coordinator ?notification_bundle - notification_bundle)
    (recipient_list_available ?recipient_list - recipient_list)
    (coordinator_has_recipient_list ?response_coordinator - response_coordinator ?recipient_list - recipient_list)
    (recipient_list_attached ?recipient_list - recipient_list)
    (recipient_list_linked_to_bundle ?recipient_list - recipient_list ?notification_bundle - notification_bundle)
    (coordinator_enrichment_allowed ?response_coordinator - response_coordinator)
    (coordinator_enrichment_complete ?response_coordinator - response_coordinator)
    (package_enrichment_finalized ?response_coordinator - response_coordinator)
    (regulatory_engagement_initiated ?response_coordinator - response_coordinator)
    (regulatory_review_completed ?response_coordinator - response_coordinator)
    (template_attached_to_package ?response_coordinator - response_coordinator)
    (internal_checks_completed ?response_coordinator - response_coordinator)
    (legal_clearance_available ?legal_clearance_document - legal_clearance_document)
    (coordinator_has_legal_clearance ?response_coordinator - response_coordinator ?legal_clearance_document - legal_clearance_document)
    (legal_clearance_assigned_to_coordinator ?response_coordinator - response_coordinator)
    (legal_clearance_validated ?response_coordinator - response_coordinator)
    (public_health_authorization_recorded ?response_coordinator - response_coordinator)
    (regulatory_body_available ?regulatory_body - regulatory_body)
    (coordinator_engaged_regulatory_body ?response_coordinator - response_coordinator ?regulatory_body - regulatory_body)
    (message_template_available ?message_template - message_template)
    (coordinator_attached_message_template ?response_coordinator - response_coordinator ?message_template - message_template)
    (escalation_approval_available ?escalation_approval - escalation_approval)
    (coordinator_attached_escalation_approval ?response_coordinator - response_coordinator ?escalation_approval - escalation_approval)
    (public_health_authorization_available ?public_health_authorization - public_health_authorization)
    (coordinator_attached_public_health_authorization ?response_coordinator - response_coordinator ?public_health_authorization - public_health_authorization)
    (eligibility_criterion_available ?eligibility_criterion - eligibility_criterion)
    (entity_eligibility_criterion_recorded ?safety_event - domain_entity ?eligibility_criterion - eligibility_criterion)
    (site_ready_for_notification_bundle ?origin_site - origin_site)
    (receiving_site_ready_for_notification_bundle ?receiving_site - receiving_site)
    (coordinator_signoff_recorded ?response_coordinator - response_coordinator)
  )
  (:action register_safety_event
    :parameters (?safety_event - domain_entity)
    :precondition
      (and
        (not
          (safety_event_recorded ?safety_event)
        )
        (not
          (eligibility_propagated ?safety_event)
        )
      )
    :effect (safety_event_recorded ?safety_event)
  )
  (:action assign_notification_channel_to_event
    :parameters (?safety_event - domain_entity ?notification_channel - notification_channel)
    :precondition
      (and
        (safety_event_recorded ?safety_event)
        (not
          (entity_notification_channel_allocated ?safety_event)
        )
        (notification_channel_available ?notification_channel)
      )
    :effect
      (and
        (entity_notification_channel_allocated ?safety_event)
        (entity_notification_channel_linked ?safety_event ?notification_channel)
        (not
          (notification_channel_available ?notification_channel)
        )
      )
  )
  (:action associate_product_lot_with_event
    :parameters (?safety_event - domain_entity ?product_lot - product_lot)
    :precondition
      (and
        (safety_event_recorded ?safety_event)
        (entity_notification_channel_allocated ?safety_event)
        (product_lot_available ?product_lot)
      )
    :effect
      (and
        (entity_product_lot_linked ?safety_event ?product_lot)
        (not
          (product_lot_available ?product_lot)
        )
      )
  )
  (:action confirm_safety_event
    :parameters (?safety_event - domain_entity ?product_lot - product_lot)
    :precondition
      (and
        (safety_event_recorded ?safety_event)
        (entity_notification_channel_allocated ?safety_event)
        (entity_product_lot_linked ?safety_event ?product_lot)
        (not
          (entity_assessment_confirmed ?safety_event)
        )
      )
    :effect (entity_assessment_confirmed ?safety_event)
  )
  (:action release_product_lot
    :parameters (?safety_event - domain_entity ?product_lot - product_lot)
    :precondition
      (and
        (entity_product_lot_linked ?safety_event ?product_lot)
      )
    :effect
      (and
        (product_lot_available ?product_lot)
        (not
          (entity_product_lot_linked ?safety_event ?product_lot)
        )
      )
  )
  (:action assign_sme_to_event
    :parameters (?safety_event - domain_entity ?subject_matter_expert - subject_matter_expert)
    :precondition
      (and
        (entity_assessment_confirmed ?safety_event)
        (subject_matter_expert_available ?subject_matter_expert)
      )
    :effect
      (and
        (entity_sme_assigned ?safety_event ?subject_matter_expert)
        (not
          (subject_matter_expert_available ?subject_matter_expert)
        )
      )
  )
  (:action unassign_sme_from_event
    :parameters (?safety_event - domain_entity ?subject_matter_expert - subject_matter_expert)
    :precondition
      (and
        (entity_sme_assigned ?safety_event ?subject_matter_expert)
      )
    :effect
      (and
        (subject_matter_expert_available ?subject_matter_expert)
        (not
          (entity_sme_assigned ?safety_event ?subject_matter_expert)
        )
      )
  )
  (:action attach_escalation_approval_to_coordinator
    :parameters (?response_coordinator - response_coordinator ?escalation_approval - escalation_approval)
    :precondition
      (and
        (entity_assessment_confirmed ?response_coordinator)
        (escalation_approval_available ?escalation_approval)
      )
    :effect
      (and
        (coordinator_attached_escalation_approval ?response_coordinator ?escalation_approval)
        (not
          (escalation_approval_available ?escalation_approval)
        )
      )
  )
  (:action detach_escalation_approval_from_coordinator
    :parameters (?response_coordinator - response_coordinator ?escalation_approval - escalation_approval)
    :precondition
      (and
        (coordinator_attached_escalation_approval ?response_coordinator ?escalation_approval)
      )
    :effect
      (and
        (escalation_approval_available ?escalation_approval)
        (not
          (coordinator_attached_escalation_approval ?response_coordinator ?escalation_approval)
        )
      )
  )
  (:action attach_public_health_authorization_to_coordinator
    :parameters (?response_coordinator - response_coordinator ?public_health_authorization - public_health_authorization)
    :precondition
      (and
        (entity_assessment_confirmed ?response_coordinator)
        (public_health_authorization_available ?public_health_authorization)
      )
    :effect
      (and
        (coordinator_attached_public_health_authorization ?response_coordinator ?public_health_authorization)
        (not
          (public_health_authorization_available ?public_health_authorization)
        )
      )
  )
  (:action detach_public_health_authorization_from_coordinator
    :parameters (?response_coordinator - response_coordinator ?public_health_authorization - public_health_authorization)
    :precondition
      (and
        (coordinator_attached_public_health_authorization ?response_coordinator ?public_health_authorization)
      )
    :effect
      (and
        (public_health_authorization_available ?public_health_authorization)
        (not
          (coordinator_attached_public_health_authorization ?response_coordinator ?public_health_authorization)
        )
      )
  )
  (:action confirm_distribution_channel_exposure
    :parameters (?origin_site - origin_site ?distribution_channel - distribution_channel ?product_lot - product_lot)
    :precondition
      (and
        (entity_assessment_confirmed ?origin_site)
        (entity_product_lot_linked ?origin_site ?product_lot)
        (origin_site_channel_linked ?origin_site ?distribution_channel)
        (not
          (distribution_channel_confirmed ?distribution_channel)
        )
        (not
          (distribution_channel_suspected ?distribution_channel)
        )
      )
    :effect (distribution_channel_confirmed ?distribution_channel)
  )
  (:action confirm_origin_site_exposure
    :parameters (?origin_site - origin_site ?distribution_channel - distribution_channel ?subject_matter_expert - subject_matter_expert)
    :precondition
      (and
        (entity_assessment_confirmed ?origin_site)
        (entity_sme_assigned ?origin_site ?subject_matter_expert)
        (origin_site_channel_linked ?origin_site ?distribution_channel)
        (distribution_channel_confirmed ?distribution_channel)
        (not
          (site_ready_for_notification_bundle ?origin_site)
        )
      )
    :effect
      (and
        (site_ready_for_notification_bundle ?origin_site)
        (origin_site_confirmed_exposure ?origin_site)
      )
  )
  (:action assign_mitigation_option_to_origin_site
    :parameters (?origin_site - origin_site ?distribution_channel - distribution_channel ?mitigation_option - mitigation_option)
    :precondition
      (and
        (entity_assessment_confirmed ?origin_site)
        (origin_site_channel_linked ?origin_site ?distribution_channel)
        (mitigation_option_available ?mitigation_option)
        (not
          (site_ready_for_notification_bundle ?origin_site)
        )
      )
    :effect
      (and
        (distribution_channel_suspected ?distribution_channel)
        (site_ready_for_notification_bundle ?origin_site)
        (origin_site_mitigation_assigned ?origin_site ?mitigation_option)
        (not
          (mitigation_option_available ?mitigation_option)
        )
      )
  )
  (:action finalize_origin_site_mitigation
    :parameters (?origin_site - origin_site ?distribution_channel - distribution_channel ?product_lot - product_lot ?mitigation_option - mitigation_option)
    :precondition
      (and
        (entity_assessment_confirmed ?origin_site)
        (entity_product_lot_linked ?origin_site ?product_lot)
        (origin_site_channel_linked ?origin_site ?distribution_channel)
        (distribution_channel_suspected ?distribution_channel)
        (origin_site_mitigation_assigned ?origin_site ?mitigation_option)
        (not
          (origin_site_confirmed_exposure ?origin_site)
        )
      )
    :effect
      (and
        (distribution_channel_confirmed ?distribution_channel)
        (origin_site_confirmed_exposure ?origin_site)
        (mitigation_option_available ?mitigation_option)
        (not
          (origin_site_mitigation_assigned ?origin_site ?mitigation_option)
        )
      )
  )
  (:action confirm_dispensing_site_exposure
    :parameters (?receiving_site - receiving_site ?dispensing_site - dispensing_site ?product_lot - product_lot)
    :precondition
      (and
        (entity_assessment_confirmed ?receiving_site)
        (entity_product_lot_linked ?receiving_site ?product_lot)
        (receiving_site_dispensing_link ?receiving_site ?dispensing_site)
        (not
          (dispensing_site_confirmed_exposure ?dispensing_site)
        )
        (not
          (dispensing_site_suspected_exposure ?dispensing_site)
        )
      )
    :effect (dispensing_site_confirmed_exposure ?dispensing_site)
  )
  (:action confirm_receiving_site_exposure
    :parameters (?receiving_site - receiving_site ?dispensing_site - dispensing_site ?subject_matter_expert - subject_matter_expert)
    :precondition
      (and
        (entity_assessment_confirmed ?receiving_site)
        (entity_sme_assigned ?receiving_site ?subject_matter_expert)
        (receiving_site_dispensing_link ?receiving_site ?dispensing_site)
        (dispensing_site_confirmed_exposure ?dispensing_site)
        (not
          (receiving_site_ready_for_notification_bundle ?receiving_site)
        )
      )
    :effect
      (and
        (receiving_site_ready_for_notification_bundle ?receiving_site)
        (receiving_site_confirmed_exposure ?receiving_site)
      )
  )
  (:action assign_mitigation_option_to_receiving_site
    :parameters (?receiving_site - receiving_site ?dispensing_site - dispensing_site ?mitigation_option - mitigation_option)
    :precondition
      (and
        (entity_assessment_confirmed ?receiving_site)
        (receiving_site_dispensing_link ?receiving_site ?dispensing_site)
        (mitigation_option_available ?mitigation_option)
        (not
          (receiving_site_ready_for_notification_bundle ?receiving_site)
        )
      )
    :effect
      (and
        (dispensing_site_suspected_exposure ?dispensing_site)
        (receiving_site_ready_for_notification_bundle ?receiving_site)
        (receiving_site_mitigation_assigned ?receiving_site ?mitigation_option)
        (not
          (mitigation_option_available ?mitigation_option)
        )
      )
  )
  (:action finalize_receiving_site_mitigation
    :parameters (?receiving_site - receiving_site ?dispensing_site - dispensing_site ?product_lot - product_lot ?mitigation_option - mitigation_option)
    :precondition
      (and
        (entity_assessment_confirmed ?receiving_site)
        (entity_product_lot_linked ?receiving_site ?product_lot)
        (receiving_site_dispensing_link ?receiving_site ?dispensing_site)
        (dispensing_site_suspected_exposure ?dispensing_site)
        (receiving_site_mitigation_assigned ?receiving_site ?mitigation_option)
        (not
          (receiving_site_confirmed_exposure ?receiving_site)
        )
      )
    :effect
      (and
        (dispensing_site_confirmed_exposure ?dispensing_site)
        (receiving_site_confirmed_exposure ?receiving_site)
        (mitigation_option_available ?mitigation_option)
        (not
          (receiving_site_mitigation_assigned ?receiving_site ?mitigation_option)
        )
      )
  )
  (:action assemble_notification_bundle
    :parameters (?origin_site - origin_site ?receiving_site - receiving_site ?distribution_channel - distribution_channel ?dispensing_site - dispensing_site ?notification_bundle - notification_bundle)
    :precondition
      (and
        (site_ready_for_notification_bundle ?origin_site)
        (receiving_site_ready_for_notification_bundle ?receiving_site)
        (origin_site_channel_linked ?origin_site ?distribution_channel)
        (receiving_site_dispensing_link ?receiving_site ?dispensing_site)
        (distribution_channel_confirmed ?distribution_channel)
        (dispensing_site_confirmed_exposure ?dispensing_site)
        (origin_site_confirmed_exposure ?origin_site)
        (receiving_site_confirmed_exposure ?receiving_site)
        (notification_bundle_available ?notification_bundle)
      )
    :effect
      (and
        (notification_bundle_registered ?notification_bundle)
        (notification_bundle_channel_linked ?notification_bundle ?distribution_channel)
        (notification_bundle_dispensing_site_linked ?notification_bundle ?dispensing_site)
        (not
          (notification_bundle_available ?notification_bundle)
        )
      )
  )
  (:action assemble_notification_bundle_channel_followup
    :parameters (?origin_site - origin_site ?receiving_site - receiving_site ?distribution_channel - distribution_channel ?dispensing_site - dispensing_site ?notification_bundle - notification_bundle)
    :precondition
      (and
        (site_ready_for_notification_bundle ?origin_site)
        (receiving_site_ready_for_notification_bundle ?receiving_site)
        (origin_site_channel_linked ?origin_site ?distribution_channel)
        (receiving_site_dispensing_link ?receiving_site ?dispensing_site)
        (distribution_channel_suspected ?distribution_channel)
        (dispensing_site_confirmed_exposure ?dispensing_site)
        (not
          (origin_site_confirmed_exposure ?origin_site)
        )
        (receiving_site_confirmed_exposure ?receiving_site)
        (notification_bundle_available ?notification_bundle)
      )
    :effect
      (and
        (notification_bundle_registered ?notification_bundle)
        (notification_bundle_channel_linked ?notification_bundle ?distribution_channel)
        (notification_bundle_dispensing_site_linked ?notification_bundle ?dispensing_site)
        (bundle_channel_followup_required ?notification_bundle)
        (not
          (notification_bundle_available ?notification_bundle)
        )
      )
  )
  (:action assemble_notification_bundle_site_followup
    :parameters (?origin_site - origin_site ?receiving_site - receiving_site ?distribution_channel - distribution_channel ?dispensing_site - dispensing_site ?notification_bundle - notification_bundle)
    :precondition
      (and
        (site_ready_for_notification_bundle ?origin_site)
        (receiving_site_ready_for_notification_bundle ?receiving_site)
        (origin_site_channel_linked ?origin_site ?distribution_channel)
        (receiving_site_dispensing_link ?receiving_site ?dispensing_site)
        (distribution_channel_confirmed ?distribution_channel)
        (dispensing_site_suspected_exposure ?dispensing_site)
        (origin_site_confirmed_exposure ?origin_site)
        (not
          (receiving_site_confirmed_exposure ?receiving_site)
        )
        (notification_bundle_available ?notification_bundle)
      )
    :effect
      (and
        (notification_bundle_registered ?notification_bundle)
        (notification_bundle_channel_linked ?notification_bundle ?distribution_channel)
        (notification_bundle_dispensing_site_linked ?notification_bundle ?dispensing_site)
        (bundle_site_followup_required ?notification_bundle)
        (not
          (notification_bundle_available ?notification_bundle)
        )
      )
  )
  (:action assemble_notification_bundle_channel_and_site_followup
    :parameters (?origin_site - origin_site ?receiving_site - receiving_site ?distribution_channel - distribution_channel ?dispensing_site - dispensing_site ?notification_bundle - notification_bundle)
    :precondition
      (and
        (site_ready_for_notification_bundle ?origin_site)
        (receiving_site_ready_for_notification_bundle ?receiving_site)
        (origin_site_channel_linked ?origin_site ?distribution_channel)
        (receiving_site_dispensing_link ?receiving_site ?dispensing_site)
        (distribution_channel_suspected ?distribution_channel)
        (dispensing_site_suspected_exposure ?dispensing_site)
        (not
          (origin_site_confirmed_exposure ?origin_site)
        )
        (not
          (receiving_site_confirmed_exposure ?receiving_site)
        )
        (notification_bundle_available ?notification_bundle)
      )
    :effect
      (and
        (notification_bundle_registered ?notification_bundle)
        (notification_bundle_channel_linked ?notification_bundle ?distribution_channel)
        (notification_bundle_dispensing_site_linked ?notification_bundle ?dispensing_site)
        (bundle_channel_followup_required ?notification_bundle)
        (bundle_site_followup_required ?notification_bundle)
        (not
          (notification_bundle_available ?notification_bundle)
        )
      )
  )
  (:action mark_bundle_ready_for_recipient_assignment
    :parameters (?notification_bundle - notification_bundle ?origin_site - origin_site ?product_lot - product_lot)
    :precondition
      (and
        (notification_bundle_registered ?notification_bundle)
        (site_ready_for_notification_bundle ?origin_site)
        (entity_product_lot_linked ?origin_site ?product_lot)
        (not
          (bundle_ready_for_recipient_assignment ?notification_bundle)
        )
      )
    :effect (bundle_ready_for_recipient_assignment ?notification_bundle)
  )
  (:action attach_recipient_list_to_bundle
    :parameters (?response_coordinator - response_coordinator ?recipient_list - recipient_list ?notification_bundle - notification_bundle)
    :precondition
      (and
        (entity_assessment_confirmed ?response_coordinator)
        (coordinator_assigned_notification_bundle ?response_coordinator ?notification_bundle)
        (coordinator_has_recipient_list ?response_coordinator ?recipient_list)
        (recipient_list_available ?recipient_list)
        (notification_bundle_registered ?notification_bundle)
        (bundle_ready_for_recipient_assignment ?notification_bundle)
        (not
          (recipient_list_attached ?recipient_list)
        )
      )
    :effect
      (and
        (recipient_list_attached ?recipient_list)
        (recipient_list_linked_to_bundle ?recipient_list ?notification_bundle)
        (not
          (recipient_list_available ?recipient_list)
        )
      )
  )
  (:action authorize_coordinator_enrichment
    :parameters (?response_coordinator - response_coordinator ?recipient_list - recipient_list ?notification_bundle - notification_bundle ?product_lot - product_lot)
    :precondition
      (and
        (entity_assessment_confirmed ?response_coordinator)
        (coordinator_has_recipient_list ?response_coordinator ?recipient_list)
        (recipient_list_attached ?recipient_list)
        (recipient_list_linked_to_bundle ?recipient_list ?notification_bundle)
        (entity_product_lot_linked ?response_coordinator ?product_lot)
        (not
          (bundle_channel_followup_required ?notification_bundle)
        )
        (not
          (coordinator_enrichment_allowed ?response_coordinator)
        )
      )
    :effect (coordinator_enrichment_allowed ?response_coordinator)
  )
  (:action engage_regulatory_body_with_coordinator
    :parameters (?response_coordinator - response_coordinator ?regulatory_body - regulatory_body)
    :precondition
      (and
        (entity_assessment_confirmed ?response_coordinator)
        (regulatory_body_available ?regulatory_body)
        (not
          (regulatory_engagement_initiated ?response_coordinator)
        )
      )
    :effect
      (and
        (regulatory_engagement_initiated ?response_coordinator)
        (coordinator_engaged_regulatory_body ?response_coordinator ?regulatory_body)
        (not
          (regulatory_body_available ?regulatory_body)
        )
      )
  )
  (:action complete_regulatory_review_and_enable_enrichment
    :parameters (?response_coordinator - response_coordinator ?recipient_list - recipient_list ?notification_bundle - notification_bundle ?product_lot - product_lot ?regulatory_body - regulatory_body)
    :precondition
      (and
        (entity_assessment_confirmed ?response_coordinator)
        (coordinator_has_recipient_list ?response_coordinator ?recipient_list)
        (recipient_list_attached ?recipient_list)
        (recipient_list_linked_to_bundle ?recipient_list ?notification_bundle)
        (entity_product_lot_linked ?response_coordinator ?product_lot)
        (bundle_channel_followup_required ?notification_bundle)
        (regulatory_engagement_initiated ?response_coordinator)
        (coordinator_engaged_regulatory_body ?response_coordinator ?regulatory_body)
        (not
          (coordinator_enrichment_allowed ?response_coordinator)
        )
      )
    :effect
      (and
        (coordinator_enrichment_allowed ?response_coordinator)
        (regulatory_review_completed ?response_coordinator)
