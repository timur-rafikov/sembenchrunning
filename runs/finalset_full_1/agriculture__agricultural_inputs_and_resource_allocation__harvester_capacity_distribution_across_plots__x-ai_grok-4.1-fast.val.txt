  (:action register_plot
    :parameters (?plot - plot)
    :precondition
      (and
        (not
          (entity_registered ?plot)
        )
        (not
          (entity_serviced ?plot)
        )
      )
    :effect (entity_registered ?plot)
  )
  (:action reserve_harvester_token_for_plot
    :parameters (?plot - plot ?harvester_resource_token - harvester_resource_token)
    :precondition
      (and
        (entity_registered ?plot)
        (not
          (entity_token_reserved ?plot)
        )
        (harvester_token_available ?harvester_resource_token)
      )
    :effect
      (and
        (entity_token_reserved ?plot)
        (has_reserved_token ?plot ?harvester_resource_token)
        (not
          (harvester_token_available ?harvester_resource_token)
        )
      )
  )
  (:action assign_time_slot_to_plot

  )
