--- domain.pddl (as passed to VAL) ---
(define (domain irrigation_water_allocation)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity - object field_container - entity resource_container - entity delivery_container - entity request_container - entity parcel_request - request_container water_source_unit - field_container irrigation_time_slot - field_container distribution_valve - field_container regulatory_permit - field_container irrigation_method - field_container contingency_reserve - field_container equipment_unit - field_container quality_approval - field_container water_quota_token - resource_container field_sensor - resource_container emergency_priority_certificate - resource_container delivery_segment - delivery_container delivery_node - delivery_container allocation_record - delivery_container priority_group_a - parcel_request priority_group_b - parcel_request high_priority_field_unit - priority_group_a secondary_priority_field_unit - priority_group_a irrigation_planner - priority_group_b)

  (:predicates
    (entity_allocation_registered ?parcel_request - parcel_request)
    (entity_allocation_confirmed ?parcel_request - parcel_request)
    (entity_allocation_source_reserved ?parcel_request - parcel_request)
    (entity_allocation_fulfilled ?parcel_request - parcel_request)
    (delivery_confirmed ?parcel_request - parcel_request)
    (entity_allocation_finalized ?parcel_request - parcel_request)
    (water_source_available ?water_source_unit - water_source_unit)
    (entity_allocation_assigned_source ?parcel_request - parcel_request ?water_source_unit - water_source_unit)
    (timeslot_available ?irrigation_time_slot - irrigation_time_slot)
    (entity_allocation_assigned_timeslot ?parcel_request - parcel_request ?irrigation_time_slot - irrigation_time_slot)
    (valve_available ?distribution_valve - distribution_valve)
    (entity_allocation_assigned_valve ?parcel_request - parcel_request ?distribution_valve - distribution_valve)
    (quota_token_available ?water_quota_token - water_quota_token)
    (high_priority_reserved_quota ?high_priority_field_unit - high_priority_field_unit ?water_quota_token - water_quota_token)
    (secondary_priority_reserved_quota ?secondary_priority_field_unit - secondary_priority_field_unit ?water_quota_token - water_quota_token)
    (high_priority_assigned_segment ?high_priority_field_unit - high_priority_field_unit ?delivery_segment - delivery_segment)
    (segment_reserved_by_high_priority ?delivery_segment - delivery_segment)
    (segment_reserved_with_quota_token ?delivery_segment - delivery_segment)
    (high_priority_unit_staged ?high_priority_field_unit - high_priority_field_unit)
    (secondary_assigned_node ?secondary_priority_field_unit - secondary_priority_field_unit ?delivery_node - delivery_node)
    (node_reserved_by_secondary ?delivery_node - delivery_node)
    (node_reserved_with_quota ?delivery_node - delivery_node)
    (secondary_unit_staged ?secondary_priority_field_unit - secondary_priority_field_unit)
    (allocation_record_available ?allocation_record - allocation_record)
    (allocation_record_assembled ?allocation_record - allocation_record)
    (record_includes_segment ?allocation_record - allocation_record ?delivery_segment - delivery_segment)
    (record_includes_node ?allocation_record - allocation_record ?delivery_node - delivery_node)
    (record_includes_high_priority ?allocation_record - allocation_record)
    (record_includes_secondary_priority ?allocation_record - allocation_record)
    (record_marked_for_validation ?allocation_record - allocation_record)
    (planner_assigned_high_unit ?irrigation_planner - irrigation_planner ?high_priority_field_unit - high_priority_field_unit)
    (planner_assigned_secondary_unit ?irrigation_planner - irrigation_planner ?secondary_priority_field_unit - secondary_priority_field_unit)
    (planner_linked_record ?irrigation_planner - irrigation_planner ?allocation_record - allocation_record)
    (sensor_available ?field_sensor - field_sensor)
    (planner_linked_sensor ?irrigation_planner - irrigation_planner ?field_sensor - field_sensor)
    (sensor_claimed ?field_sensor - field_sensor)
    (sensor_linked_record ?field_sensor - field_sensor ?allocation_record - allocation_record)
    (planner_initial_authorization ?irrigation_planner - irrigation_planner)
    (planner_equipment_assigned ?irrigation_planner - irrigation_planner)
    (planner_quality_validated ?irrigation_planner - irrigation_planner)
    (planner_has_permit ?irrigation_planner - irrigation_planner)
    (planner_permit_link_confirmed ?irrigation_planner - irrigation_planner)
    (planner_method_authorized ?irrigation_planner - irrigation_planner)
    (planner_ready_for_dispatch ?irrigation_planner - irrigation_planner)
    (emergency_certificate_available ?emergency_priority_certificate - emergency_priority_certificate)
    (planner_linked_emergency_certificate ?irrigation_planner - irrigation_planner ?emergency_priority_certificate - emergency_priority_certificate)
    (planner_emergency_certified ?irrigation_planner - irrigation_planner)
    (planner_emergency_action_authorized ?irrigation_planner - irrigation_planner)
    (planner_emergency_action_confirmed ?irrigation_planner - irrigation_planner)
    (regulatory_permit_available ?regulatory_permit - regulatory_permit)
    (planner_bound_permit ?irrigation_planner - irrigation_planner ?regulatory_permit - regulatory_permit)
    (irrigation_method_available ?irrigation_method - irrigation_method)
    (planner_assigned_method ?irrigation_planner - irrigation_planner ?irrigation_method - irrigation_method)
    (equipment_available ?equipment_unit - equipment_unit)
    (planner_assigned_equipment ?irrigation_planner - irrigation_planner ?equipment_unit - equipment_unit)
    (quality_approval_available ?quality_approval - quality_approval)
    (planner_assigned_quality_approval ?irrigation_planner - irrigation_planner ?quality_approval - quality_approval)
    (contingency_reserve_available ?contingency_reserve - contingency_reserve)
    (entity_allocation_assigned_contingency ?parcel_request - parcel_request ?contingency_reserve - contingency_reserve)
    (high_priority_unit_committed ?high_priority_field_unit - high_priority_field_unit)
    (secondary_unit_committed ?secondary_priority_field_unit - secondary_priority_field_unit)
    (planner_dispatched ?irrigation_planner - irrigation_planner)
  )
  (:action register_parcel_request
    :parameters (?parcel_request - parcel_request)
    :precondition
      (and
        (not
          (entity_allocation_registered ?parcel_request)
        )
        (not
          (entity_allocation_fulfilled ?parcel_request)
        )
      )
    :effect (entity_allocation_registered ?parcel_request)
  )
  (:action assign_request_to_source
    :parameters (?parcel_request - parcel_request ?water_source_unit - water_source_unit)
    :precondition
      (and
        (entity_allocation_registered ?parcel_request)
        (not
          (entity_allocation_source_reserved ?parcel_request)
        )
        (water_source_available ?water_source_unit)
      )
    :effect
      (and
        (entity_allocation_source_reserved ?parcel_request)
        (entity_allocation_assigned_source ?parcel_request ?water_source_unit)
        (not
          (water_source_available ?water_source_unit)
        )
      )
  )
  (:action assign_request_to_timeslot
    :parameters (?parcel_request - parcel_request ?irrigation_time_slot - irrigation_time_slot)
    :precondition
      (and
        (entity_allocation_registered ?parcel_request)
        (entity_allocation_source_reserved ?parcel_request)
        (timeslot_available ?irrigation_time_slot)
      )
    :effect
      (and
        (entity_allocation_assigned_timeslot ?parcel_request ?irrigation_time_slot)
        (not
          (timeslot_available ?irrigation_time_slot)
        )
      )
  )
  (:action confirm_request_allocation
    :parameters (?parcel_request - parcel_request ?irrigation_time_slot - irrigation_time_slot)
    :precondition
      (and
        (entity_allocation_registered ?parcel_request)
        (entity_allocation_source_reserved ?parcel_request)
        (entity_allocation_assigned_timeslot ?parcel_request ?irrigation_time_slot)
        (not
          (entity_allocation_confirmed ?parcel_request)
        )
      )
    :effect (entity_allocation_confirmed ?parcel_request)
  )
  (:action release_timeslot_from_request
    :parameters (?parcel_request - parcel_request ?irrigation_time_slot - irrigation_time_slot)
    :precondition
      (and
        (entity_allocation_assigned_timeslot ?parcel_request ?irrigation_time_slot)
      )
    :effect
      (and
        (timeslot_available ?irrigation_time_slot)
        (not
          (entity_allocation_assigned_timeslot ?parcel_request ?irrigation_time_slot)
        )
      )
  )
  (:action assign_valve_to_request
    :parameters (?parcel_request - parcel_request ?distribution_valve - distribution_valve)
    :precondition
      (and
        (entity_allocation_confirmed ?parcel_request)
        (valve_available ?distribution_valve)
      )
    :effect
      (and
        (entity_allocation_assigned_valve ?parcel_request ?distribution_valve)
        (not
          (valve_available ?distribution_valve)
        )
      )
  )
  (:action release_valve_from_request
    :parameters (?parcel_request - parcel_request ?distribution_valve - distribution_valve)
    :precondition
      (and
        (entity_allocation_assigned_valve ?parcel_request ?distribution_valve)
      )
    :effect
      (and
        (valve_available ?distribution_valve)
        (not
          (entity_allocation_assigned_valve ?parcel_request ?distribution_valve)
        )
      )
  )
  (:action assign_equipment_to_planner
    :parameters (?irrigation_planner - irrigation_planner ?equipment_unit - equipment_unit)
    :precondition
      (and
        (entity_allocation_confirmed ?irrigation_planner)
        (equipment_available ?equipment_unit)
      )
    :effect
      (and
        (planner_assigned_equipment ?irrigation_planner ?equipment_unit)
        (not
          (equipment_available ?equipment_unit)
        )
      )
  )
  (:action release_equipment_from_planner
    :parameters (?irrigation_planner - irrigation_planner ?equipment_unit - equipment_unit)
    :precondition
      (and
        (planner_assigned_equipment ?irrigation_planner ?equipment_unit)
      )
    :effect
      (and
        (equipment_available ?equipment_unit)
        (not
          (planner_assigned_equipment ?irrigation_planner ?equipment_unit)
        )
      )
  )
  (:action assign_quality_approval_to_planner
    :parameters (?irrigation_planner - irrigation_planner ?quality_approval - quality_approval)
    :precondition
      (and
        (entity_allocation_confirmed ?irrigation_planner)
        (quality_approval_available ?quality_approval)
      )
    :effect
      (and
        (planner_assigned_quality_approval ?irrigation_planner ?quality_approval)
        (not
          (quality_approval_available ?quality_approval)
        )
      )
  )
  (:action release_quality_approval_from_planner
    :parameters (?irrigation_planner - irrigation_planner ?quality_approval - quality_approval)
    :precondition
      (and
        (planner_assigned_quality_approval ?irrigation_planner ?quality_approval)
      )
    :effect
      (and
        (quality_approval_available ?quality_approval)
        (not
          (planner_assigned_quality_approval ?irrigation_planner ?quality_approval)
        )
      )
  )
  (:action claim_delivery_segment_high_priority
    :parameters (?high_priority_field_unit - high_priority_field_unit ?delivery_segment - delivery_segment ?irrigation_time_slot - irrigation_time_slot)
    :precondition
      (and
        (entity_allocation_confirmed ?high_priority_field_unit)
        (entity_allocation_assigned_timeslot ?high_priority_field_unit ?irrigation_time_slot)
        (high_priority_assigned_segment ?high_priority_field_unit ?delivery_segment)
        (not
          (segment_reserved_by_high_priority ?delivery_segment)
        )
        (not
          (segment_reserved_with_quota_token ?delivery_segment)
        )
      )
    :effect (segment_reserved_by_high_priority ?delivery_segment)
  )
  (:action commit_high_priority_claim
    :parameters (?high_priority_field_unit - high_priority_field_unit ?delivery_segment - delivery_segment ?distribution_valve - distribution_valve)
    :precondition
      (and
        (entity_allocation_confirmed ?high_priority_field_unit)
        (entity_allocation_assigned_valve ?high_priority_field_unit ?distribution_valve)
        (high_priority_assigned_segment ?high_priority_field_unit ?delivery_segment)
        (segment_reserved_by_high_priority ?delivery_segment)
        (not
          (high_priority_unit_committed ?high_priority_field_unit)
        )
      )
    :effect
      (and
        (high_priority_unit_committed ?high_priority_field_unit)
        (high_priority_unit_staged ?high_priority_field_unit)
      )
  )
  (:action reserve_quota_high_priority
    :parameters (?high_priority_field_unit - high_priority_field_unit ?delivery_segment - delivery_segment ?water_quota_token - water_quota_token)
    :precondition
      (and
        (entity_allocation_confirmed ?high_priority_field_unit)
        (high_priority_assigned_segment ?high_priority_field_unit ?delivery_segment)
        (quota_token_available ?water_quota_token)
        (not
          (high_priority_unit_committed ?high_priority_field_unit)
        )
      )
    :effect
      (and
        (segment_reserved_with_quota_token ?delivery_segment)
        (high_priority_unit_committed ?high_priority_field_unit)
        (high_priority_reserved_quota ?high_priority_field_unit ?water_quota_token)
        (not
          (quota_token_available ?water_quota_token)
        )
      )
  )
  (:action finalize_high_priority_allocation
    :parameters (?high_priority_field_unit - high_priority_field_unit ?delivery_segment - delivery_segment ?irrigation_time_slot - irrigation_time_slot ?water_quota_token - water_quota_token)
    :precondition
      (and
        (entity_allocation_confirmed ?high_priority_field_unit)
        (entity_allocation_assigned_timeslot ?high_priority_field_unit ?irrigation_time_slot)
        (high_priority_assigned_segment ?high_priority_field_unit ?delivery_segment)
        (segment_reserved_with_quota_token ?delivery_segment)
        (high_priority_reserved_quota ?high_priority_field_unit ?water_quota_token)
        (not
          (high_priority_unit_staged ?high_priority_field_unit)
        )
      )
    :effect
      (and
        (segment_reserved_by_high_priority ?delivery_segment)
        (high_priority_unit_staged ?high_priority_field_unit)
        (quota_token_available ?water_quota_token)
        (not
          (high_priority_reserved_quota ?high_priority_field_unit ?water_quota_token)
        )
      )
  )
  (:action claim_delivery_node_secondary_priority
    :parameters (?secondary_priority_field_unit - secondary_priority_field_unit ?delivery_node - delivery_node ?irrigation_time_slot - irrigation_time_slot)
    :precondition
      (and
        (entity_allocation_confirmed ?secondary_priority_field_unit)
        (entity_allocation_assigned_timeslot ?secondary_priority_field_unit ?irrigation_time_slot)
        (secondary_assigned_node ?secondary_priority_field_unit ?delivery_node)
        (not
          (node_reserved_by_secondary ?delivery_node)
        )
        (not
          (node_reserved_with_quota ?delivery_node)
        )
      )
    :effect (node_reserved_by_secondary ?delivery_node)
  )
  (:action commit_secondary_claim
    :parameters (?secondary_priority_field_unit - secondary_priority_field_unit ?delivery_node - delivery_node ?distribution_valve - distribution_valve)
    :precondition
      (and
        (entity_allocation_confirmed ?secondary_priority_field_unit)
        (entity_allocation_assigned_valve ?secondary_priority_field_unit ?distribution_valve)
        (secondary_assigned_node ?secondary_priority_field_unit ?delivery_node)
        (node_reserved_by_secondary ?delivery_node)
        (not
          (secondary_unit_committed ?secondary_priority_field_unit)
        )
      )
    :effect
      (and
        (secondary_unit_committed ?secondary_priority_field_unit)
        (secondary_unit_staged ?secondary_priority_field_unit)
      )
  )
  (:action reserve_quota_secondary_priority
    :parameters (?secondary_priority_field_unit - secondary_priority_field_unit ?delivery_node - delivery_node ?water_quota_token - water_quota_token)
    :precondition
      (and
        (entity_allocation_confirmed ?secondary_priority_field_unit)
        (secondary_assigned_node ?secondary_priority_field_unit ?delivery_node)
        (quota_token_available ?water_quota_token)
        (not
          (secondary_unit_committed ?secondary_priority_field_unit)
        )
      )
    :effect
      (and
        (node_reserved_with_quota ?delivery_node)
        (secondary_unit_committed ?secondary_priority_field_unit)
        (secondary_priority_reserved_quota ?secondary_priority_field_unit ?water_quota_token)
        (not
          (quota_token_available ?water_quota_token)
        )
      )
  )
  (:action finalize_secondary_allocation
    :parameters (?secondary_priority_field_unit - secondary_priority_field_unit ?delivery_node - delivery_node ?irrigation_time_slot - irrigation_time_slot ?water_quota_token - water_quota_token)
    :precondition
      (and
        (entity_allocation_confirmed ?secondary_priority_field_unit)
        (entity_allocation_assigned_timeslot ?secondary_priority_field_unit ?irrigation_time_slot)
        (secondary_assigned_node ?secondary_priority_field_unit ?delivery_node)
        (node_reserved_with_quota ?delivery_node)
        (secondary_priority_reserved_quota ?secondary_priority_field_unit ?water_quota_token)
        (not
          (secondary_unit_staged ?secondary_priority_field_unit)
        )
      )
    :effect
      (and
        (node_reserved_by_secondary ?delivery_node)
        (secondary_unit_staged ?secondary_priority_field_unit)
        (quota_token_available ?water_quota_token)
        (not
          (secondary_priority_reserved_quota ?secondary_priority_field_unit ?water_quota_token)
        )
      )
  )
  (:action assemble_allocation_record_full
    :parameters (?high_priority_field_unit - high_priority_field_unit ?secondary_priority_field_unit - secondary_priority_field_unit ?delivery_segment - delivery_segment ?delivery_node - delivery_node ?allocation_record - allocation_record)
    :precondition
      (and
        (high_priority_unit_committed ?high_priority_field_unit)
        (secondary_unit_committed ?secondary_priority_field_unit)
        (high_priority_assigned_segment ?high_priority_field_unit ?delivery_segment)
        (secondary_assigned_node ?secondary_priority_field_unit ?delivery_node)
        (segment_reserved_by_high_priority ?delivery_segment)
        (node_reserved_by_secondary ?delivery_node)
        (high_priority_unit_staged ?high_priority_field_unit)
        (secondary_unit_staged ?secondary_priority_field_unit)
        (allocation_record_available ?allocation_record)
      )
    :effect
      (and
        (allocation_record_assembled ?allocation_record)
        (record_includes_segment ?allocation_record ?delivery_segment)
        (record_includes_node ?allocation_record ?delivery_node)
        (not
          (allocation_record_available ?allocation_record)
        )
      )
  )
  (:action assemble_allocation_record_high_priority_quota
    :parameters (?high_priority_field_unit - high_priority_field_unit ?secondary_priority_field_unit - secondary_priority_field_unit ?delivery_segment - delivery_segment ?delivery_node - delivery_node ?allocation_record - allocation_record)
    :precondition
      (and
        (high_priority_unit_committed ?high_priority_field_unit)
        (secondary_unit_committed ?secondary_priority_field_unit)
        (high_priority_assigned_segment ?high_priority_field_unit ?delivery_segment)
        (secondary_assigned_node ?secondary_priority_field_unit ?delivery_node)
        (segment_reserved_with_quota_token ?delivery_segment)
        (node_reserved_by_secondary ?delivery_node)
        (not
          (high_priority_unit_staged ?high_priority_field_unit)
        )
        (secondary_unit_staged ?secondary_priority_field_unit)
        (allocation_record_available ?allocation_record)
      )
    :effect
      (and
        (allocation_record_assembled ?allocation_record)
        (record_includes_segment ?allocation_record ?delivery_segment)
        (record_includes_node ?allocation_record ?delivery_node)
        (record_includes_high_priority ?allocation_record)
        (not
          (allocation_record_available ?allocation_record)
        )
      )
  )
  (:action assemble_allocation_record_secondary_priority_quota
    :parameters (?high_priority_field_unit - high_priority_field_unit ?secondary_priority_field_unit - secondary_priority_field_unit ?delivery_segment - delivery_segment ?delivery_node - delivery_node ?allocation_record - allocation_record)
    :precondition
      (and
        (high_priority_unit_committed ?high_priority_field_unit)
        (secondary_unit_committed ?secondary_priority_field_unit)
        (high_priority_assigned_segment ?high_priority_field_unit ?delivery_segment)
        (secondary_assigned_node ?secondary_priority_field_unit ?delivery_node)
        (segment_reserved_by_high_priority ?delivery_segment)
        (node_reserved_with_quota ?delivery_node)
        (high_priority_unit_staged ?high_priority_field_unit)
        (not
          (secondary_unit_staged ?secondary_priority_field_unit)
        )
        (allocation_record_available ?allocation_record)
      )
    :effect
      (and
        (allocation_record_assembled ?allocation_record)
        (record_includes_segment ?allocation_record ?delivery_segment)
        (record_includes_node ?allocation_record ?delivery_node)
        (record_includes_secondary_priority ?allocation_record)
        (not
          (allocation_record_available ?allocation_record)
        )
      )
  )
  (:action assemble_allocation_record_mixed_priorities
    :parameters (?high_priority_field_unit - high_priority_field_unit ?secondary_priority_field_unit - secondary_priority_field_unit ?delivery_segment - delivery_segment ?delivery_node - delivery_node ?allocation_record - allocation_record)
    :precondition
      (and
        (high_priority_unit_committed ?high_priority_field_unit)
        (secondary_unit_committed ?secondary_priority_field_unit)
        (high_priority_assigned_segment ?high_priority_field_unit ?delivery_segment)
        (secondary_assigned_node ?secondary_priority_field_unit ?delivery_node)
        (segment_reserved_with_quota_token ?delivery_segment)
        (node_reserved_with_quota ?delivery_node)
        (not
          (high_priority_unit_staged ?high_priority_field_unit)
        )
        (not
          (secondary_unit_staged ?secondary_priority_field_unit)
        )
        (allocation_record_available ?allocation_record)
      )
    :effect
      (and
        (allocation_record_assembled ?allocation_record)
        (record_includes_segment ?allocation_record ?delivery_segment)
        (record_includes_node ?allocation_record ?delivery_node)
        (record_includes_high_priority ?allocation_record)
        (record_includes_secondary_priority ?allocation_record)
        (not
          (allocation_record_available ?allocation_record)
        )
      )
  )
  (:action mark_record_for_validation
    :parameters (?allocation_record - allocation_record ?high_priority_field_unit - high_priority_field_unit ?irrigation_time_slot - irrigation_time_slot)
    :precondition
      (and
        (allocation_record_assembled ?allocation_record)
        (high_priority_unit_committed ?high_priority_field_unit)
        (entity_allocation_assigned_timeslot ?high_priority_field_unit ?irrigation_time_slot)
        (not
          (record_marked_for_validation ?allocation_record)
        )
      )
    :effect (record_marked_for_validation ?allocation_record)
  )
  (:action assign_sensor_to_planner_and_record
    :parameters (?irrigation_planner - irrigation_planner ?field_sensor - field_sensor ?allocation_record - allocation_record)
    :precondition
      (and
        (entity_allocation_confirmed ?irrigation_planner)
        (planner_linked_record ?irrigation_planner ?allocation_record)
        (planner_linked_sensor ?irrigation_planner ?field_sensor)
        (sensor_available ?field_sensor)
        (allocation_record_assembled ?allocation_record)
        (record_marked_for_validation ?allocation_record)
        (not
          (sensor_claimed ?field_sensor)
        )
      )
    :effect
      (and
        (sensor_claimed ?field_sensor)
        (sensor_linked_record ?field_sensor ?allocation_record)
        (not
          (sensor_available ?field_sensor)
        )
      )
  )
  (:action initialize_planner_authorization
    :parameters (?irrigation_planner - irrigation_planner ?field_sensor - field_sensor ?allocation_record - allocation_record ?irrigation_time_slot - irrigation_time_slot)
    :precondition
      (and
        (entity_allocation_confirmed ?irrigation_planner)
        (planner_linked_sensor ?irrigation_planner ?field_sensor)
        (sensor_claimed ?field_sensor)
        (sensor_linked_record ?field_sensor ?allocation_record)
        (entity_allocation_assigned_timeslot ?irrigation_planner ?irrigation_time_slot)
        (not
          (record_includes_high_priority ?allocation_record)
        )
        (not
          (planner_initial_authorization ?irrigation_planner)
