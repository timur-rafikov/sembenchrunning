--- domain.pddl (as passed to VAL) ---
(define (domain multi_farm_resource_sharing_coordination_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_category - object material_category - object service_category - object domain_entity - object farm_unit - domain_entity resource_provider - resource_category input_batch - resource_category machinery_unit - resource_category land_parcel - resource_category support_grant - resource_category contingency_reserve - resource_category operational_task - resource_category special_permit - resource_category consumable_resource - material_category certification_record - material_category permission_token - material_category lender_allocation_slot - service_category borrower_allocation_slot - service_category shared_package - service_category lender_group - farm_unit borrower_group - farm_unit lender_farm - lender_group borrower_farm - lender_group coordination_entity - borrower_group)
  (:predicates
    (farm_registered ?farm_unit - farm_unit)
    (entity_validated ?farm_unit - farm_unit)
    (provisionally_matched ?farm_unit - farm_unit)
    (transfer_completed ?farm_unit - farm_unit)
    (entity_package_staged ?farm_unit - farm_unit)
    (entity_package_claimed ?farm_unit - farm_unit)
    (provider_available ?resource_provider - resource_provider)
    (assigned_provider_to_entity ?farm_unit - farm_unit ?resource_provider - resource_provider)
    (input_batch_available ?input_batch - input_batch)
    (reserved_input_for_entity ?farm_unit - farm_unit ?input_batch - input_batch)
    (machinery_available ?machinery_unit - machinery_unit)
    (machinery_assigned_to_entity ?farm_unit - farm_unit ?machinery_unit - machinery_unit)
    (consumable_available ?consumable_resource - consumable_resource)
    (lender_consumable_allocated ?lender_farm - lender_farm ?consumable_resource - consumable_resource)
    (borrower_consumable_allocated ?borrower_farm - borrower_farm ?consumable_resource - consumable_resource)
    (lender_slot_offered ?lender_farm - lender_farm ?lender_allocation_slot - lender_allocation_slot)
    (lender_slot_ready ?lender_allocation_slot - lender_allocation_slot)
    (lender_slot_supplied_with_consumable ?lender_allocation_slot - lender_allocation_slot)
    (lender_ready ?lender_farm - lender_farm)
    (borrower_slot_offered ?borrower_farm - borrower_farm ?borrower_allocation_slot - borrower_allocation_slot)
    (borrower_slot_ready ?borrower_allocation_slot - borrower_allocation_slot)
    (borrower_slot_supplied_with_consumable ?borrower_allocation_slot - borrower_allocation_slot)
    (borrower_ready ?borrower_farm - borrower_farm)
    (shared_package_available ?shared_package - shared_package)
    (shared_package_staged ?shared_package - shared_package)
    (package_linked_to_lender_slot ?shared_package - shared_package ?lender_allocation_slot - lender_allocation_slot)
    (package_linked_to_borrower_slot ?shared_package - shared_package ?borrower_allocation_slot - borrower_allocation_slot)
    (package_requires_parcel_permission ?shared_package - shared_package)
    (package_requires_special_permit ?shared_package - shared_package)
    (shared_package_verified ?shared_package - shared_package)
    (coordination_entity_manages_lender ?coordination_entity - coordination_entity ?lender_farm - lender_farm)
    (coordination_entity_manages_borrower ?coordination_entity - coordination_entity ?borrower_farm - borrower_farm)
    (coordination_entity_links_package ?coordination_entity - coordination_entity ?shared_package - shared_package)
    (certification_available ?certification_record - certification_record)
    (coordination_entity_holds_certification ?coordination_entity - coordination_entity ?certification_record - certification_record)
    (certification_validated ?certification_record - certification_record)
    (certification_attached_to_package ?certification_record - certification_record ?shared_package - shared_package)
    (entity_certification_ready ?coordination_entity - coordination_entity)
    (entity_certified_stage1 ?coordination_entity - coordination_entity)
    (entity_compliance_finalized ?coordination_entity - coordination_entity)
    (entity_has_parcel_permission ?coordination_entity - coordination_entity)
    (entity_parcel_permission_verified ?coordination_entity - coordination_entity)
    (entity_requires_additional_permission ?coordination_entity - coordination_entity)
    (entity_ready_for_dispatch ?coordination_entity - coordination_entity)
    (permission_token_available ?permission_token - permission_token)
    (coordination_entity_holds_permission_token ?coordination_entity - coordination_entity ?permission_token - permission_token)
    (entity_permission_token_committed ?coordination_entity - coordination_entity)
    (entity_permission_in_process ?coordination_entity - coordination_entity)
    (entity_permission_finalized ?coordination_entity - coordination_entity)
    (land_parcel_available ?land_parcel - land_parcel)
    (coordination_entity_assigned_parcel ?coordination_entity - coordination_entity ?land_parcel - land_parcel)
    (support_grant_available ?support_grant - support_grant)
    (coordination_entity_assigned_grant ?coordination_entity - coordination_entity ?support_grant - support_grant)
    (operational_task_available ?operational_task - operational_task)
    (coordination_entity_assigned_task ?coordination_entity - coordination_entity ?operational_task - operational_task)
    (special_permit_available ?special_permit - special_permit)
    (coordination_entity_assigned_special_permit ?coordination_entity - coordination_entity ?special_permit - special_permit)
    (contingency_reserve_available ?contingency_reserve - contingency_reserve)
    (entity_assigned_contingency_reserve ?farm_unit - farm_unit ?contingency_reserve - contingency_reserve)
    (lender_slot_owner_verified ?lender_farm - lender_farm)
    (borrower_slot_owner_verified ?borrower_farm - borrower_farm)
    (entity_allocation_locked ?coordination_entity - coordination_entity)
  )
  (:action register_farm
    :parameters (?farm_unit - farm_unit)
    :precondition
      (and
        (not
          (farm_registered ?farm_unit)
        )
        (not
          (transfer_completed ?farm_unit)
        )
      )
    :effect (farm_registered ?farm_unit)
  )
  (:action match_farm_with_provider
    :parameters (?farm_unit - farm_unit ?resource_provider - resource_provider)
    :precondition
      (and
        (farm_registered ?farm_unit)
        (not
          (provisionally_matched ?farm_unit)
        )
        (provider_available ?resource_provider)
      )
    :effect
      (and
        (provisionally_matched ?farm_unit)
        (assigned_provider_to_entity ?farm_unit ?resource_provider)
        (not
          (provider_available ?resource_provider)
        )
      )
  )
  (:action reserve_input_batch_for_farm
    :parameters (?farm_unit - farm_unit ?input_batch - input_batch)
    :precondition
      (and
        (farm_registered ?farm_unit)
        (provisionally_matched ?farm_unit)
        (input_batch_available ?input_batch)
      )
    :effect
      (and
        (reserved_input_for_entity ?farm_unit ?input_batch)
        (not
          (input_batch_available ?input_batch)
        )
      )
  )
  (:action confirm_input_reservation
    :parameters (?farm_unit - farm_unit ?input_batch - input_batch)
    :precondition
      (and
        (farm_registered ?farm_unit)
        (provisionally_matched ?farm_unit)
        (reserved_input_for_entity ?farm_unit ?input_batch)
        (not
          (entity_validated ?farm_unit)
        )
      )
    :effect (entity_validated ?farm_unit)
  )
  (:action release_reserved_input_batch
    :parameters (?farm_unit - farm_unit ?input_batch - input_batch)
    :precondition
      (and
        (reserved_input_for_entity ?farm_unit ?input_batch)
      )
    :effect
      (and
        (input_batch_available ?input_batch)
        (not
          (reserved_input_for_entity ?farm_unit ?input_batch)
        )
      )
  )
  (:action assign_machinery_to_farm
    :parameters (?farm_unit - farm_unit ?machinery_unit - machinery_unit)
    :precondition
      (and
        (entity_validated ?farm_unit)
        (machinery_available ?machinery_unit)
      )
    :effect
      (and
        (machinery_assigned_to_entity ?farm_unit ?machinery_unit)
        (not
          (machinery_available ?machinery_unit)
        )
      )
  )
  (:action release_machinery_from_farm
    :parameters (?farm_unit - farm_unit ?machinery_unit - machinery_unit)
    :precondition
      (and
        (machinery_assigned_to_entity ?farm_unit ?machinery_unit)
      )
    :effect
      (and
        (machinery_available ?machinery_unit)
        (not
          (machinery_assigned_to_entity ?farm_unit ?machinery_unit)
        )
      )
  )
  (:action assign_operational_task_to_entity
    :parameters (?coordination_entity - coordination_entity ?operational_task - operational_task)
    :precondition
      (and
        (entity_validated ?coordination_entity)
        (operational_task_available ?operational_task)
      )
    :effect
      (and
        (coordination_entity_assigned_task ?coordination_entity ?operational_task)
        (not
          (operational_task_available ?operational_task)
        )
      )
  )
  (:action release_operational_task_from_entity
    :parameters (?coordination_entity - coordination_entity ?operational_task - operational_task)
    :precondition
      (and
        (coordination_entity_assigned_task ?coordination_entity ?operational_task)
      )
    :effect
      (and
        (operational_task_available ?operational_task)
        (not
          (coordination_entity_assigned_task ?coordination_entity ?operational_task)
        )
      )
  )
  (:action assign_special_permit_to_entity
    :parameters (?coordination_entity - coordination_entity ?special_permit - special_permit)
    :precondition
      (and
        (entity_validated ?coordination_entity)
        (special_permit_available ?special_permit)
      )
    :effect
      (and
        (coordination_entity_assigned_special_permit ?coordination_entity ?special_permit)
        (not
          (special_permit_available ?special_permit)
        )
      )
  )
  (:action release_special_permit_from_entity
    :parameters (?coordination_entity - coordination_entity ?special_permit - special_permit)
    :precondition
      (and
        (coordination_entity_assigned_special_permit ?coordination_entity ?special_permit)
      )
    :effect
      (and
        (special_permit_available ?special_permit)
        (not
          (coordination_entity_assigned_special_permit ?coordination_entity ?special_permit)
        )
      )
  )
  (:action prepare_lender_allocation_slot
    :parameters (?lender_farm - lender_farm ?lender_allocation_slot - lender_allocation_slot ?input_batch - input_batch)
    :precondition
      (and
        (entity_validated ?lender_farm)
        (reserved_input_for_entity ?lender_farm ?input_batch)
        (lender_slot_offered ?lender_farm ?lender_allocation_slot)
        (not
          (lender_slot_ready ?lender_allocation_slot)
        )
        (not
          (lender_slot_supplied_with_consumable ?lender_allocation_slot)
        )
      )
    :effect (lender_slot_ready ?lender_allocation_slot)
  )
  (:action confirm_lender_slot
    :parameters (?lender_farm - lender_farm ?lender_allocation_slot - lender_allocation_slot ?machinery_unit - machinery_unit)
    :precondition
      (and
        (entity_validated ?lender_farm)
        (machinery_assigned_to_entity ?lender_farm ?machinery_unit)
        (lender_slot_offered ?lender_farm ?lender_allocation_slot)
        (lender_slot_ready ?lender_allocation_slot)
        (not
          (lender_slot_owner_verified ?lender_farm)
        )
      )
    :effect
      (and
        (lender_slot_owner_verified ?lender_farm)
        (lender_ready ?lender_farm)
      )
  )
  (:action assign_consumable_to_lender_slot
    :parameters (?lender_farm - lender_farm ?lender_allocation_slot - lender_allocation_slot ?consumable_resource - consumable_resource)
    :precondition
      (and
        (entity_validated ?lender_farm)
        (lender_slot_offered ?lender_farm ?lender_allocation_slot)
        (consumable_available ?consumable_resource)
        (not
          (lender_slot_owner_verified ?lender_farm)
        )
      )
    :effect
      (and
        (lender_slot_supplied_with_consumable ?lender_allocation_slot)
        (lender_slot_owner_verified ?lender_farm)
        (lender_consumable_allocated ?lender_farm ?consumable_resource)
        (not
          (consumable_available ?consumable_resource)
        )
      )
  )
  (:action finalize_lender_slot
    :parameters (?lender_farm - lender_farm ?lender_allocation_slot - lender_allocation_slot ?input_batch - input_batch ?consumable_resource - consumable_resource)
    :precondition
      (and
        (entity_validated ?lender_farm)
        (reserved_input_for_entity ?lender_farm ?input_batch)
        (lender_slot_offered ?lender_farm ?lender_allocation_slot)
        (lender_slot_supplied_with_consumable ?lender_allocation_slot)
        (lender_consumable_allocated ?lender_farm ?consumable_resource)
        (not
          (lender_ready ?lender_farm)
        )
      )
    :effect
      (and
        (lender_slot_ready ?lender_allocation_slot)
        (lender_ready ?lender_farm)
        (consumable_available ?consumable_resource)
        (not
          (lender_consumable_allocated ?lender_farm ?consumable_resource)
        )
      )
  )
  (:action prepare_borrower_allocation_slot
    :parameters (?borrower_farm - borrower_farm ?borrower_allocation_slot - borrower_allocation_slot ?input_batch - input_batch)
    :precondition
      (and
        (entity_validated ?borrower_farm)
        (reserved_input_for_entity ?borrower_farm ?input_batch)
        (borrower_slot_offered ?borrower_farm ?borrower_allocation_slot)
        (not
          (borrower_slot_ready ?borrower_allocation_slot)
        )
        (not
          (borrower_slot_supplied_with_consumable ?borrower_allocation_slot)
        )
      )
    :effect (borrower_slot_ready ?borrower_allocation_slot)
  )
  (:action confirm_borrower_slot
    :parameters (?borrower_farm - borrower_farm ?borrower_allocation_slot - borrower_allocation_slot ?machinery_unit - machinery_unit)
    :precondition
      (and
        (entity_validated ?borrower_farm)
        (machinery_assigned_to_entity ?borrower_farm ?machinery_unit)
        (borrower_slot_offered ?borrower_farm ?borrower_allocation_slot)
        (borrower_slot_ready ?borrower_allocation_slot)
        (not
          (borrower_slot_owner_verified ?borrower_farm)
        )
      )
    :effect
      (and
        (borrower_slot_owner_verified ?borrower_farm)
        (borrower_ready ?borrower_farm)
      )
  )
  (:action assign_consumable_to_borrower_slot
    :parameters (?borrower_farm - borrower_farm ?borrower_allocation_slot - borrower_allocation_slot ?consumable_resource - consumable_resource)
    :precondition
      (and
        (entity_validated ?borrower_farm)
        (borrower_slot_offered ?borrower_farm ?borrower_allocation_slot)
        (consumable_available ?consumable_resource)
        (not
          (borrower_slot_owner_verified ?borrower_farm)
        )
      )
    :effect
      (and
        (borrower_slot_supplied_with_consumable ?borrower_allocation_slot)
        (borrower_slot_owner_verified ?borrower_farm)
        (borrower_consumable_allocated ?borrower_farm ?consumable_resource)
        (not
          (consumable_available ?consumable_resource)
        )
      )
  )
  (:action finalize_borrower_slot
    :parameters (?borrower_farm - borrower_farm ?borrower_allocation_slot - borrower_allocation_slot ?input_batch - input_batch ?consumable_resource - consumable_resource)
    :precondition
      (and
        (entity_validated ?borrower_farm)
        (reserved_input_for_entity ?borrower_farm ?input_batch)
        (borrower_slot_offered ?borrower_farm ?borrower_allocation_slot)
        (borrower_slot_supplied_with_consumable ?borrower_allocation_slot)
        (borrower_consumable_allocated ?borrower_farm ?consumable_resource)
        (not
          (borrower_ready ?borrower_farm)
        )
      )
    :effect
      (and
        (borrower_slot_ready ?borrower_allocation_slot)
        (borrower_ready ?borrower_farm)
        (consumable_available ?consumable_resource)
        (not
          (borrower_consumable_allocated ?borrower_farm ?consumable_resource)
        )
      )
  )
  (:action stage_shared_package_basic
    :parameters (?lender_farm - lender_farm ?borrower_farm - borrower_farm ?lender_allocation_slot - lender_allocation_slot ?borrower_allocation_slot - borrower_allocation_slot ?shared_package - shared_package)
    :precondition
      (and
        (lender_slot_owner_verified ?lender_farm)
        (borrower_slot_owner_verified ?borrower_farm)
        (lender_slot_offered ?lender_farm ?lender_allocation_slot)
        (borrower_slot_offered ?borrower_farm ?borrower_allocation_slot)
        (lender_slot_ready ?lender_allocation_slot)
        (borrower_slot_ready ?borrower_allocation_slot)
        (lender_ready ?lender_farm)
        (borrower_ready ?borrower_farm)
        (shared_package_available ?shared_package)
      )
    :effect
      (and
        (shared_package_staged ?shared_package)
        (package_linked_to_lender_slot ?shared_package ?lender_allocation_slot)
        (package_linked_to_borrower_slot ?shared_package ?borrower_allocation_slot)
        (not
          (shared_package_available ?shared_package)
        )
      )
  )
  (:action stage_shared_package_with_parcel_permission
    :parameters (?lender_farm - lender_farm ?borrower_farm - borrower_farm ?lender_allocation_slot - lender_allocation_slot ?borrower_allocation_slot - borrower_allocation_slot ?shared_package - shared_package)
    :precondition
      (and
        (lender_slot_owner_verified ?lender_farm)
        (borrower_slot_owner_verified ?borrower_farm)
        (lender_slot_offered ?lender_farm ?lender_allocation_slot)
        (borrower_slot_offered ?borrower_farm ?borrower_allocation_slot)
        (lender_slot_supplied_with_consumable ?lender_allocation_slot)
        (borrower_slot_ready ?borrower_allocation_slot)
        (not
          (lender_ready ?lender_farm)
        )
        (borrower_ready ?borrower_farm)
        (shared_package_available ?shared_package)
      )
    :effect
      (and
        (shared_package_staged ?shared_package)
        (package_linked_to_lender_slot ?shared_package ?lender_allocation_slot)
        (package_linked_to_borrower_slot ?shared_package ?borrower_allocation_slot)
        (package_requires_parcel_permission ?shared_package)
        (not
          (shared_package_available ?shared_package)
        )
      )
  )
  (:action stage_shared_package_with_special_permit
    :parameters (?lender_farm - lender_farm ?borrower_farm - borrower_farm ?lender_allocation_slot - lender_allocation_slot ?borrower_allocation_slot - borrower_allocation_slot ?shared_package - shared_package)
    :precondition
      (and
        (lender_slot_owner_verified ?lender_farm)
        (borrower_slot_owner_verified ?borrower_farm)
        (lender_slot_offered ?lender_farm ?lender_allocation_slot)
        (borrower_slot_offered ?borrower_farm ?borrower_allocation_slot)
        (lender_slot_ready ?lender_allocation_slot)
        (borrower_slot_supplied_with_consumable ?borrower_allocation_slot)
        (lender_ready ?lender_farm)
        (not
          (borrower_ready ?borrower_farm)
        )
        (shared_package_available ?shared_package)
      )
    :effect
      (and
        (shared_package_staged ?shared_package)
        (package_linked_to_lender_slot ?shared_package ?lender_allocation_slot)
        (package_linked_to_borrower_slot ?shared_package ?borrower_allocation_slot)
        (package_requires_special_permit ?shared_package)
        (not
          (shared_package_available ?shared_package)
        )
      )
  )
  (:action stage_shared_package_with_both_permissions
    :parameters (?lender_farm - lender_farm ?borrower_farm - borrower_farm ?lender_allocation_slot - lender_allocation_slot ?borrower_allocation_slot - borrower_allocation_slot ?shared_package - shared_package)
    :precondition
      (and
        (lender_slot_owner_verified ?lender_farm)
        (borrower_slot_owner_verified ?borrower_farm)
        (lender_slot_offered ?lender_farm ?lender_allocation_slot)
        (borrower_slot_offered ?borrower_farm ?borrower_allocation_slot)
        (lender_slot_supplied_with_consumable ?lender_allocation_slot)
        (borrower_slot_supplied_with_consumable ?borrower_allocation_slot)
        (not
          (lender_ready ?lender_farm)
        )
        (not
          (borrower_ready ?borrower_farm)
        )
        (shared_package_available ?shared_package)
      )
    :effect
      (and
        (shared_package_staged ?shared_package)
        (package_linked_to_lender_slot ?shared_package ?lender_allocation_slot)
        (package_linked_to_borrower_slot ?shared_package ?borrower_allocation_slot)
        (package_requires_parcel_permission ?shared_package)
        (package_requires_special_permit ?shared_package)
        (not
          (shared_package_available ?shared_package)
        )
      )
  )
  (:action verify_shared_package
    :parameters (?shared_package - shared_package ?lender_farm - lender_farm ?input_batch - input_batch)
    :precondition
      (and
        (shared_package_staged ?shared_package)
        (lender_slot_owner_verified ?lender_farm)
        (reserved_input_for_entity ?lender_farm ?input_batch)
        (not
          (shared_package_verified ?shared_package)
        )
      )
    :effect (shared_package_verified ?shared_package)
  )
  (:action attach_certification_to_package
    :parameters (?coordination_entity - coordination_entity ?certification_record - certification_record ?shared_package - shared_package)
    :precondition
      (and
        (entity_validated ?coordination_entity)
        (coordination_entity_links_package ?coordination_entity ?shared_package)
        (coordination_entity_holds_certification ?coordination_entity ?certification_record)
        (certification_available ?certification_record)
        (shared_package_staged ?shared_package)
        (shared_package_verified ?shared_package)
        (not
          (certification_validated ?certification_record)
        )
      )
    :effect
      (and
        (certification_validated ?certification_record)
        (certification_attached_to_package ?certification_record ?shared_package)
        (not
          (certification_available ?certification_record)
        )
      )
  )
  (:action validate_certification_for_entity
    :parameters (?coordination_entity - coordination_entity ?certification_record - certification_record ?shared_package - shared_package ?input_batch - input_batch)
    :precondition
      (and
        (entity_validated ?coordination_entity)
        (coordination_entity_holds_certification ?coordination_entity ?certification_record)
        (certification_validated ?certification_record)
        (certification_attached_to_package ?certification_record ?shared_package)
        (reserved_input_for_entity ?coordination_entity ?input_batch)
        (not
          (package_requires_parcel_permission ?shared_package)
        )
        (not
          (entity_certification_ready ?coordination_entity)
        )
      )
    :effect (entity_certification_ready ?coordination_entity)
  )
  (:action assign_parcel_permission_to_entity
    :parameters (?coordination_entity - coordination_entity ?land_parcel - land_parcel)
    :precondition
      (and
        (entity_validated ?coordination_entity)
        (land_parcel_available ?land_parcel)
        (not
          (entity_has_parcel_permission ?coordination_entity)
        )
      )
    :effect
      (and
        (entity_has_parcel_permission ?coordination_entity)
        (coordination_entity_assigned_parcel ?coordination_entity ?land_parcel)
        (not
          (land_parcel_available ?land_parcel)
