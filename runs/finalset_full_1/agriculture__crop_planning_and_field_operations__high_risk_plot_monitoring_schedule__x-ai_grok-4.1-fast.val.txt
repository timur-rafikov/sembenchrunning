  (:action enroll_plot
    :parameters (?high_risk_plot - high_risk_plot)
    :precondition
      (and
        (not
          (plot_enrolled ?high_risk_plot)
        )
        (not
          (subject_cleared ?high_risk_plot)
        )
      )
    :effect (plot_enrolled ?high_risk_plot)
  )
  (:action assign_monitoring_asset_to_plot
    :parameters (?high_risk_plot - high_risk_plot ?monitoring_asset - monitoring_asset)
    :precondition
      (and
        (plot_enrolled ?high_risk_plot)
        (not
          (monitoring_asset_reserved ?high_risk_plot)
        )
        (monitoring_asset_available ?monitoring_asset)
      )
    :effect
      (and
        (monitoring_asset_reserved ?high_risk_plot)
        (monitoring_asset_assigned_to_subject ?high_risk_plot ?monitoring_asset)
        (not
          (monitoring_asset_available ?monitoring_asset)
        )
      )
  )
  (:action assign_field_technician_to_plot
    :parameters (?high_risk_plot - high_risk_plot ?field_technician - field_technician)
