  (:requirements :strips :typing :negative-preconditions)
  (:types role_group - object interface_type - object asset_class - object logical_scope - object entity - logical_scope log_channel - role_group event_schema - role_group operator_identity - role_group policy_template - role_group compliance_check - role_group credential - role_group encryption_key - role_group timestamp_token - role_group evidence_resource - interface_type evidence_fragment - interface_type external_reference - interface_type producer_stream - asset_class consumer_stream - asset_class audit_artifact - asset_class logical_partition - entity subpartition - entity producer_component - logical_partition consumer_component - logical_partition audit_coordinator - subpartition)
  (:predicates
    (entity_registered ?audit_subject - entity)
    (entity_prepared ?audit_subject - entity)
    (entity_channel_allocated ?audit_subject - entity)
    (entity_finalized ?audit_subject - entity)
    (eligible_for_credential_binding ?audit_subject - entity)
    (credential_bound ?audit_subject - entity)
    (channel_available ?log_channel - log_channel)
    (entity_bound_to_channel ?audit_subject - entity ?log_channel - log_channel)
    (event_schema_available ?event_schema - event_schema)
    (entity_bound_to_schema ?audit_subject - entity ?event_schema - event_schema)
    (operator_available ?operator_identity - operator_identity)
    (entity_bound_to_operator ?audit_subject - entity ?operator_identity - operator_identity)
    (evidence_resource_available ?evidence_resource - evidence_resource)
    (producer_component_evidence_attached ?producer_component - producer_component ?evidence_resource - evidence_resource)
    (consumer_component_evidence_attached ?consumer_component - consumer_component ?evidence_resource - evidence_resource)
    (producer_component_bound_to_stream ?producer_component - producer_component ?producer_stream - producer_stream)
    (producer_stream_ready ?producer_stream - producer_stream)
    (producer_stream_evidence_locked ?producer_stream - producer_stream)
    (producer_ready_for_composition ?producer_component - producer_component)
    (consumer_component_bound_to_stream ?consumer_component - consumer_component ?consumer_stream - consumer_stream)
    (consumer_stream_ready ?consumer_stream - consumer_stream)
    (consumer_stream_evidence_locked ?consumer_stream - consumer_stream)
    (consumer_ready_for_composition ?consumer_component - consumer_component)
    (artifact_pending_composition ?audit_artifact - audit_artifact)
    (artifact_composed ?audit_artifact - audit_artifact)
    (artifact_bound_to_producer_stream ?audit_artifact - audit_artifact ?producer_stream - producer_stream)
    (artifact_bound_to_consumer_stream ?audit_artifact - audit_artifact ?consumer_stream - consumer_stream)
    (artifact_request_encryption ?audit_artifact - audit_artifact)
    (artifact_request_timestamp ?audit_artifact - audit_artifact)
    (artifact_sealed ?audit_artifact - audit_artifact)
    (coordinator_bound_producer ?audit_coordinator - audit_coordinator ?producer_component - producer_component)
    (coordinator_bound_consumer ?audit_coordinator - audit_coordinator ?consumer_component - consumer_component)
    (coordinator_bound_artifact ?audit_coordinator - audit_coordinator ?audit_artifact - audit_artifact)
    (evidence_fragment_available ?evidence_fragment - evidence_fragment)
    (coordinator_has_evidence_fragment ?audit_coordinator - audit_coordinator ?evidence_fragment - evidence_fragment)
    (evidence_fragment_ingested ?evidence_fragment - evidence_fragment)
    (fragment_bound_to_artifact ?evidence_fragment - evidence_fragment ?audit_artifact - audit_artifact)
    (coordinator_policy_enforcement_started ?audit_coordinator - audit_coordinator)
    (coordinator_encryption_applied ?audit_coordinator - audit_coordinator)
    (coordinator_timestamp_applied ?audit_coordinator - audit_coordinator)
    (policy_attached ?audit_coordinator - audit_coordinator)
    (policy_attachment_recorded ?audit_coordinator - audit_coordinator)
    (policy_check_passed ?audit_coordinator - audit_coordinator)
    (coordinator_ready_for_finalization ?audit_coordinator - audit_coordinator)
    (external_reference_available ?external_reference - external_reference)
    (coordinator_bound_external_reference ?audit_coordinator - audit_coordinator ?external_reference - external_reference)
    (external_reference_attested ?audit_coordinator - audit_coordinator)
    (attestation_recorded ?audit_coordinator - audit_coordinator)
    (attestation_timestamped ?audit_coordinator - audit_coordinator)
    (policy_template_available ?policy_template - policy_template)
    (coordinator_bound_policy_template ?audit_coordinator - audit_coordinator ?policy_template - policy_template)
    (compliance_check_available ?compliance_check - compliance_check)
    (coordinator_bound_compliance_check ?audit_coordinator - audit_coordinator ?compliance_check - compliance_check)
    (encryption_key_available ?encryption_key - encryption_key)
    (coordinator_bound_encryption_key ?audit_coordinator - audit_coordinator ?encryption_key - encryption_key)
    (timestamp_token_available ?timestamp_token - timestamp_token)
    (coordinator_bound_timestamp_token ?audit_coordinator - audit_coordinator ?timestamp_token - timestamp_token)
    (credential_available ?credential - credential)
    (entity_bound_to_credential ?audit_subject - entity ?credential - credential)
    (partition_composition_locked ?producer_component - producer_component)
    (consumer_partition_composition_locked ?consumer_component - consumer_component)
    (coordinator_committed ?audit_coordinator - audit_coordinator)
  )
  (:action register_audit_entity
    :parameters (?audit_subject - entity)
    :precondition
      (and
        (not
          (entity_registered ?audit_subject)
        )
        (not
          (entity_finalized ?audit_subject)
        )
      )
    :effect (entity_registered ?audit_subject)
  )
  (:action allocate_log_channel_to_entity
    :parameters (?audit_subject - entity ?log_channel - log_channel)
    :precondition
      (and
        (entity_registered ?audit_subject)
        (not
          (entity_channel_allocated ?audit_subject)
        )
        (channel_available ?log_channel)
      )
    :effect
      (and
        (entity_channel_allocated ?audit_subject)
        (entity_bound_to_channel ?audit_subject ?log_channel)
        (not
          (channel_available ?log_channel)
        )
      )
  )
  (:action bind_event_schema_to_entity
    :parameters (?audit_subject - entity ?event_schema - event_schema)
    :precondition
      (and
        (entity_registered ?audit_subject)
        (entity_channel_allocated ?audit_subject)
        (event_schema_available ?event_schema)
      )
    :effect
      (and
        (entity_bound_to_schema ?audit_subject ?event_schema)
        (not
          (event_schema_available ?event_schema)
        )
      )
  )
  (:action authorize_entity_for_collection
    :parameters (?audit_subject - entity ?event_schema - event_schema)
    :precondition
      (and
        (entity_registered ?audit_subject)
        (entity_channel_allocated ?audit_subject)
        (entity_bound_to_schema ?audit_subject ?event_schema)
        (not
          (entity_prepared ?audit_subject)
        )
      )
    :effect (entity_prepared ?audit_subject)
  )
  (:action release_event_schema_from_entity
    :parameters (?audit_subject - entity ?event_schema - event_schema)
    :precondition
      (and
        (entity_bound_to_schema ?audit_subject ?event_schema)
      )
    :effect
      (and
        (event_schema_available ?event_schema)
        (not
          (entity_bound_to_schema ?audit_subject ?event_schema)
        )
      )
  )
  (:action bind_operator_identity_to_entity
    :parameters (?audit_subject - entity ?operator_identity - operator_identity)
    :precondition
      (and
        (entity_prepared ?audit_subject)
        (operator_available ?operator_identity)
      )
    :effect
      (and
        (entity_bound_to_operator ?audit_subject ?operator_identity)
        (not
          (operator_available ?operator_identity)
        )
      )
  )
  (:action release_operator_identity_from_entity
    :parameters (?audit_subject - entity ?operator_identity - operator_identity)
    :precondition
      (and
        (entity_bound_to_operator ?audit_subject ?operator_identity)
      )
    :effect
      (and
        (operator_available ?operator_identity)
        (not
          (entity_bound_to_operator ?audit_subject ?operator_identity)
        )
      )
  )
  (:action attach_encryption_key_to_coordinator
    :parameters (?audit_coordinator - audit_coordinator ?encryption_key - encryption_key)
    :precondition
      (and
        (entity_prepared ?audit_coordinator)
        (encryption_key_available ?encryption_key)
      )
    :effect
      (and
        (coordinator_bound_encryption_key ?audit_coordinator ?encryption_key)
        (not
          (encryption_key_available ?encryption_key)
        )
      )
  )
  (:action release_encryption_key_from_coordinator
    :parameters (?audit_coordinator - audit_coordinator ?encryption_key - encryption_key)
    :precondition
      (and
        (coordinator_bound_encryption_key ?audit_coordinator ?encryption_key)
      )
    :effect
      (and
        (encryption_key_available ?encryption_key)
        (not
          (coordinator_bound_encryption_key ?audit_coordinator ?encryption_key)
        )
      )
  )
  (:action attach_timestamp_token_to_coordinator
    :parameters (?audit_coordinator - audit_coordinator ?timestamp_token - timestamp_token)
    :precondition
      (and
        (entity_prepared ?audit_coordinator)
        (timestamp_token_available ?timestamp_token)
      )
    :effect
      (and
        (coordinator_bound_timestamp_token ?audit_coordinator ?timestamp_token)
        (not
          (timestamp_token_available ?timestamp_token)
        )
      )
  )
  (:action release_timestamp_token_from_coordinator
    :parameters (?audit_coordinator - audit_coordinator ?timestamp_token - timestamp_token)
    :precondition
      (and
        (coordinator_bound_timestamp_token ?audit_coordinator ?timestamp_token)
      )
    :effect
      (and
        (timestamp_token_available ?timestamp_token)
        (not
          (coordinator_bound_timestamp_token ?audit_coordinator ?timestamp_token)
        )
      )
  )
  (:action prepare_producer_stream_for_collection
    :parameters (?producer_component - producer_component ?producer_stream - producer_stream ?event_schema - event_schema)
    :precondition
      (and
        (entity_prepared ?producer_component)
        (entity_bound_to_schema ?producer_component ?event_schema)
        (producer_component_bound_to_stream ?producer_component ?producer_stream)
        (not
          (producer_stream_ready ?producer_stream)
        )
        (not
          (producer_stream_evidence_locked ?producer_stream)
        )
      )
    :effect (producer_stream_ready ?producer_stream)
  )
  (:action lock_producer_partition_for_composition
    :parameters (?producer_component - producer_component ?producer_stream - producer_stream ?operator_identity - operator_identity)
    :precondition
      (and
        (entity_prepared ?producer_component)
        (entity_bound_to_operator ?producer_component ?operator_identity)
        (producer_component_bound_to_stream ?producer_component ?producer_stream)
        (producer_stream_ready ?producer_stream)
        (not
          (partition_composition_locked ?producer_component)
        )
      )
    :effect
      (and
        (partition_composition_locked ?producer_component)
        (producer_ready_for_composition ?producer_component)
      )
  )
  (:action attach_evidence_to_producer_partition
    :parameters (?producer_component - producer_component ?producer_stream - producer_stream ?evidence_resource - evidence_resource)
    :precondition
      (and
        (entity_prepared ?producer_component)
        (producer_component_bound_to_stream ?producer_component ?producer_stream)
        (evidence_resource_available ?evidence_resource)
        (not
          (partition_composition_locked ?producer_component)
        )
      )
    :effect
      (and
        (producer_stream_evidence_locked ?producer_stream)
        (partition_composition_locked ?producer_component)
        (producer_component_evidence_attached ?producer_component ?evidence_resource)
        (not
          (evidence_resource_available ?evidence_resource)
        )
      )
  )
  (:action ingest_producer_evidence_into_stream
    :parameters (?producer_component - producer_component ?producer_stream - producer_stream ?event_schema - event_schema ?evidence_resource - evidence_resource)
    :precondition
      (and
        (entity_prepared ?producer_component)
        (entity_bound_to_schema ?producer_component ?event_schema)
        (producer_component_bound_to_stream ?producer_component ?producer_stream)
        (producer_stream_evidence_locked ?producer_stream)
        (producer_component_evidence_attached ?producer_component ?evidence_resource)
        (not
          (producer_ready_for_composition ?producer_component)
        )
      )
    :effect
      (and
        (producer_stream_ready ?producer_stream)
        (producer_ready_for_composition ?producer_component)
        (evidence_resource_available ?evidence_resource)
        (not
          (producer_component_evidence_attached ?producer_component ?evidence_resource)
        )
      )
  )
  (:action prepare_consumer_stream_for_collection
    :parameters (?consumer_component - consumer_component ?consumer_stream - consumer_stream ?event_schema - event_schema)
    :precondition
      (and
        (entity_prepared ?consumer_component)
        (entity_bound_to_schema ?consumer_component ?event_schema)
        (consumer_component_bound_to_stream ?consumer_component ?consumer_stream)
        (not
          (consumer_stream_ready ?consumer_stream)

        (not
          (consumer_stream_evidence_locked ?consumer_stream)
        )
      )
    :effect (consumer_stream_ready ?consumer_stream)
  )
  (:action lock_consumer_partition_for_composition
    :parameters (?consumer_component - consumer_component ?consumer_stream - consumer_stream ?operator_identity - operator_identity)

      (and

      )
  )
  (:action attach_evidence_to_consumer_partition
    :parameters (?consumer_component - consumer_component ?consumer_stream - consumer_stream ?evidence_resource - evidence_resource)
    :precondition
      (and
        (entity_prepared ?consumer_component)
        (consumer_component_bound_to_stream ?consumer_component ?consumer_stream)
        (evidence_resource_available ?evidence_resource)
        (not
          (consumer_partition_composition_locked ?consumer_component)
        )
      )
    :effect
      (and
        (consumer_stream_evidence_locked ?consumer_stream)
        (consumer_partition_composition_locked ?consumer_component)
        (consumer_component_evidence_attached ?consumer_component ?evidence_resource)
        (not
          (evidence_resource_available ?evidence_resource)
        )
      )
  )
  (:action ingest_consumer_evidence_into_stream
    :parameters (?consumer_component - consumer_component ?consumer_stream - consumer_stream ?event_schema - event_schema ?evidence_resource - evidence_resource)
    :precondition
      (and
        (entity_prepared ?consumer_component)
        (entity_bound_to_schema ?consumer_component ?event_schema)
        (consumer_component_bound_to_stream ?consumer_component ?consumer_stream)
        (consumer_stream_evidence_locked ?consumer_stream)
        (consumer_component_evidence_attached ?consumer_component ?evidence_resource)
        (not
          (consumer_ready_for_composition ?consumer_component)
        )
      )
    :effect
      (and
        (consumer_stream_ready ?consumer_stream)
        (consumer_ready_for_composition ?consumer_component)
        (evidence_resource_available ?evidence_resource)
        (not
          (consumer_component_evidence_attached ?consumer_component ?evidence_resource)
        )
      )
  )
  (:action compose_audit_artifact
    :parameters (?producer_component - producer_component ?consumer_component - consumer_component ?producer_stream - producer_stream ?consumer_stream - consumer_stream ?audit_artifact - audit_artifact)
    :precondition
      (and
        (partition_composition_locked ?producer_component)
        (consumer_partition_composition_locked ?consumer_component)
        (producer_component_bound_to_stream ?producer_component ?producer_stream)
        (consumer_component_bound_to_stream ?consumer_component ?consumer_stream)
        (producer_stream_ready ?producer_stream)
        (consumer_stream_ready ?consumer_stream)
        (producer_ready_for_composition ?producer_component)
        (consumer_ready_for_composition ?consumer_component)
        (artifact_pending_composition ?audit_artifact)
      )
    :effect
      (and
        (artifact_composed ?audit_artifact)
        (artifact_bound_to_producer_stream ?audit_artifact ?producer_stream)
        (artifact_bound_to_consumer_stream ?audit_artifact ?consumer_stream)
        (not
          (artifact_pending_composition ?audit_artifact)
        )
      )
  )
  (:action compose_audit_artifact_request_encryption
    :parameters (?producer_component - producer_component ?consumer_component - consumer_component ?producer_stream - producer_stream ?consumer_stream - consumer_stream ?audit_artifact - audit_artifact)
    :precondition
      (and
        (partition_composition_locked ?producer_component)
        (consumer_partition_composition_locked ?consumer_component)
        (producer_component_bound_to_stream ?producer_component ?producer_stream)
        (consumer_component_bound_to_stream ?consumer_component ?consumer_stream)
        (producer_stream_evidence_locked ?producer_stream)
        (consumer_stream_ready ?consumer_stream)
        (not
          (producer_ready_for_composition ?producer_component)
        )
        (consumer_ready_for_composition ?consumer_component)
        (artifact_pending_composition ?audit_artifact)
      )
    :effect
      (and
        (artifact_composed ?audit_artifact)
        (artifact_bound_to_producer_stream ?audit_artifact ?producer_stream)
        (artifact_bound_to_consumer_stream ?audit_artifact ?consumer_stream)
        (artifact_request_encryption ?audit_artifact)
        (not
          (artifact_pending_composition ?audit_artifact)
        )
      )
  )
  (:action compose_audit_artifact_request_timestamp
    :parameters (?producer_component - producer_component ?consumer_component - consumer_component ?producer_stream - producer_stream ?consumer_stream - consumer_stream ?audit_artifact - audit_artifact)
    :precondition
      (and
        (partition_composition_locked ?producer_component)
        (consumer_partition_composition_locked ?consumer_component)
        (producer_component_bound_to_stream ?producer_component ?producer_stream)
        (consumer_component_bound_to_stream ?consumer_component ?consumer_stream)
        (producer_stream_ready ?producer_stream)
        (consumer_stream_evidence_locked ?consumer_stream)
        (producer_ready_for_composition ?producer_component)
        (not
          (consumer_ready_for_composition ?consumer_component)
        )
        (artifact_pending_composition ?audit_artifact)
      )
    :effect
      (and
        (artifact_composed ?audit_artifact)
        (artifact_bound_to_producer_stream ?audit_artifact ?producer_stream)
        (artifact_bound_to_consumer_stream ?audit_artifact ?consumer_stream)
        (artifact_request_timestamp ?audit_artifact)
        (not
          (artifact_pending_composition ?audit_artifact)
        )
      )
  )
  (:action compose_audit_artifact_request_encryption_and_timestamp
    :parameters (?producer_component - producer_component ?consumer_component - consumer_component ?producer_stream - producer_stream ?consumer_stream - consumer_stream ?audit_artifact - audit_artifact)
    :precondition
      (and
        (partition_composition_locked ?producer_component)
        (consumer_partition_composition_locked ?consumer_component)
        (producer_component_bound_to_stream ?producer_component ?producer_stream)
        (consumer_component_bound_to_stream ?consumer_component ?consumer_stream)
        (producer_stream_evidence_locked ?producer_stream)
        (consumer_stream_evidence_locked ?consumer_stream)
        (not
          (producer_ready_for_composition ?producer_component)
        )
        (not
          (consumer_ready_for_composition ?consumer_component)
        )
        (artifact_pending_composition ?audit_artifact)
      )
    :effect
      (and
        (artifact_composed ?audit_artifact)
        (artifact_bound_to_producer_stream ?audit_artifact ?producer_stream)
        (artifact_bound_to_consumer_stream ?audit_artifact ?consumer_stream)
        (artifact_request_encryption ?audit_artifact)
        (artifact_request_timestamp ?audit_artifact)
        (not
          (artifact_pending_composition ?audit_artifact)
        )
      )
  )
  (:action seal_audit_artifact
    :parameters (?audit_artifact - audit_artifact ?producer_component - producer_component ?event_schema - event_schema)
    :precondition
      (and
        (artifact_composed ?audit_artifact)
        (partition_composition_locked ?producer_component)
        (entity_bound_to_schema ?producer_component ?event_schema)
        (not
          (artifact_sealed ?audit_artifact)
        )
      )
    :effect (artifact_sealed ?audit_artifact)
  )
  (:action ingest_evidence_fragment_into_coordinator

        (not
          (evidence_fragment_ingested ?evidence_fragment)

    :effect
      (and
        (evidence_fragment_ingested ?evidence_fragment)

          (evidence_fragment_available ?evidence_fragment)
        )
      )
  )
  (:action prepare_coordinator_with_fragment

        (not
          (artifact_request_encryption ?audit_artifact)
        )
        (not
          (coordinator_policy_enforcement_started ?audit_coordinator)

  (:action attach_policy_template_to_coordinator
    :parameters (?audit_coordinator - audit_coordinator ?policy_template - policy_template)
    :precondition
      (and
        (entity_prepared ?audit_coordinator)
        (policy_template_available ?policy_template)
        (not
          (policy_attached ?audit_coordinator)

      (and
        (policy_attached ?audit_coordinator)

        (not
          (policy_template_available ?policy_template)
        )
      )
  )
  (:action activate_policy_on_coordinator
    :parameters (?audit_coordinator - audit_coordinator ?evidence_fragment - evidence_fragment ?audit_artifact - audit_artifact ?event_schema - event_schema ?policy_template - policy_template)
    :precondition
      (and
        (entity_prepared ?audit_coordinator)
        (coordinator_has_evidence_fragment ?audit_coordinator ?evidence_fragment)
        (evidence_fragment_ingested ?evidence_fragment)
        (fragment_bound_to_artifact ?evidence_fragment ?audit_artifact)
        (entity_bound_to_schema ?audit_coordinator ?event_schema)
        (artifact_request_encryption ?audit_artifact)
        (policy_attached ?audit_coordinator)
        (coordinator_bound_policy_template ?audit_coordinator ?policy_template)
        (not
          (coordinator_policy_enforcement_started ?audit_coordinator)
        )
      )
    :effect
      (and
        (coordinator_policy_enforcement_started ?audit_coordinator)
        (policy_attachment_recorded ?audit_coordinator)
      )
  )
  (:action apply_encryption_to_coordinator
    :parameters (?audit_coordinator - audit_coordinator ?encryption_key - encryption_key ?operator_identity - operator_identity ?evidence_fragment - evidence_fragment ?audit_artifact - audit_artifact)
    :precondition
      (and
        (coordinator_policy_enforcement_started ?audit_coordinator)
        (coordinator_bound_encryption_key ?audit_coordinator ?encryption_key)
        (entity_bound_to_operator ?audit_coordinator ?operator_identity)
        (coordinator_has_evidence_fragment ?audit_coordinator ?evidence_fragment)
        (fragment_bound_to_artifact ?evidence_fragment ?audit_artifact)
        (not
          (artifact_request_timestamp ?audit_artifact)
        )
        (not
          (coordinator_encryption_applied ?audit_coordinator)
        )
      )
    :effect (coordinator_encryption_applied ?audit_coordinator)
  )
  (:action apply_encryption_to_coordinator_alternate
    :parameters (?audit_coordinator - audit_coordinator ?encryption_key - encryption_key ?operator_identity - operator_identity ?evidence_fragment - evidence_fragment ?audit_artifact - audit_artifact)
    :precondition
      (and
        (coordinator_policy_enforcement_started ?audit_coordinator)
        (coordinator_bound_encryption_key ?audit_coordinator ?encryption_key)
        (entity_bound_to_operator ?audit_coordinator ?operator_identity)
        (coordinator_has_evidence_fragment ?audit_coordinator ?evidence_fragment)
        (fragment_bound_to_artifact ?evidence_fragment ?audit_artifact)
        (artifact_request_timestamp ?audit_artifact)
        (not
          (coordinator_encryption_applied ?audit_coordinator)
