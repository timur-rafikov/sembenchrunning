  (:requirements :strips :typing :negative-preconditions)
  (:types domain_object - object operational_actor_or_asset - domain_object resource_or_document - domain_object location_node - domain_object case_entity_root - domain_object case_entity - case_entity_root response_team - operational_actor_or_asset forensic_sample - operational_actor_or_asset secure_hold_location - operational_actor_or_asset regulatory_clearance_document - operational_actor_or_asset supply_partner_confirmation - operational_actor_or_asset notification_template - operational_actor_or_asset logistics_provider - operational_actor_or_asset regulatory_liaison - operational_actor_or_asset mitigation_inventory - resource_or_document laboratory_report - resource_or_document public_health_authority - resource_or_document production_line - location_node distribution_center - location_node recall_event - location_node case_entity_group - case_entity organizational_unit - case_entity manufacturing_site - case_entity_group distribution_partner - case_entity_group incident_coordinating_unit - organizational_unit)

  (:predicates
    (case_opened ?affected_product_batch - case_entity)
    (entity_case_validated ?affected_product_batch - case_entity)
    (team_assigned_to_case ?affected_product_batch - case_entity)
    (execution_authorized ?affected_product_batch - case_entity)
    (operation_flagged_for_execution ?affected_product_batch - case_entity)
    (notification_committed ?affected_product_batch - case_entity)
    (response_team_available ?response_team - response_team)
    (case_assigned_response_team ?affected_product_batch - case_entity ?response_team - response_team)
    (forensic_sample_available ?forensic_sample - forensic_sample)
    (sample_associated_with_entity ?affected_product_batch - case_entity ?forensic_sample - forensic_sample)
    (secure_hold_available ?secure_hold_location - secure_hold_location)
    (entity_quarantined_at_location ?affected_product_batch - case_entity ?secure_hold_location - secure_hold_location)
    (mitigation_inventory_available ?mitigation_inventory - mitigation_inventory)
    (mitigation_inventory_allocated_to_site ?manufacturing_site - manufacturing_site ?mitigation_inventory - mitigation_inventory)
    (mitigation_inventory_allocated_to_partner ?distribution_partner - distribution_partner ?mitigation_inventory - mitigation_inventory)
    (production_line_assigned_to_site ?manufacturing_site - manufacturing_site ?production_line - production_line)
    (production_line_under_investigation ?production_line - production_line)
    (production_line_in_quarantine ?production_line - production_line)
    (site_mitigation_engaged ?manufacturing_site - manufacturing_site)
    (partner_associated_with_distribution_center ?distribution_partner - distribution_partner ?distribution_center - distribution_center)
    (distribution_center_under_investigation ?distribution_center - distribution_center)
    (distribution_center_in_quarantine ?distribution_center - distribution_center)
    (distribution_partner_mitigation_engaged ?distribution_partner - distribution_partner)
    (recall_operation_defined ?recall_operation - recall_event)
    (recall_operation_activated ?recall_operation - recall_event)
    (recall_includes_production_line ?recall_operation - recall_event ?production_line - production_line)
    (recall_includes_distribution_center ?recall_operation - recall_event ?distribution_center - distribution_center)
    (recall_requires_logistics_confirmation ?recall_operation - recall_event)
    (recall_requires_regulatory_clearance ?recall_operation - recall_event)
    (recall_ready_for_report_ingestion ?recall_operation - recall_event)
    (coordinating_unit_assigned_to_site ?incident_coordinating_unit - incident_coordinating_unit ?manufacturing_site - manufacturing_site)
    (coordinating_unit_assigned_to_partner ?incident_coordinating_unit - incident_coordinating_unit ?distribution_partner - distribution_partner)
    (coordinating_unit_responsible_for_recall ?incident_coordinating_unit - incident_coordinating_unit ?recall_operation - recall_event)
    (laboratory_report_available ?laboratory_report - laboratory_report)
    (coordinating_unit_has_report ?incident_coordinating_unit - incident_coordinating_unit ?laboratory_report - laboratory_report)
    (laboratory_report_ingested ?laboratory_report - laboratory_report)
    (laboratory_report_linked_to_recall ?laboratory_report - laboratory_report ?recall_operation - recall_event)
    (unit_authorization_phase_one ?incident_coordinating_unit - incident_coordinating_unit)
    (unit_authorization_phase_two ?incident_coordinating_unit - incident_coordinating_unit)
    (unit_execution_ready ?incident_coordinating_unit - incident_coordinating_unit)
    (unit_has_regulatory_clearance ?incident_coordinating_unit - incident_coordinating_unit)
    (unit_regulatory_clearance_acknowledged ?incident_coordinating_unit - incident_coordinating_unit)
    (unit_final_checks_passed ?incident_coordinating_unit - incident_coordinating_unit)
    (unit_tasks_assigned ?incident_coordinating_unit - incident_coordinating_unit)
    (pha_contact_available ?public_health_authority - public_health_authority)
    (coordinating_unit_has_pha_contact ?incident_coordinating_unit - incident_coordinating_unit ?public_health_authority - public_health_authority)
    (unit_pha_engaged ?incident_coordinating_unit - incident_coordinating_unit)
    (unit_pha_action_initiated ?incident_coordinating_unit - incident_coordinating_unit)
    (unit_pha_action_confirmed ?incident_coordinating_unit - incident_coordinating_unit)
    (regulatory_clearance_document_available ?regulatory_clearance_document - regulatory_clearance_document)
    (unit_has_regulatory_document ?incident_coordinating_unit - incident_coordinating_unit ?regulatory_clearance_document - regulatory_clearance_document)
    (supply_partner_confirmation_available ?supply_partner_confirmation - supply_partner_confirmation)
    (unit_has_supply_partner_confirmation ?incident_coordinating_unit - incident_coordinating_unit ?supply_partner_confirmation - supply_partner_confirmation)
    (logistics_provider_available ?logistics_provider - logistics_provider)
    (coordinating_unit_assigned_logistics_provider ?incident_coordinating_unit - incident_coordinating_unit ?logistics_provider - logistics_provider)
    (regulatory_liaison_available ?regulatory_liaison - regulatory_liaison)
    (coordinating_unit_assigned_regulatory_liaison ?incident_coordinating_unit - incident_coordinating_unit ?regulatory_liaison - regulatory_liaison)
    (notification_template_available ?notification_template - notification_template)
    (notification_template_assigned_to_entity ?affected_product_batch - case_entity ?notification_template - notification_template)
    (manufacturing_site_execution_ready ?manufacturing_site - manufacturing_site)
    (distribution_partner_execution_ready ?distribution_partner - distribution_partner)
    (unit_execution_confirmed ?incident_coordinating_unit - incident_coordinating_unit)
  )
  (:action record_tamper_evidence_case
    :parameters (?affected_product_batch - case_entity)
    :precondition
      (and
        (not
          (case_opened ?affected_product_batch)
        )
        (not
          (execution_authorized ?affected_product_batch)
        )
      )
    :effect (case_opened ?affected_product_batch)
  )
  (:action assign_response_team_to_case
    :parameters (?affected_product_batch - case_entity ?response_team - response_team)
    :precondition
      (and
        (case_opened ?affected_product_batch)
        (not
          (team_assigned_to_case ?affected_product_batch)
        )
        (response_team_available ?response_team)
      )
    :effect
      (and
        (team_assigned_to_case ?affected_product_batch)
        (case_assigned_response_team ?affected_product_batch ?response_team)
        (not
          (response_team_available ?response_team)
        )
      )
  )
  (:action associate_forensic_sample_with_case
    :parameters (?affected_product_batch - case_entity ?forensic_sample - forensic_sample)
    :precondition
      (and
        (case_opened ?affected_product_batch)
        (team_assigned_to_case ?affected_product_batch)
        (forensic_sample_available ?forensic_sample)
      )
    :effect
      (and
        (sample_associated_with_entity ?affected_product_batch ?forensic_sample)
        (not
          (forensic_sample_available ?forensic_sample)
        )
      )
  )
  (:action validate_case
    :parameters (?affected_product_batch - case_entity ?forensic_sample - forensic_sample)
    :precondition
      (and
        (case_opened ?affected_product_batch)
        (team_assigned_to_case ?affected_product_batch)
        (sample_associated_with_entity ?affected_product_batch ?forensic_sample)
        (not
          (entity_case_validated ?affected_product_batch)
        )
      )
    :effect (entity_case_validated ?affected_product_batch)
  )
  (:action release_forensic_sample
    :parameters (?affected_product_batch - case_entity ?forensic_sample - forensic_sample)
    :precondition
      (and
        (sample_associated_with_entity ?affected_product_batch ?forensic_sample)
      )
    :effect
      (and
        (forensic_sample_available ?forensic_sample)
        (not
          (sample_associated_with_entity ?affected_product_batch ?forensic_sample)
        )
      )
  )
  (:action assign_batch_to_secure_hold_location
    :parameters (?affected_product_batch - case_entity ?secure_hold_location - secure_hold_location)
    :precondition
      (and
        (entity_case_validated ?affected_product_batch)
        (secure_hold_available ?secure_hold_location)
      )
    :effect
      (and
        (entity_quarantined_at_location ?affected_product_batch ?secure_hold_location)
        (not
          (secure_hold_available ?secure_hold_location)
        )
      )
  )
  (:action release_secure_hold_location
    :parameters (?affected_product_batch - case_entity ?secure_hold_location - secure_hold_location)
    :precondition
      (and
        (entity_quarantined_at_location ?affected_product_batch ?secure_hold_location)
      )
    :effect
      (and
        (secure_hold_available ?secure_hold_location)
        (not
          (entity_quarantined_at_location ?affected_product_batch ?secure_hold_location)
        )
      )
  )
  (:action assign_logistics_provider_to_unit
    :parameters (?incident_coordinating_unit - incident_coordinating_unit ?logistics_provider - logistics_provider)
    :precondition
      (and
        (entity_case_validated ?incident_coordinating_unit)
        (logistics_provider_available ?logistics_provider)
      )
    :effect
      (and
        (coordinating_unit_assigned_logistics_provider ?incident_coordinating_unit ?logistics_provider)
        (not
          (logistics_provider_available ?logistics_provider)
        )
      )
  )
  (:action release_logistics_provider_from_unit
    :parameters (?incident_coordinating_unit - incident_coordinating_unit ?logistics_provider - logistics_provider)
    :precondition
      (and
        (coordinating_unit_assigned_logistics_provider ?incident_coordinating_unit ?logistics_provider)
      )
    :effect
      (and
        (logistics_provider_available ?logistics_provider)
        (not
          (coordinating_unit_assigned_logistics_provider ?incident_coordinating_unit ?logistics_provider)
        )
      )
  )
  (:action assign_regulatory_liaison_to_unit
    :parameters (?incident_coordinating_unit - incident_coordinating_unit ?regulatory_liaison - regulatory_liaison)
    :precondition
      (and
        (entity_case_validated ?incident_coordinating_unit)
        (regulatory_liaison_available ?regulatory_liaison)
      )
    :effect
      (and
        (coordinating_unit_assigned_regulatory_liaison ?incident_coordinating_unit ?regulatory_liaison)
        (not
          (regulatory_liaison_available ?regulatory_liaison)
        )
      )
  )
  (:action release_regulatory_liaison_from_unit
    :parameters (?incident_coordinating_unit - incident_coordinating_unit ?regulatory_liaison - regulatory_liaison)
    :precondition
      (and
        (coordinating_unit_assigned_regulatory_liaison ?incident_coordinating_unit ?regulatory_liaison)
      )
    :effect
      (and
        (regulatory_liaison_available ?regulatory_liaison)
        (not
          (coordinating_unit_assigned_regulatory_liaison ?incident_coordinating_unit ?regulatory_liaison)
        )
      )
  )
  (:action flag_production_line_for_investigation
    :parameters (?manufacturing_site - manufacturing_site ?production_line - production_line ?forensic_sample - forensic_sample)
    :precondition
      (and
        (entity_case_validated ?manufacturing_site)
        (sample_associated_with_entity ?manufacturing_site ?forensic_sample)
        (production_line_assigned_to_site ?manufacturing_site ?production_line)
        (not
          (production_line_under_investigation ?production_line)
        )
        (not
          (production_line_in_quarantine ?production_line)
        )
      )
    :effect (production_line_under_investigation ?production_line)
  )
  (:action engage_site_mitigation
    :parameters (?manufacturing_site - manufacturing_site ?production_line - production_line ?secure_hold_location - secure_hold_location)
    :precondition
      (and
        (entity_case_validated ?manufacturing_site)
        (entity_quarantined_at_location ?manufacturing_site ?secure_hold_location)
        (production_line_assigned_to_site ?manufacturing_site ?production_line)
        (production_line_under_investigation ?production_line)
        (not
          (manufacturing_site_execution_ready ?manufacturing_site)
        )
      )
    :effect
      (and
        (manufacturing_site_execution_ready ?manufacturing_site)
        (site_mitigation_engaged ?manufacturing_site)
      )
  )
  (:action allocate_mitigation_inventory_to_site
    :parameters (?manufacturing_site - manufacturing_site ?production_line - production_line ?mitigation_inventory - mitigation_inventory)
    :precondition
      (and
        (entity_case_validated ?manufacturing_site)
        (production_line_assigned_to_site ?manufacturing_site ?production_line)
        (mitigation_inventory_available ?mitigation_inventory)
        (not
          (manufacturing_site_execution_ready ?manufacturing_site)
        )
      )
    :effect
      (and
        (production_line_in_quarantine ?production_line)
        (manufacturing_site_execution_ready ?manufacturing_site)
        (mitigation_inventory_allocated_to_site ?manufacturing_site ?mitigation_inventory)
        (not
          (mitigation_inventory_available ?mitigation_inventory)
        )
      )
  )
  (:action reopen_production_line_after_mitigation
    :parameters (?manufacturing_site - manufacturing_site ?production_line - production_line ?forensic_sample - forensic_sample ?mitigation_inventory - mitigation_inventory)
    :precondition
      (and
        (entity_case_validated ?manufacturing_site)
        (sample_associated_with_entity ?manufacturing_site ?forensic_sample)
        (production_line_assigned_to_site ?manufacturing_site ?production_line)
        (production_line_in_quarantine ?production_line)
        (mitigation_inventory_allocated_to_site ?manufacturing_site ?mitigation_inventory)
        (not
          (site_mitigation_engaged ?manufacturing_site)
        )
      )
    :effect
      (and
        (production_line_under_investigation ?production_line)
        (site_mitigation_engaged ?manufacturing_site)
        (mitigation_inventory_available ?mitigation_inventory)
        (not
          (mitigation_inventory_allocated_to_site ?manufacturing_site ?mitigation_inventory)
        )
      )
  )
  (:action flag_distribution_center_for_investigation
    :parameters (?distribution_partner - distribution_partner ?distribution_center - distribution_center ?forensic_sample - forensic_sample)
    :precondition
      (and
        (entity_case_validated ?distribution_partner)
        (sample_associated_with_entity ?distribution_partner ?forensic_sample)
        (partner_associated_with_distribution_center ?distribution_partner ?distribution_center)
        (not
          (distribution_center_under_investigation ?distribution_center)
        )
        (not
          (distribution_center_in_quarantine ?distribution_center)
        )
      )
    :effect (distribution_center_under_investigation ?distribution_center)
  )
  (:action engage_partner_mitigation
    :parameters (?distribution_partner - distribution_partner ?distribution_center - distribution_center ?secure_hold_location - secure_hold_location)
    :precondition
      (and
        (entity_case_validated ?distribution_partner)
        (entity_quarantined_at_location ?distribution_partner ?secure_hold_location)
        (partner_associated_with_distribution_center ?distribution_partner ?distribution_center)
        (distribution_center_under_investigation ?distribution_center)
        (not
          (distribution_partner_execution_ready ?distribution_partner)
        )
      )
    :effect
      (and
        (distribution_partner_execution_ready ?distribution_partner)
        (distribution_partner_mitigation_engaged ?distribution_partner)
      )
  )
  (:action allocate_mitigation_inventory_to_partner
    :parameters (?distribution_partner - distribution_partner ?distribution_center - distribution_center ?mitigation_inventory - mitigation_inventory)
    :precondition
      (and
        (entity_case_validated ?distribution_partner)
        (partner_associated_with_distribution_center ?distribution_partner ?distribution_center)
        (mitigation_inventory_available ?mitigation_inventory)
        (not
          (distribution_partner_execution_ready ?distribution_partner)
        )
      )
    :effect
      (and
        (distribution_center_in_quarantine ?distribution_center)
        (distribution_partner_execution_ready ?distribution_partner)
        (mitigation_inventory_allocated_to_partner ?distribution_partner ?mitigation_inventory)
        (not
          (mitigation_inventory_available ?mitigation_inventory)
        )
      )
  )
  (:action reopen_distribution_center_after_mitigation
    :parameters (?distribution_partner - distribution_partner ?distribution_center - distribution_center ?forensic_sample - forensic_sample ?mitigation_inventory - mitigation_inventory)
    :precondition
      (and
        (entity_case_validated ?distribution_partner)
        (sample_associated_with_entity ?distribution_partner ?forensic_sample)
        (partner_associated_with_distribution_center ?distribution_partner ?distribution_center)
        (distribution_center_in_quarantine ?distribution_center)
        (mitigation_inventory_allocated_to_partner ?distribution_partner ?mitigation_inventory)
        (not
          (distribution_partner_mitigation_engaged ?distribution_partner)
        )
      )
    :effect
      (and
        (distribution_center_under_investigation ?distribution_center)
        (distribution_partner_mitigation_engaged ?distribution_partner)
        (mitigation_inventory_available ?mitigation_inventory)
        (not
          (mitigation_inventory_allocated_to_partner ?distribution_partner ?mitigation_inventory)
        )
      )
  )
  (:action form_recall_operation_standard
    :parameters (?manufacturing_site - manufacturing_site ?distribution_partner - distribution_partner ?production_line - production_line ?distribution_center - distribution_center ?recall_operation - recall_event)
    :precondition
      (and
        (manufacturing_site_execution_ready ?manufacturing_site)
        (distribution_partner_execution_ready ?distribution_partner)
        (production_line_assigned_to_site ?manufacturing_site ?production_line)
        (partner_associated_with_distribution_center ?distribution_partner ?distribution_center)
        (production_line_under_investigation ?production_line)
        (distribution_center_under_investigation ?distribution_center)
        (site_mitigation_engaged ?manufacturing_site)
        (distribution_partner_mitigation_engaged ?distribution_partner)
        (recall_operation_defined ?recall_operation)
      )
    :effect
      (and
        (recall_operation_activated ?recall_operation)
        (recall_includes_production_line ?recall_operation ?production_line)
        (recall_includes_distribution_center ?recall_operation ?distribution_center)
        (not
          (recall_operation_defined ?recall_operation)
        )
      )
  )
  (:action form_recall_operation_with_logistics_requirement
    :parameters (?manufacturing_site - manufacturing_site ?distribution_partner - distribution_partner ?production_line - production_line ?distribution_center - distribution_center ?recall_operation - recall_event)
    :precondition
      (and
        (manufacturing_site_execution_ready ?manufacturing_site)
        (distribution_partner_execution_ready ?distribution_partner)
        (production_line_assigned_to_site ?manufacturing_site ?production_line)
        (partner_associated_with_distribution_center ?distribution_partner ?distribution_center)
        (production_line_in_quarantine ?production_line)
        (distribution_center_under_investigation ?distribution_center)
        (not
          (site_mitigation_engaged ?manufacturing_site)
        )
        (distribution_partner_mitigation_engaged ?distribution_partner)
        (recall_operation_defined ?recall_operation)
      )
    :effect
      (and
        (recall_operation_activated ?recall_operation)
        (recall_includes_production_line ?recall_operation ?production_line)
        (recall_includes_distribution_center ?recall_operation ?distribution_center)
        (recall_requires_logistics_confirmation ?recall_operation)
        (not
          (recall_operation_defined ?recall_operation)
        )
      )
  )
  (:action form_recall_operation_with_regulatory_requirement
    :parameters (?manufacturing_site - manufacturing_site ?distribution_partner - distribution_partner ?production_line - production_line ?distribution_center - distribution_center ?recall_operation - recall_event)
    :precondition
      (and
        (manufacturing_site_execution_ready ?manufacturing_site)
        (distribution_partner_execution_ready ?distribution_partner)
        (production_line_assigned_to_site ?manufacturing_site ?production_line)
        (partner_associated_with_distribution_center ?distribution_partner ?distribution_center)
        (production_line_under_investigation ?production_line)
        (distribution_center_in_quarantine ?distribution_center)
        (site_mitigation_engaged ?manufacturing_site)
        (not
          (distribution_partner_mitigation_engaged ?distribution_partner)
        )
        (recall_operation_defined ?recall_operation)
      )
    :effect
      (and
        (recall_operation_activated ?recall_operation)
        (recall_includes_production_line ?recall_operation ?production_line)
        (recall_includes_distribution_center ?recall_operation ?distribution_center)
        (recall_requires_regulatory_clearance ?recall_operation)
        (not
          (recall_operation_defined ?recall_operation)
        )
      )
  )
  (:action form_recall_operation_with_full_requirements
    :parameters (?manufacturing_site - manufacturing_site ?distribution_partner - distribution_partner ?production_line - production_line ?distribution_center - distribution_center ?recall_operation - recall_event)
    :precondition
      (and
        (manufacturing_site_execution_ready ?manufacturing_site)
        (distribution_partner_execution_ready ?distribution_partner)
        (production_line_assigned_to_site ?manufacturing_site ?production_line)
        (partner_associated_with_distribution_center ?distribution_partner ?distribution_center)
        (production_line_in_quarantine ?production_line)
        (distribution_center_in_quarantine ?distribution_center)
        (not
          (site_mitigation_engaged ?manufacturing_site)
        )
        (not
          (distribution_partner_mitigation_engaged ?distribution_partner)
        )
        (recall_operation_defined ?recall_operation)
      )
    :effect
      (and
        (recall_operation_activated ?recall_operation)
        (recall_includes_production_line ?recall_operation ?production_line)
        (recall_includes_distribution_center ?recall_operation ?distribution_center)
        (recall_requires_logistics_confirmation ?recall_operation)
        (recall_requires_regulatory_clearance ?recall_operation)
        (not
          (recall_operation_defined ?recall_operation)
        )
      )
  )
  (:action mark_recall_for_report_ingestion
    :parameters (?recall_operation - recall_event ?manufacturing_site - manufacturing_site ?forensic_sample - forensic_sample)
    :precondition
      (and
        (recall_operation_activated ?recall_operation)
        (manufacturing_site_execution_ready ?manufacturing_site)
        (sample_associated_with_entity ?manufacturing_site ?forensic_sample)
        (not
          (recall_ready_for_report_ingestion ?recall_operation)
        )
      )
    :effect (recall_ready_for_report_ingestion ?recall_operation)
  )
  (:action ingest_laboratory_report
    :parameters (?incident_coordinating_unit - incident_coordinating_unit ?laboratory_report - laboratory_report ?recall_operation - recall_event)
    :precondition
      (and
        (entity_case_validated ?incident_coordinating_unit)
        (coordinating_unit_responsible_for_recall ?incident_coordinating_unit ?recall_operation)
        (coordinating_unit_has_report ?incident_coordinating_unit ?laboratory_report)
        (laboratory_report_available ?laboratory_report)
        (recall_operation_activated ?recall_operation)
        (recall_ready_for_report_ingestion ?recall_operation)
        (not
          (laboratory_report_ingested ?laboratory_report)
        )
      )
    :effect
      (and
        (laboratory_report_ingested ?laboratory_report)
        (laboratory_report_linked_to_recall ?laboratory_report ?recall_operation)
        (not
          (laboratory_report_available ?laboratory_report)
        )
      )
  )
  (:action authorize_coordinating_unit_phase_one
    :parameters (?incident_coordinating_unit - incident_coordinating_unit ?laboratory_report - laboratory_report ?recall_operation - recall_event ?forensic_sample - forensic_sample)
    :precondition
      (and
        (entity_case_validated ?incident_coordinating_unit)
        (coordinating_unit_has_report ?incident_coordinating_unit ?laboratory_report)
        (laboratory_report_ingested ?laboratory_report)
        (laboratory_report_linked_to_recall ?laboratory_report ?recall_operation)
        (sample_associated_with_entity ?incident_coordinating_unit ?forensic_sample)
        (not
          (recall_requires_logistics_confirmation ?recall_operation)
        )
        (not
          (unit_authorization_phase_one ?incident_coordinating_unit)
        )
      )
    :effect (unit_authorization_phase_one ?incident_coordinating_unit)
