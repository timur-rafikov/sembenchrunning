--- domain.pddl (as passed to VAL) ---
(define (domain healing_supply_reservation)
  (:requirements :strips :typing :negative-preconditions)
  (:types item_category - object material_category - object component_category - object entity_category - object unit - entity_category supply_template - item_category healing_method - item_category tool - item_category attachment_type_a - item_category attachment_type_b - item_category consumable - item_category equipment_module - item_category catalyst - item_category raw_material - material_category consumable_pack - material_category special_resource - material_category component_type_a - component_category component_type_b - component_category prepared_bundle - component_category unit_subtype_category - unit facility_subtype_category - unit frontline_unit - unit_subtype_category support_unit - unit_subtype_category facility - facility_subtype_category)
  (:predicates
    (nominated_unit ?unit - unit)
    (ready_for_preparation ?unit - unit)
    (template_assigned ?unit - unit)
    (reservation_finalized ?unit - unit)
    (prepared_recipient ?unit - unit)
    (consumable_reserved ?unit - unit)
    (template_available ?supply_template - supply_template)
    (assigned_template ?unit - unit ?supply_template - supply_template)
    (method_available ?healing_method - healing_method)
    (assigned_method ?unit - unit ?healing_method - healing_method)
    (tool_available ?tool - tool)
    (assigned_tool ?unit - unit ?tool - tool)
    (material_available ?raw_material - raw_material)
    (staged_material_frontline ?frontline_unit - frontline_unit ?raw_material - raw_material)
    (staged_material_support ?support_unit - support_unit ?raw_material - raw_material)
    (unit_component_a_compat ?frontline_unit - frontline_unit ?component_type_a - component_type_a)
    (component_a_staged ?component_type_a - component_type_a)
    (component_a_material_attached ?component_type_a - component_type_a)
    (frontline_unit_component_ready ?frontline_unit - frontline_unit)
    (unit_component_b_compat ?support_unit - support_unit ?component_type_b - component_type_b)
    (component_b_staged ?component_type_b - component_type_b)
    (component_b_material_attached ?component_type_b - component_type_b)
    (support_unit_component_ready ?support_unit - support_unit)
    (bundle_slot_available ?prepared_bundle - prepared_bundle)
    (bundle_assembled ?prepared_bundle - prepared_bundle)
    (bundle_has_component_a ?prepared_bundle - prepared_bundle ?component_type_a - component_type_a)
    (bundle_has_component_b ?prepared_bundle - prepared_bundle ?component_type_b - component_type_b)
    (bundle_flag_a ?prepared_bundle - prepared_bundle)
    (bundle_flag_b ?prepared_bundle - prepared_bundle)
    (bundle_activated ?prepared_bundle - prepared_bundle)
    (facility_supports_frontline_unit ?facility - facility ?frontline_unit - frontline_unit)
    (facility_supports_support_unit ?facility - facility ?support_unit - support_unit)
    (facility_has_bundle ?facility - facility ?prepared_bundle - prepared_bundle)
    (consumable_pack_available ?consumable_pack - consumable_pack)
    (facility_has_consumable_pack ?facility - facility ?consumable_pack - consumable_pack)
    (consumable_pack_engaged ?consumable_pack - consumable_pack)
    (pack_attached_to_bundle ?consumable_pack - consumable_pack ?prepared_bundle - prepared_bundle)
    (facility_pack_configured ?facility - facility)
    (facility_module_attached ?facility - facility)
    (facility_configured_for_activation ?facility - facility)
    (facility_has_primary_attachment ?facility - facility)
    (facility_secondary_attachment_configured ?facility - facility)
    (facility_secondary_attachment_attached ?facility - facility)
    (facility_final_checks_passed ?facility - facility)
    (special_resource_available ?special_resource - special_resource)
    (facility_has_special_resource ?facility - facility ?special_resource - special_resource)
    (facility_upgrade_applied ?facility - facility)
    (facility_upgrade_stage1 ?facility - facility)
    (facility_upgrade_ready ?facility - facility)
    (attachment_type_a_available ?attachment_type_a - attachment_type_a)
    (facility_has_attachment_type_a ?facility - facility ?attachment_type_a - attachment_type_a)
    (attachment_type_b_available ?attachment_type_b - attachment_type_b)
    (facility_has_attachment_type_b ?facility - facility ?attachment_type_b - attachment_type_b)
    (equipment_module_available ?equipment_module - equipment_module)
    (facility_has_equipment_module ?facility - facility ?equipment_module - equipment_module)
    (catalyst_available ?catalyst - catalyst)
    (facility_has_catalyst ?facility - facility ?catalyst - catalyst)
    (consumable_available ?consumable - consumable)
    (unit_reserved_consumable ?unit - unit ?consumable - consumable)
    (frontline_unit_staged ?frontline_unit - frontline_unit)
    (support_unit_staged ?support_unit - support_unit)
    (facility_ready ?facility - facility)
  )
  (:action nominate_unit
    :parameters (?unit - unit)
    :precondition
      (and
        (not
          (nominated_unit ?unit)
        )
        (not
          (reservation_finalized ?unit)
        )
      )
    :effect (nominated_unit ?unit)
  )
  (:action assign_supply_template
    :parameters (?unit - unit ?supply_template - supply_template)
    :precondition
      (and
        (nominated_unit ?unit)
        (not
          (template_assigned ?unit)
        )
        (template_available ?supply_template)
      )
    :effect
      (and
        (template_assigned ?unit)
        (assigned_template ?unit ?supply_template)
        (not
          (template_available ?supply_template)
        )
      )
  )
  (:action allocate_healing_method
    :parameters (?unit - unit ?healing_method - healing_method)
    :precondition
      (and
        (nominated_unit ?unit)
        (template_assigned ?unit)
        (method_available ?healing_method)
      )
    :effect
      (and
        (assigned_method ?unit ?healing_method)
        (not
          (method_available ?healing_method)
        )
      )
  )
  (:action confirm_healing_method
    :parameters (?unit - unit ?healing_method - healing_method)
    :precondition
      (and
        (nominated_unit ?unit)
        (template_assigned ?unit)
        (assigned_method ?unit ?healing_method)
        (not
          (ready_for_preparation ?unit)
        )
      )
    :effect (ready_for_preparation ?unit)
  )
  (:action release_healing_method
    :parameters (?unit - unit ?healing_method - healing_method)
    :precondition
      (and
        (assigned_method ?unit ?healing_method)
      )
    :effect
      (and
        (method_available ?healing_method)
        (not
          (assigned_method ?unit ?healing_method)
        )
      )
  )
  (:action assign_tool_to_unit
    :parameters (?unit - unit ?tool - tool)
    :precondition
      (and
        (ready_for_preparation ?unit)
        (tool_available ?tool)
      )
    :effect
      (and
        (assigned_tool ?unit ?tool)
        (not
          (tool_available ?tool)
        )
      )
  )
  (:action release_tool_from_unit
    :parameters (?unit - unit ?tool - tool)
    :precondition
      (and
        (assigned_tool ?unit ?tool)
      )
    :effect
      (and
        (tool_available ?tool)
        (not
          (assigned_tool ?unit ?tool)
        )
      )
  )
  (:action attach_equipment_module
    :parameters (?facility - facility ?equipment_module - equipment_module)
    :precondition
      (and
        (ready_for_preparation ?facility)
        (equipment_module_available ?equipment_module)
      )
    :effect
      (and
        (facility_has_equipment_module ?facility ?equipment_module)
        (not
          (equipment_module_available ?equipment_module)
        )
      )
  )
  (:action detach_equipment_module
    :parameters (?facility - facility ?equipment_module - equipment_module)
    :precondition
      (and
        (facility_has_equipment_module ?facility ?equipment_module)
      )
    :effect
      (and
        (equipment_module_available ?equipment_module)
        (not
          (facility_has_equipment_module ?facility ?equipment_module)
        )
      )
  )
  (:action attach_catalyst
    :parameters (?facility - facility ?catalyst - catalyst)
    :precondition
      (and
        (ready_for_preparation ?facility)
        (catalyst_available ?catalyst)
      )
    :effect
      (and
        (facility_has_catalyst ?facility ?catalyst)
        (not
          (catalyst_available ?catalyst)
        )
      )
  )
  (:action detach_catalyst
    :parameters (?facility - facility ?catalyst - catalyst)
    :precondition
      (and
        (facility_has_catalyst ?facility ?catalyst)
      )
    :effect
      (and
        (catalyst_available ?catalyst)
        (not
          (facility_has_catalyst ?facility ?catalyst)
        )
      )
  )
  (:action stage_component_a
    :parameters (?frontline_unit - frontline_unit ?component_type_a - component_type_a ?healing_method - healing_method)
    :precondition
      (and
        (ready_for_preparation ?frontline_unit)
        (assigned_method ?frontline_unit ?healing_method)
        (unit_component_a_compat ?frontline_unit ?component_type_a)
        (not
          (component_a_staged ?component_type_a)
        )
        (not
          (component_a_material_attached ?component_type_a)
        )
      )
    :effect (component_a_staged ?component_type_a)
  )
  (:action stage_component_a_with_tool
    :parameters (?frontline_unit - frontline_unit ?component_type_a - component_type_a ?tool - tool)
    :precondition
      (and
        (ready_for_preparation ?frontline_unit)
        (assigned_tool ?frontline_unit ?tool)
        (unit_component_a_compat ?frontline_unit ?component_type_a)
        (component_a_staged ?component_type_a)
        (not
          (frontline_unit_staged ?frontline_unit)
        )
      )
    :effect
      (and
        (frontline_unit_staged ?frontline_unit)
        (frontline_unit_component_ready ?frontline_unit)
      )
  )
  (:action attach_material_to_component_a
    :parameters (?frontline_unit - frontline_unit ?component_type_a - component_type_a ?raw_material - raw_material)
    :precondition
      (and
        (ready_for_preparation ?frontline_unit)
        (unit_component_a_compat ?frontline_unit ?component_type_a)
        (material_available ?raw_material)
        (not
          (frontline_unit_staged ?frontline_unit)
        )
      )
    :effect
      (and
        (component_a_material_attached ?component_type_a)
        (frontline_unit_staged ?frontline_unit)
        (staged_material_frontline ?frontline_unit ?raw_material)
        (not
          (material_available ?raw_material)
        )
      )
  )
  (:action finalize_component_a_staging
    :parameters (?frontline_unit - frontline_unit ?component_type_a - component_type_a ?healing_method - healing_method ?raw_material - raw_material)
    :precondition
      (and
        (ready_for_preparation ?frontline_unit)
        (assigned_method ?frontline_unit ?healing_method)
        (unit_component_a_compat ?frontline_unit ?component_type_a)
        (component_a_material_attached ?component_type_a)
        (staged_material_frontline ?frontline_unit ?raw_material)
        (not
          (frontline_unit_component_ready ?frontline_unit)
        )
      )
    :effect
      (and
        (component_a_staged ?component_type_a)
        (frontline_unit_component_ready ?frontline_unit)
        (material_available ?raw_material)
        (not
          (staged_material_frontline ?frontline_unit ?raw_material)
        )
      )
  )
  (:action stage_component_b
    :parameters (?support_unit - support_unit ?component_type_b - component_type_b ?healing_method - healing_method)
    :precondition
      (and
        (ready_for_preparation ?support_unit)
        (assigned_method ?support_unit ?healing_method)
        (unit_component_b_compat ?support_unit ?component_type_b)
        (not
          (component_b_staged ?component_type_b)
        )
        (not
          (component_b_material_attached ?component_type_b)
        )
      )
    :effect (component_b_staged ?component_type_b)
  )
  (:action stage_component_b_with_tool
    :parameters (?support_unit - support_unit ?component_type_b - component_type_b ?tool - tool)
    :precondition
      (and
        (ready_for_preparation ?support_unit)
        (assigned_tool ?support_unit ?tool)
        (unit_component_b_compat ?support_unit ?component_type_b)
        (component_b_staged ?component_type_b)
        (not
          (support_unit_staged ?support_unit)
        )
      )
    :effect
      (and
        (support_unit_staged ?support_unit)
        (support_unit_component_ready ?support_unit)
      )
  )
  (:action attach_material_to_component_b
    :parameters (?support_unit - support_unit ?component_type_b - component_type_b ?raw_material - raw_material)
    :precondition
      (and
        (ready_for_preparation ?support_unit)
        (unit_component_b_compat ?support_unit ?component_type_b)
        (material_available ?raw_material)
        (not
          (support_unit_staged ?support_unit)
        )
      )
    :effect
      (and
        (component_b_material_attached ?component_type_b)
        (support_unit_staged ?support_unit)
        (staged_material_support ?support_unit ?raw_material)
        (not
          (material_available ?raw_material)
        )
      )
  )
  (:action finalize_component_b_staging
    :parameters (?support_unit - support_unit ?component_type_b - component_type_b ?healing_method - healing_method ?raw_material - raw_material)
    :precondition
      (and
        (ready_for_preparation ?support_unit)
        (assigned_method ?support_unit ?healing_method)
        (unit_component_b_compat ?support_unit ?component_type_b)
        (component_b_material_attached ?component_type_b)
        (staged_material_support ?support_unit ?raw_material)
        (not
          (support_unit_component_ready ?support_unit)
        )
      )
    :effect
      (and
        (component_b_staged ?component_type_b)
        (support_unit_component_ready ?support_unit)
        (material_available ?raw_material)
        (not
          (staged_material_support ?support_unit ?raw_material)
        )
      )
  )
  (:action assemble_bundle_standard
    :parameters (?frontline_unit - frontline_unit ?support_unit - support_unit ?component_type_a - component_type_a ?component_type_b - component_type_b ?prepared_bundle - prepared_bundle)
    :precondition
      (and
        (frontline_unit_staged ?frontline_unit)
        (support_unit_staged ?support_unit)
        (unit_component_a_compat ?frontline_unit ?component_type_a)
        (unit_component_b_compat ?support_unit ?component_type_b)
        (component_a_staged ?component_type_a)
        (component_b_staged ?component_type_b)
        (frontline_unit_component_ready ?frontline_unit)
        (support_unit_component_ready ?support_unit)
        (bundle_slot_available ?prepared_bundle)
      )
    :effect
      (and
        (bundle_assembled ?prepared_bundle)
        (bundle_has_component_a ?prepared_bundle ?component_type_a)
        (bundle_has_component_b ?prepared_bundle ?component_type_b)
        (not
          (bundle_slot_available ?prepared_bundle)
        )
      )
  )
  (:action assemble_bundle_variant_a
    :parameters (?frontline_unit - frontline_unit ?support_unit - support_unit ?component_type_a - component_type_a ?component_type_b - component_type_b ?prepared_bundle - prepared_bundle)
    :precondition
      (and
        (frontline_unit_staged ?frontline_unit)
        (support_unit_staged ?support_unit)
        (unit_component_a_compat ?frontline_unit ?component_type_a)
        (unit_component_b_compat ?support_unit ?component_type_b)
        (component_a_material_attached ?component_type_a)
        (component_b_staged ?component_type_b)
        (not
          (frontline_unit_component_ready ?frontline_unit)
        )
        (support_unit_component_ready ?support_unit)
        (bundle_slot_available ?prepared_bundle)
      )
    :effect
      (and
        (bundle_assembled ?prepared_bundle)
        (bundle_has_component_a ?prepared_bundle ?component_type_a)
        (bundle_has_component_b ?prepared_bundle ?component_type_b)
        (bundle_flag_a ?prepared_bundle)
        (not
          (bundle_slot_available ?prepared_bundle)
        )
      )
  )
  (:action assemble_bundle_variant_b
    :parameters (?frontline_unit - frontline_unit ?support_unit - support_unit ?component_type_a - component_type_a ?component_type_b - component_type_b ?prepared_bundle - prepared_bundle)
    :precondition
      (and
        (frontline_unit_staged ?frontline_unit)
        (support_unit_staged ?support_unit)
        (unit_component_a_compat ?frontline_unit ?component_type_a)
        (unit_component_b_compat ?support_unit ?component_type_b)
        (component_a_staged ?component_type_a)
        (component_b_material_attached ?component_type_b)
        (frontline_unit_component_ready ?frontline_unit)
        (not
          (support_unit_component_ready ?support_unit)
        )
        (bundle_slot_available ?prepared_bundle)
      )
    :effect
      (and
        (bundle_assembled ?prepared_bundle)
        (bundle_has_component_a ?prepared_bundle ?component_type_a)
        (bundle_has_component_b ?prepared_bundle ?component_type_b)
        (bundle_flag_b ?prepared_bundle)
        (not
          (bundle_slot_available ?prepared_bundle)
        )
      )
  )
  (:action assemble_bundle_variant_ab
    :parameters (?frontline_unit - frontline_unit ?support_unit - support_unit ?component_type_a - component_type_a ?component_type_b - component_type_b ?prepared_bundle - prepared_bundle)
    :precondition
      (and
        (frontline_unit_staged ?frontline_unit)
        (support_unit_staged ?support_unit)
        (unit_component_a_compat ?frontline_unit ?component_type_a)
        (unit_component_b_compat ?support_unit ?component_type_b)
        (component_a_material_attached ?component_type_a)
        (component_b_material_attached ?component_type_b)
        (not
          (frontline_unit_component_ready ?frontline_unit)
        )
        (not
          (support_unit_component_ready ?support_unit)
        )
        (bundle_slot_available ?prepared_bundle)
      )
    :effect
      (and
        (bundle_assembled ?prepared_bundle)
        (bundle_has_component_a ?prepared_bundle ?component_type_a)
        (bundle_has_component_b ?prepared_bundle ?component_type_b)
        (bundle_flag_a ?prepared_bundle)
        (bundle_flag_b ?prepared_bundle)
        (not
          (bundle_slot_available ?prepared_bundle)
        )
      )
  )
  (:action activate_bundle
    :parameters (?prepared_bundle - prepared_bundle ?frontline_unit - frontline_unit ?healing_method - healing_method)
    :precondition
      (and
        (bundle_assembled ?prepared_bundle)
        (frontline_unit_staged ?frontline_unit)
        (assigned_method ?frontline_unit ?healing_method)
        (not
          (bundle_activated ?prepared_bundle)
        )
      )
    :effect (bundle_activated ?prepared_bundle)
  )
  (:action attach_consumable_pack_to_bundle
    :parameters (?facility - facility ?consumable_pack - consumable_pack ?prepared_bundle - prepared_bundle)
    :precondition
      (and
        (ready_for_preparation ?facility)
        (facility_has_bundle ?facility ?prepared_bundle)
        (facility_has_consumable_pack ?facility ?consumable_pack)
        (consumable_pack_available ?consumable_pack)
        (bundle_assembled ?prepared_bundle)
        (bundle_activated ?prepared_bundle)
        (not
          (consumable_pack_engaged ?consumable_pack)
        )
      )
    :effect
      (and
        (consumable_pack_engaged ?consumable_pack)
        (pack_attached_to_bundle ?consumable_pack ?prepared_bundle)
        (not
          (consumable_pack_available ?consumable_pack)
        )
      )
  )
  (:action configure_consumable_pack_for_facility
    :parameters (?facility - facility ?consumable_pack - consumable_pack ?prepared_bundle - prepared_bundle ?healing_method - healing_method)
    :precondition
      (and
        (ready_for_preparation ?facility)
        (facility_has_consumable_pack ?facility ?consumable_pack)
        (consumable_pack_engaged ?consumable_pack)
        (pack_attached_to_bundle ?consumable_pack ?prepared_bundle)
        (assigned_method ?facility ?healing_method)
        (not
          (bundle_flag_a ?prepared_bundle)
        )
        (not
          (facility_pack_configured ?facility)
        )
      )
    :effect (facility_pack_configured ?facility)
  )
  (:action attach_attachment_type_a
    :parameters (?facility - facility ?attachment_type_a - attachment_type_a)
    :precondition
      (and
        (ready_for_preparation ?facility)
        (attachment_type_a_available ?attachment_type_a)
        (not
          (facility_has_primary_attachment ?facility)
        )
      )
    :effect
      (and
        (facility_has_primary_attachment ?facility)
        (facility_has_attachment_type_a ?facility ?attachment_type_a)
        (not
          (attachment_type_a_available ?attachment_type_a)
        )
      )
  )
  (:action integrate_pack_and_attachment
    :parameters (?facility - facility ?consumable_pack - consumable_pack ?prepared_bundle - prepared_bundle ?healing_method - healing_method ?attachment_type_a - attachment_type_a)
    :precondition
      (and
        (ready_for_preparation ?facility)
        (facility_has_consumable_pack ?facility ?consumable_pack)
        (consumable_pack_engaged ?consumable_pack)
        (pack_attached_to_bundle ?consumable_pack ?prepared_bundle)
        (assigned_method ?facility ?healing_method)
        (bundle_flag_a ?prepared_bundle)
        (facility_has_primary_attachment ?facility)
        (facility_has_attachment_type_a ?facility ?attachment_type_a)
        (not
          (facility_pack_configured ?facility)
        )
      )
    :effect
      (and
        (facility_pack_configured ?facility)
        (facility_secondary_attachment_configured ?facility)
      )
  )
  (:action install_module_with_tool
    :parameters (?facility - facility ?equipment_module - equipment_module ?tool - tool ?consumable_pack - consumable_pack ?prepared_bundle - prepared_bundle)
    :precondition
      (and
        (facility_pack_configured ?facility)
        (facility_has_equipment_module ?facility ?equipment_module)
        (assigned_tool ?facility ?tool)
        (facility_has_consumable_pack ?facility ?consumable_pack)
        (pack_attached_to_bundle ?consumable_pack ?prepared_bundle)
        (not
          (bundle_flag_b ?prepared_bundle)
        )
        (not
          (facility_module_attached ?facility)
        )
      )
    :effect (facility_module_attached ?facility)
  )
  (:action install_module_with_tool_variant
    :parameters (?facility - facility ?equipment_module - equipment_module ?tool - tool ?consumable_pack - consumable_pack ?prepared_bundle - prepared_bundle)
    :precondition
      (and
        (facility_pack_configured ?facility)
        (facility_has_equipment_module ?facility ?equipment_module)
        (assigned_tool ?facility ?tool)
        (facility_has_consumable_pack ?facility ?consumable_pack)
        (pack_attached_to_bundle ?consumable_pack ?prepared_bundle)
        (bundle_flag_b ?prepared_bundle)
        (not
          (facility_module_attached ?facility)
        )
      )
    :effect (facility_module_attached ?facility)
  )
  (:action apply_catalyst_stage1
    :parameters (?facility - facility ?catalyst - catalyst ?consumable_pack - consumable_pack ?prepared_bundle - prepared_bundle)
    :precondition
      (and
        (facility_module_attached ?facility)
        (facility_has_catalyst ?facility ?catalyst)
        (facility_has_consumable_pack ?facility ?consumable_pack)
        (pack_attached_to_bundle ?consumable_pack ?prepared_bundle)
        (not
          (bundle_flag_a ?prepared_bundle)
        )
        (not
          (bundle_flag_b ?prepared_bundle)
        )
        (not
          (facility_configured_for_activation ?facility)
        )
      )
    :effect (facility_configured_for_activation ?facility)
  )
  (:action apply_catalyst_stage1_variant
    :parameters (?facility - facility ?catalyst - catalyst ?consumable_pack - consumable_pack ?prepared_bundle - prepared_bundle)
    :precondition
      (and
        (facility_module_attached ?facility)
        (facility_has_catalyst ?facility ?catalyst)
        (facility_has_consumable_pack ?facility ?consumable_pack)
