    (source_location_linked ?customer_order - customer_order ?source_location - source_location)

    (release_window_linked ?customer_order - customer_order ?release_window - release_window)

    (destination_item_linked ?destination_fulfillment_unit - destination_fulfillment_unit ?fulfillment_item - fulfillment_item)

    (source_fulfillment_unit_ready ?source_fulfillment_unit - source_fulfillment_unit)

    (destination_segment_open ?destination_shipment_segment - destination_shipment_segment)

    (shipment_destination_linked ?shipment - shipment ?destination_fulfillment_unit - destination_fulfillment_unit)
    (shipment_transport_unit_linked ?shipment - shipment ?transport_unit - transport_unit)

    (shipment_configuration_complete ?shipment - shipment)
    (temperature_compliant ?shipment - shipment)
    (routing_option_selected ?shipment - shipment)
    (shipment_plan_validated ?shipment - shipment)
    (stowage_planning_open ?shipment - shipment)
    (shipment_manifest_assembled ?shipment - shipment)

    (carrier_preference_selected ?shipment - shipment)
    (carrier_booking_prepared ?shipment - shipment)
    (carrier_manifest_ready ?shipment - shipment)
    (routing_option_available ?routing_option - routing_option)
    (routing_option_linked ?shipment - shipment ?routing_option - routing_option)
    (stowage_profile_available ?stowage_profile - stowage_profile)

    (equipment_requirement_available ?equipment_requirement - equipment_requirement)
    (equipment_requirement_linked ?shipment - shipment ?equipment_requirement - equipment_requirement)

    (temperature_profile_linked ?shipment - shipment ?temperature_profile - temperature_profile)
    (service_level_profile_available ?service_level_profile - service_level_profile)
    (service_level_linked ?customer_order - customer_order ?service_level_profile - service_level_profile)

    (destination_fulfillment_unit_activated ?destination_fulfillment_unit - destination_fulfillment_unit)
    (dispatch_cleared ?shipment - shipment)
  )
  (:action register_customer_order
    :parameters (?customer_order - customer_order)
    :precondition
      (and
        (not
          (order_intake_logged ?customer_order)
        )
        (not
          (outbound_closed ?customer_order)
        )
      )
    :effect (order_intake_logged ?customer_order)
  )
  (:action assign_source_location
    :parameters (?customer_order - customer_order ?source_location - source_location)
    :precondition
      (and
        (order_intake_logged ?customer_order)
        (not
          (source_location_selected ?customer_order)
        )
        (source_location_available ?source_location)
      )
    :effect
      (and
        (source_location_selected ?customer_order)
        (source_location_linked ?customer_order ?source_location)
        (not
          (source_location_available ?source_location)
        )
      )
  )
  (:action attach_release_window
    :parameters (?customer_order - customer_order ?release_window - release_window)
    :precondition
      (and
        (order_intake_logged ?customer_order)
        (source_location_selected ?customer_order)
        (release_window_available ?release_window)
      )
    :effect
      (and
        (release_window_linked ?customer_order ?release_window)
        (not
          (release_window_available ?release_window)
        )
      )
  )
  (:action authorize_processing_release
    :parameters (?customer_order - customer_order ?release_window - release_window)
    :precondition
      (and
        (order_intake_logged ?customer_order)
        (source_location_selected ?customer_order)
        (release_window_linked ?customer_order ?release_window)
        (not
          (processing_release_approved ?customer_order)
        )
      )
    :effect (processing_release_approved ?customer_order)
  )
  (:action reopen_release_window
    :parameters (?customer_order - customer_order ?release_window - release_window)
    :precondition
      (and
        (release_window_linked ?customer_order ?release_window)
      )
    :effect
      (and
        (release_window_available ?release_window)
        (not
          (release_window_linked ?customer_order ?release_window)
        )
      )
  )
  (:action allocate_handling_resource
    :parameters (?customer_order - customer_order ?handling_resource - handling_resource)
    :precondition
      (and
        (processing_release_approved ?customer_order)
        (handling_resource_available ?handling_resource)
      )
    :effect
      (and
        (handling_resource_linked ?customer_order ?handling_resource)
        (not
          (handling_resource_available ?handling_resource)
        )
      )
  )
  (:action free_handling_resource
    :parameters (?customer_order - customer_order ?handling_resource - handling_resource)
    :precondition
      (and
        (handling_resource_linked ?customer_order ?handling_resource)
      )
    :effect
      (and
        (handling_resource_available ?handling_resource)
        (not
          (handling_resource_linked ?customer_order ?handling_resource)
        )
      )
  )
  (:action attach_equipment_requirement
    :parameters (?shipment - shipment ?equipment_requirement - equipment_requirement)
    :precondition
      (and
        (processing_release_approved ?shipment)
        (equipment_requirement_available ?equipment_requirement)
      )
    :effect
      (and
        (equipment_requirement_linked ?shipment ?equipment_requirement)
        (not
          (equipment_requirement_available ?equipment_requirement)
        )
      )
  )
  (:action detach_equipment_requirement
    :parameters (?shipment - shipment ?equipment_requirement - equipment_requirement)
    :precondition
      (and
        (equipment_requirement_linked ?shipment ?equipment_requirement)
      )
    :effect
      (and
        (equipment_requirement_available ?equipment_requirement)
        (not
          (equipment_requirement_linked ?shipment ?equipment_requirement)
        )
      )
  )
  (:action attach_temperature_profile
    :parameters (?shipment - shipment ?temperature_profile - temperature_profile)
    :precondition
      (and
        (processing_release_approved ?shipment)
        (temperature_profile_available ?temperature_profile)
      )
    :effect
      (and
        (temperature_profile_linked ?shipment ?temperature_profile)
        (not
          (temperature_profile_available ?temperature_profile)
        )
      )
  )
  (:action detach_temperature_profile
    :parameters (?shipment - shipment ?temperature_profile - temperature_profile)
    :precondition
      (and
        (temperature_profile_linked ?shipment ?temperature_profile)
      )
    :effect
      (and
        (temperature_profile_available ?temperature_profile)
        (not
          (temperature_profile_linked ?shipment ?temperature_profile)
        )
      )
  )
  (:action open_source_segment
    :parameters (?source_fulfillment_unit - source_fulfillment_unit ?source_shipment_segment - source_shipment_segment ?release_window - release_window)
    :precondition
      (and
        (processing_release_approved ?source_fulfillment_unit)
        (release_window_linked ?source_fulfillment_unit ?release_window)
        (source_segment_linked ?source_fulfillment_unit ?source_shipment_segment)
        (not
          (source_segment_open ?source_shipment_segment)
        )
        (not
          (source_segment_reserved ?source_shipment_segment)
        )
      )
    :effect (source_segment_open ?source_shipment_segment)
  )
  (:action activate_source_unit
    :parameters (?source_fulfillment_unit - source_fulfillment_unit ?source_shipment_segment - source_shipment_segment ?handling_resource - handling_resource)
    :precondition
      (and
        (processing_release_approved ?source_fulfillment_unit)
        (handling_resource_linked ?source_fulfillment_unit ?handling_resource)
        (source_segment_linked ?source_fulfillment_unit ?source_shipment_segment)
        (source_segment_open ?source_shipment_segment)
        (not
          (source_fulfillment_unit_activated ?source_fulfillment_unit)
        )
      )
    :effect
      (and
        (source_fulfillment_unit_activated ?source_fulfillment_unit)
        (source_fulfillment_unit_ready ?source_fulfillment_unit)
      )
  )
  (:action reserve_source_item
    :parameters (?source_fulfillment_unit - source_fulfillment_unit ?source_shipment_segment - source_shipment_segment ?fulfillment_item - fulfillment_item)
    :precondition
      (and
        (processing_release_approved ?source_fulfillment_unit)
        (source_segment_linked ?source_fulfillment_unit ?source_shipment_segment)
        (fulfillment_item_available ?fulfillment_item)
        (not
          (source_fulfillment_unit_activated ?source_fulfillment_unit)
        )
      )
    :effect
      (and
        (source_segment_reserved ?source_shipment_segment)
        (source_fulfillment_unit_activated ?source_fulfillment_unit)
        (source_item_linked ?source_fulfillment_unit ?fulfillment_item)
        (not
          (fulfillment_item_available ?fulfillment_item)
        )
      )
  )
  (:action finalize_source_segment
    :parameters (?source_fulfillment_unit - source_fulfillment_unit ?source_shipment_segment - source_shipment_segment ?release_window - release_window ?fulfillment_item - fulfillment_item)
    :precondition
      (and
        (processing_release_approved ?source_fulfillment_unit)
        (release_window_linked ?source_fulfillment_unit ?release_window)
        (source_segment_linked ?source_fulfillment_unit ?source_shipment_segment)
        (source_segment_reserved ?source_shipment_segment)
        (source_item_linked ?source_fulfillment_unit ?fulfillment_item)
        (not
          (source_fulfillment_unit_ready ?source_fulfillment_unit)
        )
      )
    :effect
      (and
        (source_segment_open ?source_shipment_segment)
        (source_fulfillment_unit_ready ?source_fulfillment_unit)
        (fulfillment_item_available ?fulfillment_item)
        (not
          (source_item_linked ?source_fulfillment_unit ?fulfillment_item)
        )
      )
  )
  (:action open_destination_segment
    :parameters (?destination_fulfillment_unit - destination_fulfillment_unit ?destination_shipment_segment - destination_shipment_segment ?release_window - release_window)

  (:action reserve_destination_item
    :parameters (?destination_fulfillment_unit - destination_fulfillment_unit ?destination_shipment_segment - destination_shipment_segment ?fulfillment_item - fulfillment_item)
    :precondition
      (and
        (processing_release_approved ?destination_fulfillment_unit)
        (destination_segment_linked ?destination_fulfillment_unit ?destination_shipment_segment)
        (fulfillment_item_available ?fulfillment_item)
        (not
          (destination_fulfillment_unit_activated ?destination_fulfillment_unit)
        )
      )
    :effect
      (and
        (destination_segment_reserved ?destination_shipment_segment)
        (destination_fulfillment_unit_activated ?destination_fulfillment_unit)

          (fulfillment_item_available ?fulfillment_item)

      )
  )
  (:action finalize_destination_segment
    :parameters (?destination_fulfillment_unit - destination_fulfillment_unit ?destination_shipment_segment - destination_shipment_segment ?release_window - release_window ?fulfillment_item - fulfillment_item)
    :precondition
      (and
        (processing_release_approved ?destination_fulfillment_unit)
        (release_window_linked ?destination_fulfillment_unit ?release_window)
        (destination_segment_linked ?destination_fulfillment_unit ?destination_shipment_segment)
        (destination_segment_reserved ?destination_shipment_segment)
        (destination_item_linked ?destination_fulfillment_unit ?fulfillment_item)
        (not
          (destination_fulfillment_unit_ready ?destination_fulfillment_unit)
        )
      )
    :effect
      (and
        (destination_segment_open ?destination_shipment_segment)
        (destination_fulfillment_unit_ready ?destination_fulfillment_unit)
        (fulfillment_item_available ?fulfillment_item)
        (not
          (destination_item_linked ?destination_fulfillment_unit ?fulfillment_item)
        )
      )
  )
  (:action reserve_transport_unit_for_segments
    :parameters (?source_fulfillment_unit - source_fulfillment_unit ?destination_fulfillment_unit - destination_fulfillment_unit ?source_shipment_segment - source_shipment_segment ?destination_shipment_segment - destination_shipment_segment ?transport_unit - transport_unit)
    :precondition
      (and
        (source_fulfillment_unit_activated ?source_fulfillment_unit)
        (destination_fulfillment_unit_activated ?destination_fulfillment_unit)
        (source_segment_linked ?source_fulfillment_unit ?source_shipment_segment)
        (destination_segment_linked ?destination_fulfillment_unit ?destination_shipment_segment)
        (source_segment_open ?source_shipment_segment)
        (destination_segment_open ?destination_shipment_segment)
        (source_fulfillment_unit_ready ?source_fulfillment_unit)
        (destination_fulfillment_unit_ready ?destination_fulfillment_unit)
        (transport_unit_available ?transport_unit)
      )
    :effect
      (and
        (transport_unit_reserved ?transport_unit)
        (transport_source_segment_linked ?transport_unit ?source_shipment_segment)
        (transport_destination_segment_linked ?transport_unit ?destination_shipment_segment)
        (not
          (transport_unit_available ?transport_unit)
        )
      )
  )
  (:action reserve_transport_unit_source_ready
    :parameters (?source_fulfillment_unit - source_fulfillment_unit ?destination_fulfillment_unit - destination_fulfillment_unit ?source_shipment_segment - source_shipment_segment ?destination_shipment_segment - destination_shipment_segment ?transport_unit - transport_unit)
    :precondition
      (and
        (source_fulfillment_unit_activated ?source_fulfillment_unit)
        (destination_fulfillment_unit_activated ?destination_fulfillment_unit)
        (source_segment_linked ?source_fulfillment_unit ?source_shipment_segment)
        (destination_segment_linked ?destination_fulfillment_unit ?destination_shipment_segment)
        (source_segment_reserved ?source_shipment_segment)
        (destination_segment_open ?destination_shipment_segment)
        (not
          (source_fulfillment_unit_ready ?source_fulfillment_unit)
        )
        (destination_fulfillment_unit_ready ?destination_fulfillment_unit)
        (transport_unit_available ?transport_unit)
      )
    :effect
      (and
        (transport_unit_reserved ?transport_unit)
        (transport_source_segment_linked ?transport_unit ?source_shipment_segment)
        (transport_destination_segment_linked ?transport_unit ?destination_shipment_segment)
        (transport_source_leg_ready ?transport_unit)
        (not
          (transport_unit_available ?transport_unit)
        )
      )
  )
  (:action reserve_transport_unit_destination_ready
    :parameters (?source_fulfillment_unit - source_fulfillment_unit ?destination_fulfillment_unit - destination_fulfillment_unit ?source_shipment_segment - source_shipment_segment ?destination_shipment_segment - destination_shipment_segment ?transport_unit - transport_unit)
    :precondition
      (and
        (source_fulfillment_unit_activated ?source_fulfillment_unit)
        (destination_fulfillment_unit_activated ?destination_fulfillment_unit)
        (source_segment_linked ?source_fulfillment_unit ?source_shipment_segment)
        (destination_segment_linked ?destination_fulfillment_unit ?destination_shipment_segment)
        (source_segment_open ?source_shipment_segment)
        (destination_segment_reserved ?destination_shipment_segment)
        (source_fulfillment_unit_ready ?source_fulfillment_unit)
        (not
          (destination_fulfillment_unit_ready ?destination_fulfillment_unit)
        )
        (transport_unit_available ?transport_unit)
      )
    :effect
      (and
        (transport_unit_reserved ?transport_unit)
        (transport_source_segment_linked ?transport_unit ?source_shipment_segment)
        (transport_destination_segment_linked ?transport_unit ?destination_shipment_segment)
        (transport_destination_leg_ready ?transport_unit)
        (not
          (transport_unit_available ?transport_unit)
        )
      )
  )
  (:action reserve_transport_unit_bidirectional_ready
    :parameters (?source_fulfillment_unit - source_fulfillment_unit ?destination_fulfillment_unit - destination_fulfillment_unit ?source_shipment_segment - source_shipment_segment ?destination_shipment_segment - destination_shipment_segment ?transport_unit - transport_unit)
    :precondition
      (and
        (source_fulfillment_unit_activated ?source_fulfillment_unit)
        (destination_fulfillment_unit_activated ?destination_fulfillment_unit)
        (source_segment_linked ?source_fulfillment_unit ?source_shipment_segment)

          (source_fulfillment_unit_ready ?source_fulfillment_unit)

        (not
          (destination_fulfillment_unit_ready ?destination_fulfillment_unit)

      (and
        (transport_unit_reserved ?transport_unit)

        (transport_destination_segment_linked ?transport_unit ?destination_shipment_segment)
        (transport_source_leg_ready ?transport_unit)
        (transport_destination_leg_ready ?transport_unit)
        (not
          (transport_unit_available ?transport_unit)
        )
      )
  )
  (:action stage_transport_unit
    :parameters (?transport_unit - transport_unit ?source_fulfillment_unit - source_fulfillment_unit ?release_window - release_window)
    :precondition
      (and
        (transport_unit_reserved ?transport_unit)
        (source_fulfillment_unit_activated ?source_fulfillment_unit)
        (release_window_linked ?source_fulfillment_unit ?release_window)
        (not
          (transport_unit_staged ?transport_unit)
        )
      )
    :effect (transport_unit_staged ?transport_unit)
  )
  (:action reserve_pack_unit
    :parameters (?shipment - shipment ?pack_unit - pack_unit ?transport_unit - transport_unit)
    :precondition
      (and
        (processing_release_approved ?shipment)
        (shipment_transport_unit_linked ?shipment ?transport_unit)
        (shipment_pack_unit_linked ?shipment ?pack_unit)
        (pack_unit_available ?pack_unit)
        (transport_unit_reserved ?transport_unit)
        (transport_unit_staged ?transport_unit)

          (pack_unit_reserved ?pack_unit)
        )
      )
    :effect
      (and
        (pack_unit_reserved ?pack_unit)
        (pack_unit_transport_linked ?pack_unit ?transport_unit)
        (not
          (pack_unit_available ?pack_unit)
        )
      )
  )
  (:action package_shipment
    :parameters (?shipment - shipment ?pack_unit - pack_unit ?transport_unit - transport_unit ?release_window - release_window)
    :precondition
      (and
        (processing_release_approved ?shipment)
        (shipment_pack_unit_linked ?shipment ?pack_unit)
        (pack_unit_reserved ?pack_unit)
        (pack_unit_transport_linked ?pack_unit ?transport_unit)
        (release_window_linked ?shipment ?release_window)
        (not
          (transport_source_leg_ready ?transport_unit)
        )
        (not
          (shipment_packaged ?shipment)
        )
      )
    :effect (shipment_packaged ?shipment)
  )
  (:action select_routing_option
    :parameters (?shipment - shipment ?routing_option - routing_option)
    :precondition
      (and
        (processing_release_approved ?shipment)
        (routing_option_available ?routing_option)
        (not
          (routing_option_selected ?shipment)
        )
      )
    :effect
      (and
        (routing_option_selected ?shipment)
        (routing_option_linked ?shipment ?routing_option)
        (not
          (routing_option_available ?routing_option)
        )
      )
  )
  (:action validate_shipment_plan
    :parameters (?shipment - shipment ?pack_unit - pack_unit ?transport_unit - transport_unit ?release_window - release_window ?routing_option - routing_option)
    :precondition
      (and
        (processing_release_approved ?shipment)
        (shipment_pack_unit_linked ?shipment ?pack_unit)
        (pack_unit_reserved ?pack_unit)
        (pack_unit_transport_linked ?pack_unit ?transport_unit)
        (release_window_linked ?shipment ?release_window)
        (transport_source_leg_ready ?transport_unit)
        (routing_option_selected ?shipment)
        (routing_option_linked ?shipment ?routing_option)
        (not
          (shipment_packaged ?shipment)
        )
      )
    :effect
      (and
        (shipment_packaged ?shipment)
        (shipment_plan_validated ?shipment)
      )
  )
  (:action configure_shipment_equipment_without_destination_leg
    :parameters (?shipment - shipment ?equipment_requirement - equipment_requirement ?handling_resource - handling_resource ?pack_unit - pack_unit ?transport_unit - transport_unit)
    :precondition
      (and
