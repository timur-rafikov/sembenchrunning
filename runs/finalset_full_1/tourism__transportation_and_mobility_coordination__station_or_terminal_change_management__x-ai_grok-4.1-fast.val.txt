--- domain.pddl (as passed to VAL) ---
(define (domain station_transfer_coordination)
  (:requirements :strips :typing :negative-preconditions)
  (:types generic_object - object resource_category - generic_object support_resource_category - generic_object connector_category - generic_object traveler_entity - generic_object transfer_entity - traveler_entity local_vehicle_unit - resource_category transfer_option - resource_category staff_resource - resource_category priority_service_type - resource_category special_service_option - resource_category time_window - resource_category vehicle_capability - resource_category operational_constraint - resource_category baggage_or_mobility_aid - support_resource_category facility_resource - support_resource_category document_or_authorization - support_resource_category incoming_access_path - connector_category outgoing_access_path - connector_category connector_instance - connector_category segment_entity - transfer_entity case_entity - transfer_entity incoming_segment - segment_entity outgoing_segment - segment_entity transfer_case - case_entity)

  (:predicates
    (traveler_registered ?traveler - transfer_entity)
    (confirmed_for_handover ?traveler - transfer_entity)
    (vehicle_reserved_flag ?traveler - transfer_entity)
    (handover_completed ?traveler - transfer_entity)
    (ready_for_handover ?traveler - transfer_entity)
    (time_window_assigned ?traveler - transfer_entity)
    (vehicle_available ?local_vehicle - local_vehicle_unit)
    (assigned_vehicle ?traveler - transfer_entity ?local_vehicle - local_vehicle_unit)
    (transfer_option_available ?transfer_option - transfer_option)
    (transfer_option_selected ?traveler - transfer_entity ?transfer_option - transfer_option)
    (staff_available ?staff_resource - staff_resource)
    (staff_assigned ?traveler - transfer_entity ?staff_resource - staff_resource)
    (baggage_available ?baggage_item - baggage_or_mobility_aid)
    (incoming_segment_baggage_assigned ?incoming_segment - incoming_segment ?baggage_item - baggage_or_mobility_aid)
    (outgoing_segment_baggage_assigned ?outgoing_segment - outgoing_segment ?baggage_item - baggage_or_mobility_aid)
    (incoming_segment_on_access_path ?incoming_segment - incoming_segment ?incoming_access_path - incoming_access_path)
    (incoming_access_path_ready ?incoming_access_path - incoming_access_path)
    (incoming_access_path_occupied ?incoming_access_path - incoming_access_path)
    (incoming_segment_ready ?incoming_segment - incoming_segment)
    (outgoing_segment_on_access_path ?outgoing_segment - outgoing_segment ?outgoing_access_path - outgoing_access_path)
    (outgoing_access_path_ready ?outgoing_access_path - outgoing_access_path)
    (outgoing_access_path_occupied ?outgoing_access_path - outgoing_access_path)
    (outgoing_segment_ready ?outgoing_segment - outgoing_segment)
    (connector_available ?connector - connector_instance)
    (connector_reserved ?connector - connector_instance)
    (connector_assigned_incoming ?connector - connector_instance ?incoming_access_path - incoming_access_path)
    (connector_assigned_outgoing ?connector - connector_instance ?outgoing_access_path - outgoing_access_path)
    (connector_incoming_configured ?connector - connector_instance)
    (connector_outgoing_configured ?connector - connector_instance)
    (connector_boarding_ready ?connector - connector_instance)
    (case_incoming_segment ?transfer_case - transfer_case ?incoming_segment - incoming_segment)
    (case_outgoing_segment ?transfer_case - transfer_case ?outgoing_segment - outgoing_segment)
    (case_assigned_connector ?transfer_case - transfer_case ?connector - connector_instance)
    (facility_available ?facility_resource - facility_resource)
    (case_requires_facility ?transfer_case - transfer_case ?facility_resource - facility_resource)
    (facility_allocated ?facility_resource - facility_resource)
    (facility_assigned_to_connector ?facility_resource - facility_resource ?connector - connector_instance)
    (case_facility_verified ?transfer_case - transfer_case)
    (case_resource_compatibility_verified ?transfer_case - transfer_case)
    (case_operationally_compliant ?transfer_case - transfer_case)
    (case_priority_selected ?transfer_case - transfer_case)
    (case_priority_acknowledged ?transfer_case - transfer_case)
    (case_ready_for_special_service ?transfer_case - transfer_case)
    (case_preparation_finalized ?transfer_case - transfer_case)
    (document_available ?document - document_or_authorization)
    (case_document_linked ?transfer_case - transfer_case ?document - document_or_authorization)
    (case_document_locked ?transfer_case - transfer_case)
    (case_document_verified ?transfer_case - transfer_case)
    (case_document_constraint_applied ?transfer_case - transfer_case)
    (priority_service_available ?priority_service - priority_service_type)
    (case_linked_priority_service ?transfer_case - transfer_case ?priority_service - priority_service_type)
    (special_service_available ?special_service - special_service_option)
    (case_special_service_assigned ?transfer_case - transfer_case ?special_service - special_service_option)
    (vehicle_capability_available ?vehicle_capability - vehicle_capability)
    (case_vehicle_capability_assigned ?transfer_case - transfer_case ?vehicle_capability - vehicle_capability)
    (operational_constraint_available ?operational_constraint - operational_constraint)
    (case_operational_constraint_assigned ?transfer_case - transfer_case ?operational_constraint - operational_constraint)
    (time_window_available ?time_window - time_window)
    (assigned_time_window ?traveler - transfer_entity ?time_window - time_window)
    (incoming_segment_flagged ?incoming_segment - incoming_segment)
    (outgoing_segment_flagged ?outgoing_segment - outgoing_segment)
    (case_validated ?transfer_case - transfer_case)
  )
  (:action register_traveler
    :parameters (?traveler - transfer_entity)
    :precondition
      (and
        (not
          (traveler_registered ?traveler)
        )
        (not
          (handover_completed ?traveler)
        )
      )
    :effect (traveler_registered ?traveler)
  )
  (:action reserve_local_vehicle
    :parameters (?traveler - transfer_entity ?local_vehicle - local_vehicle_unit)
    :precondition
      (and
        (traveler_registered ?traveler)
        (not
          (vehicle_reserved_flag ?traveler)
        )
        (vehicle_available ?local_vehicle)
      )
    :effect
      (and
        (vehicle_reserved_flag ?traveler)
        (assigned_vehicle ?traveler ?local_vehicle)
        (not
          (vehicle_available ?local_vehicle)
        )
      )
  )
  (:action select_transfer_option
    :parameters (?traveler - transfer_entity ?transfer_option - transfer_option)
    :precondition
      (and
        (traveler_registered ?traveler)
        (vehicle_reserved_flag ?traveler)
        (transfer_option_available ?transfer_option)
      )
    :effect
      (and
        (transfer_option_selected ?traveler ?transfer_option)
        (not
          (transfer_option_available ?transfer_option)
        )
      )
  )
  (:action confirm_transfer_option
    :parameters (?traveler - transfer_entity ?transfer_option - transfer_option)
    :precondition
      (and
        (traveler_registered ?traveler)
        (vehicle_reserved_flag ?traveler)
        (transfer_option_selected ?traveler ?transfer_option)
        (not
          (confirmed_for_handover ?traveler)
        )
      )
    :effect (confirmed_for_handover ?traveler)
  )
  (:action release_transfer_option
    :parameters (?traveler - transfer_entity ?transfer_option - transfer_option)
    :precondition
      (and
        (transfer_option_selected ?traveler ?transfer_option)
      )
    :effect
      (and
        (transfer_option_available ?transfer_option)
        (not
          (transfer_option_selected ?traveler ?transfer_option)
        )
      )
  )
  (:action assign_staff
    :parameters (?traveler - transfer_entity ?staff_resource - staff_resource)
    :precondition
      (and
        (confirmed_for_handover ?traveler)
        (staff_available ?staff_resource)
      )
    :effect
      (and
        (staff_assigned ?traveler ?staff_resource)
        (not
          (staff_available ?staff_resource)
        )
      )
  )
  (:action release_staff
    :parameters (?traveler - transfer_entity ?staff_resource - staff_resource)
    :precondition
      (and
        (staff_assigned ?traveler ?staff_resource)
      )
    :effect
      (and
        (staff_available ?staff_resource)
        (not
          (staff_assigned ?traveler ?staff_resource)
        )
      )
  )
  (:action assign_vehicle_capability
    :parameters (?transfer_case - transfer_case ?vehicle_capability - vehicle_capability)
    :precondition
      (and
        (confirmed_for_handover ?transfer_case)
        (vehicle_capability_available ?vehicle_capability)
      )
    :effect
      (and
        (case_vehicle_capability_assigned ?transfer_case ?vehicle_capability)
        (not
          (vehicle_capability_available ?vehicle_capability)
        )
      )
  )
  (:action release_vehicle_capability
    :parameters (?transfer_case - transfer_case ?vehicle_capability - vehicle_capability)
    :precondition
      (and
        (case_vehicle_capability_assigned ?transfer_case ?vehicle_capability)
      )
    :effect
      (and
        (vehicle_capability_available ?vehicle_capability)
        (not
          (case_vehicle_capability_assigned ?transfer_case ?vehicle_capability)
        )
      )
  )
  (:action assign_operational_constraint
    :parameters (?transfer_case - transfer_case ?operational_constraint - operational_constraint)
    :precondition
      (and
        (confirmed_for_handover ?transfer_case)
        (operational_constraint_available ?operational_constraint)
      )
    :effect
      (and
        (case_operational_constraint_assigned ?transfer_case ?operational_constraint)
        (not
          (operational_constraint_available ?operational_constraint)
        )
      )
  )
  (:action remove_operational_constraint
    :parameters (?transfer_case - transfer_case ?operational_constraint - operational_constraint)
    :precondition
      (and
        (case_operational_constraint_assigned ?transfer_case ?operational_constraint)
      )
    :effect
      (and
        (operational_constraint_available ?operational_constraint)
        (not
          (case_operational_constraint_assigned ?transfer_case ?operational_constraint)
        )
      )
  )
  (:action mark_incoming_access_path_ready
    :parameters (?incoming_segment - incoming_segment ?incoming_access_path - incoming_access_path ?transfer_option - transfer_option)
    :precondition
      (and
        (confirmed_for_handover ?incoming_segment)
        (transfer_option_selected ?incoming_segment ?transfer_option)
        (incoming_segment_on_access_path ?incoming_segment ?incoming_access_path)
        (not
          (incoming_access_path_ready ?incoming_access_path)
        )
        (not
          (incoming_access_path_occupied ?incoming_access_path)
        )
      )
    :effect (incoming_access_path_ready ?incoming_access_path)
  )
  (:action confirm_incoming_staff_assignment
    :parameters (?incoming_segment - incoming_segment ?incoming_access_path - incoming_access_path ?staff_resource - staff_resource)
    :precondition
      (and
        (confirmed_for_handover ?incoming_segment)
        (staff_assigned ?incoming_segment ?staff_resource)
        (incoming_segment_on_access_path ?incoming_segment ?incoming_access_path)
        (incoming_access_path_ready ?incoming_access_path)
        (not
          (incoming_segment_flagged ?incoming_segment)
        )
      )
    :effect
      (and
        (incoming_segment_flagged ?incoming_segment)
        (incoming_segment_ready ?incoming_segment)
      )
  )
  (:action assign_baggage_to_incoming
    :parameters (?incoming_segment - incoming_segment ?incoming_access_path - incoming_access_path ?baggage_item - baggage_or_mobility_aid)
    :precondition
      (and
        (confirmed_for_handover ?incoming_segment)
        (incoming_segment_on_access_path ?incoming_segment ?incoming_access_path)
        (baggage_available ?baggage_item)
        (not
          (incoming_segment_flagged ?incoming_segment)
        )
      )
    :effect
      (and
        (incoming_access_path_occupied ?incoming_access_path)
        (incoming_segment_flagged ?incoming_segment)
        (incoming_segment_baggage_assigned ?incoming_segment ?baggage_item)
        (not
          (baggage_available ?baggage_item)
        )
      )
  )
  (:action finalize_incoming_baggage_allocation
    :parameters (?incoming_segment - incoming_segment ?incoming_access_path - incoming_access_path ?transfer_option - transfer_option ?baggage_item - baggage_or_mobility_aid)
    :precondition
      (and
        (confirmed_for_handover ?incoming_segment)
        (transfer_option_selected ?incoming_segment ?transfer_option)
        (incoming_segment_on_access_path ?incoming_segment ?incoming_access_path)
        (incoming_access_path_occupied ?incoming_access_path)
        (incoming_segment_baggage_assigned ?incoming_segment ?baggage_item)
        (not
          (incoming_segment_ready ?incoming_segment)
        )
      )
    :effect
      (and
        (incoming_access_path_ready ?incoming_access_path)
        (incoming_segment_ready ?incoming_segment)
        (baggage_available ?baggage_item)
        (not
          (incoming_segment_baggage_assigned ?incoming_segment ?baggage_item)
        )
      )
  )
  (:action mark_outgoing_access_path_ready
    :parameters (?outgoing_segment - outgoing_segment ?outgoing_access_path - outgoing_access_path ?transfer_option - transfer_option)
    :precondition
      (and
        (confirmed_for_handover ?outgoing_segment)
        (transfer_option_selected ?outgoing_segment ?transfer_option)
        (outgoing_segment_on_access_path ?outgoing_segment ?outgoing_access_path)
        (not
          (outgoing_access_path_ready ?outgoing_access_path)
        )
        (not
          (outgoing_access_path_occupied ?outgoing_access_path)
        )
      )
    :effect (outgoing_access_path_ready ?outgoing_access_path)
  )
  (:action assign_staff_to_outgoing
    :parameters (?outgoing_segment - outgoing_segment ?outgoing_access_path - outgoing_access_path ?staff_resource - staff_resource)
    :precondition
      (and
        (confirmed_for_handover ?outgoing_segment)
        (staff_assigned ?outgoing_segment ?staff_resource)
        (outgoing_segment_on_access_path ?outgoing_segment ?outgoing_access_path)
        (outgoing_access_path_ready ?outgoing_access_path)
        (not
          (outgoing_segment_flagged ?outgoing_segment)
        )
      )
    :effect
      (and
        (outgoing_segment_flagged ?outgoing_segment)
        (outgoing_segment_ready ?outgoing_segment)
      )
  )
  (:action assign_baggage_to_outgoing
    :parameters (?outgoing_segment - outgoing_segment ?outgoing_access_path - outgoing_access_path ?baggage_item - baggage_or_mobility_aid)
    :precondition
      (and
        (confirmed_for_handover ?outgoing_segment)
        (outgoing_segment_on_access_path ?outgoing_segment ?outgoing_access_path)
        (baggage_available ?baggage_item)
        (not
          (outgoing_segment_flagged ?outgoing_segment)
        )
      )
    :effect
      (and
        (outgoing_access_path_occupied ?outgoing_access_path)
        (outgoing_segment_flagged ?outgoing_segment)
        (outgoing_segment_baggage_assigned ?outgoing_segment ?baggage_item)
        (not
          (baggage_available ?baggage_item)
        )
      )
  )
  (:action finalize_outgoing_baggage_allocation
    :parameters (?outgoing_segment - outgoing_segment ?outgoing_access_path - outgoing_access_path ?transfer_option - transfer_option ?baggage_item - baggage_or_mobility_aid)
    :precondition
      (and
        (confirmed_for_handover ?outgoing_segment)
        (transfer_option_selected ?outgoing_segment ?transfer_option)
        (outgoing_segment_on_access_path ?outgoing_segment ?outgoing_access_path)
        (outgoing_access_path_occupied ?outgoing_access_path)
        (outgoing_segment_baggage_assigned ?outgoing_segment ?baggage_item)
        (not
          (outgoing_segment_ready ?outgoing_segment)
        )
      )
    :effect
      (and
        (outgoing_access_path_ready ?outgoing_access_path)
        (outgoing_segment_ready ?outgoing_segment)
        (baggage_available ?baggage_item)
        (not
          (outgoing_segment_baggage_assigned ?outgoing_segment ?baggage_item)
        )
      )
  )
  (:action allocate_connector
    :parameters (?incoming_segment - incoming_segment ?outgoing_segment - outgoing_segment ?incoming_access_path - incoming_access_path ?outgoing_access_path - outgoing_access_path ?connector - connector_instance)
    :precondition
      (and
        (incoming_segment_flagged ?incoming_segment)
        (outgoing_segment_flagged ?outgoing_segment)
        (incoming_segment_on_access_path ?incoming_segment ?incoming_access_path)
        (outgoing_segment_on_access_path ?outgoing_segment ?outgoing_access_path)
        (incoming_access_path_ready ?incoming_access_path)
        (outgoing_access_path_ready ?outgoing_access_path)
        (incoming_segment_ready ?incoming_segment)
        (outgoing_segment_ready ?outgoing_segment)
        (connector_available ?connector)
      )
    :effect
      (and
        (connector_reserved ?connector)
        (connector_assigned_incoming ?connector ?incoming_access_path)
        (connector_assigned_outgoing ?connector ?outgoing_access_path)
        (not
          (connector_available ?connector)
        )
      )
  )
  (:action allocate_connector_incoming_occupied
    :parameters (?incoming_segment - incoming_segment ?outgoing_segment - outgoing_segment ?incoming_access_path - incoming_access_path ?outgoing_access_path - outgoing_access_path ?connector - connector_instance)
    :precondition
      (and
        (incoming_segment_flagged ?incoming_segment)
        (outgoing_segment_flagged ?outgoing_segment)
        (incoming_segment_on_access_path ?incoming_segment ?incoming_access_path)
        (outgoing_segment_on_access_path ?outgoing_segment ?outgoing_access_path)
        (incoming_access_path_occupied ?incoming_access_path)
        (outgoing_access_path_ready ?outgoing_access_path)
        (not
          (incoming_segment_ready ?incoming_segment)
        )
        (outgoing_segment_ready ?outgoing_segment)
        (connector_available ?connector)
      )
    :effect
      (and
        (connector_reserved ?connector)
        (connector_assigned_incoming ?connector ?incoming_access_path)
        (connector_assigned_outgoing ?connector ?outgoing_access_path)
        (connector_incoming_configured ?connector)
        (not
          (connector_available ?connector)
        )
      )
  )
  (:action allocate_connector_outgoing_occupied
    :parameters (?incoming_segment - incoming_segment ?outgoing_segment - outgoing_segment ?incoming_access_path - incoming_access_path ?outgoing_access_path - outgoing_access_path ?connector - connector_instance)
    :precondition
      (and
        (incoming_segment_flagged ?incoming_segment)
        (outgoing_segment_flagged ?outgoing_segment)
        (incoming_segment_on_access_path ?incoming_segment ?incoming_access_path)
        (outgoing_segment_on_access_path ?outgoing_segment ?outgoing_access_path)
        (incoming_access_path_ready ?incoming_access_path)
        (outgoing_access_path_occupied ?outgoing_access_path)
        (incoming_segment_ready ?incoming_segment)
        (not
          (outgoing_segment_ready ?outgoing_segment)
        )
        (connector_available ?connector)
      )
    :effect
      (and
        (connector_reserved ?connector)
        (connector_assigned_incoming ?connector ?incoming_access_path)
        (connector_assigned_outgoing ?connector ?outgoing_access_path)
        (connector_outgoing_configured ?connector)
        (not
          (connector_available ?connector)
        )
      )
  )
  (:action allocate_connector_both_occupied
    :parameters (?incoming_segment - incoming_segment ?outgoing_segment - outgoing_segment ?incoming_access_path - incoming_access_path ?outgoing_access_path - outgoing_access_path ?connector - connector_instance)
    :precondition
      (and
        (incoming_segment_flagged ?incoming_segment)
        (outgoing_segment_flagged ?outgoing_segment)
        (incoming_segment_on_access_path ?incoming_segment ?incoming_access_path)
        (outgoing_segment_on_access_path ?outgoing_segment ?outgoing_access_path)
        (incoming_access_path_occupied ?incoming_access_path)
        (outgoing_access_path_occupied ?outgoing_access_path)
        (not
          (incoming_segment_ready ?incoming_segment)
        )
        (not
          (outgoing_segment_ready ?outgoing_segment)
        )
        (connector_available ?connector)
      )
    :effect
      (and
        (connector_reserved ?connector)
        (connector_assigned_incoming ?connector ?incoming_access_path)
        (connector_assigned_outgoing ?connector ?outgoing_access_path)
        (connector_incoming_configured ?connector)
        (connector_outgoing_configured ?connector)
        (not
          (connector_available ?connector)
        )
      )
  )
  (:action confirm_connector_boarding_ready
    :parameters (?connector - connector_instance ?incoming_segment - incoming_segment ?transfer_option - transfer_option)
    :precondition
      (and
        (connector_reserved ?connector)
        (incoming_segment_flagged ?incoming_segment)
        (transfer_option_selected ?incoming_segment ?transfer_option)
        (not
          (connector_boarding_ready ?connector)
        )
      )
    :effect (connector_boarding_ready ?connector)
  )
  (:action provision_facility
    :parameters (?transfer_case - transfer_case ?facility_resource - facility_resource ?connector - connector_instance)
    :precondition
      (and
        (confirmed_for_handover ?transfer_case)
        (case_assigned_connector ?transfer_case ?connector)
        (case_requires_facility ?transfer_case ?facility_resource)
        (facility_available ?facility_resource)
        (connector_reserved ?connector)
        (connector_boarding_ready ?connector)
        (not
          (facility_allocated ?facility_resource)
        )
      )
    :effect
      (and
        (facility_allocated ?facility_resource)
        (facility_assigned_to_connector ?facility_resource ?connector)
        (not
          (facility_available ?facility_resource)
        )
      )
  )
  (:action verify_facility_and_mark_case
    :parameters (?transfer_case - transfer_case ?facility_resource - facility_resource ?connector - connector_instance ?transfer_option - transfer_option)
    :precondition
      (and
        (confirmed_for_handover ?transfer_case)
        (case_requires_facility ?transfer_case ?facility_resource)
        (facility_allocated ?facility_resource)
        (facility_assigned_to_connector ?facility_resource ?connector)
        (transfer_option_selected ?transfer_case ?transfer_option)
        (not
          (connector_incoming_configured ?connector)
        )
        (not
          (case_facility_verified ?transfer_case)
        )
      )
    :effect (case_facility_verified ?transfer_case)
  )
  (:action apply_priority_service
    :parameters (?transfer_case - transfer_case ?priority_service - priority_service_type)
    :precondition
      (and
        (confirmed_for_handover ?transfer_case)
        (priority_service_available ?priority_service)
        (not
          (case_priority_selected ?transfer_case)
        )
      )
    :effect
      (and
        (case_priority_selected ?transfer_case)
        (case_linked_priority_service ?transfer_case ?priority_service)
        (not
          (priority_service_available ?priority_service)
        )
      )
  )
  (:action finalize_priority_and_prepare_case
    :parameters (?transfer_case - transfer_case ?facility_resource - facility_resource ?connector - connector_instance ?transfer_option - transfer_option ?priority_service - priority_service_type)
    :precondition
      (and
        (confirmed_for_handover ?transfer_case)
        (case_requires_facility ?transfer_case ?facility_resource)
        (facility_allocated ?facility_resource)
        (facility_assigned_to_connector ?facility_resource ?connector)
        (transfer_option_selected ?transfer_case ?transfer_option)
        (connector_incoming_configured ?connector)
        (case_priority_selected ?transfer_case)
        (case_linked_priority_service ?transfer_case ?priority_service)
        (not
          (case_facility_verified ?transfer_case)
        )
      )
    :effect
      (and
        (case_facility_verified ?transfer_case)
        (case_priority_acknowledged ?transfer_case)
      )
  )
  (:action assign_vehicle_capability_to_case
    :parameters (?transfer_case - transfer_case ?vehicle_capability - vehicle_capability ?staff_resource - staff_resource ?facility_resource - facility_resource ?connector - connector_instance)
    :precondition
      (and
        (case_facility_verified ?transfer_case)
        (case_vehicle_capability_assigned ?transfer_case ?vehicle_capability)
        (staff_assigned ?transfer_case ?staff_resource)
        (case_requires_facility ?transfer_case ?facility_resource)
        (facility_assigned_to_connector ?facility_resource ?connector)
        (not
          (connector_outgoing_configured ?connector)
        )
        (not
          (case_resource_compatibility_verified ?transfer_case)
        )
      )
    :effect (case_resource_compatibility_verified ?transfer_case)
  )
  (:action confirm_vehicle_capability_for_case
    :parameters (?transfer_case - transfer_case ?vehicle_capability - vehicle_capability ?staff_resource - staff_resource ?facility_resource - facility_resource ?connector - connector_instance)
    :precondition
      (and
        (case_facility_verified ?transfer_case)
        (case_vehicle_capability_assigned ?transfer_case ?vehicle_capability)
        (staff_assigned ?transfer_case ?staff_resource)
        (case_requires_facility ?transfer_case ?facility_resource)
        (facility_assigned_to_connector ?facility_resource ?connector)
        (connector_outgoing_configured ?connector)
        (not
          (case_resource_compatibility_verified ?transfer_case)
        )
      )
