    (case_verified ?case_record - case_record)
    (case_final_checks_completed ?case_record - case_record)

    (case_has_medication_dose ?case_record - case_record ?medication_dose - medication_dose)

    (case_has_certification ?case_record - case_record ?certification - certification)

  (:action register_entity
    :parameters (?herd_subject - herd_subject)
    :precondition
      (and
        (not
          (entity_registered ?herd_subject)
        )
        (not
          (entity_authorized_for_transfer ?herd_subject)
        )
      )
    :effect (entity_registered ?herd_subject)
  )
  (:action allocate_holding_slot
    :parameters (?herd_subject - herd_subject ?holding_slot - holding_slot)
    :precondition
      (and
        (entity_registered ?herd_subject)
        (not
          (entity_in_holding ?herd_subject)
        )
        (holding_slot_available ?holding_slot)
      )
    :effect
      (and
        (entity_in_holding ?herd_subject)
        (entity_assigned_to_holding_slot ?herd_subject ?holding_slot)
        (not
          (holding_slot_available ?holding_slot)
        )
      )
  )
  (:action assign_diagnostic_kit
    :parameters (?herd_subject - herd_subject ?diagnostic_kit - diagnostic_kit)
    :precondition
      (and
        (entity_registered ?herd_subject)
        (entity_in_holding ?herd_subject)
        (diagnostic_kit_available ?diagnostic_kit)
      )
    :effect
      (and
        (entity_assigned_diagnostic_kit ?herd_subject ?diagnostic_kit)
        (not
          (diagnostic_kit_available ?diagnostic_kit)
        )
      )
  )
  (:action complete_diagnostic_test
    :parameters (?herd_subject - herd_subject ?diagnostic_kit - diagnostic_kit)
    :precondition
      (and
        (entity_registered ?herd_subject)
        (entity_in_holding ?herd_subject)
        (entity_assigned_diagnostic_kit ?herd_subject ?diagnostic_kit)
        (not
          (entity_sample_processed ?herd_subject)
        )
      )
    :effect (entity_sample_processed ?herd_subject)
  )
  (:action release_diagnostic_kit
    :parameters (?herd_subject - herd_subject ?diagnostic_kit - diagnostic_kit)
    :precondition
      (and
        (entity_assigned_diagnostic_kit ?herd_subject ?diagnostic_kit)
      )
    :effect
      (and
        (diagnostic_kit_available ?diagnostic_kit)
        (not
          (entity_assigned_diagnostic_kit ?herd_subject ?diagnostic_kit)
        )
      )
  )
  (:action assign_technician_to_entity
    :parameters (?herd_subject - herd_subject ?technician - technician)
    :precondition
      (and
        (entity_sample_processed ?herd_subject)
        (technician_available ?technician)
      )
    :effect
      (and
        (technician_assigned_to_entity ?herd_subject ?technician)
        (not
          (technician_available ?technician)
        )
      )
  )
  (:action release_technician_from_entity
    :parameters (?herd_subject - herd_subject ?technician - technician)
    :precondition
      (and
        (technician_assigned_to_entity ?herd_subject ?technician)
      )
    :effect
      (and
        (technician_available ?technician)
        (not
          (technician_assigned_to_entity ?herd_subject ?technician)
        )
      )
  )
  (:action assign_certification_to_case

          (case_has_certification ?case_record ?certification)

    :parameters (?case_record - case_record ?vaccine_batch - vaccine_batch)

      (and
        (case_has_vaccine_batch ?case_record ?vaccine_batch)
      )
    :effect
      (and
        (vaccine_batch_available ?vaccine_batch)
        (not
          (case_has_vaccine_batch ?case_record ?vaccine_batch)
        )
      )
  )
  (:action confirm_feeding_slot
    :parameters (?quarantine_unit - quarantine_unit ?feeding_slot - feeding_slot ?diagnostic_kit - diagnostic_kit)
    :precondition
      (and
        (entity_sample_processed ?quarantine_unit)
        (entity_assigned_diagnostic_kit ?quarantine_unit ?diagnostic_kit)
        (quarantine_unit_has_feeding_slot ?quarantine_unit ?feeding_slot)
        (not
          (feeding_slot_confirmed ?feeding_slot)

        (not
          (feeding_slot_has_feed ?feeding_slot)

  )
  (:action execute_feeding_by_technician
    :parameters (?quarantine_unit - quarantine_unit ?feeding_slot - feeding_slot ?technician - technician)
    :precondition
      (and
        (entity_sample_processed ?quarantine_unit)
        (technician_assigned_to_entity ?quarantine_unit ?technician)
        (quarantine_unit_has_feeding_slot ?quarantine_unit ?feeding_slot)
        (feeding_slot_confirmed ?feeding_slot)
        (not
          (quarantine_unit_ready_for_transfer ?quarantine_unit)
        )
      )
    :effect
      (and
        (quarantine_unit_ready_for_transfer ?quarantine_unit)
        (quarantine_unit_feed_confirmed ?quarantine_unit)
      )
  )
  (:action assign_feed_to_quarantine_unit
    :parameters (?quarantine_unit - quarantine_unit ?feeding_slot - feeding_slot ?feed_type - feed_type)
    :precondition
      (and
        (entity_sample_processed ?quarantine_unit)
        (quarantine_unit_has_feeding_slot ?quarantine_unit ?feeding_slot)
        (feed_type_available ?feed_type)
        (not
          (quarantine_unit_ready_for_transfer ?quarantine_unit)
        )
      )
    :effect
      (and
        (feeding_slot_has_feed ?feeding_slot)
        (quarantine_unit_ready_for_transfer ?quarantine_unit)
        (quarantine_unit_assigned_feed ?quarantine_unit ?feed_type)
        (not
          (feed_type_available ?feed_type)
        )
      )
  )
  (:action record_feed_consumption
    :parameters (?quarantine_unit - quarantine_unit ?feeding_slot - feeding_slot ?diagnostic_kit - diagnostic_kit ?feed_type - feed_type)
    :precondition
      (and
        (entity_sample_processed ?quarantine_unit)

        (quarantine_unit_has_feeding_slot ?quarantine_unit ?feeding_slot)
        (feeding_slot_has_feed ?feeding_slot)
        (quarantine_unit_assigned_feed ?quarantine_unit ?feed_type)
        (not
          (quarantine_unit_feed_confirmed ?quarantine_unit)
        )
      )
    :effect
      (and
        (feeding_slot_confirmed ?feeding_slot)
        (quarantine_unit_feed_confirmed ?quarantine_unit)
        (feed_type_available ?feed_type)
        (not
          (quarantine_unit_assigned_feed ?quarantine_unit ?feed_type)
        )
      )
  )
  (:action confirm_watering_slot
    :parameters (?contact_unit - contact_unit ?watering_slot - watering_slot ?diagnostic_kit - diagnostic_kit)

        (entity_sample_processed ?contact_unit)

        (contact_unit_ready_for_transfer ?contact_unit)

      )
  )
  (:action assign_feed_to_contact_unit

          (contact_unit_ready_for_transfer ?contact_unit)
        )
      )
    :effect
      (and
        (watering_slot_has_water ?watering_slot)
        (contact_unit_ready_for_transfer ?contact_unit)

          (feed_type_available ?feed_type)

      )
  )
  (:action record_feed_consumption_for_contact_unit
    :parameters (?contact_unit - contact_unit ?watering_slot - watering_slot ?diagnostic_kit - diagnostic_kit ?feed_type - feed_type)
    :precondition
      (and

    :effect
      (and
        (watering_slot_confirmed ?watering_slot)
        (contact_unit_water_confirmed ?contact_unit)

          (contact_unit_assigned_feed ?contact_unit ?feed_type)

  (:action assemble_transfer_container
    :parameters (?quarantine_unit - quarantine_unit ?contact_unit - contact_unit ?feeding_slot - feeding_slot ?watering_slot - watering_slot ?transfer_container - transfer_container)

      (and
        (quarantine_unit_ready_for_transfer ?quarantine_unit)
        (contact_unit_ready_for_transfer ?contact_unit)
        (quarantine_unit_has_feeding_slot ?quarantine_unit ?feeding_slot)
        (contact_unit_has_watering_slot ?contact_unit ?watering_slot)
        (feeding_slot_confirmed ?feeding_slot)
        (watering_slot_confirmed ?watering_slot)
        (quarantine_unit_feed_confirmed ?quarantine_unit)
        (contact_unit_water_confirmed ?contact_unit)
        (transfer_container_available ?transfer_container)
      )
    :effect
      (and
        (transfer_container_allocated ?transfer_container)
        (transfer_container_has_feeding_slot ?transfer_container ?feeding_slot)
        (transfer_container_has_watering_slot ?transfer_container ?watering_slot)
        (not
          (transfer_container_available ?transfer_container)
        )
      )
  )
  (:action assemble_transfer_container_with_feed_flag
    :parameters (?quarantine_unit - quarantine_unit ?contact_unit - contact_unit ?feeding_slot - feeding_slot ?watering_slot - watering_slot ?transfer_container - transfer_container)

      (and
        (quarantine_unit_ready_for_transfer ?quarantine_unit)

        (contact_unit_has_watering_slot ?contact_unit ?watering_slot)
        (feeding_slot_has_feed ?feeding_slot)
        (watering_slot_confirmed ?watering_slot)

          (quarantine_unit_feed_confirmed ?quarantine_unit)

    :effect
      (and
        (transfer_container_allocated ?transfer_container)

        (transfer_container_flag_feeding ?transfer_container)
        (not
          (transfer_container_available ?transfer_container)
        )
      )
  )
  (:action assemble_transfer_container_with_watering_flag
    :parameters (?quarantine_unit - quarantine_unit ?contact_unit - contact_unit ?feeding_slot - feeding_slot ?watering_slot - watering_slot ?transfer_container - transfer_container)

      (and
        (quarantine_unit_ready_for_transfer ?quarantine_unit)
        (contact_unit_ready_for_transfer ?contact_unit)
        (quarantine_unit_has_feeding_slot ?quarantine_unit ?feeding_slot)

        (watering_slot_has_water ?watering_slot)
        (quarantine_unit_feed_confirmed ?quarantine_unit)
        (not
          (contact_unit_water_confirmed ?contact_unit)

        (transfer_container_available ?transfer_container)

    :effect
      (and
        (transfer_container_allocated ?transfer_container)
        (transfer_container_has_feeding_slot ?transfer_container ?feeding_slot)
        (transfer_container_has_watering_slot ?transfer_container ?watering_slot)
        (transfer_container_flag_watering ?transfer_container)
        (not
          (transfer_container_available ?transfer_container)
        )
      )
  )
  (:action assemble_transfer_container_with_feed_and_water_flags

          (transfer_container_available ?transfer_container)

        (sample_container_allocated ?sample_container)

        (entity_assigned_diagnostic_kit ?case_record ?diagnostic_kit)
        (transfer_container_flag_feeding ?transfer_container)
        (breeding_asset_assigned ?case_record)
        (case_has_breeding_asset ?case_record ?breeding_asset)
        (not
          (case_sample_prepared ?case_record)

    :effect
      (and
        (case_sample_prepared ?case_record)
        (case_breeding_confirmed ?case_record)

  )

    :parameters (?case_record - case_record ?certification - certification ?technician - technician ?sample_container - sample_container ?transfer_container - transfer_container)
    :precondition
      (and
        (case_sample_prepared ?case_record)
        (case_has_certification ?case_record ?certification)

        (case_has_sample_container ?case_record ?sample_container)

        (not
          (transfer_container_flag_watering ?transfer_container)
        )
        (not
          (case_sample_submitted ?case_record)
        )
      )
    :effect (case_sample_submitted ?case_record)
  )
  (:action submit_case_for_analysis_stage2
    :parameters (?case_record - case_record ?certification - certification ?technician - technician ?sample_container - sample_container ?transfer_container - transfer_container)
    :precondition
      (and
        (case_sample_prepared ?case_record)
        (case_has_certification ?case_record ?certification)
