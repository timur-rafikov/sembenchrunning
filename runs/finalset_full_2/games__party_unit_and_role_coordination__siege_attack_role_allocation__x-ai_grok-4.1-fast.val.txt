(define (domain siege_attack_role_allocation)
  (:requirements :strips :typing :negative-preconditions)
  (:types unit_category - object asset_category - object marker_category - object domain_object_category - object participant - domain_object_category role_token - unit_category specialist_ability - unit_category support_device - unit_category objective_marker - unit_category buff_token - unit_category mission_marker - unit_category equipment_unit - unit_category commander_unit - unit_category consumable_item - asset_category resource_cache - asset_category secondary_objective - asset_category approach_vector_alpha - marker_category approach_vector_bravo - marker_category siege_asset - marker_category assault_subgroup_type - participant assignment_cluster_type - participant assault_subgroup_a_member - assault_subgroup_type assault_subgroup_b_member - assault_subgroup_type assignment_slot - assignment_cluster_type)
  (:predicates
    (participant_selected ?participant - participant)
    (assignee_ready_for_assignment ?participant - participant)
    (role_locked ?participant - participant)
    (assignee_committed ?participant - participant)
    (assignee_ready_for_deployment ?participant - participant)
    (assignee_mission_authorized ?participant - participant)
    (role_available ?role_token - role_token)
    (role_bound_to_assignee ?participant - participant ?role_token - role_token)
    (ability_available ?specialist_ability - specialist_ability)
    (ability_bound_to_assignee ?participant - participant ?specialist_ability - specialist_ability)
    (device_available ?support_device - support_device)
    (device_attached_to_assignee ?participant - participant ?support_device - support_device)
    (item_available ?consumable_item - consumable_item)
    (subgroup_a_member_has_consumable ?assault_subgroup_a_member - assault_subgroup_a_member ?consumable_item - consumable_item)
    (subgroup_b_member_has_consumable ?assault_subgroup_b_member - assault_subgroup_b_member ?consumable_item - consumable_item)
    (assigned_to_approach_vector_alpha ?assault_subgroup_a_member - assault_subgroup_a_member ?approach_vector_a - approach_vector_alpha)
    (approach_vector_alpha_ready ?approach_vector_a - approach_vector_alpha)
    (approach_vector_alpha_has_staged_resource ?approach_vector_a - approach_vector_alpha)
    (subgroup_a_member_ready ?assault_subgroup_a_member - assault_subgroup_a_member)
    (assigned_to_approach_vector_bravo ?assault_subgroup_b_member - assault_subgroup_b_member ?approach_vector_b - approach_vector_bravo)
    (approach_vector_bravo_ready ?approach_vector_b - approach_vector_bravo)
    (approach_vector_bravo_has_staged_resource ?approach_vector_b - approach_vector_bravo)
    (subgroup_b_member_ready ?assault_subgroup_b_member - assault_subgroup_b_member)
    (asset_available ?siege_asset - siege_asset)
    (asset_reserved ?siege_asset - siege_asset)
    (asset_assigned_to_approach_vector_alpha ?siege_asset - siege_asset ?approach_vector_a - approach_vector_alpha)
    (asset_assigned_to_approach_vector_bravo ?siege_asset - siege_asset ?approach_vector_b - approach_vector_bravo)
    (asset_flag_path_alpha ?siege_asset - siege_asset)
    (asset_flag_path_bravo ?siege_asset - siege_asset)
    (asset_prepared ?siege_asset - siege_asset)
    (slot_linked_member_a ?assignment_slot - assignment_slot ?assault_subgroup_a_member - assault_subgroup_a_member)
    (slot_linked_member_b ?assignment_slot - assignment_slot ?assault_subgroup_b_member - assault_subgroup_b_member)
    (slot_assigned_to_asset ?assignment_slot - assignment_slot ?siege_asset - siege_asset)
    (cache_available ?resource_cache - resource_cache)
    (slot_linked_cache ?assignment_slot - assignment_slot ?resource_cache - resource_cache)
    (cache_staged ?resource_cache - resource_cache)
    (cache_assigned_to_asset ?resource_cache - resource_cache ?siege_asset - siege_asset)
    (slot_prepared_base ?assignment_slot - assignment_slot)
    (slot_components_loaded ?assignment_slot - assignment_slot)
    (slot_verified ?assignment_slot - assignment_slot)
    (slot_objective_attached ?assignment_slot - assignment_slot)
    (slot_extended_ready ?assignment_slot - assignment_slot)
    (slot_buff_ready ?assignment_slot - assignment_slot)
    (slot_locked ?assignment_slot - assignment_slot)
    (secondary_objective_available ?secondary_objective - secondary_objective)
    (slot_linked_secondary_objective ?assignment_slot - assignment_slot ?secondary_objective - secondary_objective)
    (slot_secondary_objective_active ?assignment_slot - assignment_slot)
    (slot_secondary_objective_started ?assignment_slot - assignment_slot)
    (slot_secondary_objective_completed ?assignment_slot - assignment_slot)
    (objective_marker_available ?objective_marker - objective_marker)
    (slot_has_objective_marker ?assignment_slot - assignment_slot ?objective_marker - objective_marker)
    (buff_token_available ?buff_token - buff_token)
    (slot_has_buff ?assignment_slot - assignment_slot ?buff_token - buff_token)
    (equipment_unit_available ?equipment_unit - equipment_unit)
    (slot_has_equipment ?assignment_slot - assignment_slot ?equipment_unit - equipment_unit)
    (commander_unit_available ?commander_unit - commander_unit)
    (slot_has_commander ?assignment_slot - assignment_slot ?commander_unit - commander_unit)
    (mission_marker_available ?mission_marker - mission_marker)
    (assignee_assigned_mission_marker ?participant - participant ?mission_marker - mission_marker)
    (subgroup_a_member_prepared ?assault_subgroup_a_member - assault_subgroup_a_member)
    (subgroup_b_member_prepared ?assault_subgroup_b_member - assault_subgroup_b_member)
    (slot_finalized ?assignment_slot - assignment_slot)
  )
  (:action select_participant
    :parameters (?participant - participant)
    :precondition
      (and
        (not
          (participant_selected ?participant)
        )
        (not
          (assignee_committed ?participant)
        )
      )
    :effect (participant_selected ?participant)
  )
  (:action assign_role_token_to_participant
    :parameters (?participant - participant ?role_token - role_token)
    :precondition
      (and
        (participant_selected ?participant)
        (not
          (role_locked ?participant)
        )
        (role_available ?role_token)
      )
    :effect
      (and
        (role_locked ?participant)
        (role_bound_to_assignee ?participant ?role_token)
        (not
          (role_available ?role_token)
        )
      )
  )
  (:action assign_specialist_ability_to_participant
    :parameters (?participant - participant ?specialist_ability - specialist_ability)
    :precondition
      (and
        (participant_selected ?participant)
        (role_locked ?participant)
        (ability_available ?specialist_ability)
      )
    :effect
      (and
        (ability_bound_to_assignee ?participant ?specialist_ability)
        (not
          (ability_available ?specialist_ability)
        )
      )
  )
  (:action confirm_participant_ready
    :parameters (?participant - participant ?specialist_ability - specialist_ability)
    :precondition
      (and
        (participant_selected ?participant)
        (role_locked ?participant)
        (ability_bound_to_assignee ?participant ?specialist_ability)
        (not
          (assignee_ready_for_assignment ?participant)
        )
      )
    :effect (assignee_ready_for_assignment ?participant)
  )
  (:action release_specialist_ability
    :parameters (?participant - participant ?specialist_ability - specialist_ability)
    :precondition
      (and
        (ability_bound_to_assignee ?participant ?specialist_ability)
      )
    :effect
      (and
        (ability_available ?specialist_ability)
        (not
          (ability_bound_to_assignee ?participant ?specialist_ability)
        )
      )
  )
  (:action attach_support_device_to_participant
    :parameters (?participant - participant ?support_device - support_device)
    :precondition
      (and
        (assignee_ready_for_assignment ?participant)
        (device_available ?support_device)
      )
    :effect
      (and
        (device_attached_to_assignee ?participant ?support_device)
        (not
          (device_available ?support_device)
        )
      )
  )
  (:action detach_support_device_from_participant
    :parameters (?participant - participant ?support_device - support_device)
    :precondition
      (and
        (device_attached_to_assignee ?participant ?support_device)
      )
    :effect
      (and
        (device_available ?support_device)
        (not
          (device_attached_to_assignee ?participant ?support_device)
        )
      )
  )

        )
      )
  )
  (:action release_equipment_from_slot
    :parameters (?assignment_slot - assignment_slot ?equipment_unit - equipment_unit)

        (slot_has_equipment ?assignment_slot ?equipment_unit)

    :parameters (?assault_subgroup_a_member - assault_subgroup_a_member ?approach_vector_a - approach_vector_alpha ?specialist_ability - specialist_ability)
    :precondition
      (and
        (assignee_ready_for_assignment ?assault_subgroup_a_member)
        (ability_bound_to_assignee ?assault_subgroup_a_member ?specialist_ability)
        (assigned_to_approach_vector_alpha ?assault_subgroup_a_member ?approach_vector_a)

          (approach_vector_alpha_ready ?approach_vector_a)
        )
        (not
          (approach_vector_alpha_has_staged_resource ?approach_vector_a)

    :effect (approach_vector_alpha_ready ?approach_vector_a)
  )
  (:action prepare_member_for_vector_a_with_device
    :parameters (?assault_subgroup_a_member - assault_subgroup_a_member ?approach_vector_a - approach_vector_alpha ?support_device - support_device)
    :precondition
      (and
        (assignee_ready_for_assignment ?assault_subgroup_a_member)
        (device_attached_to_assignee ?assault_subgroup_a_member ?support_device)
        (assigned_to_approach_vector_alpha ?assault_subgroup_a_member ?approach_vector_a)
        (approach_vector_alpha_ready ?approach_vector_a)
        (not
          (subgroup_a_member_prepared ?assault_subgroup_a_member)
        )
      )
    :effect
      (and
        (subgroup_a_member_prepared ?assault_subgroup_a_member)
        (subgroup_a_member_ready ?assault_subgroup_a_member)
      )
  )
  (:action stage_consumable_for_vector_a
    :parameters (?assault_subgroup_a_member - assault_subgroup_a_member ?approach_vector_a - approach_vector_alpha ?consumable_item - consumable_item)

        (not
          (subgroup_a_member_prepared ?assault_subgroup_a_member)
        )
      )
    :effect
      (and
        (approach_vector_alpha_has_staged_resource ?approach_vector_a)
        (subgroup_a_member_prepared ?assault_subgroup_a_member)
        (subgroup_a_member_has_consumable ?assault_subgroup_a_member ?consumable_item)
        (not
          (item_available ?consumable_item)
        )

  )
  (:action finalize_member_preparation_for_vector_a

        (not
          (subgroup_a_member_ready ?assault_subgroup_a_member)
        )
      )
    :effect
      (and
        (approach_vector_alpha_ready ?approach_vector_a)
        (subgroup_a_member_ready ?assault_subgroup_a_member)
        (item_available ?consumable_item)
        (not
