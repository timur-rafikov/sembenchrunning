--- domain.pddl (as passed to VAL) ---
(define (domain pharmaceutics_complaint_to_quality_system)
  (:requirements :strips :typing :negative-preconditions)
  (:types human_role - object entity - object asset - object quality_record - object complaint_record - quality_record investigator_assignee - human_role complainant_contact - human_role reviewer_approver - human_role risk_assessment_report - human_role release_authority - human_role regulatory_submission_document - human_role equipment_asset - human_role validation_specialist - human_role supporting_document - entity validation_protocol - entity supplier - entity impacted_product - asset impacted_process - asset change_control_package - asset investigation_subclass - complaint_record casefile_subclass - complaint_record investigation_record - investigation_subclass capa_record - investigation_subclass quality_case_file - casefile_subclass)
  (:predicates
    (record_registered ?complaint_record - complaint_record)
    (record_open ?complaint_record - complaint_record)
    (record_assigned ?complaint_record - complaint_record)
    (registered_in_quality_system ?complaint_record - complaint_record)
    (record_finalized ?complaint_record - complaint_record)
    (record_regulatory_associated ?complaint_record - complaint_record)
    (investigator_available ?investigator_assignee - investigator_assignee)
    (assigned_investigator_for_record ?complaint_record - complaint_record ?investigator_assignee - investigator_assignee)
    (complainant_contact_available ?complainant_contact - complainant_contact)
    (record_linked_complainant_contact ?complaint_record - complaint_record ?complainant_contact - complainant_contact)
    (reviewer_available ?reviewer_approver - reviewer_approver)
    (assigned_reviewer_for_record ?complaint_record - complaint_record ?reviewer_approver - reviewer_approver)
    (supporting_document_available ?supporting_document - supporting_document)
    (investigation_linked_supporting_document ?investigation_record - investigation_record ?supporting_document - supporting_document)
    (capa_linked_supporting_document ?capa_record - capa_record ?supporting_document - supporting_document)
    (investigation_impacted_product ?investigation_record - investigation_record ?impacted_product - impacted_product)
    (product_marked_for_validation ?impacted_product - impacted_product)
    (product_marked_for_change_control ?impacted_product - impacted_product)
    (investigation_ready_for_change_package ?investigation_record - investigation_record)
    (capa_impacted_process ?capa_record - capa_record ?impacted_process - impacted_process)
    (process_marked_for_validation ?impacted_process - impacted_process)
    (process_marked_for_change_control ?impacted_process - impacted_process)
    (capa_ready_for_change_package ?capa_record - capa_record)
    (change_control_package_prepared ?change_control_package - change_control_package)
    (change_control_package_compiled ?change_control_package - change_control_package)
    (change_control_package_includes_product ?change_control_package - change_control_package ?impacted_product - impacted_product)
    (change_control_package_includes_process ?change_control_package - change_control_package ?impacted_process - impacted_process)
    (change_control_package_includes_validation_dependency ?change_control_package - change_control_package)
    (change_control_package_includes_specialist_dependency ?change_control_package - change_control_package)
    (change_control_package_validation_ready ?change_control_package - change_control_package)
    (casefile_includes_investigation ?quality_case_file - quality_case_file ?investigation_record - investigation_record)
    (casefile_includes_capa ?quality_case_file - quality_case_file ?capa_record - capa_record)
    (casefile_includes_change_control_package ?quality_case_file - quality_case_file ?change_control_package - change_control_package)
    (validation_protocol_available ?validation_protocol - validation_protocol)
    (casefile_includes_validation_protocol ?quality_case_file - quality_case_file ?validation_protocol - validation_protocol)
    (validation_protocol_attached ?validation_protocol - validation_protocol)
    (validation_protocol_linked_to_change_control_package ?validation_protocol - validation_protocol ?change_control_package - change_control_package)
    (casefile_technical_review_complete ?quality_case_file - quality_case_file)
    (casefile_validation_ready ?quality_case_file - quality_case_file)
    (casefile_ready_for_approval ?quality_case_file - quality_case_file)
    (casefile_risk_assessment_assigned ?quality_case_file - quality_case_file)
    (casefile_ready_for_additional_approval ?quality_case_file - quality_case_file)
    (casefile_requires_release_authority ?quality_case_file - quality_case_file)
    (casefile_implementation_complete ?quality_case_file - quality_case_file)
    (supplier_available ?supplier - supplier)
    (casefile_linked_supplier ?quality_case_file - quality_case_file ?supplier - supplier)
    (casefile_supplier_review_initiated ?quality_case_file - quality_case_file)
    (casefile_supplier_review_completed ?quality_case_file - quality_case_file)
    (casefile_supplier_review_approved ?quality_case_file - quality_case_file)
    (risk_assessment_available ?risk_assessment - risk_assessment_report)
    (casefile_linked_risk_assessment ?quality_case_file - quality_case_file ?risk_assessment - risk_assessment_report)
    (release_authority_available ?release_authority - release_authority)
    (casefile_linked_release_authority ?quality_case_file - quality_case_file ?release_authority - release_authority)
    (equipment_available ?equipment_or_system - equipment_asset)
    (casefile_linked_equipment ?quality_case_file - quality_case_file ?equipment_or_system - equipment_asset)
    (validation_specialist_available ?validation_specialist - validation_specialist)
    (casefile_linked_validation_specialist ?quality_case_file - quality_case_file ?validation_specialist - validation_specialist)
    (regulatory_reference_available ?regulatory_submission_reference - regulatory_submission_document)
    (linked_regulatory_reference ?complaint_record - complaint_record ?regulatory_submission_reference - regulatory_submission_document)
    (investigation_ready ?investigation_record - investigation_record)
    (capa_ready ?capa_record - capa_record)
    (casefile_approved_for_closure ?quality_case_file - quality_case_file)
  )
  (:action register_complaint_record
    :parameters (?complaint_record - complaint_record)
    :precondition
      (and
        (not
          (record_registered ?complaint_record)
        )
        (not
          (registered_in_quality_system ?complaint_record)
        )
      )
    :effect (record_registered ?complaint_record)
  )
  (:action assign_investigator_to_complaint
    :parameters (?complaint_record - complaint_record ?investigator_assignee - investigator_assignee)
    :precondition
      (and
        (record_registered ?complaint_record)
        (not
          (record_assigned ?complaint_record)
        )
        (investigator_available ?investigator_assignee)
      )
    :effect
      (and
        (record_assigned ?complaint_record)
        (assigned_investigator_for_record ?complaint_record ?investigator_assignee)
        (not
          (investigator_available ?investigator_assignee)
        )
      )
  )
  (:action link_complainant_contact_to_complaint
    :parameters (?complaint_record - complaint_record ?complainant_contact - complainant_contact)
    :precondition
      (and
        (record_registered ?complaint_record)
        (record_assigned ?complaint_record)
        (complainant_contact_available ?complainant_contact)
      )
    :effect
      (and
        (record_linked_complainant_contact ?complaint_record ?complainant_contact)
        (not
          (complainant_contact_available ?complainant_contact)
        )
      )
  )
  (:action complete_complaint_triage
    :parameters (?complaint_record - complaint_record ?complainant_contact - complainant_contact)
    :precondition
      (and
        (record_registered ?complaint_record)
        (record_assigned ?complaint_record)
        (record_linked_complainant_contact ?complaint_record ?complainant_contact)
        (not
          (record_open ?complaint_record)
        )
      )
    :effect (record_open ?complaint_record)
  )
  (:action release_complainant_contact
    :parameters (?complaint_record - complaint_record ?complainant_contact - complainant_contact)
    :precondition
      (and
        (record_linked_complainant_contact ?complaint_record ?complainant_contact)
      )
    :effect
      (and
        (complainant_contact_available ?complainant_contact)
        (not
          (record_linked_complainant_contact ?complaint_record ?complainant_contact)
        )
      )
  )
  (:action assign_reviewer_to_record
    :parameters (?complaint_record - complaint_record ?reviewer_approver - reviewer_approver)
    :precondition
      (and
        (record_open ?complaint_record)
        (reviewer_available ?reviewer_approver)
      )
    :effect
      (and
        (assigned_reviewer_for_record ?complaint_record ?reviewer_approver)
        (not
          (reviewer_available ?reviewer_approver)
        )
      )
  )
  (:action release_reviewer_from_record
    :parameters (?complaint_record - complaint_record ?reviewer_approver - reviewer_approver)
    :precondition
      (and
        (assigned_reviewer_for_record ?complaint_record ?reviewer_approver)
      )
    :effect
      (and
        (reviewer_available ?reviewer_approver)
        (not
          (assigned_reviewer_for_record ?complaint_record ?reviewer_approver)
        )
      )
  )
  (:action link_equipment_to_casefile
    :parameters (?quality_case_file - quality_case_file ?equipment_or_system - equipment_asset)
    :precondition
      (and
        (record_open ?quality_case_file)
        (equipment_available ?equipment_or_system)
      )
    :effect
      (and
        (casefile_linked_equipment ?quality_case_file ?equipment_or_system)
        (not
          (equipment_available ?equipment_or_system)
        )
      )
  )
  (:action unlink_equipment_from_casefile
    :parameters (?quality_case_file - quality_case_file ?equipment_or_system - equipment_asset)
    :precondition
      (and
        (casefile_linked_equipment ?quality_case_file ?equipment_or_system)
      )
    :effect
      (and
        (equipment_available ?equipment_or_system)
        (not
          (casefile_linked_equipment ?quality_case_file ?equipment_or_system)
        )
      )
  )
  (:action link_validation_specialist_to_casefile
    :parameters (?quality_case_file - quality_case_file ?validation_specialist - validation_specialist)
    :precondition
      (and
        (record_open ?quality_case_file)
        (validation_specialist_available ?validation_specialist)
      )
    :effect
      (and
        (casefile_linked_validation_specialist ?quality_case_file ?validation_specialist)
        (not
          (validation_specialist_available ?validation_specialist)
        )
      )
  )
  (:action unlink_validation_specialist_from_casefile
    :parameters (?quality_case_file - quality_case_file ?validation_specialist - validation_specialist)
    :precondition
      (and
        (casefile_linked_validation_specialist ?quality_case_file ?validation_specialist)
      )
    :effect
      (and
        (validation_specialist_available ?validation_specialist)
        (not
          (casefile_linked_validation_specialist ?quality_case_file ?validation_specialist)
        )
      )
  )
  (:action flag_product_for_validation
    :parameters (?investigation_record - investigation_record ?impacted_product - impacted_product ?complainant_contact - complainant_contact)
    :precondition
      (and
        (record_open ?investigation_record)
        (record_linked_complainant_contact ?investigation_record ?complainant_contact)
        (investigation_impacted_product ?investigation_record ?impacted_product)
        (not
          (product_marked_for_validation ?impacted_product)
        )
        (not
          (product_marked_for_change_control ?impacted_product)
        )
      )
    :effect (product_marked_for_validation ?impacted_product)
  )
  (:action mark_investigation_ready_for_change_package
    :parameters (?investigation_record - investigation_record ?impacted_product - impacted_product ?reviewer_approver - reviewer_approver)
    :precondition
      (and
        (record_open ?investigation_record)
        (assigned_reviewer_for_record ?investigation_record ?reviewer_approver)
        (investigation_impacted_product ?investigation_record ?impacted_product)
        (product_marked_for_validation ?impacted_product)
        (not
          (investigation_ready ?investigation_record)
        )
      )
    :effect
      (and
        (investigation_ready ?investigation_record)
        (investigation_ready_for_change_package ?investigation_record)
      )
  )
  (:action attach_supporting_document_and_mark_product_for_change_control
    :parameters (?investigation_record - investigation_record ?impacted_product - impacted_product ?supporting_document - supporting_document)
    :precondition
      (and
        (record_open ?investigation_record)
        (investigation_impacted_product ?investigation_record ?impacted_product)
        (supporting_document_available ?supporting_document)
        (not
          (investigation_ready ?investigation_record)
        )
      )
    :effect
      (and
        (product_marked_for_change_control ?impacted_product)
        (investigation_ready ?investigation_record)
        (investigation_linked_supporting_document ?investigation_record ?supporting_document)
        (not
          (supporting_document_available ?supporting_document)
        )
      )
  )
  (:action convert_product_change_control_to_validation_and_update_investigation
    :parameters (?investigation_record - investigation_record ?impacted_product - impacted_product ?complainant_contact - complainant_contact ?supporting_document - supporting_document)
    :precondition
      (and
        (record_open ?investigation_record)
        (record_linked_complainant_contact ?investigation_record ?complainant_contact)
        (investigation_impacted_product ?investigation_record ?impacted_product)
        (product_marked_for_change_control ?impacted_product)
        (investigation_linked_supporting_document ?investigation_record ?supporting_document)
        (not
          (investigation_ready_for_change_package ?investigation_record)
        )
      )
    :effect
      (and
        (product_marked_for_validation ?impacted_product)
        (investigation_ready_for_change_package ?investigation_record)
        (supporting_document_available ?supporting_document)
        (not
          (investigation_linked_supporting_document ?investigation_record ?supporting_document)
        )
      )
  )
  (:action flag_impacted_process_for_validation
    :parameters (?capa_record - capa_record ?impacted_process - impacted_process ?complainant_contact - complainant_contact)
    :precondition
      (and
        (record_open ?capa_record)
        (record_linked_complainant_contact ?capa_record ?complainant_contact)
        (capa_impacted_process ?capa_record ?impacted_process)
        (not
          (process_marked_for_validation ?impacted_process)
        )
        (not
          (process_marked_for_change_control ?impacted_process)
        )
      )
    :effect (process_marked_for_validation ?impacted_process)
  )
  (:action assign_reviewer_and_mark_capa_ready_for_package
    :parameters (?capa_record - capa_record ?impacted_process - impacted_process ?reviewer_approver - reviewer_approver)
    :precondition
      (and
        (record_open ?capa_record)
        (assigned_reviewer_for_record ?capa_record ?reviewer_approver)
        (capa_impacted_process ?capa_record ?impacted_process)
        (process_marked_for_validation ?impacted_process)
        (not
          (capa_ready ?capa_record)
        )
      )
    :effect
      (and
        (capa_ready ?capa_record)
        (capa_ready_for_change_package ?capa_record)
      )
  )
  (:action attach_supporting_document_and_mark_process_for_change_control
    :parameters (?capa_record - capa_record ?impacted_process - impacted_process ?supporting_document - supporting_document)
    :precondition
      (and
        (record_open ?capa_record)
        (capa_impacted_process ?capa_record ?impacted_process)
        (supporting_document_available ?supporting_document)
        (not
          (capa_ready ?capa_record)
        )
      )
    :effect
      (and
        (process_marked_for_change_control ?impacted_process)
        (capa_ready ?capa_record)
        (capa_linked_supporting_document ?capa_record ?supporting_document)
        (not
          (supporting_document_available ?supporting_document)
        )
      )
  )
  (:action convert_process_change_to_validation_and_update_capa
    :parameters (?capa_record - capa_record ?impacted_process - impacted_process ?complainant_contact - complainant_contact ?supporting_document - supporting_document)
    :precondition
      (and
        (record_open ?capa_record)
        (record_linked_complainant_contact ?capa_record ?complainant_contact)
        (capa_impacted_process ?capa_record ?impacted_process)
        (process_marked_for_change_control ?impacted_process)
        (capa_linked_supporting_document ?capa_record ?supporting_document)
        (not
          (capa_ready_for_change_package ?capa_record)
        )
      )
    :effect
      (and
        (process_marked_for_validation ?impacted_process)
        (capa_ready_for_change_package ?capa_record)
        (supporting_document_available ?supporting_document)
        (not
          (capa_linked_supporting_document ?capa_record ?supporting_document)
        )
      )
  )
  (:action assemble_change_control_package
    :parameters (?investigation_record - investigation_record ?capa_record - capa_record ?impacted_product - impacted_product ?impacted_process - impacted_process ?change_control_package - change_control_package)
    :precondition
      (and
        (investigation_ready ?investigation_record)
        (capa_ready ?capa_record)
        (investigation_impacted_product ?investigation_record ?impacted_product)
        (capa_impacted_process ?capa_record ?impacted_process)
        (product_marked_for_validation ?impacted_product)
        (process_marked_for_validation ?impacted_process)
        (investigation_ready_for_change_package ?investigation_record)
        (capa_ready_for_change_package ?capa_record)
        (change_control_package_prepared ?change_control_package)
      )
    :effect
      (and
        (change_control_package_compiled ?change_control_package)
        (change_control_package_includes_product ?change_control_package ?impacted_product)
        (change_control_package_includes_process ?change_control_package ?impacted_process)
        (not
          (change_control_package_prepared ?change_control_package)
        )
      )
  )
  (:action assemble_change_control_package_with_validation_dependency
    :parameters (?investigation_record - investigation_record ?capa_record - capa_record ?impacted_product - impacted_product ?impacted_process - impacted_process ?change_control_package - change_control_package)
    :precondition
      (and
        (investigation_ready ?investigation_record)
        (capa_ready ?capa_record)
        (investigation_impacted_product ?investigation_record ?impacted_product)
        (capa_impacted_process ?capa_record ?impacted_process)
        (product_marked_for_change_control ?impacted_product)
        (process_marked_for_validation ?impacted_process)
        (not
          (investigation_ready_for_change_package ?investigation_record)
        )
        (capa_ready_for_change_package ?capa_record)
        (change_control_package_prepared ?change_control_package)
      )
    :effect
      (and
        (change_control_package_compiled ?change_control_package)
        (change_control_package_includes_product ?change_control_package ?impacted_product)
        (change_control_package_includes_process ?change_control_package ?impacted_process)
        (change_control_package_includes_validation_dependency ?change_control_package)
        (not
          (change_control_package_prepared ?change_control_package)
        )
      )
  )
  (:action assemble_change_control_package_with_specialist_dependency
    :parameters (?investigation_record - investigation_record ?capa_record - capa_record ?impacted_product - impacted_product ?impacted_process - impacted_process ?change_control_package - change_control_package)
    :precondition
      (and
        (investigation_ready ?investigation_record)
        (capa_ready ?capa_record)
        (investigation_impacted_product ?investigation_record ?impacted_product)
        (capa_impacted_process ?capa_record ?impacted_process)
        (product_marked_for_validation ?impacted_product)
        (process_marked_for_change_control ?impacted_process)
        (investigation_ready_for_change_package ?investigation_record)
        (not
          (capa_ready_for_change_package ?capa_record)
        )
        (change_control_package_prepared ?change_control_package)
      )
    :effect
      (and
        (change_control_package_compiled ?change_control_package)
        (change_control_package_includes_product ?change_control_package ?impacted_product)
        (change_control_package_includes_process ?change_control_package ?impacted_process)
        (change_control_package_includes_specialist_dependency ?change_control_package)
        (not
          (change_control_package_prepared ?change_control_package)
        )
      )
  )
  (:action assemble_complete_change_control_package
    :parameters (?investigation_record - investigation_record ?capa_record - capa_record ?impacted_product - impacted_product ?impacted_process - impacted_process ?change_control_package - change_control_package)
    :precondition
      (and
        (investigation_ready ?investigation_record)
        (capa_ready ?capa_record)
        (investigation_impacted_product ?investigation_record ?impacted_product)
        (capa_impacted_process ?capa_record ?impacted_process)
        (product_marked_for_change_control ?impacted_product)
        (process_marked_for_change_control ?impacted_process)
        (not
          (investigation_ready_for_change_package ?investigation_record)
        )
        (not
          (capa_ready_for_change_package ?capa_record)
        )
        (change_control_package_prepared ?change_control_package)
      )
    :effect
      (and
        (change_control_package_compiled ?change_control_package)
        (change_control_package_includes_product ?change_control_package ?impacted_product)
        (change_control_package_includes_process ?change_control_package ?impacted_process)
        (change_control_package_includes_validation_dependency ?change_control_package)
        (change_control_package_includes_specialist_dependency ?change_control_package)
        (not
          (change_control_package_prepared ?change_control_package)
        )
      )
  )
  (:action mark_change_control_package_validation_ready
    :parameters (?change_control_package - change_control_package ?investigation_record - investigation_record ?complainant_contact - complainant_contact)
    :precondition
      (and
        (change_control_package_compiled ?change_control_package)
        (investigation_ready ?investigation_record)
        (record_linked_complainant_contact ?investigation_record ?complainant_contact)
        (not
          (change_control_package_validation_ready ?change_control_package)
        )
      )
    :effect (change_control_package_validation_ready ?change_control_package)
  )
  (:action attach_validation_protocol_to_casefile_and_package
    :parameters (?quality_case_file - quality_case_file ?validation_protocol - validation_protocol ?change_control_package - change_control_package)
    :precondition
      (and
        (record_open ?quality_case_file)
        (casefile_includes_change_control_package ?quality_case_file ?change_control_package)
        (casefile_includes_validation_protocol ?quality_case_file ?validation_protocol)
        (validation_protocol_available ?validation_protocol)
        (change_control_package_compiled ?change_control_package)
        (change_control_package_validation_ready ?change_control_package)
        (not
          (validation_protocol_attached ?validation_protocol)
        )
      )
    :effect
      (and
        (validation_protocol_attached ?validation_protocol)
        (validation_protocol_linked_to_change_control_package ?validation_protocol ?change_control_package)
        (not
          (validation_protocol_available ?validation_protocol)
        )
      )
  )
  (:action complete_technical_review_without_validation_dependency
    :parameters (?quality_case_file - quality_case_file ?validation_protocol - validation_protocol ?change_control_package - change_control_package ?complainant_contact - complainant_contact)
    :precondition
      (and
        (record_open ?quality_case_file)
        (casefile_includes_validation_protocol ?quality_case_file ?validation_protocol)
        (validation_protocol_attached ?validation_protocol)
        (validation_protocol_linked_to_change_control_package ?validation_protocol ?change_control_package)
        (record_linked_complainant_contact ?quality_case_file ?complainant_contact)
        (not
          (change_control_package_includes_validation_dependency ?change_control_package)
        )
        (not
          (casefile_technical_review_complete ?quality_case_file)
        )
      )
    :effect (casefile_technical_review_complete ?quality_case_file)
  )
  (:action assign_risk_assessment_to_casefile
    :parameters (?quality_case_file - quality_case_file ?risk_assessment - risk_assessment_report)
    :precondition
      (and
        (record_open ?quality_case_file)
        (risk_assessment_available ?risk_assessment)
        (not
          (casefile_risk_assessment_assigned ?quality_case_file)
        )
      )
    :effect
      (and
        (casefile_risk_assessment_assigned ?quality_case_file)
        (casefile_linked_risk_assessment ?quality_case_file ?risk_assessment)
        (not
          (risk_assessment_available ?risk_assessment)
        )
      )
  )
  (:action complete_technical_review_with_validation_dependency_and_risk_assessment
    :parameters (?quality_case_file - quality_case_file ?validation_protocol - validation_protocol ?change_control_package - change_control_package ?complainant_contact - complainant_contact ?risk_assessment - risk_assessment_report)
    :precondition
      (and
        (record_open ?quality_case_file)
