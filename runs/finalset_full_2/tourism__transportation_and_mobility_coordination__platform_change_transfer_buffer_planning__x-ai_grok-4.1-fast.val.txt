  (:action create_transfer_event
    :parameters (?transfer_event - transfer_event)
    :precondition
      (and
        (not
          (transfer_event_initialized ?transfer_event)
        )
        (not
          (transfer_event_completed ?transfer_event)
        )
      )
    :effect (transfer_event_initialized ?transfer_event)
  )
  (:action reserve_mobility_option_for_transfer
    :parameters (?transfer_event - transfer_event ?local_mobility_option - local_mobility_option)
    :precondition
      (and
        (transfer_event_initialized ?transfer_event)
        (not
          (mobility_option_reserved ?transfer_event)
        )
        (mobility_option_available ?local_mobility_option)
      )
    :effect
      (and
        (mobility_option_reserved ?transfer_event)
        (mobility_assigned_to_transfer_event ?transfer_event ?local_mobility_option)
        (not
          (mobility_option_available ?local_mobility_option)
        )
      )
  )
  (:action assign_walking_time_estimate
    :parameters (?transfer_event - transfer_event ?walking_time_estimate - walking_time_estimate)
    :precondition
      (and
        (transfer_event_initialized ?transfer_event)
        (mobility_option_reserved ?transfer_event)
        (walking_time_available ?walking_time_estimate)
      )
    :effect
      (and
        (walking_estimate_assigned ?transfer_event ?walking_time_estimate)
        (not
          (walking_time_available ?walking_time_estimate)
        )
      )
  )
  (:action confirm_transfer_feasibility
    :parameters (?transfer_event - transfer_event ?walking_time_estimate - walking_time_estimate)
    :precondition
      (and
        (transfer_event_initialized ?transfer_event)
        (mobility_option_reserved ?transfer_event)
        (walking_estimate_assigned ?transfer_event ?walking_time_estimate)
        (not
          (transfer_event_validated ?transfer_event)
        )
      )
    :effect (transfer_event_validated ?transfer_event)
  )
  (:action release_walking_time_estimate
    :parameters (?transfer_event - transfer_event ?walking_time_estimate - walking_time_estimate)
    :precondition
      (and
        (walking_estimate_assigned ?transfer_event ?walking_time_estimate)
      )
    :effect
      (and
        (walking_time_available ?walking_time_estimate)
        (not
          (walking_estimate_assigned ?transfer_event ?walking_time_estimate)
        )
      )
  )
  (:action assign_gate_to_transfer
    :parameters (?transfer_event - transfer_event ?station_gate - station_gate)
    :precondition
      (and
        (transfer_event_validated ?transfer_event)
        (gate_available ?station_gate)
      )
    :effect
      (and
        (gate_assigned_to_transfer_event ?transfer_event ?station_gate)
        (not
          (gate_available ?station_gate)
        )
      )
  )
  (:action release_gate_from_transfer
    :parameters (?transfer_event - transfer_event ?station_gate - station_gate)
    :precondition
      (and
        (gate_assigned_to_transfer_event ?transfer_event ?station_gate)
      )
    :effect
      (and
        (gate_available ?station_gate)
        (not
          (gate_assigned_to_transfer_event ?transfer_event ?station_gate)
        )
      )
  )
  (:action assign_staff_to_plan
    :parameters (?transfer_plan - transfer_plan ?staff_resource - staff_resource)
    :precondition
      (and
        (transfer_event_validated ?transfer_plan)
        (staff_available ?staff_resource)
      )
    :effect
      (and
        (staff_assigned_to_plan ?transfer_plan ?staff_resource)
        (not
          (staff_available ?staff_resource)
        )
      )
  )
  (:action release_staff_from_plan
    :parameters (?transfer_plan - transfer_plan ?staff_resource - staff_resource)
    :precondition
      (and
        (staff_assigned_to_plan ?transfer_plan ?staff_resource)
      )
    :effect
      (and
        (staff_available ?staff_resource)
        (not
          (staff_assigned_to_plan ?transfer_plan ?staff_resource)
        )
      )
  )
  (:action assign_equipment_reservation_to_plan
    :parameters (?transfer_plan - transfer_plan ?equipment_reservation_token - equipment_reservation_token)
    :precondition
      (and
        (transfer_event_validated ?transfer_plan)
        (equipment_reservation_available ?equipment_reservation_token)
      )
    :effect
      (and
        (equipment_assigned_to_plan ?transfer_plan ?equipment_reservation_token)
        (not
          (equipment_reservation_available ?equipment_reservation_token)
        )
      )
  )
  (:action release_equipment_reservation_from_plan
    :parameters (?transfer_plan - transfer_plan ?equipment_reservation_token - equipment_reservation_token)
    :precondition
      (and
        (equipment_assigned_to_plan ?transfer_plan ?equipment_reservation_token)
      )
    :effect
      (and
        (equipment_reservation_available ?equipment_reservation_token)
        (not
          (equipment_assigned_to_plan ?transfer_plan ?equipment_reservation_token)
        )
      )
  )
  (:action claim_arrival_buffer_zone
    :parameters (?incoming_leg - incoming_leg ?arrival_buffer_zone - arrival_buffer_zone ?walking_time_estimate - walking_time_estimate)
    :precondition
      (and
        (transfer_event_validated ?incoming_leg)
        (walking_estimate_assigned ?incoming_leg ?walking_time_estimate)
        (arrival_buffer_assigned ?incoming_leg ?arrival_buffer_zone)
        (not
          (arrival_buffer_reserved ?arrival_buffer_zone)
        )
        (not
          (arrival_buffer_equipment_reserved ?arrival_buffer_zone)
        )
      )
    :effect (arrival_buffer_reserved ?arrival_buffer_zone)
  )
  (:action mark_incoming_leg_ready
    :parameters (?incoming_leg - incoming_leg ?arrival_buffer_zone - arrival_buffer_zone ?station_gate - station_gate)
    :precondition
      (and
        (transfer_event_validated ?incoming_leg)
        (gate_assigned_to_transfer_event ?incoming_leg ?station_gate)
        (arrival_buffer_assigned ?incoming_leg ?arrival_buffer_zone)
        (arrival_buffer_reserved ?arrival_buffer_zone)
        (not
          (incoming_leg_prepared ?incoming_leg)
        )
      )
    :effect
      (and
        (incoming_leg_prepared ?incoming_leg)
        (incoming_leg_ready ?incoming_leg)
      )
  )
  (:action assign_equipment_to_incoming_leg
    :parameters (?incoming_leg - incoming_leg ?arrival_buffer_zone - arrival_buffer_zone ?assistance_equipment - assistance_equipment)
    :precondition
      (and
        (transfer_event_validated ?incoming_leg)
        (arrival_buffer_assigned ?incoming_leg ?arrival_buffer_zone)
        (equipment_available ?assistance_equipment)
        (not
          (incoming_leg_prepared ?incoming_leg)
        )
      )
    :effect
      (and
        (arrival_buffer_equipment_reserved ?arrival_buffer_zone)
        (incoming_leg_prepared ?incoming_leg)
        (equipment_allocated_to_incoming_leg ?incoming_leg ?assistance_equipment)
        (not
          (equipment_available ?assistance_equipment)
        )
      )
  )
  (:action finalize_equipment_assignment_for_incoming_leg
    :parameters (?incoming_leg - incoming_leg ?arrival_buffer_zone - arrival_buffer_zone ?walking_time_estimate - walking_time_estimate ?assistance_equipment - assistance_equipment)
    :precondition
      (and
        (transfer_event_validated ?incoming_leg)
        (walking_estimate_assigned ?incoming_leg ?walking_time_estimate)
        (arrival_buffer_assigned ?incoming_leg ?arrival_buffer_zone)
        (arrival_buffer_equipment_reserved ?arrival_buffer_zone)
        (equipment_allocated_to_incoming_leg ?incoming_leg ?assistance_equipment)
        (not
          (incoming_leg_ready ?incoming_leg)
        )
      )
    :effect
      (and
        (arrival_buffer_reserved ?arrival_buffer_zone)
        (incoming_leg_ready ?incoming_leg)
        (equipment_available ?assistance_equipment)
        (not
          (equipment_allocated_to_incoming_leg ?incoming_leg ?assistance_equipment)
        )
      )
  )
  (:action claim_departure_buffer_zone
    :parameters (?outgoing_leg - outgoing_leg ?departure_buffer_zone - departure_buffer_zone ?walking_time_estimate - walking_time_estimate)
    :precondition
      (and
        (transfer_event_validated ?outgoing_leg)
        (walking_estimate_assigned ?outgoing_leg ?walking_time_estimate)
        (departure_buffer_assigned ?outgoing_leg ?departure_buffer_zone)

          (departure_buffer_reserved ?departure_buffer_zone)

    :effect (departure_buffer_reserved ?departure_buffer_zone)
  )
  (:action mark_outgoing_leg_ready
    :parameters (?outgoing_leg - outgoing_leg ?departure_buffer_zone - departure_buffer_zone ?station_gate - station_gate)
    :precondition
      (and
        (transfer_event_validated ?outgoing_leg)
        (gate_assigned_to_transfer_event ?outgoing_leg ?station_gate)
        (departure_buffer_assigned ?outgoing_leg ?departure_buffer_zone)
        (departure_buffer_reserved ?departure_buffer_zone)
        (not
          (outgoing_leg_prepared ?outgoing_leg)
        )
      )
    :effect
      (and
        (outgoing_leg_prepared ?outgoing_leg)
        (outgoing_leg_ready ?outgoing_leg)
      )
  )
  (:action assign_equipment_to_outgoing_leg
    :parameters (?outgoing_leg - outgoing_leg ?departure_buffer_zone - departure_buffer_zone ?assistance_equipment - assistance_equipment)
    :precondition
      (and
        (transfer_event_validated ?outgoing_leg)
        (departure_buffer_assigned ?outgoing_leg ?departure_buffer_zone)
        (equipment_available ?assistance_equipment)
        (not
          (outgoing_leg_prepared ?outgoing_leg)
        )
      )
    :effect
      (and
        (departure_buffer_equipment_reserved ?departure_buffer_zone)
        (outgoing_leg_prepared ?outgoing_leg)
        (equipment_allocated_to_outgoing_leg ?outgoing_leg ?assistance_equipment)
        (not
          (equipment_available ?assistance_equipment)
        )
      )
  )
  (:action finalize_equipment_assignment_for_outgoing_leg
    :parameters (?outgoing_leg - outgoing_leg ?departure_buffer_zone - departure_buffer_zone ?walking_time_estimate - walking_time_estimate ?assistance_equipment - assistance_equipment)
    :precondition
      (and
        (transfer_event_validated ?outgoing_leg)
        (walking_estimate_assigned ?outgoing_leg ?walking_time_estimate)
        (departure_buffer_assigned ?outgoing_leg ?departure_buffer_zone)
        (departure_buffer_equipment_reserved ?departure_buffer_zone)
        (equipment_allocated_to_outgoing_leg ?outgoing_leg ?assistance_equipment)
        (not
          (outgoing_leg_ready ?outgoing_leg)
        )
      )
    :effect
      (and
        (departure_buffer_reserved ?departure_buffer_zone)
        (outgoing_leg_ready ?outgoing_leg)
        (equipment_available ?assistance_equipment)
        (not
          (equipment_allocated_to_outgoing_leg ?outgoing_leg ?assistance_equipment)
        )
      )
  )
  (:action allocate_transfer_buffer
    :parameters (?incoming_leg - incoming_leg ?outgoing_leg - outgoing_leg ?arrival_buffer_zone - arrival_buffer_zone ?departure_buffer_zone - departure_buffer_zone ?buffer_allocation - buffer_allocation)
    :precondition
      (and
        (incoming_leg_prepared ?incoming_leg)
        (outgoing_leg_prepared ?outgoing_leg)
        (arrival_buffer_assigned ?incoming_leg ?arrival_buffer_zone)
        (departure_buffer_assigned ?outgoing_leg ?departure_buffer_zone)
        (arrival_buffer_reserved ?arrival_buffer_zone)
        (departure_buffer_reserved ?departure_buffer_zone)
        (incoming_leg_ready ?incoming_leg)
        (outgoing_leg_ready ?outgoing_leg)
        (buffer_available ?buffer_allocation)
      )
    :effect
      (and
        (buffer_allocated ?buffer_allocation)
        (allocation_binds_arrival_buffer ?buffer_allocation ?arrival_buffer_zone)
        (allocation_binds_departure_buffer ?buffer_allocation ?departure_buffer_zone)
        (not
          (buffer_available ?buffer_allocation)
        )
      )
  )
  (:action allocate_buffer_with_arrival_confirmation
    :parameters (?incoming_leg - incoming_leg ?outgoing_leg - outgoing_leg ?arrival_buffer_zone - arrival_buffer_zone ?departure_buffer_zone - departure_buffer_zone ?buffer_allocation - buffer_allocation)
    :precondition
      (and
        (incoming_leg_prepared ?incoming_leg)
        (outgoing_leg_prepared ?outgoing_leg)
        (arrival_buffer_assigned ?incoming_leg ?arrival_buffer_zone)
        (departure_buffer_assigned ?outgoing_leg ?departure_buffer_zone)
        (arrival_buffer_equipment_reserved ?arrival_buffer_zone)
        (departure_buffer_reserved ?departure_buffer_zone)
        (not
          (incoming_leg_ready ?incoming_leg)
        )
        (outgoing_leg_ready ?outgoing_leg)
        (buffer_available ?buffer_allocation)
      )
    :effect
      (and
        (buffer_allocated ?buffer_allocation)
        (allocation_binds_arrival_buffer ?buffer_allocation ?arrival_buffer_zone)
        (allocation_binds_departure_buffer ?buffer_allocation ?departure_buffer_zone)
        (allocation_requires_arrival_confirmation ?buffer_allocation)
        (not
          (buffer_available ?buffer_allocation)
        )
      )
  )
  (:action allocate_buffer_with_departure_confirmation
    :parameters (?incoming_leg - incoming_leg ?outgoing_leg - outgoing_leg ?arrival_buffer_zone - arrival_buffer_zone ?departure_buffer_zone - departure_buffer_zone ?buffer_allocation - buffer_allocation)
    :precondition
      (and
        (incoming_leg_prepared ?incoming_leg)
        (outgoing_leg_prepared ?outgoing_leg)
        (arrival_buffer_assigned ?incoming_leg ?arrival_buffer_zone)
        (departure_buffer_assigned ?outgoing_leg ?departure_buffer_zone)
        (arrival_buffer_reserved ?arrival_buffer_zone)
        (departure_buffer_equipment_reserved ?departure_buffer_zone)
        (incoming_leg_ready ?incoming_leg)
        (not
          (outgoing_leg_ready ?outgoing_leg)
        )
        (buffer_available ?buffer_allocation)
      )
    :effect
      (and
        (buffer_allocated ?buffer_allocation)
        (allocation_binds_arrival_buffer ?buffer_allocation ?arrival_buffer_zone)
        (allocation_binds_departure_buffer ?buffer_allocation ?departure_buffer_zone)
        (allocation_requires_departure_confirmation ?buffer_allocation)
        (not
          (buffer_available ?buffer_allocation)
        )
      )
  )
  (:action allocate_buffer_with_both_confirmations
    :parameters (?incoming_leg - incoming_leg ?outgoing_leg - outgoing_leg ?arrival_buffer_zone - arrival_buffer_zone ?departure_buffer_zone - departure_buffer_zone ?buffer_allocation - buffer_allocation)
    :precondition
      (and
        (incoming_leg_prepared ?incoming_leg)
        (outgoing_leg_prepared ?outgoing_leg)

        (departure_buffer_assigned ?outgoing_leg ?departure_buffer_zone)

          (incoming_leg_ready ?incoming_leg)
        )
        (not
          (outgoing_leg_ready ?outgoing_leg)
        )
        (buffer_available ?buffer_allocation)
      )
    :effect
      (and
        (buffer_allocated ?buffer_allocation)
        (allocation_binds_arrival_buffer ?buffer_allocation ?arrival_buffer_zone)
        (allocation_binds_departure_buffer ?buffer_allocation ?departure_buffer_zone)
        (allocation_requires_arrival_confirmation ?buffer_allocation)
        (allocation_requires_departure_confirmation ?buffer_allocation)
        (not
          (buffer_available ?buffer_allocation)
        )
      )
  )
  (:action confirm_allocation_for_arrival_leg

    :precondition
      (and
        (buffer_allocated ?buffer_allocation)
        (incoming_leg_prepared ?incoming_leg)
        (walking_estimate_assigned ?incoming_leg ?walking_time_estimate)
        (not
          (allocation_confirmed ?buffer_allocation)
        )
      )
    :effect (allocation_confirmed ?buffer_allocation)
  )
  (:action reserve_access_path_for_plan

    :precondition
      (and
        (transfer_event_validated ?transfer_plan)
        (plan_assigned_allocation ?transfer_plan ?buffer_allocation)
        (plan_assigned_access_path ?transfer_plan ?access_path)
        (access_path_available ?access_path)
        (buffer_allocated ?buffer_allocation)
        (allocation_confirmed ?buffer_allocation)
        (not
          (access_path_reserved ?access_path)
        )
      )
    :effect
      (and
        (access_path_reserved ?access_path)
        (access_path_binds_allocation ?access_path ?buffer_allocation)
        (not
          (access_path_available ?access_path)
        )
      )
  )
  (:action validate_access_path_and_mark_plan
    :parameters (?transfer_plan - transfer_plan ?access_path - access_path ?buffer_allocation - buffer_allocation ?walking_time_estimate - walking_time_estimate)
    :precondition
      (and
        (transfer_event_validated ?transfer_plan)
        (plan_assigned_access_path ?transfer_plan ?access_path)
        (access_path_reserved ?access_path)
        (access_path_binds_allocation ?access_path ?buffer_allocation)
        (walking_estimate_assigned ?transfer_plan ?walking_time_estimate)
        (not
          (allocation_requires_arrival_confirmation ?buffer_allocation)
        )
        (not
          (plan_access_confirmed ?transfer_plan)
        )
      )
    :effect (plan_access_confirmed ?transfer_plan)
  )
  (:action assign_priority_pass_to_plan
    :parameters (?transfer_plan - transfer_plan ?priority_pass - priority_pass)
    :precondition
      (and
        (transfer_event_validated ?transfer_plan)
        (priority_pass_available ?priority_pass)
        (not
          (plan_has_priority_pass ?transfer_plan)
        )
      )
    :effect
      (and
        (plan_has_priority_pass ?transfer_plan)
        (plan_bound_priority_pass ?transfer_plan ?priority_pass)
        (not
          (priority_pass_available ?priority_pass)
        )
      )
  )
  (:action confirm_plan_access_and_priority
    :parameters (?transfer_plan - transfer_plan ?access_path - access_path ?buffer_allocation - buffer_allocation ?walking_time_estimate - walking_time_estimate ?priority_pass - priority_pass)
    :precondition
      (and
        (transfer_event_validated ?transfer_plan)
        (plan_assigned_access_path ?transfer_plan ?access_path)
