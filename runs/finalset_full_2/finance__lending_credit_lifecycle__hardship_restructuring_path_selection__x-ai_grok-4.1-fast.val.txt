--- domain.pddl (as passed to VAL) ---
(define (domain finance_hardship_restructuring_path_selection)
  (:requirements :strips :typing :negative-preconditions)
  (:types case_entity - object proposal_component - object eligibility_artifact - object loan_account - object hardship_case - loan_account case_officer - case_entity financial_assessment - case_entity risk_officer - case_entity approval_policy - case_entity implementation_method - case_entity hardship_reason_code - case_entity relief_instrument - case_entity senior_approval - case_entity restructuring_option - proposal_component supporting_document - proposal_component external_confirmation - proposal_component primary_eligibility_criterion - eligibility_artifact co_obligor_eligibility_criterion - eligibility_artifact candidate_restructuring_path - eligibility_artifact account_obligor - hardship_case case_proposal_container - hardship_case primary_obligor - account_obligor co_obligor_or_guarantor - account_obligor restructuring_proposal - case_proposal_container)
  (:predicates
    (case_registered ?hardship_case - hardship_case)
    (financial_assessment_completed ?hardship_case - hardship_case)
    (case_officer_assigned ?hardship_case - hardship_case)
    (restructuring_applied ?hardship_case - hardship_case)
    (entity_approved_for_application ?hardship_case - hardship_case)
    (application_prepared ?hardship_case - hardship_case)
    (case_officer_available ?case_officer - case_officer)
    (assigned_case_officer ?hardship_case - hardship_case ?case_officer - case_officer)
    (assessment_available ?financial_assessment - financial_assessment)
    (has_financial_assessment ?hardship_case - hardship_case ?financial_assessment - financial_assessment)
    (risk_officer_available ?risk_officer - risk_officer)
    (entity_assigned_risk_officer ?hardship_case - hardship_case ?risk_officer - risk_officer)
    (restructuring_option_available ?restructuring_option - restructuring_option)
    (primary_obligor_bound_option ?primary_obligor - primary_obligor ?restructuring_option - restructuring_option)
    (co_obligor_bound_option ?co_obligor_or_guarantor - co_obligor_or_guarantor ?restructuring_option - restructuring_option)
    (primary_obligor_has_criterion ?primary_obligor - primary_obligor ?primary_eligibility_criterion - primary_eligibility_criterion)
    (primary_criterion_selected ?primary_eligibility_criterion - primary_eligibility_criterion)
    (primary_criterion_bound ?primary_eligibility_criterion - primary_eligibility_criterion)
    (primary_obligor_eligibility_confirmed ?primary_obligor - primary_obligor)
    (co_obligor_has_criterion ?co_obligor_or_guarantor - co_obligor_or_guarantor ?co_obligor_eligibility_criterion - co_obligor_eligibility_criterion)
    (co_criterion_selected ?co_obligor_eligibility_criterion - co_obligor_eligibility_criterion)
    (co_criterion_bound ?co_obligor_eligibility_criterion - co_obligor_eligibility_criterion)
    (co_obligor_eligibility_confirmed ?co_obligor_or_guarantor - co_obligor_or_guarantor)
    (candidate_path_slot_available ?candidate_restructuring_path - candidate_restructuring_path)
    (candidate_path_created ?candidate_restructuring_path - candidate_restructuring_path)
    (path_includes_primary_criterion ?candidate_restructuring_path - candidate_restructuring_path ?primary_eligibility_criterion - primary_eligibility_criterion)
    (path_includes_co_criterion ?candidate_restructuring_path - candidate_restructuring_path ?co_obligor_eligibility_criterion - co_obligor_eligibility_criterion)
    (path_primary_flag ?candidate_restructuring_path - candidate_restructuring_path)
    (path_co_flag ?candidate_restructuring_path - candidate_restructuring_path)
    (candidate_path_validated ?candidate_restructuring_path - candidate_restructuring_path)
    (proposal_bound_to_primary_obligor ?restructuring_proposal - restructuring_proposal ?primary_obligor - primary_obligor)
    (proposal_bound_to_co_obligor ?restructuring_proposal - restructuring_proposal ?co_obligor_or_guarantor - co_obligor_or_guarantor)
    (proposal_includes_candidate_path ?restructuring_proposal - restructuring_proposal ?candidate_restructuring_path - candidate_restructuring_path)
    (supporting_document_available ?supporting_document - supporting_document)
    (proposal_has_supporting_document ?restructuring_proposal - restructuring_proposal ?supporting_document - supporting_document)
    (supporting_document_attached ?supporting_document - supporting_document)
    (supporting_document_links_to_path ?supporting_document - supporting_document ?candidate_restructuring_path - candidate_restructuring_path)
    (proposal_documentation_verified ?restructuring_proposal - restructuring_proposal)
    (proposal_specialist_review_complete ?restructuring_proposal - restructuring_proposal)
    (proposal_senior_approved ?restructuring_proposal - restructuring_proposal)
    (proposal_policy_assigned ?restructuring_proposal - restructuring_proposal)
    (proposal_policy_endorsed ?restructuring_proposal - restructuring_proposal)
    (proposal_ready_for_finalization ?restructuring_proposal - restructuring_proposal)
    (proposal_final_validation_passed ?restructuring_proposal - restructuring_proposal)
    (external_confirmation_available ?external_confirmation - external_confirmation)
    (proposal_has_external_confirmation ?restructuring_proposal - restructuring_proposal ?external_confirmation - external_confirmation)
    (external_confirmation_assigned ?restructuring_proposal - restructuring_proposal)
    (proposal_specialist_check_started ?restructuring_proposal - restructuring_proposal)
    (proposal_specialist_check_passed ?restructuring_proposal - restructuring_proposal)
    (approval_policy_available ?approval_policy - approval_policy)
    (proposal_assigned_policy ?restructuring_proposal - restructuring_proposal ?approval_policy - approval_policy)
    (implementation_method_available ?implementation_method - implementation_method)
    (proposal_assigned_implementation_method ?restructuring_proposal - restructuring_proposal ?implementation_method - implementation_method)
    (relief_instrument_available ?relief_instrument - relief_instrument)
    (proposal_attached_relief_instrument ?restructuring_proposal - restructuring_proposal ?relief_instrument - relief_instrument)
    (senior_approval_available ?senior_approval - senior_approval)
    (proposal_attached_senior_approval ?restructuring_proposal - restructuring_proposal ?senior_approval - senior_approval)
    (hardship_reason_code_available ?hardship_reason_code - hardship_reason_code)
    (case_hardship_reason_assigned ?hardship_case - hardship_case ?hardship_reason_code - hardship_reason_code)
    (primary_obligor_ready ?primary_obligor - primary_obligor)
    (co_obligor_ready ?co_obligor_or_guarantor - co_obligor_or_guarantor)
    (proposal_marked_ready_for_application ?restructuring_proposal - restructuring_proposal)
  )
  (:action register_hardship_case
    :parameters (?hardship_case - hardship_case)
    :precondition
      (and
        (not
          (case_registered ?hardship_case)
        )
        (not
          (restructuring_applied ?hardship_case)
        )
      )
    :effect (case_registered ?hardship_case)
  )
  (:action assign_case_officer
    :parameters (?hardship_case - hardship_case ?case_officer - case_officer)
    :precondition
      (and
        (case_registered ?hardship_case)
        (not
          (case_officer_assigned ?hardship_case)
        )
        (case_officer_available ?case_officer)
      )
    :effect
      (and
        (case_officer_assigned ?hardship_case)
        (assigned_case_officer ?hardship_case ?case_officer)
        (not
          (case_officer_available ?case_officer)
        )
      )
  )
  (:action attach_financial_assessment
    :parameters (?hardship_case - hardship_case ?financial_assessment - financial_assessment)
    :precondition
      (and
        (case_registered ?hardship_case)
        (case_officer_assigned ?hardship_case)
        (assessment_available ?financial_assessment)
      )
    :effect
      (and
        (has_financial_assessment ?hardship_case ?financial_assessment)
        (not
          (assessment_available ?financial_assessment)
        )
      )
  )
  (:action finalize_financial_assessment
    :parameters (?hardship_case - hardship_case ?financial_assessment - financial_assessment)
    :precondition
      (and
        (case_registered ?hardship_case)
        (case_officer_assigned ?hardship_case)
        (has_financial_assessment ?hardship_case ?financial_assessment)
        (not
          (financial_assessment_completed ?hardship_case)
        )
      )
    :effect (financial_assessment_completed ?hardship_case)
  )
  (:action detach_financial_assessment
    :parameters (?hardship_case - hardship_case ?financial_assessment - financial_assessment)
    :precondition
      (and
        (has_financial_assessment ?hardship_case ?financial_assessment)
      )
    :effect
      (and
        (assessment_available ?financial_assessment)
        (not
          (has_financial_assessment ?hardship_case ?financial_assessment)
        )
      )
  )
  (:action assign_risk_officer
    :parameters (?hardship_case - hardship_case ?risk_officer - risk_officer)
    :precondition
      (and
        (financial_assessment_completed ?hardship_case)
        (risk_officer_available ?risk_officer)
      )
    :effect
      (and
        (entity_assigned_risk_officer ?hardship_case ?risk_officer)
        (not
          (risk_officer_available ?risk_officer)
        )
      )
  )
  (:action unassign_risk_officer
    :parameters (?hardship_case - hardship_case ?risk_officer - risk_officer)
    :precondition
      (and
        (entity_assigned_risk_officer ?hardship_case ?risk_officer)
      )
    :effect
      (and
        (risk_officer_available ?risk_officer)
        (not
          (entity_assigned_risk_officer ?hardship_case ?risk_officer)
        )
      )
  )
  (:action attach_relief_instrument
    :parameters (?restructuring_proposal - restructuring_proposal ?relief_instrument - relief_instrument)
    :precondition
      (and
        (financial_assessment_completed ?restructuring_proposal)
        (relief_instrument_available ?relief_instrument)
      )
    :effect
      (and
        (proposal_attached_relief_instrument ?restructuring_proposal ?relief_instrument)
        (not
          (relief_instrument_available ?relief_instrument)
        )
      )
  )
  (:action detach_relief_instrument
    :parameters (?restructuring_proposal - restructuring_proposal ?relief_instrument - relief_instrument)
    :precondition
      (and
        (proposal_attached_relief_instrument ?restructuring_proposal ?relief_instrument)
      )
    :effect
      (and
        (relief_instrument_available ?relief_instrument)
        (not
          (proposal_attached_relief_instrument ?restructuring_proposal ?relief_instrument)
        )
      )
  )
  (:action attach_senior_approval
    :parameters (?restructuring_proposal - restructuring_proposal ?senior_approval - senior_approval)
    :precondition
      (and
        (financial_assessment_completed ?restructuring_proposal)
        (senior_approval_available ?senior_approval)
      )
    :effect
      (and
        (proposal_attached_senior_approval ?restructuring_proposal ?senior_approval)
        (not
          (senior_approval_available ?senior_approval)
        )
      )
  )
  (:action detach_senior_approval
    :parameters (?restructuring_proposal - restructuring_proposal ?senior_approval - senior_approval)
    :precondition
      (and
        (proposal_attached_senior_approval ?restructuring_proposal ?senior_approval)
      )
    :effect
      (and
        (senior_approval_available ?senior_approval)
        (not
          (proposal_attached_senior_approval ?restructuring_proposal ?senior_approval)
        )
      )
  )
  (:action select_primary_eligibility_criterion
    :parameters (?primary_obligor - primary_obligor ?primary_eligibility_criterion - primary_eligibility_criterion ?financial_assessment - financial_assessment)
    :precondition
      (and
        (financial_assessment_completed ?primary_obligor)
        (has_financial_assessment ?primary_obligor ?financial_assessment)
        (primary_obligor_has_criterion ?primary_obligor ?primary_eligibility_criterion)
        (not
          (primary_criterion_selected ?primary_eligibility_criterion)
        )
        (not
          (primary_criterion_bound ?primary_eligibility_criterion)
        )
      )
    :effect (primary_criterion_selected ?primary_eligibility_criterion)
  )
  (:action confirm_primary_criterion_and_lock_obligor
    :parameters (?primary_obligor - primary_obligor ?primary_eligibility_criterion - primary_eligibility_criterion ?risk_officer - risk_officer)
    :precondition
      (and
        (financial_assessment_completed ?primary_obligor)
        (entity_assigned_risk_officer ?primary_obligor ?risk_officer)
        (primary_obligor_has_criterion ?primary_obligor ?primary_eligibility_criterion)
        (primary_criterion_selected ?primary_eligibility_criterion)
        (not
          (primary_obligor_ready ?primary_obligor)
        )
      )
    :effect
      (and
        (primary_obligor_ready ?primary_obligor)
        (primary_obligor_eligibility_confirmed ?primary_obligor)
      )
  )
  (:action bind_restructuring_option_to_primary_obligor
    :parameters (?primary_obligor - primary_obligor ?primary_eligibility_criterion - primary_eligibility_criterion ?restructuring_option - restructuring_option)
    :precondition
      (and
        (financial_assessment_completed ?primary_obligor)
        (primary_obligor_has_criterion ?primary_obligor ?primary_eligibility_criterion)
        (restructuring_option_available ?restructuring_option)
        (not
          (primary_obligor_ready ?primary_obligor)
        )
      )
    :effect
      (and
        (primary_criterion_bound ?primary_eligibility_criterion)
        (primary_obligor_ready ?primary_obligor)
        (primary_obligor_bound_option ?primary_obligor ?restructuring_option)
        (not
          (restructuring_option_available ?restructuring_option)
        )
      )
  )
  (:action swap_primary_obligor_bound_option
    :parameters (?primary_obligor - primary_obligor ?primary_eligibility_criterion - primary_eligibility_criterion ?financial_assessment - financial_assessment ?restructuring_option - restructuring_option)
    :precondition
      (and
        (financial_assessment_completed ?primary_obligor)
        (has_financial_assessment ?primary_obligor ?financial_assessment)
        (primary_obligor_has_criterion ?primary_obligor ?primary_eligibility_criterion)
        (primary_criterion_bound ?primary_eligibility_criterion)
        (primary_obligor_bound_option ?primary_obligor ?restructuring_option)
        (not
          (primary_obligor_eligibility_confirmed ?primary_obligor)
        )
      )
    :effect
      (and
        (primary_criterion_selected ?primary_eligibility_criterion)
        (primary_obligor_eligibility_confirmed ?primary_obligor)
        (restructuring_option_available ?restructuring_option)
        (not
          (primary_obligor_bound_option ?primary_obligor ?restructuring_option)
        )
      )
  )
  (:action select_co_obligor_eligibility_criterion
    :parameters (?co_obligor_or_guarantor - co_obligor_or_guarantor ?co_obligor_eligibility_criterion - co_obligor_eligibility_criterion ?financial_assessment - financial_assessment)
    :precondition
      (and
        (financial_assessment_completed ?co_obligor_or_guarantor)
        (has_financial_assessment ?co_obligor_or_guarantor ?financial_assessment)
        (co_obligor_has_criterion ?co_obligor_or_guarantor ?co_obligor_eligibility_criterion)
        (not
          (co_criterion_selected ?co_obligor_eligibility_criterion)
        )
        (not
          (co_criterion_bound ?co_obligor_eligibility_criterion)
        )
      )
    :effect (co_criterion_selected ?co_obligor_eligibility_criterion)
  )
  (:action confirm_co_criterion_and_lock_co_obligor
    :parameters (?co_obligor_or_guarantor - co_obligor_or_guarantor ?co_obligor_eligibility_criterion - co_obligor_eligibility_criterion ?risk_officer - risk_officer)
    :precondition
      (and
        (financial_assessment_completed ?co_obligor_or_guarantor)
        (entity_assigned_risk_officer ?co_obligor_or_guarantor ?risk_officer)
        (co_obligor_has_criterion ?co_obligor_or_guarantor ?co_obligor_eligibility_criterion)
        (co_criterion_selected ?co_obligor_eligibility_criterion)
        (not
          (co_obligor_ready ?co_obligor_or_guarantor)
        )
      )
    :effect
      (and
        (co_obligor_ready ?co_obligor_or_guarantor)
        (co_obligor_eligibility_confirmed ?co_obligor_or_guarantor)
      )
  )
  (:action bind_restructuring_option_to_co_obligor
    :parameters (?co_obligor_or_guarantor - co_obligor_or_guarantor ?co_obligor_eligibility_criterion - co_obligor_eligibility_criterion ?restructuring_option - restructuring_option)
    :precondition
      (and
        (financial_assessment_completed ?co_obligor_or_guarantor)
        (co_obligor_has_criterion ?co_obligor_or_guarantor ?co_obligor_eligibility_criterion)
        (restructuring_option_available ?restructuring_option)
        (not
          (co_obligor_ready ?co_obligor_or_guarantor)
        )
      )
    :effect
      (and
        (co_criterion_bound ?co_obligor_eligibility_criterion)
        (co_obligor_ready ?co_obligor_or_guarantor)
        (co_obligor_bound_option ?co_obligor_or_guarantor ?restructuring_option)
        (not
          (restructuring_option_available ?restructuring_option)
        )
      )
  )
  (:action swap_co_obligor_bound_option
    :parameters (?co_obligor_or_guarantor - co_obligor_or_guarantor ?co_obligor_eligibility_criterion - co_obligor_eligibility_criterion ?financial_assessment - financial_assessment ?restructuring_option - restructuring_option)
    :precondition
      (and
        (financial_assessment_completed ?co_obligor_or_guarantor)
        (has_financial_assessment ?co_obligor_or_guarantor ?financial_assessment)
        (co_obligor_has_criterion ?co_obligor_or_guarantor ?co_obligor_eligibility_criterion)
        (co_criterion_bound ?co_obligor_eligibility_criterion)
        (co_obligor_bound_option ?co_obligor_or_guarantor ?restructuring_option)
        (not
          (co_obligor_eligibility_confirmed ?co_obligor_or_guarantor)
        )
      )
    :effect
      (and
        (co_criterion_selected ?co_obligor_eligibility_criterion)
        (co_obligor_eligibility_confirmed ?co_obligor_or_guarantor)
        (restructuring_option_available ?restructuring_option)
        (not
          (co_obligor_bound_option ?co_obligor_or_guarantor ?restructuring_option)
        )
      )
  )
  (:action compose_candidate_path_no_bindings
    :parameters (?primary_obligor - primary_obligor ?co_obligor_or_guarantor - co_obligor_or_guarantor ?primary_eligibility_criterion - primary_eligibility_criterion ?co_obligor_eligibility_criterion - co_obligor_eligibility_criterion ?candidate_restructuring_path - candidate_restructuring_path)
    :precondition
      (and
        (primary_obligor_ready ?primary_obligor)
        (co_obligor_ready ?co_obligor_or_guarantor)
        (primary_obligor_has_criterion ?primary_obligor ?primary_eligibility_criterion)
        (co_obligor_has_criterion ?co_obligor_or_guarantor ?co_obligor_eligibility_criterion)
        (primary_criterion_selected ?primary_eligibility_criterion)
        (co_criterion_selected ?co_obligor_eligibility_criterion)
        (primary_obligor_eligibility_confirmed ?primary_obligor)
        (co_obligor_eligibility_confirmed ?co_obligor_or_guarantor)
        (candidate_path_slot_available ?candidate_restructuring_path)
      )
    :effect
      (and
        (candidate_path_created ?candidate_restructuring_path)
        (path_includes_primary_criterion ?candidate_restructuring_path ?primary_eligibility_criterion)
        (path_includes_co_criterion ?candidate_restructuring_path ?co_obligor_eligibility_criterion)
        (not
          (candidate_path_slot_available ?candidate_restructuring_path)
        )
      )
  )
  (:action compose_candidate_path_primary_bound
    :parameters (?primary_obligor - primary_obligor ?co_obligor_or_guarantor - co_obligor_or_guarantor ?primary_eligibility_criterion - primary_eligibility_criterion ?co_obligor_eligibility_criterion - co_obligor_eligibility_criterion ?candidate_restructuring_path - candidate_restructuring_path)
    :precondition
      (and
        (primary_obligor_ready ?primary_obligor)
        (co_obligor_ready ?co_obligor_or_guarantor)
        (primary_obligor_has_criterion ?primary_obligor ?primary_eligibility_criterion)
        (co_obligor_has_criterion ?co_obligor_or_guarantor ?co_obligor_eligibility_criterion)
        (primary_criterion_bound ?primary_eligibility_criterion)
        (co_criterion_selected ?co_obligor_eligibility_criterion)
        (not
          (primary_obligor_eligibility_confirmed ?primary_obligor)
        )
        (co_obligor_eligibility_confirmed ?co_obligor_or_guarantor)
        (candidate_path_slot_available ?candidate_restructuring_path)
      )
    :effect
      (and
        (candidate_path_created ?candidate_restructuring_path)
        (path_includes_primary_criterion ?candidate_restructuring_path ?primary_eligibility_criterion)
        (path_includes_co_criterion ?candidate_restructuring_path ?co_obligor_eligibility_criterion)
        (path_primary_flag ?candidate_restructuring_path)
        (not
          (candidate_path_slot_available ?candidate_restructuring_path)
        )
      )
  )
  (:action compose_candidate_path_co_bound
    :parameters (?primary_obligor - primary_obligor ?co_obligor_or_guarantor - co_obligor_or_guarantor ?primary_eligibility_criterion - primary_eligibility_criterion ?co_obligor_eligibility_criterion - co_obligor_eligibility_criterion ?candidate_restructuring_path - candidate_restructuring_path)
    :precondition
      (and
        (primary_obligor_ready ?primary_obligor)
        (co_obligor_ready ?co_obligor_or_guarantor)
        (primary_obligor_has_criterion ?primary_obligor ?primary_eligibility_criterion)
        (co_obligor_has_criterion ?co_obligor_or_guarantor ?co_obligor_eligibility_criterion)
        (primary_criterion_selected ?primary_eligibility_criterion)
        (co_criterion_bound ?co_obligor_eligibility_criterion)
        (primary_obligor_eligibility_confirmed ?primary_obligor)
        (not
          (co_obligor_eligibility_confirmed ?co_obligor_or_guarantor)
        )
        (candidate_path_slot_available ?candidate_restructuring_path)
      )
    :effect
      (and
        (candidate_path_created ?candidate_restructuring_path)
        (path_includes_primary_criterion ?candidate_restructuring_path ?primary_eligibility_criterion)
        (path_includes_co_criterion ?candidate_restructuring_path ?co_obligor_eligibility_criterion)
        (path_co_flag ?candidate_restructuring_path)
        (not
          (candidate_path_slot_available ?candidate_restructuring_path)
        )
      )
  )
  (:action compose_candidate_path_primary_and_co_bound
    :parameters (?primary_obligor - primary_obligor ?co_obligor_or_guarantor - co_obligor_or_guarantor ?primary_eligibility_criterion - primary_eligibility_criterion ?co_obligor_eligibility_criterion - co_obligor_eligibility_criterion ?candidate_restructuring_path - candidate_restructuring_path)
    :precondition
      (and
        (primary_obligor_ready ?primary_obligor)
        (co_obligor_ready ?co_obligor_or_guarantor)
        (primary_obligor_has_criterion ?primary_obligor ?primary_eligibility_criterion)
        (co_obligor_has_criterion ?co_obligor_or_guarantor ?co_obligor_eligibility_criterion)
        (primary_criterion_bound ?primary_eligibility_criterion)
        (co_criterion_bound ?co_obligor_eligibility_criterion)
        (not
          (primary_obligor_eligibility_confirmed ?primary_obligor)
        )
        (not
          (co_obligor_eligibility_confirmed ?co_obligor_or_guarantor)
