(define (domain pharmaceutics_adverse_event_case_triage_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity - object support_entity_type - entity asset_category - entity organization_entity_type - entity case_root_type - entity adverse_event_case - case_root_type response_resource - support_entity_type report_document - support_entity_type responder - support_entity_type regulatory_document - support_entity_type supplier - support_entity_type notification_template - support_entity_type technical_resource - support_entity_type external_expert - support_entity_type mitigation_resource - asset_category laboratory_sample - asset_category regulatory_authority - asset_category manufacturing_site_unit - organization_entity_type distribution_node - organization_entity_type remediation_order - organization_entity_type site_case_group - adverse_event_case workflow_case_group - adverse_event_case manufacturing_site_case - site_case_group distribution_site_case - site_case_group investigation_workstream - workflow_case_group)

  (:predicates
    (case_intake_recorded ?adverse_event_case - adverse_event_case)
    (case_under_investigation ?adverse_event_case - adverse_event_case)
    (case_triaged ?adverse_event_case - adverse_event_case)
    (case_closed ?adverse_event_case - adverse_event_case)
    (case_recovered ?adverse_event_case - adverse_event_case)
    (case_notified ?adverse_event_case - adverse_event_case)
    (response_resource_available ?response_resource - response_resource)
    (case_assigned_response_resource ?adverse_event_case - adverse_event_case ?response_resource - response_resource)
    (report_document_available ?report_document - report_document)
    (case_has_report_document ?adverse_event_case - adverse_event_case ?report_document - report_document)
    (responder_available ?responder - responder)
    (case_assigned_responder ?adverse_event_case - adverse_event_case ?responder - responder)
    (mitigation_resource_available ?mitigation_resource - mitigation_resource)
    (manufacturing_case_assigned_mitigation_resource ?manufacturing_site_case - manufacturing_site_case ?mitigation_resource - mitigation_resource)
    (distribution_case_assigned_mitigation_resource ?distribution_site_case - distribution_site_case ?mitigation_resource - mitigation_resource)
    (manufacturing_case_assigned_site_unit ?manufacturing_site_case - manufacturing_site_case ?manufacturing_site_unit - manufacturing_site_unit)
    (site_unit_flagged_for_quarantine ?manufacturing_site_unit - manufacturing_site_unit)
    (site_unit_quarantined ?manufacturing_site_unit - manufacturing_site_unit)
    (manufacturing_containment_verified ?manufacturing_site_case - manufacturing_site_case)
    (distribution_case_assigned_distribution_node ?distribution_site_case - distribution_site_case ?distribution_node - distribution_node)
    (distribution_node_flagged_for_quarantine ?distribution_node - distribution_node)
    (distribution_node_quarantined ?distribution_node - distribution_node)
    (distribution_containment_verified ?distribution_site_case - distribution_site_case)
    (remediation_order_planned ?remediation_action - remediation_order)
    (remediation_order_prepared ?remediation_action - remediation_order)
    (remediation_order_linked_to_site_unit ?remediation_action - remediation_order ?manufacturing_site_unit - manufacturing_site_unit)
    (remediation_order_linked_to_distribution_node ?remediation_action - remediation_order ?distribution_node - distribution_node)
    (manufacturing_scope_confirmed ?remediation_action - remediation_order)
    (distribution_scope_confirmed ?remediation_action - remediation_order)
    (remediation_order_issued ?remediation_action - remediation_order)
    (workstream_scoped_to_manufacturing_site_case ?investigation_workstream - investigation_workstream ?manufacturing_site_case - manufacturing_site_case)
    (workstream_scoped_to_distribution_site_case ?investigation_workstream - investigation_workstream ?distribution_site_case - distribution_site_case)
    (workstream_linked_to_remediation_order ?investigation_workstream - investigation_workstream ?remediation_action - remediation_order)
    (laboratory_sample_available ?laboratory_sample - laboratory_sample)
    (workstream_has_laboratory_sample ?investigation_workstream - investigation_workstream ?laboratory_sample - laboratory_sample)
    (laboratory_sample_collected ?laboratory_sample - laboratory_sample)
    (sample_linked_to_remediation_order ?laboratory_sample - laboratory_sample ?remediation_action - remediation_order)
    (sample_assessment_in_progress ?investigation_workstream - investigation_workstream)
    (technical_review_complete ?investigation_workstream - investigation_workstream)
    (specialist_review_complete ?investigation_workstream - investigation_workstream)
    (regulatory_document_assigned ?investigation_workstream - investigation_workstream)
    (regulatory_package_submitted ?investigation_workstream - investigation_workstream)
    (specialist_clearance_granted ?investigation_workstream - investigation_workstream)
    (closure_review_complete ?investigation_workstream - investigation_workstream)
    (regulatory_authority_available ?regulatory_authority - regulatory_authority)
    (workstream_assigned_regulatory_authority ?investigation_workstream - investigation_workstream ?regulatory_authority - regulatory_authority)
    (authority_engaged ?investigation_workstream - investigation_workstream)
    (authority_acknowledged ?investigation_workstream - investigation_workstream)
    (authority_clearance_granted ?investigation_workstream - investigation_workstream)
    (regulatory_document_available ?regulatory_document - regulatory_document)
    (workstream_assigned_regulatory_document ?investigation_workstream - investigation_workstream ?regulatory_document - regulatory_document)
    (supplier_available ?supplier - supplier)
    (workstream_assigned_supplier ?investigation_workstream - investigation_workstream ?supplier - supplier)
    (technical_resource_available ?technical_resource - technical_resource)
    (workstream_assigned_technical_resource ?investigation_workstream - investigation_workstream ?technical_resource - technical_resource)
    (external_expert_available ?external_expert - external_expert)
    (workstream_assigned_external_expert ?investigation_workstream - investigation_workstream ?external_expert - external_expert)
    (notification_template_available ?notification_template - notification_template)
    (case_notification_template_assigned ?adverse_event_case - adverse_event_case ?notification_template - notification_template)
    (manufacturing_containment_active ?manufacturing_site_case - manufacturing_site_case)
    (distribution_containment_active ?distribution_site_case - distribution_site_case)
    (recovery_finalized ?investigation_workstream - investigation_workstream)
  )
  (:action record_adverse_event_case_intake
    :parameters (?adverse_event_case - adverse_event_case)
    :precondition
      (and
        (not
          (case_intake_recorded ?adverse_event_case)
        )
        (not
          (case_closed ?adverse_event_case)
        )
      )
    :effect (case_intake_recorded ?adverse_event_case)
  )
  (:action triage_case_with_response_resource
    :parameters (?adverse_event_case - adverse_event_case ?response_resource - response_resource)
    :precondition
      (and
        (case_intake_recorded ?adverse_event_case)
        (not
          (case_triaged ?adverse_event_case)
        )
        (response_resource_available ?response_resource)
      )
    :effect
      (and
        (case_triaged ?adverse_event_case)
        (case_assigned_response_resource ?adverse_event_case ?response_resource)
        (not
          (response_resource_available ?response_resource)
        )
      )
  )
  (:action attach_report_document_to_case
    :parameters (?adverse_event_case - adverse_event_case ?report_document - report_document)
    :precondition
      (and
        (case_intake_recorded ?adverse_event_case)
        (case_triaged ?adverse_event_case)
        (report_document_available ?report_document)
      )
    :effect
      (and
        (case_has_report_document ?adverse_event_case ?report_document)
        (not
          (report_document_available ?report_document)
        )
      )
  )
  (:action escalate_case_to_investigation
    :parameters (?adverse_event_case - adverse_event_case ?report_document - report_document)
    :precondition
      (and
        (case_intake_recorded ?adverse_event_case)
        (case_triaged ?adverse_event_case)
        (case_has_report_document ?adverse_event_case ?report_document)
        (not
          (case_under_investigation ?adverse_event_case)
        )
      )
    :effect (case_under_investigation ?adverse_event_case)
  )
  (:action detach_report_document_from_case
    :parameters (?adverse_event_case - adverse_event_case ?report_document - report_document)
    :precondition
      (and
        (case_has_report_document ?adverse_event_case ?report_document)
      )
    :effect
      (and
        (report_document_available ?report_document)
        (not
          (case_has_report_document ?adverse_event_case ?report_document)
        )
      )
  )
  (:action assign_responder_to_case
    :parameters (?adverse_event_case - adverse_event_case ?responder - responder)
    :precondition
      (and
        (case_under_investigation ?adverse_event_case)
        (responder_available ?responder)
      )
    :effect
      (and
        (case_assigned_responder ?adverse_event_case ?responder)
        (not
          (responder_available ?responder)
        )
      )
  )
  (:action release_responder_from_case
    :parameters (?adverse_event_case - adverse_event_case ?responder - responder)
    :precondition
      (and
        (case_assigned_responder ?adverse_event_case ?responder)
      )
    :effect
      (and
        (responder_available ?responder)
        (not
          (case_assigned_responder ?adverse_event_case ?responder)
        )
      )
  )
  (:action assign_technical_resource_to_workstream
    :parameters (?investigation_workstream - investigation_workstream ?technical_resource - technical_resource)
    :precondition
      (and
        (case_under_investigation ?investigation_workstream)
        (technical_resource_available ?technical_resource)
      )
    :effect
      (and
        (workstream_assigned_technical_resource ?investigation_workstream ?technical_resource)
        (not
          (technical_resource_available ?technical_resource)
        )
      )
  )
  (:action release_technical_resource_from_workstream
    :parameters (?investigation_workstream - investigation_workstream ?technical_resource - technical_resource)
    :precondition
      (and
        (workstream_assigned_technical_resource ?investigation_workstream ?technical_resource)
      )
    :effect
      (and
        (technical_resource_available ?technical_resource)
        (not
          (workstream_assigned_technical_resource ?investigation_workstream ?technical_resource)
        )
      )
  )
  (:action assign_external_expert_to_workstream
    :parameters (?investigation_workstream - investigation_workstream ?external_expert - external_expert)
    :precondition
      (and
        (case_under_investigation ?investigation_workstream)
        (external_expert_available ?external_expert)
      )
    :effect
      (and
        (workstream_assigned_external_expert ?investigation_workstream ?external_expert)
        (not
          (external_expert_available ?external_expert)
        )
      )
  )
  (:action release_external_expert_from_workstream
    :parameters (?investigation_workstream - investigation_workstream ?external_expert - external_expert)
    :precondition
      (and
        (workstream_assigned_external_expert ?investigation_workstream ?external_expert)
      )
    :effect
      (and
        (external_expert_available ?external_expert)
        (not
          (workstream_assigned_external_expert ?investigation_workstream ?external_expert)
        )
      )
  )
  (:action flag_manufacturing_site_unit_for_quarantine
    :parameters (?manufacturing_site_case - manufacturing_site_case ?manufacturing_site_unit - manufacturing_site_unit ?report_document - report_document)
    :precondition
      (and
        (case_under_investigation ?manufacturing_site_case)
        (case_has_report_document ?manufacturing_site_case ?report_document)
        (manufacturing_case_assigned_site_unit ?manufacturing_site_case ?manufacturing_site_unit)
        (not
          (site_unit_flagged_for_quarantine ?manufacturing_site_unit)
        )
        (not
          (site_unit_quarantined ?manufacturing_site_unit)
        )
      )
    :effect (site_unit_flagged_for_quarantine ?manufacturing_site_unit)
  )
  (:action activate_manufacturing_containment
    :parameters (?manufacturing_site_case - manufacturing_site_case ?manufacturing_site_unit - manufacturing_site_unit ?responder - responder)
    :precondition
      (and
        (case_under_investigation ?manufacturing_site_case)
        (case_assigned_responder ?manufacturing_site_case ?responder)
        (manufacturing_case_assigned_site_unit ?manufacturing_site_case ?manufacturing_site_unit)
        (site_unit_flagged_for_quarantine ?manufacturing_site_unit)
        (not
          (manufacturing_containment_active ?manufacturing_site_case)
        )
      )
    :effect
      (and
        (manufacturing_containment_active ?manufacturing_site_case)
        (manufacturing_containment_verified ?manufacturing_site_case)
      )
  )
  (:action apply_manufacturing_quarantine
    :parameters (?manufacturing_site_case - manufacturing_site_case ?manufacturing_site_unit - manufacturing_site_unit ?mitigation_resource - mitigation_resource)
    :precondition
      (and
        (case_under_investigation ?manufacturing_site_case)
        (manufacturing_case_assigned_site_unit ?manufacturing_site_case ?manufacturing_site_unit)
        (mitigation_resource_available ?mitigation_resource)
        (not
          (manufacturing_containment_active ?manufacturing_site_case)
        )
      )
    :effect
      (and
        (site_unit_quarantined ?manufacturing_site_unit)
        (manufacturing_containment_active ?manufacturing_site_case)
        (manufacturing_case_assigned_mitigation_resource ?manufacturing_site_case ?mitigation_resource)
        (not
          (mitigation_resource_available ?mitigation_resource)
        )
      )
  )
  (:action complete_manufacturing_mitigation
    :parameters (?manufacturing_site_case - manufacturing_site_case ?manufacturing_site_unit - manufacturing_site_unit ?report_document - report_document ?mitigation_resource - mitigation_resource)
    :precondition
      (and
        (case_under_investigation ?manufacturing_site_case)
        (case_has_report_document ?manufacturing_site_case ?report_document)
        (manufacturing_case_assigned_site_unit ?manufacturing_site_case ?manufacturing_site_unit)
        (site_unit_quarantined ?manufacturing_site_unit)
        (manufacturing_case_assigned_mitigation_resource ?manufacturing_site_case ?mitigation_resource)
        (not
          (manufacturing_containment_verified ?manufacturing_site_case)
        )
      )
    :effect
      (and
        (site_unit_flagged_for_quarantine ?manufacturing_site_unit)
        (manufacturing_containment_verified ?manufacturing_site_case)
        (mitigation_resource_available ?mitigation_resource)
        (not
          (manufacturing_case_assigned_mitigation_resource ?manufacturing_site_case ?mitigation_resource)
        )
      )
  )
  (:action flag_distribution_node_for_quarantine
    :parameters (?distribution_site_case - distribution_site_case ?distribution_node - distribution_node ?report_document - report_document)
    :precondition
      (and
        (case_under_investigation ?distribution_site_case)
        (case_has_report_document ?distribution_site_case ?report_document)
        (distribution_case_assigned_distribution_node ?distribution_site_case ?distribution_node)
        (not
          (distribution_node_flagged_for_quarantine ?distribution_node)
        )
        (not
          (distribution_node_quarantined ?distribution_node)
        )
      )
    :effect (distribution_node_flagged_for_quarantine ?distribution_node)
  )
  (:action activate_distribution_containment
    :parameters (?distribution_site_case - distribution_site_case ?distribution_node - distribution_node ?responder - responder)
    :precondition
      (and
        (case_under_investigation ?distribution_site_case)
        (case_assigned_responder ?distribution_site_case ?responder)
        (distribution_case_assigned_distribution_node ?distribution_site_case ?distribution_node)
        (distribution_node_flagged_for_quarantine ?distribution_node)
        (not
          (distribution_containment_active ?distribution_site_case)
        )
      )
    :effect
      (and
        (distribution_containment_active ?distribution_site_case)
        (distribution_containment_verified ?distribution_site_case)
      )
  )
  (:action apply_distribution_quarantine
    :parameters (?distribution_site_case - distribution_site_case ?distribution_node - distribution_node ?mitigation_resource - mitigation_resource)
    :precondition
      (and
        (case_under_investigation ?distribution_site_case)
        (distribution_case_assigned_distribution_node ?distribution_site_case ?distribution_node)
        (mitigation_resource_available ?mitigation_resource)
        (not
          (distribution_containment_active ?distribution_site_case)
        )
      )
    :effect
      (and
        (distribution_node_quarantined ?distribution_node)
        (distribution_containment_active ?distribution_site_case)
        (distribution_case_assigned_mitigation_resource ?distribution_site_case ?mitigation_resource)
        (not
          (mitigation_resource_available ?mitigation_resource)
        )
      )
  )
  (:action complete_distribution_mitigation
    :parameters (?distribution_site_case - distribution_site_case ?distribution_node - distribution_node ?report_document - report_document ?mitigation_resource - mitigation_resource)
    :precondition
      (and
        (case_under_investigation ?distribution_site_case)
        (case_has_report_document ?distribution_site_case ?report_document)
        (distribution_case_assigned_distribution_node ?distribution_site_case ?distribution_node)
        (distribution_node_quarantined ?distribution_node)
        (distribution_case_assigned_mitigation_resource ?distribution_site_case ?mitigation_resource)
        (not
          (distribution_containment_verified ?distribution_site_case)
        )
      )
    :effect
      (and
        (distribution_node_flagged_for_quarantine ?distribution_node)
        (distribution_containment_verified ?distribution_site_case)
        (mitigation_resource_available ?mitigation_resource)
        (not
          (distribution_case_assigned_mitigation_resource ?distribution_site_case ?mitigation_resource)
        )
      )
  )
  (:action prepare_combined_remediation_order
    :parameters (?manufacturing_site_case - manufacturing_site_case ?distribution_site_case - distribution_site_case ?manufacturing_site_unit - manufacturing_site_unit ?distribution_node - distribution_node ?remediation_action - remediation_order)
    :precondition
      (and
        (manufacturing_containment_active ?manufacturing_site_case)
        (distribution_containment_active ?distribution_site_case)
        (manufacturing_case_assigned_site_unit ?manufacturing_site_case ?manufacturing_site_unit)
        (distribution_case_assigned_distribution_node ?distribution_site_case ?distribution_node)
        (site_unit_flagged_for_quarantine ?manufacturing_site_unit)
        (distribution_node_flagged_for_quarantine ?distribution_node)
        (manufacturing_containment_verified ?manufacturing_site_case)
        (distribution_containment_verified ?distribution_site_case)
        (remediation_order_planned ?remediation_action)
      )
    :effect
      (and
        (remediation_order_prepared ?remediation_action)
        (remediation_order_linked_to_site_unit ?remediation_action ?manufacturing_site_unit)
        (remediation_order_linked_to_distribution_node ?remediation_action ?distribution_node)
        (not
          (remediation_order_planned ?remediation_action)
        )
      )
  )
  (:action prepare_manufacturing_remediation_order
    :parameters (?manufacturing_site_case - manufacturing_site_case ?distribution_site_case - distribution_site_case ?manufacturing_site_unit - manufacturing_site_unit ?distribution_node - distribution_node ?remediation_action - remediation_order)
    :precondition
      (and
        (manufacturing_containment_active ?manufacturing_site_case)
        (distribution_containment_active ?distribution_site_case)
        (manufacturing_case_assigned_site_unit ?manufacturing_site_case ?manufacturing_site_unit)
        (distribution_case_assigned_distribution_node ?distribution_site_case ?distribution_node)
        (site_unit_quarantined ?manufacturing_site_unit)
        (distribution_node_flagged_for_quarantine ?distribution_node)
        (not
          (manufacturing_containment_verified ?manufacturing_site_case)
        )
        (distribution_containment_verified ?distribution_site_case)
        (remediation_order_planned ?remediation_action)
      )
    :effect
      (and
        (remediation_order_prepared ?remediation_action)
        (remediation_order_linked_to_site_unit ?remediation_action ?manufacturing_site_unit)
        (remediation_order_linked_to_distribution_node ?remediation_action ?distribution_node)
        (manufacturing_scope_confirmed ?remediation_action)
        (not
          (remediation_order_planned ?remediation_action)
        )
      )
  )
  (:action prepare_distribution_remediation_order
    :parameters (?manufacturing_site_case - manufacturing_site_case ?distribution_site_case - distribution_site_case ?manufacturing_site_unit - manufacturing_site_unit ?distribution_node - distribution_node ?remediation_action - remediation_order)
    :precondition
      (and
        (manufacturing_containment_active ?manufacturing_site_case)
        (distribution_containment_active ?distribution_site_case)
        (manufacturing_case_assigned_site_unit ?manufacturing_site_case ?manufacturing_site_unit)
        (distribution_case_assigned_distribution_node ?distribution_site_case ?distribution_node)
        (site_unit_flagged_for_quarantine ?manufacturing_site_unit)
        (distribution_node_quarantined ?distribution_node)
        (manufacturing_containment_verified ?manufacturing_site_case)
        (not
          (distribution_containment_verified ?distribution_site_case)
        )
        (remediation_order_planned ?remediation_action)
      )
    :effect
      (and
        (remediation_order_prepared ?remediation_action)
        (remediation_order_linked_to_site_unit ?remediation_action ?manufacturing_site_unit)
        (remediation_order_linked_to_distribution_node ?remediation_action ?distribution_node)
        (distribution_scope_confirmed ?remediation_action)
        (not
          (remediation_order_planned ?remediation_action)
        )
      )
  )
  (:action prepare_full_scope_remediation_order
    :parameters (?manufacturing_site_case - manufacturing_site_case ?distribution_site_case - distribution_site_case ?manufacturing_site_unit - manufacturing_site_unit ?distribution_node - distribution_node ?remediation_action - remediation_order)
    :precondition
      (and
        (manufacturing_containment_active ?manufacturing_site_case)
        (distribution_containment_active ?distribution_site_case)
        (manufacturing_case_assigned_site_unit ?manufacturing_site_case ?manufacturing_site_unit)
        (distribution_case_assigned_distribution_node ?distribution_site_case ?distribution_node)
        (site_unit_quarantined ?manufacturing_site_unit)
        (distribution_node_quarantined ?distribution_node)
        (not
          (manufacturing_containment_verified ?manufacturing_site_case)
        )
        (not
          (distribution_containment_verified ?distribution_site_case)
        )
        (remediation_order_planned ?remediation_action)
      )
    :effect
      (and
        (remediation_order_prepared ?remediation_action)
        (remediation_order_linked_to_site_unit ?remediation_action ?manufacturing_site_unit)
        (remediation_order_linked_to_distribution_node ?remediation_action ?distribution_node)
        (manufacturing_scope_confirmed ?remediation_action)
        (distribution_scope_confirmed ?remediation_action)
        (not
          (remediation_order_planned ?remediation_action)
        )
      )
  )
  (:action issue_remediation_order
    :parameters (?remediation_action - remediation_order ?manufacturing_site_case - manufacturing_site_case ?report_document - report_document)
    :precondition
      (and
        (remediation_order_prepared ?remediation_action)
        (manufacturing_containment_active ?manufacturing_site_case)
        (case_has_report_document ?manufacturing_site_case ?report_document)
        (not
          (remediation_order_issued ?remediation_action)
        )
      )
    :effect (remediation_order_issued ?remediation_action)
  )
  (:action collect_laboratory_sample
    :parameters (?investigation_workstream - investigation_workstream ?laboratory_sample - laboratory_sample ?remediation_action - remediation_order)
    :precondition
      (and
        (case_under_investigation ?investigation_workstream)
        (workstream_linked_to_remediation_order ?investigation_workstream ?remediation_action)
        (workstream_has_laboratory_sample ?investigation_workstream ?laboratory_sample)
        (laboratory_sample_available ?laboratory_sample)
        (remediation_order_prepared ?remediation_action)
        (remediation_order_issued ?remediation_action)
        (not
          (laboratory_sample_collected ?laboratory_sample)
        )
      )
    :effect
      (and
        (laboratory_sample_collected ?laboratory_sample)
        (sample_linked_to_remediation_order ?laboratory_sample ?remediation_action)
        (not
          (laboratory_sample_available ?laboratory_sample)
        )
      )
  )
  (:action advance_laboratory_sample_assessment
    :parameters (?investigation_workstream - investigation_workstream ?laboratory_sample - laboratory_sample ?remediation_action - remediation_order ?report_document - report_document)
    :precondition
      (and
        (case_under_investigation ?investigation_workstream)
        (workstream_has_laboratory_sample ?investigation_workstream ?laboratory_sample)
        (laboratory_sample_collected ?laboratory_sample)
        (sample_linked_to_remediation_order ?laboratory_sample ?remediation_action)
        (case_has_report_document ?investigation_workstream ?report_document)
        (not
          (manufacturing_scope_confirmed ?remediation_action)
