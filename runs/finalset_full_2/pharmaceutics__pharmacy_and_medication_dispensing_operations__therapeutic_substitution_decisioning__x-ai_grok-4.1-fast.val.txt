--- domain.pddl (as passed to VAL) ---
(define (domain pharmacy_therapeutic_substitution_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types pharmacy_domain_entity - object resource_registry_group - object asset_group - object prescription_case_root - object prescription_case - prescription_case_root drug_product_candidate - pharmacy_domain_entity pharmacist - pharmacy_domain_entity prescriber_clinician - pharmacy_domain_entity special_handling_option - pharmacy_domain_entity patient_instruction_template - pharmacy_domain_entity patient_instruction_asset - pharmacy_domain_entity clinical_check_record - pharmacy_domain_entity clinical_override_record - pharmacy_domain_entity transient_resource_token - resource_registry_group label_template - resource_registry_group insurance_authorization_record - resource_registry_group formulation_component - asset_group packaging_component - asset_group package_unit - asset_group prescription_assignment_group - prescription_case prescription_line_group - prescription_case pharmacist_dispense_task - prescription_assignment_group technician_dispense_task - prescription_assignment_group prescription_line - prescription_line_group)
  (:predicates
    (prescription_case_registered ?prescription_case - prescription_case)
    (pharmacist_review_completed_for_entity ?prescription_case - prescription_case)
    (substitution_candidate_selected ?prescription_case - prescription_case)
    (dispense_finalized_for_entity ?prescription_case - prescription_case)
    (final_verification_completed_for_entity ?prescription_case - prescription_case)
    (patient_instructions_attached_for_entity ?prescription_case - prescription_case)
    (candidate_available ?drug_product_candidate - drug_product_candidate)
    (selected_substitution_candidate_for_entity ?prescription_case - prescription_case ?drug_product_candidate - drug_product_candidate)
    (pharmacist_available ?pharmacist - pharmacist)
    (assigned_pharmacist_for_entity ?prescription_case - prescription_case ?pharmacist - pharmacist)
    (prescriber_available ?prescriber_clinician - prescriber_clinician)
    (authorization_request_pending_for_entity ?prescription_case - prescription_case ?prescriber_clinician - prescriber_clinician)
    (transient_resource_available ?transient_resource_token - transient_resource_token)
    (reserved_resource_for_pharmacist_task ?pharmacist_dispense_task - pharmacist_dispense_task ?transient_resource_token - transient_resource_token)
    (reserved_resource_for_technician_task ?technician_dispense_task - technician_dispense_task ?transient_resource_token - transient_resource_token)
    (dispense_task_requires_component ?pharmacist_dispense_task - pharmacist_dispense_task ?formulation_component - formulation_component)
    (component_reserved ?formulation_component - formulation_component)
    (component_staged ?formulation_component - formulation_component)
    (formulation_component_confirmed_for_task ?pharmacist_dispense_task - pharmacist_dispense_task)
    (technician_task_requires_packaging_component ?technician_dispense_task - technician_dispense_task ?packaging_component - packaging_component)
    (packaging_component_reserved ?packaging_component - packaging_component)
    (packaging_component_staged ?packaging_component - packaging_component)
    (packaging_component_confirmed_for_task ?technician_dispense_task - technician_dispense_task)
    (package_unit_available ?package_unit - package_unit)
    (package_unit_allocated ?package_unit - package_unit)
    (package_unit_has_formulation_component ?package_unit - package_unit ?formulation_component - formulation_component)
    (package_unit_has_packaging_component ?package_unit - package_unit ?packaging_component - packaging_component)
    (package_unit_requires_special_handling ?package_unit - package_unit)
    (package_unit_requires_clinical_check ?package_unit - package_unit)
    (package_unit_packaging_complete ?package_unit - package_unit)
    (prescription_line_assigned_to_pharmacist_task ?prescription_line - prescription_line ?pharmacist_dispense_task - pharmacist_dispense_task)
    (prescription_line_assigned_to_technician_task ?prescription_line - prescription_line ?technician_dispense_task - technician_dispense_task)
    (prescription_line_assigned_package_unit ?prescription_line - prescription_line ?package_unit - package_unit)
    (label_template_available ?label_template - label_template)
    (line_label_template_association ?prescription_line - prescription_line ?label_template - label_template)
    (label_template_bound ?label_template - label_template)
    (label_template_bound_to_package_unit ?label_template - label_template ?package_unit - package_unit)
    (label_binding_ready ?prescription_line - prescription_line)
    (label_applied_to_line ?prescription_line - prescription_line)
    (clinical_check_passed ?prescription_line - prescription_line)
    (special_handling_assigned ?prescription_line - prescription_line)
    (special_handling_applied ?prescription_line - prescription_line)
    (label_and_instruction_binding_completed ?prescription_line - prescription_line)
    (final_checks_completed_for_line ?prescription_line - prescription_line)
    (authorization_record_available ?insurance_authorization_record - insurance_authorization_record)
    (prescription_line_has_authorization_record ?prescription_line - prescription_line ?insurance_authorization_record - insurance_authorization_record)
    (authorization_applied_to_line ?prescription_line - prescription_line)
    (authorization_in_progress_for_line ?prescription_line - prescription_line)
    (prescriber_response_recorded ?prescription_line - prescription_line)
    (special_handling_option_available ?special_handling_option - special_handling_option)
    (line_special_handling_association ?prescription_line - prescription_line ?special_handling_option - special_handling_option)
    (patient_instruction_template_available ?patient_instruction_template - patient_instruction_template)
    (line_bound_patient_instruction_template ?prescription_line - prescription_line ?patient_instruction_template - patient_instruction_template)
    (clinical_check_record_available ?clinical_check_record - clinical_check_record)
    (line_clinical_check_bound ?prescription_line - prescription_line ?clinical_check_record - clinical_check_record)
    (clinical_override_record_available ?clinical_override_record - clinical_override_record)
    (line_override_record_bound ?prescription_line - prescription_line ?clinical_override_record - clinical_override_record)
    (patient_instruction_asset_available ?patient_instruction_asset - patient_instruction_asset)
    (has_patient_instruction_asset_for_entity ?prescription_case - prescription_case ?patient_instruction_asset - patient_instruction_asset)
    (dispense_task_components_ready ?pharmacist_dispense_task - pharmacist_dispense_task)
    (technician_task_components_ready ?technician_dispense_task - technician_dispense_task)
    (prescription_line_verification_completed ?prescription_line - prescription_line)
  )
  (:action register_prescription_case
    :parameters (?prescription_case - prescription_case)
    :precondition
      (and
        (not
          (prescription_case_registered ?prescription_case)
        )
        (not
          (dispense_finalized_for_entity ?prescription_case)
        )
      )
    :effect (prescription_case_registered ?prescription_case)
  )
  (:action select_substitution_candidate_for_case
    :parameters (?prescription_case - prescription_case ?drug_product_candidate - drug_product_candidate)
    :precondition
      (and
        (prescription_case_registered ?prescription_case)
        (not
          (substitution_candidate_selected ?prescription_case)
        )
        (candidate_available ?drug_product_candidate)
      )
    :effect
      (and
        (substitution_candidate_selected ?prescription_case)
        (selected_substitution_candidate_for_entity ?prescription_case ?drug_product_candidate)
        (not
          (candidate_available ?drug_product_candidate)
        )
      )
  )
  (:action assign_pharmacist_to_case
    :parameters (?prescription_case - prescription_case ?pharmacist - pharmacist)
    :precondition
      (and
        (prescription_case_registered ?prescription_case)
        (substitution_candidate_selected ?prescription_case)
        (pharmacist_available ?pharmacist)
      )
    :effect
      (and
        (assigned_pharmacist_for_entity ?prescription_case ?pharmacist)
        (not
          (pharmacist_available ?pharmacist)
        )
      )
  )
  (:action complete_pharmacist_review
    :parameters (?prescription_case - prescription_case ?pharmacist - pharmacist)
    :precondition
      (and
        (prescription_case_registered ?prescription_case)
        (substitution_candidate_selected ?prescription_case)
        (assigned_pharmacist_for_entity ?prescription_case ?pharmacist)
        (not
          (pharmacist_review_completed_for_entity ?prescription_case)
        )
      )
    :effect (pharmacist_review_completed_for_entity ?prescription_case)
  )
  (:action release_pharmacist_assignment
    :parameters (?prescription_case - prescription_case ?pharmacist - pharmacist)
    :precondition
      (and
        (assigned_pharmacist_for_entity ?prescription_case ?pharmacist)
      )
    :effect
      (and
        (pharmacist_available ?pharmacist)
        (not
          (assigned_pharmacist_for_entity ?prescription_case ?pharmacist)
        )
      )
  )
  (:action request_prescriber_authorization_for_case
    :parameters (?prescription_case - prescription_case ?prescriber_clinician - prescriber_clinician)
    :precondition
      (and
        (pharmacist_review_completed_for_entity ?prescription_case)
        (prescriber_available ?prescriber_clinician)
      )
    :effect
      (and
        (authorization_request_pending_for_entity ?prescription_case ?prescriber_clinician)
        (not
          (prescriber_available ?prescriber_clinician)
        )
      )
  )
  (:action cancel_prescriber_authorization_request
    :parameters (?prescription_case - prescription_case ?prescriber_clinician - prescriber_clinician)
    :precondition
      (and
        (authorization_request_pending_for_entity ?prescription_case ?prescriber_clinician)
      )
    :effect
      (and
        (prescriber_available ?prescriber_clinician)
        (not
          (authorization_request_pending_for_entity ?prescription_case ?prescriber_clinician)
        )
      )
  )
  (:action assign_clinical_check_to_line
    :parameters (?prescription_line - prescription_line ?clinical_check_record - clinical_check_record)
    :precondition
      (and
        (pharmacist_review_completed_for_entity ?prescription_line)
        (clinical_check_record_available ?clinical_check_record)
      )
    :effect
      (and
        (line_clinical_check_bound ?prescription_line ?clinical_check_record)
        (not
          (clinical_check_record_available ?clinical_check_record)
        )
      )
  )
  (:action release_clinical_check_from_line
    :parameters (?prescription_line - prescription_line ?clinical_check_record - clinical_check_record)
    :precondition
      (and
        (line_clinical_check_bound ?prescription_line ?clinical_check_record)
      )
    :effect
      (and
        (clinical_check_record_available ?clinical_check_record)
        (not
          (line_clinical_check_bound ?prescription_line ?clinical_check_record)
        )
      )
  )
  (:action assign_override_record_to_line
    :parameters (?prescription_line - prescription_line ?clinical_override_record - clinical_override_record)
    :precondition
      (and
        (pharmacist_review_completed_for_entity ?prescription_line)
        (clinical_override_record_available ?clinical_override_record)
      )
    :effect
      (and
        (line_override_record_bound ?prescription_line ?clinical_override_record)
        (not
          (clinical_override_record_available ?clinical_override_record)
        )
      )
  )
  (:action release_override_record_from_line
    :parameters (?prescription_line - prescription_line ?clinical_override_record - clinical_override_record)
    :precondition
      (and
        (line_override_record_bound ?prescription_line ?clinical_override_record)
      )
    :effect
      (and
        (clinical_override_record_available ?clinical_override_record)
        (not
          (line_override_record_bound ?prescription_line ?clinical_override_record)
        )
      )
  )
  (:action reserve_formulation_component_for_task
    :parameters (?pharmacist_dispense_task - pharmacist_dispense_task ?formulation_component - formulation_component ?pharmacist - pharmacist)
    :precondition
      (and
        (pharmacist_review_completed_for_entity ?pharmacist_dispense_task)
        (assigned_pharmacist_for_entity ?pharmacist_dispense_task ?pharmacist)
        (dispense_task_requires_component ?pharmacist_dispense_task ?formulation_component)
        (not
          (component_reserved ?formulation_component)
        )
        (not
          (component_staged ?formulation_component)
        )
      )
    :effect (component_reserved ?formulation_component)
  )
  (:action finalize_formulation_component_with_prescriber
    :parameters (?pharmacist_dispense_task - pharmacist_dispense_task ?formulation_component - formulation_component ?prescriber_clinician - prescriber_clinician)
    :precondition
      (and
        (pharmacist_review_completed_for_entity ?pharmacist_dispense_task)
        (authorization_request_pending_for_entity ?pharmacist_dispense_task ?prescriber_clinician)
        (dispense_task_requires_component ?pharmacist_dispense_task ?formulation_component)
        (component_reserved ?formulation_component)
        (not
          (dispense_task_components_ready ?pharmacist_dispense_task)
        )
      )
    :effect
      (and
        (dispense_task_components_ready ?pharmacist_dispense_task)
        (formulation_component_confirmed_for_task ?pharmacist_dispense_task)
      )
  )
  (:action stage_formulation_component_with_resource
    :parameters (?pharmacist_dispense_task - pharmacist_dispense_task ?formulation_component - formulation_component ?transient_resource_token - transient_resource_token)
    :precondition
      (and
        (pharmacist_review_completed_for_entity ?pharmacist_dispense_task)
        (dispense_task_requires_component ?pharmacist_dispense_task ?formulation_component)
        (transient_resource_available ?transient_resource_token)
        (not
          (dispense_task_components_ready ?pharmacist_dispense_task)
        )
      )
    :effect
      (and
        (component_staged ?formulation_component)
        (dispense_task_components_ready ?pharmacist_dispense_task)
        (reserved_resource_for_pharmacist_task ?pharmacist_dispense_task ?transient_resource_token)
        (not
          (transient_resource_available ?transient_resource_token)
        )
      )
  )
  (:action finalize_staged_formulation_component
    :parameters (?pharmacist_dispense_task - pharmacist_dispense_task ?formulation_component - formulation_component ?pharmacist - pharmacist ?transient_resource_token - transient_resource_token)
    :precondition
      (and
        (pharmacist_review_completed_for_entity ?pharmacist_dispense_task)
        (assigned_pharmacist_for_entity ?pharmacist_dispense_task ?pharmacist)
        (dispense_task_requires_component ?pharmacist_dispense_task ?formulation_component)
        (component_staged ?formulation_component)
        (reserved_resource_for_pharmacist_task ?pharmacist_dispense_task ?transient_resource_token)
        (not
          (formulation_component_confirmed_for_task ?pharmacist_dispense_task)
        )
      )
    :effect
      (and
        (component_reserved ?formulation_component)
        (formulation_component_confirmed_for_task ?pharmacist_dispense_task)
        (transient_resource_available ?transient_resource_token)
        (not
          (reserved_resource_for_pharmacist_task ?pharmacist_dispense_task ?transient_resource_token)
        )
      )
  )
  (:action reserve_packaging_component_for_task
    :parameters (?technician_dispense_task - technician_dispense_task ?packaging_component - packaging_component ?pharmacist - pharmacist)
    :precondition
      (and
        (pharmacist_review_completed_for_entity ?technician_dispense_task)
        (assigned_pharmacist_for_entity ?technician_dispense_task ?pharmacist)
        (technician_task_requires_packaging_component ?technician_dispense_task ?packaging_component)
        (not
          (packaging_component_reserved ?packaging_component)
        )
        (not
          (packaging_component_staged ?packaging_component)
        )
      )
    :effect (packaging_component_reserved ?packaging_component)
  )
  (:action finalize_packaging_component_with_prescriber
    :parameters (?technician_dispense_task - technician_dispense_task ?packaging_component - packaging_component ?prescriber_clinician - prescriber_clinician)
    :precondition
      (and
        (pharmacist_review_completed_for_entity ?technician_dispense_task)
        (authorization_request_pending_for_entity ?technician_dispense_task ?prescriber_clinician)
        (technician_task_requires_packaging_component ?technician_dispense_task ?packaging_component)
        (packaging_component_reserved ?packaging_component)
        (not
          (technician_task_components_ready ?technician_dispense_task)
        )
      )
    :effect
      (and
        (technician_task_components_ready ?technician_dispense_task)
        (packaging_component_confirmed_for_task ?technician_dispense_task)
      )
  )
  (:action stage_packaging_component_with_resource
    :parameters (?technician_dispense_task - technician_dispense_task ?packaging_component - packaging_component ?transient_resource_token - transient_resource_token)
    :precondition
      (and
        (pharmacist_review_completed_for_entity ?technician_dispense_task)
        (technician_task_requires_packaging_component ?technician_dispense_task ?packaging_component)
        (transient_resource_available ?transient_resource_token)
        (not
          (technician_task_components_ready ?technician_dispense_task)
        )
      )
    :effect
      (and
        (packaging_component_staged ?packaging_component)
        (technician_task_components_ready ?technician_dispense_task)
        (reserved_resource_for_technician_task ?technician_dispense_task ?transient_resource_token)
        (not
          (transient_resource_available ?transient_resource_token)
        )
      )
  )
  (:action finalize_staged_packaging_component
    :parameters (?technician_dispense_task - technician_dispense_task ?packaging_component - packaging_component ?pharmacist - pharmacist ?transient_resource_token - transient_resource_token)
    :precondition
      (and
        (pharmacist_review_completed_for_entity ?technician_dispense_task)
        (assigned_pharmacist_for_entity ?technician_dispense_task ?pharmacist)
        (technician_task_requires_packaging_component ?technician_dispense_task ?packaging_component)
        (packaging_component_staged ?packaging_component)
        (reserved_resource_for_technician_task ?technician_dispense_task ?transient_resource_token)
        (not
          (packaging_component_confirmed_for_task ?technician_dispense_task)
        )
      )
    :effect
      (and
        (packaging_component_reserved ?packaging_component)
        (packaging_component_confirmed_for_task ?technician_dispense_task)
        (transient_resource_available ?transient_resource_token)
        (not
          (reserved_resource_for_technician_task ?technician_dispense_task ?transient_resource_token)
        )
      )
  )
  (:action assemble_package_unit_from_components
    :parameters (?pharmacist_dispense_task - pharmacist_dispense_task ?technician_dispense_task - technician_dispense_task ?formulation_component - formulation_component ?packaging_component - packaging_component ?package_unit - package_unit)
    :precondition
      (and
        (dispense_task_components_ready ?pharmacist_dispense_task)
        (technician_task_components_ready ?technician_dispense_task)
        (dispense_task_requires_component ?pharmacist_dispense_task ?formulation_component)
        (technician_task_requires_packaging_component ?technician_dispense_task ?packaging_component)
        (component_reserved ?formulation_component)
        (packaging_component_reserved ?packaging_component)
        (formulation_component_confirmed_for_task ?pharmacist_dispense_task)
        (packaging_component_confirmed_for_task ?technician_dispense_task)
        (package_unit_available ?package_unit)
      )
    :effect
      (and
        (package_unit_allocated ?package_unit)
        (package_unit_has_formulation_component ?package_unit ?formulation_component)
        (package_unit_has_packaging_component ?package_unit ?packaging_component)
        (not
          (package_unit_available ?package_unit)
        )
      )
  )
  (:action assemble_package_unit_with_formulation_staged
    :parameters (?pharmacist_dispense_task - pharmacist_dispense_task ?technician_dispense_task - technician_dispense_task ?formulation_component - formulation_component ?packaging_component - packaging_component ?package_unit - package_unit)
    :precondition
      (and
        (dispense_task_components_ready ?pharmacist_dispense_task)
        (technician_task_components_ready ?technician_dispense_task)
        (dispense_task_requires_component ?pharmacist_dispense_task ?formulation_component)
        (technician_task_requires_packaging_component ?technician_dispense_task ?packaging_component)
        (component_staged ?formulation_component)
        (packaging_component_reserved ?packaging_component)
        (not
          (formulation_component_confirmed_for_task ?pharmacist_dispense_task)
        )
        (packaging_component_confirmed_for_task ?technician_dispense_task)
        (package_unit_available ?package_unit)
      )
    :effect
      (and
        (package_unit_allocated ?package_unit)
        (package_unit_has_formulation_component ?package_unit ?formulation_component)
        (package_unit_has_packaging_component ?package_unit ?packaging_component)
        (package_unit_requires_special_handling ?package_unit)
        (not
          (package_unit_available ?package_unit)
        )
      )
  )
  (:action assemble_package_unit_with_packaging_staged
    :parameters (?pharmacist_dispense_task - pharmacist_dispense_task ?technician_dispense_task - technician_dispense_task ?formulation_component - formulation_component ?packaging_component - packaging_component ?package_unit - package_unit)
    :precondition
      (and
        (dispense_task_components_ready ?pharmacist_dispense_task)
        (technician_task_components_ready ?technician_dispense_task)
        (dispense_task_requires_component ?pharmacist_dispense_task ?formulation_component)
        (technician_task_requires_packaging_component ?technician_dispense_task ?packaging_component)
        (component_reserved ?formulation_component)
        (packaging_component_staged ?packaging_component)
        (formulation_component_confirmed_for_task ?pharmacist_dispense_task)
        (not
          (packaging_component_confirmed_for_task ?technician_dispense_task)
        )
        (package_unit_available ?package_unit)
      )
    :effect
      (and
        (package_unit_allocated ?package_unit)
        (package_unit_has_formulation_component ?package_unit ?formulation_component)
        (package_unit_has_packaging_component ?package_unit ?packaging_component)
        (package_unit_requires_clinical_check ?package_unit)
        (not
          (package_unit_available ?package_unit)
        )
      )
  )
  (:action assemble_package_unit_with_all_components_staged
    :parameters (?pharmacist_dispense_task - pharmacist_dispense_task ?technician_dispense_task - technician_dispense_task ?formulation_component - formulation_component ?packaging_component - packaging_component ?package_unit - package_unit)
    :precondition
      (and
        (dispense_task_components_ready ?pharmacist_dispense_task)
        (technician_task_components_ready ?technician_dispense_task)
        (dispense_task_requires_component ?pharmacist_dispense_task ?formulation_component)
        (technician_task_requires_packaging_component ?technician_dispense_task ?packaging_component)
        (component_staged ?formulation_component)
        (packaging_component_staged ?packaging_component)
        (not
          (formulation_component_confirmed_for_task ?pharmacist_dispense_task)
        )
        (not
          (packaging_component_confirmed_for_task ?technician_dispense_task)
        )
        (package_unit_available ?package_unit)
      )
    :effect
      (and
        (package_unit_allocated ?package_unit)
        (package_unit_has_formulation_component ?package_unit ?formulation_component)
        (package_unit_has_packaging_component ?package_unit ?packaging_component)
        (package_unit_requires_special_handling ?package_unit)
        (package_unit_requires_clinical_check ?package_unit)
        (not
          (package_unit_available ?package_unit)
        )
      )
  )
  (:action complete_package_unit_packaging
    :parameters (?package_unit - package_unit ?pharmacist_dispense_task - pharmacist_dispense_task ?pharmacist - pharmacist)
    :precondition
      (and
        (package_unit_allocated ?package_unit)
        (dispense_task_components_ready ?pharmacist_dispense_task)
        (assigned_pharmacist_for_entity ?pharmacist_dispense_task ?pharmacist)
        (not
          (package_unit_packaging_complete ?package_unit)
        )
      )
    :effect (package_unit_packaging_complete ?package_unit)
  )
  (:action bind_label_template_to_package_unit
    :parameters (?prescription_line - prescription_line ?label_template - label_template ?package_unit - package_unit)
    :precondition
      (and
        (pharmacist_review_completed_for_entity ?prescription_line)
        (prescription_line_assigned_package_unit ?prescription_line ?package_unit)
        (line_label_template_association ?prescription_line ?label_template)
        (label_template_available ?label_template)
        (package_unit_allocated ?package_unit)
        (package_unit_packaging_complete ?package_unit)
        (not
          (label_template_bound ?label_template)
        )
      )
    :effect
      (and
        (label_template_bound ?label_template)
        (label_template_bound_to_package_unit ?label_template ?package_unit)
        (not
          (label_template_available ?label_template)
        )
      )
  )
  (:action prepare_label_binding_for_line
    :parameters (?prescription_line - prescription_line ?label_template - label_template ?package_unit - package_unit ?pharmacist - pharmacist)
    :precondition
      (and
