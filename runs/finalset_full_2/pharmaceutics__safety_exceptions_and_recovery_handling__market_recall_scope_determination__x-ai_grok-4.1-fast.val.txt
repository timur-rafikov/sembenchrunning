  (:action register_recall_case
    :parameters (?recall_case - case_participation)
    :precondition
      (and
        (not
          (case_participation_registered ?recall_case)
        )
        (not
          (case_participation_execution_assignment_confirmed ?recall_case)
        )
      )
    :effect (case_participation_registered ?recall_case)
  )
  (:action assign_logistics_provider_to_case
    :parameters (?recall_case - case_participation ?logistics_provider - logistics_provider)
    :precondition
      (and
        (case_participation_registered ?recall_case)
        (not
          (case_participation_provider_assigned_flag ?recall_case)
        )
        (logistics_provider_available ?logistics_provider)
      )
    :effect
      (and
        (case_participation_provider_assigned_flag ?recall_case)
        (case_participation_assigned_to_logistics_provider ?recall_case ?logistics_provider)
        (not
          (logistics_provider_available ?logistics_provider)
        )
      )
  )
  (:action assign_lab_test_to_case
    :parameters (?recall_case - case_participation ?lab_test - lab_test)
    :precondition
      (and
        (case_participation_registered ?recall_case)
        (case_participation_provider_assigned_flag ?recall_case)
        (lab_test_available ?lab_test)
      )
    :effect
      (and
        (case_participation_assigned_to_lab_test ?recall_case ?lab_test)
        (not
          (lab_test_available ?lab_test)
        )
      )
  )
  (:action confirm_lab_assignment_for_case
    :parameters (?recall_case - case_participation ?lab_test - lab_test)
    :precondition
      (and
        (case_participation_registered ?recall_case)
        (case_participation_provider_assigned_flag ?recall_case)
        (case_participation_assigned_to_lab_test ?recall_case ?lab_test)
        (not
          (case_participation_triaged ?recall_case)
        )
      )
    :effect (case_participation_triaged ?recall_case)
  )
  (:action release_lab_test_assignment
    :parameters (?recall_case - case_participation ?lab_test - lab_test)
    :precondition
      (and
        (case_participation_assigned_to_lab_test ?recall_case ?lab_test)
      )
    :effect
      (and
        (lab_test_available ?lab_test)
        (not
          (case_participation_assigned_to_lab_test ?recall_case ?lab_test)
        )
      )
  )
  (:action engage_regulatory_authority_with_case
    :parameters (?recall_case - case_participation ?regulatory_authority - regulatory_authority)
    :precondition
      (and
        (case_participation_triaged ?recall_case)
        (regulatory_contact_available ?regulatory_authority)
      )
    :effect
      (and
        (case_participation_linked_to_regulatory_authority ?recall_case ?regulatory_authority)
        (not
          (regulatory_contact_available ?regulatory_authority)
        )
      )
  )
  (:action release_regulatory_authority_from_case
    :parameters (?recall_case - case_participation ?regulatory_authority - regulatory_authority)
    :precondition
      (and
        (case_participation_linked_to_regulatory_authority ?recall_case ?regulatory_authority)
      )
    :effect
      (and
        (regulatory_contact_available ?regulatory_authority)
        (not
          (case_participation_linked_to_regulatory_authority ?recall_case ?regulatory_authority)
        )
      )

  (:action flag_manufacturer_lot
    :parameters (?manufacturer_site - manufacturer_site ?manufacturer_lot - manufacturer_lot ?lab_test - lab_test)
    :precondition
      (and
        (case_participation_triaged ?manufacturer_site)
        (case_participation_assigned_to_lab_test ?manufacturer_site ?lab_test)
        (site_contains_manufacturer_lot ?manufacturer_site ?manufacturer_lot)
        (not
          (manufacturer_lot_flagged ?manufacturer_lot)

        (not
          (manufacturer_lot_isolated ?manufacturer_lot)

    :effect (manufacturer_lot_flagged ?manufacturer_lot)
  )
  (:action mark_manufacturer_site_contained
    :parameters (?manufacturer_site - manufacturer_site ?manufacturer_lot - manufacturer_lot ?regulatory_authority - regulatory_authority)
    :precondition
      (and
        (case_participation_triaged ?manufacturer_site)
        (case_participation_linked_to_regulatory_authority ?manufacturer_site ?regulatory_authority)
        (site_contains_manufacturer_lot ?manufacturer_site ?manufacturer_lot)
        (manufacturer_lot_flagged ?manufacturer_lot)
        (not
          (manufacturer_site_ready_for_execution ?manufacturer_site)
        )
      )
    :effect
      (and
        (manufacturer_site_ready_for_execution ?manufacturer_site)
        (manufacturer_site_contained ?manufacturer_site)
      )
  )
  (:action isolate_manufacturer_lot_and_assign_sample

          (manufacturer_site_ready_for_execution ?manufacturer_site)

        (manufacturer_lot_isolated ?manufacturer_lot)

  (:action initiate_recall_scope_record

    :precondition
      (and
        (manufacturer_site_ready_for_execution ?manufacturer_site)
        (distributor_site_ready_for_execution ?distributor_site)
        (site_contains_manufacturer_lot ?manufacturer_site ?manufacturer_lot)
        (site_contains_distributor_lot ?distributor_site ?distributor_lot)
        (manufacturer_lot_flagged ?manufacturer_lot)
        (distributor_lot_flagged ?distributor_lot)
        (manufacturer_site_contained ?manufacturer_site)
        (distributor_site_contained ?distributor_site)
        (recall_scope_record_available ?recall_scope_record)
      )
    :effect
      (and
        (recall_scope_record_initiated ?recall_scope_record)
        (recall_scope_includes_manufacturer_lot ?recall_scope_record ?manufacturer_lot)
        (recall_scope_includes_distributor_lot ?recall_scope_record ?distributor_lot)
        (not
          (recall_scope_record_available ?recall_scope_record)
        )
      )
  )
  (:action initiate_recall_scope_with_manufacturer_inclusion

      (and
        (manufacturer_site_ready_for_execution ?manufacturer_site)

        (site_contains_distributor_lot ?distributor_site ?distributor_lot)
        (manufacturer_lot_isolated ?manufacturer_lot)
        (distributor_lot_flagged ?distributor_lot)

          (manufacturer_site_contained ?manufacturer_site)
        )
        (distributor_site_contained ?distributor_site)
        (recall_scope_record_available ?recall_scope_record)
      )
    :effect
      (and
        (recall_scope_record_initiated ?recall_scope_record)
        (recall_scope_includes_manufacturer_lot ?recall_scope_record ?manufacturer_lot)
        (recall_scope_includes_distributor_lot ?recall_scope_record ?distributor_lot)

          (recall_scope_record_available ?recall_scope_record)

          (distributor_site_contained ?distributor_site)

        (recall_scope_record_available ?recall_scope_record)

      (and
        (recall_scope_record_initiated ?recall_scope_record)

        (recall_scope_includes_distributor_lot ?recall_scope_record ?distributor_lot)
        (recall_scope_distributor_marker ?recall_scope_record)
        (not
          (recall_scope_record_available ?recall_scope_record)

  )

      (and
        (analytical_report_consumed ?analytical_report)

          (analytical_report_available ?analytical_report)

  (:action review_analytical_evidence_for_product

        (case_participation_triaged ?product_participation)

        (analytical_report_consumed ?analytical_report)
        (analytical_report_linked_to_scope ?analytical_report ?recall_scope_record)
        (case_participation_assigned_to_lab_test ?product_participation ?lab_test)
        (not
          (recall_scope_manufacturer_marker ?recall_scope_record)
        )
        (not
          (product_marked_for_evidence_review ?product_participation)
        )
      )
    :effect (product_marked_for_evidence_review ?product_participation)
  )
  (:action attach_adverse_event_report_to_product
    :parameters (?product_participation - product_participation ?adverse_event_report - adverse_event_report)
    :precondition
      (and
        (case_participation_triaged ?product_participation)
        (adverse_event_report_available ?adverse_event_report)
