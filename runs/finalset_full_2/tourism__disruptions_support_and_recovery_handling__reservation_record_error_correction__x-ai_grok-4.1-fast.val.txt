--- domain.pddl (as passed to VAL) ---
(define (domain reservation_record_error_correction)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity_group - object communication_group - object resource_group - object record_type - object reservation_record - record_type support_resource - entity_group verification_event - entity_group partner_contact - entity_group policy - entity_group approval_token - entity_group payment_instrument - entity_group refund_offer - entity_group compliance_check - entity_group evidence_document - communication_group documentation_package - communication_group escalation_contact - communication_group resolution_path - resource_group compensation_path - resource_group repair_proposal - resource_group booking_group - reservation_record booking_subgroup - reservation_record primary_booking_item - booking_group secondary_booking_item - booking_group correction_case - booking_subgroup)
  (:predicates
    (entity_error_reported ?reservation_record - reservation_record)
    (entity_verification_completed ?reservation_record - reservation_record)
    (entity_support_resource_assigned_flag ?reservation_record - reservation_record)
    (entity_correction_applied ?reservation_record - reservation_record)
    (entity_ready_for_execution ?reservation_record - reservation_record)
    (entity_settlement_authorized ?reservation_record - reservation_record)
    (support_resource_available ?support_resource - support_resource)
    (entity_support_resource_assigned_to ?reservation_record - reservation_record ?support_resource - support_resource)
    (verification_event_available ?verification_event - verification_event)
    (entity_has_verification_event ?reservation_record - reservation_record ?verification_event - verification_event)
    (partner_contact_available ?partner_contact - partner_contact)
    (entity_assigned_partner_contact ?reservation_record - reservation_record ?partner_contact - partner_contact)
    (evidence_document_available ?evidence_document - evidence_document)
    (primary_booking_item_evidence_reserved ?primary_booking_item - primary_booking_item ?evidence_document - evidence_document)
    (secondary_booking_item_evidence_reserved ?secondary_booking_item - secondary_booking_item ?evidence_document - evidence_document)
    (booking_item_resolution_path ?primary_booking_item - primary_booking_item ?resolution_path - resolution_path)
    (resolution_path_validated ?resolution_path - resolution_path)
    (resolution_path_evidence_attached ?resolution_path - resolution_path)
    (primary_booking_item_ready_for_proposal ?primary_booking_item - primary_booking_item)
    (secondary_booking_item_compensation_path ?secondary_booking_item - secondary_booking_item ?compensation_path - compensation_path)
    (compensation_path_validated ?compensation_path - compensation_path)
    (compensation_path_evidence_attached ?compensation_path - compensation_path)
    (secondary_booking_item_ready_for_proposal ?secondary_booking_item - secondary_booking_item)
    (repair_proposal_available ?repair_proposal - repair_proposal)
    (repair_proposal_generated ?repair_proposal - repair_proposal)
    (repair_proposal_resolution_path ?repair_proposal - repair_proposal ?resolution_path - resolution_path)
    (repair_proposal_compensation_path ?repair_proposal - repair_proposal ?compensation_path - compensation_path)
    (repair_proposal_requires_policy_attachment ?repair_proposal - repair_proposal)
    (repair_proposal_requires_escalation ?repair_proposal - repair_proposal)
    (repair_proposal_locked ?repair_proposal - repair_proposal)
    (case_contains_primary_booking_item ?correction_case - correction_case ?primary_booking_item - primary_booking_item)
    (case_contains_secondary_booking_item ?correction_case - correction_case ?secondary_booking_item - secondary_booking_item)
    (case_has_repair_proposal ?correction_case - correction_case ?repair_proposal - repair_proposal)
    (documentation_package_available ?documentation_package - documentation_package)
    (case_has_documentation_package ?correction_case - correction_case ?documentation_package - documentation_package)
    (documentation_package_assigned ?documentation_package - documentation_package)
    (documentation_package_linked_to_proposal ?documentation_package - documentation_package ?repair_proposal - repair_proposal)
    (case_documentation_validated ?correction_case - correction_case)
    (case_approval_requested ?correction_case - correction_case)
    (case_approval_granted ?correction_case - correction_case)
    (case_policy_attached ?correction_case - correction_case)
    (case_policy_validated ?correction_case - correction_case)
    (case_requires_approval_token ?correction_case - correction_case)
    (case_final_checks_passed ?correction_case - correction_case)
    (escalation_contact_available ?escalation_contact - escalation_contact)
    (case_escalation_contact_assigned ?correction_case - correction_case ?escalation_contact - escalation_contact)
    (case_escalated ?correction_case - correction_case)
    (escalation_contact_acknowledged ?correction_case - correction_case)
    (escalation_compliance_check_passed ?correction_case - correction_case)
    (policy_available ?policy - policy)
    (case_policy_linked ?correction_case - correction_case ?policy - policy)
    (approval_token_available ?approval_token - approval_token)
    (case_approval_token_attached ?correction_case - correction_case ?approval_token - approval_token)
    (refund_offer_available ?refund_offer - refund_offer)
    (case_refund_offer_attached ?correction_case - correction_case ?refund_offer - refund_offer)
    (compliance_check_available ?compliance_check - compliance_check)
    (case_compliance_check_assigned ?correction_case - correction_case ?compliance_check - compliance_check)
    (payment_instrument_available ?payment_instrument - payment_instrument)
    (entity_payment_instrument_selected ?reservation_record - reservation_record ?payment_instrument - payment_instrument)
    (primary_booking_item_ready ?primary_booking_item - primary_booking_item)
    (secondary_booking_item_ready ?secondary_booking_item - secondary_booking_item)
    (case_finalized ?correction_case - correction_case)
  )
  (:action log_reservation_error
    :parameters (?reservation_record - reservation_record)
    :precondition
      (and
        (not
          (entity_error_reported ?reservation_record)
        )
        (not
          (entity_correction_applied ?reservation_record)
        )
      )
    :effect (entity_error_reported ?reservation_record)
  )
  (:action assign_support_resource_to_reservation
    :parameters (?reservation_record - reservation_record ?support_resource - support_resource)
    :precondition
      (and
        (entity_error_reported ?reservation_record)
        (not
          (entity_support_resource_assigned_flag ?reservation_record)
        )
        (support_resource_available ?support_resource)
      )
    :effect
      (and
        (entity_support_resource_assigned_flag ?reservation_record)
        (entity_support_resource_assigned_to ?reservation_record ?support_resource)
        (not
          (support_resource_available ?support_resource)
        )
      )
  )
  (:action initiate_verification_event
    :parameters (?reservation_record - reservation_record ?verification_event - verification_event)
    :precondition
      (and
        (entity_error_reported ?reservation_record)
        (entity_support_resource_assigned_flag ?reservation_record)
        (verification_event_available ?verification_event)
      )
    :effect
      (and
        (entity_has_verification_event ?reservation_record ?verification_event)
        (not
          (verification_event_available ?verification_event)
        )
      )
  )
  (:action confirm_verification
    :parameters (?reservation_record - reservation_record ?verification_event - verification_event)
    :precondition
      (and
        (entity_error_reported ?reservation_record)
        (entity_support_resource_assigned_flag ?reservation_record)
        (entity_has_verification_event ?reservation_record ?verification_event)
        (not
          (entity_verification_completed ?reservation_record)
        )
      )
    :effect (entity_verification_completed ?reservation_record)
  )
  (:action release_verification_event
    :parameters (?reservation_record - reservation_record ?verification_event - verification_event)
    :precondition
      (and
        (entity_has_verification_event ?reservation_record ?verification_event)
      )
    :effect
      (and
        (verification_event_available ?verification_event)
        (not
          (entity_has_verification_event ?reservation_record ?verification_event)
        )
      )
  )
  (:action engage_partner_contact
    :parameters (?reservation_record - reservation_record ?partner_contact - partner_contact)
    :precondition
      (and
        (entity_verification_completed ?reservation_record)
        (partner_contact_available ?partner_contact)
      )
    :effect
      (and
        (entity_assigned_partner_contact ?reservation_record ?partner_contact)
        (not
          (partner_contact_available ?partner_contact)
        )
      )
  )
  (:action release_partner_contact
    :parameters (?reservation_record - reservation_record ?partner_contact - partner_contact)
    :precondition
      (and
        (entity_assigned_partner_contact ?reservation_record ?partner_contact)
      )
    :effect
      (and
        (partner_contact_available ?partner_contact)
        (not
          (entity_assigned_partner_contact ?reservation_record ?partner_contact)
        )
      )
  )
  (:action attach_refund_offer_to_case
    :parameters (?correction_case - correction_case ?refund_offer - refund_offer)
    :precondition
      (and
        (entity_verification_completed ?correction_case)
        (refund_offer_available ?refund_offer)
      )
    :effect
      (and
        (case_refund_offer_attached ?correction_case ?refund_offer)
        (not
          (refund_offer_available ?refund_offer)
        )
      )
  )
  (:action detach_refund_offer_from_case
    :parameters (?correction_case - correction_case ?refund_offer - refund_offer)
    :precondition
      (and
        (case_refund_offer_attached ?correction_case ?refund_offer)
      )
    :effect
      (and
        (refund_offer_available ?refund_offer)
        (not
          (case_refund_offer_attached ?correction_case ?refund_offer)
        )
      )
  )
  (:action attach_compliance_check_to_case
    :parameters (?correction_case - correction_case ?compliance_check - compliance_check)
    :precondition
      (and
        (entity_verification_completed ?correction_case)
        (compliance_check_available ?compliance_check)
      )
    :effect
      (and
        (case_compliance_check_assigned ?correction_case ?compliance_check)
        (not
          (compliance_check_available ?compliance_check)
        )
      )
  )
  (:action detach_compliance_check_from_case
    :parameters (?correction_case - correction_case ?compliance_check - compliance_check)
    :precondition
      (and
        (case_compliance_check_assigned ?correction_case ?compliance_check)
      )
    :effect
      (and
        (compliance_check_available ?compliance_check)
        (not
          (case_compliance_check_assigned ?correction_case ?compliance_check)
        )
      )
  )
  (:action validate_resolution_path_for_booking_item
    :parameters (?primary_booking_item - primary_booking_item ?resolution_path - resolution_path ?verification_event - verification_event)
    :precondition
      (and
        (entity_verification_completed ?primary_booking_item)
        (entity_has_verification_event ?primary_booking_item ?verification_event)
        (booking_item_resolution_path ?primary_booking_item ?resolution_path)
        (not
          (resolution_path_validated ?resolution_path)
        )
        (not
          (resolution_path_evidence_attached ?resolution_path)
        )
      )
    :effect (resolution_path_validated ?resolution_path)
  )
  (:action coordinate_partner_for_resolution
    :parameters (?primary_booking_item - primary_booking_item ?resolution_path - resolution_path ?partner_contact - partner_contact)
    :precondition
      (and
        (entity_verification_completed ?primary_booking_item)
        (entity_assigned_partner_contact ?primary_booking_item ?partner_contact)
        (booking_item_resolution_path ?primary_booking_item ?resolution_path)
        (resolution_path_validated ?resolution_path)
        (not
          (primary_booking_item_ready ?primary_booking_item)
        )
      )
    :effect
      (and
        (primary_booking_item_ready ?primary_booking_item)
        (primary_booking_item_ready_for_proposal ?primary_booking_item)
      )
  )
  (:action attach_evidence_to_booking_item_for_resolution
    :parameters (?primary_booking_item - primary_booking_item ?resolution_path - resolution_path ?evidence_document - evidence_document)
    :precondition
      (and
        (entity_verification_completed ?primary_booking_item)
        (booking_item_resolution_path ?primary_booking_item ?resolution_path)
        (evidence_document_available ?evidence_document)
        (not
          (primary_booking_item_ready ?primary_booking_item)
        )
      )
    :effect
      (and
        (resolution_path_evidence_attached ?resolution_path)
        (primary_booking_item_ready ?primary_booking_item)
        (primary_booking_item_evidence_reserved ?primary_booking_item ?evidence_document)
        (not
          (evidence_document_available ?evidence_document)
        )
      )
  )
  (:action process_evidence_and_validate_resolution
    :parameters (?primary_booking_item - primary_booking_item ?resolution_path - resolution_path ?verification_event - verification_event ?evidence_document - evidence_document)
    :precondition
      (and
        (entity_verification_completed ?primary_booking_item)
        (entity_has_verification_event ?primary_booking_item ?verification_event)
        (booking_item_resolution_path ?primary_booking_item ?resolution_path)
        (resolution_path_evidence_attached ?resolution_path)
        (primary_booking_item_evidence_reserved ?primary_booking_item ?evidence_document)
        (not
          (primary_booking_item_ready_for_proposal ?primary_booking_item)
        )
      )
    :effect
      (and
        (resolution_path_validated ?resolution_path)
        (primary_booking_item_ready_for_proposal ?primary_booking_item)
        (evidence_document_available ?evidence_document)
        (not
          (primary_booking_item_evidence_reserved ?primary_booking_item ?evidence_document)
        )
      )
  )
  (:action validate_compensation_path_for_secondary_item
    :parameters (?secondary_booking_item - secondary_booking_item ?compensation_path - compensation_path ?verification_event - verification_event)
    :precondition
      (and
        (entity_verification_completed ?secondary_booking_item)
        (entity_has_verification_event ?secondary_booking_item ?verification_event)
        (secondary_booking_item_compensation_path ?secondary_booking_item ?compensation_path)
        (not
          (compensation_path_validated ?compensation_path)
        )
        (not
          (compensation_path_evidence_attached ?compensation_path)
        )
      )
    :effect (compensation_path_validated ?compensation_path)
  )
  (:action coordinate_partner_for_compensation
    :parameters (?secondary_booking_item - secondary_booking_item ?compensation_path - compensation_path ?partner_contact - partner_contact)
    :precondition
      (and
        (entity_verification_completed ?secondary_booking_item)
        (entity_assigned_partner_contact ?secondary_booking_item ?partner_contact)
        (secondary_booking_item_compensation_path ?secondary_booking_item ?compensation_path)
        (compensation_path_validated ?compensation_path)
        (not
          (secondary_booking_item_ready ?secondary_booking_item)
        )
      )
    :effect
      (and
        (secondary_booking_item_ready ?secondary_booking_item)
        (secondary_booking_item_ready_for_proposal ?secondary_booking_item)
      )
  )
  (:action attach_evidence_to_secondary_item_for_compensation
    :parameters (?secondary_booking_item - secondary_booking_item ?compensation_path - compensation_path ?evidence_document - evidence_document)
    :precondition
      (and
        (entity_verification_completed ?secondary_booking_item)
        (secondary_booking_item_compensation_path ?secondary_booking_item ?compensation_path)
        (evidence_document_available ?evidence_document)
        (not
          (secondary_booking_item_ready ?secondary_booking_item)
        )
      )
    :effect
      (and
        (compensation_path_evidence_attached ?compensation_path)
        (secondary_booking_item_ready ?secondary_booking_item)
        (secondary_booking_item_evidence_reserved ?secondary_booking_item ?evidence_document)
        (not
          (evidence_document_available ?evidence_document)
        )
      )
  )
  (:action process_compensation_evidence_and_validate
    :parameters (?secondary_booking_item - secondary_booking_item ?compensation_path - compensation_path ?verification_event - verification_event ?evidence_document - evidence_document)
    :precondition
      (and
        (entity_verification_completed ?secondary_booking_item)
        (entity_has_verification_event ?secondary_booking_item ?verification_event)
        (secondary_booking_item_compensation_path ?secondary_booking_item ?compensation_path)
        (compensation_path_evidence_attached ?compensation_path)
        (secondary_booking_item_evidence_reserved ?secondary_booking_item ?evidence_document)
        (not
          (secondary_booking_item_ready_for_proposal ?secondary_booking_item)
        )
      )
    :effect
      (and
        (compensation_path_validated ?compensation_path)
        (secondary_booking_item_ready_for_proposal ?secondary_booking_item)
        (evidence_document_available ?evidence_document)
        (not
          (secondary_booking_item_evidence_reserved ?secondary_booking_item ?evidence_document)
        )
      )
  )
  (:action generate_repair_proposal
    :parameters (?primary_booking_item - primary_booking_item ?secondary_booking_item - secondary_booking_item ?resolution_path - resolution_path ?compensation_path - compensation_path ?repair_proposal - repair_proposal)
    :precondition
      (and
        (primary_booking_item_ready ?primary_booking_item)
        (secondary_booking_item_ready ?secondary_booking_item)
        (booking_item_resolution_path ?primary_booking_item ?resolution_path)
        (secondary_booking_item_compensation_path ?secondary_booking_item ?compensation_path)
        (resolution_path_validated ?resolution_path)
        (compensation_path_validated ?compensation_path)
        (primary_booking_item_ready_for_proposal ?primary_booking_item)
        (secondary_booking_item_ready_for_proposal ?secondary_booking_item)
        (repair_proposal_available ?repair_proposal)
      )
    :effect
      (and
        (repair_proposal_generated ?repair_proposal)
        (repair_proposal_resolution_path ?repair_proposal ?resolution_path)
        (repair_proposal_compensation_path ?repair_proposal ?compensation_path)
        (not
          (repair_proposal_available ?repair_proposal)
        )
      )
  )
  (:action generate_repair_proposal_policy_flagged
    :parameters (?primary_booking_item - primary_booking_item ?secondary_booking_item - secondary_booking_item ?resolution_path - resolution_path ?compensation_path - compensation_path ?repair_proposal - repair_proposal)
    :precondition
      (and
        (primary_booking_item_ready ?primary_booking_item)
        (secondary_booking_item_ready ?secondary_booking_item)
        (booking_item_resolution_path ?primary_booking_item ?resolution_path)
        (secondary_booking_item_compensation_path ?secondary_booking_item ?compensation_path)
        (resolution_path_evidence_attached ?resolution_path)
        (compensation_path_validated ?compensation_path)
        (not
          (primary_booking_item_ready_for_proposal ?primary_booking_item)
        )
        (secondary_booking_item_ready_for_proposal ?secondary_booking_item)
        (repair_proposal_available ?repair_proposal)
      )
    :effect
      (and
        (repair_proposal_generated ?repair_proposal)
        (repair_proposal_resolution_path ?repair_proposal ?resolution_path)
        (repair_proposal_compensation_path ?repair_proposal ?compensation_path)
        (repair_proposal_requires_policy_attachment ?repair_proposal)
        (not
          (repair_proposal_available ?repair_proposal)
        )
      )
  )
  (:action generate_repair_proposal_escalation_flagged
    :parameters (?primary_booking_item - primary_booking_item ?secondary_booking_item - secondary_booking_item ?resolution_path - resolution_path ?compensation_path - compensation_path ?repair_proposal - repair_proposal)
    :precondition
      (and
        (primary_booking_item_ready ?primary_booking_item)
        (secondary_booking_item_ready ?secondary_booking_item)
        (booking_item_resolution_path ?primary_booking_item ?resolution_path)
        (secondary_booking_item_compensation_path ?secondary_booking_item ?compensation_path)
        (resolution_path_validated ?resolution_path)
        (compensation_path_evidence_attached ?compensation_path)
        (primary_booking_item_ready_for_proposal ?primary_booking_item)
        (not
          (secondary_booking_item_ready_for_proposal ?secondary_booking_item)
        )
        (repair_proposal_available ?repair_proposal)
      )
    :effect
      (and
        (repair_proposal_generated ?repair_proposal)
        (repair_proposal_resolution_path ?repair_proposal ?resolution_path)
        (repair_proposal_compensation_path ?repair_proposal ?compensation_path)
        (repair_proposal_requires_escalation ?repair_proposal)
        (not
          (repair_proposal_available ?repair_proposal)
        )
      )
  )
  (:action generate_repair_proposal_with_all_flags
    :parameters (?primary_booking_item - primary_booking_item ?secondary_booking_item - secondary_booking_item ?resolution_path - resolution_path ?compensation_path - compensation_path ?repair_proposal - repair_proposal)
    :precondition
      (and
        (primary_booking_item_ready ?primary_booking_item)
        (secondary_booking_item_ready ?secondary_booking_item)
        (booking_item_resolution_path ?primary_booking_item ?resolution_path)
        (secondary_booking_item_compensation_path ?secondary_booking_item ?compensation_path)
        (resolution_path_evidence_attached ?resolution_path)
        (compensation_path_evidence_attached ?compensation_path)
        (not
          (primary_booking_item_ready_for_proposal ?primary_booking_item)
        )
        (not
          (secondary_booking_item_ready_for_proposal ?secondary_booking_item)
        )
        (repair_proposal_available ?repair_proposal)
      )
    :effect
      (and
        (repair_proposal_generated ?repair_proposal)
        (repair_proposal_resolution_path ?repair_proposal ?resolution_path)
        (repair_proposal_compensation_path ?repair_proposal ?compensation_path)
        (repair_proposal_requires_policy_attachment ?repair_proposal)
        (repair_proposal_requires_escalation ?repair_proposal)
        (not
          (repair_proposal_available ?repair_proposal)
        )
      )
  )
  (:action lock_repair_proposal
    :parameters (?repair_proposal - repair_proposal ?primary_booking_item - primary_booking_item ?verification_event - verification_event)
    :precondition
      (and
        (repair_proposal_generated ?repair_proposal)
        (primary_booking_item_ready ?primary_booking_item)
        (entity_has_verification_event ?primary_booking_item ?verification_event)
        (not
          (repair_proposal_locked ?repair_proposal)
        )
      )
    :effect (repair_proposal_locked ?repair_proposal)
  )
  (:action attach_documentation_to_proposal
    :parameters (?correction_case - correction_case ?documentation_package - documentation_package ?repair_proposal - repair_proposal)
    :precondition
      (and
        (entity_verification_completed ?correction_case)
        (case_has_repair_proposal ?correction_case ?repair_proposal)
        (case_has_documentation_package ?correction_case ?documentation_package)
        (documentation_package_available ?documentation_package)
        (repair_proposal_generated ?repair_proposal)
        (repair_proposal_locked ?repair_proposal)
        (not
          (documentation_package_assigned ?documentation_package)
        )
      )
    :effect
      (and
        (documentation_package_assigned ?documentation_package)
        (documentation_package_linked_to_proposal ?documentation_package ?repair_proposal)
        (not
          (documentation_package_available ?documentation_package)
        )
      )
  )
  (:action validate_documentation_package_for_case
    :parameters (?correction_case - correction_case ?documentation_package - documentation_package ?repair_proposal - repair_proposal ?verification_event - verification_event)
    :precondition
      (and
        (entity_verification_completed ?correction_case)
        (case_has_documentation_package ?correction_case ?documentation_package)
        (documentation_package_assigned ?documentation_package)
        (documentation_package_linked_to_proposal ?documentation_package ?repair_proposal)
        (entity_has_verification_event ?correction_case ?verification_event)
        (not
          (repair_proposal_requires_policy_attachment ?repair_proposal)
        )
        (not
          (case_documentation_validated ?correction_case)
        )
      )
    :effect (case_documentation_validated ?correction_case)
  )
  (:action attach_policy_to_case
    :parameters (?correction_case - correction_case ?policy - policy)
    :precondition
      (and
        (entity_verification_completed ?correction_case)
        (policy_available ?policy)
        (not
          (case_policy_attached ?correction_case)
        )
      )
    :effect
      (and
        (case_policy_attached ?correction_case)
        (case_policy_linked ?correction_case ?policy)
        (not
          (policy_available ?policy)
        )
      )
