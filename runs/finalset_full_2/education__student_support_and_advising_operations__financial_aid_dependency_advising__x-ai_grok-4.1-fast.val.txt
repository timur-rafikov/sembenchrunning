  (:requirements :strips :typing :negative-preconditions)

  (:predicates
    (case_intake_opened ?student_record - student_record)
    (entity_active ?student_record - student_record)
    (case_has_assigned_advisor ?student_record - student_record)
    (routed ?student_record - student_record)
    (ready_for_outcome ?student_record - student_record)

    (appointment_link ?student_record - student_record ?appointment_slot - appointment_time_slot)
    (support_staff_available ?support_staff - support_staff)
    (support_staff_assignment ?student_record - student_record ?support_staff - support_staff)
    (external_resource_available ?external_resource - external_resource)
    (primary_case_external_resource_link ?primary_student_case - primary_student_case ?external_resource - external_resource)
    (secondary_case_external_resource_link ?secondary_student_case - secondary_student_case ?external_resource - external_resource)
    (case_linked_to_dependency_issue ?primary_student_case - primary_student_case ?dependency_issue_flag - dependency_issue_flag)
    (dependency_issue_flag_open ?dependency_issue_flag - dependency_issue_flag)
    (dependency_issue_flag_has_resource ?dependency_issue_flag - dependency_issue_flag)
    (case_ready_for_documentation ?primary_student_case - primary_student_case)
    (secondary_case_financial_concern_link ?secondary_student_case - secondary_student_case ?financial_concern_record - financial_concern_record)
    (financial_record_ready_for_processing ?financial_concern_record - financial_concern_record)
    (financial_record_response_received ?financial_concern_record - financial_concern_record)
    (secondary_case_ready_for_documentation ?secondary_student_case - secondary_student_case)
    (documentation_package_available ?documentation_package - documentation_package)
    (documentation_package_created ?documentation_package - documentation_package)
    (documentation_package_linked_to_issue ?documentation_package - documentation_package ?dependency_issue_flag - dependency_issue_flag)
    (documentation_package_linked_to_financial_record ?documentation_package - documentation_package ?financial_concern_record - financial_concern_record)
    (documentation_package_in_review_stage1 ?documentation_package - documentation_package)
    (documentation_package_in_review_stage2 ?documentation_package - documentation_package)
    (documentation_package_finalized ?documentation_package - documentation_package)
    (manager_assigned_to_primary_case ?case_manager - case_manager ?primary_student_case - primary_student_case)
    (manager_assigned_to_secondary_case ?case_manager - case_manager ?secondary_student_case - secondary_student_case)
    (manager_linked_to_documentation_package ?case_manager - case_manager ?documentation_package - documentation_package)
    (evidence_document_available ?evidence_document - evidence_document)
    (manager_has_evidence_document ?case_manager - case_manager ?evidence_document - evidence_document)
    (evidence_document_processed ?evidence_document - evidence_document)
    (evidence_document_linked_to_package ?evidence_document - evidence_document ?documentation_package - documentation_package)
    (manager_ready_for_decision_review ?case_manager - case_manager)
    (manager_recommendation_recorded ?case_manager - case_manager)
    (manager_final_review_started ?case_manager - case_manager)
    (policy_exception_request_assigned ?case_manager - case_manager)
    (policy_exception_review_in_progress ?case_manager - case_manager)
    (manager_documentation_verified ?case_manager - case_manager)
    (manager_completed_final_checks ?case_manager - case_manager)
    (consent_form_available ?consent_form - consent_form)
    (manager_linked_to_consent_form ?case_manager - case_manager ?consent_form - consent_form)
    (consent_acknowledged_by_manager ?case_manager - case_manager)
    (manager_prepared_for_partner_contact ?case_manager - case_manager)
    (manager_partner_contact_initiated ?case_manager - case_manager)
    (policy_exception_request_available ?policy_exception_request - policy_exception_request_document)
    (manager_linked_to_policy_exception_request ?case_manager - case_manager ?policy_exception_request - policy_exception_request_document)
    (service_option_available ?service_option - service_option_resource)
    (manager_linked_to_service_option ?case_manager - case_manager ?service_option - service_option_resource)
    (intervention_plan_available ?intervention_plan - intervention_plan_document)
    (intervention_plan_assigned_to_manager ?case_manager - case_manager ?intervention_plan - intervention_plan_document)
    (external_partner_available ?external_partner - external_partner_org)
    (manager_linked_to_external_partner ?case_manager - case_manager ?external_partner - external_partner_org)
    (appeal_record_available ?appeal_record - appeal_record_document)
    (linked_to_appeal_record ?student_record - student_record ?appeal_record - appeal_record_document)
    (primary_case_triage_completed ?primary_student_case - primary_student_case)
    (secondary_case_triage_completed ?secondary_student_case - secondary_student_case)
    (manager_signed_off_for_outcome ?case_manager - case_manager)
  )
  (:action open_student_intake
    :parameters (?student_record - student_record)
    :precondition
      (and
        (not
          (case_intake_opened ?student_record)
        )
        (not
          (routed ?student_record)
        )
      )
    :effect (case_intake_opened ?student_record)
  )
  (:action assign_advisor_to_student
    :parameters (?student_record - student_record ?advisor - advisor)
    :precondition
      (and
        (case_intake_opened ?student_record)
        (not
          (case_has_assigned_advisor ?student_record)
        )
        (advisor_available ?advisor)
      )
    :effect
      (and
        (case_has_assigned_advisor ?student_record)
        (assigned_advisor_link ?student_record ?advisor)
        (not
          (advisor_available ?advisor)
        )
      )
  )
  (:action schedule_appointment_for_student
    :parameters (?student_record - student_record ?appointment_slot - appointment_time_slot)
    :precondition
      (and
        (case_intake_opened ?student_record)
        (case_has_assigned_advisor ?student_record)
        (appointment_slot_available ?appointment_slot)
      )
    :effect
      (and
        (appointment_link ?student_record ?appointment_slot)
        (not
          (appointment_slot_available ?appointment_slot)
        )
      )
  )
  (:action activate_case_for_student
    :parameters (?student_record - student_record ?appointment_slot - appointment_time_slot)
    :precondition
      (and
        (case_intake_opened ?student_record)
        (case_has_assigned_advisor ?student_record)
        (appointment_link ?student_record ?appointment_slot)
        (not
          (entity_active ?student_record)
        )
      )
    :effect (entity_active ?student_record)
  )
  (:action release_appointment_slot_for_student
    :parameters (?student_record - student_record ?appointment_slot - appointment_time_slot)
    :precondition
      (and
        (appointment_link ?student_record ?appointment_slot)
      )
    :effect
      (and
        (appointment_slot_available ?appointment_slot)
        (not
          (appointment_link ?student_record ?appointment_slot)
        )
      )
  )
  (:action assign_support_staff_to_student
    :parameters (?student_record - student_record ?support_staff - support_staff)
    :precondition
      (and
        (entity_active ?student_record)
        (support_staff_available ?support_staff)
      )
    :effect
      (and
        (support_staff_assignment ?student_record ?support_staff)
        (not
          (support_staff_available ?support_staff)
        )
      )
  )
  (:action release_support_staff_assignment
    :parameters (?student_record - student_record ?support_staff - support_staff)
    :precondition
      (and
        (support_staff_assignment ?student_record ?support_staff)
      )
    :effect
      (and
        (support_staff_available ?support_staff)
        (not
          (support_staff_assignment ?student_record ?support_staff)
        )
      )
  )
  (:action assign_intervention_plan_to_manager
    :parameters (?case_manager - case_manager ?intervention_plan - intervention_plan_document)
    :precondition
      (and
        (entity_active ?case_manager)
        (intervention_plan_available ?intervention_plan)
      )
    :effect
      (and
        (intervention_plan_assigned_to_manager ?case_manager ?intervention_plan)
        (not
          (intervention_plan_available ?intervention_plan)
        )
      )
  )
  (:action withdraw_intervention_plan_from_manager
    :parameters (?case_manager - case_manager ?intervention_plan - intervention_plan_document)
    :precondition
      (and
        (intervention_plan_assigned_to_manager ?case_manager ?intervention_plan)
      )
    :effect
      (and
        (intervention_plan_available ?intervention_plan)
        (not
          (intervention_plan_assigned_to_manager ?case_manager ?intervention_plan)
        )
      )
  )
  (:action assign_external_partner_to_manager
    :parameters (?case_manager - case_manager ?external_partner - external_partner_org)
    :precondition
      (and
        (entity_active ?case_manager)
        (external_partner_available ?external_partner)
      )
    :effect
      (and
        (manager_linked_to_external_partner ?case_manager ?external_partner)
        (not
          (external_partner_available ?external_partner)
        )
      )
  )
  (:action remove_external_partner_assignment_from_manager
    :parameters (?case_manager - case_manager ?external_partner - external_partner_org)
    :precondition
      (and
        (manager_linked_to_external_partner ?case_manager ?external_partner)
      )
    :effect
      (and
        (external_partner_available ?external_partner)
        (not
          (manager_linked_to_external_partner ?case_manager ?external_partner)
        )
      )
  )
  (:action open_dependency_issue_flag
    :parameters (?primary_student_case - primary_student_case ?dependency_issue_flag - dependency_issue_flag ?appointment_slot - appointment_time_slot)
    :precondition
      (and
        (entity_active ?primary_student_case)
        (appointment_link ?primary_student_case ?appointment_slot)
        (case_linked_to_dependency_issue ?primary_student_case ?dependency_issue_flag)
        (not
          (dependency_issue_flag_open ?dependency_issue_flag)
        )
        (not
          (dependency_issue_flag_has_resource ?dependency_issue_flag)
        )
      )
    :effect (dependency_issue_flag_open ?dependency_issue_flag)
  )
  (:action triage_dependency_issue_by_support_staff
    :parameters (?primary_student_case - primary_student_case ?dependency_issue_flag - dependency_issue_flag ?support_staff - support_staff)
    :precondition
      (and
        (entity_active ?primary_student_case)
        (support_staff_assignment ?primary_student_case ?support_staff)
        (case_linked_to_dependency_issue ?primary_student_case ?dependency_issue_flag)
        (dependency_issue_flag_open ?dependency_issue_flag)
        (not
          (primary_case_triage_completed ?primary_student_case)
        )
      )
    :effect
      (and
        (primary_case_triage_completed ?primary_student_case)
        (case_ready_for_documentation ?primary_student_case)
      )
  )
  (:action request_external_resource_for_primary_case
    :parameters (?primary_student_case - primary_student_case ?dependency_issue_flag - dependency_issue_flag ?external_resource - external_resource)
    :precondition
      (and
        (entity_active ?primary_student_case)
        (case_linked_to_dependency_issue ?primary_student_case ?dependency_issue_flag)
        (external_resource_available ?external_resource)
        (not
          (primary_case_triage_completed ?primary_student_case)
        )
      )
    :effect
      (and
        (dependency_issue_flag_has_resource ?dependency_issue_flag)
        (primary_case_triage_completed ?primary_student_case)
        (primary_case_external_resource_link ?primary_student_case ?external_resource)
        (not
          (external_resource_available ?external_resource)
        )
      )
  )
  (:action process_external_resource_response_for_primary_case
    :parameters (?primary_student_case - primary_student_case ?dependency_issue_flag - dependency_issue_flag ?appointment_slot - appointment_time_slot ?external_resource - external_resource)
    :precondition
      (and
        (entity_active ?primary_student_case)
        (appointment_link ?primary_student_case ?appointment_slot)
        (case_linked_to_dependency_issue ?primary_student_case ?dependency_issue_flag)
        (dependency_issue_flag_has_resource ?dependency_issue_flag)
        (primary_case_external_resource_link ?primary_student_case ?external_resource)
        (not
          (case_ready_for_documentation ?primary_student_case)
        )
      )
    :effect
      (and
        (dependency_issue_flag_open ?dependency_issue_flag)
        (case_ready_for_documentation ?primary_student_case)
        (external_resource_available ?external_resource)
        (not
          (primary_case_external_resource_link ?primary_student_case ?external_resource)
        )
      )
  )
  (:action mark_financial_record_ready_for_processing
    :parameters (?secondary_student_case - secondary_student_case ?financial_concern_record - financial_concern_record ?appointment_slot - appointment_time_slot)
    :precondition
      (and
        (entity_active ?secondary_student_case)
        (appointment_link ?secondary_student_case ?appointment_slot)
        (secondary_case_financial_concern_link ?secondary_student_case ?financial_concern_record)
        (not
          (financial_record_ready_for_processing ?financial_concern_record)
        )
        (not
          (financial_record_response_received ?financial_concern_record)
        )
      )
    :effect (financial_record_ready_for_processing ?financial_concern_record)
  )
  (:action assign_support_staff_to_secondary_case
    :parameters (?secondary_student_case - secondary_student_case ?financial_concern_record - financial_concern_record ?support_staff - support_staff)
    :precondition
      (and
        (entity_active ?secondary_student_case)
        (support_staff_assignment ?secondary_student_case ?support_staff)
        (secondary_case_financial_concern_link ?secondary_student_case ?financial_concern_record)
        (financial_record_ready_for_processing ?financial_concern_record)
        (not
          (secondary_case_triage_completed ?secondary_student_case)
        )
      )
    :effect
      (and
        (secondary_case_triage_completed ?secondary_student_case)
        (secondary_case_ready_for_documentation ?secondary_student_case)
      )
  )
  (:action request_external_resource_for_secondary_case
    :parameters (?secondary_student_case - secondary_student_case ?financial_concern_record - financial_concern_record ?external_resource - external_resource)
    :precondition
      (and
        (entity_active ?secondary_student_case)
        (secondary_case_financial_concern_link ?secondary_student_case ?financial_concern_record)
        (external_resource_available ?external_resource)
        (not
          (secondary_case_triage_completed ?secondary_student_case)
        )
      )
    :effect
      (and
        (financial_record_response_received ?financial_concern_record)
        (secondary_case_triage_completed ?secondary_student_case)
        (secondary_case_external_resource_link ?secondary_student_case ?external_resource)
        (not
          (external_resource_available ?external_resource)
        )
      )
  )
  (:action process_external_resource_response_for_secondary_case
    :parameters (?secondary_student_case - secondary_student_case ?financial_concern_record - financial_concern_record ?appointment_slot - appointment_time_slot ?external_resource - external_resource)
    :precondition
      (and
        (entity_active ?secondary_student_case)
        (appointment_link ?secondary_student_case ?appointment_slot)
        (secondary_case_financial_concern_link ?secondary_student_case ?financial_concern_record)
        (financial_record_response_received ?financial_concern_record)
        (secondary_case_external_resource_link ?secondary_student_case ?external_resource)
        (not
          (secondary_case_ready_for_documentation ?secondary_student_case)
        )
      )
    :effect
      (and
        (financial_record_ready_for_processing ?financial_concern_record)
        (secondary_case_ready_for_documentation ?secondary_student_case)
        (external_resource_available ?external_resource)
        (not
          (secondary_case_external_resource_link ?secondary_student_case ?external_resource)
        )
      )
  )
  (:action create_documentation_package
    :parameters (?primary_student_case - primary_student_case ?secondary_student_case - secondary_student_case ?dependency_issue_flag - dependency_issue_flag ?financial_concern_record - financial_concern_record ?documentation_package - documentation_package)
    :precondition
      (and
        (primary_case_triage_completed ?primary_student_case)
        (secondary_case_triage_completed ?secondary_student_case)
        (case_linked_to_dependency_issue ?primary_student_case ?dependency_issue_flag)
        (secondary_case_financial_concern_link ?secondary_student_case ?financial_concern_record)
        (dependency_issue_flag_open ?dependency_issue_flag)
        (financial_record_ready_for_processing ?financial_concern_record)
        (case_ready_for_documentation ?primary_student_case)
        (secondary_case_ready_for_documentation ?secondary_student_case)
        (documentation_package_available ?documentation_package)
      )
    :effect
      (and
        (documentation_package_created ?documentation_package)
        (documentation_package_linked_to_issue ?documentation_package ?dependency_issue_flag)
        (documentation_package_linked_to_financial_record ?documentation_package ?financial_concern_record)
        (not
          (documentation_package_available ?documentation_package)
        )
      )
  )
  (:action create_documentation_package_with_stage1
    :parameters (?primary_student_case - primary_student_case ?secondary_student_case - secondary_student_case ?dependency_issue_flag - dependency_issue_flag ?financial_concern_record - financial_concern_record ?documentation_package - documentation_package)
    :precondition
      (and
        (primary_case_triage_completed ?primary_student_case)
        (secondary_case_triage_completed ?secondary_student_case)
        (case_linked_to_dependency_issue ?primary_student_case ?dependency_issue_flag)
        (secondary_case_financial_concern_link ?secondary_student_case ?financial_concern_record)
        (dependency_issue_flag_has_resource ?dependency_issue_flag)
        (financial_record_ready_for_processing ?financial_concern_record)
        (not
          (case_ready_for_documentation ?primary_student_case)
        )
        (secondary_case_ready_for_documentation ?secondary_student_case)
        (documentation_package_available ?documentation_package)
      )
    :effect
      (and
        (documentation_package_created ?documentation_package)
        (documentation_package_linked_to_issue ?documentation_package ?dependency_issue_flag)
        (documentation_package_linked_to_financial_record ?documentation_package ?financial_concern_record)
        (documentation_package_in_review_stage1 ?documentation_package)
        (not
          (documentation_package_available ?documentation_package)
        )
      )
  )
  (:action create_documentation_package_with_stage2
    :parameters (?primary_student_case - primary_student_case ?secondary_student_case - secondary_student_case ?dependency_issue_flag - dependency_issue_flag ?financial_concern_record - financial_concern_record ?documentation_package - documentation_package)
    :precondition
      (and
        (primary_case_triage_completed ?primary_student_case)
        (secondary_case_triage_completed ?secondary_student_case)
        (case_linked_to_dependency_issue ?primary_student_case ?dependency_issue_flag)
        (secondary_case_financial_concern_link ?secondary_student_case ?financial_concern_record)
        (dependency_issue_flag_open ?dependency_issue_flag)
        (financial_record_response_received ?financial_concern_record)
        (case_ready_for_documentation ?primary_student_case)
        (not
          (secondary_case_ready_for_documentation ?secondary_student_case)
        )
        (documentation_package_available ?documentation_package)
      )
    :effect
      (and
        (documentation_package_created ?documentation_package)
        (documentation_package_linked_to_issue ?documentation_package ?dependency_issue_flag)
        (documentation_package_linked_to_financial_record ?documentation_package ?financial_concern_record)
        (documentation_package_in_review_stage2 ?documentation_package)
        (not
          (documentation_package_available ?documentation_package)
        )
      )
  )
  (:action create_full_documentation_package
    :parameters (?primary_student_case - primary_student_case ?secondary_student_case - secondary_student_case ?dependency_issue_flag - dependency_issue_flag ?financial_concern_record - financial_concern_record ?documentation_package - documentation_package)
    :precondition
      (and
        (primary_case_triage_completed ?primary_student_case)
        (secondary_case_triage_completed ?secondary_student_case)
        (case_linked_to_dependency_issue ?primary_student_case ?dependency_issue_flag)
        (secondary_case_financial_concern_link ?secondary_student_case ?financial_concern_record)
        (dependency_issue_flag_has_resource ?dependency_issue_flag)

        (not
          (case_ready_for_documentation ?primary_student_case)
        )
        (not
          (secondary_case_ready_for_documentation ?secondary_student_case)
        )
        (documentation_package_available ?documentation_package)
      )
    :effect
      (and
        (documentation_package_created ?documentation_package)
        (documentation_package_linked_to_issue ?documentation_package ?dependency_issue_flag)
        (documentation_package_linked_to_financial_record ?documentation_package ?financial_concern_record)
        (documentation_package_in_review_stage1 ?documentation_package)
        (documentation_package_in_review_stage2 ?documentation_package)
        (not
          (documentation_package_available ?documentation_package)
        )
      )
  )
  (:action finalize_documentation_package
    :parameters (?documentation_package - documentation_package ?primary_student_case - primary_student_case ?appointment_slot - appointment_time_slot)
    :precondition
      (and
        (documentation_package_created ?documentation_package)
        (primary_case_triage_completed ?primary_student_case)
        (appointment_link ?primary_student_case ?appointment_slot)
        (not
          (documentation_package_finalized ?documentation_package)
        )
      )
    :effect (documentation_package_finalized ?documentation_package)
  )
  (:action attach_evidence_document_to_package
    :parameters (?case_manager - case_manager ?evidence_document - evidence_document ?documentation_package - documentation_package)
    :precondition
      (and
        (entity_active ?case_manager)
        (manager_linked_to_documentation_package ?case_manager ?documentation_package)
        (manager_has_evidence_document ?case_manager ?evidence_document)
        (evidence_document_available ?evidence_document)
        (documentation_package_created ?documentation_package)
        (documentation_package_finalized ?documentation_package)
        (not
          (evidence_document_processed ?evidence_document)
        )
      )
    :effect
      (and
        (evidence_document_processed ?evidence_document)
        (evidence_document_linked_to_package ?evidence_document ?documentation_package)
        (not
          (evidence_document_available ?evidence_document)
        )
      )
  )
  (:action verify_evidence_document_for_package
    :parameters (?case_manager - case_manager ?evidence_document - evidence_document ?documentation_package - documentation_package ?appointment_slot - appointment_time_slot)
    :precondition
      (and
        (entity_active ?case_manager)
        (manager_has_evidence_document ?case_manager ?evidence_document)
        (evidence_document_processed ?evidence_document)
        (evidence_document_linked_to_package ?evidence_document ?documentation_package)
        (appointment_link ?case_manager ?appointment_slot)
        (not
          (documentation_package_in_review_stage1 ?documentation_package)
        )
        (not
          (manager_ready_for_decision_review ?case_manager)
        )
      )
    :effect (manager_ready_for_decision_review ?case_manager)
  )
  (:action file_policy_exception_request
    :parameters (?case_manager - case_manager ?policy_exception_request - policy_exception_request_document)
    :precondition
      (and
        (entity_active ?case_manager)
        (policy_exception_request_available ?policy_exception_request)
        (not
          (policy_exception_request_assigned ?case_manager)
        )
      )
    :effect
      (and
        (policy_exception_request_assigned ?case_manager)
        (manager_linked_to_policy_exception_request ?case_manager ?policy_exception_request)
        (not
          (policy_exception_request_available ?policy_exception_request)
        )
      )
  )
  (:action review_policy_exception_with_package
    :parameters (?case_manager - case_manager ?evidence_document - evidence_document ?documentation_package - documentation_package ?appointment_slot - appointment_time_slot ?policy_exception_request - policy_exception_request_document)
    :precondition
      (and
        (entity_active ?case_manager)
        (manager_has_evidence_document ?case_manager ?evidence_document)
        (evidence_document_processed ?evidence_document)
