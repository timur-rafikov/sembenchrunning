--- domain.pddl (as passed to VAL) ---
(define (domain rescue_mission_dependency_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types operational_asset_pool - object logistics_pool - object routing_pool - object mission_category - object mission_element - mission_category responder_unit - operational_asset_pool specialist_unit - operational_asset_pool support_npc - operational_asset_pool optional_module - operational_asset_pool upgrade_module - operational_asset_pool mission_token - operational_asset_pool equipment_unit - operational_asset_pool event_flag - operational_asset_pool consumable_resource - logistics_pool asset_component - logistics_pool unique_key_item - logistics_pool approach_route - routing_pool extraction_route - routing_pool extraction_asset - routing_pool site_role_group - mission_element operation_role_group - mission_element rescue_site - site_role_group support_site - site_role_group mission_instance - operation_role_group)
  (:predicates
    (element_active ?mission_element - mission_element)
    (element_secured ?mission_element - mission_element)
    (element_assigned ?mission_element - mission_element)
    (element_resolved ?mission_element - mission_element)
    (element_finalization_unlocked ?mission_element - mission_element)
    (element_extraction_ready ?mission_element - mission_element)
    (responder_available ?responder_unit - responder_unit)
    (responder_assigned_to_element ?mission_element - mission_element ?responder_unit - responder_unit)
    (specialist_available ?specialist_unit - specialist_unit)
    (specialist_assigned_to_element ?mission_element - mission_element ?specialist_unit - specialist_unit)
    (support_npc_available ?support_npc - support_npc)
    (support_npc_assigned_to_element ?mission_element - mission_element ?support_npc - support_npc)
    (consumable_available ?consumable_resource - consumable_resource)
    (rescue_site_consumable_reserved ?rescue_site - rescue_site ?consumable_resource - consumable_resource)
    (support_site_consumable_reserved ?support_site - support_site ?consumable_resource - consumable_resource)
    (approach_route_assigned_to_site ?rescue_site - rescue_site ?approach_route - approach_route)
    (approach_route_staged ?approach_route - approach_route)
    (approach_route_prepared ?approach_route - approach_route)
    (rescue_site_verified ?rescue_site - rescue_site)
    (extraction_route_assigned_to_support_site ?support_site - support_site ?extraction_route - extraction_route)
    (extraction_route_staged ?extraction_route - extraction_route)
    (extraction_route_prepared ?extraction_route - extraction_route)
    (support_site_verified ?support_site - support_site)
    (extraction_asset_available ?extraction_asset - extraction_asset)
    (extraction_asset_configured ?extraction_asset - extraction_asset)
    (extraction_asset_bound_to_approach_route ?extraction_asset - extraction_asset ?approach_route - approach_route)
    (extraction_asset_bound_to_extraction_route ?extraction_asset - extraction_asset ?extraction_route - extraction_route)
    (extraction_asset_attachment_stage_one ?extraction_asset - extraction_asset)
    (extraction_asset_attachment_stage_two ?extraction_asset - extraction_asset)
    (extraction_asset_ready ?extraction_asset - extraction_asset)
    (instance_includes_rescue_site ?mission_instance - mission_instance ?rescue_site - rescue_site)
    (instance_includes_support_site ?mission_instance - mission_instance ?support_site - support_site)
    (instance_includes_extraction_asset ?mission_instance - mission_instance ?extraction_asset - extraction_asset)
    (asset_component_available ?asset_component - asset_component)
    (instance_includes_component ?mission_instance - mission_instance ?asset_component - asset_component)
    (asset_component_installed ?asset_component - asset_component)
    (component_attached_to_extraction_asset ?asset_component - asset_component ?extraction_asset - extraction_asset)
    (instance_base_configured ?mission_instance - mission_instance)
    (instance_module_attached ?mission_instance - mission_instance)
    (instance_ready_for_finalization ?mission_instance - mission_instance)
    (instance_optional_module_attached ?mission_instance - mission_instance)
    (instance_optional_module_activated ?mission_instance - mission_instance)
    (instance_upgrade_installed ?mission_instance - mission_instance)
    (instance_final_checks_passed ?mission_instance - mission_instance)
    (unique_key_item_available ?unique_key_item - unique_key_item)
    (instance_has_key_item ?mission_instance - mission_instance ?unique_key_item - unique_key_item)
    (instance_alternate_unlocked ?mission_instance - mission_instance)
    (instance_option_enabled ?mission_instance - mission_instance)
    (instance_option_activated ?mission_instance - mission_instance)
    (optional_module_available ?optional_module - optional_module)
    (instance_has_optional_module ?mission_instance - mission_instance ?optional_module - optional_module)
    (upgrade_module_available ?upgrade_module - upgrade_module)
    (instance_has_upgrade_module ?mission_instance - mission_instance ?upgrade_module - upgrade_module)
    (equipment_unit_available ?equipment_unit - equipment_unit)
    (instance_has_equipment_unit ?mission_instance - mission_instance ?equipment_unit - equipment_unit)
    (event_flag_available ?event_flag - event_flag)
    (instance_has_event_flag ?mission_instance - mission_instance ?event_flag - event_flag)
    (mission_token_available ?mission_token - mission_token)
    (element_token_bound ?mission_element - mission_element ?mission_token - mission_token)
    (rescue_site_prepped ?rescue_site - rescue_site)
    (support_site_prepped ?support_site - support_site)
    (instance_reward_granted ?mission_instance - mission_instance)
  )
  (:action activate_mission_element
    :parameters (?mission_element - mission_element)
    :precondition
      (and
        (not
          (element_active ?mission_element)
        )
        (not
          (element_resolved ?mission_element)
        )
      )
    :effect (element_active ?mission_element)
  )
  (:action assign_responder_to_element
    :parameters (?mission_element - mission_element ?responder_unit - responder_unit)
    :precondition
      (and
        (element_active ?mission_element)
        (not
          (element_assigned ?mission_element)
        )
        (responder_available ?responder_unit)
      )
    :effect
      (and
        (element_assigned ?mission_element)
        (responder_assigned_to_element ?mission_element ?responder_unit)
        (not
          (responder_available ?responder_unit)
        )
      )
  )
  (:action assign_specialist_to_element
    :parameters (?mission_element - mission_element ?specialist_unit - specialist_unit)
    :precondition
      (and
        (element_active ?mission_element)
        (element_assigned ?mission_element)
        (specialist_available ?specialist_unit)
      )
    :effect
      (and
        (specialist_assigned_to_element ?mission_element ?specialist_unit)
        (not
          (specialist_available ?specialist_unit)
        )
      )
  )
  (:action secure_mission_element
    :parameters (?mission_element - mission_element ?specialist_unit - specialist_unit)
    :precondition
      (and
        (element_active ?mission_element)
        (element_assigned ?mission_element)
        (specialist_assigned_to_element ?mission_element ?specialist_unit)
        (not
          (element_secured ?mission_element)
        )
      )
    :effect (element_secured ?mission_element)
  )
  (:action release_specialist_from_element
    :parameters (?mission_element - mission_element ?specialist_unit - specialist_unit)
    :precondition
      (and
        (specialist_assigned_to_element ?mission_element ?specialist_unit)
      )
    :effect
      (and
        (specialist_available ?specialist_unit)
        (not
          (specialist_assigned_to_element ?mission_element ?specialist_unit)
        )
      )
  )
  (:action assign_support_npc_to_element
    :parameters (?mission_element - mission_element ?support_npc - support_npc)
    :precondition
      (and
        (element_secured ?mission_element)
        (support_npc_available ?support_npc)
      )
    :effect
      (and
        (support_npc_assigned_to_element ?mission_element ?support_npc)
        (not
          (support_npc_available ?support_npc)
        )
      )
  )
  (:action release_support_npc_from_element
    :parameters (?mission_element - mission_element ?support_npc - support_npc)
    :precondition
      (and
        (support_npc_assigned_to_element ?mission_element ?support_npc)
      )
    :effect
      (and
        (support_npc_available ?support_npc)
        (not
          (support_npc_assigned_to_element ?mission_element ?support_npc)
        )
      )
  )
  (:action attach_equipment_to_instance
    :parameters (?mission_instance - mission_instance ?equipment_unit - equipment_unit)
    :precondition
      (and
        (element_secured ?mission_instance)
        (equipment_unit_available ?equipment_unit)
      )
    :effect
      (and
        (instance_has_equipment_unit ?mission_instance ?equipment_unit)
        (not
          (equipment_unit_available ?equipment_unit)
        )
      )
  )
  (:action detach_equipment_from_instance
    :parameters (?mission_instance - mission_instance ?equipment_unit - equipment_unit)
    :precondition
      (and
        (instance_has_equipment_unit ?mission_instance ?equipment_unit)
      )
    :effect
      (and
        (equipment_unit_available ?equipment_unit)
        (not
          (instance_has_equipment_unit ?mission_instance ?equipment_unit)
        )
      )
  )
  (:action attach_event_flag_to_instance
    :parameters (?mission_instance - mission_instance ?event_flag - event_flag)
    :precondition
      (and
        (element_secured ?mission_instance)
        (event_flag_available ?event_flag)
      )
    :effect
      (and
        (instance_has_event_flag ?mission_instance ?event_flag)
        (not
          (event_flag_available ?event_flag)
        )
      )
  )
  (:action detach_event_flag_from_instance
    :parameters (?mission_instance - mission_instance ?event_flag - event_flag)
    :precondition
      (and
        (instance_has_event_flag ?mission_instance ?event_flag)
      )
    :effect
      (and
        (event_flag_available ?event_flag)
        (not
          (instance_has_event_flag ?mission_instance ?event_flag)
        )
      )
  )
  (:action stage_approach_route
    :parameters (?rescue_site - rescue_site ?approach_route - approach_route ?specialist_unit - specialist_unit)
    :precondition
      (and
        (element_secured ?rescue_site)
        (specialist_assigned_to_element ?rescue_site ?specialist_unit)
        (approach_route_assigned_to_site ?rescue_site ?approach_route)
        (not
          (approach_route_staged ?approach_route)
        )
        (not
          (approach_route_prepared ?approach_route)
        )
      )
    :effect (approach_route_staged ?approach_route)
  )
  (:action confirm_rescue_site_readiness
    :parameters (?rescue_site - rescue_site ?approach_route - approach_route ?support_npc - support_npc)
    :precondition
      (and
        (element_secured ?rescue_site)
        (support_npc_assigned_to_element ?rescue_site ?support_npc)
        (approach_route_assigned_to_site ?rescue_site ?approach_route)
        (approach_route_staged ?approach_route)
        (not
          (rescue_site_prepped ?rescue_site)
        )
      )
    :effect
      (and
        (rescue_site_prepped ?rescue_site)
        (rescue_site_verified ?rescue_site)
      )
  )
  (:action apply_consumable_to_site
    :parameters (?rescue_site - rescue_site ?approach_route - approach_route ?consumable_resource - consumable_resource)
    :precondition
      (and
        (element_secured ?rescue_site)
        (approach_route_assigned_to_site ?rescue_site ?approach_route)
        (consumable_available ?consumable_resource)
        (not
          (rescue_site_prepped ?rescue_site)
        )
      )
    :effect
      (and
        (approach_route_prepared ?approach_route)
        (rescue_site_prepped ?rescue_site)
        (rescue_site_consumable_reserved ?rescue_site ?consumable_resource)
        (not
          (consumable_available ?consumable_resource)
        )
      )
  )
  (:action finalize_rescue_site_preparation
    :parameters (?rescue_site - rescue_site ?approach_route - approach_route ?specialist_unit - specialist_unit ?consumable_resource - consumable_resource)
    :precondition
      (and
        (element_secured ?rescue_site)
        (specialist_assigned_to_element ?rescue_site ?specialist_unit)
        (approach_route_assigned_to_site ?rescue_site ?approach_route)
        (approach_route_prepared ?approach_route)
        (rescue_site_consumable_reserved ?rescue_site ?consumable_resource)
        (not
          (rescue_site_verified ?rescue_site)
        )
      )
    :effect
      (and
        (approach_route_staged ?approach_route)
        (rescue_site_verified ?rescue_site)
        (consumable_available ?consumable_resource)
        (not
          (rescue_site_consumable_reserved ?rescue_site ?consumable_resource)
        )
      )
  )
  (:action stage_extraction_route
    :parameters (?support_site - support_site ?extraction_route - extraction_route ?specialist_unit - specialist_unit)
    :precondition
      (and
        (element_secured ?support_site)
        (specialist_assigned_to_element ?support_site ?specialist_unit)
        (extraction_route_assigned_to_support_site ?support_site ?extraction_route)
        (not
          (extraction_route_staged ?extraction_route)
        )
        (not
          (extraction_route_prepared ?extraction_route)
        )
      )
    :effect (extraction_route_staged ?extraction_route)
  )
  (:action confirm_support_site_readiness
    :parameters (?support_site - support_site ?extraction_route - extraction_route ?support_npc - support_npc)
    :precondition
      (and
        (element_secured ?support_site)
        (support_npc_assigned_to_element ?support_site ?support_npc)
        (extraction_route_assigned_to_support_site ?support_site ?extraction_route)
        (extraction_route_staged ?extraction_route)
        (not
          (support_site_prepped ?support_site)
        )
      )
    :effect
      (and
        (support_site_prepped ?support_site)
        (support_site_verified ?support_site)
      )
  )
  (:action apply_consumable_to_support_site
    :parameters (?support_site - support_site ?extraction_route - extraction_route ?consumable_resource - consumable_resource)
    :precondition
      (and
        (element_secured ?support_site)
        (extraction_route_assigned_to_support_site ?support_site ?extraction_route)
        (consumable_available ?consumable_resource)
        (not
          (support_site_prepped ?support_site)
        )
      )
    :effect
      (and
        (extraction_route_prepared ?extraction_route)
        (support_site_prepped ?support_site)
        (support_site_consumable_reserved ?support_site ?consumable_resource)
        (not
          (consumable_available ?consumable_resource)
        )
      )
  )
  (:action finalize_support_site_preparation
    :parameters (?support_site - support_site ?extraction_route - extraction_route ?specialist_unit - specialist_unit ?consumable_resource - consumable_resource)
    :precondition
      (and
        (element_secured ?support_site)
        (specialist_assigned_to_element ?support_site ?specialist_unit)
        (extraction_route_assigned_to_support_site ?support_site ?extraction_route)
        (extraction_route_prepared ?extraction_route)
        (support_site_consumable_reserved ?support_site ?consumable_resource)
        (not
          (support_site_verified ?support_site)
        )
      )
    :effect
      (and
        (extraction_route_staged ?extraction_route)
        (support_site_verified ?support_site)
        (consumable_available ?consumable_resource)
        (not
          (support_site_consumable_reserved ?support_site ?consumable_resource)
        )
      )
  )
  (:action assemble_extraction_asset
    :parameters (?rescue_site - rescue_site ?support_site - support_site ?approach_route - approach_route ?extraction_route - extraction_route ?extraction_asset - extraction_asset)
    :precondition
      (and
        (rescue_site_prepped ?rescue_site)
        (support_site_prepped ?support_site)
        (approach_route_assigned_to_site ?rescue_site ?approach_route)
        (extraction_route_assigned_to_support_site ?support_site ?extraction_route)
        (approach_route_staged ?approach_route)
        (extraction_route_staged ?extraction_route)
        (rescue_site_verified ?rescue_site)
        (support_site_verified ?support_site)
        (extraction_asset_available ?extraction_asset)
      )
    :effect
      (and
        (extraction_asset_configured ?extraction_asset)
        (extraction_asset_bound_to_approach_route ?extraction_asset ?approach_route)
        (extraction_asset_bound_to_extraction_route ?extraction_asset ?extraction_route)
        (not
          (extraction_asset_available ?extraction_asset)
        )
      )
  )
  (:action assemble_asset_with_stage_one
    :parameters (?rescue_site - rescue_site ?support_site - support_site ?approach_route - approach_route ?extraction_route - extraction_route ?extraction_asset - extraction_asset)
    :precondition
      (and
        (rescue_site_prepped ?rescue_site)
        (support_site_prepped ?support_site)
        (approach_route_assigned_to_site ?rescue_site ?approach_route)
        (extraction_route_assigned_to_support_site ?support_site ?extraction_route)
        (approach_route_prepared ?approach_route)
        (extraction_route_staged ?extraction_route)
        (not
          (rescue_site_verified ?rescue_site)
        )
        (support_site_verified ?support_site)
        (extraction_asset_available ?extraction_asset)
      )
    :effect
      (and
        (extraction_asset_configured ?extraction_asset)
        (extraction_asset_bound_to_approach_route ?extraction_asset ?approach_route)
        (extraction_asset_bound_to_extraction_route ?extraction_asset ?extraction_route)
        (extraction_asset_attachment_stage_one ?extraction_asset)
        (not
          (extraction_asset_available ?extraction_asset)
        )
      )
  )
  (:action assemble_asset_with_stage_two
    :parameters (?rescue_site - rescue_site ?support_site - support_site ?approach_route - approach_route ?extraction_route - extraction_route ?extraction_asset - extraction_asset)
    :precondition
      (and
        (rescue_site_prepped ?rescue_site)
        (support_site_prepped ?support_site)
        (approach_route_assigned_to_site ?rescue_site ?approach_route)
        (extraction_route_assigned_to_support_site ?support_site ?extraction_route)
        (approach_route_staged ?approach_route)
        (extraction_route_prepared ?extraction_route)
        (rescue_site_verified ?rescue_site)
        (not
          (support_site_verified ?support_site)
        )
        (extraction_asset_available ?extraction_asset)
      )
    :effect
      (and
        (extraction_asset_configured ?extraction_asset)
        (extraction_asset_bound_to_approach_route ?extraction_asset ?approach_route)
        (extraction_asset_bound_to_extraction_route ?extraction_asset ?extraction_route)
        (extraction_asset_attachment_stage_two ?extraction_asset)
        (not
          (extraction_asset_available ?extraction_asset)
        )
      )
  )
  (:action assemble_asset_with_both_stages
    :parameters (?rescue_site - rescue_site ?support_site - support_site ?approach_route - approach_route ?extraction_route - extraction_route ?extraction_asset - extraction_asset)
    :precondition
      (and
        (rescue_site_prepped ?rescue_site)
        (support_site_prepped ?support_site)
        (approach_route_assigned_to_site ?rescue_site ?approach_route)
        (extraction_route_assigned_to_support_site ?support_site ?extraction_route)
        (approach_route_prepared ?approach_route)
        (extraction_route_prepared ?extraction_route)
        (not
          (rescue_site_verified ?rescue_site)
        )
        (not
          (support_site_verified ?support_site)
        )
        (extraction_asset_available ?extraction_asset)
      )
    :effect
      (and
        (extraction_asset_configured ?extraction_asset)
        (extraction_asset_bound_to_approach_route ?extraction_asset ?approach_route)
        (extraction_asset_bound_to_extraction_route ?extraction_asset ?extraction_route)
        (extraction_asset_attachment_stage_one ?extraction_asset)
        (extraction_asset_attachment_stage_two ?extraction_asset)
        (not
          (extraction_asset_available ?extraction_asset)
        )
      )
  )
  (:action mark_asset_ready
    :parameters (?extraction_asset - extraction_asset ?rescue_site - rescue_site ?specialist_unit - specialist_unit)
    :precondition
      (and
        (extraction_asset_configured ?extraction_asset)
        (rescue_site_prepped ?rescue_site)
        (specialist_assigned_to_element ?rescue_site ?specialist_unit)
        (not
          (extraction_asset_ready ?extraction_asset)
        )
      )
    :effect (extraction_asset_ready ?extraction_asset)
  )
  (:action install_component_on_instance
    :parameters (?mission_instance - mission_instance ?asset_component - asset_component ?extraction_asset - extraction_asset)
    :precondition
      (and
        (element_secured ?mission_instance)
        (instance_includes_extraction_asset ?mission_instance ?extraction_asset)
        (instance_includes_component ?mission_instance ?asset_component)
        (asset_component_available ?asset_component)
        (extraction_asset_configured ?extraction_asset)
        (extraction_asset_ready ?extraction_asset)
        (not
          (asset_component_installed ?asset_component)
        )
      )
    :effect
      (and
        (asset_component_installed ?asset_component)
        (component_attached_to_extraction_asset ?asset_component ?extraction_asset)
        (not
          (asset_component_available ?asset_component)
        )
      )
  )
  (:action complete_component_integration
    :parameters (?mission_instance - mission_instance ?asset_component - asset_component ?extraction_asset - extraction_asset ?specialist_unit - specialist_unit)
    :precondition
      (and
        (element_secured ?mission_instance)
        (instance_includes_component ?mission_instance ?asset_component)
        (asset_component_installed ?asset_component)
        (component_attached_to_extraction_asset ?asset_component ?extraction_asset)
        (specialist_assigned_to_element ?mission_instance ?specialist_unit)
        (not
          (extraction_asset_attachment_stage_one ?extraction_asset)
        )
        (not
          (instance_base_configured ?mission_instance)
        )
      )
    :effect (instance_base_configured ?mission_instance)
  )
  (:action attach_optional_module_to_instance
    :parameters (?mission_instance - mission_instance ?optional_module - optional_module)
    :precondition
      (and
        (element_secured ?mission_instance)
        (optional_module_available ?optional_module)
        (not
          (instance_optional_module_attached ?mission_instance)
        )
      )
    :effect
      (and
        (instance_optional_module_attached ?mission_instance)
        (instance_has_optional_module ?mission_instance ?optional_module)
        (not
          (optional_module_available ?optional_module)
        )
      )
  )
  (:action activate_optional_module_on_instance
    :parameters (?mission_instance - mission_instance ?asset_component - asset_component ?extraction_asset - extraction_asset ?specialist_unit - specialist_unit ?optional_module - optional_module)
    :precondition
      (and
        (element_secured ?mission_instance)
        (instance_includes_component ?mission_instance ?asset_component)
        (asset_component_installed ?asset_component)
        (component_attached_to_extraction_asset ?asset_component ?extraction_asset)
        (specialist_assigned_to_element ?mission_instance ?specialist_unit)
        (extraction_asset_attachment_stage_one ?extraction_asset)
        (instance_optional_module_attached ?mission_instance)
        (instance_has_optional_module ?mission_instance ?optional_module)
        (not
          (instance_base_configured ?mission_instance)
        )
      )
    :effect
      (and
        (instance_base_configured ?mission_instance)
        (instance_optional_module_activated ?mission_instance)
      )
  )
  (:action attach_equipment_and_finalize_module
    :parameters (?mission_instance - mission_instance ?equipment_unit - equipment_unit ?support_npc - support_npc ?asset_component - asset_component ?extraction_asset - extraction_asset)
    :precondition
      (and
        (instance_base_configured ?mission_instance)
        (instance_has_equipment_unit ?mission_instance ?equipment_unit)
        (support_npc_assigned_to_element ?mission_instance ?support_npc)
        (instance_includes_component ?mission_instance ?asset_component)
        (component_attached_to_extraction_asset ?asset_component ?extraction_asset)
        (not
          (extraction_asset_attachment_stage_two ?extraction_asset)
        )
        (not
          (instance_module_attached ?mission_instance)
        )
      )
    :effect (instance_module_attached ?mission_instance)
  )
