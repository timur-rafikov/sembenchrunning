    (biosecurity_staff_assigned_to_unit ?affected_field_unit - affected_field_unit ?biosecurity_trained_staff - biosecurity_trained_staff)
    (resource_kit_available ?portable_resource_kit - portable_resource_kit)
    (kit_assigned_to_supervisor ?field_supervisor - field_supervisor ?portable_resource_kit - portable_resource_kit)

    (supervisor_assigned_shift ?field_supervisor - field_supervisor ?task_shift - task_shift)
    (task_shift_reserved ?task_shift - task_shift)
    (task_shift_kit_allocated ?task_shift - task_shift)
    (supervisor_ready ?field_supervisor - field_supervisor)
    (section_assigned_operation_window ?section_lead - section_lead ?operation_window - operation_window)
    (operation_window_reserved ?operation_window - operation_window)
    (operation_window_kit_allocated ?operation_window - operation_window)
    (section_ready ?section_lead - section_lead)
    (salvage_slot_available ?salvage_operation_slot - salvage_operation_slot)
    (salvage_slot_allocated ?salvage_operation_slot - salvage_operation_slot)
    (salvage_slot_assigned_shift ?salvage_operation_slot - salvage_operation_slot ?task_shift - task_shift)
    (salvage_slot_assigned_window ?salvage_operation_slot - salvage_operation_slot ?operation_window - operation_window)
    (salvage_slot_shift_ready ?salvage_operation_slot - salvage_operation_slot)
    (salvage_slot_window_ready ?salvage_operation_slot - salvage_operation_slot)
    (salvage_slot_confirmed ?salvage_operation_slot - salvage_operation_slot)
    (manager_oversees_supervisor ?response_manager - response_manager ?field_supervisor - field_supervisor)
    (manager_oversees_section ?response_manager - response_manager ?section_lead - section_lead)
    (manager_assigned_salvage_slot ?response_manager - response_manager ?salvage_operation_slot - salvage_operation_slot)
    (produce_batch_available ?produce_batch - produce_batch)
    (manager_assigned_produce_batch ?response_manager - response_manager ?produce_batch - produce_batch)
    (produce_batch_processed ?produce_batch - produce_batch)
    (produce_batch_assigned_to_slot ?produce_batch - produce_batch ?salvage_operation_slot - salvage_operation_slot)
    (manager_processing_in_progress ?response_manager - response_manager)
    (manager_processed_batch ?response_manager - response_manager)
    (manager_ready_for_mobilization ?response_manager - response_manager)
    (manager_equipment_assigned ?response_manager - response_manager)
    (manager_equipment_confirmed ?response_manager - response_manager)
    (manager_authorized_for_contractor ?response_manager - response_manager)
    (manager_team_mobilized ?response_manager - response_manager)
    (external_liaison_available ?external_liaison - external_liaison)
    (manager_assigned_liaison ?response_manager - response_manager ?external_liaison - external_liaison)

    (manager_equipment_ready ?response_manager - response_manager)
    (manager_authorization_initiated ?response_manager - response_manager)
    (special_equipment_available ?special_equipment - special_equipment)
    (manager_assigned_special_equipment ?response_manager - response_manager ?special_equipment - special_equipment)
    (contractor_service_available ?contractor_service - contractor_service)
    (manager_assigned_contractor ?response_manager - response_manager ?contractor_service - contractor_service)
    (manager_certification_available ?manager_certification - manager_certification)
    (manager_has_certification ?response_manager - response_manager ?manager_certification - manager_certification)
    (regulatory_authorization_available ?regulatory_authorization - regulatory_authorization)
    (manager_has_regulatory_authorization ?response_manager - response_manager ?regulatory_authorization - regulatory_authorization)

  (:action report_affected_field_unit
    :parameters (?affected_field_unit - affected_field_unit)
    :precondition
      (and
        (not
          (affected_unit_declared ?affected_field_unit)
        )
        (not
          (unit_recovery_complete ?affected_field_unit)
        )
      )
    :effect (affected_unit_declared ?affected_field_unit)
  )
  (:action allocate_contingency_labor_to_unit
    :parameters (?affected_field_unit - affected_field_unit ?contingency_labour_pool - contingency_labor_pool)
    :precondition
      (and
        (affected_unit_declared ?affected_field_unit)
        (not
          (contingency_assigned ?affected_field_unit)
        )
        (contingency_labor_pool_available ?contingency_labour_pool)
      )
    :effect
      (and
        (contingency_assigned ?affected_field_unit)
        (unit_assigned_contingency_labor_pool ?affected_field_unit ?contingency_labour_pool)
        (not
          (contingency_labor_pool_available ?contingency_labour_pool)
        )
      )
  )
  (:action assign_crew_to_field_unit
    :parameters (?affected_field_unit - affected_field_unit ?crew_member - crew_member)
    :precondition
      (and
        (affected_unit_declared ?affected_field_unit)
        (contingency_assigned ?affected_field_unit)
        (crew_member_available ?crew_member)
      )
    :effect
      (and
        (crew_assigned_to_unit ?affected_field_unit ?crew_member)
        (not
          (crew_member_available ?crew_member)
        )
      )
  )
  (:action complete_initial_assessment_for_field_unit
    :parameters (?affected_field_unit - affected_field_unit ?crew_member - crew_member)
    :precondition
      (and
        (affected_unit_declared ?affected_field_unit)
        (contingency_assigned ?affected_field_unit)
        (crew_assigned_to_unit ?affected_field_unit ?crew_member)
        (not
          (assessment_complete ?affected_field_unit)
        )
      )
    :effect (assessment_complete ?affected_field_unit)
  )
  (:action release_crew_from_field_unit
    :parameters (?affected_field_unit - affected_field_unit ?crew_member - crew_member)
    :precondition
      (and
        (crew_assigned_to_unit ?affected_field_unit ?crew_member)
      )
    :effect
      (and
        (crew_member_available ?crew_member)
        (not
          (crew_assigned_to_unit ?affected_field_unit ?crew_member)
        )
      )
  )
  (:action assign_biosecurity_staff_to_field_unit
    :parameters (?affected_field_unit - affected_field_unit ?biosecurity_trained_staff - biosecurity_trained_staff)
    :precondition
      (and
        (assessment_complete ?affected_field_unit)
        (biosecurity_staff_available ?biosecurity_trained_staff)
      )
    :effect
      (and
        (biosecurity_staff_assigned_to_unit ?affected_field_unit ?biosecurity_trained_staff)
        (not
          (biosecurity_staff_available ?biosecurity_trained_staff)
        )
      )
  )
  (:action release_biosecurity_staff_from_field_unit
    :parameters (?affected_field_unit - affected_field_unit ?biosecurity_trained_staff - biosecurity_trained_staff)
    :precondition
      (and
        (biosecurity_staff_assigned_to_unit ?affected_field_unit ?biosecurity_trained_staff)
      )
    :effect
      (and
        (biosecurity_staff_available ?biosecurity_trained_staff)
        (not
          (biosecurity_staff_assigned_to_unit ?affected_field_unit ?biosecurity_trained_staff)
        )
      )
  )
  (:action assign_manager_certification
    :parameters (?response_manager - response_manager ?manager_certification - manager_certification)
    :precondition
      (and
        (assessment_complete ?response_manager)
        (manager_certification_available ?manager_certification)
      )
    :effect
      (and
        (manager_has_certification ?response_manager ?manager_certification)
        (not
          (manager_certification_available ?manager_certification)
        )
      )
  )
  (:action remove_manager_certification
    :parameters (?response_manager - response_manager ?manager_certification - manager_certification)
    :precondition
      (and
        (manager_has_certification ?response_manager ?manager_certification)
      )
    :effect
      (and
        (manager_certification_available ?manager_certification)
        (not
          (manager_has_certification ?response_manager ?manager_certification)
        )
      )
  )
  (:action assign_regulatory_authorization
    :parameters (?response_manager - response_manager ?regulatory_authorization - regulatory_authorization)
    :precondition
      (and
        (assessment_complete ?response_manager)
        (regulatory_authorization_available ?regulatory_authorization)
      )
    :effect
      (and
        (manager_has_regulatory_authorization ?response_manager ?regulatory_authorization)
        (not
          (regulatory_authorization_available ?regulatory_authorization)
        )
      )
  )
  (:action remove_regulatory_authorization
    :parameters (?response_manager - response_manager ?regulatory_authorization - regulatory_authorization)
    :precondition
      (and
        (manager_has_regulatory_authorization ?response_manager ?regulatory_authorization)
      )
    :effect
      (and
        (regulatory_authorization_available ?regulatory_authorization)
        (not
          (manager_has_regulatory_authorization ?response_manager ?regulatory_authorization)
        )
      )
  )
  (:action reserve_task_shift_for_supervisor
    :parameters (?field_supervisor - field_supervisor ?task_shift - task_shift ?crew_member - crew_member)
    :precondition
      (and
        (assessment_complete ?field_supervisor)
        (crew_assigned_to_unit ?field_supervisor ?crew_member)
        (supervisor_assigned_shift ?field_supervisor ?task_shift)
        (not
          (task_shift_reserved ?task_shift)
        )
        (not
          (task_shift_kit_allocated ?task_shift)
        )
      )
    :effect (task_shift_reserved ?task_shift)
  )
  (:action stage_supervisor_with_biosecurity_staff
    :parameters (?field_supervisor - field_supervisor ?task_shift - task_shift ?biosecurity_trained_staff - biosecurity_trained_staff)
    :precondition
      (and
        (assessment_complete ?field_supervisor)
        (biosecurity_staff_assigned_to_unit ?field_supervisor ?biosecurity_trained_staff)
        (supervisor_assigned_shift ?field_supervisor ?task_shift)
        (task_shift_reserved ?task_shift)
        (not
          (supervisor_scheduled ?field_supervisor)
        )
      )
    :effect
      (and
        (supervisor_scheduled ?field_supervisor)
        (supervisor_ready ?field_supervisor)
      )
  )
  (:action assign_resource_kit_to_supervisor
    :parameters (?field_supervisor - field_supervisor ?task_shift - task_shift ?portable_resource_kit - portable_resource_kit)
    :precondition
      (and
        (assessment_complete ?field_supervisor)
        (supervisor_assigned_shift ?field_supervisor ?task_shift)
        (resource_kit_available ?portable_resource_kit)
        (not
          (supervisor_scheduled ?field_supervisor)
        )
      )
    :effect
      (and
        (task_shift_kit_allocated ?task_shift)
        (supervisor_scheduled ?field_supervisor)
        (kit_assigned_to_supervisor ?field_supervisor ?portable_resource_kit)
        (not
          (resource_kit_available ?portable_resource_kit)
        )
      )
  )
  (:action finalize_supervisor_shift_allocation
    :parameters (?field_supervisor - field_supervisor ?task_shift - task_shift ?crew_member - crew_member ?portable_resource_kit - portable_resource_kit)
    :precondition
      (and
        (assessment_complete ?field_supervisor)
        (crew_assigned_to_unit ?field_supervisor ?crew_member)
        (supervisor_assigned_shift ?field_supervisor ?task_shift)
        (task_shift_kit_allocated ?task_shift)
        (kit_assigned_to_supervisor ?field_supervisor ?portable_resource_kit)
        (not
          (supervisor_ready ?field_supervisor)
        )
      )
    :effect
      (and
        (task_shift_reserved ?task_shift)
        (supervisor_ready ?field_supervisor)
        (resource_kit_available ?portable_resource_kit)
        (not
          (kit_assigned_to_supervisor ?field_supervisor ?portable_resource_kit)
        )
      )
  )
  (:action reserve_operation_window_for_section
    :parameters (?section_lead - section_lead ?operation_window - operation_window ?crew_member - crew_member)
    :precondition
      (and
        (assessment_complete ?section_lead)
        (crew_assigned_to_unit ?section_lead ?crew_member)
        (section_assigned_operation_window ?section_lead ?operation_window)
        (not
          (operation_window_reserved ?operation_window)
        )
        (not
          (operation_window_kit_allocated ?operation_window)
        )
      )
    :effect (operation_window_reserved ?operation_window)
  )
  (:action stage_section_with_biosecurity_staff
    :parameters (?section_lead - section_lead ?operation_window - operation_window ?biosecurity_trained_staff - biosecurity_trained_staff)
    :precondition
      (and
        (assessment_complete ?section_lead)
        (biosecurity_staff_assigned_to_unit ?section_lead ?biosecurity_trained_staff)
        (section_assigned_operation_window ?section_lead ?operation_window)
        (operation_window_reserved ?operation_window)
        (not
          (section_scheduled ?section_lead)
        )
      )
    :effect
      (and
        (section_scheduled ?section_lead)
        (section_ready ?section_lead)
      )
  )
  (:action assign_resource_kit_to_section
    :parameters (?section_lead - section_lead ?operation_window - operation_window ?portable_resource_kit - portable_resource_kit)
    :precondition
      (and
        (assessment_complete ?section_lead)
        (section_assigned_operation_window ?section_lead ?operation_window)
        (resource_kit_available ?portable_resource_kit)
        (not
          (section_scheduled ?section_lead)
        )
      )
    :effect
      (and
        (operation_window_kit_allocated ?operation_window)
        (section_scheduled ?section_lead)
        (kit_assigned_to_section ?section_lead ?portable_resource_kit)
        (not
          (resource_kit_available ?portable_resource_kit)
        )
      )
  )
  (:action finalize_section_operation_allocation
    :parameters (?section_lead - section_lead ?operation_window - operation_window ?crew_member - crew_member ?portable_resource_kit - portable_resource_kit)
    :precondition
      (and
        (assessment_complete ?section_lead)
        (crew_assigned_to_unit ?section_lead ?crew_member)
        (section_assigned_operation_window ?section_lead ?operation_window)
        (operation_window_kit_allocated ?operation_window)
        (kit_assigned_to_section ?section_lead ?portable_resource_kit)
        (not
          (section_ready ?section_lead)
        )
      )
    :effect
      (and
        (operation_window_reserved ?operation_window)
        (section_ready ?section_lead)
        (resource_kit_available ?portable_resource_kit)
        (not
          (kit_assigned_to_section ?section_lead ?portable_resource_kit)
        )
      )
  )
  (:action assemble_salvage_operation_slot
    :parameters (?field_supervisor - field_supervisor ?section_lead - section_lead ?task_shift - task_shift ?operation_window - operation_window ?salvage_operation_slot - salvage_operation_slot)
    :precondition
      (and
        (supervisor_scheduled ?field_supervisor)
        (section_scheduled ?section_lead)
        (supervisor_assigned_shift ?field_supervisor ?task_shift)
        (section_assigned_operation_window ?section_lead ?operation_window)
        (task_shift_reserved ?task_shift)
        (operation_window_reserved ?operation_window)
        (supervisor_ready ?field_supervisor)
        (section_ready ?section_lead)
        (salvage_slot_available ?salvage_operation_slot)
      )
    :effect
      (and
        (salvage_slot_allocated ?salvage_operation_slot)
        (salvage_slot_assigned_shift ?salvage_operation_slot ?task_shift)
        (salvage_slot_assigned_window ?salvage_operation_slot ?operation_window)
        (not
          (salvage_slot_available ?salvage_operation_slot)
        )
      )
  )
  (:action assemble_salvage_operation_with_shift_ready
    :parameters (?field_supervisor - field_supervisor ?section_lead - section_lead ?task_shift - task_shift ?operation_window - operation_window ?salvage_operation_slot - salvage_operation_slot)
    :precondition
      (and
        (supervisor_scheduled ?field_supervisor)
        (section_scheduled ?section_lead)
        (supervisor_assigned_shift ?field_supervisor ?task_shift)
        (section_assigned_operation_window ?section_lead ?operation_window)
        (task_shift_kit_allocated ?task_shift)
        (operation_window_reserved ?operation_window)
        (not
          (supervisor_ready ?field_supervisor)
        )
        (section_ready ?section_lead)
        (salvage_slot_available ?salvage_operation_slot)
      )
    :effect
      (and
        (salvage_slot_allocated ?salvage_operation_slot)
        (salvage_slot_assigned_shift ?salvage_operation_slot ?task_shift)
        (salvage_slot_assigned_window ?salvage_operation_slot ?operation_window)
        (salvage_slot_shift_ready ?salvage_operation_slot)
        (not
          (salvage_slot_available ?salvage_operation_slot)
        )
      )
  )
  (:action assemble_salvage_operation_with_window_ready
    :parameters (?field_supervisor - field_supervisor ?section_lead - section_lead ?task_shift - task_shift ?operation_window - operation_window ?salvage_operation_slot - salvage_operation_slot)
    :precondition
      (and
        (supervisor_scheduled ?field_supervisor)
        (section_scheduled ?section_lead)
        (supervisor_assigned_shift ?field_supervisor ?task_shift)
        (section_assigned_operation_window ?section_lead ?operation_window)
        (task_shift_reserved ?task_shift)
        (operation_window_kit_allocated ?operation_window)
        (supervisor_ready ?field_supervisor)
        (not
          (section_ready ?section_lead)
        )
        (salvage_slot_available ?salvage_operation_slot)
      )
    :effect
      (and
        (salvage_slot_allocated ?salvage_operation_slot)
        (salvage_slot_assigned_shift ?salvage_operation_slot ?task_shift)
        (salvage_slot_assigned_window ?salvage_operation_slot ?operation_window)
        (salvage_slot_window_ready ?salvage_operation_slot)
        (not
          (salvage_slot_available ?salvage_operation_slot)
        )
      )
  )
  (:action assemble_full_salvage_operation
    :parameters (?field_supervisor - field_supervisor ?section_lead - section_lead ?task_shift - task_shift ?operation_window - operation_window ?salvage_operation_slot - salvage_operation_slot)
    :precondition
      (and
        (supervisor_scheduled ?field_supervisor)
        (section_scheduled ?section_lead)
        (supervisor_assigned_shift ?field_supervisor ?task_shift)
        (section_assigned_operation_window ?section_lead ?operation_window)
        (task_shift_kit_allocated ?task_shift)
        (operation_window_kit_allocated ?operation_window)
        (not
          (supervisor_ready ?field_supervisor)
        )
        (not
          (section_ready ?section_lead)
        )
        (salvage_slot_available ?salvage_operation_slot)
      )
    :effect
      (and
        (salvage_slot_allocated ?salvage_operation_slot)
        (salvage_slot_assigned_shift ?salvage_operation_slot ?task_shift)
        (salvage_slot_assigned_window ?salvage_operation_slot ?operation_window)
        (salvage_slot_shift_ready ?salvage_operation_slot)
        (salvage_slot_window_ready ?salvage_operation_slot)
        (not
          (salvage_slot_available ?salvage_operation_slot)
        )
      )
  )
  (:action confirm_salvage_slot
    :parameters (?salvage_operation_slot - salvage_operation_slot ?field_supervisor - field_supervisor ?crew_member - crew_member)
    :precondition
      (and
        (salvage_slot_allocated ?salvage_operation_slot)
        (supervisor_scheduled ?field_supervisor)
        (crew_assigned_to_unit ?field_supervisor ?crew_member)
        (not
          (salvage_slot_confirmed ?salvage_operation_slot)
        )
      )
    :effect (salvage_slot_confirmed ?salvage_operation_slot)
  )
  (:action triage_produce_batch
    :parameters (?response_manager - response_manager ?produce_batch - produce_batch ?salvage_operation_slot - salvage_operation_slot)
    :precondition
      (and
        (assessment_complete ?response_manager)
        (manager_assigned_salvage_slot ?response_manager ?salvage_operation_slot)
        (manager_assigned_produce_batch ?response_manager ?produce_batch)
        (produce_batch_available ?produce_batch)
        (salvage_slot_allocated ?salvage_operation_slot)
        (salvage_slot_confirmed ?salvage_operation_slot)
        (not
          (produce_batch_processed ?produce_batch)
        )
      )
    :effect
      (and
        (produce_batch_processed ?produce_batch)
        (produce_batch_assigned_to_slot ?produce_batch ?salvage_operation_slot)
        (not
          (produce_batch_available ?produce_batch)
        )
      )
  )
  (:action start_manager_batch_processing
    :parameters (?response_manager - response_manager ?produce_batch - produce_batch ?salvage_operation_slot - salvage_operation_slot ?crew_member - crew_member)
    :precondition
      (and
        (assessment_complete ?response_manager)
        (manager_assigned_produce_batch ?response_manager ?produce_batch)
        (produce_batch_processed ?produce_batch)
        (produce_batch_assigned_to_slot ?produce_batch ?salvage_operation_slot)
        (crew_assigned_to_unit ?response_manager ?crew_member)
        (not
          (salvage_slot_shift_ready ?salvage_operation_slot)
        )
        (not
          (manager_processing_in_progress ?response_manager)
        )
      )
    :effect (manager_processing_in_progress ?response_manager)
  )
  (:action assign_special_equipment_to_manager
    :parameters (?response_manager - response_manager ?special_equipment - special_equipment)
    :precondition
      (and
        (assessment_complete ?response_manager)
        (special_equipment_available ?special_equipment)
        (not
          (manager_equipment_assigned ?response_manager)
        )
      )
    :effect
      (and
        (manager_equipment_assigned ?response_manager)
        (manager_assigned_special_equipment ?response_manager ?special_equipment)
        (not
          (special_equipment_available ?special_equipment)
        )
      )
  )
  (:action prepare_manager_equipment_and_start_processing
    :parameters (?response_manager - response_manager ?produce_batch - produce_batch ?salvage_operation_slot - salvage_operation_slot ?crew_member - crew_member ?special_equipment - special_equipment)
    :precondition
      (and
        (assessment_complete ?response_manager)
