--- domain.pddl (as passed to VAL) ---
(define (domain default_workout_option_selection_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types workout_resource - object workout_category - object workout_artifact - object loan_case_category - object distressed_loan_case - loan_case_category workout_option_template - workout_resource assessment_resource - workout_resource negotiation_resource - workout_resource external_approval_document - workout_resource restructuring_attribute - workout_resource evidence_item - workout_resource policy_constraint - workout_resource legal_constraint - workout_resource remediation_template - workout_category collateral_asset - workout_category approval_role - workout_category primary_strategy - workout_artifact co_obligor_strategy - workout_artifact workout_plan - workout_artifact case_role_group - distressed_loan_case case_role_group_2 - distressed_loan_case primary_obligor_case - case_role_group co_obligor_case - case_role_group workout_decision - case_role_group_2)
  (:predicates
    (decision_opened ?distressed_loan_case - distressed_loan_case)
    (decision_screened ?distressed_loan_case - distressed_loan_case)
    (options_presented ?distressed_loan_case - distressed_loan_case)
    (ready_for_execution ?distressed_loan_case - distressed_loan_case)
    (decision_finalization_ready_flag ?distressed_loan_case - distressed_loan_case)
    (evidence_attached ?distressed_loan_case - distressed_loan_case)
    (option_template_available ?workout_option_template - workout_option_template)
    (decision_assigned_option_template ?distressed_loan_case - distressed_loan_case ?workout_option_template - workout_option_template)
    (assessment_available ?assessment_resource - assessment_resource)
    (decision_assigned_assessment ?distressed_loan_case - distressed_loan_case ?assessment_resource - assessment_resource)
    (negotiation_resource_available ?negotiation_resource - negotiation_resource)
    (decision_assigned_negotiation_resource ?distressed_loan_case - distressed_loan_case ?negotiation_resource - negotiation_resource)
    (remediation_template_available ?remediation_template - remediation_template)
    (primary_case_assigned_remediation ?primary_obligor_case - primary_obligor_case ?remediation_template - remediation_template)
    (co_obligor_case_assigned_remediation ?co_obligor_case - co_obligor_case ?remediation_template - remediation_template)
    (primary_case_assigned_strategy ?primary_obligor_case - primary_obligor_case ?primary_strategy - primary_strategy)
    (primary_strategy_selected ?primary_strategy - primary_strategy)
    (primary_strategy_remediation_attached ?primary_strategy - primary_strategy)
    (primary_strategy_confirmed ?primary_obligor_case - primary_obligor_case)
    (co_obligor_assigned_strategy ?co_obligor_case - co_obligor_case ?co_obligor_strategy - co_obligor_strategy)
    (co_strategy_selected ?co_obligor_strategy - co_obligor_strategy)
    (co_strategy_remediation_attached ?co_obligor_strategy - co_obligor_strategy)
    (co_strategy_confirmed ?co_obligor_case - co_obligor_case)
    (workout_plan_available ?workout_plan - workout_plan)
    (workout_plan_created ?workout_plan - workout_plan)
    (workout_plan_includes_primary_strategy ?workout_plan - workout_plan ?primary_strategy - primary_strategy)
    (workout_plan_includes_co_strategy ?workout_plan - workout_plan ?co_obligor_strategy - co_obligor_strategy)
    (workout_plan_flag_primary_remediation ?workout_plan - workout_plan)
    (workout_plan_flag_co_remediation ?workout_plan - workout_plan)
    (workout_plan_scheduled ?workout_plan - workout_plan)
    (decision_includes_primary_case ?workout_decision - workout_decision ?primary_obligor_case - primary_obligor_case)
    (decision_includes_co_obligor_case ?workout_decision - workout_decision ?co_obligor_case - co_obligor_case)
    (decision_includes_workout_plan ?workout_decision - workout_decision ?workout_plan - workout_plan)
    (collateral_asset_available ?collateral_asset - collateral_asset)
    (decision_includes_collateral_asset ?workout_decision - workout_decision ?collateral_asset - collateral_asset)
    (collateral_asset_committed ?collateral_asset - collateral_asset)
    (collateral_assigned_to_workout_plan ?collateral_asset - collateral_asset ?workout_plan - workout_plan)
    (decision_has_committed_collateral ?workout_decision - workout_decision)
    (approval_sequence_started ?workout_decision - workout_decision)
    (approval_requirement_met ?workout_decision - workout_decision)
    (decision_external_document_attached ?workout_decision - workout_decision)
    (decision_external_document_verified ?workout_decision - workout_decision)
    (decision_has_additional_approvals ?workout_decision - workout_decision)
    (decision_additional_approvals_completed ?workout_decision - workout_decision)
    (approval_role_available ?approval_role - approval_role)
    (decision_assigned_approval_role ?workout_decision - workout_decision ?approval_role - approval_role)
    (decision_approval_role_engaged ?workout_decision - workout_decision)
    (decision_approval_role_actioned ?workout_decision - workout_decision)
    (approval_role_completed ?workout_decision - workout_decision)
    (external_document_available ?external_approval_document - external_approval_document)
    (decision_attached_external_document ?workout_decision - workout_decision ?external_approval_document - external_approval_document)
    (restructuring_attribute_available ?restructuring_attribute - restructuring_attribute)
    (decision_assigned_restructuring_attribute ?workout_decision - workout_decision ?restructuring_attribute - restructuring_attribute)
    (policy_constraint_available ?policy_constraint_var - policy_constraint)
    (decision_assigned_policy_constraint ?workout_decision - workout_decision ?policy_constraint_var - policy_constraint)
    (legal_constraint_available ?legal_constraint_var - legal_constraint)
    (decision_assigned_legal_constraint ?workout_decision - workout_decision ?legal_constraint_var - legal_constraint)
    (evidence_available ?evidence_item - evidence_item)
    (decision_attached_evidence ?distressed_loan_case - distressed_loan_case ?evidence_item - evidence_item)
    (primary_case_ready_for_plan ?primary_obligor_case - primary_obligor_case)
    (co_obligor_case_ready_for_plan ?co_obligor_case - co_obligor_case)
    (decision_finalized_flag ?workout_decision - workout_decision)
  )
  (:action open_distressed_loan_case
    :parameters (?distressed_loan_case - distressed_loan_case)
    :precondition
      (and
        (not
          (decision_opened ?distressed_loan_case)
        )
        (not
          (ready_for_execution ?distressed_loan_case)
        )
      )
    :effect (decision_opened ?distressed_loan_case)
  )
  (:action assign_workout_option_template_to_case
    :parameters (?distressed_loan_case - distressed_loan_case ?workout_option_template - workout_option_template)
    :precondition
      (and
        (decision_opened ?distressed_loan_case)
        (not
          (options_presented ?distressed_loan_case)
        )
        (option_template_available ?workout_option_template)
      )
    :effect
      (and
        (options_presented ?distressed_loan_case)
        (decision_assigned_option_template ?distressed_loan_case ?workout_option_template)
        (not
          (option_template_available ?workout_option_template)
        )
      )
  )
  (:action allocate_assessment_to_case
    :parameters (?distressed_loan_case - distressed_loan_case ?assessment_resource - assessment_resource)
    :precondition
      (and
        (decision_opened ?distressed_loan_case)
        (options_presented ?distressed_loan_case)
        (assessment_available ?assessment_resource)
      )
    :effect
      (and
        (decision_assigned_assessment ?distressed_loan_case ?assessment_resource)
        (not
          (assessment_available ?assessment_resource)
        )
      )
  )
  (:action complete_case_screening
    :parameters (?distressed_loan_case - distressed_loan_case ?assessment_resource - assessment_resource)
    :precondition
      (and
        (decision_opened ?distressed_loan_case)
        (options_presented ?distressed_loan_case)
        (decision_assigned_assessment ?distressed_loan_case ?assessment_resource)
        (not
          (decision_screened ?distressed_loan_case)
        )
      )
    :effect (decision_screened ?distressed_loan_case)
  )
  (:action release_assessment_resource
    :parameters (?distressed_loan_case - distressed_loan_case ?assessment_resource - assessment_resource)
    :precondition
      (and
        (decision_assigned_assessment ?distressed_loan_case ?assessment_resource)
      )
    :effect
      (and
        (assessment_available ?assessment_resource)
        (not
          (decision_assigned_assessment ?distressed_loan_case ?assessment_resource)
        )
      )
  )
  (:action assign_negotiation_resource_to_case
    :parameters (?distressed_loan_case - distressed_loan_case ?negotiation_resource - negotiation_resource)
    :precondition
      (and
        (decision_screened ?distressed_loan_case)
        (negotiation_resource_available ?negotiation_resource)
      )
    :effect
      (and
        (decision_assigned_negotiation_resource ?distressed_loan_case ?negotiation_resource)
        (not
          (negotiation_resource_available ?negotiation_resource)
        )
      )
  )
  (:action release_negotiation_resource_from_case
    :parameters (?distressed_loan_case - distressed_loan_case ?negotiation_resource - negotiation_resource)
    :precondition
      (and
        (decision_assigned_negotiation_resource ?distressed_loan_case ?negotiation_resource)
      )
    :effect
      (and
        (negotiation_resource_available ?negotiation_resource)
        (not
          (decision_assigned_negotiation_resource ?distressed_loan_case ?negotiation_resource)
        )
      )
  )
  (:action attach_policy_constraint_to_decision
    :parameters (?workout_decision - workout_decision ?policy_constraint_var - policy_constraint)
    :precondition
      (and
        (decision_screened ?workout_decision)
        (policy_constraint_available ?policy_constraint_var)
      )
    :effect
      (and
        (decision_assigned_policy_constraint ?workout_decision ?policy_constraint_var)
        (not
          (policy_constraint_available ?policy_constraint_var)
        )
      )
  )
  (:action detach_policy_constraint_from_decision
    :parameters (?workout_decision - workout_decision ?policy_constraint_var - policy_constraint)
    :precondition
      (and
        (decision_assigned_policy_constraint ?workout_decision ?policy_constraint_var)
      )
    :effect
      (and
        (policy_constraint_available ?policy_constraint_var)
        (not
          (decision_assigned_policy_constraint ?workout_decision ?policy_constraint_var)
        )
      )
  )
  (:action attach_legal_constraint_to_decision
    :parameters (?workout_decision - workout_decision ?legal_constraint_var - legal_constraint)
    :precondition
      (and
        (decision_screened ?workout_decision)
        (legal_constraint_available ?legal_constraint_var)
      )
    :effect
      (and
        (decision_assigned_legal_constraint ?workout_decision ?legal_constraint_var)
        (not
          (legal_constraint_available ?legal_constraint_var)
        )
      )
  )
  (:action detach_legal_constraint_from_decision
    :parameters (?workout_decision - workout_decision ?legal_constraint_var - legal_constraint)
    :precondition
      (and
        (decision_assigned_legal_constraint ?workout_decision ?legal_constraint_var)
      )
    :effect
      (and
        (legal_constraint_available ?legal_constraint_var)
        (not
          (decision_assigned_legal_constraint ?workout_decision ?legal_constraint_var)
        )
      )
  )
  (:action select_primary_strategy_for_case
    :parameters (?primary_obligor_case - primary_obligor_case ?primary_strategy - primary_strategy ?assessment_resource - assessment_resource)
    :precondition
      (and
        (decision_screened ?primary_obligor_case)
        (decision_assigned_assessment ?primary_obligor_case ?assessment_resource)
        (primary_case_assigned_strategy ?primary_obligor_case ?primary_strategy)
        (not
          (primary_strategy_selected ?primary_strategy)
        )
        (not
          (primary_strategy_remediation_attached ?primary_strategy)
        )
      )
    :effect (primary_strategy_selected ?primary_strategy)
  )
  (:action confirm_primary_strategy_after_negotiation
    :parameters (?primary_obligor_case - primary_obligor_case ?primary_strategy - primary_strategy ?negotiation_resource - negotiation_resource)
    :precondition
      (and
        (decision_screened ?primary_obligor_case)
        (decision_assigned_negotiation_resource ?primary_obligor_case ?negotiation_resource)
        (primary_case_assigned_strategy ?primary_obligor_case ?primary_strategy)
        (primary_strategy_selected ?primary_strategy)
        (not
          (primary_case_ready_for_plan ?primary_obligor_case)
        )
      )
    :effect
      (and
        (primary_case_ready_for_plan ?primary_obligor_case)
        (primary_strategy_confirmed ?primary_obligor_case)
      )
  )
  (:action attach_remediation_to_primary_strategy
    :parameters (?primary_obligor_case - primary_obligor_case ?primary_strategy - primary_strategy ?remediation_template - remediation_template)
    :precondition
      (and
        (decision_screened ?primary_obligor_case)
        (primary_case_assigned_strategy ?primary_obligor_case ?primary_strategy)
        (remediation_template_available ?remediation_template)
        (not
          (primary_case_ready_for_plan ?primary_obligor_case)
        )
      )
    :effect
      (and
        (primary_strategy_remediation_attached ?primary_strategy)
        (primary_case_ready_for_plan ?primary_obligor_case)
        (primary_case_assigned_remediation ?primary_obligor_case ?remediation_template)
        (not
          (remediation_template_available ?remediation_template)
        )
      )
  )
  (:action finalize_primary_strategy_with_remediation
    :parameters (?primary_obligor_case - primary_obligor_case ?primary_strategy - primary_strategy ?assessment_resource - assessment_resource ?remediation_template - remediation_template)
    :precondition
      (and
        (decision_screened ?primary_obligor_case)
        (decision_assigned_assessment ?primary_obligor_case ?assessment_resource)
        (primary_case_assigned_strategy ?primary_obligor_case ?primary_strategy)
        (primary_strategy_remediation_attached ?primary_strategy)
        (primary_case_assigned_remediation ?primary_obligor_case ?remediation_template)
        (not
          (primary_strategy_confirmed ?primary_obligor_case)
        )
      )
    :effect
      (and
        (primary_strategy_selected ?primary_strategy)
        (primary_strategy_confirmed ?primary_obligor_case)
        (remediation_template_available ?remediation_template)
        (not
          (primary_case_assigned_remediation ?primary_obligor_case ?remediation_template)
        )
      )
  )
  (:action select_co_obligor_strategy_for_case
    :parameters (?co_obligor_case - co_obligor_case ?co_obligor_strategy - co_obligor_strategy ?assessment_resource - assessment_resource)
    :precondition
      (and
        (decision_screened ?co_obligor_case)
        (decision_assigned_assessment ?co_obligor_case ?assessment_resource)
        (co_obligor_assigned_strategy ?co_obligor_case ?co_obligor_strategy)
        (not
          (co_strategy_selected ?co_obligor_strategy)
        )
        (not
          (co_strategy_remediation_attached ?co_obligor_strategy)
        )
      )
    :effect (co_strategy_selected ?co_obligor_strategy)
  )
  (:action confirm_co_obligor_strategy_after_negotiation
    :parameters (?co_obligor_case - co_obligor_case ?co_obligor_strategy - co_obligor_strategy ?negotiation_resource - negotiation_resource)
    :precondition
      (and
        (decision_screened ?co_obligor_case)
        (decision_assigned_negotiation_resource ?co_obligor_case ?negotiation_resource)
        (co_obligor_assigned_strategy ?co_obligor_case ?co_obligor_strategy)
        (co_strategy_selected ?co_obligor_strategy)
        (not
          (co_obligor_case_ready_for_plan ?co_obligor_case)
        )
      )
    :effect
      (and
        (co_obligor_case_ready_for_plan ?co_obligor_case)
        (co_strategy_confirmed ?co_obligor_case)
      )
  )
  (:action attach_remediation_to_co_obligor_strategy
    :parameters (?co_obligor_case - co_obligor_case ?co_obligor_strategy - co_obligor_strategy ?remediation_template - remediation_template)
    :precondition
      (and
        (decision_screened ?co_obligor_case)
        (co_obligor_assigned_strategy ?co_obligor_case ?co_obligor_strategy)
        (remediation_template_available ?remediation_template)
        (not
          (co_obligor_case_ready_for_plan ?co_obligor_case)
        )
      )
    :effect
      (and
        (co_strategy_remediation_attached ?co_obligor_strategy)
        (co_obligor_case_ready_for_plan ?co_obligor_case)
        (co_obligor_case_assigned_remediation ?co_obligor_case ?remediation_template)
        (not
          (remediation_template_available ?remediation_template)
        )
      )
  )
  (:action finalize_co_obligor_strategy_with_remediation
    :parameters (?co_obligor_case - co_obligor_case ?co_obligor_strategy - co_obligor_strategy ?assessment_resource - assessment_resource ?remediation_template - remediation_template)
    :precondition
      (and
        (decision_screened ?co_obligor_case)
        (decision_assigned_assessment ?co_obligor_case ?assessment_resource)
        (co_obligor_assigned_strategy ?co_obligor_case ?co_obligor_strategy)
        (co_strategy_remediation_attached ?co_obligor_strategy)
        (co_obligor_case_assigned_remediation ?co_obligor_case ?remediation_template)
        (not
          (co_strategy_confirmed ?co_obligor_case)
        )
      )
    :effect
      (and
        (co_strategy_selected ?co_obligor_strategy)
        (co_strategy_confirmed ?co_obligor_case)
        (remediation_template_available ?remediation_template)
        (not
          (co_obligor_case_assigned_remediation ?co_obligor_case ?remediation_template)
        )
      )
  )
  (:action compose_workout_plan_from_strategies
    :parameters (?primary_obligor_case - primary_obligor_case ?co_obligor_case - co_obligor_case ?primary_strategy - primary_strategy ?co_obligor_strategy - co_obligor_strategy ?workout_plan - workout_plan)
    :precondition
      (and
        (primary_case_ready_for_plan ?primary_obligor_case)
        (co_obligor_case_ready_for_plan ?co_obligor_case)
        (primary_case_assigned_strategy ?primary_obligor_case ?primary_strategy)
        (co_obligor_assigned_strategy ?co_obligor_case ?co_obligor_strategy)
        (primary_strategy_selected ?primary_strategy)
        (co_strategy_selected ?co_obligor_strategy)
        (primary_strategy_confirmed ?primary_obligor_case)
        (co_strategy_confirmed ?co_obligor_case)
        (workout_plan_available ?workout_plan)
      )
    :effect
      (and
        (workout_plan_created ?workout_plan)
        (workout_plan_includes_primary_strategy ?workout_plan ?primary_strategy)
        (workout_plan_includes_co_strategy ?workout_plan ?co_obligor_strategy)
        (not
          (workout_plan_available ?workout_plan)
        )
      )
  )
  (:action compose_workout_plan_with_primary_remediation
    :parameters (?primary_obligor_case - primary_obligor_case ?co_obligor_case - co_obligor_case ?primary_strategy - primary_strategy ?co_obligor_strategy - co_obligor_strategy ?workout_plan - workout_plan)
    :precondition
      (and
        (primary_case_ready_for_plan ?primary_obligor_case)
        (co_obligor_case_ready_for_plan ?co_obligor_case)
        (primary_case_assigned_strategy ?primary_obligor_case ?primary_strategy)
        (co_obligor_assigned_strategy ?co_obligor_case ?co_obligor_strategy)
        (primary_strategy_remediation_attached ?primary_strategy)
        (co_strategy_selected ?co_obligor_strategy)
        (not
          (primary_strategy_confirmed ?primary_obligor_case)
        )
        (co_strategy_confirmed ?co_obligor_case)
        (workout_plan_available ?workout_plan)
      )
    :effect
      (and
        (workout_plan_created ?workout_plan)
        (workout_plan_includes_primary_strategy ?workout_plan ?primary_strategy)
        (workout_plan_includes_co_strategy ?workout_plan ?co_obligor_strategy)
        (workout_plan_flag_primary_remediation ?workout_plan)
        (not
          (workout_plan_available ?workout_plan)
        )
      )
  )
  (:action compose_workout_plan_with_co_remediation
    :parameters (?primary_obligor_case - primary_obligor_case ?co_obligor_case - co_obligor_case ?primary_strategy - primary_strategy ?co_obligor_strategy - co_obligor_strategy ?workout_plan - workout_plan)
    :precondition
      (and
        (primary_case_ready_for_plan ?primary_obligor_case)
        (co_obligor_case_ready_for_plan ?co_obligor_case)
        (primary_case_assigned_strategy ?primary_obligor_case ?primary_strategy)
        (co_obligor_assigned_strategy ?co_obligor_case ?co_obligor_strategy)
        (primary_strategy_selected ?primary_strategy)
        (co_strategy_remediation_attached ?co_obligor_strategy)
        (primary_strategy_confirmed ?primary_obligor_case)
        (not
          (co_strategy_confirmed ?co_obligor_case)
        )
        (workout_plan_available ?workout_plan)
      )
    :effect
      (and
        (workout_plan_created ?workout_plan)
        (workout_plan_includes_primary_strategy ?workout_plan ?primary_strategy)
        (workout_plan_includes_co_strategy ?workout_plan ?co_obligor_strategy)
        (workout_plan_flag_co_remediation ?workout_plan)
        (not
          (workout_plan_available ?workout_plan)
        )
      )
  )
  (:action compose_workout_plan_with_primary_and_co_remediation
    :parameters (?primary_obligor_case - primary_obligor_case ?co_obligor_case - co_obligor_case ?primary_strategy - primary_strategy ?co_obligor_strategy - co_obligor_strategy ?workout_plan - workout_plan)
    :precondition
      (and
        (primary_case_ready_for_plan ?primary_obligor_case)
        (co_obligor_case_ready_for_plan ?co_obligor_case)
        (primary_case_assigned_strategy ?primary_obligor_case ?primary_strategy)
        (co_obligor_assigned_strategy ?co_obligor_case ?co_obligor_strategy)
        (primary_strategy_remediation_attached ?primary_strategy)
        (co_strategy_remediation_attached ?co_obligor_strategy)
        (not
          (primary_strategy_confirmed ?primary_obligor_case)
        )
        (not
          (co_strategy_confirmed ?co_obligor_case)
        )
        (workout_plan_available ?workout_plan)
      )
    :effect
      (and
        (workout_plan_created ?workout_plan)
        (workout_plan_includes_primary_strategy ?workout_plan ?primary_strategy)
        (workout_plan_includes_co_strategy ?workout_plan ?co_obligor_strategy)
        (workout_plan_flag_primary_remediation ?workout_plan)
        (workout_plan_flag_co_remediation ?workout_plan)
        (not
          (workout_plan_available ?workout_plan)
        )
      )
  )
  (:action schedule_workout_plan
    :parameters (?workout_plan - workout_plan ?primary_obligor_case - primary_obligor_case ?assessment_resource - assessment_resource)
    :precondition
      (and
        (workout_plan_created ?workout_plan)
        (primary_case_ready_for_plan ?primary_obligor_case)
        (decision_assigned_assessment ?primary_obligor_case ?assessment_resource)
        (not
          (workout_plan_scheduled ?workout_plan)
        )
      )
    :effect (workout_plan_scheduled ?workout_plan)
  )
  (:action allocate_collateral_asset_to_plan
    :parameters (?workout_decision - workout_decision ?collateral_asset - collateral_asset ?workout_plan - workout_plan)
    :precondition
      (and
        (decision_screened ?workout_decision)
        (decision_includes_workout_plan ?workout_decision ?workout_plan)
        (decision_includes_collateral_asset ?workout_decision ?collateral_asset)
        (collateral_asset_available ?collateral_asset)
        (workout_plan_created ?workout_plan)
