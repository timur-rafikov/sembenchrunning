--- domain.pddl (as passed to VAL) ---
(define (domain pharmaceutics_scaleup_batch_readiness)
  (:requirements :strips :typing :negative-preconditions)
  (:types human_or_document_resource - object technical_resource - object equipment - object project_case - object scaleup_entity - project_case manufacturing_site - human_or_document_resource analytical_scientist - human_or_document_resource process_engineer - human_or_document_resource regulatory_checklist - human_or_document_resource qa_assessment - human_or_document_resource evidence_artifact - human_or_document_resource manufacturing_procedure - human_or_document_resource sponsor_approval - human_or_document_resource analytical_method - technical_resource stability_plan - technical_resource risk_assessment - technical_resource primary_equipment_config - equipment secondary_equipment_config - equipment scaleup_package - equipment auxiliary_resource - scaleup_entity auxiliary_document - scaleup_entity primary_process - auxiliary_resource secondary_process - auxiliary_resource formulation_dossier - auxiliary_document)
  (:predicates
    (entity_registered ?development_candidate - scaleup_entity)
    (assessment_completed_for_entity ?development_candidate - scaleup_entity)
    (site_assigned_to_entity ?development_candidate - scaleup_entity)
    (is_scaleup_ready ?development_candidate - scaleup_entity)
    (operational_signoff ?development_candidate - scaleup_entity)
    (evidence_bound_to_entity ?development_candidate - scaleup_entity)
    (site_available ?manufacturing_site - manufacturing_site)
    (assigned_manufacturing_site_to_entity ?development_candidate - scaleup_entity ?manufacturing_site - manufacturing_site)
    (analytical_scientist_available ?analytical_scientist - analytical_scientist)
    (assigned_analytical_scientist ?development_candidate - scaleup_entity ?analytical_scientist - analytical_scientist)
    (engineer_available ?process_engineer - process_engineer)
    (assigned_process_engineer ?development_candidate - scaleup_entity ?process_engineer - process_engineer)
    (analytical_method_available ?analytical_method - analytical_method)
    (primary_process_method_link ?primary_process - primary_process ?analytical_method - analytical_method)
    (secondary_process_method_link ?secondary_process - secondary_process ?analytical_method - analytical_method)
    (primary_process_equipment_match ?primary_process - primary_process ?primary_equipment_config - primary_equipment_config)
    (primary_equipment_feasible ?primary_equipment_config - primary_equipment_config)
    (primary_equipment_concern_flag ?primary_equipment_config - primary_equipment_config)
    (primary_process_feasible ?primary_process - primary_process)
    (secondary_process_equipment_match ?secondary_process - secondary_process ?secondary_equipment_config - secondary_equipment_config)
    (secondary_equipment_feasible ?secondary_equipment_config - secondary_equipment_config)
    (secondary_equipment_concern_flag ?secondary_equipment_config - secondary_equipment_config)
    (secondary_process_feasible ?secondary_process - secondary_process)
    (scaleup_package_draft_available ?scaleup_package - scaleup_package)
    (scaleup_package_created ?scaleup_package - scaleup_package)
    (package_links_primary_equipment ?scaleup_package - scaleup_package ?primary_equipment_config - primary_equipment_config)
    (package_links_secondary_equipment ?scaleup_package - scaleup_package ?secondary_equipment_config - secondary_equipment_config)
    (package_primary_option_flag ?scaleup_package - scaleup_package)
    (package_secondary_option_flag ?scaleup_package - scaleup_package)
    (package_option_validated ?scaleup_package - scaleup_package)
    (dossier_links_primary_process ?formulation_dossier - formulation_dossier ?primary_process - primary_process)
    (dossier_links_secondary_process ?formulation_dossier - formulation_dossier ?secondary_process - secondary_process)
    (dossier_links_package ?formulation_dossier - formulation_dossier ?scaleup_package - scaleup_package)
    (stability_plan_available ?stability_plan - stability_plan)
    (dossier_links_stability_plan ?formulation_dossier - formulation_dossier ?stability_plan - stability_plan)
    (stability_plan_attached ?stability_plan - stability_plan)
    (stability_plan_linked_to_package ?stability_plan - stability_plan ?scaleup_package - scaleup_package)
    (dossier_stability_regulatory_checks_passed ?formulation_dossier - formulation_dossier)
    (regulatory_review_initiated ?formulation_dossier - formulation_dossier)
    (regulatory_review_completed ?formulation_dossier - formulation_dossier)
    (regulatory_checklist_attached ?formulation_dossier - formulation_dossier)
    (regulatory_checklist_validated ?formulation_dossier - formulation_dossier)
    (qa_sequence_ready ?formulation_dossier - formulation_dossier)
    (signoff_prepared ?formulation_dossier - formulation_dossier)
    (risk_assessment_available ?risk_assessment - risk_assessment)
    (dossier_linked_risk_assessment ?formulation_dossier - formulation_dossier ?risk_assessment - risk_assessment)
    (risk_assessment_applied ?formulation_dossier - formulation_dossier)
    (risk_assessment_reviewed ?formulation_dossier - formulation_dossier)
    (risk_assessment_approved ?formulation_dossier - formulation_dossier)
    (regulatory_checklist_available ?regulatory_checklist - regulatory_checklist)
    (dossier_regulatory_checklist_link ?formulation_dossier - formulation_dossier ?regulatory_checklist - regulatory_checklist)
    (qa_assessment_available ?qa_assessment - qa_assessment)
    (dossier_qa_link ?formulation_dossier - formulation_dossier ?qa_assessment - qa_assessment)
    (manufacturing_procedure_available ?manufacturing_procedure - manufacturing_procedure)
    (dossier_procedure_link ?formulation_dossier - formulation_dossier ?manufacturing_procedure - manufacturing_procedure)
    (sponsor_approval_available ?sponsor_approval - sponsor_approval)
    (dossier_sponsor_approval_link ?formulation_dossier - formulation_dossier ?sponsor_approval - sponsor_approval)
    (evidence_artifact_available ?evidence_artifact - evidence_artifact)
    (entity_evidence_link ?development_candidate - scaleup_entity ?evidence_artifact - evidence_artifact)
    (primary_process_evaluated ?primary_process - primary_process)
    (secondary_process_evaluated ?secondary_process - secondary_process)
    (final_signoff_recorded ?formulation_dossier - formulation_dossier)
  )
  (:action register_development_candidate
    :parameters (?development_candidate - scaleup_entity)
    :precondition
      (and
        (not
          (entity_registered ?development_candidate)
        )
        (not
          (is_scaleup_ready ?development_candidate)
        )
      )
    :effect (entity_registered ?development_candidate)
  )
  (:action allocate_manufacturing_site
    :parameters (?development_candidate - scaleup_entity ?manufacturing_site - manufacturing_site)
    :precondition
      (and
        (entity_registered ?development_candidate)
        (not
          (site_assigned_to_entity ?development_candidate)
        )
        (site_available ?manufacturing_site)
      )
    :effect
      (and
        (site_assigned_to_entity ?development_candidate)
        (assigned_manufacturing_site_to_entity ?development_candidate ?manufacturing_site)
        (not
          (site_available ?manufacturing_site)
        )
      )
  )
  (:action assign_analytical_scientist
    :parameters (?development_candidate - scaleup_entity ?analytical_scientist - analytical_scientist)
    :precondition
      (and
        (entity_registered ?development_candidate)
        (site_assigned_to_entity ?development_candidate)
        (analytical_scientist_available ?analytical_scientist)
      )
    :effect
      (and
        (assigned_analytical_scientist ?development_candidate ?analytical_scientist)
        (not
          (analytical_scientist_available ?analytical_scientist)
        )
      )
  )
  (:action complete_analytical_assessment
    :parameters (?development_candidate - scaleup_entity ?analytical_scientist - analytical_scientist)
    :precondition
      (and
        (entity_registered ?development_candidate)
        (site_assigned_to_entity ?development_candidate)
        (assigned_analytical_scientist ?development_candidate ?analytical_scientist)
        (not
          (assessment_completed_for_entity ?development_candidate)
        )
      )
    :effect (assessment_completed_for_entity ?development_candidate)
  )
  (:action unassign_analyst_from_candidate
    :parameters (?development_candidate - scaleup_entity ?analytical_scientist - analytical_scientist)
    :precondition
      (and
        (assigned_analytical_scientist ?development_candidate ?analytical_scientist)
      )
    :effect
      (and
        (analytical_scientist_available ?analytical_scientist)
        (not
          (assigned_analytical_scientist ?development_candidate ?analytical_scientist)
        )
      )
  )
  (:action assign_process_engineer
    :parameters (?development_candidate - scaleup_entity ?process_engineer - process_engineer)
    :precondition
      (and
        (assessment_completed_for_entity ?development_candidate)
        (engineer_available ?process_engineer)
      )
    :effect
      (and
        (assigned_process_engineer ?development_candidate ?process_engineer)
        (not
          (engineer_available ?process_engineer)
        )
      )
  )
  (:action unassign_process_engineer
    :parameters (?development_candidate - scaleup_entity ?process_engineer - process_engineer)
    :precondition
      (and
        (assigned_process_engineer ?development_candidate ?process_engineer)
      )
    :effect
      (and
        (engineer_available ?process_engineer)
        (not
          (assigned_process_engineer ?development_candidate ?process_engineer)
        )
      )
  )
  (:action attach_manufacturing_procedure_to_dossier
    :parameters (?formulation_dossier - formulation_dossier ?manufacturing_procedure - manufacturing_procedure)
    :precondition
      (and
        (assessment_completed_for_entity ?formulation_dossier)
        (manufacturing_procedure_available ?manufacturing_procedure)
      )
    :effect
      (and
        (dossier_procedure_link ?formulation_dossier ?manufacturing_procedure)
        (not
          (manufacturing_procedure_available ?manufacturing_procedure)
        )
      )
  )
  (:action detach_manufacturing_procedure_from_dossier
    :parameters (?formulation_dossier - formulation_dossier ?manufacturing_procedure - manufacturing_procedure)
    :precondition
      (and
        (dossier_procedure_link ?formulation_dossier ?manufacturing_procedure)
      )
    :effect
      (and
        (manufacturing_procedure_available ?manufacturing_procedure)
        (not
          (dossier_procedure_link ?formulation_dossier ?manufacturing_procedure)
        )
      )
  )
  (:action attach_sponsor_approval
    :parameters (?formulation_dossier - formulation_dossier ?sponsor_approval - sponsor_approval)
    :precondition
      (and
        (assessment_completed_for_entity ?formulation_dossier)
        (sponsor_approval_available ?sponsor_approval)
      )
    :effect
      (and
        (dossier_sponsor_approval_link ?formulation_dossier ?sponsor_approval)
        (not
          (sponsor_approval_available ?sponsor_approval)
        )
      )
  )
  (:action detach_sponsor_approval
    :parameters (?formulation_dossier - formulation_dossier ?sponsor_approval - sponsor_approval)
    :precondition
      (and
        (dossier_sponsor_approval_link ?formulation_dossier ?sponsor_approval)
      )
    :effect
      (and
        (sponsor_approval_available ?sponsor_approval)
        (not
          (dossier_sponsor_approval_link ?formulation_dossier ?sponsor_approval)
        )
      )
  )
  (:action assess_primary_equipment_feasibility
    :parameters (?primary_process - primary_process ?primary_equipment_config - primary_equipment_config ?analytical_scientist - analytical_scientist)
    :precondition
      (and
        (assessment_completed_for_entity ?primary_process)
        (assigned_analytical_scientist ?primary_process ?analytical_scientist)
        (primary_process_equipment_match ?primary_process ?primary_equipment_config)
        (not
          (primary_equipment_feasible ?primary_equipment_config)
        )
        (not
          (primary_equipment_concern_flag ?primary_equipment_config)
        )
      )
    :effect (primary_equipment_feasible ?primary_equipment_config)
  )
  (:action confirm_primary_process_feasibility
    :parameters (?primary_process - primary_process ?primary_equipment_config - primary_equipment_config ?process_engineer - process_engineer)
    :precondition
      (and
        (assessment_completed_for_entity ?primary_process)
        (assigned_process_engineer ?primary_process ?process_engineer)
        (primary_process_equipment_match ?primary_process ?primary_equipment_config)
        (primary_equipment_feasible ?primary_equipment_config)
        (not
          (primary_process_evaluated ?primary_process)
        )
      )
    :effect
      (and
        (primary_process_evaluated ?primary_process)
        (primary_process_feasible ?primary_process)
      )
  )
  (:action attach_method_to_primary_process_and_evaluate
    :parameters (?primary_process - primary_process ?primary_equipment_config - primary_equipment_config ?analytical_method - analytical_method)
    :precondition
      (and
        (assessment_completed_for_entity ?primary_process)
        (primary_process_equipment_match ?primary_process ?primary_equipment_config)
        (analytical_method_available ?analytical_method)
        (not
          (primary_process_evaluated ?primary_process)
        )
      )
    :effect
      (and
        (primary_equipment_concern_flag ?primary_equipment_config)
        (primary_process_evaluated ?primary_process)
        (primary_process_method_link ?primary_process ?analytical_method)
        (not
          (analytical_method_available ?analytical_method)
        )
      )
  )
  (:action finalize_primary_process_evaluation
    :parameters (?primary_process - primary_process ?primary_equipment_config - primary_equipment_config ?analytical_scientist - analytical_scientist ?analytical_method - analytical_method)
    :precondition
      (and
        (assessment_completed_for_entity ?primary_process)
        (assigned_analytical_scientist ?primary_process ?analytical_scientist)
        (primary_process_equipment_match ?primary_process ?primary_equipment_config)
        (primary_equipment_concern_flag ?primary_equipment_config)
        (primary_process_method_link ?primary_process ?analytical_method)
        (not
          (primary_process_feasible ?primary_process)
        )
      )
    :effect
      (and
        (primary_equipment_feasible ?primary_equipment_config)
        (primary_process_feasible ?primary_process)
        (analytical_method_available ?analytical_method)
        (not
          (primary_process_method_link ?primary_process ?analytical_method)
        )
      )
  )
  (:action assess_secondary_equipment_feasibility
    :parameters (?secondary_process - secondary_process ?secondary_equipment_config - secondary_equipment_config ?analytical_scientist - analytical_scientist)
    :precondition
      (and
        (assessment_completed_for_entity ?secondary_process)
        (assigned_analytical_scientist ?secondary_process ?analytical_scientist)
        (secondary_process_equipment_match ?secondary_process ?secondary_equipment_config)
        (not
          (secondary_equipment_feasible ?secondary_equipment_config)
        )
        (not
          (secondary_equipment_concern_flag ?secondary_equipment_config)
        )
      )
    :effect (secondary_equipment_feasible ?secondary_equipment_config)
  )
  (:action confirm_secondary_process_feasibility
    :parameters (?secondary_process - secondary_process ?secondary_equipment_config - secondary_equipment_config ?process_engineer - process_engineer)
    :precondition
      (and
        (assessment_completed_for_entity ?secondary_process)
        (assigned_process_engineer ?secondary_process ?process_engineer)
        (secondary_process_equipment_match ?secondary_process ?secondary_equipment_config)
        (secondary_equipment_feasible ?secondary_equipment_config)
        (not
          (secondary_process_evaluated ?secondary_process)
        )
      )
    :effect
      (and
        (secondary_process_evaluated ?secondary_process)
        (secondary_process_feasible ?secondary_process)
      )
  )
  (:action attach_method_to_secondary_process_and_evaluate
    :parameters (?secondary_process - secondary_process ?secondary_equipment_config - secondary_equipment_config ?analytical_method - analytical_method)
    :precondition
      (and
        (assessment_completed_for_entity ?secondary_process)
        (secondary_process_equipment_match ?secondary_process ?secondary_equipment_config)
        (analytical_method_available ?analytical_method)
        (not
          (secondary_process_evaluated ?secondary_process)
        )
      )
    :effect
      (and
        (secondary_equipment_concern_flag ?secondary_equipment_config)
        (secondary_process_evaluated ?secondary_process)
        (secondary_process_method_link ?secondary_process ?analytical_method)
        (not
          (analytical_method_available ?analytical_method)
        )
      )
  )
  (:action finalize_secondary_process_evaluation
    :parameters (?secondary_process - secondary_process ?secondary_equipment_config - secondary_equipment_config ?analytical_scientist - analytical_scientist ?analytical_method - analytical_method)
    :precondition
      (and
        (assessment_completed_for_entity ?secondary_process)
        (assigned_analytical_scientist ?secondary_process ?analytical_scientist)
        (secondary_process_equipment_match ?secondary_process ?secondary_equipment_config)
        (secondary_equipment_concern_flag ?secondary_equipment_config)
        (secondary_process_method_link ?secondary_process ?analytical_method)
        (not
          (secondary_process_feasible ?secondary_process)
        )
      )
    :effect
      (and
        (secondary_equipment_feasible ?secondary_equipment_config)
        (secondary_process_feasible ?secondary_process)
        (analytical_method_available ?analytical_method)
        (not
          (secondary_process_method_link ?secondary_process ?analytical_method)
        )
      )
  )
  (:action assemble_scaleup_package_option
    :parameters (?primary_process - primary_process ?secondary_process - secondary_process ?primary_equipment_config - primary_equipment_config ?secondary_equipment_config - secondary_equipment_config ?scaleup_package - scaleup_package)
    :precondition
      (and
        (primary_process_evaluated ?primary_process)
        (secondary_process_evaluated ?secondary_process)
        (primary_process_equipment_match ?primary_process ?primary_equipment_config)
        (secondary_process_equipment_match ?secondary_process ?secondary_equipment_config)
        (primary_equipment_feasible ?primary_equipment_config)
        (secondary_equipment_feasible ?secondary_equipment_config)
        (primary_process_feasible ?primary_process)
        (secondary_process_feasible ?secondary_process)
        (scaleup_package_draft_available ?scaleup_package)
      )
    :effect
      (and
        (scaleup_package_created ?scaleup_package)
        (package_links_primary_equipment ?scaleup_package ?primary_equipment_config)
        (package_links_secondary_equipment ?scaleup_package ?secondary_equipment_config)
        (not
          (scaleup_package_draft_available ?scaleup_package)
        )
      )
  )
  (:action assemble_scaleup_package_option_with_primary_integration
    :parameters (?primary_process - primary_process ?secondary_process - secondary_process ?primary_equipment_config - primary_equipment_config ?secondary_equipment_config - secondary_equipment_config ?scaleup_package - scaleup_package)
    :precondition
      (and
        (primary_process_evaluated ?primary_process)
        (secondary_process_evaluated ?secondary_process)
        (primary_process_equipment_match ?primary_process ?primary_equipment_config)
        (secondary_process_equipment_match ?secondary_process ?secondary_equipment_config)
        (primary_equipment_concern_flag ?primary_equipment_config)
        (secondary_equipment_feasible ?secondary_equipment_config)
        (not
          (primary_process_feasible ?primary_process)
        )
        (secondary_process_feasible ?secondary_process)
        (scaleup_package_draft_available ?scaleup_package)
      )
    :effect
      (and
        (scaleup_package_created ?scaleup_package)
        (package_links_primary_equipment ?scaleup_package ?primary_equipment_config)
        (package_links_secondary_equipment ?scaleup_package ?secondary_equipment_config)
        (package_primary_option_flag ?scaleup_package)
        (not
          (scaleup_package_draft_available ?scaleup_package)
        )
      )
  )
  (:action assemble_scaleup_package_option_with_secondary_integration
    :parameters (?primary_process - primary_process ?secondary_process - secondary_process ?primary_equipment_config - primary_equipment_config ?secondary_equipment_config - secondary_equipment_config ?scaleup_package - scaleup_package)
    :precondition
      (and
        (primary_process_evaluated ?primary_process)
        (secondary_process_evaluated ?secondary_process)
        (primary_process_equipment_match ?primary_process ?primary_equipment_config)
        (secondary_process_equipment_match ?secondary_process ?secondary_equipment_config)
        (primary_equipment_feasible ?primary_equipment_config)
        (secondary_equipment_concern_flag ?secondary_equipment_config)
        (primary_process_feasible ?primary_process)
        (not
          (secondary_process_feasible ?secondary_process)
        )
        (scaleup_package_draft_available ?scaleup_package)
      )
    :effect
      (and
        (scaleup_package_created ?scaleup_package)
        (package_links_primary_equipment ?scaleup_package ?primary_equipment_config)
        (package_links_secondary_equipment ?scaleup_package ?secondary_equipment_config)
        (package_secondary_option_flag ?scaleup_package)
        (not
          (scaleup_package_draft_available ?scaleup_package)
        )
      )
  )
  (:action assemble_scaleup_package_option_with_both_integrations
    :parameters (?primary_process - primary_process ?secondary_process - secondary_process ?primary_equipment_config - primary_equipment_config ?secondary_equipment_config - secondary_equipment_config ?scaleup_package - scaleup_package)
    :precondition
      (and
        (primary_process_evaluated ?primary_process)
        (secondary_process_evaluated ?secondary_process)
        (primary_process_equipment_match ?primary_process ?primary_equipment_config)
        (secondary_process_equipment_match ?secondary_process ?secondary_equipment_config)
        (primary_equipment_concern_flag ?primary_equipment_config)
        (secondary_equipment_concern_flag ?secondary_equipment_config)
        (not
          (primary_process_feasible ?primary_process)
        )
        (not
          (secondary_process_feasible ?secondary_process)
        )
        (scaleup_package_draft_available ?scaleup_package)
      )
    :effect
      (and
        (scaleup_package_created ?scaleup_package)
        (package_links_primary_equipment ?scaleup_package ?primary_equipment_config)
        (package_links_secondary_equipment ?scaleup_package ?secondary_equipment_config)
        (package_primary_option_flag ?scaleup_package)
        (package_secondary_option_flag ?scaleup_package)
        (not
          (scaleup_package_draft_available ?scaleup_package)
        )
      )
  )
  (:action validate_package_option
    :parameters (?scaleup_package - scaleup_package ?primary_process - primary_process ?analytical_scientist - analytical_scientist)
    :precondition
      (and
        (scaleup_package_created ?scaleup_package)
        (primary_process_evaluated ?primary_process)
        (assigned_analytical_scientist ?primary_process ?analytical_scientist)
        (not
          (package_option_validated ?scaleup_package)
        )
      )
    :effect (package_option_validated ?scaleup_package)
  )
  (:action integrate_stability_plan_into_dossier
    :parameters (?formulation_dossier - formulation_dossier ?stability_plan - stability_plan ?scaleup_package - scaleup_package)
    :precondition
      (and
        (assessment_completed_for_entity ?formulation_dossier)
        (dossier_links_package ?formulation_dossier ?scaleup_package)
        (dossier_links_stability_plan ?formulation_dossier ?stability_plan)
        (stability_plan_available ?stability_plan)
        (scaleup_package_created ?scaleup_package)
        (package_option_validated ?scaleup_package)
        (not
          (stability_plan_attached ?stability_plan)
        )
      )
    :effect
      (and
        (stability_plan_attached ?stability_plan)
        (stability_plan_linked_to_package ?stability_plan ?scaleup_package)
        (not
          (stability_plan_available ?stability_plan)
        )
      )
  )
  (:action confirm_stability_plan_integration_for_dossier
    :parameters (?formulation_dossier - formulation_dossier ?stability_plan - stability_plan ?scaleup_package - scaleup_package ?analytical_scientist - analytical_scientist)
    :precondition
      (and
        (assessment_completed_for_entity ?formulation_dossier)
        (dossier_links_stability_plan ?formulation_dossier ?stability_plan)
        (stability_plan_attached ?stability_plan)
        (stability_plan_linked_to_package ?stability_plan ?scaleup_package)
        (assigned_analytical_scientist ?formulation_dossier ?analytical_scientist)
        (not
          (package_primary_option_flag ?scaleup_package)
        )
        (not
          (dossier_stability_regulatory_checks_passed ?formulation_dossier)
        )
      )
    :effect (dossier_stability_regulatory_checks_passed ?formulation_dossier)
  )
  (:action apply_regulatory_checklist_to_dossier
    :parameters (?formulation_dossier - formulation_dossier ?regulatory_checklist - regulatory_checklist)
    :precondition
      (and
        (assessment_completed_for_entity ?formulation_dossier)
        (regulatory_checklist_available ?regulatory_checklist)
        (not
          (regulatory_checklist_attached ?formulation_dossier)
        )
      )
    :effect
      (and
        (regulatory_checklist_attached ?formulation_dossier)
        (dossier_regulatory_checklist_link ?formulation_dossier ?regulatory_checklist)
        (not
          (regulatory_checklist_available ?regulatory_checklist)
        )
      )
  )
  (:action execute_regulatory_and_stability_review
    :parameters (?formulation_dossier - formulation_dossier ?stability_plan - stability_plan ?scaleup_package - scaleup_package ?analytical_scientist - analytical_scientist ?regulatory_checklist - regulatory_checklist)
    :precondition
