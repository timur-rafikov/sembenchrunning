--- domain.pddl (as passed to VAL) ---
(define (domain postharvest_destination_market_channel_assignment)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_category - object infrastructure_category - object logistics_category - object inventory_category - object batch - inventory_category market_channel_option - resource_category quality_test_resource - resource_category conditioning_station - resource_category label_template - resource_category dispatch_document_template - resource_category destination_attribute - resource_category packaging_spec - resource_category certification_document - resource_category packaging_material - infrastructure_category sample_unit - infrastructure_category buyer - infrastructure_category temperature_zone - logistics_category transit_route - logistics_category consignment_unit - logistics_category storage_category - batch product_category - batch storage_slot - storage_category staging_slot - storage_category product_sku - product_category)
  (:predicates
    (received ?batch - batch)
    (quality_approved ?batch - batch)
    (channel_nominated ?batch - batch)
    (released_for_destination ?batch - batch)
    (ready_for_dispatch ?batch - batch)
    (destination_assigned ?batch - batch)
    (market_channel_available ?market_channel - market_channel_option)
    (channel_assignment ?batch - batch ?market_channel - market_channel_option)
    (test_resource_available ?test_resource - quality_test_resource)
    (test_assignment ?batch - batch ?test_resource - quality_test_resource)
    (conditioning_station_available ?conditioning_station - conditioning_station)
    (conditioning_assignment ?batch - batch ?conditioning_station - conditioning_station)
    (packaging_material_available ?packaging_material - packaging_material)
    (storage_slot_packaging_assignment ?storage_slot - storage_slot ?packaging_material - packaging_material)
    (staging_slot_packaging_assignment ?staging_slot - staging_slot ?packaging_material - packaging_material)
    (storage_slot_in_temperature_zone ?storage_slot - storage_slot ?temperature_zone - temperature_zone)
    (temperature_zone_reserved ?temperature_zone - temperature_zone)
    (temperature_zone_material_loaded ?temperature_zone - temperature_zone)
    (storage_slot_occupied ?storage_slot - storage_slot)
    (staging_slot_transit_assignment ?staging_slot - staging_slot ?transit_route - transit_route)
    (transit_route_reserved ?transit_route - transit_route)
    (transit_route_material_loaded ?transit_route - transit_route)
    (staging_slot_occupied ?staging_slot - staging_slot)
    (consignment_unit_available ?consignment_unit - consignment_unit)
    (consignment_reserved ?consignment_unit - consignment_unit)
    (consignment_in_temperature_zone ?consignment_unit - consignment_unit ?temperature_zone - temperature_zone)
    (consignment_transit_assignment ?consignment_unit - consignment_unit ?transit_route - transit_route)
    (consignment_temperature_allocated ?consignment_unit - consignment_unit)
    (consignment_transit_allocated ?consignment_unit - consignment_unit)
    (consignment_inspected ?consignment_unit - consignment_unit)
    (sku_assigned_storage_slot ?sku - product_sku ?storage_slot - storage_slot)
    (sku_assigned_staging_slot ?sku - product_sku ?staging_slot - staging_slot)
    (sku_assigned_consignment ?sku - product_sku ?consignment_unit - consignment_unit)
    (sample_unit_available ?sample_unit - sample_unit)
    (sku_sample_association ?sku - product_sku ?sample_unit - sample_unit)
    (sample_unit_reserved ?sample_unit - sample_unit)
    (sample_assigned_consignment ?sample_unit - sample_unit ?consignment_unit - consignment_unit)
    (sku_inspected ?sku - product_sku)
    (sku_label_applied ?sku - product_sku)
    (sku_prepared_for_release ?sku - product_sku)
    (sku_has_label_template ?sku - product_sku)
    (sku_label_committed ?sku - product_sku)
    (sku_dispatch_documents_attached ?sku - product_sku)
    (sku_packing_confirmed ?sku - product_sku)
    (buyer_available ?buyer - buyer)
    (sku_buyer_assignment ?sku - product_sku ?buyer - buyer)
    (sku_buyer_selected ?sku - product_sku)
    (sku_buyer_assignment_confirmed ?sku - product_sku)
    (sku_certification_attached ?sku - product_sku)
    (label_template_available ?label_template - label_template)
    (sku_label_template_assigned ?sku - product_sku ?label_template - label_template)
    (dispatch_template_available ?dispatch_template - dispatch_document_template)
    (sku_dispatch_template_assigned ?sku - product_sku ?dispatch_template - dispatch_document_template)
    (packaging_spec_available ?packaging_spec - packaging_spec)
    (sku_packaging_spec_assigned ?sku - product_sku ?packaging_spec - packaging_spec)
    (certification_document_available ?certification_document - certification_document)
    (sku_certification_assigned ?sku - product_sku ?certification_document - certification_document)
    (destination_attribute_available ?destination_attribute - destination_attribute)
    (inventory_destination_attribute_assignment ?batch - batch ?destination_attribute - destination_attribute)
    (storage_slot_prepared ?storage_slot - storage_slot)
    (staging_slot_prepared ?staging_slot - staging_slot)
    (final_release_marker ?sku - product_sku)
  )
  (:action register_batch_receipt
    :parameters (?batch - batch)
    :precondition
      (and
        (not
          (received ?batch)
        )
        (not
          (released_for_destination ?batch)
        )
      )
    :effect (received ?batch)
  )
  (:action nominate_batch_market_channel
    :parameters (?batch - batch ?market_channel - market_channel_option)
    :precondition
      (and
        (received ?batch)
        (not
          (channel_nominated ?batch)
        )
        (market_channel_available ?market_channel)
      )
    :effect
      (and
        (channel_nominated ?batch)
        (channel_assignment ?batch ?market_channel)
        (not
          (market_channel_available ?market_channel)
        )
      )
  )
  (:action assign_test_resource_to_batch
    :parameters (?batch - batch ?test_resource - quality_test_resource)
    :precondition
      (and
        (received ?batch)
        (channel_nominated ?batch)
        (test_resource_available ?test_resource)
      )
    :effect
      (and
        (test_assignment ?batch ?test_resource)
        (not
          (test_resource_available ?test_resource)
        )
      )
  )
  (:action approve_batch_quality_test
    :parameters (?batch - batch ?test_resource - quality_test_resource)
    :precondition
      (and
        (received ?batch)
        (channel_nominated ?batch)
        (test_assignment ?batch ?test_resource)
        (not
          (quality_approved ?batch)
        )
      )
    :effect (quality_approved ?batch)
  )
  (:action release_test_resource_from_batch
    :parameters (?batch - batch ?test_resource - quality_test_resource)
    :precondition
      (and
        (test_assignment ?batch ?test_resource)
      )
    :effect
      (and
        (test_resource_available ?test_resource)
        (not
          (test_assignment ?batch ?test_resource)
        )
      )
  )
  (:action assign_conditioning_station_to_batch
    :parameters (?batch - batch ?conditioning_station - conditioning_station)
    :precondition
      (and
        (quality_approved ?batch)
        (conditioning_station_available ?conditioning_station)
      )
    :effect
      (and
        (conditioning_assignment ?batch ?conditioning_station)
        (not
          (conditioning_station_available ?conditioning_station)
        )
      )
  )
  (:action release_conditioning_station_from_batch
    :parameters (?batch - batch ?conditioning_station - conditioning_station)
    :precondition
      (and
        (conditioning_assignment ?batch ?conditioning_station)
      )
    :effect
      (and
        (conditioning_station_available ?conditioning_station)
        (not
          (conditioning_assignment ?batch ?conditioning_station)
        )
      )
  )
  (:action assign_packaging_spec_to_sku
    :parameters (?sku - product_sku ?packaging_spec - packaging_spec)
    :precondition
      (and
        (quality_approved ?sku)
        (packaging_spec_available ?packaging_spec)
      )
    :effect
      (and
        (sku_packaging_spec_assigned ?sku ?packaging_spec)
        (not
          (packaging_spec_available ?packaging_spec)
        )
      )
  )
  (:action release_packaging_spec_from_sku
    :parameters (?sku - product_sku ?packaging_spec - packaging_spec)
    :precondition
      (and
        (sku_packaging_spec_assigned ?sku ?packaging_spec)
      )
    :effect
      (and
        (packaging_spec_available ?packaging_spec)
        (not
          (sku_packaging_spec_assigned ?sku ?packaging_spec)
        )
      )
  )
  (:action assign_certification_document_to_sku
    :parameters (?sku - product_sku ?certification_document - certification_document)
    :precondition
      (and
        (quality_approved ?sku)
        (certification_document_available ?certification_document)
      )
    :effect
      (and
        (sku_certification_assigned ?sku ?certification_document)
        (not
          (certification_document_available ?certification_document)
        )
      )
  )
  (:action release_certification_document_from_sku
    :parameters (?sku - product_sku ?certification_document - certification_document)
    :precondition
      (and
        (sku_certification_assigned ?sku ?certification_document)
      )
    :effect
      (and
        (certification_document_available ?certification_document)
        (not
          (sku_certification_assigned ?sku ?certification_document)
        )
      )
  )
  (:action reserve_temperature_zone_for_storage_slot
    :parameters (?storage_slot - storage_slot ?temperature_zone - temperature_zone ?test_resource - quality_test_resource)
    :precondition
      (and
        (quality_approved ?storage_slot)
        (test_assignment ?storage_slot ?test_resource)
        (storage_slot_in_temperature_zone ?storage_slot ?temperature_zone)
        (not
          (temperature_zone_reserved ?temperature_zone)
        )
        (not
          (temperature_zone_material_loaded ?temperature_zone)
        )
      )
    :effect (temperature_zone_reserved ?temperature_zone)
  )
  (:action apply_conditioning_to_storage_slot
    :parameters (?storage_slot - storage_slot ?temperature_zone - temperature_zone ?conditioning_station - conditioning_station)
    :precondition
      (and
        (quality_approved ?storage_slot)
        (conditioning_assignment ?storage_slot ?conditioning_station)
        (storage_slot_in_temperature_zone ?storage_slot ?temperature_zone)
        (temperature_zone_reserved ?temperature_zone)
        (not
          (storage_slot_prepared ?storage_slot)
        )
      )
    :effect
      (and
        (storage_slot_prepared ?storage_slot)
        (storage_slot_occupied ?storage_slot)
      )
  )
  (:action assign_packaging_to_storage_slot
    :parameters (?storage_slot - storage_slot ?temperature_zone - temperature_zone ?packaging_material - packaging_material)
    :precondition
      (and
        (quality_approved ?storage_slot)
        (storage_slot_in_temperature_zone ?storage_slot ?temperature_zone)
        (packaging_material_available ?packaging_material)
        (not
          (storage_slot_prepared ?storage_slot)
        )
      )
    :effect
      (and
        (temperature_zone_material_loaded ?temperature_zone)
        (storage_slot_prepared ?storage_slot)
        (storage_slot_packaging_assignment ?storage_slot ?packaging_material)
        (not
          (packaging_material_available ?packaging_material)
        )
      )
  )
  (:action store_batch_in_storage_slot
    :parameters (?storage_slot - storage_slot ?temperature_zone - temperature_zone ?test_resource - quality_test_resource ?packaging_material - packaging_material)
    :precondition
      (and
        (quality_approved ?storage_slot)
        (test_assignment ?storage_slot ?test_resource)
        (storage_slot_in_temperature_zone ?storage_slot ?temperature_zone)
        (temperature_zone_material_loaded ?temperature_zone)
        (storage_slot_packaging_assignment ?storage_slot ?packaging_material)
        (not
          (storage_slot_occupied ?storage_slot)
        )
      )
    :effect
      (and
        (temperature_zone_reserved ?temperature_zone)
        (storage_slot_occupied ?storage_slot)
        (packaging_material_available ?packaging_material)
        (not
          (storage_slot_packaging_assignment ?storage_slot ?packaging_material)
        )
      )
  )
  (:action reserve_transit_route_for_staging_slot
    :parameters (?staging_slot - staging_slot ?transit_route - transit_route ?test_resource - quality_test_resource)
    :precondition
      (and
        (quality_approved ?staging_slot)
        (test_assignment ?staging_slot ?test_resource)
        (staging_slot_transit_assignment ?staging_slot ?transit_route)
        (not
          (transit_route_reserved ?transit_route)
        )
        (not
          (transit_route_material_loaded ?transit_route)
        )
      )
    :effect (transit_route_reserved ?transit_route)
  )
  (:action apply_conditioning_to_staging_slot
    :parameters (?staging_slot - staging_slot ?transit_route - transit_route ?conditioning_station - conditioning_station)
    :precondition
      (and
        (quality_approved ?staging_slot)
        (conditioning_assignment ?staging_slot ?conditioning_station)
        (staging_slot_transit_assignment ?staging_slot ?transit_route)
        (transit_route_reserved ?transit_route)
        (not
          (staging_slot_prepared ?staging_slot)
        )
      )
    :effect
      (and
        (staging_slot_prepared ?staging_slot)
        (staging_slot_occupied ?staging_slot)
      )
  )
  (:action assign_packaging_to_staging_slot
    :parameters (?staging_slot - staging_slot ?transit_route - transit_route ?packaging_material - packaging_material)
    :precondition
      (and
        (quality_approved ?staging_slot)
        (staging_slot_transit_assignment ?staging_slot ?transit_route)
        (packaging_material_available ?packaging_material)
        (not
          (staging_slot_prepared ?staging_slot)
        )
      )
    :effect
      (and
        (transit_route_material_loaded ?transit_route)
        (staging_slot_prepared ?staging_slot)
        (staging_slot_packaging_assignment ?staging_slot ?packaging_material)
        (not
          (packaging_material_available ?packaging_material)
        )
      )
  )
  (:action prepare_staging_slot_for_transit
    :parameters (?staging_slot - staging_slot ?transit_route - transit_route ?test_resource - quality_test_resource ?packaging_material - packaging_material)
    :precondition
      (and
        (quality_approved ?staging_slot)
        (test_assignment ?staging_slot ?test_resource)
        (staging_slot_transit_assignment ?staging_slot ?transit_route)
        (transit_route_material_loaded ?transit_route)
        (staging_slot_packaging_assignment ?staging_slot ?packaging_material)
        (not
          (staging_slot_occupied ?staging_slot)
        )
      )
    :effect
      (and
        (transit_route_reserved ?transit_route)
        (staging_slot_occupied ?staging_slot)
        (packaging_material_available ?packaging_material)
        (not
          (staging_slot_packaging_assignment ?staging_slot ?packaging_material)
        )
      )
  )
  (:action assemble_consignment_from_reserved_storage_and_staging
    :parameters (?storage_slot - storage_slot ?staging_slot - staging_slot ?temperature_zone - temperature_zone ?transit_route - transit_route ?consignment_unit - consignment_unit)
    :precondition
      (and
        (storage_slot_prepared ?storage_slot)
        (staging_slot_prepared ?staging_slot)
        (storage_slot_in_temperature_zone ?storage_slot ?temperature_zone)
        (staging_slot_transit_assignment ?staging_slot ?transit_route)
        (temperature_zone_reserved ?temperature_zone)
        (transit_route_reserved ?transit_route)
        (storage_slot_occupied ?storage_slot)
        (staging_slot_occupied ?staging_slot)
        (consignment_unit_available ?consignment_unit)
      )
    :effect
      (and
        (consignment_reserved ?consignment_unit)
        (consignment_in_temperature_zone ?consignment_unit ?temperature_zone)
        (consignment_transit_assignment ?consignment_unit ?transit_route)
        (not
          (consignment_unit_available ?consignment_unit)
        )
      )
  )
  (:action assemble_consignment_with_zone_material_loaded
    :parameters (?storage_slot - storage_slot ?staging_slot - staging_slot ?temperature_zone - temperature_zone ?transit_route - transit_route ?consignment_unit - consignment_unit)
    :precondition
      (and
        (storage_slot_prepared ?storage_slot)
        (staging_slot_prepared ?staging_slot)
        (storage_slot_in_temperature_zone ?storage_slot ?temperature_zone)
        (staging_slot_transit_assignment ?staging_slot ?transit_route)
        (temperature_zone_material_loaded ?temperature_zone)
        (transit_route_reserved ?transit_route)
        (not
          (storage_slot_occupied ?storage_slot)
        )
        (staging_slot_occupied ?staging_slot)
        (consignment_unit_available ?consignment_unit)
      )
    :effect
      (and
        (consignment_reserved ?consignment_unit)
        (consignment_in_temperature_zone ?consignment_unit ?temperature_zone)
        (consignment_transit_assignment ?consignment_unit ?transit_route)
        (consignment_temperature_allocated ?consignment_unit)
        (not
          (consignment_unit_available ?consignment_unit)
        )
      )
  )
  (:action assemble_consignment_with_route_material_loaded
    :parameters (?storage_slot - storage_slot ?staging_slot - staging_slot ?temperature_zone - temperature_zone ?transit_route - transit_route ?consignment_unit - consignment_unit)
    :precondition
      (and
        (storage_slot_prepared ?storage_slot)
        (staging_slot_prepared ?staging_slot)
        (storage_slot_in_temperature_zone ?storage_slot ?temperature_zone)
        (staging_slot_transit_assignment ?staging_slot ?transit_route)
        (temperature_zone_reserved ?temperature_zone)
        (transit_route_material_loaded ?transit_route)
        (storage_slot_occupied ?storage_slot)
        (not
          (staging_slot_occupied ?staging_slot)
        )
        (consignment_unit_available ?consignment_unit)
      )
    :effect
      (and
        (consignment_reserved ?consignment_unit)
        (consignment_in_temperature_zone ?consignment_unit ?temperature_zone)
        (consignment_transit_assignment ?consignment_unit ?transit_route)
        (consignment_transit_allocated ?consignment_unit)
        (not
          (consignment_unit_available ?consignment_unit)
        )
      )
  )
  (:action assemble_consignment_with_zone_and_route_material_loaded
    :parameters (?storage_slot - storage_slot ?staging_slot - staging_slot ?temperature_zone - temperature_zone ?transit_route - transit_route ?consignment_unit - consignment_unit)
    :precondition
      (and
        (storage_slot_prepared ?storage_slot)
        (staging_slot_prepared ?staging_slot)
        (storage_slot_in_temperature_zone ?storage_slot ?temperature_zone)
        (staging_slot_transit_assignment ?staging_slot ?transit_route)
        (temperature_zone_material_loaded ?temperature_zone)
        (transit_route_material_loaded ?transit_route)
        (not
          (storage_slot_occupied ?storage_slot)
        )
        (not
          (staging_slot_occupied ?staging_slot)
        )
        (consignment_unit_available ?consignment_unit)
      )
    :effect
      (and
        (consignment_reserved ?consignment_unit)
        (consignment_in_temperature_zone ?consignment_unit ?temperature_zone)
        (consignment_transit_assignment ?consignment_unit ?transit_route)
        (consignment_temperature_allocated ?consignment_unit)
        (consignment_transit_allocated ?consignment_unit)
        (not
          (consignment_unit_available ?consignment_unit)
        )
      )
  )
  (:action inspect_consignment
    :parameters (?consignment_unit - consignment_unit ?storage_slot - storage_slot ?test_resource - quality_test_resource)
    :precondition
      (and
        (consignment_reserved ?consignment_unit)
        (storage_slot_prepared ?storage_slot)
        (test_assignment ?storage_slot ?test_resource)
        (not
          (consignment_inspected ?consignment_unit)
        )
      )
    :effect (consignment_inspected ?consignment_unit)
  )
  (:action bind_sample_to_consignment
    :parameters (?sku - product_sku ?sample_unit - sample_unit ?consignment_unit - consignment_unit)
    :precondition
      (and
        (quality_approved ?sku)
        (sku_assigned_consignment ?sku ?consignment_unit)
        (sku_sample_association ?sku ?sample_unit)
        (sample_unit_available ?sample_unit)
        (consignment_reserved ?consignment_unit)
        (consignment_inspected ?consignment_unit)
        (not
          (sample_unit_reserved ?sample_unit)
        )
      )
    :effect
      (and
        (sample_unit_reserved ?sample_unit)
        (sample_assigned_consignment ?sample_unit ?consignment_unit)
        (not
          (sample_unit_available ?sample_unit)
        )
      )
  )
  (:action record_sku_inspection
    :parameters (?sku - product_sku ?sample_unit - sample_unit ?consignment_unit - consignment_unit ?test_resource - quality_test_resource)
    :precondition
      (and
        (quality_approved ?sku)
        (sku_sample_association ?sku ?sample_unit)
        (sample_unit_reserved ?sample_unit)
        (sample_assigned_consignment ?sample_unit ?consignment_unit)
        (test_assignment ?sku ?test_resource)
        (not
          (consignment_temperature_allocated ?consignment_unit)
        )
        (not
          (sku_inspected ?sku)
        )
      )
    :effect (sku_inspected ?sku)
  )
  (:action apply_label_template_to_sku
    :parameters (?sku - product_sku ?label_template - label_template)
    :precondition
      (and
        (quality_approved ?sku)
        (label_template_available ?label_template)
        (not
          (sku_has_label_template ?sku)
        )
      )
    :effect
      (and
        (sku_has_label_template ?sku)
        (sku_label_template_assigned ?sku ?label_template)
        (not
          (label_template_available ?label_template)
        )
      )
  )
  (:action assign_label_and_prepare_sku
    :parameters (?sku - product_sku ?sample_unit - sample_unit ?consignment_unit - consignment_unit ?test_resource - quality_test_resource ?label_template - label_template)
    :precondition
      (and
        (quality_approved ?sku)
        (sku_sample_association ?sku ?sample_unit)
        (sample_unit_reserved ?sample_unit)
        (sample_assigned_consignment ?sample_unit ?consignment_unit)
        (test_assignment ?sku ?test_resource)
        (consignment_temperature_allocated ?consignment_unit)
        (sku_has_label_template ?sku)
        (sku_label_template_assigned ?sku ?label_template)
        (not
          (sku_inspected ?sku)
        )
      )
    :effect
      (and
        (sku_inspected ?sku)
        (sku_label_committed ?sku)
      )
  )
  (:action apply_label_and_seal_sku_stage_a
    :parameters (?sku - product_sku ?packaging_spec - packaging_spec ?conditioning_station - conditioning_station ?sample_unit - sample_unit ?consignment_unit - consignment_unit)
    :precondition
      (and
        (sku_inspected ?sku)
        (sku_packaging_spec_assigned ?sku ?packaging_spec)
        (conditioning_assignment ?sku ?conditioning_station)
        (sku_sample_association ?sku ?sample_unit)
        (sample_assigned_consignment ?sample_unit ?consignment_unit)
        (not
          (consignment_transit_allocated ?consignment_unit)
        )
        (not
          (sku_label_applied ?sku)
        )
      )
    :effect (sku_label_applied ?sku)
  )
  (:action apply_label_and_seal_sku_stage_b
    :parameters (?sku - product_sku ?packaging_spec - packaging_spec ?conditioning_station - conditioning_station ?sample_unit - sample_unit ?consignment_unit - consignment_unit)
    :precondition
      (and
        (sku_inspected ?sku)
        (sku_packaging_spec_assigned ?sku ?packaging_spec)
        (conditioning_assignment ?sku ?conditioning_station)
        (sku_sample_association ?sku ?sample_unit)
        (sample_assigned_consignment ?sample_unit ?consignment_unit)
        (consignment_transit_allocated ?consignment_unit)
        (not
          (sku_label_applied ?sku)
        )
      )
    :effect (sku_label_applied ?sku)
  )
  (:action initiate_sku_release_approval
    :parameters (?sku - product_sku ?certification_document - certification_document ?sample_unit - sample_unit ?consignment_unit - consignment_unit)
    :precondition
      (and
        (sku_label_applied ?sku)
        (sku_certification_assigned ?sku ?certification_document)
        (sku_sample_association ?sku ?sample_unit)
        (sample_assigned_consignment ?sample_unit ?consignment_unit)
        (not
          (consignment_temperature_allocated ?consignment_unit)
        )
        (not
          (consignment_transit_allocated ?consignment_unit)
        )
        (not
          (sku_prepared_for_release ?sku)
        )
      )
