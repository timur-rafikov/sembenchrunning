  (:requirements :strips :typing :negative-preconditions)
  (:types operational_resource - object document_resource - object checkpoint_resource_group - object case_category - object entity_record - case_category investigation_resource - operational_resource tracking_source - operational_resource carrier_contact - operational_resource escalation_template - operational_resource authorization_credential - operational_resource loss_cause_code - operational_resource insurance_policy - operational_resource external_authority - operational_resource contingency_option - document_resource evidence_document - document_resource claim_reference - document_resource origin_checkpoint - checkpoint_resource_group destination_checkpoint - checkpoint_resource_group recovery_shipment - checkpoint_resource_group logistics_actor - entity_record support_party_group - entity_record origin_logistics_actor - logistics_actor destination_logistics_actor - logistics_actor operations_center - support_party_group)
  (:predicates
    (shipment_case_opened ?shipment_case - entity_record)
    (entity_validated ?shipment_case - entity_record)
    (entity_investigation_assigned ?shipment_case - entity_record)
    (recovery_action_initiated ?shipment_case - entity_record)
    (entity_closure_recorded ?shipment_case - entity_record)
    (entity_claim_initiated ?shipment_case - entity_record)
    (investigation_resource_available ?investigation_resource - investigation_resource)
    (entity_assigned_investigation_resource ?shipment_case - entity_record ?investigation_resource - investigation_resource)
    (tracking_source_available ?tracking_source - tracking_source)
    (entity_has_tracking_source ?shipment_case - entity_record ?tracking_source - tracking_source)
    (carrier_contact_available ?carrier_contact - carrier_contact)
    (entity_assigned_carrier_contact ?shipment_case - entity_record ?carrier_contact - carrier_contact)
    (contingency_option_available ?contingency_option - contingency_option)
    (origin_actor_assigned_contingency_option ?origin_actor - origin_logistics_actor ?contingency_option - contingency_option)
    (destination_actor_assigned_contingency_option ?destination_actor - destination_logistics_actor ?contingency_option - contingency_option)
    (origin_actor_has_checkpoint ?origin_actor - origin_logistics_actor ?origin_checkpoint - origin_checkpoint)
    (origin_checkpoint_verified ?origin_checkpoint - origin_checkpoint)
    (origin_checkpoint_flagged_for_contingency ?origin_checkpoint - origin_checkpoint)
    (origin_actor_recovery_ready ?origin_actor - origin_logistics_actor)
    (destination_actor_has_checkpoint ?destination_actor - destination_logistics_actor ?destination_checkpoint - destination_checkpoint)
    (destination_checkpoint_verified ?destination_checkpoint - destination_checkpoint)
    (destination_checkpoint_flagged_for_contingency ?destination_checkpoint - destination_checkpoint)
    (destination_actor_recovery_ready ?destination_actor - destination_logistics_actor)
    (recovery_shipment_pending ?recovery_shipment - recovery_shipment)
    (recovery_shipment_created ?recovery_shipment - recovery_shipment)
    (recovery_shipment_assigned_origin_checkpoint ?recovery_shipment - recovery_shipment ?origin_checkpoint - origin_checkpoint)
    (recovery_shipment_assigned_destination_checkpoint ?recovery_shipment - recovery_shipment ?destination_checkpoint - destination_checkpoint)
    (recovery_shipment_origin_requires_authorization ?recovery_shipment - recovery_shipment)
    (recovery_shipment_destination_requires_authorization ?recovery_shipment - recovery_shipment)
    (recovery_shipment_origin_checkin_complete ?recovery_shipment - recovery_shipment)
    (operations_center_manages_origin_actor ?operations_center - operations_center ?origin_actor - origin_logistics_actor)
    (operations_center_manages_destination_actor ?operations_center - operations_center ?destination_actor - destination_logistics_actor)
    (operations_center_has_recovery_shipment ?operations_center - operations_center ?recovery_shipment - recovery_shipment)
    (evidence_document_available ?evidence_document - evidence_document)
    (operations_center_has_evidence_document ?operations_center - operations_center ?evidence_document - evidence_document)
    (evidence_document_attached ?evidence_document - evidence_document)
    (evidence_document_linked_to_recovery_shipment ?evidence_document - evidence_document ?recovery_shipment - recovery_shipment)
    (operations_center_documentation_initiated ?operations_center - operations_center)
    (operations_center_documents_reviewed ?operations_center - operations_center)
    (operations_center_finalization_pending ?operations_center - operations_center)
    (operations_center_has_escalation_template ?operations_center - operations_center)
    (operations_center_escalation_triggered ?operations_center - operations_center)
    (operations_center_authorizations_collected ?operations_center - operations_center)
    (operations_center_approval_recorded ?operations_center - operations_center)
    (claim_reference_available ?claim_reference - claim_reference)
    (operations_center_has_claim_reference ?operations_center - operations_center ?claim_reference - claim_reference)
    (operations_center_claim_registered ?operations_center - operations_center)
    (operations_center_claim_process_started ?operations_center - operations_center)
    (operations_center_claim_authorized ?operations_center - operations_center)
    (escalation_template_available ?escalation_template - escalation_template)
    (operations_center_assigned_escalation_template ?operations_center - operations_center ?escalation_template - escalation_template)

    (operations_center_has_authorization_credential ?operations_center - operations_center ?authorization_credential - authorization_credential)
    (insurance_policy_available ?insurance_policy - insurance_policy)
    (operations_center_has_insurance_policy ?operations_center - operations_center ?insurance_policy - insurance_policy)
    (external_authority_available ?external_authority - external_authority)
    (operations_center_has_external_authority_contact ?operations_center - operations_center ?external_authority - external_authority)
    (loss_cause_code_available ?loss_cause_code - loss_cause_code)
    (entity_has_loss_cause_code ?shipment_case - entity_record ?loss_cause_code - loss_cause_code)

    (destination_actor_activated ?destination_actor - destination_logistics_actor)
    (operations_center_case_closed ?operations_center - operations_center)
  )
  (:action open_shipment_case
    :parameters (?shipment_case - entity_record)
    :precondition
      (and
        (not
          (shipment_case_opened ?shipment_case)
        )
        (not
          (recovery_action_initiated ?shipment_case)
        )
      )
    :effect (shipment_case_opened ?shipment_case)
  )
  (:action assign_investigation_resource
    :parameters (?shipment_case - entity_record ?investigation_resource - investigation_resource)
    :precondition
      (and
        (shipment_case_opened ?shipment_case)
        (not
          (entity_investigation_assigned ?shipment_case)
        )
        (investigation_resource_available ?investigation_resource)
      )
    :effect
      (and
        (entity_investigation_assigned ?shipment_case)
        (entity_assigned_investigation_resource ?shipment_case ?investigation_resource)
        (not
          (investigation_resource_available ?investigation_resource)
        )
      )
  )
  (:action attach_tracking_source_to_case
    :parameters (?shipment_case - entity_record ?tracking_source - tracking_source)
    :precondition
      (and
        (shipment_case_opened ?shipment_case)
        (entity_investigation_assigned ?shipment_case)
        (tracking_source_available ?tracking_source)
      )
    :effect
      (and
        (entity_has_tracking_source ?shipment_case ?tracking_source)
        (not
          (tracking_source_available ?tracking_source)
        )
      )
  )
  (:action confirm_tracking_evidence_for_case
    :parameters (?shipment_case - entity_record ?tracking_source - tracking_source)
    :precondition
      (and
        (shipment_case_opened ?shipment_case)
        (entity_investigation_assigned ?shipment_case)
        (entity_has_tracking_source ?shipment_case ?tracking_source)
        (not
          (entity_validated ?shipment_case)
        )
      )
    :effect (entity_validated ?shipment_case)
  )
  (:action release_tracking_source
    :parameters (?shipment_case - entity_record ?tracking_source - tracking_source)
    :precondition
      (and
        (entity_has_tracking_source ?shipment_case ?tracking_source)
      )
    :effect
      (and
        (tracking_source_available ?tracking_source)
        (not
          (entity_has_tracking_source ?shipment_case ?tracking_source)
        )
      )
  )
  (:action assign_carrier_contact_to_case
    :parameters (?shipment_case - entity_record ?carrier_contact - carrier_contact)
    :precondition
      (and
        (entity_validated ?shipment_case)
        (carrier_contact_available ?carrier_contact)
      )
    :effect
      (and
        (entity_assigned_carrier_contact ?shipment_case ?carrier_contact)
        (not
          (carrier_contact_available ?carrier_contact)
        )
      )
  )
  (:action release_carrier_contact_from_case
    :parameters (?shipment_case - entity_record ?carrier_contact - carrier_contact)
    :precondition
      (and
        (entity_assigned_carrier_contact ?shipment_case ?carrier_contact)
      )
    :effect
      (and
        (carrier_contact_available ?carrier_contact)
        (not
          (entity_assigned_carrier_contact ?shipment_case ?carrier_contact)
        )
      )
  )
  (:action assign_insurance_policy_to_ops_center
    :parameters (?operations_center - operations_center ?insurance_policy - insurance_policy)
    :precondition
      (and
        (entity_validated ?operations_center)
        (insurance_policy_available ?insurance_policy)

    :parameters (?operations_center - operations_center ?insurance_policy - insurance_policy)

    :effect

      )

  (:action assign_external_authority_contact

      )
    :effect
      (and
        (operations_center_has_external_authority_contact ?operations_center ?external_authority)
        (not
          (external_authority_available ?external_authority)
        )
      )
  )
  (:action release_external_authority_contact
    :parameters (?operations_center - operations_center ?external_authority - external_authority)
    :precondition
      (and
        (operations_center_has_external_authority_contact ?operations_center ?external_authority)
      )
    :effect
      (and
        (external_authority_available ?external_authority)
        (not
          (operations_center_has_external_authority_contact ?operations_center ?external_authority)
        )
      )
  )
  (:action verify_origin_checkpoint_with_tracking
    :parameters (?origin_actor - origin_logistics_actor ?origin_checkpoint - origin_checkpoint ?tracking_source - tracking_source)
    :precondition
      (and
        (entity_validated ?origin_actor)
        (entity_has_tracking_source ?origin_actor ?tracking_source)
        (origin_actor_has_checkpoint ?origin_actor ?origin_checkpoint)
        (not
          (origin_checkpoint_verified ?origin_checkpoint)
        )
        (not
          (origin_checkpoint_flagged_for_contingency ?origin_checkpoint)
        )
      )
    :effect (origin_checkpoint_verified ?origin_checkpoint)
  )
  (:action acknowledge_origin_recovery_readiness
    :parameters (?origin_actor - origin_logistics_actor ?origin_checkpoint - origin_checkpoint ?carrier_contact - carrier_contact)
    :precondition
      (and
        (entity_validated ?origin_actor)
        (entity_assigned_carrier_contact ?origin_actor ?carrier_contact)
        (origin_actor_has_checkpoint ?origin_actor ?origin_checkpoint)
        (origin_checkpoint_verified ?origin_checkpoint)
        (not
          (origin_actor_activated ?origin_actor)
        )
      )
    :effect
      (and
        (origin_actor_activated ?origin_actor)
        (origin_actor_recovery_ready ?origin_actor)
      )
  )
  (:action assign_contingency_option_to_origin_actor
    :parameters (?origin_actor - origin_logistics_actor ?origin_checkpoint - origin_checkpoint ?contingency_option - contingency_option)
    :precondition
      (and
        (entity_validated ?origin_actor)
        (origin_actor_has_checkpoint ?origin_actor ?origin_checkpoint)
        (contingency_option_available ?contingency_option)
        (not
          (origin_actor_activated ?origin_actor)
        )
      )
    :effect
      (and
        (origin_checkpoint_flagged_for_contingency ?origin_checkpoint)
        (origin_actor_activated ?origin_actor)
        (origin_actor_assigned_contingency_option ?origin_actor ?contingency_option)
        (not
          (contingency_option_available ?contingency_option)
        )
      )
  )
  (:action finalize_origin_checkpoint_contingency
    :parameters (?origin_actor - origin_logistics_actor ?origin_checkpoint - origin_checkpoint ?tracking_source - tracking_source ?contingency_option - contingency_option)

        (origin_actor_assigned_contingency_option ?origin_actor ?contingency_option)

          (origin_actor_recovery_ready ?origin_actor)

    :effect
      (and
        (origin_checkpoint_verified ?origin_checkpoint)
        (origin_actor_recovery_ready ?origin_actor)
        (contingency_option_available ?contingency_option)
        (not
          (origin_actor_assigned_contingency_option ?origin_actor ?contingency_option)
        )
      )
  )
  (:action verify_destination_checkpoint_with_tracking

  (:action assign_contingency_option_to_destination_actor
    :parameters (?destination_actor - destination_logistics_actor ?destination_checkpoint - destination_checkpoint ?contingency_option - contingency_option)
    :precondition
      (and
        (entity_validated ?destination_actor)
        (destination_actor_has_checkpoint ?destination_actor ?destination_checkpoint)
        (contingency_option_available ?contingency_option)
        (not
          (destination_actor_activated ?destination_actor)
        )
      )
    :effect
      (and
        (destination_checkpoint_flagged_for_contingency ?destination_checkpoint)
        (destination_actor_activated ?destination_actor)
        (destination_actor_assigned_contingency_option ?destination_actor ?contingency_option)

          (contingency_option_available ?contingency_option)

  (:action finalize_destination_checkpoint_contingency
    :parameters (?destination_actor - destination_logistics_actor ?destination_checkpoint - destination_checkpoint ?tracking_source - tracking_source ?contingency_option - contingency_option)
    :precondition
      (and

        (destination_actor_has_checkpoint ?destination_actor ?destination_checkpoint)
        (destination_checkpoint_flagged_for_contingency ?destination_checkpoint)
        (destination_actor_assigned_contingency_option ?destination_actor ?contingency_option)
        (not
          (destination_actor_recovery_ready ?destination_actor)

    :effect
      (and
        (destination_checkpoint_verified ?destination_checkpoint)
        (destination_actor_recovery_ready ?destination_actor)
        (contingency_option_available ?contingency_option)

          (destination_actor_assigned_contingency_option ?destination_actor ?contingency_option)

  (:action instantiate_recovery_shipment_standard
    :parameters (?origin_actor - origin_logistics_actor ?destination_actor - destination_logistics_actor ?origin_checkpoint - origin_checkpoint ?destination_checkpoint - destination_checkpoint ?recovery_shipment - recovery_shipment)
    :precondition
      (and
        (origin_actor_activated ?origin_actor)
        (destination_actor_activated ?destination_actor)
        (origin_actor_has_checkpoint ?origin_actor ?origin_checkpoint)
        (destination_actor_has_checkpoint ?destination_actor ?destination_checkpoint)
        (origin_checkpoint_verified ?origin_checkpoint)
        (destination_checkpoint_verified ?destination_checkpoint)
        (origin_actor_recovery_ready ?origin_actor)
        (destination_actor_recovery_ready ?destination_actor)
        (recovery_shipment_pending ?recovery_shipment)
      )
    :effect
      (and
        (recovery_shipment_created ?recovery_shipment)
        (recovery_shipment_assigned_origin_checkpoint ?recovery_shipment ?origin_checkpoint)
        (recovery_shipment_assigned_destination_checkpoint ?recovery_shipment ?destination_checkpoint)
        (not
          (recovery_shipment_pending ?recovery_shipment)
        )
      )
  )
  (:action instantiate_recovery_shipment_origin_requires_authorization
    :parameters (?origin_actor - origin_logistics_actor ?destination_actor - destination_logistics_actor ?origin_checkpoint - origin_checkpoint ?destination_checkpoint - destination_checkpoint ?recovery_shipment - recovery_shipment)
    :precondition
      (and
        (origin_actor_activated ?origin_actor)
        (destination_actor_activated ?destination_actor)
        (origin_actor_has_checkpoint ?origin_actor ?origin_checkpoint)
        (destination_actor_has_checkpoint ?destination_actor ?destination_checkpoint)
        (origin_checkpoint_flagged_for_contingency ?origin_checkpoint)
        (destination_checkpoint_verified ?destination_checkpoint)
        (not
          (origin_actor_recovery_ready ?origin_actor)
        )
        (destination_actor_recovery_ready ?destination_actor)
        (recovery_shipment_pending ?recovery_shipment)
      )
    :effect
      (and
        (recovery_shipment_created ?recovery_shipment)
        (recovery_shipment_assigned_origin_checkpoint ?recovery_shipment ?origin_checkpoint)
        (recovery_shipment_assigned_destination_checkpoint ?recovery_shipment ?destination_checkpoint)
        (recovery_shipment_origin_requires_authorization ?recovery_shipment)
        (not
          (recovery_shipment_pending ?recovery_shipment)
        )
      )
  )
  (:action instantiate_recovery_shipment_destination_requires_authorization
    :parameters (?origin_actor - origin_logistics_actor ?destination_actor - destination_logistics_actor ?origin_checkpoint - origin_checkpoint ?destination_checkpoint - destination_checkpoint ?recovery_shipment - recovery_shipment)
    :precondition
      (and
        (origin_actor_activated ?origin_actor)
        (destination_actor_activated ?destination_actor)
        (origin_actor_has_checkpoint ?origin_actor ?origin_checkpoint)
        (destination_actor_has_checkpoint ?destination_actor ?destination_checkpoint)
        (origin_checkpoint_verified ?origin_checkpoint)
        (destination_checkpoint_flagged_for_contingency ?destination_checkpoint)
        (origin_actor_recovery_ready ?origin_actor)
        (not
          (destination_actor_recovery_ready ?destination_actor)
        )
        (recovery_shipment_pending ?recovery_shipment)
      )
    :effect
      (and
        (recovery_shipment_created ?recovery_shipment)
        (recovery_shipment_assigned_origin_checkpoint ?recovery_shipment ?origin_checkpoint)
        (recovery_shipment_assigned_destination_checkpoint ?recovery_shipment ?destination_checkpoint)
        (recovery_shipment_destination_requires_authorization ?recovery_shipment)
        (not
          (recovery_shipment_pending ?recovery_shipment)
        )
      )
  )
  (:action instantiate_recovery_shipment_both_require_authorization
    :parameters (?origin_actor - origin_logistics_actor ?destination_actor - destination_logistics_actor ?origin_checkpoint - origin_checkpoint ?destination_checkpoint - destination_checkpoint ?recovery_shipment - recovery_shipment)

      (and
        (origin_actor_activated ?origin_actor)
        (destination_actor_activated ?destination_actor)

        (destination_actor_has_checkpoint ?destination_actor ?destination_checkpoint)
        (origin_checkpoint_flagged_for_contingency ?origin_checkpoint)
        (destination_checkpoint_flagged_for_contingency ?destination_checkpoint)
        (not
          (origin_actor_recovery_ready ?origin_actor)

          (destination_actor_recovery_ready ?destination_actor)

      (and
        (recovery_shipment_created ?recovery_shipment)

        (recovery_shipment_destination_requires_authorization ?recovery_shipment)
        (not
          (recovery_shipment_pending ?recovery_shipment)
        )
      )
  )
  (:action checkin_recovery_shipment_at_origin
    :parameters (?recovery_shipment - recovery_shipment ?origin_actor - origin_logistics_actor ?tracking_source - tracking_source)

      (and
        (recovery_shipment_created ?recovery_shipment)

        (not
          (recovery_shipment_origin_checkin_complete ?recovery_shipment)

  )
  (:action attach_evidence_to_recovery_shipment

      (and

        (not
          (evidence_document_attached ?evidence_document)

      )
    :effect
      (and
        (evidence_document_attached ?evidence_document)
        (evidence_document_linked_to_recovery_shipment ?evidence_document ?recovery_shipment)
        (not
          (evidence_document_available ?evidence_document)
        )
      )
  )
  (:action initiate_documentation_preparation

      (and
        (entity_validated ?operations_center)
        (operations_center_has_evidence_document ?operations_center ?evidence_document)
        (evidence_document_attached ?evidence_document)
        (evidence_document_linked_to_recovery_shipment ?evidence_document ?recovery_shipment)
        (entity_has_tracking_source ?operations_center ?tracking_source)
        (not
          (recovery_shipment_origin_requires_authorization ?recovery_shipment)
        )
        (not
          (operations_center_documentation_initiated ?operations_center)
        )
      )
    :effect (operations_center_documentation_initiated ?operations_center)
  )
  (:action assign_escalation_template_to_ops_center
    :parameters (?operations_center - operations_center ?escalation_template - escalation_template)
    :precondition
      (and
        (entity_validated ?operations_center)
        (escalation_template_available ?escalation_template)
        (not
          (operations_center_has_escalation_template ?operations_center)
        )
      )
    :effect
      (and
        (operations_center_has_escalation_template ?operations_center)
        (operations_center_assigned_escalation_template ?operations_center ?escalation_template)
        (not
          (escalation_template_available ?escalation_template)
        )
      )
  )
  (:action prepare_documentation_and_assign_escalation_template
    :parameters (?operations_center - operations_center ?evidence_document - evidence_document ?recovery_shipment - recovery_shipment ?tracking_source - tracking_source ?escalation_template - escalation_template)
    :precondition
      (and
        (entity_validated ?operations_center)
