  (:action initialize_rental_case
    :parameters (?rental_case - rental_case)
    :precondition
      (and
        (not
          (entity_created ?rental_case)
        )
        (not
          (eligibility_assigned ?rental_case)
        )
      )
    :effect (entity_created ?rental_case)
  )
  (:action select_and_reserve_deposit_card
    :parameters (?rental_case - rental_case ?deposit_card_type - deposit_card_type)
    :precondition
      (and
        (entity_created ?rental_case)
        (not
          (card_selected ?rental_case)
        )
        (deposit_card_type_available ?deposit_card_type)
      )
    :effect
      (and
        (card_selected ?rental_case)
        (entity_reserved_card ?rental_case ?deposit_card_type)
        (not
          (deposit_card_type_available ?deposit_card_type)
        )
      )
  )
  (:action attach_document_to_case
    :parameters (?rental_case - rental_case ?document - document)
    :precondition
      (and
        (entity_created ?rental_case)
        (card_selected ?rental_case)
        (document_available ?document)
      )
    :effect
      (and
        (document_attached_to_entity ?rental_case ?document)
        (not
          (document_available ?document)
        )
      )
  )
  (:action validate_attached_document
    :parameters (?rental_case - rental_case ?document - document)
    :precondition
      (and
        (entity_created ?rental_case)
        (card_selected ?rental_case)
        (document_attached_to_entity ?rental_case ?document)
        (not
          (entity_active ?rental_case)
        )
      )
    :effect (entity_active ?rental_case)
  )
  (:action detach_document_from_case
    :parameters (?rental_case - rental_case ?document - document)
    :precondition
      (and
        (document_attached_to_entity ?rental_case ?document)
      )
    :effect
      (and
        (document_available ?document)
        (not
          (document_attached_to_entity ?rental_case ?document)
        )
      )
  )
  (:action link_policy_rule_to_case
    :parameters (?rental_case - rental_case ?policy_rule - policy_rule)
    :precondition
      (and
        (entity_active ?rental_case)
        (policy_rule_available ?policy_rule)
      )
    :effect
      (and
        (rule_linked_to_entity ?rental_case ?policy_rule)
        (not
          (policy_rule_available ?policy_rule)
        )
      )
  )
  (:action unlink_policy_rule_from_case
    :parameters (?rental_case - rental_case ?policy_rule - policy_rule)
    :precondition
      (and
        (rule_linked_to_entity ?rental_case ?policy_rule)
      )
    :effect
      (and
        (policy_rule_available ?policy_rule)
        (not
          (rule_linked_to_entity ?rental_case ?policy_rule)
        )
      )
  )
  (:action assign_payment_method_to_processor

        (processor_linked_temporal_constraint ?eligibility_processor ?temporal_constraint)
      )

        (temporal_constraint_available ?temporal_constraint)
        (not
          (processor_linked_temporal_constraint ?eligibility_processor ?temporal_constraint)

    :parameters (?primary_applicant_profile - primary_applicant_profile ?applicant_jurisdiction - applicant_jurisdiction ?document - document)

        (document_attached_to_entity ?primary_applicant_profile ?document)

        (not
          (jurisdiction_marked ?applicant_jurisdiction)

          (jurisdiction_evidence_flag ?applicant_jurisdiction)

  )
  (:action complete_primary_profile_verification
    :parameters (?primary_applicant_profile - primary_applicant_profile ?applicant_jurisdiction - applicant_jurisdiction ?policy_rule - policy_rule)
    :precondition
      (and
        (entity_active ?primary_applicant_profile)
        (rule_linked_to_entity ?primary_applicant_profile ?policy_rule)
        (primary_profile_jurisdiction ?primary_applicant_profile ?applicant_jurisdiction)
        (jurisdiction_marked ?applicant_jurisdiction)
        (not
          (primary_profile_ready ?primary_applicant_profile)

    :effect
      (and
        (primary_profile_ready ?primary_applicant_profile)
        (primary_profile_verification_complete ?primary_applicant_profile)

    :parameters (?primary_applicant_profile - primary_applicant_profile ?applicant_jurisdiction - applicant_jurisdiction ?supporting_evidence - supporting_evidence)

          (primary_profile_ready ?primary_applicant_profile)

        (jurisdiction_evidence_flag ?applicant_jurisdiction)

          (evidence_available ?supporting_evidence)

          (primary_profile_attached_evidence ?primary_applicant_profile ?supporting_evidence)
