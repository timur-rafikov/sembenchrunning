(define (domain insufficient_level_or_power_rebuild_plan)
  (:requirements :strips :typing :negative-preconditions)
  (:types item_family - object resource_group - object location_group - object deployable_category - object deployable_entity - deployable_category loadout_marker - item_family upgrade_module - item_family equipment_slot - item_family configuration_tag - item_family strategy_modifier - item_family repair_material - item_family special_attachment - item_family special_module - item_family consumable_resource - resource_group upgrade_state - resource_group communication_marker - resource_group mission_node_a - location_group mission_node_b - location_group mission_asset - location_group combat_agent_role - deployable_entity controller_role_group - deployable_entity player_agent - combat_agent_role npc_agent - combat_agent_role controller - controller_role_group)
  (:predicates
    (assessed_for_redeployment ?deployable_entity - deployable_entity)
    (configuration_validated ?deployable_entity - deployable_entity)
    (entity_loadout_flag ?deployable_entity - deployable_entity)
    (redeployed ?deployable_entity - deployable_entity)
    (rebuild_commit_registered ?deployable_entity - deployable_entity)
    (rebuild_commit_confirmed ?deployable_entity - deployable_entity)
    (loadout_marker_available ?loadout_marker - loadout_marker)
    (loadout_assigned ?deployable_entity - deployable_entity ?loadout_marker - loadout_marker)
    (upgrade_module_available ?upgrade_module - upgrade_module)
    (upgrade_module_attached ?deployable_entity - deployable_entity ?upgrade_module - upgrade_module)
    (equipment_slot_available ?equipment_slot - equipment_slot)
    (equipment_slot_bound ?deployable_entity - deployable_entity ?equipment_slot - equipment_slot)
    (consumable_available ?consumable_resource - consumable_resource)
    (player_consumable_reserved ?player_agent - player_agent ?consumable_resource - consumable_resource)
    (npc_consumable_reserved ?npc_agent - npc_agent ?consumable_resource - consumable_resource)
    (mission_node_a_assigned ?player_agent - player_agent ?mission_node_a - mission_node_a)
    (mission_node_a_primary_ready ?mission_node_a - mission_node_a)
    (mission_node_a_secondary_ready ?mission_node_a - mission_node_a)
    (player_node_a_ready ?player_agent - player_agent)
    (mission_node_b_assigned ?npc_agent - npc_agent ?mission_node_b - mission_node_b)
    (mission_node_b_primary_ready ?mission_node_b - mission_node_b)
    (mission_node_b_secondary_ready ?mission_node_b - mission_node_b)
    (npc_node_b_ready ?npc_agent - npc_agent)
    (mission_asset_available ?mission_asset - mission_asset)
    (mission_asset_staged ?mission_asset - mission_asset)
    (asset_includes_node_a ?mission_asset - mission_asset ?mission_node_a - mission_node_a)
    (asset_includes_node_b ?mission_asset - mission_asset ?mission_node_b - mission_node_b)
    (mission_asset_outcome_primary ?mission_asset - mission_asset)
    (mission_asset_outcome_secondary ?mission_asset - mission_asset)
    (mission_asset_validated ?mission_asset - mission_asset)
    (controller_assigned_player ?recovery_agent - controller ?player_agent - player_agent)
    (controller_assigned_npc ?recovery_agent - controller ?npc_agent - npc_agent)
    (controller_has_mission_asset ?recovery_agent - controller ?mission_asset - mission_asset)
    (upgrade_state_available ?upgrade_state - upgrade_state)
    (controller_has_upgrade_state ?recovery_agent - controller ?upgrade_state - upgrade_state)
    (upgrade_state_initialized ?upgrade_state - upgrade_state)
    (upgrade_state_applied_to_asset ?upgrade_state - upgrade_state ?mission_asset - mission_asset)
    (controller_primed_for_activation ?recovery_agent - controller)
    (controller_activation_engaged ?recovery_agent - controller)
    (controller_activation_finalized ?recovery_agent - controller)
    (controller_config_tag_present ?recovery_agent - controller)
    (controller_config_tag_applied ?recovery_agent - controller)
    (controller_strategy_modifier_attached ?recovery_agent - controller)
    (controller_team_ready ?recovery_agent - controller)
    (communication_marker_available ?communication_marker - communication_marker)
    (controller_has_communication_marker ?recovery_agent - controller ?communication_marker - communication_marker)
    (controller_comm_configured ?recovery_agent - controller)
    (controller_comm_slot_open ?recovery_agent - controller)
    (controller_comm_activated ?recovery_agent - controller)
    (configuration_tag_available ?configuration_tag - configuration_tag)
    (controller_configuration_tag_assigned ?recovery_agent - controller ?configuration_tag - configuration_tag)
    (strategy_modifier_available ?strategy_modifier - strategy_modifier)
    (controller_strategy_modifier_assigned ?recovery_agent - controller ?strategy_modifier - strategy_modifier)
    (special_attachment_available ?special_attachment - special_attachment)
    (controller_has_special_attachment ?recovery_agent - controller ?special_attachment - special_attachment)
    (special_module_available ?special_module - special_module)
    (controller_has_special_module ?recovery_agent - controller ?special_module - special_module)
    (repair_material_available ?repair_material - repair_material)
    (repair_material_assigned ?deployable_entity - deployable_entity ?repair_material - repair_material)
    (player_ready_for_asset ?player_agent - player_agent)
    (npc_ready_for_asset ?npc_agent - npc_agent)
    (rebuild_commit_marker_set ?recovery_agent - controller)
  )
  (:action initialize_assessment
    :parameters (?deployable_entity - deployable_entity)
    :precondition
      (and
        (not
          (assessed_for_redeployment ?deployable_entity)
        )
        (not
          (redeployed ?deployable_entity)
        )
      )
    :effect (assessed_for_redeployment ?deployable_entity)
  )
  (:action assign_loadout_marker
    :parameters (?deployable_entity - deployable_entity ?loadout_marker - loadout_marker)
    :precondition
      (and
        (assessed_for_redeployment ?deployable_entity)
        (not
          (entity_loadout_flag ?deployable_entity)
        )
        (loadout_marker_available ?loadout_marker)
      )
    :effect
      (and
        (entity_loadout_flag ?deployable_entity)
        (loadout_assigned ?deployable_entity ?loadout_marker)
        (not
          (loadout_marker_available ?loadout_marker)
        )
      )
  )
  (:action attach_upgrade_module
    :parameters (?deployable_entity - deployable_entity ?upgrade_module - upgrade_module)
    :precondition
      (and
        (assessed_for_redeployment ?deployable_entity)
        (entity_loadout_flag ?deployable_entity)
        (upgrade_module_available ?upgrade_module)
      )
    :effect
      (and
        (upgrade_module_attached ?deployable_entity ?upgrade_module)
        (not
          (upgrade_module_available ?upgrade_module)
        )
      )
  )
  (:action confirm_module_attachment
    :parameters (?deployable_entity - deployable_entity ?upgrade_module - upgrade_module)
    :precondition
      (and
        (assessed_for_redeployment ?deployable_entity)
        (entity_loadout_flag ?deployable_entity)
        (upgrade_module_attached ?deployable_entity ?upgrade_module)
        (not
          (configuration_validated ?deployable_entity)
        )
      )
    :effect (configuration_validated ?deployable_entity)
  )
  (:action detach_upgrade_module
    :parameters (?deployable_entity - deployable_entity ?upgrade_module - upgrade_module)
    :precondition
      (and
        (upgrade_module_attached ?deployable_entity ?upgrade_module)
      )
    :effect
      (and
        (upgrade_module_available ?upgrade_module)
        (not
          (upgrade_module_attached ?deployable_entity ?upgrade_module)
        )
      )
  )
  (:action bind_equipment_slot
    :parameters (?deployable_entity - deployable_entity ?equipment_slot - equipment_slot)
    :precondition
      (and
        (configuration_validated ?deployable_entity)
        (equipment_slot_available ?equipment_slot)
      )
    :effect
      (and
        (equipment_slot_bound ?deployable_entity ?equipment_slot)
        (not
          (equipment_slot_available ?equipment_slot)
        )
      )
  )
  (:action release_equipment_slot
    :parameters (?deployable_entity - deployable_entity ?equipment_slot - equipment_slot)
    :precondition
      (and
        (equipment_slot_bound ?deployable_entity ?equipment_slot)
      )
    :effect
      (and
        (equipment_slot_available ?equipment_slot)
        (not
          (equipment_slot_bound ?deployable_entity ?equipment_slot)
        )
      )
  )
  (:action attach_special_attachment
    :parameters (?recovery_agent - controller ?special_attachment - special_attachment)
    :precondition
      (and
        (configuration_validated ?recovery_agent)
        (special_attachment_available ?special_attachment)
      )
    :effect
      (and
        (controller_has_special_attachment ?recovery_agent ?special_attachment)
        (not
          (special_attachment_available ?special_attachment)
        )
      )
  )
  (:action detach_special_attachment
    :parameters (?recovery_agent - controller ?special_attachment - special_attachment)
    :precondition
      (and
        (controller_has_special_attachment ?recovery_agent ?special_attachment)
      )
    :effect
      (and
        (special_attachment_available ?special_attachment)
        (not
          (controller_has_special_attachment ?recovery_agent ?special_attachment)
        )
      )
  )
  (:action attach_special_module
    :parameters (?recovery_agent - controller ?special_module - special_module)
    :precondition
      (and
        (configuration_validated ?recovery_agent)
        (special_module_available ?special_module)
      )
    :effect
      (and
        (controller_has_special_module ?recovery_agent ?special_module)
        (not
          (special_module_available ?special_module)
        )
      )
  )
  (:action detach_special_module
    :parameters (?recovery_agent - controller ?special_module - special_module)
    :precondition
      (and
        (controller_has_special_module ?recovery_agent ?special_module)
      )
    :effect
      (and
        (special_module_available ?special_module)
        (not
          (controller_has_special_module ?recovery_agent ?special_module)
        )
      )
  )
  (:action reserve_and_prepare_node_a
    :parameters (?player_agent - player_agent ?mission_node_a - mission_node_a ?upgrade_module - upgrade_module)
    :precondition
      (and
        (configuration_validated ?player_agent)
        (upgrade_module_attached ?player_agent ?upgrade_module)
        (mission_node_a_assigned ?player_agent ?mission_node_a)
        (not
          (mission_node_a_primary_ready ?mission_node_a)
        )
        (not
          (mission_node_a_secondary_ready ?mission_node_a)
        )
      )
    :effect (mission_node_a_primary_ready ?mission_node_a)
  )
  (:action apply_equipment_slot_to_node_a
    :parameters (?player_agent - player_agent ?mission_node_a - mission_node_a ?equipment_slot - equipment_slot)
    :precondition
      (and
        (configuration_validated ?player_agent)
        (equipment_slot_bound ?player_agent ?equipment_slot)
        (mission_node_a_assigned ?player_agent ?mission_node_a)
        (mission_node_a_primary_ready ?mission_node_a)
        (not
          (player_ready_for_asset ?player_agent)
        )
      )
    :effect
      (and
        (player_ready_for_asset ?player_agent)
        (player_node_a_ready ?player_agent)
      )
  )
  (:action consume_resource_for_node_a
    :parameters (?player_agent - player_agent ?mission_node_a - mission_node_a ?consumable_resource - consumable_resource)
    :precondition
      (and
        (configuration_validated ?player_agent)
        (mission_node_a_assigned ?player_agent ?mission_node_a)
        (consumable_available ?consumable_resource)
        (not
          (player_ready_for_asset ?player_agent)
        )
      )
    :effect
      (and
        (mission_node_a_secondary_ready ?mission_node_a)
        (player_ready_for_asset ?player_agent)
        (player_consumable_reserved ?player_agent ?consumable_resource)
        (not
          (consumable_available ?consumable_resource)
        )
      )
  )
  (:action finalize_node_a_preparation
    :parameters (?player_agent - player_agent ?mission_node_a - mission_node_a ?upgrade_module - upgrade_module ?consumable_resource - consumable_resource)
    :precondition
      (and
        (configuration_validated ?player_agent)
        (upgrade_module_attached ?player_agent ?upgrade_module)
        (mission_node_a_assigned ?player_agent ?mission_node_a)
        (mission_node_a_secondary_ready ?mission_node_a)
        (player_consumable_reserved ?player_agent ?consumable_resource)
        (not
          (player_node_a_ready ?player_agent)
        )
      )
    :effect
      (and
        (mission_node_a_primary_ready ?mission_node_a)
        (player_node_a_ready ?player_agent)
        (consumable_available ?consumable_resource)
        (not
          (player_consumable_reserved ?player_agent ?consumable_resource)
        )
      )
  )
  (:action reserve_and_prepare_node_b
    :parameters (?npc_agent - npc_agent ?mission_node_b - mission_node_b ?upgrade_module - upgrade_module)
    :precondition
      (and
        (configuration_validated ?npc_agent)
        (upgrade_module_attached ?npc_agent ?upgrade_module)
        (mission_node_b_assigned ?npc_agent ?mission_node_b)
        (not
          (mission_node_b_primary_ready ?mission_node_b)
        )
        (not
          (mission_node_b_secondary_ready ?mission_node_b)
        )
      )
    :effect (mission_node_b_primary_ready ?mission_node_b)
  )
  (:action apply_equipment_slot_to_node_b
    :parameters (?npc_agent - npc_agent ?mission_node_b - mission_node_b ?equipment_slot - equipment_slot)
    :precondition
      (and
        (configuration_validated ?npc_agent)
        (equipment_slot_bound ?npc_agent ?equipment_slot)
        (mission_node_b_assigned ?npc_agent ?mission_node_b)
        (mission_node_b_primary_ready ?mission_node_b)
        (not
          (npc_ready_for_asset ?npc_agent)
        )
      )
    :effect
      (and
        (npc_ready_for_asset ?npc_agent)
        (npc_node_b_ready ?npc_agent)
      )
  )
  (:action consume_resource_for_node_b
    :parameters (?npc_agent - npc_agent ?mission_node_b - mission_node_b ?consumable_resource - consumable_resource)
    :precondition
      (and
        (configuration_validated ?npc_agent)
        (mission_node_b_assigned ?npc_agent ?mission_node_b)
        (consumable_available ?consumable_resource)
        (not
          (npc_ready_for_asset ?npc_agent)
        )
      )
    :effect
      (and
        (mission_node_b_secondary_ready ?mission_node_b)
        (npc_ready_for_asset ?npc_agent)
        (npc_consumable_reserved ?npc_agent ?consumable_resource)
        (not
          (consumable_available ?consumable_resource)
        )
      )
  )
  (:action finalize_node_b_preparation
    :parameters (?npc_agent - npc_agent ?mission_node_b - mission_node_b ?upgrade_module - upgrade_module ?consumable_resource - consumable_resource)
    :precondition
      (and
        (configuration_validated ?npc_agent)
        (upgrade_module_attached ?npc_agent ?upgrade_module)
        (mission_node_b_assigned ?npc_agent ?mission_node_b)
        (mission_node_b_secondary_ready ?mission_node_b)
        (npc_consumable_reserved ?npc_agent ?consumable_resource)
        (not
          (npc_node_b_ready ?npc_agent)
        )
      )
    :effect
      (and
        (mission_node_b_primary_ready ?mission_node_b)
        (npc_node_b_ready ?npc_agent)
        (consumable_available ?consumable_resource)
        (not
          (npc_consumable_reserved ?npc_agent ?consumable_resource)
        )
      )
  )
  (:action compose_and_stage_mission_asset
    :parameters (?player_agent - player_agent ?npc_agent - npc_agent ?mission_node_a - mission_node_a ?mission_node_b - mission_node_b ?mission_asset - mission_asset)
    :precondition
      (and
        (player_ready_for_asset ?player_agent)
        (npc_ready_for_asset ?npc_agent)
        (mission_node_a_assigned ?player_agent ?mission_node_a)
        (mission_node_b_assigned ?npc_agent ?mission_node_b)
        (mission_node_a_primary_ready ?mission_node_a)
        (mission_node_b_primary_ready ?mission_node_b)
        (player_node_a_ready ?player_agent)
        (npc_node_b_ready ?npc_agent)
        (mission_asset_available ?mission_asset)
      )
    :effect
      (and
        (mission_asset_staged ?mission_asset)
        (asset_includes_node_a ?mission_asset ?mission_node_a)
        (asset_includes_node_b ?mission_asset ?mission_node_b)
        (not
          (mission_asset_available ?mission_asset)
        )
      )
  )
  (:action stage_mission_asset_outcome_primary
    :parameters (?player_agent - player_agent ?npc_agent - npc_agent ?mission_node_a - mission_node_a ?mission_node_b - mission_node_b ?mission_asset - mission_asset)
    :precondition
      (and
        (player_ready_for_asset ?player_agent)
        (npc_ready_for_asset ?npc_agent)
        (mission_node_a_assigned ?player_agent ?mission_node_a)
        (mission_node_b_assigned ?npc_agent ?mission_node_b)
        (mission_node_a_secondary_ready ?mission_node_a)
        (mission_node_b_primary_ready ?mission_node_b)
        (not
          (player_node_a_ready ?player_agent)
        )
        (npc_node_b_ready ?npc_agent)
        (mission_asset_available ?mission_asset)
      )
    :effect
      (and
        (mission_asset_staged ?mission_asset)
        (asset_includes_node_a ?mission_asset ?mission_node_a)
        (asset_includes_node_b ?mission_asset ?mission_node_b)
        (mission_asset_outcome_primary ?mission_asset)
        (not
          (mission_asset_available ?mission_asset)
        )
      )
  )
  (:action stage_mission_asset_outcome_secondary
    :parameters (?player_agent - player_agent ?npc_agent - npc_agent ?mission_node_a - mission_node_a ?mission_node_b - mission_node_b ?mission_asset - mission_asset)
    :precondition
      (and
        (player_ready_for_asset ?player_agent)
        (npc_ready_for_asset ?npc_agent)
        (mission_node_a_assigned ?player_agent ?mission_node_a)
        (mission_node_b_assigned ?npc_agent ?mission_node_b)
        (mission_node_a_primary_ready ?mission_node_a)
        (mission_node_b_secondary_ready ?mission_node_b)
        (player_node_a_ready ?player_agent)
        (not
          (npc_node_b_ready ?npc_agent)
        )
        (mission_asset_available ?mission_asset)
      )
    :effect
      (and
        (mission_asset_staged ?mission_asset)
        (asset_includes_node_a ?mission_asset ?mission_node_a)
        (asset_includes_node_b ?mission_asset ?mission_node_b)
        (mission_asset_outcome_secondary ?mission_asset)
        (not
          (mission_asset_available ?mission_asset)
        )
      )
  )
  (:action stage_mission_asset_outcome_both
    :parameters (?player_agent - player_agent ?npc_agent - npc_agent ?mission_node_a - mission_node_a ?mission_node_b - mission_node_b ?mission_asset - mission_asset)
    :precondition
      (and
        (player_ready_for_asset ?player_agent)
        (npc_ready_for_asset ?npc_agent)
        (mission_node_a_assigned ?player_agent ?mission_node_a)
        (mission_node_b_assigned ?npc_agent ?mission_node_b)
        (mission_node_a_secondary_ready ?mission_node_a)
        (mission_node_b_secondary_ready ?mission_node_b)
        (not
          (player_node_a_ready ?player_agent)
        )
        (not
          (npc_node_b_ready ?npc_agent)
        )
        (mission_asset_available ?mission_asset)
      )
    :effect
      (and
        (mission_asset_staged ?mission_asset)
        (asset_includes_node_a ?mission_asset ?mission_node_a)
        (asset_includes_node_b ?mission_asset ?mission_node_b)
        (mission_asset_outcome_primary ?mission_asset)
        (mission_asset_outcome_secondary ?mission_asset)
        (not
          (mission_asset_available ?mission_asset)
        )
      )
  )
  (:action validate_mission_asset
    :parameters (?mission_asset - mission_asset ?player_agent - player_agent ?upgrade_module - upgrade_module)
    :precondition
      (and
        (mission_asset_staged ?mission_asset)
        (player_ready_for_asset ?player_agent)
        (upgrade_module_attached ?player_agent ?upgrade_module)
        (not
          (mission_asset_validated ?mission_asset)
        )
      )
    :effect (mission_asset_validated ?mission_asset)
  )
  (:action initialize_upgrade_state
    :parameters (?recovery_agent - controller ?upgrade_state - upgrade_state ?mission_asset - mission_asset)
    :precondition
      (and
        (configuration_validated ?recovery_agent)
        (controller_has_mission_asset ?recovery_agent ?mission_asset)
        (controller_has_upgrade_state ?recovery_agent ?upgrade_state)
        (upgrade_state_available ?upgrade_state)
        (mission_asset_staged ?mission_asset)
        (mission_asset_validated ?mission_asset)
        (not
          (upgrade_state_initialized ?upgrade_state)
        )
      )
    :effect
      (and
        (upgrade_state_initialized ?upgrade_state)
        (upgrade_state_applied_to_asset ?upgrade_state ?mission_asset)
        (not
          (upgrade_state_available ?upgrade_state)
        )
      )
  )
  (:action prime_controller_for_activation
    :parameters (?recovery_agent - controller ?upgrade_state - upgrade_state ?mission_asset - mission_asset ?upgrade_module - upgrade_module)
    :precondition
      (and
        (configuration_validated ?recovery_agent)
        (controller_has_upgrade_state ?recovery_agent ?upgrade_state)
        (upgrade_state_initialized ?upgrade_state)
        (upgrade_state_applied_to_asset ?upgrade_state ?mission_asset)
        (upgrade_module_attached ?recovery_agent ?upgrade_module)
        (not
          (mission_asset_outcome_primary ?mission_asset)
        )
        (not
          (controller_primed_for_activation ?recovery_agent)
        )
      )
    :effect (controller_primed_for_activation ?recovery_agent)
  )
  (:action assign_configuration_tag
    :parameters (?recovery_agent - controller ?configuration_tag - configuration_tag)
    :precondition
      (and
        (configuration_validated ?recovery_agent)
        (configuration_tag_available ?configuration_tag)
        (not
          (controller_config_tag_present ?recovery_agent)
        )
      )
    :effect
      (and
        (controller_config_tag_present ?recovery_agent)
        (controller_configuration_tag_assigned ?recovery_agent ?configuration_tag)
        (not
          (configuration_tag_available ?configuration_tag)
        )
      )
  )
  (:action apply_configuration_tag_and_prime
    :parameters (?recovery_agent - controller ?upgrade_state - upgrade_state ?mission_asset - mission_asset ?upgrade_module - upgrade_module ?configuration_tag - configuration_tag)
    :precondition
      (and
        (configuration_validated ?recovery_agent)
        (controller_has_upgrade_state ?recovery_agent ?upgrade_state)
        (upgrade_state_initialized ?upgrade_state)
        (upgrade_state_applied_to_asset ?upgrade_state ?mission_asset)
        (upgrade_module_attached ?recovery_agent ?upgrade_module)
        (mission_asset_outcome_primary ?mission_asset)
        (controller_config_tag_present ?recovery_agent)
        (controller_configuration_tag_assigned ?recovery_agent ?configuration_tag)
        (not
          (controller_primed_for_activation ?recovery_agent)
        )
      )
    :effect
      (and
        (controller_primed_for_activation ?recovery_agent)
        (controller_config_tag_applied ?recovery_agent)
      )
  )
  (:action engage_activation_sequence_variant_1
    :parameters (?recovery_agent - controller ?special_attachment - special_attachment ?equipment_slot - equipment_slot ?upgrade_state - upgrade_state ?mission_asset - mission_asset)
    :precondition
      (and
        (controller_primed_for_activation ?recovery_agent)
        (controller_has_special_attachment ?recovery_agent ?special_attachment)
        (equipment_slot_bound ?recovery_agent ?equipment_slot)
        (controller_has_upgrade_state ?recovery_agent ?upgrade_state)
        (upgrade_state_applied_to_asset ?upgrade_state ?mission_asset)
        (not
          (mission_asset_outcome_secondary ?mission_asset)
        )
        (not
          (controller_activation_engaged ?recovery_agent)
        )
      )
    :effect (controller_activation_engaged ?recovery_agent)
  )
  (:action engage_activation_sequence_variant_2
    :parameters (?recovery_agent - controller ?special_attachment - special_attachment ?equipment_slot - equipment_slot ?upgrade_state - upgrade_state ?mission_asset - mission_asset)
    :precondition
