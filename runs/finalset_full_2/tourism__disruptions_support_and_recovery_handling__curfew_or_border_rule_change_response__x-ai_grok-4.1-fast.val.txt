  (:action open_disruption_case
    :parameters (?disruption_case - disruption_case)
    :precondition
      (and
        (not
          (case_initiated ?disruption_case)
        )
        (not
          (subject_settlement_initiated ?disruption_case)
        )
      )
    :effect (case_initiated ?disruption_case)
  )
  (:action reserve_service_option_for_case
    :parameters (?disruption_case - disruption_case ?service_option - service_option)
    :precondition
      (and
        (case_initiated ?disruption_case)
        (not
          (option_reserved ?disruption_case)
        )
        (service_option_available ?service_option)
      )
    :effect
      (and
        (option_reserved ?disruption_case)
        (subject_reserved_service_option ?disruption_case ?service_option)
        (not
          (service_option_available ?service_option)
        )
      )
  )
  (:action attach_booking_to_case
    :parameters (?disruption_case - disruption_case ?booking_record - booking_record)
    :precondition
      (and
        (case_initiated ?disruption_case)
        (option_reserved ?disruption_case)
        (booking_available ?booking_record)
      )
    :effect
      (and
        (subject_booking_associated ?disruption_case ?booking_record)
        (not
          (booking_available ?booking_record)
        )
      )
  )
  (:action verify_booking_attachment
    :parameters (?disruption_case - disruption_case ?booking_record - booking_record)
    :precondition
      (and
        (case_initiated ?disruption_case)
        (option_reserved ?disruption_case)
        (subject_booking_associated ?disruption_case ?booking_record)
        (not
          (subject_booking_verified ?disruption_case)
        )
      )
    :effect (subject_booking_verified ?disruption_case)
  )
  (:action release_booking_association
    :parameters (?disruption_case - disruption_case ?booking_record - booking_record)
    :precondition
      (and
        (subject_booking_associated ?disruption_case ?booking_record)
      )
    :effect
      (and
        (booking_available ?booking_record)
        (not
          (subject_booking_associated ?disruption_case ?booking_record)
        )
      )
  )
  (:action assign_supplier_contact
    :parameters (?disruption_case - disruption_case ?supplier_contact - supplier_contact)

        (supplier_contact_assigned ?disruption_case ?supplier_contact)
      )
    :effect
      (and
        (supplier_contact_available ?supplier_contact)

          (supplier_contact_assigned ?disruption_case ?supplier_contact)
        )
      )
  )
  (:action assign_financial_account_to_handler

        (associated_eligibility_confirmed ?associated_traveler)
        (remediation_offer_available ?remediation_offer)

    :effect
      (and
        (remediation_offer_compiled ?remediation_offer)
        (offer_includes_affected_location ?remediation_offer ?affected_location)
        (offer_includes_travel_segment ?remediation_offer ?travel_segment)
        (not
          (remediation_offer_available ?remediation_offer)
        )
      )
  )
  (:action assemble_remediation_offer_requires_token
    :parameters (?lead_traveler - lead_traveler ?associated_traveler - associated_traveler ?affected_location - affected_location ?travel_segment - travel_segment ?remediation_offer - remediation_offer)

        (remediation_offer_compiled ?remediation_offer)

        (not
          (remediation_offer_available ?remediation_offer)
        )

          (remediation_offer_available ?remediation_offer)
