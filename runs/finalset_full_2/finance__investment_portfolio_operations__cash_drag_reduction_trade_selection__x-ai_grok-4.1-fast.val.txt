  (:action register_trade_case
    :parameters (?trade_case - case_subject)
    :precondition
      (and
        (not
          (trade_entity_registered ?trade_case)
        )
        (not
          (post_trade_settled ?trade_case)
        )
      )
    :effect (trade_entity_registered ?trade_case)
  )
  (:action apply_selection_strategy
    :parameters (?trade_case - case_subject ?selection_strategy - selection_strategy)
    :precondition
      (and
        (trade_entity_registered ?trade_case)
        (not
          (selection_strategy_applied ?trade_case)
        )
        (selection_strategy_available ?selection_strategy)
      )
    :effect
      (and
        (selection_strategy_applied ?trade_case)
        (entity_selection_strategy_binding ?trade_case ?selection_strategy)
        (not
          (selection_strategy_available ?selection_strategy)
        )
      )
  )
  (:action select_instrument
    :parameters (?trade_case - case_subject ?instrument - instrument)
    :precondition
      (and
        (trade_entity_registered ?trade_case)
        (selection_strategy_applied ?trade_case)
        (instrument_available ?instrument)
      )
    :effect
      (and
        (instrument_selected ?trade_case ?instrument)
        (not
          (instrument_available ?instrument)
        )
      )
  )
  (:action confirm_instrument_screening
    :parameters (?trade_case - case_subject ?instrument - instrument)
    :precondition
      (and
        (trade_entity_registered ?trade_case)
        (selection_strategy_applied ?trade_case)
        (instrument_selected ?trade_case ?instrument)
        (not
          (trade_entity_qualified ?trade_case)
        )
      )
    :effect (trade_entity_qualified ?trade_case)
  )
  (:action release_instrument_selection
    :parameters (?trade_case - case_subject ?instrument - instrument)
    :precondition
      (and
        (instrument_selected ?trade_case ?instrument)
      )
    :effect
      (and
        (instrument_available ?instrument)
        (not
          (instrument_selected ?trade_case ?instrument)
        )
      )
  )
  (:action reserve_trading_desk
    :parameters (?trade_case - case_subject ?trading_desk - trading_desk)
    :precondition
      (and
        (trade_entity_qualified ?trade_case)
        (trading_desk_available ?trading_desk)
      )
    :effect
      (and
        (trading_desk_reserved ?trade_case ?trading_desk)
        (not
          (trading_desk_available ?trading_desk)
        )
      )
  )
  (:action release_trading_desk
    :parameters (?trade_case - case_subject ?trading_desk - trading_desk)
    :precondition
      (and
        (trading_desk_reserved ?trade_case ?trading_desk)
      )
    :effect
      (and
        (trading_desk_available ?trading_desk)
        (not
          (trading_desk_reserved ?trade_case ?trading_desk)
        )
      )
  )
  (:action reserve_execution_instruction_template

          (portfolio_execution_template_linked ?portfolio ?execution_instruction_template)

        (portfolio_settlement_parameter_linked ?portfolio ?settlement_parameter)
      )

        (settlement_parameter_available ?settlement_parameter)

          (portfolio_settlement_parameter_linked ?portfolio ?settlement_parameter)

  (:action screen_source_liquidity
    :parameters (?source_tranche - source_tranche ?source_liquidity_profile - source_liquidity_profile ?instrument - instrument)
    :precondition
      (and
        (trade_entity_qualified ?source_tranche)
        (instrument_selected ?source_tranche ?instrument)
        (source_liquidity_profile_linked ?source_tranche ?source_liquidity_profile)
        (not
          (source_liquidity_screened ?source_liquidity_profile)

          (source_liquidity_bound ?source_liquidity_profile)

  )
  (:action commit_source_tranche
    :parameters (?source_tranche - source_tranche ?source_liquidity_profile - source_liquidity_profile ?trading_desk - trading_desk)
    :precondition
      (and
        (trade_entity_qualified ?source_tranche)
        (trading_desk_reserved ?source_tranche ?trading_desk)
        (source_liquidity_profile_linked ?source_tranche ?source_liquidity_profile)
        (source_liquidity_screened ?source_liquidity_profile)
        (not
          (source_tranche_execution_ready ?source_tranche)
        )
      )
    :effect
      (and
        (source_tranche_execution_ready ?source_tranche)
        (source_tranche_committed ?source_tranche)

  (:action bind_source_liquidity_template
    :parameters (?source_tranche - source_tranche ?source_liquidity_profile - source_liquidity_profile ?liquidity_template - liquidity_profile_template)
    :precondition
      (and
        (trade_entity_qualified ?source_tranche)
        (source_liquidity_profile_linked ?source_tranche ?source_liquidity_profile)
        (liquidity_template_available ?liquidity_template)
        (not
          (source_tranche_execution_ready ?source_tranche)
        )
      )
    :effect
      (and
        (source_liquidity_bound ?source_liquidity_profile)
        (source_tranche_execution_ready ?source_tranche)
        (source_liquidity_template_bound ?source_tranche ?liquidity_template)
        (not
          (liquidity_template_available ?liquidity_template)
        )

        (source_liquidity_template_bound ?source_tranche ?liquidity_template)
        (not
          (source_tranche_committed ?source_tranche)

    :effect
      (and
        (source_liquidity_screened ?source_liquidity_profile)
        (source_tranche_committed ?source_tranche)
        (liquidity_template_available ?liquidity_template)

          (source_liquidity_template_bound ?source_tranche ?liquidity_template)
        )
      )
  )

        (liquidity_template_available ?liquidity_template)
        (not
          (destination_tranche_execution_ready ?destination_tranche)

    :effect
      (and
        (destination_liquidity_bound ?destination_liquidity_profile)

          (liquidity_template_available ?liquidity_template)

  (:action finalize_destination_tranche_selection
    :parameters (?destination_tranche - destination_tranche ?destination_liquidity_profile - destination_liquidity_profile ?instrument - instrument ?liquidity_template - liquidity_profile_template)

        (destination_liquidity_template_bound ?destination_tranche ?liquidity_template)
        (not
          (destination_tranche_committed ?destination_tranche)
        )

    :effect
      (and
        (destination_liquidity_screened ?destination_liquidity_profile)
        (destination_tranche_committed ?destination_tranche)
        (liquidity_template_available ?liquidity_template)
        (not
          (destination_liquidity_template_bound ?destination_tranche ?liquidity_template)

  (:action assemble_trade_proposal

    :precondition
      (and
        (source_tranche_execution_ready ?source_tranche)
        (destination_tranche_execution_ready ?destination_tranche)
        (source_liquidity_profile_linked ?source_tranche ?source_liquidity_profile)
        (destination_liquidity_profile_linked ?destination_tranche ?destination_liquidity_profile)
        (source_liquidity_screened ?source_liquidity_profile)
        (destination_liquidity_screened ?destination_liquidity_profile)
        (source_tranche_committed ?source_tranche)
        (destination_tranche_committed ?destination_tranche)
        (trade_proposal_available ?trade_proposal)
      )
    :effect
      (and
        (trade_proposal_assembled ?trade_proposal)
        (proposal_source_liquidity_linked ?trade_proposal ?source_liquidity_profile)
        (proposal_destination_liquidity_linked ?trade_proposal ?destination_liquidity_profile)
        (not
          (trade_proposal_available ?trade_proposal)
        )
      )
  )
  (:action assemble_trade_proposal_source_aligned

    :precondition
      (and
        (source_tranche_execution_ready ?source_tranche)
        (destination_tranche_execution_ready ?destination_tranche)
        (source_liquidity_profile_linked ?source_tranche ?source_liquidity_profile)
        (destination_liquidity_profile_linked ?destination_tranche ?destination_liquidity_profile)
        (source_liquidity_bound ?source_liquidity_profile)
        (destination_liquidity_screened ?destination_liquidity_profile)
        (not
          (source_tranche_committed ?source_tranche)
        )
        (destination_tranche_committed ?destination_tranche)
        (trade_proposal_available ?trade_proposal)
      )
    :effect
      (and
        (trade_proposal_assembled ?trade_proposal)
        (proposal_source_liquidity_linked ?trade_proposal ?source_liquidity_profile)
        (proposal_destination_liquidity_linked ?trade_proposal ?destination_liquidity_profile)
        (source_leg_enriched ?trade_proposal)
        (not
          (trade_proposal_available ?trade_proposal)
        )
      )
  )
  (:action assemble_trade_proposal_destination_aligned
    :parameters (?source_tranche - source_tranche ?destination_tranche - destination_tranche ?source_liquidity_profile - source_liquidity_profile ?destination_liquidity_profile - destination_liquidity_profile ?trade_proposal - trade_proposal)
    :precondition
      (and
        (source_tranche_execution_ready ?source_tranche)
        (destination_tranche_execution_ready ?destination_tranche)
        (source_liquidity_profile_linked ?source_tranche ?source_liquidity_profile)
        (destination_liquidity_profile_linked ?destination_tranche ?destination_liquidity_profile)
        (source_liquidity_screened ?source_liquidity_profile)
        (destination_liquidity_bound ?destination_liquidity_profile)
        (source_tranche_committed ?source_tranche)
        (not
          (destination_tranche_committed ?destination_tranche)
        )
        (trade_proposal_available ?trade_proposal)
      )
    :effect
      (and
        (trade_proposal_assembled ?trade_proposal)
        (proposal_source_liquidity_linked ?trade_proposal ?source_liquidity_profile)
        (proposal_destination_liquidity_linked ?trade_proposal ?destination_liquidity_profile)
        (destination_leg_enriched ?trade_proposal)
        (not
          (trade_proposal_available ?trade_proposal)
        )
      )
  )
  (:action assemble_trade_proposal_fully_aligned
    :parameters (?source_tranche - source_tranche ?destination_tranche - destination_tranche ?source_liquidity_profile - source_liquidity_profile ?destination_liquidity_profile - destination_liquidity_profile ?trade_proposal - trade_proposal)
    :precondition
      (and
        (source_tranche_execution_ready ?source_tranche)
        (destination_tranche_execution_ready ?destination_tranche)

        (not
          (source_tranche_committed ?source_tranche)
        )
        (not
          (destination_tranche_committed ?destination_tranche)
        )
        (trade_proposal_available ?trade_proposal)

    :effect
      (and
        (trade_proposal_assembled ?trade_proposal)
        (proposal_source_liquidity_linked ?trade_proposal ?source_liquidity_profile)
        (proposal_destination_liquidity_linked ?trade_proposal ?destination_liquidity_profile)
        (source_leg_enriched ?trade_proposal)
        (destination_leg_enriched ?trade_proposal)

          (trade_proposal_available ?trade_proposal)

          (trade_proposal_authorized ?trade_proposal)
        )
      )

  )

        (not
          (holding_lot_allocated ?holding_lot)
        )
      )
    :effect
      (and
        (holding_lot_allocated ?holding_lot)
        (holding_lot_bound_to_proposal ?holding_lot ?trade_proposal)
        (not
          (holding_lot_available ?holding_lot)
        )
      )
  )
  (:action stage_portfolio_proposal

    :precondition
      (and
        (trade_entity_qualified ?portfolio)

        (holding_lot_allocated ?holding_lot)
        (holding_lot_bound_to_proposal ?holding_lot ?trade_proposal)

          (source_leg_enriched ?trade_proposal)
        )
        (not
          (portfolio_proposal_staged ?portfolio)
        )

    :precondition
      (and
        (trade_entity_qualified ?portfolio)
        (order_constraint_available ?order_constraint)
        (not
          (portfolio_order_constraint_bound ?portfolio)
        )
      )
    :effect
      (and
        (portfolio_order_constraint_bound ?portfolio)
        (portfolio_order_constraint_linked ?portfolio ?order_constraint)
        (not
          (order_constraint_available ?order_constraint)
        )
      )
  )
  (:action confirm_order_constraint_bundle
    :parameters (?portfolio - portfolio ?holding_lot - holding_lot ?trade_proposal - trade_proposal ?instrument - instrument ?order_constraint - order_constraint)
    :precondition
      (and
        (trade_entity_qualified ?portfolio)
        (portfolio_holding_lot_linked ?portfolio ?holding_lot)
        (holding_lot_allocated ?holding_lot)
        (holding_lot_bound_to_proposal ?holding_lot ?trade_proposal)
        (instrument_selected ?portfolio ?instrument)
        (source_leg_enriched ?trade_proposal)
        (portfolio_order_constraint_bound ?portfolio)
        (portfolio_order_constraint_linked ?portfolio ?order_constraint)
        (not
          (portfolio_proposal_staged ?portfolio)
        )
      )
    :effect
      (and
        (portfolio_proposal_staged ?portfolio)
        (order_constraint_bundle_reviewed ?portfolio)
      )
  )
  (:action package_execution_instructions
    :parameters (?portfolio - portfolio ?execution_instruction_template - execution_instruction_template ?trading_desk - trading_desk ?holding_lot - holding_lot ?trade_proposal - trade_proposal)
    :precondition
      (and
        (portfolio_proposal_staged ?portfolio)
        (portfolio_execution_template_linked ?portfolio ?execution_instruction_template)
        (trading_desk_reserved ?portfolio ?trading_desk)
        (portfolio_holding_lot_linked ?portfolio ?holding_lot)
        (holding_lot_bound_to_proposal ?holding_lot ?trade_proposal)
        (not
          (destination_leg_enriched ?trade_proposal)
