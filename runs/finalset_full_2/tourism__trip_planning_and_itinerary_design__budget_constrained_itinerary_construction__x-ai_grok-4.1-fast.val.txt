--- domain.pddl (as passed to VAL) ---
(define (domain budget_constrained_itinerary_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_option_superclass - object time_slot_superclass - object booking_slot_superclass - object itinerary_element_superclass - object itinerary_stop - itinerary_element_superclass transport_option - resource_option_superclass activity - resource_option_superclass venue_supplier - resource_option_superclass promotional_offer - resource_option_superclass ancillary_service - resource_option_superclass budget_token - resource_option_superclass supplier_service_variant - resource_option_superclass special_requirement_token - resource_option_superclass consumable_ticket - time_slot_superclass accommodation_option - time_slot_superclass preference_token - time_slot_superclass time_window_a - booking_slot_superclass time_window_b - booking_slot_superclass booking_candidate - booking_slot_superclass day_slot_group_a - itinerary_stop day_slot_group_b - itinerary_stop primary_day_slot - day_slot_group_a secondary_day_slot - day_slot_group_a itinerary_instance - day_slot_group_b)
  (:predicates
    (itinerary_element_initialized ?itinerary_stop - itinerary_stop)
    (itinerary_element_ready_for_venue_assignment ?itinerary_stop - itinerary_stop)
    (itinerary_element_transport_assigned ?itinerary_stop - itinerary_stop)
    (budget_allocated ?itinerary_stop - itinerary_stop)
    (itinerary_element_finalized ?itinerary_stop - itinerary_stop)
    (budget_authorized ?itinerary_stop - itinerary_stop)
    (transport_available ?transport_option - transport_option)
    (itinerary_element_transport_assignment ?itinerary_stop - itinerary_stop ?transport_option - transport_option)
    (activity_available ?activity - activity)
    (itinerary_element_has_activity ?itinerary_stop - itinerary_stop ?activity - activity)
    (venue_available ?venue_supplier - venue_supplier)
    (itinerary_element_has_venue ?itinerary_stop - itinerary_stop ?venue_supplier - venue_supplier)
    (ticket_available ?consumable_ticket - consumable_ticket)
    (primary_day_slot_ticket_allocated ?primary_day_slot - primary_day_slot ?consumable_ticket - consumable_ticket)
    (secondary_day_slot_ticket_allocated ?secondary_day_slot - secondary_day_slot ?consumable_ticket - consumable_ticket)
    (primary_day_slot_time_window_a_assigned ?primary_day_slot - primary_day_slot ?time_window_a - time_window_a)
    (time_window_a_selected ?time_window_a - time_window_a)
    (time_window_a_has_ticket ?time_window_a - time_window_a)
    (primary_day_slot_ready ?primary_day_slot - primary_day_slot)
    (secondary_day_slot_time_window_b_assigned ?secondary_day_slot - secondary_day_slot ?time_window_b - time_window_b)
    (time_window_b_selected ?time_window_b - time_window_b)
    (time_window_b_has_ticket ?time_window_b - time_window_b)
    (secondary_day_slot_ready ?secondary_day_slot - secondary_day_slot)
    (booking_candidate_available ?booking_candidate - booking_candidate)
    (booking_candidate_ready ?booking_candidate - booking_candidate)
    (booking_candidate_time_window_a_assigned ?booking_candidate - booking_candidate ?time_window_a - time_window_a)
    (booking_candidate_time_window_b_assigned ?booking_candidate - booking_candidate ?time_window_b - time_window_b)
    (booking_candidate_includes_time_window_a ?booking_candidate - booking_candidate)
    (booking_candidate_includes_time_window_b ?booking_candidate - booking_candidate)
    (booking_candidate_time_confirmed ?booking_candidate - booking_candidate)
    (itinerary_primary_day_slot_association ?itinerary_instance - itinerary_instance ?primary_day_slot - primary_day_slot)
    (itinerary_secondary_day_slot_association ?itinerary_instance - itinerary_instance ?secondary_day_slot - secondary_day_slot)
    (itinerary_has_booking_candidate ?itinerary_instance - itinerary_instance ?booking_candidate - booking_candidate)
    (accommodation_available ?accommodation_option - accommodation_option)
    (itinerary_has_accommodation_option ?itinerary_instance - itinerary_instance ?accommodation_option - accommodation_option)
    (accommodation_reserved ?accommodation_option - accommodation_option)
    (accommodation_allocated_to_candidate ?accommodation_option - accommodation_option ?booking_candidate - booking_candidate)
    (accommodation_validation_passed ?itinerary_instance - itinerary_instance)
    (supplier_availability_confirmed ?itinerary_instance - itinerary_instance)
    (supplier_confirmation_ready ?itinerary_instance - itinerary_instance)
    (promotion_attached ?itinerary_instance - itinerary_instance)
    (promotion_validation_passed ?itinerary_instance - itinerary_instance)
    (ancillary_attachment_allowed ?itinerary_instance - itinerary_instance)
    (final_validation_passed ?itinerary_instance - itinerary_instance)
    (preference_available ?preference_token - preference_token)
    (itinerary_preference_assignment ?itinerary_instance - itinerary_instance ?preference_token - preference_token)
    (itinerary_has_preference ?itinerary_instance - itinerary_instance)
    (preference_validation_stage1 ?itinerary_instance - itinerary_instance)
    (preference_validation_stage2 ?itinerary_instance - itinerary_instance)
    (promotional_offer_available ?promotional_offer - promotional_offer)
    (itinerary_promotion_assignment ?itinerary_instance - itinerary_instance ?promotional_offer - promotional_offer)
    (ancillary_service_available ?ancillary_service - ancillary_service)
    (itinerary_has_ancillary_service ?itinerary_instance - itinerary_instance ?ancillary_service - ancillary_service)
    (supplier_service_variant_available ?supplier_service_variant - supplier_service_variant)
    (itinerary_has_supplier_service_variant ?itinerary_instance - itinerary_instance ?supplier_service_variant - supplier_service_variant)
    (special_requirement_token_available ?special_requirement_token - special_requirement_token)
    (itinerary_has_special_requirement ?itinerary_instance - itinerary_instance ?special_requirement_token - special_requirement_token)
    (budget_token_available ?budget_token - budget_token)
    (itinerary_element_has_budget_token ?itinerary_stop - itinerary_stop ?budget_token - budget_token)
    (primary_day_slot_locked ?primary_day_slot - primary_day_slot)
    (secondary_day_slot_locked ?secondary_day_slot - secondary_day_slot)
    (finalization_acknowledged ?itinerary_instance - itinerary_instance)
  )
  (:action create_itinerary_stop
    :parameters (?itinerary_stop - itinerary_stop)
    :precondition
      (and
        (not
          (itinerary_element_initialized ?itinerary_stop)
        )
        (not
          (budget_allocated ?itinerary_stop)
        )
      )
    :effect (itinerary_element_initialized ?itinerary_stop)
  )
  (:action assign_transport_to_stop
    :parameters (?itinerary_stop - itinerary_stop ?transport_option - transport_option)
    :precondition
      (and
        (itinerary_element_initialized ?itinerary_stop)
        (not
          (itinerary_element_transport_assigned ?itinerary_stop)
        )
        (transport_available ?transport_option)
      )
    :effect
      (and
        (itinerary_element_transport_assigned ?itinerary_stop)
        (itinerary_element_transport_assignment ?itinerary_stop ?transport_option)
        (not
          (transport_available ?transport_option)
        )
      )
  )
  (:action attach_activity_to_stop
    :parameters (?itinerary_stop - itinerary_stop ?activity - activity)
    :precondition
      (and
        (itinerary_element_initialized ?itinerary_stop)
        (itinerary_element_transport_assigned ?itinerary_stop)
        (activity_available ?activity)
      )
    :effect
      (and
        (itinerary_element_has_activity ?itinerary_stop ?activity)
        (not
          (activity_available ?activity)
        )
      )
  )
  (:action confirm_activity_assignment
    :parameters (?itinerary_stop - itinerary_stop ?activity - activity)
    :precondition
      (and
        (itinerary_element_initialized ?itinerary_stop)
        (itinerary_element_transport_assigned ?itinerary_stop)
        (itinerary_element_has_activity ?itinerary_stop ?activity)
        (not
          (itinerary_element_ready_for_venue_assignment ?itinerary_stop)
        )
      )
    :effect (itinerary_element_ready_for_venue_assignment ?itinerary_stop)
  )
  (:action retract_activity_from_stop
    :parameters (?itinerary_stop - itinerary_stop ?activity - activity)
    :precondition
      (and
        (itinerary_element_has_activity ?itinerary_stop ?activity)
      )
    :effect
      (and
        (activity_available ?activity)
        (not
          (itinerary_element_has_activity ?itinerary_stop ?activity)
        )
      )
  )
  (:action reserve_venue_for_stop
    :parameters (?itinerary_stop - itinerary_stop ?venue_supplier - venue_supplier)
    :precondition
      (and
        (itinerary_element_ready_for_venue_assignment ?itinerary_stop)
        (venue_available ?venue_supplier)
      )
    :effect
      (and
        (itinerary_element_has_venue ?itinerary_stop ?venue_supplier)
        (not
          (venue_available ?venue_supplier)
        )
      )
  )
  (:action release_venue_from_stop
    :parameters (?itinerary_stop - itinerary_stop ?venue_supplier - venue_supplier)
    :precondition
      (and
        (itinerary_element_has_venue ?itinerary_stop ?venue_supplier)
      )
    :effect
      (and
        (venue_available ?venue_supplier)
        (not
          (itinerary_element_has_venue ?itinerary_stop ?venue_supplier)
        )
      )
  )
  (:action attach_supplier_service_variant
    :parameters (?itinerary_instance - itinerary_instance ?supplier_service_variant - supplier_service_variant)
    :precondition
      (and
        (itinerary_element_ready_for_venue_assignment ?itinerary_instance)
        (supplier_service_variant_available ?supplier_service_variant)
      )
    :effect
      (and
        (itinerary_has_supplier_service_variant ?itinerary_instance ?supplier_service_variant)
        (not
          (supplier_service_variant_available ?supplier_service_variant)
        )
      )
  )
  (:action release_supplier_service_variant
    :parameters (?itinerary_instance - itinerary_instance ?supplier_service_variant - supplier_service_variant)
    :precondition
      (and
        (itinerary_has_supplier_service_variant ?itinerary_instance ?supplier_service_variant)
      )
    :effect
      (and
        (supplier_service_variant_available ?supplier_service_variant)
        (not
          (itinerary_has_supplier_service_variant ?itinerary_instance ?supplier_service_variant)
        )
      )
  )
  (:action attach_special_requirement_token
    :parameters (?itinerary_instance - itinerary_instance ?special_requirement_token - special_requirement_token)
    :precondition
      (and
        (itinerary_element_ready_for_venue_assignment ?itinerary_instance)
        (special_requirement_token_available ?special_requirement_token)
      )
    :effect
      (and
        (itinerary_has_special_requirement ?itinerary_instance ?special_requirement_token)
        (not
          (special_requirement_token_available ?special_requirement_token)
        )
      )
  )
  (:action release_special_requirement_token
    :parameters (?itinerary_instance - itinerary_instance ?special_requirement_token - special_requirement_token)
    :precondition
      (and
        (itinerary_has_special_requirement ?itinerary_instance ?special_requirement_token)
      )
    :effect
      (and
        (special_requirement_token_available ?special_requirement_token)
        (not
          (itinerary_has_special_requirement ?itinerary_instance ?special_requirement_token)
        )
      )
  )
  (:action select_time_window_for_primary_day_slot
    :parameters (?primary_day_slot - primary_day_slot ?time_window_a - time_window_a ?activity - activity)
    :precondition
      (and
        (itinerary_element_ready_for_venue_assignment ?primary_day_slot)
        (itinerary_element_has_activity ?primary_day_slot ?activity)
        (primary_day_slot_time_window_a_assigned ?primary_day_slot ?time_window_a)
        (not
          (time_window_a_selected ?time_window_a)
        )
        (not
          (time_window_a_has_ticket ?time_window_a)
        )
      )
    :effect (time_window_a_selected ?time_window_a)
  )
  (:action confirm_primary_day_slot_with_venue
    :parameters (?primary_day_slot - primary_day_slot ?time_window_a - time_window_a ?venue_supplier - venue_supplier)
    :precondition
      (and
        (itinerary_element_ready_for_venue_assignment ?primary_day_slot)
        (itinerary_element_has_venue ?primary_day_slot ?venue_supplier)
        (primary_day_slot_time_window_a_assigned ?primary_day_slot ?time_window_a)
        (time_window_a_selected ?time_window_a)
        (not
          (primary_day_slot_locked ?primary_day_slot)
        )
      )
    :effect
      (and
        (primary_day_slot_locked ?primary_day_slot)
        (primary_day_slot_ready ?primary_day_slot)
      )
  )
  (:action allocate_ticket_to_primary_day_slot
    :parameters (?primary_day_slot - primary_day_slot ?time_window_a - time_window_a ?consumable_ticket - consumable_ticket)
    :precondition
      (and
        (itinerary_element_ready_for_venue_assignment ?primary_day_slot)
        (primary_day_slot_time_window_a_assigned ?primary_day_slot ?time_window_a)
        (ticket_available ?consumable_ticket)
        (not
          (primary_day_slot_locked ?primary_day_slot)
        )
      )
    :effect
      (and
        (time_window_a_has_ticket ?time_window_a)
        (primary_day_slot_locked ?primary_day_slot)
        (primary_day_slot_ticket_allocated ?primary_day_slot ?consumable_ticket)
        (not
          (ticket_available ?consumable_ticket)
        )
      )
  )
  (:action finalize_primary_day_slot_ticket_assignment
    :parameters (?primary_day_slot - primary_day_slot ?time_window_a - time_window_a ?activity - activity ?consumable_ticket - consumable_ticket)
    :precondition
      (and
        (itinerary_element_ready_for_venue_assignment ?primary_day_slot)
        (itinerary_element_has_activity ?primary_day_slot ?activity)
        (primary_day_slot_time_window_a_assigned ?primary_day_slot ?time_window_a)
        (time_window_a_has_ticket ?time_window_a)
        (primary_day_slot_ticket_allocated ?primary_day_slot ?consumable_ticket)
        (not
          (primary_day_slot_ready ?primary_day_slot)
        )
      )
    :effect
      (and
        (time_window_a_selected ?time_window_a)
        (primary_day_slot_ready ?primary_day_slot)
        (ticket_available ?consumable_ticket)
        (not
          (primary_day_slot_ticket_allocated ?primary_day_slot ?consumable_ticket)
        )
      )
  )
  (:action select_time_window_for_secondary_day_slot
    :parameters (?secondary_day_slot - secondary_day_slot ?time_window_b - time_window_b ?activity - activity)
    :precondition
      (and
        (itinerary_element_ready_for_venue_assignment ?secondary_day_slot)
        (itinerary_element_has_activity ?secondary_day_slot ?activity)
        (secondary_day_slot_time_window_b_assigned ?secondary_day_slot ?time_window_b)
        (not
          (time_window_b_selected ?time_window_b)
        )
        (not
          (time_window_b_has_ticket ?time_window_b)
        )
      )
    :effect (time_window_b_selected ?time_window_b)
  )
  (:action confirm_secondary_day_slot_with_venue
    :parameters (?secondary_day_slot - secondary_day_slot ?time_window_b - time_window_b ?venue_supplier - venue_supplier)
    :precondition
      (and
        (itinerary_element_ready_for_venue_assignment ?secondary_day_slot)
        (itinerary_element_has_venue ?secondary_day_slot ?venue_supplier)
        (secondary_day_slot_time_window_b_assigned ?secondary_day_slot ?time_window_b)
        (time_window_b_selected ?time_window_b)
        (not
          (secondary_day_slot_locked ?secondary_day_slot)
        )
      )
    :effect
      (and
        (secondary_day_slot_locked ?secondary_day_slot)
        (secondary_day_slot_ready ?secondary_day_slot)
      )
  )
  (:action allocate_ticket_to_secondary_day_slot
    :parameters (?secondary_day_slot - secondary_day_slot ?time_window_b - time_window_b ?consumable_ticket - consumable_ticket)
    :precondition
      (and
        (itinerary_element_ready_for_venue_assignment ?secondary_day_slot)
        (secondary_day_slot_time_window_b_assigned ?secondary_day_slot ?time_window_b)
        (ticket_available ?consumable_ticket)
        (not
          (secondary_day_slot_locked ?secondary_day_slot)
        )
      )
    :effect
      (and
        (time_window_b_has_ticket ?time_window_b)
        (secondary_day_slot_locked ?secondary_day_slot)
        (secondary_day_slot_ticket_allocated ?secondary_day_slot ?consumable_ticket)
        (not
          (ticket_available ?consumable_ticket)
        )
      )
  )
  (:action finalize_secondary_day_slot_ticket_assignment
    :parameters (?secondary_day_slot - secondary_day_slot ?time_window_b - time_window_b ?activity - activity ?consumable_ticket - consumable_ticket)
    :precondition
      (and
        (itinerary_element_ready_for_venue_assignment ?secondary_day_slot)
        (itinerary_element_has_activity ?secondary_day_slot ?activity)
        (secondary_day_slot_time_window_b_assigned ?secondary_day_slot ?time_window_b)
        (time_window_b_has_ticket ?time_window_b)
        (secondary_day_slot_ticket_allocated ?secondary_day_slot ?consumable_ticket)
        (not
          (secondary_day_slot_ready ?secondary_day_slot)
        )
      )
    :effect
      (and
        (time_window_b_selected ?time_window_b)
        (secondary_day_slot_ready ?secondary_day_slot)
        (ticket_available ?consumable_ticket)
        (not
          (secondary_day_slot_ticket_allocated ?secondary_day_slot ?consumable_ticket)
        )
      )
  )
  (:action assemble_booking_candidate_basic
    :parameters (?primary_day_slot - primary_day_slot ?secondary_day_slot - secondary_day_slot ?time_window_a - time_window_a ?time_window_b - time_window_b ?booking_candidate - booking_candidate)
    :precondition
      (and
        (primary_day_slot_locked ?primary_day_slot)
        (secondary_day_slot_locked ?secondary_day_slot)
        (primary_day_slot_time_window_a_assigned ?primary_day_slot ?time_window_a)
        (secondary_day_slot_time_window_b_assigned ?secondary_day_slot ?time_window_b)
        (time_window_a_selected ?time_window_a)
        (time_window_b_selected ?time_window_b)
        (primary_day_slot_ready ?primary_day_slot)
        (secondary_day_slot_ready ?secondary_day_slot)
        (booking_candidate_available ?booking_candidate)
      )
    :effect
      (and
        (booking_candidate_ready ?booking_candidate)
        (booking_candidate_time_window_a_assigned ?booking_candidate ?time_window_a)
        (booking_candidate_time_window_b_assigned ?booking_candidate ?time_window_b)
        (not
          (booking_candidate_available ?booking_candidate)
        )
      )
  )
  (:action assemble_booking_candidate_with_time_window_a
    :parameters (?primary_day_slot - primary_day_slot ?secondary_day_slot - secondary_day_slot ?time_window_a - time_window_a ?time_window_b - time_window_b ?booking_candidate - booking_candidate)
    :precondition
      (and
        (primary_day_slot_locked ?primary_day_slot)
        (secondary_day_slot_locked ?secondary_day_slot)
        (primary_day_slot_time_window_a_assigned ?primary_day_slot ?time_window_a)
        (secondary_day_slot_time_window_b_assigned ?secondary_day_slot ?time_window_b)
        (time_window_a_has_ticket ?time_window_a)
        (time_window_b_selected ?time_window_b)
        (not
          (primary_day_slot_ready ?primary_day_slot)
        )
        (secondary_day_slot_ready ?secondary_day_slot)
        (booking_candidate_available ?booking_candidate)
      )
    :effect
      (and
        (booking_candidate_ready ?booking_candidate)
        (booking_candidate_time_window_a_assigned ?booking_candidate ?time_window_a)
        (booking_candidate_time_window_b_assigned ?booking_candidate ?time_window_b)
        (booking_candidate_includes_time_window_a ?booking_candidate)
        (not
          (booking_candidate_available ?booking_candidate)
        )
      )
  )
  (:action assemble_booking_candidate_with_time_window_b
    :parameters (?primary_day_slot - primary_day_slot ?secondary_day_slot - secondary_day_slot ?time_window_a - time_window_a ?time_window_b - time_window_b ?booking_candidate - booking_candidate)
    :precondition
      (and
        (primary_day_slot_locked ?primary_day_slot)
        (secondary_day_slot_locked ?secondary_day_slot)
        (primary_day_slot_time_window_a_assigned ?primary_day_slot ?time_window_a)
        (secondary_day_slot_time_window_b_assigned ?secondary_day_slot ?time_window_b)
        (time_window_a_selected ?time_window_a)
        (time_window_b_has_ticket ?time_window_b)
        (primary_day_slot_ready ?primary_day_slot)
        (not
          (secondary_day_slot_ready ?secondary_day_slot)
        )
        (booking_candidate_available ?booking_candidate)
      )
    :effect
      (and
        (booking_candidate_ready ?booking_candidate)
        (booking_candidate_time_window_a_assigned ?booking_candidate ?time_window_a)
        (booking_candidate_time_window_b_assigned ?booking_candidate ?time_window_b)
        (booking_candidate_includes_time_window_b ?booking_candidate)
        (not
          (booking_candidate_available ?booking_candidate)
        )
      )
  )
  (:action assemble_booking_candidate_with_both_time_windows
    :parameters (?primary_day_slot - primary_day_slot ?secondary_day_slot - secondary_day_slot ?time_window_a - time_window_a ?time_window_b - time_window_b ?booking_candidate - booking_candidate)
    :precondition
      (and
        (primary_day_slot_locked ?primary_day_slot)
        (secondary_day_slot_locked ?secondary_day_slot)
        (primary_day_slot_time_window_a_assigned ?primary_day_slot ?time_window_a)
        (secondary_day_slot_time_window_b_assigned ?secondary_day_slot ?time_window_b)
        (time_window_a_has_ticket ?time_window_a)
        (time_window_b_has_ticket ?time_window_b)
        (not
          (primary_day_slot_ready ?primary_day_slot)
        )
        (not
          (secondary_day_slot_ready ?secondary_day_slot)
        )
        (booking_candidate_available ?booking_candidate)
      )
    :effect
      (and
        (booking_candidate_ready ?booking_candidate)
        (booking_candidate_time_window_a_assigned ?booking_candidate ?time_window_a)
        (booking_candidate_time_window_b_assigned ?booking_candidate ?time_window_b)
        (booking_candidate_includes_time_window_a ?booking_candidate)
        (booking_candidate_includes_time_window_b ?booking_candidate)
        (not
          (booking_candidate_available ?booking_candidate)
        )
      )
  )
  (:action confirm_booking_candidate_time_assignment
    :parameters (?booking_candidate - booking_candidate ?primary_day_slot - primary_day_slot ?activity - activity)
    :precondition
      (and
        (booking_candidate_ready ?booking_candidate)
        (primary_day_slot_locked ?primary_day_slot)
        (itinerary_element_has_activity ?primary_day_slot ?activity)
        (not
          (booking_candidate_time_confirmed ?booking_candidate)
        )
      )
    :effect (booking_candidate_time_confirmed ?booking_candidate)
  )
  (:action reserve_accommodation_for_booking_candidate
    :parameters (?itinerary_instance - itinerary_instance ?accommodation_option - accommodation_option ?booking_candidate - booking_candidate)
    :precondition
      (and
        (itinerary_element_ready_for_venue_assignment ?itinerary_instance)
        (itinerary_has_booking_candidate ?itinerary_instance ?booking_candidate)
        (itinerary_has_accommodation_option ?itinerary_instance ?accommodation_option)
        (accommodation_available ?accommodation_option)
        (booking_candidate_ready ?booking_candidate)
        (booking_candidate_time_confirmed ?booking_candidate)
        (not
          (accommodation_reserved ?accommodation_option)
