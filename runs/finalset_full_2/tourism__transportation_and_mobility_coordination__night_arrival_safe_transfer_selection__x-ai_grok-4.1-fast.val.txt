--- domain.pddl (as passed to VAL) ---
(define (domain night_arrival_safe_transfer_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types transfer_option_category - object transfer_attribute - object operational_entity - object transfer_record_supertype - object transfer_record - transfer_record_supertype vehicle_proposal_option - transfer_option_category local_transport_option - transfer_option_category escort_staff - transfer_option_category passenger_profile - transfer_option_category assignment_preference - transfer_option_category transfer_safety_resource - transfer_option_category vehicle_equipment_spec - transfer_option_category operator_clearance - transfer_option_category assistive_service_option - transfer_attribute route_leg - transfer_attribute special_handling_tag - transfer_attribute pickup_node - operational_entity dropoff_node - operational_entity transfer_vehicle_asset - operational_entity segment_group - transfer_record segment_group_variant - transfer_record incoming_segment - segment_group outgoing_segment - segment_group transfer_plan - segment_group_variant)
  (:predicates
    (transfer_record_registered ?arrival_transfer_record - transfer_record)
    (transfer_allocation_ready ?arrival_transfer_record - transfer_record)
    (transfer_vehicle_proposal_attached ?arrival_transfer_record - transfer_record)
    (transfer_confirmed ?arrival_transfer_record - transfer_record)
    (transfer_ready ?arrival_transfer_record - transfer_record)
    (transfer_safety_assigned ?arrival_transfer_record - transfer_record)
    (transfer_vehicle_proposal_available ?vehicle_proposal_option - vehicle_proposal_option)
    (transfer_vehicle_proposal_link ?arrival_transfer_record - transfer_record ?vehicle_proposal_option - vehicle_proposal_option)
    (transfer_local_option_available ?local_transport_option - local_transport_option)
    (transfer_local_option_link ?arrival_transfer_record - transfer_record ?local_transport_option - local_transport_option)
    (transfer_escort_available ?escort_staff - escort_staff)
    (transfer_escort_assignment ?arrival_transfer_record - transfer_record ?escort_staff - escort_staff)
    (transfer_assistive_service_available ?assistive_service_option - assistive_service_option)
    (transfer_assistive_service_link_incoming ?incoming_segment - incoming_segment ?assistive_service_option - assistive_service_option)
    (transfer_assistive_service_link_outgoing ?outgoing_segment - outgoing_segment ?assistive_service_option - assistive_service_option)
    (transfer_segment_pickup_node_link ?incoming_segment - incoming_segment ?pickup_node - pickup_node)
    (transfer_pickup_node_reserved ?pickup_node - pickup_node)
    (transfer_pickup_node_assist_reserved ?pickup_node - pickup_node)
    (incoming_segment_personnel_assigned ?incoming_segment - incoming_segment)
    (transfer_segment_dropoff_node_link ?outgoing_segment - outgoing_segment ?dropoff_node - dropoff_node)
    (transfer_dropoff_node_reserved ?dropoff_node - dropoff_node)
    (transfer_dropoff_node_assist_reserved ?dropoff_node - dropoff_node)
    (outgoing_segment_personnel_assigned ?outgoing_segment - outgoing_segment)
    (transfer_vehicle_asset_available ?transfer_vehicle_asset - transfer_vehicle_asset)
    (transfer_vehicle_allocated ?transfer_vehicle_asset - transfer_vehicle_asset)
    (transfer_vehicle_pickup_node_link ?transfer_vehicle_asset - transfer_vehicle_asset ?pickup_node - pickup_node)
    (transfer_vehicle_dropoff_node_link ?transfer_vehicle_asset - transfer_vehicle_asset ?dropoff_node - dropoff_node)
    (transfer_vehicle_approved_for_pickup ?transfer_vehicle_asset - transfer_vehicle_asset)
    (transfer_vehicle_approved_for_dropoff ?transfer_vehicle_asset - transfer_vehicle_asset)
    (transfer_vehicle_ready_for_assignment ?transfer_vehicle_asset - transfer_vehicle_asset)
    (transfer_plan_includes_incoming_segment ?transfer_plan - transfer_plan ?incoming_segment - incoming_segment)
    (transfer_plan_includes_outgoing_segment ?transfer_plan - transfer_plan ?outgoing_segment - outgoing_segment)
    (transfer_plan_assigned_vehicle_asset ?transfer_plan - transfer_plan ?transfer_vehicle_asset - transfer_vehicle_asset)
    (transfer_route_leg_available ?route_leg - route_leg)
    (transfer_plan_has_route_leg ?transfer_plan - transfer_plan ?route_leg - route_leg)
    (transfer_route_leg_confirmed ?route_leg - route_leg)
    (transfer_route_leg_assigned_vehicle ?route_leg - route_leg ?transfer_vehicle_asset - transfer_vehicle_asset)
    (transfer_plan_equipment_confirmed ?transfer_plan - transfer_plan)
    (transfer_plan_equipment_verified ?transfer_plan - transfer_plan)
    (transfer_plan_clearance_ready ?transfer_plan - transfer_plan)
    (transfer_plan_special_handling_flag ?transfer_plan - transfer_plan)
    (transfer_plan_special_handling_recorded ?transfer_plan - transfer_plan)
    (transfer_plan_final_verification_passed ?transfer_plan - transfer_plan)
    (transfer_plan_dispatch_ready ?transfer_plan - transfer_plan)
    (special_handling_tag_available ?special_handling_tag - special_handling_tag)
    (transfer_plan_special_handling_link ?transfer_plan - transfer_plan ?special_handling_tag - special_handling_tag)
    (transfer_plan_special_handling_assigned ?transfer_plan - transfer_plan)
    (transfer_plan_special_handling_ready ?transfer_plan - transfer_plan)
    (transfer_plan_operator_clearance_granted ?transfer_plan - transfer_plan)
    (passenger_profile_available ?passenger_profile - passenger_profile)
    (transfer_plan_passenger_profile_link ?transfer_plan - transfer_plan ?passenger_profile - passenger_profile)
    (assignment_preference_available ?assignment_preference - assignment_preference)
    (transfer_plan_assignment_preference_link ?transfer_plan - transfer_plan ?assignment_preference - assignment_preference)
    (vehicle_equipment_available ?vehicle_equipment_spec - vehicle_equipment_spec)
    (transfer_plan_has_equipment_spec ?transfer_plan - transfer_plan ?vehicle_equipment_spec - vehicle_equipment_spec)
    (operator_clearance_available ?operator_clearance - operator_clearance)
    (transfer_plan_operator_clearance_link ?transfer_plan - transfer_plan ?operator_clearance - operator_clearance)
    (transfer_safety_resource_available ?transfer_safety_resource - transfer_safety_resource)
    (transfer_safety_resource_link ?arrival_transfer_record - transfer_record ?transfer_safety_resource - transfer_safety_resource)
    (incoming_segment_ready_for_vehicle_selection ?incoming_segment - incoming_segment)
    (outgoing_segment_ready_for_vehicle_selection ?outgoing_segment - outgoing_segment)
    (transfer_plan_committed ?transfer_plan - transfer_plan)
  )
  (:action register_arrival_transfer_record
    :parameters (?arrival_transfer_record - transfer_record)
    :precondition
      (and
        (not
          (transfer_record_registered ?arrival_transfer_record)
        )
        (not
          (transfer_confirmed ?arrival_transfer_record)
        )
      )
    :effect (transfer_record_registered ?arrival_transfer_record)
  )
  (:action assign_vehicle_proposal_to_transfer_record
    :parameters (?arrival_transfer_record - transfer_record ?vehicle_proposal_option - vehicle_proposal_option)
    :precondition
      (and
        (transfer_record_registered ?arrival_transfer_record)
        (not
          (transfer_vehicle_proposal_attached ?arrival_transfer_record)
        )
        (transfer_vehicle_proposal_available ?vehicle_proposal_option)
      )
    :effect
      (and
        (transfer_vehicle_proposal_attached ?arrival_transfer_record)
        (transfer_vehicle_proposal_link ?arrival_transfer_record ?vehicle_proposal_option)
        (not
          (transfer_vehicle_proposal_available ?vehicle_proposal_option)
        )
      )
  )
  (:action bind_local_transport_option_to_transfer_record
    :parameters (?arrival_transfer_record - transfer_record ?local_transport_option - local_transport_option)
    :precondition
      (and
        (transfer_record_registered ?arrival_transfer_record)
        (transfer_vehicle_proposal_attached ?arrival_transfer_record)
        (transfer_local_option_available ?local_transport_option)
      )
    :effect
      (and
        (transfer_local_option_link ?arrival_transfer_record ?local_transport_option)
        (not
          (transfer_local_option_available ?local_transport_option)
        )
      )
  )
  (:action confirm_preliminary_transfer_match
    :parameters (?arrival_transfer_record - transfer_record ?local_transport_option - local_transport_option)
    :precondition
      (and
        (transfer_record_registered ?arrival_transfer_record)
        (transfer_vehicle_proposal_attached ?arrival_transfer_record)
        (transfer_local_option_link ?arrival_transfer_record ?local_transport_option)
        (not
          (transfer_allocation_ready ?arrival_transfer_record)
        )
      )
    :effect (transfer_allocation_ready ?arrival_transfer_record)
  )
  (:action release_local_transport_option_from_record
    :parameters (?arrival_transfer_record - transfer_record ?local_transport_option - local_transport_option)
    :precondition
      (and
        (transfer_local_option_link ?arrival_transfer_record ?local_transport_option)
      )
    :effect
      (and
        (transfer_local_option_available ?local_transport_option)
        (not
          (transfer_local_option_link ?arrival_transfer_record ?local_transport_option)
        )
      )
  )
  (:action assign_escort_to_transfer_record
    :parameters (?arrival_transfer_record - transfer_record ?escort_staff - escort_staff)
    :precondition
      (and
        (transfer_allocation_ready ?arrival_transfer_record)
        (transfer_escort_available ?escort_staff)
      )
    :effect
      (and
        (transfer_escort_assignment ?arrival_transfer_record ?escort_staff)
        (not
          (transfer_escort_available ?escort_staff)
        )
      )
  )
  (:action release_escort_from_transfer_record
    :parameters (?arrival_transfer_record - transfer_record ?escort_staff - escort_staff)
    :precondition
      (and
        (transfer_escort_assignment ?arrival_transfer_record ?escort_staff)
      )
    :effect
      (and
        (transfer_escort_available ?escort_staff)
        (not
          (transfer_escort_assignment ?arrival_transfer_record ?escort_staff)
        )
      )
  )
  (:action assign_equipment_spec_to_transfer_plan
    :parameters (?transfer_plan - transfer_plan ?vehicle_equipment_spec - vehicle_equipment_spec)
    :precondition
      (and
        (transfer_allocation_ready ?transfer_plan)
        (vehicle_equipment_available ?vehicle_equipment_spec)
      )
    :effect
      (and
        (transfer_plan_has_equipment_spec ?transfer_plan ?vehicle_equipment_spec)
        (not
          (vehicle_equipment_available ?vehicle_equipment_spec)
        )
      )
  )
  (:action release_equipment_spec_from_transfer_plan
    :parameters (?transfer_plan - transfer_plan ?vehicle_equipment_spec - vehicle_equipment_spec)
    :precondition
      (and
        (transfer_plan_has_equipment_spec ?transfer_plan ?vehicle_equipment_spec)
      )
    :effect
      (and
        (vehicle_equipment_available ?vehicle_equipment_spec)
        (not
          (transfer_plan_has_equipment_spec ?transfer_plan ?vehicle_equipment_spec)
        )
      )
  )
  (:action assign_operator_clearance_to_transfer_plan
    :parameters (?transfer_plan - transfer_plan ?operator_clearance - operator_clearance)
    :precondition
      (and
        (transfer_allocation_ready ?transfer_plan)
        (operator_clearance_available ?operator_clearance)
      )
    :effect
      (and
        (transfer_plan_operator_clearance_link ?transfer_plan ?operator_clearance)
        (not
          (operator_clearance_available ?operator_clearance)
        )
      )
  )
  (:action release_operator_clearance_from_transfer_plan
    :parameters (?transfer_plan - transfer_plan ?operator_clearance - operator_clearance)
    :precondition
      (and
        (transfer_plan_operator_clearance_link ?transfer_plan ?operator_clearance)
      )
    :effect
      (and
        (operator_clearance_available ?operator_clearance)
        (not
          (transfer_plan_operator_clearance_link ?transfer_plan ?operator_clearance)
        )
      )
  )
  (:action reserve_pickup_node_for_incoming_segment
    :parameters (?incoming_segment - incoming_segment ?pickup_node - pickup_node ?local_transport_option - local_transport_option)
    :precondition
      (and
        (transfer_allocation_ready ?incoming_segment)
        (transfer_local_option_link ?incoming_segment ?local_transport_option)
        (transfer_segment_pickup_node_link ?incoming_segment ?pickup_node)
        (not
          (transfer_pickup_node_reserved ?pickup_node)
        )
        (not
          (transfer_pickup_node_assist_reserved ?pickup_node)
        )
      )
    :effect (transfer_pickup_node_reserved ?pickup_node)
  )
  (:action confirm_incoming_segment_personnel_assignment
    :parameters (?incoming_segment - incoming_segment ?pickup_node - pickup_node ?escort_staff - escort_staff)
    :precondition
      (and
        (transfer_allocation_ready ?incoming_segment)
        (transfer_escort_assignment ?incoming_segment ?escort_staff)
        (transfer_segment_pickup_node_link ?incoming_segment ?pickup_node)
        (transfer_pickup_node_reserved ?pickup_node)
        (not
          (incoming_segment_ready_for_vehicle_selection ?incoming_segment)
        )
      )
    :effect
      (and
        (incoming_segment_ready_for_vehicle_selection ?incoming_segment)
        (incoming_segment_personnel_assigned ?incoming_segment)
      )
  )
  (:action assign_assistive_service_to_incoming_segment
    :parameters (?incoming_segment - incoming_segment ?pickup_node - pickup_node ?assistive_service_option - assistive_service_option)
    :precondition
      (and
        (transfer_allocation_ready ?incoming_segment)
        (transfer_segment_pickup_node_link ?incoming_segment ?pickup_node)
        (transfer_assistive_service_available ?assistive_service_option)
        (not
          (incoming_segment_ready_for_vehicle_selection ?incoming_segment)
        )
      )
    :effect
      (and
        (transfer_pickup_node_assist_reserved ?pickup_node)
        (incoming_segment_ready_for_vehicle_selection ?incoming_segment)
        (transfer_assistive_service_link_incoming ?incoming_segment ?assistive_service_option)
        (not
          (transfer_assistive_service_available ?assistive_service_option)
        )
      )
  )
  (:action confirm_incoming_assistive_service_assignment
    :parameters (?incoming_segment - incoming_segment ?pickup_node - pickup_node ?local_transport_option - local_transport_option ?assistive_service_option - assistive_service_option)
    :precondition
      (and
        (transfer_allocation_ready ?incoming_segment)
        (transfer_local_option_link ?incoming_segment ?local_transport_option)
        (transfer_segment_pickup_node_link ?incoming_segment ?pickup_node)
        (transfer_pickup_node_assist_reserved ?pickup_node)
        (transfer_assistive_service_link_incoming ?incoming_segment ?assistive_service_option)
        (not
          (incoming_segment_personnel_assigned ?incoming_segment)
        )
      )
    :effect
      (and
        (transfer_pickup_node_reserved ?pickup_node)
        (incoming_segment_personnel_assigned ?incoming_segment)
        (transfer_assistive_service_available ?assistive_service_option)
        (not
          (transfer_assistive_service_link_incoming ?incoming_segment ?assistive_service_option)
        )
      )
  )
  (:action reserve_dropoff_node_for_outgoing_segment
    :parameters (?outgoing_segment - outgoing_segment ?dropoff_node - dropoff_node ?local_transport_option - local_transport_option)
    :precondition
      (and
        (transfer_allocation_ready ?outgoing_segment)
        (transfer_local_option_link ?outgoing_segment ?local_transport_option)
        (transfer_segment_dropoff_node_link ?outgoing_segment ?dropoff_node)
        (not
          (transfer_dropoff_node_reserved ?dropoff_node)
        )
        (not
          (transfer_dropoff_node_assist_reserved ?dropoff_node)
        )
      )
    :effect (transfer_dropoff_node_reserved ?dropoff_node)
  )
  (:action confirm_outgoing_segment_personnel_assignment
    :parameters (?outgoing_segment - outgoing_segment ?dropoff_node - dropoff_node ?escort_staff - escort_staff)
    :precondition
      (and
        (transfer_allocation_ready ?outgoing_segment)
        (transfer_escort_assignment ?outgoing_segment ?escort_staff)
        (transfer_segment_dropoff_node_link ?outgoing_segment ?dropoff_node)
        (transfer_dropoff_node_reserved ?dropoff_node)
        (not
          (outgoing_segment_ready_for_vehicle_selection ?outgoing_segment)
        )
      )
    :effect
      (and
        (outgoing_segment_ready_for_vehicle_selection ?outgoing_segment)
        (outgoing_segment_personnel_assigned ?outgoing_segment)
      )
  )
  (:action assign_assistive_service_to_outgoing_segment
    :parameters (?outgoing_segment - outgoing_segment ?dropoff_node - dropoff_node ?assistive_service_option - assistive_service_option)
    :precondition
      (and
        (transfer_allocation_ready ?outgoing_segment)
        (transfer_segment_dropoff_node_link ?outgoing_segment ?dropoff_node)
        (transfer_assistive_service_available ?assistive_service_option)
        (not
          (outgoing_segment_ready_for_vehicle_selection ?outgoing_segment)
        )
      )
    :effect
      (and
        (transfer_dropoff_node_assist_reserved ?dropoff_node)
        (outgoing_segment_ready_for_vehicle_selection ?outgoing_segment)
        (transfer_assistive_service_link_outgoing ?outgoing_segment ?assistive_service_option)
        (not
          (transfer_assistive_service_available ?assistive_service_option)
        )
      )
  )
  (:action confirm_outgoing_assistive_service_assignment
    :parameters (?outgoing_segment - outgoing_segment ?dropoff_node - dropoff_node ?local_transport_option - local_transport_option ?assistive_service_option - assistive_service_option)
    :precondition
      (and
        (transfer_allocation_ready ?outgoing_segment)
        (transfer_local_option_link ?outgoing_segment ?local_transport_option)
        (transfer_segment_dropoff_node_link ?outgoing_segment ?dropoff_node)
        (transfer_dropoff_node_assist_reserved ?dropoff_node)
        (transfer_assistive_service_link_outgoing ?outgoing_segment ?assistive_service_option)
        (not
          (outgoing_segment_personnel_assigned ?outgoing_segment)
        )
      )
    :effect
      (and
        (transfer_dropoff_node_reserved ?dropoff_node)
        (outgoing_segment_personnel_assigned ?outgoing_segment)
        (transfer_assistive_service_available ?assistive_service_option)
        (not
          (transfer_assistive_service_link_outgoing ?outgoing_segment ?assistive_service_option)
        )
      )
  )
  (:action allocate_vehicle_asset_to_transfer
    :parameters (?incoming_segment - incoming_segment ?outgoing_segment - outgoing_segment ?pickup_node - pickup_node ?dropoff_node - dropoff_node ?transfer_vehicle_asset - transfer_vehicle_asset)
    :precondition
      (and
        (incoming_segment_ready_for_vehicle_selection ?incoming_segment)
        (outgoing_segment_ready_for_vehicle_selection ?outgoing_segment)
        (transfer_segment_pickup_node_link ?incoming_segment ?pickup_node)
        (transfer_segment_dropoff_node_link ?outgoing_segment ?dropoff_node)
        (transfer_pickup_node_reserved ?pickup_node)
        (transfer_dropoff_node_reserved ?dropoff_node)
        (incoming_segment_personnel_assigned ?incoming_segment)
        (outgoing_segment_personnel_assigned ?outgoing_segment)
        (transfer_vehicle_asset_available ?transfer_vehicle_asset)
      )
    :effect
      (and
        (transfer_vehicle_allocated ?transfer_vehicle_asset)
        (transfer_vehicle_pickup_node_link ?transfer_vehicle_asset ?pickup_node)
        (transfer_vehicle_dropoff_node_link ?transfer_vehicle_asset ?dropoff_node)
        (not
          (transfer_vehicle_asset_available ?transfer_vehicle_asset)
        )
      )
  )
  (:action allocate_vehicle_asset_with_pickup_assist
    :parameters (?incoming_segment - incoming_segment ?outgoing_segment - outgoing_segment ?pickup_node - pickup_node ?dropoff_node - dropoff_node ?transfer_vehicle_asset - transfer_vehicle_asset)
    :precondition
      (and
        (incoming_segment_ready_for_vehicle_selection ?incoming_segment)
        (outgoing_segment_ready_for_vehicle_selection ?outgoing_segment)
        (transfer_segment_pickup_node_link ?incoming_segment ?pickup_node)
        (transfer_segment_dropoff_node_link ?outgoing_segment ?dropoff_node)
        (transfer_pickup_node_assist_reserved ?pickup_node)
        (transfer_dropoff_node_reserved ?dropoff_node)
        (not
          (incoming_segment_personnel_assigned ?incoming_segment)
        )
        (outgoing_segment_personnel_assigned ?outgoing_segment)
        (transfer_vehicle_asset_available ?transfer_vehicle_asset)
      )
    :effect
      (and
        (transfer_vehicle_allocated ?transfer_vehicle_asset)
        (transfer_vehicle_pickup_node_link ?transfer_vehicle_asset ?pickup_node)
        (transfer_vehicle_dropoff_node_link ?transfer_vehicle_asset ?dropoff_node)
        (transfer_vehicle_approved_for_pickup ?transfer_vehicle_asset)
        (not
          (transfer_vehicle_asset_available ?transfer_vehicle_asset)
        )
      )
  )
  (:action allocate_vehicle_asset_with_dropoff_assist
    :parameters (?incoming_segment - incoming_segment ?outgoing_segment - outgoing_segment ?pickup_node - pickup_node ?dropoff_node - dropoff_node ?transfer_vehicle_asset - transfer_vehicle_asset)
    :precondition
      (and
        (incoming_segment_ready_for_vehicle_selection ?incoming_segment)
        (outgoing_segment_ready_for_vehicle_selection ?outgoing_segment)
        (transfer_segment_pickup_node_link ?incoming_segment ?pickup_node)
        (transfer_segment_dropoff_node_link ?outgoing_segment ?dropoff_node)
        (transfer_pickup_node_reserved ?pickup_node)
        (transfer_dropoff_node_assist_reserved ?dropoff_node)
        (incoming_segment_personnel_assigned ?incoming_segment)
        (not
          (outgoing_segment_personnel_assigned ?outgoing_segment)
        )
        (transfer_vehicle_asset_available ?transfer_vehicle_asset)
      )
    :effect
      (and
        (transfer_vehicle_allocated ?transfer_vehicle_asset)
        (transfer_vehicle_pickup_node_link ?transfer_vehicle_asset ?pickup_node)
        (transfer_vehicle_dropoff_node_link ?transfer_vehicle_asset ?dropoff_node)
        (transfer_vehicle_approved_for_dropoff ?transfer_vehicle_asset)
        (not
          (transfer_vehicle_asset_available ?transfer_vehicle_asset)
        )
      )
  )
  (:action allocate_vehicle_asset_with_both_assists
    :parameters (?incoming_segment - incoming_segment ?outgoing_segment - outgoing_segment ?pickup_node - pickup_node ?dropoff_node - dropoff_node ?transfer_vehicle_asset - transfer_vehicle_asset)
    :precondition
      (and
        (incoming_segment_ready_for_vehicle_selection ?incoming_segment)
        (outgoing_segment_ready_for_vehicle_selection ?outgoing_segment)
        (transfer_segment_pickup_node_link ?incoming_segment ?pickup_node)
        (transfer_segment_dropoff_node_link ?outgoing_segment ?dropoff_node)
        (transfer_pickup_node_assist_reserved ?pickup_node)
        (transfer_dropoff_node_assist_reserved ?dropoff_node)
        (not
          (incoming_segment_personnel_assigned ?incoming_segment)
        )
        (not
          (outgoing_segment_personnel_assigned ?outgoing_segment)
        )
        (transfer_vehicle_asset_available ?transfer_vehicle_asset)
      )
    :effect
      (and
        (transfer_vehicle_allocated ?transfer_vehicle_asset)
        (transfer_vehicle_pickup_node_link ?transfer_vehicle_asset ?pickup_node)
        (transfer_vehicle_dropoff_node_link ?transfer_vehicle_asset ?dropoff_node)
        (transfer_vehicle_approved_for_pickup ?transfer_vehicle_asset)
        (transfer_vehicle_approved_for_dropoff ?transfer_vehicle_asset)
        (not
          (transfer_vehicle_asset_available ?transfer_vehicle_asset)
        )
      )
  )
  (:action mark_vehicle_asset_ready_for_route_assignment
    :parameters (?transfer_vehicle_asset - transfer_vehicle_asset ?incoming_segment - incoming_segment ?local_transport_option - local_transport_option)
    :precondition
      (and
        (transfer_vehicle_allocated ?transfer_vehicle_asset)
        (incoming_segment_ready_for_vehicle_selection ?incoming_segment)
        (transfer_local_option_link ?incoming_segment ?local_transport_option)
        (not
          (transfer_vehicle_ready_for_assignment ?transfer_vehicle_asset)
        )
      )
    :effect (transfer_vehicle_ready_for_assignment ?transfer_vehicle_asset)
  )
  (:action assign_route_leg_to_transfer_plan
    :parameters (?transfer_plan - transfer_plan ?route_leg - route_leg ?transfer_vehicle_asset - transfer_vehicle_asset)
    :precondition
      (and
        (transfer_allocation_ready ?transfer_plan)
        (transfer_plan_assigned_vehicle_asset ?transfer_plan ?transfer_vehicle_asset)
        (transfer_plan_has_route_leg ?transfer_plan ?route_leg)
        (transfer_route_leg_available ?route_leg)
        (transfer_vehicle_allocated ?transfer_vehicle_asset)
        (transfer_vehicle_ready_for_assignment ?transfer_vehicle_asset)
        (not
          (transfer_route_leg_confirmed ?route_leg)
        )
      )
    :effect
      (and
        (transfer_route_leg_confirmed ?route_leg)
        (transfer_route_leg_assigned_vehicle ?route_leg ?transfer_vehicle_asset)
        (not
          (transfer_route_leg_available ?route_leg)
        )
      )
  )
  (:action lock_transfer_plan_equipment
    :parameters (?transfer_plan - transfer_plan ?route_leg - route_leg ?transfer_vehicle_asset - transfer_vehicle_asset ?local_transport_option - local_transport_option)
    :precondition
