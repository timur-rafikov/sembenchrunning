--- domain.pddl (as passed to VAL) ---
(define (domain stealth_assault_team_partitioning)
  (:requirements :strips :typing :negative-preconditions)
  (:types asset_category - object mission_item_group - object spatial_entity_group - object unit_category - object team_member - unit_category role_token - asset_category tactic - asset_category gear - asset_category secondary_objective - asset_category sync_marker - asset_category signal_marker - asset_category breach_charge - asset_category extraction_marker - asset_category utility_item - mission_item_group objective_marker - mission_item_group target_tag - mission_item_group entry_route - spatial_entity_group cover_sector - spatial_entity_group subteam - spatial_entity_group element_slot_group - team_member coordinator_slot_group - team_member assault_element - element_slot_group support_element - element_slot_group team_coordinator - coordinator_slot_group)
  (:predicates
    (member_enrolled ?member - team_member)
    (member_configured ?member - team_member)
    (member_has_role ?member - team_member)
    (member_committed ?unit - team_member)
    (member_authorized_for_execution ?unit - team_member)
    (initiator_designated ?member - team_member)
    (role_token_available ?role_token - role_token)
    (member_role_binding ?member - team_member ?role_token - role_token)
    (tactic_available ?tactic - tactic)
    (member_tactic_binding ?member - team_member ?tactic - tactic)
    (gear_available ?gear - gear)
    (member_gear_binding ?member - team_member ?gear - gear)
    (utility_available ?utility_item - utility_item)
    (assault_utility_assigned ?assault_element - assault_element ?utility_item - utility_item)
    (support_utility_assigned ?support_element - support_element ?utility_item - utility_item)
    (element_assigned_route ?assault_element - assault_element ?entry_route - entry_route)
    (route_reserved ?entry_route - entry_route)
    (route_has_utility ?entry_route - entry_route)
    (element_confirmed ?assault_element - assault_element)
    (support_assigned_sector ?support_element - support_element ?cover_sector - cover_sector)
    (sector_reserved ?cover_sector - cover_sector)
    (sector_has_utility ?cover_sector - cover_sector)
    (support_confirmed ?support_element - support_element)
    (subteam_available ?subteam - subteam)
    (subteam_formed ?subteam - subteam)
    (subteam_assigned_route ?subteam - subteam ?entry_route - entry_route)
    (subteam_assigned_sector ?subteam - subteam ?cover_sector - cover_sector)
    (subteam_route_ready ?subteam - subteam)
    (subteam_sector_ready ?subteam - subteam)
    (subteam_ready ?subteam - subteam)
    (coordinator_assigned_assault_element ?coordinator - team_coordinator ?assault_element - assault_element)
    (coordinator_assigned_support_element ?coordinator - team_coordinator ?support_element - support_element)
    (coordinator_assigned_subteam ?coordinator - team_coordinator ?subteam - subteam)
    (objective_marker_available ?objective_marker - objective_marker)
    (coordinator_objective_binding ?coordinator - team_coordinator ?objective_marker - objective_marker)
    (objective_marker_engaged ?objective_marker - objective_marker)
    (objective_marker_assigned_to_subteam ?objective_marker - objective_marker ?subteam - subteam)
    (coordinator_prepared ?coordinator - team_coordinator)
    (coordinator_loadout_configured ?coordinator - team_coordinator)
    (coordinator_sync_ready ?coordinator - team_coordinator)
    (coordinator_has_secondary_objective ?coordinator - team_coordinator)
    (coordinator_secondary_confirmed ?coordinator - team_coordinator)
    (coordinator_sync_marker_attached ?coordinator - team_coordinator)
    (coordinator_finalized ?coordinator - team_coordinator)
    (target_tag_available ?target_tag - target_tag)
    (coordinator_target_tag_binding ?coordinator - team_coordinator ?target_tag - target_tag)
    (coordinator_target_tag_active ?coordinator - team_coordinator)
    (coordinator_target_gear_ready ?coordinator - team_coordinator)
    (coordinator_target_confirmed ?coordinator - team_coordinator)
    (secondary_objective_available ?secondary_objective - secondary_objective)
    (coordinator_secondary_binding ?coordinator - team_coordinator ?secondary_objective - secondary_objective)
    (sync_marker_available ?sync_marker - sync_marker)
    (coordinator_sync_binding ?coordinator - team_coordinator ?sync_marker - sync_marker)
    (breach_charge_available ?breach_charge - breach_charge)
    (coordinator_breach_charge_binding ?coordinator - team_coordinator ?breach_charge - breach_charge)
    (extraction_marker_available ?extraction_marker - extraction_marker)
    (coordinator_extraction_marker_binding ?coordinator - team_coordinator ?extraction_marker - extraction_marker)
    (signal_marker_available ?signal_marker - signal_marker)
    (member_signal_binding ?member - team_member ?signal_marker - signal_marker)
    (assault_element_prepared ?assault_element - assault_element)
    (support_element_prepared ?support_element - support_element)
    (coordinator_authorized ?coordinator - team_coordinator)
  )
  (:action enroll_member
    :parameters (?member - team_member)
    :precondition
      (and
        (not
          (member_enrolled ?member)
        )
        (not
          (member_committed ?member)
        )
      )
    :effect (member_enrolled ?member)
  )
  (:action assign_role_to_member
    :parameters (?member - team_member ?role_token - role_token)
    :precondition
      (and
        (member_enrolled ?member)
        (not
          (member_has_role ?member)
        )
        (role_token_available ?role_token)
      )
    :effect
      (and
        (member_has_role ?member)
        (member_role_binding ?member ?role_token)
        (not
          (role_token_available ?role_token)
        )
      )
  )
  (:action assign_tactic_to_member
    :parameters (?member - team_member ?tactic - tactic)
    :precondition
      (and
        (member_enrolled ?member)
        (member_has_role ?member)
        (tactic_available ?tactic)
      )
    :effect
      (and
        (member_tactic_binding ?member ?tactic)
        (not
          (tactic_available ?tactic)
        )
      )
  )
  (:action confirm_member_configuration
    :parameters (?member - team_member ?tactic - tactic)
    :precondition
      (and
        (member_enrolled ?member)
        (member_has_role ?member)
        (member_tactic_binding ?member ?tactic)
        (not
          (member_configured ?member)
        )
      )
    :effect (member_configured ?member)
  )
  (:action remove_tactic_from_member
    :parameters (?member - team_member ?tactic - tactic)
    :precondition
      (and
        (member_tactic_binding ?member ?tactic)
      )
    :effect
      (and
        (tactic_available ?tactic)
        (not
          (member_tactic_binding ?member ?tactic)
        )
      )
  )
  (:action assign_gear_to_member
    :parameters (?member - team_member ?gear - gear)
    :precondition
      (and
        (member_configured ?member)
        (gear_available ?gear)
      )
    :effect
      (and
        (member_gear_binding ?member ?gear)
        (not
          (gear_available ?gear)
        )
      )
  )
  (:action remove_gear_from_member
    :parameters (?member - team_member ?gear - gear)
    :precondition
      (and
        (member_gear_binding ?member ?gear)
      )
    :effect
      (and
        (gear_available ?gear)
        (not
          (member_gear_binding ?member ?gear)
        )
      )
  )
  (:action coordinator_attach_breach_charge
    :parameters (?coordinator - team_coordinator ?breach_charge - breach_charge)
    :precondition
      (and
        (member_configured ?coordinator)
        (breach_charge_available ?breach_charge)
      )
    :effect
      (and
        (coordinator_breach_charge_binding ?coordinator ?breach_charge)
        (not
          (breach_charge_available ?breach_charge)
        )
      )
  )
  (:action coordinator_detach_breach_charge
    :parameters (?coordinator - team_coordinator ?breach_charge - breach_charge)
    :precondition
      (and
        (coordinator_breach_charge_binding ?coordinator ?breach_charge)
      )
    :effect
      (and
        (breach_charge_available ?breach_charge)
        (not
          (coordinator_breach_charge_binding ?coordinator ?breach_charge)
        )
      )
  )
  (:action coordinator_attach_extraction_marker
    :parameters (?coordinator - team_coordinator ?extraction_marker - extraction_marker)
    :precondition
      (and
        (member_configured ?coordinator)
        (extraction_marker_available ?extraction_marker)
      )
    :effect
      (and
        (coordinator_extraction_marker_binding ?coordinator ?extraction_marker)
        (not
          (extraction_marker_available ?extraction_marker)
        )
      )
  )
  (:action coordinator_detach_extraction_marker
    :parameters (?coordinator - team_coordinator ?extraction_marker - extraction_marker)
    :precondition
      (and
        (coordinator_extraction_marker_binding ?coordinator ?extraction_marker)
      )
    :effect
      (and
        (extraction_marker_available ?extraction_marker)
        (not
          (coordinator_extraction_marker_binding ?coordinator ?extraction_marker)
        )
      )
  )
  (:action reserve_route_for_element
    :parameters (?assault_element - assault_element ?entry_route - entry_route ?tactic - tactic)
    :precondition
      (and
        (member_configured ?assault_element)
        (member_tactic_binding ?assault_element ?tactic)
        (element_assigned_route ?assault_element ?entry_route)
        (not
          (route_reserved ?entry_route)
        )
        (not
          (route_has_utility ?entry_route)
        )
      )
    :effect (route_reserved ?entry_route)
  )
  (:action confirm_element_route_with_gear
    :parameters (?assault_element - assault_element ?entry_route - entry_route ?gear - gear)
    :precondition
      (and
        (member_configured ?assault_element)
        (member_gear_binding ?assault_element ?gear)
        (element_assigned_route ?assault_element ?entry_route)
        (route_reserved ?entry_route)
        (not
          (assault_element_prepared ?assault_element)
        )
      )
    :effect
      (and
        (assault_element_prepared ?assault_element)
        (element_confirmed ?assault_element)
      )
  )
  (:action assign_utility_to_element_route
    :parameters (?assault_element - assault_element ?entry_route - entry_route ?utility_item - utility_item)
    :precondition
      (and
        (member_configured ?assault_element)
        (element_assigned_route ?assault_element ?entry_route)
        (utility_available ?utility_item)
        (not
          (assault_element_prepared ?assault_element)
        )
      )
    :effect
      (and
        (route_has_utility ?entry_route)
        (assault_element_prepared ?assault_element)
        (assault_utility_assigned ?assault_element ?utility_item)
        (not
          (utility_available ?utility_item)
        )
      )
  )
  (:action finalize_element_route_with_utility
    :parameters (?assault_element - assault_element ?entry_route - entry_route ?tactic - tactic ?utility_item - utility_item)
    :precondition
      (and
        (member_configured ?assault_element)
        (member_tactic_binding ?assault_element ?tactic)
        (element_assigned_route ?assault_element ?entry_route)
        (route_has_utility ?entry_route)
        (assault_utility_assigned ?assault_element ?utility_item)
        (not
          (element_confirmed ?assault_element)
        )
      )
    :effect
      (and
        (route_reserved ?entry_route)
        (element_confirmed ?assault_element)
        (utility_available ?utility_item)
        (not
          (assault_utility_assigned ?assault_element ?utility_item)
        )
      )
  )
  (:action reserve_sector_for_support
    :parameters (?support_element - support_element ?cover_sector - cover_sector ?tactic - tactic)
    :precondition
      (and
        (member_configured ?support_element)
        (member_tactic_binding ?support_element ?tactic)
        (support_assigned_sector ?support_element ?cover_sector)
        (not
          (sector_reserved ?cover_sector)
        )
        (not
          (sector_has_utility ?cover_sector)
        )
      )
    :effect (sector_reserved ?cover_sector)
  )
  (:action confirm_support_sector_with_gear
    :parameters (?support_element - support_element ?cover_sector - cover_sector ?gear - gear)
    :precondition
      (and
        (member_configured ?support_element)
        (member_gear_binding ?support_element ?gear)
        (support_assigned_sector ?support_element ?cover_sector)
        (sector_reserved ?cover_sector)
        (not
          (support_element_prepared ?support_element)
        )
      )
    :effect
      (and
        (support_element_prepared ?support_element)
        (support_confirmed ?support_element)
      )
  )
  (:action assign_utility_to_support_sector
    :parameters (?support_element - support_element ?cover_sector - cover_sector ?utility_item - utility_item)
    :precondition
      (and
        (member_configured ?support_element)
        (support_assigned_sector ?support_element ?cover_sector)
        (utility_available ?utility_item)
        (not
          (support_element_prepared ?support_element)
        )
      )
    :effect
      (and
        (sector_has_utility ?cover_sector)
        (support_element_prepared ?support_element)
        (support_utility_assigned ?support_element ?utility_item)
        (not
          (utility_available ?utility_item)
        )
      )
  )
  (:action finalize_support_sector_with_utility
    :parameters (?support_element - support_element ?cover_sector - cover_sector ?tactic - tactic ?utility_item - utility_item)
    :precondition
      (and
        (member_configured ?support_element)
        (member_tactic_binding ?support_element ?tactic)
        (support_assigned_sector ?support_element ?cover_sector)
        (sector_has_utility ?cover_sector)
        (support_utility_assigned ?support_element ?utility_item)
        (not
          (support_confirmed ?support_element)
        )
      )
    :effect
      (and
        (sector_reserved ?cover_sector)
        (support_confirmed ?support_element)
        (utility_available ?utility_item)
        (not
          (support_utility_assigned ?support_element ?utility_item)
        )
      )
  )
  (:action assemble_subteam
    :parameters (?assault_element - assault_element ?support_element - support_element ?entry_route - entry_route ?cover_sector - cover_sector ?subteam - subteam)
    :precondition
      (and
        (assault_element_prepared ?assault_element)
        (support_element_prepared ?support_element)
        (element_assigned_route ?assault_element ?entry_route)
        (support_assigned_sector ?support_element ?cover_sector)
        (route_reserved ?entry_route)
        (sector_reserved ?cover_sector)
        (element_confirmed ?assault_element)
        (support_confirmed ?support_element)
        (subteam_available ?subteam)
      )
    :effect
      (and
        (subteam_formed ?subteam)
        (subteam_assigned_route ?subteam ?entry_route)
        (subteam_assigned_sector ?subteam ?cover_sector)
        (not
          (subteam_available ?subteam)
        )
      )
  )
  (:action assemble_subteam_with_route_sync
    :parameters (?assault_element - assault_element ?support_element - support_element ?entry_route - entry_route ?cover_sector - cover_sector ?subteam - subteam)
    :precondition
      (and
        (assault_element_prepared ?assault_element)
        (support_element_prepared ?support_element)
        (element_assigned_route ?assault_element ?entry_route)
        (support_assigned_sector ?support_element ?cover_sector)
        (route_has_utility ?entry_route)
        (sector_reserved ?cover_sector)
        (not
          (element_confirmed ?assault_element)
        )
        (support_confirmed ?support_element)
        (subteam_available ?subteam)
      )
    :effect
      (and
        (subteam_formed ?subteam)
        (subteam_assigned_route ?subteam ?entry_route)
        (subteam_assigned_sector ?subteam ?cover_sector)
        (subteam_route_ready ?subteam)
        (not
          (subteam_available ?subteam)
        )
      )
  )
  (:action assemble_subteam_with_sector_sync
    :parameters (?assault_element - assault_element ?support_element - support_element ?entry_route - entry_route ?cover_sector - cover_sector ?subteam - subteam)
    :precondition
      (and
        (assault_element_prepared ?assault_element)
        (support_element_prepared ?support_element)
        (element_assigned_route ?assault_element ?entry_route)
        (support_assigned_sector ?support_element ?cover_sector)
        (route_reserved ?entry_route)
        (sector_has_utility ?cover_sector)
        (element_confirmed ?assault_element)
        (not
          (support_confirmed ?support_element)
        )
        (subteam_available ?subteam)
      )
    :effect
      (and
        (subteam_formed ?subteam)
        (subteam_assigned_route ?subteam ?entry_route)
        (subteam_assigned_sector ?subteam ?cover_sector)
        (subteam_sector_ready ?subteam)
        (not
          (subteam_available ?subteam)
        )
      )
  )
  (:action assemble_subteam_full
    :parameters (?assault_element - assault_element ?support_element - support_element ?entry_route - entry_route ?cover_sector - cover_sector ?subteam - subteam)
    :precondition
      (and
        (assault_element_prepared ?assault_element)
        (support_element_prepared ?support_element)
        (element_assigned_route ?assault_element ?entry_route)
        (support_assigned_sector ?support_element ?cover_sector)
        (route_has_utility ?entry_route)
        (sector_has_utility ?cover_sector)
        (not
          (element_confirmed ?assault_element)
        )
        (not
          (support_confirmed ?support_element)
        )
        (subteam_available ?subteam)
      )
    :effect
      (and
        (subteam_formed ?subteam)
        (subteam_assigned_route ?subteam ?entry_route)
        (subteam_assigned_sector ?subteam ?cover_sector)
        (subteam_route_ready ?subteam)
        (subteam_sector_ready ?subteam)
        (not
          (subteam_available ?subteam)
        )
      )
  )
  (:action provision_subteam
    :parameters (?subteam - subteam ?assault_element - assault_element ?tactic - tactic)
    :precondition
      (and
        (subteam_formed ?subteam)
        (assault_element_prepared ?assault_element)
        (member_tactic_binding ?assault_element ?tactic)
        (not
          (subteam_ready ?subteam)
        )
      )
    :effect (subteam_ready ?subteam)
  )
  (:action attach_objective_marker_to_subteam
    :parameters (?coordinator - team_coordinator ?objective_marker - objective_marker ?subteam - subteam)
    :precondition
      (and
        (member_configured ?coordinator)
        (coordinator_assigned_subteam ?coordinator ?subteam)
        (coordinator_objective_binding ?coordinator ?objective_marker)
        (objective_marker_available ?objective_marker)
        (subteam_formed ?subteam)
        (subteam_ready ?subteam)
        (not
          (objective_marker_engaged ?objective_marker)
        )
      )
    :effect
      (and
        (objective_marker_engaged ?objective_marker)
        (objective_marker_assigned_to_subteam ?objective_marker ?subteam)
        (not
          (objective_marker_available ?objective_marker)
        )
      )
  )
  (:action validate_subteam_readiness
    :parameters (?coordinator - team_coordinator ?objective_marker - objective_marker ?subteam - subteam ?tactic - tactic)
    :precondition
      (and
        (member_configured ?coordinator)
        (coordinator_objective_binding ?coordinator ?objective_marker)
        (objective_marker_engaged ?objective_marker)
        (objective_marker_assigned_to_subteam ?objective_marker ?subteam)
        (member_tactic_binding ?coordinator ?tactic)
        (not
          (subteam_route_ready ?subteam)
        )
        (not
          (coordinator_prepared ?coordinator)
        )
      )
    :effect (coordinator_prepared ?coordinator)
  )
  (:action coordinator_claim_secondary_objective
    :parameters (?coordinator - team_coordinator ?secondary_objective - secondary_objective)
    :precondition
      (and
        (member_configured ?coordinator)
        (secondary_objective_available ?secondary_objective)
        (not
          (coordinator_has_secondary_objective ?coordinator)
        )
      )
    :effect
      (and
        (coordinator_has_secondary_objective ?coordinator)
        (coordinator_secondary_binding ?coordinator ?secondary_objective)
        (not
          (secondary_objective_available ?secondary_objective)
        )
      )
  )
  (:action coordinator_configure_loadout
    :parameters (?coordinator - team_coordinator ?objective_marker - objective_marker ?subteam - subteam ?tactic - tactic ?secondary_objective - secondary_objective)
    :precondition
      (and
        (member_configured ?coordinator)
        (coordinator_objective_binding ?coordinator ?objective_marker)
        (objective_marker_engaged ?objective_marker)
        (objective_marker_assigned_to_subteam ?objective_marker ?subteam)
        (member_tactic_binding ?coordinator ?tactic)
        (subteam_route_ready ?subteam)
        (coordinator_has_secondary_objective ?coordinator)
        (coordinator_secondary_binding ?coordinator ?secondary_objective)
        (not
          (coordinator_prepared ?coordinator)
        )
      )
    :effect
      (and
        (coordinator_prepared ?coordinator)
        (coordinator_secondary_confirmed ?coordinator)
      )
  )
  (:action coordinator_prepare_with_breach
    :parameters (?coordinator - team_coordinator ?breach_charge - breach_charge ?gear - gear ?objective_marker - objective_marker ?subteam - subteam)
    :precondition
      (and
        (coordinator_prepared ?coordinator)
        (coordinator_breach_charge_binding ?coordinator ?breach_charge)
        (member_gear_binding ?coordinator ?gear)
        (coordinator_objective_binding ?coordinator ?objective_marker)
        (objective_marker_assigned_to_subteam ?objective_marker ?subteam)
        (not
          (subteam_sector_ready ?subteam)
        )
        (not
          (coordinator_loadout_configured ?coordinator)
        )
      )
    :effect (coordinator_loadout_configured ?coordinator)
  )
  (:action coordinator_confirm_breach_loadout
    :parameters (?coordinator - team_coordinator ?breach_charge - breach_charge ?gear - gear ?objective_marker - objective_marker ?subteam - subteam)
    :precondition
      (and
        (coordinator_prepared ?coordinator)
        (coordinator_breach_charge_binding ?coordinator ?breach_charge)
        (member_gear_binding ?coordinator ?gear)
        (coordinator_objective_binding ?coordinator ?objective_marker)
        (objective_marker_assigned_to_subteam ?objective_marker ?subteam)
        (subteam_sector_ready ?subteam)
        (not
          (coordinator_loadout_configured ?coordinator)
        )
      )
    :effect (coordinator_loadout_configured ?coordinator)
  )
  (:action coordinator_activate_extraction_marker
    :parameters (?coordinator - team_coordinator ?extraction_marker - extraction_marker ?objective_marker - objective_marker ?subteam - subteam)
    :precondition
      (and
        (coordinator_loadout_configured ?coordinator)
        (coordinator_extraction_marker_binding ?coordinator ?extraction_marker)
        (coordinator_objective_binding ?coordinator ?objective_marker)
        (objective_marker_assigned_to_subteam ?objective_marker ?subteam)
        (not
          (subteam_route_ready ?subteam)
        )
        (not
          (subteam_sector_ready ?subteam)
        )
        (not
          (coordinator_sync_ready ?coordinator)
        )
      )
    :effect (coordinator_sync_ready ?coordinator)
  )
  (:action coordinator_activate_extraction_with_route_flag
    :parameters (?coordinator - team_coordinator ?extraction_marker - extraction_marker ?objective_marker - objective_marker ?subteam - subteam)
    :precondition
      (and
        (coordinator_loadout_configured ?coordinator)
        (coordinator_extraction_marker_binding ?coordinator ?extraction_marker)
        (coordinator_objective_binding ?coordinator ?objective_marker)
        (objective_marker_assigned_to_subteam ?objective_marker ?subteam)
        (subteam_route_ready ?subteam)
        (not
          (subteam_sector_ready ?subteam)
        )
        (not
          (coordinator_sync_ready ?coordinator)
        )
      )
    :effect
      (and
        (coordinator_sync_ready ?coordinator)
        (coordinator_sync_marker_attached ?coordinator)
      )
  )
  (:action coordinator_activate_extraction_with_sector_flag
    :parameters (?coordinator - team_coordinator ?extraction_marker - extraction_marker ?objective_marker - objective_marker ?subteam - subteam)
    :precondition
      (and
        (coordinator_loadout_configured ?coordinator)
        (coordinator_extraction_marker_binding ?coordinator ?extraction_marker)
        (coordinator_objective_binding ?coordinator ?objective_marker)
