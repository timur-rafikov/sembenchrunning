(define (domain low_transfer_fatigue_route_design)
  (:requirements :strips :typing :negative-preconditions)
  (:types unit_option - object stop_option - object proposal_component - object itinerary_element - object itinerary_unit - itinerary_element transport_option - unit_option time_profile - unit_option service_resource - unit_option theme_preference - unit_option traveler_preference - unit_option fatigue_profile - unit_option service_option - unit_option temporal_constraint - unit_option activity_option - stop_option accommodation_option - stop_option site_permit - stop_option transfer_in - proposal_component transfer_out - proposal_component itinerary_proposal - proposal_component segment_category - itinerary_unit stop_category - itinerary_unit origin_segment - segment_category destination_segment - segment_category stop_plan - stop_category)
  (:predicates
    (itinerary_unit_selected ?itinerary_unit - itinerary_unit)
    (itinerary_unit_confirmed ?itinerary_unit - itinerary_unit)
    (itinerary_unit_has_transport_candidate ?itinerary_unit - itinerary_unit)
    (transfer_assigned ?itinerary_unit - itinerary_unit)
    (itinerary_unit_ready ?itinerary_unit - itinerary_unit)
    (fatigue_profile_assigned ?itinerary_unit - itinerary_unit)
    (transport_available ?transport_option - transport_option)
    (itinerary_unit_assigned_transport ?itinerary_unit - itinerary_unit ?transport_option - transport_option)
    (time_profile_available ?time_profile - time_profile)
    (itinerary_unit_assigned_time_profile ?itinerary_unit - itinerary_unit ?time_profile - time_profile)
    (service_resource_available ?service_resource - service_resource)
    (itinerary_unit_assigned_service_resource ?itinerary_unit - itinerary_unit ?service_resource - service_resource)
    (activity_available ?activity_option - activity_option)
    (origin_segment_assigned_activity ?origin_segment - origin_segment ?activity_option - activity_option)
    (destination_segment_assigned_activity ?destination_segment - destination_segment ?activity_option - activity_option)
    (origin_segment_has_transfer_in ?origin_segment - origin_segment ?transfer_in - transfer_in)
    (transfer_in_selected ?transfer_in - transfer_in)
    (transfer_in_activity_marked ?transfer_in - transfer_in)
    (origin_segment_transfer_confirmed ?origin_segment - origin_segment)
    (destination_segment_has_transfer_out ?destination_segment - destination_segment ?transfer_out - transfer_out)
    (transfer_out_selected ?transfer_out - transfer_out)
    (transfer_out_activity_marked ?transfer_out - transfer_out)
    (destination_segment_transfer_confirmed ?destination_segment - destination_segment)
    (itinerary_proposal_available ?itinerary_proposal - itinerary_proposal)
    (itinerary_proposal_active ?itinerary_proposal - itinerary_proposal)
    (proposal_links_inbound ?itinerary_proposal - itinerary_proposal ?transfer_in - transfer_in)
    (proposal_links_outbound ?itinerary_proposal - itinerary_proposal ?transfer_out - transfer_out)
    (proposal_variant_inbound_activity ?itinerary_proposal - itinerary_proposal)
    (proposal_variant_outbound_activity ?itinerary_proposal - itinerary_proposal)
    (itinerary_proposal_ready ?itinerary_proposal - itinerary_proposal)
    (stop_plan_origin_segment_link ?stop_plan - stop_plan ?origin_segment - origin_segment)
    (stop_plan_destination_segment_link ?stop_plan - stop_plan ?destination_segment - destination_segment)
    (stop_plan_in_proposal ?stop_plan - stop_plan ?itinerary_proposal - itinerary_proposal)
    (accommodation_available ?accommodation_option - accommodation_option)
    (stop_plan_has_accommodation_option ?stop_plan - stop_plan ?accommodation_option - accommodation_option)
    (accommodation_reserved ?accommodation_option - accommodation_option)
    (accommodation_assigned_to_proposal ?accommodation_option - accommodation_option ?itinerary_proposal - itinerary_proposal)
    (stop_plan_basic_validated ?stop_plan - stop_plan)
    (stop_plan_services_confirmed ?stop_plan - stop_plan)
    (stop_plan_constraints_satisfied ?stop_plan - stop_plan)
    (stop_plan_theme_reserved ?stop_plan - stop_plan)
    (stop_plan_theme_validated ?stop_plan - stop_plan)
    (stop_plan_preferences_attached ?stop_plan - stop_plan)
    (stop_plan_detailed_checks_passed ?stop_plan - stop_plan)
    (site_permit_available ?site_permit - site_permit)
    (stop_plan_has_site_permit ?stop_plan - stop_plan ?site_permit - site_permit)
    (site_permit_reserved_for_stop ?stop_plan - stop_plan)
    (stop_plan_permit_validated ?stop_plan - stop_plan)
    (stop_plan_permit_finalized ?stop_plan - stop_plan)
    (theme_preference_available ?theme_preference - theme_preference)
    (stop_plan_assigned_theme_preference ?stop_plan - stop_plan ?theme_preference - theme_preference)
    (traveler_preference_available ?traveler_preference - traveler_preference)
    (stop_plan_assigned_traveler_preference ?stop_plan - stop_plan ?traveler_preference - traveler_preference)
    (service_option_available ?service_option - service_option)
    (stop_plan_assigned_service_option ?stop_plan - stop_plan ?service_option - service_option)
    (temporal_constraint_available ?temporal_constraint - temporal_constraint)
    (stop_plan_assigned_temporal_constraint ?stop_plan - stop_plan ?temporal_constraint - temporal_constraint)
    (fatigue_profile_available ?fatigue_profile - fatigue_profile)
    (itinerary_unit_assigned_fatigue_profile ?itinerary_unit - itinerary_unit ?fatigue_profile - fatigue_profile)
    (origin_segment_ready ?origin_segment - origin_segment)
    (destination_segment_ready ?destination_segment - destination_segment)
    (stop_plan_occupancy_confirmed ?stop_plan - stop_plan)
  )
  (:action select_itinerary_unit
    :parameters (?itinerary_unit - itinerary_unit)
    :precondition
      (and
        (not
          (itinerary_unit_selected ?itinerary_unit)
        )
        (not
          (transfer_assigned ?itinerary_unit)
        )
      )
    :effect (itinerary_unit_selected ?itinerary_unit)
  )
  (:action bind_transport_candidate_to_unit
    :parameters (?itinerary_unit - itinerary_unit ?transport_option - transport_option)
    :precondition
      (and
        (itinerary_unit_selected ?itinerary_unit)
        (not
          (itinerary_unit_has_transport_candidate ?itinerary_unit)
        )
        (transport_available ?transport_option)
      )
    :effect
      (and
        (itinerary_unit_has_transport_candidate ?itinerary_unit)
        (itinerary_unit_assigned_transport ?itinerary_unit ?transport_option)
        (not
          (transport_available ?transport_option)
        )
      )
  )
  (:action assign_time_profile_to_unit
    :parameters (?itinerary_unit - itinerary_unit ?time_profile - time_profile)
    :precondition
      (and
        (itinerary_unit_selected ?itinerary_unit)
        (itinerary_unit_has_transport_candidate ?itinerary_unit)
        (time_profile_available ?time_profile)
      )
    :effect
      (and
        (itinerary_unit_assigned_time_profile ?itinerary_unit ?time_profile)
        (not
          (time_profile_available ?time_profile)
        )
      )
  )
  (:action confirm_unit_time_assignment
    :parameters (?itinerary_unit - itinerary_unit ?time_profile - time_profile)
    :precondition
      (and
        (itinerary_unit_selected ?itinerary_unit)
        (itinerary_unit_has_transport_candidate ?itinerary_unit)
        (itinerary_unit_assigned_time_profile ?itinerary_unit ?time_profile)
        (not
          (itinerary_unit_confirmed ?itinerary_unit)
        )
      )
    :effect (itinerary_unit_confirmed ?itinerary_unit)
  )
  (:action release_time_profile_from_unit
    :parameters (?itinerary_unit - itinerary_unit ?time_profile - time_profile)
    :precondition
      (and
        (itinerary_unit_assigned_time_profile ?itinerary_unit ?time_profile)
      )
    :effect
      (and
        (time_profile_available ?time_profile)
        (not
          (itinerary_unit_assigned_time_profile ?itinerary_unit ?time_profile)
        )
      )
  )
  (:action assign_service_resource_to_unit
    :parameters (?itinerary_unit - itinerary_unit ?service_resource - service_resource)
    :precondition
      (and
        (itinerary_unit_confirmed ?itinerary_unit)
        (service_resource_available ?service_resource)
      )
    :effect
      (and
        (itinerary_unit_assigned_service_resource ?itinerary_unit ?service_resource)
        (not
          (service_resource_available ?service_resource)
        )
      )
  )
  (:action release_service_resource_from_unit
    :parameters (?itinerary_unit - itinerary_unit ?service_resource - service_resource)
    :precondition
      (and
        (itinerary_unit_assigned_service_resource ?itinerary_unit ?service_resource)
      )
    :effect
      (and
        (service_resource_available ?service_resource)
        (not
          (itinerary_unit_assigned_service_resource ?itinerary_unit ?service_resource)
        )
      )
  )
  (:action assign_service_option_to_stop_plan
    :parameters (?stop_plan - stop_plan ?service_option - service_option)
    :precondition
      (and
        (itinerary_unit_confirmed ?stop_plan)
        (service_option_available ?service_option)
      )
    :effect
      (and
        (stop_plan_assigned_service_option ?stop_plan ?service_option)
        (not
          (service_option_available ?service_option)
        )
      )
  )
  (:action release_service_option_from_stop_plan
    :parameters (?stop_plan - stop_plan ?service_option - service_option)
    :precondition
      (and
        (stop_plan_assigned_service_option ?stop_plan ?service_option)
      )
    :effect
      (and
        (service_option_available ?service_option)
        (not
          (stop_plan_assigned_service_option ?stop_plan ?service_option)
        )
      )
  )
  (:action assign_temporal_constraint_to_stop_plan
    :parameters (?stop_plan - stop_plan ?temporal_constraint - temporal_constraint)
    :precondition
      (and
        (itinerary_unit_confirmed ?stop_plan)
        (temporal_constraint_available ?temporal_constraint)
      )
    :effect
      (and
        (stop_plan_assigned_temporal_constraint ?stop_plan ?temporal_constraint)
        (not
          (temporal_constraint_available ?temporal_constraint)
        )
      )
  )
  (:action release_temporal_constraint_from_stop_plan
    :parameters (?stop_plan - stop_plan ?temporal_constraint - temporal_constraint)
    :precondition
      (and
        (stop_plan_assigned_temporal_constraint ?stop_plan ?temporal_constraint)
      )
    :effect
      (and
        (temporal_constraint_available ?temporal_constraint)
        (not
          (stop_plan_assigned_temporal_constraint ?stop_plan ?temporal_constraint)
        )
      )
  )
  (:action mark_inbound_transfer_candidate_for_origin_segment
    :parameters (?origin_segment - origin_segment ?transfer_in - transfer_in ?time_profile - time_profile)
    :precondition
      (and
        (itinerary_unit_confirmed ?origin_segment)
        (itinerary_unit_assigned_time_profile ?origin_segment ?time_profile)
        (origin_segment_has_transfer_in ?origin_segment ?transfer_in)
        (not
          (transfer_in_selected ?transfer_in)
        )
        (not
          (transfer_in_activity_marked ?transfer_in)
        )
      )
    :effect (transfer_in_selected ?transfer_in)
  )
  (:action confirm_inbound_transfer_with_service_for_origin_segment
    :parameters (?origin_segment - origin_segment ?transfer_in - transfer_in ?service_resource - service_resource)
    :precondition
      (and
        (itinerary_unit_confirmed ?origin_segment)
        (itinerary_unit_assigned_service_resource ?origin_segment ?service_resource)
        (origin_segment_has_transfer_in ?origin_segment ?transfer_in)
        (transfer_in_selected ?transfer_in)
        (not
          (origin_segment_ready ?origin_segment)
        )
      )
    :effect
      (and
        (origin_segment_ready ?origin_segment)
        (origin_segment_transfer_confirmed ?origin_segment)
      )
  )
  (:action assign_activity_and_mark_inbound_transfer_for_origin_segment
    :parameters (?origin_segment - origin_segment ?transfer_in - transfer_in ?activity_option - activity_option)
    :precondition
      (and
        (itinerary_unit_confirmed ?origin_segment)
        (origin_segment_has_transfer_in ?origin_segment ?transfer_in)
        (activity_available ?activity_option)
        (not
          (origin_segment_ready ?origin_segment)
        )
      )
    :effect
      (and
        (transfer_in_activity_marked ?transfer_in)
        (origin_segment_ready ?origin_segment)
        (origin_segment_assigned_activity ?origin_segment ?activity_option)
        (not
          (activity_available ?activity_option)
        )
      )
  )
  (:action confirm_activity_and_inbound_transfer_for_origin_segment
    :parameters (?origin_segment - origin_segment ?transfer_in - transfer_in ?time_profile - time_profile ?activity_option - activity_option)
    :precondition
      (and
        (itinerary_unit_confirmed ?origin_segment)
        (itinerary_unit_assigned_time_profile ?origin_segment ?time_profile)
        (origin_segment_has_transfer_in ?origin_segment ?transfer_in)
        (transfer_in_activity_marked ?transfer_in)
        (origin_segment_assigned_activity ?origin_segment ?activity_option)
        (not
          (origin_segment_transfer_confirmed ?origin_segment)
        )
      )
    :effect
      (and
        (transfer_in_selected ?transfer_in)
        (origin_segment_transfer_confirmed ?origin_segment)
        (activity_available ?activity_option)
        (not
          (origin_segment_assigned_activity ?origin_segment ?activity_option)
        )
      )
  )

  (:action confirm_outbound_transfer_with_service_for_destination_segment
    :parameters (?destination_segment - destination_segment ?transfer_out - transfer_out ?service_resource - service_resource)

      (and
        (itinerary_unit_confirmed ?destination_segment)

        (destination_segment_has_transfer_out ?destination_segment ?transfer_out)
        (transfer_out_selected ?transfer_out)
        (not
          (destination_segment_ready ?destination_segment)
        )
      )
    :effect
      (and
        (destination_segment_ready ?destination_segment)
        (destination_segment_transfer_confirmed ?destination_segment)
      )
  )
  (:action assign_activity_and_mark_outbound_transfer_for_destination_segment
    :parameters (?destination_segment - destination_segment ?transfer_out - transfer_out ?activity_option - activity_option)
    :precondition
      (and
        (itinerary_unit_confirmed ?destination_segment)
        (destination_segment_has_transfer_out ?destination_segment ?transfer_out)
        (activity_available ?activity_option)
        (not
          (destination_segment_ready ?destination_segment)
        )
      )
    :effect
      (and
        (transfer_out_activity_marked ?transfer_out)
        (destination_segment_ready ?destination_segment)
        (destination_segment_assigned_activity ?destination_segment ?activity_option)
        (not
          (activity_available ?activity_option)
        )
      )
  )
  (:action confirm_activity_and_outbound_transfer_for_destination_segment
    :parameters (?destination_segment - destination_segment ?transfer_out - transfer_out ?time_profile - time_profile ?activity_option - activity_option)
    :precondition
      (and
        (itinerary_unit_confirmed ?destination_segment)
        (itinerary_unit_assigned_time_profile ?destination_segment ?time_profile)
        (destination_segment_has_transfer_out ?destination_segment ?transfer_out)
        (transfer_out_activity_marked ?transfer_out)
        (destination_segment_assigned_activity ?destination_segment ?activity_option)
        (not
          (destination_segment_transfer_confirmed ?destination_segment)
        )
      )
    :effect
      (and
        (transfer_out_selected ?transfer_out)
        (destination_segment_transfer_confirmed ?destination_segment)
        (activity_available ?activity_option)
        (not
          (destination_segment_assigned_activity ?destination_segment ?activity_option)
        )
      )
  )
  (:action assemble_itinerary_proposal_variant
    :parameters (?origin_segment - origin_segment ?destination_segment - destination_segment ?transfer_in - transfer_in ?transfer_out - transfer_out ?itinerary_proposal - itinerary_proposal)
    :precondition
      (and
        (origin_segment_ready ?origin_segment)
        (destination_segment_ready ?destination_segment)
        (origin_segment_has_transfer_in ?origin_segment ?transfer_in)
        (destination_segment_has_transfer_out ?destination_segment ?transfer_out)
        (transfer_in_selected ?transfer_in)
        (transfer_out_selected ?transfer_out)
        (origin_segment_transfer_confirmed ?origin_segment)
        (destination_segment_transfer_confirmed ?destination_segment)
        (itinerary_proposal_available ?itinerary_proposal)
      )
    :effect
      (and
        (itinerary_proposal_active ?itinerary_proposal)
        (proposal_links_inbound ?itinerary_proposal ?transfer_in)
        (proposal_links_outbound ?itinerary_proposal ?transfer_out)
        (not
          (itinerary_proposal_available ?itinerary_proposal)
        )
      )
  )
  (:action assemble_itinerary_proposal_variant_including_inbound_activity
    :parameters (?origin_segment - origin_segment ?destination_segment - destination_segment ?transfer_in - transfer_in ?transfer_out - transfer_out ?itinerary_proposal - itinerary_proposal)
    :precondition
      (and
        (origin_segment_ready ?origin_segment)
        (destination_segment_ready ?destination_segment)
        (origin_segment_has_transfer_in ?origin_segment ?transfer_in)
        (destination_segment_has_transfer_out ?destination_segment ?transfer_out)
        (transfer_in_activity_marked ?transfer_in)
        (transfer_out_selected ?transfer_out)
        (not
          (origin_segment_transfer_confirmed ?origin_segment)
        )
        (destination_segment_transfer_confirmed ?destination_segment)
        (itinerary_proposal_available ?itinerary_proposal)
      )
    :effect
      (and
        (itinerary_proposal_active ?itinerary_proposal)
        (proposal_links_inbound ?itinerary_proposal ?transfer_in)
        (proposal_links_outbound ?itinerary_proposal ?transfer_out)
        (proposal_variant_inbound_activity ?itinerary_proposal)
        (not
          (itinerary_proposal_available ?itinerary_proposal)
        )
      )
  )
  (:action assemble_itinerary_proposal_variant_including_outbound_activity
    :parameters (?origin_segment - origin_segment ?destination_segment - destination_segment ?transfer_in - transfer_in ?transfer_out - transfer_out ?itinerary_proposal - itinerary_proposal)
    :precondition
      (and
        (origin_segment_ready ?origin_segment)
        (destination_segment_ready ?destination_segment)
        (origin_segment_has_transfer_in ?origin_segment ?transfer_in)
        (destination_segment_has_transfer_out ?destination_segment ?transfer_out)
        (transfer_in_selected ?transfer_in)
        (transfer_out_activity_marked ?transfer_out)
        (origin_segment_transfer_confirmed ?origin_segment)
        (not
          (destination_segment_transfer_confirmed ?destination_segment)
        )
        (itinerary_proposal_available ?itinerary_proposal)
      )
    :effect
      (and
        (itinerary_proposal_active ?itinerary_proposal)
        (proposal_links_inbound ?itinerary_proposal ?transfer_in)
        (proposal_links_outbound ?itinerary_proposal ?transfer_out)
        (proposal_variant_outbound_activity ?itinerary_proposal)
        (not
          (itinerary_proposal_available ?itinerary_proposal)
        )
      )
  )
  (:action assemble_itinerary_proposal_variant_including_both_activities
    :parameters (?origin_segment - origin_segment ?destination_segment - destination_segment ?transfer_in - transfer_in ?transfer_out - transfer_out ?itinerary_proposal - itinerary_proposal)
    :precondition
      (and
        (origin_segment_ready ?origin_segment)
        (destination_segment_ready ?destination_segment)
        (origin_segment_has_transfer_in ?origin_segment ?transfer_in)
        (destination_segment_has_transfer_out ?destination_segment ?transfer_out)
        (transfer_in_activity_marked ?transfer_in)
        (transfer_out_activity_marked ?transfer_out)
        (not
          (origin_segment_transfer_confirmed ?origin_segment)
        )
        (not
          (destination_segment_transfer_confirmed ?destination_segment)
        )
        (itinerary_proposal_available ?itinerary_proposal)
      )
    :effect
      (and
        (itinerary_proposal_active ?itinerary_proposal)
        (proposal_links_inbound ?itinerary_proposal ?transfer_in)
        (proposal_links_outbound ?itinerary_proposal ?transfer_out)
        (proposal_variant_inbound_activity ?itinerary_proposal)
        (proposal_variant_outbound_activity ?itinerary_proposal)
        (not
          (itinerary_proposal_available ?itinerary_proposal)
        )
      )
  )
  (:action mark_itinerary_proposal_ready
    :parameters (?itinerary_proposal - itinerary_proposal ?origin_segment - origin_segment ?time_profile - time_profile)
    :precondition
      (and
        (itinerary_proposal_active ?itinerary_proposal)
        (origin_segment_ready ?origin_segment)
        (itinerary_unit_assigned_time_profile ?origin_segment ?time_profile)
        (not
          (itinerary_proposal_ready ?itinerary_proposal)
        )
      )
    :effect (itinerary_proposal_ready ?itinerary_proposal)
  )
  (:action reserve_accommodation_for_stop_plan
    :parameters (?stop_plan - stop_plan ?accommodation_option - accommodation_option ?itinerary_proposal - itinerary_proposal)
    :precondition
      (and
        (itinerary_unit_confirmed ?stop_plan)
        (stop_plan_in_proposal ?stop_plan ?itinerary_proposal)
        (stop_plan_has_accommodation_option ?stop_plan ?accommodation_option)
        (accommodation_available ?accommodation_option)
        (itinerary_proposal_active ?itinerary_proposal)
        (itinerary_proposal_ready ?itinerary_proposal)
        (not
          (accommodation_reserved ?accommodation_option)
        )
      )
    :effect
      (and
        (accommodation_reserved ?accommodation_option)
        (accommodation_assigned_to_proposal ?accommodation_option ?itinerary_proposal)
        (not
          (accommodation_available ?accommodation_option)
        )
      )
  )
  (:action validate_stop_plan_accommodation_and_time
    :parameters (?stop_plan - stop_plan ?accommodation_option - accommodation_option ?itinerary_proposal - itinerary_proposal ?time_profile - time_profile)
    :precondition
      (and
        (itinerary_unit_confirmed ?stop_plan)
        (stop_plan_has_accommodation_option ?stop_plan ?accommodation_option)
        (accommodation_reserved ?accommodation_option)
        (accommodation_assigned_to_proposal ?accommodation_option ?itinerary_proposal)
        (itinerary_unit_assigned_time_profile ?stop_plan ?time_profile)
        (not
          (proposal_variant_inbound_activity ?itinerary_proposal)
        )
        (not
          (stop_plan_basic_validated ?stop_plan)
        )
      )
    :effect (stop_plan_basic_validated ?stop_plan)
  )
  (:action reserve_theme_for_stop_plan
    :parameters (?stop_plan - stop_plan ?theme_preference - theme_preference)
    :precondition
      (and
        (itinerary_unit_confirmed ?stop_plan)
        (theme_preference_available ?theme_preference)
        (not
          (stop_plan_theme_reserved ?stop_plan)
        )
      )
    :effect
      (and
        (stop_plan_theme_reserved ?stop_plan)
        (stop_plan_assigned_theme_preference ?stop_plan ?theme_preference)
        (not
          (theme_preference_available ?theme_preference)
        )
      )
  )
  (:action validate_stop_plan_with_theme_and_accommodation
    :parameters (?stop_plan - stop_plan ?accommodation_option - accommodation_option ?itinerary_proposal - itinerary_proposal ?time_profile - time_profile ?theme_preference - theme_preference)
    :precondition
      (and
        (itinerary_unit_confirmed ?stop_plan)
        (stop_plan_has_accommodation_option ?stop_plan ?accommodation_option)
        (accommodation_reserved ?accommodation_option)
        (accommodation_assigned_to_proposal ?accommodation_option ?itinerary_proposal)
        (itinerary_unit_assigned_time_profile ?stop_plan ?time_profile)
        (proposal_variant_inbound_activity ?itinerary_proposal)
        (stop_plan_theme_reserved ?stop_plan)
        (stop_plan_assigned_theme_preference ?stop_plan ?theme_preference)
        (not
          (stop_plan_basic_validated ?stop_plan)
        )
      )
