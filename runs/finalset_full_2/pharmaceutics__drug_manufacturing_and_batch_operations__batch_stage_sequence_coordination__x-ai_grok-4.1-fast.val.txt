--- domain.pddl (as passed to VAL) ---
(define (domain pharmaceutics_batch_stage_sequence_coordination)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_category - object asset_category - object component_category - object batch_root_type - object processing_entity - batch_root_type line_resource - resource_category operator - resource_category auxiliary_equipment - resource_category quality_document - resource_category qc_test_kit - resource_category release_document - resource_category cleaning_certificate - resource_category final_quality_approval - resource_category material_lot - asset_category container_unit - asset_category special_authorization - asset_category upstream_stage_slot - component_category downstream_stage_slot - component_category production_order - component_category stage_batch_instance_type - processing_entity stage_batch_definition_type - processing_entity upstream_stage_batch_instance - stage_batch_instance_type downstream_stage_batch_instance - stage_batch_instance_type process_stage - stage_batch_definition_type)
  (:predicates
    (batch_record_created ?manufacturing_batch - processing_entity)
    (active_for_processing ?manufacturing_batch - processing_entity)
    (line_allocation_recorded ?manufacturing_batch - processing_entity)
    (released_for_processing ?manufacturing_batch - processing_entity)
    (qa_finalized ?manufacturing_batch - processing_entity)
    (release_document_attached ?manufacturing_batch - processing_entity)
    (line_resource_available ?line_resource - line_resource)
    (allocated_to_line ?manufacturing_batch - processing_entity ?line_resource - line_resource)
    (operator_available ?operator - operator)
    (operator_assigned_to_entity ?manufacturing_batch - processing_entity ?operator - operator)
    (aux_equipment_available ?auxiliary_equipment - auxiliary_equipment)
    (aux_equipment_assigned ?manufacturing_batch - processing_entity ?auxiliary_equipment - auxiliary_equipment)
    (material_lot_available ?material_lot - material_lot)
    (material_allocated_to_upstream_instance ?upstream_stage_batch_instance - upstream_stage_batch_instance ?material_lot - material_lot)
    (material_allocated_to_downstream_instance ?downstream_stage_batch_instance - downstream_stage_batch_instance ?material_lot - material_lot)
    (upstream_instance_assigned_to_slot ?upstream_stage_batch_instance - upstream_stage_batch_instance ?upstream_stage_slot - upstream_stage_slot)
    (upstream_slot_ready ?upstream_stage_slot - upstream_stage_slot)
    (upstream_slot_occupied ?upstream_stage_slot - upstream_stage_slot)
    (upstream_instance_ready ?upstream_stage_batch_instance - upstream_stage_batch_instance)
    (downstream_instance_assigned_to_slot ?downstream_stage_batch_instance - downstream_stage_batch_instance ?downstream_stage_slot - downstream_stage_slot)
    (downstream_slot_ready ?downstream_stage_slot - downstream_stage_slot)
    (downstream_slot_occupied ?downstream_stage_slot - downstream_stage_slot)
    (downstream_instance_ready ?downstream_stage_batch_instance - downstream_stage_batch_instance)
    (production_order_slot_available ?production_order - production_order)
    (production_order_assembled ?production_order - production_order)
    (production_order_attached_to_upstream_slot ?production_order - production_order ?upstream_stage_slot - upstream_stage_slot)
    (production_order_attached_to_downstream_slot ?production_order - production_order ?downstream_stage_slot - downstream_stage_slot)
    (production_order_upstream_inspection_flag ?production_order - production_order)
    (production_order_downstream_inspection_flag ?production_order - production_order)
    (production_order_inspection_complete ?production_order - production_order)
    (stage_linked_to_upstream_instance ?stage_definition - process_stage ?upstream_stage_batch_instance - upstream_stage_batch_instance)
    (stage_linked_to_downstream_instance ?stage_definition - process_stage ?downstream_stage_batch_instance - downstream_stage_batch_instance)
    (stage_linked_to_production_order ?stage_definition - process_stage ?production_order - production_order)
    (container_available ?container_unit - container_unit)
    (stage_has_container_unit ?stage_definition - process_stage ?container_unit - container_unit)
    (container_allocated ?container_unit - container_unit)
    (container_assigned_to_order ?container_unit - container_unit ?production_order - production_order)
    (stage_container_commissioned ?stage_definition - process_stage)
    (stage_integration_complete ?stage_definition - process_stage)
    (stage_ready_for_qa ?stage_definition - process_stage)
    (quality_document_attached_to_stage ?stage_definition - process_stage)
    (stage_container_quality_verified ?stage_definition - process_stage)
    (stage_qa_evidence_attached ?stage_definition - process_stage)
    (stage_qa_signoff_performed ?stage_definition - process_stage)
    (special_authorization_available ?special_authorization - special_authorization)
    (stage_has_special_authorization ?stage_definition - process_stage ?special_authorization - special_authorization)
    (stage_special_authorized ?stage_definition - process_stage)
    (stage_authorization_confirmed ?stage_definition - process_stage)
    (stage_final_authorization_recorded ?stage_definition - process_stage)
    (quality_document_available ?quality_document - quality_document)
    (stage_quality_document_linked ?stage_definition - process_stage ?quality_document - quality_document)
    (qc_test_kit_available ?qc_test_kit - qc_test_kit)
    (stage_has_qc_test_kit ?stage_definition - process_stage ?qc_test_kit - qc_test_kit)
    (cleaning_certificate_available ?cleaning_certificate - cleaning_certificate)
    (stage_has_cleaning_certificate ?stage_definition - process_stage ?cleaning_certificate - cleaning_certificate)
    (final_quality_approval_available ?final_quality_approval - final_quality_approval)
    (stage_has_final_quality_approval ?stage_definition - process_stage ?final_quality_approval - final_quality_approval)
    (release_document_available ?release_document - release_document)
    (release_document_assigned_to_batch ?manufacturing_batch - processing_entity ?release_document - release_document)
    (upstream_instance_cleared ?upstream_stage_batch_instance - upstream_stage_batch_instance)
    (downstream_instance_cleared ?downstream_stage_batch_instance - downstream_stage_batch_instance)
    (stage_activation_recorded ?stage_definition - process_stage)
  )
  (:action create_batch_record
    :parameters (?manufacturing_batch - processing_entity)
    :precondition
      (and
        (not
          (batch_record_created ?manufacturing_batch)
        )
        (not
          (released_for_processing ?manufacturing_batch)
        )
      )
    :effect (batch_record_created ?manufacturing_batch)
  )
  (:action assign_line_resource_to_batch
    :parameters (?manufacturing_batch - processing_entity ?line_resource - line_resource)
    :precondition
      (and
        (batch_record_created ?manufacturing_batch)
        (not
          (line_allocation_recorded ?manufacturing_batch)
        )
        (line_resource_available ?line_resource)
      )
    :effect
      (and
        (line_allocation_recorded ?manufacturing_batch)
        (allocated_to_line ?manufacturing_batch ?line_resource)
        (not
          (line_resource_available ?line_resource)
        )
      )
  )
  (:action assign_operator_to_batch
    :parameters (?manufacturing_batch - processing_entity ?operator - operator)
    :precondition
      (and
        (batch_record_created ?manufacturing_batch)
        (line_allocation_recorded ?manufacturing_batch)
        (operator_available ?operator)
      )
    :effect
      (and
        (operator_assigned_to_entity ?manufacturing_batch ?operator)
        (not
          (operator_available ?operator)
        )
      )
  )
  (:action set_batch_stage_ready
    :parameters (?manufacturing_batch - processing_entity ?operator - operator)
    :precondition
      (and
        (batch_record_created ?manufacturing_batch)
        (line_allocation_recorded ?manufacturing_batch)
        (operator_assigned_to_entity ?manufacturing_batch ?operator)
        (not
          (active_for_processing ?manufacturing_batch)
        )
      )
    :effect (active_for_processing ?manufacturing_batch)
  )
  (:action release_operator_from_batch
    :parameters (?manufacturing_batch - processing_entity ?operator - operator)
    :precondition
      (and
        (operator_assigned_to_entity ?manufacturing_batch ?operator)
      )
    :effect
      (and
        (operator_available ?operator)
        (not
          (operator_assigned_to_entity ?manufacturing_batch ?operator)
        )
      )
  )
  (:action assign_auxiliary_equipment_to_batch
    :parameters (?manufacturing_batch - processing_entity ?auxiliary_equipment - auxiliary_equipment)
    :precondition
      (and
        (active_for_processing ?manufacturing_batch)
        (aux_equipment_available ?auxiliary_equipment)
      )
    :effect
      (and
        (aux_equipment_assigned ?manufacturing_batch ?auxiliary_equipment)
        (not
          (aux_equipment_available ?auxiliary_equipment)
        )
      )
  )
  (:action release_auxiliary_equipment_from_batch
    :parameters (?manufacturing_batch - processing_entity ?auxiliary_equipment - auxiliary_equipment)
    :precondition
      (and
        (aux_equipment_assigned ?manufacturing_batch ?auxiliary_equipment)
      )
    :effect
      (and
        (aux_equipment_available ?auxiliary_equipment)
        (not
          (aux_equipment_assigned ?manufacturing_batch ?auxiliary_equipment)
        )
      )
  )
  (:action attach_cleaning_certificate_to_stage
    :parameters (?stage_definition - process_stage ?cleaning_certificate - cleaning_certificate)
    :precondition
      (and
        (active_for_processing ?stage_definition)
        (cleaning_certificate_available ?cleaning_certificate)
      )
    :effect
      (and
        (stage_has_cleaning_certificate ?stage_definition ?cleaning_certificate)
        (not
          (cleaning_certificate_available ?cleaning_certificate)
        )
      )
  )
  (:action detach_cleaning_certificate_from_stage
    :parameters (?stage_definition - process_stage ?cleaning_certificate - cleaning_certificate)
    :precondition
      (and
        (stage_has_cleaning_certificate ?stage_definition ?cleaning_certificate)
      )
    :effect
      (and
        (cleaning_certificate_available ?cleaning_certificate)
        (not
          (stage_has_cleaning_certificate ?stage_definition ?cleaning_certificate)
        )
      )
  )
  (:action attach_final_quality_approval_to_stage
    :parameters (?stage_definition - process_stage ?final_quality_approval - final_quality_approval)
    :precondition
      (and
        (active_for_processing ?stage_definition)
        (final_quality_approval_available ?final_quality_approval)
      )
    :effect
      (and
        (stage_has_final_quality_approval ?stage_definition ?final_quality_approval)
        (not
          (final_quality_approval_available ?final_quality_approval)
        )
      )
  )
  (:action detach_final_quality_approval_from_stage
    :parameters (?stage_definition - process_stage ?final_quality_approval - final_quality_approval)
    :precondition
      (and
        (stage_has_final_quality_approval ?stage_definition ?final_quality_approval)
      )
    :effect
      (and
        (final_quality_approval_available ?final_quality_approval)
        (not
          (stage_has_final_quality_approval ?stage_definition ?final_quality_approval)
        )
      )
  )
  (:action mark_upstream_slot_ready
    :parameters (?upstream_stage_batch_instance - upstream_stage_batch_instance ?upstream_stage_slot - upstream_stage_slot ?operator - operator)
    :precondition
      (and
        (active_for_processing ?upstream_stage_batch_instance)
        (operator_assigned_to_entity ?upstream_stage_batch_instance ?operator)
        (upstream_instance_assigned_to_slot ?upstream_stage_batch_instance ?upstream_stage_slot)
        (not
          (upstream_slot_ready ?upstream_stage_slot)
        )
        (not
          (upstream_slot_occupied ?upstream_stage_slot)
        )
      )
    :effect (upstream_slot_ready ?upstream_stage_slot)
  )
  (:action confirm_upstream_instance
    :parameters (?upstream_stage_batch_instance - upstream_stage_batch_instance ?upstream_stage_slot - upstream_stage_slot ?auxiliary_equipment - auxiliary_equipment)
    :precondition
      (and
        (active_for_processing ?upstream_stage_batch_instance)
        (aux_equipment_assigned ?upstream_stage_batch_instance ?auxiliary_equipment)
        (upstream_instance_assigned_to_slot ?upstream_stage_batch_instance ?upstream_stage_slot)
        (upstream_slot_ready ?upstream_stage_slot)
        (not
          (upstream_instance_cleared ?upstream_stage_batch_instance)
        )
      )
    :effect
      (and
        (upstream_instance_cleared ?upstream_stage_batch_instance)
        (upstream_instance_ready ?upstream_stage_batch_instance)
      )
  )
  (:action assign_material_to_upstream_instance
    :parameters (?upstream_stage_batch_instance - upstream_stage_batch_instance ?upstream_stage_slot - upstream_stage_slot ?material_lot - material_lot)
    :precondition
      (and
        (active_for_processing ?upstream_stage_batch_instance)
        (upstream_instance_assigned_to_slot ?upstream_stage_batch_instance ?upstream_stage_slot)
        (material_lot_available ?material_lot)
        (not
          (upstream_instance_cleared ?upstream_stage_batch_instance)
        )
      )
    :effect
      (and
        (upstream_slot_occupied ?upstream_stage_slot)
        (upstream_instance_cleared ?upstream_stage_batch_instance)
        (material_allocated_to_upstream_instance ?upstream_stage_batch_instance ?material_lot)
        (not
          (material_lot_available ?material_lot)
        )
      )
  )
  (:action complete_upstream_processing
    :parameters (?upstream_stage_batch_instance - upstream_stage_batch_instance ?upstream_stage_slot - upstream_stage_slot ?operator - operator ?material_lot - material_lot)
    :precondition
      (and
        (active_for_processing ?upstream_stage_batch_instance)
        (operator_assigned_to_entity ?upstream_stage_batch_instance ?operator)
        (upstream_instance_assigned_to_slot ?upstream_stage_batch_instance ?upstream_stage_slot)
        (upstream_slot_occupied ?upstream_stage_slot)
        (material_allocated_to_upstream_instance ?upstream_stage_batch_instance ?material_lot)
        (not
          (upstream_instance_ready ?upstream_stage_batch_instance)
        )
      )
    :effect
      (and
        (upstream_slot_ready ?upstream_stage_slot)
        (upstream_instance_ready ?upstream_stage_batch_instance)
        (material_lot_available ?material_lot)
        (not
          (material_allocated_to_upstream_instance ?upstream_stage_batch_instance ?material_lot)
        )
      )
  )
  (:action mark_downstream_slot_ready
    :parameters (?downstream_stage_batch_instance - downstream_stage_batch_instance ?downstream_stage_slot - downstream_stage_slot ?operator - operator)
    :precondition
      (and
        (active_for_processing ?downstream_stage_batch_instance)
        (operator_assigned_to_entity ?downstream_stage_batch_instance ?operator)
        (downstream_instance_assigned_to_slot ?downstream_stage_batch_instance ?downstream_stage_slot)
        (not
          (downstream_slot_ready ?downstream_stage_slot)
        )
        (not
          (downstream_slot_occupied ?downstream_stage_slot)
        )
      )
    :effect (downstream_slot_ready ?downstream_stage_slot)
  )
  (:action confirm_downstream_instance
    :parameters (?downstream_stage_batch_instance - downstream_stage_batch_instance ?downstream_stage_slot - downstream_stage_slot ?auxiliary_equipment - auxiliary_equipment)
    :precondition
      (and
        (active_for_processing ?downstream_stage_batch_instance)
        (aux_equipment_assigned ?downstream_stage_batch_instance ?auxiliary_equipment)
        (downstream_instance_assigned_to_slot ?downstream_stage_batch_instance ?downstream_stage_slot)
        (downstream_slot_ready ?downstream_stage_slot)
        (not
          (downstream_instance_cleared ?downstream_stage_batch_instance)
        )
      )
    :effect
      (and
        (downstream_instance_cleared ?downstream_stage_batch_instance)
        (downstream_instance_ready ?downstream_stage_batch_instance)
      )
  )
  (:action assign_material_to_downstream_instance
    :parameters (?downstream_stage_batch_instance - downstream_stage_batch_instance ?downstream_stage_slot - downstream_stage_slot ?material_lot - material_lot)
    :precondition
      (and
        (active_for_processing ?downstream_stage_batch_instance)
        (downstream_instance_assigned_to_slot ?downstream_stage_batch_instance ?downstream_stage_slot)
        (material_lot_available ?material_lot)
        (not
          (downstream_instance_cleared ?downstream_stage_batch_instance)
        )
      )
    :effect
      (and
        (downstream_slot_occupied ?downstream_stage_slot)
        (downstream_instance_cleared ?downstream_stage_batch_instance)
        (material_allocated_to_downstream_instance ?downstream_stage_batch_instance ?material_lot)
        (not
          (material_lot_available ?material_lot)
        )
      )
  )
  (:action complete_downstream_processing
    :parameters (?downstream_stage_batch_instance - downstream_stage_batch_instance ?downstream_stage_slot - downstream_stage_slot ?operator - operator ?material_lot - material_lot)
    :precondition
      (and
        (active_for_processing ?downstream_stage_batch_instance)
        (operator_assigned_to_entity ?downstream_stage_batch_instance ?operator)
        (downstream_instance_assigned_to_slot ?downstream_stage_batch_instance ?downstream_stage_slot)
        (downstream_slot_occupied ?downstream_stage_slot)
        (material_allocated_to_downstream_instance ?downstream_stage_batch_instance ?material_lot)
        (not
          (downstream_instance_ready ?downstream_stage_batch_instance)
        )
      )
    :effect
      (and
        (downstream_slot_ready ?downstream_stage_slot)
        (downstream_instance_ready ?downstream_stage_batch_instance)
        (material_lot_available ?material_lot)
        (not
          (material_allocated_to_downstream_instance ?downstream_stage_batch_instance ?material_lot)
        )
      )
  )
  (:action assemble_production_order
    :parameters (?upstream_stage_batch_instance - upstream_stage_batch_instance ?downstream_stage_batch_instance - downstream_stage_batch_instance ?upstream_stage_slot - upstream_stage_slot ?downstream_stage_slot - downstream_stage_slot ?production_order - production_order)
    :precondition
      (and
        (upstream_instance_cleared ?upstream_stage_batch_instance)
        (downstream_instance_cleared ?downstream_stage_batch_instance)
        (upstream_instance_assigned_to_slot ?upstream_stage_batch_instance ?upstream_stage_slot)
        (downstream_instance_assigned_to_slot ?downstream_stage_batch_instance ?downstream_stage_slot)
        (upstream_slot_ready ?upstream_stage_slot)
        (downstream_slot_ready ?downstream_stage_slot)
        (upstream_instance_ready ?upstream_stage_batch_instance)
        (downstream_instance_ready ?downstream_stage_batch_instance)
        (production_order_slot_available ?production_order)
      )
    :effect
      (and
        (production_order_assembled ?production_order)
        (production_order_attached_to_upstream_slot ?production_order ?upstream_stage_slot)
        (production_order_attached_to_downstream_slot ?production_order ?downstream_stage_slot)
        (not
          (production_order_slot_available ?production_order)
        )
      )
  )
  (:action assemble_production_order_mark_upstream_inspection
    :parameters (?upstream_stage_batch_instance - upstream_stage_batch_instance ?downstream_stage_batch_instance - downstream_stage_batch_instance ?upstream_stage_slot - upstream_stage_slot ?downstream_stage_slot - downstream_stage_slot ?production_order - production_order)
    :precondition
      (and
        (upstream_instance_cleared ?upstream_stage_batch_instance)
        (downstream_instance_cleared ?downstream_stage_batch_instance)
        (upstream_instance_assigned_to_slot ?upstream_stage_batch_instance ?upstream_stage_slot)
        (downstream_instance_assigned_to_slot ?downstream_stage_batch_instance ?downstream_stage_slot)
        (upstream_slot_occupied ?upstream_stage_slot)
        (downstream_slot_ready ?downstream_stage_slot)
        (not
          (upstream_instance_ready ?upstream_stage_batch_instance)
        )
        (downstream_instance_ready ?downstream_stage_batch_instance)
        (production_order_slot_available ?production_order)
      )
    :effect
      (and
        (production_order_assembled ?production_order)
        (production_order_attached_to_upstream_slot ?production_order ?upstream_stage_slot)
        (production_order_attached_to_downstream_slot ?production_order ?downstream_stage_slot)
        (production_order_upstream_inspection_flag ?production_order)
        (not
          (production_order_slot_available ?production_order)
        )
      )
  )
  (:action assemble_production_order_mark_downstream_inspection
    :parameters (?upstream_stage_batch_instance - upstream_stage_batch_instance ?downstream_stage_batch_instance - downstream_stage_batch_instance ?upstream_stage_slot - upstream_stage_slot ?downstream_stage_slot - downstream_stage_slot ?production_order - production_order)
    :precondition
      (and
        (upstream_instance_cleared ?upstream_stage_batch_instance)
        (downstream_instance_cleared ?downstream_stage_batch_instance)
        (upstream_instance_assigned_to_slot ?upstream_stage_batch_instance ?upstream_stage_slot)
        (downstream_instance_assigned_to_slot ?downstream_stage_batch_instance ?downstream_stage_slot)
        (upstream_slot_ready ?upstream_stage_slot)
        (downstream_slot_occupied ?downstream_stage_slot)
        (upstream_instance_ready ?upstream_stage_batch_instance)
        (not
          (downstream_instance_ready ?downstream_stage_batch_instance)
        )
        (production_order_slot_available ?production_order)
      )
    :effect
      (and
        (production_order_assembled ?production_order)
        (production_order_attached_to_upstream_slot ?production_order ?upstream_stage_slot)
        (production_order_attached_to_downstream_slot ?production_order ?downstream_stage_slot)
        (production_order_downstream_inspection_flag ?production_order)
        (not
          (production_order_slot_available ?production_order)
        )
      )
  )
  (:action assemble_production_order_mark_both_inspections
    :parameters (?upstream_stage_batch_instance - upstream_stage_batch_instance ?downstream_stage_batch_instance - downstream_stage_batch_instance ?upstream_stage_slot - upstream_stage_slot ?downstream_stage_slot - downstream_stage_slot ?production_order - production_order)
    :precondition
      (and
        (upstream_instance_cleared ?upstream_stage_batch_instance)
        (downstream_instance_cleared ?downstream_stage_batch_instance)
        (upstream_instance_assigned_to_slot ?upstream_stage_batch_instance ?upstream_stage_slot)
        (downstream_instance_assigned_to_slot ?downstream_stage_batch_instance ?downstream_stage_slot)
        (upstream_slot_occupied ?upstream_stage_slot)
        (downstream_slot_occupied ?downstream_stage_slot)
        (not
          (upstream_instance_ready ?upstream_stage_batch_instance)
        )
        (not
          (downstream_instance_ready ?downstream_stage_batch_instance)
        )
        (production_order_slot_available ?production_order)
      )
    :effect
      (and
        (production_order_assembled ?production_order)
        (production_order_attached_to_upstream_slot ?production_order ?upstream_stage_slot)
        (production_order_attached_to_downstream_slot ?production_order ?downstream_stage_slot)
        (production_order_upstream_inspection_flag ?production_order)
        (production_order_downstream_inspection_flag ?production_order)
        (not
          (production_order_slot_available ?production_order)
        )
      )
  )
  (:action complete_production_order_inspection
    :parameters (?production_order - production_order ?upstream_stage_batch_instance - upstream_stage_batch_instance ?operator - operator)
    :precondition
      (and
        (production_order_assembled ?production_order)
        (upstream_instance_cleared ?upstream_stage_batch_instance)
        (operator_assigned_to_entity ?upstream_stage_batch_instance ?operator)
        (not
          (production_order_inspection_complete ?production_order)
        )
      )
    :effect (production_order_inspection_complete ?production_order)
  )
  (:action allocate_container_to_stage_and_order
    :parameters (?stage_definition - process_stage ?container_unit - container_unit ?production_order - production_order)
    :precondition
      (and
        (active_for_processing ?stage_definition)
        (stage_linked_to_production_order ?stage_definition ?production_order)
        (stage_has_container_unit ?stage_definition ?container_unit)
        (container_available ?container_unit)
        (production_order_assembled ?production_order)
        (production_order_inspection_complete ?production_order)
        (not
          (container_allocated ?container_unit)
        )
      )
    :effect
      (and
        (container_allocated ?container_unit)
        (container_assigned_to_order ?container_unit ?production_order)
        (not
          (container_available ?container_unit)
        )
      )
  )
  (:action commission_stage_container
    :parameters (?stage_definition - process_stage ?container_unit - container_unit ?production_order - production_order ?operator - operator)
    :precondition
      (and
        (active_for_processing ?stage_definition)
        (stage_has_container_unit ?stage_definition ?container_unit)
        (container_allocated ?container_unit)
        (container_assigned_to_order ?container_unit ?production_order)
        (operator_assigned_to_entity ?stage_definition ?operator)
        (not
          (production_order_upstream_inspection_flag ?production_order)
        )
        (not
          (stage_container_commissioned ?stage_definition)
        )
      )
    :effect (stage_container_commissioned ?stage_definition)
  )
  (:action attach_quality_document_to_stage
    :parameters (?stage_definition - process_stage ?quality_document - quality_document)
