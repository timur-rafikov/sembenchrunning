--- domain.pddl (as passed to VAL) ---
(define (domain travel_policy_exception_escalation)
  (:requirements :strips :typing :negative-preconditions)
  (:types policy_entity - object rule_group - object resource_group - object case_root_type - object traveler_case - case_root_type policy_rule - policy_entity document_type - policy_entity reviewer_role - policy_entity supporting_requirement - policy_entity approver_authority - policy_entity exception_token - policy_entity specialist_reviewer - policy_entity senior_reviewer - policy_entity evidence_item - rule_group document_template - rule_group escalation_category - rule_group entry_requirement_rule - resource_group fare_eligibility_rule - resource_group travel_option - resource_group travel_segment - traveler_case escalation_related_type - traveler_case origin_segment - travel_segment destination_segment - travel_segment policy_exception_request - escalation_related_type)
  (:predicates
    (case_registered ?traveler_case - traveler_case)
    (case_documents_validated ?traveler_case - traveler_case)
    (case_policy_rule_assigned_flag ?traveler_case - traveler_case)
    (ready_for_travel ?traveler_case - traveler_case)
    (approval_granted_flag ?traveler_case - traveler_case)
    (case_exception_token_assigned_flag ?traveler_case - traveler_case)
    (policy_rule_available ?policy_rule - policy_rule)
    (case_has_policy_rule ?traveler_case - traveler_case ?policy_rule - policy_rule)
    (document_type_available ?document_type - document_type)
    (case_has_document ?traveler_case - traveler_case ?document_type - document_type)
    (reviewer_available ?reviewer_role - reviewer_role)
    (case_assigned_reviewer ?traveler_case - traveler_case ?reviewer_role - reviewer_role)
    (evidence_available ?evidence_item - evidence_item)
    (origin_segment_evidence_link ?origin_segment - origin_segment ?evidence_item - evidence_item)
    (destination_segment_evidence_link ?destination_segment - destination_segment ?evidence_item - evidence_item)
    (segment_entry_requirement_link ?origin_segment - origin_segment ?entry_requirement_rule - entry_requirement_rule)
    (entry_requirement_flagged ?entry_requirement_rule - entry_requirement_rule)
    (entry_requirement_supported ?entry_requirement_rule - entry_requirement_rule)
    (origin_segment_requirement_satisfied ?origin_segment - origin_segment)
    (destination_segment_fare_rule_link ?destination_segment - destination_segment ?fare_eligibility_rule - fare_eligibility_rule)
    (fare_requirement_flagged ?fare_eligibility_rule - fare_eligibility_rule)
    (fare_requirement_supported ?fare_eligibility_rule - fare_eligibility_rule)
    (destination_segment_requirement_satisfied ?destination_segment - destination_segment)
    (travel_option_available ?travel_option - travel_option)
    (travel_option_assembled ?travel_option - travel_option)
    (travel_option_entry_requirement_link ?travel_option - travel_option ?entry_requirement_rule - entry_requirement_rule)
    (travel_option_fare_rule_link ?travel_option - travel_option ?fare_eligibility_rule - fare_eligibility_rule)
    (travel_option_origin_eligible ?travel_option - travel_option)
    (travel_option_destination_eligible ?travel_option - travel_option)
    (travel_option_eligibility_confirmed ?travel_option - travel_option)
    (request_origin_link ?policy_exception_request - policy_exception_request ?origin_segment - origin_segment)
    (request_destination_link ?policy_exception_request - policy_exception_request ?destination_segment - destination_segment)
    (request_option_link ?policy_exception_request - policy_exception_request ?travel_option - travel_option)
    (document_template_available ?document_template - document_template)
    (request_template_form_link ?policy_exception_request - policy_exception_request ?document_template - document_template)
    (document_template_reserved ?document_template - document_template)
    (template_form_bound_to_travel_option ?document_template - document_template ?travel_option - travel_option)
    (request_template_matched ?policy_exception_request - policy_exception_request)
    (request_specialist_assigned_flag ?policy_exception_request - policy_exception_request)
    (specialist_review_complete ?policy_exception_request - policy_exception_request)
    (supporting_requirement_reserved ?policy_exception_request - policy_exception_request)
    (supporting_requirement_bound ?policy_exception_request - policy_exception_request)
    (request_approver_assigned_flag ?policy_exception_request - policy_exception_request)
    (request_ready_for_finalization ?policy_exception_request - policy_exception_request)
    (escalation_category_available ?escalation_category - escalation_category)
    (request_escalation_category_link ?policy_exception_request - policy_exception_request ?escalation_category - escalation_category)
    (request_categorized ?policy_exception_request - policy_exception_request)
    (review_stage_one_complete ?policy_exception_request - policy_exception_request)
    (review_stage_two_complete ?policy_exception_request - policy_exception_request)
    (supporting_requirement_available ?supporting_requirement - supporting_requirement)
    (request_supporting_requirement_link ?policy_exception_request - policy_exception_request ?supporting_requirement - supporting_requirement)
    (approver_authority_available ?approver_authority - approver_authority)
    (request_approver_assigned ?policy_exception_request - policy_exception_request ?approver_authority - approver_authority)
    (specialist_available ?specialist_reviewer - specialist_reviewer)
    (request_specialist_assigned ?policy_exception_request - policy_exception_request ?specialist_reviewer - specialist_reviewer)
    (senior_available ?senior_reviewer - senior_reviewer)
    (request_senior_assigned ?policy_exception_request - policy_exception_request ?senior_reviewer - senior_reviewer)
    (exception_token_available ?exception_token - exception_token)
    (case_exception_token_link ?traveler_case - traveler_case ?exception_token - exception_token)
    (origin_segment_checked ?origin_segment - origin_segment)
    (destination_segment_checked ?destination_segment - destination_segment)
    (request_finalized ?policy_exception_request - policy_exception_request)
  )
  (:action create_traveler_case
    :parameters (?traveler_case - traveler_case)
    :precondition
      (and
        (not
          (case_registered ?traveler_case)
        )
        (not
          (ready_for_travel ?traveler_case)
        )
      )
    :effect (case_registered ?traveler_case)
  )
  (:action link_policy_rule_to_case
    :parameters (?traveler_case - traveler_case ?policy_rule - policy_rule)
    :precondition
      (and
        (case_registered ?traveler_case)
        (not
          (case_policy_rule_assigned_flag ?traveler_case)
        )
        (policy_rule_available ?policy_rule)
      )
    :effect
      (and
        (case_policy_rule_assigned_flag ?traveler_case)
        (case_has_policy_rule ?traveler_case ?policy_rule)
        (not
          (policy_rule_available ?policy_rule)
        )
      )
  )
  (:action attach_document_to_case
    :parameters (?traveler_case - traveler_case ?document_type - document_type)
    :precondition
      (and
        (case_registered ?traveler_case)
        (case_policy_rule_assigned_flag ?traveler_case)
        (document_type_available ?document_type)
      )
    :effect
      (and
        (case_has_document ?traveler_case ?document_type)
        (not
          (document_type_available ?document_type)
        )
      )
  )
  (:action validate_case_documents
    :parameters (?traveler_case - traveler_case ?document_type - document_type)
    :precondition
      (and
        (case_registered ?traveler_case)
        (case_policy_rule_assigned_flag ?traveler_case)
        (case_has_document ?traveler_case ?document_type)
        (not
          (case_documents_validated ?traveler_case)
        )
      )
    :effect (case_documents_validated ?traveler_case)
  )
  (:action detach_document_from_case
    :parameters (?traveler_case - traveler_case ?document_type - document_type)
    :precondition
      (and
        (case_has_document ?traveler_case ?document_type)
      )
    :effect
      (and
        (document_type_available ?document_type)
        (not
          (case_has_document ?traveler_case ?document_type)
        )
      )
  )
  (:action assign_reviewer_to_case
    :parameters (?traveler_case - traveler_case ?reviewer_role - reviewer_role)
    :precondition
      (and
        (case_documents_validated ?traveler_case)
        (reviewer_available ?reviewer_role)
      )
    :effect
      (and
        (case_assigned_reviewer ?traveler_case ?reviewer_role)
        (not
          (reviewer_available ?reviewer_role)
        )
      )
  )
  (:action release_reviewer_from_case
    :parameters (?traveler_case - traveler_case ?reviewer_role - reviewer_role)
    :precondition
      (and
        (case_assigned_reviewer ?traveler_case ?reviewer_role)
      )
    :effect
      (and
        (reviewer_available ?reviewer_role)
        (not
          (case_assigned_reviewer ?traveler_case ?reviewer_role)
        )
      )
  )
  (:action assign_specialist_to_request
    :parameters (?policy_exception_request - policy_exception_request ?specialist_reviewer - specialist_reviewer)
    :precondition
      (and
        (case_documents_validated ?policy_exception_request)
        (specialist_available ?specialist_reviewer)
      )
    :effect
      (and
        (request_specialist_assigned ?policy_exception_request ?specialist_reviewer)
        (not
          (specialist_available ?specialist_reviewer)
        )
      )
  )
  (:action release_specialist_from_request
    :parameters (?policy_exception_request - policy_exception_request ?specialist_reviewer - specialist_reviewer)
    :precondition
      (and
        (request_specialist_assigned ?policy_exception_request ?specialist_reviewer)
      )
    :effect
      (and
        (specialist_available ?specialist_reviewer)
        (not
          (request_specialist_assigned ?policy_exception_request ?specialist_reviewer)
        )
      )
  )
  (:action assign_senior_to_request
    :parameters (?policy_exception_request - policy_exception_request ?senior_reviewer - senior_reviewer)
    :precondition
      (and
        (case_documents_validated ?policy_exception_request)
        (senior_available ?senior_reviewer)
      )
    :effect
      (and
        (request_senior_assigned ?policy_exception_request ?senior_reviewer)
        (not
          (senior_available ?senior_reviewer)
        )
      )
  )
  (:action release_senior_from_request
    :parameters (?policy_exception_request - policy_exception_request ?senior_reviewer - senior_reviewer)
    :precondition
      (and
        (request_senior_assigned ?policy_exception_request ?senior_reviewer)
      )
    :effect
      (and
        (senior_available ?senior_reviewer)
        (not
          (request_senior_assigned ?policy_exception_request ?senior_reviewer)
        )
      )
  )
  (:action flag_entry_requirement
    :parameters (?origin_segment - origin_segment ?entry_requirement_rule - entry_requirement_rule ?document_type - document_type)
    :precondition
      (and
        (case_documents_validated ?origin_segment)
        (case_has_document ?origin_segment ?document_type)
        (segment_entry_requirement_link ?origin_segment ?entry_requirement_rule)
        (not
          (entry_requirement_flagged ?entry_requirement_rule)
        )
        (not
          (entry_requirement_supported ?entry_requirement_rule)
        )
      )
    :effect (entry_requirement_flagged ?entry_requirement_rule)
  )
  (:action confirm_origin_requirement_by_reviewer
    :parameters (?origin_segment - origin_segment ?entry_requirement_rule - entry_requirement_rule ?reviewer_role - reviewer_role)
    :precondition
      (and
        (case_documents_validated ?origin_segment)
        (case_assigned_reviewer ?origin_segment ?reviewer_role)
        (segment_entry_requirement_link ?origin_segment ?entry_requirement_rule)
        (entry_requirement_flagged ?entry_requirement_rule)
        (not
          (origin_segment_checked ?origin_segment)
        )
      )
    :effect
      (and
        (origin_segment_checked ?origin_segment)
        (origin_segment_requirement_satisfied ?origin_segment)
      )
  )
  (:action attach_evidence_to_origin_and_mark_supported
    :parameters (?origin_segment - origin_segment ?entry_requirement_rule - entry_requirement_rule ?evidence_item - evidence_item)
    :precondition
      (and
        (case_documents_validated ?origin_segment)
        (segment_entry_requirement_link ?origin_segment ?entry_requirement_rule)
        (evidence_available ?evidence_item)
        (not
          (origin_segment_checked ?origin_segment)
        )
      )
    :effect
      (and
        (entry_requirement_supported ?entry_requirement_rule)
        (origin_segment_checked ?origin_segment)
        (origin_segment_evidence_link ?origin_segment ?evidence_item)
        (not
          (evidence_available ?evidence_item)
        )
      )
  )
  (:action process_origin_evidence_and_update_requirement_status
    :parameters (?origin_segment - origin_segment ?entry_requirement_rule - entry_requirement_rule ?document_type - document_type ?evidence_item - evidence_item)
    :precondition
      (and
        (case_documents_validated ?origin_segment)
        (case_has_document ?origin_segment ?document_type)
        (segment_entry_requirement_link ?origin_segment ?entry_requirement_rule)
        (entry_requirement_supported ?entry_requirement_rule)
        (origin_segment_evidence_link ?origin_segment ?evidence_item)
        (not
          (origin_segment_requirement_satisfied ?origin_segment)
        )
      )
    :effect
      (and
        (entry_requirement_flagged ?entry_requirement_rule)
        (origin_segment_requirement_satisfied ?origin_segment)
        (evidence_available ?evidence_item)
        (not
          (origin_segment_evidence_link ?origin_segment ?evidence_item)
        )
      )
  )
  (:action flag_fare_requirement
    :parameters (?destination_segment - destination_segment ?fare_eligibility_rule - fare_eligibility_rule ?document_type - document_type)
    :precondition
      (and
        (case_documents_validated ?destination_segment)
        (case_has_document ?destination_segment ?document_type)
        (destination_segment_fare_rule_link ?destination_segment ?fare_eligibility_rule)
        (not
          (fare_requirement_flagged ?fare_eligibility_rule)
        )
        (not
          (fare_requirement_supported ?fare_eligibility_rule)
        )
      )
    :effect (fare_requirement_flagged ?fare_eligibility_rule)
  )
  (:action confirm_destination_requirement_by_reviewer
    :parameters (?destination_segment - destination_segment ?fare_eligibility_rule - fare_eligibility_rule ?reviewer_role - reviewer_role)
    :precondition
      (and
        (case_documents_validated ?destination_segment)
        (case_assigned_reviewer ?destination_segment ?reviewer_role)
        (destination_segment_fare_rule_link ?destination_segment ?fare_eligibility_rule)
        (fare_requirement_flagged ?fare_eligibility_rule)
        (not
          (destination_segment_checked ?destination_segment)
        )
      )
    :effect
      (and
        (destination_segment_checked ?destination_segment)
        (destination_segment_requirement_satisfied ?destination_segment)
      )
  )
  (:action attach_evidence_to_destination_and_mark_supported
    :parameters (?destination_segment - destination_segment ?fare_eligibility_rule - fare_eligibility_rule ?evidence_item - evidence_item)
    :precondition
      (and
        (case_documents_validated ?destination_segment)
        (destination_segment_fare_rule_link ?destination_segment ?fare_eligibility_rule)
        (evidence_available ?evidence_item)
        (not
          (destination_segment_checked ?destination_segment)
        )
      )
    :effect
      (and
        (fare_requirement_supported ?fare_eligibility_rule)
        (destination_segment_checked ?destination_segment)
        (destination_segment_evidence_link ?destination_segment ?evidence_item)
        (not
          (evidence_available ?evidence_item)
        )
      )
  )
  (:action process_destination_evidence_and_update_requirement_status
    :parameters (?destination_segment - destination_segment ?fare_eligibility_rule - fare_eligibility_rule ?document_type - document_type ?evidence_item - evidence_item)
    :precondition
      (and
        (case_documents_validated ?destination_segment)
        (case_has_document ?destination_segment ?document_type)
        (destination_segment_fare_rule_link ?destination_segment ?fare_eligibility_rule)
        (fare_requirement_supported ?fare_eligibility_rule)
        (destination_segment_evidence_link ?destination_segment ?evidence_item)
        (not
          (destination_segment_requirement_satisfied ?destination_segment)
        )
      )
    :effect
      (and
        (fare_requirement_flagged ?fare_eligibility_rule)
        (destination_segment_requirement_satisfied ?destination_segment)
        (evidence_available ?evidence_item)
        (not
          (destination_segment_evidence_link ?destination_segment ?evidence_item)
        )
      )
  )
  (:action create_travel_option
    :parameters (?origin_segment - origin_segment ?destination_segment - destination_segment ?entry_requirement_rule - entry_requirement_rule ?fare_eligibility_rule - fare_eligibility_rule ?travel_option - travel_option)
    :precondition
      (and
        (origin_segment_checked ?origin_segment)
        (destination_segment_checked ?destination_segment)
        (segment_entry_requirement_link ?origin_segment ?entry_requirement_rule)
        (destination_segment_fare_rule_link ?destination_segment ?fare_eligibility_rule)
        (entry_requirement_flagged ?entry_requirement_rule)
        (fare_requirement_flagged ?fare_eligibility_rule)
        (origin_segment_requirement_satisfied ?origin_segment)
        (destination_segment_requirement_satisfied ?destination_segment)
        (travel_option_available ?travel_option)
      )
    :effect
      (and
        (travel_option_assembled ?travel_option)
        (travel_option_entry_requirement_link ?travel_option ?entry_requirement_rule)
        (travel_option_fare_rule_link ?travel_option ?fare_eligibility_rule)
        (not
          (travel_option_available ?travel_option)
        )
      )
  )
  (:action create_travel_option_with_origin_support
    :parameters (?origin_segment - origin_segment ?destination_segment - destination_segment ?entry_requirement_rule - entry_requirement_rule ?fare_eligibility_rule - fare_eligibility_rule ?travel_option - travel_option)
    :precondition
      (and
        (origin_segment_checked ?origin_segment)
        (destination_segment_checked ?destination_segment)
        (segment_entry_requirement_link ?origin_segment ?entry_requirement_rule)
        (destination_segment_fare_rule_link ?destination_segment ?fare_eligibility_rule)
        (entry_requirement_supported ?entry_requirement_rule)
        (fare_requirement_flagged ?fare_eligibility_rule)
        (not
          (origin_segment_requirement_satisfied ?origin_segment)
        )
        (destination_segment_requirement_satisfied ?destination_segment)
        (travel_option_available ?travel_option)
      )
    :effect
      (and
        (travel_option_assembled ?travel_option)
        (travel_option_entry_requirement_link ?travel_option ?entry_requirement_rule)
        (travel_option_fare_rule_link ?travel_option ?fare_eligibility_rule)
        (travel_option_origin_eligible ?travel_option)
        (not
          (travel_option_available ?travel_option)
        )
      )
  )
  (:action create_travel_option_with_destination_support
    :parameters (?origin_segment - origin_segment ?destination_segment - destination_segment ?entry_requirement_rule - entry_requirement_rule ?fare_eligibility_rule - fare_eligibility_rule ?travel_option - travel_option)
    :precondition
      (and
        (origin_segment_checked ?origin_segment)
        (destination_segment_checked ?destination_segment)
        (segment_entry_requirement_link ?origin_segment ?entry_requirement_rule)
        (destination_segment_fare_rule_link ?destination_segment ?fare_eligibility_rule)
        (entry_requirement_flagged ?entry_requirement_rule)
        (fare_requirement_supported ?fare_eligibility_rule)
        (origin_segment_requirement_satisfied ?origin_segment)
        (not
          (destination_segment_requirement_satisfied ?destination_segment)
        )
        (travel_option_available ?travel_option)
      )
    :effect
      (and
        (travel_option_assembled ?travel_option)
        (travel_option_entry_requirement_link ?travel_option ?entry_requirement_rule)
        (travel_option_fare_rule_link ?travel_option ?fare_eligibility_rule)
        (travel_option_destination_eligible ?travel_option)
        (not
          (travel_option_available ?travel_option)
        )
      )
  )
  (:action create_travel_option_with_both_supports
    :parameters (?origin_segment - origin_segment ?destination_segment - destination_segment ?entry_requirement_rule - entry_requirement_rule ?fare_eligibility_rule - fare_eligibility_rule ?travel_option - travel_option)
    :precondition
      (and
        (origin_segment_checked ?origin_segment)
        (destination_segment_checked ?destination_segment)
        (segment_entry_requirement_link ?origin_segment ?entry_requirement_rule)
        (destination_segment_fare_rule_link ?destination_segment ?fare_eligibility_rule)
        (entry_requirement_supported ?entry_requirement_rule)
        (fare_requirement_supported ?fare_eligibility_rule)
        (not
          (origin_segment_requirement_satisfied ?origin_segment)
        )
        (not
          (destination_segment_requirement_satisfied ?destination_segment)
        )
        (travel_option_available ?travel_option)
      )
    :effect
      (and
        (travel_option_assembled ?travel_option)
        (travel_option_entry_requirement_link ?travel_option ?entry_requirement_rule)
        (travel_option_fare_rule_link ?travel_option ?fare_eligibility_rule)
        (travel_option_origin_eligible ?travel_option)
        (travel_option_destination_eligible ?travel_option)
        (not
          (travel_option_available ?travel_option)
        )
      )
  )
  (:action confirm_travel_option_eligibility
    :parameters (?travel_option - travel_option ?origin_segment - origin_segment ?document_type - document_type)
    :precondition
      (and
        (travel_option_assembled ?travel_option)
        (origin_segment_checked ?origin_segment)
        (case_has_document ?origin_segment ?document_type)
        (not
          (travel_option_eligibility_confirmed ?travel_option)
        )
      )
    :effect (travel_option_eligibility_confirmed ?travel_option)
  )
  (:action reserve_template_bind_to_option
    :parameters (?policy_exception_request - policy_exception_request ?document_template - document_template ?travel_option - travel_option)
    :precondition
      (and
        (case_documents_validated ?policy_exception_request)
        (request_option_link ?policy_exception_request ?travel_option)
        (request_template_form_link ?policy_exception_request ?document_template)
        (document_template_available ?document_template)
        (travel_option_assembled ?travel_option)
        (travel_option_eligibility_confirmed ?travel_option)
        (not
          (document_template_reserved ?document_template)
        )
      )
    :effect
      (and
        (document_template_reserved ?document_template)
        (template_form_bound_to_travel_option ?document_template ?travel_option)
        (not
          (document_template_available ?document_template)
        )
      )
  )
  (:action confirm_template_match_for_request
    :parameters (?policy_exception_request - policy_exception_request ?document_template - document_template ?travel_option - travel_option ?document_type - document_type)
    :precondition
      (and
        (case_documents_validated ?policy_exception_request)
        (request_template_form_link ?policy_exception_request ?document_template)
        (document_template_reserved ?document_template)
        (template_form_bound_to_travel_option ?document_template ?travel_option)
        (case_has_document ?policy_exception_request ?document_type)
        (not
          (travel_option_origin_eligible ?travel_option)
        )
        (not
          (request_template_matched ?policy_exception_request)
        )
      )
    :effect (request_template_matched ?policy_exception_request)
  )
  (:action reserve_supporting_requirement_for_request
    :parameters (?policy_exception_request - policy_exception_request ?supporting_requirement - supporting_requirement)
    :precondition
      (and
        (case_documents_validated ?policy_exception_request)
        (supporting_requirement_available ?supporting_requirement)
        (not
          (supporting_requirement_reserved ?policy_exception_request)
        )
      )
    :effect
      (and
        (supporting_requirement_reserved ?policy_exception_request)
        (request_supporting_requirement_link ?policy_exception_request ?supporting_requirement)
        (not
          (supporting_requirement_available ?supporting_requirement)
        )
      )
  )
  (:action bind_supporting_requirement_to_request
    :parameters (?policy_exception_request - policy_exception_request ?document_template - document_template ?travel_option - travel_option ?document_type - document_type ?supporting_requirement - supporting_requirement)
    :precondition
      (and
        (case_documents_validated ?policy_exception_request)
        (request_template_form_link ?policy_exception_request ?document_template)
        (document_template_reserved ?document_template)
        (template_form_bound_to_travel_option ?document_template ?travel_option)
        (case_has_document ?policy_exception_request ?document_type)
        (travel_option_origin_eligible ?travel_option)
        (supporting_requirement_reserved ?policy_exception_request)
        (request_supporting_requirement_link ?policy_exception_request ?supporting_requirement)
        (not
          (request_template_matched ?policy_exception_request)
        )
      )
    :effect
      (and
        (request_template_matched ?policy_exception_request)
        (supporting_requirement_bound ?policy_exception_request)
      )
  )
  (:action mark_specialist_review_complete_initial
    :parameters (?policy_exception_request - policy_exception_request ?specialist_reviewer - specialist_reviewer ?reviewer_role - reviewer_role ?document_template - document_template ?travel_option - travel_option)
    :precondition
