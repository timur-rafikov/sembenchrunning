--- domain.pddl (as passed to VAL) ---
(define (domain animal_group_reassignment_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types grouping_category - object resource_category - object staff_category - object animal_root - object animal - animal_root pen - grouping_category handling_station - grouping_category handler - grouping_category identification_tag - grouping_category vaccination_dose - grouping_category movement_document - grouping_category sedation_dose - grouping_category pregnancy_test - grouping_category feed_ration - resource_category load_crate - resource_category breeding_record - resource_category movement_slot - staff_category reception_slot - staff_category transport_vehicle - staff_category group_role - animal group_subrole - animal source_group - group_role destination_group - group_role reassignment_task - group_subrole)
  (:predicates
    (animal_enrolled ?animal - animal)
    (entity_in_reassignment_workflow ?animal - animal)
    (allocated_to_pen ?animal - animal)
    (entity_reassignment_completed ?animal - animal)
    (entity_cleared_for_movement ?animal - animal)
    (entity_movement_authorized ?animal - animal)
    (pen_available ?pen - pen)
    (entity_assigned_to_pen ?animal - animal ?pen - pen)
    (handling_station_available ?handling_station - handling_station)
    (entity_assigned_to_handling_station ?animal - animal ?handling_station - handling_station)
    (handler_available ?handler - handler)
    (entity_assigned_handler ?animal - animal ?handler - handler)
    (feed_ration_available ?feed_ration - feed_ration)
    (group_feed_allocated ?source_group - source_group ?feed_ration - feed_ration)
    (destgroup_feed_allocated ?destination_group - destination_group ?feed_ration - feed_ration)
    (group_movement_slot_assigned ?source_group - source_group ?movement_slot - movement_slot)
    (movement_slot_reserved ?movement_slot - movement_slot)
    (movement_slot_on_hold ?movement_slot - movement_slot)
    (group_ready_for_loading ?source_group - source_group)
    (destgroup_reception_slot_assigned ?destination_group - destination_group ?reception_slot - reception_slot)
    (reception_slot_reserved ?reception_slot - reception_slot)
    (reception_slot_on_hold ?reception_slot - reception_slot)
    (destgroup_ready_for_reception ?destination_group - destination_group)
    (vehicle_available ?transport_vehicle - transport_vehicle)
    (vehicle_prepared ?transport_vehicle - transport_vehicle)
    (vehicle_assigned_movement_slot ?transport_vehicle - transport_vehicle ?movement_slot - movement_slot)
    (vehicle_assigned_reception_slot ?transport_vehicle - transport_vehicle ?reception_slot - reception_slot)
    (vehicle_flag_restricted_loading ?transport_vehicle - transport_vehicle)
    (vehicle_flag_extended_loading ?transport_vehicle - transport_vehicle)
    (vehicle_loaded ?transport_vehicle - transport_vehicle)
    (task_source_group ?reassignment_task - reassignment_task ?source_group - source_group)
    (task_destination_group ?reassignment_task - reassignment_task ?destination_group - destination_group)
    (task_assigned_vehicle ?reassignment_task - reassignment_task ?transport_vehicle - transport_vehicle)
    (crate_available ?load_crate - load_crate)
    (task_has_crate ?reassignment_task - reassignment_task ?load_crate - load_crate)
    (crate_validated ?load_crate - load_crate)
    (crate_assigned_vehicle ?load_crate - load_crate ?transport_vehicle - transport_vehicle)
    (task_prechecks_passed ?reassignment_task - reassignment_task)
    (task_confirmed_for_assessment ?reassignment_task - reassignment_task)
    (task_checks_complete ?reassignment_task - reassignment_task)
    (task_identification_tag_attached ?reassignment_task - reassignment_task)
    (task_tag_verified ?reassignment_task - reassignment_task)
    (vaccination_required ?reassignment_task - reassignment_task)
    (task_ready_for_batch_execution ?reassignment_task - reassignment_task)
    (breeding_record_present ?breeding_record - breeding_record)
    (task_has_breeding_record ?reassignment_task - reassignment_task ?breeding_record - breeding_record)
    (task_breeding_check_initiated ?reassignment_task - reassignment_task)
    (task_breeding_check_pending ?reassignment_task - reassignment_task)
    (task_breeding_check_passed ?reassignment_task - reassignment_task)
    (identification_tag_available ?identification_tag - identification_tag)
    (task_identification_tag_link ?reassignment_task - reassignment_task ?identification_tag - identification_tag)
    (vaccination_dose_available ?vaccination_dose - vaccination_dose)
    (task_vaccination_allocated ?reassignment_task - reassignment_task ?vaccination_dose - vaccination_dose)
    (sedation_dose_available ?sedation_dose - sedation_dose)
    (task_sedation_allocated ?reassignment_task - reassignment_task ?sedation_dose - sedation_dose)
    (pregnancy_test_available ?pregnancy_test - pregnancy_test)
    (task_pregnancy_test_allocated ?reassignment_task - reassignment_task ?pregnancy_test - pregnancy_test)
    (movement_document_available ?movement_document - movement_document)
    (entity_movement_document_attached ?animal - animal ?movement_document - movement_document)
    (source_group_preparation_complete ?source_group - source_group)
    (destination_group_preparation_complete ?destination_group - destination_group)
    (task_finalized ?reassignment_task - reassignment_task)
  )
  (:action enroll_animal
    :parameters (?animal - animal)
    :precondition
      (and
        (not
          (animal_enrolled ?animal)
        )
        (not
          (entity_reassignment_completed ?animal)
        )
      )
    :effect (animal_enrolled ?animal)
  )
  (:action assign_animal_to_pen
    :parameters (?animal - animal ?pen - pen)
    :precondition
      (and
        (animal_enrolled ?animal)
        (not
          (allocated_to_pen ?animal)
        )
        (pen_available ?pen)
      )
    :effect
      (and
        (allocated_to_pen ?animal)
        (entity_assigned_to_pen ?animal ?pen)
        (not
          (pen_available ?pen)
        )
      )
  )
  (:action assign_animal_to_handling_station
    :parameters (?animal - animal ?handling_station - handling_station)
    :precondition
      (and
        (animal_enrolled ?animal)
        (allocated_to_pen ?animal)
        (handling_station_available ?handling_station)
      )
    :effect
      (and
        (entity_assigned_to_handling_station ?animal ?handling_station)
        (not
          (handling_station_available ?handling_station)
        )
      )
  )
  (:action finalize_animal_enrollment
    :parameters (?animal - animal ?handling_station - handling_station)
    :precondition
      (and
        (animal_enrolled ?animal)
        (allocated_to_pen ?animal)
        (entity_assigned_to_handling_station ?animal ?handling_station)
        (not
          (entity_in_reassignment_workflow ?animal)
        )
      )
    :effect (entity_in_reassignment_workflow ?animal)
  )
  (:action release_handling_station
    :parameters (?animal - animal ?handling_station - handling_station)
    :precondition
      (and
        (entity_assigned_to_handling_station ?animal ?handling_station)
      )
    :effect
      (and
        (handling_station_available ?handling_station)
        (not
          (entity_assigned_to_handling_station ?animal ?handling_station)
        )
      )
  )
  (:action assign_handler_to_animal
    :parameters (?animal - animal ?handler - handler)
    :precondition
      (and
        (entity_in_reassignment_workflow ?animal)
        (handler_available ?handler)
      )
    :effect
      (and
        (entity_assigned_handler ?animal ?handler)
        (not
          (handler_available ?handler)
        )
      )
  )
  (:action release_handler_from_animal
    :parameters (?animal - animal ?handler - handler)
    :precondition
      (and
        (entity_assigned_handler ?animal ?handler)
      )
    :effect
      (and
        (handler_available ?handler)
        (not
          (entity_assigned_handler ?animal ?handler)
        )
      )
  )
  (:action assign_sedation_to_task
    :parameters (?reassignment_task - reassignment_task ?sedation_dose - sedation_dose)
    :precondition
      (and
        (entity_in_reassignment_workflow ?reassignment_task)
        (sedation_dose_available ?sedation_dose)
      )
    :effect
      (and
        (task_sedation_allocated ?reassignment_task ?sedation_dose)
        (not
          (sedation_dose_available ?sedation_dose)
        )
      )
  )
  (:action release_sedation_from_task
    :parameters (?reassignment_task - reassignment_task ?sedation_dose - sedation_dose)
    :precondition
      (and
        (task_sedation_allocated ?reassignment_task ?sedation_dose)
      )
    :effect
      (and
        (sedation_dose_available ?sedation_dose)
        (not
          (task_sedation_allocated ?reassignment_task ?sedation_dose)
        )
      )
  )
  (:action assign_pregnancy_test
    :parameters (?reassignment_task - reassignment_task ?pregnancy_test - pregnancy_test)
    :precondition
      (and
        (entity_in_reassignment_workflow ?reassignment_task)
        (pregnancy_test_available ?pregnancy_test)
      )
    :effect
      (and
        (task_pregnancy_test_allocated ?reassignment_task ?pregnancy_test)
        (not
          (pregnancy_test_available ?pregnancy_test)
        )
      )
  )
  (:action release_pregnancy_test
    :parameters (?reassignment_task - reassignment_task ?pregnancy_test - pregnancy_test)
    :precondition
      (and
        (task_pregnancy_test_allocated ?reassignment_task ?pregnancy_test)
      )
    :effect
      (and
        (pregnancy_test_available ?pregnancy_test)
        (not
          (task_pregnancy_test_allocated ?reassignment_task ?pregnancy_test)
        )
      )
  )
  (:action reserve_movement_slot
    :parameters (?source_group - source_group ?movement_slot - movement_slot ?handling_station - handling_station)
    :precondition
      (and
        (entity_in_reassignment_workflow ?source_group)
        (entity_assigned_to_handling_station ?source_group ?handling_station)
        (group_movement_slot_assigned ?source_group ?movement_slot)
        (not
          (movement_slot_reserved ?movement_slot)
        )
        (not
          (movement_slot_on_hold ?movement_slot)
        )
      )
    :effect (movement_slot_reserved ?movement_slot)
  )
  (:action prepare_source_group_with_handler
    :parameters (?source_group - source_group ?movement_slot - movement_slot ?handler - handler)
    :precondition
      (and
        (entity_in_reassignment_workflow ?source_group)
        (entity_assigned_handler ?source_group ?handler)
        (group_movement_slot_assigned ?source_group ?movement_slot)
        (movement_slot_reserved ?movement_slot)
        (not
          (source_group_preparation_complete ?source_group)
        )
      )
    :effect
      (and
        (source_group_preparation_complete ?source_group)
        (group_ready_for_loading ?source_group)
      )
  )
  (:action allocate_feed_to_group
    :parameters (?source_group - source_group ?movement_slot - movement_slot ?feed_ration - feed_ration)
    :precondition
      (and
        (entity_in_reassignment_workflow ?source_group)
        (group_movement_slot_assigned ?source_group ?movement_slot)
        (feed_ration_available ?feed_ration)
        (not
          (source_group_preparation_complete ?source_group)
        )
      )
    :effect
      (and
        (movement_slot_on_hold ?movement_slot)
        (source_group_preparation_complete ?source_group)
        (group_feed_allocated ?source_group ?feed_ration)
        (not
          (feed_ration_available ?feed_ration)
        )
      )
  )
  (:action finalize_group_preparation_with_feed
    :parameters (?source_group - source_group ?movement_slot - movement_slot ?handling_station - handling_station ?feed_ration - feed_ration)
    :precondition
      (and
        (entity_in_reassignment_workflow ?source_group)
        (entity_assigned_to_handling_station ?source_group ?handling_station)
        (group_movement_slot_assigned ?source_group ?movement_slot)
        (movement_slot_on_hold ?movement_slot)
        (group_feed_allocated ?source_group ?feed_ration)
        (not
          (group_ready_for_loading ?source_group)
        )
      )
    :effect
      (and
        (movement_slot_reserved ?movement_slot)
        (group_ready_for_loading ?source_group)
        (feed_ration_available ?feed_ration)
        (not
          (group_feed_allocated ?source_group ?feed_ration)
        )
      )
  )
  (:action reserve_reception_slot
    :parameters (?destination_group - destination_group ?reception_slot - reception_slot ?handling_station - handling_station)
    :precondition
      (and
        (entity_in_reassignment_workflow ?destination_group)
        (entity_assigned_to_handling_station ?destination_group ?handling_station)
        (destgroup_reception_slot_assigned ?destination_group ?reception_slot)
        (not
          (reception_slot_reserved ?reception_slot)
        )
        (not
          (reception_slot_on_hold ?reception_slot)
        )
      )
    :effect (reception_slot_reserved ?reception_slot)
  )
  (:action prepare_destination_group_with_handler
    :parameters (?destination_group - destination_group ?reception_slot - reception_slot ?handler - handler)
    :precondition
      (and
        (entity_in_reassignment_workflow ?destination_group)
        (entity_assigned_handler ?destination_group ?handler)
        (destgroup_reception_slot_assigned ?destination_group ?reception_slot)
        (reception_slot_reserved ?reception_slot)
        (not
          (destination_group_preparation_complete ?destination_group)
        )
      )
    :effect
      (and
        (destination_group_preparation_complete ?destination_group)
        (destgroup_ready_for_reception ?destination_group)
      )
  )
  (:action allocate_feed_to_destination_group
    :parameters (?destination_group - destination_group ?reception_slot - reception_slot ?feed_ration - feed_ration)
    :precondition
      (and
        (entity_in_reassignment_workflow ?destination_group)
        (destgroup_reception_slot_assigned ?destination_group ?reception_slot)
        (feed_ration_available ?feed_ration)
        (not
          (destination_group_preparation_complete ?destination_group)
        )
      )
    :effect
      (and
        (reception_slot_on_hold ?reception_slot)
        (destination_group_preparation_complete ?destination_group)
        (destgroup_feed_allocated ?destination_group ?feed_ration)
        (not
          (feed_ration_available ?feed_ration)
        )
      )
  )
  (:action finalize_destination_group_preparation_with_feed
    :parameters (?destination_group - destination_group ?reception_slot - reception_slot ?handling_station - handling_station ?feed_ration - feed_ration)
    :precondition
      (and
        (entity_in_reassignment_workflow ?destination_group)
        (entity_assigned_to_handling_station ?destination_group ?handling_station)
        (destgroup_reception_slot_assigned ?destination_group ?reception_slot)
        (reception_slot_on_hold ?reception_slot)
        (destgroup_feed_allocated ?destination_group ?feed_ration)
        (not
          (destgroup_ready_for_reception ?destination_group)
        )
      )
    :effect
      (and
        (reception_slot_reserved ?reception_slot)
        (destgroup_ready_for_reception ?destination_group)
        (feed_ration_available ?feed_ration)
        (not
          (destgroup_feed_allocated ?destination_group ?feed_ration)
        )
      )
  )
  (:action prepare_transport_vehicle_for_task
    :parameters (?source_group - source_group ?destination_group - destination_group ?movement_slot - movement_slot ?reception_slot - reception_slot ?transport_vehicle - transport_vehicle)
    :precondition
      (and
        (source_group_preparation_complete ?source_group)
        (destination_group_preparation_complete ?destination_group)
        (group_movement_slot_assigned ?source_group ?movement_slot)
        (destgroup_reception_slot_assigned ?destination_group ?reception_slot)
        (movement_slot_reserved ?movement_slot)
        (reception_slot_reserved ?reception_slot)
        (group_ready_for_loading ?source_group)
        (destgroup_ready_for_reception ?destination_group)
        (vehicle_available ?transport_vehicle)
      )
    :effect
      (and
        (vehicle_prepared ?transport_vehicle)
        (vehicle_assigned_movement_slot ?transport_vehicle ?movement_slot)
        (vehicle_assigned_reception_slot ?transport_vehicle ?reception_slot)
        (not
          (vehicle_available ?transport_vehicle)
        )
      )
  )
  (:action prepare_vehicle_for_restricted_loading
    :parameters (?source_group - source_group ?destination_group - destination_group ?movement_slot - movement_slot ?reception_slot - reception_slot ?transport_vehicle - transport_vehicle)
    :precondition
      (and
        (source_group_preparation_complete ?source_group)
        (destination_group_preparation_complete ?destination_group)
        (group_movement_slot_assigned ?source_group ?movement_slot)
        (destgroup_reception_slot_assigned ?destination_group ?reception_slot)
        (movement_slot_on_hold ?movement_slot)
        (reception_slot_reserved ?reception_slot)
        (not
          (group_ready_for_loading ?source_group)
        )
        (destgroup_ready_for_reception ?destination_group)
        (vehicle_available ?transport_vehicle)
      )
    :effect
      (and
        (vehicle_prepared ?transport_vehicle)
        (vehicle_assigned_movement_slot ?transport_vehicle ?movement_slot)
        (vehicle_assigned_reception_slot ?transport_vehicle ?reception_slot)
        (vehicle_flag_restricted_loading ?transport_vehicle)
        (not
          (vehicle_available ?transport_vehicle)
        )
      )
  )
  (:action prepare_vehicle_for_extended_loading
    :parameters (?source_group - source_group ?destination_group - destination_group ?movement_slot - movement_slot ?reception_slot - reception_slot ?transport_vehicle - transport_vehicle)
    :precondition
      (and
        (source_group_preparation_complete ?source_group)
        (destination_group_preparation_complete ?destination_group)
        (group_movement_slot_assigned ?source_group ?movement_slot)
        (destgroup_reception_slot_assigned ?destination_group ?reception_slot)
        (movement_slot_reserved ?movement_slot)
        (reception_slot_on_hold ?reception_slot)
        (group_ready_for_loading ?source_group)
        (not
          (destgroup_ready_for_reception ?destination_group)
        )
        (vehicle_available ?transport_vehicle)
      )
    :effect
      (and
        (vehicle_prepared ?transport_vehicle)
        (vehicle_assigned_movement_slot ?transport_vehicle ?movement_slot)
        (vehicle_assigned_reception_slot ?transport_vehicle ?reception_slot)
        (vehicle_flag_extended_loading ?transport_vehicle)
        (not
          (vehicle_available ?transport_vehicle)
        )
      )
  )
  (:action prepare_transport_vehicle_comprehensive_preparation
    :parameters (?source_group - source_group ?destination_group - destination_group ?movement_slot - movement_slot ?reception_slot - reception_slot ?transport_vehicle - transport_vehicle)
    :precondition
      (and
        (source_group_preparation_complete ?source_group)
        (destination_group_preparation_complete ?destination_group)
        (group_movement_slot_assigned ?source_group ?movement_slot)
        (destgroup_reception_slot_assigned ?destination_group ?reception_slot)
        (movement_slot_on_hold ?movement_slot)
        (reception_slot_on_hold ?reception_slot)
        (not
          (group_ready_for_loading ?source_group)
        )
        (not
          (destgroup_ready_for_reception ?destination_group)
        )
        (vehicle_available ?transport_vehicle)
      )
    :effect
      (and
        (vehicle_prepared ?transport_vehicle)
        (vehicle_assigned_movement_slot ?transport_vehicle ?movement_slot)
        (vehicle_assigned_reception_slot ?transport_vehicle ?reception_slot)
        (vehicle_flag_restricted_loading ?transport_vehicle)
        (vehicle_flag_extended_loading ?transport_vehicle)
        (not
          (vehicle_available ?transport_vehicle)
        )
      )
  )
  (:action confirm_vehicle_load_ready
    :parameters (?transport_vehicle - transport_vehicle ?source_group - source_group ?handling_station - handling_station)
    :precondition
      (and
        (vehicle_prepared ?transport_vehicle)
        (source_group_preparation_complete ?source_group)
        (entity_assigned_to_handling_station ?source_group ?handling_station)
        (not
          (vehicle_loaded ?transport_vehicle)
        )
      )
    :effect (vehicle_loaded ?transport_vehicle)
  )
  (:action assign_crate_to_task
    :parameters (?reassignment_task - reassignment_task ?load_crate - load_crate ?transport_vehicle - transport_vehicle)
    :precondition
      (and
        (entity_in_reassignment_workflow ?reassignment_task)
        (task_assigned_vehicle ?reassignment_task ?transport_vehicle)
        (task_has_crate ?reassignment_task ?load_crate)
        (crate_available ?load_crate)
        (vehicle_prepared ?transport_vehicle)
        (vehicle_loaded ?transport_vehicle)
        (not
          (crate_validated ?load_crate)
        )
      )
    :effect
      (and
        (crate_validated ?load_crate)
        (crate_assigned_vehicle ?load_crate ?transport_vehicle)
        (not
          (crate_available ?load_crate)
        )
      )
  )
  (:action validate_crate_for_task
    :parameters (?reassignment_task - reassignment_task ?load_crate - load_crate ?transport_vehicle - transport_vehicle ?handling_station - handling_station)
    :precondition
      (and
        (entity_in_reassignment_workflow ?reassignment_task)
        (task_has_crate ?reassignment_task ?load_crate)
        (crate_validated ?load_crate)
        (crate_assigned_vehicle ?load_crate ?transport_vehicle)
        (entity_assigned_to_handling_station ?reassignment_task ?handling_station)
        (not
          (vehicle_flag_restricted_loading ?transport_vehicle)
        )
        (not
          (task_prechecks_passed ?reassignment_task)
        )
      )
    :effect (task_prechecks_passed ?reassignment_task)
  )
  (:action attach_identification_tag_to_task
    :parameters (?reassignment_task - reassignment_task ?identification_tag - identification_tag)
    :precondition
      (and
        (entity_in_reassignment_workflow ?reassignment_task)
        (identification_tag_available ?identification_tag)
        (not
          (task_identification_tag_attached ?reassignment_task)
        )
      )
    :effect
      (and
        (task_identification_tag_attached ?reassignment_task)
        (task_identification_tag_link ?reassignment_task ?identification_tag)
        (not
          (identification_tag_available ?identification_tag)
        )
      )
  )
  (:action verify_tag_and_crate_for_task
    :parameters (?reassignment_task - reassignment_task ?load_crate - load_crate ?transport_vehicle - transport_vehicle ?handling_station - handling_station ?identification_tag - identification_tag)
    :precondition
      (and
        (entity_in_reassignment_workflow ?reassignment_task)
        (task_has_crate ?reassignment_task ?load_crate)
        (crate_validated ?load_crate)
        (crate_assigned_vehicle ?load_crate ?transport_vehicle)
        (entity_assigned_to_handling_station ?reassignment_task ?handling_station)
        (vehicle_flag_restricted_loading ?transport_vehicle)
        (task_identification_tag_attached ?reassignment_task)
        (task_identification_tag_link ?reassignment_task ?identification_tag)
        (not
          (task_prechecks_passed ?reassignment_task)
        )
      )
    :effect
      (and
        (task_prechecks_passed ?reassignment_task)
        (task_tag_verified ?reassignment_task)
      )
  )
  (:action complete_crate_and_sedation_check
    :parameters (?reassignment_task - reassignment_task ?sedation_dose - sedation_dose ?handler - handler ?load_crate - load_crate ?transport_vehicle - transport_vehicle)
    :precondition
      (and
        (task_prechecks_passed ?reassignment_task)
        (task_sedation_allocated ?reassignment_task ?sedation_dose)
        (entity_assigned_handler ?reassignment_task ?handler)
        (task_has_crate ?reassignment_task ?load_crate)
        (crate_assigned_vehicle ?load_crate ?transport_vehicle)
        (not
          (vehicle_flag_extended_loading ?transport_vehicle)
        )
        (not
          (task_confirmed_for_assessment ?reassignment_task)
        )
      )
    :effect (task_confirmed_for_assessment ?reassignment_task)
  )
  (:action complete_crate_and_sedation_recheck
    :parameters (?reassignment_task - reassignment_task ?sedation_dose - sedation_dose ?handler - handler ?load_crate - load_crate ?transport_vehicle - transport_vehicle)
    :precondition
      (and
        (task_prechecks_passed ?reassignment_task)
        (task_sedation_allocated ?reassignment_task ?sedation_dose)
        (entity_assigned_handler ?reassignment_task ?handler)
        (task_has_crate ?reassignment_task ?load_crate)
        (crate_assigned_vehicle ?load_crate ?transport_vehicle)
        (vehicle_flag_extended_loading ?transport_vehicle)
        (not
          (task_confirmed_for_assessment ?reassignment_task)
        )
      )
