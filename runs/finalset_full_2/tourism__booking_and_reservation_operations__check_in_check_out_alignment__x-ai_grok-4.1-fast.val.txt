  (:action create_provisional_booking
    :parameters (?booking_request - booking_request)
    :precondition
      (and
        (not
          (reservation_provisional ?booking_request)
        )
        (not
          (reservation_finalized ?booking_request)
        )
      )
    :effect (reservation_provisional ?booking_request)
  )
  (:action allocate_offer_to_booking
    :parameters (?booking_request - booking_request ?service_offer - service_offer)
    :precondition
      (and
        (reservation_provisional ?booking_request)
        (not
          (reservation_has_hold ?booking_request)
        )
        (service_offer_available ?service_offer)
      )
    :effect
      (and
        (reservation_has_hold ?booking_request)
        (reservation_holds_offer ?booking_request ?service_offer)
        (not
          (service_offer_available ?service_offer)
        )
      )
  )
