--- domain.pddl (as passed to VAL) ---
(define (domain summon_pet_control_prioritization)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity - object capability_group - entity attachment_group - entity summon_family - entity controller_root - entity controller_slot - controller_root summon_command_token - capability_group target_marker - capability_group support_ability_token - capability_group trait_token - capability_group assignment_modifier_token - capability_group priority_token - capability_group equipment_buff_token - capability_group tactical_module_token - capability_group consumable_resource - attachment_group companion_equipment - attachment_group tactical_marker - attachment_group summon_template_tank - summon_family summon_template_dps - summon_family summon_instance - summon_family role_slot_family - controller_slot party_member_family - controller_slot tank_role_slot - role_slot_family dps_role_slot - role_slot_family party_coordinator - party_member_family)

  (:predicates
    (controller_registered ?controller_slot - controller_slot)
    (entity_engaged_for_assignment ?controller_slot - controller_slot)
    (entity_summon_bound ?controller_slot - controller_slot)
    (entity_committed ?controller_slot - controller_slot)
    (entity_finalized ?controller_slot - controller_slot)
    (entity_ready_for_commit ?controller_slot - controller_slot)
    (summon_token_available ?summon_command_token - summon_command_token)
    (entity_summon_binding ?controller_slot - controller_slot ?summon_command_token - summon_command_token)
    (target_available ?target_marker - target_marker)
    (entity_target_binding ?controller_slot - controller_slot ?target_marker - target_marker)
    (support_ability_available ?support_ability_token - support_ability_token)
    (entity_support_binding ?controller_slot - controller_slot ?support_ability_token - support_ability_token)
    (consumable_available ?consumable_resource - consumable_resource)
    (tank_role_consumable_binding ?tank_role_slot - tank_role_slot ?consumable_resource - consumable_resource)
    (dps_consumable_binding ?dps_role_slot - dps_role_slot ?consumable_resource - consumable_resource)
    (tank_role_template_compatible ?tank_role_slot - tank_role_slot ?summon_template_tank - summon_template_tank)
    (tank_template_selected ?summon_template_tank - summon_template_tank)
    (tank_template_buffered ?summon_template_tank - summon_template_tank)
    (tank_role_executed_assignment ?tank_role_slot - tank_role_slot)
    (dps_role_template_binding ?dps_role_slot - dps_role_slot ?summon_template_dps - summon_template_dps)
    (dps_template_selected ?summon_template_dps - summon_template_dps)
    (dps_template_buffered ?summon_template_dps - summon_template_dps)
    (dps_role_executed_assignment ?dps_role_slot - dps_role_slot)
    (summon_instance_available ?summon_instance - summon_instance)
    (summon_instance_reserved ?summon_instance - summon_instance)
    (instance_tank_template_link ?summon_instance - summon_instance ?summon_template_tank - summon_template_tank)
    (instance_dps_template_link ?summon_instance - summon_instance ?summon_template_dps - summon_template_dps)
    (instance_stage_a ?summon_instance - summon_instance)
    (instance_stage_b ?summon_instance - summon_instance)
    (instance_deploy_ready ?summon_instance - summon_instance)
    (coordinator_tank_role_link ?party_coordinator - party_coordinator ?tank_role_slot - tank_role_slot)
    (coordinator_dps_role_link ?party_coordinator - party_coordinator ?dps_role_slot - dps_role_slot)
    (coordinator_instance_preference ?party_coordinator - party_coordinator ?summon_instance - summon_instance)
    (equipment_available ?companion_equipment - companion_equipment)
    (coordinator_equipment_binding ?party_coordinator - party_coordinator ?companion_equipment - companion_equipment)
    (equipment_attached ?companion_equipment - companion_equipment)
    (equipment_instance_binding ?companion_equipment - companion_equipment ?summon_instance - summon_instance)
    (coordinator_ready_for_specialization ?party_coordinator - party_coordinator)
    (coordinator_specialization_in_progress ?party_coordinator - party_coordinator)
    (coordinator_specialization_complete ?party_coordinator - party_coordinator)
    (coordinator_trait_available ?party_coordinator - party_coordinator)
    (coordinator_trait_applied ?party_coordinator - party_coordinator)
    (coordinator_ready_for_execution ?party_coordinator - party_coordinator)
    (coordinator_executable_marker ?party_coordinator - party_coordinator)
    (tactical_marker_available ?tactical_marker - tactical_marker)
    (coordinator_tactical_marker_binding ?party_coordinator - party_coordinator ?tactical_marker - tactical_marker)
    (coordinator_trait_locked ?party_coordinator - party_coordinator)
    (trait_stage_ready ?party_coordinator - party_coordinator)
    (trait_finalized ?party_coordinator - party_coordinator)
    (optional_trait_available ?trait_token - trait_token)
    (coordinator_optional_trait_binding ?party_coordinator - party_coordinator ?trait_token - trait_token)
    (assignment_modifier_available ?assignment_modifier_token - assignment_modifier_token)
    (coordinator_assignment_modifier_binding ?party_coordinator - party_coordinator ?assignment_modifier_token - assignment_modifier_token)
    (equipment_buff_available ?equipment_buff_token - equipment_buff_token)
    (coordinator_equipment_buff_binding ?party_coordinator - party_coordinator ?equipment_buff_token - equipment_buff_token)
    (tactical_module_available ?tactical_module_token - tactical_module_token)
    (coordinator_tactical_module_binding ?party_coordinator - party_coordinator ?tactical_module_token - tactical_module_token)
    (priority_token_available ?priority_token - priority_token)
    (entity_priority_binding ?controller_slot - controller_slot ?priority_token - priority_token)
    (tank_role_staged ?tank_role_slot - tank_role_slot)
    (dps_role_staged ?dps_role_slot - dps_role_slot)
    (activation_claimed ?party_coordinator - party_coordinator)
  )
  (:action register_controller_slot
    :parameters (?controller_slot - controller_slot)
    :precondition
      (and
        (not
          (controller_registered ?controller_slot)
        )
        (not
          (entity_committed ?controller_slot)
        )
      )
    :effect (controller_registered ?controller_slot)
  )
  (:action bind_summon_token_to_controller
    :parameters (?controller_slot - controller_slot ?summon_command_token - summon_command_token)
    :precondition
      (and
        (controller_registered ?controller_slot)
        (not
          (entity_summon_bound ?controller_slot)
        )
        (summon_token_available ?summon_command_token)
      )
    :effect
      (and
        (entity_summon_bound ?controller_slot)
        (entity_summon_binding ?controller_slot ?summon_command_token)
        (not
          (summon_token_available ?summon_command_token)
        )
      )
  )
  (:action bind_target_marker_to_controller
    :parameters (?controller_slot - controller_slot ?target_marker - target_marker)
    :precondition
      (and
        (controller_registered ?controller_slot)
        (entity_summon_bound ?controller_slot)
        (target_available ?target_marker)
      )
    :effect
      (and
        (entity_target_binding ?controller_slot ?target_marker)
        (not
          (target_available ?target_marker)
        )
      )
  )
  (:action engage_controller_for_assignment
    :parameters (?controller_slot - controller_slot ?target_marker - target_marker)
    :precondition
      (and
        (controller_registered ?controller_slot)
        (entity_summon_bound ?controller_slot)
        (entity_target_binding ?controller_slot ?target_marker)
        (not
          (entity_engaged_for_assignment ?controller_slot)
        )
      )
    :effect (entity_engaged_for_assignment ?controller_slot)
  )
  (:action release_target_for_controller
    :parameters (?controller_slot - controller_slot ?target_marker - target_marker)
    :precondition
      (and
        (entity_target_binding ?controller_slot ?target_marker)
      )
    :effect
      (and
        (target_available ?target_marker)
        (not
          (entity_target_binding ?controller_slot ?target_marker)
        )
      )
  )
  (:action bind_support_to_controller
    :parameters (?controller_slot - controller_slot ?support_ability_token - support_ability_token)
    :precondition
      (and
        (entity_engaged_for_assignment ?controller_slot)
        (support_ability_available ?support_ability_token)
      )
    :effect
      (and
        (entity_support_binding ?controller_slot ?support_ability_token)
        (not
          (support_ability_available ?support_ability_token)
        )
      )
  )
  (:action release_support_from_controller
    :parameters (?controller_slot - controller_slot ?support_ability_token - support_ability_token)
    :precondition
      (and
        (entity_support_binding ?controller_slot ?support_ability_token)
      )
    :effect
      (and
        (support_ability_available ?support_ability_token)
        (not
          (entity_support_binding ?controller_slot ?support_ability_token)
        )
      )
  )
  (:action bind_equipment_buff_to_coordinator
    :parameters (?party_coordinator - party_coordinator ?equipment_buff_token - equipment_buff_token)
    :precondition
      (and
        (entity_engaged_for_assignment ?party_coordinator)
        (equipment_buff_available ?equipment_buff_token)
      )
    :effect
      (and
        (coordinator_equipment_buff_binding ?party_coordinator ?equipment_buff_token)
        (not
          (equipment_buff_available ?equipment_buff_token)
        )
      )
  )
  (:action unbind_equipment_buff_from_coordinator
    :parameters (?party_coordinator - party_coordinator ?equipment_buff_token - equipment_buff_token)
    :precondition
      (and
        (coordinator_equipment_buff_binding ?party_coordinator ?equipment_buff_token)
      )
    :effect
      (and
        (equipment_buff_available ?equipment_buff_token)
        (not
          (coordinator_equipment_buff_binding ?party_coordinator ?equipment_buff_token)
        )
      )
  )
  (:action bind_tactical_module_to_coordinator
    :parameters (?party_coordinator - party_coordinator ?tactical_module_token - tactical_module_token)
    :precondition
      (and
        (entity_engaged_for_assignment ?party_coordinator)
        (tactical_module_available ?tactical_module_token)
      )
    :effect
      (and
        (coordinator_tactical_module_binding ?party_coordinator ?tactical_module_token)
        (not
          (tactical_module_available ?tactical_module_token)
        )
      )
  )
  (:action unbind_tactical_module_from_coordinator
    :parameters (?party_coordinator - party_coordinator ?tactical_module_token - tactical_module_token)
    :precondition
      (and
        (coordinator_tactical_module_binding ?party_coordinator ?tactical_module_token)
      )
    :effect
      (and
        (tactical_module_available ?tactical_module_token)
        (not
          (coordinator_tactical_module_binding ?party_coordinator ?tactical_module_token)
        )
      )
  )
  (:action select_tank_template
    :parameters (?tank_role_slot - tank_role_slot ?summon_template_tank - summon_template_tank ?target_marker - target_marker)
    :precondition
      (and
        (entity_engaged_for_assignment ?tank_role_slot)
        (entity_target_binding ?tank_role_slot ?target_marker)
        (tank_role_template_compatible ?tank_role_slot ?summon_template_tank)
        (not
          (tank_template_selected ?summon_template_tank)
        )
        (not
          (tank_template_buffered ?summon_template_tank)
        )
      )
    :effect (tank_template_selected ?summon_template_tank)
  )
  (:action finalize_tank_role_assignment_with_support
    :parameters (?tank_role_slot - tank_role_slot ?summon_template_tank - summon_template_tank ?support_ability_token - support_ability_token)
    :precondition
      (and
        (entity_engaged_for_assignment ?tank_role_slot)
        (entity_support_binding ?tank_role_slot ?support_ability_token)
        (tank_role_template_compatible ?tank_role_slot ?summon_template_tank)
        (tank_template_selected ?summon_template_tank)
        (not
          (tank_role_staged ?tank_role_slot)
        )
      )
    :effect
      (and
        (tank_role_staged ?tank_role_slot)
        (tank_role_executed_assignment ?tank_role_slot)
      )
  )
  (:action buffer_tank_template_with_consumable
    :parameters (?tank_role_slot - tank_role_slot ?summon_template_tank - summon_template_tank ?consumable_resource - consumable_resource)
    :precondition
      (and
        (entity_engaged_for_assignment ?tank_role_slot)
        (tank_role_template_compatible ?tank_role_slot ?summon_template_tank)
        (consumable_available ?consumable_resource)
        (not
          (tank_role_staged ?tank_role_slot)
        )
      )
    :effect
      (and
        (tank_template_buffered ?summon_template_tank)
        (tank_role_staged ?tank_role_slot)
        (tank_role_consumable_binding ?tank_role_slot ?consumable_resource)
        (not
          (consumable_available ?consumable_resource)
        )
      )
  )
  (:action activate_buffered_tank_template
    :parameters (?tank_role_slot - tank_role_slot ?summon_template_tank - summon_template_tank ?target_marker - target_marker ?consumable_resource - consumable_resource)
    :precondition
      (and
        (entity_engaged_for_assignment ?tank_role_slot)
        (entity_target_binding ?tank_role_slot ?target_marker)
        (tank_role_template_compatible ?tank_role_slot ?summon_template_tank)
        (tank_template_buffered ?summon_template_tank)
        (tank_role_consumable_binding ?tank_role_slot ?consumable_resource)
        (not
          (tank_role_executed_assignment ?tank_role_slot)
        )
      )
    :effect
      (and
        (tank_template_selected ?summon_template_tank)
        (tank_role_executed_assignment ?tank_role_slot)
        (consumable_available ?consumable_resource)
        (not
          (tank_role_consumable_binding ?tank_role_slot ?consumable_resource)
        )
      )
  )
  (:action select_dps_template
    :parameters (?dps_role_slot - dps_role_slot ?summon_template_dps - summon_template_dps ?target_marker - target_marker)
    :precondition
      (and
        (entity_engaged_for_assignment ?dps_role_slot)
        (entity_target_binding ?dps_role_slot ?target_marker)
        (dps_role_template_binding ?dps_role_slot ?summon_template_dps)
        (not
          (dps_template_selected ?summon_template_dps)
        )
        (not
          (dps_template_buffered ?summon_template_dps)
        )
      )
    :effect (dps_template_selected ?summon_template_dps)
  )
  (:action finalize_dps_role_assignment_with_support
    :parameters (?dps_role_slot - dps_role_slot ?summon_template_dps - summon_template_dps ?support_ability_token - support_ability_token)
    :precondition
      (and
        (entity_engaged_for_assignment ?dps_role_slot)
        (entity_support_binding ?dps_role_slot ?support_ability_token)
        (dps_role_template_binding ?dps_role_slot ?summon_template_dps)
        (dps_template_selected ?summon_template_dps)
        (not
          (dps_role_staged ?dps_role_slot)
        )
      )
    :effect
      (and
        (dps_role_staged ?dps_role_slot)
        (dps_role_executed_assignment ?dps_role_slot)
      )
  )
  (:action buffer_dps_template_with_consumable
    :parameters (?dps_role_slot - dps_role_slot ?summon_template_dps - summon_template_dps ?consumable_resource - consumable_resource)
    :precondition
      (and
        (entity_engaged_for_assignment ?dps_role_slot)
        (dps_role_template_binding ?dps_role_slot ?summon_template_dps)
        (consumable_available ?consumable_resource)
        (not
          (dps_role_staged ?dps_role_slot)
        )
      )
    :effect
      (and
        (dps_template_buffered ?summon_template_dps)
        (dps_role_staged ?dps_role_slot)
        (dps_consumable_binding ?dps_role_slot ?consumable_resource)
        (not
          (consumable_available ?consumable_resource)
        )
      )
  )
  (:action activate_buffered_dps_template
    :parameters (?dps_role_slot - dps_role_slot ?summon_template_dps - summon_template_dps ?target_marker - target_marker ?consumable_resource - consumable_resource)
    :precondition
      (and
        (entity_engaged_for_assignment ?dps_role_slot)
        (entity_target_binding ?dps_role_slot ?target_marker)
        (dps_role_template_binding ?dps_role_slot ?summon_template_dps)
        (dps_template_buffered ?summon_template_dps)
        (dps_consumable_binding ?dps_role_slot ?consumable_resource)
        (not
          (dps_role_executed_assignment ?dps_role_slot)
        )
      )
    :effect
      (and
        (dps_template_selected ?summon_template_dps)
        (dps_role_executed_assignment ?dps_role_slot)
        (consumable_available ?consumable_resource)
        (not
          (dps_consumable_binding ?dps_role_slot ?consumable_resource)
        )
      )
  )
  (:action reserve_and_pair_summon_instance
    :parameters (?tank_role_slot - tank_role_slot ?dps_role_slot - dps_role_slot ?summon_template_tank - summon_template_tank ?summon_template_dps - summon_template_dps ?summon_instance - summon_instance)
    :precondition
      (and
        (tank_role_staged ?tank_role_slot)
        (dps_role_staged ?dps_role_slot)
        (tank_role_template_compatible ?tank_role_slot ?summon_template_tank)
        (dps_role_template_binding ?dps_role_slot ?summon_template_dps)
        (tank_template_selected ?summon_template_tank)
        (dps_template_selected ?summon_template_dps)
        (tank_role_executed_assignment ?tank_role_slot)
        (dps_role_executed_assignment ?dps_role_slot)
        (summon_instance_available ?summon_instance)
      )
    :effect
      (and
        (summon_instance_reserved ?summon_instance)
        (instance_tank_template_link ?summon_instance ?summon_template_tank)
        (instance_dps_template_link ?summon_instance ?summon_template_dps)
        (not
          (summon_instance_available ?summon_instance)
        )
      )
  )
  (:action reserve_and_pair_summon_instance_set_stage_a
    :parameters (?tank_role_slot - tank_role_slot ?dps_role_slot - dps_role_slot ?summon_template_tank - summon_template_tank ?summon_template_dps - summon_template_dps ?summon_instance - summon_instance)
    :precondition
      (and
        (tank_role_staged ?tank_role_slot)
        (dps_role_staged ?dps_role_slot)
        (tank_role_template_compatible ?tank_role_slot ?summon_template_tank)
        (dps_role_template_binding ?dps_role_slot ?summon_template_dps)
        (tank_template_buffered ?summon_template_tank)
        (dps_template_selected ?summon_template_dps)
        (not
          (tank_role_executed_assignment ?tank_role_slot)
        )
        (dps_role_executed_assignment ?dps_role_slot)
        (summon_instance_available ?summon_instance)
      )
    :effect
      (and
        (summon_instance_reserved ?summon_instance)
        (instance_tank_template_link ?summon_instance ?summon_template_tank)
        (instance_dps_template_link ?summon_instance ?summon_template_dps)
        (instance_stage_a ?summon_instance)
        (not
          (summon_instance_available ?summon_instance)
        )
      )
  )
  (:action reserve_and_pair_summon_instance_set_stage_b
    :parameters (?tank_role_slot - tank_role_slot ?dps_role_slot - dps_role_slot ?summon_template_tank - summon_template_tank ?summon_template_dps - summon_template_dps ?summon_instance - summon_instance)
    :precondition
      (and
        (tank_role_staged ?tank_role_slot)
        (dps_role_staged ?dps_role_slot)
        (tank_role_template_compatible ?tank_role_slot ?summon_template_tank)
        (dps_role_template_binding ?dps_role_slot ?summon_template_dps)
        (tank_template_selected ?summon_template_tank)
        (dps_template_buffered ?summon_template_dps)
        (tank_role_executed_assignment ?tank_role_slot)
        (not
          (dps_role_executed_assignment ?dps_role_slot)
        )
        (summon_instance_available ?summon_instance)
      )
    :effect
      (and
        (summon_instance_reserved ?summon_instance)
        (instance_tank_template_link ?summon_instance ?summon_template_tank)
        (instance_dps_template_link ?summon_instance ?summon_template_dps)
        (instance_stage_b ?summon_instance)
        (not
          (summon_instance_available ?summon_instance)
        )
      )
  )
  (:action reserve_and_pair_summon_instance_set_both_stages
    :parameters (?tank_role_slot - tank_role_slot ?dps_role_slot - dps_role_slot ?summon_template_tank - summon_template_tank ?summon_template_dps - summon_template_dps ?summon_instance - summon_instance)
    :precondition
      (and
        (tank_role_staged ?tank_role_slot)
        (dps_role_staged ?dps_role_slot)
        (tank_role_template_compatible ?tank_role_slot ?summon_template_tank)
        (dps_role_template_binding ?dps_role_slot ?summon_template_dps)
        (tank_template_buffered ?summon_template_tank)
        (dps_template_buffered ?summon_template_dps)
        (not
          (tank_role_executed_assignment ?tank_role_slot)
        )
        (not
          (dps_role_executed_assignment ?dps_role_slot)
        )
        (summon_instance_available ?summon_instance)
      )
    :effect
      (and
        (summon_instance_reserved ?summon_instance)
        (instance_tank_template_link ?summon_instance ?summon_template_tank)
        (instance_dps_template_link ?summon_instance ?summon_template_dps)
        (instance_stage_a ?summon_instance)
        (instance_stage_b ?summon_instance)
        (not
          (summon_instance_available ?summon_instance)
        )
      )
  )
  (:action mark_instance_deploy_ready
    :parameters (?summon_instance - summon_instance ?tank_role_slot - tank_role_slot ?target_marker - target_marker)
    :precondition
      (and
        (summon_instance_reserved ?summon_instance)
        (tank_role_staged ?tank_role_slot)
        (entity_target_binding ?tank_role_slot ?target_marker)
        (not
          (instance_deploy_ready ?summon_instance)
        )
      )
    :effect (instance_deploy_ready ?summon_instance)
  )
  (:action attach_equipment_to_instance
    :parameters (?party_coordinator - party_coordinator ?companion_equipment - companion_equipment ?summon_instance - summon_instance)
    :precondition
      (and
        (entity_engaged_for_assignment ?party_coordinator)
        (coordinator_instance_preference ?party_coordinator ?summon_instance)
        (coordinator_equipment_binding ?party_coordinator ?companion_equipment)
        (equipment_available ?companion_equipment)
        (summon_instance_reserved ?summon_instance)
        (instance_deploy_ready ?summon_instance)
        (not
          (equipment_attached ?companion_equipment)
        )
      )
    :effect
      (and
        (equipment_attached ?companion_equipment)
        (equipment_instance_binding ?companion_equipment ?summon_instance)
        (not
          (equipment_available ?companion_equipment)
        )
      )
  )
  (:action mark_coordinator_ready_for_specialization
    :parameters (?party_coordinator - party_coordinator ?companion_equipment - companion_equipment ?summon_instance - summon_instance ?target_marker - target_marker)
    :precondition
      (and
        (entity_engaged_for_assignment ?party_coordinator)
        (coordinator_equipment_binding ?party_coordinator ?companion_equipment)
        (equipment_attached ?companion_equipment)
        (equipment_instance_binding ?companion_equipment ?summon_instance)
        (entity_target_binding ?party_coordinator ?target_marker)
        (not
          (instance_stage_a ?summon_instance)
        )
        (not
          (coordinator_ready_for_specialization ?party_coordinator)
        )
      )
    :effect (coordinator_ready_for_specialization ?party_coordinator)
  )
