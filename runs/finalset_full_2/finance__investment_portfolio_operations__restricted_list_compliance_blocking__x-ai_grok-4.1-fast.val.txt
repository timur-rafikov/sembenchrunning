  (:action register_investment_entity
    :parameters (?investment_entity - investment_entity)
    :precondition
      (and
        (not
          (entity_registered ?investment_entity)
        )
        (not
          (block_propagated ?investment_entity)
        )
      )
    :effect (entity_registered ?investment_entity)
  )
  (:action assign_restriction_entry
    :parameters (?investment_entity - investment_entity ?restricted_list_entry - restricted_list_entry)
    :precondition
      (and
        (entity_registered ?investment_entity)
        (not
          (has_restriction ?investment_entity)
        )
        (restricted_entry_available ?restricted_list_entry)
      )
    :effect
      (and
        (has_restriction ?investment_entity)
        (restricted_entry_assigned ?investment_entity ?restricted_list_entry)
        (not
          (restricted_entry_available ?restricted_list_entry)
        )
      )
  )
  (:action assign_screening_resource
    :parameters (?investment_entity - investment_entity ?screening_resource - screening_resource)
    :precondition
      (and
        (entity_registered ?investment_entity)

        (screening_resource_available ?screening_resource)
      )
    :effect
      (and
        (screened_with_resource ?investment_entity ?screening_resource)
        (not
          (screening_resource_available ?screening_resource)
        )
      )
  )
  (:action confirm_screening_clearance

        (entity_registered ?investment_entity)

          (screening_cleared_entity ?investment_entity)
        )
      )
    :effect (screening_cleared_entity ?investment_entity)
  )
  (:action release_screening_resource
    :parameters (?investment_entity - investment_entity ?screening_resource - screening_resource)

      (and
        (screened_with_resource ?investment_entity ?screening_resource)
      )
    :effect
      (and
        (screening_resource_available ?screening_resource)
        (not
          (screened_with_resource ?investment_entity ?screening_resource)
        )
      )
  )
  (:action assign_approver_to_entity
