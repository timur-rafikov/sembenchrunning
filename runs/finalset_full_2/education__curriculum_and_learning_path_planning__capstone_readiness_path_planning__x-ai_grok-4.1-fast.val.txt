--- domain.pddl (as passed to VAL) ---
(define (domain capstone_readiness_path_planning)
  (:requirements :strips :typing :negative-preconditions)
  (:types academic_asset_group - object scheduling_group - object administrative_category - object plan_template - object learner_plan - plan_template program_track - academic_asset_group course_offering - academic_asset_group faculty_advisor - academic_asset_group approval_document - academic_asset_group resource_allocation - academic_asset_group timeline_slot - academic_asset_group equipment_resource - academic_asset_group external_reviewer - academic_asset_group permission_token - scheduling_group assessment_artifact - scheduling_group mentor_role - scheduling_group prerequisite_requirement - administrative_category milestone_type - administrative_category capstone_proposal - administrative_category path_template - learner_plan path_variant - learner_plan learner_path_a - path_template learner_path_b - path_template capstone_component - path_variant)
  (:predicates
    (entity_initialized ?learner_plan - learner_plan)
    (entity_eligibility_confirmed ?learner_plan - learner_plan)
    (entity_has_assigned_track ?learner_plan - learner_plan)
    (entity_ready_for_capstone ?learner_plan - learner_plan)
    (entity_internally_certified ?learner_plan - learner_plan)
    (entity_capstone_readiness_activated ?learner_plan - learner_plan)
    (program_track_available ?program_track - program_track)
    (assigned_program_track ?learner_plan - learner_plan ?program_track - program_track)
    (course_seat_available ?course_offering - course_offering)
    (has_course_reservation ?learner_plan - learner_plan ?course_offering - course_offering)
    (faculty_advisor_available ?faculty_advisor - faculty_advisor)
    (assigned_faculty_advisor ?learner_plan - learner_plan ?faculty_advisor - faculty_advisor)
    (permission_token_available ?permission_token - permission_token)
    (path_a_permission_token_assigned ?learner_path_a - learner_path_a ?permission_token - permission_token)
    (path_b_permission_token_assigned ?learner_path_b - learner_path_b ?permission_token - permission_token)
    (path_a_requires_prerequisite ?learner_path_a - learner_path_a ?prerequisite_requirement - prerequisite_requirement)
    (prerequisite_validated ?prerequisite_requirement - prerequisite_requirement)
    (prerequisite_in_progress ?prerequisite_requirement - prerequisite_requirement)
    (path_a_prerequisites_met ?learner_path_a - learner_path_a)
    (path_b_has_milestone ?learner_path_b - learner_path_b ?milestone_type - milestone_type)
    (milestone_reserved ?milestone_type - milestone_type)
    (milestone_in_progress ?milestone_type - milestone_type)
    (path_b_prerequisites_met ?learner_path_b - learner_path_b)
    (capstone_proposal_slot_available ?capstone_proposal - capstone_proposal)
    (capstone_proposal_reserved ?capstone_proposal - capstone_proposal)
    (proposal_associated_prerequisite ?capstone_proposal - capstone_proposal ?prerequisite_requirement - prerequisite_requirement)
    (proposal_associated_milestone ?capstone_proposal - capstone_proposal ?milestone_type - milestone_type)
    (capstone_proposal_flag_a ?capstone_proposal - capstone_proposal)
    (capstone_proposal_flag_b ?capstone_proposal - capstone_proposal)
    (capstone_proposal_confirmed ?capstone_proposal - capstone_proposal)
    (linked_to_path_a ?capstone_component - capstone_component ?learner_path_a - learner_path_a)
    (linked_to_path_b ?capstone_component - capstone_component ?learner_path_b - learner_path_b)
    (linked_to_proposal ?capstone_component - capstone_component ?capstone_proposal - capstone_proposal)
    (assessment_artifact_template_available ?assessment_artifact - assessment_artifact)
    (has_assessment_artifact_template ?capstone_component - capstone_component ?assessment_artifact - assessment_artifact)
    (assessment_artifact_submitted ?assessment_artifact - assessment_artifact)
    (assessment_artifact_associated_with_proposal ?assessment_artifact - assessment_artifact ?capstone_proposal - capstone_proposal)
    (review_stage_one_complete ?capstone_component - capstone_component)
    (review_stage_two_complete ?capstone_component - capstone_component)
    (review_stage_three_complete ?capstone_component - capstone_component)
    (has_any_approval_document ?capstone_component - capstone_component)
    (secondary_approval_flag ?capstone_component - capstone_component)
    (resource_allocation_eligible ?capstone_component - capstone_component)
    (ready_for_final_certification ?capstone_component - capstone_component)
    (mentor_available ?mentor_assignment - mentor_role)
    (component_preassigned_mentor ?capstone_component - capstone_component ?mentor_assignment - mentor_role)
    (mentor_engagement_confirmed ?capstone_component - capstone_component)
    (mentor_review_step_one_done ?capstone_component - capstone_component)
    (mentor_review_step_two_done ?capstone_component - capstone_component)
    (approval_document_available ?approval_document - approval_document)
    (approval_document_assigned ?capstone_component - capstone_component ?approval_document - approval_document)
    (resource_allocation_available ?resource_allocation - resource_allocation)
    (resource_allocation_assigned ?capstone_component - capstone_component ?resource_allocation - resource_allocation)
    (equipment_resource_available ?equipment_resource - equipment_resource)
    (equipment_allocated ?capstone_component - capstone_component ?equipment_resource - equipment_resource)
    (external_reviewer_available ?external_reviewer - external_reviewer)
    (external_reviewer_assigned ?capstone_component - capstone_component ?external_reviewer - external_reviewer)
    (timeline_slot_available ?timeline_slot - timeline_slot)
    (entity_plan_assigned_timeline ?learner_plan - learner_plan ?timeline_slot - timeline_slot)
    (path_a_processing_hold ?learner_path_a - learner_path_a)
    (path_b_processing_hold ?learner_path_b - learner_path_b)
    (internal_verification_recorded ?capstone_component - capstone_component)
  )
  (:action initialize_learner_plan
    :parameters (?learner_plan - learner_plan)
    :precondition
      (and
        (not
          (entity_initialized ?learner_plan)
        )
        (not
          (entity_ready_for_capstone ?learner_plan)
        )
      )
    :effect (entity_initialized ?learner_plan)
  )
  (:action assign_program_track_to_plan
    :parameters (?learner_plan - learner_plan ?program_track - program_track)
    :precondition
      (and
        (entity_initialized ?learner_plan)
        (not
          (entity_has_assigned_track ?learner_plan)
        )
        (program_track_available ?program_track)
      )
    :effect
      (and
        (entity_has_assigned_track ?learner_plan)
        (assigned_program_track ?learner_plan ?program_track)
        (not
          (program_track_available ?program_track)
        )
      )
  )
  (:action reserve_course_seat_for_plan
    :parameters (?learner_plan - learner_plan ?course_offering - course_offering)
    :precondition
      (and
        (entity_initialized ?learner_plan)
        (entity_has_assigned_track ?learner_plan)
        (course_seat_available ?course_offering)
      )
    :effect
      (and
        (has_course_reservation ?learner_plan ?course_offering)
        (not
          (course_seat_available ?course_offering)
        )
      )
  )
  (:action confirm_course_enrollment
    :parameters (?learner_plan - learner_plan ?course_offering - course_offering)
    :precondition
      (and
        (entity_initialized ?learner_plan)
        (entity_has_assigned_track ?learner_plan)
        (has_course_reservation ?learner_plan ?course_offering)
        (not
          (entity_eligibility_confirmed ?learner_plan)
        )
      )
    :effect (entity_eligibility_confirmed ?learner_plan)
  )
  (:action release_course_reservation
    :parameters (?learner_plan - learner_plan ?course_offering - course_offering)
    :precondition
      (and
        (has_course_reservation ?learner_plan ?course_offering)
      )
    :effect
      (and
        (course_seat_available ?course_offering)
        (not
          (has_course_reservation ?learner_plan ?course_offering)
        )
      )
  )
  (:action assign_faculty_advisor_to_plan
    :parameters (?learner_plan - learner_plan ?faculty_advisor - faculty_advisor)
    :precondition
      (and
        (entity_eligibility_confirmed ?learner_plan)
        (faculty_advisor_available ?faculty_advisor)
      )
    :effect
      (and
        (assigned_faculty_advisor ?learner_plan ?faculty_advisor)
        (not
          (faculty_advisor_available ?faculty_advisor)
        )
      )
  )
  (:action release_faculty_advisor_from_plan
    :parameters (?learner_plan - learner_plan ?faculty_advisor - faculty_advisor)
    :precondition
      (and
        (assigned_faculty_advisor ?learner_plan ?faculty_advisor)
      )
    :effect
      (and
        (faculty_advisor_available ?faculty_advisor)
        (not
          (assigned_faculty_advisor ?learner_plan ?faculty_advisor)
        )
      )
  )
  (:action allocate_equipment_to_component
    :parameters (?capstone_component - capstone_component ?equipment_resource - equipment_resource)
    :precondition
      (and
        (entity_eligibility_confirmed ?capstone_component)
        (equipment_resource_available ?equipment_resource)
      )
    :effect
      (and
        (equipment_allocated ?capstone_component ?equipment_resource)
        (not
          (equipment_resource_available ?equipment_resource)
        )
      )
  )
  (:action release_equipment_from_component
    :parameters (?capstone_component - capstone_component ?equipment_resource - equipment_resource)
    :precondition
      (and
        (equipment_allocated ?capstone_component ?equipment_resource)
      )
    :effect
      (and
        (equipment_resource_available ?equipment_resource)
        (not
          (equipment_allocated ?capstone_component ?equipment_resource)
        )
      )
  )
  (:action assign_external_reviewer_to_component
    :parameters (?capstone_component - capstone_component ?external_reviewer - external_reviewer)
    :precondition
      (and
        (entity_eligibility_confirmed ?capstone_component)
        (external_reviewer_available ?external_reviewer)
      )
    :effect
      (and
        (external_reviewer_assigned ?capstone_component ?external_reviewer)
        (not
          (external_reviewer_available ?external_reviewer)
        )
      )
  )
  (:action release_external_reviewer_from_component
    :parameters (?capstone_component - capstone_component ?external_reviewer - external_reviewer)
    :precondition
      (and
        (external_reviewer_assigned ?capstone_component ?external_reviewer)
      )
    :effect
      (and
        (external_reviewer_available ?external_reviewer)
        (not
          (external_reviewer_assigned ?capstone_component ?external_reviewer)
        )
      )
  )
  (:action validate_prerequisite_for_path_a
    :parameters (?learner_path_a - learner_path_a ?prerequisite_requirement - prerequisite_requirement ?course_offering - course_offering)
    :precondition
      (and
        (entity_eligibility_confirmed ?learner_path_a)
        (has_course_reservation ?learner_path_a ?course_offering)
        (path_a_requires_prerequisite ?learner_path_a ?prerequisite_requirement)
        (not
          (prerequisite_validated ?prerequisite_requirement)
        )
        (not
          (prerequisite_in_progress ?prerequisite_requirement)
        )
      )
    :effect (prerequisite_validated ?prerequisite_requirement)
  )
  (:action advisor_signoff_prerequisite_for_path_a
    :parameters (?learner_path_a - learner_path_a ?prerequisite_requirement - prerequisite_requirement ?faculty_advisor - faculty_advisor)
    :precondition
      (and
        (entity_eligibility_confirmed ?learner_path_a)
        (assigned_faculty_advisor ?learner_path_a ?faculty_advisor)
        (path_a_requires_prerequisite ?learner_path_a ?prerequisite_requirement)
        (prerequisite_validated ?prerequisite_requirement)
        (not
          (path_a_processing_hold ?learner_path_a)
        )
      )
    :effect
      (and
        (path_a_processing_hold ?learner_path_a)
        (path_a_prerequisites_met ?learner_path_a)
      )
  )
  (:action apply_permission_token_to_path_a
    :parameters (?learner_path_a - learner_path_a ?prerequisite_requirement - prerequisite_requirement ?permission_token - permission_token)
    :precondition
      (and
        (entity_eligibility_confirmed ?learner_path_a)
        (path_a_requires_prerequisite ?learner_path_a ?prerequisite_requirement)
        (permission_token_available ?permission_token)
        (not
          (path_a_processing_hold ?learner_path_a)
        )
      )
    :effect
      (and
        (prerequisite_in_progress ?prerequisite_requirement)
        (path_a_processing_hold ?learner_path_a)
        (path_a_permission_token_assigned ?learner_path_a ?permission_token)
        (not
          (permission_token_available ?permission_token)
        )
      )
  )
  (:action finalize_prerequisite_completion_for_path_a
    :parameters (?learner_path_a - learner_path_a ?prerequisite_requirement - prerequisite_requirement ?course_offering - course_offering ?permission_token - permission_token)
    :precondition
      (and
        (entity_eligibility_confirmed ?learner_path_a)
        (has_course_reservation ?learner_path_a ?course_offering)
        (path_a_requires_prerequisite ?learner_path_a ?prerequisite_requirement)
        (prerequisite_in_progress ?prerequisite_requirement)
        (path_a_permission_token_assigned ?learner_path_a ?permission_token)
        (not
          (path_a_prerequisites_met ?learner_path_a)
        )
      )
    :effect
      (and
        (prerequisite_validated ?prerequisite_requirement)
        (path_a_prerequisites_met ?learner_path_a)
        (permission_token_available ?permission_token)
        (not
          (path_a_permission_token_assigned ?learner_path_a ?permission_token)
        )
      )
  )
  (:action validate_prerequisite_for_path_b
    :parameters (?learner_path_b - learner_path_b ?milestone_type - milestone_type ?course_offering - course_offering)
    :precondition
      (and
        (entity_eligibility_confirmed ?learner_path_b)
        (has_course_reservation ?learner_path_b ?course_offering)
        (path_b_has_milestone ?learner_path_b ?milestone_type)
        (not
          (milestone_reserved ?milestone_type)
        )
        (not
          (milestone_in_progress ?milestone_type)
        )
      )
    :effect (milestone_reserved ?milestone_type)
  )
  (:action faculty_approve_milestone_for_path_b
    :parameters (?learner_path_b - learner_path_b ?milestone_type - milestone_type ?faculty_advisor - faculty_advisor)
    :precondition
      (and
        (entity_eligibility_confirmed ?learner_path_b)
        (assigned_faculty_advisor ?learner_path_b ?faculty_advisor)
        (path_b_has_milestone ?learner_path_b ?milestone_type)
        (milestone_reserved ?milestone_type)
        (not
          (path_b_processing_hold ?learner_path_b)
        )
      )
    :effect
      (and
        (path_b_processing_hold ?learner_path_b)
        (path_b_prerequisites_met ?learner_path_b)
      )
  )
  (:action apply_permission_token_to_path_b
    :parameters (?learner_path_b - learner_path_b ?milestone_type - milestone_type ?permission_token - permission_token)
    :precondition
      (and
        (entity_eligibility_confirmed ?learner_path_b)
        (path_b_has_milestone ?learner_path_b ?milestone_type)
        (permission_token_available ?permission_token)
        (not
          (path_b_processing_hold ?learner_path_b)
        )
      )
    :effect
      (and
        (milestone_in_progress ?milestone_type)
        (path_b_processing_hold ?learner_path_b)
        (path_b_permission_token_assigned ?learner_path_b ?permission_token)
        (not
          (permission_token_available ?permission_token)
        )
      )
  )
  (:action finalize_prerequisite_completion_for_path_b
    :parameters (?learner_path_b - learner_path_b ?milestone_type - milestone_type ?course_offering - course_offering ?permission_token - permission_token)
    :precondition
      (and
        (entity_eligibility_confirmed ?learner_path_b)
        (has_course_reservation ?learner_path_b ?course_offering)
        (path_b_has_milestone ?learner_path_b ?milestone_type)
        (milestone_in_progress ?milestone_type)
        (path_b_permission_token_assigned ?learner_path_b ?permission_token)
        (not
          (path_b_prerequisites_met ?learner_path_b)
        )
      )
    :effect
      (and
        (milestone_reserved ?milestone_type)
        (path_b_prerequisites_met ?learner_path_b)
        (permission_token_available ?permission_token)
        (not
          (path_b_permission_token_assigned ?learner_path_b ?permission_token)
        )
      )
  )
  (:action create_proposal_from_validated_paths
    :parameters (?learner_path_a - learner_path_a ?learner_path_b - learner_path_b ?prerequisite_requirement - prerequisite_requirement ?milestone_type - milestone_type ?capstone_proposal - capstone_proposal)
    :precondition
      (and
        (path_a_processing_hold ?learner_path_a)
        (path_b_processing_hold ?learner_path_b)
        (path_a_requires_prerequisite ?learner_path_a ?prerequisite_requirement)
        (path_b_has_milestone ?learner_path_b ?milestone_type)
        (prerequisite_validated ?prerequisite_requirement)
        (milestone_reserved ?milestone_type)
        (path_a_prerequisites_met ?learner_path_a)
        (path_b_prerequisites_met ?learner_path_b)
        (capstone_proposal_slot_available ?capstone_proposal)
      )
    :effect
      (and
        (capstone_proposal_reserved ?capstone_proposal)
        (proposal_associated_prerequisite ?capstone_proposal ?prerequisite_requirement)
        (proposal_associated_milestone ?capstone_proposal ?milestone_type)
        (not
          (capstone_proposal_slot_available ?capstone_proposal)
        )
      )
  )
  (:action create_proposal_with_prereq_in_progress
    :parameters (?learner_path_a - learner_path_a ?learner_path_b - learner_path_b ?prerequisite_requirement - prerequisite_requirement ?milestone_type - milestone_type ?capstone_proposal - capstone_proposal)
    :precondition
      (and
        (path_a_processing_hold ?learner_path_a)
        (path_b_processing_hold ?learner_path_b)
        (path_a_requires_prerequisite ?learner_path_a ?prerequisite_requirement)
        (path_b_has_milestone ?learner_path_b ?milestone_type)
        (prerequisite_in_progress ?prerequisite_requirement)
        (milestone_reserved ?milestone_type)
        (not
          (path_a_prerequisites_met ?learner_path_a)
        )
        (path_b_prerequisites_met ?learner_path_b)
        (capstone_proposal_slot_available ?capstone_proposal)
      )
    :effect
      (and
        (capstone_proposal_reserved ?capstone_proposal)
        (proposal_associated_prerequisite ?capstone_proposal ?prerequisite_requirement)
        (proposal_associated_milestone ?capstone_proposal ?milestone_type)
        (capstone_proposal_flag_a ?capstone_proposal)
        (not
          (capstone_proposal_slot_available ?capstone_proposal)
        )
      )
  )
  (:action create_proposal_with_milestone_in_progress
    :parameters (?learner_path_a - learner_path_a ?learner_path_b - learner_path_b ?prerequisite_requirement - prerequisite_requirement ?milestone_type - milestone_type ?capstone_proposal - capstone_proposal)
    :precondition
      (and
        (path_a_processing_hold ?learner_path_a)
        (path_b_processing_hold ?learner_path_b)
        (path_a_requires_prerequisite ?learner_path_a ?prerequisite_requirement)
        (path_b_has_milestone ?learner_path_b ?milestone_type)
        (prerequisite_validated ?prerequisite_requirement)
        (milestone_in_progress ?milestone_type)
        (path_a_prerequisites_met ?learner_path_a)
        (not
          (path_b_prerequisites_met ?learner_path_b)
        )
        (capstone_proposal_slot_available ?capstone_proposal)
      )
    :effect
      (and
        (capstone_proposal_reserved ?capstone_proposal)
        (proposal_associated_prerequisite ?capstone_proposal ?prerequisite_requirement)
        (proposal_associated_milestone ?capstone_proposal ?milestone_type)
        (capstone_proposal_flag_b ?capstone_proposal)
        (not
          (capstone_proposal_slot_available ?capstone_proposal)
        )
      )
  )
  (:action create_proposal_with_both_in_progress
    :parameters (?learner_path_a - learner_path_a ?learner_path_b - learner_path_b ?prerequisite_requirement - prerequisite_requirement ?milestone_type - milestone_type ?capstone_proposal - capstone_proposal)
    :precondition
      (and
        (path_a_processing_hold ?learner_path_a)
        (path_b_processing_hold ?learner_path_b)
        (path_a_requires_prerequisite ?learner_path_a ?prerequisite_requirement)
        (path_b_has_milestone ?learner_path_b ?milestone_type)
        (prerequisite_in_progress ?prerequisite_requirement)
        (milestone_in_progress ?milestone_type)
        (not
          (path_a_prerequisites_met ?learner_path_a)
        )
        (not
          (path_b_prerequisites_met ?learner_path_b)
        )
        (capstone_proposal_slot_available ?capstone_proposal)
      )
    :effect
      (and
        (capstone_proposal_reserved ?capstone_proposal)
        (proposal_associated_prerequisite ?capstone_proposal ?prerequisite_requirement)
        (proposal_associated_milestone ?capstone_proposal ?milestone_type)
        (capstone_proposal_flag_a ?capstone_proposal)
        (capstone_proposal_flag_b ?capstone_proposal)
        (not
          (capstone_proposal_slot_available ?capstone_proposal)
        )
      )
  )
  (:action confirm_capstone_proposal_slot
    :parameters (?capstone_proposal - capstone_proposal ?learner_path_a - learner_path_a ?course_offering - course_offering)
    :precondition
      (and
        (capstone_proposal_reserved ?capstone_proposal)
        (path_a_processing_hold ?learner_path_a)
        (has_course_reservation ?learner_path_a ?course_offering)
        (not
          (capstone_proposal_confirmed ?capstone_proposal)
        )
      )
    :effect (capstone_proposal_confirmed ?capstone_proposal)
  )
  (:action submit_assessment_artifact_to_component_proposal
    :parameters (?capstone_component - capstone_component ?assessment_artifact - assessment_artifact ?capstone_proposal - capstone_proposal)
    :precondition
      (and
        (entity_eligibility_confirmed ?capstone_component)
        (linked_to_proposal ?capstone_component ?capstone_proposal)
        (has_assessment_artifact_template ?capstone_component ?assessment_artifact)
        (assessment_artifact_template_available ?assessment_artifact)
        (capstone_proposal_reserved ?capstone_proposal)
        (capstone_proposal_confirmed ?capstone_proposal)
        (not
          (assessment_artifact_submitted ?assessment_artifact)
        )
      )
    :effect
      (and
        (assessment_artifact_submitted ?assessment_artifact)
        (assessment_artifact_associated_with_proposal ?assessment_artifact ?capstone_proposal)
        (not
          (assessment_artifact_template_available ?assessment_artifact)
        )
      )
  )
  (:action start_artifact_review_for_component
    :parameters (?capstone_component - capstone_component ?assessment_artifact - assessment_artifact ?capstone_proposal - capstone_proposal ?course_offering - course_offering)
    :precondition
      (and
        (entity_eligibility_confirmed ?capstone_component)
        (has_assessment_artifact_template ?capstone_component ?assessment_artifact)
        (assessment_artifact_submitted ?assessment_artifact)
        (assessment_artifact_associated_with_proposal ?assessment_artifact ?capstone_proposal)
        (has_course_reservation ?capstone_component ?course_offering)
        (not
          (capstone_proposal_flag_a ?capstone_proposal)
        )
        (not
          (review_stage_one_complete ?capstone_component)
        )
      )
    :effect (review_stage_one_complete ?capstone_component)
  )
  (:action attach_approval_document_to_component
    :parameters (?capstone_component - capstone_component ?approval_document - approval_document)
    :precondition
      (and
        (entity_eligibility_confirmed ?capstone_component)
