--- domain.pddl (as passed to VAL) ---
(define (domain pharmacy_allergy_interaction_alert_resolution)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity - object catalog_resource_category - entity template_category - entity packaging_label_category - entity clinical_artifact - entity prescription_order - clinical_artifact medication_option - catalog_resource_category clinical_alert_rule - catalog_resource_category pharmacy_staff - catalog_resource_category auxiliary_instruction_template - catalog_resource_category patient_instruction_template - catalog_resource_category counseling_script - catalog_resource_category clinical_reason_code - catalog_resource_category authorization_reason - catalog_resource_category substitution_proposal - template_category label_template - template_category authorization_template - template_category packaging_choice - packaging_label_category label_choice - packaging_label_category package_unit - packaging_label_category order_item_group - prescription_order packaging_group - prescription_order dispensing_line_item - order_item_group verification_line_item - order_item_group prescription_processing_record - packaging_group)

  (:predicates
    (prescription_screened ?prescription_order - prescription_order)
    (work_item_in_review ?prescription_order - prescription_order)
    (formulary_check_completed ?prescription_order - prescription_order)
    (work_item_alert_resolved ?prescription_order - prescription_order)
    (work_item_ready_for_dispense ?prescription_order - prescription_order)
    (work_item_clinical_review_completed ?prescription_order - prescription_order)
    (medication_option_available ?medication_option - medication_option)
    (work_item_selected_medication ?prescription_order - prescription_order ?medication_option - medication_option)
    (alert_rule_available ?clinical_alert_rule - clinical_alert_rule)
    (work_item_has_alert_rule ?prescription_order - prescription_order ?clinical_alert_rule - clinical_alert_rule)
    (staff_available ?pharmacy_staff - pharmacy_staff)
    (staff_assigned_to_work_item ?prescription_order - prescription_order ?pharmacy_staff - pharmacy_staff)
    (substitution_proposal_available ?substitution_proposal - substitution_proposal)
    (line_item_selected_substitution ?dispensing_line_item - dispensing_line_item ?substitution_proposal - substitution_proposal)
    (verification_selected_substitution ?verification_line_item - verification_line_item ?substitution_proposal - substitution_proposal)
    (line_item_packaging_compatible ?dispensing_line_item - dispensing_line_item ?packaging_choice - packaging_choice)
    (packaging_choice_reserved ?packaging_choice - packaging_choice)
    (packaging_choice_requires_substitution ?packaging_choice - packaging_choice)
    (line_item_packaging_confirmed ?dispensing_line_item - dispensing_line_item)
    (verification_label_compatible ?verification_line_item - verification_line_item ?label_choice - label_choice)
    (label_choice_reserved ?label_choice - label_choice)
    (label_choice_requires_template_change ?label_choice - label_choice)
    (verification_confirms_selection ?verification_line_item - verification_line_item)
    (package_unit_available ?package_unit - package_unit)
    (package_unit_reserved ?package_unit - package_unit)
    (package_has_packaging_choice ?package_unit - package_unit ?packaging_choice - packaging_choice)
    (package_has_label_choice ?package_unit - package_unit ?label_choice - label_choice)
    (package_requires_additional_labeling ?package_unit - package_unit)
    (package_requires_auxiliary_materials ?package_unit - package_unit)
    (package_ready_for_label_application ?package_unit - package_unit)
    (processing_record_has_dispensing_line_item ?prescription_processing_record - prescription_processing_record ?dispensing_line_item - dispensing_line_item)
    (processing_record_has_verification_line_item ?prescription_processing_record - prescription_processing_record ?verification_line_item - verification_line_item)
    (processing_record_has_reserved_package_unit ?prescription_processing_record - prescription_processing_record ?package_unit - package_unit)
    (label_template_available ?label_template - label_template)
    (processing_record_references_label_template ?prescription_processing_record - prescription_processing_record ?label_template - label_template)
    (label_template_selected ?label_template - label_template)
    (label_template_applied_to_package_unit ?label_template - label_template ?package_unit - package_unit)
    (processing_record_authorized_for_packaging ?prescription_processing_record - prescription_processing_record)
    (processing_record_packaging_ready ?prescription_processing_record - prescription_processing_record)
    (processing_record_clinical_cleared ?prescription_processing_record - prescription_processing_record)
    (processing_record_auxiliary_instruction_assigned ?prescription_processing_record - prescription_processing_record)
    (auxiliary_instruction_flagged ?prescription_processing_record - prescription_processing_record)
    (processing_record_patient_instructions_attached ?prescription_processing_record - prescription_processing_record)
    (processing_record_packaging_finalized ?prescription_processing_record - prescription_processing_record)
    (authorization_template_available ?authorization_template - authorization_template)
    (processing_record_linked_to_authorization_template ?prescription_processing_record - prescription_processing_record ?authorization_template - authorization_template)
    (authorization_request_created ?prescription_processing_record - prescription_processing_record)
    (authorization_approved ?prescription_processing_record - prescription_processing_record)
    (authorization_recorded_for_audit ?prescription_processing_record - prescription_processing_record)
    (auxiliary_instruction_template_available ?auxiliary_instruction_template - auxiliary_instruction_template)
    (processing_record_references_auxiliary_instruction_template ?prescription_processing_record - prescription_processing_record ?auxiliary_instruction_template - auxiliary_instruction_template)
    (patient_instruction_template_available ?patient_instruction_template - patient_instruction_template)
    (processing_record_references_patient_instruction_template ?prescription_processing_record - prescription_processing_record ?patient_instruction_template - patient_instruction_template)
    (clinical_reason_code_available ?clinical_reason_code - clinical_reason_code)
    (processing_record_linked_to_clinical_reason_code ?prescription_processing_record - prescription_processing_record ?clinical_reason_code - clinical_reason_code)
    (authorization_reason_available ?authorization_reason - authorization_reason)
    (processing_record_linked_to_authorization_reason ?prescription_processing_record - prescription_processing_record ?authorization_reason - authorization_reason)
    (counseling_script_available ?counseling_script - counseling_script)
    (prescription_linked_to_counseling_script ?prescription_order - prescription_order ?counseling_script - counseling_script)
    (line_item_packaging_asserted ?dispensing_line_item - dispensing_line_item)
    (verification_line_item_packaging_asserted ?verification_line_item - verification_line_item)
    (processing_record_finalized ?prescription_processing_record - prescription_processing_record)
  )
  (:action mark_prescription_screened
    :parameters (?prescription_order - prescription_order)
    :precondition
      (and
        (not
          (prescription_screened ?prescription_order)
        )
        (not
          (work_item_alert_resolved ?prescription_order)
        )
      )
    :effect (prescription_screened ?prescription_order)
  )
  (:action select_medication_option_for_prescription
    :parameters (?prescription_order - prescription_order ?medication_option - medication_option)
    :precondition
      (and
        (prescription_screened ?prescription_order)
        (not
          (formulary_check_completed ?prescription_order)
        )
        (medication_option_available ?medication_option)
      )
    :effect
      (and
        (formulary_check_completed ?prescription_order)
        (work_item_selected_medication ?prescription_order ?medication_option)
        (not
          (medication_option_available ?medication_option)
        )
      )
  )
  (:action assign_alert_rule_to_prescription
    :parameters (?prescription_order - prescription_order ?clinical_alert_rule - clinical_alert_rule)
    :precondition
      (and
        (prescription_screened ?prescription_order)
        (formulary_check_completed ?prescription_order)
        (alert_rule_available ?clinical_alert_rule)
      )
    :effect
      (and
        (work_item_has_alert_rule ?prescription_order ?clinical_alert_rule)
        (not
          (alert_rule_available ?clinical_alert_rule)
        )
      )
  )
  (:action flag_prescription_for_clinical_review
    :parameters (?prescription_order - prescription_order ?clinical_alert_rule - clinical_alert_rule)
    :precondition
      (and
        (prescription_screened ?prescription_order)
        (formulary_check_completed ?prescription_order)
        (work_item_has_alert_rule ?prescription_order ?clinical_alert_rule)
        (not
          (work_item_in_review ?prescription_order)
        )
      )
    :effect (work_item_in_review ?prescription_order)
  )
  (:action unassign_alert_rule_from_prescription
    :parameters (?prescription_order - prescription_order ?clinical_alert_rule - clinical_alert_rule)
    :precondition
      (and
        (work_item_has_alert_rule ?prescription_order ?clinical_alert_rule)
      )
    :effect
      (and
        (alert_rule_available ?clinical_alert_rule)
        (not
          (work_item_has_alert_rule ?prescription_order ?clinical_alert_rule)
        )
      )
  )
  (:action assign_staff_to_prescription
    :parameters (?prescription_order - prescription_order ?pharmacy_staff - pharmacy_staff)
    :precondition
      (and
        (work_item_in_review ?prescription_order)
        (staff_available ?pharmacy_staff)
      )
    :effect
      (and
        (staff_assigned_to_work_item ?prescription_order ?pharmacy_staff)
        (not
          (staff_available ?pharmacy_staff)
        )
      )
  )
  (:action unassign_staff_from_prescription
    :parameters (?prescription_order - prescription_order ?pharmacy_staff - pharmacy_staff)
    :precondition
      (and
        (staff_assigned_to_work_item ?prescription_order ?pharmacy_staff)
      )
    :effect
      (and
        (staff_available ?pharmacy_staff)
        (not
          (staff_assigned_to_work_item ?prescription_order ?pharmacy_staff)
        )
      )
  )
  (:action link_clinical_reason_code_to_processing_record
    :parameters (?prescription_processing_record - prescription_processing_record ?clinical_reason_code - clinical_reason_code)
    :precondition
      (and
        (work_item_in_review ?prescription_processing_record)
        (clinical_reason_code_available ?clinical_reason_code)
      )
    :effect
      (and
        (processing_record_linked_to_clinical_reason_code ?prescription_processing_record ?clinical_reason_code)
        (not
          (clinical_reason_code_available ?clinical_reason_code)
        )
      )
  )
  (:action unlink_clinical_reason_code_from_processing_record
    :parameters (?prescription_processing_record - prescription_processing_record ?clinical_reason_code - clinical_reason_code)
    :precondition
      (and
        (processing_record_linked_to_clinical_reason_code ?prescription_processing_record ?clinical_reason_code)
      )
    :effect
      (and
        (clinical_reason_code_available ?clinical_reason_code)
        (not
          (processing_record_linked_to_clinical_reason_code ?prescription_processing_record ?clinical_reason_code)
        )
      )
  )
  (:action link_authorization_reason_to_processing_record
    :parameters (?prescription_processing_record - prescription_processing_record ?authorization_reason - authorization_reason)
    :precondition
      (and
        (work_item_in_review ?prescription_processing_record)
        (authorization_reason_available ?authorization_reason)
      )
    :effect
      (and
        (processing_record_linked_to_authorization_reason ?prescription_processing_record ?authorization_reason)
        (not
          (authorization_reason_available ?authorization_reason)
        )
      )
  )
  (:action unlink_authorization_reason_from_processing_record
    :parameters (?prescription_processing_record - prescription_processing_record ?authorization_reason - authorization_reason)
    :precondition
      (and
        (processing_record_linked_to_authorization_reason ?prescription_processing_record ?authorization_reason)
      )
    :effect
      (and
        (authorization_reason_available ?authorization_reason)
        (not
          (processing_record_linked_to_authorization_reason ?prescription_processing_record ?authorization_reason)
        )
      )
  )
  (:action reserve_packaging_choice_for_line_item
    :parameters (?dispensing_line_item - dispensing_line_item ?packaging_choice - packaging_choice ?clinical_alert_rule - clinical_alert_rule)
    :precondition
      (and
        (work_item_in_review ?dispensing_line_item)
        (work_item_has_alert_rule ?dispensing_line_item ?clinical_alert_rule)
        (line_item_packaging_compatible ?dispensing_line_item ?packaging_choice)
        (not
          (packaging_choice_reserved ?packaging_choice)
        )
        (not
          (packaging_choice_requires_substitution ?packaging_choice)
        )
      )
    :effect (packaging_choice_reserved ?packaging_choice)
  )
  (:action staff_confirm_packaging_for_line_item
    :parameters (?dispensing_line_item - dispensing_line_item ?packaging_choice - packaging_choice ?pharmacy_staff - pharmacy_staff)
    :precondition
      (and
        (work_item_in_review ?dispensing_line_item)
        (staff_assigned_to_work_item ?dispensing_line_item ?pharmacy_staff)
        (line_item_packaging_compatible ?dispensing_line_item ?packaging_choice)
        (packaging_choice_reserved ?packaging_choice)
        (not
          (line_item_packaging_asserted ?dispensing_line_item)
        )
      )
    :effect
      (and
        (line_item_packaging_asserted ?dispensing_line_item)
        (line_item_packaging_confirmed ?dispensing_line_item)
      )
  )
  (:action propose_substitution_and_reserve_packaging_for_line_item
    :parameters (?dispensing_line_item - dispensing_line_item ?packaging_choice - packaging_choice ?substitution_proposal - substitution_proposal)
    :precondition
      (and
        (work_item_in_review ?dispensing_line_item)
        (line_item_packaging_compatible ?dispensing_line_item ?packaging_choice)
        (substitution_proposal_available ?substitution_proposal)
        (not
          (line_item_packaging_asserted ?dispensing_line_item)
        )
      )
    :effect
      (and
        (packaging_choice_requires_substitution ?packaging_choice)
        (line_item_packaging_asserted ?dispensing_line_item)
        (line_item_selected_substitution ?dispensing_line_item ?substitution_proposal)
        (not
          (substitution_proposal_available ?substitution_proposal)
        )
      )
  )
  (:action accept_substitution_and_confirm_packaging_for_line_item
    :parameters (?dispensing_line_item - dispensing_line_item ?packaging_choice - packaging_choice ?clinical_alert_rule - clinical_alert_rule ?substitution_proposal - substitution_proposal)
    :precondition
      (and
        (work_item_in_review ?dispensing_line_item)
        (work_item_has_alert_rule ?dispensing_line_item ?clinical_alert_rule)
        (line_item_packaging_compatible ?dispensing_line_item ?packaging_choice)
        (packaging_choice_requires_substitution ?packaging_choice)
        (line_item_selected_substitution ?dispensing_line_item ?substitution_proposal)
        (not
          (line_item_packaging_confirmed ?dispensing_line_item)
        )
      )
    :effect
      (and
        (packaging_choice_reserved ?packaging_choice)
        (line_item_packaging_confirmed ?dispensing_line_item)
        (substitution_proposal_available ?substitution_proposal)
        (not
          (line_item_selected_substitution ?dispensing_line_item ?substitution_proposal)
        )
      )
  )
  (:action reserve_label_choice_for_verification_line_item
    :parameters (?verification_line_item - verification_line_item ?label_choice - label_choice ?clinical_alert_rule - clinical_alert_rule)
    :precondition
      (and
        (work_item_in_review ?verification_line_item)
        (work_item_has_alert_rule ?verification_line_item ?clinical_alert_rule)
        (verification_label_compatible ?verification_line_item ?label_choice)
        (not
          (label_choice_reserved ?label_choice)
        )
        (not
          (label_choice_requires_template_change ?label_choice)
        )
      )
    :effect (label_choice_reserved ?label_choice)
  )
  (:action staff_confirm_label_for_verification_line_item
    :parameters (?verification_line_item - verification_line_item ?label_choice - label_choice ?pharmacy_staff - pharmacy_staff)
    :precondition
      (and
        (work_item_in_review ?verification_line_item)
        (staff_assigned_to_work_item ?verification_line_item ?pharmacy_staff)
        (verification_label_compatible ?verification_line_item ?label_choice)
        (label_choice_reserved ?label_choice)
        (not
          (verification_line_item_packaging_asserted ?verification_line_item)
        )
      )
    :effect
      (and
        (verification_line_item_packaging_asserted ?verification_line_item)
        (verification_confirms_selection ?verification_line_item)
      )
  )
  (:action propose_label_substitution_for_verification_line_item
    :parameters (?verification_line_item - verification_line_item ?label_choice - label_choice ?substitution_proposal - substitution_proposal)
    :precondition
      (and
        (work_item_in_review ?verification_line_item)
        (verification_label_compatible ?verification_line_item ?label_choice)
        (substitution_proposal_available ?substitution_proposal)
        (not
          (verification_line_item_packaging_asserted ?verification_line_item)
        )
      )
    :effect
      (and
        (label_choice_requires_template_change ?label_choice)
        (verification_line_item_packaging_asserted ?verification_line_item)
        (verification_selected_substitution ?verification_line_item ?substitution_proposal)
        (not
          (substitution_proposal_available ?substitution_proposal)
        )
      )
  )
  (:action accept_label_substitution_and_confirm_verification_line_item
    :parameters (?verification_line_item - verification_line_item ?label_choice - label_choice ?clinical_alert_rule - clinical_alert_rule ?substitution_proposal - substitution_proposal)
    :precondition
      (and
        (work_item_in_review ?verification_line_item)
        (work_item_has_alert_rule ?verification_line_item ?clinical_alert_rule)
        (verification_label_compatible ?verification_line_item ?label_choice)
        (label_choice_requires_template_change ?label_choice)
        (verification_selected_substitution ?verification_line_item ?substitution_proposal)
        (not
          (verification_confirms_selection ?verification_line_item)
        )
      )
    :effect
      (and
        (label_choice_reserved ?label_choice)
        (verification_confirms_selection ?verification_line_item)
        (substitution_proposal_available ?substitution_proposal)
        (not
          (verification_selected_substitution ?verification_line_item ?substitution_proposal)
        )
      )
  )
  (:action reserve_package_and_assign_packaging_and_label
    :parameters (?dispensing_line_item - dispensing_line_item ?verification_line_item - verification_line_item ?packaging_choice - packaging_choice ?label_choice - label_choice ?package_unit - package_unit)
    :precondition
      (and
        (line_item_packaging_asserted ?dispensing_line_item)
        (verification_line_item_packaging_asserted ?verification_line_item)
        (line_item_packaging_compatible ?dispensing_line_item ?packaging_choice)
        (verification_label_compatible ?verification_line_item ?label_choice)
        (packaging_choice_reserved ?packaging_choice)
        (label_choice_reserved ?label_choice)
        (line_item_packaging_confirmed ?dispensing_line_item)
        (verification_confirms_selection ?verification_line_item)
        (package_unit_available ?package_unit)
      )
    :effect
      (and
        (package_unit_reserved ?package_unit)
        (package_has_packaging_choice ?package_unit ?packaging_choice)
        (package_has_label_choice ?package_unit ?label_choice)
        (not
          (package_unit_available ?package_unit)
        )
      )
  )
  (:action reserve_package_with_additional_labeling_required
    :parameters (?dispensing_line_item - dispensing_line_item ?verification_line_item - verification_line_item ?packaging_choice - packaging_choice ?label_choice - label_choice ?package_unit - package_unit)
    :precondition
      (and
        (line_item_packaging_asserted ?dispensing_line_item)
        (verification_line_item_packaging_asserted ?verification_line_item)
        (line_item_packaging_compatible ?dispensing_line_item ?packaging_choice)
        (verification_label_compatible ?verification_line_item ?label_choice)
        (packaging_choice_requires_substitution ?packaging_choice)
        (label_choice_reserved ?label_choice)
        (not
          (line_item_packaging_confirmed ?dispensing_line_item)
        )
        (verification_confirms_selection ?verification_line_item)
        (package_unit_available ?package_unit)
      )
    :effect
      (and
        (package_unit_reserved ?package_unit)
        (package_has_packaging_choice ?package_unit ?packaging_choice)
        (package_has_label_choice ?package_unit ?label_choice)
        (package_requires_additional_labeling ?package_unit)
        (not
          (package_unit_available ?package_unit)
        )
      )
  )
  (:action reserve_package_with_auxiliary_instructions_required
    :parameters (?dispensing_line_item - dispensing_line_item ?verification_line_item - verification_line_item ?packaging_choice - packaging_choice ?label_choice - label_choice ?package_unit - package_unit)
    :precondition
      (and
        (line_item_packaging_asserted ?dispensing_line_item)
        (verification_line_item_packaging_asserted ?verification_line_item)
        (line_item_packaging_compatible ?dispensing_line_item ?packaging_choice)
        (verification_label_compatible ?verification_line_item ?label_choice)
        (packaging_choice_reserved ?packaging_choice)
        (label_choice_requires_template_change ?label_choice)
        (line_item_packaging_confirmed ?dispensing_line_item)
        (not
          (verification_confirms_selection ?verification_line_item)
        )
        (package_unit_available ?package_unit)
      )
    :effect
      (and
        (package_unit_reserved ?package_unit)
        (package_has_packaging_choice ?package_unit ?packaging_choice)
        (package_has_label_choice ?package_unit ?label_choice)
        (package_requires_auxiliary_materials ?package_unit)
        (not
          (package_unit_available ?package_unit)
        )
      )
  )
  (:action reserve_package_with_labeling_and_auxiliary_required
    :parameters (?dispensing_line_item - dispensing_line_item ?verification_line_item - verification_line_item ?packaging_choice - packaging_choice ?label_choice - label_choice ?package_unit - package_unit)
    :precondition
      (and
        (line_item_packaging_asserted ?dispensing_line_item)
        (verification_line_item_packaging_asserted ?verification_line_item)
        (line_item_packaging_compatible ?dispensing_line_item ?packaging_choice)
        (verification_label_compatible ?verification_line_item ?label_choice)
        (packaging_choice_requires_substitution ?packaging_choice)
        (label_choice_requires_template_change ?label_choice)
        (not
          (line_item_packaging_confirmed ?dispensing_line_item)
        )
        (not
          (verification_confirms_selection ?verification_line_item)
        )
        (package_unit_available ?package_unit)
      )
    :effect
      (and
        (package_unit_reserved ?package_unit)
        (package_has_packaging_choice ?package_unit ?packaging_choice)
        (package_has_label_choice ?package_unit ?label_choice)
        (package_requires_additional_labeling ?package_unit)
        (package_requires_auxiliary_materials ?package_unit)
        (not
          (package_unit_available ?package_unit)
        )
      )
  )
  (:action mark_package_ready_for_label_application
    :parameters (?package_unit - package_unit ?dispensing_line_item - dispensing_line_item ?clinical_alert_rule - clinical_alert_rule)
    :precondition
      (and
        (package_unit_reserved ?package_unit)
        (line_item_packaging_asserted ?dispensing_line_item)
        (work_item_has_alert_rule ?dispensing_line_item ?clinical_alert_rule)
        (not
          (package_ready_for_label_application ?package_unit)
        )
      )
    :effect (package_ready_for_label_application ?package_unit)
  )
  (:action select_label_template_and_apply_to_package
    :parameters (?prescription_processing_record - prescription_processing_record ?label_template - label_template ?package_unit - package_unit)
    :precondition
      (and
        (work_item_in_review ?prescription_processing_record)
        (processing_record_has_reserved_package_unit ?prescription_processing_record ?package_unit)
        (processing_record_references_label_template ?prescription_processing_record ?label_template)
        (label_template_available ?label_template)
        (package_unit_reserved ?package_unit)
        (package_ready_for_label_application ?package_unit)
        (not
          (label_template_selected ?label_template)
        )
      )
    :effect
      (and
        (label_template_selected ?label_template)
        (label_template_applied_to_package_unit ?label_template ?package_unit)
        (not
          (label_template_available ?label_template)
        )
      )
  )
  (:action authorize_processing_record_for_packaging_operations
    :parameters (?prescription_processing_record - prescription_processing_record ?label_template - label_template ?package_unit - package_unit ?clinical_alert_rule - clinical_alert_rule)
    :precondition
      (and
        (work_item_in_review ?prescription_processing_record)
        (processing_record_references_label_template ?prescription_processing_record ?label_template)
        (label_template_selected ?label_template)
        (label_template_applied_to_package_unit ?label_template ?package_unit)
        (work_item_has_alert_rule ?prescription_processing_record ?clinical_alert_rule)
        (not
          (package_requires_additional_labeling ?package_unit)
        )
        (not
          (processing_record_authorized_for_packaging ?prescription_processing_record)
