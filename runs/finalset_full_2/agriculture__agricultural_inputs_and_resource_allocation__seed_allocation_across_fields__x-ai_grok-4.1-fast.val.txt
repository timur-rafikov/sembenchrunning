  (:predicates
    (plot_request_registered ?field_plot_request - allocation_unit)
    (unit_allocation_approved ?field_plot_request - allocation_unit)
    (request_seed_reserved ?field_plot_request - allocation_unit)
    (unit_allocation_confirmed ?field_plot_request - allocation_unit)
    (unit_allocation_finalized ?field_plot_request - allocation_unit)
    (contingency_assigned ?field_plot_request - allocation_unit)
    (seed_batch_available ?seed_batch - seed_batch)
    (seed_batch_allocated_to_request ?field_plot_request - allocation_unit ?seed_batch - seed_batch)
    (seed_variety_available ?seed_variety - seed_variety)
    (assigned_seed_variety_for_unit ?field_plot_request - allocation_unit ?seed_variety - seed_variety)
    (machinery_available ?machinery_unit - machinery_unit)
    (machinery_assigned_to_unit ?field_plot_request - allocation_unit ?machinery_unit - machinery_unit)
    (fertilizer_available ?fertilizer_batch - fertilizer_batch)
    (fertilizer_assigned_to_irrigated_unit ?irrigated_field_unit - irrigated_field_unit ?fertilizer_batch - fertilizer_batch)
    (fertilizer_assigned_to_rainfed_unit ?rainfed_field_unit - rainfed_field_unit ?fertilizer_batch - fertilizer_batch)
    (allocation_window_assigned_to_irrigated_unit ?irrigated_field_unit - irrigated_field_unit ?allocation_window_irrigated - allocation_window_irrigated)
    (allocation_window_irrigated_ready ?allocation_window_irrigated - allocation_window_irrigated)
    (allocation_window_irrigated_engaged ?allocation_window_irrigated - allocation_window_irrigated)
    (irrigated_unit_resources_confirmed ?irrigated_field_unit - irrigated_field_unit)
    (allocation_window_assigned_to_rainfed_unit ?rainfed_field_unit - rainfed_field_unit ?allocation_window_rainfed - allocation_window_rainfed)
    (allocation_window_rainfed_ready ?allocation_window_rainfed - allocation_window_rainfed)
    (allocation_window_rainfed_engaged ?allocation_window_rainfed - allocation_window_rainfed)
    (rainfed_unit_resources_confirmed ?rainfed_field_unit - rainfed_field_unit)
    (allocation_package_available ?allocation_package - allocation_package)
    (allocation_package_staged ?allocation_package - allocation_package)
    (package_allocated_to_allocation_window_irrigated ?allocation_package - allocation_package ?allocation_window_irrigated - allocation_window_irrigated)
    (package_allocated_to_allocation_window_rainfed ?allocation_package - allocation_package ?allocation_window_rainfed - allocation_window_rainfed)
    (allocation_package_priority_flag ?allocation_package - allocation_package)
    (allocation_package_secondary_flag ?allocation_package - allocation_package)
    (allocation_package_inspected ?allocation_package - allocation_package)
    (farm_site_manages_irrigated_unit ?farm_site - farm_site ?irrigated_field_unit - irrigated_field_unit)
    (farm_site_manages_rainfed_unit ?farm_site - farm_site ?rainfed_field_unit - rainfed_field_unit)
    (farm_site_has_package ?farm_site - farm_site ?allocation_package - allocation_package)
    (storage_container_available ?storage_container - storage_container)
    (farm_site_has_storage_container ?farm_site - farm_site ?storage_container - storage_container)
    (storage_container_in_use ?storage_container - storage_container)
    (storage_container_contains_package ?storage_container - storage_container ?allocation_package - allocation_package)

  (:action register_plot_request
    :parameters (?field_plot_request - allocation_unit)
    :precondition
      (and
        (not
          (plot_request_registered ?field_plot_request)
        )
        (not
          (unit_allocation_confirmed ?field_plot_request)
        )
      )
    :effect (plot_request_registered ?field_plot_request)
  )
  (:action assign_seed_batch_to_request
    :parameters (?field_plot_request - allocation_unit ?seed_batch - seed_batch)
    :precondition
      (and
        (plot_request_registered ?field_plot_request)
        (not
          (request_seed_reserved ?field_plot_request)
        )
        (seed_batch_available ?seed_batch)
      )
    :effect
      (and
        (request_seed_reserved ?field_plot_request)
        (seed_batch_allocated_to_request ?field_plot_request ?seed_batch)
        (not
          (seed_batch_available ?seed_batch)
        )
      )
  )
  (:action assign_seed_variety_to_request
    :parameters (?field_plot_request - allocation_unit ?seed_variety - seed_variety)
    :precondition
      (and
        (plot_request_registered ?field_plot_request)
        (request_seed_reserved ?field_plot_request)
        (seed_variety_available ?seed_variety)
      )
    :effect
      (and
        (assigned_seed_variety_for_unit ?field_plot_request ?seed_variety)
        (not
          (seed_variety_available ?seed_variety)
        )
      )
  )
  (:action approve_variety_assignment
    :parameters (?field_plot_request - allocation_unit ?seed_variety - seed_variety)
    :precondition
      (and
        (plot_request_registered ?field_plot_request)
        (request_seed_reserved ?field_plot_request)
        (assigned_seed_variety_for_unit ?field_plot_request ?seed_variety)
        (not
          (unit_allocation_approved ?field_plot_request)
        )
      )
    :effect (unit_allocation_approved ?field_plot_request)
  )
  (:action unassign_seed_variety_from_request
    :parameters (?field_plot_request - allocation_unit ?seed_variety - seed_variety)
    :precondition
      (and
        (assigned_seed_variety_for_unit ?field_plot_request ?seed_variety)
      )
    :effect
      (and
        (seed_variety_available ?seed_variety)
        (not
          (assigned_seed_variety_for_unit ?field_plot_request ?seed_variety)
        )
      )
  )
  (:action assign_machinery_to_request
    :parameters (?field_plot_request - allocation_unit ?machinery_unit - machinery_unit)
    :precondition
      (and
        (unit_allocation_approved ?field_plot_request)
        (machinery_available ?machinery_unit)
      )
    :effect
      (and
        (machinery_assigned_to_unit ?field_plot_request ?machinery_unit)
        (not
          (machinery_available ?machinery_unit)
        )
      )
  )
  (:action release_machinery_from_request
    :parameters (?field_plot_request - allocation_unit ?machinery_unit - machinery_unit)
    :precondition
      (and
        (machinery_assigned_to_unit ?field_plot_request ?machinery_unit)
      )
    :effect
      (and
        (machinery_available ?machinery_unit)
        (not
          (machinery_assigned_to_unit ?field_plot_request ?machinery_unit)
        )
      )
  )

    :parameters (?irrigated_field_unit - irrigated_field_unit ?allocation_window_irrigated - allocation_window_irrigated ?seed_variety - seed_variety)

        (not
          (allocation_window_irrigated_ready ?allocation_window_irrigated)
        )
        (not
          (allocation_window_irrigated_engaged ?allocation_window_irrigated)

  )

          (irrigated_unit_preparation_checked ?irrigated_field_unit)

      )
