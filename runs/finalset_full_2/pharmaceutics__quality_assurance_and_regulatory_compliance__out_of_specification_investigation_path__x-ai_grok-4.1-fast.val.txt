--- domain.pddl (as passed to VAL) ---
(define (domain pharmaceutics_oos_investigation_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types personnel_or_resource_group - object resource_group - object document_group - object entity_group - object record_component - entity_group investigator - personnel_or_resource_group laboratory_analyst - personnel_or_resource_group technical_reviewer - personnel_or_resource_group approval_type - personnel_or_resource_group approver - personnel_or_resource_group disposition_resource - personnel_or_resource_group validation_activity - personnel_or_resource_group regulatory_document - personnel_or_resource_group evidence_item - resource_group supporting_document - resource_group capa_record - resource_group lab_result_primary - document_group lab_result_retest - document_group quality_record - document_group case_subcomponent_group - record_component record_subtype_group - record_component production_sample - case_subcomponent_group production_batch - case_subcomponent_group investigation_record - record_subtype_group)
  (:predicates
    (record_initiated ?case_component - record_component)
    (record_active ?case_component - record_component)
    (record_assigned ?case_component - record_component)
    (record_released_for_disposition ?case_component - record_component)
    (record_approval_requested ?case_component - record_component)
    (record_approval_granted ?case_component - record_component)
    (investigator_available ?assignee - investigator)
    (investigator_assigned_to_record ?case_component - record_component ?assignee - investigator)
    (analyst_available ?laboratory_analyst - laboratory_analyst)
    (laboratory_analyst_assigned_to_record ?case_component - record_component ?laboratory_analyst - laboratory_analyst)
    (reviewer_available ?technical_reviewer - technical_reviewer)
    (technical_reviewer_assigned_to_record ?case_component - record_component ?technical_reviewer - technical_reviewer)
    (evidence_available ?evidence_item - evidence_item)
    (record_sample_has_evidence ?production_sample - production_sample ?evidence_item - evidence_item)
    (record_batch_has_evidence ?production_batch - production_batch ?evidence_item - evidence_item)
    (record_sample_has_primary_result ?production_sample - production_sample ?lab_result_primary - lab_result_primary)
    (primary_result_reported ?lab_result_primary - lab_result_primary)
    (primary_result_confirmed ?lab_result_primary - lab_result_primary)
    (record_sample_reviewed ?production_sample - production_sample)
    (record_batch_has_retest_result ?production_batch - production_batch ?lab_result_retest - lab_result_retest)
    (retest_result_reported ?lab_result_retest - lab_result_retest)
    (retest_result_confirmed ?lab_result_retest - lab_result_retest)
    (record_batch_reviewed ?production_batch - production_batch)
    (quality_record_prepared ?quality_record - quality_record)
    (quality_record_generated ?quality_record - quality_record)
    (quality_record_includes_primary_result ?quality_record - quality_record ?lab_result_primary - lab_result_primary)
    (quality_record_includes_retest_result ?quality_record - quality_record ?lab_result_retest - lab_result_retest)
    (quality_record_requires_validation ?quality_record - quality_record)
    (quality_record_requires_capa ?quality_record - quality_record)
    (quality_record_lab_review_complete ?quality_record - quality_record)
    (record_links_sample ?investigation_record - investigation_record ?production_sample - production_sample)
    (record_links_batch ?investigation_record - investigation_record ?production_batch - production_batch)
    (record_investigation_has_quality_record ?investigation_record - investigation_record ?quality_record - quality_record)
    (supporting_document_available ?supporting_document - supporting_document)
    (record_includes_supporting_document ?investigation_record - investigation_record ?supporting_document - supporting_document)
    (supporting_document_attached ?supporting_document - supporting_document)
    (supporting_document_linked_to_quality_record ?supporting_document - supporting_document ?quality_record - quality_record)
    (record_evidence_locked ?investigation_record - investigation_record)
    (record_package_completed ?investigation_record - investigation_record)
    (record_ready_for_approval ?investigation_record - investigation_record)
    (record_approval_route_assigned ?investigation_record - investigation_record)
    (record_signoff_initialized ?investigation_record - investigation_record)
    (record_approval_round_initialized ?investigation_record - investigation_record)
    (record_cross_functional_review_complete ?investigation_record - investigation_record)
    (capa_available ?capa_record - capa_record)
    (record_linked_to_capa ?investigation_record - investigation_record ?capa_record - capa_record)
    (record_capa_associated_with_record ?investigation_record - investigation_record)
    (record_capa_review_in_progress ?investigation_record - investigation_record)
    (record_capa_review_complete ?investigation_record - investigation_record)
    (approval_type_available ?approval_type - approval_type)
    (approval_type_assigned_to_record ?investigation_record - investigation_record ?approval_type - approval_type)
    (approver_available ?approver - approver)
    (approver_assigned_to_record ?investigation_record - investigation_record ?approver - approver)
    (validation_activity_available ?validation_activity - validation_activity)
    (validation_activity_assigned_to_record ?investigation_record - investigation_record ?validation_activity - validation_activity)
    (regulatory_document_available ?regulatory_correspondence - regulatory_document)
    (record_linked_regulatory_document ?investigation_record - investigation_record ?regulatory_correspondence - regulatory_document)
    (disposition_resource_available ?disposition_resource - disposition_resource)
    (record_associated_disposition_resource ?case_component - record_component ?disposition_resource - disposition_resource)
    (record_sample_testing_initialized ?production_sample - production_sample)
    (record_batch_testing_initialized ?production_batch - production_batch)
    (record_final_signoff_logged ?investigation_record - investigation_record)
  )
  (:action initiate_investigation_case
    :parameters (?case_component - record_component)
    :precondition
      (and
        (not
          (record_initiated ?case_component)
        )
        (not
          (record_released_for_disposition ?case_component)
        )
      )
    :effect (record_initiated ?case_component)
  )
  (:action assign_investigator
    :parameters (?case_component - record_component ?assignee - investigator)
    :precondition
      (and
        (record_initiated ?case_component)
        (not
          (record_assigned ?case_component)
        )
        (investigator_available ?assignee)
      )
    :effect
      (and
        (record_assigned ?case_component)
        (investigator_assigned_to_record ?case_component ?assignee)
        (not
          (investigator_available ?assignee)
        )
      )
  )
  (:action assign_lab_analyst
    :parameters (?case_component - record_component ?laboratory_analyst - laboratory_analyst)
    :precondition
      (and
        (record_initiated ?case_component)
        (record_assigned ?case_component)
        (analyst_available ?laboratory_analyst)
      )
    :effect
      (and
        (laboratory_analyst_assigned_to_record ?case_component ?laboratory_analyst)
        (not
          (analyst_available ?laboratory_analyst)
        )
      )
  )
  (:action mark_case_active
    :parameters (?case_component - record_component ?laboratory_analyst - laboratory_analyst)
    :precondition
      (and
        (record_initiated ?case_component)
        (record_assigned ?case_component)
        (laboratory_analyst_assigned_to_record ?case_component ?laboratory_analyst)
        (not
          (record_active ?case_component)
        )
      )
    :effect (record_active ?case_component)
  )
  (:action release_lab_analyst
    :parameters (?case_component - record_component ?laboratory_analyst - laboratory_analyst)
    :precondition
      (and
        (laboratory_analyst_assigned_to_record ?case_component ?laboratory_analyst)
      )
    :effect
      (and
        (analyst_available ?laboratory_analyst)
        (not
          (laboratory_analyst_assigned_to_record ?case_component ?laboratory_analyst)
        )
      )
  )
  (:action assign_technical_reviewer
    :parameters (?case_component - record_component ?technical_reviewer - technical_reviewer)
    :precondition
      (and
        (record_active ?case_component)
        (reviewer_available ?technical_reviewer)
      )
    :effect
      (and
        (technical_reviewer_assigned_to_record ?case_component ?technical_reviewer)
        (not
          (reviewer_available ?technical_reviewer)
        )
      )
  )
  (:action release_technical_reviewer
    :parameters (?case_component - record_component ?technical_reviewer - technical_reviewer)
    :precondition
      (and
        (technical_reviewer_assigned_to_record ?case_component ?technical_reviewer)
      )
    :effect
      (and
        (reviewer_available ?technical_reviewer)
        (not
          (technical_reviewer_assigned_to_record ?case_component ?technical_reviewer)
        )
      )
  )
  (:action reserve_validation_activity_for_record
    :parameters (?investigation_record - investigation_record ?validation_activity - validation_activity)
    :precondition
      (and
        (record_active ?investigation_record)
        (validation_activity_available ?validation_activity)
      )
    :effect
      (and
        (validation_activity_assigned_to_record ?investigation_record ?validation_activity)
        (not
          (validation_activity_available ?validation_activity)
        )
      )
  )
  (:action release_validation_activity_from_record
    :parameters (?investigation_record - investigation_record ?validation_activity - validation_activity)
    :precondition
      (and
        (validation_activity_assigned_to_record ?investigation_record ?validation_activity)
      )
    :effect
      (and
        (validation_activity_available ?validation_activity)
        (not
          (validation_activity_assigned_to_record ?investigation_record ?validation_activity)
        )
      )
  )
  (:action reserve_regulatory_document_for_record
    :parameters (?investigation_record - investigation_record ?regulatory_correspondence - regulatory_document)
    :precondition
      (and
        (record_active ?investigation_record)
        (regulatory_document_available ?regulatory_correspondence)
      )
    :effect
      (and
        (record_linked_regulatory_document ?investigation_record ?regulatory_correspondence)
        (not
          (regulatory_document_available ?regulatory_correspondence)
        )
      )
  )
  (:action release_regulatory_document_from_record
    :parameters (?investigation_record - investigation_record ?regulatory_correspondence - regulatory_document)
    :precondition
      (and
        (record_linked_regulatory_document ?investigation_record ?regulatory_correspondence)
      )
    :effect
      (and
        (regulatory_document_available ?regulatory_correspondence)
        (not
          (record_linked_regulatory_document ?investigation_record ?regulatory_correspondence)
        )
      )
  )
  (:action report_primary_result
    :parameters (?production_sample - production_sample ?lab_result_primary - lab_result_primary ?laboratory_analyst - laboratory_analyst)
    :precondition
      (and
        (record_active ?production_sample)
        (laboratory_analyst_assigned_to_record ?production_sample ?laboratory_analyst)
        (record_sample_has_primary_result ?production_sample ?lab_result_primary)
        (not
          (primary_result_reported ?lab_result_primary)
        )
        (not
          (primary_result_confirmed ?lab_result_primary)
        )
      )
    :effect (primary_result_reported ?lab_result_primary)
  )
  (:action technical_review_of_primary_result
    :parameters (?production_sample - production_sample ?lab_result_primary - lab_result_primary ?technical_reviewer - technical_reviewer)
    :precondition
      (and
        (record_active ?production_sample)
        (technical_reviewer_assigned_to_record ?production_sample ?technical_reviewer)
        (record_sample_has_primary_result ?production_sample ?lab_result_primary)
        (primary_result_reported ?lab_result_primary)
        (not
          (record_sample_testing_initialized ?production_sample)
        )
      )
    :effect
      (and
        (record_sample_testing_initialized ?production_sample)
        (record_sample_reviewed ?production_sample)
      )
  )
  (:action attach_evidence_and_flag_confirmatory
    :parameters (?production_sample - production_sample ?lab_result_primary - lab_result_primary ?evidence_item - evidence_item)
    :precondition
      (and
        (record_active ?production_sample)
        (record_sample_has_primary_result ?production_sample ?lab_result_primary)
        (evidence_available ?evidence_item)
        (not
          (record_sample_testing_initialized ?production_sample)
        )
      )
    :effect
      (and
        (primary_result_confirmed ?lab_result_primary)
        (record_sample_testing_initialized ?production_sample)
        (record_sample_has_evidence ?production_sample ?evidence_item)
        (not
          (evidence_available ?evidence_item)
        )
      )
  )
  (:action process_confirmatory_result
    :parameters (?production_sample - production_sample ?lab_result_primary - lab_result_primary ?laboratory_analyst - laboratory_analyst ?evidence_item - evidence_item)
    :precondition
      (and
        (record_active ?production_sample)
        (laboratory_analyst_assigned_to_record ?production_sample ?laboratory_analyst)
        (record_sample_has_primary_result ?production_sample ?lab_result_primary)
        (primary_result_confirmed ?lab_result_primary)
        (record_sample_has_evidence ?production_sample ?evidence_item)
        (not
          (record_sample_reviewed ?production_sample)
        )
      )
    :effect
      (and
        (primary_result_reported ?lab_result_primary)
        (record_sample_reviewed ?production_sample)
        (evidence_available ?evidence_item)
        (not
          (record_sample_has_evidence ?production_sample ?evidence_item)
        )
      )
  )
  (:action report_batch_retest_result
    :parameters (?production_batch - production_batch ?lab_result_retest - lab_result_retest ?laboratory_analyst - laboratory_analyst)
    :precondition
      (and
        (record_active ?production_batch)
        (laboratory_analyst_assigned_to_record ?production_batch ?laboratory_analyst)
        (record_batch_has_retest_result ?production_batch ?lab_result_retest)
        (not
          (retest_result_reported ?lab_result_retest)
        )
        (not
          (retest_result_confirmed ?lab_result_retest)
        )
      )
    :effect (retest_result_reported ?lab_result_retest)
  )
  (:action technical_review_of_batch_result
    :parameters (?production_batch - production_batch ?lab_result_retest - lab_result_retest ?technical_reviewer - technical_reviewer)
    :precondition
      (and
        (record_active ?production_batch)
        (technical_reviewer_assigned_to_record ?production_batch ?technical_reviewer)
        (record_batch_has_retest_result ?production_batch ?lab_result_retest)
        (retest_result_reported ?lab_result_retest)
        (not
          (record_batch_testing_initialized ?production_batch)
        )
      )
    :effect
      (and
        (record_batch_testing_initialized ?production_batch)
        (record_batch_reviewed ?production_batch)
      )
  )
  (:action attach_batch_evidence_and_flag_confirmatory
    :parameters (?production_batch - production_batch ?lab_result_retest - lab_result_retest ?evidence_item - evidence_item)
    :precondition
      (and
        (record_active ?production_batch)
        (record_batch_has_retest_result ?production_batch ?lab_result_retest)
        (evidence_available ?evidence_item)
        (not
          (record_batch_testing_initialized ?production_batch)
        )
      )
    :effect
      (and
        (retest_result_confirmed ?lab_result_retest)
        (record_batch_testing_initialized ?production_batch)
        (record_batch_has_evidence ?production_batch ?evidence_item)
        (not
          (evidence_available ?evidence_item)
        )
      )
  )
  (:action process_batch_confirmatory_result
    :parameters (?production_batch - production_batch ?lab_result_retest - lab_result_retest ?laboratory_analyst - laboratory_analyst ?evidence_item - evidence_item)
    :precondition
      (and
        (record_active ?production_batch)
        (laboratory_analyst_assigned_to_record ?production_batch ?laboratory_analyst)
        (record_batch_has_retest_result ?production_batch ?lab_result_retest)
        (retest_result_confirmed ?lab_result_retest)
        (record_batch_has_evidence ?production_batch ?evidence_item)
        (not
          (record_batch_reviewed ?production_batch)
        )
      )
    :effect
      (and
        (retest_result_reported ?lab_result_retest)
        (record_batch_reviewed ?production_batch)
        (evidence_available ?evidence_item)
        (not
          (record_batch_has_evidence ?production_batch ?evidence_item)
        )
      )
  )
  (:action assemble_quality_record_with_results
    :parameters (?production_sample - production_sample ?production_batch - production_batch ?lab_result_primary - lab_result_primary ?lab_result_retest - lab_result_retest ?quality_record - quality_record)
    :precondition
      (and
        (record_sample_testing_initialized ?production_sample)
        (record_batch_testing_initialized ?production_batch)
        (record_sample_has_primary_result ?production_sample ?lab_result_primary)
        (record_batch_has_retest_result ?production_batch ?lab_result_retest)
        (primary_result_reported ?lab_result_primary)
        (retest_result_reported ?lab_result_retest)
        (record_sample_reviewed ?production_sample)
        (record_batch_reviewed ?production_batch)
        (quality_record_prepared ?quality_record)
      )
    :effect
      (and
        (quality_record_generated ?quality_record)
        (quality_record_includes_primary_result ?quality_record ?lab_result_primary)
        (quality_record_includes_retest_result ?quality_record ?lab_result_retest)
        (not
          (quality_record_prepared ?quality_record)
        )
      )
  )
  (:action assemble_quality_record_with_validation_flag
    :parameters (?production_sample - production_sample ?production_batch - production_batch ?lab_result_primary - lab_result_primary ?lab_result_retest - lab_result_retest ?quality_record - quality_record)
    :precondition
      (and
        (record_sample_testing_initialized ?production_sample)
        (record_batch_testing_initialized ?production_batch)
        (record_sample_has_primary_result ?production_sample ?lab_result_primary)
        (record_batch_has_retest_result ?production_batch ?lab_result_retest)
        (primary_result_confirmed ?lab_result_primary)
        (retest_result_reported ?lab_result_retest)
        (not
          (record_sample_reviewed ?production_sample)
        )
        (record_batch_reviewed ?production_batch)
        (quality_record_prepared ?quality_record)
      )
    :effect
      (and
        (quality_record_generated ?quality_record)
        (quality_record_includes_primary_result ?quality_record ?lab_result_primary)
        (quality_record_includes_retest_result ?quality_record ?lab_result_retest)
        (quality_record_requires_validation ?quality_record)
        (not
          (quality_record_prepared ?quality_record)
        )
      )
  )
  (:action assemble_quality_record_and_flag_for_capa
    :parameters (?production_sample - production_sample ?production_batch - production_batch ?lab_result_primary - lab_result_primary ?lab_result_retest - lab_result_retest ?quality_record - quality_record)
    :precondition
      (and
        (record_sample_testing_initialized ?production_sample)
        (record_batch_testing_initialized ?production_batch)
        (record_sample_has_primary_result ?production_sample ?lab_result_primary)
        (record_batch_has_retest_result ?production_batch ?lab_result_retest)
        (primary_result_reported ?lab_result_primary)
        (retest_result_confirmed ?lab_result_retest)
        (record_sample_reviewed ?production_sample)
        (not
          (record_batch_reviewed ?production_batch)
        )
        (quality_record_prepared ?quality_record)
      )
    :effect
      (and
        (quality_record_generated ?quality_record)
        (quality_record_includes_primary_result ?quality_record ?lab_result_primary)
        (quality_record_includes_retest_result ?quality_record ?lab_result_retest)
        (quality_record_requires_capa ?quality_record)
        (not
          (quality_record_prepared ?quality_record)
        )
      )
  )
  (:action assemble_quality_record_and_flag_both
    :parameters (?production_sample - production_sample ?production_batch - production_batch ?lab_result_primary - lab_result_primary ?lab_result_retest - lab_result_retest ?quality_record - quality_record)
    :precondition
      (and
        (record_sample_testing_initialized ?production_sample)
        (record_batch_testing_initialized ?production_batch)
        (record_sample_has_primary_result ?production_sample ?lab_result_primary)
        (record_batch_has_retest_result ?production_batch ?lab_result_retest)
        (primary_result_confirmed ?lab_result_primary)
        (retest_result_confirmed ?lab_result_retest)
        (not
          (record_sample_reviewed ?production_sample)
        )
        (not
          (record_batch_reviewed ?production_batch)
        )
        (quality_record_prepared ?quality_record)
      )
    :effect
      (and
        (quality_record_generated ?quality_record)
        (quality_record_includes_primary_result ?quality_record ?lab_result_primary)
        (quality_record_includes_retest_result ?quality_record ?lab_result_retest)
        (quality_record_requires_validation ?quality_record)
        (quality_record_requires_capa ?quality_record)
        (not
          (quality_record_prepared ?quality_record)
        )
      )
  )
  (:action finalize_quality_record_for_sample_review
    :parameters (?quality_record - quality_record ?production_sample - production_sample ?laboratory_analyst - laboratory_analyst)
    :precondition
      (and
        (quality_record_generated ?quality_record)
        (record_sample_testing_initialized ?production_sample)
        (laboratory_analyst_assigned_to_record ?production_sample ?laboratory_analyst)
        (not
          (quality_record_lab_review_complete ?quality_record)
        )
      )
    :effect (quality_record_lab_review_complete ?quality_record)
  )
  (:action attach_supporting_document
    :parameters (?investigation_record - investigation_record ?supporting_document - supporting_document ?quality_record - quality_record)
    :precondition
      (and
        (record_active ?investigation_record)
        (record_investigation_has_quality_record ?investigation_record ?quality_record)
        (record_includes_supporting_document ?investigation_record ?supporting_document)
        (supporting_document_available ?supporting_document)
        (quality_record_generated ?quality_record)
        (quality_record_lab_review_complete ?quality_record)
        (not
          (supporting_document_attached ?supporting_document)
        )
      )
    :effect
      (and
        (supporting_document_attached ?supporting_document)
        (supporting_document_linked_to_quality_record ?supporting_document ?quality_record)
        (not
          (supporting_document_available ?supporting_document)
        )
      )
  )
  (:action lock_evidence_set
    :parameters (?investigation_record - investigation_record ?supporting_document - supporting_document ?quality_record - quality_record ?laboratory_analyst - laboratory_analyst)
    :precondition
      (and
        (record_active ?investigation_record)
        (record_includes_supporting_document ?investigation_record ?supporting_document)
        (supporting_document_attached ?supporting_document)
        (supporting_document_linked_to_quality_record ?supporting_document ?quality_record)
        (laboratory_analyst_assigned_to_record ?investigation_record ?laboratory_analyst)
        (not
          (quality_record_requires_validation ?quality_record)
        )
        (not
          (record_evidence_locked ?investigation_record)
        )
      )
    :effect (record_evidence_locked ?investigation_record)
  )
  (:action assign_approval_type_to_record
    :parameters (?investigation_record - investigation_record ?approval_type - approval_type)
    :precondition
      (and
        (record_active ?investigation_record)
        (approval_type_available ?approval_type)
        (not
          (record_approval_route_assigned ?investigation_record)
        )
      )
    :effect
      (and
        (record_approval_route_assigned ?investigation_record)
        (approval_type_assigned_to_record ?investigation_record ?approval_type)
        (not
          (approval_type_available ?approval_type)
        )
      )
  )
  (:action prepare_record_for_signoff
    :parameters (?investigation_record - investigation_record ?supporting_document - supporting_document ?quality_record - quality_record ?laboratory_analyst - laboratory_analyst ?approval_type - approval_type)
    :precondition
      (and
        (record_active ?investigation_record)
        (record_includes_supporting_document ?investigation_record ?supporting_document)
        (supporting_document_attached ?supporting_document)
        (supporting_document_linked_to_quality_record ?supporting_document ?quality_record)
        (laboratory_analyst_assigned_to_record ?investigation_record ?laboratory_analyst)
        (quality_record_requires_validation ?quality_record)
        (record_approval_route_assigned ?investigation_record)
        (approval_type_assigned_to_record ?investigation_record ?approval_type)
        (not
          (record_evidence_locked ?investigation_record)
        )
      )
    :effect
      (and
        (record_evidence_locked ?investigation_record)
        (record_signoff_initialized ?investigation_record)
      )
  )
  (:action complete_document_package_with_validation
    :parameters (?investigation_record - investigation_record ?validation_activity - validation_activity ?technical_reviewer - technical_reviewer ?supporting_document - supporting_document ?quality_record - quality_record)
    :precondition
      (and
