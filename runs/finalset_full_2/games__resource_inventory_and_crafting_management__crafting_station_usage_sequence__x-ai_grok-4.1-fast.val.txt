--- domain.pddl (as passed to VAL) ---
(define (domain crafting_station_usage)
  (:requirements :strips :typing :negative-preconditions)
  (:types item_subtype - object consumable_subtype - object material_subtype - object entity_supertype - object crafting_entity - entity_supertype inventory_stack - item_subtype tool_item - item_subtype equipment_attachment - item_subtype optional_upgrade_token - item_subtype maintenance_kit - item_subtype preservation_token - item_subtype quality_module - item_subtype stabilizer_unit - item_subtype consumable_item - consumable_subtype component_item - consumable_subtype special_fixture - consumable_subtype primary_material_batch - material_subtype secondary_material_batch - material_subtype workpiece - material_subtype station_role_group - crafting_entity job_role_group - crafting_entity station_instance - station_role_group support_station_instance - station_role_group assembly_job - job_role_group)
  (:predicates
    (entity_reserved ?crafting_entity - crafting_entity)
    (entity_primed ?crafting_entity - crafting_entity)
    (material_selected ?crafting_entity - crafting_entity)
    (entity_released ?crafting_entity - crafting_entity)
    (entity_finalized ?crafting_entity - crafting_entity)
    (preservation_applied ?crafting_entity - crafting_entity)
    (inventory_stack_available ?inventory_stack - inventory_stack)
    (assigned_inventory_stack ?crafting_entity - crafting_entity ?inventory_stack - inventory_stack)
    (tool_available ?tool_item - tool_item)
    (tool_attached ?crafting_entity - crafting_entity ?tool_item - tool_item)
    (attachment_available ?equipment_attachment - equipment_attachment)
    (entity_attached_equipment ?crafting_entity - crafting_entity ?equipment_attachment - equipment_attachment)
    (consumable_available ?consumable_item - consumable_item)
    (station_staged_consumable ?station_instance - station_instance ?consumable_item - consumable_item)
    (support_station_staged_consumable ?support_station_instance - support_station_instance ?consumable_item - consumable_item)
    (station_primary_batch_assigned ?station_instance - station_instance ?primary_material_batch - primary_material_batch)
    (primary_batch_reserved ?primary_material_batch - primary_material_batch)
    (primary_batch_staged ?primary_material_batch - primary_material_batch)
    (station_primary_ready ?station_instance - station_instance)
    (support_station_secondary_batch_assigned ?support_station_instance - support_station_instance ?secondary_material_batch - secondary_material_batch)
    (secondary_batch_reserved ?secondary_material_batch - secondary_material_batch)
    (secondary_batch_staged ?secondary_material_batch - secondary_material_batch)
    (support_station_ready ?support_station_instance - support_station_instance)
    (workpiece_slot_available ?workpiece - workpiece)
    (workpiece_created ?workpiece - workpiece)
    (workpiece_has_primary_batch ?workpiece - workpiece ?primary_material_batch - primary_material_batch)
    (workpiece_has_secondary_batch ?workpiece - workpiece ?secondary_material_batch - secondary_material_batch)
    (workpiece_primary_marker ?workpiece - workpiece)
    (workpiece_secondary_marker ?workpiece - workpiece)
    (workpiece_processed ?workpiece - workpiece)
    (assembly_job_bound_to_primary_station ?recipe_blueprint - assembly_job ?station_instance - station_instance)
    (assembly_job_bound_to_support_station ?recipe_blueprint - assembly_job ?support_station_instance - support_station_instance)
    (assembly_job_has_workpiece ?recipe_blueprint - assembly_job ?workpiece - workpiece)
    (component_available ?component_item - component_item)
    (assembly_job_has_component ?recipe_blueprint - assembly_job ?component_item - component_item)
    (component_prepared ?component_item - component_item)
    (component_bound_to_workpiece ?component_item - component_item ?workpiece - workpiece)
    (assembly_job_component_slot_locked ?recipe_blueprint - assembly_job)
    (assembly_job_component_staged ?recipe_blueprint - assembly_job)
    (assembly_job_ready_for_finalization ?recipe_blueprint - assembly_job)
    (assembly_job_upgrade_applied ?recipe_blueprint - assembly_job)
    (assembly_job_upgrade_marked ?recipe_blueprint - assembly_job)
    (assembly_job_has_attached_quality_module ?recipe_blueprint - assembly_job)
    (assembly_job_finalized ?recipe_blueprint - assembly_job)
    (fixture_available ?special_fixture - special_fixture)
    (assembly_job_attached_fixture ?recipe_blueprint - assembly_job ?special_fixture - special_fixture)
    (assembly_job_fixture_installed ?recipe_blueprint - assembly_job)
    (assembly_job_fixture_locked ?recipe_blueprint - assembly_job)
    (assembly_job_has_attached_stabilizer ?recipe_blueprint - assembly_job)
    (optional_upgrade_token_available ?optional_upgrade_token - optional_upgrade_token)
    (assembly_job_has_upgrade_token ?recipe_blueprint - assembly_job ?optional_upgrade_token - optional_upgrade_token)
    (maintenance_kit_available ?maintenance_kit - maintenance_kit)
    (assembly_job_bound_maintenance_kit ?recipe_blueprint - assembly_job ?maintenance_kit - maintenance_kit)
    (quality_module_available ?quality_module - quality_module)
    (assembly_job_attached_quality_module ?recipe_blueprint - assembly_job ?quality_module - quality_module)
    (stabilizer_unit_available ?stabilizer_unit - stabilizer_unit)
    (assembly_job_attached_stabilizer ?recipe_blueprint - assembly_job ?stabilizer_unit - stabilizer_unit)
    (preservation_token_available ?preservation_token - preservation_token)
    (entity_preservation_token_assigned ?crafting_entity - crafting_entity ?preservation_token - preservation_token)
    (station_primary_processing_started ?station_instance - station_instance)
    (support_station_secondary_processing_started ?support_station_instance - support_station_instance)
    (assembly_job_completion_recorded ?recipe_blueprint - assembly_job)
  )
  (:action reserve_crafting_entity
    :parameters (?crafting_entity - crafting_entity)
    :precondition
      (and
        (not
          (entity_reserved ?crafting_entity)
        )
        (not
          (entity_released ?crafting_entity)
        )
      )
    :effect (entity_reserved ?crafting_entity)
  )
  (:action assign_inventory_stack_to_entity
    :parameters (?crafting_entity - crafting_entity ?inventory_stack - inventory_stack)
    :precondition
      (and
        (entity_reserved ?crafting_entity)
        (not
          (material_selected ?crafting_entity)
        )
        (inventory_stack_available ?inventory_stack)
      )
    :effect
      (and
        (material_selected ?crafting_entity)
        (assigned_inventory_stack ?crafting_entity ?inventory_stack)
        (not
          (inventory_stack_available ?inventory_stack)
        )
      )
  )
  (:action attach_tool_to_entity
    :parameters (?crafting_entity - crafting_entity ?tool_item - tool_item)
    :precondition
      (and
        (entity_reserved ?crafting_entity)
        (material_selected ?crafting_entity)
        (tool_available ?tool_item)
      )
    :effect
      (and
        (tool_attached ?crafting_entity ?tool_item)
        (not
          (tool_available ?tool_item)
        )
      )
  )
  (:action prime_crafting_entity
    :parameters (?crafting_entity - crafting_entity ?tool_item - tool_item)
    :precondition
      (and
        (entity_reserved ?crafting_entity)
        (material_selected ?crafting_entity)
        (tool_attached ?crafting_entity ?tool_item)
        (not
          (entity_primed ?crafting_entity)
        )
      )
    :effect (entity_primed ?crafting_entity)
  )
  (:action detach_tool_from_entity
    :parameters (?crafting_entity - crafting_entity ?tool_item - tool_item)
    :precondition
      (and
        (tool_attached ?crafting_entity ?tool_item)
      )
    :effect
      (and
        (tool_available ?tool_item)
        (not
          (tool_attached ?crafting_entity ?tool_item)
        )
      )
  )
  (:action attach_equipment_to_entity
    :parameters (?crafting_entity - crafting_entity ?equipment_attachment - equipment_attachment)
    :precondition
      (and
        (entity_primed ?crafting_entity)
        (attachment_available ?equipment_attachment)
      )
    :effect
      (and
        (entity_attached_equipment ?crafting_entity ?equipment_attachment)
        (not
          (attachment_available ?equipment_attachment)
        )
      )
  )
  (:action detach_equipment_from_entity
    :parameters (?crafting_entity - crafting_entity ?equipment_attachment - equipment_attachment)
    :precondition
      (and
        (entity_attached_equipment ?crafting_entity ?equipment_attachment)
      )
    :effect
      (and
        (attachment_available ?equipment_attachment)
        (not
          (entity_attached_equipment ?crafting_entity ?equipment_attachment)
        )
      )
  )
  (:action attach_quality_module_to_assembly_job
    :parameters (?recipe_blueprint - assembly_job ?quality_module - quality_module)
    :precondition
      (and
        (entity_primed ?recipe_blueprint)
        (quality_module_available ?quality_module)
      )
    :effect
      (and
        (assembly_job_attached_quality_module ?recipe_blueprint ?quality_module)
        (not
          (quality_module_available ?quality_module)
        )
      )
  )
  (:action detach_quality_module_from_assembly_job
    :parameters (?recipe_blueprint - assembly_job ?quality_module - quality_module)
    :precondition
      (and
        (assembly_job_attached_quality_module ?recipe_blueprint ?quality_module)
      )
    :effect
      (and
        (quality_module_available ?quality_module)
        (not
          (assembly_job_attached_quality_module ?recipe_blueprint ?quality_module)
        )
      )
  )
  (:action attach_stabilizer_to_assembly_job
    :parameters (?recipe_blueprint - assembly_job ?stabilizer_unit - stabilizer_unit)
    :precondition
      (and
        (entity_primed ?recipe_blueprint)
        (stabilizer_unit_available ?stabilizer_unit)
      )
    :effect
      (and
        (assembly_job_attached_stabilizer ?recipe_blueprint ?stabilizer_unit)
        (not
          (stabilizer_unit_available ?stabilizer_unit)
        )
      )
  )
  (:action detach_stabilizer_from_assembly_job
    :parameters (?recipe_blueprint - assembly_job ?stabilizer_unit - stabilizer_unit)
    :precondition
      (and
        (assembly_job_attached_stabilizer ?recipe_blueprint ?stabilizer_unit)
      )
    :effect
      (and
        (stabilizer_unit_available ?stabilizer_unit)
        (not
          (assembly_job_attached_stabilizer ?recipe_blueprint ?stabilizer_unit)
        )
      )
  )
  (:action reserve_primary_batch_for_station
    :parameters (?station_instance - station_instance ?primary_material_batch - primary_material_batch ?tool_item - tool_item)
    :precondition
      (and
        (entity_primed ?station_instance)
        (tool_attached ?station_instance ?tool_item)
        (station_primary_batch_assigned ?station_instance ?primary_material_batch)
        (not
          (primary_batch_reserved ?primary_material_batch)
        )
        (not
          (primary_batch_staged ?primary_material_batch)
        )
      )
    :effect (primary_batch_reserved ?primary_material_batch)
  )
  (:action prepare_station_primary
    :parameters (?station_instance - station_instance ?primary_material_batch - primary_material_batch ?equipment_attachment - equipment_attachment)
    :precondition
      (and
        (entity_primed ?station_instance)
        (entity_attached_equipment ?station_instance ?equipment_attachment)
        (station_primary_batch_assigned ?station_instance ?primary_material_batch)
        (primary_batch_reserved ?primary_material_batch)
        (not
          (station_primary_processing_started ?station_instance)
        )
      )
    :effect
      (and
        (station_primary_processing_started ?station_instance)
        (station_primary_ready ?station_instance)
      )
  )
  (:action stage_consumable_on_station
    :parameters (?station_instance - station_instance ?primary_material_batch - primary_material_batch ?consumable_item - consumable_item)
    :precondition
      (and
        (entity_primed ?station_instance)
        (station_primary_batch_assigned ?station_instance ?primary_material_batch)
        (consumable_available ?consumable_item)
        (not
          (station_primary_processing_started ?station_instance)
        )
      )
    :effect
      (and
        (primary_batch_staged ?primary_material_batch)
        (station_primary_processing_started ?station_instance)
        (station_staged_consumable ?station_instance ?consumable_item)
        (not
          (consumable_available ?consumable_item)
        )
      )
  )
  (:action process_station_primary_stage
    :parameters (?station_instance - station_instance ?primary_material_batch - primary_material_batch ?tool_item - tool_item ?consumable_item - consumable_item)
    :precondition
      (and
        (entity_primed ?station_instance)
        (tool_attached ?station_instance ?tool_item)
        (station_primary_batch_assigned ?station_instance ?primary_material_batch)
        (primary_batch_staged ?primary_material_batch)
        (station_staged_consumable ?station_instance ?consumable_item)
        (not
          (station_primary_ready ?station_instance)
        )
      )
    :effect
      (and
        (primary_batch_reserved ?primary_material_batch)
        (station_primary_ready ?station_instance)
        (consumable_available ?consumable_item)
        (not
          (station_staged_consumable ?station_instance ?consumable_item)
        )
      )
  )
  (:action reserve_secondary_batch_for_support_station
    :parameters (?support_station_instance - support_station_instance ?secondary_material_batch - secondary_material_batch ?tool_item - tool_item)
    :precondition
      (and
        (entity_primed ?support_station_instance)
        (tool_attached ?support_station_instance ?tool_item)
        (support_station_secondary_batch_assigned ?support_station_instance ?secondary_material_batch)
        (not
          (secondary_batch_reserved ?secondary_material_batch)
        )
        (not
          (secondary_batch_staged ?secondary_material_batch)
        )
      )
    :effect (secondary_batch_reserved ?secondary_material_batch)
  )
  (:action prepare_support_station_secondary
    :parameters (?support_station_instance - support_station_instance ?secondary_material_batch - secondary_material_batch ?equipment_attachment - equipment_attachment)
    :precondition
      (and
        (entity_primed ?support_station_instance)
        (entity_attached_equipment ?support_station_instance ?equipment_attachment)
        (support_station_secondary_batch_assigned ?support_station_instance ?secondary_material_batch)
        (secondary_batch_reserved ?secondary_material_batch)
        (not
          (support_station_secondary_processing_started ?support_station_instance)
        )
      )
    :effect
      (and
        (support_station_secondary_processing_started ?support_station_instance)
        (support_station_ready ?support_station_instance)
      )
  )
  (:action stage_consumable_on_support_station
    :parameters (?support_station_instance - support_station_instance ?secondary_material_batch - secondary_material_batch ?consumable_item - consumable_item)
    :precondition
      (and
        (entity_primed ?support_station_instance)
        (support_station_secondary_batch_assigned ?support_station_instance ?secondary_material_batch)
        (consumable_available ?consumable_item)
        (not
          (support_station_secondary_processing_started ?support_station_instance)
        )
      )
    :effect
      (and
        (secondary_batch_staged ?secondary_material_batch)
        (support_station_secondary_processing_started ?support_station_instance)
        (support_station_staged_consumable ?support_station_instance ?consumable_item)
        (not
          (consumable_available ?consumable_item)
        )
      )
  )
  (:action process_support_station_secondary_stage
    :parameters (?support_station_instance - support_station_instance ?secondary_material_batch - secondary_material_batch ?tool_item - tool_item ?consumable_item - consumable_item)
    :precondition
      (and
        (entity_primed ?support_station_instance)
        (tool_attached ?support_station_instance ?tool_item)
        (support_station_secondary_batch_assigned ?support_station_instance ?secondary_material_batch)
        (secondary_batch_staged ?secondary_material_batch)
        (support_station_staged_consumable ?support_station_instance ?consumable_item)
        (not
          (support_station_ready ?support_station_instance)
        )
      )
    :effect
      (and
        (secondary_batch_reserved ?secondary_material_batch)
        (support_station_ready ?support_station_instance)
        (consumable_available ?consumable_item)
        (not
          (support_station_staged_consumable ?support_station_instance ?consumable_item)
        )
      )
  )
  (:action create_workpiece_from_stations
    :parameters (?station_instance - station_instance ?support_station_instance - support_station_instance ?primary_material_batch - primary_material_batch ?secondary_material_batch - secondary_material_batch ?workpiece - workpiece)
    :precondition
      (and
        (station_primary_processing_started ?station_instance)
        (support_station_secondary_processing_started ?support_station_instance)
        (station_primary_batch_assigned ?station_instance ?primary_material_batch)
        (support_station_secondary_batch_assigned ?support_station_instance ?secondary_material_batch)
        (primary_batch_reserved ?primary_material_batch)
        (secondary_batch_reserved ?secondary_material_batch)
        (station_primary_ready ?station_instance)
        (support_station_ready ?support_station_instance)
        (workpiece_slot_available ?workpiece)
      )
    :effect
      (and
        (workpiece_created ?workpiece)
        (workpiece_has_primary_batch ?workpiece ?primary_material_batch)
        (workpiece_has_secondary_batch ?workpiece ?secondary_material_batch)
        (not
          (workpiece_slot_available ?workpiece)
        )
      )
  )
  (:action create_workpiece_with_primary_marker
    :parameters (?station_instance - station_instance ?support_station_instance - support_station_instance ?primary_material_batch - primary_material_batch ?secondary_material_batch - secondary_material_batch ?workpiece - workpiece)
    :precondition
      (and
        (station_primary_processing_started ?station_instance)
        (support_station_secondary_processing_started ?support_station_instance)
        (station_primary_batch_assigned ?station_instance ?primary_material_batch)
        (support_station_secondary_batch_assigned ?support_station_instance ?secondary_material_batch)
        (primary_batch_staged ?primary_material_batch)
        (secondary_batch_reserved ?secondary_material_batch)
        (not
          (station_primary_ready ?station_instance)
        )
        (support_station_ready ?support_station_instance)
        (workpiece_slot_available ?workpiece)
      )
    :effect
      (and
        (workpiece_created ?workpiece)
        (workpiece_has_primary_batch ?workpiece ?primary_material_batch)
        (workpiece_has_secondary_batch ?workpiece ?secondary_material_batch)
        (workpiece_primary_marker ?workpiece)
        (not
          (workpiece_slot_available ?workpiece)
        )
      )
  )
  (:action create_workpiece_with_secondary_marker
    :parameters (?station_instance - station_instance ?support_station_instance - support_station_instance ?primary_material_batch - primary_material_batch ?secondary_material_batch - secondary_material_batch ?workpiece - workpiece)
    :precondition
      (and
        (station_primary_processing_started ?station_instance)
        (support_station_secondary_processing_started ?support_station_instance)
        (station_primary_batch_assigned ?station_instance ?primary_material_batch)
        (support_station_secondary_batch_assigned ?support_station_instance ?secondary_material_batch)
        (primary_batch_reserved ?primary_material_batch)
        (secondary_batch_staged ?secondary_material_batch)
        (station_primary_ready ?station_instance)
        (not
          (support_station_ready ?support_station_instance)
        )
        (workpiece_slot_available ?workpiece)
      )
    :effect
      (and
        (workpiece_created ?workpiece)
        (workpiece_has_primary_batch ?workpiece ?primary_material_batch)
        (workpiece_has_secondary_batch ?workpiece ?secondary_material_batch)
        (workpiece_secondary_marker ?workpiece)
        (not
          (workpiece_slot_available ?workpiece)
        )
      )
  )
  (:action create_workpiece_with_both_markers
    :parameters (?station_instance - station_instance ?support_station_instance - support_station_instance ?primary_material_batch - primary_material_batch ?secondary_material_batch - secondary_material_batch ?workpiece - workpiece)
    :precondition
      (and
        (station_primary_processing_started ?station_instance)
        (support_station_secondary_processing_started ?support_station_instance)
        (station_primary_batch_assigned ?station_instance ?primary_material_batch)
        (support_station_secondary_batch_assigned ?support_station_instance ?secondary_material_batch)
        (primary_batch_staged ?primary_material_batch)
        (secondary_batch_staged ?secondary_material_batch)
        (not
          (station_primary_ready ?station_instance)
        )
        (not
          (support_station_ready ?support_station_instance)
        )
        (workpiece_slot_available ?workpiece)
      )
    :effect
      (and
        (workpiece_created ?workpiece)
        (workpiece_has_primary_batch ?workpiece ?primary_material_batch)
        (workpiece_has_secondary_batch ?workpiece ?secondary_material_batch)
        (workpiece_primary_marker ?workpiece)
        (workpiece_secondary_marker ?workpiece)
        (not
          (workpiece_slot_available ?workpiece)
        )
      )
  )
  (:action mark_workpiece_processed
    :parameters (?workpiece - workpiece ?station_instance - station_instance ?tool_item - tool_item)
    :precondition
      (and
        (workpiece_created ?workpiece)
        (station_primary_processing_started ?station_instance)
        (tool_attached ?station_instance ?tool_item)
        (not
          (workpiece_processed ?workpiece)
        )
      )
    :effect (workpiece_processed ?workpiece)
  )
  (:action stage_component_and_bind_to_workpiece
    :parameters (?recipe_blueprint - assembly_job ?component_item - component_item ?workpiece - workpiece)
    :precondition
      (and
        (entity_primed ?recipe_blueprint)
        (assembly_job_has_workpiece ?recipe_blueprint ?workpiece)
        (assembly_job_has_component ?recipe_blueprint ?component_item)
        (component_available ?component_item)
        (workpiece_created ?workpiece)
        (workpiece_processed ?workpiece)
        (not
          (component_prepared ?component_item)
        )
      )
    :effect
      (and
        (component_prepared ?component_item)
        (component_bound_to_workpiece ?component_item ?workpiece)
        (not
          (component_available ?component_item)
        )
      )
  )
  (:action lock_component_slot_for_recipe
    :parameters (?recipe_blueprint - assembly_job ?component_item - component_item ?workpiece - workpiece ?tool_item - tool_item)
    :precondition
      (and
        (entity_primed ?recipe_blueprint)
        (assembly_job_has_component ?recipe_blueprint ?component_item)
        (component_prepared ?component_item)
        (component_bound_to_workpiece ?component_item ?workpiece)
        (tool_attached ?recipe_blueprint ?tool_item)
        (not
          (workpiece_primary_marker ?workpiece)
        )
        (not
          (assembly_job_component_slot_locked ?recipe_blueprint)
        )
      )
    :effect (assembly_job_component_slot_locked ?recipe_blueprint)
  )
  (:action apply_optional_upgrade_to_assembly_job
    :parameters (?recipe_blueprint - assembly_job ?optional_upgrade_token - optional_upgrade_token)
    :precondition
      (and
        (entity_primed ?recipe_blueprint)
        (optional_upgrade_token_available ?optional_upgrade_token)
        (not
          (assembly_job_upgrade_applied ?recipe_blueprint)
        )
      )
    :effect
      (and
        (assembly_job_upgrade_applied ?recipe_blueprint)
        (assembly_job_has_upgrade_token ?recipe_blueprint ?optional_upgrade_token)
        (not
          (optional_upgrade_token_available ?optional_upgrade_token)
        )
      )
  )
  (:action apply_upgrade_and_lock_component_slot
    :parameters (?recipe_blueprint - assembly_job ?component_item - component_item ?workpiece - workpiece ?tool_item - tool_item ?optional_upgrade_token - optional_upgrade_token)
    :precondition
      (and
        (entity_primed ?recipe_blueprint)
        (assembly_job_has_component ?recipe_blueprint ?component_item)
        (component_prepared ?component_item)
        (component_bound_to_workpiece ?component_item ?workpiece)
        (tool_attached ?recipe_blueprint ?tool_item)
        (workpiece_primary_marker ?workpiece)
        (assembly_job_upgrade_applied ?recipe_blueprint)
        (assembly_job_has_upgrade_token ?recipe_blueprint ?optional_upgrade_token)
        (not
          (assembly_job_component_slot_locked ?recipe_blueprint)
        )
      )
    :effect
      (and
        (assembly_job_component_slot_locked ?recipe_blueprint)
        (assembly_job_upgrade_marked ?recipe_blueprint)
      )
  )
  (:action install_quality_module_and_stage_component
    :parameters (?recipe_blueprint - assembly_job ?quality_module - quality_module ?equipment_attachment - equipment_attachment ?component_item - component_item ?workpiece - workpiece)
    :precondition
      (and
        (assembly_job_component_slot_locked ?recipe_blueprint)
        (assembly_job_attached_quality_module ?recipe_blueprint ?quality_module)
        (entity_attached_equipment ?recipe_blueprint ?equipment_attachment)
        (assembly_job_has_component ?recipe_blueprint ?component_item)
        (component_bound_to_workpiece ?component_item ?workpiece)
        (not
          (workpiece_secondary_marker ?workpiece)
        )
        (not
          (assembly_job_component_staged ?recipe_blueprint)
        )
      )
    :effect (assembly_job_component_staged ?recipe_blueprint)
