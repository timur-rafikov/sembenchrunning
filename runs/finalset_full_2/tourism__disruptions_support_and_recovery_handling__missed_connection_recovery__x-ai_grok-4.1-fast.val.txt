--- domain.pddl (as passed to VAL) ---
(define (domain missed_connection_recovery_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types operational_resource - object transport_category - object service_category - object case_identifier - object recovery_case - case_identifier agent_resource_slot - operational_resource flight_leg - operational_resource service_desk - operational_resource compensation_scheme - operational_resource followup_task - operational_resource contact_channel - operational_resource compensation_offer - operational_resource assistance_resource - operational_resource consumable_voucher - transport_category document_bundle - transport_category policy_reference - transport_category alternative_option_a - service_category alternative_option_b - service_category recovery_package - service_category itinerary_component - recovery_case passenger_profile - recovery_case inbound_flight_segment - itinerary_component outbound_flight_segment - itinerary_component support_agent_team - passenger_profile)
  (:predicates
    (work_item_created ?recovery_case - recovery_case)
    (work_item_triaged ?recovery_case - recovery_case)
    (work_item_has_assigned_agent ?recovery_case - recovery_case)
    (work_item_confirmed ?recovery_case - recovery_case)
    (work_item_ready_for_notification ?recovery_case - recovery_case)
    (work_item_notification_sent ?recovery_case - recovery_case)
    (agent_slot_available ?agent_resource_slot - agent_resource_slot)
    (work_item_allocated_agent_slot ?recovery_case - recovery_case ?agent_resource_slot - agent_resource_slot)
    (flight_leg_available ?flight_leg - flight_leg)
    (work_item_linked_to_flight_leg ?recovery_case - recovery_case ?flight_leg - flight_leg)
    (service_desk_available ?service_desk - service_desk)
    (work_item_assigned_service_desk ?recovery_case - recovery_case ?service_desk - service_desk)
    (voucher_available ?consumable_voucher - consumable_voucher)
    (inbound_voucher_assigned ?inbound_flight_segment - inbound_flight_segment ?consumable_voucher - consumable_voucher)
    (outbound_voucher_assigned ?outbound_flight_segment - outbound_flight_segment ?consumable_voucher - consumable_voucher)
    (inbound_has_alternative_a ?inbound_flight_segment - inbound_flight_segment ?alternative_option_a - alternative_option_a)
    (alternative_a_approved ?alternative_option_a - alternative_option_a)
    (alternative_a_requires_voucher ?alternative_option_a - alternative_option_a)
    (inbound_option_finalized ?inbound_flight_segment - inbound_flight_segment)
    (outbound_has_alternative_b ?outbound_flight_segment - outbound_flight_segment ?alternative_option_b - alternative_option_b)
    (alternative_b_approved ?alternative_option_b - alternative_option_b)
    (alternative_b_requires_voucher ?alternative_option_b - alternative_option_b)
    (outbound_option_finalized ?outbound_flight_segment - outbound_flight_segment)
    (recovery_package_pending ?recovery_package - recovery_package)
    (recovery_package_assembled ?recovery_package - recovery_package)
    (recovery_package_includes_option_a ?recovery_package - recovery_package ?alternative_option_a - alternative_option_a)
    (recovery_package_includes_option_b ?recovery_package - recovery_package ?alternative_option_b - alternative_option_b)
    (recovery_package_priority_flag_a ?recovery_package - recovery_package)
    (recovery_package_priority_flag_b ?recovery_package - recovery_package)
    (recovery_package_ready ?recovery_package - recovery_package)
    (team_assigned_to_inbound_segment ?support_agent_team - support_agent_team ?inbound_flight_segment - inbound_flight_segment)
    (team_assigned_to_outbound_segment ?support_agent_team - support_agent_team ?outbound_flight_segment - outbound_flight_segment)
    (team_assigned_to_recovery_package ?support_agent_team - support_agent_team ?recovery_package - recovery_package)
    (document_bundle_available ?document_bundle - document_bundle)
    (team_has_document_bundle ?support_agent_team - support_agent_team ?document_bundle - document_bundle)
    (document_bundle_attached ?document_bundle - document_bundle)
    (document_bundle_attached_to_package ?document_bundle - document_bundle ?recovery_package - recovery_package)
    (team_documents_validated ?support_agent_team - support_agent_team)
    (team_compensation_offer_attached ?support_agent_team - support_agent_team)
    (team_assistance_resource_applied ?support_agent_team - support_agent_team)
    (team_has_compensation_scheme ?support_agent_team - support_agent_team)
    (team_compensation_endorsement ?support_agent_team - support_agent_team)
    (team_enrichment_complete ?support_agent_team - support_agent_team)
    (team_execution_performed ?support_agent_team - support_agent_team)
    (policy_reference_available ?policy_reference - policy_reference)
    (team_has_policy_reference ?support_agent_team - support_agent_team ?policy_reference - policy_reference)
    (team_policy_attached ?support_agent_team - support_agent_team)
    (team_policy_validated ?support_agent_team - support_agent_team)
    (team_policy_approved ?support_agent_team - support_agent_team)
    (compensation_scheme_available ?compensation_scheme - compensation_scheme)
    (team_assigned_compensation_scheme ?support_agent_team - support_agent_team ?compensation_scheme - compensation_scheme)
    (followup_task_available ?followup_task - followup_task)
    (team_assigned_followup_task ?support_agent_team - support_agent_team ?followup_task - followup_task)
    (compensation_offer_available ?compensation_offer - compensation_offer)
    (team_has_compensation_offer ?support_agent_team - support_agent_team ?compensation_offer - compensation_offer)
    (assistance_resource_available ?assistance_resource - assistance_resource)
    (team_allocated_assistance_resource ?support_agent_team - support_agent_team ?assistance_resource - assistance_resource)
    (contact_channel_available ?contact_channel - contact_channel)
    (work_item_contact_channel_registered ?recovery_case - recovery_case ?contact_channel - contact_channel)
    (inbound_processing_flag ?inbound_flight_segment - inbound_flight_segment)
    (outbound_processing_flag ?outbound_flight_segment - outbound_flight_segment)
    (team_finalization_flag ?support_agent_team - support_agent_team)
  )
  (:action create_recovery_case
    :parameters (?recovery_case - recovery_case)
    :precondition
      (and
        (not
          (work_item_created ?recovery_case)
        )
        (not
          (work_item_confirmed ?recovery_case)
        )
      )
    :effect (work_item_created ?recovery_case)
  )
  (:action allocate_agent_slot_to_case
    :parameters (?recovery_case - recovery_case ?agent_resource_slot - agent_resource_slot)
    :precondition
      (and
        (work_item_created ?recovery_case)
        (not
          (work_item_has_assigned_agent ?recovery_case)
        )
        (agent_slot_available ?agent_resource_slot)
      )
    :effect
      (and
        (work_item_has_assigned_agent ?recovery_case)
        (work_item_allocated_agent_slot ?recovery_case ?agent_resource_slot)
        (not
          (agent_slot_available ?agent_resource_slot)
        )
      )
  )
  (:action attach_flight_leg_to_case
    :parameters (?recovery_case - recovery_case ?flight_leg - flight_leg)
    :precondition
      (and
        (work_item_created ?recovery_case)
        (work_item_has_assigned_agent ?recovery_case)
        (flight_leg_available ?flight_leg)
      )
    :effect
      (and
        (work_item_linked_to_flight_leg ?recovery_case ?flight_leg)
        (not
          (flight_leg_available ?flight_leg)
        )
      )
  )
  (:action finalize_case_triage
    :parameters (?recovery_case - recovery_case ?flight_leg - flight_leg)
    :precondition
      (and
        (work_item_created ?recovery_case)
        (work_item_has_assigned_agent ?recovery_case)
        (work_item_linked_to_flight_leg ?recovery_case ?flight_leg)
        (not
          (work_item_triaged ?recovery_case)
        )
      )
    :effect (work_item_triaged ?recovery_case)
  )
  (:action unassign_flight_leg_from_case
    :parameters (?recovery_case - recovery_case ?flight_leg - flight_leg)
    :precondition
      (and
        (work_item_linked_to_flight_leg ?recovery_case ?flight_leg)
      )
    :effect
      (and
        (flight_leg_available ?flight_leg)
        (not
          (work_item_linked_to_flight_leg ?recovery_case ?flight_leg)
        )
      )
  )
  (:action assign_service_desk_to_case
    :parameters (?recovery_case - recovery_case ?service_desk - service_desk)
    :precondition
      (and
        (work_item_triaged ?recovery_case)
        (service_desk_available ?service_desk)
      )
    :effect
      (and
        (work_item_assigned_service_desk ?recovery_case ?service_desk)
        (not
          (service_desk_available ?service_desk)
        )
      )
  )
  (:action release_service_desk_from_case
    :parameters (?recovery_case - recovery_case ?service_desk - service_desk)
    :precondition
      (and
        (work_item_assigned_service_desk ?recovery_case ?service_desk)
      )
    :effect
      (and
        (service_desk_available ?service_desk)
        (not
          (work_item_assigned_service_desk ?recovery_case ?service_desk)
        )
      )
  )
  (:action attach_compensation_offer_to_team
    :parameters (?support_agent_team - support_agent_team ?compensation_offer - compensation_offer)
    :precondition
      (and
        (work_item_triaged ?support_agent_team)
        (compensation_offer_available ?compensation_offer)
      )
    :effect
      (and
        (team_has_compensation_offer ?support_agent_team ?compensation_offer)
        (not
          (compensation_offer_available ?compensation_offer)
        )
      )
  )
  (:action detach_compensation_offer_from_team
    :parameters (?support_agent_team - support_agent_team ?compensation_offer - compensation_offer)
    :precondition
      (and
        (team_has_compensation_offer ?support_agent_team ?compensation_offer)
      )
    :effect
      (and
        (compensation_offer_available ?compensation_offer)
        (not
          (team_has_compensation_offer ?support_agent_team ?compensation_offer)
        )
      )
  )
  (:action assign_assistance_resource_to_team
    :parameters (?support_agent_team - support_agent_team ?assistance_resource - assistance_resource)
    :precondition
      (and
        (work_item_triaged ?support_agent_team)
        (assistance_resource_available ?assistance_resource)
      )
    :effect
      (and
        (team_allocated_assistance_resource ?support_agent_team ?assistance_resource)
        (not
          (assistance_resource_available ?assistance_resource)
        )
      )
  )
  (:action remove_assistance_resource_from_team
    :parameters (?support_agent_team - support_agent_team ?assistance_resource - assistance_resource)
    :precondition
      (and
        (team_allocated_assistance_resource ?support_agent_team ?assistance_resource)
      )
    :effect
      (and
        (assistance_resource_available ?assistance_resource)
        (not
          (team_allocated_assistance_resource ?support_agent_team ?assistance_resource)
        )
      )
  )
  (:action propose_inbound_option_a
    :parameters (?inbound_flight_segment - inbound_flight_segment ?alternative_option_a - alternative_option_a ?flight_leg - flight_leg)
    :precondition
      (and
        (work_item_triaged ?inbound_flight_segment)
        (work_item_linked_to_flight_leg ?inbound_flight_segment ?flight_leg)
        (inbound_has_alternative_a ?inbound_flight_segment ?alternative_option_a)
        (not
          (alternative_a_approved ?alternative_option_a)
        )
        (not
          (alternative_a_requires_voucher ?alternative_option_a)
        )
      )
    :effect (alternative_a_approved ?alternative_option_a)
  )
  (:action confirm_inbound_option_a
    :parameters (?inbound_flight_segment - inbound_flight_segment ?alternative_option_a - alternative_option_a ?service_desk - service_desk)
    :precondition
      (and
        (work_item_triaged ?inbound_flight_segment)
        (work_item_assigned_service_desk ?inbound_flight_segment ?service_desk)
        (inbound_has_alternative_a ?inbound_flight_segment ?alternative_option_a)
        (alternative_a_approved ?alternative_option_a)
        (not
          (inbound_processing_flag ?inbound_flight_segment)
        )
      )
    :effect
      (and
        (inbound_processing_flag ?inbound_flight_segment)
        (inbound_option_finalized ?inbound_flight_segment)
      )
  )
  (:action assign_voucher_for_inbound_option_a
    :parameters (?inbound_flight_segment - inbound_flight_segment ?alternative_option_a - alternative_option_a ?consumable_voucher - consumable_voucher)
    :precondition
      (and
        (work_item_triaged ?inbound_flight_segment)
        (inbound_has_alternative_a ?inbound_flight_segment ?alternative_option_a)
        (voucher_available ?consumable_voucher)
        (not
          (inbound_processing_flag ?inbound_flight_segment)
        )
      )
    :effect
      (and
        (alternative_a_requires_voucher ?alternative_option_a)
        (inbound_processing_flag ?inbound_flight_segment)
        (inbound_voucher_assigned ?inbound_flight_segment ?consumable_voucher)
        (not
          (voucher_available ?consumable_voucher)
        )
      )
  )
  (:action validate_inbound_option_a_with_voucher
    :parameters (?inbound_flight_segment - inbound_flight_segment ?alternative_option_a - alternative_option_a ?flight_leg - flight_leg ?consumable_voucher - consumable_voucher)
    :precondition
      (and
        (work_item_triaged ?inbound_flight_segment)
        (work_item_linked_to_flight_leg ?inbound_flight_segment ?flight_leg)
        (inbound_has_alternative_a ?inbound_flight_segment ?alternative_option_a)
        (alternative_a_requires_voucher ?alternative_option_a)
        (inbound_voucher_assigned ?inbound_flight_segment ?consumable_voucher)
        (not
          (inbound_option_finalized ?inbound_flight_segment)
        )
      )
    :effect
      (and
        (alternative_a_approved ?alternative_option_a)
        (inbound_option_finalized ?inbound_flight_segment)
        (voucher_available ?consumable_voucher)
        (not
          (inbound_voucher_assigned ?inbound_flight_segment ?consumable_voucher)
        )
      )
  )
  (:action propose_outbound_option_b
    :parameters (?outbound_flight_segment - outbound_flight_segment ?alternative_option_b - alternative_option_b ?flight_leg - flight_leg)
    :precondition
      (and
        (work_item_triaged ?outbound_flight_segment)
        (work_item_linked_to_flight_leg ?outbound_flight_segment ?flight_leg)
        (outbound_has_alternative_b ?outbound_flight_segment ?alternative_option_b)
        (not
          (alternative_b_approved ?alternative_option_b)
        )
        (not
          (alternative_b_requires_voucher ?alternative_option_b)
        )
      )
    :effect (alternative_b_approved ?alternative_option_b)
  )
  (:action confirm_outbound_option_b
    :parameters (?outbound_flight_segment - outbound_flight_segment ?alternative_option_b - alternative_option_b ?service_desk - service_desk)
    :precondition
      (and
        (work_item_triaged ?outbound_flight_segment)
        (work_item_assigned_service_desk ?outbound_flight_segment ?service_desk)
        (outbound_has_alternative_b ?outbound_flight_segment ?alternative_option_b)
        (alternative_b_approved ?alternative_option_b)
        (not
          (outbound_processing_flag ?outbound_flight_segment)
        )
      )
    :effect
      (and
        (outbound_processing_flag ?outbound_flight_segment)
        (outbound_option_finalized ?outbound_flight_segment)
      )
  )
  (:action assign_voucher_for_outbound_option_b
    :parameters (?outbound_flight_segment - outbound_flight_segment ?alternative_option_b - alternative_option_b ?consumable_voucher - consumable_voucher)
    :precondition
      (and
        (work_item_triaged ?outbound_flight_segment)
        (outbound_has_alternative_b ?outbound_flight_segment ?alternative_option_b)
        (voucher_available ?consumable_voucher)
        (not
          (outbound_processing_flag ?outbound_flight_segment)
        )
      )
    :effect
      (and
        (alternative_b_requires_voucher ?alternative_option_b)
        (outbound_processing_flag ?outbound_flight_segment)
        (outbound_voucher_assigned ?outbound_flight_segment ?consumable_voucher)
        (not
          (voucher_available ?consumable_voucher)
        )
      )
  )
  (:action validate_outbound_option_b_with_voucher
    :parameters (?outbound_flight_segment - outbound_flight_segment ?alternative_option_b - alternative_option_b ?flight_leg - flight_leg ?consumable_voucher - consumable_voucher)
    :precondition
      (and
        (work_item_triaged ?outbound_flight_segment)
        (work_item_linked_to_flight_leg ?outbound_flight_segment ?flight_leg)
        (outbound_has_alternative_b ?outbound_flight_segment ?alternative_option_b)
        (alternative_b_requires_voucher ?alternative_option_b)
        (outbound_voucher_assigned ?outbound_flight_segment ?consumable_voucher)
        (not
          (outbound_option_finalized ?outbound_flight_segment)
        )
      )
    :effect
      (and
        (alternative_b_approved ?alternative_option_b)
        (outbound_option_finalized ?outbound_flight_segment)
        (voucher_available ?consumable_voucher)
        (not
          (outbound_voucher_assigned ?outbound_flight_segment ?consumable_voucher)
        )
      )
  )
  (:action assemble_recovery_package
    :parameters (?inbound_flight_segment - inbound_flight_segment ?outbound_flight_segment - outbound_flight_segment ?alternative_option_a - alternative_option_a ?alternative_option_b - alternative_option_b ?recovery_package - recovery_package)
    :precondition
      (and
        (inbound_processing_flag ?inbound_flight_segment)
        (outbound_processing_flag ?outbound_flight_segment)
        (inbound_has_alternative_a ?inbound_flight_segment ?alternative_option_a)
        (outbound_has_alternative_b ?outbound_flight_segment ?alternative_option_b)
        (alternative_a_approved ?alternative_option_a)
        (alternative_b_approved ?alternative_option_b)
        (inbound_option_finalized ?inbound_flight_segment)
        (outbound_option_finalized ?outbound_flight_segment)
        (recovery_package_pending ?recovery_package)
      )
    :effect
      (and
        (recovery_package_assembled ?recovery_package)
        (recovery_package_includes_option_a ?recovery_package ?alternative_option_a)
        (recovery_package_includes_option_b ?recovery_package ?alternative_option_b)
        (not
          (recovery_package_pending ?recovery_package)
        )
      )
  )
  (:action assemble_recovery_package_with_option_a_priority
    :parameters (?inbound_flight_segment - inbound_flight_segment ?outbound_flight_segment - outbound_flight_segment ?alternative_option_a - alternative_option_a ?alternative_option_b - alternative_option_b ?recovery_package - recovery_package)
    :precondition
      (and
        (inbound_processing_flag ?inbound_flight_segment)
        (outbound_processing_flag ?outbound_flight_segment)
        (inbound_has_alternative_a ?inbound_flight_segment ?alternative_option_a)
        (outbound_has_alternative_b ?outbound_flight_segment ?alternative_option_b)
        (alternative_a_requires_voucher ?alternative_option_a)
        (alternative_b_approved ?alternative_option_b)
        (not
          (inbound_option_finalized ?inbound_flight_segment)
        )
        (outbound_option_finalized ?outbound_flight_segment)
        (recovery_package_pending ?recovery_package)
      )
    :effect
      (and
        (recovery_package_assembled ?recovery_package)
        (recovery_package_includes_option_a ?recovery_package ?alternative_option_a)
        (recovery_package_includes_option_b ?recovery_package ?alternative_option_b)
        (recovery_package_priority_flag_a ?recovery_package)
        (not
          (recovery_package_pending ?recovery_package)
        )
      )
  )
  (:action assemble_recovery_package_with_option_b_priority
    :parameters (?inbound_flight_segment - inbound_flight_segment ?outbound_flight_segment - outbound_flight_segment ?alternative_option_a - alternative_option_a ?alternative_option_b - alternative_option_b ?recovery_package - recovery_package)
    :precondition
      (and
        (inbound_processing_flag ?inbound_flight_segment)
        (outbound_processing_flag ?outbound_flight_segment)
        (inbound_has_alternative_a ?inbound_flight_segment ?alternative_option_a)
        (outbound_has_alternative_b ?outbound_flight_segment ?alternative_option_b)
        (alternative_a_approved ?alternative_option_a)
        (alternative_b_requires_voucher ?alternative_option_b)
        (inbound_option_finalized ?inbound_flight_segment)
        (not
          (outbound_option_finalized ?outbound_flight_segment)
        )
        (recovery_package_pending ?recovery_package)
      )
    :effect
      (and
        (recovery_package_assembled ?recovery_package)
        (recovery_package_includes_option_a ?recovery_package ?alternative_option_a)
        (recovery_package_includes_option_b ?recovery_package ?alternative_option_b)
        (recovery_package_priority_flag_b ?recovery_package)
        (not
          (recovery_package_pending ?recovery_package)
        )
      )
  )
  (:action assemble_recovery_package_with_both_priorities
    :parameters (?inbound_flight_segment - inbound_flight_segment ?outbound_flight_segment - outbound_flight_segment ?alternative_option_a - alternative_option_a ?alternative_option_b - alternative_option_b ?recovery_package - recovery_package)
    :precondition
      (and
        (inbound_processing_flag ?inbound_flight_segment)
        (outbound_processing_flag ?outbound_flight_segment)
        (inbound_has_alternative_a ?inbound_flight_segment ?alternative_option_a)
        (outbound_has_alternative_b ?outbound_flight_segment ?alternative_option_b)
        (alternative_a_requires_voucher ?alternative_option_a)
        (alternative_b_requires_voucher ?alternative_option_b)
        (not
          (inbound_option_finalized ?inbound_flight_segment)
        )
        (not
          (outbound_option_finalized ?outbound_flight_segment)
        )
        (recovery_package_pending ?recovery_package)
      )
    :effect
      (and
        (recovery_package_assembled ?recovery_package)
        (recovery_package_includes_option_a ?recovery_package ?alternative_option_a)
        (recovery_package_includes_option_b ?recovery_package ?alternative_option_b)
        (recovery_package_priority_flag_a ?recovery_package)
        (recovery_package_priority_flag_b ?recovery_package)
        (not
          (recovery_package_pending ?recovery_package)
        )
      )
  )
  (:action mark_recovery_package_ready
    :parameters (?recovery_package - recovery_package ?inbound_flight_segment - inbound_flight_segment ?flight_leg - flight_leg)
    :precondition
      (and
        (recovery_package_assembled ?recovery_package)
        (inbound_processing_flag ?inbound_flight_segment)
        (work_item_linked_to_flight_leg ?inbound_flight_segment ?flight_leg)
        (not
          (recovery_package_ready ?recovery_package)
        )
      )
    :effect (recovery_package_ready ?recovery_package)
  )
  (:action attach_document_bundle_to_package
    :parameters (?support_agent_team - support_agent_team ?document_bundle - document_bundle ?recovery_package - recovery_package)
    :precondition
      (and
        (work_item_triaged ?support_agent_team)
        (team_assigned_to_recovery_package ?support_agent_team ?recovery_package)
        (team_has_document_bundle ?support_agent_team ?document_bundle)
        (document_bundle_available ?document_bundle)
        (recovery_package_assembled ?recovery_package)
        (recovery_package_ready ?recovery_package)
        (not
          (document_bundle_attached ?document_bundle)
        )
      )
    :effect
      (and
        (document_bundle_attached ?document_bundle)
        (document_bundle_attached_to_package ?document_bundle ?recovery_package)
        (not
          (document_bundle_available ?document_bundle)
        )
      )
  )
  (:action validate_document_bundle_for_package
    :parameters (?support_agent_team - support_agent_team ?document_bundle - document_bundle ?recovery_package - recovery_package ?flight_leg - flight_leg)
    :precondition
      (and
        (work_item_triaged ?support_agent_team)
        (team_has_document_bundle ?support_agent_team ?document_bundle)
        (document_bundle_attached ?document_bundle)
