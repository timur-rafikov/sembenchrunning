--- domain.pddl (as passed to VAL) ---
(define (domain critical_api_supply_disruption_response_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity - object external_actor_category - entity inventory_resource_category - entity product_family - entity event_record_type - entity incident_event - event_record_type supplier - external_actor_category laboratory_resource - external_actor_category quarantine_location - external_actor_category communication_template - external_actor_category logistics_option - external_actor_category regulatory_authority - external_actor_category quality_assurance_resource - external_actor_category regulatory_package - external_actor_category mitigation_resource - inventory_resource_category testing_lab - inventory_resource_category external_authority_contact - inventory_resource_category api_product - product_family distribution_channel - product_family response_plan - product_family manufacturer_role_container - incident_event distributor_role_container - incident_event manufacturing_site - manufacturer_role_container distribution_site - manufacturer_role_container response_team_unit - distributor_role_container)

  (:predicates
    (incident_reported ?incident_event - incident_event)
    (entity_response_authorization_granted ?incident_event - incident_event)
    (entity_supplier_assignment_recorded ?incident_event - incident_event)
    (entity_supplier_assignment_committed ?incident_event - incident_event)
    (execution_authorized ?incident_event - incident_event)
    (recovery_approved ?incident_event - incident_event)
    (supplier_available ?supplier - supplier)
    (entity_assigned_supplier ?incident_event - incident_event ?supplier - supplier)
    (laboratory_resource_available ?laboratory_resource - laboratory_resource)
    (entity_assigned_laboratory ?incident_event - incident_event ?laboratory_resource - laboratory_resource)
    (quarantine_location_available ?quarantine_location - quarantine_location)
    (entity_quarantined_at ?incident_event - incident_event ?quarantine_location - quarantine_location)
    (mitigation_resource_available ?mitigation_resource - mitigation_resource)
    (manufacturing_site_allocated_mitigation ?manufacturing_site - manufacturing_site ?mitigation_resource - mitigation_resource)
    (distribution_site_allocated_mitigation ?distribution_site - distribution_site ?mitigation_resource - mitigation_resource)
    (manufacturing_site_impacted_product ?manufacturing_site - manufacturing_site ?api_product - api_product)
    (api_product_hold_flag ?api_product - api_product)
    (api_product_mitigation_deployed ?api_product - api_product)
    (manufacturing_site_containment_confirmed ?manufacturing_site - manufacturing_site)
    (distribution_site_serves_channel ?distribution_site - distribution_site ?distribution_channel - distribution_channel)
    (distribution_channel_hold_flag ?distribution_channel - distribution_channel)
    (distribution_channel_mitigation_deployed ?distribution_channel - distribution_channel)
    (distribution_site_containment_confirmed ?distribution_site - distribution_site)
    (response_plan_available ?response_plan - response_plan)
    (response_plan_selected ?response_plan - response_plan)
    (response_plan_covers_product ?response_plan - response_plan ?api_product - api_product)
    (response_plan_covers_channel ?response_plan - response_plan ?distribution_channel - distribution_channel)
    (response_plan_requires_qa_approval ?response_plan - response_plan)
    (response_plan_requires_regulatory_approval ?response_plan - response_plan)
    (response_plan_lab_activated ?response_plan - response_plan)
    (team_assigned_to_manufacturing_site ?response_team_unit - response_team_unit ?manufacturing_site - manufacturing_site)
    (team_assigned_to_distribution_site ?response_team_unit - response_team_unit ?distribution_site - distribution_site)
    (team_bound_to_response_plan ?response_team_unit - response_team_unit ?response_plan - response_plan)
    (testing_lab_available ?testing_lab - testing_lab)
    (team_bound_to_testing_lab ?response_team_unit - response_team_unit ?testing_lab - testing_lab)
    (testing_lab_committed ?testing_lab - testing_lab)
    (testing_lab_assigned_to_plan ?testing_lab - testing_lab ?response_plan - response_plan)
    (team_qa_review_initiated ?response_team_unit - response_team_unit)
    (team_qa_approval_obtained ?response_team_unit - response_team_unit)
    (regulatory_package_submitted ?response_team_unit - response_team_unit)
    (communication_template_committed ?response_team_unit - response_team_unit)
    (team_communications_prepared ?response_team_unit - response_team_unit)
    (regulatory_clearance_obtained ?response_team_unit - response_team_unit)
    (team_ready_for_execution ?response_team_unit - response_team_unit)
    (external_contact_available ?external_authority_contact - external_authority_contact)
    (team_has_external_contact_assigned ?response_team_unit - response_team_unit ?external_authority_contact - external_authority_contact)
    (team_external_contact_engaged ?response_team_unit - response_team_unit)
    (external_notification_initiated ?response_team_unit - response_team_unit)
    (external_notification_completed ?response_team_unit - response_team_unit)
    (communication_template_available ?communication_template - communication_template)
    (team_assigned_communication_template ?response_team_unit - response_team_unit ?communication_template - communication_template)
    (logistics_option_available ?logistics_option - logistics_option)
    (team_assigned_logistics_option ?response_team_unit - response_team_unit ?logistics_option - logistics_option)
    (quality_assurance_resource_available ?quality_assurance_resource - quality_assurance_resource)
    (team_assigned_qa_resource ?response_team_unit - response_team_unit ?quality_assurance_resource - quality_assurance_resource)
    (regulatory_package_available ?regulatory_package - regulatory_package)
    (team_assigned_regulatory_package ?response_team_unit - response_team_unit ?regulatory_package - regulatory_package)
    (regulatory_authority_available ?regulatory_authority - regulatory_authority)
    (incident_notified_regulatory_authority ?incident_event - incident_event ?regulatory_authority - regulatory_authority)
    (manufacturing_site_remediation_ready ?manufacturing_site - manufacturing_site)
    (distribution_site_remediation_ready ?distribution_site - distribution_site)
    (team_final_authorization_recorded ?response_team_unit - response_team_unit)
  )
  (:action initialize_incident_record
    :parameters (?incident_event - incident_event)
    :precondition
      (and
        (not
          (incident_reported ?incident_event)
        )
        (not
          (entity_supplier_assignment_committed ?incident_event)
        )
      )
    :effect (incident_reported ?incident_event)
  )
  (:action assign_supplier_to_incident
    :parameters (?incident_event - incident_event ?supplier - supplier)
    :precondition
      (and
        (incident_reported ?incident_event)
        (not
          (entity_supplier_assignment_recorded ?incident_event)
        )
        (supplier_available ?supplier)
      )
    :effect
      (and
        (entity_supplier_assignment_recorded ?incident_event)
        (entity_assigned_supplier ?incident_event ?supplier)
        (not
          (supplier_available ?supplier)
        )
      )
  )
  (:action assign_lab_to_incident
    :parameters (?incident_event - incident_event ?laboratory_resource - laboratory_resource)
    :precondition
      (and
        (incident_reported ?incident_event)
        (entity_supplier_assignment_recorded ?incident_event)
        (laboratory_resource_available ?laboratory_resource)
      )
    :effect
      (and
        (entity_assigned_laboratory ?incident_event ?laboratory_resource)
        (not
          (laboratory_resource_available ?laboratory_resource)
        )
      )
  )
  (:action confirm_triage_for_incident
    :parameters (?incident_event - incident_event ?laboratory_resource - laboratory_resource)
    :precondition
      (and
        (incident_reported ?incident_event)
        (entity_supplier_assignment_recorded ?incident_event)
        (entity_assigned_laboratory ?incident_event ?laboratory_resource)
        (not
          (entity_response_authorization_granted ?incident_event)
        )
      )
    :effect (entity_response_authorization_granted ?incident_event)
  )
  (:action release_lab_assignment
    :parameters (?incident_event - incident_event ?laboratory_resource - laboratory_resource)
    :precondition
      (and
        (entity_assigned_laboratory ?incident_event ?laboratory_resource)
      )
    :effect
      (and
        (laboratory_resource_available ?laboratory_resource)
        (not
          (entity_assigned_laboratory ?incident_event ?laboratory_resource)
        )
      )
  )
  (:action assign_quarantine_location_to_incident
    :parameters (?incident_event - incident_event ?quarantine_location - quarantine_location)
    :precondition
      (and
        (entity_response_authorization_granted ?incident_event)
        (quarantine_location_available ?quarantine_location)
      )
    :effect
      (and
        (entity_quarantined_at ?incident_event ?quarantine_location)
        (not
          (quarantine_location_available ?quarantine_location)
        )
      )
  )
  (:action release_quarantine_assignment
    :parameters (?incident_event - incident_event ?quarantine_location - quarantine_location)
    :precondition
      (and
        (entity_quarantined_at ?incident_event ?quarantine_location)
      )
    :effect
      (and
        (quarantine_location_available ?quarantine_location)
        (not
          (entity_quarantined_at ?incident_event ?quarantine_location)
        )
      )
  )
  (:action assign_qa_resource_to_team
    :parameters (?response_team_unit - response_team_unit ?quality_assurance_resource - quality_assurance_resource)
    :precondition
      (and
        (entity_response_authorization_granted ?response_team_unit)
        (quality_assurance_resource_available ?quality_assurance_resource)
      )
    :effect
      (and
        (team_assigned_qa_resource ?response_team_unit ?quality_assurance_resource)
        (not
          (quality_assurance_resource_available ?quality_assurance_resource)
        )
      )
  )
  (:action release_qa_resource_from_team
    :parameters (?response_team_unit - response_team_unit ?quality_assurance_resource - quality_assurance_resource)
    :precondition
      (and
        (team_assigned_qa_resource ?response_team_unit ?quality_assurance_resource)
      )
    :effect
      (and
        (quality_assurance_resource_available ?quality_assurance_resource)
        (not
          (team_assigned_qa_resource ?response_team_unit ?quality_assurance_resource)
        )
      )
  )
  (:action assign_regulatory_package_to_team
    :parameters (?response_team_unit - response_team_unit ?regulatory_package - regulatory_package)
    :precondition
      (and
        (entity_response_authorization_granted ?response_team_unit)
        (regulatory_package_available ?regulatory_package)
      )
    :effect
      (and
        (team_assigned_regulatory_package ?response_team_unit ?regulatory_package)
        (not
          (regulatory_package_available ?regulatory_package)
        )
      )
  )
  (:action release_regulatory_package_from_team
    :parameters (?response_team_unit - response_team_unit ?regulatory_package - regulatory_package)
    :precondition
      (and
        (team_assigned_regulatory_package ?response_team_unit ?regulatory_package)
      )
    :effect
      (and
        (regulatory_package_available ?regulatory_package)
        (not
          (team_assigned_regulatory_package ?response_team_unit ?regulatory_package)
        )
      )
  )
  (:action initiate_product_hold_for_site
    :parameters (?manufacturing_site - manufacturing_site ?api_product - api_product ?laboratory_resource - laboratory_resource)
    :precondition
      (and
        (entity_response_authorization_granted ?manufacturing_site)
        (entity_assigned_laboratory ?manufacturing_site ?laboratory_resource)
        (manufacturing_site_impacted_product ?manufacturing_site ?api_product)
        (not
          (api_product_hold_flag ?api_product)
        )
        (not
          (api_product_mitigation_deployed ?api_product)
        )
      )
    :effect (api_product_hold_flag ?api_product)
  )
  (:action confirm_site_containment
    :parameters (?manufacturing_site - manufacturing_site ?api_product - api_product ?quarantine_location - quarantine_location)
    :precondition
      (and
        (entity_response_authorization_granted ?manufacturing_site)
        (entity_quarantined_at ?manufacturing_site ?quarantine_location)
        (manufacturing_site_impacted_product ?manufacturing_site ?api_product)
        (api_product_hold_flag ?api_product)
        (not
          (manufacturing_site_remediation_ready ?manufacturing_site)
        )
      )
    :effect
      (and
        (manufacturing_site_remediation_ready ?manufacturing_site)
        (manufacturing_site_containment_confirmed ?manufacturing_site)
      )
  )
  (:action deploy_mitigation_resource_to_site
    :parameters (?manufacturing_site - manufacturing_site ?api_product - api_product ?mitigation_resource - mitigation_resource)
    :precondition
      (and
        (entity_response_authorization_granted ?manufacturing_site)
        (manufacturing_site_impacted_product ?manufacturing_site ?api_product)
        (mitigation_resource_available ?mitigation_resource)
        (not
          (manufacturing_site_remediation_ready ?manufacturing_site)
        )
      )
    :effect
      (and
        (api_product_mitigation_deployed ?api_product)
        (manufacturing_site_remediation_ready ?manufacturing_site)
        (manufacturing_site_allocated_mitigation ?manufacturing_site ?mitigation_resource)
        (not
          (mitigation_resource_available ?mitigation_resource)
        )
      )
  )
  (:action finalize_site_mitigation
    :parameters (?manufacturing_site - manufacturing_site ?api_product - api_product ?laboratory_resource - laboratory_resource ?mitigation_resource - mitigation_resource)
    :precondition
      (and
        (entity_response_authorization_granted ?manufacturing_site)
        (entity_assigned_laboratory ?manufacturing_site ?laboratory_resource)
        (manufacturing_site_impacted_product ?manufacturing_site ?api_product)
        (api_product_mitigation_deployed ?api_product)
        (manufacturing_site_allocated_mitigation ?manufacturing_site ?mitigation_resource)
        (not
          (manufacturing_site_containment_confirmed ?manufacturing_site)
        )
      )
    :effect
      (and
        (api_product_hold_flag ?api_product)
        (manufacturing_site_containment_confirmed ?manufacturing_site)
        (mitigation_resource_available ?mitigation_resource)
        (not
          (manufacturing_site_allocated_mitigation ?manufacturing_site ?mitigation_resource)
        )
      )
  )
  (:action initiate_distribution_site_hold
    :parameters (?distribution_site - distribution_site ?distribution_channel - distribution_channel ?laboratory_resource - laboratory_resource)
    :precondition
      (and
        (entity_response_authorization_granted ?distribution_site)
        (entity_assigned_laboratory ?distribution_site ?laboratory_resource)
        (distribution_site_serves_channel ?distribution_site ?distribution_channel)
        (not
          (distribution_channel_hold_flag ?distribution_channel)
        )
        (not
          (distribution_channel_mitigation_deployed ?distribution_channel)
        )
      )
    :effect (distribution_channel_hold_flag ?distribution_channel)
  )
  (:action confirm_distribution_site_containment
    :parameters (?distribution_site - distribution_site ?distribution_channel - distribution_channel ?quarantine_location - quarantine_location)
    :precondition
      (and
        (entity_response_authorization_granted ?distribution_site)
        (entity_quarantined_at ?distribution_site ?quarantine_location)
        (distribution_site_serves_channel ?distribution_site ?distribution_channel)
        (distribution_channel_hold_flag ?distribution_channel)
        (not
          (distribution_site_remediation_ready ?distribution_site)
        )
      )
    :effect
      (and
        (distribution_site_remediation_ready ?distribution_site)
        (distribution_site_containment_confirmed ?distribution_site)
      )
  )
  (:action deploy_mitigation_resource_to_distribution_site
    :parameters (?distribution_site - distribution_site ?distribution_channel - distribution_channel ?mitigation_resource - mitigation_resource)
    :precondition
      (and
        (entity_response_authorization_granted ?distribution_site)
        (distribution_site_serves_channel ?distribution_site ?distribution_channel)
        (mitigation_resource_available ?mitigation_resource)
        (not
          (distribution_site_remediation_ready ?distribution_site)
        )
      )
    :effect
      (and
        (distribution_channel_mitigation_deployed ?distribution_channel)
        (distribution_site_remediation_ready ?distribution_site)
        (distribution_site_allocated_mitigation ?distribution_site ?mitigation_resource)
        (not
          (mitigation_resource_available ?mitigation_resource)
        )
      )
  )
  (:action finalize_distribution_site_mitigation
    :parameters (?distribution_site - distribution_site ?distribution_channel - distribution_channel ?laboratory_resource - laboratory_resource ?mitigation_resource - mitigation_resource)
    :precondition
      (and
        (entity_response_authorization_granted ?distribution_site)
        (entity_assigned_laboratory ?distribution_site ?laboratory_resource)
        (distribution_site_serves_channel ?distribution_site ?distribution_channel)
        (distribution_channel_mitigation_deployed ?distribution_channel)
        (distribution_site_allocated_mitigation ?distribution_site ?mitigation_resource)
        (not
          (distribution_site_containment_confirmed ?distribution_site)
        )
      )
    :effect
      (and
        (distribution_channel_hold_flag ?distribution_channel)
        (distribution_site_containment_confirmed ?distribution_site)
        (mitigation_resource_available ?mitigation_resource)
        (not
          (distribution_site_allocated_mitigation ?distribution_site ?mitigation_resource)
        )
      )
  )
  (:action select_response_plan
    :parameters (?manufacturing_site - manufacturing_site ?distribution_site - distribution_site ?api_product - api_product ?distribution_channel - distribution_channel ?response_plan - response_plan)
    :precondition
      (and
        (manufacturing_site_remediation_ready ?manufacturing_site)
        (distribution_site_remediation_ready ?distribution_site)
        (manufacturing_site_impacted_product ?manufacturing_site ?api_product)
        (distribution_site_serves_channel ?distribution_site ?distribution_channel)
        (api_product_hold_flag ?api_product)
        (distribution_channel_hold_flag ?distribution_channel)
        (manufacturing_site_containment_confirmed ?manufacturing_site)
        (distribution_site_containment_confirmed ?distribution_site)
        (response_plan_available ?response_plan)
      )
    :effect
      (and
        (response_plan_selected ?response_plan)
        (response_plan_covers_product ?response_plan ?api_product)
        (response_plan_covers_channel ?response_plan ?distribution_channel)
        (not
          (response_plan_available ?response_plan)
        )
      )
  )
  (:action select_response_plan_with_qa_requirement
    :parameters (?manufacturing_site - manufacturing_site ?distribution_site - distribution_site ?api_product - api_product ?distribution_channel - distribution_channel ?response_plan - response_plan)
    :precondition
      (and
        (manufacturing_site_remediation_ready ?manufacturing_site)
        (distribution_site_remediation_ready ?distribution_site)
        (manufacturing_site_impacted_product ?manufacturing_site ?api_product)
        (distribution_site_serves_channel ?distribution_site ?distribution_channel)
        (api_product_mitigation_deployed ?api_product)
        (distribution_channel_hold_flag ?distribution_channel)
        (not
          (manufacturing_site_containment_confirmed ?manufacturing_site)
        )
        (distribution_site_containment_confirmed ?distribution_site)
        (response_plan_available ?response_plan)
      )
    :effect
      (and
        (response_plan_selected ?response_plan)
        (response_plan_covers_product ?response_plan ?api_product)
        (response_plan_covers_channel ?response_plan ?distribution_channel)
        (response_plan_requires_qa_approval ?response_plan)
        (not
          (response_plan_available ?response_plan)
        )
      )
  )
  (:action select_response_plan_with_regulatory_requirement
    :parameters (?manufacturing_site - manufacturing_site ?distribution_site - distribution_site ?api_product - api_product ?distribution_channel - distribution_channel ?response_plan - response_plan)
    :precondition
      (and
        (manufacturing_site_remediation_ready ?manufacturing_site)
        (distribution_site_remediation_ready ?distribution_site)
        (manufacturing_site_impacted_product ?manufacturing_site ?api_product)
        (distribution_site_serves_channel ?distribution_site ?distribution_channel)
        (api_product_hold_flag ?api_product)
        (distribution_channel_mitigation_deployed ?distribution_channel)
        (manufacturing_site_containment_confirmed ?manufacturing_site)
        (not
          (distribution_site_containment_confirmed ?distribution_site)
        )
        (response_plan_available ?response_plan)
      )
    :effect
      (and
        (response_plan_selected ?response_plan)
        (response_plan_covers_product ?response_plan ?api_product)
        (response_plan_covers_channel ?response_plan ?distribution_channel)
        (response_plan_requires_regulatory_approval ?response_plan)
        (not
          (response_plan_available ?response_plan)
        )
      )
  )
  (:action select_response_plan_with_qa_and_regulatory_requirements
    :parameters (?manufacturing_site - manufacturing_site ?distribution_site - distribution_site ?api_product - api_product ?distribution_channel - distribution_channel ?response_plan - response_plan)
    :precondition
      (and
        (manufacturing_site_remediation_ready ?manufacturing_site)
        (distribution_site_remediation_ready ?distribution_site)
        (manufacturing_site_impacted_product ?manufacturing_site ?api_product)
        (distribution_site_serves_channel ?distribution_site ?distribution_channel)
        (api_product_mitigation_deployed ?api_product)
        (distribution_channel_mitigation_deployed ?distribution_channel)
        (not
          (manufacturing_site_containment_confirmed ?manufacturing_site)
        )
        (not
          (distribution_site_containment_confirmed ?distribution_site)
        )
        (response_plan_available ?response_plan)
      )
    :effect
      (and
        (response_plan_selected ?response_plan)
        (response_plan_covers_product ?response_plan ?api_product)
        (response_plan_covers_channel ?response_plan ?distribution_channel)
        (response_plan_requires_qa_approval ?response_plan)
        (response_plan_requires_regulatory_approval ?response_plan)
        (not
          (response_plan_available ?response_plan)
        )
      )
  )
  (:action activate_response_plan
    :parameters (?response_plan - response_plan ?manufacturing_site - manufacturing_site ?laboratory_resource - laboratory_resource)
    :precondition
      (and
        (response_plan_selected ?response_plan)
        (manufacturing_site_remediation_ready ?manufacturing_site)
        (entity_assigned_laboratory ?manufacturing_site ?laboratory_resource)
        (not
          (response_plan_lab_activated ?response_plan)
        )
      )
    :effect (response_plan_lab_activated ?response_plan)
  )
  (:action bind_testing_lab_to_team_and_plan
    :parameters (?response_team_unit - response_team_unit ?testing_lab - testing_lab ?response_plan - response_plan)
    :precondition
      (and
        (entity_response_authorization_granted ?response_team_unit)
        (team_bound_to_response_plan ?response_team_unit ?response_plan)
        (team_bound_to_testing_lab ?response_team_unit ?testing_lab)
        (testing_lab_available ?testing_lab)
        (response_plan_selected ?response_plan)
        (response_plan_lab_activated ?response_plan)
        (not
          (testing_lab_committed ?testing_lab)
        )
      )
    :effect
      (and
        (testing_lab_committed ?testing_lab)
        (testing_lab_assigned_to_plan ?testing_lab ?response_plan)
        (not
          (testing_lab_available ?testing_lab)
        )
      )
  )
  (:action initiate_team_qa_review
    :parameters (?response_team_unit - response_team_unit ?testing_lab - testing_lab ?response_plan - response_plan ?laboratory_resource - laboratory_resource)
    :precondition
      (and
        (entity_response_authorization_granted ?response_team_unit)
        (team_bound_to_testing_lab ?response_team_unit ?testing_lab)
        (testing_lab_committed ?testing_lab)
        (testing_lab_assigned_to_plan ?testing_lab ?response_plan)
        (entity_assigned_laboratory ?response_team_unit ?laboratory_resource)
        (not
          (response_plan_requires_qa_approval ?response_plan)
        )
        (not
          (team_qa_review_initiated ?response_team_unit)
        )
      )
    :effect (team_qa_review_initiated ?response_team_unit)
  )
  (:action assign_communication_template_to_team
    :parameters (?response_team_unit - response_team_unit ?communication_template - communication_template)
    :precondition
      (and
        (entity_response_authorization_granted ?response_team_unit)
        (communication_template_available ?communication_template)
        (not
          (communication_template_committed ?response_team_unit)
        )
      )
    :effect
      (and
        (communication_template_committed ?response_team_unit)
        (team_assigned_communication_template ?response_team_unit ?communication_template)
        (not
          (communication_template_available ?communication_template)
        )
      )
  )
  (:action finalize_testing_and_communications_for_team
    :parameters (?response_team_unit - response_team_unit ?testing_lab - testing_lab ?response_plan - response_plan ?laboratory_resource - laboratory_resource ?communication_template - communication_template)
    :precondition
      (and
        (entity_response_authorization_granted ?response_team_unit)
        (team_bound_to_testing_lab ?response_team_unit ?testing_lab)
        (testing_lab_committed ?testing_lab)
        (testing_lab_assigned_to_plan ?testing_lab ?response_plan)
        (entity_assigned_laboratory ?response_team_unit ?laboratory_resource)
