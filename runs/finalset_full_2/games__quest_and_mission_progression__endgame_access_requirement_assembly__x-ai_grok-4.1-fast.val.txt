--- domain.pddl (as passed to VAL) ---
(define (domain endgame_access_requirement_assembly)
  (:requirements :strips :typing :negative-preconditions)
  (:types base_object - object actor_or_token - base_object item_category - base_object requirement_category - base_object sequence_container - base_object progression_slot - sequence_container requirement_token - actor_or_token encounter - actor_or_token agent - actor_or_token optional_modifier - actor_or_token upgrade_token - actor_or_token identity_seal - actor_or_token attribute_module - actor_or_token influence_token - actor_or_token consumable_item - item_category component - item_category story_flag - item_category subnode_a - requirement_category subnode_b - requirement_category requirement_package - requirement_category cohort_container_a - progression_slot cohort_container_b - progression_slot cohort_a - cohort_container_a cohort_b - cohort_container_a endgame_gate - cohort_container_b)

  (:predicates
    (enabled ?progression_slot - progression_slot)
    (validated ?progression_slot - progression_slot)
    (token_marked ?progression_slot - progression_slot)
    (access_granted ?progression_slot - progression_slot)
    (ready ?progression_slot - progression_slot)
    (finalized ?progression_slot - progression_slot)
    (token_available ?requirement_token - requirement_token)
    (bound_token ?progression_slot - progression_slot ?requirement_token - requirement_token)
    (encounter_available ?encounter - encounter)
    (assigned_encounter ?progression_slot - progression_slot ?encounter - encounter)
    (agent_available ?agent - agent)
    (assigned_agent ?progression_slot - progression_slot ?agent - agent)
    (consumable_available ?consumable_item - consumable_item)
    (cohort_a_bound_item ?cohort_a - cohort_a ?consumable_item - consumable_item)
    (cohort_b_bound_item ?cohort_b - cohort_b ?consumable_item - consumable_item)
    (cohort_a_subnode_link ?cohort_a - cohort_a ?subnode_a - subnode_a)
    (subnode_a_completed_by_encounter ?subnode_a - subnode_a)
    (subnode_a_completed_by_item ?subnode_a - subnode_a)
    (cohort_a_checkpoint_completed ?cohort_a - cohort_a)
    (cohort_b_subnode_link ?cohort_b - cohort_b ?subnode_b - subnode_b)
    (subnode_b_completed_by_encounter ?subnode_b - subnode_b)
    (subnode_b_completed_by_item ?subnode_b - subnode_b)
    (cohort_b_checkpoint_completed ?cohort_b - cohort_b)
    (package_available ?requirement_package - requirement_package)
    (package_assembled ?requirement_package - requirement_package)
    (package_includes_subnode_a ?requirement_package - requirement_package ?subnode_a - subnode_a)
    (package_includes_subnode_b ?requirement_package - requirement_package ?subnode_b - subnode_b)
    (package_has_subnode_a_item ?requirement_package - requirement_package)
    (package_has_subnode_b_item ?requirement_package - requirement_package)
    (package_validated ?requirement_package - requirement_package)
    (gate_linked_cohort_a ?endgame_gate - endgame_gate ?cohort_a - cohort_a)
    (gate_linked_cohort_b ?endgame_gate - endgame_gate ?cohort_b - cohort_b)
    (gate_linked_package ?endgame_gate - endgame_gate ?requirement_package - requirement_package)
    (component_available ?component - component)
    (gate_component_compatibility ?endgame_gate - endgame_gate ?component - component)
    (component_consumed ?component - component)
    (component_added_to_package ?component - component ?requirement_package - requirement_package)
    (gate_components_integrated ?endgame_gate - endgame_gate)
    (gate_attribute_module_attached ?endgame_gate - endgame_gate)
    (gate_influence_module_attached ?endgame_gate - endgame_gate)
    (gate_modifier_attached ?endgame_gate - endgame_gate)
    (gate_modifier_validated ?endgame_gate - endgame_gate)
    (gate_upgrade_applied ?endgame_gate - endgame_gate)
    (gate_authorization_ready ?endgame_gate - endgame_gate)
    (story_flag_available ?story_flag - story_flag)
    (gate_has_story_flag ?endgame_gate - endgame_gate ?story_flag - story_flag)
    (gate_story_flag_applied ?endgame_gate - endgame_gate)
    (gate_story_flag_validated ?endgame_gate - endgame_gate)
    (gate_story_influence_applied ?endgame_gate - endgame_gate)
    (optional_modifier_available ?optional_modifier - optional_modifier)
    (gate_modifier_linked ?endgame_gate - endgame_gate ?optional_modifier - optional_modifier)
    (upgrade_token_available ?upgrade_token - upgrade_token)
    (gate_upgrade_linked ?endgame_gate - endgame_gate ?upgrade_token - upgrade_token)
    (attribute_module_available ?attribute_module - attribute_module)
    (gate_attribute_module_linked ?endgame_gate - endgame_gate ?attribute_module - attribute_module)
    (influence_token_available ?influence_token - influence_token)
    (gate_influence_token_linked ?endgame_gate - endgame_gate ?influence_token - influence_token)
    (identity_seal_available ?identity_seal - identity_seal)
    (bound_identity_seal ?progression_slot - progression_slot ?identity_seal - identity_seal)
    (cohort_a_checkpoint_flag ?cohort_a - cohort_a)
    (cohort_b_checkpoint_flag ?cohort_b - cohort_b)
    (gate_authorized ?endgame_gate - endgame_gate)
  )
  (:action activate_progression_slot
    :parameters (?progression_slot - progression_slot)
    :precondition
      (and
        (not
          (enabled ?progression_slot)
        )
        (not
          (access_granted ?progression_slot)
        )
      )
    :effect (enabled ?progression_slot)
  )
  (:action assign_requirement_token_to_slot
    :parameters (?progression_slot - progression_slot ?requirement_token - requirement_token)
    :precondition
      (and
        (enabled ?progression_slot)
        (not
          (token_marked ?progression_slot)
        )
        (token_available ?requirement_token)
      )
    :effect
      (and
        (token_marked ?progression_slot)
        (bound_token ?progression_slot ?requirement_token)
        (not
          (token_available ?requirement_token)
        )
      )
  )
  (:action assign_encounter_to_slot
    :parameters (?progression_slot - progression_slot ?encounter - encounter)
    :precondition
      (and
        (enabled ?progression_slot)
        (token_marked ?progression_slot)
        (encounter_available ?encounter)
      )
    :effect
      (and
        (assigned_encounter ?progression_slot ?encounter)
        (not
          (encounter_available ?encounter)
        )
      )
  )
  (:action validate_progression_slot
    :parameters (?progression_slot - progression_slot ?encounter - encounter)
    :precondition
      (and
        (enabled ?progression_slot)
        (token_marked ?progression_slot)
        (assigned_encounter ?progression_slot ?encounter)
        (not
          (validated ?progression_slot)
        )
      )
    :effect (validated ?progression_slot)
  )
  (:action release_encounter_from_slot
    :parameters (?progression_slot - progression_slot ?encounter - encounter)
    :precondition
      (and
        (assigned_encounter ?progression_slot ?encounter)
      )
    :effect
      (and
        (encounter_available ?encounter)
        (not
          (assigned_encounter ?progression_slot ?encounter)
        )
      )
  )
  (:action assign_agent_to_slot
    :parameters (?progression_slot - progression_slot ?agent - agent)
    :precondition
      (and
        (validated ?progression_slot)
        (agent_available ?agent)
      )
    :effect
      (and
        (assigned_agent ?progression_slot ?agent)
        (not
          (agent_available ?agent)
        )
      )
  )
  (:action release_agent_from_slot
    :parameters (?progression_slot - progression_slot ?agent - agent)
    :precondition
      (and
        (assigned_agent ?progression_slot ?agent)
      )
    :effect
      (and
        (agent_available ?agent)
        (not
          (assigned_agent ?progression_slot ?agent)
        )
      )
  )
  (:action attach_attribute_module
    :parameters (?endgame_gate - endgame_gate ?attribute_module - attribute_module)
    :precondition
      (and
        (validated ?endgame_gate)
        (attribute_module_available ?attribute_module)
      )
    :effect
      (and
        (gate_attribute_module_linked ?endgame_gate ?attribute_module)
        (not
          (attribute_module_available ?attribute_module)
        )
      )
  )
  (:action detach_attribute_module
    :parameters (?endgame_gate - endgame_gate ?attribute_module - attribute_module)
    :precondition
      (and
        (gate_attribute_module_linked ?endgame_gate ?attribute_module)
      )
    :effect
      (and
        (attribute_module_available ?attribute_module)
        (not
          (gate_attribute_module_linked ?endgame_gate ?attribute_module)
        )
      )
  )
  (:action attach_influence_token
    :parameters (?endgame_gate - endgame_gate ?influence_token - influence_token)
    :precondition
      (and
        (validated ?endgame_gate)
        (influence_token_available ?influence_token)
      )
    :effect
      (and
        (gate_influence_token_linked ?endgame_gate ?influence_token)
        (not
          (influence_token_available ?influence_token)
        )
      )
  )
  (:action detach_influence_token
    :parameters (?endgame_gate - endgame_gate ?influence_token - influence_token)
    :precondition
      (and
        (gate_influence_token_linked ?endgame_gate ?influence_token)
      )
    :effect
      (and
        (influence_token_available ?influence_token)
        (not
          (gate_influence_token_linked ?endgame_gate ?influence_token)
        )
      )
  )
  (:action complete_subnode_a_by_encounter
    :parameters (?cohort_a - cohort_a ?subnode_a - subnode_a ?encounter - encounter)
    :precondition
      (and
        (validated ?cohort_a)
        (assigned_encounter ?cohort_a ?encounter)
        (cohort_a_subnode_link ?cohort_a ?subnode_a)
        (not
          (subnode_a_completed_by_encounter ?subnode_a)
        )
        (not
          (subnode_a_completed_by_item ?subnode_a)
        )
      )
    :effect (subnode_a_completed_by_encounter ?subnode_a)
  )
  (:action confirm_subnode_a_with_agent
    :parameters (?cohort_a - cohort_a ?subnode_a - subnode_a ?agent - agent)
    :precondition
      (and
        (validated ?cohort_a)
        (assigned_agent ?cohort_a ?agent)
        (cohort_a_subnode_link ?cohort_a ?subnode_a)
        (subnode_a_completed_by_encounter ?subnode_a)
        (not
          (cohort_a_checkpoint_flag ?cohort_a)
        )
      )
    :effect
      (and
        (cohort_a_checkpoint_flag ?cohort_a)
        (cohort_a_checkpoint_completed ?cohort_a)
      )
  )
  (:action complete_subnode_a_with_consumable
    :parameters (?cohort_a - cohort_a ?subnode_a - subnode_a ?consumable_item - consumable_item)
    :precondition
      (and
        (validated ?cohort_a)
        (cohort_a_subnode_link ?cohort_a ?subnode_a)
        (consumable_available ?consumable_item)
        (not
          (cohort_a_checkpoint_flag ?cohort_a)
        )
      )
    :effect
      (and
        (subnode_a_completed_by_item ?subnode_a)
        (cohort_a_checkpoint_flag ?cohort_a)
        (cohort_a_bound_item ?cohort_a ?consumable_item)
        (not
          (consumable_available ?consumable_item)
        )
      )
  )
  (:action finalize_subnode_a
    :parameters (?cohort_a - cohort_a ?subnode_a - subnode_a ?encounter - encounter ?consumable_item - consumable_item)
    :precondition
      (and
        (validated ?cohort_a)
        (assigned_encounter ?cohort_a ?encounter)
        (cohort_a_subnode_link ?cohort_a ?subnode_a)
        (subnode_a_completed_by_item ?subnode_a)
        (cohort_a_bound_item ?cohort_a ?consumable_item)
        (not
          (cohort_a_checkpoint_completed ?cohort_a)
        )
      )
    :effect
      (and
        (subnode_a_completed_by_encounter ?subnode_a)
        (cohort_a_checkpoint_completed ?cohort_a)
        (consumable_available ?consumable_item)
        (not
          (cohort_a_bound_item ?cohort_a ?consumable_item)
        )
      )
  )
  (:action complete_subnode_b_by_encounter
    :parameters (?cohort_b - cohort_b ?subnode_b - subnode_b ?encounter - encounter)
    :precondition
      (and
        (validated ?cohort_b)
        (assigned_encounter ?cohort_b ?encounter)
        (cohort_b_subnode_link ?cohort_b ?subnode_b)
        (not
          (subnode_b_completed_by_encounter ?subnode_b)
        )
        (not
          (subnode_b_completed_by_item ?subnode_b)
        )
      )
    :effect (subnode_b_completed_by_encounter ?subnode_b)
  )
  (:action confirm_subnode_b_with_agent
    :parameters (?cohort_b - cohort_b ?subnode_b - subnode_b ?agent - agent)
    :precondition
      (and
        (validated ?cohort_b)
        (assigned_agent ?cohort_b ?agent)
        (cohort_b_subnode_link ?cohort_b ?subnode_b)
        (subnode_b_completed_by_encounter ?subnode_b)
        (not
          (cohort_b_checkpoint_flag ?cohort_b)
        )
      )
    :effect
      (and
        (cohort_b_checkpoint_flag ?cohort_b)
        (cohort_b_checkpoint_completed ?cohort_b)
      )
  )
  (:action complete_subnode_b_with_consumable
    :parameters (?cohort_b - cohort_b ?subnode_b - subnode_b ?consumable_item - consumable_item)
    :precondition
      (and
        (validated ?cohort_b)
        (cohort_b_subnode_link ?cohort_b ?subnode_b)
        (consumable_available ?consumable_item)
        (not
          (cohort_b_checkpoint_flag ?cohort_b)
        )
      )
    :effect
      (and
        (subnode_b_completed_by_item ?subnode_b)
        (cohort_b_checkpoint_flag ?cohort_b)
        (cohort_b_bound_item ?cohort_b ?consumable_item)
        (not
          (consumable_available ?consumable_item)
        )
      )
  )
  (:action finalize_subnode_b
    :parameters (?cohort_b - cohort_b ?subnode_b - subnode_b ?encounter - encounter ?consumable_item - consumable_item)
    :precondition
      (and
        (validated ?cohort_b)
        (assigned_encounter ?cohort_b ?encounter)
        (cohort_b_subnode_link ?cohort_b ?subnode_b)
        (subnode_b_completed_by_item ?subnode_b)
        (cohort_b_bound_item ?cohort_b ?consumable_item)
        (not
          (cohort_b_checkpoint_completed ?cohort_b)
        )
      )
    :effect
      (and
        (subnode_b_completed_by_encounter ?subnode_b)
        (cohort_b_checkpoint_completed ?cohort_b)
        (consumable_available ?consumable_item)
        (not
          (cohort_b_bound_item ?cohort_b ?consumable_item)
        )
      )
  )
  (:action assemble_requirement_package
    :parameters (?cohort_a - cohort_a ?cohort_b - cohort_b ?subnode_a - subnode_a ?subnode_b - subnode_b ?requirement_package - requirement_package)
    :precondition
      (and
        (cohort_a_checkpoint_flag ?cohort_a)
        (cohort_b_checkpoint_flag ?cohort_b)
        (cohort_a_subnode_link ?cohort_a ?subnode_a)
        (cohort_b_subnode_link ?cohort_b ?subnode_b)
        (subnode_a_completed_by_encounter ?subnode_a)
        (subnode_b_completed_by_encounter ?subnode_b)
        (cohort_a_checkpoint_completed ?cohort_a)
        (cohort_b_checkpoint_completed ?cohort_b)
        (package_available ?requirement_package)
      )
    :effect
      (and
        (package_assembled ?requirement_package)
        (package_includes_subnode_a ?requirement_package ?subnode_a)
        (package_includes_subnode_b ?requirement_package ?subnode_b)
        (not
          (package_available ?requirement_package)
        )
      )
  )
  (:action assemble_requirement_package_with_a_item
    :parameters (?cohort_a - cohort_a ?cohort_b - cohort_b ?subnode_a - subnode_a ?subnode_b - subnode_b ?requirement_package - requirement_package)
    :precondition
      (and
        (cohort_a_checkpoint_flag ?cohort_a)
        (cohort_b_checkpoint_flag ?cohort_b)
        (cohort_a_subnode_link ?cohort_a ?subnode_a)
        (cohort_b_subnode_link ?cohort_b ?subnode_b)
        (subnode_a_completed_by_item ?subnode_a)
        (subnode_b_completed_by_encounter ?subnode_b)
        (not
          (cohort_a_checkpoint_completed ?cohort_a)
        )
        (cohort_b_checkpoint_completed ?cohort_b)
        (package_available ?requirement_package)
      )
    :effect
      (and
        (package_assembled ?requirement_package)
        (package_includes_subnode_a ?requirement_package ?subnode_a)
        (package_includes_subnode_b ?requirement_package ?subnode_b)
        (package_has_subnode_a_item ?requirement_package)
        (not
          (package_available ?requirement_package)
        )
      )
  )
  (:action assemble_requirement_package_with_b_item
    :parameters (?cohort_a - cohort_a ?cohort_b - cohort_b ?subnode_a - subnode_a ?subnode_b - subnode_b ?requirement_package - requirement_package)
    :precondition
      (and
        (cohort_a_checkpoint_flag ?cohort_a)
        (cohort_b_checkpoint_flag ?cohort_b)
        (cohort_a_subnode_link ?cohort_a ?subnode_a)
        (cohort_b_subnode_link ?cohort_b ?subnode_b)
        (subnode_a_completed_by_encounter ?subnode_a)
        (subnode_b_completed_by_item ?subnode_b)
        (cohort_a_checkpoint_completed ?cohort_a)
        (not
          (cohort_b_checkpoint_completed ?cohort_b)
        )
        (package_available ?requirement_package)
      )
    :effect
      (and
        (package_assembled ?requirement_package)
        (package_includes_subnode_a ?requirement_package ?subnode_a)
        (package_includes_subnode_b ?requirement_package ?subnode_b)
        (package_has_subnode_b_item ?requirement_package)
        (not
          (package_available ?requirement_package)
        )
      )
  )
  (:action assemble_requirement_package_with_both_items
    :parameters (?cohort_a - cohort_a ?cohort_b - cohort_b ?subnode_a - subnode_a ?subnode_b - subnode_b ?requirement_package - requirement_package)
    :precondition
      (and
        (cohort_a_checkpoint_flag ?cohort_a)
        (cohort_b_checkpoint_flag ?cohort_b)
        (cohort_a_subnode_link ?cohort_a ?subnode_a)
        (cohort_b_subnode_link ?cohort_b ?subnode_b)
        (subnode_a_completed_by_item ?subnode_a)
        (subnode_b_completed_by_item ?subnode_b)
        (not
          (cohort_a_checkpoint_completed ?cohort_a)
        )
        (not
          (cohort_b_checkpoint_completed ?cohort_b)
        )
        (package_available ?requirement_package)
      )
    :effect
      (and
        (package_assembled ?requirement_package)
        (package_includes_subnode_a ?requirement_package ?subnode_a)
        (package_includes_subnode_b ?requirement_package ?subnode_b)
        (package_has_subnode_a_item ?requirement_package)
        (package_has_subnode_b_item ?requirement_package)
        (not
          (package_available ?requirement_package)
        )
      )
  )
  (:action validate_requirement_package
    :parameters (?requirement_package - requirement_package ?cohort_a - cohort_a ?encounter - encounter)
    :precondition
      (and
        (package_assembled ?requirement_package)
        (cohort_a_checkpoint_flag ?cohort_a)
        (assigned_encounter ?cohort_a ?encounter)
        (not
          (package_validated ?requirement_package)
        )
      )
    :effect (package_validated ?requirement_package)
  )
  (:action attach_component_to_package
    :parameters (?endgame_gate - endgame_gate ?component - component ?requirement_package - requirement_package)
    :precondition
      (and
        (validated ?endgame_gate)
        (gate_linked_package ?endgame_gate ?requirement_package)
        (gate_component_compatibility ?endgame_gate ?component)
        (component_available ?component)
        (package_assembled ?requirement_package)
        (package_validated ?requirement_package)
        (not
          (component_consumed ?component)
        )
      )
    :effect
      (and
        (component_consumed ?component)
        (component_added_to_package ?component ?requirement_package)
        (not
          (component_available ?component)
        )
      )
  )
  (:action finalize_gate_component_integration
    :parameters (?endgame_gate - endgame_gate ?component - component ?requirement_package - requirement_package ?encounter - encounter)
    :precondition
      (and
        (validated ?endgame_gate)
        (gate_component_compatibility ?endgame_gate ?component)
        (component_consumed ?component)
        (component_added_to_package ?component ?requirement_package)
        (assigned_encounter ?endgame_gate ?encounter)
        (not
          (package_has_subnode_a_item ?requirement_package)
        )
        (not
          (gate_components_integrated ?endgame_gate)
        )
      )
    :effect (gate_components_integrated ?endgame_gate)
  )
  (:action apply_optional_modifier_to_gate
    :parameters (?endgame_gate - endgame_gate ?optional_modifier - optional_modifier)
    :precondition
      (and
        (validated ?endgame_gate)
        (optional_modifier_available ?optional_modifier)
        (not
          (gate_modifier_attached ?endgame_gate)
        )
      )
    :effect
      (and
        (gate_modifier_attached ?endgame_gate)
        (gate_modifier_linked ?endgame_gate ?optional_modifier)
        (not
          (optional_modifier_available ?optional_modifier)
        )
      )
  )
  (:action validate_gate_modifier_integration
    :parameters (?endgame_gate - endgame_gate ?component - component ?requirement_package - requirement_package ?encounter - encounter ?optional_modifier - optional_modifier)
    :precondition
      (and
        (validated ?endgame_gate)
        (gate_component_compatibility ?endgame_gate ?component)
        (component_consumed ?component)
        (component_added_to_package ?component ?requirement_package)
        (assigned_encounter ?endgame_gate ?encounter)
        (package_has_subnode_a_item ?requirement_package)
        (gate_modifier_attached ?endgame_gate)
        (gate_modifier_linked ?endgame_gate ?optional_modifier)
        (not
          (gate_components_integrated ?endgame_gate)
        )
      )
    :effect
      (and
        (gate_components_integrated ?endgame_gate)
        (gate_modifier_validated ?endgame_gate)
      )
  )
  (:action attach_attribute_module_stage1
    :parameters (?endgame_gate - endgame_gate ?attribute_module - attribute_module ?agent - agent ?component - component ?requirement_package - requirement_package)
    :precondition
      (and
        (gate_components_integrated ?endgame_gate)
        (gate_attribute_module_linked ?endgame_gate ?attribute_module)
        (assigned_agent ?endgame_gate ?agent)
        (gate_component_compatibility ?endgame_gate ?component)
        (component_added_to_package ?component ?requirement_package)
        (not
          (package_has_subnode_b_item ?requirement_package)
        )
        (not
          (gate_attribute_module_attached ?endgame_gate)
        )
      )
    :effect (gate_attribute_module_attached ?endgame_gate)
  )
  (:action attach_attribute_module_stage1_variant
    :parameters (?endgame_gate - endgame_gate ?attribute_module - attribute_module ?agent - agent ?component - component ?requirement_package - requirement_package)
    :precondition
      (and
        (gate_components_integrated ?endgame_gate)
        (gate_attribute_module_linked ?endgame_gate ?attribute_module)
        (assigned_agent ?endgame_gate ?agent)
        (gate_component_compatibility ?endgame_gate ?component)
        (component_added_to_package ?component ?requirement_package)
        (package_has_subnode_b_item ?requirement_package)
        (not
          (gate_attribute_module_attached ?endgame_gate)
        )
      )
