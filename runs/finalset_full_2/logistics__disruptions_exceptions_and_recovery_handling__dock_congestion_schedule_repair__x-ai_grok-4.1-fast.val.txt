--- domain.pddl (as passed to VAL) ---
(define (domain dock_congestion_schedule_repair_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types object_root - object asset_category - object_root service_category - object_root plan_category - object_root entity_root - object_root affected_entity - entity_root resource_unit - asset_category inspection_activity - asset_category work_team - asset_category authorization_token - asset_category third_party_service - asset_category claim_record - asset_category spare_part - asset_category contractor_resource - asset_category contingency_option - service_category inventory_item - service_category external_provider - service_category berth_slot - plan_category route_option - plan_category repair_schedule - plan_category subunit_type_a - affected_entity subunit_type_b - affected_entity origin_node - subunit_type_a destination_node - subunit_type_a recovery_case - subunit_type_b)

  (:predicates
    (exception_reported ?affected_entity - affected_entity)
    (exception_validated ?affected_entity - affected_entity)
    (exception_triaged ?affected_entity - affected_entity)
    (service_restored ?affected_entity - affected_entity)
    (service_available ?affected_entity - affected_entity)
    (repair_authorized ?affected_entity - affected_entity)
    (resource_available ?resource_unit - resource_unit)
    (resource_assigned ?affected_entity - affected_entity ?resource_unit - resource_unit)
    (inspection_available ?inspection_activity - inspection_activity)
    (inspection_assigned ?affected_entity - affected_entity ?inspection_activity - inspection_activity)
    (team_available ?work_team - work_team)
    (team_assigned ?affected_entity - affected_entity ?work_team - work_team)
    (contingency_available ?contingency_option - contingency_option)
    (contingency_allocated_to_origin ?origin_node - origin_node ?contingency_option - contingency_option)
    (contingency_allocated_to_destination ?destination_node - destination_node ?contingency_option - contingency_option)
    (berth_candidate_for_origin ?origin_node - origin_node ?berth_slot - berth_slot)
    (berth_qualified ?berth_slot - berth_slot)
    (berth_staged ?berth_slot - berth_slot)
    (origin_qualification_confirmed ?origin_node - origin_node)
    (route_candidate_for_destination ?destination_node - destination_node ?route_option - route_option)
    (route_qualified ?route_option - route_option)
    (route_staged ?route_option - route_option)
    (destination_qualification_confirmed ?destination_node - destination_node)
    (schedule_pending ?repair_schedule - repair_schedule)
    (schedule_reserved ?repair_schedule - repair_schedule)
    (schedule_linked_to_berth ?repair_schedule - repair_schedule ?berth_slot - berth_slot)
    (schedule_linked_to_route ?repair_schedule - repair_schedule ?route_option - route_option)
    (schedule_flag_berth_restriction ?repair_schedule - repair_schedule)
    (schedule_flag_route_restriction ?repair_schedule - repair_schedule)
    (schedule_confirmed ?repair_schedule - repair_schedule)
    (case_linked_to_origin ?recovery_case - recovery_case ?origin_node - origin_node)
    (case_linked_to_destination ?recovery_case - recovery_case ?destination_node - destination_node)
    (case_linked_to_schedule ?recovery_case - recovery_case ?repair_schedule - repair_schedule)
    (inventory_in_stock ?inventory_item - inventory_item)
    (item_staged_for_case ?recovery_case - recovery_case ?inventory_item - inventory_item)
    (inventory_allocated ?inventory_item - inventory_item)
    (inventory_assigned_to_schedule ?inventory_item - inventory_item ?repair_schedule - repair_schedule)
    (case_prepared_for_staging ?recovery_case - recovery_case)
    (case_staged ?recovery_case - recovery_case)
    (case_quality_checked ?recovery_case - recovery_case)
    (authorization_granted ?recovery_case - recovery_case)
    (authorization_confirmed ?recovery_case - recovery_case)
    (spares_staged ?recovery_case - recovery_case)
    (repair_sequence_finalized ?recovery_case - recovery_case)
    (external_provider_available ?external_provider - external_provider)
    (provider_assigned_to_case ?recovery_case - recovery_case ?external_provider - external_provider)
    (external_provider_linked ?recovery_case - recovery_case)
    (contractor_assigned ?recovery_case - recovery_case)
    (contractor_deployed ?recovery_case - recovery_case)
    (authorization_token_available ?authorization_token - authorization_token)
    (authorization_linked_to_case ?recovery_case - recovery_case ?authorization_token - authorization_token)
    (third_party_service_available ?third_party_service - third_party_service)
    (third_party_service_linked ?recovery_case - recovery_case ?third_party_service - third_party_service)
    (spare_part_available ?spare_part - spare_part)
    (spare_part_allocated ?recovery_case - recovery_case ?spare_part - spare_part)
    (contractor_resource_available ?contractor_resource - contractor_resource)
    (contractor_assigned_to_case ?recovery_case - recovery_case ?contractor_resource - contractor_resource)
    (claim_record_available ?claim_record - claim_record)
    (claim_linked_to_unit ?affected_entity - affected_entity ?claim_record - claim_record)
    (origin_ready_for_consolidation ?origin_node - origin_node)
    (destination_ready_for_consolidation ?destination_node - destination_node)
    (case_execution_authorized ?recovery_case - recovery_case)
  )
  (:action report_dock_congestion_exception
    :parameters (?affected_entity - affected_entity)
    :precondition
      (and
        (not
          (exception_reported ?affected_entity)
        )
        (not
          (service_restored ?affected_entity)
        )
      )
    :effect (exception_reported ?affected_entity)
  )
  (:action reserve_handling_resource_for_triage
    :parameters (?affected_entity - affected_entity ?resource_unit - resource_unit)
    :precondition
      (and
        (exception_reported ?affected_entity)
        (not
          (exception_triaged ?affected_entity)
        )
        (resource_available ?resource_unit)
      )
    :effect
      (and
        (exception_triaged ?affected_entity)
        (resource_assigned ?affected_entity ?resource_unit)
        (not
          (resource_available ?resource_unit)
        )
      )
  )
  (:action allocate_inspection_activity_to_unit
    :parameters (?affected_entity - affected_entity ?inspection_activity - inspection_activity)
    :precondition
      (and
        (exception_reported ?affected_entity)
        (exception_triaged ?affected_entity)
        (inspection_available ?inspection_activity)
      )
    :effect
      (and
        (inspection_assigned ?affected_entity ?inspection_activity)
        (not
          (inspection_available ?inspection_activity)
        )
      )
  )
  (:action validate_exception_after_inspection
    :parameters (?affected_entity - affected_entity ?inspection_activity - inspection_activity)
    :precondition
      (and
        (exception_reported ?affected_entity)
        (exception_triaged ?affected_entity)
        (inspection_assigned ?affected_entity ?inspection_activity)
        (not
          (exception_validated ?affected_entity)
        )
      )
    :effect (exception_validated ?affected_entity)
  )
  (:action release_inspection_activity
    :parameters (?affected_entity - affected_entity ?inspection_activity - inspection_activity)
    :precondition
      (and
        (inspection_assigned ?affected_entity ?inspection_activity)
      )
    :effect
      (and
        (inspection_available ?inspection_activity)
        (not
          (inspection_assigned ?affected_entity ?inspection_activity)
        )
      )
  )
  (:action assign_work_team_to_validated_exception
    :parameters (?affected_entity - affected_entity ?work_team - work_team)
    :precondition
      (and
        (exception_validated ?affected_entity)
        (team_available ?work_team)
      )
    :effect
      (and
        (team_assigned ?affected_entity ?work_team)
        (not
          (team_available ?work_team)
        )
      )
  )
  (:action release_work_team
    :parameters (?affected_entity - affected_entity ?work_team - work_team)
    :precondition
      (and
        (team_assigned ?affected_entity ?work_team)
      )
    :effect
      (and
        (team_available ?work_team)
        (not
          (team_assigned ?affected_entity ?work_team)
        )
      )
  )
  (:action allocate_spare_part_to_recovery_case
    :parameters (?recovery_case - recovery_case ?spare_part - spare_part)
    :precondition
      (and
        (exception_validated ?recovery_case)
        (spare_part_available ?spare_part)
      )
    :effect
      (and
        (spare_part_allocated ?recovery_case ?spare_part)
        (not
          (spare_part_available ?spare_part)
        )
      )
  )
  (:action release_spare_part_from_recovery_case
    :parameters (?recovery_case - recovery_case ?spare_part - spare_part)
    :precondition
      (and
        (spare_part_allocated ?recovery_case ?spare_part)
      )
    :effect
      (and
        (spare_part_available ?spare_part)
        (not
          (spare_part_allocated ?recovery_case ?spare_part)
        )
      )
  )
  (:action assign_contractor_resource_to_case
    :parameters (?recovery_case - recovery_case ?contractor_resource - contractor_resource)
    :precondition
      (and
        (exception_validated ?recovery_case)
        (contractor_resource_available ?contractor_resource)
      )
    :effect
      (and
        (contractor_assigned_to_case ?recovery_case ?contractor_resource)
        (not
          (contractor_resource_available ?contractor_resource)
        )
      )
  )
  (:action release_contractor_from_case
    :parameters (?recovery_case - recovery_case ?contractor_resource - contractor_resource)
    :precondition
      (and
        (contractor_assigned_to_case ?recovery_case ?contractor_resource)
      )
    :effect
      (and
        (contractor_resource_available ?contractor_resource)
        (not
          (contractor_assigned_to_case ?recovery_case ?contractor_resource)
        )
      )
  )
  (:action qualify_berth_slot_for_origin
    :parameters (?origin_node - origin_node ?berth_slot - berth_slot ?inspection_activity - inspection_activity)
    :precondition
      (and
        (exception_validated ?origin_node)
        (inspection_assigned ?origin_node ?inspection_activity)
        (berth_candidate_for_origin ?origin_node ?berth_slot)
        (not
          (berth_qualified ?berth_slot)
        )
        (not
          (berth_staged ?berth_slot)
        )
      )
    :effect (berth_qualified ?berth_slot)
  )
  (:action confirm_origin_berth_qualification
    :parameters (?origin_node - origin_node ?berth_slot - berth_slot ?work_team - work_team)
    :precondition
      (and
        (exception_validated ?origin_node)
        (team_assigned ?origin_node ?work_team)
        (berth_candidate_for_origin ?origin_node ?berth_slot)
        (berth_qualified ?berth_slot)
        (not
          (origin_ready_for_consolidation ?origin_node)
        )
      )
    :effect
      (and
        (origin_ready_for_consolidation ?origin_node)
        (origin_qualification_confirmed ?origin_node)
      )
  )
  (:action stage_berth_with_contingency
    :parameters (?origin_node - origin_node ?berth_slot - berth_slot ?contingency_option - contingency_option)
    :precondition
      (and
        (exception_validated ?origin_node)
        (berth_candidate_for_origin ?origin_node ?berth_slot)
        (contingency_available ?contingency_option)
        (not
          (origin_ready_for_consolidation ?origin_node)
        )
      )
    :effect
      (and
        (berth_staged ?berth_slot)
        (origin_ready_for_consolidation ?origin_node)
        (contingency_allocated_to_origin ?origin_node ?contingency_option)
        (not
          (contingency_available ?contingency_option)
        )
      )
  )
  (:action activate_berth_after_inspection
    :parameters (?origin_node - origin_node ?berth_slot - berth_slot ?inspection_activity - inspection_activity ?contingency_option - contingency_option)
    :precondition
      (and
        (exception_validated ?origin_node)
        (inspection_assigned ?origin_node ?inspection_activity)
        (berth_candidate_for_origin ?origin_node ?berth_slot)
        (berth_staged ?berth_slot)
        (contingency_allocated_to_origin ?origin_node ?contingency_option)
        (not
          (origin_qualification_confirmed ?origin_node)
        )
      )
    :effect
      (and
        (berth_qualified ?berth_slot)
        (origin_qualification_confirmed ?origin_node)
        (contingency_available ?contingency_option)
        (not
          (contingency_allocated_to_origin ?origin_node ?contingency_option)
        )
      )
  )
  (:action qualify_route_option_for_destination
    :parameters (?destination_node - destination_node ?route_option - route_option ?inspection_activity - inspection_activity)
    :precondition
      (and
        (exception_validated ?destination_node)
        (inspection_assigned ?destination_node ?inspection_activity)
        (route_candidate_for_destination ?destination_node ?route_option)
        (not
          (route_qualified ?route_option)
        )
        (not
          (route_staged ?route_option)
        )
      )
    :effect (route_qualified ?route_option)
  )
  (:action confirm_destination_route_with_team
    :parameters (?destination_node - destination_node ?route_option - route_option ?work_team - work_team)
    :precondition
      (and
        (exception_validated ?destination_node)
        (team_assigned ?destination_node ?work_team)
        (route_candidate_for_destination ?destination_node ?route_option)
        (route_qualified ?route_option)
        (not
          (destination_ready_for_consolidation ?destination_node)
        )
      )
    :effect
      (and
        (destination_ready_for_consolidation ?destination_node)
        (destination_qualification_confirmed ?destination_node)
      )
  )
  (:action stage_route_with_contingency
    :parameters (?destination_node - destination_node ?route_option - route_option ?contingency_option - contingency_option)
    :precondition
      (and
        (exception_validated ?destination_node)
        (route_candidate_for_destination ?destination_node ?route_option)
        (contingency_available ?contingency_option)
        (not
          (destination_ready_for_consolidation ?destination_node)
        )
      )
    :effect
      (and
        (route_staged ?route_option)
        (destination_ready_for_consolidation ?destination_node)
        (contingency_allocated_to_destination ?destination_node ?contingency_option)
        (not
          (contingency_available ?contingency_option)
        )
      )
  )
  (:action activate_route_after_inspection
    :parameters (?destination_node - destination_node ?route_option - route_option ?inspection_activity - inspection_activity ?contingency_option - contingency_option)
    :precondition
      (and
        (exception_validated ?destination_node)
        (inspection_assigned ?destination_node ?inspection_activity)
        (route_candidate_for_destination ?destination_node ?route_option)
        (route_staged ?route_option)
        (contingency_allocated_to_destination ?destination_node ?contingency_option)
        (not
          (destination_qualification_confirmed ?destination_node)
        )
      )
    :effect
      (and
        (route_qualified ?route_option)
        (destination_qualification_confirmed ?destination_node)
        (contingency_available ?contingency_option)
        (not
          (contingency_allocated_to_destination ?destination_node ?contingency_option)
        )
      )
  )
  (:action consolidate_and_reserve_repair_schedule
    :parameters (?origin_node - origin_node ?destination_node - destination_node ?berth_slot - berth_slot ?route_option - route_option ?repair_schedule - repair_schedule)
    :precondition
      (and
        (origin_ready_for_consolidation ?origin_node)
        (destination_ready_for_consolidation ?destination_node)
        (berth_candidate_for_origin ?origin_node ?berth_slot)
        (route_candidate_for_destination ?destination_node ?route_option)
        (berth_qualified ?berth_slot)
        (route_qualified ?route_option)
        (origin_qualification_confirmed ?origin_node)
        (destination_qualification_confirmed ?destination_node)
        (schedule_pending ?repair_schedule)
      )
    :effect
      (and
        (schedule_reserved ?repair_schedule)
        (schedule_linked_to_berth ?repair_schedule ?berth_slot)
        (schedule_linked_to_route ?repair_schedule ?route_option)
        (not
          (schedule_pending ?repair_schedule)
        )
      )
  )
  (:action consolidate_and_reserve_with_staged_berth
    :parameters (?origin_node - origin_node ?destination_node - destination_node ?berth_slot - berth_slot ?route_option - route_option ?repair_schedule - repair_schedule)
    :precondition
      (and
        (origin_ready_for_consolidation ?origin_node)
        (destination_ready_for_consolidation ?destination_node)
        (berth_candidate_for_origin ?origin_node ?berth_slot)
        (route_candidate_for_destination ?destination_node ?route_option)
        (berth_staged ?berth_slot)
        (route_qualified ?route_option)
        (not
          (origin_qualification_confirmed ?origin_node)
        )
        (destination_qualification_confirmed ?destination_node)
        (schedule_pending ?repair_schedule)
      )
    :effect
      (and
        (schedule_reserved ?repair_schedule)
        (schedule_linked_to_berth ?repair_schedule ?berth_slot)
        (schedule_linked_to_route ?repair_schedule ?route_option)
        (schedule_flag_berth_restriction ?repair_schedule)
        (not
          (schedule_pending ?repair_schedule)
        )
      )
  )
  (:action consolidate_and_reserve_with_qualified_berth
    :parameters (?origin_node - origin_node ?destination_node - destination_node ?berth_slot - berth_slot ?route_option - route_option ?repair_schedule - repair_schedule)
    :precondition
      (and
        (origin_ready_for_consolidation ?origin_node)
        (destination_ready_for_consolidation ?destination_node)
        (berth_candidate_for_origin ?origin_node ?berth_slot)
        (route_candidate_for_destination ?destination_node ?route_option)
        (berth_qualified ?berth_slot)
        (route_staged ?route_option)
        (origin_qualification_confirmed ?origin_node)
        (not
          (destination_qualification_confirmed ?destination_node)
        )
        (schedule_pending ?repair_schedule)
      )
    :effect
      (and
        (schedule_reserved ?repair_schedule)
        (schedule_linked_to_berth ?repair_schedule ?berth_slot)
        (schedule_linked_to_route ?repair_schedule ?route_option)
        (schedule_flag_route_restriction ?repair_schedule)
        (not
          (schedule_pending ?repair_schedule)
        )
      )
  )
  (:action consolidate_and_reserve_with_staged_berth_and_route
    :parameters (?origin_node - origin_node ?destination_node - destination_node ?berth_slot - berth_slot ?route_option - route_option ?repair_schedule - repair_schedule)
    :precondition
      (and
        (origin_ready_for_consolidation ?origin_node)
        (destination_ready_for_consolidation ?destination_node)
        (berth_candidate_for_origin ?origin_node ?berth_slot)
        (route_candidate_for_destination ?destination_node ?route_option)
        (berth_staged ?berth_slot)
        (route_staged ?route_option)
        (not
          (origin_qualification_confirmed ?origin_node)
        )
        (not
          (destination_qualification_confirmed ?destination_node)
        )
        (schedule_pending ?repair_schedule)
      )
    :effect
      (and
        (schedule_reserved ?repair_schedule)
        (schedule_linked_to_berth ?repair_schedule ?berth_slot)
        (schedule_linked_to_route ?repair_schedule ?route_option)
        (schedule_flag_berth_restriction ?repair_schedule)
        (schedule_flag_route_restriction ?repair_schedule)
        (not
          (schedule_pending ?repair_schedule)
        )
      )
  )
  (:action confirm_repair_schedule_at_origin
    :parameters (?repair_schedule - repair_schedule ?origin_node - origin_node ?inspection_activity - inspection_activity)
    :precondition
      (and
        (schedule_reserved ?repair_schedule)
        (origin_ready_for_consolidation ?origin_node)
        (inspection_assigned ?origin_node ?inspection_activity)
        (not
          (schedule_confirmed ?repair_schedule)
        )
      )
    :effect (schedule_confirmed ?repair_schedule)
  )
  (:action stage_inventory_item_for_schedule
    :parameters (?recovery_case - recovery_case ?inventory_item - inventory_item ?repair_schedule - repair_schedule)
    :precondition
      (and
        (exception_validated ?recovery_case)
        (case_linked_to_schedule ?recovery_case ?repair_schedule)
        (item_staged_for_case ?recovery_case ?inventory_item)
        (inventory_in_stock ?inventory_item)
        (schedule_reserved ?repair_schedule)
        (schedule_confirmed ?repair_schedule)
        (not
          (inventory_allocated ?inventory_item)
        )
      )
    :effect
      (and
        (inventory_allocated ?inventory_item)
        (inventory_assigned_to_schedule ?inventory_item ?repair_schedule)
        (not
          (inventory_in_stock ?inventory_item)
        )
      )
  )
  (:action mark_case_prepared_for_staging
    :parameters (?recovery_case - recovery_case ?inventory_item - inventory_item ?repair_schedule - repair_schedule ?inspection_activity - inspection_activity)
    :precondition
      (and
        (exception_validated ?recovery_case)
        (item_staged_for_case ?recovery_case ?inventory_item)
        (inventory_allocated ?inventory_item)
        (inventory_assigned_to_schedule ?inventory_item ?repair_schedule)
        (inspection_assigned ?recovery_case ?inspection_activity)
        (not
          (schedule_flag_berth_restriction ?repair_schedule)
        )
        (not
          (case_prepared_for_staging ?recovery_case)
        )
      )
    :effect (case_prepared_for_staging ?recovery_case)
  )
  (:action assign_authorization_token_to_case
    :parameters (?recovery_case - recovery_case ?authorization_token - authorization_token)
    :precondition
      (and
        (exception_validated ?recovery_case)
        (authorization_token_available ?authorization_token)
        (not
          (authorization_granted ?recovery_case)
        )
      )
    :effect
      (and
        (authorization_granted ?recovery_case)
        (authorization_linked_to_case ?recovery_case ?authorization_token)
        (not
          (authorization_token_available ?authorization_token)
        )
      )
  )
  (:action confirm_authorization_and_prepare_case
    :parameters (?recovery_case - recovery_case ?inventory_item - inventory_item ?repair_schedule - repair_schedule ?inspection_activity - inspection_activity ?authorization_token - authorization_token)
    :precondition
      (and
        (exception_validated ?recovery_case)
        (item_staged_for_case ?recovery_case ?inventory_item)
        (inventory_allocated ?inventory_item)
        (inventory_assigned_to_schedule ?inventory_item ?repair_schedule)
        (inspection_assigned ?recovery_case ?inspection_activity)
        (schedule_flag_berth_restriction ?repair_schedule)
        (authorization_granted ?recovery_case)
        (authorization_linked_to_case ?recovery_case ?authorization_token)
        (not
          (case_prepared_for_staging ?recovery_case)
        )
      )
    :effect
      (and
        (case_prepared_for_staging ?recovery_case)
        (authorization_confirmed ?recovery_case)
      )
  )
  (:action assign_team_and_stage_case
    :parameters (?recovery_case - recovery_case ?spare_part - spare_part ?work_team - work_team ?inventory_item - inventory_item ?repair_schedule - repair_schedule)
    :precondition
      (and
        (case_prepared_for_staging ?recovery_case)
        (spare_part_allocated ?recovery_case ?spare_part)
        (team_assigned ?recovery_case ?work_team)
        (item_staged_for_case ?recovery_case ?inventory_item)
        (inventory_assigned_to_schedule ?inventory_item ?repair_schedule)
        (not
          (schedule_flag_route_restriction ?repair_schedule)
        )
        (not
          (case_staged ?recovery_case)
        )
      )
    :effect (case_staged ?recovery_case)
  )
  (:action assign_team_and_stage_case_alternative
    :parameters (?recovery_case - recovery_case ?spare_part - spare_part ?work_team - work_team ?inventory_item - inventory_item ?repair_schedule - repair_schedule)
    :precondition
      (and
        (case_prepared_for_staging ?recovery_case)
        (spare_part_allocated ?recovery_case ?spare_part)
        (team_assigned ?recovery_case ?work_team)
        (item_staged_for_case ?recovery_case ?inventory_item)
        (inventory_assigned_to_schedule ?inventory_item ?repair_schedule)
        (schedule_flag_route_restriction ?repair_schedule)
        (not
          (case_staged ?recovery_case)
        )
      )
    :effect (case_staged ?recovery_case)
  )
  (:action perform_case_quality_check
    :parameters (?recovery_case - recovery_case ?contractor_resource - contractor_resource ?inventory_item - inventory_item ?repair_schedule - repair_schedule)
