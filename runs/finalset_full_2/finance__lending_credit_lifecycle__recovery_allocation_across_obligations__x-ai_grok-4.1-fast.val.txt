--- domain.pddl (as passed to VAL) ---
(define (domain finance_recovery_allocation_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types operational_token_group - object administrative_token_group - object control_token_group - object domain_root - object case_entity - domain_root recovery_fund_bucket - operational_token_group valuation_model - operational_token_group collateral_record - operational_token_group external_approval - operational_token_group settlement_option - operational_token_group recovery_fund_token - operational_token_group valuation_adjustment - operational_token_group allocation_strategy - operational_token_group entitlement_marker - administrative_token_group collateral_item - administrative_token_group legal_remedy - administrative_token_group recovery_resource - control_token_group resource_share_group - control_token_group allocation_record - control_token_group obligation_category - case_entity case_category - case_entity primary_obligation - obligation_category co_obligation - obligation_category recovery_case - case_category)
  (:predicates
    (entity_registered ?entity - case_entity)
    (entity_configured ?entity - case_entity)
    (entity_fund_assignment_recorded ?entity - case_entity)
    (entity_recovered ?entity - case_entity)
    (entity_ready_for_reservation ?entity - case_entity)
    (entity_reservation_confirmed ?entity - case_entity)
    (fund_bucket_available ?fund_bucket - recovery_fund_bucket)
    (entity_linked_fund_bucket ?entity - case_entity ?fund_bucket - recovery_fund_bucket)
    (valuation_model_available ?valuation_model - valuation_model)
    (entity_valuation_model_attached ?entity - case_entity ?valuation_model - valuation_model)
    (collateral_record_available ?collateral_record - collateral_record)
    (entity_attached_collateral_record ?entity - case_entity ?collateral_record - collateral_record)
    (entitlement_marker_available ?entitlement_marker - entitlement_marker)
    (primary_obligation_entitlement_attached ?primary_obligation - primary_obligation ?entitlement_marker - entitlement_marker)
    (co_obligation_entitlement_attached ?co_obligation - co_obligation ?entitlement_marker - entitlement_marker)
    (primary_obligation_linked_resource ?primary_obligation - primary_obligation ?recovery_resource - recovery_resource)
    (recovery_resource_preselected ?recovery_resource - recovery_resource)
    (recovery_resource_entitlement_locked ?recovery_resource - recovery_resource)
    (primary_obligation_resource_confirmed ?primary_obligation - primary_obligation)
    (co_obligation_linked_share_group ?co_obligation - co_obligation ?resource_share_group - resource_share_group)
    (resource_share_group_preselected ?resource_share_group - resource_share_group)
    (resource_share_group_entitlement_locked ?resource_share_group - resource_share_group)
    (co_obligation_resource_confirmed ?co_obligation - co_obligation)
    (allocation_record_available ?allocation_record - allocation_record)
    (allocation_record_prepared ?allocation_record - allocation_record)
    (allocation_record_includes_resource ?allocation_record - allocation_record ?recovery_resource - recovery_resource)
    (allocation_record_includes_share_group ?allocation_record - allocation_record ?resource_share_group - resource_share_group)
    (allocation_record_primary_flag ?allocation_record - allocation_record)
    (allocation_record_secondary_flag ?allocation_record - allocation_record)
    (allocation_record_valuation_validated ?allocation_record - allocation_record)
    (case_has_primary_obligation ?recovery_case - recovery_case ?primary_obligation - primary_obligation)
    (case_has_co_obligation ?recovery_case - recovery_case ?co_obligation - co_obligation)
    (case_has_allocation_record ?recovery_case - recovery_case ?allocation_record - allocation_record)
    (collateral_item_available ?collateral_item - collateral_item)
    (case_has_collateral_item ?recovery_case - recovery_case ?collateral_item - collateral_item)
    (collateral_item_consumed ?collateral_item - collateral_item)
    (collateral_item_allocated_to_record ?collateral_item - collateral_item ?allocation_record - allocation_record)
    (case_collateral_validated ?recovery_case - recovery_case)
    (case_strategy_staged ?recovery_case - recovery_case)
    (case_strategy_applied ?recovery_case - recovery_case)
    (case_external_approval_attached ?recovery_case - recovery_case)
    (case_external_approval_verified ?recovery_case - recovery_case)
    (case_ready_for_settlement_option ?recovery_case - recovery_case)
    (case_ready_for_finalization ?recovery_case - recovery_case)
    (legal_remedy_available ?legal_remedy - legal_remedy)
    (case_has_legal_remedy ?recovery_case - recovery_case ?legal_remedy - legal_remedy)
    (case_legal_remedy_applied ?recovery_case - recovery_case)
    (case_legal_remedy_verified ?recovery_case - recovery_case)
    (case_legal_remedy_executed ?recovery_case - recovery_case)
    (external_approval_available ?external_approval - external_approval)
    (case_has_external_approval ?recovery_case - recovery_case ?external_approval - external_approval)
    (settlement_option_available ?settlement_option - settlement_option)
    (case_settlement_option_attached ?recovery_case - recovery_case ?settlement_option - settlement_option)
    (valuation_adjustment_available ?valuation_adjustment - valuation_adjustment)
    (case_valuation_adjustment_attached ?recovery_case - recovery_case ?valuation_adjustment - valuation_adjustment)
    (allocation_strategy_available ?allocation_strategy - allocation_strategy)
    (case_allocation_strategy_attached ?recovery_case - recovery_case ?allocation_strategy - allocation_strategy)
    (recovery_fund_token_available ?recovery_fund_token - recovery_fund_token)
    (entity_bound_fund_token ?entity - case_entity ?recovery_fund_token - recovery_fund_token)
    (primary_obligation_ready ?primary_obligation - primary_obligation)
    (co_obligation_ready ?co_obligation - co_obligation)
    (case_closed ?recovery_case - recovery_case)
  )
  (:action register_case_entity
    :parameters (?entity - case_entity)
    :precondition
      (and
        (not
          (entity_registered ?entity)
        )
        (not
          (entity_recovered ?entity)
        )
      )
    :effect (entity_registered ?entity)
  )
  (:action assign_fund_bucket_to_entity
    :parameters (?entity - case_entity ?fund_bucket - recovery_fund_bucket)
    :precondition
      (and
        (entity_registered ?entity)
        (not
          (entity_fund_assignment_recorded ?entity)
        )
        (fund_bucket_available ?fund_bucket)
      )
    :effect
      (and
        (entity_fund_assignment_recorded ?entity)
        (entity_linked_fund_bucket ?entity ?fund_bucket)
        (not
          (fund_bucket_available ?fund_bucket)
        )
      )
  )
  (:action attach_valuation_model_to_entity
    :parameters (?entity - case_entity ?valuation_model - valuation_model)
    :precondition
      (and
        (entity_registered ?entity)
        (entity_fund_assignment_recorded ?entity)
        (valuation_model_available ?valuation_model)
      )
    :effect
      (and
        (entity_valuation_model_attached ?entity ?valuation_model)
        (not
          (valuation_model_available ?valuation_model)
        )
      )
  )
  (:action complete_entity_configuration
    :parameters (?entity - case_entity ?valuation_model - valuation_model)
    :precondition
      (and
        (entity_registered ?entity)
        (entity_fund_assignment_recorded ?entity)
        (entity_valuation_model_attached ?entity ?valuation_model)
        (not
          (entity_configured ?entity)
        )
      )
    :effect (entity_configured ?entity)
  )
  (:action release_valuation_model
    :parameters (?entity - case_entity ?valuation_model - valuation_model)
    :precondition
      (and
        (entity_valuation_model_attached ?entity ?valuation_model)
      )
    :effect
      (and
        (valuation_model_available ?valuation_model)
        (not
          (entity_valuation_model_attached ?entity ?valuation_model)
        )
      )
  )
  (:action attach_collateral_record_to_entity
    :parameters (?entity - case_entity ?collateral_record - collateral_record)
    :precondition
      (and
        (entity_configured ?entity)
        (collateral_record_available ?collateral_record)
      )
    :effect
      (and
        (entity_attached_collateral_record ?entity ?collateral_record)
        (not
          (collateral_record_available ?collateral_record)
        )
      )
  )
  (:action detach_collateral_record_from_entity
    :parameters (?entity - case_entity ?collateral_record - collateral_record)
    :precondition
      (and
        (entity_attached_collateral_record ?entity ?collateral_record)
      )
    :effect
      (and
        (collateral_record_available ?collateral_record)
        (not
          (entity_attached_collateral_record ?entity ?collateral_record)
        )
      )
  )
  (:action attach_valuation_adjustment_to_case
    :parameters (?recovery_case - recovery_case ?valuation_adjustment - valuation_adjustment)
    :precondition
      (and
        (entity_configured ?recovery_case)
        (valuation_adjustment_available ?valuation_adjustment)
      )
    :effect
      (and
        (case_valuation_adjustment_attached ?recovery_case ?valuation_adjustment)
        (not
          (valuation_adjustment_available ?valuation_adjustment)
        )
      )
  )
  (:action detach_valuation_adjustment_from_case
    :parameters (?recovery_case - recovery_case ?valuation_adjustment - valuation_adjustment)
    :precondition
      (and
        (case_valuation_adjustment_attached ?recovery_case ?valuation_adjustment)
      )
    :effect
      (and
        (valuation_adjustment_available ?valuation_adjustment)
        (not
          (case_valuation_adjustment_attached ?recovery_case ?valuation_adjustment)
        )
      )
  )
  (:action attach_allocation_strategy_to_case
    :parameters (?recovery_case - recovery_case ?allocation_strategy - allocation_strategy)
    :precondition
      (and
        (entity_configured ?recovery_case)
        (allocation_strategy_available ?allocation_strategy)
      )
    :effect
      (and
        (case_allocation_strategy_attached ?recovery_case ?allocation_strategy)
        (not
          (allocation_strategy_available ?allocation_strategy)
        )
      )
  )
  (:action detach_allocation_strategy_from_case
    :parameters (?recovery_case - recovery_case ?allocation_strategy - allocation_strategy)
    :precondition
      (and
        (case_allocation_strategy_attached ?recovery_case ?allocation_strategy)
      )
    :effect
      (and
        (allocation_strategy_available ?allocation_strategy)
        (not
          (case_allocation_strategy_attached ?recovery_case ?allocation_strategy)
        )
      )
  )
  (:action preselect_recovery_resource_for_primary_obligation
    :parameters (?primary_obligation - primary_obligation ?recovery_resource - recovery_resource ?valuation_model - valuation_model)
    :precondition
      (and
        (entity_configured ?primary_obligation)
        (entity_valuation_model_attached ?primary_obligation ?valuation_model)
        (primary_obligation_linked_resource ?primary_obligation ?recovery_resource)
        (not
          (recovery_resource_preselected ?recovery_resource)
        )
        (not
          (recovery_resource_entitlement_locked ?recovery_resource)
        )
      )
    :effect (recovery_resource_preselected ?recovery_resource)
  )
  (:action confirm_primary_obligation_resource_with_collateral
    :parameters (?primary_obligation - primary_obligation ?recovery_resource - recovery_resource ?collateral_record - collateral_record)
    :precondition
      (and
        (entity_configured ?primary_obligation)
        (entity_attached_collateral_record ?primary_obligation ?collateral_record)
        (primary_obligation_linked_resource ?primary_obligation ?recovery_resource)
        (recovery_resource_preselected ?recovery_resource)
        (not
          (primary_obligation_ready ?primary_obligation)
        )
      )
    :effect
      (and
        (primary_obligation_ready ?primary_obligation)
        (primary_obligation_resource_confirmed ?primary_obligation)
      )
  )
  (:action attach_entitlement_to_primary_obligation_resource
    :parameters (?primary_obligation - primary_obligation ?recovery_resource - recovery_resource ?entitlement_marker - entitlement_marker)
    :precondition
      (and
        (entity_configured ?primary_obligation)
        (primary_obligation_linked_resource ?primary_obligation ?recovery_resource)
        (entitlement_marker_available ?entitlement_marker)
        (not
          (primary_obligation_ready ?primary_obligation)
        )
      )
    :effect
      (and
        (recovery_resource_entitlement_locked ?recovery_resource)
        (primary_obligation_ready ?primary_obligation)
        (primary_obligation_entitlement_attached ?primary_obligation ?entitlement_marker)
        (not
          (entitlement_marker_available ?entitlement_marker)
        )
      )
  )
  (:action reconcile_primary_obligation_resource_selection
    :parameters (?primary_obligation - primary_obligation ?recovery_resource - recovery_resource ?valuation_model - valuation_model ?entitlement_marker - entitlement_marker)
    :precondition
      (and
        (entity_configured ?primary_obligation)
        (entity_valuation_model_attached ?primary_obligation ?valuation_model)
        (primary_obligation_linked_resource ?primary_obligation ?recovery_resource)
        (recovery_resource_entitlement_locked ?recovery_resource)
        (primary_obligation_entitlement_attached ?primary_obligation ?entitlement_marker)
        (not
          (primary_obligation_resource_confirmed ?primary_obligation)
        )
      )
    :effect
      (and
        (recovery_resource_preselected ?recovery_resource)
        (primary_obligation_resource_confirmed ?primary_obligation)
        (entitlement_marker_available ?entitlement_marker)
        (not
          (primary_obligation_entitlement_attached ?primary_obligation ?entitlement_marker)
        )
      )
  )
  (:action preselect_share_group_for_co_obligation
    :parameters (?co_obligation - co_obligation ?resource_share_group - resource_share_group ?valuation_model - valuation_model)
    :precondition
      (and
        (entity_configured ?co_obligation)
        (entity_valuation_model_attached ?co_obligation ?valuation_model)
        (co_obligation_linked_share_group ?co_obligation ?resource_share_group)
        (not
          (resource_share_group_preselected ?resource_share_group)
        )
        (not
          (resource_share_group_entitlement_locked ?resource_share_group)
        )
      )
    :effect (resource_share_group_preselected ?resource_share_group)
  )
  (:action confirm_co_obligation_share_group_with_collateral
    :parameters (?co_obligation - co_obligation ?resource_share_group - resource_share_group ?collateral_record - collateral_record)
    :precondition
      (and
        (entity_configured ?co_obligation)
        (entity_attached_collateral_record ?co_obligation ?collateral_record)
        (co_obligation_linked_share_group ?co_obligation ?resource_share_group)
        (resource_share_group_preselected ?resource_share_group)
        (not
          (co_obligation_ready ?co_obligation)
        )
      )
    :effect
      (and
        (co_obligation_ready ?co_obligation)
        (co_obligation_resource_confirmed ?co_obligation)
      )
  )
  (:action attach_entitlement_to_co_obligation_share_group
    :parameters (?co_obligation - co_obligation ?resource_share_group - resource_share_group ?entitlement_marker - entitlement_marker)
    :precondition
      (and
        (entity_configured ?co_obligation)
        (co_obligation_linked_share_group ?co_obligation ?resource_share_group)
        (entitlement_marker_available ?entitlement_marker)
        (not
          (co_obligation_ready ?co_obligation)
        )
      )
    :effect
      (and
        (resource_share_group_entitlement_locked ?resource_share_group)
        (co_obligation_ready ?co_obligation)
        (co_obligation_entitlement_attached ?co_obligation ?entitlement_marker)
        (not
          (entitlement_marker_available ?entitlement_marker)
        )
      )
  )
  (:action reconcile_co_obligation_share_group_selection
    :parameters (?co_obligation - co_obligation ?resource_share_group - resource_share_group ?valuation_model - valuation_model ?entitlement_marker - entitlement_marker)
    :precondition
      (and
        (entity_configured ?co_obligation)
        (entity_valuation_model_attached ?co_obligation ?valuation_model)
        (co_obligation_linked_share_group ?co_obligation ?resource_share_group)
        (resource_share_group_entitlement_locked ?resource_share_group)
        (co_obligation_entitlement_attached ?co_obligation ?entitlement_marker)
        (not
          (co_obligation_resource_confirmed ?co_obligation)
        )
      )
    :effect
      (and
        (resource_share_group_preselected ?resource_share_group)
        (co_obligation_resource_confirmed ?co_obligation)
        (entitlement_marker_available ?entitlement_marker)
        (not
          (co_obligation_entitlement_attached ?co_obligation ?entitlement_marker)
        )
      )
  )
  (:action create_allocation_record_basic
    :parameters (?primary_obligation - primary_obligation ?co_obligation - co_obligation ?recovery_resource - recovery_resource ?resource_share_group - resource_share_group ?allocation_record - allocation_record)
    :precondition
      (and
        (primary_obligation_ready ?primary_obligation)
        (co_obligation_ready ?co_obligation)
        (primary_obligation_linked_resource ?primary_obligation ?recovery_resource)
        (co_obligation_linked_share_group ?co_obligation ?resource_share_group)
        (recovery_resource_preselected ?recovery_resource)
        (resource_share_group_preselected ?resource_share_group)
        (primary_obligation_resource_confirmed ?primary_obligation)
        (co_obligation_resource_confirmed ?co_obligation)
        (allocation_record_available ?allocation_record)
      )
    :effect
      (and
        (allocation_record_prepared ?allocation_record)
        (allocation_record_includes_resource ?allocation_record ?recovery_resource)
        (allocation_record_includes_share_group ?allocation_record ?resource_share_group)
        (not
          (allocation_record_available ?allocation_record)
        )
      )
  )
  (:action create_allocation_record_with_primary_indicator
    :parameters (?primary_obligation - primary_obligation ?co_obligation - co_obligation ?recovery_resource - recovery_resource ?resource_share_group - resource_share_group ?allocation_record - allocation_record)
    :precondition
      (and
        (primary_obligation_ready ?primary_obligation)
        (co_obligation_ready ?co_obligation)
        (primary_obligation_linked_resource ?primary_obligation ?recovery_resource)
        (co_obligation_linked_share_group ?co_obligation ?resource_share_group)
        (recovery_resource_entitlement_locked ?recovery_resource)
        (resource_share_group_preselected ?resource_share_group)
        (not
          (primary_obligation_resource_confirmed ?primary_obligation)
        )
        (co_obligation_resource_confirmed ?co_obligation)
        (allocation_record_available ?allocation_record)
      )
    :effect
      (and
        (allocation_record_prepared ?allocation_record)
        (allocation_record_includes_resource ?allocation_record ?recovery_resource)
        (allocation_record_includes_share_group ?allocation_record ?resource_share_group)
        (allocation_record_primary_flag ?allocation_record)
        (not
          (allocation_record_available ?allocation_record)
        )
      )
  )
  (:action create_allocation_record_with_co_indicator
    :parameters (?primary_obligation - primary_obligation ?co_obligation - co_obligation ?recovery_resource - recovery_resource ?resource_share_group - resource_share_group ?allocation_record - allocation_record)
    :precondition
      (and
        (primary_obligation_ready ?primary_obligation)
        (co_obligation_ready ?co_obligation)
        (primary_obligation_linked_resource ?primary_obligation ?recovery_resource)
        (co_obligation_linked_share_group ?co_obligation ?resource_share_group)
        (recovery_resource_preselected ?recovery_resource)
        (resource_share_group_entitlement_locked ?resource_share_group)
        (primary_obligation_resource_confirmed ?primary_obligation)
        (not
          (co_obligation_resource_confirmed ?co_obligation)
        )
        (allocation_record_available ?allocation_record)
      )
    :effect
      (and
        (allocation_record_prepared ?allocation_record)
        (allocation_record_includes_resource ?allocation_record ?recovery_resource)
        (allocation_record_includes_share_group ?allocation_record ?resource_share_group)
        (allocation_record_secondary_flag ?allocation_record)
        (not
          (allocation_record_available ?allocation_record)
        )
      )
  )
  (:action create_allocation_record_with_both_indicators
    :parameters (?primary_obligation - primary_obligation ?co_obligation - co_obligation ?recovery_resource - recovery_resource ?resource_share_group - resource_share_group ?allocation_record - allocation_record)
    :precondition
      (and
        (primary_obligation_ready ?primary_obligation)
        (co_obligation_ready ?co_obligation)
        (primary_obligation_linked_resource ?primary_obligation ?recovery_resource)
        (co_obligation_linked_share_group ?co_obligation ?resource_share_group)
        (recovery_resource_entitlement_locked ?recovery_resource)
        (resource_share_group_entitlement_locked ?resource_share_group)
        (not
          (primary_obligation_resource_confirmed ?primary_obligation)
        )
        (not
          (co_obligation_resource_confirmed ?co_obligation)
        )
        (allocation_record_available ?allocation_record)
      )
    :effect
      (and
        (allocation_record_prepared ?allocation_record)
        (allocation_record_includes_resource ?allocation_record ?recovery_resource)
        (allocation_record_includes_share_group ?allocation_record ?resource_share_group)
        (allocation_record_primary_flag ?allocation_record)
        (allocation_record_secondary_flag ?allocation_record)
        (not
          (allocation_record_available ?allocation_record)
        )
      )
  )
  (:action validate_allocation_record_valuation
    :parameters (?allocation_record - allocation_record ?primary_obligation - primary_obligation ?valuation_model - valuation_model)
    :precondition
      (and
        (allocation_record_prepared ?allocation_record)
        (primary_obligation_ready ?primary_obligation)
        (entity_valuation_model_attached ?primary_obligation ?valuation_model)
        (not
          (allocation_record_valuation_validated ?allocation_record)
        )
      )
    :effect (allocation_record_valuation_validated ?allocation_record)
  )
  (:action consume_collateral_item_for_allocation
    :parameters (?recovery_case - recovery_case ?collateral_item - collateral_item ?allocation_record - allocation_record)
    :precondition
      (and
        (entity_configured ?recovery_case)
        (case_has_allocation_record ?recovery_case ?allocation_record)
        (case_has_collateral_item ?recovery_case ?collateral_item)
        (collateral_item_available ?collateral_item)
        (allocation_record_prepared ?allocation_record)
        (allocation_record_valuation_validated ?allocation_record)
        (not
          (collateral_item_consumed ?collateral_item)
        )
      )
    :effect
      (and
        (collateral_item_consumed ?collateral_item)
        (collateral_item_allocated_to_record ?collateral_item ?allocation_record)
        (not
          (collateral_item_available ?collateral_item)
        )
      )
  )
  (:action validate_case_collateral_for_allocation
    :parameters (?recovery_case - recovery_case ?collateral_item - collateral_item ?allocation_record - allocation_record ?valuation_model - valuation_model)
    :precondition
      (and
        (entity_configured ?recovery_case)
        (case_has_collateral_item ?recovery_case ?collateral_item)
        (collateral_item_consumed ?collateral_item)
        (collateral_item_allocated_to_record ?collateral_item ?allocation_record)
        (entity_valuation_model_attached ?recovery_case ?valuation_model)
        (not
          (allocation_record_primary_flag ?allocation_record)
        )
        (not
          (case_collateral_validated ?recovery_case)
        )
      )
    :effect (case_collateral_validated ?recovery_case)
  )
  (:action attach_external_approval_to_case
    :parameters (?recovery_case - recovery_case ?external_approval - external_approval)
    :precondition
      (and
        (entity_configured ?recovery_case)
        (external_approval_available ?external_approval)
        (not
          (case_external_approval_attached ?recovery_case)
        )
      )
    :effect
      (and
        (case_external_approval_attached ?recovery_case)
        (case_has_external_approval ?recovery_case ?external_approval)
        (not
          (external_approval_available ?external_approval)
        )
      )
