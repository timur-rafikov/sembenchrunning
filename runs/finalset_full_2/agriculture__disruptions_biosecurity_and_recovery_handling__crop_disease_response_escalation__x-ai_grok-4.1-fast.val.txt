--- domain.pddl (as passed to VAL) ---
(define (domain crop_disease_response_escalation)
  (:requirements :strips :typing :negative-preconditions)
  (:types operational_resource - object material_resource - object spatial_or_facility_resource - object domain_entity - object crop_unit - domain_entity response_team - operational_resource diagnostic_capacity - operational_resource containment_team - operational_resource special_equipment - operational_resource restoration_package - operational_resource clearance_document - operational_resource technical_expert - operational_resource external_consultant - operational_resource treatment_batch - material_resource replant_stock - material_resource regulatory_permit - material_resource control_zone - spatial_or_facility_resource salvage_zone - spatial_or_facility_resource processing_facility - spatial_or_facility_resource field - crop_unit management_unit - crop_unit infected_plot - field adjacent_plot - field farm_operation - management_unit)
  (:predicates
    (incident_reported ?crop_unit - crop_unit)
    (incident_confirmed ?crop_unit - crop_unit)
    (response_deployed ?crop_unit - crop_unit)
    (cleared_for_production ?crop_unit - crop_unit)
    (ready_for_verification ?crop_unit - crop_unit)
    (verification_completed ?crop_unit - crop_unit)
    (response_team_available ?response_team - response_team)
    (response_team_assigned ?crop_unit - crop_unit ?response_team - response_team)
    (diagnostic_capacity_available ?diagnostic_capacity - diagnostic_capacity)
    (diagnostic_assigned ?crop_unit - crop_unit ?diagnostic_capacity - diagnostic_capacity)
    (containment_team_available ?containment_team - containment_team)
    (containment_team_assigned ?crop_unit - crop_unit ?containment_team - containment_team)
    (treatment_batch_available ?treatment_batch - treatment_batch)
    (treatment_allocated_to_infected_plot ?infected_plot - infected_plot ?treatment_batch - treatment_batch)
    (treatment_allocated_to_adjacent_plot ?adjacent_plot - adjacent_plot ?treatment_batch - treatment_batch)
    (plot_in_control_zone ?infected_plot - infected_plot ?control_zone - control_zone)
    (control_zone_active ?control_zone - control_zone)
    (control_zone_treatment_active ?control_zone - control_zone)
    (containment_confirmed ?infected_plot - infected_plot)
    (adjacent_plot_assigned_salvage_zone ?adjacent_plot - adjacent_plot ?salvage_zone - salvage_zone)
    (salvage_zone_reserved ?salvage_zone - salvage_zone)
    (salvage_zone_treatment_active ?salvage_zone - salvage_zone)
    (adjacent_containment_confirmed ?adjacent_plot - adjacent_plot)
    (processing_facility_available ?processing_facility - processing_facility)
    (processing_facility_reserved ?processing_facility - processing_facility)
    (facility_linked_to_control_zone ?processing_facility - processing_facility ?control_zone - control_zone)
    (facility_linked_to_salvage_zone ?processing_facility - processing_facility ?salvage_zone - salvage_zone)
    (facility_configured_for_control_processing ?processing_facility - processing_facility)
    (facility_configured_for_salvage_processing ?processing_facility - processing_facility)
    (facility_processing_activated ?processing_facility - processing_facility)
    (operation_manages_infected_plot ?farm_operation - farm_operation ?infected_plot - infected_plot)
    (operation_manages_adjacent_plot ?farm_operation - farm_operation ?adjacent_plot - adjacent_plot)
    (operation_associated_with_facility ?farm_operation - farm_operation ?processing_facility - processing_facility)
    (replant_stock_available ?replant_stock - replant_stock)
    (operation_has_replant_stock ?farm_operation - farm_operation ?replant_stock - replant_stock)
    (replant_stock_allocated ?replant_stock - replant_stock)
    (replant_stock_allocated_to_facility ?replant_stock - replant_stock ?processing_facility - processing_facility)
    (operation_prepared_for_recovery ?farm_operation - farm_operation)
    (recovery_materials_staged ?farm_operation - farm_operation)
    (consultant_review_completed ?farm_operation - farm_operation)
    (operation_equipment_allocated_flag ?farm_operation - farm_operation)
    (equipment_deployment_verified ?farm_operation - farm_operation)
    (recovery_plan_approved ?farm_operation - farm_operation)
    (recovery_execution_authorized ?farm_operation - farm_operation)
    (regulatory_permit_available ?regulatory_permit - regulatory_permit)
    (operation_has_regulatory_permit ?farm_operation - farm_operation ?regulatory_permit - regulatory_permit)
    (permit_assigned_to_operation ?farm_operation - farm_operation)
    (permit_validated ?farm_operation - farm_operation)
    (permit_authorization_completed ?farm_operation - farm_operation)
    (special_equipment_available ?special_equipment - special_equipment)
    (operation_allocated_equipment ?farm_operation - farm_operation ?special_equipment - special_equipment)
    (restoration_package_available ?restoration_package - restoration_package)
    (operation_allocated_restoration_package ?farm_operation - farm_operation ?restoration_package - restoration_package)
    (technical_expert_available ?technical_expert - technical_expert)
    (operation_allocated_technical_expert ?farm_operation - farm_operation ?technical_expert - technical_expert)
    (external_consultant_available ?external_consultant - external_consultant)
    (consultant_assigned_to_operation ?farm_operation - farm_operation ?external_consultant - external_consultant)
    (clearance_document_available ?clearance_document - clearance_document)
    (clearance_assigned_to_crop_unit ?crop_unit - crop_unit ?clearance_document - clearance_document)
    (plot_eligible_for_salvage ?infected_plot - infected_plot)
    (adjacent_plot_eligible_for_salvage ?adjacent_plot - adjacent_plot)
    (recovery_finalized ?farm_operation - farm_operation)
  )
  (:action detect_and_report_case
    :parameters (?crop_unit - crop_unit)
    :precondition
      (and
        (not
          (incident_reported ?crop_unit)
        )
        (not
          (cleared_for_production ?crop_unit)
        )
      )
    :effect (incident_reported ?crop_unit)
  )
  (:action deploy_response_team
    :parameters (?crop_unit - crop_unit ?response_team - response_team)
    :precondition
      (and
        (incident_reported ?crop_unit)
        (not
          (response_deployed ?crop_unit)
        )
        (response_team_available ?response_team)
      )
    :effect
      (and
        (response_deployed ?crop_unit)
        (response_team_assigned ?crop_unit ?response_team)
        (not
          (response_team_available ?response_team)
        )
      )
  )
  (:action allocate_diagnostic_capacity
    :parameters (?crop_unit - crop_unit ?diagnostic_capacity - diagnostic_capacity)
    :precondition
      (and
        (incident_reported ?crop_unit)
        (response_deployed ?crop_unit)
        (diagnostic_capacity_available ?diagnostic_capacity)
      )
    :effect
      (and
        (diagnostic_assigned ?crop_unit ?diagnostic_capacity)
        (not
          (diagnostic_capacity_available ?diagnostic_capacity)
        )
      )
  )
  (:action confirm_case_diagnosis
    :parameters (?crop_unit - crop_unit ?diagnostic_capacity - diagnostic_capacity)
    :precondition
      (and
        (incident_reported ?crop_unit)
        (response_deployed ?crop_unit)
        (diagnostic_assigned ?crop_unit ?diagnostic_capacity)
        (not
          (incident_confirmed ?crop_unit)
        )
      )
    :effect (incident_confirmed ?crop_unit)
  )
  (:action release_diagnostic_capacity
    :parameters (?crop_unit - crop_unit ?diagnostic_capacity - diagnostic_capacity)
    :precondition
      (and
        (diagnostic_assigned ?crop_unit ?diagnostic_capacity)
      )
    :effect
      (and
        (diagnostic_capacity_available ?diagnostic_capacity)
        (not
          (diagnostic_assigned ?crop_unit ?diagnostic_capacity)
        )
      )
  )
  (:action assign_containment_team_to_case
    :parameters (?crop_unit - crop_unit ?containment_team - containment_team)
    :precondition
      (and
        (incident_confirmed ?crop_unit)
        (containment_team_available ?containment_team)
      )
    :effect
      (and
        (containment_team_assigned ?crop_unit ?containment_team)
        (not
          (containment_team_available ?containment_team)
        )
      )
  )
  (:action release_containment_team_from_case
    :parameters (?crop_unit - crop_unit ?containment_team - containment_team)
    :precondition
      (and
        (containment_team_assigned ?crop_unit ?containment_team)
      )
    :effect
      (and
        (containment_team_available ?containment_team)
        (not
          (containment_team_assigned ?crop_unit ?containment_team)
        )
      )
  )
  (:action assign_technical_expert_to_operation
    :parameters (?farm_operation - farm_operation ?technical_expert - technical_expert)
    :precondition
      (and
        (incident_confirmed ?farm_operation)
        (technical_expert_available ?technical_expert)
      )
    :effect
      (and
        (operation_allocated_technical_expert ?farm_operation ?technical_expert)
        (not
          (technical_expert_available ?technical_expert)
        )
      )
  )
  (:action release_technical_expert_from_operation
    :parameters (?farm_operation - farm_operation ?technical_expert - technical_expert)
    :precondition
      (and
        (operation_allocated_technical_expert ?farm_operation ?technical_expert)
      )
    :effect
      (and
        (technical_expert_available ?technical_expert)
        (not
          (operation_allocated_technical_expert ?farm_operation ?technical_expert)
        )
      )
  )
  (:action assign_external_consultant_to_operation
    :parameters (?farm_operation - farm_operation ?external_consultant - external_consultant)
    :precondition
      (and
        (incident_confirmed ?farm_operation)
        (external_consultant_available ?external_consultant)
      )
    :effect
      (and
        (consultant_assigned_to_operation ?farm_operation ?external_consultant)
        (not
          (external_consultant_available ?external_consultant)
        )
      )
  )
  (:action release_external_consultant_from_operation
    :parameters (?farm_operation - farm_operation ?external_consultant - external_consultant)
    :precondition
      (and
        (consultant_assigned_to_operation ?farm_operation ?external_consultant)
      )
    :effect
      (and
        (external_consultant_available ?external_consultant)
        (not
          (consultant_assigned_to_operation ?farm_operation ?external_consultant)
        )
      )
  )
  (:action activate_control_zone_for_infected_plot
    :parameters (?infected_plot - infected_plot ?control_zone - control_zone ?diagnostic_capacity - diagnostic_capacity)
    :precondition
      (and
        (incident_confirmed ?infected_plot)
        (diagnostic_assigned ?infected_plot ?diagnostic_capacity)
        (plot_in_control_zone ?infected_plot ?control_zone)
        (not
          (control_zone_active ?control_zone)
        )
        (not
          (control_zone_treatment_active ?control_zone)
        )
      )
    :effect (control_zone_active ?control_zone)
  )
  (:action containment_team_execute_containment_for_plot
    :parameters (?infected_plot - infected_plot ?control_zone - control_zone ?containment_team - containment_team)
    :precondition
      (and
        (incident_confirmed ?infected_plot)
        (containment_team_assigned ?infected_plot ?containment_team)
        (plot_in_control_zone ?infected_plot ?control_zone)
        (control_zone_active ?control_zone)
        (not
          (plot_eligible_for_salvage ?infected_plot)
        )
      )
    :effect
      (and
        (plot_eligible_for_salvage ?infected_plot)
        (containment_confirmed ?infected_plot)
      )
  )
  (:action allocate_and_dispatch_treatment_batch_to_plot
    :parameters (?infected_plot - infected_plot ?control_zone - control_zone ?treatment_batch - treatment_batch)
    :precondition
      (and
        (incident_confirmed ?infected_plot)
        (plot_in_control_zone ?infected_plot ?control_zone)
        (treatment_batch_available ?treatment_batch)
        (not
          (plot_eligible_for_salvage ?infected_plot)
        )
      )
    :effect
      (and
        (control_zone_treatment_active ?control_zone)
        (plot_eligible_for_salvage ?infected_plot)
        (treatment_allocated_to_infected_plot ?infected_plot ?treatment_batch)
        (not
          (treatment_batch_available ?treatment_batch)
        )
      )
  )
  (:action apply_treatment_and_update_zone
    :parameters (?infected_plot - infected_plot ?control_zone - control_zone ?diagnostic_capacity - diagnostic_capacity ?treatment_batch - treatment_batch)
    :precondition
      (and
        (incident_confirmed ?infected_plot)
        (diagnostic_assigned ?infected_plot ?diagnostic_capacity)
        (plot_in_control_zone ?infected_plot ?control_zone)
        (control_zone_treatment_active ?control_zone)
        (treatment_allocated_to_infected_plot ?infected_plot ?treatment_batch)
        (not
          (containment_confirmed ?infected_plot)
        )
      )
    :effect
      (and
        (control_zone_active ?control_zone)
        (containment_confirmed ?infected_plot)
        (treatment_batch_available ?treatment_batch)
        (not
          (treatment_allocated_to_infected_plot ?infected_plot ?treatment_batch)
        )
      )
  )
  (:action reserve_salvage_zone_for_adjacent_plot
    :parameters (?adjacent_plot - adjacent_plot ?salvage_zone - salvage_zone ?diagnostic_capacity - diagnostic_capacity)
    :precondition
      (and
        (incident_confirmed ?adjacent_plot)
        (diagnostic_assigned ?adjacent_plot ?diagnostic_capacity)
        (adjacent_plot_assigned_salvage_zone ?adjacent_plot ?salvage_zone)
        (not
          (salvage_zone_reserved ?salvage_zone)
        )
        (not
          (salvage_zone_treatment_active ?salvage_zone)
        )
      )
    :effect (salvage_zone_reserved ?salvage_zone)
  )
  (:action containment_team_execute_adjacent_containment
    :parameters (?adjacent_plot - adjacent_plot ?salvage_zone - salvage_zone ?containment_team - containment_team)
    :precondition
      (and
        (incident_confirmed ?adjacent_plot)
        (containment_team_assigned ?adjacent_plot ?containment_team)
        (adjacent_plot_assigned_salvage_zone ?adjacent_plot ?salvage_zone)
        (salvage_zone_reserved ?salvage_zone)
        (not
          (adjacent_plot_eligible_for_salvage ?adjacent_plot)
        )
      )
    :effect
      (and
        (adjacent_plot_eligible_for_salvage ?adjacent_plot)
        (adjacent_containment_confirmed ?adjacent_plot)
      )
  )
  (:action allocate_treatment_to_adjacent_plot
    :parameters (?adjacent_plot - adjacent_plot ?salvage_zone - salvage_zone ?treatment_batch - treatment_batch)
    :precondition
      (and
        (incident_confirmed ?adjacent_plot)
        (adjacent_plot_assigned_salvage_zone ?adjacent_plot ?salvage_zone)
        (treatment_batch_available ?treatment_batch)
        (not
          (adjacent_plot_eligible_for_salvage ?adjacent_plot)
        )
      )
    :effect
      (and
        (salvage_zone_treatment_active ?salvage_zone)
        (adjacent_plot_eligible_for_salvage ?adjacent_plot)
        (treatment_allocated_to_adjacent_plot ?adjacent_plot ?treatment_batch)
        (not
          (treatment_batch_available ?treatment_batch)
        )
      )
  )
  (:action apply_treatment_to_adjacent_plot
    :parameters (?adjacent_plot - adjacent_plot ?salvage_zone - salvage_zone ?diagnostic_capacity - diagnostic_capacity ?treatment_batch - treatment_batch)
    :precondition
      (and
        (incident_confirmed ?adjacent_plot)
        (diagnostic_assigned ?adjacent_plot ?diagnostic_capacity)
        (adjacent_plot_assigned_salvage_zone ?adjacent_plot ?salvage_zone)
        (salvage_zone_treatment_active ?salvage_zone)
        (treatment_allocated_to_adjacent_plot ?adjacent_plot ?treatment_batch)
        (not
          (adjacent_containment_confirmed ?adjacent_plot)
        )
      )
    :effect
      (and
        (salvage_zone_reserved ?salvage_zone)
        (adjacent_containment_confirmed ?adjacent_plot)
        (treatment_batch_available ?treatment_batch)
        (not
          (treatment_allocated_to_adjacent_plot ?adjacent_plot ?treatment_batch)
        )
      )
  )
  (:action reserve_and_link_processing_facility
    :parameters (?infected_plot - infected_plot ?adjacent_plot - adjacent_plot ?control_zone - control_zone ?salvage_zone - salvage_zone ?processing_facility - processing_facility)
    :precondition
      (and
        (plot_eligible_for_salvage ?infected_plot)
        (adjacent_plot_eligible_for_salvage ?adjacent_plot)
        (plot_in_control_zone ?infected_plot ?control_zone)
        (adjacent_plot_assigned_salvage_zone ?adjacent_plot ?salvage_zone)
        (control_zone_active ?control_zone)
        (salvage_zone_reserved ?salvage_zone)
        (containment_confirmed ?infected_plot)
        (adjacent_containment_confirmed ?adjacent_plot)
        (processing_facility_available ?processing_facility)
      )
    :effect
      (and
        (processing_facility_reserved ?processing_facility)
        (facility_linked_to_control_zone ?processing_facility ?control_zone)
        (facility_linked_to_salvage_zone ?processing_facility ?salvage_zone)
        (not
          (processing_facility_available ?processing_facility)
        )
      )
  )
  (:action reserve_and_configure_facility_with_control_capability
    :parameters (?infected_plot - infected_plot ?adjacent_plot - adjacent_plot ?control_zone - control_zone ?salvage_zone - salvage_zone ?processing_facility - processing_facility)
    :precondition
      (and
        (plot_eligible_for_salvage ?infected_plot)
        (adjacent_plot_eligible_for_salvage ?adjacent_plot)
        (plot_in_control_zone ?infected_plot ?control_zone)
        (adjacent_plot_assigned_salvage_zone ?adjacent_plot ?salvage_zone)
        (control_zone_treatment_active ?control_zone)
        (salvage_zone_reserved ?salvage_zone)
        (not
          (containment_confirmed ?infected_plot)
        )
        (adjacent_containment_confirmed ?adjacent_plot)
        (processing_facility_available ?processing_facility)
      )
    :effect
      (and
        (processing_facility_reserved ?processing_facility)
        (facility_linked_to_control_zone ?processing_facility ?control_zone)
        (facility_linked_to_salvage_zone ?processing_facility ?salvage_zone)
        (facility_configured_for_control_processing ?processing_facility)
        (not
          (processing_facility_available ?processing_facility)
        )
      )
  )
  (:action reserve_and_configure_facility_with_salvage_capability
    :parameters (?infected_plot - infected_plot ?adjacent_plot - adjacent_plot ?control_zone - control_zone ?salvage_zone - salvage_zone ?processing_facility - processing_facility)
    :precondition
      (and
        (plot_eligible_for_salvage ?infected_plot)
        (adjacent_plot_eligible_for_salvage ?adjacent_plot)
        (plot_in_control_zone ?infected_plot ?control_zone)
        (adjacent_plot_assigned_salvage_zone ?adjacent_plot ?salvage_zone)
        (control_zone_active ?control_zone)
        (salvage_zone_treatment_active ?salvage_zone)
        (containment_confirmed ?infected_plot)
        (not
          (adjacent_containment_confirmed ?adjacent_plot)
        )
        (processing_facility_available ?processing_facility)
      )
    :effect
      (and
        (processing_facility_reserved ?processing_facility)
        (facility_linked_to_control_zone ?processing_facility ?control_zone)
        (facility_linked_to_salvage_zone ?processing_facility ?salvage_zone)
        (facility_configured_for_salvage_processing ?processing_facility)
        (not
          (processing_facility_available ?processing_facility)
        )
      )
  )
  (:action reserve_and_configure_facility_for_both_modes
    :parameters (?infected_plot - infected_plot ?adjacent_plot - adjacent_plot ?control_zone - control_zone ?salvage_zone - salvage_zone ?processing_facility - processing_facility)
    :precondition
      (and
        (plot_eligible_for_salvage ?infected_plot)
        (adjacent_plot_eligible_for_salvage ?adjacent_plot)
        (plot_in_control_zone ?infected_plot ?control_zone)
        (adjacent_plot_assigned_salvage_zone ?adjacent_plot ?salvage_zone)
        (control_zone_treatment_active ?control_zone)
        (salvage_zone_treatment_active ?salvage_zone)
        (not
          (containment_confirmed ?infected_plot)
        )
        (not
          (adjacent_containment_confirmed ?adjacent_plot)
        )
        (processing_facility_available ?processing_facility)
      )
    :effect
      (and
        (processing_facility_reserved ?processing_facility)
        (facility_linked_to_control_zone ?processing_facility ?control_zone)
        (facility_linked_to_salvage_zone ?processing_facility ?salvage_zone)
        (facility_configured_for_control_processing ?processing_facility)
        (facility_configured_for_salvage_processing ?processing_facility)
        (not
          (processing_facility_available ?processing_facility)
        )
      )
  )
  (:action activate_processing_at_facility_for_plot
    :parameters (?processing_facility - processing_facility ?infected_plot - infected_plot ?diagnostic_capacity - diagnostic_capacity)
    :precondition
      (and
        (processing_facility_reserved ?processing_facility)
        (plot_eligible_for_salvage ?infected_plot)
        (diagnostic_assigned ?infected_plot ?diagnostic_capacity)
        (not
          (facility_processing_activated ?processing_facility)
        )
      )
    :effect (facility_processing_activated ?processing_facility)
  )
  (:action stage_replant_stock_at_facility
    :parameters (?farm_operation - farm_operation ?replant_stock - replant_stock ?processing_facility - processing_facility)
    :precondition
      (and
        (incident_confirmed ?farm_operation)
        (operation_associated_with_facility ?farm_operation ?processing_facility)
        (operation_has_replant_stock ?farm_operation ?replant_stock)
        (replant_stock_available ?replant_stock)
        (processing_facility_reserved ?processing_facility)
        (facility_processing_activated ?processing_facility)
        (not
          (replant_stock_allocated ?replant_stock)
        )
      )
    :effect
      (and
        (replant_stock_allocated ?replant_stock)
        (replant_stock_allocated_to_facility ?replant_stock ?processing_facility)
        (not
          (replant_stock_available ?replant_stock)
        )
      )
  )
  (:action prepare_operation_for_replant
    :parameters (?farm_operation - farm_operation ?replant_stock - replant_stock ?processing_facility - processing_facility ?diagnostic_capacity - diagnostic_capacity)
    :precondition
      (and
        (incident_confirmed ?farm_operation)
        (operation_has_replant_stock ?farm_operation ?replant_stock)
        (replant_stock_allocated ?replant_stock)
        (replant_stock_allocated_to_facility ?replant_stock ?processing_facility)
        (diagnostic_assigned ?farm_operation ?diagnostic_capacity)
        (not
          (facility_configured_for_control_processing ?processing_facility)
        )
        (not
          (operation_prepared_for_recovery ?farm_operation)
        )
      )
    :effect (operation_prepared_for_recovery ?farm_operation)
  )
  (:action allocate_special_equipment_to_operation
    :parameters (?farm_operation - farm_operation ?special_equipment - special_equipment)
    :precondition
      (and
        (incident_confirmed ?farm_operation)
        (special_equipment_available ?special_equipment)
        (not
          (operation_equipment_allocated_flag ?farm_operation)
        )
      )
    :effect
      (and
        (operation_equipment_allocated_flag ?farm_operation)
        (operation_allocated_equipment ?farm_operation ?special_equipment)
        (not
          (special_equipment_available ?special_equipment)
        )
      )
  )
  (:action prepare_operation_with_equipment_and_stock
    :parameters (?farm_operation - farm_operation ?replant_stock - replant_stock ?processing_facility - processing_facility ?diagnostic_capacity - diagnostic_capacity ?special_equipment - special_equipment)
    :precondition
      (and
        (incident_confirmed ?farm_operation)
        (operation_has_replant_stock ?farm_operation ?replant_stock)
        (replant_stock_allocated ?replant_stock)
        (replant_stock_allocated_to_facility ?replant_stock ?processing_facility)
        (diagnostic_assigned ?farm_operation ?diagnostic_capacity)
        (facility_configured_for_control_processing ?processing_facility)
        (operation_equipment_allocated_flag ?farm_operation)
        (operation_allocated_equipment ?farm_operation ?special_equipment)
        (not
          (operation_prepared_for_recovery ?farm_operation)
        )
      )
    :effect
      (and
        (operation_prepared_for_recovery ?farm_operation)
