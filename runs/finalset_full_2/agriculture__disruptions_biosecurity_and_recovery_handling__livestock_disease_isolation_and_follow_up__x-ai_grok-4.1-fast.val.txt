--- domain.pddl (as passed to VAL) ---
(define (domain livestock_disease_isolation_and_follow_up)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_category - object supply_category - object logistics_category - object case_class - object epidemiological_unit - case_class isolation_resource - resource_category diagnostic_kit - resource_category responder_team - resource_category permit_document - resource_category contracted_service - resource_category clearance_document - resource_category treatment_product - resource_category specialist_service - resource_category consumable_supply - supply_category supply_batch - supply_category inspection_certificate - supply_category isolation_zone - logistics_category transport_unit - logistics_category contingency_kit - logistics_category herd_grouping - epidemiological_unit management_grouping - epidemiological_unit primary_herd - herd_grouping contact_herd - herd_grouping management_unit - management_grouping)
  (:predicates
    (reported_suspect_epidemiological_unit ?epidemiological_unit - epidemiological_unit)
    (confirmed_infected_epidemiological_unit ?epidemiological_unit - epidemiological_unit)
    (epidemiological_unit_isolation_assigned ?epidemiological_unit - epidemiological_unit)
    (clearance_granted ?epidemiological_unit - epidemiological_unit)
    (recovery_validated ?epidemiological_unit - epidemiological_unit)
    (epidemiological_unit_clearance_issued ?epidemiological_unit - epidemiological_unit)
    (isolation_resource_available ?isolation_resource - isolation_resource)
    (epidemiological_unit_assigned_isolation_resource ?epidemiological_unit - epidemiological_unit ?isolation_resource - isolation_resource)
    (diagnostic_kit_available ?diagnostic_kit - diagnostic_kit)
    (epidemiological_unit_assigned_diagnostic_kit ?epidemiological_unit - epidemiological_unit ?diagnostic_kit - diagnostic_kit)
    (responder_team_available ?responder_team - responder_team)
    (epidemiological_unit_assigned_responder_team ?epidemiological_unit - epidemiological_unit ?responder_team - responder_team)
    (consumable_available ?consumable_supply - consumable_supply)
    (primary_herd_allocated_consumable_supply ?primary_herd - primary_herd ?consumable_supply - consumable_supply)
    (consumable_allocated_to_contact_herd ?contact_herd - contact_herd ?consumable_supply - consumable_supply)
    (herd_assigned_isolation_zone ?primary_herd - primary_herd ?isolation_zone - isolation_zone)
    (isolation_zone_sampled ?isolation_zone - isolation_zone)
    (isolation_zone_supplied ?isolation_zone - isolation_zone)
    (primary_herd_triaged ?primary_herd - primary_herd)
    (contact_herd_assigned_transport_unit ?contact_herd - contact_herd ?transport_unit - transport_unit)
    (transport_unit_sampled ?transport_unit - transport_unit)
    (transport_unit_supplied ?transport_unit - transport_unit)
    (contact_herd_triaged ?contact_herd - contact_herd)
    (contingency_kit_available ?contingency_kit - contingency_kit)
    (contingency_kit_staged ?contingency_kit - contingency_kit)
    (contingency_kit_assigned_to_isolation_zone ?contingency_kit - contingency_kit ?isolation_zone - isolation_zone)
    (contingency_kit_assigned_to_transport_unit ?contingency_kit - contingency_kit ?transport_unit - transport_unit)
    (contingency_kit_marked_standard_deployment ?contingency_kit - contingency_kit)
    (contingency_kit_marked_enhanced_deployment ?contingency_kit - contingency_kit)
    (contingency_kit_lab_processed ?contingency_kit - contingency_kit)
    (management_has_primary_herd ?management_unit - management_unit ?primary_herd - primary_herd)
    (management_has_contact_herd ?management_unit - management_unit ?contact_herd - contact_herd)
    (management_has_contingency_kit ?management_unit - management_unit ?contingency_kit - contingency_kit)
    (supply_batch_available ?supply_batch - supply_batch)
    (management_has_supply_batch ?management_unit - management_unit ?supply_batch - supply_batch)
    (supply_batch_processed ?supply_batch - supply_batch)
    (supply_batch_linked_to_contingency_kit ?supply_batch - supply_batch ?contingency_kit - contingency_kit)
    (management_unit_prepared_for_treatment ?management_unit - management_unit)
    (treatment_ready ?management_unit - management_unit)
    (intervention_performed ?management_unit - management_unit)
    (management_unit_has_permit ?management_unit - management_unit)
    (permit_acknowledged ?management_unit - management_unit)
    (post_intervention_validation ?management_unit - management_unit)
    (recovery_checks_completed ?management_unit - management_unit)
    (inspection_certificate_available ?inspection_certificate - inspection_certificate)
    (management_unit_assigned_inspection_certificate ?management_unit - management_unit ?inspection_certificate - inspection_certificate)
    (inspection_recorded ?management_unit - management_unit)
    (inspection_started ?management_unit - management_unit)
    (inspection_passed ?management_unit - management_unit)
    (permit_document_available ?permit_document - permit_document)
    (management_unit_assigned_permit_document ?management_unit - management_unit ?permit_document - permit_document)
    (contracted_service_available ?contracted_service - contracted_service)
    (management_unit_assigned_contracted_service ?management_unit - management_unit ?contracted_service - contracted_service)
    (treatment_product_available ?treatment_product - treatment_product)
    (management_unit_assigned_treatment_product ?management_unit - management_unit ?treatment_product - treatment_product)
    (specialist_service_available ?specialist_service - specialist_service)
    (management_unit_assigned_specialist_service ?management_unit - management_unit ?specialist_service - specialist_service)
    (clearance_document_available ?clearance_document - clearance_document)
    (epidemiological_unit_assigned_clearance_document ?epidemiological_unit - epidemiological_unit ?clearance_document - clearance_document)
    (primary_herd_ready ?primary_herd - primary_herd)
    (contact_herd_ready ?contact_herd - contact_herd)
    (management_unit_finalized ?management_unit - management_unit)
  )
  (:action register_suspected_epidemiological_unit
    :parameters (?epidemiological_unit - epidemiological_unit)
    :precondition
      (and
        (not
          (reported_suspect_epidemiological_unit ?epidemiological_unit)
        )
        (not
          (clearance_granted ?epidemiological_unit)
        )
      )
    :effect (reported_suspect_epidemiological_unit ?epidemiological_unit)
  )
  (:action assign_isolation_resource_to_epidemiological_unit
    :parameters (?epidemiological_unit - epidemiological_unit ?isolation_resource - isolation_resource)
    :precondition
      (and
        (reported_suspect_epidemiological_unit ?epidemiological_unit)
        (not
          (epidemiological_unit_isolation_assigned ?epidemiological_unit)
        )
        (isolation_resource_available ?isolation_resource)
      )
    :effect
      (and
        (epidemiological_unit_isolation_assigned ?epidemiological_unit)
        (epidemiological_unit_assigned_isolation_resource ?epidemiological_unit ?isolation_resource)
        (not
          (isolation_resource_available ?isolation_resource)
        )
      )
  )
  (:action collect_sample_and_assign_diagnostic_kit
    :parameters (?epidemiological_unit - epidemiological_unit ?diagnostic_kit - diagnostic_kit)
    :precondition
      (and
        (reported_suspect_epidemiological_unit ?epidemiological_unit)
        (epidemiological_unit_isolation_assigned ?epidemiological_unit)
        (diagnostic_kit_available ?diagnostic_kit)
      )
    :effect
      (and
        (epidemiological_unit_assigned_diagnostic_kit ?epidemiological_unit ?diagnostic_kit)
        (not
          (diagnostic_kit_available ?diagnostic_kit)
        )
      )
  )
  (:action confirm_infection_for_epidemiological_unit
    :parameters (?epidemiological_unit - epidemiological_unit ?diagnostic_kit - diagnostic_kit)
    :precondition
      (and
        (reported_suspect_epidemiological_unit ?epidemiological_unit)
        (epidemiological_unit_isolation_assigned ?epidemiological_unit)
        (epidemiological_unit_assigned_diagnostic_kit ?epidemiological_unit ?diagnostic_kit)
        (not
          (confirmed_infected_epidemiological_unit ?epidemiological_unit)
        )
      )
    :effect (confirmed_infected_epidemiological_unit ?epidemiological_unit)
  )
  (:action release_diagnostic_kit_from_epidemiological_unit
    :parameters (?epidemiological_unit - epidemiological_unit ?diagnostic_kit - diagnostic_kit)
    :precondition
      (and
        (epidemiological_unit_assigned_diagnostic_kit ?epidemiological_unit ?diagnostic_kit)
      )
    :effect
      (and
        (diagnostic_kit_available ?diagnostic_kit)
        (not
          (epidemiological_unit_assigned_diagnostic_kit ?epidemiological_unit ?diagnostic_kit)
        )
      )
  )
  (:action assign_responder_team_to_epidemiological_unit
    :parameters (?epidemiological_unit - epidemiological_unit ?responder_team - responder_team)
    :precondition
      (and
        (confirmed_infected_epidemiological_unit ?epidemiological_unit)
        (responder_team_available ?responder_team)
      )
    :effect
      (and
        (epidemiological_unit_assigned_responder_team ?epidemiological_unit ?responder_team)
        (not
          (responder_team_available ?responder_team)
        )
      )
  )
  (:action release_responder_team_from_epidemiological_unit
    :parameters (?epidemiological_unit - epidemiological_unit ?responder_team - responder_team)
    :precondition
      (and
        (epidemiological_unit_assigned_responder_team ?epidemiological_unit ?responder_team)
      )
    :effect
      (and
        (responder_team_available ?responder_team)
        (not
          (epidemiological_unit_assigned_responder_team ?epidemiological_unit ?responder_team)
        )
      )
  )
  (:action assign_treatment_product_to_management_unit
    :parameters (?management_unit - management_unit ?treatment_product - treatment_product)
    :precondition
      (and
        (confirmed_infected_epidemiological_unit ?management_unit)
        (treatment_product_available ?treatment_product)
      )
    :effect
      (and
        (management_unit_assigned_treatment_product ?management_unit ?treatment_product)
        (not
          (treatment_product_available ?treatment_product)
        )
      )
  )
  (:action release_treatment_product_from_management_unit
    :parameters (?management_unit - management_unit ?treatment_product - treatment_product)
    :precondition
      (and
        (management_unit_assigned_treatment_product ?management_unit ?treatment_product)
      )
    :effect
      (and
        (treatment_product_available ?treatment_product)
        (not
          (management_unit_assigned_treatment_product ?management_unit ?treatment_product)
        )
      )
  )
  (:action assign_specialist_service_to_management_unit
    :parameters (?management_unit - management_unit ?specialist_service - specialist_service)
    :precondition
      (and
        (confirmed_infected_epidemiological_unit ?management_unit)
        (specialist_service_available ?specialist_service)
      )
    :effect
      (and
        (management_unit_assigned_specialist_service ?management_unit ?specialist_service)
        (not
          (specialist_service_available ?specialist_service)
        )
      )
  )
  (:action release_specialist_service_from_management_unit
    :parameters (?management_unit - management_unit ?specialist_service - specialist_service)
    :precondition
      (and
        (management_unit_assigned_specialist_service ?management_unit ?specialist_service)
      )
    :effect
      (and
        (specialist_service_available ?specialist_service)
        (not
          (management_unit_assigned_specialist_service ?management_unit ?specialist_service)
        )
      )
  )
  (:action process_sample_and_mark_zone_sampled
    :parameters (?primary_herd - primary_herd ?isolation_zone - isolation_zone ?diagnostic_kit - diagnostic_kit)
    :precondition
      (and
        (confirmed_infected_epidemiological_unit ?primary_herd)
        (epidemiological_unit_assigned_diagnostic_kit ?primary_herd ?diagnostic_kit)
        (herd_assigned_isolation_zone ?primary_herd ?isolation_zone)
        (not
          (isolation_zone_sampled ?isolation_zone)
        )
        (not
          (isolation_zone_supplied ?isolation_zone)
        )
      )
    :effect (isolation_zone_sampled ?isolation_zone)
  )
  (:action triage_primary_herd_with_team
    :parameters (?primary_herd - primary_herd ?isolation_zone - isolation_zone ?responder_team - responder_team)
    :precondition
      (and
        (confirmed_infected_epidemiological_unit ?primary_herd)
        (epidemiological_unit_assigned_responder_team ?primary_herd ?responder_team)
        (herd_assigned_isolation_zone ?primary_herd ?isolation_zone)
        (isolation_zone_sampled ?isolation_zone)
        (not
          (primary_herd_ready ?primary_herd)
        )
      )
    :effect
      (and
        (primary_herd_ready ?primary_herd)
        (primary_herd_triaged ?primary_herd)
      )
  )
  (:action supply_consumables_to_primary_herd
    :parameters (?primary_herd - primary_herd ?isolation_zone - isolation_zone ?consumable_supply - consumable_supply)
    :precondition
      (and
        (confirmed_infected_epidemiological_unit ?primary_herd)
        (herd_assigned_isolation_zone ?primary_herd ?isolation_zone)
        (consumable_available ?consumable_supply)
        (not
          (primary_herd_ready ?primary_herd)
        )
      )
    :effect
      (and
        (isolation_zone_supplied ?isolation_zone)
        (primary_herd_ready ?primary_herd)
        (primary_herd_allocated_consumable_supply ?primary_herd ?consumable_supply)
        (not
          (consumable_available ?consumable_supply)
        )
      )
  )
  (:action triage_primary_herd_with_consumables
    :parameters (?primary_herd - primary_herd ?isolation_zone - isolation_zone ?diagnostic_kit - diagnostic_kit ?consumable_supply - consumable_supply)
    :precondition
      (and
        (confirmed_infected_epidemiological_unit ?primary_herd)
        (epidemiological_unit_assigned_diagnostic_kit ?primary_herd ?diagnostic_kit)
        (herd_assigned_isolation_zone ?primary_herd ?isolation_zone)
        (isolation_zone_supplied ?isolation_zone)
        (primary_herd_allocated_consumable_supply ?primary_herd ?consumable_supply)
        (not
          (primary_herd_triaged ?primary_herd)
        )
      )
    :effect
      (and
        (isolation_zone_sampled ?isolation_zone)
        (primary_herd_triaged ?primary_herd)
        (consumable_available ?consumable_supply)
        (not
          (primary_herd_allocated_consumable_supply ?primary_herd ?consumable_supply)
        )
      )
  )
  (:action initiate_contact_herd_sampling
    :parameters (?contact_herd - contact_herd ?transport_unit - transport_unit ?diagnostic_kit - diagnostic_kit)
    :precondition
      (and
        (confirmed_infected_epidemiological_unit ?contact_herd)
        (epidemiological_unit_assigned_diagnostic_kit ?contact_herd ?diagnostic_kit)
        (contact_herd_assigned_transport_unit ?contact_herd ?transport_unit)
        (not
          (transport_unit_sampled ?transport_unit)
        )
        (not
          (transport_unit_supplied ?transport_unit)
        )
      )
    :effect (transport_unit_sampled ?transport_unit)
  )
  (:action assign_responder_team_to_contact_herd
    :parameters (?contact_herd - contact_herd ?transport_unit - transport_unit ?responder_team - responder_team)
    :precondition
      (and
        (confirmed_infected_epidemiological_unit ?contact_herd)
        (epidemiological_unit_assigned_responder_team ?contact_herd ?responder_team)
        (contact_herd_assigned_transport_unit ?contact_herd ?transport_unit)
        (transport_unit_sampled ?transport_unit)
        (not
          (contact_herd_ready ?contact_herd)
        )
      )
    :effect
      (and
        (contact_herd_ready ?contact_herd)
        (contact_herd_triaged ?contact_herd)
      )
  )
  (:action supply_consumables_to_contact_herd
    :parameters (?contact_herd - contact_herd ?transport_unit - transport_unit ?consumable_supply - consumable_supply)
    :precondition
      (and
        (confirmed_infected_epidemiological_unit ?contact_herd)
        (contact_herd_assigned_transport_unit ?contact_herd ?transport_unit)
        (consumable_available ?consumable_supply)
        (not
          (contact_herd_ready ?contact_herd)
        )
      )
    :effect
      (and
        (transport_unit_supplied ?transport_unit)
        (contact_herd_ready ?contact_herd)
        (consumable_allocated_to_contact_herd ?contact_herd ?consumable_supply)
        (not
          (consumable_available ?consumable_supply)
        )
      )
  )
  (:action triage_contact_herd_with_consumables
    :parameters (?contact_herd - contact_herd ?transport_unit - transport_unit ?diagnostic_kit - diagnostic_kit ?consumable_supply - consumable_supply)
    :precondition
      (and
        (confirmed_infected_epidemiological_unit ?contact_herd)
        (epidemiological_unit_assigned_diagnostic_kit ?contact_herd ?diagnostic_kit)
        (contact_herd_assigned_transport_unit ?contact_herd ?transport_unit)
        (transport_unit_supplied ?transport_unit)
        (consumable_allocated_to_contact_herd ?contact_herd ?consumable_supply)
        (not
          (contact_herd_triaged ?contact_herd)
        )
      )
    :effect
      (and
        (transport_unit_sampled ?transport_unit)
        (contact_herd_triaged ?contact_herd)
        (consumable_available ?consumable_supply)
        (not
          (consumable_allocated_to_contact_herd ?contact_herd ?consumable_supply)
        )
      )
  )
  (:action assemble_contingency_kit
    :parameters (?primary_herd - primary_herd ?contact_herd - contact_herd ?isolation_zone - isolation_zone ?transport_unit - transport_unit ?contingency_kit - contingency_kit)
    :precondition
      (and
        (primary_herd_ready ?primary_herd)
        (contact_herd_ready ?contact_herd)
        (herd_assigned_isolation_zone ?primary_herd ?isolation_zone)
        (contact_herd_assigned_transport_unit ?contact_herd ?transport_unit)
        (isolation_zone_sampled ?isolation_zone)
        (transport_unit_sampled ?transport_unit)
        (primary_herd_triaged ?primary_herd)
        (contact_herd_triaged ?contact_herd)
        (contingency_kit_available ?contingency_kit)
      )
    :effect
      (and
        (contingency_kit_staged ?contingency_kit)
        (contingency_kit_assigned_to_isolation_zone ?contingency_kit ?isolation_zone)
        (contingency_kit_assigned_to_transport_unit ?contingency_kit ?transport_unit)
        (not
          (contingency_kit_available ?contingency_kit)
        )
      )
  )
  (:action assemble_contingency_kit_mark_standard
    :parameters (?primary_herd - primary_herd ?contact_herd - contact_herd ?isolation_zone - isolation_zone ?transport_unit - transport_unit ?contingency_kit - contingency_kit)
    :precondition
      (and
        (primary_herd_ready ?primary_herd)
        (contact_herd_ready ?contact_herd)
        (herd_assigned_isolation_zone ?primary_herd ?isolation_zone)
        (contact_herd_assigned_transport_unit ?contact_herd ?transport_unit)
        (isolation_zone_supplied ?isolation_zone)
        (transport_unit_sampled ?transport_unit)
        (not
          (primary_herd_triaged ?primary_herd)
        )
        (contact_herd_triaged ?contact_herd)
        (contingency_kit_available ?contingency_kit)
      )
    :effect
      (and
        (contingency_kit_staged ?contingency_kit)
        (contingency_kit_assigned_to_isolation_zone ?contingency_kit ?isolation_zone)
        (contingency_kit_assigned_to_transport_unit ?contingency_kit ?transport_unit)
        (contingency_kit_marked_standard_deployment ?contingency_kit)
        (not
          (contingency_kit_available ?contingency_kit)
        )
      )
  )
  (:action assemble_contingency_kit_mark_enhanced
    :parameters (?primary_herd - primary_herd ?contact_herd - contact_herd ?isolation_zone - isolation_zone ?transport_unit - transport_unit ?contingency_kit - contingency_kit)
    :precondition
      (and
        (primary_herd_ready ?primary_herd)
        (contact_herd_ready ?contact_herd)
        (herd_assigned_isolation_zone ?primary_herd ?isolation_zone)
        (contact_herd_assigned_transport_unit ?contact_herd ?transport_unit)
        (isolation_zone_sampled ?isolation_zone)
        (transport_unit_supplied ?transport_unit)
        (primary_herd_triaged ?primary_herd)
        (not
          (contact_herd_triaged ?contact_herd)
        )
        (contingency_kit_available ?contingency_kit)
      )
    :effect
      (and
        (contingency_kit_staged ?contingency_kit)
        (contingency_kit_assigned_to_isolation_zone ?contingency_kit ?isolation_zone)
        (contingency_kit_assigned_to_transport_unit ?contingency_kit ?transport_unit)
        (contingency_kit_marked_enhanced_deployment ?contingency_kit)
        (not
          (contingency_kit_available ?contingency_kit)
        )
      )
  )
  (:action assemble_contingency_kit_mark_both
    :parameters (?primary_herd - primary_herd ?contact_herd - contact_herd ?isolation_zone - isolation_zone ?transport_unit - transport_unit ?contingency_kit - contingency_kit)
    :precondition
      (and
        (primary_herd_ready ?primary_herd)
        (contact_herd_ready ?contact_herd)
        (herd_assigned_isolation_zone ?primary_herd ?isolation_zone)
        (contact_herd_assigned_transport_unit ?contact_herd ?transport_unit)
        (isolation_zone_supplied ?isolation_zone)
        (transport_unit_supplied ?transport_unit)
        (not
          (primary_herd_triaged ?primary_herd)
        )
        (not
          (contact_herd_triaged ?contact_herd)
        )
        (contingency_kit_available ?contingency_kit)
      )
    :effect
      (and
        (contingency_kit_staged ?contingency_kit)
        (contingency_kit_assigned_to_isolation_zone ?contingency_kit ?isolation_zone)
        (contingency_kit_assigned_to_transport_unit ?contingency_kit ?transport_unit)
        (contingency_kit_marked_standard_deployment ?contingency_kit)
        (contingency_kit_marked_enhanced_deployment ?contingency_kit)
        (not
          (contingency_kit_available ?contingency_kit)
        )
      )
  )
  (:action process_contingency_kit_in_lab
    :parameters (?contingency_kit - contingency_kit ?primary_herd - primary_herd ?diagnostic_kit - diagnostic_kit)
    :precondition
      (and
        (contingency_kit_staged ?contingency_kit)
        (primary_herd_ready ?primary_herd)
        (epidemiological_unit_assigned_diagnostic_kit ?primary_herd ?diagnostic_kit)
        (not
          (contingency_kit_lab_processed ?contingency_kit)
        )
      )
    :effect (contingency_kit_lab_processed ?contingency_kit)
  )
  (:action process_supply_batch_and_link_to_kit
    :parameters (?management_unit - management_unit ?supply_batch - supply_batch ?contingency_kit - contingency_kit)
    :precondition
      (and
        (confirmed_infected_epidemiological_unit ?management_unit)
        (management_has_contingency_kit ?management_unit ?contingency_kit)
        (management_has_supply_batch ?management_unit ?supply_batch)
        (supply_batch_available ?supply_batch)
        (contingency_kit_staged ?contingency_kit)
        (contingency_kit_lab_processed ?contingency_kit)
        (not
          (supply_batch_processed ?supply_batch)
        )
      )
    :effect
      (and
        (supply_batch_processed ?supply_batch)
        (supply_batch_linked_to_contingency_kit ?supply_batch ?contingency_kit)
        (not
          (supply_batch_available ?supply_batch)
        )
      )
  )
  (:action prepare_management_unit_for_treatment
    :parameters (?management_unit - management_unit ?supply_batch - supply_batch ?contingency_kit - contingency_kit ?diagnostic_kit - diagnostic_kit)
    :precondition
      (and
        (confirmed_infected_epidemiological_unit ?management_unit)
        (management_has_supply_batch ?management_unit ?supply_batch)
        (supply_batch_processed ?supply_batch)
        (supply_batch_linked_to_contingency_kit ?supply_batch ?contingency_kit)
        (epidemiological_unit_assigned_diagnostic_kit ?management_unit ?diagnostic_kit)
        (not
          (contingency_kit_marked_standard_deployment ?contingency_kit)
        )
        (not
          (management_unit_prepared_for_treatment ?management_unit)
        )
      )
    :effect (management_unit_prepared_for_treatment ?management_unit)
  )
  (:action issue_permit_to_management_unit
    :parameters (?management_unit - management_unit ?permit_document - permit_document)
    :precondition
      (and
        (confirmed_infected_epidemiological_unit ?management_unit)
        (permit_document_available ?permit_document)
        (not
          (management_unit_has_permit ?management_unit)
        )
      )
    :effect
