--- domain.pddl (as passed to VAL) ---
(define (domain instructor_change_adjustment_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types domain_entity - object document - object resource - object case_root - object adjustment_case - case_root replacement_candidate - domain_entity syllabus_document - domain_entity support_staff - domain_entity classroom_resource_asset - domain_entity instructional_material_resource - domain_entity approval_token_artifact - domain_entity faculty_credential_document - domain_entity external_certificate_document - domain_entity administrative_document - document curriculum_module - document authorization_document - document primary_time_slot - resource secondary_time_slot - resource recovery_plan - resource section_group - adjustment_case section_variant - adjustment_case primary_section - section_group secondary_section - section_group course_record - section_variant)
  (:predicates
    (entity_registered ?adjustment_case - adjustment_case)
    (entity_validated ?adjustment_case - adjustment_case)
    (candidate_nominated ?adjustment_case - adjustment_case)
    (administrative_closure ?adjustment_case - adjustment_case)
    (readiness_certification ?adjustment_case - adjustment_case)
    (entity_committed ?adjustment_case - adjustment_case)
    (candidate_available ?replacement_candidate - replacement_candidate)
    (candidate_allocated ?adjustment_case - adjustment_case ?replacement_candidate - replacement_candidate)
    (syllabus_available ?syllabus_version - syllabus_document)
    (syllabus_linked ?adjustment_case - adjustment_case ?syllabus_version - syllabus_document)
    (support_staff_available ?support_staff - support_staff)
    (support_staff_assigned ?adjustment_case - adjustment_case ?support_staff - support_staff)
    (administrative_document_available ?administrative_document - administrative_document)
    (primary_section_document_bound ?primary_section - primary_section ?administrative_document - administrative_document)
    (secondary_section_document_bound ?secondary_section - secondary_section ?administrative_document - administrative_document)
    (primary_section_has_time_slot ?primary_section - primary_section ?primary_time_slot - primary_time_slot)
    (primary_time_slot_reserved ?primary_time_slot - primary_time_slot)
    (primary_time_slot_encumbered ?primary_time_slot - primary_time_slot)
    (primary_section_slot_confirmed ?primary_section - primary_section)
    (secondary_section_has_time_slot ?secondary_section - secondary_section ?secondary_time_slot - secondary_time_slot)
    (secondary_time_slot_reserved ?secondary_time_slot - secondary_time_slot)
    (secondary_time_slot_encumbered ?secondary_time_slot - secondary_time_slot)
    (secondary_section_slot_confirmed ?secondary_section - secondary_section)
    (recovery_plan_available ?recovery_plan - recovery_plan)
    (recovery_plan_prepared ?recovery_plan - recovery_plan)
    (recovery_plan_bound_primary_slot ?recovery_plan - recovery_plan ?primary_time_slot - primary_time_slot)
    (recovery_plan_bound_secondary_slot ?recovery_plan - recovery_plan ?secondary_time_slot - secondary_time_slot)
    (recovery_plan_primary_confirmation ?recovery_plan - recovery_plan)
    (recovery_plan_secondary_confirmation ?recovery_plan - recovery_plan)
    (recovery_plan_slot_validated ?recovery_plan - recovery_plan)
    (course_record_has_primary_section ?course_record - course_record ?primary_section - primary_section)
    (course_record_has_secondary_section ?course_record - course_record ?secondary_section - secondary_section)
    (course_record_linked_to_recovery_plan ?course_record - course_record ?recovery_plan - recovery_plan)
    (curriculum_module_available ?curriculum_module - curriculum_module)
    (course_record_includes_module ?course_record - course_record ?curriculum_module - curriculum_module)
    (curriculum_module_locked ?curriculum_module - curriculum_module)
    (curriculum_module_bound_to_plan ?curriculum_module - curriculum_module ?recovery_plan - recovery_plan)
    (course_record_prepared ?course_record - course_record)
    (course_record_ready_for_signoff ?course_record - course_record)
    (course_record_signoff_initiated ?course_record - course_record)
    (course_record_classroom_allocated ?course_record - course_record)
    (course_record_sequenced ?course_record - course_record)
    (course_record_instructional_material_bound ?course_record - course_record)
    (course_record_internal_checks_completed ?course_record - course_record)
    (authorization_document_available ?authorization_document - authorization_document)
    (course_record_has_authorization_document ?course_record - course_record ?authorization_document - authorization_document)
    (authorization_applied_to_course_record ?course_record - course_record)
    (authorization_processed ?course_record - course_record)
    (authorization_finalized ?course_record - course_record)
    (classroom_resource_available ?classroom_resource - classroom_resource_asset)
    (course_record_has_classroom_resource ?course_record - course_record ?classroom_resource - classroom_resource_asset)
    (instructional_material_available ?instructional_material - instructional_material_resource)
    (course_record_has_instructional_material ?course_record - course_record ?instructional_material - instructional_material_resource)
    (faculty_credential_available ?faculty_credential - faculty_credential_document)
    (course_record_has_faculty_credential ?course_record - course_record ?faculty_credential - faculty_credential_document)
    (external_certificate_available ?external_certificate - external_certificate_document)
    (course_record_has_external_certificate ?course_record - course_record ?external_certificate - external_certificate_document)
    (approval_token_available ?approval_token - approval_token_artifact)
    (entity_has_approval_token ?adjustment_case - adjustment_case ?approval_token - approval_token_artifact)
    (primary_section_ready ?primary_section - primary_section)
    (secondary_section_ready ?secondary_section - secondary_section)
    (course_record_internally_certified ?course_record - course_record)
  )
  (:action register_adjustment_case
    :parameters (?adjustment_case - adjustment_case)
    :precondition
      (and
        (not
          (entity_registered ?adjustment_case)
        )
        (not
          (administrative_closure ?adjustment_case)
        )
      )
    :effect (entity_registered ?adjustment_case)
  )
  (:action nominate_replacement_candidate
    :parameters (?adjustment_case - adjustment_case ?replacement_candidate - replacement_candidate)
    :precondition
      (and
        (entity_registered ?adjustment_case)
        (not
          (candidate_nominated ?adjustment_case)
        )
        (candidate_available ?replacement_candidate)
      )
    :effect
      (and
        (candidate_nominated ?adjustment_case)
        (candidate_allocated ?adjustment_case ?replacement_candidate)
        (not
          (candidate_available ?replacement_candidate)
        )
      )
  )
  (:action link_syllabus_to_case
    :parameters (?adjustment_case - adjustment_case ?syllabus_version - syllabus_document)
    :precondition
      (and
        (entity_registered ?adjustment_case)
        (candidate_nominated ?adjustment_case)
        (syllabus_available ?syllabus_version)
      )
    :effect
      (and
        (syllabus_linked ?adjustment_case ?syllabus_version)
        (not
          (syllabus_available ?syllabus_version)
        )
      )
  )
  (:action confirm_case_eligibility
    :parameters (?adjustment_case - adjustment_case ?syllabus_version - syllabus_document)
    :precondition
      (and
        (entity_registered ?adjustment_case)
        (candidate_nominated ?adjustment_case)
        (syllabus_linked ?adjustment_case ?syllabus_version)
        (not
          (entity_validated ?adjustment_case)
        )
      )
    :effect (entity_validated ?adjustment_case)
  )
  (:action release_syllabus_binding
    :parameters (?adjustment_case - adjustment_case ?syllabus_version - syllabus_document)
    :precondition
      (and
        (syllabus_linked ?adjustment_case ?syllabus_version)
      )
    :effect
      (and
        (syllabus_available ?syllabus_version)
        (not
          (syllabus_linked ?adjustment_case ?syllabus_version)
        )
      )
  )
  (:action assign_support_staff
    :parameters (?adjustment_case - adjustment_case ?support_staff - support_staff)
    :precondition
      (and
        (entity_validated ?adjustment_case)
        (support_staff_available ?support_staff)
      )
    :effect
      (and
        (support_staff_assigned ?adjustment_case ?support_staff)
        (not
          (support_staff_available ?support_staff)
        )
      )
  )
  (:action release_support_staff
    :parameters (?adjustment_case - adjustment_case ?support_staff - support_staff)
    :precondition
      (and
        (support_staff_assigned ?adjustment_case ?support_staff)
      )
    :effect
      (and
        (support_staff_available ?support_staff)
        (not
          (support_staff_assigned ?adjustment_case ?support_staff)
        )
      )
  )
  (:action attach_faculty_credential
    :parameters (?course_record - course_record ?faculty_credential - faculty_credential_document)
    :precondition
      (and
        (entity_validated ?course_record)
        (faculty_credential_available ?faculty_credential)
      )
    :effect
      (and
        (course_record_has_faculty_credential ?course_record ?faculty_credential)
        (not
          (faculty_credential_available ?faculty_credential)
        )
      )
  )
  (:action detach_faculty_credential
    :parameters (?course_record - course_record ?faculty_credential - faculty_credential_document)
    :precondition
      (and
        (course_record_has_faculty_credential ?course_record ?faculty_credential)
      )
    :effect
      (and
        (faculty_credential_available ?faculty_credential)
        (not
          (course_record_has_faculty_credential ?course_record ?faculty_credential)
        )
      )
  )
  (:action attach_external_certificate
    :parameters (?course_record - course_record ?external_certificate - external_certificate_document)
    :precondition
      (and
        (entity_validated ?course_record)
        (external_certificate_available ?external_certificate)
      )
    :effect
      (and
        (course_record_has_external_certificate ?course_record ?external_certificate)
        (not
          (external_certificate_available ?external_certificate)
        )
      )
  )
  (:action detach_external_certificate
    :parameters (?course_record - course_record ?external_certificate - external_certificate_document)
    :precondition
      (and
        (course_record_has_external_certificate ?course_record ?external_certificate)
      )
    :effect
      (and
        (external_certificate_available ?external_certificate)
        (not
          (course_record_has_external_certificate ?course_record ?external_certificate)
        )
      )
  )
  (:action reserve_primary_time_slot
    :parameters (?primary_section - primary_section ?primary_time_slot - primary_time_slot ?syllabus_version - syllabus_document)
    :precondition
      (and
        (entity_validated ?primary_section)
        (syllabus_linked ?primary_section ?syllabus_version)
        (primary_section_has_time_slot ?primary_section ?primary_time_slot)
        (not
          (primary_time_slot_reserved ?primary_time_slot)
        )
        (not
          (primary_time_slot_encumbered ?primary_time_slot)
        )
      )
    :effect (primary_time_slot_reserved ?primary_time_slot)
  )
  (:action confirm_primary_section_slot_with_staff
    :parameters (?primary_section - primary_section ?primary_time_slot - primary_time_slot ?support_staff - support_staff)
    :precondition
      (and
        (entity_validated ?primary_section)
        (support_staff_assigned ?primary_section ?support_staff)
        (primary_section_has_time_slot ?primary_section ?primary_time_slot)
        (primary_time_slot_reserved ?primary_time_slot)
        (not
          (primary_section_ready ?primary_section)
        )
      )
    :effect
      (and
        (primary_section_ready ?primary_section)
        (primary_section_slot_confirmed ?primary_section)
      )
  )
  (:action attach_administrative_document_to_primary_section
    :parameters (?primary_section - primary_section ?primary_time_slot - primary_time_slot ?administrative_document - administrative_document)
    :precondition
      (and
        (entity_validated ?primary_section)
        (primary_section_has_time_slot ?primary_section ?primary_time_slot)
        (administrative_document_available ?administrative_document)
        (not
          (primary_section_ready ?primary_section)
        )
      )
    :effect
      (and
        (primary_time_slot_encumbered ?primary_time_slot)
        (primary_section_ready ?primary_section)
        (primary_section_document_bound ?primary_section ?administrative_document)
        (not
          (administrative_document_available ?administrative_document)
        )
      )
  )
  (:action finalize_primary_section_slot_and_restore_document
    :parameters (?primary_section - primary_section ?primary_time_slot - primary_time_slot ?syllabus_version - syllabus_document ?administrative_document - administrative_document)
    :precondition
      (and
        (entity_validated ?primary_section)
        (syllabus_linked ?primary_section ?syllabus_version)
        (primary_section_has_time_slot ?primary_section ?primary_time_slot)
        (primary_time_slot_encumbered ?primary_time_slot)
        (primary_section_document_bound ?primary_section ?administrative_document)
        (not
          (primary_section_slot_confirmed ?primary_section)
        )
      )
    :effect
      (and
        (primary_time_slot_reserved ?primary_time_slot)
        (primary_section_slot_confirmed ?primary_section)
        (administrative_document_available ?administrative_document)
        (not
          (primary_section_document_bound ?primary_section ?administrative_document)
        )
      )
  )
  (:action reserve_secondary_time_slot
    :parameters (?secondary_section - secondary_section ?secondary_time_slot - secondary_time_slot ?syllabus_version - syllabus_document)
    :precondition
      (and
        (entity_validated ?secondary_section)
        (syllabus_linked ?secondary_section ?syllabus_version)
        (secondary_section_has_time_slot ?secondary_section ?secondary_time_slot)
        (not
          (secondary_time_slot_reserved ?secondary_time_slot)
        )
        (not
          (secondary_time_slot_encumbered ?secondary_time_slot)
        )
      )
    :effect (secondary_time_slot_reserved ?secondary_time_slot)
  )
  (:action confirm_secondary_section_slot_with_staff
    :parameters (?secondary_section - secondary_section ?secondary_time_slot - secondary_time_slot ?support_staff - support_staff)
    :precondition
      (and
        (entity_validated ?secondary_section)
        (support_staff_assigned ?secondary_section ?support_staff)
        (secondary_section_has_time_slot ?secondary_section ?secondary_time_slot)
        (secondary_time_slot_reserved ?secondary_time_slot)
        (not
          (secondary_section_ready ?secondary_section)
        )
      )
    :effect
      (and
        (secondary_section_ready ?secondary_section)
        (secondary_section_slot_confirmed ?secondary_section)
      )
  )
  (:action attach_administrative_document_to_secondary_section
    :parameters (?secondary_section - secondary_section ?secondary_time_slot - secondary_time_slot ?administrative_document - administrative_document)
    :precondition
      (and
        (entity_validated ?secondary_section)
        (secondary_section_has_time_slot ?secondary_section ?secondary_time_slot)
        (administrative_document_available ?administrative_document)
        (not
          (secondary_section_ready ?secondary_section)
        )
      )
    :effect
      (and
        (secondary_time_slot_encumbered ?secondary_time_slot)
        (secondary_section_ready ?secondary_section)
        (secondary_section_document_bound ?secondary_section ?administrative_document)
        (not
          (administrative_document_available ?administrative_document)
        )
      )
  )
  (:action finalize_secondary_section_slot_and_restore_document
    :parameters (?secondary_section - secondary_section ?secondary_time_slot - secondary_time_slot ?syllabus_version - syllabus_document ?administrative_document - administrative_document)
    :precondition
      (and
        (entity_validated ?secondary_section)
        (syllabus_linked ?secondary_section ?syllabus_version)
        (secondary_section_has_time_slot ?secondary_section ?secondary_time_slot)
        (secondary_time_slot_encumbered ?secondary_time_slot)
        (secondary_section_document_bound ?secondary_section ?administrative_document)
        (not
          (secondary_section_slot_confirmed ?secondary_section)
        )
      )
    :effect
      (and
        (secondary_time_slot_reserved ?secondary_time_slot)
        (secondary_section_slot_confirmed ?secondary_section)
        (administrative_document_available ?administrative_document)
        (not
          (secondary_section_document_bound ?secondary_section ?administrative_document)
        )
      )
  )
  (:action assemble_recovery_plan
    :parameters (?primary_section - primary_section ?secondary_section - secondary_section ?primary_time_slot - primary_time_slot ?secondary_time_slot - secondary_time_slot ?recovery_plan - recovery_plan)
    :precondition
      (and
        (primary_section_ready ?primary_section)
        (secondary_section_ready ?secondary_section)
        (primary_section_has_time_slot ?primary_section ?primary_time_slot)
        (secondary_section_has_time_slot ?secondary_section ?secondary_time_slot)
        (primary_time_slot_reserved ?primary_time_slot)
        (secondary_time_slot_reserved ?secondary_time_slot)
        (primary_section_slot_confirmed ?primary_section)
        (secondary_section_slot_confirmed ?secondary_section)
        (recovery_plan_available ?recovery_plan)
      )
    :effect
      (and
        (recovery_plan_prepared ?recovery_plan)
        (recovery_plan_bound_primary_slot ?recovery_plan ?primary_time_slot)
        (recovery_plan_bound_secondary_slot ?recovery_plan ?secondary_time_slot)
        (not
          (recovery_plan_available ?recovery_plan)
        )
      )
  )
  (:action assemble_recovery_plan_with_primary_confirmation
    :parameters (?primary_section - primary_section ?secondary_section - secondary_section ?primary_time_slot - primary_time_slot ?secondary_time_slot - secondary_time_slot ?recovery_plan - recovery_plan)
    :precondition
      (and
        (primary_section_ready ?primary_section)
        (secondary_section_ready ?secondary_section)
        (primary_section_has_time_slot ?primary_section ?primary_time_slot)
        (secondary_section_has_time_slot ?secondary_section ?secondary_time_slot)
        (primary_time_slot_encumbered ?primary_time_slot)
        (secondary_time_slot_reserved ?secondary_time_slot)
        (not
          (primary_section_slot_confirmed ?primary_section)
        )
        (secondary_section_slot_confirmed ?secondary_section)
        (recovery_plan_available ?recovery_plan)
      )
    :effect
      (and
        (recovery_plan_prepared ?recovery_plan)
        (recovery_plan_bound_primary_slot ?recovery_plan ?primary_time_slot)
        (recovery_plan_bound_secondary_slot ?recovery_plan ?secondary_time_slot)
        (recovery_plan_primary_confirmation ?recovery_plan)
        (not
          (recovery_plan_available ?recovery_plan)
        )
      )
  )
  (:action assemble_recovery_plan_with_secondary_confirmation
    :parameters (?primary_section - primary_section ?secondary_section - secondary_section ?primary_time_slot - primary_time_slot ?secondary_time_slot - secondary_time_slot ?recovery_plan - recovery_plan)
    :precondition
      (and
        (primary_section_ready ?primary_section)
        (secondary_section_ready ?secondary_section)
        (primary_section_has_time_slot ?primary_section ?primary_time_slot)
        (secondary_section_has_time_slot ?secondary_section ?secondary_time_slot)
        (primary_time_slot_reserved ?primary_time_slot)
        (secondary_time_slot_encumbered ?secondary_time_slot)
        (primary_section_slot_confirmed ?primary_section)
        (not
          (secondary_section_slot_confirmed ?secondary_section)
        )
        (recovery_plan_available ?recovery_plan)
      )
    :effect
      (and
        (recovery_plan_prepared ?recovery_plan)
        (recovery_plan_bound_primary_slot ?recovery_plan ?primary_time_slot)
        (recovery_plan_bound_secondary_slot ?recovery_plan ?secondary_time_slot)
        (recovery_plan_secondary_confirmation ?recovery_plan)
        (not
          (recovery_plan_available ?recovery_plan)
        )
      )
  )
  (:action assemble_recovery_plan_with_both_confirmations
    :parameters (?primary_section - primary_section ?secondary_section - secondary_section ?primary_time_slot - primary_time_slot ?secondary_time_slot - secondary_time_slot ?recovery_plan - recovery_plan)
    :precondition
      (and
        (primary_section_ready ?primary_section)
        (secondary_section_ready ?secondary_section)
        (primary_section_has_time_slot ?primary_section ?primary_time_slot)
        (secondary_section_has_time_slot ?secondary_section ?secondary_time_slot)
        (primary_time_slot_encumbered ?primary_time_slot)
        (secondary_time_slot_encumbered ?secondary_time_slot)
        (not
          (primary_section_slot_confirmed ?primary_section)
        )
        (not
          (secondary_section_slot_confirmed ?secondary_section)
        )
        (recovery_plan_available ?recovery_plan)
      )
    :effect
      (and
        (recovery_plan_prepared ?recovery_plan)
        (recovery_plan_bound_primary_slot ?recovery_plan ?primary_time_slot)
        (recovery_plan_bound_secondary_slot ?recovery_plan ?secondary_time_slot)
        (recovery_plan_primary_confirmation ?recovery_plan)
        (recovery_plan_secondary_confirmation ?recovery_plan)
        (not
          (recovery_plan_available ?recovery_plan)
        )
      )
  )
  (:action validate_recovery_plan_slot
    :parameters (?recovery_plan - recovery_plan ?primary_section - primary_section ?syllabus_version - syllabus_document)
    :precondition
      (and
        (recovery_plan_prepared ?recovery_plan)
        (primary_section_ready ?primary_section)
        (syllabus_linked ?primary_section ?syllabus_version)
        (not
          (recovery_plan_slot_validated ?recovery_plan)
        )
      )
    :effect (recovery_plan_slot_validated ?recovery_plan)
  )
  (:action lock_curriculum_module_into_plan
    :parameters (?course_record - course_record ?curriculum_module - curriculum_module ?recovery_plan - recovery_plan)
    :precondition
      (and
        (entity_validated ?course_record)
        (course_record_linked_to_recovery_plan ?course_record ?recovery_plan)
        (course_record_includes_module ?course_record ?curriculum_module)
        (curriculum_module_available ?curriculum_module)
        (recovery_plan_prepared ?recovery_plan)
        (recovery_plan_slot_validated ?recovery_plan)
        (not
          (curriculum_module_locked ?curriculum_module)
        )
      )
    :effect
      (and
        (curriculum_module_locked ?curriculum_module)
        (curriculum_module_bound_to_plan ?curriculum_module ?recovery_plan)
        (not
          (curriculum_module_available ?curriculum_module)
        )
      )
  )
  (:action finalize_module_binding
    :parameters (?course_record - course_record ?curriculum_module - curriculum_module ?recovery_plan - recovery_plan ?syllabus_version - syllabus_document)
    :precondition
      (and
        (entity_validated ?course_record)
        (course_record_includes_module ?course_record ?curriculum_module)
        (curriculum_module_locked ?curriculum_module)
        (curriculum_module_bound_to_plan ?curriculum_module ?recovery_plan)
        (syllabus_linked ?course_record ?syllabus_version)
        (not
          (recovery_plan_primary_confirmation ?recovery_plan)
        )
        (not
          (course_record_prepared ?course_record)
        )
      )
    :effect (course_record_prepared ?course_record)
  )
  (:action allocate_classroom_resource
    :parameters (?course_record - course_record ?classroom_resource - classroom_resource_asset)
    :precondition
      (and
        (entity_validated ?course_record)
        (classroom_resource_available ?classroom_resource)
        (not
          (course_record_classroom_allocated ?course_record)
        )
      )
    :effect
      (and
        (course_record_classroom_allocated ?course_record)
        (course_record_has_classroom_resource ?course_record ?classroom_resource)
        (not
          (classroom_resource_available ?classroom_resource)
        )
      )
  )
  (:action sequence_administrative_preparations
    :parameters (?course_record - course_record ?curriculum_module - curriculum_module ?recovery_plan - recovery_plan ?syllabus_version - syllabus_document ?classroom_resource - classroom_resource_asset)
    :precondition
      (and
        (entity_validated ?course_record)
        (course_record_includes_module ?course_record ?curriculum_module)
        (curriculum_module_locked ?curriculum_module)
        (curriculum_module_bound_to_plan ?curriculum_module ?recovery_plan)
        (syllabus_linked ?course_record ?syllabus_version)
        (recovery_plan_primary_confirmation ?recovery_plan)
        (course_record_classroom_allocated ?course_record)
        (course_record_has_classroom_resource ?course_record ?classroom_resource)
        (not
          (course_record_prepared ?course_record)
        )
      )
    :effect
      (and
        (course_record_prepared ?course_record)
        (course_record_sequenced ?course_record)
      )
  )
  (:action prepare_course_record_credentials_initial
    :parameters (?course_record - course_record ?faculty_credential - faculty_credential_document ?support_staff - support_staff ?curriculum_module - curriculum_module ?recovery_plan - recovery_plan)
    :precondition
