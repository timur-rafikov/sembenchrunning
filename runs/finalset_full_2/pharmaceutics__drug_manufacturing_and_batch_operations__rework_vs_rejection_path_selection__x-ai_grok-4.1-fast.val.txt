--- domain.pddl (as passed to VAL) ---
(define (domain pharmaceutical_batch_rework_rejection_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource - object artifact - object control_point_category - object production_entity_root - object production_batch - production_entity_root manufacturing_line - resource process_step - resource qualified_operator - resource label_asset - resource packaging_material_batch - resource disposition_document - resource validation_document - resource qa_signatory - resource work_order - artifact qc_sample - artifact quality_representative - artifact rework_hold_point - control_point_category rejection_hold_point - control_point_category packaging_unit_category - control_point_category processing_branch - production_batch documentation_branch - production_batch primary_stage_instance - processing_branch secondary_stage_instance - processing_branch batch_record - documentation_branch)
  (:predicates
    (batch_registered ?production_batch - production_batch)
    (entity_open ?production_batch - production_batch)
    (batch_allocated ?production_batch - production_batch)
    (entity_finalized ?production_batch - production_batch)
    (entity_release_ready ?production_batch - production_batch)
    (entity_disposition_recorded ?production_batch - production_batch)
    (manufacturing_line_available ?manufacturing_line - manufacturing_line)
    (entity_assigned_to_line ?production_batch - production_batch ?manufacturing_line - manufacturing_line)
    (process_step_available ?process_step - process_step)
    (entity_assigned_to_process_step ?production_batch - production_batch ?process_step - process_step)
    (operator_available ?qualified_operator - qualified_operator)
    (operator_assigned_to_entity ?production_batch - production_batch ?qualified_operator - qualified_operator)
    (work_order_available ?rework_workorder - work_order)
    (work_order_for_primary_stage ?primary_stage_instance - primary_stage_instance ?rework_workorder - work_order)
    (work_order_for_secondary_stage ?secondary_stage_instance - secondary_stage_instance ?rework_workorder - work_order)
    (stage_has_rework_hold_point ?primary_stage_instance - primary_stage_instance ?rework_hold_point - rework_hold_point)
    (rework_hold_raised ?rework_hold_point - rework_hold_point)
    (work_order_issued ?rework_hold_point - rework_hold_point)
    (primary_stage_completed ?primary_stage_instance - primary_stage_instance)
    (stage_has_rejection_hold_point ?secondary_stage_instance - secondary_stage_instance ?rejection_hold_point - rejection_hold_point)
    (rejection_hold_raised ?rejection_hold_point - rejection_hold_point)
    (rejection_workorder_issued ?rejection_hold_point - rejection_hold_point)
    (secondary_stage_completed ?secondary_stage_instance - secondary_stage_instance)
    (packaging_unit_available ?finished_packaging_unit - packaging_unit_category)
    (packaging_unit_allocated ?finished_packaging_unit - packaging_unit_category)
    (packaging_unit_assigned_rework_hold ?finished_packaging_unit - packaging_unit_category ?rework_hold_point - rework_hold_point)
    (packaging_unit_assigned_rejection_hold ?finished_packaging_unit - packaging_unit_category ?rejection_hold_point - rejection_hold_point)
    (packaging_unit_rework_flag ?finished_packaging_unit - packaging_unit_category)
    (packaging_unit_rejection_flag ?finished_packaging_unit - packaging_unit_category)
    (packaging_unit_packaged ?finished_packaging_unit - packaging_unit_category)
    (record_covers_primary_stage ?batch_record - batch_record ?primary_stage_instance - primary_stage_instance)
    (record_covers_secondary_stage ?batch_record - batch_record ?secondary_stage_instance - secondary_stage_instance)
    (record_covers_packaging_unit ?batch_record - batch_record ?finished_packaging_unit - packaging_unit_category)
    (qc_sample_available ?qc_sample - qc_sample)
    (record_has_sample ?batch_record - batch_record ?qc_sample - qc_sample)
    (qc_sample_tested ?qc_sample - qc_sample)
    (qc_sample_assigned_to_packaging_unit ?qc_sample - qc_sample ?finished_packaging_unit - packaging_unit_category)
    (record_marked_for_analysis ?batch_record - batch_record)
    (analysis_ready ?batch_record - batch_record)
    (qa_review_started ?batch_record - batch_record)
    (label_assigned_to_record ?batch_record - batch_record)
    (label_registered_on_record ?batch_record - batch_record)
    (qa_review_completed ?batch_record - batch_record)
    (release_checklist_completed ?batch_record - batch_record)
    (quality_representative_available ?quality_representative - quality_representative)
    (record_has_quality_representative ?batch_record - batch_record ?quality_representative - quality_representative)
    (quality_representative_assigned ?batch_record - batch_record)
    (quality_rep_tasks_completed ?batch_record - batch_record)
    (qa_signatory_applied ?batch_record - batch_record)
    (label_asset_available ?label_asset - label_asset)
    (record_has_label_asset ?batch_record - batch_record ?label_asset - label_asset)
    (packaging_material_available ?packaging_material_batch - packaging_material_batch)
    (record_has_packaging_material ?batch_record - batch_record ?packaging_material_batch - packaging_material_batch)
    (validation_document_available ?validation_document - validation_document)
    (record_has_validation_document ?batch_record - batch_record ?validation_document - validation_document)
    (qa_signatory_available ?qa_signatory - qa_signatory)
    (record_has_qa_signatory ?batch_record - batch_record ?qa_signatory - qa_signatory)
    (disposition_document_available ?disposition_document - disposition_document)
    (entity_linked_to_disposition_document ?production_batch - production_batch ?disposition_document - disposition_document)
    (primary_stage_prepared ?primary_stage_instance - primary_stage_instance)
    (secondary_stage_prepared ?secondary_stage_instance - secondary_stage_instance)
    (record_signoff_complete ?batch_record - batch_record)
  )
  (:action intake_batch
    :parameters (?production_batch - production_batch)
    :precondition
      (and
        (not
          (batch_registered ?production_batch)
        )
        (not
          (entity_finalized ?production_batch)
        )
      )
    :effect (batch_registered ?production_batch)
  )
  (:action assign_batch_to_line
    :parameters (?production_batch - production_batch ?manufacturing_line - manufacturing_line)
    :precondition
      (and
        (batch_registered ?production_batch)
        (not
          (batch_allocated ?production_batch)
        )
        (manufacturing_line_available ?manufacturing_line)
      )
    :effect
      (and
        (batch_allocated ?production_batch)
        (entity_assigned_to_line ?production_batch ?manufacturing_line)
        (not
          (manufacturing_line_available ?manufacturing_line)
        )
      )
  )
  (:action assign_batch_to_process_step
    :parameters (?production_batch - production_batch ?process_step - process_step)
    :precondition
      (and
        (batch_registered ?production_batch)
        (batch_allocated ?production_batch)
        (process_step_available ?process_step)
      )
    :effect
      (and
        (entity_assigned_to_process_step ?production_batch ?process_step)
        (not
          (process_step_available ?process_step)
        )
      )
  )
  (:action open_batch_record
    :parameters (?production_batch - production_batch ?process_step - process_step)
    :precondition
      (and
        (batch_registered ?production_batch)
        (batch_allocated ?production_batch)
        (entity_assigned_to_process_step ?production_batch ?process_step)
        (not
          (entity_open ?production_batch)
        )
      )
    :effect (entity_open ?production_batch)
  )
  (:action release_process_step
    :parameters (?production_batch - production_batch ?process_step - process_step)
    :precondition
      (and
        (entity_assigned_to_process_step ?production_batch ?process_step)
      )
    :effect
      (and
        (process_step_available ?process_step)
        (not
          (entity_assigned_to_process_step ?production_batch ?process_step)
        )
      )
  )
  (:action assign_operator_to_batch
    :parameters (?production_batch - production_batch ?qualified_operator - qualified_operator)
    :precondition
      (and
        (entity_open ?production_batch)
        (operator_available ?qualified_operator)
      )
    :effect
      (and
        (operator_assigned_to_entity ?production_batch ?qualified_operator)
        (not
          (operator_available ?qualified_operator)
        )
      )
  )
  (:action release_operator_from_batch
    :parameters (?production_batch - production_batch ?qualified_operator - qualified_operator)
    :precondition
      (and
        (operator_assigned_to_entity ?production_batch ?qualified_operator)
      )
    :effect
      (and
        (operator_available ?qualified_operator)
        (not
          (operator_assigned_to_entity ?production_batch ?qualified_operator)
        )
      )
  )
  (:action attach_validation_document_to_record
    :parameters (?batch_record - batch_record ?validation_document - validation_document)
    :precondition
      (and
        (entity_open ?batch_record)
        (validation_document_available ?validation_document)
      )
    :effect
      (and
        (record_has_validation_document ?batch_record ?validation_document)
        (not
          (validation_document_available ?validation_document)
        )
      )
  )
  (:action detach_validation_document_from_record
    :parameters (?batch_record - batch_record ?validation_document - validation_document)
    :precondition
      (and
        (record_has_validation_document ?batch_record ?validation_document)
      )
    :effect
      (and
        (validation_document_available ?validation_document)
        (not
          (record_has_validation_document ?batch_record ?validation_document)
        )
      )
  )
  (:action attach_qa_signatory_to_record
    :parameters (?batch_record - batch_record ?qa_signatory - qa_signatory)
    :precondition
      (and
        (entity_open ?batch_record)
        (qa_signatory_available ?qa_signatory)
      )
    :effect
      (and
        (record_has_qa_signatory ?batch_record ?qa_signatory)
        (not
          (qa_signatory_available ?qa_signatory)
        )
      )
  )
  (:action detach_qa_signatory_from_record
    :parameters (?batch_record - batch_record ?qa_signatory - qa_signatory)
    :precondition
      (and
        (record_has_qa_signatory ?batch_record ?qa_signatory)
      )
    :effect
      (and
        (qa_signatory_available ?qa_signatory)
        (not
          (record_has_qa_signatory ?batch_record ?qa_signatory)
        )
      )
  )
  (:action raise_rework_hold_point
    :parameters (?primary_stage_instance - primary_stage_instance ?rework_hold_point - rework_hold_point ?process_step - process_step)
    :precondition
      (and
        (entity_open ?primary_stage_instance)
        (entity_assigned_to_process_step ?primary_stage_instance ?process_step)
        (stage_has_rework_hold_point ?primary_stage_instance ?rework_hold_point)
        (not
          (rework_hold_raised ?rework_hold_point)
        )
        (not
          (work_order_issued ?rework_hold_point)
        )
      )
    :effect (rework_hold_raised ?rework_hold_point)
  )
  (:action acknowledge_rework_hold_and_prepare_stage
    :parameters (?primary_stage_instance - primary_stage_instance ?rework_hold_point - rework_hold_point ?qualified_operator - qualified_operator)
    :precondition
      (and
        (entity_open ?primary_stage_instance)
        (operator_assigned_to_entity ?primary_stage_instance ?qualified_operator)
        (stage_has_rework_hold_point ?primary_stage_instance ?rework_hold_point)
        (rework_hold_raised ?rework_hold_point)
        (not
          (primary_stage_prepared ?primary_stage_instance)
        )
      )
    :effect
      (and
        (primary_stage_prepared ?primary_stage_instance)
        (primary_stage_completed ?primary_stage_instance)
      )
  )
  (:action issue_rework_workorder_for_stage
    :parameters (?primary_stage_instance - primary_stage_instance ?rework_hold_point - rework_hold_point ?rework_workorder - work_order)
    :precondition
      (and
        (entity_open ?primary_stage_instance)
        (stage_has_rework_hold_point ?primary_stage_instance ?rework_hold_point)
        (work_order_available ?rework_workorder)
        (not
          (primary_stage_prepared ?primary_stage_instance)
        )
      )
    :effect
      (and
        (work_order_issued ?rework_hold_point)
        (primary_stage_prepared ?primary_stage_instance)
        (work_order_for_primary_stage ?primary_stage_instance ?rework_workorder)
        (not
          (work_order_available ?rework_workorder)
        )
      )
  )
  (:action complete_rework_and_update_stage
    :parameters (?primary_stage_instance - primary_stage_instance ?rework_hold_point - rework_hold_point ?process_step - process_step ?rework_workorder - work_order)
    :precondition
      (and
        (entity_open ?primary_stage_instance)
        (entity_assigned_to_process_step ?primary_stage_instance ?process_step)
        (stage_has_rework_hold_point ?primary_stage_instance ?rework_hold_point)
        (work_order_issued ?rework_hold_point)
        (work_order_for_primary_stage ?primary_stage_instance ?rework_workorder)
        (not
          (primary_stage_completed ?primary_stage_instance)
        )
      )
    :effect
      (and
        (rework_hold_raised ?rework_hold_point)
        (primary_stage_completed ?primary_stage_instance)
        (work_order_available ?rework_workorder)
        (not
          (work_order_for_primary_stage ?primary_stage_instance ?rework_workorder)
        )
      )
  )
  (:action raise_rejection_hold_point
    :parameters (?secondary_stage_instance - secondary_stage_instance ?rejection_hold_point - rejection_hold_point ?process_step - process_step)
    :precondition
      (and
        (entity_open ?secondary_stage_instance)
        (entity_assigned_to_process_step ?secondary_stage_instance ?process_step)
        (stage_has_rejection_hold_point ?secondary_stage_instance ?rejection_hold_point)
        (not
          (rejection_hold_raised ?rejection_hold_point)
        )
        (not
          (rejection_workorder_issued ?rejection_hold_point)
        )
      )
    :effect (rejection_hold_raised ?rejection_hold_point)
  )
  (:action acknowledge_rejection_hold_and_prepare_stage
    :parameters (?secondary_stage_instance - secondary_stage_instance ?rejection_hold_point - rejection_hold_point ?qualified_operator - qualified_operator)
    :precondition
      (and
        (entity_open ?secondary_stage_instance)
        (operator_assigned_to_entity ?secondary_stage_instance ?qualified_operator)
        (stage_has_rejection_hold_point ?secondary_stage_instance ?rejection_hold_point)
        (rejection_hold_raised ?rejection_hold_point)
        (not
          (secondary_stage_prepared ?secondary_stage_instance)
        )
      )
    :effect
      (and
        (secondary_stage_prepared ?secondary_stage_instance)
        (secondary_stage_completed ?secondary_stage_instance)
      )
  )
  (:action issue_rejection_workorder_for_stage
    :parameters (?secondary_stage_instance - secondary_stage_instance ?rejection_hold_point - rejection_hold_point ?rework_workorder - work_order)
    :precondition
      (and
        (entity_open ?secondary_stage_instance)
        (stage_has_rejection_hold_point ?secondary_stage_instance ?rejection_hold_point)
        (work_order_available ?rework_workorder)
        (not
          (secondary_stage_prepared ?secondary_stage_instance)
        )
      )
    :effect
      (and
        (rejection_workorder_issued ?rejection_hold_point)
        (secondary_stage_prepared ?secondary_stage_instance)
        (work_order_for_secondary_stage ?secondary_stage_instance ?rework_workorder)
        (not
          (work_order_available ?rework_workorder)
        )
      )
  )
  (:action complete_rejection_and_update_stage
    :parameters (?secondary_stage_instance - secondary_stage_instance ?rejection_hold_point - rejection_hold_point ?process_step - process_step ?rework_workorder - work_order)
    :precondition
      (and
        (entity_open ?secondary_stage_instance)
        (entity_assigned_to_process_step ?secondary_stage_instance ?process_step)
        (stage_has_rejection_hold_point ?secondary_stage_instance ?rejection_hold_point)
        (rejection_workorder_issued ?rejection_hold_point)
        (work_order_for_secondary_stage ?secondary_stage_instance ?rework_workorder)
        (not
          (secondary_stage_completed ?secondary_stage_instance)
        )
      )
    :effect
      (and
        (rejection_hold_raised ?rejection_hold_point)
        (secondary_stage_completed ?secondary_stage_instance)
        (work_order_available ?rework_workorder)
        (not
          (work_order_for_secondary_stage ?secondary_stage_instance ?rework_workorder)
        )
      )
  )
  (:action allocate_packaging_unit_for_finalization
    :parameters (?primary_stage_instance - primary_stage_instance ?secondary_stage_instance - secondary_stage_instance ?rework_hold_point - rework_hold_point ?rejection_hold_point - rejection_hold_point ?finished_packaging_unit - packaging_unit_category)
    :precondition
      (and
        (primary_stage_prepared ?primary_stage_instance)
        (secondary_stage_prepared ?secondary_stage_instance)
        (stage_has_rework_hold_point ?primary_stage_instance ?rework_hold_point)
        (stage_has_rejection_hold_point ?secondary_stage_instance ?rejection_hold_point)
        (rework_hold_raised ?rework_hold_point)
        (rejection_hold_raised ?rejection_hold_point)
        (primary_stage_completed ?primary_stage_instance)
        (secondary_stage_completed ?secondary_stage_instance)
        (packaging_unit_available ?finished_packaging_unit)
      )
    :effect
      (and
        (packaging_unit_allocated ?finished_packaging_unit)
        (packaging_unit_assigned_rework_hold ?finished_packaging_unit ?rework_hold_point)
        (packaging_unit_assigned_rejection_hold ?finished_packaging_unit ?rejection_hold_point)
        (not
          (packaging_unit_available ?finished_packaging_unit)
        )
      )
  )
  (:action allocate_packaging_unit_with_rework_flag
    :parameters (?primary_stage_instance - primary_stage_instance ?secondary_stage_instance - secondary_stage_instance ?rework_hold_point - rework_hold_point ?rejection_hold_point - rejection_hold_point ?finished_packaging_unit - packaging_unit_category)
    :precondition
      (and
        (primary_stage_prepared ?primary_stage_instance)
        (secondary_stage_prepared ?secondary_stage_instance)
        (stage_has_rework_hold_point ?primary_stage_instance ?rework_hold_point)
        (stage_has_rejection_hold_point ?secondary_stage_instance ?rejection_hold_point)
        (work_order_issued ?rework_hold_point)
        (rejection_hold_raised ?rejection_hold_point)
        (not
          (primary_stage_completed ?primary_stage_instance)
        )
        (secondary_stage_completed ?secondary_stage_instance)
        (packaging_unit_available ?finished_packaging_unit)
      )
    :effect
      (and
        (packaging_unit_allocated ?finished_packaging_unit)
        (packaging_unit_assigned_rework_hold ?finished_packaging_unit ?rework_hold_point)
        (packaging_unit_assigned_rejection_hold ?finished_packaging_unit ?rejection_hold_point)
        (packaging_unit_rework_flag ?finished_packaging_unit)
        (not
          (packaging_unit_available ?finished_packaging_unit)
        )
      )
  )
  (:action allocate_packaging_unit_with_rejection_flag
    :parameters (?primary_stage_instance - primary_stage_instance ?secondary_stage_instance - secondary_stage_instance ?rework_hold_point - rework_hold_point ?rejection_hold_point - rejection_hold_point ?finished_packaging_unit - packaging_unit_category)
    :precondition
      (and
        (primary_stage_prepared ?primary_stage_instance)
        (secondary_stage_prepared ?secondary_stage_instance)
        (stage_has_rework_hold_point ?primary_stage_instance ?rework_hold_point)
        (stage_has_rejection_hold_point ?secondary_stage_instance ?rejection_hold_point)
        (rework_hold_raised ?rework_hold_point)
        (rejection_workorder_issued ?rejection_hold_point)
        (primary_stage_completed ?primary_stage_instance)
        (not
          (secondary_stage_completed ?secondary_stage_instance)
        )
        (packaging_unit_available ?finished_packaging_unit)
      )
    :effect
      (and
        (packaging_unit_allocated ?finished_packaging_unit)
        (packaging_unit_assigned_rework_hold ?finished_packaging_unit ?rework_hold_point)
        (packaging_unit_assigned_rejection_hold ?finished_packaging_unit ?rejection_hold_point)
        (packaging_unit_rejection_flag ?finished_packaging_unit)
        (not
          (packaging_unit_available ?finished_packaging_unit)
        )
      )
  )
  (:action allocate_packaging_unit_with_both_flags
    :parameters (?primary_stage_instance - primary_stage_instance ?secondary_stage_instance - secondary_stage_instance ?rework_hold_point - rework_hold_point ?rejection_hold_point - rejection_hold_point ?finished_packaging_unit - packaging_unit_category)
    :precondition
      (and
        (primary_stage_prepared ?primary_stage_instance)
        (secondary_stage_prepared ?secondary_stage_instance)
        (stage_has_rework_hold_point ?primary_stage_instance ?rework_hold_point)
        (stage_has_rejection_hold_point ?secondary_stage_instance ?rejection_hold_point)
        (work_order_issued ?rework_hold_point)
        (rejection_workorder_issued ?rejection_hold_point)
        (not
          (primary_stage_completed ?primary_stage_instance)
        )
        (not
          (secondary_stage_completed ?secondary_stage_instance)
        )
        (packaging_unit_available ?finished_packaging_unit)
      )
    :effect
      (and
        (packaging_unit_allocated ?finished_packaging_unit)
        (packaging_unit_assigned_rework_hold ?finished_packaging_unit ?rework_hold_point)
        (packaging_unit_assigned_rejection_hold ?finished_packaging_unit ?rejection_hold_point)
        (packaging_unit_rework_flag ?finished_packaging_unit)
        (packaging_unit_rejection_flag ?finished_packaging_unit)
        (not
          (packaging_unit_available ?finished_packaging_unit)
        )
      )
  )
  (:action seal_packaging_unit
    :parameters (?finished_packaging_unit - packaging_unit_category ?primary_stage_instance - primary_stage_instance ?process_step - process_step)
    :precondition
      (and
        (packaging_unit_allocated ?finished_packaging_unit)
        (primary_stage_prepared ?primary_stage_instance)
        (entity_assigned_to_process_step ?primary_stage_instance ?process_step)
        (not
          (packaging_unit_packaged ?finished_packaging_unit)
        )
      )
    :effect (packaging_unit_packaged ?finished_packaging_unit)
  )
  (:action collect_and_register_qc_sample
    :parameters (?batch_record - batch_record ?qc_sample - qc_sample ?finished_packaging_unit - packaging_unit_category)
    :precondition
      (and
        (entity_open ?batch_record)
        (record_covers_packaging_unit ?batch_record ?finished_packaging_unit)
        (record_has_sample ?batch_record ?qc_sample)
        (qc_sample_available ?qc_sample)
        (packaging_unit_allocated ?finished_packaging_unit)
        (packaging_unit_packaged ?finished_packaging_unit)
        (not
          (qc_sample_tested ?qc_sample)
        )
      )
    :effect
      (and
        (qc_sample_tested ?qc_sample)
        (qc_sample_assigned_to_packaging_unit ?qc_sample ?finished_packaging_unit)
        (not
          (qc_sample_available ?qc_sample)
        )
      )
  )
  (:action initiate_sample_analysis
    :parameters (?batch_record - batch_record ?qc_sample - qc_sample ?finished_packaging_unit - packaging_unit_category ?process_step - process_step)
    :precondition
      (and
        (entity_open ?batch_record)
        (record_has_sample ?batch_record ?qc_sample)
        (qc_sample_tested ?qc_sample)
        (qc_sample_assigned_to_packaging_unit ?qc_sample ?finished_packaging_unit)
        (entity_assigned_to_process_step ?batch_record ?process_step)
        (not
          (packaging_unit_rework_flag ?finished_packaging_unit)
        )
        (not
          (record_marked_for_analysis ?batch_record)
        )
      )
    :effect (record_marked_for_analysis ?batch_record)
  )
  (:action assign_label_asset_to_record
    :parameters (?batch_record - batch_record ?label_asset - label_asset)
    :precondition
      (and
        (entity_open ?batch_record)
        (label_asset_available ?label_asset)
        (not
          (label_assigned_to_record ?batch_record)
        )
      )
    :effect
      (and
        (label_assigned_to_record ?batch_record)
        (record_has_label_asset ?batch_record ?label_asset)
        (not
          (label_asset_available ?label_asset)
        )
      )
  )
  (:action apply_label_and_prepare_record_for_analysis
    :parameters (?batch_record - batch_record ?qc_sample - qc_sample ?finished_packaging_unit - packaging_unit_category ?process_step - process_step ?label_asset - label_asset)
    :precondition
      (and
        (entity_open ?batch_record)
        (record_has_sample ?batch_record ?qc_sample)
        (qc_sample_tested ?qc_sample)
        (qc_sample_assigned_to_packaging_unit ?qc_sample ?finished_packaging_unit)
        (entity_assigned_to_process_step ?batch_record ?process_step)
