--- domain.pddl (as passed to VAL) ---
(define (domain frontline_and_support_positioning)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity - object role_asset - entity resource_asset - entity position_asset - entity member_slot_parent - entity party_member_slot - member_slot_parent role_token - role_asset capability_token - role_asset assignment_module - role_asset tactic_marker - role_asset specialization_token - role_asset consumable_item - role_asset support_power - role_asset status_token - role_asset shared_resource - resource_asset equipment_module - resource_asset target_marker - resource_asset frontline_position - position_asset support_position - position_asset encounter_objective - position_asset unit_slot - party_member_slot support_slot - party_member_slot frontline_unit - unit_slot support_unit - unit_slot coordinator_node - support_slot)

  (:predicates
    (actor_onboarded ?party_member_slot - party_member_slot)
    (assignment_confirmed ?party_member_slot - party_member_slot)
    (role_token_bound ?party_member_slot - party_member_slot)
    (actor_deployed ?party_member_slot - party_member_slot)
    (actor_activation_flag ?party_member_slot - party_member_slot)
    (actor_consumable_applied ?party_member_slot - party_member_slot)
    (role_token_available ?role_token - role_token)
    (role_token_assigned ?party_member_slot - party_member_slot ?role_token - role_token)
    (capability_available ?capability_token - capability_token)
    (actor_has_capability ?party_member_slot - party_member_slot ?capability_token - capability_token)
    (assignment_module_available ?assignment_module - assignment_module)
    (module_assigned ?party_member_slot - party_member_slot ?assignment_module - assignment_module)
    (shared_resource_available ?shared_resource - shared_resource)
    (frontline_has_resource ?frontline_unit - frontline_unit ?shared_resource - shared_resource)
    (support_has_resource ?support_unit - support_unit ?shared_resource - shared_resource)
    (frontline_position_assigned ?frontline_unit - frontline_unit ?frontline_position - frontline_position)
    (position_hold_flag ?frontline_position - frontline_position)
    (position_assist_flag ?frontline_position - frontline_position)
    (frontline_ready ?frontline_unit - frontline_unit)
    (support_position_assigned ?support_unit - support_unit ?support_position - support_position)
    (support_hold_flag ?support_position - support_position)
    (support_assist_flag ?support_position - support_position)
    (support_ready ?support_unit - support_unit)
    (objective_available ?encounter_objective - encounter_objective)
    (objective_engaged ?encounter_objective - encounter_objective)
    (objective_link_frontline ?encounter_objective - encounter_objective ?frontline_position - frontline_position)
    (objective_link_support ?encounter_objective - encounter_objective ?support_position - support_position)
    (objective_stage_alpha ?encounter_objective - encounter_objective)
    (objective_stage_beta ?encounter_objective - encounter_objective)
    (objective_committed ?encounter_objective - encounter_objective)
    (coordinator_frontline_link ?coordinator_node - coordinator_node ?frontline_unit - frontline_unit)
    (coordinator_support_link ?coordinator_node - coordinator_node ?support_unit - support_unit)
    (coordinator_objective_link ?coordinator_node - coordinator_node ?encounter_objective - encounter_objective)
    (equipment_available ?equipment_module - equipment_module)
    (coordinator_has_equipment ?coordinator_node - coordinator_node ?equipment_module - equipment_module)
    (equipment_allocated ?equipment_module - equipment_module)
    (equipment_linked_objective ?equipment_module - equipment_module ?encounter_objective - encounter_objective)
    (coordinator_equipment_ready ?coordinator_node - coordinator_node)
    (coordinator_power_prepared ?coordinator_node - coordinator_node)
    (coordinator_activation_prereq_one ?coordinator_node - coordinator_node)
    (coordinator_tactic_enabled ?coordinator_node - coordinator_node)
    (coordinator_tactic_stage ?coordinator_node - coordinator_node)
    (coordinator_activation_prereq_two ?coordinator_node - coordinator_node)
    (coordinator_fully_prepared ?coordinator_node - coordinator_node)
    (target_marker_available ?target_marker - target_marker)
    (coordinator_target_link ?coordinator_node - coordinator_node ?target_marker - target_marker)
    (coordinator_target_locked ?coordinator_node - coordinator_node)
    (coordinator_target_prepped ?coordinator_node - coordinator_node)
    (coordinator_target_finalized ?coordinator_node - coordinator_node)
    (tactic_marker_available ?tactic_marker - tactic_marker)
    (coordinator_has_tactic_marker ?coordinator_node - coordinator_node ?tactic_marker - tactic_marker)
    (specialization_available ?specialization_token - specialization_token)
    (coordinator_specialization_assigned ?coordinator_node - coordinator_node ?specialization_token - specialization_token)
    (support_power_available ?support_power - support_power)
    (coordinator_has_support_power ?coordinator_node - coordinator_node ?support_power - support_power)
    (status_token_available ?status_token - status_token)
    (coordinator_status_assigned ?coordinator_node - coordinator_node ?status_token - status_token)
    (consumable_available ?consumable_item - consumable_item)
    (actor_has_consumable ?party_member_slot - party_member_slot ?consumable_item - consumable_item)
    (frontline_staged ?frontline_unit - frontline_unit)
    (support_staged ?support_unit - support_unit)
    (coordinator_committed ?coordinator_node - coordinator_node)
  )
  (:action onboard_member
    :parameters (?party_member_slot - party_member_slot)
    :precondition
      (and
        (not
          (actor_onboarded ?party_member_slot)
        )
        (not
          (actor_deployed ?party_member_slot)
        )
      )
    :effect (actor_onboarded ?party_member_slot)
  )
  (:action assign_role_token
    :parameters (?party_member_slot - party_member_slot ?role_token - role_token)
    :precondition
      (and
        (actor_onboarded ?party_member_slot)
        (not
          (role_token_bound ?party_member_slot)
        )
        (role_token_available ?role_token)
      )
    :effect
      (and
        (role_token_bound ?party_member_slot)
        (role_token_assigned ?party_member_slot ?role_token)
        (not
          (role_token_available ?role_token)
        )
      )
  )
  (:action verify_and_assign_capability
    :parameters (?party_member_slot - party_member_slot ?capability_token - capability_token)
    :precondition
      (and
        (actor_onboarded ?party_member_slot)
        (role_token_bound ?party_member_slot)
        (capability_available ?capability_token)
      )
    :effect
      (and
        (actor_has_capability ?party_member_slot ?capability_token)
        (not
          (capability_available ?capability_token)
        )
      )
  )
  (:action confirm_role_assignment
    :parameters (?party_member_slot - party_member_slot ?capability_token - capability_token)
    :precondition
      (and
        (actor_onboarded ?party_member_slot)
        (role_token_bound ?party_member_slot)
        (actor_has_capability ?party_member_slot ?capability_token)
        (not
          (assignment_confirmed ?party_member_slot)
        )
      )
    :effect (assignment_confirmed ?party_member_slot)
  )
  (:action unassign_capability
    :parameters (?party_member_slot - party_member_slot ?capability_token - capability_token)
    :precondition
      (and
        (actor_has_capability ?party_member_slot ?capability_token)
      )
    :effect
      (and
        (capability_available ?capability_token)
        (not
          (actor_has_capability ?party_member_slot ?capability_token)
        )
      )
  )
  (:action attach_assignment_module
    :parameters (?party_member_slot - party_member_slot ?assignment_module - assignment_module)
    :precondition
      (and
        (assignment_confirmed ?party_member_slot)
        (assignment_module_available ?assignment_module)
      )
    :effect
      (and
        (module_assigned ?party_member_slot ?assignment_module)
        (not
          (assignment_module_available ?assignment_module)
        )
      )
  )
  (:action detach_assignment_module
    :parameters (?party_member_slot - party_member_slot ?assignment_module - assignment_module)
    :precondition
      (and
        (module_assigned ?party_member_slot ?assignment_module)
      )
    :effect
      (and
        (assignment_module_available ?assignment_module)
        (not
          (module_assigned ?party_member_slot ?assignment_module)
        )
      )
  )
  (:action attach_support_power
    :parameters (?coordinator_node - coordinator_node ?support_power - support_power)
    :precondition
      (and
        (assignment_confirmed ?coordinator_node)
        (support_power_available ?support_power)
      )
    :effect
      (and
        (coordinator_has_support_power ?coordinator_node ?support_power)
        (not
          (support_power_available ?support_power)
        )
      )
  )
  (:action detach_support_power
    :parameters (?coordinator_node - coordinator_node ?support_power - support_power)
    :precondition
      (and
        (coordinator_has_support_power ?coordinator_node ?support_power)
      )
    :effect
      (and
        (support_power_available ?support_power)
        (not
          (coordinator_has_support_power ?coordinator_node ?support_power)
        )
      )
  )
  (:action attach_status_token
    :parameters (?coordinator_node - coordinator_node ?status_token - status_token)
    :precondition
      (and
        (assignment_confirmed ?coordinator_node)
        (status_token_available ?status_token)
      )
    :effect
      (and
        (coordinator_status_assigned ?coordinator_node ?status_token)
        (not
          (status_token_available ?status_token)
        )
      )
  )
  (:action detach_status_token
    :parameters (?coordinator_node - coordinator_node ?status_token - status_token)
    :precondition
      (and
        (coordinator_status_assigned ?coordinator_node ?status_token)
      )
    :effect
      (and
        (status_token_available ?status_token)
        (not
          (coordinator_status_assigned ?coordinator_node ?status_token)
        )
      )
  )
  (:action claim_frontline_position
    :parameters (?frontline_unit - frontline_unit ?frontline_position - frontline_position ?capability_token - capability_token)
    :precondition
      (and
        (assignment_confirmed ?frontline_unit)
        (actor_has_capability ?frontline_unit ?capability_token)
        (frontline_position_assigned ?frontline_unit ?frontline_position)
        (not
          (position_hold_flag ?frontline_position)
        )
        (not
          (position_assist_flag ?frontline_position)
        )
      )
    :effect (position_hold_flag ?frontline_position)
  )
  (:action secure_frontline_position
    :parameters (?frontline_unit - frontline_unit ?frontline_position - frontline_position ?assignment_module - assignment_module)
    :precondition
      (and
        (assignment_confirmed ?frontline_unit)
        (module_assigned ?frontline_unit ?assignment_module)
        (frontline_position_assigned ?frontline_unit ?frontline_position)
        (position_hold_flag ?frontline_position)
        (not
          (frontline_staged ?frontline_unit)
        )
      )
    :effect
      (and
        (frontline_staged ?frontline_unit)
        (frontline_ready ?frontline_unit)
      )
  )
  (:action allocate_shared_resource_to_frontline
    :parameters (?frontline_unit - frontline_unit ?frontline_position - frontline_position ?shared_resource - shared_resource)
    :precondition
      (and
        (assignment_confirmed ?frontline_unit)
        (frontline_position_assigned ?frontline_unit ?frontline_position)
        (shared_resource_available ?shared_resource)
        (not
          (frontline_staged ?frontline_unit)
        )
      )
    :effect
      (and
        (position_assist_flag ?frontline_position)
        (frontline_staged ?frontline_unit)
        (frontline_has_resource ?frontline_unit ?shared_resource)
        (not
          (shared_resource_available ?shared_resource)
        )
      )
  )
  (:action execute_resource_assisted_hold
    :parameters (?frontline_unit - frontline_unit ?frontline_position - frontline_position ?capability_token - capability_token ?shared_resource - shared_resource)
    :precondition
      (and
        (assignment_confirmed ?frontline_unit)
        (actor_has_capability ?frontline_unit ?capability_token)
        (frontline_position_assigned ?frontline_unit ?frontline_position)
        (position_assist_flag ?frontline_position)
        (frontline_has_resource ?frontline_unit ?shared_resource)
        (not
          (frontline_ready ?frontline_unit)
        )
      )
    :effect
      (and
        (position_hold_flag ?frontline_position)
        (frontline_ready ?frontline_unit)
        (shared_resource_available ?shared_resource)
        (not
          (frontline_has_resource ?frontline_unit ?shared_resource)
        )
      )
  )
  (:action claim_support_position
    :parameters (?support_unit - support_unit ?support_position - support_position ?capability_token - capability_token)
    :precondition
      (and
        (assignment_confirmed ?support_unit)
        (actor_has_capability ?support_unit ?capability_token)
        (support_position_assigned ?support_unit ?support_position)
        (not
          (support_hold_flag ?support_position)
        )
        (not
          (support_assist_flag ?support_position)
        )
      )
    :effect (support_hold_flag ?support_position)
  )
  (:action secure_support_position
    :parameters (?support_unit - support_unit ?support_position - support_position ?assignment_module - assignment_module)
    :precondition
      (and
        (assignment_confirmed ?support_unit)
        (module_assigned ?support_unit ?assignment_module)
        (support_position_assigned ?support_unit ?support_position)
        (support_hold_flag ?support_position)
        (not
          (support_staged ?support_unit)
        )
      )
    :effect
      (and
        (support_staged ?support_unit)
        (support_ready ?support_unit)
      )
  )
  (:action allocate_shared_resource_to_support
    :parameters (?support_unit - support_unit ?support_position - support_position ?shared_resource - shared_resource)
    :precondition
      (and
        (assignment_confirmed ?support_unit)
        (support_position_assigned ?support_unit ?support_position)
        (shared_resource_available ?shared_resource)
        (not
          (support_staged ?support_unit)
        )
      )
    :effect
      (and
        (support_assist_flag ?support_position)
        (support_staged ?support_unit)
        (support_has_resource ?support_unit ?shared_resource)
        (not
          (shared_resource_available ?shared_resource)
        )
      )
  )
  (:action execute_resource_assisted_support
    :parameters (?support_unit - support_unit ?support_position - support_position ?capability_token - capability_token ?shared_resource - shared_resource)
    :precondition
      (and
        (assignment_confirmed ?support_unit)
        (actor_has_capability ?support_unit ?capability_token)
        (support_position_assigned ?support_unit ?support_position)
        (support_assist_flag ?support_position)
        (support_has_resource ?support_unit ?shared_resource)
        (not
          (support_ready ?support_unit)
        )
      )
    :effect
      (and
        (support_hold_flag ?support_position)
        (support_ready ?support_unit)
        (shared_resource_available ?shared_resource)
        (not
          (support_has_resource ?support_unit ?shared_resource)
        )
      )
  )
  (:action assemble_formation_primary
    :parameters (?frontline_unit - frontline_unit ?support_unit - support_unit ?frontline_position - frontline_position ?support_position - support_position ?encounter_objective - encounter_objective)
    :precondition
      (and
        (frontline_staged ?frontline_unit)
        (support_staged ?support_unit)
        (frontline_position_assigned ?frontline_unit ?frontline_position)
        (support_position_assigned ?support_unit ?support_position)
        (position_hold_flag ?frontline_position)
        (support_hold_flag ?support_position)
        (frontline_ready ?frontline_unit)
        (support_ready ?support_unit)
        (objective_available ?encounter_objective)
      )
    :effect
      (and
        (objective_engaged ?encounter_objective)
        (objective_link_frontline ?encounter_objective ?frontline_position)
        (objective_link_support ?encounter_objective ?support_position)
        (not
          (objective_available ?encounter_objective)
        )
      )
  )
  (:action assemble_formation_secondary
    :parameters (?frontline_unit - frontline_unit ?support_unit - support_unit ?frontline_position - frontline_position ?support_position - support_position ?encounter_objective - encounter_objective)
    :precondition
      (and
        (frontline_staged ?frontline_unit)
        (support_staged ?support_unit)
        (frontline_position_assigned ?frontline_unit ?frontline_position)
        (support_position_assigned ?support_unit ?support_position)
        (position_assist_flag ?frontline_position)
        (support_hold_flag ?support_position)
        (not
          (frontline_ready ?frontline_unit)
        )
        (support_ready ?support_unit)
        (objective_available ?encounter_objective)
      )
    :effect
      (and
        (objective_engaged ?encounter_objective)
        (objective_link_frontline ?encounter_objective ?frontline_position)
        (objective_link_support ?encounter_objective ?support_position)
        (objective_stage_alpha ?encounter_objective)
        (not
          (objective_available ?encounter_objective)
        )
      )
  )
  (:action assemble_formation_tertiary
    :parameters (?frontline_unit - frontline_unit ?support_unit - support_unit ?frontline_position - frontline_position ?support_position - support_position ?encounter_objective - encounter_objective)
    :precondition
      (and
        (frontline_staged ?frontline_unit)
        (support_staged ?support_unit)
        (frontline_position_assigned ?frontline_unit ?frontline_position)
        (support_position_assigned ?support_unit ?support_position)
        (position_hold_flag ?frontline_position)
        (support_assist_flag ?support_position)
        (frontline_ready ?frontline_unit)
        (not
          (support_ready ?support_unit)
        )
        (objective_available ?encounter_objective)
      )
    :effect
      (and
        (objective_engaged ?encounter_objective)
        (objective_link_frontline ?encounter_objective ?frontline_position)
        (objective_link_support ?encounter_objective ?support_position)
        (objective_stage_beta ?encounter_objective)
        (not
          (objective_available ?encounter_objective)
        )
      )
  )
  (:action assemble_formation_full
    :parameters (?frontline_unit - frontline_unit ?support_unit - support_unit ?frontline_position - frontline_position ?support_position - support_position ?encounter_objective - encounter_objective)
    :precondition
      (and
        (frontline_staged ?frontline_unit)
        (support_staged ?support_unit)
        (frontline_position_assigned ?frontline_unit ?frontline_position)
        (support_position_assigned ?support_unit ?support_position)
        (position_assist_flag ?frontline_position)
        (support_assist_flag ?support_position)
        (not
          (frontline_ready ?frontline_unit)
        )
        (not
          (support_ready ?support_unit)
        )
        (objective_available ?encounter_objective)
      )
    :effect
      (and
        (objective_engaged ?encounter_objective)
        (objective_link_frontline ?encounter_objective ?frontline_position)
        (objective_link_support ?encounter_objective ?support_position)
        (objective_stage_alpha ?encounter_objective)
        (objective_stage_beta ?encounter_objective)
        (not
          (objective_available ?encounter_objective)
        )
      )
  )
  (:action commit_encounter_objective
    :parameters (?encounter_objective - encounter_objective ?frontline_unit - frontline_unit ?capability_token - capability_token)
    :precondition
      (and
        (objective_engaged ?encounter_objective)
        (frontline_staged ?frontline_unit)
        (actor_has_capability ?frontline_unit ?capability_token)
        (not
          (objective_committed ?encounter_objective)
        )
      )
    :effect (objective_committed ?encounter_objective)
  )
  (:action attach_equipment_module
    :parameters (?coordinator_node - coordinator_node ?equipment_module - equipment_module ?encounter_objective - encounter_objective)
    :precondition
      (and
        (assignment_confirmed ?coordinator_node)
        (coordinator_objective_link ?coordinator_node ?encounter_objective)
        (coordinator_has_equipment ?coordinator_node ?equipment_module)
        (equipment_available ?equipment_module)
        (objective_engaged ?encounter_objective)
        (objective_committed ?encounter_objective)
        (not
          (equipment_allocated ?equipment_module)
        )
      )
    :effect
      (and
        (equipment_allocated ?equipment_module)
        (equipment_linked_objective ?equipment_module ?encounter_objective)
        (not
          (equipment_available ?equipment_module)
        )
      )
  )
  (:action validate_equipment_and_coordinator
    :parameters (?coordinator_node - coordinator_node ?equipment_module - equipment_module ?encounter_objective - encounter_objective ?capability_token - capability_token)
    :precondition
      (and
        (assignment_confirmed ?coordinator_node)
        (coordinator_has_equipment ?coordinator_node ?equipment_module)
        (equipment_allocated ?equipment_module)
        (equipment_linked_objective ?equipment_module ?encounter_objective)
        (actor_has_capability ?coordinator_node ?capability_token)
        (not
          (objective_stage_alpha ?encounter_objective)
        )
        (not
          (coordinator_equipment_ready ?coordinator_node)
        )
      )
    :effect (coordinator_equipment_ready ?coordinator_node)
  )
  (:action bind_tactic_marker_to_coordinator
    :parameters (?coordinator_node - coordinator_node ?tactic_marker - tactic_marker)
    :precondition
      (and
        (assignment_confirmed ?coordinator_node)
        (tactic_marker_available ?tactic_marker)
        (not
          (coordinator_tactic_enabled ?coordinator_node)
        )
      )
    :effect
      (and
        (coordinator_tactic_enabled ?coordinator_node)
        (coordinator_has_tactic_marker ?coordinator_node ?tactic_marker)
        (not
          (tactic_marker_available ?tactic_marker)
        )
      )
  )
  (:action apply_tactic_sequence
    :parameters (?coordinator_node - coordinator_node ?equipment_module - equipment_module ?encounter_objective - encounter_objective ?capability_token - capability_token ?tactic_marker - tactic_marker)
    :precondition
      (and
        (assignment_confirmed ?coordinator_node)
        (coordinator_has_equipment ?coordinator_node ?equipment_module)
        (equipment_allocated ?equipment_module)
        (equipment_linked_objective ?equipment_module ?encounter_objective)
        (actor_has_capability ?coordinator_node ?capability_token)
        (objective_stage_alpha ?encounter_objective)
        (coordinator_tactic_enabled ?coordinator_node)
        (coordinator_has_tactic_marker ?coordinator_node ?tactic_marker)
        (not
          (coordinator_equipment_ready ?coordinator_node)
        )
      )
    :effect
      (and
        (coordinator_equipment_ready ?coordinator_node)
        (coordinator_tactic_stage ?coordinator_node)
      )
  )
  (:action prepare_support_power_execution
    :parameters (?coordinator_node - coordinator_node ?support_power - support_power ?assignment_module - assignment_module ?equipment_module - equipment_module ?encounter_objective - encounter_objective)
    :precondition
      (and
        (coordinator_equipment_ready ?coordinator_node)
        (coordinator_has_support_power ?coordinator_node ?support_power)
        (module_assigned ?coordinator_node ?assignment_module)
        (coordinator_has_equipment ?coordinator_node ?equipment_module)
        (equipment_linked_objective ?equipment_module ?encounter_objective)
        (not
          (objective_stage_beta ?encounter_objective)
        )
        (not
          (coordinator_power_prepared ?coordinator_node)
        )
      )
    :effect (coordinator_power_prepared ?coordinator_node)
  )
  (:action confirm_support_power_execution
    :parameters (?coordinator_node - coordinator_node ?support_power - support_power ?assignment_module - assignment_module ?equipment_module - equipment_module ?encounter_objective - encounter_objective)
    :precondition
      (and
        (coordinator_equipment_ready ?coordinator_node)
        (coordinator_has_support_power ?coordinator_node ?support_power)
        (module_assigned ?coordinator_node ?assignment_module)
        (coordinator_has_equipment ?coordinator_node ?equipment_module)
        (equipment_linked_objective ?equipment_module ?encounter_objective)
        (objective_stage_beta ?encounter_objective)
        (not
          (coordinator_power_prepared ?coordinator_node)
        )
      )
    :effect (coordinator_power_prepared ?coordinator_node)
  )
  (:action prime_status_token_for_coordinator
    :parameters (?coordinator_node - coordinator_node ?status_token - status_token ?equipment_module - equipment_module ?encounter_objective - encounter_objective)
    :precondition
      (and
