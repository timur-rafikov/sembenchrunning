--- domain.pddl (as passed to VAL) ---
(define (domain reservation_modification_coordination)
  (:requirements :strips :typing :negative-preconditions)
  (:types booking_auxiliary - object itinerary_artifact - object provider_artifact - object booking_root - object booking - booking_root offer - booking_auxiliary payment_method - booking_auxiliary supplier_slot - booking_auxiliary optional_addon - booking_auxiliary fulfillment_profile - booking_auxiliary modification_token - booking_auxiliary payment_provider - booking_auxiliary billing_document - booking_auxiliary amendment_payload - itinerary_artifact fulfillment_token - itinerary_artifact loyalty_credential - itinerary_artifact provider_resource_primary - provider_artifact provider_resource_secondary - provider_artifact booking_document - provider_artifact booking_component - booking itinerary_component - booking service_instance_primary - booking_component service_instance_secondary - booking_component master_itinerary - itinerary_component)
  (:predicates
    (entity_open ?reservation - booking)
    (entity_confirmed ?reservation - booking)
    (entity_offer_selected ?reservation - booking)
    (entity_committed ?reservation - booking)
    (entity_finalized ?reservation - booking)
    (entity_modification_registered ?reservation - booking)
    (offer_available ?offer - offer)
    (entity_bound_offer ?reservation - booking ?offer - offer)
    (payment_method_available ?payment_method - payment_method)
    (entity_payment_attached ?reservation - booking ?payment_method - payment_method)
    (supplier_slot_available ?supplier_slot - supplier_slot)
    (entity_supplier_slot_reserved ?reservation - booking ?supplier_slot - supplier_slot)
    (amendment_payload_available ?amendment_payload - amendment_payload)
    (service_primary_amendment_applied ?service_instance_primary - service_instance_primary ?amendment_payload - amendment_payload)
    (service_secondary_amendment_applied ?service_instance_secondary - service_instance_secondary ?amendment_payload - amendment_payload)
    (service_primary_bound_provider_resource ?service_instance_primary - service_instance_primary ?provider_resource_primary - provider_resource_primary)
    (provider_resource_primary_provisionally_allocated ?provider_resource_primary - provider_resource_primary)
    (provider_resource_primary_pending ?provider_resource_primary - provider_resource_primary)
    (service_instance_primary_confirmed ?service_instance_primary - service_instance_primary)
    (service_secondary_bound_provider_resource ?service_instance_secondary - service_instance_secondary ?provider_resource_secondary - provider_resource_secondary)
    (provider_resource_secondary_provisionally_allocated ?provider_resource_secondary - provider_resource_secondary)
    (provider_resource_secondary_pending ?provider_resource_secondary - provider_resource_secondary)
    (service_instance_secondary_confirmed ?service_instance_secondary - service_instance_secondary)
    (booking_document_draft ?booking_document - booking_document)
    (booking_document_prepared ?booking_document - booking_document)
    (booking_document_resource_primary_linked ?booking_document - booking_document ?provider_resource_primary - provider_resource_primary)
    (booking_document_resource_secondary_linked ?booking_document - booking_document ?provider_resource_secondary - provider_resource_secondary)
    (booking_document_resource_primary_provisional ?booking_document - booking_document)
    (booking_document_resource_secondary_provisional ?booking_document - booking_document)
    (booking_document_issued ?booking_document - booking_document)
    (itinerary_includes_service_primary ?master_itinerary - master_itinerary ?service_instance_primary - service_instance_primary)
    (itinerary_includes_service_secondary ?master_itinerary - master_itinerary ?service_instance_secondary - service_instance_secondary)
    (itinerary_links_booking_document ?master_itinerary - master_itinerary ?booking_document - booking_document)
    (fulfillment_token_available ?fulfillment_token - fulfillment_token)
    (itinerary_has_fulfillment_token ?master_itinerary - master_itinerary ?fulfillment_token - fulfillment_token)
    (fulfillment_token_allocated ?fulfillment_token - fulfillment_token)
    (fulfillment_token_linked_booking_document ?fulfillment_token - fulfillment_token ?booking_document - booking_document)
    (itinerary_ready_for_supplier_checks ?master_itinerary - master_itinerary)
    (itinerary_supplier_confirmation_started ?master_itinerary - master_itinerary)
    (itinerary_supplier_checks_completed ?master_itinerary - master_itinerary)
    (itinerary_addon_attached ?master_itinerary - master_itinerary)
    (itinerary_addon_registered ?master_itinerary - master_itinerary)
    (itinerary_requires_profile_assignment ?master_itinerary - master_itinerary)
    (itinerary_profile_assigned ?master_itinerary - master_itinerary)
    (loyalty_credential_available ?loyalty_credential - loyalty_credential)
    (itinerary_loyalty_credential_linked ?master_itinerary - master_itinerary ?loyalty_credential - loyalty_credential)
    (itinerary_loyalty_applied ?master_itinerary - master_itinerary)
    (itinerary_loyalty_validated ?master_itinerary - master_itinerary)
    (itinerary_loyalty_finalized ?master_itinerary - master_itinerary)
    (optional_addon_available ?optional_addon - optional_addon)
    (itinerary_addon_linked ?master_itinerary - master_itinerary ?optional_addon - optional_addon)
    (fulfillment_profile_available ?fulfillment_profile - fulfillment_profile)
    (itinerary_fulfillment_profile_assigned ?master_itinerary - master_itinerary ?fulfillment_profile - fulfillment_profile)
    (payment_provider_available ?payment_provider - payment_provider)
    (itinerary_payment_provider_attached ?master_itinerary - master_itinerary ?payment_provider - payment_provider)
    (billing_document_available ?billing_document - billing_document)
    (itinerary_billing_document_attached ?master_itinerary - master_itinerary ?billing_document - billing_document)
    (modification_token_available ?modification_token - modification_token)
    (entity_modification_token_linked ?reservation - booking ?modification_token - modification_token)
    (service_instance_primary_ready ?service_instance_primary - service_instance_primary)
    (service_instance_secondary_ready ?service_instance_secondary - service_instance_secondary)
    (itinerary_finalized ?master_itinerary - master_itinerary)
  )
  (:action initiate_reservation
    :parameters (?reservation - booking)
    :precondition
      (and
        (not
          (entity_open ?reservation)
        )
        (not
          (entity_committed ?reservation)
        )
      )
    :effect (entity_open ?reservation)
  )
  (:action select_and_bind_offer
    :parameters (?reservation - booking ?offer - offer)
    :precondition
      (and
        (entity_open ?reservation)
        (not
          (entity_offer_selected ?reservation)
        )
        (offer_available ?offer)
      )
    :effect
      (and
        (entity_offer_selected ?reservation)
        (entity_bound_offer ?reservation ?offer)
        (not
          (offer_available ?offer)
        )
      )
  )
  (:action attach_payment_method_to_reservation
    :parameters (?reservation - booking ?payment_method - payment_method)
    :precondition
      (and
        (entity_open ?reservation)
        (entity_offer_selected ?reservation)
        (payment_method_available ?payment_method)
      )
    :effect
      (and
        (entity_payment_attached ?reservation ?payment_method)
        (not
          (payment_method_available ?payment_method)
        )
      )
  )
  (:action confirm_payment_for_reservation
    :parameters (?reservation - booking ?payment_method - payment_method)
    :precondition
      (and
        (entity_open ?reservation)
        (entity_offer_selected ?reservation)
        (entity_payment_attached ?reservation ?payment_method)
        (not
          (entity_confirmed ?reservation)
        )
      )
    :effect (entity_confirmed ?reservation)
  )
  (:action release_payment_method_from_reservation
    :parameters (?reservation - booking ?payment_method - payment_method)
    :precondition
      (and
        (entity_payment_attached ?reservation ?payment_method)
      )
    :effect
      (and
        (payment_method_available ?payment_method)
        (not
          (entity_payment_attached ?reservation ?payment_method)
        )
      )
  )
  (:action reserve_supplier_slot_for_reservation
    :parameters (?reservation - booking ?supplier_slot - supplier_slot)
    :precondition
      (and
        (entity_confirmed ?reservation)
        (supplier_slot_available ?supplier_slot)
      )
    :effect
      (and
        (entity_supplier_slot_reserved ?reservation ?supplier_slot)
        (not
          (supplier_slot_available ?supplier_slot)
        )
      )
  )
  (:action release_supplier_slot_from_reservation
    :parameters (?reservation - booking ?supplier_slot - supplier_slot)
    :precondition
      (and
        (entity_supplier_slot_reserved ?reservation ?supplier_slot)
      )
    :effect
      (and
        (supplier_slot_available ?supplier_slot)
        (not
          (entity_supplier_slot_reserved ?reservation ?supplier_slot)
        )
      )
  )
  (:action attach_payment_provider_to_itinerary
    :parameters (?master_itinerary - master_itinerary ?payment_provider - payment_provider)
    :precondition
      (and
        (entity_confirmed ?master_itinerary)
        (payment_provider_available ?payment_provider)
      )
    :effect
      (and
        (itinerary_payment_provider_attached ?master_itinerary ?payment_provider)
        (not
          (payment_provider_available ?payment_provider)
        )
      )
  )
  (:action detach_payment_provider_from_itinerary
    :parameters (?master_itinerary - master_itinerary ?payment_provider - payment_provider)
    :precondition
      (and
        (itinerary_payment_provider_attached ?master_itinerary ?payment_provider)
      )
    :effect
      (and
        (payment_provider_available ?payment_provider)
        (not
          (itinerary_payment_provider_attached ?master_itinerary ?payment_provider)
        )
      )
  )
  (:action attach_billing_document_to_itinerary
    :parameters (?master_itinerary - master_itinerary ?billing_document - billing_document)
    :precondition
      (and
        (entity_confirmed ?master_itinerary)
        (billing_document_available ?billing_document)
      )
    :effect
      (and
        (itinerary_billing_document_attached ?master_itinerary ?billing_document)
        (not
          (billing_document_available ?billing_document)
        )
      )
  )
  (:action detach_billing_document_from_itinerary
    :parameters (?master_itinerary - master_itinerary ?billing_document - billing_document)
    :precondition
      (and
        (itinerary_billing_document_attached ?master_itinerary ?billing_document)
      )
    :effect
      (and
        (billing_document_available ?billing_document)
        (not
          (itinerary_billing_document_attached ?master_itinerary ?billing_document)
        )
      )
  )
  (:action initiate_provider_resource_primary_hold
    :parameters (?service_instance_primary - service_instance_primary ?provider_resource_primary - provider_resource_primary ?payment_method - payment_method)
    :precondition
      (and
        (entity_confirmed ?service_instance_primary)
        (entity_payment_attached ?service_instance_primary ?payment_method)
        (service_primary_bound_provider_resource ?service_instance_primary ?provider_resource_primary)
        (not
          (provider_resource_primary_provisionally_allocated ?provider_resource_primary)
        )
        (not
          (provider_resource_primary_pending ?provider_resource_primary)
        )
      )
    :effect (provider_resource_primary_provisionally_allocated ?provider_resource_primary)
  )
  (:action confirm_supplier_slot_for_service_primary
    :parameters (?service_instance_primary - service_instance_primary ?provider_resource_primary - provider_resource_primary ?supplier_slot - supplier_slot)
    :precondition
      (and
        (entity_confirmed ?service_instance_primary)
        (entity_supplier_slot_reserved ?service_instance_primary ?supplier_slot)
        (service_primary_bound_provider_resource ?service_instance_primary ?provider_resource_primary)
        (provider_resource_primary_provisionally_allocated ?provider_resource_primary)
        (not
          (service_instance_primary_ready ?service_instance_primary)
        )
      )
    :effect
      (and
        (service_instance_primary_ready ?service_instance_primary)
        (service_instance_primary_confirmed ?service_instance_primary)
      )
  )
  (:action apply_amendment_payload_to_service_primary
    :parameters (?service_instance_primary - service_instance_primary ?provider_resource_primary - provider_resource_primary ?amendment_payload - amendment_payload)
    :precondition
      (and
        (entity_confirmed ?service_instance_primary)
        (service_primary_bound_provider_resource ?service_instance_primary ?provider_resource_primary)
        (amendment_payload_available ?amendment_payload)
        (not
          (service_instance_primary_ready ?service_instance_primary)
        )
      )
    :effect
      (and
        (provider_resource_primary_pending ?provider_resource_primary)
        (service_instance_primary_ready ?service_instance_primary)
        (service_primary_amendment_applied ?service_instance_primary ?amendment_payload)
        (not
          (amendment_payload_available ?amendment_payload)
        )
      )
  )
  (:action finalize_amendment_for_service_primary
    :parameters (?service_instance_primary - service_instance_primary ?provider_resource_primary - provider_resource_primary ?payment_method - payment_method ?amendment_payload - amendment_payload)
    :precondition
      (and
        (entity_confirmed ?service_instance_primary)
        (entity_payment_attached ?service_instance_primary ?payment_method)
        (service_primary_bound_provider_resource ?service_instance_primary ?provider_resource_primary)
        (provider_resource_primary_pending ?provider_resource_primary)
        (service_primary_amendment_applied ?service_instance_primary ?amendment_payload)
        (not
          (service_instance_primary_confirmed ?service_instance_primary)
        )
      )
    :effect
      (and
        (provider_resource_primary_provisionally_allocated ?provider_resource_primary)
        (service_instance_primary_confirmed ?service_instance_primary)
        (amendment_payload_available ?amendment_payload)
        (not
          (service_primary_amendment_applied ?service_instance_primary ?amendment_payload)
        )
      )
  )
  (:action initiate_provider_resource_secondary_hold
    :parameters (?service_instance_secondary - service_instance_secondary ?provider_resource_secondary - provider_resource_secondary ?payment_method - payment_method)
    :precondition
      (and
        (entity_confirmed ?service_instance_secondary)
        (entity_payment_attached ?service_instance_secondary ?payment_method)
        (service_secondary_bound_provider_resource ?service_instance_secondary ?provider_resource_secondary)
        (not
          (provider_resource_secondary_provisionally_allocated ?provider_resource_secondary)
        )
        (not
          (provider_resource_secondary_pending ?provider_resource_secondary)
        )
      )
    :effect (provider_resource_secondary_provisionally_allocated ?provider_resource_secondary)
  )
  (:action confirm_supplier_slot_for_service_secondary
    :parameters (?service_instance_secondary - service_instance_secondary ?provider_resource_secondary - provider_resource_secondary ?supplier_slot - supplier_slot)
    :precondition
      (and
        (entity_confirmed ?service_instance_secondary)
        (entity_supplier_slot_reserved ?service_instance_secondary ?supplier_slot)
        (service_secondary_bound_provider_resource ?service_instance_secondary ?provider_resource_secondary)
        (provider_resource_secondary_provisionally_allocated ?provider_resource_secondary)
        (not
          (service_instance_secondary_ready ?service_instance_secondary)
        )
      )
    :effect
      (and
        (service_instance_secondary_ready ?service_instance_secondary)
        (service_instance_secondary_confirmed ?service_instance_secondary)
      )
  )
  (:action apply_amendment_payload_to_service_secondary
    :parameters (?service_instance_secondary - service_instance_secondary ?provider_resource_secondary - provider_resource_secondary ?amendment_payload - amendment_payload)
    :precondition
      (and
        (entity_confirmed ?service_instance_secondary)
        (service_secondary_bound_provider_resource ?service_instance_secondary ?provider_resource_secondary)
        (amendment_payload_available ?amendment_payload)
        (not
          (service_instance_secondary_ready ?service_instance_secondary)
        )
      )
    :effect
      (and
        (provider_resource_secondary_pending ?provider_resource_secondary)
        (service_instance_secondary_ready ?service_instance_secondary)
        (service_secondary_amendment_applied ?service_instance_secondary ?amendment_payload)
        (not
          (amendment_payload_available ?amendment_payload)
        )
      )
  )
  (:action finalize_amendment_for_service_secondary
    :parameters (?service_instance_secondary - service_instance_secondary ?provider_resource_secondary - provider_resource_secondary ?payment_method - payment_method ?amendment_payload - amendment_payload)
    :precondition
      (and
        (entity_confirmed ?service_instance_secondary)
        (entity_payment_attached ?service_instance_secondary ?payment_method)
        (service_secondary_bound_provider_resource ?service_instance_secondary ?provider_resource_secondary)
        (provider_resource_secondary_pending ?provider_resource_secondary)
        (service_secondary_amendment_applied ?service_instance_secondary ?amendment_payload)
        (not
          (service_instance_secondary_confirmed ?service_instance_secondary)
        )
      )
    :effect
      (and
        (provider_resource_secondary_provisionally_allocated ?provider_resource_secondary)
        (service_instance_secondary_confirmed ?service_instance_secondary)
        (amendment_payload_available ?amendment_payload)
        (not
          (service_secondary_amendment_applied ?service_instance_secondary ?amendment_payload)
        )
      )
  )
  (:action construct_booking_document_with_confirmed_resources
    :parameters (?service_instance_primary - service_instance_primary ?service_instance_secondary - service_instance_secondary ?provider_resource_primary - provider_resource_primary ?provider_resource_secondary - provider_resource_secondary ?booking_document - booking_document)
    :precondition
      (and
        (service_instance_primary_ready ?service_instance_primary)
        (service_instance_secondary_ready ?service_instance_secondary)
        (service_primary_bound_provider_resource ?service_instance_primary ?provider_resource_primary)
        (service_secondary_bound_provider_resource ?service_instance_secondary ?provider_resource_secondary)
        (provider_resource_primary_provisionally_allocated ?provider_resource_primary)
        (provider_resource_secondary_provisionally_allocated ?provider_resource_secondary)
        (service_instance_primary_confirmed ?service_instance_primary)
        (service_instance_secondary_confirmed ?service_instance_secondary)
        (booking_document_draft ?booking_document)
      )
    :effect
      (and
        (booking_document_prepared ?booking_document)
        (booking_document_resource_primary_linked ?booking_document ?provider_resource_primary)
        (booking_document_resource_secondary_linked ?booking_document ?provider_resource_secondary)
        (not
          (booking_document_draft ?booking_document)
        )
      )
  )
  (:action construct_booking_document_with_provisional_resource_primary
    :parameters (?service_instance_primary - service_instance_primary ?service_instance_secondary - service_instance_secondary ?provider_resource_primary - provider_resource_primary ?provider_resource_secondary - provider_resource_secondary ?booking_document - booking_document)
    :precondition
      (and
        (service_instance_primary_ready ?service_instance_primary)
        (service_instance_secondary_ready ?service_instance_secondary)
        (service_primary_bound_provider_resource ?service_instance_primary ?provider_resource_primary)
        (service_secondary_bound_provider_resource ?service_instance_secondary ?provider_resource_secondary)
        (provider_resource_primary_pending ?provider_resource_primary)
        (provider_resource_secondary_provisionally_allocated ?provider_resource_secondary)
        (not
          (service_instance_primary_confirmed ?service_instance_primary)
        )
        (service_instance_secondary_confirmed ?service_instance_secondary)
        (booking_document_draft ?booking_document)
      )
    :effect
      (and
        (booking_document_prepared ?booking_document)
        (booking_document_resource_primary_linked ?booking_document ?provider_resource_primary)
        (booking_document_resource_secondary_linked ?booking_document ?provider_resource_secondary)
        (booking_document_resource_primary_provisional ?booking_document)
        (not
          (booking_document_draft ?booking_document)
        )
      )
  )
  (:action construct_booking_document_with_provisional_resource_secondary
    :parameters (?service_instance_primary - service_instance_primary ?service_instance_secondary - service_instance_secondary ?provider_resource_primary - provider_resource_primary ?provider_resource_secondary - provider_resource_secondary ?booking_document - booking_document)
    :precondition
      (and
        (service_instance_primary_ready ?service_instance_primary)
        (service_instance_secondary_ready ?service_instance_secondary)
        (service_primary_bound_provider_resource ?service_instance_primary ?provider_resource_primary)
        (service_secondary_bound_provider_resource ?service_instance_secondary ?provider_resource_secondary)
        (provider_resource_primary_provisionally_allocated ?provider_resource_primary)
        (provider_resource_secondary_pending ?provider_resource_secondary)
        (service_instance_primary_confirmed ?service_instance_primary)
        (not
          (service_instance_secondary_confirmed ?service_instance_secondary)
        )
        (booking_document_draft ?booking_document)
      )
    :effect
      (and
        (booking_document_prepared ?booking_document)
        (booking_document_resource_primary_linked ?booking_document ?provider_resource_primary)
        (booking_document_resource_secondary_linked ?booking_document ?provider_resource_secondary)
        (booking_document_resource_secondary_provisional ?booking_document)
        (not
          (booking_document_draft ?booking_document)
        )
      )
  )
  (:action construct_booking_document_with_provisional_resources
    :parameters (?service_instance_primary - service_instance_primary ?service_instance_secondary - service_instance_secondary ?provider_resource_primary - provider_resource_primary ?provider_resource_secondary - provider_resource_secondary ?booking_document - booking_document)
    :precondition
      (and
        (service_instance_primary_ready ?service_instance_primary)
        (service_instance_secondary_ready ?service_instance_secondary)
        (service_primary_bound_provider_resource ?service_instance_primary ?provider_resource_primary)
        (service_secondary_bound_provider_resource ?service_instance_secondary ?provider_resource_secondary)
        (provider_resource_primary_pending ?provider_resource_primary)
        (provider_resource_secondary_pending ?provider_resource_secondary)
        (not
          (service_instance_primary_confirmed ?service_instance_primary)
        )
        (not
          (service_instance_secondary_confirmed ?service_instance_secondary)
        )
        (booking_document_draft ?booking_document)
      )
    :effect
      (and
        (booking_document_prepared ?booking_document)
        (booking_document_resource_primary_linked ?booking_document ?provider_resource_primary)
        (booking_document_resource_secondary_linked ?booking_document ?provider_resource_secondary)
        (booking_document_resource_primary_provisional ?booking_document)
        (booking_document_resource_secondary_provisional ?booking_document)
        (not
          (booking_document_draft ?booking_document)
        )
      )
  )
  (:action issue_booking_document
    :parameters (?booking_document - booking_document ?service_instance_primary - service_instance_primary ?payment_method - payment_method)
    :precondition
      (and
        (booking_document_prepared ?booking_document)
        (service_instance_primary_ready ?service_instance_primary)
        (entity_payment_attached ?service_instance_primary ?payment_method)
        (not
          (booking_document_issued ?booking_document)
        )
      )
    :effect (booking_document_issued ?booking_document)
  )
  (:action allocate_fulfillment_token_to_itinerary
    :parameters (?master_itinerary - master_itinerary ?fulfillment_token - fulfillment_token ?booking_document - booking_document)
    :precondition
      (and
        (entity_confirmed ?master_itinerary)
        (itinerary_links_booking_document ?master_itinerary ?booking_document)
        (itinerary_has_fulfillment_token ?master_itinerary ?fulfillment_token)
        (fulfillment_token_available ?fulfillment_token)
        (booking_document_prepared ?booking_document)
        (booking_document_issued ?booking_document)
        (not
          (fulfillment_token_allocated ?fulfillment_token)
        )
      )
    :effect
      (and
        (fulfillment_token_allocated ?fulfillment_token)
        (fulfillment_token_linked_booking_document ?fulfillment_token ?booking_document)
        (not
          (fulfillment_token_available ?fulfillment_token)
        )
      )
  )
  (:action allocate_and_prepare_fulfillment_token
    :parameters (?master_itinerary - master_itinerary ?fulfillment_token - fulfillment_token ?booking_document - booking_document ?payment_method - payment_method)
    :precondition
      (and
        (entity_confirmed ?master_itinerary)
        (itinerary_has_fulfillment_token ?master_itinerary ?fulfillment_token)
        (fulfillment_token_allocated ?fulfillment_token)
        (fulfillment_token_linked_booking_document ?fulfillment_token ?booking_document)
        (entity_payment_attached ?master_itinerary ?payment_method)
        (not
          (booking_document_resource_primary_provisional ?booking_document)
        )
        (not
          (itinerary_ready_for_supplier_checks ?master_itinerary)
        )
      )
    :effect (itinerary_ready_for_supplier_checks ?master_itinerary)
  )
  (:action attach_optional_addon_to_itinerary
    :parameters (?master_itinerary - master_itinerary ?optional_addon - optional_addon)
    :precondition
      (and
        (entity_confirmed ?master_itinerary)
        (optional_addon_available ?optional_addon)
        (not
          (itinerary_addon_attached ?master_itinerary)
        )
      )
    :effect
      (and
        (itinerary_addon_attached ?master_itinerary)
        (itinerary_addon_linked ?master_itinerary ?optional_addon)
        (not
          (optional_addon_available ?optional_addon)
        )
      )
  )
  (:action register_optional_addon_on_itinerary
