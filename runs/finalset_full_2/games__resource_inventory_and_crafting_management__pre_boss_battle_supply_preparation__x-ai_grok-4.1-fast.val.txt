--- domain.pddl (as passed to VAL) ---
(define (domain pre_boss_supply_preparation)
  (:requirements :strips :typing :negative-preconditions)
  (:types item_category - object supply_family - object modifier_family - object preparation_target_type - object preparation_target - preparation_target_type resource_token - item_category consumable_class - item_category tool_class - item_category optional_modifier - item_category performance_chip - item_category consumable_container - item_category weapon_attachment - item_category augment_module - item_category consumable_instance - supply_family material_unit - supply_family unique_token - supply_family attachment_point - modifier_family upgrade_node - modifier_family gear_piece - modifier_family primary_role_group - preparation_target support_role_group - preparation_target primary_unit - primary_role_group secondary_unit - primary_role_group loadout_profile - support_role_group)
  (:predicates
    (preparation_target_registered ?preparation_target - preparation_target)
    (preparation_ready ?preparation_target - preparation_target)
    (resource_reserved ?preparation_target - preparation_target)
    (prepared_for_battle ?preparation_target - preparation_target)
    (equipment_applied_to_target ?preparation_target - preparation_target)
    (consumable_applied_to_target ?preparation_target - preparation_target)
    (resource_token_available ?resource_token - resource_token)
    (resource_allocated_to_target ?preparation_target - preparation_target ?resource_token - resource_token)
    (consumable_available ?consumable_class - consumable_class)
    (consumable_assigned_to_target ?preparation_target - preparation_target ?consumable_class - consumable_class)
    (tool_available ?tool_class - tool_class)
    (tool_attached_to_target ?preparation_target - preparation_target ?tool_class - tool_class)
    (consumable_instance_available ?consumable_instance - consumable_instance)
    (primary_consumable_allocated ?primary_unit - primary_unit ?consumable_instance - consumable_instance)
    (secondary_consumable_allocated ?secondary_unit - secondary_unit ?consumable_instance - consumable_instance)
    (primary_has_attachment_point ?primary_unit - primary_unit ?attachment_point - attachment_point)
    (attachment_point_reserved ?attachment_point - attachment_point)
    (attachment_point_material_loaded ?attachment_point - attachment_point)
    (primary_attachment_confirmed ?primary_unit - primary_unit)
    (secondary_has_upgrade_node ?secondary_unit - secondary_unit ?upgrade_node - upgrade_node)
    (upgrade_node_reserved ?upgrade_node - upgrade_node)
    (upgrade_node_material_loaded ?upgrade_node - upgrade_node)
    (secondary_attachment_confirmed ?secondary_unit - secondary_unit)
    (gear_pending_assembly ?gear_piece - gear_piece)
    (gear_assembly_started ?gear_piece - gear_piece)
    (gear_bound_to_attachment_point ?gear_piece - gear_piece ?attachment_point - attachment_point)
    (gear_bound_to_upgrade_node ?gear_piece - gear_piece ?upgrade_node - upgrade_node)
    (gear_quality_flag_1 ?gear_piece - gear_piece)
    (gear_quality_flag_2 ?gear_piece - gear_piece)
    (gear_finalized ?gear_piece - gear_piece)
    (profile_links_primary_unit ?loadout_profile - loadout_profile ?primary_unit - primary_unit)
    (profile_links_secondary_unit ?loadout_profile - loadout_profile ?secondary_unit - secondary_unit)
    (profile_contains_gear ?loadout_profile - loadout_profile ?gear_piece - gear_piece)
    (material_available ?material_unit - material_unit)
    (profile_has_material ?loadout_profile - loadout_profile ?material_unit - material_unit)
    (material_consumed ?material_unit - material_unit)
    (material_bound_to_gear ?material_unit - material_unit ?gear_piece - gear_piece)
    (profile_materials_locked ?loadout_profile - loadout_profile)
    (modifier_stage_ready ?loadout_profile - loadout_profile)
    (modifier_final_stage_ready ?loadout_profile - loadout_profile)
    (optional_modifier_attached ?loadout_profile - loadout_profile)
    (optional_modifier_stage_two_applied ?loadout_profile - loadout_profile)
    (performance_chip_staged ?loadout_profile - loadout_profile)
    (modifications_integrated ?loadout_profile - loadout_profile)
    (unique_token_available ?unique_token - unique_token)
    (profile_linked_unique_token ?loadout_profile - loadout_profile ?unique_token - unique_token)
    (unique_token_installed ?loadout_profile - loadout_profile)
    (unique_token_staged ?loadout_profile - loadout_profile)
    (unique_token_finalized ?loadout_profile - loadout_profile)
    (optional_modifier_available ?optional_modifier - optional_modifier)
    (profile_has_optional_modifier ?loadout_profile - loadout_profile ?optional_modifier - optional_modifier)
    (performance_chip_available ?performance_chip - performance_chip)
    (profile_has_performance_chip ?loadout_profile - loadout_profile ?performance_chip - performance_chip)
    (weapon_attachment_available ?weapon_attachment - weapon_attachment)
    (profile_has_weapon_attachment ?loadout_profile - loadout_profile ?weapon_attachment - weapon_attachment)
    (augment_module_available ?augment_module - augment_module)
    (profile_has_augment_module ?loadout_profile - loadout_profile ?augment_module - augment_module)
    (consumable_container_available ?consumable_container - consumable_container)
    (consumable_container_assigned ?preparation_target - preparation_target ?consumable_container - consumable_container)
    (primary_unit_ready ?primary_unit - primary_unit)
    (secondary_unit_ready ?secondary_unit - secondary_unit)
    (profile_confirmed ?loadout_profile - loadout_profile)
  )
  (:action register_preparation_target
    :parameters (?preparation_target - preparation_target)
    :precondition
      (and
        (not
          (preparation_target_registered ?preparation_target)
        )
        (not
          (prepared_for_battle ?preparation_target)
        )
      )
    :effect (preparation_target_registered ?preparation_target)
  )
  (:action reserve_resource_for_preparation_target
    :parameters (?preparation_target - preparation_target ?resource_token - resource_token)
    :precondition
      (and
        (preparation_target_registered ?preparation_target)
        (not
          (resource_reserved ?preparation_target)
        )
        (resource_token_available ?resource_token)
      )
    :effect
      (and
        (resource_reserved ?preparation_target)
        (resource_allocated_to_target ?preparation_target ?resource_token)
        (not
          (resource_token_available ?resource_token)
        )
      )
  )
  (:action assign_consumable_to_preparation_target
    :parameters (?preparation_target - preparation_target ?consumable_class - consumable_class)
    :precondition
      (and
        (preparation_target_registered ?preparation_target)
        (resource_reserved ?preparation_target)
        (consumable_available ?consumable_class)
      )
    :effect
      (and
        (consumable_assigned_to_target ?preparation_target ?consumable_class)
        (not
          (consumable_available ?consumable_class)
        )
      )
  )
  (:action confirm_consumable_assignment
    :parameters (?preparation_target - preparation_target ?consumable_class - consumable_class)
    :precondition
      (and
        (preparation_target_registered ?preparation_target)
        (resource_reserved ?preparation_target)
        (consumable_assigned_to_target ?preparation_target ?consumable_class)
        (not
          (preparation_ready ?preparation_target)
        )
      )
    :effect (preparation_ready ?preparation_target)
  )
  (:action unassign_consumable_from_preparation_target
    :parameters (?preparation_target - preparation_target ?consumable_class - consumable_class)
    :precondition
      (and
        (consumable_assigned_to_target ?preparation_target ?consumable_class)
      )
    :effect
      (and
        (consumable_available ?consumable_class)
        (not
          (consumable_assigned_to_target ?preparation_target ?consumable_class)
        )
      )
  )
  (:action equip_tool_on_preparation_target
    :parameters (?preparation_target - preparation_target ?tool_class - tool_class)
    :precondition
      (and
        (preparation_ready ?preparation_target)
        (tool_available ?tool_class)
      )
    :effect
      (and
        (tool_attached_to_target ?preparation_target ?tool_class)
        (not
          (tool_available ?tool_class)
        )
      )
  )
  (:action unequip_tool_from_preparation_target
    :parameters (?preparation_target - preparation_target ?tool_class - tool_class)
    :precondition
      (and
        (tool_attached_to_target ?preparation_target ?tool_class)
      )
    :effect
      (and
        (tool_available ?tool_class)
        (not
          (tool_attached_to_target ?preparation_target ?tool_class)
        )
      )
  )
  (:action attach_weapon_attachment_to_profile
    :parameters (?loadout_profile - loadout_profile ?weapon_attachment - weapon_attachment)
    :precondition
      (and
        (preparation_ready ?loadout_profile)
        (weapon_attachment_available ?weapon_attachment)
      )
    :effect
      (and
        (profile_has_weapon_attachment ?loadout_profile ?weapon_attachment)
        (not
          (weapon_attachment_available ?weapon_attachment)
        )
      )
  )
  (:action detach_weapon_attachment_from_profile
    :parameters (?loadout_profile - loadout_profile ?weapon_attachment - weapon_attachment)
    :precondition
      (and
        (profile_has_weapon_attachment ?loadout_profile ?weapon_attachment)
      )
    :effect
      (and
        (weapon_attachment_available ?weapon_attachment)
        (not
          (profile_has_weapon_attachment ?loadout_profile ?weapon_attachment)
        )
      )
  )
  (:action attach_augment_module_to_profile
    :parameters (?loadout_profile - loadout_profile ?augment_module - augment_module)
    :precondition
      (and
        (preparation_ready ?loadout_profile)
        (augment_module_available ?augment_module)
      )
    :effect
      (and
        (profile_has_augment_module ?loadout_profile ?augment_module)
        (not
          (augment_module_available ?augment_module)
        )
      )
  )
  (:action detach_augment_module_from_profile
    :parameters (?loadout_profile - loadout_profile ?augment_module - augment_module)
    :precondition
      (and
        (profile_has_augment_module ?loadout_profile ?augment_module)
      )
    :effect
      (and
        (augment_module_available ?augment_module)
        (not
          (profile_has_augment_module ?loadout_profile ?augment_module)
        )
      )
  )
  (:action reserve_attachment_point_for_primary
    :parameters (?primary_unit - primary_unit ?attachment_point - attachment_point ?consumable_class - consumable_class)
    :precondition
      (and
        (preparation_ready ?primary_unit)
        (consumable_assigned_to_target ?primary_unit ?consumable_class)
        (primary_has_attachment_point ?primary_unit ?attachment_point)
        (not
          (attachment_point_reserved ?attachment_point)
        )
        (not
          (attachment_point_material_loaded ?attachment_point)
        )
      )
    :effect (attachment_point_reserved ?attachment_point)
  )
  (:action equip_tool_for_primary_attachment_point
    :parameters (?primary_unit - primary_unit ?attachment_point - attachment_point ?tool_class - tool_class)
    :precondition
      (and
        (preparation_ready ?primary_unit)
        (tool_attached_to_target ?primary_unit ?tool_class)
        (primary_has_attachment_point ?primary_unit ?attachment_point)
        (attachment_point_reserved ?attachment_point)
        (not
          (primary_unit_ready ?primary_unit)
        )
      )
    :effect
      (and
        (primary_unit_ready ?primary_unit)
        (primary_attachment_confirmed ?primary_unit)
      )
  )
  (:action load_material_on_primary_attachment_point
    :parameters (?primary_unit - primary_unit ?attachment_point - attachment_point ?consumable_instance - consumable_instance)
    :precondition
      (and
        (preparation_ready ?primary_unit)
        (primary_has_attachment_point ?primary_unit ?attachment_point)
        (consumable_instance_available ?consumable_instance)
        (not
          (primary_unit_ready ?primary_unit)
        )
      )
    :effect
      (and
        (attachment_point_material_loaded ?attachment_point)
        (primary_unit_ready ?primary_unit)
        (primary_consumable_allocated ?primary_unit ?consumable_instance)
        (not
          (consumable_instance_available ?consumable_instance)
        )
      )
  )
  (:action finalize_primary_attachment
    :parameters (?primary_unit - primary_unit ?attachment_point - attachment_point ?consumable_class - consumable_class ?consumable_instance - consumable_instance)
    :precondition
      (and
        (preparation_ready ?primary_unit)
        (consumable_assigned_to_target ?primary_unit ?consumable_class)
        (primary_has_attachment_point ?primary_unit ?attachment_point)
        (attachment_point_material_loaded ?attachment_point)
        (primary_consumable_allocated ?primary_unit ?consumable_instance)
        (not
          (primary_attachment_confirmed ?primary_unit)
        )
      )
    :effect
      (and
        (attachment_point_reserved ?attachment_point)
        (primary_attachment_confirmed ?primary_unit)
        (consumable_instance_available ?consumable_instance)
        (not
          (primary_consumable_allocated ?primary_unit ?consumable_instance)
        )
      )
  )
  (:action reserve_upgrade_node_for_secondary
    :parameters (?secondary_unit - secondary_unit ?upgrade_node - upgrade_node ?consumable_class - consumable_class)
    :precondition
      (and
        (preparation_ready ?secondary_unit)
        (consumable_assigned_to_target ?secondary_unit ?consumable_class)
        (secondary_has_upgrade_node ?secondary_unit ?upgrade_node)
        (not
          (upgrade_node_reserved ?upgrade_node)
        )
        (not
          (upgrade_node_material_loaded ?upgrade_node)
        )
      )
    :effect (upgrade_node_reserved ?upgrade_node)
  )
  (:action equip_tool_for_secondary_upgrade_node
    :parameters (?secondary_unit - secondary_unit ?upgrade_node - upgrade_node ?tool_class - tool_class)
    :precondition
      (and
        (preparation_ready ?secondary_unit)
        (tool_attached_to_target ?secondary_unit ?tool_class)
        (secondary_has_upgrade_node ?secondary_unit ?upgrade_node)
        (upgrade_node_reserved ?upgrade_node)
        (not
          (secondary_unit_ready ?secondary_unit)
        )
      )
    :effect
      (and
        (secondary_unit_ready ?secondary_unit)
        (secondary_attachment_confirmed ?secondary_unit)
      )
  )
  (:action load_material_on_secondary_upgrade_node
    :parameters (?secondary_unit - secondary_unit ?upgrade_node - upgrade_node ?consumable_instance - consumable_instance)
    :precondition
      (and
        (preparation_ready ?secondary_unit)
        (secondary_has_upgrade_node ?secondary_unit ?upgrade_node)
        (consumable_instance_available ?consumable_instance)
        (not
          (secondary_unit_ready ?secondary_unit)
        )
      )
    :effect
      (and
        (upgrade_node_material_loaded ?upgrade_node)
        (secondary_unit_ready ?secondary_unit)
        (secondary_consumable_allocated ?secondary_unit ?consumable_instance)
        (not
          (consumable_instance_available ?consumable_instance)
        )
      )
  )
  (:action finalize_secondary_upgrade
    :parameters (?secondary_unit - secondary_unit ?upgrade_node - upgrade_node ?consumable_class - consumable_class ?consumable_instance - consumable_instance)
    :precondition
      (and
        (preparation_ready ?secondary_unit)
        (consumable_assigned_to_target ?secondary_unit ?consumable_class)
        (secondary_has_upgrade_node ?secondary_unit ?upgrade_node)
        (upgrade_node_material_loaded ?upgrade_node)
        (secondary_consumable_allocated ?secondary_unit ?consumable_instance)
        (not
          (secondary_attachment_confirmed ?secondary_unit)
        )
      )
    :effect
      (and
        (upgrade_node_reserved ?upgrade_node)
        (secondary_attachment_confirmed ?secondary_unit)
        (consumable_instance_available ?consumable_instance)
        (not
          (secondary_consumable_allocated ?secondary_unit ?consumable_instance)
        )
      )
  )
  (:action assemble_gear_standard
    :parameters (?primary_unit - primary_unit ?secondary_unit - secondary_unit ?attachment_point - attachment_point ?upgrade_node - upgrade_node ?gear_piece - gear_piece)
    :precondition
      (and
        (primary_unit_ready ?primary_unit)
        (secondary_unit_ready ?secondary_unit)
        (primary_has_attachment_point ?primary_unit ?attachment_point)
        (secondary_has_upgrade_node ?secondary_unit ?upgrade_node)
        (attachment_point_reserved ?attachment_point)
        (upgrade_node_reserved ?upgrade_node)
        (primary_attachment_confirmed ?primary_unit)
        (secondary_attachment_confirmed ?secondary_unit)
        (gear_pending_assembly ?gear_piece)
      )
    :effect
      (and
        (gear_assembly_started ?gear_piece)
        (gear_bound_to_attachment_point ?gear_piece ?attachment_point)
        (gear_bound_to_upgrade_node ?gear_piece ?upgrade_node)
        (not
          (gear_pending_assembly ?gear_piece)
        )
      )
  )
  (:action assemble_gear_quality_a
    :parameters (?primary_unit - primary_unit ?secondary_unit - secondary_unit ?attachment_point - attachment_point ?upgrade_node - upgrade_node ?gear_piece - gear_piece)
    :precondition
      (and
        (primary_unit_ready ?primary_unit)
        (secondary_unit_ready ?secondary_unit)
        (primary_has_attachment_point ?primary_unit ?attachment_point)
        (secondary_has_upgrade_node ?secondary_unit ?upgrade_node)
        (attachment_point_material_loaded ?attachment_point)
        (upgrade_node_reserved ?upgrade_node)
        (not
          (primary_attachment_confirmed ?primary_unit)
        )
        (secondary_attachment_confirmed ?secondary_unit)
        (gear_pending_assembly ?gear_piece)
      )
    :effect
      (and
        (gear_assembly_started ?gear_piece)
        (gear_bound_to_attachment_point ?gear_piece ?attachment_point)
        (gear_bound_to_upgrade_node ?gear_piece ?upgrade_node)
        (gear_quality_flag_1 ?gear_piece)
        (not
          (gear_pending_assembly ?gear_piece)
        )
      )
  )
  (:action assemble_gear_quality_b
    :parameters (?primary_unit - primary_unit ?secondary_unit - secondary_unit ?attachment_point - attachment_point ?upgrade_node - upgrade_node ?gear_piece - gear_piece)
    :precondition
      (and
        (primary_unit_ready ?primary_unit)
        (secondary_unit_ready ?secondary_unit)
        (primary_has_attachment_point ?primary_unit ?attachment_point)
        (secondary_has_upgrade_node ?secondary_unit ?upgrade_node)
        (attachment_point_reserved ?attachment_point)
        (upgrade_node_material_loaded ?upgrade_node)
        (primary_attachment_confirmed ?primary_unit)
        (not
          (secondary_attachment_confirmed ?secondary_unit)
        )
        (gear_pending_assembly ?gear_piece)
      )
    :effect
      (and
        (gear_assembly_started ?gear_piece)
        (gear_bound_to_attachment_point ?gear_piece ?attachment_point)
        (gear_bound_to_upgrade_node ?gear_piece ?upgrade_node)
        (gear_quality_flag_2 ?gear_piece)
        (not
          (gear_pending_assembly ?gear_piece)
        )
      )
  )
  (:action assemble_gear_quality_a_and_b
    :parameters (?primary_unit - primary_unit ?secondary_unit - secondary_unit ?attachment_point - attachment_point ?upgrade_node - upgrade_node ?gear_piece - gear_piece)
    :precondition
      (and
        (primary_unit_ready ?primary_unit)
        (secondary_unit_ready ?secondary_unit)
        (primary_has_attachment_point ?primary_unit ?attachment_point)
        (secondary_has_upgrade_node ?secondary_unit ?upgrade_node)
        (attachment_point_material_loaded ?attachment_point)
        (upgrade_node_material_loaded ?upgrade_node)
        (not
          (primary_attachment_confirmed ?primary_unit)
        )
        (not
          (secondary_attachment_confirmed ?secondary_unit)
        )
        (gear_pending_assembly ?gear_piece)
      )
    :effect
      (and
        (gear_assembly_started ?gear_piece)
        (gear_bound_to_attachment_point ?gear_piece ?attachment_point)
        (gear_bound_to_upgrade_node ?gear_piece ?upgrade_node)
        (gear_quality_flag_1 ?gear_piece)
        (gear_quality_flag_2 ?gear_piece)
        (not
          (gear_pending_assembly ?gear_piece)
        )
      )
  )
  (:action finalize_gear_processing
    :parameters (?gear_piece - gear_piece ?primary_unit - primary_unit ?consumable_class - consumable_class)
    :precondition
      (and
        (gear_assembly_started ?gear_piece)
        (primary_unit_ready ?primary_unit)
        (consumable_assigned_to_target ?primary_unit ?consumable_class)
        (not
          (gear_finalized ?gear_piece)
        )
      )
    :effect (gear_finalized ?gear_piece)
  )
  (:action consume_material_for_gear
    :parameters (?loadout_profile - loadout_profile ?material_unit - material_unit ?gear_piece - gear_piece)
    :precondition
      (and
        (preparation_ready ?loadout_profile)
        (profile_contains_gear ?loadout_profile ?gear_piece)
        (profile_has_material ?loadout_profile ?material_unit)
        (material_available ?material_unit)
        (gear_assembly_started ?gear_piece)
        (gear_finalized ?gear_piece)
        (not
          (material_consumed ?material_unit)
        )
      )
    :effect
      (and
        (material_consumed ?material_unit)
        (material_bound_to_gear ?material_unit ?gear_piece)
        (not
          (material_available ?material_unit)
        )
      )
  )
  (:action finalize_material_in_profile
    :parameters (?loadout_profile - loadout_profile ?material_unit - material_unit ?gear_piece - gear_piece ?consumable_class - consumable_class)
    :precondition
      (and
        (preparation_ready ?loadout_profile)
        (profile_has_material ?loadout_profile ?material_unit)
        (material_consumed ?material_unit)
        (material_bound_to_gear ?material_unit ?gear_piece)
        (consumable_assigned_to_target ?loadout_profile ?consumable_class)
        (not
          (gear_quality_flag_1 ?gear_piece)
        )
        (not
          (profile_materials_locked ?loadout_profile)
        )
      )
    :effect (profile_materials_locked ?loadout_profile)
  )
  (:action attach_optional_modifier_to_profile
    :parameters (?loadout_profile - loadout_profile ?optional_modifier - optional_modifier)
    :precondition
      (and
        (preparation_ready ?loadout_profile)
        (optional_modifier_available ?optional_modifier)
        (not
          (optional_modifier_attached ?loadout_profile)
        )
      )
    :effect
      (and
        (optional_modifier_attached ?loadout_profile)
        (profile_has_optional_modifier ?loadout_profile ?optional_modifier)
        (not
          (optional_modifier_available ?optional_modifier)
        )
      )
  )
  (:action stage_optional_modifier_installation
    :parameters (?loadout_profile - loadout_profile ?material_unit - material_unit ?gear_piece - gear_piece ?consumable_class - consumable_class ?optional_modifier - optional_modifier)
    :precondition
      (and
        (preparation_ready ?loadout_profile)
        (profile_has_material ?loadout_profile ?material_unit)
        (material_consumed ?material_unit)
        (material_bound_to_gear ?material_unit ?gear_piece)
        (consumable_assigned_to_target ?loadout_profile ?consumable_class)
        (gear_quality_flag_1 ?gear_piece)
        (optional_modifier_attached ?loadout_profile)
        (profile_has_optional_modifier ?loadout_profile ?optional_modifier)
        (not
          (profile_materials_locked ?loadout_profile)
        )
      )
    :effect
      (and
        (profile_materials_locked ?loadout_profile)
        (optional_modifier_stage_two_applied ?loadout_profile)
      )
  )
  (:action apply_modifier_weapon_attachment_step1
    :parameters (?loadout_profile - loadout_profile ?weapon_attachment - weapon_attachment ?tool_class - tool_class ?material_unit - material_unit ?gear_piece - gear_piece)
    :precondition
      (and
        (profile_materials_locked ?loadout_profile)
        (profile_has_weapon_attachment ?loadout_profile ?weapon_attachment)
        (tool_attached_to_target ?loadout_profile ?tool_class)
        (profile_has_material ?loadout_profile ?material_unit)
        (material_bound_to_gear ?material_unit ?gear_piece)
        (not
          (gear_quality_flag_2 ?gear_piece)
        )
        (not
          (modifier_stage_ready ?loadout_profile)
        )
      )
    :effect (modifier_stage_ready ?loadout_profile)
  )
  (:action apply_modifier_weapon_attachment_step2
    :parameters (?loadout_profile - loadout_profile ?weapon_attachment - weapon_attachment ?tool_class - tool_class ?material_unit - material_unit ?gear_piece - gear_piece)
    :precondition
      (and
        (profile_materials_locked ?loadout_profile)
        (profile_has_weapon_attachment ?loadout_profile ?weapon_attachment)
        (tool_attached_to_target ?loadout_profile ?tool_class)
        (profile_has_material ?loadout_profile ?material_unit)
        (material_bound_to_gear ?material_unit ?gear_piece)
        (gear_quality_flag_2 ?gear_piece)
        (not
          (modifier_stage_ready ?loadout_profile)
        )
      )
    :effect (modifier_stage_ready ?loadout_profile)
