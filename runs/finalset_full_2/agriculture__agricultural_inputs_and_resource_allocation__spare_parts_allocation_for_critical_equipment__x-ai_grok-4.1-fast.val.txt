--- domain.pddl (as passed to VAL) ---
(define (domain agriculture_spare_parts_allocation)
  (:requirements :strips :typing :negative-preconditions)
  (:types inventory_category - object supply_batch_category - object logistics_category - object asset_hierarchy - object critical_equipment - asset_hierarchy spare_part_item - inventory_category fault_code - inventory_category technician - inventory_category priority_tag - inventory_category tool_set - inventory_category contingency_reserve - inventory_category critical_subpart - inventory_category emergency_part - inventory_category consumable_kit - supply_batch_category quality_certificate - supply_batch_category vendor_authorization - supply_batch_category local_transport - logistics_category regional_transport - logistics_category repair_kit - logistics_category field_equipment_group - critical_equipment support_entity - critical_equipment primary_equipment_unit - field_equipment_group secondary_equipment_unit - field_equipment_group service_station - support_entity)
  (:predicates
    (diagnostic_report_created ?critical_equipment - critical_equipment)
    (entity_maintenance_authorization_granted ?critical_equipment - critical_equipment)
    (spare_part_source_assigned ?critical_equipment - critical_equipment)
    (entity_operational ?critical_equipment - critical_equipment)
    (entity_ready_for_redeployment ?critical_equipment - critical_equipment)
    (entity_approved_for_redeployment ?critical_equipment - critical_equipment)
    (spare_part_available ?spare_part_item - spare_part_item)
    (spare_reserved_for_entity ?critical_equipment - critical_equipment ?spare_part_item - spare_part_item)
    (unlinked_fault_report ?fault_code - fault_code)
    (fault_assigned_to_entity ?critical_equipment - critical_equipment ?fault_code - fault_code)
    (technician_available ?technician - technician)
    (technician_assigned_to_entity ?critical_equipment - critical_equipment ?technician - technician)
    (consumable_available ?consumable_kit - consumable_kit)
    (consumable_allocated_to_primary_unit ?primary_equipment_unit - primary_equipment_unit ?consumable_kit - consumable_kit)
    (consumable_allocated_to_secondary_unit ?secondary_equipment_unit - secondary_equipment_unit ?consumable_kit - consumable_kit)
    (primary_unit_assigned_local_transport ?primary_equipment_unit - primary_equipment_unit ?local_transport - local_transport)
    (local_transport_staged ?local_transport - local_transport)
    (local_transport_reserved ?local_transport - local_transport)
    (primary_unit_local_transport_confirmed ?primary_equipment_unit - primary_equipment_unit)
    (secondary_unit_assigned_regional_transport ?secondary_equipment_unit - secondary_equipment_unit ?regional_transport - regional_transport)
    (regional_transport_staged ?regional_transport - regional_transport)
    (regional_transport_reserved ?regional_transport - regional_transport)
    (secondary_unit_regional_transport_confirmed ?secondary_equipment_unit - secondary_equipment_unit)
    (repair_kit_available ?repair_kit - repair_kit)
    (repair_kit_assembled ?repair_kit - repair_kit)
    (repair_kit_assigned_local_transport ?repair_kit - repair_kit ?local_transport - local_transport)
    (repair_kit_assigned_regional_transport ?repair_kit - repair_kit ?regional_transport - regional_transport)
    (repair_kit_local_route_confirmed ?repair_kit - repair_kit)
    (repair_kit_regional_route_confirmed ?repair_kit - repair_kit)
    (repair_kit_locked_for_dispatch ?repair_kit - repair_kit)
    (service_station_supports_primary_unit ?service_station - service_station ?primary_equipment_unit - primary_equipment_unit)
    (service_station_supports_secondary_unit ?service_station - service_station ?secondary_equipment_unit - secondary_equipment_unit)
    (service_station_stores_repair_kit ?service_station - service_station ?repair_kit - repair_kit)
    (quality_certificate_available ?quality_certificate - quality_certificate)
    (service_station_has_certificate ?service_station - service_station ?quality_certificate - quality_certificate)
    (quality_certificate_validated ?quality_certificate - quality_certificate)
    (certificate_attached_to_repair_kit ?quality_certificate - quality_certificate ?repair_kit - repair_kit)
    (inspection_stage_certificate_verified ?service_station - service_station)
    (inspection_stage_subpart_verified ?service_station - service_station)
    (inspection_passed ?service_station - service_station)
    (station_priority_assigned ?service_station - service_station)
    (inspection_record_created ?service_station - service_station)
    (inspection_authorization_applied ?service_station - service_station)
    (maintenance_executed ?service_station - service_station)
    (vendor_authorization_available ?vendor_authorization - vendor_authorization)
    (vendor_authorization_linked_to_station ?service_station - service_station ?vendor_authorization - vendor_authorization)
    (vendor_clearance_acknowledged ?service_station - service_station)
    (vendor_task_assigned ?service_station - service_station)
    (vendor_task_ready_for_completion ?service_station - service_station)
    (priority_tag_available ?priority_tag - priority_tag)
    (priority_tag_assigned_to_station ?service_station - service_station ?priority_tag - priority_tag)
    (tool_set_available ?tool_set - tool_set)
    (tool_set_allocated_to_station ?service_station - service_station ?tool_set - tool_set)
    (critical_subpart_available ?critical_subpart - critical_subpart)
    (critical_subpart_allocated_to_station ?service_station - service_station ?critical_subpart - critical_subpart)
    (emergency_part_available ?emergency_part - emergency_part)
    (emergency_part_allocated_to_station ?service_station - service_station ?emergency_part - emergency_part)
    (contingency_reserve_available ?contingency_reserve - contingency_reserve)
    (contingency_reserved_for_entity ?critical_equipment - critical_equipment ?contingency_reserve - contingency_reserve)
    (primary_unit_prepared ?primary_equipment_unit - primary_equipment_unit)
    (secondary_unit_prepared ?secondary_equipment_unit - secondary_equipment_unit)
    (station_ready_for_deployment ?service_station - service_station)
  )
  (:action open_diagnostic_report
    :parameters (?critical_equipment - critical_equipment)
    :precondition
      (and
        (not
          (diagnostic_report_created ?critical_equipment)
        )
        (not
          (entity_operational ?critical_equipment)
        )
      )
    :effect (diagnostic_report_created ?critical_equipment)
  )
  (:action reserve_spare_part_source_for_equipment
    :parameters (?critical_equipment - critical_equipment ?spare_part_item - spare_part_item)
    :precondition
      (and
        (diagnostic_report_created ?critical_equipment)
        (not
          (spare_part_source_assigned ?critical_equipment)
        )
        (spare_part_available ?spare_part_item)
      )
    :effect
      (and
        (spare_part_source_assigned ?critical_equipment)
        (spare_reserved_for_entity ?critical_equipment ?spare_part_item)
        (not
          (spare_part_available ?spare_part_item)
        )
      )
  )
  (:action assign_fault_to_diagnostic_report
    :parameters (?critical_equipment - critical_equipment ?fault_code - fault_code)
    :precondition
      (and
        (diagnostic_report_created ?critical_equipment)
        (spare_part_source_assigned ?critical_equipment)
        (unlinked_fault_report ?fault_code)
      )
    :effect
      (and
        (fault_assigned_to_entity ?critical_equipment ?fault_code)
        (not
          (unlinked_fault_report ?fault_code)
        )
      )
  )
  (:action authorize_maintenance
    :parameters (?critical_equipment - critical_equipment ?fault_code - fault_code)
    :precondition
      (and
        (diagnostic_report_created ?critical_equipment)
        (spare_part_source_assigned ?critical_equipment)
        (fault_assigned_to_entity ?critical_equipment ?fault_code)
        (not
          (entity_maintenance_authorization_granted ?critical_equipment)
        )
      )
    :effect (entity_maintenance_authorization_granted ?critical_equipment)
  )
  (:action release_fault_assignment
    :parameters (?critical_equipment - critical_equipment ?fault_code - fault_code)
    :precondition
      (and
        (fault_assigned_to_entity ?critical_equipment ?fault_code)
      )
    :effect
      (and
        (unlinked_fault_report ?fault_code)
        (not
          (fault_assigned_to_entity ?critical_equipment ?fault_code)
        )
      )
  )
  (:action assign_technician
    :parameters (?critical_equipment - critical_equipment ?technician - technician)
    :precondition
      (and
        (entity_maintenance_authorization_granted ?critical_equipment)
        (technician_available ?technician)
      )
    :effect
      (and
        (technician_assigned_to_entity ?critical_equipment ?technician)
        (not
          (technician_available ?technician)
        )
      )
  )
  (:action release_technician
    :parameters (?critical_equipment - critical_equipment ?technician - technician)
    :precondition
      (and
        (technician_assigned_to_entity ?critical_equipment ?technician)
      )
    :effect
      (and
        (technician_available ?technician)
        (not
          (technician_assigned_to_entity ?critical_equipment ?technician)
        )
      )
  )
  (:action allocate_critical_subpart
    :parameters (?service_station - service_station ?critical_subpart - critical_subpart)
    :precondition
      (and
        (entity_maintenance_authorization_granted ?service_station)
        (critical_subpart_available ?critical_subpart)
      )
    :effect
      (and
        (critical_subpart_allocated_to_station ?service_station ?critical_subpart)
        (not
          (critical_subpart_available ?critical_subpart)
        )
      )
  )
  (:action release_critical_subpart
    :parameters (?service_station - service_station ?critical_subpart - critical_subpart)
    :precondition
      (and
        (critical_subpart_allocated_to_station ?service_station ?critical_subpart)
      )
    :effect
      (and
        (critical_subpart_available ?critical_subpart)
        (not
          (critical_subpart_allocated_to_station ?service_station ?critical_subpart)
        )
      )
  )
  (:action allocate_emergency_part
    :parameters (?service_station - service_station ?emergency_part - emergency_part)
    :precondition
      (and
        (entity_maintenance_authorization_granted ?service_station)
        (emergency_part_available ?emergency_part)
      )
    :effect
      (and
        (emergency_part_allocated_to_station ?service_station ?emergency_part)
        (not
          (emergency_part_available ?emergency_part)
        )
      )
  )
  (:action release_emergency_part
    :parameters (?service_station - service_station ?emergency_part - emergency_part)
    :precondition
      (and
        (emergency_part_allocated_to_station ?service_station ?emergency_part)
      )
    :effect
      (and
        (emergency_part_available ?emergency_part)
        (not
          (emergency_part_allocated_to_station ?service_station ?emergency_part)
        )
      )
  )
  (:action stage_local_transport_for_primary_unit
    :parameters (?primary_equipment_unit - primary_equipment_unit ?local_transport - local_transport ?fault_code - fault_code)
    :precondition
      (and
        (entity_maintenance_authorization_granted ?primary_equipment_unit)
        (fault_assigned_to_entity ?primary_equipment_unit ?fault_code)
        (primary_unit_assigned_local_transport ?primary_equipment_unit ?local_transport)
        (not
          (local_transport_staged ?local_transport)
        )
        (not
          (local_transport_reserved ?local_transport)
        )
      )
    :effect (local_transport_staged ?local_transport)
  )
  (:action confirm_transport_and_assign_technician_for_primary_unit
    :parameters (?primary_equipment_unit - primary_equipment_unit ?local_transport - local_transport ?technician - technician)
    :precondition
      (and
        (entity_maintenance_authorization_granted ?primary_equipment_unit)
        (technician_assigned_to_entity ?primary_equipment_unit ?technician)
        (primary_unit_assigned_local_transport ?primary_equipment_unit ?local_transport)
        (local_transport_staged ?local_transport)
        (not
          (primary_unit_prepared ?primary_equipment_unit)
        )
      )
    :effect
      (and
        (primary_unit_prepared ?primary_equipment_unit)
        (primary_unit_local_transport_confirmed ?primary_equipment_unit)
      )
  )
  (:action allocate_consumable_and_reserve_local_transport
    :parameters (?primary_equipment_unit - primary_equipment_unit ?local_transport - local_transport ?consumable_kit - consumable_kit)
    :precondition
      (and
        (entity_maintenance_authorization_granted ?primary_equipment_unit)
        (primary_unit_assigned_local_transport ?primary_equipment_unit ?local_transport)
        (consumable_available ?consumable_kit)
        (not
          (primary_unit_prepared ?primary_equipment_unit)
        )
      )
    :effect
      (and
        (local_transport_reserved ?local_transport)
        (primary_unit_prepared ?primary_equipment_unit)
        (consumable_allocated_to_primary_unit ?primary_equipment_unit ?consumable_kit)
        (not
          (consumable_available ?consumable_kit)
        )
      )
  )
  (:action apply_consumable_and_restage_transport_for_primary_unit
    :parameters (?primary_equipment_unit - primary_equipment_unit ?local_transport - local_transport ?fault_code - fault_code ?consumable_kit - consumable_kit)
    :precondition
      (and
        (entity_maintenance_authorization_granted ?primary_equipment_unit)
        (fault_assigned_to_entity ?primary_equipment_unit ?fault_code)
        (primary_unit_assigned_local_transport ?primary_equipment_unit ?local_transport)
        (local_transport_reserved ?local_transport)
        (consumable_allocated_to_primary_unit ?primary_equipment_unit ?consumable_kit)
        (not
          (primary_unit_local_transport_confirmed ?primary_equipment_unit)
        )
      )
    :effect
      (and
        (local_transport_staged ?local_transport)
        (primary_unit_local_transport_confirmed ?primary_equipment_unit)
        (consumable_available ?consumable_kit)
        (not
          (consumable_allocated_to_primary_unit ?primary_equipment_unit ?consumable_kit)
        )
      )
  )
  (:action stage_regional_transport_for_secondary_unit
    :parameters (?secondary_equipment_unit - secondary_equipment_unit ?regional_transport - regional_transport ?fault_code - fault_code)
    :precondition
      (and
        (entity_maintenance_authorization_granted ?secondary_equipment_unit)
        (fault_assigned_to_entity ?secondary_equipment_unit ?fault_code)
        (secondary_unit_assigned_regional_transport ?secondary_equipment_unit ?regional_transport)
        (not
          (regional_transport_staged ?regional_transport)
        )
        (not
          (regional_transport_reserved ?regional_transport)
        )
      )
    :effect (regional_transport_staged ?regional_transport)
  )
  (:action confirm_transport_and_assign_technician_for_secondary_unit
    :parameters (?secondary_equipment_unit - secondary_equipment_unit ?regional_transport - regional_transport ?technician - technician)
    :precondition
      (and
        (entity_maintenance_authorization_granted ?secondary_equipment_unit)
        (technician_assigned_to_entity ?secondary_equipment_unit ?technician)
        (secondary_unit_assigned_regional_transport ?secondary_equipment_unit ?regional_transport)
        (regional_transport_staged ?regional_transport)
        (not
          (secondary_unit_prepared ?secondary_equipment_unit)
        )
      )
    :effect
      (and
        (secondary_unit_prepared ?secondary_equipment_unit)
        (secondary_unit_regional_transport_confirmed ?secondary_equipment_unit)
      )
  )
  (:action allocate_consumable_and_reserve_regional_transport
    :parameters (?secondary_equipment_unit - secondary_equipment_unit ?regional_transport - regional_transport ?consumable_kit - consumable_kit)
    :precondition
      (and
        (entity_maintenance_authorization_granted ?secondary_equipment_unit)
        (secondary_unit_assigned_regional_transport ?secondary_equipment_unit ?regional_transport)
        (consumable_available ?consumable_kit)
        (not
          (secondary_unit_prepared ?secondary_equipment_unit)
        )
      )
    :effect
      (and
        (regional_transport_reserved ?regional_transport)
        (secondary_unit_prepared ?secondary_equipment_unit)
        (consumable_allocated_to_secondary_unit ?secondary_equipment_unit ?consumable_kit)
        (not
          (consumable_available ?consumable_kit)
        )
      )
  )
  (:action apply_consumable_and_restage_transport_for_secondary_unit
    :parameters (?secondary_equipment_unit - secondary_equipment_unit ?regional_transport - regional_transport ?fault_code - fault_code ?consumable_kit - consumable_kit)
    :precondition
      (and
        (entity_maintenance_authorization_granted ?secondary_equipment_unit)
        (fault_assigned_to_entity ?secondary_equipment_unit ?fault_code)
        (secondary_unit_assigned_regional_transport ?secondary_equipment_unit ?regional_transport)
        (regional_transport_reserved ?regional_transport)
        (consumable_allocated_to_secondary_unit ?secondary_equipment_unit ?consumable_kit)
        (not
          (secondary_unit_regional_transport_confirmed ?secondary_equipment_unit)
        )
      )
    :effect
      (and
        (regional_transport_staged ?regional_transport)
        (secondary_unit_regional_transport_confirmed ?secondary_equipment_unit)
        (consumable_available ?consumable_kit)
        (not
          (consumable_allocated_to_secondary_unit ?secondary_equipment_unit ?consumable_kit)
        )
      )
  )
  (:action assemble_repair_kit_for_units
    :parameters (?primary_equipment_unit - primary_equipment_unit ?secondary_equipment_unit - secondary_equipment_unit ?local_transport - local_transport ?regional_transport - regional_transport ?repair_kit - repair_kit)
    :precondition
      (and
        (primary_unit_prepared ?primary_equipment_unit)
        (secondary_unit_prepared ?secondary_equipment_unit)
        (primary_unit_assigned_local_transport ?primary_equipment_unit ?local_transport)
        (secondary_unit_assigned_regional_transport ?secondary_equipment_unit ?regional_transport)
        (local_transport_staged ?local_transport)
        (regional_transport_staged ?regional_transport)
        (primary_unit_local_transport_confirmed ?primary_equipment_unit)
        (secondary_unit_regional_transport_confirmed ?secondary_equipment_unit)
        (repair_kit_available ?repair_kit)
      )
    :effect
      (and
        (repair_kit_assembled ?repair_kit)
        (repair_kit_assigned_local_transport ?repair_kit ?local_transport)
        (repair_kit_assigned_regional_transport ?repair_kit ?regional_transport)
        (not
          (repair_kit_available ?repair_kit)
        )
      )
  )
  (:action assemble_repair_kit_for_deployment
    :parameters (?primary_equipment_unit - primary_equipment_unit ?secondary_equipment_unit - secondary_equipment_unit ?local_transport - local_transport ?regional_transport - regional_transport ?repair_kit - repair_kit)
    :precondition
      (and
        (primary_unit_prepared ?primary_equipment_unit)
        (secondary_unit_prepared ?secondary_equipment_unit)
        (primary_unit_assigned_local_transport ?primary_equipment_unit ?local_transport)
        (secondary_unit_assigned_regional_transport ?secondary_equipment_unit ?regional_transport)
        (local_transport_reserved ?local_transport)
        (regional_transport_staged ?regional_transport)
        (not
          (primary_unit_local_transport_confirmed ?primary_equipment_unit)
        )
        (secondary_unit_regional_transport_confirmed ?secondary_equipment_unit)
        (repair_kit_available ?repair_kit)
      )
    :effect
      (and
        (repair_kit_assembled ?repair_kit)
        (repair_kit_assigned_local_transport ?repair_kit ?local_transport)
        (repair_kit_assigned_regional_transport ?repair_kit ?regional_transport)
        (repair_kit_local_route_confirmed ?repair_kit)
        (not
          (repair_kit_available ?repair_kit)
        )
      )
  )
  (:action assemble_repair_kit_for_alternate_route
    :parameters (?primary_equipment_unit - primary_equipment_unit ?secondary_equipment_unit - secondary_equipment_unit ?local_transport - local_transport ?regional_transport - regional_transport ?repair_kit - repair_kit)
    :precondition
      (and
        (primary_unit_prepared ?primary_equipment_unit)
        (secondary_unit_prepared ?secondary_equipment_unit)
        (primary_unit_assigned_local_transport ?primary_equipment_unit ?local_transport)
        (secondary_unit_assigned_regional_transport ?secondary_equipment_unit ?regional_transport)
        (local_transport_staged ?local_transport)
        (regional_transport_reserved ?regional_transport)
        (primary_unit_local_transport_confirmed ?primary_equipment_unit)
        (not
          (secondary_unit_regional_transport_confirmed ?secondary_equipment_unit)
        )
        (repair_kit_available ?repair_kit)
      )
    :effect
      (and
        (repair_kit_assembled ?repair_kit)
        (repair_kit_assigned_local_transport ?repair_kit ?local_transport)
        (repair_kit_assigned_regional_transport ?repair_kit ?regional_transport)
        (repair_kit_regional_route_confirmed ?repair_kit)
        (not
          (repair_kit_available ?repair_kit)
        )
      )
  )
  (:action assemble_repair_kit_comprehensive
    :parameters (?primary_equipment_unit - primary_equipment_unit ?secondary_equipment_unit - secondary_equipment_unit ?local_transport - local_transport ?regional_transport - regional_transport ?repair_kit - repair_kit)
    :precondition
      (and
        (primary_unit_prepared ?primary_equipment_unit)
        (secondary_unit_prepared ?secondary_equipment_unit)
        (primary_unit_assigned_local_transport ?primary_equipment_unit ?local_transport)
        (secondary_unit_assigned_regional_transport ?secondary_equipment_unit ?regional_transport)
        (local_transport_reserved ?local_transport)
        (regional_transport_reserved ?regional_transport)
        (not
          (primary_unit_local_transport_confirmed ?primary_equipment_unit)
        )
        (not
          (secondary_unit_regional_transport_confirmed ?secondary_equipment_unit)
        )
        (repair_kit_available ?repair_kit)
      )
    :effect
      (and
        (repair_kit_assembled ?repair_kit)
        (repair_kit_assigned_local_transport ?repair_kit ?local_transport)
        (repair_kit_assigned_regional_transport ?repair_kit ?regional_transport)
        (repair_kit_local_route_confirmed ?repair_kit)
        (repair_kit_regional_route_confirmed ?repair_kit)
        (not
          (repair_kit_available ?repair_kit)
        )
      )
  )
  (:action lock_repair_kit_for_dispatch
    :parameters (?repair_kit - repair_kit ?primary_equipment_unit - primary_equipment_unit ?fault_code - fault_code)
    :precondition
      (and
        (repair_kit_assembled ?repair_kit)
        (primary_unit_prepared ?primary_equipment_unit)
        (fault_assigned_to_entity ?primary_equipment_unit ?fault_code)
        (not
          (repair_kit_locked_for_dispatch ?repair_kit)
        )
      )
    :effect (repair_kit_locked_for_dispatch ?repair_kit)
  )
  (:action certify_and_attach_certificate_to_kit
    :parameters (?service_station - service_station ?quality_certificate - quality_certificate ?repair_kit - repair_kit)
    :precondition
      (and
        (entity_maintenance_authorization_granted ?service_station)
        (service_station_stores_repair_kit ?service_station ?repair_kit)
        (service_station_has_certificate ?service_station ?quality_certificate)
        (quality_certificate_available ?quality_certificate)
        (repair_kit_assembled ?repair_kit)
        (repair_kit_locked_for_dispatch ?repair_kit)
        (not
          (quality_certificate_validated ?quality_certificate)
        )
      )
    :effect
      (and
        (quality_certificate_validated ?quality_certificate)
        (certificate_attached_to_repair_kit ?quality_certificate ?repair_kit)
        (not
          (quality_certificate_available ?quality_certificate)
        )
      )
  )
  (:action perform_quality_inspection_stage_one
    :parameters (?service_station - service_station ?quality_certificate - quality_certificate ?repair_kit - repair_kit ?fault_code - fault_code)
    :precondition
      (and
        (entity_maintenance_authorization_granted ?service_station)
        (service_station_has_certificate ?service_station ?quality_certificate)
        (quality_certificate_validated ?quality_certificate)
        (certificate_attached_to_repair_kit ?quality_certificate ?repair_kit)
        (fault_assigned_to_entity ?service_station ?fault_code)
        (not
          (repair_kit_local_route_confirmed ?repair_kit)
        )
        (not
          (inspection_stage_certificate_verified ?service_station)
        )
      )
    :effect (inspection_stage_certificate_verified ?service_station)
  )
  (:action assign_priority_and_vendor_authorization
    :parameters (?service_station - service_station ?priority_tag - priority_tag)
    :precondition
      (and
        (entity_maintenance_authorization_granted ?service_station)
        (priority_tag_available ?priority_tag)
        (not
          (station_priority_assigned ?service_station)
        )
      )
    :effect
      (and
        (station_priority_assigned ?service_station)
        (priority_tag_assigned_to_station ?service_station ?priority_tag)
        (not
          (priority_tag_available ?priority_tag)
        )
