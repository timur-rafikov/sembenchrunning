  (:predicates

    (entity_evaluation_enabled ?entity_item - entity_item)

    (entity_finalized ?entity_item - entity_item)
    (entity_output_selected ?entity_item - entity_item)
    (entity_committed_for_placement ?entity_item - entity_item)

    (entity_bound_tie_breaker_token ?entity_item - entity_item ?resource_tiebreak_token - resource_tie_breaker_token)

    (group_a_assigned_attribute_token ?entity_group_a - entity_group_a ?attribute_token_primary - attribute_token_primary)
    (group_b_assigned_attribute_token ?entity_group_b - entity_group_b ?attribute_token_primary - attribute_token_primary)
    (group_a_slot_primary_link ?entity_group_a - entity_group_a ?attribute_slot_primary - attribute_slot_primary)
    (attribute_slot_primary_flag_provisional ?attribute_slot_primary - attribute_slot_primary)
    (attribute_slot_primary_flag_confirmed ?attribute_slot_primary - attribute_slot_primary)
    (group_a_attribute_finalized ?entity_group_a - entity_group_a)
    (group_b_slot_secondary_link ?entity_group_b - entity_group_b ?attribute_slot_secondary - attribute_slot_secondary)
    (attribute_slot_secondary_flag_provisional ?attribute_slot_secondary - attribute_slot_secondary)
    (attribute_slot_secondary_flag_confirmed ?attribute_slot_secondary - attribute_slot_secondary)

    (evaluation_record_slot_secondary_link ?evaluation_record - evaluation_record ?attribute_slot_secondary - attribute_slot_secondary)
    (evaluation_record_aux_comp1_present ?evaluation_record - evaluation_record)
    (evaluation_record_aux_comp2_present ?evaluation_record - evaluation_record)
    (evaluation_record_processing_marker ?evaluation_record - evaluation_record)
    (bucket_group_a_link ?entity_bucket - entity_bucket ?entity_group_a - entity_group_a)
    (bucket_group_b_link ?entity_bucket - entity_bucket ?entity_group_b - entity_group_b)
    (bucket_evaluation_record_link ?entity_bucket - entity_bucket ?evaluation_record - evaluation_record)
    (attribute_source_available ?attribute_source - attribute_source)
    (bucket_attached_attribute_source ?entity_bucket - entity_bucket ?attribute_source - attribute_source)
    (attribute_source_consumed ?attribute_source - attribute_source)
    (attribute_source_attached_to_record ?attribute_source - attribute_source ?evaluation_record - evaluation_record)

  (:action register_entity
    :parameters (?entity_item - entity_item)
    :precondition
      (and
        (not
          (entity_registered ?entity_item)
        )
        (not
          (entity_finalized ?entity_item)
        )
      )
    :effect (entity_registered ?entity_item)
  )
  (:action assign_tie_breaker_to_entity
    :parameters (?entity_item - entity_item ?resource_tiebreak_token - resource_tie_breaker_token)
    :precondition
      (and
        (entity_registered ?entity_item)
        (not
          (entity_tiebreak_assigned ?entity_item)
        )
        (resource_tie_breaker_available ?resource_tiebreak_token)
      )
    :effect
      (and
        (entity_tiebreak_assigned ?entity_item)
        (entity_bound_tie_breaker_token ?entity_item ?resource_tiebreak_token)
        (not
          (resource_tie_breaker_available ?resource_tiebreak_token)
        )
      )
  )
  (:action assign_criterion_to_entity
    :parameters (?entity_item - entity_item ?resource_criterion_token - resource_criterion_token)
    :precondition
      (and
        (entity_registered ?entity_item)
        (entity_tiebreak_assigned ?entity_item)
        (resource_criterion_available ?resource_criterion_token)
      )
    :effect
      (and
        (entity_bound_criterion_token ?entity_item ?resource_criterion_token)
        (not
          (resource_criterion_available ?resource_criterion_token)
        )
      )
  )
  (:action enable_entity_evaluation
    :parameters (?entity_item - entity_item ?resource_criterion_token - resource_criterion_token)
    :precondition
      (and
        (entity_registered ?entity_item)
        (entity_tiebreak_assigned ?entity_item)
        (entity_bound_criterion_token ?entity_item ?resource_criterion_token)
        (not
          (entity_evaluation_enabled ?entity_item)
        )
      )
    :effect (entity_evaluation_enabled ?entity_item)
  )
  (:action release_criterion_from_entity
    :parameters (?entity_item - entity_item ?resource_criterion_token - resource_criterion_token)
    :precondition
      (and
        (entity_bound_criterion_token ?entity_item ?resource_criterion_token)
      )
    :effect
      (and
        (resource_criterion_available ?resource_criterion_token)
        (not
          (entity_bound_criterion_token ?entity_item ?resource_criterion_token)
        )
      )
  )
  (:action assign_slot_to_entity
    :parameters (?entity_item - entity_item ?resource_slot - resource_slot)
    :precondition
      (and
        (entity_evaluation_enabled ?entity_item)
        (resource_slot_available ?resource_slot)
      )
    :effect
      (and
        (entity_assigned_slot ?entity_item ?resource_slot)
        (not
          (resource_slot_available ?resource_slot)
        )
      )
  )
  (:action release_slot_from_entity
    :parameters (?entity_item - entity_item ?resource_slot - resource_slot)
    :precondition
      (and
        (entity_assigned_slot ?entity_item ?resource_slot)
      )
    :effect
      (and
        (resource_slot_available ?resource_slot)
        (not
          (entity_assigned_slot ?entity_item ?resource_slot)
        )
      )
  )
  (:action assign_comparator_aux1_to_bucket
    :parameters (?entity_bucket - entity_bucket ?comparator_aux_token_1 - comparator_aux_token_1)
    :precondition
      (and
        (entity_evaluation_enabled ?entity_bucket)
        (comparator_aux1_available ?comparator_aux_token_1)
      )
    :effect
      (and
        (bucket_bound_comparator_aux1 ?entity_bucket ?comparator_aux_token_1)
        (not
          (comparator_aux1_available ?comparator_aux_token_1)
        )
      )
  )
  (:action release_comparator_aux1_from_bucket
    :parameters (?entity_bucket - entity_bucket ?comparator_aux_token_1 - comparator_aux_token_1)
    :precondition
      (and
        (bucket_bound_comparator_aux1 ?entity_bucket ?comparator_aux_token_1)
      )
    :effect
      (and
        (comparator_aux1_available ?comparator_aux_token_1)
        (not
          (bucket_bound_comparator_aux1 ?entity_bucket ?comparator_aux_token_1)
        )
      )
  )
  (:action assign_comparator_aux3_to_bucket
    :parameters (?entity_bucket - entity_bucket ?comparator_aux_token_3 - comparator_aux_token_3)
    :precondition
      (and
        (entity_evaluation_enabled ?entity_bucket)
        (comparator_aux3_available ?comparator_aux_token_3)
      )
    :effect
      (and
        (bucket_bound_comparator_aux3 ?entity_bucket ?comparator_aux_token_3)
        (not
          (comparator_aux3_available ?comparator_aux_token_3)
        )
      )
  )
  (:action release_comparator_aux3_from_bucket
    :parameters (?entity_bucket - entity_bucket ?comparator_aux_token_3 - comparator_aux_token_3)

      (and
        (bucket_bound_comparator_aux3 ?entity_bucket ?comparator_aux_token_3)
      )
    :effect
      (and
        (comparator_aux3_available ?comparator_aux_token_3)
        (not
          (bucket_bound_comparator_aux3 ?entity_bucket ?comparator_aux_token_3)
        )
      )
  )
  (:action set_slot_primary_provisional
    :parameters (?entity_group_a - entity_group_a ?attribute_slot_primary - attribute_slot_primary ?resource_criterion_token - resource_criterion_token)
    :precondition
      (and
        (entity_evaluation_enabled ?entity_group_a)
        (entity_bound_criterion_token ?entity_group_a ?resource_criterion_token)
        (group_a_slot_primary_link ?entity_group_a ?attribute_slot_primary)
        (not
          (attribute_slot_primary_flag_provisional ?attribute_slot_primary)
        )
        (not
          (attribute_slot_primary_flag_confirmed ?attribute_slot_primary)
        )
      )
    :effect (attribute_slot_primary_flag_provisional ?attribute_slot_primary)
  )
  (:action group_a_confirm_primary_slot
    :parameters (?entity_group_a - entity_group_a ?attribute_slot_primary - attribute_slot_primary ?resource_slot - resource_slot)
    :precondition
      (and
        (entity_evaluation_enabled ?entity_group_a)
        (entity_assigned_slot ?entity_group_a ?resource_slot)
        (group_a_slot_primary_link ?entity_group_a ?attribute_slot_primary)
        (attribute_slot_primary_flag_provisional ?attribute_slot_primary)
        (not
          (group_a_attribute_claimed ?entity_group_a)
        )
      )
    :effect
      (and
        (group_a_attribute_claimed ?entity_group_a)
        (group_a_attribute_finalized ?entity_group_a)
      )
  )
  (:action group_a_attach_attribute_token
    :parameters (?entity_group_a - entity_group_a ?attribute_slot_primary - attribute_slot_primary ?attribute_token_primary - attribute_token_primary)
    :precondition
      (and
        (entity_evaluation_enabled ?entity_group_a)
        (group_a_slot_primary_link ?entity_group_a ?attribute_slot_primary)
        (attribute_token_primary_available ?attribute_token_primary)
        (not
          (group_a_attribute_claimed ?entity_group_a)
        )
      )
    :effect
      (and
        (attribute_slot_primary_flag_confirmed ?attribute_slot_primary)
        (group_a_attribute_claimed ?entity_group_a)
        (group_a_assigned_attribute_token ?entity_group_a ?attribute_token_primary)
        (not
          (attribute_token_primary_available ?attribute_token_primary)
        )
      )
  )
  (:action group_a_resolve_primary_assignment
    :parameters (?entity_group_a - entity_group_a ?attribute_slot_primary - attribute_slot_primary ?resource_criterion_token - resource_criterion_token ?attribute_token_primary - attribute_token_primary)
    :precondition
      (and
        (entity_evaluation_enabled ?entity_group_a)

          (group_a_assigned_attribute_token ?entity_group_a ?attribute_token_primary)

  )
  (:action group_b_confirm_secondary_slot
    :parameters (?entity_group_b - entity_group_b ?attribute_slot_secondary - attribute_slot_secondary ?resource_slot - resource_slot)

    :parameters (?entity_group_b - entity_group_b ?attribute_slot_secondary - attribute_slot_secondary ?attribute_token_primary - attribute_token_primary)
    :precondition
      (and
        (entity_evaluation_enabled ?entity_group_b)

        (not
          (group_b_attribute_claimed ?entity_group_b)

    :effect
      (and
        (attribute_slot_secondary_flag_confirmed ?attribute_slot_secondary)

        (attribute_slot_secondary_flag_provisional ?attribute_slot_secondary)

        (attribute_token_primary_available ?attribute_token_primary)
        (not
          (group_b_assigned_attribute_token ?entity_group_b ?attribute_token_primary)
        )
      )
  )

      (and
        (group_a_attribute_claimed ?entity_group_a)

        (group_b_slot_secondary_link ?entity_group_b ?attribute_slot_secondary)

      (and

          (evaluation_record_available ?evaluation_record)

    :precondition
      (and
        (entity_evaluation_enabled ?entity_bucket)
