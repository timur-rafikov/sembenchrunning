--- domain.pddl (as passed to VAL) ---
(define (domain crowd_control_assignment_across_targets)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity_super - object token_super - object item_super - object slot_super - object unit_slot - slot_super cc_token - entity_super cc_ability - entity_super support_actor - entity_super priority_marker - entity_super talent_slot - entity_super assignment_tag - entity_super auxiliary_support - entity_super phase_token - entity_super consumable_resource - token_super tactical_marker - token_super priority_token - token_super target_cluster_a - item_super target_cluster_b - item_super assignment_bundle - item_super slot_group - unit_slot slot_group_alt - unit_slot primary_cc_unit - slot_group secondary_cc_unit - slot_group assignment_controller - slot_group_alt)
  (:predicates
    (assignment_selected ?unit_slot - unit_slot)
    (assignment_validated ?unit_slot - unit_slot)
    (assignment_has_token ?unit_slot - unit_slot)
    (assignment_assigned ?unit_slot - unit_slot)
    (assignment_readiness_flag ?unit_slot - unit_slot)
    (assignment_staged ?unit_slot - unit_slot)
    (cc_token_available ?cc_token - cc_token)
    (assignment_token_bound ?unit_slot - unit_slot ?cc_token - cc_token)
    (ability_available ?cc_ability - cc_ability)
    (assignment_ability_bound ?unit_slot - unit_slot ?cc_ability - cc_ability)
    (support_available ?support_actor - support_actor)
    (support_bound_to_assignment ?unit_slot - unit_slot ?support_actor - support_actor)
    (resource_available ?consumable_resource - consumable_resource)
    (primary_resource_bound ?primary_cc_unit - primary_cc_unit ?consumable_resource - consumable_resource)
    (secondary_resource_bound ?secondary_cc_unit - secondary_cc_unit ?consumable_resource - consumable_resource)
    (primary_bound_to_cluster_a ?primary_cc_unit - primary_cc_unit ?target_cluster_a - target_cluster_a)
    (cluster_a_reserved ?target_cluster_a - target_cluster_a)
    (cluster_a_resource_reserved ?target_cluster_a - target_cluster_a)
    (primary_confirmed ?primary_cc_unit - primary_cc_unit)
    (secondary_bound_to_cluster_b ?secondary_cc_unit - secondary_cc_unit ?target_cluster_b - target_cluster_b)
    (cluster_b_reserved ?target_cluster_b - target_cluster_b)
    (cluster_b_resource_reserved ?target_cluster_b - target_cluster_b)
    (secondary_confirmed ?secondary_cc_unit - secondary_cc_unit)
    (bundle_available ?assignment_bundle - assignment_bundle)
    (bundle_claimed ?assignment_bundle - assignment_bundle)
    (bundle_bound_to_cluster_a ?assignment_bundle - assignment_bundle ?target_cluster_a - target_cluster_a)
    (bundle_bound_to_cluster_b ?assignment_bundle - assignment_bundle ?target_cluster_b - target_cluster_b)
    (bundle_flag_a_resource ?assignment_bundle - assignment_bundle)
    (bundle_flag_b_resource ?assignment_bundle - assignment_bundle)
    (bundle_marker_ready ?assignment_bundle - assignment_bundle)
    (controller_assigned_primary ?assignment_controller - assignment_controller ?primary_cc_unit - primary_cc_unit)
    (controller_assigned_secondary ?assignment_controller - assignment_controller ?secondary_cc_unit - secondary_cc_unit)
    (controller_bound_to_bundle ?assignment_controller - assignment_controller ?assignment_bundle - assignment_bundle)
    (marker_available ?tactical_marker - tactical_marker)
    (controller_marker_bound ?assignment_controller - assignment_controller ?tactical_marker - tactical_marker)
    (marker_consumed ?tactical_marker - tactical_marker)
    (marker_bound_to_bundle ?tactical_marker - tactical_marker ?assignment_bundle - assignment_bundle)
    (controller_phase_marker_applied ?assignment_controller - assignment_controller)
    (controller_aux_supports_ready ?assignment_controller - assignment_controller)
    (controller_talent_ready ?assignment_controller - assignment_controller)
    (controller_priority_active ?assignment_controller - assignment_controller)
    (controller_priority_followup_set ?assignment_controller - assignment_controller)
    (controller_talent_bound ?assignment_controller - assignment_controller)
    (controller_execution_committed ?assignment_controller - assignment_controller)
    (priority_token_available ?priority_token - priority_token)
    (controller_priority_bound ?assignment_controller - assignment_controller ?priority_token - priority_token)
    (controller_priority_reserved ?assignment_controller - assignment_controller)
    (controller_priority_acknowledged ?assignment_controller - assignment_controller)
    (controller_priority_sequence_ready ?assignment_controller - assignment_controller)
    (priority_marker_available ?priority_marker - priority_marker)
    (controller_priority_marker_bound ?assignment_controller - assignment_controller ?priority_marker - priority_marker)
    (talent_slot_available ?talent_slot - talent_slot)
    (controller_talent_slot_bound ?assignment_controller - assignment_controller ?talent_slot - talent_slot)
    (auxiliary_support_available ?auxiliary_support - auxiliary_support)
    (controller_auxiliary_support_bound ?assignment_controller - assignment_controller ?auxiliary_support - auxiliary_support)
    (phase_token_available ?phase_token - phase_token)
    (controller_phase_token_bound ?assignment_controller - assignment_controller ?phase_token - phase_token)
    (assignment_tag_available ?assignment_tag - assignment_tag)
    (assignment_tag_bound ?unit_slot - unit_slot ?assignment_tag - assignment_tag)
    (primary_ready ?primary_cc_unit - primary_cc_unit)
    (secondary_ready ?secondary_cc_unit - secondary_cc_unit)
    (controller_finalized ?assignment_controller - assignment_controller)
  )
  (:action select_unit_into_slot
    :parameters (?unit_slot - unit_slot)
    :precondition
      (and
        (not
          (assignment_selected ?unit_slot)
        )
        (not
          (assignment_assigned ?unit_slot)
        )
      )
    :effect (assignment_selected ?unit_slot)
  )
  (:action attach_cc_token_to_slot
    :parameters (?unit_slot - unit_slot ?cc_token - cc_token)
    :precondition
      (and
        (assignment_selected ?unit_slot)
        (not
          (assignment_has_token ?unit_slot)
        )
        (cc_token_available ?cc_token)
      )
    :effect
      (and
        (assignment_has_token ?unit_slot)
        (assignment_token_bound ?unit_slot ?cc_token)
        (not
          (cc_token_available ?cc_token)
        )
      )
  )
  (:action attach_ability_to_slot
    :parameters (?unit_slot - unit_slot ?cc_ability - cc_ability)
    :precondition
      (and
        (assignment_selected ?unit_slot)
        (assignment_has_token ?unit_slot)
        (ability_available ?cc_ability)
      )
    :effect
      (and
        (assignment_ability_bound ?unit_slot ?cc_ability)
        (not
          (ability_available ?cc_ability)
        )
      )
  )
  (:action confirm_slot_with_ability
    :parameters (?unit_slot - unit_slot ?cc_ability - cc_ability)
    :precondition
      (and
        (assignment_selected ?unit_slot)
        (assignment_has_token ?unit_slot)
        (assignment_ability_bound ?unit_slot ?cc_ability)
        (not
          (assignment_validated ?unit_slot)
        )
      )
    :effect (assignment_validated ?unit_slot)
  )
  (:action release_ability_from_slot
    :parameters (?unit_slot - unit_slot ?cc_ability - cc_ability)
    :precondition
      (and
        (assignment_ability_bound ?unit_slot ?cc_ability)
      )
    :effect
      (and
        (ability_available ?cc_ability)
        (not
          (assignment_ability_bound ?unit_slot ?cc_ability)
        )
      )
  )
  (:action attach_support_to_slot
    :parameters (?unit_slot - unit_slot ?support_actor - support_actor)
    :precondition
      (and
        (assignment_validated ?unit_slot)
        (support_available ?support_actor)
      )
    :effect
      (and
        (support_bound_to_assignment ?unit_slot ?support_actor)
        (not
          (support_available ?support_actor)
        )
      )
  )
  (:action detach_support_from_slot
    :parameters (?unit_slot - unit_slot ?support_actor - support_actor)
    :precondition
      (and
        (support_bound_to_assignment ?unit_slot ?support_actor)
      )
    :effect
      (and
        (support_available ?support_actor)
        (not
          (support_bound_to_assignment ?unit_slot ?support_actor)
        )
      )
  )
  (:action attach_aux_support_to_controller
    :parameters (?assignment_controller - assignment_controller ?auxiliary_support - auxiliary_support)
    :precondition
      (and
        (assignment_validated ?assignment_controller)
        (auxiliary_support_available ?auxiliary_support)
      )
    :effect
      (and
        (controller_auxiliary_support_bound ?assignment_controller ?auxiliary_support)
        (not
          (auxiliary_support_available ?auxiliary_support)
        )
      )
  )
  (:action detach_aux_support_from_controller
    :parameters (?assignment_controller - assignment_controller ?auxiliary_support - auxiliary_support)
    :precondition
      (and
        (controller_auxiliary_support_bound ?assignment_controller ?auxiliary_support)
      )
    :effect
      (and
        (auxiliary_support_available ?auxiliary_support)
        (not
          (controller_auxiliary_support_bound ?assignment_controller ?auxiliary_support)
        )
      )
  )
  (:action attach_phase_token_to_controller
    :parameters (?assignment_controller - assignment_controller ?phase_token - phase_token)
    :precondition
      (and
        (assignment_validated ?assignment_controller)
        (phase_token_available ?phase_token)
      )
    :effect
      (and
        (controller_phase_token_bound ?assignment_controller ?phase_token)
        (not
          (phase_token_available ?phase_token)
        )
      )
  )
  (:action detach_phase_token_from_controller
    :parameters (?assignment_controller - assignment_controller ?phase_token - phase_token)
    :precondition
      (and
        (controller_phase_token_bound ?assignment_controller ?phase_token)
      )
    :effect
      (and
        (phase_token_available ?phase_token)
        (not
          (controller_phase_token_bound ?assignment_controller ?phase_token)
        )
      )
  )
  (:action reserve_cluster_a
    :parameters (?primary_cc_unit - primary_cc_unit ?target_cluster_a - target_cluster_a ?cc_ability - cc_ability)
    :precondition
      (and
        (assignment_validated ?primary_cc_unit)
        (assignment_ability_bound ?primary_cc_unit ?cc_ability)
        (primary_bound_to_cluster_a ?primary_cc_unit ?target_cluster_a)
        (not
          (cluster_a_reserved ?target_cluster_a)
        )
        (not
          (cluster_a_resource_reserved ?target_cluster_a)
        )
      )
    :effect (cluster_a_reserved ?target_cluster_a)
  )
  (:action confirm_primary_reservation_with_support
    :parameters (?primary_cc_unit - primary_cc_unit ?target_cluster_a - target_cluster_a ?support_actor - support_actor)
    :precondition
      (and
        (assignment_validated ?primary_cc_unit)
        (support_bound_to_assignment ?primary_cc_unit ?support_actor)
        (primary_bound_to_cluster_a ?primary_cc_unit ?target_cluster_a)
        (cluster_a_reserved ?target_cluster_a)
        (not
          (primary_ready ?primary_cc_unit)
        )
      )
    :effect
      (and
        (primary_ready ?primary_cc_unit)
        (primary_confirmed ?primary_cc_unit)
      )
  )
  (:action reserve_cluster_a_with_resource
    :parameters (?primary_cc_unit - primary_cc_unit ?target_cluster_a - target_cluster_a ?consumable_resource - consumable_resource)
    :precondition
      (and
        (assignment_validated ?primary_cc_unit)
        (primary_bound_to_cluster_a ?primary_cc_unit ?target_cluster_a)
        (resource_available ?consumable_resource)
        (not
          (primary_ready ?primary_cc_unit)
        )
      )
    :effect
      (and
        (cluster_a_resource_reserved ?target_cluster_a)
        (primary_ready ?primary_cc_unit)
        (primary_resource_bound ?primary_cc_unit ?consumable_resource)
        (not
          (resource_available ?consumable_resource)
        )
      )
  )
  (:action finalize_cluster_a_reservation
    :parameters (?primary_cc_unit - primary_cc_unit ?target_cluster_a - target_cluster_a ?cc_ability - cc_ability ?consumable_resource - consumable_resource)
    :precondition
      (and
        (assignment_validated ?primary_cc_unit)
        (assignment_ability_bound ?primary_cc_unit ?cc_ability)
        (primary_bound_to_cluster_a ?primary_cc_unit ?target_cluster_a)
        (cluster_a_resource_reserved ?target_cluster_a)
        (primary_resource_bound ?primary_cc_unit ?consumable_resource)
        (not
          (primary_confirmed ?primary_cc_unit)
        )
      )
    :effect
      (and
        (cluster_a_reserved ?target_cluster_a)
        (primary_confirmed ?primary_cc_unit)
        (resource_available ?consumable_resource)
        (not
          (primary_resource_bound ?primary_cc_unit ?consumable_resource)
        )
      )
  )
  (:action reserve_cluster_b
    :parameters (?secondary_cc_unit - secondary_cc_unit ?target_cluster_b - target_cluster_b ?cc_ability - cc_ability)
    :precondition
      (and
        (assignment_validated ?secondary_cc_unit)
        (assignment_ability_bound ?secondary_cc_unit ?cc_ability)
        (secondary_bound_to_cluster_b ?secondary_cc_unit ?target_cluster_b)
        (not
          (cluster_b_reserved ?target_cluster_b)
        )
        (not
          (cluster_b_resource_reserved ?target_cluster_b)
        )
      )
    :effect (cluster_b_reserved ?target_cluster_b)
  )
  (:action confirm_secondary_reservation_with_support
    :parameters (?secondary_cc_unit - secondary_cc_unit ?target_cluster_b - target_cluster_b ?support_actor - support_actor)
    :precondition
      (and
        (assignment_validated ?secondary_cc_unit)
        (support_bound_to_assignment ?secondary_cc_unit ?support_actor)
        (secondary_bound_to_cluster_b ?secondary_cc_unit ?target_cluster_b)
        (cluster_b_reserved ?target_cluster_b)
        (not
          (secondary_ready ?secondary_cc_unit)
        )
      )
    :effect
      (and
        (secondary_ready ?secondary_cc_unit)
        (secondary_confirmed ?secondary_cc_unit)
      )
  )
  (:action reserve_cluster_b_with_resource
    :parameters (?secondary_cc_unit - secondary_cc_unit ?target_cluster_b - target_cluster_b ?consumable_resource - consumable_resource)
    :precondition
      (and
        (assignment_validated ?secondary_cc_unit)
        (secondary_bound_to_cluster_b ?secondary_cc_unit ?target_cluster_b)
        (resource_available ?consumable_resource)
        (not
          (secondary_ready ?secondary_cc_unit)
        )
      )
    :effect
      (and
        (cluster_b_resource_reserved ?target_cluster_b)
        (secondary_ready ?secondary_cc_unit)
        (secondary_resource_bound ?secondary_cc_unit ?consumable_resource)
        (not
          (resource_available ?consumable_resource)
        )
      )
  )
  (:action finalize_cluster_b_reservation
    :parameters (?secondary_cc_unit - secondary_cc_unit ?target_cluster_b - target_cluster_b ?cc_ability - cc_ability ?consumable_resource - consumable_resource)
    :precondition
      (and
        (assignment_validated ?secondary_cc_unit)
        (assignment_ability_bound ?secondary_cc_unit ?cc_ability)
        (secondary_bound_to_cluster_b ?secondary_cc_unit ?target_cluster_b)
        (cluster_b_resource_reserved ?target_cluster_b)
        (secondary_resource_bound ?secondary_cc_unit ?consumable_resource)
        (not
          (secondary_confirmed ?secondary_cc_unit)
        )
      )
    :effect
      (and
        (cluster_b_reserved ?target_cluster_b)
        (secondary_confirmed ?secondary_cc_unit)
        (resource_available ?consumable_resource)
        (not
          (secondary_resource_bound ?secondary_cc_unit ?consumable_resource)
        )
      )
  )
  (:action aggregate_reservations_into_bundle
    :parameters (?primary_cc_unit - primary_cc_unit ?secondary_cc_unit - secondary_cc_unit ?target_cluster_a - target_cluster_a ?target_cluster_b - target_cluster_b ?assignment_bundle - assignment_bundle)
    :precondition
      (and
        (primary_ready ?primary_cc_unit)
        (secondary_ready ?secondary_cc_unit)
        (primary_bound_to_cluster_a ?primary_cc_unit ?target_cluster_a)
        (secondary_bound_to_cluster_b ?secondary_cc_unit ?target_cluster_b)
        (cluster_a_reserved ?target_cluster_a)
        (cluster_b_reserved ?target_cluster_b)
        (primary_confirmed ?primary_cc_unit)
        (secondary_confirmed ?secondary_cc_unit)
        (bundle_available ?assignment_bundle)
      )
    :effect
      (and
        (bundle_claimed ?assignment_bundle)
        (bundle_bound_to_cluster_a ?assignment_bundle ?target_cluster_a)
        (bundle_bound_to_cluster_b ?assignment_bundle ?target_cluster_b)
        (not
          (bundle_available ?assignment_bundle)
        )
      )
  )
  (:action aggregate_reservations_into_bundle_variant_a
    :parameters (?primary_cc_unit - primary_cc_unit ?secondary_cc_unit - secondary_cc_unit ?target_cluster_a - target_cluster_a ?target_cluster_b - target_cluster_b ?assignment_bundle - assignment_bundle)
    :precondition
      (and
        (primary_ready ?primary_cc_unit)
        (secondary_ready ?secondary_cc_unit)
        (primary_bound_to_cluster_a ?primary_cc_unit ?target_cluster_a)
        (secondary_bound_to_cluster_b ?secondary_cc_unit ?target_cluster_b)
        (cluster_a_resource_reserved ?target_cluster_a)
        (cluster_b_reserved ?target_cluster_b)
        (not
          (primary_confirmed ?primary_cc_unit)
        )
        (secondary_confirmed ?secondary_cc_unit)
        (bundle_available ?assignment_bundle)
      )
    :effect
      (and
        (bundle_claimed ?assignment_bundle)
        (bundle_bound_to_cluster_a ?assignment_bundle ?target_cluster_a)
        (bundle_bound_to_cluster_b ?assignment_bundle ?target_cluster_b)
        (bundle_flag_a_resource ?assignment_bundle)
        (not
          (bundle_available ?assignment_bundle)
        )
      )
  )
  (:action aggregate_reservations_into_bundle_variant_b
    :parameters (?primary_cc_unit - primary_cc_unit ?secondary_cc_unit - secondary_cc_unit ?target_cluster_a - target_cluster_a ?target_cluster_b - target_cluster_b ?assignment_bundle - assignment_bundle)
    :precondition
      (and
        (primary_ready ?primary_cc_unit)
        (secondary_ready ?secondary_cc_unit)
        (primary_bound_to_cluster_a ?primary_cc_unit ?target_cluster_a)
        (secondary_bound_to_cluster_b ?secondary_cc_unit ?target_cluster_b)
        (cluster_a_reserved ?target_cluster_a)
        (cluster_b_resource_reserved ?target_cluster_b)
        (primary_confirmed ?primary_cc_unit)
        (not
          (secondary_confirmed ?secondary_cc_unit)
        )
        (bundle_available ?assignment_bundle)
      )
    :effect
      (and
        (bundle_claimed ?assignment_bundle)
        (bundle_bound_to_cluster_a ?assignment_bundle ?target_cluster_a)
        (bundle_bound_to_cluster_b ?assignment_bundle ?target_cluster_b)
        (bundle_flag_b_resource ?assignment_bundle)
        (not
          (bundle_available ?assignment_bundle)
        )
      )
  )
  (:action aggregate_reservations_into_bundle_variant_both
    :parameters (?primary_cc_unit - primary_cc_unit ?secondary_cc_unit - secondary_cc_unit ?target_cluster_a - target_cluster_a ?target_cluster_b - target_cluster_b ?assignment_bundle - assignment_bundle)
    :precondition
      (and
        (primary_ready ?primary_cc_unit)
        (secondary_ready ?secondary_cc_unit)
        (primary_bound_to_cluster_a ?primary_cc_unit ?target_cluster_a)
        (secondary_bound_to_cluster_b ?secondary_cc_unit ?target_cluster_b)
        (cluster_a_resource_reserved ?target_cluster_a)
        (cluster_b_resource_reserved ?target_cluster_b)
        (not
          (primary_confirmed ?primary_cc_unit)
        )
        (not
          (secondary_confirmed ?secondary_cc_unit)
        )
        (bundle_available ?assignment_bundle)
      )
    :effect
      (and
        (bundle_claimed ?assignment_bundle)
        (bundle_bound_to_cluster_a ?assignment_bundle ?target_cluster_a)
        (bundle_bound_to_cluster_b ?assignment_bundle ?target_cluster_b)
        (bundle_flag_a_resource ?assignment_bundle)
        (bundle_flag_b_resource ?assignment_bundle)
        (not
          (bundle_available ?assignment_bundle)
        )
      )
  )
  (:action activate_bundle_marker_ready
    :parameters (?assignment_bundle - assignment_bundle ?primary_cc_unit - primary_cc_unit ?cc_ability - cc_ability)
    :precondition
      (and
        (bundle_claimed ?assignment_bundle)
        (primary_ready ?primary_cc_unit)
        (assignment_ability_bound ?primary_cc_unit ?cc_ability)
        (not
          (bundle_marker_ready ?assignment_bundle)
        )
      )
    :effect (bundle_marker_ready ?assignment_bundle)
  )
  (:action bind_tactical_marker_to_controller_and_bundle
    :parameters (?assignment_controller - assignment_controller ?tactical_marker - tactical_marker ?assignment_bundle - assignment_bundle)
    :precondition
      (and
        (assignment_validated ?assignment_controller)
        (controller_bound_to_bundle ?assignment_controller ?assignment_bundle)
        (controller_marker_bound ?assignment_controller ?tactical_marker)
        (marker_available ?tactical_marker)
        (bundle_claimed ?assignment_bundle)
        (bundle_marker_ready ?assignment_bundle)
        (not
          (marker_consumed ?tactical_marker)
        )
      )
    :effect
      (and
        (marker_consumed ?tactical_marker)
        (marker_bound_to_bundle ?tactical_marker ?assignment_bundle)
        (not
          (marker_available ?tactical_marker)
        )
      )
  )
  (:action controller_process_bound_marker
    :parameters (?assignment_controller - assignment_controller ?tactical_marker - tactical_marker ?assignment_bundle - assignment_bundle ?cc_ability - cc_ability)
    :precondition
      (and
        (assignment_validated ?assignment_controller)
        (controller_marker_bound ?assignment_controller ?tactical_marker)
        (marker_consumed ?tactical_marker)
        (marker_bound_to_bundle ?tactical_marker ?assignment_bundle)
        (assignment_ability_bound ?assignment_controller ?cc_ability)
        (not
          (bundle_flag_a_resource ?assignment_bundle)
        )
        (not
          (controller_phase_marker_applied ?assignment_controller)
        )
      )
    :effect (controller_phase_marker_applied ?assignment_controller)
  )
  (:action attach_priority_marker_to_controller
    :parameters (?assignment_controller - assignment_controller ?priority_marker - priority_marker)
    :precondition
      (and
        (assignment_validated ?assignment_controller)
        (priority_marker_available ?priority_marker)
        (not
          (controller_priority_active ?assignment_controller)
        )
      )
    :effect
      (and
        (controller_priority_active ?assignment_controller)
        (controller_priority_marker_bound ?assignment_controller ?priority_marker)
        (not
          (priority_marker_available ?priority_marker)
        )
      )
  )
  (:action controller_apply_marker_and_priority
    :parameters (?assignment_controller - assignment_controller ?tactical_marker - tactical_marker ?assignment_bundle - assignment_bundle ?cc_ability - cc_ability ?priority_marker - priority_marker)
    :precondition
      (and
        (assignment_validated ?assignment_controller)
        (controller_marker_bound ?assignment_controller ?tactical_marker)
        (marker_consumed ?tactical_marker)
        (marker_bound_to_bundle ?tactical_marker ?assignment_bundle)
        (assignment_ability_bound ?assignment_controller ?cc_ability)
        (bundle_flag_a_resource ?assignment_bundle)
        (controller_priority_active ?assignment_controller)
        (controller_priority_marker_bound ?assignment_controller ?priority_marker)
        (not
          (controller_phase_marker_applied ?assignment_controller)
        )
      )
    :effect
      (and
        (controller_phase_marker_applied ?assignment_controller)
        (controller_priority_followup_set ?assignment_controller)
      )
  )
  (:action controller_consume_aux_support_phase1
    :parameters (?assignment_controller - assignment_controller ?auxiliary_support - auxiliary_support ?support_actor - support_actor ?tactical_marker - tactical_marker ?assignment_bundle - assignment_bundle)
    :precondition
      (and
        (controller_phase_marker_applied ?assignment_controller)
        (controller_auxiliary_support_bound ?assignment_controller ?auxiliary_support)
        (support_bound_to_assignment ?assignment_controller ?support_actor)
        (controller_marker_bound ?assignment_controller ?tactical_marker)
        (marker_bound_to_bundle ?tactical_marker ?assignment_bundle)
        (not
          (bundle_flag_b_resource ?assignment_bundle)
        )
        (not
          (controller_aux_supports_ready ?assignment_controller)
        )
      )
    :effect (controller_aux_supports_ready ?assignment_controller)
  )
  (:action controller_consume_aux_support_phase2
    :parameters (?assignment_controller - assignment_controller ?auxiliary_support - auxiliary_support ?support_actor - support_actor ?tactical_marker - tactical_marker ?assignment_bundle - assignment_bundle)
    :precondition
