--- domain.pddl (as passed to VAL) ---
(define (domain crafting_dependency_chain)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity - object resource_supertype - entity material_supertype - entity component_supertype - entity task_supertype - entity craft_task - task_supertype raw_resource_batch - resource_supertype material_type - resource_supertype tool - resource_supertype special_component - resource_supertype calibration_token - resource_supertype preservative_item - resource_supertype station_upgrade_a - resource_supertype station_upgrade_c - resource_supertype consumable - material_supertype module_slot - material_supertype station_upgrade_b - material_supertype material_variant_a - component_supertype material_variant_b - component_supertype assembly_core - component_supertype subassembly_supertype - craft_task subassembly_variant_supertype - craft_task subassembly_a - subassembly_supertype subassembly_b - subassembly_supertype workstation_task - subassembly_variant_supertype)

  (:predicates
    (entity_registered ?craft_task - craft_task)
    (entity_material_ready ?craft_task - craft_task)
    (entity_resource_allocated ?craft_task - craft_task)
    (entity_goal_satisfied ?craft_task - craft_task)
    (entity_release_token ?craft_task - craft_task)
    (entity_preservation_applied ?craft_task - craft_task)
    (resource_batch_available ?raw_resource_batch - raw_resource_batch)
    (resource_reserved_for_entity ?craft_task - craft_task ?raw_resource_batch - raw_resource_batch)
    (material_type_available ?material_type - material_type)
    (entity_material_binding ?craft_task - craft_task ?material_type - material_type)
    (tool_available ?tool - tool)
    (entity_tool_binding ?craft_task - craft_task ?tool - tool)
    (consumable_available ?consumable - consumable)
    (subassembly_a_consumable_reserved ?subassembly_a - subassembly_a ?consumable - consumable)
    (subassembly_b_consumable_reserved ?subassembly_b - subassembly_b ?consumable - consumable)
    (subassembly_a_variant_compat ?subassembly_a - subassembly_a ?material_variant_a - material_variant_a)
    (variant_a_reserved ?material_variant_a - material_variant_a)
    (variant_a_allocated ?material_variant_a - material_variant_a)
    (subassembly_a_ready ?subassembly_a - subassembly_a)
    (subassembly_b_variant_compat ?subassembly_b - subassembly_b ?material_variant_b - material_variant_b)
    (variant_b_reserved ?material_variant_b - material_variant_b)
    (variant_b_allocated ?material_variant_b - material_variant_b)
    (subassembly_b_ready ?subassembly_b - subassembly_b)
    (assembly_core_idle ?assembly_core - assembly_core)
    (assembly_core_reserved ?assembly_core - assembly_core)
    (core_has_variant_a ?assembly_core - assembly_core ?material_variant_a - material_variant_a)
    (core_has_variant_b ?assembly_core - assembly_core ?material_variant_b - material_variant_b)
    (core_variant_a_verified ?assembly_core - assembly_core)
    (core_variant_b_verified ?assembly_core - assembly_core)
    (core_preassembly_checked ?assembly_core - assembly_core)
    (workstation_supports_subassembly_a ?workstation - workstation_task ?subassembly_a - subassembly_a)
    (workstation_supports_subassembly_b ?workstation - workstation_task ?subassembly_b - subassembly_b)
    (workstation_core_compatible ?workstation - workstation_task ?assembly_core - assembly_core)
    (module_slot_available ?module_slot - module_slot)
    (workstation_has_module_slot ?workstation - workstation_task ?module_slot - module_slot)
    (module_installed ?module_slot - module_slot)
    (module_slot_bound_to_core ?module_slot - module_slot ?assembly_core - assembly_core)
    (workstation_config_stage_1 ?workstation - workstation_task)
    (workstation_activation_stage_configured ?workstation - workstation_task)
    (workstation_validation_passed ?workstation - workstation_task)
    (special_component_installed ?workstation - workstation_task)
    (workstation_module_bindings_verified ?workstation - workstation_task)
    (workstation_calibrated ?workstation - workstation_task)
    (workstation_final_checks_passed ?workstation - workstation_task)
    (upgrade_b_available ?station_upgrade_b - station_upgrade_b)
    (workstation_upgrade_b_attached ?workstation - workstation_task ?station_upgrade_b - station_upgrade_b)
    (workstation_upgrade_b_active ?workstation - workstation_task)
    (workstation_calibration_prep_stage_configured ?workstation - workstation_task)
    (workstation_calibration_complete ?workstation - workstation_task)
    (special_component_available ?special_component - special_component)
    (workstation_has_special_component ?workstation - workstation_task ?special_component - special_component)
    (calibration_token_available ?calibration_token - calibration_token)
    (workstation_has_calibration_token ?workstation - workstation_task ?calibration_token - calibration_token)
    (upgrade_a_available ?station_upgrade_a - station_upgrade_a)
    (workstation_has_upgrade_a ?workstation - workstation_task ?station_upgrade_a - station_upgrade_a)
    (upgrade_c_available ?station_upgrade_c - station_upgrade_c)
    (workstation_has_upgrade_c ?workstation - workstation_task ?station_upgrade_c - station_upgrade_c)
    (preservative_item_available ?preservative_item - preservative_item)
    (entity_preservative_bound ?craft_task - craft_task ?preservative_item - preservative_item)
    (subassembly_a_processed ?subassembly_a - subassembly_a)
    (subassembly_b_processed ?subassembly_b - subassembly_b)
    (workstation_ready_token ?workstation - workstation_task)
  )
  (:action create_craft_task
    :parameters (?craft_task - craft_task)
    :precondition
      (and
        (not
          (entity_registered ?craft_task)
        )
        (not
          (entity_goal_satisfied ?craft_task)
        )
      )
    :effect (entity_registered ?craft_task)
  )
  (:action assign_resource_batch_to_task
    :parameters (?craft_task - craft_task ?raw_resource_batch - raw_resource_batch)
    :precondition
      (and
        (entity_registered ?craft_task)
        (not
          (entity_resource_allocated ?craft_task)
        )
        (resource_batch_available ?raw_resource_batch)
      )
    :effect
      (and
        (entity_resource_allocated ?craft_task)
        (resource_reserved_for_entity ?craft_task ?raw_resource_batch)
        (not
          (resource_batch_available ?raw_resource_batch)
        )
      )
  )
  (:action assign_material_type_to_task
    :parameters (?craft_task - craft_task ?material_type - material_type)
    :precondition
      (and
        (entity_registered ?craft_task)
        (entity_resource_allocated ?craft_task)
        (material_type_available ?material_type)
      )
    :effect
      (and
        (entity_material_binding ?craft_task ?material_type)
        (not
          (material_type_available ?material_type)
        )
      )
  )
  (:action mark_task_material_ready
    :parameters (?craft_task - craft_task ?material_type - material_type)
    :precondition
      (and
        (entity_registered ?craft_task)
        (entity_resource_allocated ?craft_task)
        (entity_material_binding ?craft_task ?material_type)
        (not
          (entity_material_ready ?craft_task)
        )
      )
    :effect (entity_material_ready ?craft_task)
  )
  (:action release_material_binding
    :parameters (?craft_task - craft_task ?material_type - material_type)
    :precondition
      (and
        (entity_material_binding ?craft_task ?material_type)
      )
    :effect
      (and
        (material_type_available ?material_type)
        (not
          (entity_material_binding ?craft_task ?material_type)
        )
      )
  )
  (:action attach_tool_to_task
    :parameters (?craft_task - craft_task ?tool - tool)
    :precondition
      (and
        (entity_material_ready ?craft_task)
        (tool_available ?tool)
      )
    :effect
      (and
        (entity_tool_binding ?craft_task ?tool)
        (not
          (tool_available ?tool)
        )
      )
  )
  (:action detach_tool_from_task
    :parameters (?craft_task - craft_task ?tool - tool)
    :precondition
      (and
        (entity_tool_binding ?craft_task ?tool)
      )
    :effect
      (and
        (tool_available ?tool)
        (not
          (entity_tool_binding ?craft_task ?tool)
        )
      )
  )
  (:action install_station_upgrade_a
    :parameters (?workstation - workstation_task ?station_upgrade_a - station_upgrade_a)
    :precondition
      (and
        (entity_material_ready ?workstation)
        (upgrade_a_available ?station_upgrade_a)
      )
    :effect
      (and
        (workstation_has_upgrade_a ?workstation ?station_upgrade_a)
        (not
          (upgrade_a_available ?station_upgrade_a)
        )
      )
  )
  (:action remove_station_upgrade_a
    :parameters (?workstation - workstation_task ?station_upgrade_a - station_upgrade_a)
    :precondition
      (and
        (workstation_has_upgrade_a ?workstation ?station_upgrade_a)
      )
    :effect
      (and
        (upgrade_a_available ?station_upgrade_a)
        (not
          (workstation_has_upgrade_a ?workstation ?station_upgrade_a)
        )
      )
  )
  (:action install_station_upgrade_c
    :parameters (?workstation - workstation_task ?station_upgrade_c - station_upgrade_c)
    :precondition
      (and
        (entity_material_ready ?workstation)
        (upgrade_c_available ?station_upgrade_c)
      )
    :effect
      (and
        (workstation_has_upgrade_c ?workstation ?station_upgrade_c)
        (not
          (upgrade_c_available ?station_upgrade_c)
        )
      )
  )
  (:action remove_station_upgrade_c
    :parameters (?workstation - workstation_task ?station_upgrade_c - station_upgrade_c)
    :precondition
      (and
        (workstation_has_upgrade_c ?workstation ?station_upgrade_c)
      )
    :effect
      (and
        (upgrade_c_available ?station_upgrade_c)
        (not
          (workstation_has_upgrade_c ?workstation ?station_upgrade_c)
        )
      )
  )
  (:action reserve_material_variant_a_for_subassembly
    :parameters (?subassembly_a - subassembly_a ?material_variant_a - material_variant_a ?material_type - material_type)
    :precondition
      (and
        (entity_material_ready ?subassembly_a)
        (entity_material_binding ?subassembly_a ?material_type)
        (subassembly_a_variant_compat ?subassembly_a ?material_variant_a)
        (not
          (variant_a_reserved ?material_variant_a)
        )
        (not
          (variant_a_allocated ?material_variant_a)
        )
      )
    :effect (variant_a_reserved ?material_variant_a)
  )
  (:action start_processing_subassembly_a_with_tool
    :parameters (?subassembly_a - subassembly_a ?material_variant_a - material_variant_a ?tool - tool)
    :precondition
      (and
        (entity_material_ready ?subassembly_a)
        (entity_tool_binding ?subassembly_a ?tool)
        (subassembly_a_variant_compat ?subassembly_a ?material_variant_a)
        (variant_a_reserved ?material_variant_a)
        (not
          (subassembly_a_processed ?subassembly_a)
        )
      )
    :effect
      (and
        (subassembly_a_processed ?subassembly_a)
        (subassembly_a_ready ?subassembly_a)
      )
  )
  (:action assign_consumable_to_subassembly_a
    :parameters (?subassembly_a - subassembly_a ?material_variant_a - material_variant_a ?consumable - consumable)
    :precondition
      (and
        (entity_material_ready ?subassembly_a)
        (subassembly_a_variant_compat ?subassembly_a ?material_variant_a)
        (consumable_available ?consumable)
        (not
          (subassembly_a_processed ?subassembly_a)
        )
      )
    :effect
      (and
        (variant_a_allocated ?material_variant_a)
        (subassembly_a_processed ?subassembly_a)
        (subassembly_a_consumable_reserved ?subassembly_a ?consumable)
        (not
          (consumable_available ?consumable)
        )
      )
  )
  (:action finalize_subassembly_a_preparation
    :parameters (?subassembly_a - subassembly_a ?material_variant_a - material_variant_a ?material_type - material_type ?consumable - consumable)
    :precondition
      (and
        (entity_material_ready ?subassembly_a)
        (entity_material_binding ?subassembly_a ?material_type)
        (subassembly_a_variant_compat ?subassembly_a ?material_variant_a)
        (variant_a_allocated ?material_variant_a)
        (subassembly_a_consumable_reserved ?subassembly_a ?consumable)
        (not
          (subassembly_a_ready ?subassembly_a)
        )
      )
    :effect
      (and
        (variant_a_reserved ?material_variant_a)
        (subassembly_a_ready ?subassembly_a)
        (consumable_available ?consumable)
        (not
          (subassembly_a_consumable_reserved ?subassembly_a ?consumable)
        )
      )
  )
  (:action reserve_material_variant_b_for_subassembly
    :parameters (?subassembly_b - subassembly_b ?material_variant_b - material_variant_b ?material_type - material_type)
    :precondition
      (and
        (entity_material_ready ?subassembly_b)
        (entity_material_binding ?subassembly_b ?material_type)
        (subassembly_b_variant_compat ?subassembly_b ?material_variant_b)
        (not
          (variant_b_reserved ?material_variant_b)
        )
        (not
          (variant_b_allocated ?material_variant_b)
        )
      )
    :effect (variant_b_reserved ?material_variant_b)
  )
  (:action start_processing_subassembly_b_with_tool
    :parameters (?subassembly_b - subassembly_b ?material_variant_b - material_variant_b ?tool - tool)
    :precondition
      (and
        (entity_material_ready ?subassembly_b)
        (entity_tool_binding ?subassembly_b ?tool)
        (subassembly_b_variant_compat ?subassembly_b ?material_variant_b)
        (variant_b_reserved ?material_variant_b)
        (not
          (subassembly_b_processed ?subassembly_b)
        )
      )
    :effect
      (and
        (subassembly_b_processed ?subassembly_b)
        (subassembly_b_ready ?subassembly_b)
      )
  )
  (:action assign_consumable_to_subassembly_b
    :parameters (?subassembly_b - subassembly_b ?material_variant_b - material_variant_b ?consumable - consumable)
    :precondition
      (and
        (entity_material_ready ?subassembly_b)
        (subassembly_b_variant_compat ?subassembly_b ?material_variant_b)
        (consumable_available ?consumable)
        (not
          (subassembly_b_processed ?subassembly_b)
        )
      )
    :effect
      (and
        (variant_b_allocated ?material_variant_b)
        (subassembly_b_processed ?subassembly_b)
        (subassembly_b_consumable_reserved ?subassembly_b ?consumable)
        (not
          (consumable_available ?consumable)
        )
      )
  )
  (:action finalize_subassembly_b_preparation
    :parameters (?subassembly_b - subassembly_b ?material_variant_b - material_variant_b ?material_type - material_type ?consumable - consumable)
    :precondition
      (and
        (entity_material_ready ?subassembly_b)
        (entity_material_binding ?subassembly_b ?material_type)
        (subassembly_b_variant_compat ?subassembly_b ?material_variant_b)
        (variant_b_allocated ?material_variant_b)
        (subassembly_b_consumable_reserved ?subassembly_b ?consumable)
        (not
          (subassembly_b_ready ?subassembly_b)
        )
      )
    :effect
      (and
        (variant_b_reserved ?material_variant_b)
        (subassembly_b_ready ?subassembly_b)
        (consumable_available ?consumable)
        (not
          (subassembly_b_consumable_reserved ?subassembly_b ?consumable)
        )
      )
  )
  (:action claim_core_and_attach_variants
    :parameters (?subassembly_a - subassembly_a ?subassembly_b - subassembly_b ?material_variant_a - material_variant_a ?material_variant_b - material_variant_b ?assembly_core - assembly_core)
    :precondition
      (and
        (subassembly_a_processed ?subassembly_a)
        (subassembly_b_processed ?subassembly_b)
        (subassembly_a_variant_compat ?subassembly_a ?material_variant_a)
        (subassembly_b_variant_compat ?subassembly_b ?material_variant_b)
        (variant_a_reserved ?material_variant_a)
        (variant_b_reserved ?material_variant_b)
        (subassembly_a_ready ?subassembly_a)
        (subassembly_b_ready ?subassembly_b)
        (assembly_core_idle ?assembly_core)
      )
    :effect
      (and
        (assembly_core_reserved ?assembly_core)
        (core_has_variant_a ?assembly_core ?material_variant_a)
        (core_has_variant_b ?assembly_core ?material_variant_b)
        (not
          (assembly_core_idle ?assembly_core)
        )
      )
  )
  (:action claim_core_attach_and_mark_variant_a_verified
    :parameters (?subassembly_a - subassembly_a ?subassembly_b - subassembly_b ?material_variant_a - material_variant_a ?material_variant_b - material_variant_b ?assembly_core - assembly_core)
    :precondition
      (and
        (subassembly_a_processed ?subassembly_a)
        (subassembly_b_processed ?subassembly_b)
        (subassembly_a_variant_compat ?subassembly_a ?material_variant_a)
        (subassembly_b_variant_compat ?subassembly_b ?material_variant_b)
        (variant_a_allocated ?material_variant_a)
        (variant_b_reserved ?material_variant_b)
        (not
          (subassembly_a_ready ?subassembly_a)
        )
        (subassembly_b_ready ?subassembly_b)
        (assembly_core_idle ?assembly_core)
      )
    :effect
      (and
        (assembly_core_reserved ?assembly_core)
        (core_has_variant_a ?assembly_core ?material_variant_a)
        (core_has_variant_b ?assembly_core ?material_variant_b)
        (core_variant_a_verified ?assembly_core)
        (not
          (assembly_core_idle ?assembly_core)
        )
      )
  )
  (:action claim_core_attach_and_mark_variant_b_verified
    :parameters (?subassembly_a - subassembly_a ?subassembly_b - subassembly_b ?material_variant_a - material_variant_a ?material_variant_b - material_variant_b ?assembly_core - assembly_core)
    :precondition
      (and
        (subassembly_a_processed ?subassembly_a)
        (subassembly_b_processed ?subassembly_b)
        (subassembly_a_variant_compat ?subassembly_a ?material_variant_a)
        (subassembly_b_variant_compat ?subassembly_b ?material_variant_b)
        (variant_a_reserved ?material_variant_a)
        (variant_b_allocated ?material_variant_b)
        (subassembly_a_ready ?subassembly_a)
        (not
          (subassembly_b_ready ?subassembly_b)
        )
        (assembly_core_idle ?assembly_core)
      )
    :effect
      (and
        (assembly_core_reserved ?assembly_core)
        (core_has_variant_a ?assembly_core ?material_variant_a)
        (core_has_variant_b ?assembly_core ?material_variant_b)
        (core_variant_b_verified ?assembly_core)
        (not
          (assembly_core_idle ?assembly_core)
        )
      )
  )
  (:action claim_core_attach_and_mark_both_variants_verified
    :parameters (?subassembly_a - subassembly_a ?subassembly_b - subassembly_b ?material_variant_a - material_variant_a ?material_variant_b - material_variant_b ?assembly_core - assembly_core)
    :precondition
      (and
        (subassembly_a_processed ?subassembly_a)
        (subassembly_b_processed ?subassembly_b)
        (subassembly_a_variant_compat ?subassembly_a ?material_variant_a)
        (subassembly_b_variant_compat ?subassembly_b ?material_variant_b)
        (variant_a_allocated ?material_variant_a)
        (variant_b_allocated ?material_variant_b)
        (not
          (subassembly_a_ready ?subassembly_a)
        )
        (not
          (subassembly_b_ready ?subassembly_b)
        )
        (assembly_core_idle ?assembly_core)
      )
    :effect
      (and
        (assembly_core_reserved ?assembly_core)
        (core_has_variant_a ?assembly_core ?material_variant_a)
        (core_has_variant_b ?assembly_core ?material_variant_b)
        (core_variant_a_verified ?assembly_core)
        (core_variant_b_verified ?assembly_core)
        (not
          (assembly_core_idle ?assembly_core)
        )
      )
  )
  (:action verify_core_preassembly
    :parameters (?assembly_core - assembly_core ?subassembly_a - subassembly_a ?material_type - material_type)
    :precondition
      (and
        (assembly_core_reserved ?assembly_core)
        (subassembly_a_processed ?subassembly_a)
        (entity_material_binding ?subassembly_a ?material_type)
        (not
          (core_preassembly_checked ?assembly_core)
        )
      )
    :effect (core_preassembly_checked ?assembly_core)
  )
  (:action install_module_and_bind_to_core
    :parameters (?workstation - workstation_task ?module_slot - module_slot ?assembly_core - assembly_core)
    :precondition
      (and
        (entity_material_ready ?workstation)
        (workstation_core_compatible ?workstation ?assembly_core)
        (workstation_has_module_slot ?workstation ?module_slot)
        (module_slot_available ?module_slot)
        (assembly_core_reserved ?assembly_core)
        (core_preassembly_checked ?assembly_core)
        (not
          (module_installed ?module_slot)
        )
      )
    :effect
      (and
        (module_installed ?module_slot)
        (module_slot_bound_to_core ?module_slot ?assembly_core)
        (not
          (module_slot_available ?module_slot)
        )
      )
  )
  (:action configure_workstation_after_module_install
    :parameters (?workstation - workstation_task ?module_slot - module_slot ?assembly_core - assembly_core ?material_type - material_type)
    :precondition
      (and
        (entity_material_ready ?workstation)
        (workstation_has_module_slot ?workstation ?module_slot)
        (module_installed ?module_slot)
        (module_slot_bound_to_core ?module_slot ?assembly_core)
        (entity_material_binding ?workstation ?material_type)
        (not
          (core_variant_a_verified ?assembly_core)
        )
        (not
          (workstation_config_stage_1 ?workstation)
        )
      )
    :effect (workstation_config_stage_1 ?workstation)
  )
  (:action attach_special_component_to_workstation
    :parameters (?workstation - workstation_task ?special_component - special_component)
    :precondition
      (and
        (entity_material_ready ?workstation)
        (special_component_available ?special_component)
        (not
          (special_component_installed ?workstation)
        )
      )
    :effect
      (and
        (special_component_installed ?workstation)
        (workstation_has_special_component ?workstation ?special_component)
        (not
          (special_component_available ?special_component)
        )
      )
  )
  (:action finalize_workstation_module_and_component_binding
    :parameters (?workstation - workstation_task ?module_slot - module_slot ?assembly_core - assembly_core ?material_type - material_type ?special_component - special_component)
    :precondition
      (and
        (entity_material_ready ?workstation)
        (workstation_has_module_slot ?workstation ?module_slot)
        (module_installed ?module_slot)
        (module_slot_bound_to_core ?module_slot ?assembly_core)
        (entity_material_binding ?workstation ?material_type)
        (core_variant_a_verified ?assembly_core)
        (special_component_installed ?workstation)
        (workstation_has_special_component ?workstation ?special_component)
        (not
          (workstation_config_stage_1 ?workstation)
        )
      )
    :effect
      (and
        (workstation_config_stage_1 ?workstation)
        (workstation_module_bindings_verified ?workstation)
      )
  )
  (:action activate_workstation_with_upgrade_a
    :parameters (?workstation - workstation_task ?station_upgrade_a - station_upgrade_a ?tool - tool ?module_slot - module_slot ?assembly_core - assembly_core)
    :precondition
      (and
        (workstation_config_stage_1 ?workstation)
        (workstation_has_upgrade_a ?workstation ?station_upgrade_a)
        (entity_tool_binding ?workstation ?tool)
        (workstation_has_module_slot ?workstation ?module_slot)
        (module_slot_bound_to_core ?module_slot ?assembly_core)
        (not
          (core_variant_b_verified ?assembly_core)
        )
        (not
          (workstation_activation_stage_configured ?workstation)
        )
      )
    :effect (workstation_activation_stage_configured ?workstation)
  )
  (:action finalize_workstation_upgrade_a_application
    :parameters (?workstation - workstation_task ?station_upgrade_a - station_upgrade_a ?tool - tool ?module_slot - module_slot ?assembly_core - assembly_core)
    :precondition
      (and
        (workstation_config_stage_1 ?workstation)
        (workstation_has_upgrade_a ?workstation ?station_upgrade_a)
        (entity_tool_binding ?workstation ?tool)
        (workstation_has_module_slot ?workstation ?module_slot)
        (module_slot_bound_to_core ?module_slot ?assembly_core)
        (core_variant_b_verified ?assembly_core)
