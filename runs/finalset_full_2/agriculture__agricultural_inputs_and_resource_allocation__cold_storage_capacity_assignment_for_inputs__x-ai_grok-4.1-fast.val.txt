  (:action register_incoming_batch
    :parameters (?allocation_entity - allocation_entity)
    :precondition
      (and
        (not
          (allocation_entity_registered ?allocation_entity)
        )
        (not
          (capacity_assigned ?allocation_entity)
        )
      )
    :effect (allocation_entity_registered ?allocation_entity)
  )
  (:action pair_batch_with_transport
    :parameters (?allocation_entity - allocation_entity ?transport_asset - transport_asset)
    :precondition
      (and
        (allocation_entity_registered ?allocation_entity)
        (not
          (transport_assignment_confirmed ?allocation_entity)
        )
        (transport_available ?transport_asset)
      )
    :effect
      (and
        (transport_assignment_confirmed ?allocation_entity)
        (allocation_entity_assigned_transport ?allocation_entity ?transport_asset)
        (not
          (transport_available ?transport_asset)
        )
      )
  )
  (:action assign_temperature_setting
    :parameters (?allocation_entity - allocation_entity ?temperature_setting - temperature_setting)
    :precondition
      (and
        (allocation_entity_registered ?allocation_entity)
        (transport_assignment_confirmed ?allocation_entity)
        (temperature_setting_available ?temperature_setting)
      )
    :effect
      (and
        (allocation_entity_assigned_temperature ?allocation_entity ?temperature_setting)
        (not
          (temperature_setting_available ?temperature_setting)
        )
      )
  )
  (:action confirm_intake_validation
    :parameters (?allocation_entity - allocation_entity ?temperature_setting - temperature_setting)
    :precondition
      (and
        (allocation_entity_registered ?allocation_entity)
        (transport_assignment_confirmed ?allocation_entity)
        (allocation_entity_assigned_temperature ?allocation_entity ?temperature_setting)
        (not
          (allocation_entity_validated ?allocation_entity)
        )
      )
    :effect (allocation_entity_validated ?allocation_entity)
  )
  (:action release_temperature_setting
    :parameters (?allocation_entity - allocation_entity ?temperature_setting - temperature_setting)
    :precondition
      (and
        (allocation_entity_assigned_temperature ?allocation_entity ?temperature_setting)
      )
    :effect
      (and
        (temperature_setting_available ?temperature_setting)
        (not
          (allocation_entity_assigned_temperature ?allocation_entity ?temperature_setting)
        )
      )
  )
  (:action assign_technician_for_intake
    :parameters (?allocation_entity - allocation_entity ?technician - technician)

        (facility_assigned_quality_certificate ?cold_storage_facility ?quality_certificate)
      )

        (allocation_entity_validated ?primary_supply_site)
        (allocation_entity_assigned_temperature ?primary_supply_site ?temperature_setting)
        (primary_site_assigned_zone_a ?primary_supply_site ?storage_zone_a)
        (not
          (zone_a_marked_for_loading ?storage_zone_a)
        )
        (not
          (zone_a_staged ?storage_zone_a)
        )

    :effect (zone_a_marked_for_loading ?storage_zone_a)
  )
  (:action secure_primary_site_for_loading
    :parameters (?primary_supply_site - primary_supply_site ?storage_zone_a - storage_zone_a ?technician - technician)
    :precondition
      (and
        (allocation_entity_validated ?primary_supply_site)
        (allocation_entity_assigned_technician ?primary_supply_site ?technician)
        (primary_site_assigned_zone_a ?primary_supply_site ?storage_zone_a)
        (zone_a_marked_for_loading ?storage_zone_a)
        (not
          (primary_site_processing_ready ?primary_supply_site)
        )

        (primary_site_processing_ready ?primary_supply_site)

      )

        (allocation_entity_validated ?cold_storage_facility)

        (not
          (container_contains_zone_a_load ?storage_container)
        )
        (not
          (facility_pallet_slot_assignment_recorded ?cold_storage_facility)
        )
      )
    :effect (facility_pallet_slot_assignment_recorded ?cold_storage_facility)
