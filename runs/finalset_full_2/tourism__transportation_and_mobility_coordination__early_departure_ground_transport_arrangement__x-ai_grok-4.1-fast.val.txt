--- domain.pddl (as passed to VAL) ---
(define (domain early_departure_ground_transport_arrangement)
  (:requirements :strips :typing :negative-preconditions)
  (:types operational_resource - object support_resource - object spatial_type - object domain_container - object journey_element - domain_container vehicle_unit - operational_resource time_slot - operational_resource driver - operational_resource passenger_preference - operational_resource capacity_profile - operational_resource location_reference - operational_resource vehicle_feature - operational_resource operator_certificate - operational_resource auxiliary_resource - support_resource equipment_asset - support_resource special_permit - support_resource pickup_zone - spatial_type dropoff_zone - spatial_type location_reservation - spatial_type origin_destination_category - journey_element dispatch_category - journey_element pickup_unit - origin_destination_category dropoff_unit - origin_destination_category dispatch_record - dispatch_category)
  (:predicates
    (element_arrangement_requested ?journey_element - journey_element)
    (element_arrangement_confirmed ?journey_element - journey_element)
    (vehicle_allocated ?journey_element - journey_element)
    (planning_completed ?journey_element - journey_element)
    (ready_for_dispatch ?journey_element - journey_element)
    (handoff_completed ?journey_element - journey_element)
    (vehicle_available ?vehicle_unit - vehicle_unit)
    (element_allocated_vehicle ?journey_element - journey_element ?vehicle_unit - vehicle_unit)
    (time_slot_available ?time_slot - time_slot)
    (element_time_slot_assigned ?journey_element - journey_element ?time_slot - time_slot)
    (driver_available ?driver - driver)
    (element_assigned_driver ?journey_element - journey_element ?driver - driver)
    (auxiliary_resource_available ?auxiliary_resource - auxiliary_resource)
    (pickup_resource_assigned ?pickup_unit - pickup_unit ?auxiliary_resource - auxiliary_resource)
    (dropoff_resource_assigned ?dropoff_unit - dropoff_unit ?auxiliary_resource - auxiliary_resource)
    (pickup_zone_compatible ?pickup_unit - pickup_unit ?pickup_zone - pickup_zone)
    (pickup_zone_reserved ?pickup_zone - pickup_zone)
    (pickup_zone_tentative_hold ?pickup_zone - pickup_zone)
    (pickup_unit_staged ?pickup_unit - pickup_unit)
    (dropoff_assigned_zone ?dropoff_unit - dropoff_unit ?dropoff_zone - dropoff_zone)
    (dropoff_zone_reserved ?dropoff_zone - dropoff_zone)
    (dropoff_zone_tentative_hold ?dropoff_zone - dropoff_zone)
    (dropoff_unit_staged ?dropoff_unit - dropoff_unit)
    (vehicle_reservation_unallocated ?location_reservation - location_reservation)
    (vehicle_reservation_active ?location_reservation - location_reservation)
    (reservation_pickup_zone ?location_reservation - location_reservation ?pickup_zone - pickup_zone)
    (reservation_dropoff_zone ?location_reservation - location_reservation ?dropoff_zone - dropoff_zone)
    (reservation_attribute_a ?location_reservation - location_reservation)
    (reservation_attribute_b ?location_reservation - location_reservation)
    (reservation_ready_for_verification ?location_reservation - location_reservation)
    (dispatch_to_pickup ?dispatch_record - dispatch_record ?pickup_unit - pickup_unit)
    (dispatch_to_dropoff ?dispatch_record - dispatch_record ?dropoff_unit - dropoff_unit)
    (dispatch_to_reservation ?dispatch_record - dispatch_record ?location_reservation - location_reservation)
    (equipment_asset_available ?equipment_asset - equipment_asset)
    (dispatch_planned_equipment ?dispatch_record - dispatch_record ?equipment_asset - equipment_asset)
    (equipment_asset_staged ?equipment_asset - equipment_asset)
    (equipment_assigned_to_reservation ?equipment_asset - equipment_asset ?location_reservation - location_reservation)
    (dispatch_manifest_preliminary ?dispatch_record - dispatch_record)
    (dispatch_manifest_confirmed ?dispatch_record - dispatch_record)
    (dispatch_advanced_readiness ?dispatch_record - dispatch_record)
    (dispatch_preference_consumed ?dispatch_record - dispatch_record)
    (dispatch_preference_acknowledged ?dispatch_record - dispatch_record)
    (dispatch_capacity_validated ?dispatch_record - dispatch_record)
    (dispatch_run_assigned ?dispatch_record - dispatch_record)
    (special_permit_available ?special_permit - special_permit)
    (dispatch_assigned_permit ?dispatch_record - dispatch_record ?special_permit - special_permit)
    (dispatch_permit_applied ?dispatch_record - dispatch_record)
    (dispatch_permit_step_completed ?dispatch_record - dispatch_record)
    (dispatch_permit_certified ?dispatch_record - dispatch_record)
    (passenger_preference_available ?passenger_preference - passenger_preference)
    (dispatch_assigned_passenger_preference ?dispatch_record - dispatch_record ?passenger_preference - passenger_preference)
    (capacity_profile_available ?capacity_profile - capacity_profile)
    (dispatch_assigned_capacity_profile ?dispatch_record - dispatch_record ?capacity_profile - capacity_profile)
    (vehicle_feature_available ?vehicle_feature - vehicle_feature)
    (dispatch_assigned_vehicle_feature ?dispatch_record - dispatch_record ?vehicle_feature - vehicle_feature)
    (operator_certificate_available ?operator_certificate - operator_certificate)
    (dispatch_assigned_operator_certificate ?dispatch_record - dispatch_record ?operator_certificate - operator_certificate)
    (location_reference_available ?location_reference - location_reference)
    (element_location_reference ?journey_element - journey_element ?location_reference - location_reference)
    (pickup_unit_ready_for_run ?pickup_unit - pickup_unit)
    (dropoff_unit_ready_for_run ?dropoff_unit - dropoff_unit)
    (dispatch_published ?dispatch_record - dispatch_record)
  )
  (:action create_arrangement_request
    :parameters (?journey_element - journey_element)
    :precondition
      (and
        (not
          (element_arrangement_requested ?journey_element)
        )
        (not
          (planning_completed ?journey_element)
        )
      )
    :effect (element_arrangement_requested ?journey_element)
  )
  (:action provisionally_allocate_vehicle
    :parameters (?journey_element - journey_element ?vehicle_unit - vehicle_unit)
    :precondition
      (and
        (element_arrangement_requested ?journey_element)
        (not
          (vehicle_allocated ?journey_element)
        )
        (vehicle_available ?vehicle_unit)
      )
    :effect
      (and
        (vehicle_allocated ?journey_element)
        (element_allocated_vehicle ?journey_element ?vehicle_unit)
        (not
          (vehicle_available ?vehicle_unit)
        )
      )
  )
  (:action provisionally_assign_time_slot
    :parameters (?journey_element - journey_element ?time_slot - time_slot)
    :precondition
      (and
        (element_arrangement_requested ?journey_element)
        (vehicle_allocated ?journey_element)
        (time_slot_available ?time_slot)
      )
    :effect
      (and
        (element_time_slot_assigned ?journey_element ?time_slot)
        (not
          (time_slot_available ?time_slot)
        )
      )
  )
  (:action confirm_arrangement
    :parameters (?journey_element - journey_element ?time_slot - time_slot)
    :precondition
      (and
        (element_arrangement_requested ?journey_element)
        (vehicle_allocated ?journey_element)
        (element_time_slot_assigned ?journey_element ?time_slot)
        (not
          (element_arrangement_confirmed ?journey_element)
        )
      )
    :effect (element_arrangement_confirmed ?journey_element)
  )
  (:action release_provisional_time_slot
    :parameters (?journey_element - journey_element ?time_slot - time_slot)
    :precondition
      (and
        (element_time_slot_assigned ?journey_element ?time_slot)
      )
    :effect
      (and
        (time_slot_available ?time_slot)
        (not
          (element_time_slot_assigned ?journey_element ?time_slot)
        )
      )
  )
  (:action assign_driver
    :parameters (?journey_element - journey_element ?driver - driver)
    :precondition
      (and
        (element_arrangement_confirmed ?journey_element)
        (driver_available ?driver)
      )
    :effect
      (and
        (element_assigned_driver ?journey_element ?driver)
        (not
          (driver_available ?driver)
        )
      )
  )
  (:action unassign_driver
    :parameters (?journey_element - journey_element ?driver - driver)
    :precondition
      (and
        (element_assigned_driver ?journey_element ?driver)
      )
    :effect
      (and
        (driver_available ?driver)
        (not
          (element_assigned_driver ?journey_element ?driver)
        )
      )
  )
  (:action assign_vehicle_feature_to_dispatch
    :parameters (?dispatch_record - dispatch_record ?vehicle_feature - vehicle_feature)
    :precondition
      (and
        (element_arrangement_confirmed ?dispatch_record)
        (vehicle_feature_available ?vehicle_feature)
      )
    :effect
      (and
        (dispatch_assigned_vehicle_feature ?dispatch_record ?vehicle_feature)
        (not
          (vehicle_feature_available ?vehicle_feature)
        )
      )
  )
  (:action unassign_vehicle_feature_from_dispatch
    :parameters (?dispatch_record - dispatch_record ?vehicle_feature - vehicle_feature)
    :precondition
      (and
        (dispatch_assigned_vehicle_feature ?dispatch_record ?vehicle_feature)
      )
    :effect
      (and
        (vehicle_feature_available ?vehicle_feature)
        (not
          (dispatch_assigned_vehicle_feature ?dispatch_record ?vehicle_feature)
        )
      )
  )
  (:action assign_operator_certificate
    :parameters (?dispatch_record - dispatch_record ?operator_certificate - operator_certificate)
    :precondition
      (and
        (element_arrangement_confirmed ?dispatch_record)
        (operator_certificate_available ?operator_certificate)
      )
    :effect
      (and
        (dispatch_assigned_operator_certificate ?dispatch_record ?operator_certificate)
        (not
          (operator_certificate_available ?operator_certificate)
        )
      )
  )
  (:action remove_operator_certificate_from_dispatch
    :parameters (?dispatch_record - dispatch_record ?operator_certificate - operator_certificate)
    :precondition
      (and
        (dispatch_assigned_operator_certificate ?dispatch_record ?operator_certificate)
      )
    :effect
      (and
        (operator_certificate_available ?operator_certificate)
        (not
          (dispatch_assigned_operator_certificate ?dispatch_record ?operator_certificate)
        )
      )
  )
  (:action reserve_pickup_zone
    :parameters (?pickup_unit - pickup_unit ?pickup_zone - pickup_zone ?time_slot - time_slot)
    :precondition
      (and
        (element_arrangement_confirmed ?pickup_unit)
        (element_time_slot_assigned ?pickup_unit ?time_slot)
        (pickup_zone_compatible ?pickup_unit ?pickup_zone)
        (not
          (pickup_zone_reserved ?pickup_zone)
        )
        (not
          (pickup_zone_tentative_hold ?pickup_zone)
        )
      )
    :effect (pickup_zone_reserved ?pickup_zone)
  )
  (:action confirm_pickup_staging
    :parameters (?pickup_unit - pickup_unit ?pickup_zone - pickup_zone ?driver - driver)
    :precondition
      (and
        (element_arrangement_confirmed ?pickup_unit)
        (element_assigned_driver ?pickup_unit ?driver)
        (pickup_zone_compatible ?pickup_unit ?pickup_zone)
        (pickup_zone_reserved ?pickup_zone)
        (not
          (pickup_unit_ready_for_run ?pickup_unit)
        )
      )
    :effect
      (and
        (pickup_unit_ready_for_run ?pickup_unit)
        (pickup_unit_staged ?pickup_unit)
      )
  )
  (:action reserve_auxiliary_for_pickup
    :parameters (?pickup_unit - pickup_unit ?pickup_zone - pickup_zone ?auxiliary_resource - auxiliary_resource)
    :precondition
      (and
        (element_arrangement_confirmed ?pickup_unit)
        (pickup_zone_compatible ?pickup_unit ?pickup_zone)
        (auxiliary_resource_available ?auxiliary_resource)
        (not
          (pickup_unit_ready_for_run ?pickup_unit)
        )
      )
    :effect
      (and
        (pickup_zone_tentative_hold ?pickup_zone)
        (pickup_unit_ready_for_run ?pickup_unit)
        (pickup_resource_assigned ?pickup_unit ?auxiliary_resource)
        (not
          (auxiliary_resource_available ?auxiliary_resource)
        )
      )
  )
  (:action complete_pickup_staging
    :parameters (?pickup_unit - pickup_unit ?pickup_zone - pickup_zone ?time_slot - time_slot ?auxiliary_resource - auxiliary_resource)
    :precondition
      (and
        (element_arrangement_confirmed ?pickup_unit)
        (element_time_slot_assigned ?pickup_unit ?time_slot)
        (pickup_zone_compatible ?pickup_unit ?pickup_zone)
        (pickup_zone_tentative_hold ?pickup_zone)
        (pickup_resource_assigned ?pickup_unit ?auxiliary_resource)
        (not
          (pickup_unit_staged ?pickup_unit)
        )
      )
    :effect
      (and
        (pickup_zone_reserved ?pickup_zone)
        (pickup_unit_staged ?pickup_unit)
        (auxiliary_resource_available ?auxiliary_resource)
        (not
          (pickup_resource_assigned ?pickup_unit ?auxiliary_resource)
        )
      )
  )
  (:action reserve_dropoff_zone
    :parameters (?dropoff_unit - dropoff_unit ?dropoff_zone - dropoff_zone ?time_slot - time_slot)
    :precondition
      (and
        (element_arrangement_confirmed ?dropoff_unit)
        (element_time_slot_assigned ?dropoff_unit ?time_slot)
        (dropoff_assigned_zone ?dropoff_unit ?dropoff_zone)
        (not
          (dropoff_zone_reserved ?dropoff_zone)
        )
        (not
          (dropoff_zone_tentative_hold ?dropoff_zone)
        )
      )
    :effect (dropoff_zone_reserved ?dropoff_zone)
  )
  (:action confirm_dropoff_staging
    :parameters (?dropoff_unit - dropoff_unit ?dropoff_zone - dropoff_zone ?driver - driver)
    :precondition
      (and
        (element_arrangement_confirmed ?dropoff_unit)
        (element_assigned_driver ?dropoff_unit ?driver)
        (dropoff_assigned_zone ?dropoff_unit ?dropoff_zone)
        (dropoff_zone_reserved ?dropoff_zone)
        (not
          (dropoff_unit_ready_for_run ?dropoff_unit)
        )
      )
    :effect
      (and
        (dropoff_unit_ready_for_run ?dropoff_unit)
        (dropoff_unit_staged ?dropoff_unit)
      )
  )
  (:action reserve_auxiliary_for_dropoff
    :parameters (?dropoff_unit - dropoff_unit ?dropoff_zone - dropoff_zone ?auxiliary_resource - auxiliary_resource)
    :precondition
      (and
        (element_arrangement_confirmed ?dropoff_unit)
        (dropoff_assigned_zone ?dropoff_unit ?dropoff_zone)
        (auxiliary_resource_available ?auxiliary_resource)
        (not
          (dropoff_unit_ready_for_run ?dropoff_unit)
        )
      )
    :effect
      (and
        (dropoff_zone_tentative_hold ?dropoff_zone)
        (dropoff_unit_ready_for_run ?dropoff_unit)
        (dropoff_resource_assigned ?dropoff_unit ?auxiliary_resource)
        (not
          (auxiliary_resource_available ?auxiliary_resource)
        )
      )
  )
  (:action complete_dropoff_staging
    :parameters (?dropoff_unit - dropoff_unit ?dropoff_zone - dropoff_zone ?time_slot - time_slot ?auxiliary_resource - auxiliary_resource)
    :precondition
      (and
        (element_arrangement_confirmed ?dropoff_unit)
        (element_time_slot_assigned ?dropoff_unit ?time_slot)
        (dropoff_assigned_zone ?dropoff_unit ?dropoff_zone)
        (dropoff_zone_tentative_hold ?dropoff_zone)
        (dropoff_resource_assigned ?dropoff_unit ?auxiliary_resource)
        (not
          (dropoff_unit_staged ?dropoff_unit)
        )
      )
    :effect
      (and
        (dropoff_zone_reserved ?dropoff_zone)
        (dropoff_unit_staged ?dropoff_unit)
        (auxiliary_resource_available ?auxiliary_resource)
        (not
          (dropoff_resource_assigned ?dropoff_unit ?auxiliary_resource)
        )
      )
  )
  (:action activate_vehicle_reservation
    :parameters (?pickup_unit - pickup_unit ?dropoff_unit - dropoff_unit ?pickup_zone - pickup_zone ?dropoff_zone - dropoff_zone ?location_reservation - location_reservation)
    :precondition
      (and
        (pickup_unit_ready_for_run ?pickup_unit)
        (dropoff_unit_ready_for_run ?dropoff_unit)
        (pickup_zone_compatible ?pickup_unit ?pickup_zone)
        (dropoff_assigned_zone ?dropoff_unit ?dropoff_zone)
        (pickup_zone_reserved ?pickup_zone)
        (dropoff_zone_reserved ?dropoff_zone)
        (pickup_unit_staged ?pickup_unit)
        (dropoff_unit_staged ?dropoff_unit)
        (vehicle_reservation_unallocated ?location_reservation)
      )
    :effect
      (and
        (vehicle_reservation_active ?location_reservation)
        (reservation_pickup_zone ?location_reservation ?pickup_zone)
        (reservation_dropoff_zone ?location_reservation ?dropoff_zone)
        (not
          (vehicle_reservation_unallocated ?location_reservation)
        )
      )
  )
  (:action activate_vehicle_reservation_with_attr_a
    :parameters (?pickup_unit - pickup_unit ?dropoff_unit - dropoff_unit ?pickup_zone - pickup_zone ?dropoff_zone - dropoff_zone ?location_reservation - location_reservation)
    :precondition
      (and
        (pickup_unit_ready_for_run ?pickup_unit)
        (dropoff_unit_ready_for_run ?dropoff_unit)
        (pickup_zone_compatible ?pickup_unit ?pickup_zone)
        (dropoff_assigned_zone ?dropoff_unit ?dropoff_zone)
        (pickup_zone_tentative_hold ?pickup_zone)
        (dropoff_zone_reserved ?dropoff_zone)
        (not
          (pickup_unit_staged ?pickup_unit)
        )
        (dropoff_unit_staged ?dropoff_unit)
        (vehicle_reservation_unallocated ?location_reservation)
      )
    :effect
      (and
        (vehicle_reservation_active ?location_reservation)
        (reservation_pickup_zone ?location_reservation ?pickup_zone)
        (reservation_dropoff_zone ?location_reservation ?dropoff_zone)
        (reservation_attribute_a ?location_reservation)
        (not
          (vehicle_reservation_unallocated ?location_reservation)
        )
      )
  )
  (:action activate_vehicle_reservation_with_attr_b
    :parameters (?pickup_unit - pickup_unit ?dropoff_unit - dropoff_unit ?pickup_zone - pickup_zone ?dropoff_zone - dropoff_zone ?location_reservation - location_reservation)
    :precondition
      (and
        (pickup_unit_ready_for_run ?pickup_unit)
        (dropoff_unit_ready_for_run ?dropoff_unit)
        (pickup_zone_compatible ?pickup_unit ?pickup_zone)
        (dropoff_assigned_zone ?dropoff_unit ?dropoff_zone)
        (pickup_zone_reserved ?pickup_zone)
        (dropoff_zone_tentative_hold ?dropoff_zone)
        (pickup_unit_staged ?pickup_unit)
        (not
          (dropoff_unit_staged ?dropoff_unit)
        )
        (vehicle_reservation_unallocated ?location_reservation)
      )
    :effect
      (and
        (vehicle_reservation_active ?location_reservation)
        (reservation_pickup_zone ?location_reservation ?pickup_zone)
        (reservation_dropoff_zone ?location_reservation ?dropoff_zone)
        (reservation_attribute_b ?location_reservation)
        (not
          (vehicle_reservation_unallocated ?location_reservation)
        )
      )
  )
  (:action activate_vehicle_reservation_with_attrs_a_b
    :parameters (?pickup_unit - pickup_unit ?dropoff_unit - dropoff_unit ?pickup_zone - pickup_zone ?dropoff_zone - dropoff_zone ?location_reservation - location_reservation)
    :precondition
      (and
        (pickup_unit_ready_for_run ?pickup_unit)
        (dropoff_unit_ready_for_run ?dropoff_unit)
        (pickup_zone_compatible ?pickup_unit ?pickup_zone)
        (dropoff_assigned_zone ?dropoff_unit ?dropoff_zone)
        (pickup_zone_tentative_hold ?pickup_zone)
        (dropoff_zone_tentative_hold ?dropoff_zone)
        (not
          (pickup_unit_staged ?pickup_unit)
        )
        (not
          (dropoff_unit_staged ?dropoff_unit)
        )
        (vehicle_reservation_unallocated ?location_reservation)
      )
    :effect
      (and
        (vehicle_reservation_active ?location_reservation)
        (reservation_pickup_zone ?location_reservation ?pickup_zone)
        (reservation_dropoff_zone ?location_reservation ?dropoff_zone)
        (reservation_attribute_a ?location_reservation)
        (reservation_attribute_b ?location_reservation)
        (not
          (vehicle_reservation_unallocated ?location_reservation)
        )
      )
  )
  (:action clear_reservation_for_final_verification
    :parameters (?location_reservation - location_reservation ?pickup_unit - pickup_unit ?time_slot - time_slot)
    :precondition
      (and
        (vehicle_reservation_active ?location_reservation)
        (pickup_unit_ready_for_run ?pickup_unit)
        (element_time_slot_assigned ?pickup_unit ?time_slot)
        (not
          (reservation_ready_for_verification ?location_reservation)
        )
      )
    :effect (reservation_ready_for_verification ?location_reservation)
  )
  (:action stage_equipment_for_reservation
    :parameters (?dispatch_record - dispatch_record ?equipment_asset - equipment_asset ?location_reservation - location_reservation)
    :precondition
      (and
        (element_arrangement_confirmed ?dispatch_record)
        (dispatch_to_reservation ?dispatch_record ?location_reservation)
        (dispatch_planned_equipment ?dispatch_record ?equipment_asset)
        (equipment_asset_available ?equipment_asset)
        (vehicle_reservation_active ?location_reservation)
        (reservation_ready_for_verification ?location_reservation)
        (not
          (equipment_asset_staged ?equipment_asset)
        )
      )
    :effect
      (and
        (equipment_asset_staged ?equipment_asset)
        (equipment_assigned_to_reservation ?equipment_asset ?location_reservation)
        (not
          (equipment_asset_available ?equipment_asset)
        )
      )
  )
  (:action confirm_preliminary_manifest
    :parameters (?dispatch_record - dispatch_record ?equipment_asset - equipment_asset ?location_reservation - location_reservation ?time_slot - time_slot)
    :precondition
      (and
        (element_arrangement_confirmed ?dispatch_record)
        (dispatch_planned_equipment ?dispatch_record ?equipment_asset)
        (equipment_asset_staged ?equipment_asset)
        (equipment_assigned_to_reservation ?equipment_asset ?location_reservation)
        (element_time_slot_assigned ?dispatch_record ?time_slot)
        (not
          (reservation_attribute_a ?location_reservation)
        )
        (not
          (dispatch_manifest_preliminary ?dispatch_record)
        )
      )
    :effect (dispatch_manifest_preliminary ?dispatch_record)
  )
  (:action assign_passenger_preference
    :parameters (?dispatch_record - dispatch_record ?passenger_preference - passenger_preference)
    :precondition
      (and
        (element_arrangement_confirmed ?dispatch_record)
        (passenger_preference_available ?passenger_preference)
        (not
          (dispatch_preference_consumed ?dispatch_record)
        )
      )
    :effect
      (and
        (dispatch_preference_consumed ?dispatch_record)
        (dispatch_assigned_passenger_preference ?dispatch_record ?passenger_preference)
        (not
          (passenger_preference_available ?passenger_preference)
        )
      )
  )
  (:action apply_preferences_and_advance_manifest
    :parameters (?dispatch_record - dispatch_record ?equipment_asset - equipment_asset ?location_reservation - location_reservation ?time_slot - time_slot ?passenger_preference - passenger_preference)
    :precondition
      (and
        (element_arrangement_confirmed ?dispatch_record)
        (dispatch_planned_equipment ?dispatch_record ?equipment_asset)
        (equipment_asset_staged ?equipment_asset)
        (equipment_assigned_to_reservation ?equipment_asset ?location_reservation)
        (element_time_slot_assigned ?dispatch_record ?time_slot)
        (reservation_attribute_a ?location_reservation)
        (dispatch_preference_consumed ?dispatch_record)
        (dispatch_assigned_passenger_preference ?dispatch_record ?passenger_preference)
        (not
          (dispatch_manifest_preliminary ?dispatch_record)
        )
      )
    :effect
      (and
        (dispatch_manifest_preliminary ?dispatch_record)
        (dispatch_preference_acknowledged ?dispatch_record)
      )
  )
  (:action confirm_manifest_with_features
    :parameters (?dispatch_record - dispatch_record ?vehicle_feature - vehicle_feature ?driver - driver ?equipment_asset - equipment_asset ?location_reservation - location_reservation)
    :precondition
      (and
        (dispatch_manifest_preliminary ?dispatch_record)
        (dispatch_assigned_vehicle_feature ?dispatch_record ?vehicle_feature)
        (element_assigned_driver ?dispatch_record ?driver)
        (dispatch_planned_equipment ?dispatch_record ?equipment_asset)
        (equipment_assigned_to_reservation ?equipment_asset ?location_reservation)
        (not
          (reservation_attribute_b ?location_reservation)
        )
        (not
          (dispatch_manifest_confirmed ?dispatch_record)
        )
      )
    :effect (dispatch_manifest_confirmed ?dispatch_record)
