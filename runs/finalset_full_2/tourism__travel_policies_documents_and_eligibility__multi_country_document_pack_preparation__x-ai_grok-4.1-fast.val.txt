--- domain.pddl (as passed to VAL) ---
(define (domain multi_country_document_pack_preparation_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource - object document_category - object requirement_category - object case_container - object travel_case - case_container validator_instance - resource rule_interpretation - resource authority_stakeholder - resource document_format_option - resource traveler_attribute - resource evidence_document - resource verification_check - resource legal_reference - resource required_document_type - document_category country_form_instance - document_category regulatory_clause - document_category origin_requirement - requirement_category destination_requirement - requirement_category document_pack_template - requirement_category journey_segment_group - travel_case processing_group - travel_case departure_context - journey_segment_group arrival_context - journey_segment_group assessment_session - processing_group)
  (:predicates
    (assessment_registered ?travel_case - travel_case)
    (assessment_under_review ?travel_case - travel_case)
    (assessment_has_validator ?travel_case - travel_case)
    (pack_allocated ?travel_case - travel_case)
    (assessment_approved ?travel_case - travel_case)
    (eligibility_confirmed ?travel_case - travel_case)
    (validator_available ?validator_instance - validator_instance)
    (assessment_assigned_validator ?travel_case - travel_case ?validator_instance - validator_instance)
    (rule_interpretation_available ?rule_interpretation - rule_interpretation)
    (assessment_rule_interpretation ?travel_case - travel_case ?rule_interpretation - rule_interpretation)
    (authority_available ?authority_stakeholder - authority_stakeholder)
    (assessment_assigned_authority ?travel_case - travel_case ?authority_stakeholder - authority_stakeholder)
    (document_type_available ?required_document_type - required_document_type)
    (departure_has_document_type ?departure_context - departure_context ?required_document_type - required_document_type)
    (arrival_has_document_type ?arrival_context - arrival_context ?required_document_type - required_document_type)
    (departure_has_origin_requirement ?departure_context - departure_context ?origin_requirement - origin_requirement)
    (origin_requirement_flagged ?origin_requirement - origin_requirement)
    (origin_requirement_documented ?origin_requirement - origin_requirement)
    (departure_requirement_confirmed ?departure_context - departure_context)
    (arrival_has_destination_requirement ?arrival_context - arrival_context ?destination_requirement - destination_requirement)
    (destination_requirement_flagged ?destination_requirement - destination_requirement)
    (destination_requirement_documented ?destination_requirement - destination_requirement)
    (arrival_requirement_confirmed ?arrival_context - arrival_context)
    (template_available ?document_pack_template - document_pack_template)
    (template_selected ?document_pack_template - document_pack_template)
    (template_origin_requirement_link ?document_pack_template - document_pack_template ?origin_requirement - origin_requirement)
    (template_destination_requirement_link ?document_pack_template - document_pack_template ?destination_requirement - destination_requirement)
    (template_origin_flag ?document_pack_template - document_pack_template)
    (template_destination_flag ?document_pack_template - document_pack_template)
    (template_ready ?document_pack_template - document_pack_template)
    (session_departure_link ?assessment_session - assessment_session ?departure_context - departure_context)
    (session_arrival_link ?assessment_session - assessment_session ?arrival_context - arrival_context)
    (session_template_link ?assessment_session - assessment_session ?document_pack_template - document_pack_template)
    (country_form_available ?country_form_instance - country_form_instance)
    (session_has_country_form ?assessment_session - assessment_session ?country_form_instance - country_form_instance)
    (country_form_attached ?country_form_instance - country_form_instance)
    (country_form_template_link ?country_form_instance - country_form_instance ?document_pack_template - document_pack_template)
    (session_format_confirmed ?assessment_session - assessment_session)
    (session_format_ready ?assessment_session - assessment_session)
    (session_checks_started ?assessment_session - assessment_session)
    (session_has_format_option ?assessment_session - assessment_session)
    (session_format_option_confirmed ?assessment_session - assessment_session)
    (session_checks_completed ?assessment_session - assessment_session)
    (session_ready_for_finalization ?assessment_session - assessment_session)
    (regulatory_clause_available ?regulatory_clause - regulatory_clause)
    (session_regulatory_clause_link ?assessment_session - assessment_session ?regulatory_clause - regulatory_clause)
    (session_clause_applied ?assessment_session - assessment_session)
    (session_clause_verified ?assessment_session - assessment_session)
    (session_clause_confirmed ?assessment_session - assessment_session)
    (format_option_available ?document_format_option - document_format_option)
    (session_format_option_link ?assessment_session - assessment_session ?document_format_option - document_format_option)
    (traveler_attribute_available ?traveler_attribute - traveler_attribute)
    (session_has_traveler_attribute ?assessment_session - assessment_session ?traveler_attribute - traveler_attribute)
    (verification_check_available ?verification_check - verification_check)
    (session_has_verification_check ?assessment_session - assessment_session ?verification_check - verification_check)
    (legal_reference_available ?legal_reference - legal_reference)
    (session_has_legal_reference ?assessment_session - assessment_session ?legal_reference - legal_reference)
    (evidence_document_available ?evidence_document - evidence_document)
    (assessment_has_evidence ?travel_case - travel_case ?evidence_document - evidence_document)
    (departure_ready ?departure_context - departure_context)
    (arrival_ready ?arrival_context - arrival_context)
    (session_finalized ?assessment_session - assessment_session)
  )
  (:action register_travel_case
    :parameters (?travel_case - travel_case)
    :precondition
      (and
        (not
          (assessment_registered ?travel_case)
        )
        (not
          (pack_allocated ?travel_case)
        )
      )
    :effect (assessment_registered ?travel_case)
  )
  (:action assign_validator_to_case
    :parameters (?travel_case - travel_case ?validator_instance - validator_instance)
    :precondition
      (and
        (assessment_registered ?travel_case)
        (not
          (assessment_has_validator ?travel_case)
        )
        (validator_available ?validator_instance)
      )
    :effect
      (and
        (assessment_has_validator ?travel_case)
        (assessment_assigned_validator ?travel_case ?validator_instance)
        (not
          (validator_available ?validator_instance)
        )
      )
  )
  (:action attach_rule_interpretation
    :parameters (?travel_case - travel_case ?rule_interpretation - rule_interpretation)
    :precondition
      (and
        (assessment_registered ?travel_case)
        (assessment_has_validator ?travel_case)
        (rule_interpretation_available ?rule_interpretation)
      )
    :effect
      (and
        (assessment_rule_interpretation ?travel_case ?rule_interpretation)
        (not
          (rule_interpretation_available ?rule_interpretation)
        )
      )
  )
  (:action finalize_rule_interpretation
    :parameters (?travel_case - travel_case ?rule_interpretation - rule_interpretation)
    :precondition
      (and
        (assessment_registered ?travel_case)
        (assessment_has_validator ?travel_case)
        (assessment_rule_interpretation ?travel_case ?rule_interpretation)
        (not
          (assessment_under_review ?travel_case)
        )
      )
    :effect (assessment_under_review ?travel_case)
  )
  (:action detach_rule_interpretation_from_case
    :parameters (?travel_case - travel_case ?rule_interpretation - rule_interpretation)
    :precondition
      (and
        (assessment_rule_interpretation ?travel_case ?rule_interpretation)
      )
    :effect
      (and
        (rule_interpretation_available ?rule_interpretation)
        (not
          (assessment_rule_interpretation ?travel_case ?rule_interpretation)
        )
      )
  )
  (:action assign_authority_to_case
    :parameters (?travel_case - travel_case ?authority_stakeholder - authority_stakeholder)
    :precondition
      (and
        (assessment_under_review ?travel_case)
        (authority_available ?authority_stakeholder)
      )
    :effect
      (and
        (assessment_assigned_authority ?travel_case ?authority_stakeholder)
        (not
          (authority_available ?authority_stakeholder)
        )
      )
  )
  (:action release_authority_from_case
    :parameters (?travel_case - travel_case ?authority_stakeholder - authority_stakeholder)
    :precondition
      (and
        (assessment_assigned_authority ?travel_case ?authority_stakeholder)
      )
    :effect
      (and
        (authority_available ?authority_stakeholder)
        (not
          (assessment_assigned_authority ?travel_case ?authority_stakeholder)
        )
      )
  )
  (:action attach_verification_check_to_session
    :parameters (?assessment_session - assessment_session ?verification_check - verification_check)
    :precondition
      (and
        (assessment_under_review ?assessment_session)
        (verification_check_available ?verification_check)
      )
    :effect
      (and
        (session_has_verification_check ?assessment_session ?verification_check)
        (not
          (verification_check_available ?verification_check)
        )
      )
  )
  (:action detach_verification_check_from_session
    :parameters (?assessment_session - assessment_session ?verification_check - verification_check)
    :precondition
      (and
        (session_has_verification_check ?assessment_session ?verification_check)
      )
    :effect
      (and
        (verification_check_available ?verification_check)
        (not
          (session_has_verification_check ?assessment_session ?verification_check)
        )
      )
  )
  (:action attach_legal_reference_to_session
    :parameters (?assessment_session - assessment_session ?legal_reference - legal_reference)
    :precondition
      (and
        (assessment_under_review ?assessment_session)
        (legal_reference_available ?legal_reference)
      )
    :effect
      (and
        (session_has_legal_reference ?assessment_session ?legal_reference)
        (not
          (legal_reference_available ?legal_reference)
        )
      )
  )
  (:action detach_legal_reference_from_session
    :parameters (?assessment_session - assessment_session ?legal_reference - legal_reference)
    :precondition
      (and
        (session_has_legal_reference ?assessment_session ?legal_reference)
      )
    :effect
      (and
        (legal_reference_available ?legal_reference)
        (not
          (session_has_legal_reference ?assessment_session ?legal_reference)
        )
      )
  )
  (:action flag_origin_requirement_for_context
    :parameters (?departure_context - departure_context ?origin_requirement - origin_requirement ?rule_interpretation - rule_interpretation)
    :precondition
      (and
        (assessment_under_review ?departure_context)
        (assessment_rule_interpretation ?departure_context ?rule_interpretation)
        (departure_has_origin_requirement ?departure_context ?origin_requirement)
        (not
          (origin_requirement_flagged ?origin_requirement)
        )
        (not
          (origin_requirement_documented ?origin_requirement)
        )
      )
    :effect (origin_requirement_flagged ?origin_requirement)
  )
  (:action verify_origin_requirement_with_authority
    :parameters (?departure_context - departure_context ?origin_requirement - origin_requirement ?authority_stakeholder - authority_stakeholder)
    :precondition
      (and
        (assessment_under_review ?departure_context)
        (assessment_assigned_authority ?departure_context ?authority_stakeholder)
        (departure_has_origin_requirement ?departure_context ?origin_requirement)
        (origin_requirement_flagged ?origin_requirement)
        (not
          (departure_ready ?departure_context)
        )
      )
    :effect
      (and
        (departure_ready ?departure_context)
        (departure_requirement_confirmed ?departure_context)
      )
  )
  (:action record_origin_document_for_requirement
    :parameters (?departure_context - departure_context ?origin_requirement - origin_requirement ?required_document_type - required_document_type)
    :precondition
      (and
        (assessment_under_review ?departure_context)
        (departure_has_origin_requirement ?departure_context ?origin_requirement)
        (document_type_available ?required_document_type)
        (not
          (departure_ready ?departure_context)
        )
      )
    :effect
      (and
        (origin_requirement_documented ?origin_requirement)
        (departure_ready ?departure_context)
        (departure_has_document_type ?departure_context ?required_document_type)
        (not
          (document_type_available ?required_document_type)
        )
      )
  )
  (:action reconcile_origin_requirement_submission
    :parameters (?departure_context - departure_context ?origin_requirement - origin_requirement ?rule_interpretation - rule_interpretation ?required_document_type - required_document_type)
    :precondition
      (and
        (assessment_under_review ?departure_context)
        (assessment_rule_interpretation ?departure_context ?rule_interpretation)
        (departure_has_origin_requirement ?departure_context ?origin_requirement)
        (origin_requirement_documented ?origin_requirement)
        (departure_has_document_type ?departure_context ?required_document_type)
        (not
          (departure_requirement_confirmed ?departure_context)
        )
      )
    :effect
      (and
        (origin_requirement_flagged ?origin_requirement)
        (departure_requirement_confirmed ?departure_context)
        (document_type_available ?required_document_type)
        (not
          (departure_has_document_type ?departure_context ?required_document_type)
        )
      )
  )
  (:action flag_destination_requirement_for_context
    :parameters (?arrival_context - arrival_context ?destination_requirement - destination_requirement ?rule_interpretation - rule_interpretation)
    :precondition
      (and
        (assessment_under_review ?arrival_context)
        (assessment_rule_interpretation ?arrival_context ?rule_interpretation)
        (arrival_has_destination_requirement ?arrival_context ?destination_requirement)
        (not
          (destination_requirement_flagged ?destination_requirement)
        )
        (not
          (destination_requirement_documented ?destination_requirement)
        )
      )
    :effect (destination_requirement_flagged ?destination_requirement)
  )
  (:action verify_destination_requirement_with_authority
    :parameters (?arrival_context - arrival_context ?destination_requirement - destination_requirement ?authority_stakeholder - authority_stakeholder)
    :precondition
      (and
        (assessment_under_review ?arrival_context)
        (assessment_assigned_authority ?arrival_context ?authority_stakeholder)
        (arrival_has_destination_requirement ?arrival_context ?destination_requirement)
        (destination_requirement_flagged ?destination_requirement)
        (not
          (arrival_ready ?arrival_context)
        )
      )
    :effect
      (and
        (arrival_ready ?arrival_context)
        (arrival_requirement_confirmed ?arrival_context)
      )
  )
  (:action record_destination_document_for_requirement
    :parameters (?arrival_context - arrival_context ?destination_requirement - destination_requirement ?required_document_type - required_document_type)
    :precondition
      (and
        (assessment_under_review ?arrival_context)
        (arrival_has_destination_requirement ?arrival_context ?destination_requirement)
        (document_type_available ?required_document_type)
        (not
          (arrival_ready ?arrival_context)
        )
      )
    :effect
      (and
        (destination_requirement_documented ?destination_requirement)
        (arrival_ready ?arrival_context)
        (arrival_has_document_type ?arrival_context ?required_document_type)
        (not
          (document_type_available ?required_document_type)
        )
      )
  )
  (:action reconcile_destination_requirement_submission
    :parameters (?arrival_context - arrival_context ?destination_requirement - destination_requirement ?rule_interpretation - rule_interpretation ?required_document_type - required_document_type)
    :precondition
      (and
        (assessment_under_review ?arrival_context)
        (assessment_rule_interpretation ?arrival_context ?rule_interpretation)
        (arrival_has_destination_requirement ?arrival_context ?destination_requirement)
        (destination_requirement_documented ?destination_requirement)
        (arrival_has_document_type ?arrival_context ?required_document_type)
        (not
          (arrival_requirement_confirmed ?arrival_context)
        )
      )
    :effect
      (and
        (destination_requirement_flagged ?destination_requirement)
        (arrival_requirement_confirmed ?arrival_context)
        (document_type_available ?required_document_type)
        (not
          (arrival_has_document_type ?arrival_context ?required_document_type)
        )
      )
  )
  (:action select_and_link_template
    :parameters (?departure_context - departure_context ?arrival_context - arrival_context ?origin_requirement - origin_requirement ?destination_requirement - destination_requirement ?document_pack_template - document_pack_template)
    :precondition
      (and
        (departure_ready ?departure_context)
        (arrival_ready ?arrival_context)
        (departure_has_origin_requirement ?departure_context ?origin_requirement)
        (arrival_has_destination_requirement ?arrival_context ?destination_requirement)
        (origin_requirement_flagged ?origin_requirement)
        (destination_requirement_flagged ?destination_requirement)
        (departure_requirement_confirmed ?departure_context)
        (arrival_requirement_confirmed ?arrival_context)
        (template_available ?document_pack_template)
      )
    :effect
      (and
        (template_selected ?document_pack_template)
        (template_origin_requirement_link ?document_pack_template ?origin_requirement)
        (template_destination_requirement_link ?document_pack_template ?destination_requirement)
        (not
          (template_available ?document_pack_template)
        )
      )
  )
  (:action select_template_with_origin_enrichment
    :parameters (?departure_context - departure_context ?arrival_context - arrival_context ?origin_requirement - origin_requirement ?destination_requirement - destination_requirement ?document_pack_template - document_pack_template)
    :precondition
      (and
        (departure_ready ?departure_context)
        (arrival_ready ?arrival_context)
        (departure_has_origin_requirement ?departure_context ?origin_requirement)
        (arrival_has_destination_requirement ?arrival_context ?destination_requirement)
        (origin_requirement_documented ?origin_requirement)
        (destination_requirement_flagged ?destination_requirement)
        (not
          (departure_requirement_confirmed ?departure_context)
        )
        (arrival_requirement_confirmed ?arrival_context)
        (template_available ?document_pack_template)
      )
    :effect
      (and
        (template_selected ?document_pack_template)
        (template_origin_requirement_link ?document_pack_template ?origin_requirement)
        (template_destination_requirement_link ?document_pack_template ?destination_requirement)
        (template_origin_flag ?document_pack_template)
        (not
          (template_available ?document_pack_template)
        )
      )
  )
  (:action select_template_with_destination_enrichment
    :parameters (?departure_context - departure_context ?arrival_context - arrival_context ?origin_requirement - origin_requirement ?destination_requirement - destination_requirement ?document_pack_template - document_pack_template)
    :precondition
      (and
        (departure_ready ?departure_context)
        (arrival_ready ?arrival_context)
        (departure_has_origin_requirement ?departure_context ?origin_requirement)
        (arrival_has_destination_requirement ?arrival_context ?destination_requirement)
        (origin_requirement_flagged ?origin_requirement)
        (destination_requirement_documented ?destination_requirement)
        (departure_requirement_confirmed ?departure_context)
        (not
          (arrival_requirement_confirmed ?arrival_context)
        )
        (template_available ?document_pack_template)
      )
    :effect
      (and
        (template_selected ?document_pack_template)
        (template_origin_requirement_link ?document_pack_template ?origin_requirement)
        (template_destination_requirement_link ?document_pack_template ?destination_requirement)
        (template_destination_flag ?document_pack_template)
        (not
          (template_available ?document_pack_template)
        )
      )
  )
  (:action select_template_with_full_enrichment
    :parameters (?departure_context - departure_context ?arrival_context - arrival_context ?origin_requirement - origin_requirement ?destination_requirement - destination_requirement ?document_pack_template - document_pack_template)
    :precondition
      (and
        (departure_ready ?departure_context)
        (arrival_ready ?arrival_context)
        (departure_has_origin_requirement ?departure_context ?origin_requirement)
        (arrival_has_destination_requirement ?arrival_context ?destination_requirement)
        (origin_requirement_documented ?origin_requirement)
        (destination_requirement_documented ?destination_requirement)
        (not
          (departure_requirement_confirmed ?departure_context)
        )
        (not
          (arrival_requirement_confirmed ?arrival_context)
        )
        (template_available ?document_pack_template)
      )
    :effect
      (and
        (template_selected ?document_pack_template)
        (template_origin_requirement_link ?document_pack_template ?origin_requirement)
        (template_destination_requirement_link ?document_pack_template ?destination_requirement)
        (template_origin_flag ?document_pack_template)
        (template_destination_flag ?document_pack_template)
        (not
          (template_available ?document_pack_template)
        )
      )
  )
  (:action mark_template_as_ready
    :parameters (?document_pack_template - document_pack_template ?departure_context - departure_context ?rule_interpretation - rule_interpretation)
    :precondition
      (and
        (template_selected ?document_pack_template)
        (departure_ready ?departure_context)
        (assessment_rule_interpretation ?departure_context ?rule_interpretation)
        (not
          (template_ready ?document_pack_template)
        )
      )
    :effect (template_ready ?document_pack_template)
  )
  (:action attach_country_form_to_session
    :parameters (?assessment_session - assessment_session ?country_form_instance - country_form_instance ?document_pack_template - document_pack_template)
    :precondition
      (and
        (assessment_under_review ?assessment_session)
        (session_template_link ?assessment_session ?document_pack_template)
        (session_has_country_form ?assessment_session ?country_form_instance)
        (country_form_available ?country_form_instance)
        (template_selected ?document_pack_template)
        (template_ready ?document_pack_template)
        (not
          (country_form_attached ?country_form_instance)
        )
      )
    :effect
      (and
        (country_form_attached ?country_form_instance)
        (country_form_template_link ?country_form_instance ?document_pack_template)
        (not
          (country_form_available ?country_form_instance)
        )
      )
  )
  (:action activate_country_form_in_session
    :parameters (?assessment_session - assessment_session ?country_form_instance - country_form_instance ?document_pack_template - document_pack_template ?rule_interpretation - rule_interpretation)
    :precondition
      (and
        (assessment_under_review ?assessment_session)
        (session_has_country_form ?assessment_session ?country_form_instance)
        (country_form_attached ?country_form_instance)
        (country_form_template_link ?country_form_instance ?document_pack_template)
        (assessment_rule_interpretation ?assessment_session ?rule_interpretation)
        (not
          (template_origin_flag ?document_pack_template)
        )
        (not
          (session_format_confirmed ?assessment_session)
        )
      )
    :effect (session_format_confirmed ?assessment_session)
  )
  (:action attach_format_option_to_session
    :parameters (?assessment_session - assessment_session ?document_format_option - document_format_option)
    :precondition
      (and
        (assessment_under_review ?assessment_session)
        (format_option_available ?document_format_option)
        (not
          (session_has_format_option ?assessment_session)
        )
      )
    :effect
      (and
        (session_has_format_option ?assessment_session)
        (session_format_option_link ?assessment_session ?document_format_option)
        (not
          (format_option_available ?document_format_option)
        )
      )
  )
  (:action apply_format_option_with_form_and_template
    :parameters (?assessment_session - assessment_session ?country_form_instance - country_form_instance ?document_pack_template - document_pack_template ?rule_interpretation - rule_interpretation ?document_format_option - document_format_option)
    :precondition
      (and
        (assessment_under_review ?assessment_session)
        (session_has_country_form ?assessment_session ?country_form_instance)
        (country_form_attached ?country_form_instance)
        (country_form_template_link ?country_form_instance ?document_pack_template)
        (assessment_rule_interpretation ?assessment_session ?rule_interpretation)
        (template_origin_flag ?document_pack_template)
        (session_has_format_option ?assessment_session)
        (session_format_option_link ?assessment_session ?document_format_option)
        (not
          (session_format_confirmed ?assessment_session)
        )
      )
    :effect
      (and
        (session_format_confirmed ?assessment_session)
        (session_format_option_confirmed ?assessment_session)
      )
  )
  (:action initiate_format_and_verification_checks
    :parameters (?assessment_session - assessment_session ?verification_check - verification_check ?authority_stakeholder - authority_stakeholder ?country_form_instance - country_form_instance ?document_pack_template - document_pack_template)
    :precondition
