--- domain.pddl (as passed to VAL) ---
(define (domain agricultural_labor_shift_allocation_peak_period)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity - object operational_resource_category - entity material_resource_category - entity logistics_resource_category - entity domain_root - entity labor_entity - domain_root equipment_unit - operational_resource_category skill_certification - operational_resource_category supervisor - operational_resource_category authorization_document - operational_resource_category toolkit - operational_resource_category contingency_reserve - operational_resource_category specialized_equipment - operational_resource_category weather_constraint - operational_resource_category consumable_supply - material_resource_category work_package - material_resource_category incentive_voucher - material_resource_category field_parcel - logistics_resource_category transport_route - logistics_resource_category shift_slot - logistics_resource_category labor_category - labor_entity labor_subcategory - labor_entity field_worker - labor_category support_worker - labor_category work_crew - labor_subcategory)

  (:predicates
    (labor_registered ?labor_entity_var - labor_entity)
    (assignment_cleared ?labor_entity_var - labor_entity)
    (equipment_linked ?labor_entity_var - labor_entity)
    (labor_returned_to_pool ?labor_entity_var - labor_entity)
    (on_shift ?labor_entity_var - labor_entity)
    (shift_completed ?labor_entity_var - labor_entity)
    (equipment_available ?equipment_unit_var - equipment_unit)
    (labor_equipment_pair ?labor_entity_var - labor_entity ?equipment_unit_var - equipment_unit)
    (skill_cert_available ?skill_certification_var - skill_certification)
    (labor_skill_pair ?labor_entity_var - labor_entity ?skill_certification_var - skill_certification)
    (supervisor_available ?supervisor_var - supervisor)
    (labor_supervisor_pair ?labor_entity_var - labor_entity ?supervisor_var - supervisor)
    (consumable_available ?consumable_supply_var - consumable_supply)
    (field_worker_supply_pair ?field_worker_var - field_worker ?consumable_supply_var - consumable_supply)
    (support_worker_supply_pair ?support_worker_var - support_worker ?consumable_supply_var - consumable_supply)
    (worker_parcel_pair ?field_worker_var - field_worker ?field_parcel_var - field_parcel)
    (parcel_reserved ?field_parcel_var - field_parcel)
    (parcel_buffered ?field_parcel_var - field_parcel)
    (field_worker_deployed ?field_worker_var - field_worker)
    (support_worker_route_pair ?support_worker_var - support_worker ?transport_route_var - transport_route)
    (route_reserved ?transport_route_var - transport_route)
    (route_buffered ?transport_route_var - transport_route)
    (support_worker_deployed ?support_worker_var - support_worker)
    (shift_open ?shift_slot_var - shift_slot)
    (shift_staged ?shift_slot_var - shift_slot)
    (shift_parcel_linked ?shift_slot_var - shift_slot ?field_parcel_var - field_parcel)
    (shift_route_linked ?shift_slot_var - shift_slot ?transport_route_var - transport_route)
    (shift_flag_primary ?shift_slot_var - shift_slot)
    (shift_flag_secondary ?shift_slot_var - shift_slot)
    (shift_locked ?shift_slot_var - shift_slot)
    (crew_field_worker_pair ?work_crew_var - work_crew ?field_worker_var - field_worker)
    (crew_support_worker_pair ?work_crew_var - work_crew ?support_worker_var - support_worker)
    (crew_shift_pair ?work_crew_var - work_crew ?shift_slot_var - shift_slot)
    (work_package_available ?work_package_var - work_package)
    (crew_work_package_pair ?work_crew_var - work_crew ?work_package_var - work_package)
    (work_package_committed ?work_package_var - work_package)
    (package_shift_pair ?work_package_var - work_package ?shift_slot_var - shift_slot)
    (crew_verified ?work_crew_var - work_crew)
    (crew_approved ?work_crew_var - work_crew)
    (crew_ready ?work_crew_var - work_crew)
    (crew_authorization_document_linked ?work_crew_var - work_crew)
    (crew_clearance_recorded ?work_crew_var - work_crew)
    (crew_resourced ?work_crew_var - work_crew)
    (crew_mobilized ?work_crew_var - work_crew)
    (incentive_voucher_available ?incentive_voucher_var - incentive_voucher)
    (crew_voucher_pair ?work_crew_var - work_crew ?incentive_voucher_var - incentive_voucher)
    (crew_incentivized ?work_crew_var - work_crew)
    (crew_toolkit_ready ?work_crew_var - work_crew)
    (crew_weather_reviewed ?work_crew_var - work_crew)
    (authorization_document_available ?authorization_document_var - authorization_document)
    (crew_authorization_pair ?work_crew_var - work_crew ?authorization_document_var - authorization_document)
    (toolkit_available ?toolkit_var - toolkit)
    (crew_toolkit_pair ?work_crew_var - work_crew ?toolkit_var - toolkit)
    (specialized_equipment_available ?specialized_equipment_var - specialized_equipment)
    (crew_specialized_equipment_pair ?work_crew_var - work_crew ?specialized_equipment_var - specialized_equipment)
    (weather_constraint_active ?weather_constraint_var - weather_constraint)
    (crew_weather_pair ?work_crew_var - work_crew ?weather_constraint_var - weather_constraint)
    (contingency_reserve_available ?contingency_reserve_var - contingency_reserve)
    (labor_reserve_pair ?labor_entity_var - labor_entity ?contingency_reserve_var - contingency_reserve)
    (field_worker_staged ?field_worker_var - field_worker)
    (support_worker_staged ?support_worker_var - support_worker)
    (crew_active ?work_crew_var - work_crew)
  )
  (:action register_labor_entity
    :parameters (?labor_entity_var - labor_entity)
    :precondition
      (and
        (not
          (labor_registered ?labor_entity_var)
        )
        (not
          (labor_returned_to_pool ?labor_entity_var)
        )
      )
    :effect (labor_registered ?labor_entity_var)
  )
  (:action allocate_equipment_to_labor_entity
    :parameters (?labor_entity_var - labor_entity ?equipment_unit_var - equipment_unit)
    :precondition
      (and
        (labor_registered ?labor_entity_var)
        (not
          (equipment_linked ?labor_entity_var)
        )
        (equipment_available ?equipment_unit_var)
      )
    :effect
      (and
        (equipment_linked ?labor_entity_var)
        (labor_equipment_pair ?labor_entity_var ?equipment_unit_var)
        (not
          (equipment_available ?equipment_unit_var)
        )
      )
  )
  (:action assign_skill_certification_to_labor_entity
    :parameters (?labor_entity_var - labor_entity ?skill_certification_var - skill_certification)
    :precondition
      (and
        (labor_registered ?labor_entity_var)
        (equipment_linked ?labor_entity_var)
        (skill_cert_available ?skill_certification_var)
      )
    :effect
      (and
        (labor_skill_pair ?labor_entity_var ?skill_certification_var)
        (not
          (skill_cert_available ?skill_certification_var)
        )
      )
  )
  (:action clear_labor_for_assignment
    :parameters (?labor_entity_var - labor_entity ?skill_certification_var - skill_certification)
    :precondition
      (and
        (labor_registered ?labor_entity_var)
        (equipment_linked ?labor_entity_var)
        (labor_skill_pair ?labor_entity_var ?skill_certification_var)
        (not
          (assignment_cleared ?labor_entity_var)
        )
      )
    :effect (assignment_cleared ?labor_entity_var)
  )
  (:action release_skill_certification_from_labor_entity
    :parameters (?labor_entity_var - labor_entity ?skill_certification_var - skill_certification)
    :precondition
      (and
        (labor_skill_pair ?labor_entity_var ?skill_certification_var)
      )
    :effect
      (and
        (skill_cert_available ?skill_certification_var)
        (not
          (labor_skill_pair ?labor_entity_var ?skill_certification_var)
        )
      )
  )
  (:action assign_supervisor_to_labor_entity
    :parameters (?labor_entity_var - labor_entity ?supervisor_var - supervisor)
    :precondition
      (and
        (assignment_cleared ?labor_entity_var)
        (supervisor_available ?supervisor_var)
      )
    :effect
      (and
        (labor_supervisor_pair ?labor_entity_var ?supervisor_var)
        (not
          (supervisor_available ?supervisor_var)
        )
      )
  )
  (:action release_supervisor_from_labor_entity
    :parameters (?labor_entity_var - labor_entity ?supervisor_var - supervisor)
    :precondition
      (and
        (labor_supervisor_pair ?labor_entity_var ?supervisor_var)
      )
    :effect
      (and
        (supervisor_available ?supervisor_var)
        (not
          (labor_supervisor_pair ?labor_entity_var ?supervisor_var)
        )
      )
  )
  (:action allocate_specialized_equipment_to_crew
    :parameters (?work_crew_var - work_crew ?specialized_equipment_var - specialized_equipment)
    :precondition
      (and
        (assignment_cleared ?work_crew_var)
        (specialized_equipment_available ?specialized_equipment_var)
      )
    :effect
      (and
        (crew_specialized_equipment_pair ?work_crew_var ?specialized_equipment_var)
        (not
          (specialized_equipment_available ?specialized_equipment_var)
        )
      )
  )
  (:action release_specialized_equipment_from_crew
    :parameters (?work_crew_var - work_crew ?specialized_equipment_var - specialized_equipment)
    :precondition
      (and
        (crew_specialized_equipment_pair ?work_crew_var ?specialized_equipment_var)
      )
    :effect
      (and
        (specialized_equipment_available ?specialized_equipment_var)
        (not
          (crew_specialized_equipment_pair ?work_crew_var ?specialized_equipment_var)
        )
      )
  )
  (:action bind_weather_constraint_to_crew
    :parameters (?work_crew_var - work_crew ?weather_constraint_var - weather_constraint)
    :precondition
      (and
        (assignment_cleared ?work_crew_var)
        (weather_constraint_active ?weather_constraint_var)
      )
    :effect
      (and
        (crew_weather_pair ?work_crew_var ?weather_constraint_var)
        (not
          (weather_constraint_active ?weather_constraint_var)
        )
      )
  )
  (:action release_weather_constraint_from_crew
    :parameters (?work_crew_var - work_crew ?weather_constraint_var - weather_constraint)
    :precondition
      (and
        (crew_weather_pair ?work_crew_var ?weather_constraint_var)
      )
    :effect
      (and
        (weather_constraint_active ?weather_constraint_var)
        (not
          (crew_weather_pair ?work_crew_var ?weather_constraint_var)
        )
      )
  )
  (:action stage_field_worker_at_parcel
    :parameters (?field_worker_var - field_worker ?field_parcel_var - field_parcel ?skill_certification_var - skill_certification)
    :precondition
      (and
        (assignment_cleared ?field_worker_var)
        (labor_skill_pair ?field_worker_var ?skill_certification_var)
        (worker_parcel_pair ?field_worker_var ?field_parcel_var)
        (not
          (parcel_reserved ?field_parcel_var)
        )
        (not
          (parcel_buffered ?field_parcel_var)
        )
      )
    :effect (parcel_reserved ?field_parcel_var)
  )
  (:action dispatch_field_worker_on_shift
    :parameters (?field_worker_var - field_worker ?field_parcel_var - field_parcel ?supervisor_var - supervisor)
    :precondition
      (and
        (assignment_cleared ?field_worker_var)
        (labor_supervisor_pair ?field_worker_var ?supervisor_var)
        (worker_parcel_pair ?field_worker_var ?field_parcel_var)
        (parcel_reserved ?field_parcel_var)
        (not
          (field_worker_staged ?field_worker_var)
        )
      )
    :effect
      (and
        (field_worker_staged ?field_worker_var)
        (field_worker_deployed ?field_worker_var)
      )
  )
  (:action provision_field_worker_consumables
    :parameters (?field_worker_var - field_worker ?field_parcel_var - field_parcel ?consumable_supply_var - consumable_supply)
    :precondition
      (and
        (assignment_cleared ?field_worker_var)
        (worker_parcel_pair ?field_worker_var ?field_parcel_var)
        (consumable_available ?consumable_supply_var)
        (not
          (field_worker_staged ?field_worker_var)
        )
      )
    :effect
      (and
        (parcel_buffered ?field_parcel_var)
        (field_worker_staged ?field_worker_var)
        (field_worker_supply_pair ?field_worker_var ?consumable_supply_var)
        (not
          (consumable_available ?consumable_supply_var)
        )
      )
  )
  (:action replenish_field_worker_consumables
    :parameters (?field_worker_var - field_worker ?field_parcel_var - field_parcel ?skill_certification_var - skill_certification ?consumable_supply_var - consumable_supply)
    :precondition
      (and
        (assignment_cleared ?field_worker_var)
        (labor_skill_pair ?field_worker_var ?skill_certification_var)
        (worker_parcel_pair ?field_worker_var ?field_parcel_var)
        (parcel_buffered ?field_parcel_var)
        (field_worker_supply_pair ?field_worker_var ?consumable_supply_var)
        (not
          (field_worker_deployed ?field_worker_var)
        )
      )
    :effect
      (and
        (parcel_reserved ?field_parcel_var)
        (field_worker_deployed ?field_worker_var)
        (consumable_available ?consumable_supply_var)
        (not
          (field_worker_supply_pair ?field_worker_var ?consumable_supply_var)
        )
      )
  )
  (:action stage_support_worker_on_route
    :parameters (?support_worker_var - support_worker ?transport_route_var - transport_route ?skill_certification_var - skill_certification)
    :precondition
      (and
        (assignment_cleared ?support_worker_var)
        (labor_skill_pair ?support_worker_var ?skill_certification_var)
        (support_worker_route_pair ?support_worker_var ?transport_route_var)
        (not
          (route_reserved ?transport_route_var)
        )
        (not
          (route_buffered ?transport_route_var)
        )
      )
    :effect (route_reserved ?transport_route_var)
  )
  (:action dispatch_support_worker_on_shift
    :parameters (?support_worker_var - support_worker ?transport_route_var - transport_route ?supervisor_var - supervisor)
    :precondition
      (and
        (assignment_cleared ?support_worker_var)
        (labor_supervisor_pair ?support_worker_var ?supervisor_var)
        (support_worker_route_pair ?support_worker_var ?transport_route_var)
        (route_reserved ?transport_route_var)
        (not
          (support_worker_staged ?support_worker_var)
        )
      )
    :effect
      (and
        (support_worker_staged ?support_worker_var)
        (support_worker_deployed ?support_worker_var)
      )
  )
  (:action provision_support_worker_consumables
    :parameters (?support_worker_var - support_worker ?transport_route_var - transport_route ?consumable_supply_var - consumable_supply)
    :precondition
      (and
        (assignment_cleared ?support_worker_var)
        (support_worker_route_pair ?support_worker_var ?transport_route_var)
        (consumable_available ?consumable_supply_var)
        (not
          (support_worker_staged ?support_worker_var)
        )
      )
    :effect
      (and
        (route_buffered ?transport_route_var)
        (support_worker_staged ?support_worker_var)
        (support_worker_supply_pair ?support_worker_var ?consumable_supply_var)
        (not
          (consumable_available ?consumable_supply_var)
        )
      )
  )
  (:action replenish_support_worker_consumables
    :parameters (?support_worker_var - support_worker ?transport_route_var - transport_route ?skill_certification_var - skill_certification ?consumable_supply_var - consumable_supply)
    :precondition
      (and
        (assignment_cleared ?support_worker_var)
        (labor_skill_pair ?support_worker_var ?skill_certification_var)
        (support_worker_route_pair ?support_worker_var ?transport_route_var)
        (route_buffered ?transport_route_var)
        (support_worker_supply_pair ?support_worker_var ?consumable_supply_var)
        (not
          (support_worker_deployed ?support_worker_var)
        )
      )
    :effect
      (and
        (route_reserved ?transport_route_var)
        (support_worker_deployed ?support_worker_var)
        (consumable_available ?consumable_supply_var)
        (not
          (support_worker_supply_pair ?support_worker_var ?consumable_supply_var)
        )
      )
  )
  (:action open_shift_slot
    :parameters (?field_worker_var - field_worker ?support_worker_var - support_worker ?field_parcel_var - field_parcel ?transport_route_var - transport_route ?shift_slot_var - shift_slot)
    :precondition
      (and
        (field_worker_staged ?field_worker_var)
        (support_worker_staged ?support_worker_var)
        (worker_parcel_pair ?field_worker_var ?field_parcel_var)
        (support_worker_route_pair ?support_worker_var ?transport_route_var)
        (parcel_reserved ?field_parcel_var)
        (route_reserved ?transport_route_var)
        (field_worker_deployed ?field_worker_var)
        (support_worker_deployed ?support_worker_var)
        (shift_open ?shift_slot_var)
      )
    :effect
      (and
        (shift_staged ?shift_slot_var)
        (shift_parcel_linked ?shift_slot_var ?field_parcel_var)
        (shift_route_linked ?shift_slot_var ?transport_route_var)
        (not
          (shift_open ?shift_slot_var)
        )
      )
  )
  (:action book_shift_slot_with_reserved_parcel
    :parameters (?field_worker_var - field_worker ?support_worker_var - support_worker ?field_parcel_var - field_parcel ?transport_route_var - transport_route ?shift_slot_var - shift_slot)
    :precondition
      (and
        (field_worker_staged ?field_worker_var)
        (support_worker_staged ?support_worker_var)
        (worker_parcel_pair ?field_worker_var ?field_parcel_var)
        (support_worker_route_pair ?support_worker_var ?transport_route_var)
        (parcel_buffered ?field_parcel_var)
        (route_reserved ?transport_route_var)
        (not
          (field_worker_deployed ?field_worker_var)
        )
        (support_worker_deployed ?support_worker_var)
        (shift_open ?shift_slot_var)
      )
    :effect
      (and
        (shift_staged ?shift_slot_var)
        (shift_parcel_linked ?shift_slot_var ?field_parcel_var)
        (shift_route_linked ?shift_slot_var ?transport_route_var)
        (shift_flag_primary ?shift_slot_var)
        (not
          (shift_open ?shift_slot_var)
        )
      )
  )
  (:action book_shift_slot_with_buffered_route
    :parameters (?field_worker_var - field_worker ?support_worker_var - support_worker ?field_parcel_var - field_parcel ?transport_route_var - transport_route ?shift_slot_var - shift_slot)
    :precondition
      (and
        (field_worker_staged ?field_worker_var)
        (support_worker_staged ?support_worker_var)
        (worker_parcel_pair ?field_worker_var ?field_parcel_var)
        (support_worker_route_pair ?support_worker_var ?transport_route_var)
        (parcel_reserved ?field_parcel_var)
        (route_buffered ?transport_route_var)
        (field_worker_deployed ?field_worker_var)
        (not
          (support_worker_deployed ?support_worker_var)
        )
        (shift_open ?shift_slot_var)
      )
    :effect
      (and
        (shift_staged ?shift_slot_var)
        (shift_parcel_linked ?shift_slot_var ?field_parcel_var)
        (shift_route_linked ?shift_slot_var ?transport_route_var)
        (shift_flag_secondary ?shift_slot_var)
        (not
          (shift_open ?shift_slot_var)
        )
      )
  )
  (:action book_shift_slot_with_buffered_parcel_and_route
    :parameters (?field_worker_var - field_worker ?support_worker_var - support_worker ?field_parcel_var - field_parcel ?transport_route_var - transport_route ?shift_slot_var - shift_slot)
    :precondition
      (and
        (field_worker_staged ?field_worker_var)
        (support_worker_staged ?support_worker_var)
        (worker_parcel_pair ?field_worker_var ?field_parcel_var)
        (support_worker_route_pair ?support_worker_var ?transport_route_var)
        (parcel_buffered ?field_parcel_var)
        (route_buffered ?transport_route_var)
        (not
          (field_worker_deployed ?field_worker_var)
        )
        (not
          (support_worker_deployed ?support_worker_var)
        )
        (shift_open ?shift_slot_var)
      )
    :effect
      (and
        (shift_staged ?shift_slot_var)
        (shift_parcel_linked ?shift_slot_var ?field_parcel_var)
        (shift_route_linked ?shift_slot_var ?transport_route_var)
        (shift_flag_primary ?shift_slot_var)
        (shift_flag_secondary ?shift_slot_var)
        (not
          (shift_open ?shift_slot_var)
        )
      )
  )
  (:action lock_shift_slot
    :parameters (?shift_slot_var - shift_slot ?field_worker_var - field_worker ?skill_certification_var - skill_certification)
    :precondition
      (and
        (shift_staged ?shift_slot_var)
        (field_worker_staged ?field_worker_var)
        (labor_skill_pair ?field_worker_var ?skill_certification_var)
        (not
          (shift_locked ?shift_slot_var)
        )
      )
    :effect (shift_locked ?shift_slot_var)
  )
  (:action stage_work_package_to_shift_slot
    :parameters (?work_crew_var - work_crew ?work_package_var - work_package ?shift_slot_var - shift_slot)
    :precondition
      (and
        (assignment_cleared ?work_crew_var)
        (crew_shift_pair ?work_crew_var ?shift_slot_var)
        (crew_work_package_pair ?work_crew_var ?work_package_var)
        (work_package_available ?work_package_var)
        (shift_staged ?shift_slot_var)
        (shift_locked ?shift_slot_var)
        (not
          (work_package_committed ?work_package_var)
        )
      )
    :effect
      (and
        (work_package_committed ?work_package_var)
        (package_shift_pair ?work_package_var ?shift_slot_var)
        (not
          (work_package_available ?work_package_var)
        )
      )
  )
  (:action confirm_work_package_allocation
    :parameters (?work_crew_var - work_crew ?work_package_var - work_package ?shift_slot_var - shift_slot ?skill_certification_var - skill_certification)
    :precondition
      (and
        (assignment_cleared ?work_crew_var)
        (crew_work_package_pair ?work_crew_var ?work_package_var)
        (work_package_committed ?work_package_var)
        (package_shift_pair ?work_package_var ?shift_slot_var)
        (labor_skill_pair ?work_crew_var ?skill_certification_var)
        (not
          (shift_flag_primary ?shift_slot_var)
        )
        (not
          (crew_verified ?work_crew_var)
        )
      )
    :effect (crew_verified ?work_crew_var)
  )
  (:action attach_authorization_document_to_crew
    :parameters (?work_crew_var - work_crew ?authorization_document_var - authorization_document)
    :precondition
      (and
        (assignment_cleared ?work_crew_var)
        (authorization_document_available ?authorization_document_var)
        (not
          (crew_authorization_document_linked ?work_crew_var)
        )
      )
    :effect
      (and
        (crew_authorization_document_linked ?work_crew_var)
        (crew_authorization_pair ?work_crew_var ?authorization_document_var)
        (not
          (authorization_document_available ?authorization_document_var)
        )
      )
  )
  (:action verify_crew_authorization
