    (case_registered ?accreditation_case - accreditation_change_case)

    (handler_assigned ?accreditation_case - accreditation_change_case)

    (approval_recorded_entity ?accreditation_case - accreditation_change_case)
    (authorization_granted_entity ?accreditation_case - accreditation_change_case)
    (admin_slot_available ?admin_slot - administrative_officer_slot)
    (assigned_to_admin_slot_entity ?accreditation_case - accreditation_change_case ?admin_slot - administrative_officer_slot)
    (change_notification_available ?change_notification - change_notification_document)
    (attached_change_notification_entity ?accreditation_case - accreditation_change_case ?change_notification - change_notification_document)
    (faculty_available ?faculty_reviewer - faculty_reviewer)
    (assigned_faculty_reviewer_entity ?accreditation_case - accreditation_change_case ?faculty_reviewer - faculty_reviewer)
    (remediation_option_available ?remediation_option - remediation_option)
    (student_assigned_remediation_option ?student - student_record ?remediation_option - remediation_option)
    (cohort_assigned_remediation_option ?cohort - cohort_record ?remediation_option - remediation_option)
    (student_term_association ?student - student_record ?term - academic_term)
    (term_primary_flag ?term - academic_term)
    (term_secondary_flag ?term - academic_term)
    (student_eligibility_confirmed ?student - student_record)
    (cohort_delivery_association ?cohort - cohort_record ?delivery_option - delivery_option)
    (delivery_option_primary ?delivery_option - delivery_option)
    (delivery_option_secondary ?delivery_option - delivery_option)
    (cohort_eligibility_confirmed ?cohort - cohort_record)
    (repair_plan_registered ?repair_plan - repair_plan_proposal)

    (plan_assigned_delivery_option ?repair_plan - repair_plan_proposal ?delivery_option - delivery_option)
    (plan_flag_term_covered ?repair_plan - repair_plan_proposal)
    (plan_flag_delivery_covered ?repair_plan - repair_plan_proposal)
    (plan_marked_ready ?repair_plan - repair_plan_proposal)
    (program_has_student ?program - academic_program ?student - student_record)
    (program_has_cohort ?program - academic_program ?cohort - cohort_record)

    (course_request_linked_to_plan ?course_request - course_change_request ?repair_plan - repair_plan_proposal)
    (documents_compiled ?program - academic_program)
    (committee_routed ?program - academic_program)
    (committee_review_complete ?program - academic_program)
    (program_constraint_attached ?program - academic_program)

    (committee_review_in_progress ?program - academic_program)
    (committee_review_signed_off ?program - academic_program)
    (accreditation_constraint_available ?accreditation_constraint - accreditation_constraint)

  (:action register_accreditation_case
    :parameters (?accreditation_case - accreditation_change_case)
    :precondition
      (and
        (not
          (case_registered ?accreditation_case)
        )
        (not
          (record_reconciled ?accreditation_case)
        )
      )
    :effect (case_registered ?accreditation_case)
  )
  (:action assign_admin_slot_to_case
    :parameters (?accreditation_case - accreditation_change_case ?admin_slot - administrative_officer_slot)
    :precondition
      (and
        (case_registered ?accreditation_case)
        (not
          (handler_assigned ?accreditation_case)
        )
        (admin_slot_available ?admin_slot)
      )
    :effect
      (and
        (handler_assigned ?accreditation_case)
        (assigned_to_admin_slot_entity ?accreditation_case ?admin_slot)
        (not
          (admin_slot_available ?admin_slot)
        )
      )
  )
  (:action attach_change_notification_to_case
    :parameters (?accreditation_case - accreditation_change_case ?change_notification - change_notification_document)
    :precondition
      (and
        (case_registered ?accreditation_case)
        (handler_assigned ?accreditation_case)
        (change_notification_available ?change_notification)
      )
    :effect
      (and
        (attached_change_notification_entity ?accreditation_case ?change_notification)
        (not
          (change_notification_available ?change_notification)
        )
      )
  )
  (:action mark_case_ready_for_assessment
    :parameters (?accreditation_case - accreditation_change_case ?change_notification - change_notification_document)
    :precondition
      (and
        (case_registered ?accreditation_case)
        (handler_assigned ?accreditation_case)
        (attached_change_notification_entity ?accreditation_case ?change_notification)
        (not
          (ready_for_assessment_entity ?accreditation_case)
        )
      )
    :effect (ready_for_assessment_entity ?accreditation_case)
  )
  (:action detach_change_notification_from_case
    :parameters (?accreditation_case - accreditation_change_case ?change_notification - change_notification_document)
    :precondition
      (and
        (attached_change_notification_entity ?accreditation_case ?change_notification)
      )
    :effect
      (and
        (change_notification_available ?change_notification)
        (not
          (attached_change_notification_entity ?accreditation_case ?change_notification)
        )
      )
  )
  (:action assign_faculty_reviewer_to_case
    :parameters (?accreditation_case - accreditation_change_case ?faculty_reviewer - faculty_reviewer)
    :precondition
      (and
        (ready_for_assessment_entity ?accreditation_case)
        (faculty_available ?faculty_reviewer)
      )
    :effect
      (and
        (assigned_faculty_reviewer_entity ?accreditation_case ?faculty_reviewer)
        (not
          (faculty_available ?faculty_reviewer)
        )
      )
  )
  (:action release_faculty_reviewer_from_case
    :parameters (?accreditation_case - accreditation_change_case ?faculty_reviewer - faculty_reviewer)
    :precondition
      (and
        (assigned_faculty_reviewer_entity ?accreditation_case ?faculty_reviewer)
      )
    :effect
      (and
        (faculty_available ?faculty_reviewer)
        (not
          (assigned_faculty_reviewer_entity ?accreditation_case ?faculty_reviewer)
        )
      )
  )
  (:action allocate_resource_to_program
    :parameters (?program - academic_program ?allocated_resource - allocated_resource)
    :precondition
      (and
        (ready_for_assessment_entity ?program)
        (allocated_resource_available ?allocated_resource)
      )
    :effect
      (and
        (program_allocated_resource ?program ?allocated_resource)
        (not
          (allocated_resource_available ?allocated_resource)
        )
      )
  )
  (:action release_allocated_resource_from_program
    :parameters (?program - academic_program ?allocated_resource - allocated_resource)
    :precondition
      (and
        (program_allocated_resource ?program ?allocated_resource)
      )
    :effect
      (and
        (allocated_resource_available ?allocated_resource)
        (not
          (program_allocated_resource ?program ?allocated_resource)
        )
      )
  )
  (:action record_accreditor_observation_for_program
    :parameters (?program - academic_program ?accreditor_observation - accreditor_observation)
    :precondition
      (and
        (ready_for_assessment_entity ?program)
        (accreditor_observation_available ?accreditor_observation)
      )
    :effect
      (and
        (program_has_accreditor_observation ?program ?accreditor_observation)
        (not
          (accreditor_observation_available ?accreditor_observation)
        )
      )
  )
  (:action clear_accreditor_observation_for_program
    :parameters (?program - academic_program ?accreditor_observation - accreditor_observation)
    :precondition
      (and
        (program_has_accreditor_observation ?program ?accreditor_observation)
      )
    :effect
      (and
        (accreditor_observation_available ?accreditor_observation)
        (not
          (program_has_accreditor_observation ?program ?accreditor_observation)
        )
      )
  )
  (:action reserve_academic_term_for_student
    :parameters (?student - student_record ?term - academic_term ?change_notification - change_notification_document)
    :precondition
      (and
        (ready_for_assessment_entity ?student)
        (attached_change_notification_entity ?student ?change_notification)
        (student_term_association ?student ?term)
        (not
          (term_primary_flag ?term)
        )
        (not
          (term_secondary_flag ?term)
        )
      )
    :effect (term_primary_flag ?term)
  )
  (:action review_student_term_eligibility
    :parameters (?student - student_record ?term - academic_term ?faculty_reviewer - faculty_reviewer)
    :precondition
      (and
        (ready_for_assessment_entity ?student)
        (assigned_faculty_reviewer_entity ?student ?faculty_reviewer)
        (student_term_association ?student ?term)
        (term_primary_flag ?term)
        (not
          (student_ready_for_synthesis ?student)
        )
      )
    :effect
      (and
        (student_ready_for_synthesis ?student)
        (student_eligibility_confirmed ?student)
      )
  )
  (:action assign_remediation_option_to_student
    :parameters (?student - student_record ?term - academic_term ?remediation_option - remediation_option)
    :precondition
      (and
        (ready_for_assessment_entity ?student)
        (student_term_association ?student ?term)
        (remediation_option_available ?remediation_option)
        (not
          (student_ready_for_synthesis ?student)
        )
      )
    :effect
      (and
        (term_secondary_flag ?term)
        (student_ready_for_synthesis ?student)
        (student_assigned_remediation_option ?student ?remediation_option)
        (not
          (remediation_option_available ?remediation_option)
        )
      )
  )
  (:action commit_remediation_for_student
    :parameters (?student - student_record ?term - academic_term ?change_notification - change_notification_document ?remediation_option - remediation_option)
    :precondition
      (and
        (ready_for_assessment_entity ?student)
        (attached_change_notification_entity ?student ?change_notification)
        (student_term_association ?student ?term)
        (term_secondary_flag ?term)
        (student_assigned_remediation_option ?student ?remediation_option)
        (not
          (student_eligibility_confirmed ?student)
        )
      )
    :effect
      (and
        (term_primary_flag ?term)
        (student_eligibility_confirmed ?student)
        (remediation_option_available ?remediation_option)
        (not
          (student_assigned_remediation_option ?student ?remediation_option)
        )
      )
  )
  (:action reserve_delivery_option_for_cohort
    :parameters (?cohort - cohort_record ?delivery_option - delivery_option ?change_notification - change_notification_document)
    :precondition
      (and
        (ready_for_assessment_entity ?cohort)
        (attached_change_notification_entity ?cohort ?change_notification)
        (cohort_delivery_association ?cohort ?delivery_option)
        (not
          (delivery_option_primary ?delivery_option)
        )
        (not
          (delivery_option_secondary ?delivery_option)
        )
      )
    :effect (delivery_option_primary ?delivery_option)
  )
  (:action assign_faculty_reviewer_for_cohort
    :parameters (?cohort - cohort_record ?delivery_option - delivery_option ?faculty_reviewer - faculty_reviewer)
    :precondition
      (and
        (ready_for_assessment_entity ?cohort)

          (cohort_ready_for_synthesis ?cohort)

      )
    :effect
      (and
        (cohort_ready_for_synthesis ?cohort)
        (cohort_eligibility_confirmed ?cohort)
      )
  )
  (:action assign_remediation_option_for_cohort
    :parameters (?cohort - cohort_record ?delivery_option - delivery_option ?remediation_option - remediation_option)
    :precondition
      (and
        (ready_for_assessment_entity ?cohort)
        (cohort_delivery_association ?cohort ?delivery_option)
        (remediation_option_available ?remediation_option)
        (not
          (cohort_ready_for_synthesis ?cohort)
        )
      )
    :effect
      (and
        (delivery_option_secondary ?delivery_option)
        (cohort_ready_for_synthesis ?cohort)
        (cohort_assigned_remediation_option ?cohort ?remediation_option)
        (not
          (remediation_option_available ?remediation_option)
        )
      )
  )
  (:action commit_remediation_for_cohort
    :parameters (?cohort - cohort_record ?delivery_option - delivery_option ?change_notification - change_notification_document ?remediation_option - remediation_option)
    :precondition
      (and
        (ready_for_assessment_entity ?cohort)
        (attached_change_notification_entity ?cohort ?change_notification)
        (cohort_delivery_association ?cohort ?delivery_option)
        (delivery_option_secondary ?delivery_option)
        (cohort_assigned_remediation_option ?cohort ?remediation_option)
        (not
          (cohort_eligibility_confirmed ?cohort)
        )
      )
    :effect
      (and
        (delivery_option_primary ?delivery_option)
        (cohort_eligibility_confirmed ?cohort)
        (remediation_option_available ?remediation_option)
        (not
          (cohort_assigned_remediation_option ?cohort ?remediation_option)
        )
      )
  )
  (:action synthesize_repair_plan_proposal
    :parameters (?student - student_record ?cohort - cohort_record ?term - academic_term ?delivery_option - delivery_option ?repair_plan - repair_plan_proposal)
    :precondition
      (and
        (student_ready_for_synthesis ?student)
        (cohort_ready_for_synthesis ?cohort)
        (student_term_association ?student ?term)
        (cohort_delivery_association ?cohort ?delivery_option)
        (term_primary_flag ?term)
        (delivery_option_primary ?delivery_option)
        (student_eligibility_confirmed ?student)
        (cohort_eligibility_confirmed ?cohort)
        (repair_plan_registered ?repair_plan)
      )
    :effect
      (and
        (repair_plan_active ?repair_plan)
        (plan_assigned_term ?repair_plan ?term)
        (plan_assigned_delivery_option ?repair_plan ?delivery_option)
        (not
          (repair_plan_registered ?repair_plan)
        )
      )
  )
  (:action synthesize_repair_plan_proposal_variant
    :parameters (?student - student_record ?cohort - cohort_record ?term - academic_term ?delivery_option - delivery_option ?repair_plan - repair_plan_proposal)
    :precondition
      (and
        (student_ready_for_synthesis ?student)
        (cohort_ready_for_synthesis ?cohort)
        (student_term_association ?student ?term)
        (cohort_delivery_association ?cohort ?delivery_option)
        (term_secondary_flag ?term)
        (delivery_option_primary ?delivery_option)
        (not
          (student_eligibility_confirmed ?student)
        )
        (cohort_eligibility_confirmed ?cohort)
        (repair_plan_registered ?repair_plan)
      )
    :effect
      (and
        (repair_plan_active ?repair_plan)
        (plan_assigned_term ?repair_plan ?term)
        (plan_assigned_delivery_option ?repair_plan ?delivery_option)
        (plan_flag_term_covered ?repair_plan)
        (not
          (repair_plan_registered ?repair_plan)
        )
      )
  )
  (:action synthesize_repair_plan_proposal_alternative
    :parameters (?student - student_record ?cohort - cohort_record ?term - academic_term ?delivery_option - delivery_option ?repair_plan - repair_plan_proposal)
    :precondition
      (and
        (student_ready_for_synthesis ?student)
        (cohort_ready_for_synthesis ?cohort)
        (student_term_association ?student ?term)
        (cohort_delivery_association ?cohort ?delivery_option)
        (term_primary_flag ?term)
        (delivery_option_secondary ?delivery_option)
        (student_eligibility_confirmed ?student)
        (not
          (cohort_eligibility_confirmed ?cohort)
        )
        (repair_plan_registered ?repair_plan)
      )
    :effect
      (and
        (repair_plan_active ?repair_plan)
        (plan_assigned_term ?repair_plan ?term)
        (plan_assigned_delivery_option ?repair_plan ?delivery_option)
        (plan_flag_delivery_covered ?repair_plan)
        (not
          (repair_plan_registered ?repair_plan)
        )
      )
  )
  (:action synthesize_repair_plan_proposal_combined
    :parameters (?student - student_record ?cohort - cohort_record ?term - academic_term ?delivery_option - delivery_option ?repair_plan - repair_plan_proposal)
    :precondition
      (and
        (student_ready_for_synthesis ?student)
        (cohort_ready_for_synthesis ?cohort)
        (student_term_association ?student ?term)
        (cohort_delivery_association ?cohort ?delivery_option)
        (term_secondary_flag ?term)
        (delivery_option_secondary ?delivery_option)
        (not
          (student_eligibility_confirmed ?student)
        )
        (not
          (cohort_eligibility_confirmed ?cohort)
        )
        (repair_plan_registered ?repair_plan)
      )
    :effect
      (and
        (repair_plan_active ?repair_plan)
        (plan_assigned_term ?repair_plan ?term)
        (plan_assigned_delivery_option ?repair_plan ?delivery_option)
        (plan_flag_term_covered ?repair_plan)
        (plan_flag_delivery_covered ?repair_plan)
        (not
          (repair_plan_registered ?repair_plan)
        )
      )
  )
  (:action finalize_repair_plan_proposal_for_review
    :parameters (?repair_plan - repair_plan_proposal ?student - student_record ?change_notification - change_notification_document)
    :precondition
      (and
        (repair_plan_active ?repair_plan)
        (student_ready_for_synthesis ?student)
        (attached_change_notification_entity ?student ?change_notification)
        (not
          (plan_marked_ready ?repair_plan)
        )
      )
    :effect (plan_marked_ready ?repair_plan)
  )
  (:action consolidate_course_request_into_repair_plan
    :parameters (?program - academic_program ?course_request - course_change_request ?repair_plan - repair_plan_proposal)
    :precondition
      (and
        (ready_for_assessment_entity ?program)
        (program_linked_to_proposal ?program ?repair_plan)
        (program_linked_course_request ?program ?course_request)
        (course_request_available ?course_request)
        (repair_plan_active ?repair_plan)
        (plan_marked_ready ?repair_plan)
        (not
          (course_request_processed ?course_request)
        )
      )
    :effect
      (and
        (course_request_processed ?course_request)
        (course_request_linked_to_plan ?course_request ?repair_plan)
        (not
          (course_request_available ?course_request)
        )
      )
  )
  (:action assemble_program_document_bundle
    :parameters (?program - academic_program ?course_request - course_change_request ?repair_plan - repair_plan_proposal ?change_notification - change_notification_document)
    :precondition
      (and
        (ready_for_assessment_entity ?program)
        (program_linked_course_request ?program ?course_request)
        (course_request_processed ?course_request)
        (course_request_linked_to_plan ?course_request ?repair_plan)
        (attached_change_notification_entity ?program ?change_notification)
        (not
          (plan_flag_term_covered ?repair_plan)
        )
        (not
          (documents_compiled ?program)
        )
      )
    :effect (documents_compiled ?program)
  )
  (:action attach_accreditation_constraint_to_program
    :parameters (?program - academic_program ?accreditation_constraint - accreditation_constraint)
    :precondition
      (and
        (ready_for_assessment_entity ?program)
        (accreditation_constraint_available ?accreditation_constraint)
        (not
          (program_constraint_attached ?program)
        )
      )
    :effect
      (and
        (program_constraint_attached ?program)
        (program_assigned_constraint ?program ?accreditation_constraint)
        (not
          (accreditation_constraint_available ?accreditation_constraint)
        )
      )
  )
  (:action validate_program_consolidation_against_constraint
    :parameters (?program - academic_program ?course_request - course_change_request ?repair_plan - repair_plan_proposal ?change_notification - change_notification_document ?accreditation_constraint - accreditation_constraint)
    :precondition
      (and
        (ready_for_assessment_entity ?program)
        (program_linked_course_request ?program ?course_request)
        (course_request_processed ?course_request)
        (course_request_linked_to_plan ?course_request ?repair_plan)
        (attached_change_notification_entity ?program ?change_notification)
        (plan_flag_term_covered ?repair_plan)
        (program_constraint_attached ?program)
        (program_assigned_constraint ?program ?accreditation_constraint)
        (not
          (documents_compiled ?program)
        )
      )
    :effect
      (and
        (documents_compiled ?program)
        (consolidation_confirmed ?program)
      )
  )
  (:action route_program_to_committee_initial
    :parameters (?program - academic_program ?allocated_resource - allocated_resource ?faculty_reviewer - faculty_reviewer ?course_request - course_change_request ?repair_plan - repair_plan_proposal)
    :precondition
      (and
        (documents_compiled ?program)
        (program_allocated_resource ?program ?allocated_resource)
        (assigned_faculty_reviewer_entity ?program ?faculty_reviewer)
        (program_linked_course_request ?program ?course_request)
        (course_request_linked_to_plan ?course_request ?repair_plan)
        (not
          (plan_flag_delivery_covered ?repair_plan)
        )
        (not
          (committee_routed ?program)
        )
      )
    :effect (committee_routed ?program)
  )
  (:action route_program_to_committee_subsequent
    :parameters (?program - academic_program ?allocated_resource - allocated_resource ?faculty_reviewer - faculty_reviewer ?course_request - course_change_request ?repair_plan - repair_plan_proposal)
    :precondition
      (and
        (documents_compiled ?program)
        (program_allocated_resource ?program ?allocated_resource)
        (assigned_faculty_reviewer_entity ?program ?faculty_reviewer)
        (program_linked_course_request ?program ?course_request)
        (course_request_linked_to_plan ?course_request ?repair_plan)
        (plan_flag_delivery_covered ?repair_plan)
        (not
          (committee_routed ?program)
        )
      )
    :effect (committee_routed ?program)
