    (lead_booking_has_ancillary ?lead_service_booking - lead_service_booking ?ancillary_option - ancillary_option)
    (dependent_booking_has_ancillary ?dependent_service_booking - dependent_service_booking ?ancillary_option - ancillary_option)
    (service_booking_has_availability ?lead_service_booking - lead_service_booking ?provider_availability - provider_availability)

    (reservation_has_availability ?reservation_record - reservation_record ?provider_availability - provider_availability)
    (reservation_has_inventory_allocation ?reservation_record - reservation_record ?inventory_allocation - inventory_allocation)
    (reservation_includes_lead_component ?reservation_record - reservation_record)

    (reservation_record_verified ?reservation_record - reservation_record)

    (booking_context_rate_plan_marked ?booking_context - booking_unit_context)

    (booking_context_requires_payment ?booking_context - booking_unit_context)
    (booking_context_checks_passed ?booking_context - booking_unit_context)
    (vendor_contract_available ?vendor_contract - vendor_contract)
    (booking_context_has_vendor_contract ?booking_context - booking_unit_context ?vendor_contract - vendor_contract)
    (vendor_contract_accepted ?booking_context - booking_unit_context)
    (vendor_contract_prepared ?booking_context - booking_unit_context)
    (vendor_contract_authorized ?booking_context - booking_unit_context)
    (rate_plan_available ?rate_plan - rate_plan)
    (booking_context_has_rate_plan ?booking_context - booking_unit_context ?rate_plan - rate_plan)
    (payment_method_available ?payment_method - payment_method)
    (booking_context_has_payment_method ?booking_context - booking_unit_context ?payment_method - payment_method)
    (supplier_credential_available ?supplier_credential - supplier_credential)
    (booking_context_has_supplier_credential ?booking_context - booking_unit_context ?supplier_credential - supplier_credential)
    (customer_reference_available ?customer_reference - customer_reference)
    (booking_context_has_customer_reference ?booking_context - booking_unit_context ?customer_reference - customer_reference)

  (:action select_booking_unit
    :parameters (?booking_unit - booking_unit)
    :precondition
      (and
        (not
          (booking_unit_selected ?booking_unit)
        )
        (not
          (booking_committed ?booking_unit)
        )
      )
    :effect (booking_unit_selected ?booking_unit)
  )
  (:action allocate_supplier_option
    :parameters (?booking_unit - booking_unit ?supplier_option - supplier_option)
    :precondition
      (and
        (booking_unit_selected ?booking_unit)
        (not
          (booking_has_supplier_allocation ?booking_unit)
        )
        (supplier_option_available ?supplier_option)
      )
    :effect
      (and
        (booking_has_supplier_allocation ?booking_unit)
        (booking_has_supplier_option ?booking_unit ?supplier_option)
        (not
          (supplier_option_available ?supplier_option)
        )
      )
  )
  (:action hold_service_variant
    :parameters (?booking_unit - booking_unit ?service_variant_quote - service_variant_quote)
    :precondition
      (and
        (booking_unit_selected ?booking_unit)
        (booking_has_supplier_allocation ?booking_unit)
        (service_variant_available ?service_variant_quote)
      )
    :effect
      (and
        (booking_has_variant_hold ?booking_unit ?service_variant_quote)
        (not
          (service_variant_available ?service_variant_quote)
        )
      )
  )
  (:action confirm_variant_hold
    :parameters (?booking_unit - booking_unit ?service_variant_quote - service_variant_quote)
    :precondition
      (and
        (booking_unit_selected ?booking_unit)
        (booking_has_supplier_allocation ?booking_unit)
        (booking_has_variant_hold ?booking_unit ?service_variant_quote)
        (not
          (booking_unit_confirmed ?booking_unit)
        )
      )
    :effect (booking_unit_confirmed ?booking_unit)
  )
  (:action release_variant_hold
    :parameters (?booking_unit - booking_unit ?service_variant_quote - service_variant_quote)
    :precondition
      (and
        (booking_has_variant_hold ?booking_unit ?service_variant_quote)
      )
    :effect
      (and
        (service_variant_available ?service_variant_quote)
        (not
          (booking_has_variant_hold ?booking_unit ?service_variant_quote)
        )
      )
  )
  (:action attach_provider_resource
    :parameters (?booking_unit - booking_unit ?provider_resource - provider_resource)
    :precondition
      (and
        (booking_unit_confirmed ?booking_unit)
        (provider_resource_available ?provider_resource)
      )
    :effect
      (and
        (booking_attached_provider_resource ?booking_unit ?provider_resource)
        (not
          (provider_resource_available ?provider_resource)
        )
      )
  )
  (:action detach_provider_resource
    :parameters (?booking_unit - booking_unit ?provider_resource - provider_resource)
    :precondition
      (and
        (booking_attached_provider_resource ?booking_unit ?provider_resource)
      )
    :effect
      (and
        (provider_resource_available ?provider_resource)
        (not
          (booking_attached_provider_resource ?booking_unit ?provider_resource)
        )
      )
  )
  (:action bind_supplier_credential
    :parameters (?booking_context - booking_unit_context ?supplier_credential - supplier_credential)
    :precondition
      (and
        (booking_unit_confirmed ?booking_context)
        (supplier_credential_available ?supplier_credential)
      )
    :effect
      (and
        (booking_context_has_supplier_credential ?booking_context ?supplier_credential)
        (not
          (supplier_credential_available ?supplier_credential)
        )
      )
  )
  (:action unbind_supplier_credential
    :parameters (?booking_context - booking_unit_context ?supplier_credential - supplier_credential)
    :precondition
      (and
        (booking_context_has_supplier_credential ?booking_context ?supplier_credential)
      )
    :effect
      (and
        (supplier_credential_available ?supplier_credential)
        (not
          (booking_context_has_supplier_credential ?booking_context ?supplier_credential)
        )
      )
  )
  (:action bind_customer_reference
    :parameters (?booking_context - booking_unit_context ?customer_reference - customer_reference)
    :precondition
      (and
        (booking_unit_confirmed ?booking_context)
        (customer_reference_available ?customer_reference)
      )
    :effect
      (and
        (booking_context_has_customer_reference ?booking_context ?customer_reference)
        (not
          (customer_reference_available ?customer_reference)
        )
      )
  )
  (:action unbind_customer_reference
    :parameters (?booking_context - booking_unit_context ?customer_reference - customer_reference)
    :precondition
      (and
        (booking_context_has_customer_reference ?booking_context ?customer_reference)
      )
    :effect
      (and
        (customer_reference_available ?customer_reference)
        (not
          (booking_context_has_customer_reference ?booking_context ?customer_reference)
        )
      )
  )
  (:action reserve_provider_availability_for_lead
    :parameters (?lead_service_booking - lead_service_booking ?provider_availability - provider_availability ?service_variant_quote - service_variant_quote)
    :precondition
      (and
        (booking_unit_confirmed ?lead_service_booking)
        (booking_has_variant_hold ?lead_service_booking ?service_variant_quote)
        (service_booking_has_availability ?lead_service_booking ?provider_availability)
        (not
          (availability_reserved ?provider_availability)
        )
        (not
          (availability_with_ancillary ?provider_availability)
        )
      )
    :effect (availability_reserved ?provider_availability)
  )
  (:action process_lead_booking_provider_attach
    :parameters (?lead_service_booking - lead_service_booking ?provider_availability - provider_availability ?provider_resource - provider_resource)
    :precondition
      (and
        (booking_unit_confirmed ?lead_service_booking)
        (booking_attached_provider_resource ?lead_service_booking ?provider_resource)
        (service_booking_has_availability ?lead_service_booking ?provider_availability)
        (availability_reserved ?provider_availability)
        (not
          (lead_booking_initialized ?lead_service_booking)
        )
      )
    :effect
      (and
        (lead_booking_initialized ?lead_service_booking)
        (lead_booking_processed ?lead_service_booking)
      )
  )
  (:action attach_ancillary_to_lead_booking
    :parameters (?lead_service_booking - lead_service_booking ?provider_availability - provider_availability ?ancillary_option - ancillary_option)
    :precondition
      (and
        (booking_unit_confirmed ?lead_service_booking)
        (service_booking_has_availability ?lead_service_booking ?provider_availability)
        (ancillary_option_available ?ancillary_option)
        (not
          (lead_booking_initialized ?lead_service_booking)
        )
      )
    :effect
      (and
        (availability_with_ancillary ?provider_availability)
        (lead_booking_initialized ?lead_service_booking)
        (lead_booking_has_ancillary ?lead_service_booking ?ancillary_option)
        (not
          (ancillary_option_available ?ancillary_option)
        )
      )
  )
  (:action finalize_lead_ancillary_processing
    :parameters (?lead_service_booking - lead_service_booking ?provider_availability - provider_availability ?service_variant_quote - service_variant_quote ?ancillary_option - ancillary_option)
    :precondition
      (and
        (booking_unit_confirmed ?lead_service_booking)
        (booking_has_variant_hold ?lead_service_booking ?service_variant_quote)
        (service_booking_has_availability ?lead_service_booking ?provider_availability)
        (availability_with_ancillary ?provider_availability)
        (lead_booking_has_ancillary ?lead_service_booking ?ancillary_option)
        (not
          (lead_booking_processed ?lead_service_booking)
        )
      )
    :effect
      (and
        (availability_reserved ?provider_availability)
        (lead_booking_processed ?lead_service_booking)
        (ancillary_option_available ?ancillary_option)
        (not
          (lead_booking_has_ancillary ?lead_service_booking ?ancillary_option)
        )
      )
  )
  (:action reserve_inventory_for_dependent
    :parameters (?dependent_service_booking - dependent_service_booking ?inventory_allocation - inventory_allocation ?service_variant_quote - service_variant_quote)
    :precondition
      (and
        (booking_unit_confirmed ?dependent_service_booking)
        (booking_has_variant_hold ?dependent_service_booking ?service_variant_quote)
        (dependent_booking_has_inventory_allocation ?dependent_service_booking ?inventory_allocation)
        (not
          (inventory_allocation_reserved ?inventory_allocation)
        )
        (not
          (inventory_allocation_with_ancillary ?inventory_allocation)
        )
      )
    :effect (inventory_allocation_reserved ?inventory_allocation)
  )
  (:action process_dependent_booking_provider_attach
    :parameters (?dependent_service_booking - dependent_service_booking ?inventory_allocation - inventory_allocation ?provider_resource - provider_resource)
    :precondition
      (and
        (booking_unit_confirmed ?dependent_service_booking)
        (booking_attached_provider_resource ?dependent_service_booking ?provider_resource)
        (dependent_booking_has_inventory_allocation ?dependent_service_booking ?inventory_allocation)
        (inventory_allocation_reserved ?inventory_allocation)
        (not
          (dependent_booking_initialized ?dependent_service_booking)
        )
      )
    :effect
      (and
        (dependent_booking_initialized ?dependent_service_booking)
        (dependent_booking_processed ?dependent_service_booking)
      )
  )
  (:action attach_ancillary_to_dependent_booking
    :parameters (?dependent_service_booking - dependent_service_booking ?inventory_allocation - inventory_allocation ?ancillary_option - ancillary_option)
    :precondition
      (and
        (booking_unit_confirmed ?dependent_service_booking)
        (dependent_booking_has_inventory_allocation ?dependent_service_booking ?inventory_allocation)
        (ancillary_option_available ?ancillary_option)
        (not
          (dependent_booking_initialized ?dependent_service_booking)
        )
      )
    :effect
      (and
        (inventory_allocation_with_ancillary ?inventory_allocation)
        (dependent_booking_initialized ?dependent_service_booking)
        (dependent_booking_has_ancillary ?dependent_service_booking ?ancillary_option)
        (not
          (ancillary_option_available ?ancillary_option)
        )
      )
  )
  (:action finalize_dependent_ancillary_processing
    :parameters (?dependent_service_booking - dependent_service_booking ?inventory_allocation - inventory_allocation ?service_variant_quote - service_variant_quote ?ancillary_option - ancillary_option)
    :precondition
      (and
        (booking_unit_confirmed ?dependent_service_booking)
        (booking_has_variant_hold ?dependent_service_booking ?service_variant_quote)
        (dependent_booking_has_inventory_allocation ?dependent_service_booking ?inventory_allocation)
        (inventory_allocation_with_ancillary ?inventory_allocation)
        (dependent_booking_has_ancillary ?dependent_service_booking ?ancillary_option)
        (not
          (dependent_booking_processed ?dependent_service_booking)
        )
      )
    :effect
      (and
        (inventory_allocation_reserved ?inventory_allocation)
        (dependent_booking_processed ?dependent_service_booking)
        (ancillary_option_available ?ancillary_option)
        (not
          (dependent_booking_has_ancillary ?dependent_service_booking ?ancillary_option)
        )
      )
  )
  (:action compose_reservation_record_both_processed
    :parameters (?lead_service_booking - lead_service_booking ?dependent_service_booking - dependent_service_booking ?provider_availability - provider_availability ?inventory_allocation - inventory_allocation ?reservation_record - reservation_record)
    :precondition
      (and
        (lead_booking_initialized ?lead_service_booking)
        (dependent_booking_initialized ?dependent_service_booking)
        (service_booking_has_availability ?lead_service_booking ?provider_availability)
        (dependent_booking_has_inventory_allocation ?dependent_service_booking ?inventory_allocation)
        (availability_reserved ?provider_availability)
        (inventory_allocation_reserved ?inventory_allocation)
        (lead_booking_processed ?lead_service_booking)
        (dependent_booking_processed ?dependent_service_booking)
        (reservation_record_pending ?reservation_record)
      )
    :effect
      (and
        (reservation_record_created ?reservation_record)
        (reservation_has_availability ?reservation_record ?provider_availability)
        (reservation_has_inventory_allocation ?reservation_record ?inventory_allocation)
        (not
          (reservation_record_pending ?reservation_record)
        )
      )
  )
  (:action compose_reservation_record_lead_unprocessed
    :parameters (?lead_service_booking - lead_service_booking ?dependent_service_booking - dependent_service_booking ?provider_availability - provider_availability ?inventory_allocation - inventory_allocation ?reservation_record - reservation_record)
    :precondition
      (and
        (lead_booking_initialized ?lead_service_booking)
        (dependent_booking_initialized ?dependent_service_booking)
        (service_booking_has_availability ?lead_service_booking ?provider_availability)
        (dependent_booking_has_inventory_allocation ?dependent_service_booking ?inventory_allocation)
        (availability_with_ancillary ?provider_availability)
        (inventory_allocation_reserved ?inventory_allocation)
        (not
          (lead_booking_processed ?lead_service_booking)
        )
        (dependent_booking_processed ?dependent_service_booking)
        (reservation_record_pending ?reservation_record)
      )
    :effect
      (and
        (reservation_record_created ?reservation_record)
        (reservation_has_availability ?reservation_record ?provider_availability)
        (reservation_has_inventory_allocation ?reservation_record ?inventory_allocation)
        (reservation_includes_lead_component ?reservation_record)
        (not
          (reservation_record_pending ?reservation_record)
        )
      )
  )
  (:action compose_reservation_record_dependent_unprocessed
    :parameters (?lead_service_booking - lead_service_booking ?dependent_service_booking - dependent_service_booking ?provider_availability - provider_availability ?inventory_allocation - inventory_allocation ?reservation_record - reservation_record)
    :precondition
      (and
        (lead_booking_initialized ?lead_service_booking)
        (dependent_booking_initialized ?dependent_service_booking)
        (service_booking_has_availability ?lead_service_booking ?provider_availability)
        (dependent_booking_has_inventory_allocation ?dependent_service_booking ?inventory_allocation)
        (availability_reserved ?provider_availability)
        (inventory_allocation_with_ancillary ?inventory_allocation)
        (lead_booking_processed ?lead_service_booking)
        (not
          (dependent_booking_processed ?dependent_service_booking)
        )
        (reservation_record_pending ?reservation_record)
      )
    :effect
      (and
        (reservation_record_created ?reservation_record)
        (reservation_has_availability ?reservation_record ?provider_availability)
        (reservation_has_inventory_allocation ?reservation_record ?inventory_allocation)
        (reservation_includes_dependent_component ?reservation_record)
        (not
          (reservation_record_pending ?reservation_record)
        )
      )
  )
  (:action compose_reservation_record_both_unprocessed
    :parameters (?lead_service_booking - lead_service_booking ?dependent_service_booking - dependent_service_booking ?provider_availability - provider_availability ?inventory_allocation - inventory_allocation ?reservation_record - reservation_record)
    :precondition
      (and
        (lead_booking_initialized ?lead_service_booking)
        (dependent_booking_initialized ?dependent_service_booking)
        (service_booking_has_availability ?lead_service_booking ?provider_availability)
        (dependent_booking_has_inventory_allocation ?dependent_service_booking ?inventory_allocation)
        (availability_with_ancillary ?provider_availability)
        (inventory_allocation_with_ancillary ?inventory_allocation)
        (not
          (lead_booking_processed ?lead_service_booking)
        )
        (not
          (dependent_booking_processed ?dependent_service_booking)
        )
        (reservation_record_pending ?reservation_record)
      )
    :effect
      (and
        (reservation_record_created ?reservation_record)
        (reservation_has_availability ?reservation_record ?provider_availability)
        (reservation_has_inventory_allocation ?reservation_record ?inventory_allocation)
        (reservation_includes_lead_component ?reservation_record)
        (reservation_includes_dependent_component ?reservation_record)
        (not
          (reservation_record_pending ?reservation_record)
        )
      )
  )
  (:action verify_reservation_record_for_lead
    :parameters (?reservation_record - reservation_record ?lead_service_booking - lead_service_booking ?service_variant_quote - service_variant_quote)
    :precondition
      (and
        (reservation_record_created ?reservation_record)
        (lead_booking_initialized ?lead_service_booking)
        (booking_has_variant_hold ?lead_service_booking ?service_variant_quote)
        (not
          (reservation_record_verified ?reservation_record)
        )
      )
    :effect (reservation_record_verified ?reservation_record)
  )
  (:action issue_fulfillment_document
    :parameters (?booking_context - booking_unit_context ?fulfillment_document - fulfillment_document ?reservation_record - reservation_record)
    :precondition
      (and
        (booking_unit_confirmed ?booking_context)
        (booking_context_has_reservation_record ?booking_context ?reservation_record)
        (booking_context_has_fulfillment_document ?booking_context ?fulfillment_document)
        (fulfillment_document_available ?fulfillment_document)
        (reservation_record_created ?reservation_record)
        (reservation_record_verified ?reservation_record)
        (not
          (fulfillment_document_issued ?fulfillment_document)
        )
      )
    :effect
      (and
        (fulfillment_document_issued ?fulfillment_document)
        (fulfillment_document_attached_to_reservation ?fulfillment_document ?reservation_record)
        (not
          (fulfillment_document_available ?fulfillment_document)
        )
      )
  )
  (:action bind_fulfillment_document
    :parameters (?booking_context - booking_unit_context ?fulfillment_document - fulfillment_document ?reservation_record - reservation_record ?service_variant_quote - service_variant_quote)
    :precondition
      (and
        (booking_unit_confirmed ?booking_context)
        (booking_context_has_fulfillment_document ?booking_context ?fulfillment_document)
        (fulfillment_document_issued ?fulfillment_document)
        (fulfillment_document_attached_to_reservation ?fulfillment_document ?reservation_record)
        (booking_has_variant_hold ?booking_context ?service_variant_quote)
        (not
          (reservation_includes_lead_component ?reservation_record)
        )
        (not
          (booking_context_documents_bound ?booking_context)
        )
      )
    :effect (booking_context_documents_bound ?booking_context)
  )
  (:action apply_rate_plan_to_context
    :parameters (?booking_context - booking_unit_context ?rate_plan - rate_plan)
    :precondition
      (and
        (booking_unit_confirmed ?booking_context)
        (rate_plan_available ?rate_plan)
        (not
          (booking_context_rate_plan_marked ?booking_context)
        )
