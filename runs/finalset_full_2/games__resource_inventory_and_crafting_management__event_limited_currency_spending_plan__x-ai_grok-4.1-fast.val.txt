--- domain.pddl (as passed to VAL) ---
(define (domain event_limited_currency_spending)
  (:requirements :strips :typing :negative-preconditions)
  (:types base_object - object purchasable_root - base_object resource_root - base_object item_root - base_object event_scope - base_object purchase_target - event_scope currency_option - purchasable_root material - purchasable_root tool - purchasable_root cosmetic_upgrade_token - purchasable_root special_material - purchasable_root voucher - purchasable_root optional_augment - purchasable_root unique_catalyst - purchasable_root consumable - resource_root component - resource_root cosmetic_token - resource_root resource_source_a - item_root resource_source_b - item_root bundle - item_root physical_slot_category - purchase_target loadout_slot_category - purchase_target slot_a - physical_slot_category slot_b - physical_slot_category loadout - loadout_slot_category)

  (:predicates
    (purchase_target_selected ?entity_slot - purchase_target)
    (purchase_target_reserved ?entity_slot - purchase_target)
    (purchase_target_currency_bound ?entity_slot - purchase_target)
    (purchase_target_acquired ?entity_slot - purchase_target)
    (purchase_target_ready_for_event ?entity_slot - purchase_target)
    (purchase_target_spend_authorized ?entity_slot - purchase_target)
    (currency_available ?currency_option - currency_option)
    (currency_assigned_to_purchase_target ?entity_slot - purchase_target ?currency_option - currency_option)
    (material_available ?material - material)
    (material_assigned_to_purchase_target ?entity_slot - purchase_target ?material - material)
    (tool_available ?tool - tool)
    (tool_assigned_to_purchase_target ?entity_slot - purchase_target ?tool - tool)
    (consumable_available ?consumable - consumable)
    (consumable_reserved_for_slot_a ?slot_a - slot_a ?consumable - consumable)
    (consumable_reserved_for_slot_b ?slot_b - slot_b ?consumable - consumable)
    (slot_a_resource_source ?slot_a - slot_a ?resource_source_a - resource_source_a)
    (resource_source_a_reserved ?resource_source_a - resource_source_a)
    (resource_source_a_tagged ?resource_source_a - resource_source_a)
    (slot_a_ready ?slot_a - slot_a)
    (slot_b_resource_source ?slot_b - slot_b ?resource_source_b - resource_source_b)
    (resource_source_b_reserved ?resource_source_b - resource_source_b)
    (resource_source_b_tagged ?resource_source_b - resource_source_b)
    (slot_b_ready ?slot_b - slot_b)
    (bundle_available ?bundle - bundle)
    (bundle_assembled ?bundle - bundle)
    (bundle_linked_to_source_a ?bundle - bundle ?resource_source_a - resource_source_a)
    (bundle_linked_to_source_b ?bundle - bundle ?resource_source_b - resource_source_b)
    (bundle_variant_alpha ?bundle - bundle)
    (bundle_variant_beta ?bundle - bundle)
    (bundle_finalized ?bundle - bundle)
    (loadout_has_slot_a ?loadout - loadout ?slot_a - slot_a)
    (loadout_has_slot_b ?loadout - loadout ?slot_b - slot_b)
    (loadout_linked_bundle ?loadout - loadout ?bundle - bundle)
    (component_available ?component - component)
    (loadout_has_component ?loadout - loadout ?component - component)
    (component_staged ?component - component)
    (component_assigned_to_bundle ?component - component ?bundle - bundle)
    (loadout_modifier_ready ?loadout - loadout)
    (loadout_modifier_applied ?loadout - loadout)
    (loadout_ready_for_finalization ?loadout - loadout)
    (loadout_cosmetic_upgrade_enabled ?loadout - loadout)
    (loadout_cosmetic_stage_complete ?loadout - loadout)
    (loadout_has_augments ?loadout - loadout)
    (loadout_modifiers_satisfied ?loadout - loadout)
    (cosmetic_token_available ?cosmetic_token - cosmetic_token)
    (loadout_cosmetic_assigned ?loadout - loadout ?cosmetic_token - cosmetic_token)
    (loadout_cosmetic_reserved ?loadout - loadout)
    (loadout_cosmetic_applied ?loadout - loadout)
    (loadout_cosmetic_finalized ?loadout - loadout)
    (cosmetic_upgrade_token_available ?cosmetic_upgrade_token - cosmetic_upgrade_token)
    (loadout_cosmetic_upgrade_token_assigned ?loadout - loadout ?cosmetic_upgrade_token - cosmetic_upgrade_token)
    (special_material_available ?special_material - special_material)
    (loadout_special_material_attached ?loadout - loadout ?special_material - special_material)
    (optional_augment_available ?optional_augment - optional_augment)
    (loadout_optional_augment_attached ?loadout - loadout ?optional_augment - optional_augment)
    (unique_catalyst_available ?unique_catalyst - unique_catalyst)
    (loadout_unique_catalyst_attached ?loadout - loadout ?unique_catalyst - unique_catalyst)
    (voucher_available ?voucher - voucher)
    (voucher_assigned_to_purchase_target ?entity_slot - purchase_target ?voucher - voucher)
    (slot_a_locked ?slot_a - slot_a)
    (slot_b_locked ?slot_b - slot_b)
    (loadout_finalized ?loadout - loadout)
  )
  (:action select_candidate_slot
    :parameters (?entity_slot - purchase_target)
    :precondition
      (and
        (not
          (purchase_target_selected ?entity_slot)
        )
        (not
          (purchase_target_acquired ?entity_slot)
        )
      )
    :effect (purchase_target_selected ?entity_slot)
  )
  (:action assign_currency_option_to_slot
    :parameters (?entity_slot - purchase_target ?currency_option - currency_option)
    :precondition
      (and
        (purchase_target_selected ?entity_slot)
        (not
          (purchase_target_currency_bound ?entity_slot)
        )
        (currency_available ?currency_option)
      )
    :effect
      (and
        (purchase_target_currency_bound ?entity_slot)
        (currency_assigned_to_purchase_target ?entity_slot ?currency_option)
        (not
          (currency_available ?currency_option)
        )
      )
  )
  (:action assign_material_to_slot
    :parameters (?entity_slot - purchase_target ?material - material)
    :precondition
      (and
        (purchase_target_selected ?entity_slot)
        (purchase_target_currency_bound ?entity_slot)
        (material_available ?material)
      )
    :effect
      (and
        (material_assigned_to_purchase_target ?entity_slot ?material)
        (not
          (material_available ?material)
        )
      )
  )
  (:action confirm_slot_reservation
    :parameters (?entity_slot - purchase_target ?material - material)
    :precondition
      (and
        (purchase_target_selected ?entity_slot)
        (purchase_target_currency_bound ?entity_slot)
        (material_assigned_to_purchase_target ?entity_slot ?material)
        (not
          (purchase_target_reserved ?entity_slot)
        )
      )
    :effect (purchase_target_reserved ?entity_slot)
  )
  (:action release_material_from_slot
    :parameters (?entity_slot - purchase_target ?material - material)
    :precondition
      (and
        (material_assigned_to_purchase_target ?entity_slot ?material)
      )
    :effect
      (and
        (material_available ?material)
        (not
          (material_assigned_to_purchase_target ?entity_slot ?material)
        )
      )
  )
  (:action attach_tool_to_slot
    :parameters (?entity_slot - purchase_target ?tool - tool)
    :precondition
      (and
        (purchase_target_reserved ?entity_slot)
        (tool_available ?tool)
      )
    :effect
      (and
        (tool_assigned_to_purchase_target ?entity_slot ?tool)
        (not
          (tool_available ?tool)
        )
      )
  )
  (:action detach_tool_from_slot
    :parameters (?entity_slot - purchase_target ?tool - tool)
    :precondition
      (and
        (tool_assigned_to_purchase_target ?entity_slot ?tool)
      )
    :effect
      (and
        (tool_available ?tool)
        (not
          (tool_assigned_to_purchase_target ?entity_slot ?tool)
        )
      )
  )
  (:action attach_optional_augment_to_loadout
    :parameters (?loadout - loadout ?optional_augment - optional_augment)
    :precondition
      (and
        (purchase_target_reserved ?loadout)
        (optional_augment_available ?optional_augment)
      )
    :effect
      (and
        (loadout_optional_augment_attached ?loadout ?optional_augment)
        (not
          (optional_augment_available ?optional_augment)
        )
      )
  )
  (:action detach_optional_augment_from_loadout
    :parameters (?loadout - loadout ?optional_augment - optional_augment)
    :precondition
      (and
        (loadout_optional_augment_attached ?loadout ?optional_augment)
      )
    :effect
      (and
        (optional_augment_available ?optional_augment)
        (not
          (loadout_optional_augment_attached ?loadout ?optional_augment)
        )
      )
  )
  (:action attach_unique_catalyst_to_loadout
    :parameters (?loadout - loadout ?unique_catalyst - unique_catalyst)
    :precondition
      (and
        (purchase_target_reserved ?loadout)
        (unique_catalyst_available ?unique_catalyst)
      )
    :effect
      (and
        (loadout_unique_catalyst_attached ?loadout ?unique_catalyst)
        (not
          (unique_catalyst_available ?unique_catalyst)
        )
      )
  )
  (:action detach_unique_catalyst_from_loadout
    :parameters (?loadout - loadout ?unique_catalyst - unique_catalyst)
    :precondition
      (and
        (loadout_unique_catalyst_attached ?loadout ?unique_catalyst)
      )
    :effect
      (and
        (unique_catalyst_available ?unique_catalyst)
        (not
          (loadout_unique_catalyst_attached ?loadout ?unique_catalyst)
        )
      )
  )
  (:action reserve_resource_source_for_slot_a
    :parameters (?slot_a - slot_a ?resource_source_a - resource_source_a ?material - material)
    :precondition
      (and
        (purchase_target_reserved ?slot_a)
        (material_assigned_to_purchase_target ?slot_a ?material)
        (slot_a_resource_source ?slot_a ?resource_source_a)
        (not
          (resource_source_a_reserved ?resource_source_a)
        )
        (not
          (resource_source_a_tagged ?resource_source_a)
        )
      )
    :effect (resource_source_a_reserved ?resource_source_a)
  )
  (:action prepare_slot_a_with_tool
    :parameters (?slot_a - slot_a ?resource_source_a - resource_source_a ?tool - tool)
    :precondition
      (and
        (purchase_target_reserved ?slot_a)
        (tool_assigned_to_purchase_target ?slot_a ?tool)
        (slot_a_resource_source ?slot_a ?resource_source_a)
        (resource_source_a_reserved ?resource_source_a)
        (not
          (slot_a_locked ?slot_a)
        )
      )
    :effect
      (and
        (slot_a_locked ?slot_a)
        (slot_a_ready ?slot_a)
      )
  )
  (:action reserve_consumable_for_slot_a
    :parameters (?slot_a - slot_a ?resource_source_a - resource_source_a ?consumable - consumable)
    :precondition
      (and
        (purchase_target_reserved ?slot_a)
        (slot_a_resource_source ?slot_a ?resource_source_a)
        (consumable_available ?consumable)
        (not
          (slot_a_locked ?slot_a)
        )
      )
    :effect
      (and
        (resource_source_a_tagged ?resource_source_a)
        (slot_a_locked ?slot_a)
        (consumable_reserved_for_slot_a ?slot_a ?consumable)
        (not
          (consumable_available ?consumable)
        )
      )
  )
  (:action finalize_slot_a_reservation
    :parameters (?slot_a - slot_a ?resource_source_a - resource_source_a ?material - material ?consumable - consumable)
    :precondition
      (and
        (purchase_target_reserved ?slot_a)
        (material_assigned_to_purchase_target ?slot_a ?material)
        (slot_a_resource_source ?slot_a ?resource_source_a)
        (resource_source_a_tagged ?resource_source_a)
        (consumable_reserved_for_slot_a ?slot_a ?consumable)
        (not
          (slot_a_ready ?slot_a)
        )
      )
    :effect
      (and
        (resource_source_a_reserved ?resource_source_a)
        (slot_a_ready ?slot_a)
        (consumable_available ?consumable)
        (not
          (consumable_reserved_for_slot_a ?slot_a ?consumable)
        )
      )
  )
  (:action reserve_resource_source_for_slot_b
    :parameters (?slot_b - slot_b ?resource_source_b - resource_source_b ?material - material)
    :precondition
      (and
        (purchase_target_reserved ?slot_b)
        (material_assigned_to_purchase_target ?slot_b ?material)
        (slot_b_resource_source ?slot_b ?resource_source_b)
        (not
          (resource_source_b_reserved ?resource_source_b)
        )
        (not
          (resource_source_b_tagged ?resource_source_b)
        )
      )
    :effect (resource_source_b_reserved ?resource_source_b)
  )
  (:action prepare_slot_b_with_tool
    :parameters (?slot_b - slot_b ?resource_source_b - resource_source_b ?tool - tool)
    :precondition
      (and
        (purchase_target_reserved ?slot_b)
        (tool_assigned_to_purchase_target ?slot_b ?tool)
        (slot_b_resource_source ?slot_b ?resource_source_b)
        (resource_source_b_reserved ?resource_source_b)
        (not
          (slot_b_locked ?slot_b)
        )
      )
    :effect
      (and
        (slot_b_locked ?slot_b)
        (slot_b_ready ?slot_b)
      )
  )
  (:action reserve_consumable_for_slot_b
    :parameters (?slot_b - slot_b ?resource_source_b - resource_source_b ?consumable - consumable)
    :precondition
      (and
        (purchase_target_reserved ?slot_b)
        (slot_b_resource_source ?slot_b ?resource_source_b)
        (consumable_available ?consumable)
        (not
          (slot_b_locked ?slot_b)
        )
      )
    :effect
      (and
        (resource_source_b_tagged ?resource_source_b)
        (slot_b_locked ?slot_b)
        (consumable_reserved_for_slot_b ?slot_b ?consumable)
        (not
          (consumable_available ?consumable)
        )
      )
  )
  (:action finalize_slot_b_reservation
    :parameters (?slot_b - slot_b ?resource_source_b - resource_source_b ?material - material ?consumable - consumable)
    :precondition
      (and
        (purchase_target_reserved ?slot_b)
        (material_assigned_to_purchase_target ?slot_b ?material)
        (slot_b_resource_source ?slot_b ?resource_source_b)
        (resource_source_b_tagged ?resource_source_b)
        (consumable_reserved_for_slot_b ?slot_b ?consumable)
        (not
          (slot_b_ready ?slot_b)
        )
      )
    :effect
      (and
        (resource_source_b_reserved ?resource_source_b)
        (slot_b_ready ?slot_b)
        (consumable_available ?consumable)
        (not
          (consumable_reserved_for_slot_b ?slot_b ?consumable)
        )
      )
  )
  (:action assemble_bundle_from_slots
    :parameters (?slot_a - slot_a ?slot_b - slot_b ?resource_source_a - resource_source_a ?resource_source_b - resource_source_b ?bundle - bundle)
    :precondition
      (and
        (slot_a_locked ?slot_a)
        (slot_b_locked ?slot_b)
        (slot_a_resource_source ?slot_a ?resource_source_a)
        (slot_b_resource_source ?slot_b ?resource_source_b)
        (resource_source_a_reserved ?resource_source_a)
        (resource_source_b_reserved ?resource_source_b)
        (slot_a_ready ?slot_a)
        (slot_b_ready ?slot_b)
        (bundle_available ?bundle)
      )
    :effect
      (and
        (bundle_assembled ?bundle)
        (bundle_linked_to_source_a ?bundle ?resource_source_a)
        (bundle_linked_to_source_b ?bundle ?resource_source_b)
        (not
          (bundle_available ?bundle)
        )
      )
  )
  (:action assemble_bundle_with_variant_alpha
    :parameters (?slot_a - slot_a ?slot_b - slot_b ?resource_source_a - resource_source_a ?resource_source_b - resource_source_b ?bundle - bundle)
    :precondition
      (and
        (slot_a_locked ?slot_a)
        (slot_b_locked ?slot_b)
        (slot_a_resource_source ?slot_a ?resource_source_a)
        (slot_b_resource_source ?slot_b ?resource_source_b)
        (resource_source_a_tagged ?resource_source_a)
        (resource_source_b_reserved ?resource_source_b)
        (not
          (slot_a_ready ?slot_a)
        )
        (slot_b_ready ?slot_b)
        (bundle_available ?bundle)
      )
    :effect
      (and
        (bundle_assembled ?bundle)
        (bundle_linked_to_source_a ?bundle ?resource_source_a)
        (bundle_linked_to_source_b ?bundle ?resource_source_b)
        (bundle_variant_alpha ?bundle)
        (not
          (bundle_available ?bundle)
        )
      )
  )
  (:action assemble_bundle_with_variant_beta
    :parameters (?slot_a - slot_a ?slot_b - slot_b ?resource_source_a - resource_source_a ?resource_source_b - resource_source_b ?bundle - bundle)
    :precondition
      (and
        (slot_a_locked ?slot_a)
        (slot_b_locked ?slot_b)
        (slot_a_resource_source ?slot_a ?resource_source_a)
        (slot_b_resource_source ?slot_b ?resource_source_b)
        (resource_source_a_reserved ?resource_source_a)
        (resource_source_b_tagged ?resource_source_b)
        (slot_a_ready ?slot_a)
        (not
          (slot_b_ready ?slot_b)
        )
        (bundle_available ?bundle)
      )
    :effect
      (and
        (bundle_assembled ?bundle)
        (bundle_linked_to_source_a ?bundle ?resource_source_a)
        (bundle_linked_to_source_b ?bundle ?resource_source_b)
        (bundle_variant_beta ?bundle)
        (not
          (bundle_available ?bundle)
        )
      )
  )
  (:action assemble_bundle_with_all_variants
    :parameters (?slot_a - slot_a ?slot_b - slot_b ?resource_source_a - resource_source_a ?resource_source_b - resource_source_b ?bundle - bundle)
    :precondition
      (and
        (slot_a_locked ?slot_a)
        (slot_b_locked ?slot_b)
        (slot_a_resource_source ?slot_a ?resource_source_a)
        (slot_b_resource_source ?slot_b ?resource_source_b)
        (resource_source_a_tagged ?resource_source_a)
        (resource_source_b_tagged ?resource_source_b)
        (not
          (slot_a_ready ?slot_a)
        )
        (not
          (slot_b_ready ?slot_b)
        )
        (bundle_available ?bundle)
      )
    :effect
      (and
        (bundle_assembled ?bundle)
        (bundle_linked_to_source_a ?bundle ?resource_source_a)
        (bundle_linked_to_source_b ?bundle ?resource_source_b)
        (bundle_variant_alpha ?bundle)
        (bundle_variant_beta ?bundle)
        (not
          (bundle_available ?bundle)
        )
      )
  )
  (:action finalize_bundle_for_slot_a
    :parameters (?bundle - bundle ?slot_a - slot_a ?material - material)
    :precondition
      (and
        (bundle_assembled ?bundle)
        (slot_a_locked ?slot_a)
        (material_assigned_to_purchase_target ?slot_a ?material)
        (not
          (bundle_finalized ?bundle)
        )
      )
    :effect (bundle_finalized ?bundle)
  )
  (:action stage_component_for_loadout_bundle
    :parameters (?loadout - loadout ?component - component ?bundle - bundle)
    :precondition
      (and
        (purchase_target_reserved ?loadout)
        (loadout_linked_bundle ?loadout ?bundle)
        (loadout_has_component ?loadout ?component)
        (component_available ?component)
        (bundle_assembled ?bundle)
        (bundle_finalized ?bundle)
        (not
          (component_staged ?component)
        )
      )
    :effect
      (and
        (component_staged ?component)
        (component_assigned_to_bundle ?component ?bundle)
        (not
          (component_available ?component)
        )
      )
  )
  (:action activate_loadout_modifier_sequence
    :parameters (?loadout - loadout ?component - component ?bundle - bundle ?material - material)
    :precondition
      (and
        (purchase_target_reserved ?loadout)
        (loadout_has_component ?loadout ?component)
        (component_staged ?component)
        (component_assigned_to_bundle ?component ?bundle)
        (material_assigned_to_purchase_target ?loadout ?material)
        (not
          (bundle_variant_alpha ?bundle)
        )
        (not
          (loadout_modifier_ready ?loadout)
        )
      )
    :effect (loadout_modifier_ready ?loadout)
  )
  (:action assign_cosmetic_upgrade_token_to_loadout
    :parameters (?loadout - loadout ?cosmetic_upgrade_token - cosmetic_upgrade_token)
    :precondition
      (and
        (purchase_target_reserved ?loadout)
        (cosmetic_upgrade_token_available ?cosmetic_upgrade_token)
        (not
          (loadout_cosmetic_upgrade_enabled ?loadout)
        )
      )
    :effect
      (and
        (loadout_cosmetic_upgrade_enabled ?loadout)
        (loadout_cosmetic_upgrade_token_assigned ?loadout ?cosmetic_upgrade_token)
        (not
          (cosmetic_upgrade_token_available ?cosmetic_upgrade_token)
        )
      )
  )
  (:action attach_component_and_cosmetic_to_loadout
    :parameters (?loadout - loadout ?component - component ?bundle - bundle ?material - material ?cosmetic_upgrade_token - cosmetic_upgrade_token)
    :precondition
      (and
        (purchase_target_reserved ?loadout)
        (loadout_has_component ?loadout ?component)
        (component_staged ?component)
        (component_assigned_to_bundle ?component ?bundle)
        (material_assigned_to_purchase_target ?loadout ?material)
        (bundle_variant_alpha ?bundle)
        (loadout_cosmetic_upgrade_enabled ?loadout)
        (loadout_cosmetic_upgrade_token_assigned ?loadout ?cosmetic_upgrade_token)
        (not
          (loadout_modifier_ready ?loadout)
        )
      )
    :effect
      (and
        (loadout_modifier_ready ?loadout)
        (loadout_cosmetic_stage_complete ?loadout)
      )
  )
  (:action execute_modifier_sequence_branch_a
    :parameters (?loadout - loadout ?optional_augment - optional_augment ?tool - tool ?component - component ?bundle - bundle)
    :precondition
      (and
        (loadout_modifier_ready ?loadout)
        (loadout_optional_augment_attached ?loadout ?optional_augment)
        (tool_assigned_to_purchase_target ?loadout ?tool)
        (loadout_has_component ?loadout ?component)
        (component_assigned_to_bundle ?component ?bundle)
        (not
          (bundle_variant_beta ?bundle)
        )
        (not
          (loadout_modifier_applied ?loadout)
        )
      )
    :effect (loadout_modifier_applied ?loadout)
  )
  (:action execute_modifier_sequence_branch_b
    :parameters (?loadout - loadout ?optional_augment - optional_augment ?tool - tool ?component - component ?bundle - bundle)
    :precondition
      (and
        (loadout_modifier_ready ?loadout)
        (loadout_optional_augment_attached ?loadout ?optional_augment)
        (tool_assigned_to_purchase_target ?loadout ?tool)
        (loadout_has_component ?loadout ?component)
        (component_assigned_to_bundle ?component ?bundle)
        (bundle_variant_beta ?bundle)
        (not
          (loadout_modifier_applied ?loadout)
        )
      )
    :effect (loadout_modifier_applied ?loadout)
  )
  (:action execute_modifier_sequence_stage_three
    :parameters (?loadout - loadout ?unique_catalyst - unique_catalyst ?component - component ?bundle - bundle)
    :precondition
      (and
        (loadout_modifier_applied ?loadout)
        (loadout_unique_catalyst_attached ?loadout ?unique_catalyst)
        (loadout_has_component ?loadout ?component)
        (component_assigned_to_bundle ?component ?bundle)
        (not
          (bundle_variant_alpha ?bundle)
        )
        (not
          (bundle_variant_beta ?bundle)
        )
        (not
          (loadout_ready_for_finalization ?loadout)
        )
      )
    :effect (loadout_ready_for_finalization ?loadout)
  )
  (:action execute_modifier_sequence_stage_four
    :parameters (?loadout - loadout ?unique_catalyst - unique_catalyst ?component - component ?bundle - bundle)
    :precondition
      (and
        (loadout_modifier_applied ?loadout)
        (loadout_unique_catalyst_attached ?loadout ?unique_catalyst)
        (loadout_has_component ?loadout ?component)
