--- domain.pddl (as passed to VAL) ---
(define (domain feed_distribution)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_category - object supply_category - object logistics_category - object organizational_category - object animal_group - organizational_category feed_outlet - resource_category feed_lot - resource_category handler - resource_category additive - resource_category packaging_unit - resource_category reserve_stock - resource_category test_kit - resource_category lab_sample_kit - resource_category supplement_item - supply_category ingredient_batch - supply_category certificate - supply_category local_zone - logistics_category local_feeding_point - logistics_category feed_load - logistics_category animal_partition - animal_group operational_unit - animal_group feeding_point - animal_partition mobile_feeding_point - animal_partition distribution_task - operational_unit)
  (:predicates
    (feeding_unit_registered ?animal_group - animal_group)
    (feeding_unit_prepared_for_feeding ?animal_group - animal_group)
    (feeding_unit_has_outlet_assignment ?animal_group - animal_group)
    (delivery_confirmed ?animal_group - animal_group)
    (completion_recorded ?animal_group - animal_group)
    (feeding_unit_nutrition_assessed ?animal_group - animal_group)
    (feed_outlet_available ?feed_outlet - feed_outlet)
    (feeding_unit_assigned_to_outlet ?animal_group - animal_group ?feed_outlet - feed_outlet)
    (feed_lot_available ?feed_lot - feed_lot)
    (feeding_unit_assigned_to_lot ?animal_group - animal_group ?feed_lot - feed_lot)
    (handler_available ?handler - handler)
    (feeding_unit_assigned_handler ?animal_group - animal_group ?handler - handler)
    (supplement_available ?supplement_item - supplement_item)
    (feeding_point_assigned_supplement ?feeding_point - feeding_point ?supplement_item - supplement_item)
    (mobile_feeding_point_assigned_supplement ?mobile_feeding_point - mobile_feeding_point ?supplement_item - supplement_item)
    (feeding_point_zone_assignment ?feeding_point - feeding_point ?local_zone - local_zone)
    (zone_ready ?local_zone - local_zone)
    (zone_supplement_prepared ?local_zone - local_zone)
    (feeding_point_confirmed ?feeding_point - feeding_point)
    (mobile_feeding_point_local_assignment ?mobile_feeding_point - mobile_feeding_point ?local_feeding_point - local_feeding_point)
    (local_feeding_point_ready ?local_feeding_point - local_feeding_point)
    (local_feeding_point_supplement_prepared ?local_feeding_point - local_feeding_point)
    (mobile_feeding_point_confirmed ?mobile_feeding_point - mobile_feeding_point)
    (feed_load_available ?feed_load - feed_load)
    (feed_load_reserved ?feed_load - feed_load)
    (feed_load_linked_to_zone ?feed_load - feed_load ?local_zone - local_zone)
    (feed_load_linked_to_local_point ?feed_load - feed_load ?local_feeding_point - local_feeding_point)
    (feed_load_testkit_attached ?feed_load - feed_load)
    (feed_load_certificate_attached ?feed_load - feed_load)
    (feed_load_processing_marked ?feed_load - feed_load)
    (task_assigned_feeding_point ?distribution_task - distribution_task ?feeding_point - feeding_point)
    (task_assigned_mobile_feeding_point ?distribution_task - distribution_task ?mobile_feeding_point - mobile_feeding_point)
    (task_assigned_feed_load ?distribution_task - distribution_task ?feed_load - feed_load)
    (ingredient_batch_available ?ingredient_batch - ingredient_batch)
    (task_has_ingredient ?distribution_task - distribution_task ?ingredient_batch - ingredient_batch)
    (ingredient_committed ?ingredient_batch - ingredient_batch)
    (ingredient_committed_to_load ?ingredient_batch - ingredient_batch ?feed_load - feed_load)
    (task_ingredients_committed ?distribution_task - distribution_task)
    (task_qc_prepared ?distribution_task - distribution_task)
    (task_qc_completed ?distribution_task - distribution_task)
    (task_additive_assigned ?distribution_task - distribution_task)
    (task_additive_confirmed ?distribution_task - distribution_task)
    (task_requires_packaging ?distribution_task - distribution_task)
    (task_packaging_completed ?distribution_task - distribution_task)
    (certificate_available ?certificate - certificate)
    (task_assigned_certificate ?distribution_task - distribution_task ?certificate - certificate)
    (task_certificate_applied_flag ?distribution_task - distribution_task)
    (task_certificate_verified_flag ?distribution_task - distribution_task)
    (task_certificate_finalized_flag ?distribution_task - distribution_task)
    (additive_available ?additive - additive)
    (task_assigned_additive ?distribution_task - distribution_task ?additive - additive)
    (packaging_unit_available ?packaging_unit - packaging_unit)
    (task_assigned_packaging_unit ?distribution_task - distribution_task ?packaging_unit - packaging_unit)
    (test_kit_available ?test_kit - test_kit)
    (task_assigned_test_kit ?distribution_task - distribution_task ?test_kit - test_kit)
    (lab_sample_kit_available ?lab_sample_kit - lab_sample_kit)
    (task_assigned_lab_sample_kit ?distribution_task - distribution_task ?lab_sample_kit - lab_sample_kit)
    (reserve_stock_available ?reserve_stock - reserve_stock)
    (feeding_unit_assigned_reserve_stock ?animal_group - animal_group ?reserve_stock - reserve_stock)
    (feeding_point_ready ?feeding_point - feeding_point)
    (mobile_feeding_point_ready ?mobile_feeding_point - mobile_feeding_point)
    (task_finalized ?distribution_task - distribution_task)
  )
  (:action register_feeding_unit
    :parameters (?animal_group - animal_group)
    :precondition
      (and
        (not
          (feeding_unit_registered ?animal_group)
        )
        (not
          (delivery_confirmed ?animal_group)
        )
      )
    :effect (feeding_unit_registered ?animal_group)
  )
  (:action assign_feed_outlet_to_feeding_unit
    :parameters (?animal_group - animal_group ?feed_outlet - feed_outlet)
    :precondition
      (and
        (feeding_unit_registered ?animal_group)
        (not
          (feeding_unit_has_outlet_assignment ?animal_group)
        )
        (feed_outlet_available ?feed_outlet)
      )
    :effect
      (and
        (feeding_unit_has_outlet_assignment ?animal_group)
        (feeding_unit_assigned_to_outlet ?animal_group ?feed_outlet)
        (not
          (feed_outlet_available ?feed_outlet)
        )
      )
  )
  (:action assign_feed_lot_to_feeding_unit
    :parameters (?animal_group - animal_group ?feed_lot - feed_lot)
    :precondition
      (and
        (feeding_unit_registered ?animal_group)
        (feeding_unit_has_outlet_assignment ?animal_group)
        (feed_lot_available ?feed_lot)
      )
    :effect
      (and
        (feeding_unit_assigned_to_lot ?animal_group ?feed_lot)
        (not
          (feed_lot_available ?feed_lot)
        )
      )
  )
  (:action confirm_feeding_unit_readiness
    :parameters (?animal_group - animal_group ?feed_lot - feed_lot)
    :precondition
      (and
        (feeding_unit_registered ?animal_group)
        (feeding_unit_has_outlet_assignment ?animal_group)
        (feeding_unit_assigned_to_lot ?animal_group ?feed_lot)
        (not
          (feeding_unit_prepared_for_feeding ?animal_group)
        )
      )
    :effect (feeding_unit_prepared_for_feeding ?animal_group)
  )
  (:action release_feed_lot_from_feeding_unit
    :parameters (?animal_group - animal_group ?feed_lot - feed_lot)
    :precondition
      (and
        (feeding_unit_assigned_to_lot ?animal_group ?feed_lot)
      )
    :effect
      (and
        (feed_lot_available ?feed_lot)
        (not
          (feeding_unit_assigned_to_lot ?animal_group ?feed_lot)
        )
      )
  )
  (:action assign_handler_to_feeding_unit
    :parameters (?animal_group - animal_group ?handler - handler)
    :precondition
      (and
        (feeding_unit_prepared_for_feeding ?animal_group)
        (handler_available ?handler)
      )
    :effect
      (and
        (feeding_unit_assigned_handler ?animal_group ?handler)
        (not
          (handler_available ?handler)
        )
      )
  )
  (:action release_handler_from_feeding_unit
    :parameters (?animal_group - animal_group ?handler - handler)
    :precondition
      (and
        (feeding_unit_assigned_handler ?animal_group ?handler)
      )
    :effect
      (and
        (handler_available ?handler)
        (not
          (feeding_unit_assigned_handler ?animal_group ?handler)
        )
      )
  )
  (:action assign_testkit_to_task
    :parameters (?distribution_task - distribution_task ?test_kit - test_kit)
    :precondition
      (and
        (feeding_unit_prepared_for_feeding ?distribution_task)
        (test_kit_available ?test_kit)
      )
    :effect
      (and
        (task_assigned_test_kit ?distribution_task ?test_kit)
        (not
          (test_kit_available ?test_kit)
        )
      )
  )
  (:action unassign_testkit_from_task
    :parameters (?distribution_task - distribution_task ?test_kit - test_kit)
    :precondition
      (and
        (task_assigned_test_kit ?distribution_task ?test_kit)
      )
    :effect
      (and
        (test_kit_available ?test_kit)
        (not
          (task_assigned_test_kit ?distribution_task ?test_kit)
        )
      )
  )
  (:action assign_lab_sample_kit_to_task
    :parameters (?distribution_task - distribution_task ?lab_sample_kit - lab_sample_kit)
    :precondition
      (and
        (feeding_unit_prepared_for_feeding ?distribution_task)
        (lab_sample_kit_available ?lab_sample_kit)
      )
    :effect
      (and
        (task_assigned_lab_sample_kit ?distribution_task ?lab_sample_kit)
        (not
          (lab_sample_kit_available ?lab_sample_kit)
        )
      )
  )
  (:action unassign_lab_sample_kit_from_task
    :parameters (?distribution_task - distribution_task ?lab_sample_kit - lab_sample_kit)
    :precondition
      (and
        (task_assigned_lab_sample_kit ?distribution_task ?lab_sample_kit)
      )
    :effect
      (and
        (lab_sample_kit_available ?lab_sample_kit)
        (not
          (task_assigned_lab_sample_kit ?distribution_task ?lab_sample_kit)
        )
      )
  )
  (:action prepare_local_zone_for_delivery
    :parameters (?feeding_point - feeding_point ?local_zone - local_zone ?feed_lot - feed_lot)
    :precondition
      (and
        (feeding_unit_prepared_for_feeding ?feeding_point)
        (feeding_unit_assigned_to_lot ?feeding_point ?feed_lot)
        (feeding_point_zone_assignment ?feeding_point ?local_zone)
        (not
          (zone_ready ?local_zone)
        )
        (not
          (zone_supplement_prepared ?local_zone)
        )
      )
    :effect (zone_ready ?local_zone)
  )
  (:action handler_confirm_feeding_point_ready
    :parameters (?feeding_point - feeding_point ?local_zone - local_zone ?handler - handler)
    :precondition
      (and
        (feeding_unit_prepared_for_feeding ?feeding_point)
        (feeding_unit_assigned_handler ?feeding_point ?handler)
        (feeding_point_zone_assignment ?feeding_point ?local_zone)
        (zone_ready ?local_zone)
        (not
          (feeding_point_ready ?feeding_point)
        )
      )
    :effect
      (and
        (feeding_point_ready ?feeding_point)
        (feeding_point_confirmed ?feeding_point)
      )
  )
  (:action assign_supplement_to_feeding_point
    :parameters (?feeding_point - feeding_point ?local_zone - local_zone ?supplement_item - supplement_item)
    :precondition
      (and
        (feeding_unit_prepared_for_feeding ?feeding_point)
        (feeding_point_zone_assignment ?feeding_point ?local_zone)
        (supplement_available ?supplement_item)
        (not
          (feeding_point_ready ?feeding_point)
        )
      )
    :effect
      (and
        (zone_supplement_prepared ?local_zone)
        (feeding_point_ready ?feeding_point)
        (feeding_point_assigned_supplement ?feeding_point ?supplement_item)
        (not
          (supplement_available ?supplement_item)
        )
      )
  )
  (:action confirm_feeding_point_with_lot_and_supplement
    :parameters (?feeding_point - feeding_point ?local_zone - local_zone ?feed_lot - feed_lot ?supplement_item - supplement_item)
    :precondition
      (and
        (feeding_unit_prepared_for_feeding ?feeding_point)
        (feeding_unit_assigned_to_lot ?feeding_point ?feed_lot)
        (feeding_point_zone_assignment ?feeding_point ?local_zone)
        (zone_supplement_prepared ?local_zone)
        (feeding_point_assigned_supplement ?feeding_point ?supplement_item)
        (not
          (feeding_point_confirmed ?feeding_point)
        )
      )
    :effect
      (and
        (zone_ready ?local_zone)
        (feeding_point_confirmed ?feeding_point)
        (supplement_available ?supplement_item)
        (not
          (feeding_point_assigned_supplement ?feeding_point ?supplement_item)
        )
      )
  )
  (:action prepare_local_feeding_point_for_mobile_delivery
    :parameters (?mobile_feeding_point - mobile_feeding_point ?local_feeding_point - local_feeding_point ?feed_lot - feed_lot)
    :precondition
      (and
        (feeding_unit_prepared_for_feeding ?mobile_feeding_point)
        (feeding_unit_assigned_to_lot ?mobile_feeding_point ?feed_lot)
        (mobile_feeding_point_local_assignment ?mobile_feeding_point ?local_feeding_point)
        (not
          (local_feeding_point_ready ?local_feeding_point)
        )
        (not
          (local_feeding_point_supplement_prepared ?local_feeding_point)
        )
      )
    :effect (local_feeding_point_ready ?local_feeding_point)
  )
  (:action assign_handler_to_mobile_feeding_point
    :parameters (?mobile_feeding_point - mobile_feeding_point ?local_feeding_point - local_feeding_point ?handler - handler)
    :precondition
      (and
        (feeding_unit_prepared_for_feeding ?mobile_feeding_point)
        (feeding_unit_assigned_handler ?mobile_feeding_point ?handler)
        (mobile_feeding_point_local_assignment ?mobile_feeding_point ?local_feeding_point)
        (local_feeding_point_ready ?local_feeding_point)
        (not
          (mobile_feeding_point_ready ?mobile_feeding_point)
        )
      )
    :effect
      (and
        (mobile_feeding_point_ready ?mobile_feeding_point)
        (mobile_feeding_point_confirmed ?mobile_feeding_point)
      )
  )
  (:action assign_supplement_to_mobile_feeding_point
    :parameters (?mobile_feeding_point - mobile_feeding_point ?local_feeding_point - local_feeding_point ?supplement_item - supplement_item)
    :precondition
      (and
        (feeding_unit_prepared_for_feeding ?mobile_feeding_point)
        (mobile_feeding_point_local_assignment ?mobile_feeding_point ?local_feeding_point)
        (supplement_available ?supplement_item)
        (not
          (mobile_feeding_point_ready ?mobile_feeding_point)
        )
      )
    :effect
      (and
        (local_feeding_point_supplement_prepared ?local_feeding_point)
        (mobile_feeding_point_ready ?mobile_feeding_point)
        (mobile_feeding_point_assigned_supplement ?mobile_feeding_point ?supplement_item)
        (not
          (supplement_available ?supplement_item)
        )
      )
  )
  (:action confirm_mobile_feeding_point_with_lot_and_supplement
    :parameters (?mobile_feeding_point - mobile_feeding_point ?local_feeding_point - local_feeding_point ?feed_lot - feed_lot ?supplement_item - supplement_item)
    :precondition
      (and
        (feeding_unit_prepared_for_feeding ?mobile_feeding_point)
        (feeding_unit_assigned_to_lot ?mobile_feeding_point ?feed_lot)
        (mobile_feeding_point_local_assignment ?mobile_feeding_point ?local_feeding_point)
        (local_feeding_point_supplement_prepared ?local_feeding_point)
        (mobile_feeding_point_assigned_supplement ?mobile_feeding_point ?supplement_item)
        (not
          (mobile_feeding_point_confirmed ?mobile_feeding_point)
        )
      )
    :effect
      (and
        (local_feeding_point_ready ?local_feeding_point)
        (mobile_feeding_point_confirmed ?mobile_feeding_point)
        (supplement_available ?supplement_item)
        (not
          (mobile_feeding_point_assigned_supplement ?mobile_feeding_point ?supplement_item)
        )
      )
  )
  (:action assemble_feed_load
    :parameters (?feeding_point - feeding_point ?mobile_feeding_point - mobile_feeding_point ?local_zone - local_zone ?local_feeding_point - local_feeding_point ?feed_load - feed_load)
    :precondition
      (and
        (feeding_point_ready ?feeding_point)
        (mobile_feeding_point_ready ?mobile_feeding_point)
        (feeding_point_zone_assignment ?feeding_point ?local_zone)
        (mobile_feeding_point_local_assignment ?mobile_feeding_point ?local_feeding_point)
        (zone_ready ?local_zone)
        (local_feeding_point_ready ?local_feeding_point)
        (feeding_point_confirmed ?feeding_point)
        (mobile_feeding_point_confirmed ?mobile_feeding_point)
        (feed_load_available ?feed_load)
      )
    :effect
      (and
        (feed_load_reserved ?feed_load)
        (feed_load_linked_to_zone ?feed_load ?local_zone)
        (feed_load_linked_to_local_point ?feed_load ?local_feeding_point)
        (not
          (feed_load_available ?feed_load)
        )
      )
  )
  (:action assemble_feed_load_with_zone_supplement
    :parameters (?feeding_point - feeding_point ?mobile_feeding_point - mobile_feeding_point ?local_zone - local_zone ?local_feeding_point - local_feeding_point ?feed_load - feed_load)
    :precondition
      (and
        (feeding_point_ready ?feeding_point)
        (mobile_feeding_point_ready ?mobile_feeding_point)
        (feeding_point_zone_assignment ?feeding_point ?local_zone)
        (mobile_feeding_point_local_assignment ?mobile_feeding_point ?local_feeding_point)
        (zone_supplement_prepared ?local_zone)
        (local_feeding_point_ready ?local_feeding_point)
        (not
          (feeding_point_confirmed ?feeding_point)
        )
        (mobile_feeding_point_confirmed ?mobile_feeding_point)
        (feed_load_available ?feed_load)
      )
    :effect
      (and
        (feed_load_reserved ?feed_load)
        (feed_load_linked_to_zone ?feed_load ?local_zone)
        (feed_load_linked_to_local_point ?feed_load ?local_feeding_point)
        (feed_load_testkit_attached ?feed_load)
        (not
          (feed_load_available ?feed_load)
        )
      )
  )
  (:action assemble_feed_load_with_local_supplement
    :parameters (?feeding_point - feeding_point ?mobile_feeding_point - mobile_feeding_point ?local_zone - local_zone ?local_feeding_point - local_feeding_point ?feed_load - feed_load)
    :precondition
      (and
        (feeding_point_ready ?feeding_point)
        (mobile_feeding_point_ready ?mobile_feeding_point)
        (feeding_point_zone_assignment ?feeding_point ?local_zone)
        (mobile_feeding_point_local_assignment ?mobile_feeding_point ?local_feeding_point)
        (zone_ready ?local_zone)
        (local_feeding_point_supplement_prepared ?local_feeding_point)
        (feeding_point_confirmed ?feeding_point)
        (not
          (mobile_feeding_point_confirmed ?mobile_feeding_point)
        )
        (feed_load_available ?feed_load)
      )
    :effect
      (and
        (feed_load_reserved ?feed_load)
        (feed_load_linked_to_zone ?feed_load ?local_zone)
        (feed_load_linked_to_local_point ?feed_load ?local_feeding_point)
        (feed_load_certificate_attached ?feed_load)
        (not
          (feed_load_available ?feed_load)
        )
      )
  )
  (:action assemble_feed_load_full
    :parameters (?feeding_point - feeding_point ?mobile_feeding_point - mobile_feeding_point ?local_zone - local_zone ?local_feeding_point - local_feeding_point ?feed_load - feed_load)
    :precondition
      (and
        (feeding_point_ready ?feeding_point)
        (mobile_feeding_point_ready ?mobile_feeding_point)
        (feeding_point_zone_assignment ?feeding_point ?local_zone)
        (mobile_feeding_point_local_assignment ?mobile_feeding_point ?local_feeding_point)
        (zone_supplement_prepared ?local_zone)
        (local_feeding_point_supplement_prepared ?local_feeding_point)
        (not
          (feeding_point_confirmed ?feeding_point)
        )
        (not
          (mobile_feeding_point_confirmed ?mobile_feeding_point)
        )
        (feed_load_available ?feed_load)
      )
    :effect
      (and
        (feed_load_reserved ?feed_load)
        (feed_load_linked_to_zone ?feed_load ?local_zone)
        (feed_load_linked_to_local_point ?feed_load ?local_feeding_point)
        (feed_load_testkit_attached ?feed_load)
        (feed_load_certificate_attached ?feed_load)
        (not
          (feed_load_available ?feed_load)
        )
      )
  )
  (:action mark_feed_load_for_processing
    :parameters (?feed_load - feed_load ?feeding_point - feeding_point ?feed_lot - feed_lot)
    :precondition
      (and
        (feed_load_reserved ?feed_load)
        (feeding_point_ready ?feeding_point)
        (feeding_unit_assigned_to_lot ?feeding_point ?feed_lot)
        (not
          (feed_load_processing_marked ?feed_load)
        )
      )
    :effect (feed_load_processing_marked ?feed_load)
  )
  (:action commit_ingredient_batch_to_load
    :parameters (?distribution_task - distribution_task ?ingredient_batch - ingredient_batch ?feed_load - feed_load)
    :precondition
      (and
        (feeding_unit_prepared_for_feeding ?distribution_task)
        (task_assigned_feed_load ?distribution_task ?feed_load)
        (task_has_ingredient ?distribution_task ?ingredient_batch)
        (ingredient_batch_available ?ingredient_batch)
        (feed_load_reserved ?feed_load)
        (feed_load_processing_marked ?feed_load)
        (not
          (ingredient_committed ?ingredient_batch)
        )
      )
    :effect
      (and
        (ingredient_committed ?ingredient_batch)
        (ingredient_committed_to_load ?ingredient_batch ?feed_load)
        (not
          (ingredient_batch_available ?ingredient_batch)
        )
      )
  )
  (:action finalize_ingredient_commit_and_prepare_task
    :parameters (?distribution_task - distribution_task ?ingredient_batch - ingredient_batch ?feed_load - feed_load ?feed_lot - feed_lot)
    :precondition
      (and
        (feeding_unit_prepared_for_feeding ?distribution_task)
        (task_has_ingredient ?distribution_task ?ingredient_batch)
        (ingredient_committed ?ingredient_batch)
        (ingredient_committed_to_load ?ingredient_batch ?feed_load)
        (feeding_unit_assigned_to_lot ?distribution_task ?feed_lot)
        (not
          (feed_load_testkit_attached ?feed_load)
        )
        (not
          (task_ingredients_committed ?distribution_task)
        )
      )
    :effect (task_ingredients_committed ?distribution_task)
  )
  (:action assign_additive_to_task
    :parameters (?distribution_task - distribution_task ?additive - additive)
    :precondition
      (and
        (feeding_unit_prepared_for_feeding ?distribution_task)
        (additive_available ?additive)
        (not
          (task_additive_assigned ?distribution_task)
        )
      )
    :effect
      (and
        (task_additive_assigned ?distribution_task)
        (task_assigned_additive ?distribution_task ?additive)
        (not
          (additive_available ?additive)
        )
      )
  )
  (:action apply_additive_and_prepare_task
    :parameters (?distribution_task - distribution_task ?ingredient_batch - ingredient_batch ?feed_load - feed_load ?feed_lot - feed_lot ?additive - additive)
    :precondition
      (and
        (feeding_unit_prepared_for_feeding ?distribution_task)
        (task_has_ingredient ?distribution_task ?ingredient_batch)
        (ingredient_committed ?ingredient_batch)
        (ingredient_committed_to_load ?ingredient_batch ?feed_load)
        (feeding_unit_assigned_to_lot ?distribution_task ?feed_lot)
        (feed_load_testkit_attached ?feed_load)
        (task_additive_assigned ?distribution_task)
        (task_assigned_additive ?distribution_task ?additive)
        (not
          (task_ingredients_committed ?distribution_task)
        )
      )
    :effect
      (and
        (task_ingredients_committed ?distribution_task)
        (task_additive_confirmed ?distribution_task)
      )
  )
  (:action allocate_testkit_and_handler_for_qc
    :parameters (?distribution_task - distribution_task ?test_kit - test_kit ?handler - handler ?ingredient_batch - ingredient_batch ?feed_load - feed_load)
    :precondition
      (and
        (task_ingredients_committed ?distribution_task)
        (task_assigned_test_kit ?distribution_task ?test_kit)
        (feeding_unit_assigned_handler ?distribution_task ?handler)
        (task_has_ingredient ?distribution_task ?ingredient_batch)
        (ingredient_committed_to_load ?ingredient_batch ?feed_load)
        (not
          (feed_load_certificate_attached ?feed_load)
        )
        (not
          (task_qc_prepared ?distribution_task)
        )
      )
