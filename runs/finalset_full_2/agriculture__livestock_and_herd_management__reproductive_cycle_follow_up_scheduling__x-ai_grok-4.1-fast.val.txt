  (:action register_animal
    :parameters (?animal - animal)
    :precondition
      (and
        (not
          (animal_registered ?animal)
        )
        (not
          (outcome_recorded ?animal)
        )
      )
    :effect (animal_registered ?animal)
  )
  (:action reserve_breeding_slot
    :parameters (?animal - animal ?breeding_slot - breeding_slot)
    :precondition
      (and
        (animal_registered ?animal)
        (not
          (breeding_slot_reserved ?animal)
        )
        (breeding_slot_available ?breeding_slot)
      )
    :effect
      (and
        (breeding_slot_reserved ?animal)
        (entity_assigned_breeding_slot ?animal ?breeding_slot)
        (not
          (breeding_slot_available ?breeding_slot)
        )
      )
  )

        )
      )

        )
      )
  )
