(define (domain exposure_limit_enforcement_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_grouping - object dimension_grouping - object proposal_grouping - object portfolio_root - object portfolio - portfolio_root limit_profile - resource_grouping security - resource_grouping broker - resource_grouping liquidity_slot - resource_grouping settlement_instruction - resource_grouping client_account - resource_grouping pretrade_approval - resource_grouping execution_authorization - resource_grouping adjustment_instruction - dimension_grouping allocation_template - dimension_grouping strategy_tag - dimension_grouping source_exposure_bucket - proposal_grouping target_exposure_bucket - proposal_grouping trade_proposal - proposal_grouping portfolio_subtype - portfolio plan_subtype - portfolio source_portfolio - portfolio_subtype target_portfolio - portfolio_subtype execution_plan - plan_subtype)
  (:predicates
    (entity_registered ?portfolio - portfolio)
    (entity_ready ?portfolio - portfolio)
    (entity_limit_attached ?portfolio - portfolio)
    (entity_finalized ?portfolio - portfolio)
    (entity_recorded ?portfolio - portfolio)
    (entity_post_trade_assignment_attached ?portfolio - portfolio)
    (limit_profile_available ?limit_profile - limit_profile)
    (entity_assigned_limit_profile ?portfolio - portfolio ?limit_profile - limit_profile)
    (security_available ?security - security)
    (entity_selected_security ?portfolio - portfolio ?security - security)
    (broker_available ?broker - broker)
    (entity_assigned_broker ?portfolio - portfolio ?broker - broker)
    (adjustment_instruction_available ?adjustment_instruction - adjustment_instruction)
    (source_portfolio_bound_adjustment ?source_portfolio - source_portfolio ?adjustment_instruction - adjustment_instruction)
    (target_portfolio_bound_adjustment ?target_portfolio - target_portfolio ?adjustment_instruction - adjustment_instruction)
    (source_portfolio_in_bucket ?source_portfolio - source_portfolio ?source_exposure_bucket - source_exposure_bucket)
    (source_bucket_selected ?source_exposure_bucket - source_exposure_bucket)
    (source_bucket_adjustment_marked ?source_exposure_bucket - source_exposure_bucket)
    (source_portfolio_selection_confirmed ?source_portfolio - source_portfolio)
    (target_portfolio_in_bucket ?target_portfolio - target_portfolio ?target_exposure_bucket - target_exposure_bucket)
    (target_bucket_selected ?target_exposure_bucket - target_exposure_bucket)
    (target_bucket_adjustment_marked ?target_exposure_bucket - target_exposure_bucket)
    (target_portfolio_selection_confirmed ?target_portfolio - target_portfolio)
    (proposal_available ?trade_proposal - trade_proposal)
    (proposal_reserved ?trade_proposal - trade_proposal)
    (proposal_linked_source_bucket ?trade_proposal - trade_proposal ?source_exposure_bucket - source_exposure_bucket)
    (proposal_linked_target_bucket ?trade_proposal - trade_proposal ?target_exposure_bucket - target_exposure_bucket)
    (proposal_requires_liquidity_slot ?trade_proposal - trade_proposal)
    (proposal_requires_additional_liquidity ?trade_proposal - trade_proposal)
    (proposal_committed ?trade_proposal - trade_proposal)
    (plan_linked_to_source_portfolio ?execution_plan - execution_plan ?source_portfolio - source_portfolio)
    (plan_linked_to_target_portfolio ?execution_plan - execution_plan ?target_portfolio - target_portfolio)
    (plan_has_trade_proposal ?execution_plan - execution_plan ?trade_proposal - trade_proposal)
    (allocation_template_available ?allocation_template - allocation_template)
    (plan_bound_allocation_template ?execution_plan - execution_plan ?allocation_template - allocation_template)
    (allocation_template_reserved ?allocation_template - allocation_template)
    (allocation_template_bound_to_proposal ?allocation_template - allocation_template ?trade_proposal - trade_proposal)
    (plan_resources_reserved ?execution_plan - execution_plan)
    (plan_liquidity_confirmed ?execution_plan - execution_plan)
    (plan_authorized ?execution_plan - execution_plan)
    (plan_liquidity_reserved ?execution_plan - execution_plan)
    (plan_liquidity_binding_acknowledged ?execution_plan - execution_plan)
    (plan_settlement_attached ?execution_plan - execution_plan)
    (plan_final_checks_passed ?execution_plan - execution_plan)
    (strategy_tag_available ?strategy_tag - strategy_tag)
    (plan_assigned_strategy_tag ?execution_plan - execution_plan ?strategy_tag - strategy_tag)
    (plan_strategy_tag_attached ?execution_plan - execution_plan)
    (plan_broker_assigned ?execution_plan - execution_plan)
    (plan_broker_authorized ?execution_plan - execution_plan)
    (liquidity_slot_available ?liquidity_slot - liquidity_slot)
    (plan_assigned_liquidity_slot ?execution_plan - execution_plan ?liquidity_slot - liquidity_slot)
    (settlement_instruction_available ?settlement_instruction - settlement_instruction)
    (plan_assigned_settlement_instruction ?execution_plan - execution_plan ?settlement_instruction - settlement_instruction)
    (pretrade_approval_available ?pretrade_approval - pretrade_approval)
    (plan_assigned_pretrade_approval ?execution_plan - execution_plan ?pretrade_approval - pretrade_approval)
    (execution_authorization_available ?execution_authorization - execution_authorization)
    (plan_assigned_execution_authorization ?execution_plan - execution_plan ?execution_authorization - execution_authorization)
    (client_account_available ?client_account - client_account)
    (entity_assigned_client_account ?portfolio - portfolio ?client_account - client_account)
    (source_portfolio_ready ?source_portfolio - source_portfolio)
    (target_portfolio_ready ?target_portfolio - target_portfolio)
    (plan_post_trade_assignment_recorded ?execution_plan - execution_plan)
  )
  (:action register_portfolio
    :parameters (?portfolio - portfolio)
    :precondition
      (and
        (not
          (entity_registered ?portfolio)
        )
        (not
          (entity_finalized ?portfolio)
        )
      )
    :effect (entity_registered ?portfolio)
  )
  (:action attach_limit_profile_to_portfolio
    :parameters (?portfolio - portfolio ?limit_profile - limit_profile)
    :precondition
      (and
        (entity_registered ?portfolio)
        (not
          (entity_limit_attached ?portfolio)
        )
        (limit_profile_available ?limit_profile)
      )
    :effect
      (and
        (entity_limit_attached ?portfolio)
        (entity_assigned_limit_profile ?portfolio ?limit_profile)
        (not
          (limit_profile_available ?limit_profile)
        )
      )
  )
  (:action nominate_security_for_portfolio
    :parameters (?portfolio - portfolio ?security - security)
    :precondition
      (and
        (entity_registered ?portfolio)
        (entity_limit_attached ?portfolio)
        (security_available ?security)
      )
    :effect
      (and
        (entity_selected_security ?portfolio ?security)
        (not
          (security_available ?security)
        )
      )
  )
  (:action finalize_security_nomination
    :parameters (?portfolio - portfolio ?security - security)
    :precondition
      (and
        (entity_registered ?portfolio)
        (entity_limit_attached ?portfolio)
        (entity_selected_security ?portfolio ?security)
        (not
          (entity_ready ?portfolio)
        )
      )
    :effect (entity_ready ?portfolio)
  )
  (:action release_security_nomination
    :parameters (?portfolio - portfolio ?security - security)
    :precondition
      (and
        (entity_selected_security ?portfolio ?security)
      )
    :effect
      (and
        (security_available ?security)
        (not
          (entity_selected_security ?portfolio ?security)
        )
      )
  )
  (:action assign_broker_to_portfolio
    :parameters (?portfolio - portfolio ?broker - broker)
    :precondition
      (and
        (entity_ready ?portfolio)
        (broker_available ?broker)
      )
    :effect
      (and
        (entity_assigned_broker ?portfolio ?broker)
        (not
          (broker_available ?broker)
        )
      )
  )
  (:action unassign_broker_from_portfolio
    :parameters (?portfolio - portfolio ?broker - broker)
    :precondition
      (and
        (entity_assigned_broker ?portfolio ?broker)
      )
    :effect
      (and
        (broker_available ?broker)
        (not
          (entity_assigned_broker ?portfolio ?broker)
        )
      )
  )
  (:action attach_pretrade_approval_to_plan
    :parameters (?execution_plan - execution_plan ?pretrade_approval - pretrade_approval)
    :precondition
      (and
        (entity_ready ?execution_plan)
        (pretrade_approval_available ?pretrade_approval)
      )
    :effect
      (and
        (plan_assigned_pretrade_approval ?execution_plan ?pretrade_approval)
        (not
          (pretrade_approval_available ?pretrade_approval)
        )
      )
  )
  (:action detach_pretrade_approval_from_plan
    :parameters (?execution_plan - execution_plan ?pretrade_approval - pretrade_approval)
    :precondition
      (and
        (plan_assigned_pretrade_approval ?execution_plan ?pretrade_approval)
      )
    :effect
      (and
        (pretrade_approval_available ?pretrade_approval)
        (not
          (plan_assigned_pretrade_approval ?execution_plan ?pretrade_approval)
        )
      )
  )
  (:action attach_execution_authorization_to_plan
    :parameters (?execution_plan - execution_plan ?execution_authorization - execution_authorization)
    :precondition
      (and
        (entity_ready ?execution_plan)
        (execution_authorization_available ?execution_authorization)
      )
    :effect
      (and
        (plan_assigned_execution_authorization ?execution_plan ?execution_authorization)
        (not
          (execution_authorization_available ?execution_authorization)
        )
      )
  )
  (:action detach_execution_authorization_from_plan
    :parameters (?execution_plan - execution_plan ?execution_authorization - execution_authorization)
    :precondition
      (and
        (plan_assigned_execution_authorization ?execution_plan ?execution_authorization)
      )
    :effect
      (and
        (execution_authorization_available ?execution_authorization)
        (not
          (plan_assigned_execution_authorization ?execution_plan ?execution_authorization)
        )
      )
  )
  (:action select_source_bucket_for_security
    :parameters (?source_portfolio - source_portfolio ?source_exposure_bucket - source_exposure_bucket ?security - security)
    :precondition
      (and
        (entity_ready ?source_portfolio)
        (entity_selected_security ?source_portfolio ?security)
        (source_portfolio_in_bucket ?source_portfolio ?source_exposure_bucket)
        (not
          (source_bucket_selected ?source_exposure_bucket)
        )
        (not
          (source_bucket_adjustment_marked ?source_exposure_bucket)
        )
      )
    :effect (source_bucket_selected ?source_exposure_bucket)
  )
  (:action confirm_source_bucket_selection
    :parameters (?source_portfolio - source_portfolio ?source_exposure_bucket - source_exposure_bucket ?broker - broker)
    :precondition
      (and
        (entity_ready ?source_portfolio)
        (entity_assigned_broker ?source_portfolio ?broker)
        (source_portfolio_in_bucket ?source_portfolio ?source_exposure_bucket)
        (source_bucket_selected ?source_exposure_bucket)
        (not
          (source_portfolio_ready ?source_portfolio)
        )
      )
    :effect
      (and
        (source_portfolio_ready ?source_portfolio)
        (source_portfolio_selection_confirmed ?source_portfolio)
      )
  )
  (:action apply_adjustment_to_source_bucket
    :parameters (?source_portfolio - source_portfolio ?source_exposure_bucket - source_exposure_bucket ?adjustment_instruction - adjustment_instruction)
    :precondition
      (and
        (entity_ready ?source_portfolio)
        (source_portfolio_in_bucket ?source_portfolio ?source_exposure_bucket)
        (adjustment_instruction_available ?adjustment_instruction)
        (not
          (source_portfolio_ready ?source_portfolio)
        )
      )
    :effect
      (and
        (source_bucket_adjustment_marked ?source_exposure_bucket)
        (source_portfolio_ready ?source_portfolio)
        (source_portfolio_bound_adjustment ?source_portfolio ?adjustment_instruction)
        (not
          (adjustment_instruction_available ?adjustment_instruction)
        )
      )
  )
  (:action consolidate_source_bucket_selection_with_adjustment
    :parameters (?source_portfolio - source_portfolio ?source_exposure_bucket - source_exposure_bucket ?security - security ?adjustment_instruction - adjustment_instruction)
    :precondition
      (and
        (entity_ready ?source_portfolio)
        (entity_selected_security ?source_portfolio ?security)
        (source_portfolio_in_bucket ?source_portfolio ?source_exposure_bucket)
        (source_bucket_adjustment_marked ?source_exposure_bucket)
        (source_portfolio_bound_adjustment ?source_portfolio ?adjustment_instruction)
        (not
          (source_portfolio_selection_confirmed ?source_portfolio)
        )
      )
    :effect
      (and
        (source_bucket_selected ?source_exposure_bucket)
        (source_portfolio_selection_confirmed ?source_portfolio)
        (adjustment_instruction_available ?adjustment_instruction)
        (not
          (source_portfolio_bound_adjustment ?source_portfolio ?adjustment_instruction)
        )
      )
  )
  (:action select_target_bucket_for_security
    :parameters (?target_portfolio - target_portfolio ?target_exposure_bucket - target_exposure_bucket ?security - security)
    :precondition
      (and
        (entity_ready ?target_portfolio)
        (entity_selected_security ?target_portfolio ?security)
        (target_portfolio_in_bucket ?target_portfolio ?target_exposure_bucket)
        (not
          (target_bucket_selected ?target_exposure_bucket)
        )
        (not
          (target_bucket_adjustment_marked ?target_exposure_bucket)
        )
      )
    :effect (target_bucket_selected ?target_exposure_bucket)
  )
  (:action confirm_target_bucket_selection
    :parameters (?target_portfolio - target_portfolio ?target_exposure_bucket - target_exposure_bucket ?broker - broker)
    :precondition
      (and
        (entity_ready ?target_portfolio)
        (entity_assigned_broker ?target_portfolio ?broker)
        (target_portfolio_in_bucket ?target_portfolio ?target_exposure_bucket)
        (target_bucket_selected ?target_exposure_bucket)
        (not
          (target_portfolio_ready ?target_portfolio)
        )
      )
    :effect
      (and
        (target_portfolio_ready ?target_portfolio)
        (target_portfolio_selection_confirmed ?target_portfolio)
      )
  )
  (:action apply_adjustment_to_target_bucket
    :parameters (?target_portfolio - target_portfolio ?target_exposure_bucket - target_exposure_bucket ?adjustment_instruction - adjustment_instruction)
    :precondition
      (and
        (entity_ready ?target_portfolio)
        (target_portfolio_in_bucket ?target_portfolio ?target_exposure_bucket)
        (adjustment_instruction_available ?adjustment_instruction)
        (not
          (target_portfolio_ready ?target_portfolio)
        )
      )
    :effect
      (and
        (target_bucket_adjustment_marked ?target_exposure_bucket)
        (target_portfolio_ready ?target_portfolio)
        (target_portfolio_bound_adjustment ?target_portfolio ?adjustment_instruction)
        (not
          (adjustment_instruction_available ?adjustment_instruction)
        )
      )
  )
  (:action consolidate_target_bucket_selection_with_adjustment
    :parameters (?target_portfolio - target_portfolio ?target_exposure_bucket - target_exposure_bucket ?security - security ?adjustment_instruction - adjustment_instruction)
    :precondition
      (and
        (entity_ready ?target_portfolio)
        (entity_selected_security ?target_portfolio ?security)
        (target_portfolio_in_bucket ?target_portfolio ?target_exposure_bucket)
        (target_bucket_adjustment_marked ?target_exposure_bucket)
        (target_portfolio_bound_adjustment ?target_portfolio ?adjustment_instruction)
        (not
          (target_portfolio_selection_confirmed ?target_portfolio)
        )
      )
    :effect
      (and
        (target_bucket_selected ?target_exposure_bucket)
        (target_portfolio_selection_confirmed ?target_portfolio)
        (adjustment_instruction_available ?adjustment_instruction)
        (not
          (target_portfolio_bound_adjustment ?target_portfolio ?adjustment_instruction)
        )
      )
  )
  (:action create_trade_proposal_standard
    :parameters (?source_portfolio - source_portfolio ?target_portfolio - target_portfolio ?source_exposure_bucket - source_exposure_bucket ?target_exposure_bucket - target_exposure_bucket ?trade_proposal - trade_proposal)
    :precondition
      (and
        (source_portfolio_ready ?source_portfolio)
        (target_portfolio_ready ?target_portfolio)
        (source_portfolio_in_bucket ?source_portfolio ?source_exposure_bucket)
        (target_portfolio_in_bucket ?target_portfolio ?target_exposure_bucket)
        (source_bucket_selected ?source_exposure_bucket)
        (target_bucket_selected ?target_exposure_bucket)
        (source_portfolio_selection_confirmed ?source_portfolio)
        (target_portfolio_selection_confirmed ?target_portfolio)
        (proposal_available ?trade_proposal)
      )
    :effect
      (and
        (proposal_reserved ?trade_proposal)
        (proposal_linked_source_bucket ?trade_proposal ?source_exposure_bucket)
        (proposal_linked_target_bucket ?trade_proposal ?target_exposure_bucket)
        (not
          (proposal_available ?trade_proposal)
        )
      )
  )
  (:action create_trade_proposal_with_source_adjustment
    :parameters (?source_portfolio - source_portfolio ?target_portfolio - target_portfolio ?source_exposure_bucket - source_exposure_bucket ?target_exposure_bucket - target_exposure_bucket ?trade_proposal - trade_proposal)
    :precondition
      (and
        (source_portfolio_ready ?source_portfolio)
        (target_portfolio_ready ?target_portfolio)
        (source_portfolio_in_bucket ?source_portfolio ?source_exposure_bucket)
        (target_portfolio_in_bucket ?target_portfolio ?target_exposure_bucket)
        (source_bucket_adjustment_marked ?source_exposure_bucket)
        (target_bucket_selected ?target_exposure_bucket)
        (not
          (source_portfolio_selection_confirmed ?source_portfolio)
        )
        (target_portfolio_selection_confirmed ?target_portfolio)
        (proposal_available ?trade_proposal)
      )
    :effect
      (and
        (proposal_reserved ?trade_proposal)
        (proposal_linked_source_bucket ?trade_proposal ?source_exposure_bucket)
        (proposal_linked_target_bucket ?trade_proposal ?target_exposure_bucket)
        (proposal_requires_liquidity_slot ?trade_proposal)
        (not
          (proposal_available ?trade_proposal)
        )
      )
  )
  (:action create_trade_proposal_with_target_adjustment
    :parameters (?source_portfolio - source_portfolio ?target_portfolio - target_portfolio ?source_exposure_bucket - source_exposure_bucket ?target_exposure_bucket - target_exposure_bucket ?trade_proposal - trade_proposal)
    :precondition
      (and
        (source_portfolio_ready ?source_portfolio)
        (target_portfolio_ready ?target_portfolio)
        (source_portfolio_in_bucket ?source_portfolio ?source_exposure_bucket)
        (target_portfolio_in_bucket ?target_portfolio ?target_exposure_bucket)
        (source_bucket_selected ?source_exposure_bucket)
        (target_bucket_adjustment_marked ?target_exposure_bucket)
        (source_portfolio_selection_confirmed ?source_portfolio)
        (not
          (target_portfolio_selection_confirmed ?target_portfolio)
        )
        (proposal_available ?trade_proposal)
      )
    :effect
      (and
        (proposal_reserved ?trade_proposal)
        (proposal_linked_source_bucket ?trade_proposal ?source_exposure_bucket)
        (proposal_linked_target_bucket ?trade_proposal ?target_exposure_bucket)
        (proposal_requires_additional_liquidity ?trade_proposal)
        (not
          (proposal_available ?trade_proposal)
        )
      )
  )
  (:action create_trade_proposal_with_both_adjustments
    :parameters (?source_portfolio - source_portfolio ?target_portfolio - target_portfolio ?source_exposure_bucket - source_exposure_bucket ?target_exposure_bucket - target_exposure_bucket ?trade_proposal - trade_proposal)
    :precondition
      (and
        (source_portfolio_ready ?source_portfolio)
        (target_portfolio_ready ?target_portfolio)
        (source_portfolio_in_bucket ?source_portfolio ?source_exposure_bucket)
        (target_portfolio_in_bucket ?target_portfolio ?target_exposure_bucket)
        (source_bucket_adjustment_marked ?source_exposure_bucket)
        (target_bucket_adjustment_marked ?target_exposure_bucket)
        (not
          (source_portfolio_selection_confirmed ?source_portfolio)
        )
        (not
          (target_portfolio_selection_confirmed ?target_portfolio)
        )
        (proposal_available ?trade_proposal)
      )
    :effect
      (and
        (proposal_reserved ?trade_proposal)
        (proposal_linked_source_bucket ?trade_proposal ?source_exposure_bucket)
        (proposal_linked_target_bucket ?trade_proposal ?target_exposure_bucket)
        (proposal_requires_liquidity_slot ?trade_proposal)
        (proposal_requires_additional_liquidity ?trade_proposal)
        (not
          (proposal_available ?trade_proposal)
        )
      )
  )
  (:action commit_trade_proposal
    :parameters (?trade_proposal - trade_proposal ?source_portfolio - source_portfolio ?security - security)
    :precondition
      (and
        (proposal_reserved ?trade_proposal)
        (source_portfolio_ready ?source_portfolio)
        (entity_selected_security ?source_portfolio ?security)
        (not
          (proposal_committed ?trade_proposal)
        )
      )
    :effect (proposal_committed ?trade_proposal)
  )
  (:action bind_allocation_template_to_plan_and_proposal
    :parameters (?execution_plan - execution_plan ?allocation_template - allocation_template ?trade_proposal - trade_proposal)
    :precondition
      (and
        (entity_ready ?execution_plan)
        (plan_has_trade_proposal ?execution_plan ?trade_proposal)
        (plan_bound_allocation_template ?execution_plan ?allocation_template)
        (allocation_template_available ?allocation_template)
        (proposal_reserved ?trade_proposal)
        (proposal_committed ?trade_proposal)
        (not
          (allocation_template_reserved ?allocation_template)
        )
      )
    :effect
      (and
        (allocation_template_reserved ?allocation_template)
        (allocation_template_bound_to_proposal ?allocation_template ?trade_proposal)
        (not
          (allocation_template_available ?allocation_template)
        )
      )
  )
  (:action reserve_plan_resources
    :parameters (?execution_plan - execution_plan ?allocation_template - allocation_template ?trade_proposal - trade_proposal ?security - security)
    :precondition
      (and
        (entity_ready ?execution_plan)
        (plan_bound_allocation_template ?execution_plan ?allocation_template)
        (allocation_template_reserved ?allocation_template)
        (allocation_template_bound_to_proposal ?allocation_template ?trade_proposal)
        (entity_selected_security ?execution_plan ?security)
        (not
          (proposal_requires_liquidity_slot ?trade_proposal)
        )
        (not
          (plan_resources_reserved ?execution_plan)
        )
      )
    :effect (plan_resources_reserved ?execution_plan)
  )
  (:action reserve_liquidity_slot_for_plan
    :parameters (?execution_plan - execution_plan ?liquidity_slot - liquidity_slot)
    :precondition
      (and
        (entity_ready ?execution_plan)
        (liquidity_slot_available ?liquidity_slot)
        (not
          (plan_liquidity_reserved ?execution_plan)
        )
      )
    :effect
      (and
        (plan_liquidity_reserved ?execution_plan)
        (plan_assigned_liquidity_slot ?execution_plan ?liquidity_slot)
        (not
          (liquidity_slot_available ?liquidity_slot)
        )
      )
  )
  (:action associate_liquidity_slot_and_confirm_plan
    :parameters (?execution_plan - execution_plan ?allocation_template - allocation_template ?trade_proposal - trade_proposal ?security - security ?liquidity_slot - liquidity_slot)
    :precondition
      (and
        (entity_ready ?execution_plan)
        (plan_bound_allocation_template ?execution_plan ?allocation_template)
        (allocation_template_reserved ?allocation_template)
        (allocation_template_bound_to_proposal ?allocation_template ?trade_proposal)
        (entity_selected_security ?execution_plan ?security)
        (proposal_requires_liquidity_slot ?trade_proposal)
        (plan_liquidity_reserved ?execution_plan)
        (plan_assigned_liquidity_slot ?execution_plan ?liquidity_slot)
        (not
          (plan_resources_reserved ?execution_plan)
        )
      )
    :effect
      (and
        (plan_resources_reserved ?execution_plan)
        (plan_liquidity_binding_acknowledged ?execution_plan)
      )
  )
  (:action reserve_broker_and_approval_for_plan
    :parameters (?execution_plan - execution_plan ?pretrade_approval - pretrade_approval ?broker - broker ?allocation_template - allocation_template ?trade_proposal - trade_proposal)
    :precondition
      (and
        (plan_resources_reserved ?execution_plan)
        (plan_assigned_pretrade_approval ?execution_plan ?pretrade_approval)
        (entity_assigned_broker ?execution_plan ?broker)
        (plan_bound_allocation_template ?execution_plan ?allocation_template)
        (allocation_template_bound_to_proposal ?allocation_template ?trade_proposal)
        (not
          (proposal_requires_additional_liquidity ?trade_proposal)
        )
        (not
          (plan_liquidity_confirmed ?execution_plan)
        )
      )
    :effect (plan_liquidity_confirmed ?execution_plan)
  )
  (:action confirm_broker_and_approval_for_plan
    :parameters (?execution_plan - execution_plan ?pretrade_approval - pretrade_approval ?broker - broker ?allocation_template - allocation_template ?trade_proposal - trade_proposal)
    :precondition
      (and
        (plan_resources_reserved ?execution_plan)
