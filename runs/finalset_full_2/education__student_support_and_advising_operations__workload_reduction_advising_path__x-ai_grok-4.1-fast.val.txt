  (:action open_student_case
    :parameters (?student - student)
    :precondition
      (and
        (not
          (student_case_initiated ?student)
        )
        (not
          (actor_workload_adjustment_acknowledged ?student)
        )
      )
    :effect (student_case_initiated ?student)
  )
  (:action schedule_intake_appointment
    :parameters (?student - student ?appointment_slot - appointment_slot)
    :precondition
      (and
        (student_case_initiated ?student)
        (not
          (student_appointment_scheduled ?student)
        )
        (appointment_slot_available ?appointment_slot)
      )
    :effect
      (and
        (student_appointment_scheduled ?student)
        (actor_scheduled_slot ?student ?appointment_slot)
        (not
          (appointment_slot_available ?appointment_slot)
        )
      )
  )
  (:action conduct_intake_assessment
    :parameters (?student - student ?intake_advisor - intake_advisor)
    :precondition
      (and
        (student_case_initiated ?student)
        (student_appointment_scheduled ?student)
        (intake_advisor_available ?intake_advisor)
      )
    :effect
      (and
        (actor_assigned_intake_advisor ?student ?intake_advisor)
        (not
          (intake_advisor_available ?intake_advisor)
        )
      )
  )
  (:action complete_intake_assessment
    :parameters (?student - student ?intake_advisor - intake_advisor)
    :precondition
      (and
        (student_case_initiated ?student)
        (student_appointment_scheduled ?student)
        (actor_assigned_intake_advisor ?student ?intake_advisor)
        (not
          (intake_assessment_completed_for_participant ?student)
        )
      )
    :effect (intake_assessment_completed_for_participant ?student)
  )
  (:action release_intake_advisor
    :parameters (?student - student ?intake_advisor - intake_advisor)
    :precondition
      (and
        (actor_assigned_intake_advisor ?student ?intake_advisor)
      )
    :effect
      (and
        (intake_advisor_available ?intake_advisor)
        (not
          (actor_assigned_intake_advisor ?student ?intake_advisor)
        )
      )
  )
  (:action refer_student_to_service_provider
    :parameters (?student - student ?service_provider - service_provider)

      (and
        (intake_assessment_completed_for_participant ?student)
        (service_provider_available ?service_provider)

        (actor_referred_to_service_provider ?student ?service_provider)

          (service_provider_available ?service_provider)
        )
      )
  )

        (actor_referred_to_service_provider ?student ?service_provider)

    :effect
      (and
        (service_provider_available ?service_provider)

          (actor_referred_to_service_provider ?student ?service_provider)
        )
      )
  )
  (:action claim_approval_token

    :parameters (?case_manager - case_manager ?approval_token - approval_token)

      (and
        (case_manager_holds_approval_token ?case_manager ?approval_token)

    :effect
      (and
        (approval_token_available ?approval_token)

          (case_manager_holds_approval_token ?case_manager ?approval_token)

    :parameters (?case_manager - case_manager ?external_authorization - external_authorization)

      (and
        (case_manager_has_external_authorization ?case_manager ?external_authorization)
      )
    :effect
      (and
        (external_authorization_available ?external_authorization)
        (not
          (case_manager_has_external_authorization ?case_manager ?external_authorization)
        )
      )
  )
  (:action request_section_review

    :parameters (?instructor - instructor ?course_section - course_section ?service_provider - service_provider)

        (course_section_resource_assigned ?course_section)

        (intake_assessment_completed_for_participant ?academic_advisor)
