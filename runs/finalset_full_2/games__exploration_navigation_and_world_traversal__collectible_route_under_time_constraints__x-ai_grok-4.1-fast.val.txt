--- domain.pddl (as passed to VAL) ---
(define (domain collectible_route_under_time_constraints)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity_root - object item_group - object route_group - object location_root - object location - location_root collector_unit - entity_root movement_token - entity_root gate_key - entity_root bonus_node - entity_root upgrade_module - entity_root time_credit - entity_root modifier_item - entity_root condition_marker - entity_root consumable_resource - item_group scannable_subzone - item_group clue_token - item_group route_branch_a - route_group route_branch_b - route_group checkpoint_hub - route_group site_subgroup - location objective_subgroup - location primary_site - site_subgroup secondary_site - site_subgroup objective_site - objective_subgroup)
  (:predicates
    (site_discovered ?location - location)
    (site_activated ?location - location)
    (site_assigned ?location - location)
    (site_collected ?location - location)
    (ready_for_collection ?location - location)
    (collection_in_progress ?location - location)
    (collector_available ?collector_unit - collector_unit)
    (collector_assigned_to_site ?location - location ?collector_unit - collector_unit)
    (movement_token_available ?movement_token - movement_token)
    (movement_token_attached_to_site ?location - location ?movement_token - movement_token)
    (gate_key_available ?gate_key - gate_key)
    (gate_key_attached_to_site ?location - location ?gate_key - gate_key)
    (consumable_available ?consumable - consumable_resource)
    (primary_site_tagged_with_consumable ?primary_site - primary_site ?consumable - consumable_resource)
    (secondary_site_tagged_with_consumable ?secondary_site - secondary_site ?consumable - consumable_resource)
    (primary_site_has_branch_a ?primary_site - primary_site ?branch_a - route_branch_a)
    (branch_a_flag ?branch_a - route_branch_a)
    (branch_a_tagged ?branch_a - route_branch_a)
    (primary_site_ready_for_hub ?primary_site - primary_site)
    (secondary_site_has_branch_b ?secondary_site - secondary_site ?branch_b - route_branch_b)
    (branch_b_flag ?branch_b - route_branch_b)
    (branch_b_tagged ?branch_b - route_branch_b)
    (secondary_site_ready_for_hub ?secondary_site - secondary_site)
    (checkpoint_available ?checkpoint - checkpoint_hub)
    (checkpoint_assembled ?checkpoint - checkpoint_hub)
    (checkpoint_records_branch_a ?checkpoint - checkpoint_hub ?branch_a - route_branch_a)
    (checkpoint_records_branch_b ?checkpoint - checkpoint_hub ?branch_b - route_branch_b)
    (checkpoint_variant_a ?checkpoint - checkpoint_hub)
    (checkpoint_variant_b ?checkpoint - checkpoint_hub)
    (checkpoint_primed ?checkpoint - checkpoint_hub)
    (objective_has_primary_site ?objective - objective_site ?primary_site - primary_site)
    (objective_has_secondary_site ?objective - objective_site ?secondary_site - secondary_site)
    (objective_linked_to_checkpoint ?objective - objective_site ?checkpoint - checkpoint_hub)
    (subzone_available ?subzone - scannable_subzone)
    (objective_has_subzone ?objective - objective_site ?subzone - scannable_subzone)
    (subzone_scanned ?subzone - scannable_subzone)
    (subzone_scanned_for_checkpoint ?subzone - scannable_subzone ?checkpoint - checkpoint_hub)
    (objective_ready_for_modifiers ?objective - objective_site)
    (objective_modifier_applied ?objective - objective_site)
    (objective_condition_met ?objective - objective_site)
    (objective_has_active_bonus ?objective - objective_site)
    (objective_bonus_applied ?objective - objective_site)
    (objective_upgraded ?objective - objective_site)
    (objective_processed ?objective - objective_site)
    (clue_available ?clue_token - clue_token)
    (objective_has_clue ?objective - objective_site ?clue_token - clue_token)
    (objective_clue_registered ?objective - objective_site)
    (objective_clue_handled ?objective - objective_site)
    (objective_clue_finalized ?objective - objective_site)
    (bonus_available ?bonus_node - bonus_node)
    (objective_linked_to_bonus ?objective - objective_site ?bonus_node - bonus_node)
    (upgrade_available ?upgrade_module - upgrade_module)
    (objective_has_upgrade ?objective - objective_site ?upgrade_module - upgrade_module)
    (modifier_available ?modifier_item - modifier_item)
    (objective_has_modifier ?objective - objective_site ?modifier_item - modifier_item)
    (condition_marker_available ?condition_marker - condition_marker)
    (objective_has_condition_marker ?objective - objective_site ?condition_marker - condition_marker)
    (time_credit_available ?time_credit - time_credit)
    (site_time_credit_assigned ?location - location ?time_credit - time_credit)
    (primary_site_flag_present ?primary_site - primary_site)
    (secondary_site_flag_present ?secondary_site - secondary_site)
    (objective_finalized ?objective - objective_site)
  )
  (:action discover_site
    :parameters (?location - location)
    :precondition
      (and
        (not
          (site_discovered ?location)
        )
        (not
          (site_collected ?location)
        )
      )
    :effect (site_discovered ?location)
  )
  (:action assign_collector_to_site
    :parameters (?location - location ?collector_unit - collector_unit)
    :precondition
      (and
        (site_discovered ?location)
        (not
          (site_assigned ?location)
        )
        (collector_available ?collector_unit)
      )
    :effect
      (and
        (site_assigned ?location)
        (collector_assigned_to_site ?location ?collector_unit)
        (not
          (collector_available ?collector_unit)
        )
      )
  )
  (:action attach_movement_token_to_site
    :parameters (?location - location ?movement_token - movement_token)
    :precondition
      (and
        (site_discovered ?location)
        (site_assigned ?location)
        (movement_token_available ?movement_token)
      )
    :effect
      (and
        (movement_token_attached_to_site ?location ?movement_token)
        (not
          (movement_token_available ?movement_token)
        )
      )
  )
  (:action activate_site
    :parameters (?location - location ?movement_token - movement_token)
    :precondition
      (and
        (site_discovered ?location)
        (site_assigned ?location)
        (movement_token_attached_to_site ?location ?movement_token)
        (not
          (site_activated ?location)
        )
      )
    :effect (site_activated ?location)
  )
  (:action return_movement_token
    :parameters (?location - location ?movement_token - movement_token)
    :precondition
      (and
        (movement_token_attached_to_site ?location ?movement_token)
      )
    :effect
      (and
        (movement_token_available ?movement_token)
        (not
          (movement_token_attached_to_site ?location ?movement_token)
        )
      )
  )
  (:action attach_gate_key_to_site
    :parameters (?location - location ?gate_key - gate_key)
    :precondition
      (and
        (site_activated ?location)
        (gate_key_available ?gate_key)
      )
    :effect
      (and
        (gate_key_attached_to_site ?location ?gate_key)
        (not
          (gate_key_available ?gate_key)
        )
      )
  )
  (:action detach_gate_key_from_site
    :parameters (?location - location ?gate_key - gate_key)
    :precondition
      (and
        (gate_key_attached_to_site ?location ?gate_key)
      )
    :effect
      (and
        (gate_key_available ?gate_key)
        (not
          (gate_key_attached_to_site ?location ?gate_key)
        )
      )
  )
  (:action attach_modifier_to_objective
    :parameters (?objective - objective_site ?modifier_item - modifier_item)
    :precondition
      (and
        (site_activated ?objective)
        (modifier_available ?modifier_item)
      )
    :effect
      (and
        (objective_has_modifier ?objective ?modifier_item)
        (not
          (modifier_available ?modifier_item)
        )
      )
  )
  (:action detach_modifier_from_objective
    :parameters (?objective - objective_site ?modifier_item - modifier_item)
    :precondition
      (and
        (objective_has_modifier ?objective ?modifier_item)
      )
    :effect
      (and
        (modifier_available ?modifier_item)
        (not
          (objective_has_modifier ?objective ?modifier_item)
        )
      )
  )
  (:action attach_condition_to_objective
    :parameters (?objective - objective_site ?condition_marker - condition_marker)
    :precondition
      (and
        (site_activated ?objective)
        (condition_marker_available ?condition_marker)
      )
    :effect
      (and
        (objective_has_condition_marker ?objective ?condition_marker)
        (not
          (condition_marker_available ?condition_marker)
        )
      )
  )
  (:action detach_condition_from_objective
    :parameters (?objective - objective_site ?condition_marker - condition_marker)
    :precondition
      (and
        (objective_has_condition_marker ?objective ?condition_marker)
      )
    :effect
      (and
        (condition_marker_available ?condition_marker)
        (not
          (objective_has_condition_marker ?objective ?condition_marker)
        )
      )
  )
  (:action flag_primary_branch_a
    :parameters (?primary_site - primary_site ?branch_a - route_branch_a ?movement_token - movement_token)
    :precondition
      (and
        (site_activated ?primary_site)
        (movement_token_attached_to_site ?primary_site ?movement_token)
        (primary_site_has_branch_a ?primary_site ?branch_a)
        (not
          (branch_a_flag ?branch_a)
        )
        (not
          (branch_a_tagged ?branch_a)
        )
      )
    :effect (branch_a_flag ?branch_a)
  )
  (:action process_primary_branch_a_with_key
    :parameters (?primary_site - primary_site ?branch_a - route_branch_a ?gate_key - gate_key)
    :precondition
      (and
        (site_activated ?primary_site)
        (gate_key_attached_to_site ?primary_site ?gate_key)
        (primary_site_has_branch_a ?primary_site ?branch_a)
        (branch_a_flag ?branch_a)
        (not
          (primary_site_flag_present ?primary_site)
        )
      )
    :effect
      (and
        (primary_site_flag_present ?primary_site)
        (primary_site_ready_for_hub ?primary_site)
      )
  )
  (:action tag_primary_branch_a_with_consumable
    :parameters (?primary_site - primary_site ?branch_a - route_branch_a ?consumable - consumable_resource)
    :precondition
      (and
        (site_activated ?primary_site)
        (primary_site_has_branch_a ?primary_site ?branch_a)
        (consumable_available ?consumable)
        (not
          (primary_site_flag_present ?primary_site)
        )
      )
    :effect
      (and
        (branch_a_tagged ?branch_a)
        (primary_site_flag_present ?primary_site)
        (primary_site_tagged_with_consumable ?primary_site ?consumable)
        (not
          (consumable_available ?consumable)
        )
      )
  )
  (:action finalize_primary_branch_a_with_consumable
    :parameters (?primary_site - primary_site ?branch_a - route_branch_a ?movement_token - movement_token ?consumable - consumable_resource)
    :precondition
      (and
        (site_activated ?primary_site)
        (movement_token_attached_to_site ?primary_site ?movement_token)
        (primary_site_has_branch_a ?primary_site ?branch_a)
        (branch_a_tagged ?branch_a)
        (primary_site_tagged_with_consumable ?primary_site ?consumable)
        (not
          (primary_site_ready_for_hub ?primary_site)
        )
      )
    :effect
      (and
        (branch_a_flag ?branch_a)
        (primary_site_ready_for_hub ?primary_site)
        (consumable_available ?consumable)
        (not
          (primary_site_tagged_with_consumable ?primary_site ?consumable)
        )
      )
  )
  (:action flag_secondary_branch_b
    :parameters (?secondary_site - secondary_site ?branch_b - route_branch_b ?movement_token - movement_token)
    :precondition
      (and
        (site_activated ?secondary_site)
        (movement_token_attached_to_site ?secondary_site ?movement_token)
        (secondary_site_has_branch_b ?secondary_site ?branch_b)
        (not
          (branch_b_flag ?branch_b)
        )
        (not
          (branch_b_tagged ?branch_b)
        )
      )
    :effect (branch_b_flag ?branch_b)
  )
  (:action process_secondary_branch_b_with_key
    :parameters (?secondary_site - secondary_site ?branch_b - route_branch_b ?gate_key - gate_key)
    :precondition
      (and
        (site_activated ?secondary_site)
        (gate_key_attached_to_site ?secondary_site ?gate_key)
        (secondary_site_has_branch_b ?secondary_site ?branch_b)
        (branch_b_flag ?branch_b)
        (not
          (secondary_site_flag_present ?secondary_site)
        )
      )
    :effect
      (and
        (secondary_site_flag_present ?secondary_site)
        (secondary_site_ready_for_hub ?secondary_site)
      )
  )
  (:action tag_secondary_branch_b_with_consumable
    :parameters (?secondary_site - secondary_site ?branch_b - route_branch_b ?consumable - consumable_resource)
    :precondition
      (and
        (site_activated ?secondary_site)
        (secondary_site_has_branch_b ?secondary_site ?branch_b)
        (consumable_available ?consumable)
        (not
          (secondary_site_flag_present ?secondary_site)
        )
      )
    :effect
      (and
        (branch_b_tagged ?branch_b)
        (secondary_site_flag_present ?secondary_site)
        (secondary_site_tagged_with_consumable ?secondary_site ?consumable)
        (not
          (consumable_available ?consumable)
        )
      )
  )
  (:action finalize_secondary_branch_b_with_consumable
    :parameters (?secondary_site - secondary_site ?branch_b - route_branch_b ?movement_token - movement_token ?consumable - consumable_resource)
    :precondition
      (and
        (site_activated ?secondary_site)
        (movement_token_attached_to_site ?secondary_site ?movement_token)
        (secondary_site_has_branch_b ?secondary_site ?branch_b)
        (branch_b_tagged ?branch_b)
        (secondary_site_tagged_with_consumable ?secondary_site ?consumable)
        (not
          (secondary_site_ready_for_hub ?secondary_site)
        )
      )
    :effect
      (and
        (branch_b_flag ?branch_b)
        (secondary_site_ready_for_hub ?secondary_site)
        (consumable_available ?consumable)
        (not
          (secondary_site_tagged_with_consumable ?secondary_site ?consumable)
        )
      )
  )
  (:action assemble_checkpoint_hub
    :parameters (?primary_site - primary_site ?secondary_site - secondary_site ?branch_a - route_branch_a ?branch_b - route_branch_b ?checkpoint - checkpoint_hub)
    :precondition
      (and
        (primary_site_flag_present ?primary_site)
        (secondary_site_flag_present ?secondary_site)
        (primary_site_has_branch_a ?primary_site ?branch_a)
        (secondary_site_has_branch_b ?secondary_site ?branch_b)
        (branch_a_flag ?branch_a)
        (branch_b_flag ?branch_b)
        (primary_site_ready_for_hub ?primary_site)
        (secondary_site_ready_for_hub ?secondary_site)
        (checkpoint_available ?checkpoint)
      )
    :effect
      (and
        (checkpoint_assembled ?checkpoint)
        (checkpoint_records_branch_a ?checkpoint ?branch_a)
        (checkpoint_records_branch_b ?checkpoint ?branch_b)
        (not
          (checkpoint_available ?checkpoint)
        )
      )
  )
  (:action assemble_checkpoint_hub_variant_a
    :parameters (?primary_site - primary_site ?secondary_site - secondary_site ?branch_a - route_branch_a ?branch_b - route_branch_b ?checkpoint - checkpoint_hub)
    :precondition
      (and
        (primary_site_flag_present ?primary_site)
        (secondary_site_flag_present ?secondary_site)
        (primary_site_has_branch_a ?primary_site ?branch_a)
        (secondary_site_has_branch_b ?secondary_site ?branch_b)
        (branch_a_tagged ?branch_a)
        (branch_b_flag ?branch_b)
        (not
          (primary_site_ready_for_hub ?primary_site)
        )
        (secondary_site_ready_for_hub ?secondary_site)
        (checkpoint_available ?checkpoint)
      )
    :effect
      (and
        (checkpoint_assembled ?checkpoint)
        (checkpoint_records_branch_a ?checkpoint ?branch_a)
        (checkpoint_records_branch_b ?checkpoint ?branch_b)
        (checkpoint_variant_a ?checkpoint)
        (not
          (checkpoint_available ?checkpoint)
        )
      )
  )
  (:action assemble_checkpoint_hub_variant_b
    :parameters (?primary_site - primary_site ?secondary_site - secondary_site ?branch_a - route_branch_a ?branch_b - route_branch_b ?checkpoint - checkpoint_hub)
    :precondition
      (and
        (primary_site_flag_present ?primary_site)
        (secondary_site_flag_present ?secondary_site)
        (primary_site_has_branch_a ?primary_site ?branch_a)
        (secondary_site_has_branch_b ?secondary_site ?branch_b)
        (branch_a_flag ?branch_a)
        (branch_b_tagged ?branch_b)
        (primary_site_ready_for_hub ?primary_site)
        (not
          (secondary_site_ready_for_hub ?secondary_site)
        )
        (checkpoint_available ?checkpoint)
      )
    :effect
      (and
        (checkpoint_assembled ?checkpoint)
        (checkpoint_records_branch_a ?checkpoint ?branch_a)
        (checkpoint_records_branch_b ?checkpoint ?branch_b)
        (checkpoint_variant_b ?checkpoint)
        (not
          (checkpoint_available ?checkpoint)
        )
      )
  )
  (:action assemble_checkpoint_hub_variant_ab
    :parameters (?primary_site - primary_site ?secondary_site - secondary_site ?branch_a - route_branch_a ?branch_b - route_branch_b ?checkpoint - checkpoint_hub)
    :precondition
      (and
        (primary_site_flag_present ?primary_site)
        (secondary_site_flag_present ?secondary_site)
        (primary_site_has_branch_a ?primary_site ?branch_a)
        (secondary_site_has_branch_b ?secondary_site ?branch_b)
        (branch_a_tagged ?branch_a)
        (branch_b_tagged ?branch_b)
        (not
          (primary_site_ready_for_hub ?primary_site)
        )
        (not
          (secondary_site_ready_for_hub ?secondary_site)
        )
        (checkpoint_available ?checkpoint)
      )
    :effect
      (and
        (checkpoint_assembled ?checkpoint)
        (checkpoint_records_branch_a ?checkpoint ?branch_a)
        (checkpoint_records_branch_b ?checkpoint ?branch_b)
        (checkpoint_variant_a ?checkpoint)
        (checkpoint_variant_b ?checkpoint)
        (not
          (checkpoint_available ?checkpoint)
        )
      )
  )
  (:action prime_checkpoint_for_subzone
    :parameters (?checkpoint - checkpoint_hub ?primary_site - primary_site ?movement_token - movement_token)
    :precondition
      (and
        (checkpoint_assembled ?checkpoint)
        (primary_site_flag_present ?primary_site)
        (movement_token_attached_to_site ?primary_site ?movement_token)
        (not
          (checkpoint_primed ?checkpoint)
        )
      )
    :effect (checkpoint_primed ?checkpoint)
  )
  (:action scan_subzone_into_checkpoint
    :parameters (?objective - objective_site ?subzone - scannable_subzone ?checkpoint - checkpoint_hub)
    :precondition
      (and
        (site_activated ?objective)
        (objective_linked_to_checkpoint ?objective ?checkpoint)
        (objective_has_subzone ?objective ?subzone)
        (subzone_available ?subzone)
        (checkpoint_assembled ?checkpoint)
        (checkpoint_primed ?checkpoint)
        (not
          (subzone_scanned ?subzone)
        )
      )
    :effect
      (and
        (subzone_scanned ?subzone)
        (subzone_scanned_for_checkpoint ?subzone ?checkpoint)
        (not
          (subzone_available ?subzone)
        )
      )
  )
  (:action prepare_objective_after_subzone_scan
    :parameters (?objective - objective_site ?subzone - scannable_subzone ?checkpoint - checkpoint_hub ?movement_token - movement_token)
    :precondition
      (and
        (site_activated ?objective)
        (objective_has_subzone ?objective ?subzone)
        (subzone_scanned ?subzone)
        (subzone_scanned_for_checkpoint ?subzone ?checkpoint)
        (movement_token_attached_to_site ?objective ?movement_token)
        (not
          (checkpoint_variant_a ?checkpoint)
        )
        (not
          (objective_ready_for_modifiers ?objective)
        )
      )
    :effect (objective_ready_for_modifiers ?objective)
  )
  (:action attach_bonus_to_objective
    :parameters (?objective - objective_site ?bonus_node - bonus_node)
    :precondition
      (and
        (site_activated ?objective)
        (bonus_available ?bonus_node)
        (not
          (objective_has_active_bonus ?objective)
        )
      )
    :effect
      (and
        (objective_has_active_bonus ?objective)
        (objective_linked_to_bonus ?objective ?bonus_node)
        (not
          (bonus_available ?bonus_node)
        )
      )
  )
  (:action apply_bonus_variant_to_objective
    :parameters (?objective - objective_site ?subzone - scannable_subzone ?checkpoint - checkpoint_hub ?movement_token - movement_token ?bonus_node - bonus_node)
    :precondition
      (and
        (site_activated ?objective)
        (objective_has_subzone ?objective ?subzone)
        (subzone_scanned ?subzone)
        (subzone_scanned_for_checkpoint ?subzone ?checkpoint)
        (movement_token_attached_to_site ?objective ?movement_token)
        (checkpoint_variant_a ?checkpoint)
        (objective_has_active_bonus ?objective)
        (objective_linked_to_bonus ?objective ?bonus_node)
        (not
          (objective_ready_for_modifiers ?objective)
        )
      )
    :effect
      (and
        (objective_ready_for_modifiers ?objective)
        (objective_bonus_applied ?objective)
      )
  )
  (:action apply_modifier_to_objective_variant_a
    :parameters (?objective - objective_site ?modifier_item - modifier_item ?gate_key - gate_key ?subzone - scannable_subzone ?checkpoint - checkpoint_hub)
    :precondition
      (and
        (objective_ready_for_modifiers ?objective)
        (objective_has_modifier ?objective ?modifier_item)
        (gate_key_attached_to_site ?objective ?gate_key)
        (objective_has_subzone ?objective ?subzone)
        (subzone_scanned_for_checkpoint ?subzone ?checkpoint)
        (not
          (checkpoint_variant_b ?checkpoint)
        )
        (not
          (objective_modifier_applied ?objective)
        )
      )
    :effect (objective_modifier_applied ?objective)
  )
  (:action apply_modifier_to_objective_variant_b
    :parameters (?objective - objective_site ?modifier_item - modifier_item ?gate_key - gate_key ?subzone - scannable_subzone ?checkpoint - checkpoint_hub)
    :precondition
      (and
        (objective_ready_for_modifiers ?objective)
        (objective_has_modifier ?objective ?modifier_item)
        (gate_key_attached_to_site ?objective ?gate_key)
        (objective_has_subzone ?objective ?subzone)
        (subzone_scanned_for_checkpoint ?subzone ?checkpoint)
        (checkpoint_variant_b ?checkpoint)
        (not
          (objective_modifier_applied ?objective)
        )
      )
    :effect (objective_modifier_applied ?objective)
  )
  (:action apply_condition_marker_to_objective
    :parameters (?objective - objective_site ?condition_marker - condition_marker ?subzone - scannable_subzone ?checkpoint - checkpoint_hub)
    :precondition
      (and
        (objective_modifier_applied ?objective)
        (objective_has_condition_marker ?objective ?condition_marker)
        (objective_has_subzone ?objective ?subzone)
        (subzone_scanned_for_checkpoint ?subzone ?checkpoint)
        (not
          (checkpoint_variant_a ?checkpoint)
        )
        (not
          (checkpoint_variant_b ?checkpoint)
        )
        (not
          (objective_condition_met ?objective)
        )
      )
    :effect (objective_condition_met ?objective)
  )
  (:action apply_condition_with_variant_a
    :parameters (?objective - objective_site ?condition_marker - condition_marker ?subzone - scannable_subzone ?checkpoint - checkpoint_hub)
    :precondition
      (and
        (objective_modifier_applied ?objective)
