--- domain.pddl (as passed to VAL) ---
(define (domain health_related_absence_course_replanning)
  (:requirements :strips :typing :negative-preconditions)
  (:types case_resource - object case_attachment - object scheduling_and_proposal_resource - object case_category - object case_subject - case_category case_handler_staff - case_resource medical_document - case_resource faculty_representative - case_resource policy_reference - case_resource approval_token - case_resource medical_certificate_type - case_resource availability_token - case_resource external_authority_reference - case_resource supporting_evidence - case_attachment approval_document - case_attachment policy_attachment - case_attachment schedule_slot_type_a - scheduling_and_proposal_resource schedule_slot_type_b - scheduling_and_proposal_resource replan_proposal - scheduling_and_proposal_resource enrollment_parent - case_subject enrollment_variant - case_subject student_enrollment - enrollment_parent module_enrollment - enrollment_parent administrative_case_record - enrollment_variant)
  (:predicates
    (case_registered ?case_subject - case_subject)
    (case_validated ?case_subject - case_subject)
    (case_assigned ?case_subject - case_subject)
    (replan_applied ?case_subject - case_subject)
    (ready_for_certificate_binding ?case_subject - case_subject)
    (medical_certificate_attached ?case_subject - case_subject)
    (handler_available ?case_handler_staff - case_handler_staff)
    (case_assigned_to ?case_subject - case_subject ?case_handler_staff - case_handler_staff)
    (medical_document_available ?medical_document - medical_document)
    (attached_medical_document ?case_subject - case_subject ?medical_document - medical_document)
    (faculty_available ?faculty_representative - faculty_representative)
    (faculty_assigned_to_case ?case_subject - case_subject ?faculty_representative - faculty_representative)
    (supporting_evidence_available ?supporting_evidence - supporting_evidence)
    (student_evidence_attached ?student_enrollment - student_enrollment ?supporting_evidence - supporting_evidence)
    (module_evidence_attached ?module_enrollment - module_enrollment ?supporting_evidence - supporting_evidence)
    (enrollment_slot_type_a_reserved ?student_enrollment - student_enrollment ?schedule_slot_type_a - schedule_slot_type_a)
    (slot_a_confirmed ?schedule_slot_type_a - schedule_slot_type_a)
    (slot_a_tentatively_reserved ?schedule_slot_type_a - schedule_slot_type_a)
    (student_negotiation_confirmed ?student_enrollment - student_enrollment)
    (enrollment_slot_type_b_reserved ?module_enrollment - module_enrollment ?schedule_slot_type_b - schedule_slot_type_b)
    (slot_b_confirmed ?schedule_slot_type_b - schedule_slot_type_b)
    (slot_b_tentatively_reserved ?schedule_slot_type_b - schedule_slot_type_b)
    (module_negotiation_confirmed ?module_enrollment - module_enrollment)
    (proposal_placeholder_available ?replan_proposal - replan_proposal)
    (proposal_reserved ?replan_proposal - replan_proposal)
    (proposal_includes_slot_a ?replan_proposal - replan_proposal ?schedule_slot_type_a - schedule_slot_type_a)
    (proposal_includes_slot_b ?replan_proposal - replan_proposal ?schedule_slot_type_b - schedule_slot_type_b)
    (proposal_option_a_locked ?replan_proposal - replan_proposal)
    (proposal_option_b_locked ?replan_proposal - replan_proposal)
    (proposal_finalized ?replan_proposal - replan_proposal)
    (case_linked_to_student_enrollment ?administrative_case_record - administrative_case_record ?student_enrollment - student_enrollment)
    (case_linked_to_module_enrollment ?administrative_case_record - administrative_case_record ?module_enrollment - module_enrollment)
    (case_linked_to_proposal ?administrative_case_record - administrative_case_record ?replan_proposal - replan_proposal)
    (approval_document_available ?approval_document - approval_document)
    (case_has_approval_document ?administrative_case_record - administrative_case_record ?approval_document - approval_document)
    (approval_document_consumed ?approval_document - approval_document)
    (approval_document_linked_to_proposal ?approval_document - approval_document ?replan_proposal - replan_proposal)
    (compliance_stage_one_passed ?administrative_case_record - administrative_case_record)
    (compliance_stage_two_passed ?administrative_case_record - administrative_case_record)
    (compliance_ready_for_final_review ?administrative_case_record - administrative_case_record)
    (policy_reference_assigned ?administrative_case_record - administrative_case_record)
    (policy_attachment_verified ?administrative_case_record - administrative_case_record)
    (compliance_documentation_verified ?administrative_case_record - administrative_case_record)
    (final_compliance_check_passed ?administrative_case_record - administrative_case_record)
    (policy_attachment_available ?policy_attachment - policy_attachment)
    (case_has_policy_attachment ?administrative_case_record - administrative_case_record ?policy_attachment - policy_attachment)
    (policy_validation_initiated ?administrative_case_record - administrative_case_record)
    (policy_review_assigned ?administrative_case_record - administrative_case_record)
    (policy_review_completed ?administrative_case_record - administrative_case_record)
    (policy_reference_available ?policy_reference - policy_reference)
    (case_has_policy_reference ?administrative_case_record - administrative_case_record ?policy_reference - policy_reference)
    (approval_token_available ?approval_token - approval_token)
    (case_has_approval_token ?administrative_case_record - administrative_case_record ?approval_token - approval_token)
    (availability_token_available ?availability_token - availability_token)
    (case_has_availability_token ?administrative_case_record - administrative_case_record ?availability_token - availability_token)
    (external_authority_available ?external_authority_reference - external_authority_reference)
    (external_authority_linked_to_case ?administrative_case_record - administrative_case_record ?external_authority_reference - external_authority_reference)
    (medical_certificate_type_available ?medical_certificate_type - medical_certificate_type)
    (case_bound_to_medical_certificate_type ?case_subject - case_subject ?medical_certificate_type - medical_certificate_type)
    (student_negotiation_initiated ?student_enrollment - student_enrollment)
    (module_negotiation_initiated ?module_enrollment - module_enrollment)
    (case_finalized ?administrative_case_record - administrative_case_record)
  )
  (:action register_case
    :parameters (?case_subject - case_subject)
    :precondition
      (and
        (not
          (case_registered ?case_subject)
        )
        (not
          (replan_applied ?case_subject)
        )
      )
    :effect (case_registered ?case_subject)
  )
  (:action assign_case_handler
    :parameters (?case_subject - case_subject ?case_handler_staff - case_handler_staff)
    :precondition
      (and
        (case_registered ?case_subject)
        (not
          (case_assigned ?case_subject)
        )
        (handler_available ?case_handler_staff)
      )
    :effect
      (and
        (case_assigned ?case_subject)
        (case_assigned_to ?case_subject ?case_handler_staff)
        (not
          (handler_available ?case_handler_staff)
        )
      )
  )
  (:action attach_medical_document
    :parameters (?case_subject - case_subject ?medical_document - medical_document)
    :precondition
      (and
        (case_registered ?case_subject)
        (case_assigned ?case_subject)
        (medical_document_available ?medical_document)
      )
    :effect
      (and
        (attached_medical_document ?case_subject ?medical_document)
        (not
          (medical_document_available ?medical_document)
        )
      )
  )
  (:action validate_medical_evidence
    :parameters (?case_subject - case_subject ?medical_document - medical_document)
    :precondition
      (and
        (case_registered ?case_subject)
        (case_assigned ?case_subject)
        (attached_medical_document ?case_subject ?medical_document)
        (not
          (case_validated ?case_subject)
        )
      )
    :effect (case_validated ?case_subject)
  )
  (:action unlink_medical_document
    :parameters (?case_subject - case_subject ?medical_document - medical_document)
    :precondition
      (and
        (attached_medical_document ?case_subject ?medical_document)
      )
    :effect
      (and
        (medical_document_available ?medical_document)
        (not
          (attached_medical_document ?case_subject ?medical_document)
        )
      )
  )
  (:action assign_faculty_to_case
    :parameters (?case_subject - case_subject ?faculty_representative - faculty_representative)
    :precondition
      (and
        (case_validated ?case_subject)
        (faculty_available ?faculty_representative)
      )
    :effect
      (and
        (faculty_assigned_to_case ?case_subject ?faculty_representative)
        (not
          (faculty_available ?faculty_representative)
        )
      )
  )
  (:action unassign_faculty_from_case
    :parameters (?case_subject - case_subject ?faculty_representative - faculty_representative)
    :precondition
      (and
        (faculty_assigned_to_case ?case_subject ?faculty_representative)
      )
    :effect
      (and
        (faculty_available ?faculty_representative)
        (not
          (faculty_assigned_to_case ?case_subject ?faculty_representative)
        )
      )
  )
  (:action assign_availability_token_to_case
    :parameters (?administrative_case_record - administrative_case_record ?availability_token - availability_token)
    :precondition
      (and
        (case_validated ?administrative_case_record)
        (availability_token_available ?availability_token)
      )
    :effect
      (and
        (case_has_availability_token ?administrative_case_record ?availability_token)
        (not
          (availability_token_available ?availability_token)
        )
      )
  )
  (:action release_availability_token_from_case
    :parameters (?administrative_case_record - administrative_case_record ?availability_token - availability_token)
    :precondition
      (and
        (case_has_availability_token ?administrative_case_record ?availability_token)
      )
    :effect
      (and
        (availability_token_available ?availability_token)
        (not
          (case_has_availability_token ?administrative_case_record ?availability_token)
        )
      )
  )
  (:action link_external_authority_reference
    :parameters (?administrative_case_record - administrative_case_record ?external_authority_reference - external_authority_reference)
    :precondition
      (and
        (case_validated ?administrative_case_record)
        (external_authority_available ?external_authority_reference)
      )
    :effect
      (and
        (external_authority_linked_to_case ?administrative_case_record ?external_authority_reference)
        (not
          (external_authority_available ?external_authority_reference)
        )
      )
  )
  (:action unlink_external_authority_reference
    :parameters (?administrative_case_record - administrative_case_record ?external_authority_reference - external_authority_reference)
    :precondition
      (and
        (external_authority_linked_to_case ?administrative_case_record ?external_authority_reference)
      )
    :effect
      (and
        (external_authority_available ?external_authority_reference)
        (not
          (external_authority_linked_to_case ?administrative_case_record ?external_authority_reference)
        )
      )
  )
  (:action confirm_student_slot_a
    :parameters (?student_enrollment - student_enrollment ?schedule_slot_type_a - schedule_slot_type_a ?medical_document - medical_document)
    :precondition
      (and
        (case_validated ?student_enrollment)
        (attached_medical_document ?student_enrollment ?medical_document)
        (enrollment_slot_type_a_reserved ?student_enrollment ?schedule_slot_type_a)
        (not
          (slot_a_confirmed ?schedule_slot_type_a)
        )
        (not
          (slot_a_tentatively_reserved ?schedule_slot_type_a)
        )
      )
    :effect (slot_a_confirmed ?schedule_slot_type_a)
  )
  (:action finalize_student_negotiation_with_faculty
    :parameters (?student_enrollment - student_enrollment ?schedule_slot_type_a - schedule_slot_type_a ?faculty_representative - faculty_representative)
    :precondition
      (and
        (case_validated ?student_enrollment)
        (faculty_assigned_to_case ?student_enrollment ?faculty_representative)
        (enrollment_slot_type_a_reserved ?student_enrollment ?schedule_slot_type_a)
        (slot_a_confirmed ?schedule_slot_type_a)
        (not
          (student_negotiation_initiated ?student_enrollment)
        )
      )
    :effect
      (and
        (student_negotiation_initiated ?student_enrollment)
        (student_negotiation_confirmed ?student_enrollment)
      )
  )
  (:action submit_supporting_evidence_student
    :parameters (?student_enrollment - student_enrollment ?schedule_slot_type_a - schedule_slot_type_a ?supporting_evidence - supporting_evidence)
    :precondition
      (and
        (case_validated ?student_enrollment)
        (enrollment_slot_type_a_reserved ?student_enrollment ?schedule_slot_type_a)
        (supporting_evidence_available ?supporting_evidence)
        (not
          (student_negotiation_initiated ?student_enrollment)
        )
      )
    :effect
      (and
        (slot_a_tentatively_reserved ?schedule_slot_type_a)
        (student_negotiation_initiated ?student_enrollment)
        (student_evidence_attached ?student_enrollment ?supporting_evidence)
        (not
          (supporting_evidence_available ?supporting_evidence)
        )
      )
  )
  (:action progress_student_negotiation
    :parameters (?student_enrollment - student_enrollment ?schedule_slot_type_a - schedule_slot_type_a ?medical_document - medical_document ?supporting_evidence - supporting_evidence)
    :precondition
      (and
        (case_validated ?student_enrollment)
        (attached_medical_document ?student_enrollment ?medical_document)
        (enrollment_slot_type_a_reserved ?student_enrollment ?schedule_slot_type_a)
        (slot_a_tentatively_reserved ?schedule_slot_type_a)
        (student_evidence_attached ?student_enrollment ?supporting_evidence)
        (not
          (student_negotiation_confirmed ?student_enrollment)
        )
      )
    :effect
      (and
        (slot_a_confirmed ?schedule_slot_type_a)
        (student_negotiation_confirmed ?student_enrollment)
        (supporting_evidence_available ?supporting_evidence)
        (not
          (student_evidence_attached ?student_enrollment ?supporting_evidence)
        )
      )
  )
  (:action confirm_module_slot_b
    :parameters (?module_enrollment - module_enrollment ?schedule_slot_type_b - schedule_slot_type_b ?medical_document - medical_document)
    :precondition
      (and
        (case_validated ?module_enrollment)
        (attached_medical_document ?module_enrollment ?medical_document)
        (enrollment_slot_type_b_reserved ?module_enrollment ?schedule_slot_type_b)
        (not
          (slot_b_confirmed ?schedule_slot_type_b)
        )
        (not
          (slot_b_tentatively_reserved ?schedule_slot_type_b)
        )
      )
    :effect (slot_b_confirmed ?schedule_slot_type_b)
  )
  (:action finalize_module_negotiation_with_faculty
    :parameters (?module_enrollment - module_enrollment ?schedule_slot_type_b - schedule_slot_type_b ?faculty_representative - faculty_representative)
    :precondition
      (and
        (case_validated ?module_enrollment)
        (faculty_assigned_to_case ?module_enrollment ?faculty_representative)
        (enrollment_slot_type_b_reserved ?module_enrollment ?schedule_slot_type_b)
        (slot_b_confirmed ?schedule_slot_type_b)
        (not
          (module_negotiation_initiated ?module_enrollment)
        )
      )
    :effect
      (and
        (module_negotiation_initiated ?module_enrollment)
        (module_negotiation_confirmed ?module_enrollment)
      )
  )
  (:action submit_supporting_evidence_module
    :parameters (?module_enrollment - module_enrollment ?schedule_slot_type_b - schedule_slot_type_b ?supporting_evidence - supporting_evidence)
    :precondition
      (and
        (case_validated ?module_enrollment)
        (enrollment_slot_type_b_reserved ?module_enrollment ?schedule_slot_type_b)
        (supporting_evidence_available ?supporting_evidence)
        (not
          (module_negotiation_initiated ?module_enrollment)
        )
      )
    :effect
      (and
        (slot_b_tentatively_reserved ?schedule_slot_type_b)
        (module_negotiation_initiated ?module_enrollment)
        (module_evidence_attached ?module_enrollment ?supporting_evidence)
        (not
          (supporting_evidence_available ?supporting_evidence)
        )
      )
  )
  (:action progress_module_negotiation
    :parameters (?module_enrollment - module_enrollment ?schedule_slot_type_b - schedule_slot_type_b ?medical_document - medical_document ?supporting_evidence - supporting_evidence)
    :precondition
      (and
        (case_validated ?module_enrollment)
        (attached_medical_document ?module_enrollment ?medical_document)
        (enrollment_slot_type_b_reserved ?module_enrollment ?schedule_slot_type_b)
        (slot_b_tentatively_reserved ?schedule_slot_type_b)
        (module_evidence_attached ?module_enrollment ?supporting_evidence)
        (not
          (module_negotiation_confirmed ?module_enrollment)
        )
      )
    :effect
      (and
        (slot_b_confirmed ?schedule_slot_type_b)
        (module_negotiation_confirmed ?module_enrollment)
        (supporting_evidence_available ?supporting_evidence)
        (not
          (module_evidence_attached ?module_enrollment ?supporting_evidence)
        )
      )
  )
  (:action synthesize_proposal_standard
    :parameters (?student_enrollment - student_enrollment ?module_enrollment - module_enrollment ?schedule_slot_type_a - schedule_slot_type_a ?schedule_slot_type_b - schedule_slot_type_b ?replan_proposal - replan_proposal)
    :precondition
      (and
        (student_negotiation_initiated ?student_enrollment)
        (module_negotiation_initiated ?module_enrollment)
        (enrollment_slot_type_a_reserved ?student_enrollment ?schedule_slot_type_a)
        (enrollment_slot_type_b_reserved ?module_enrollment ?schedule_slot_type_b)
        (slot_a_confirmed ?schedule_slot_type_a)
        (slot_b_confirmed ?schedule_slot_type_b)
        (student_negotiation_confirmed ?student_enrollment)
        (module_negotiation_confirmed ?module_enrollment)
        (proposal_placeholder_available ?replan_proposal)
      )
    :effect
      (and
        (proposal_reserved ?replan_proposal)
        (proposal_includes_slot_a ?replan_proposal ?schedule_slot_type_a)
        (proposal_includes_slot_b ?replan_proposal ?schedule_slot_type_b)
        (not
          (proposal_placeholder_available ?replan_proposal)
        )
      )
  )
  (:action synthesize_proposal_with_option_a
    :parameters (?student_enrollment - student_enrollment ?module_enrollment - module_enrollment ?schedule_slot_type_a - schedule_slot_type_a ?schedule_slot_type_b - schedule_slot_type_b ?replan_proposal - replan_proposal)
    :precondition
      (and
        (student_negotiation_initiated ?student_enrollment)
        (module_negotiation_initiated ?module_enrollment)
        (enrollment_slot_type_a_reserved ?student_enrollment ?schedule_slot_type_a)
        (enrollment_slot_type_b_reserved ?module_enrollment ?schedule_slot_type_b)
        (slot_a_tentatively_reserved ?schedule_slot_type_a)
        (slot_b_confirmed ?schedule_slot_type_b)
        (not
          (student_negotiation_confirmed ?student_enrollment)
        )
        (module_negotiation_confirmed ?module_enrollment)
        (proposal_placeholder_available ?replan_proposal)
      )
    :effect
      (and
        (proposal_reserved ?replan_proposal)
        (proposal_includes_slot_a ?replan_proposal ?schedule_slot_type_a)
        (proposal_includes_slot_b ?replan_proposal ?schedule_slot_type_b)
        (proposal_option_a_locked ?replan_proposal)
        (not
          (proposal_placeholder_available ?replan_proposal)
        )
      )
  )
  (:action synthesize_proposal_with_option_b
    :parameters (?student_enrollment - student_enrollment ?module_enrollment - module_enrollment ?schedule_slot_type_a - schedule_slot_type_a ?schedule_slot_type_b - schedule_slot_type_b ?replan_proposal - replan_proposal)
    :precondition
      (and
        (student_negotiation_initiated ?student_enrollment)
        (module_negotiation_initiated ?module_enrollment)
        (enrollment_slot_type_a_reserved ?student_enrollment ?schedule_slot_type_a)
        (enrollment_slot_type_b_reserved ?module_enrollment ?schedule_slot_type_b)
        (slot_a_confirmed ?schedule_slot_type_a)
        (slot_b_tentatively_reserved ?schedule_slot_type_b)
        (student_negotiation_confirmed ?student_enrollment)
        (not
          (module_negotiation_confirmed ?module_enrollment)
        )
        (proposal_placeholder_available ?replan_proposal)
      )
    :effect
      (and
        (proposal_reserved ?replan_proposal)
        (proposal_includes_slot_a ?replan_proposal ?schedule_slot_type_a)
        (proposal_includes_slot_b ?replan_proposal ?schedule_slot_type_b)
        (proposal_option_b_locked ?replan_proposal)
        (not
          (proposal_placeholder_available ?replan_proposal)
        )
      )
  )
  (:action synthesize_proposal_with_options_ab
    :parameters (?student_enrollment - student_enrollment ?module_enrollment - module_enrollment ?schedule_slot_type_a - schedule_slot_type_a ?schedule_slot_type_b - schedule_slot_type_b ?replan_proposal - replan_proposal)
    :precondition
      (and
        (student_negotiation_initiated ?student_enrollment)
        (module_negotiation_initiated ?module_enrollment)
        (enrollment_slot_type_a_reserved ?student_enrollment ?schedule_slot_type_a)
        (enrollment_slot_type_b_reserved ?module_enrollment ?schedule_slot_type_b)
        (slot_a_tentatively_reserved ?schedule_slot_type_a)
        (slot_b_tentatively_reserved ?schedule_slot_type_b)
        (not
          (student_negotiation_confirmed ?student_enrollment)
        )
        (not
          (module_negotiation_confirmed ?module_enrollment)
        )
        (proposal_placeholder_available ?replan_proposal)
      )
    :effect
      (and
        (proposal_reserved ?replan_proposal)
        (proposal_includes_slot_a ?replan_proposal ?schedule_slot_type_a)
        (proposal_includes_slot_b ?replan_proposal ?schedule_slot_type_b)
        (proposal_option_a_locked ?replan_proposal)
        (proposal_option_b_locked ?replan_proposal)
        (not
          (proposal_placeholder_available ?replan_proposal)
        )
      )
  )
  (:action finalize_proposal
    :parameters (?replan_proposal - replan_proposal ?student_enrollment - student_enrollment ?medical_document - medical_document)
    :precondition
      (and
        (proposal_reserved ?replan_proposal)
        (student_negotiation_initiated ?student_enrollment)
        (attached_medical_document ?student_enrollment ?medical_document)
        (not
          (proposal_finalized ?replan_proposal)
        )
      )
    :effect (proposal_finalized ?replan_proposal)
  )
  (:action attach_approval_document_to_proposal
    :parameters (?administrative_case_record - administrative_case_record ?approval_document - approval_document ?replan_proposal - replan_proposal)
    :precondition
      (and
        (case_validated ?administrative_case_record)
        (case_linked_to_proposal ?administrative_case_record ?replan_proposal)
        (case_has_approval_document ?administrative_case_record ?approval_document)
        (approval_document_available ?approval_document)
        (proposal_reserved ?replan_proposal)
        (proposal_finalized ?replan_proposal)
        (not
          (approval_document_consumed ?approval_document)
        )
      )
    :effect
      (and
        (approval_document_consumed ?approval_document)
        (approval_document_linked_to_proposal ?approval_document ?replan_proposal)
        (not
          (approval_document_available ?approval_document)
        )
      )
  )
  (:action run_compliance_stage_one_check
    :parameters (?administrative_case_record - administrative_case_record ?approval_document - approval_document ?replan_proposal - replan_proposal ?medical_document - medical_document)
    :precondition
      (and
        (case_validated ?administrative_case_record)
        (case_has_approval_document ?administrative_case_record ?approval_document)
        (approval_document_consumed ?approval_document)
        (approval_document_linked_to_proposal ?approval_document ?replan_proposal)
        (attached_medical_document ?administrative_case_record ?medical_document)
        (not
          (proposal_option_a_locked ?replan_proposal)
        )
        (not
          (compliance_stage_one_passed ?administrative_case_record)
        )
      )
    :effect (compliance_stage_one_passed ?administrative_case_record)
  )
  (:action assign_policy_reference_to_case
    :parameters (?administrative_case_record - administrative_case_record ?policy_reference - policy_reference)
    :precondition
      (and
        (case_validated ?administrative_case_record)
