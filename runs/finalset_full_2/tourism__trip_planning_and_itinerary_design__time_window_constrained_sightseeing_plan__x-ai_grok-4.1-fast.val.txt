--- domain.pddl (as passed to VAL) ---
(define (domain time_window_constrained_sightseeing_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity - object infrastructure_asset - entity resource_family - entity scheduling_family - entity itinerary_component_supertype - entity itinerary_item - itinerary_component_supertype transport_option - infrastructure_asset activity - infrastructure_asset onsite_service_slot - infrastructure_asset premium_option - infrastructure_asset auxiliary_service - infrastructure_asset temporal_marker - infrastructure_asset preference_token - infrastructure_asset location_attribute - infrastructure_asset consumable_resource - resource_family resource_option - resource_family constraint_token - resource_family time_window_morning_type - scheduling_family time_window_afternoon_type - scheduling_family visit_candidate - scheduling_family day_slot_supertype - itinerary_item itinerary_supertype - itinerary_item day_slot_morning - day_slot_supertype day_slot_afternoon - day_slot_supertype itinerary_instance - itinerary_supertype)

  (:predicates
    (itinerary_item_selected ?itinerary_component - itinerary_item)
    (itinerary_item_activity_confirmed ?itinerary_component - itinerary_item)
    (itinerary_item_transport_reserved ?itinerary_component - itinerary_item)
    (itinerary_item_finalized ?itinerary_component - itinerary_item)
    (itinerary_item_validated ?itinerary_component - itinerary_item)
    (itinerary_item_time_anchored ?itinerary_component - itinerary_item)
    (transport_option_available ?transport_option - transport_option)
    (itinerary_item_assigned_transport ?itinerary_component - itinerary_item ?transport_option - transport_option)
    (activity_available ?activity - activity)
    (activity_assigned_to_itinerary_item ?itinerary_component - itinerary_item ?activity - activity)
    (service_slot_available ?onsite_service_slot - onsite_service_slot)
    (service_slot_assigned_to_itinerary_item ?itinerary_component - itinerary_item ?onsite_service_slot - onsite_service_slot)
    (consumable_available ?consumable_resource - consumable_resource)
    (day_slot_morning_consumable_assigned ?day_slot_a - day_slot_morning ?consumable_resource - consumable_resource)
    (day_slot_afternoon_consumable_assigned ?day_slot_b - day_slot_afternoon ?consumable_resource - consumable_resource)
    (day_slot_morning_time_window_link ?day_slot_a - day_slot_morning ?time_window_a - time_window_morning_type)
    (time_window_morning_booked ?time_window_a - time_window_morning_type)
    (time_window_morning_pending ?time_window_a - time_window_morning_type)
    (day_slot_morning_completed ?day_slot_a - day_slot_morning)
    (day_slot_afternoon_time_window_link ?day_slot_b - day_slot_afternoon ?time_window_b - time_window_afternoon_type)
    (time_window_afternoon_booked ?time_window_b - time_window_afternoon_type)
    (time_window_afternoon_pending ?time_window_b - time_window_afternoon_type)
    (day_slot_afternoon_completed ?day_slot_b - day_slot_afternoon)
    (visit_candidate_available ?visit_candidate - visit_candidate)
    (visit_candidate_reserved ?visit_candidate - visit_candidate)
    (visit_candidate_has_time_window_morning ?visit_candidate - visit_candidate ?time_window_a - time_window_morning_type)
    (visit_candidate_has_time_window_afternoon ?visit_candidate - visit_candidate ?time_window_b - time_window_afternoon_type)
    (visit_candidate_flag_option_a ?visit_candidate - visit_candidate)
    (visit_candidate_flag_option_b ?visit_candidate - visit_candidate)
    (visit_candidate_timing_confirmed ?visit_candidate - visit_candidate)
    (itinerary_includes_day_slot_morning ?itinerary_instance - itinerary_instance ?day_slot_a - day_slot_morning)
    (itinerary_includes_day_slot_afternoon ?itinerary_instance - itinerary_instance ?day_slot_b - day_slot_afternoon)
    (itinerary_includes_visit_candidate ?itinerary_instance - itinerary_instance ?visit_candidate - visit_candidate)
    (resource_option_available ?resource_option - resource_option)
    (itinerary_has_resource_option ?itinerary_instance - itinerary_instance ?resource_option - resource_option)
    (resource_option_locked ?resource_option - resource_option)
    (resource_option_attached_to_visit_candidate ?resource_option - resource_option ?visit_candidate - visit_candidate)
    (itinerary_option_applied ?itinerary_instance - itinerary_instance)
    (itinerary_option_stage2_applied ?itinerary_instance - itinerary_instance)
    (itinerary_option_stage3_applied ?itinerary_instance - itinerary_instance)
    (premium_option_selected ?itinerary_instance - itinerary_instance)
    (itinerary_option_augmented ?itinerary_instance - itinerary_instance)
    (itinerary_auxiliary_flag ?itinerary_instance - itinerary_instance)
    (itinerary_ready_for_finalization ?itinerary_instance - itinerary_instance)
    (constraint_token_available ?constraint_token - constraint_token)
    (itinerary_has_constraint_token ?itinerary_instance - itinerary_instance ?constraint_token - constraint_token)
    (constraint_applied_to_itinerary ?itinerary_instance - itinerary_instance)
    (itinerary_constraint_service_prepared ?itinerary_instance - itinerary_instance)
    (itinerary_constraint_service_confirmed ?itinerary_instance - itinerary_instance)
    (premium_option_available ?premium_option - premium_option)
    (itinerary_has_premium_option ?itinerary_instance - itinerary_instance ?premium_option - premium_option)
    (auxiliary_service_available ?auxiliary_service - auxiliary_service)
    (itinerary_has_auxiliary_service ?itinerary_instance - itinerary_instance ?auxiliary_service - auxiliary_service)
    (preference_token_available ?preference_token - preference_token)
    (itinerary_has_preference_token ?itinerary_instance - itinerary_instance ?preference_token - preference_token)
    (location_attribute_available ?location_attribute - location_attribute)
    (itinerary_has_location_attribute ?itinerary_instance - itinerary_instance ?location_attribute - location_attribute)
    (temporal_marker_available ?temporal_marker - temporal_marker)
    (itinerary_item_has_temporal_marker ?itinerary_component - itinerary_item ?temporal_marker - temporal_marker)
    (day_slot_morning_ready ?day_slot_a - day_slot_morning)
    (day_slot_afternoon_ready ?day_slot_b - day_slot_afternoon)
    (itinerary_finalized ?itinerary_instance - itinerary_instance)
  )
  (:action select_itinerary_component
    :parameters (?itinerary_component - itinerary_item)
    :precondition
      (and
        (not
          (itinerary_item_selected ?itinerary_component)
        )
        (not
          (itinerary_item_finalized ?itinerary_component)
        )
      )
    :effect (itinerary_item_selected ?itinerary_component)
  )
  (:action assign_transport_to_component
    :parameters (?itinerary_component - itinerary_item ?transport_option - transport_option)
    :precondition
      (and
        (itinerary_item_selected ?itinerary_component)
        (not
          (itinerary_item_transport_reserved ?itinerary_component)
        )
        (transport_option_available ?transport_option)
      )
    :effect
      (and
        (itinerary_item_transport_reserved ?itinerary_component)
        (itinerary_item_assigned_transport ?itinerary_component ?transport_option)
        (not
          (transport_option_available ?transport_option)
        )
      )
  )
  (:action propose_activity_for_component
    :parameters (?itinerary_component - itinerary_item ?activity - activity)
    :precondition
      (and
        (itinerary_item_selected ?itinerary_component)
        (itinerary_item_transport_reserved ?itinerary_component)
        (activity_available ?activity)
      )
    :effect
      (and
        (activity_assigned_to_itinerary_item ?itinerary_component ?activity)
        (not
          (activity_available ?activity)
        )
      )
  )
  (:action confirm_activity_for_component
    :parameters (?itinerary_component - itinerary_item ?activity - activity)
    :precondition
      (and
        (itinerary_item_selected ?itinerary_component)
        (itinerary_item_transport_reserved ?itinerary_component)
        (activity_assigned_to_itinerary_item ?itinerary_component ?activity)
        (not
          (itinerary_item_activity_confirmed ?itinerary_component)
        )
      )
    :effect (itinerary_item_activity_confirmed ?itinerary_component)
  )
  (:action rollback_activity_assignment
    :parameters (?itinerary_component - itinerary_item ?activity - activity)
    :precondition
      (and
        (activity_assigned_to_itinerary_item ?itinerary_component ?activity)
      )
    :effect
      (and
        (activity_available ?activity)
        (not
          (activity_assigned_to_itinerary_item ?itinerary_component ?activity)
        )
      )
  )
  (:action book_onsite_service_for_component
    :parameters (?itinerary_component - itinerary_item ?onsite_service_slot - onsite_service_slot)
    :precondition
      (and
        (itinerary_item_activity_confirmed ?itinerary_component)
        (service_slot_available ?onsite_service_slot)
      )
    :effect
      (and
        (service_slot_assigned_to_itinerary_item ?itinerary_component ?onsite_service_slot)
        (not
          (service_slot_available ?onsite_service_slot)
        )
      )
  )
  (:action release_onsite_service_from_component
    :parameters (?itinerary_component - itinerary_item ?onsite_service_slot - onsite_service_slot)
    :precondition
      (and
        (service_slot_assigned_to_itinerary_item ?itinerary_component ?onsite_service_slot)
      )
    :effect
      (and
        (service_slot_available ?onsite_service_slot)
        (not
          (service_slot_assigned_to_itinerary_item ?itinerary_component ?onsite_service_slot)
        )
      )
  )
  (:action attach_preference_token_to_itinerary
    :parameters (?itinerary_instance - itinerary_instance ?preference_token - preference_token)
    :precondition
      (and
        (itinerary_item_activity_confirmed ?itinerary_instance)
        (preference_token_available ?preference_token)
      )
    :effect
      (and
        (itinerary_has_preference_token ?itinerary_instance ?preference_token)
        (not
          (preference_token_available ?preference_token)
        )
      )
  )
  (:action detach_preference_token_from_itinerary
    :parameters (?itinerary_instance - itinerary_instance ?preference_token - preference_token)
    :precondition
      (and
        (itinerary_has_preference_token ?itinerary_instance ?preference_token)
      )
    :effect
      (and
        (preference_token_available ?preference_token)
        (not
          (itinerary_has_preference_token ?itinerary_instance ?preference_token)
        )
      )
  )
  (:action attach_location_attribute_to_itinerary
    :parameters (?itinerary_instance - itinerary_instance ?location_attribute - location_attribute)
    :precondition
      (and
        (itinerary_item_activity_confirmed ?itinerary_instance)
        (location_attribute_available ?location_attribute)
      )
    :effect
      (and
        (itinerary_has_location_attribute ?itinerary_instance ?location_attribute)
        (not
          (location_attribute_available ?location_attribute)
        )
      )
  )
  (:action detach_location_attribute_from_itinerary
    :parameters (?itinerary_instance - itinerary_instance ?location_attribute - location_attribute)
    :precondition
      (and
        (itinerary_has_location_attribute ?itinerary_instance ?location_attribute)
      )
    :effect
      (and
        (location_attribute_available ?location_attribute)
        (not
          (itinerary_has_location_attribute ?itinerary_instance ?location_attribute)
        )
      )
  )
  (:action claim_time_window_for_day_slot_morning
    :parameters (?day_slot_a - day_slot_morning ?time_window_a - time_window_morning_type ?activity - activity)
    :precondition
      (and
        (itinerary_item_activity_confirmed ?day_slot_a)
        (activity_assigned_to_itinerary_item ?day_slot_a ?activity)
        (day_slot_morning_time_window_link ?day_slot_a ?time_window_a)
        (not
          (time_window_morning_booked ?time_window_a)
        )
        (not
          (time_window_morning_pending ?time_window_a)
        )
      )
    :effect (time_window_morning_booked ?time_window_a)
  )
  (:action confirm_day_slot_morning_service
    :parameters (?day_slot_a - day_slot_morning ?time_window_a - time_window_morning_type ?onsite_service_slot - onsite_service_slot)
    :precondition
      (and
        (itinerary_item_activity_confirmed ?day_slot_a)
        (service_slot_assigned_to_itinerary_item ?day_slot_a ?onsite_service_slot)
        (day_slot_morning_time_window_link ?day_slot_a ?time_window_a)
        (time_window_morning_booked ?time_window_a)
        (not
          (day_slot_morning_ready ?day_slot_a)
        )
      )
    :effect
      (and
        (day_slot_morning_ready ?day_slot_a)
        (day_slot_morning_completed ?day_slot_a)
      )
  )
  (:action attach_consumable_to_day_slot_morning
    :parameters (?day_slot_a - day_slot_morning ?time_window_a - time_window_morning_type ?consumable_resource - consumable_resource)
    :precondition
      (and
        (itinerary_item_activity_confirmed ?day_slot_a)
        (day_slot_morning_time_window_link ?day_slot_a ?time_window_a)
        (consumable_available ?consumable_resource)
        (not
          (day_slot_morning_ready ?day_slot_a)
        )
      )
    :effect
      (and
        (time_window_morning_pending ?time_window_a)
        (day_slot_morning_ready ?day_slot_a)
        (day_slot_morning_consumable_assigned ?day_slot_a ?consumable_resource)
        (not
          (consumable_available ?consumable_resource)
        )
      )
  )
  (:action finalize_day_slot_morning_consumable
    :parameters (?day_slot_a - day_slot_morning ?time_window_a - time_window_morning_type ?activity - activity ?consumable_resource - consumable_resource)
    :precondition
      (and
        (itinerary_item_activity_confirmed ?day_slot_a)
        (activity_assigned_to_itinerary_item ?day_slot_a ?activity)
        (day_slot_morning_time_window_link ?day_slot_a ?time_window_a)
        (time_window_morning_pending ?time_window_a)
        (day_slot_morning_consumable_assigned ?day_slot_a ?consumable_resource)
        (not
          (day_slot_morning_completed ?day_slot_a)
        )
      )
    :effect
      (and
        (time_window_morning_booked ?time_window_a)
        (day_slot_morning_completed ?day_slot_a)
        (consumable_available ?consumable_resource)
        (not
          (day_slot_morning_consumable_assigned ?day_slot_a ?consumable_resource)
        )
      )
  )
  (:action claim_time_window_for_day_slot_afternoon
    :parameters (?day_slot_b - day_slot_afternoon ?time_window_b - time_window_afternoon_type ?activity - activity)
    :precondition
      (and
        (itinerary_item_activity_confirmed ?day_slot_b)
        (activity_assigned_to_itinerary_item ?day_slot_b ?activity)
        (day_slot_afternoon_time_window_link ?day_slot_b ?time_window_b)
        (not
          (time_window_afternoon_booked ?time_window_b)
        )
        (not
          (time_window_afternoon_pending ?time_window_b)
        )
      )
    :effect (time_window_afternoon_booked ?time_window_b)
  )
  (:action confirm_day_slot_afternoon_service
    :parameters (?day_slot_b - day_slot_afternoon ?time_window_b - time_window_afternoon_type ?onsite_service_slot - onsite_service_slot)
    :precondition
      (and
        (itinerary_item_activity_confirmed ?day_slot_b)
        (service_slot_assigned_to_itinerary_item ?day_slot_b ?onsite_service_slot)
        (day_slot_afternoon_time_window_link ?day_slot_b ?time_window_b)
        (time_window_afternoon_booked ?time_window_b)
        (not
          (day_slot_afternoon_ready ?day_slot_b)
        )
      )
    :effect
      (and
        (day_slot_afternoon_ready ?day_slot_b)
        (day_slot_afternoon_completed ?day_slot_b)
      )
  )
  (:action attach_consumable_to_day_slot_afternoon
    :parameters (?day_slot_b - day_slot_afternoon ?time_window_b - time_window_afternoon_type ?consumable_resource - consumable_resource)
    :precondition
      (and
        (itinerary_item_activity_confirmed ?day_slot_b)
        (day_slot_afternoon_time_window_link ?day_slot_b ?time_window_b)
        (consumable_available ?consumable_resource)
        (not
          (day_slot_afternoon_ready ?day_slot_b)
        )
      )
    :effect
      (and
        (time_window_afternoon_pending ?time_window_b)
        (day_slot_afternoon_ready ?day_slot_b)
        (day_slot_afternoon_consumable_assigned ?day_slot_b ?consumable_resource)
        (not
          (consumable_available ?consumable_resource)
        )
      )
  )
  (:action finalize_day_slot_afternoon_consumable
    :parameters (?day_slot_b - day_slot_afternoon ?time_window_b - time_window_afternoon_type ?activity - activity ?consumable_resource - consumable_resource)
    :precondition
      (and
        (itinerary_item_activity_confirmed ?day_slot_b)
        (activity_assigned_to_itinerary_item ?day_slot_b ?activity)
        (day_slot_afternoon_time_window_link ?day_slot_b ?time_window_b)
        (time_window_afternoon_pending ?time_window_b)
        (day_slot_afternoon_consumable_assigned ?day_slot_b ?consumable_resource)
        (not
          (day_slot_afternoon_completed ?day_slot_b)
        )
      )
    :effect
      (and
        (time_window_afternoon_booked ?time_window_b)
        (day_slot_afternoon_completed ?day_slot_b)
        (consumable_available ?consumable_resource)
        (not
          (day_slot_afternoon_consumable_assigned ?day_slot_b ?consumable_resource)
        )
      )
  )
  (:action compose_visit_candidate
    :parameters (?day_slot_a - day_slot_morning ?day_slot_b - day_slot_afternoon ?time_window_a - time_window_morning_type ?time_window_b - time_window_afternoon_type ?visit_candidate - visit_candidate)
    :precondition
      (and
        (day_slot_morning_ready ?day_slot_a)
        (day_slot_afternoon_ready ?day_slot_b)
        (day_slot_morning_time_window_link ?day_slot_a ?time_window_a)
        (day_slot_afternoon_time_window_link ?day_slot_b ?time_window_b)
        (time_window_morning_booked ?time_window_a)
        (time_window_afternoon_booked ?time_window_b)
        (day_slot_morning_completed ?day_slot_a)
        (day_slot_afternoon_completed ?day_slot_b)
        (visit_candidate_available ?visit_candidate)
      )
    :effect
      (and
        (visit_candidate_reserved ?visit_candidate)
        (visit_candidate_has_time_window_morning ?visit_candidate ?time_window_a)
        (visit_candidate_has_time_window_afternoon ?visit_candidate ?time_window_b)
        (not
          (visit_candidate_available ?visit_candidate)
        )
      )
  )
  (:action compose_visit_candidate_with_window_morning_pending
    :parameters (?day_slot_a - day_slot_morning ?day_slot_b - day_slot_afternoon ?time_window_a - time_window_morning_type ?time_window_b - time_window_afternoon_type ?visit_candidate - visit_candidate)
    :precondition
      (and
        (day_slot_morning_ready ?day_slot_a)
        (day_slot_afternoon_ready ?day_slot_b)
        (day_slot_morning_time_window_link ?day_slot_a ?time_window_a)
        (day_slot_afternoon_time_window_link ?day_slot_b ?time_window_b)
        (time_window_morning_pending ?time_window_a)
        (time_window_afternoon_booked ?time_window_b)
        (not
          (day_slot_morning_completed ?day_slot_a)
        )
        (day_slot_afternoon_completed ?day_slot_b)
        (visit_candidate_available ?visit_candidate)
      )
    :effect
      (and
        (visit_candidate_reserved ?visit_candidate)
        (visit_candidate_has_time_window_morning ?visit_candidate ?time_window_a)
        (visit_candidate_has_time_window_afternoon ?visit_candidate ?time_window_b)
        (visit_candidate_flag_option_a ?visit_candidate)
        (not
          (visit_candidate_available ?visit_candidate)
        )
      )
  )
  (:action compose_visit_candidate_with_window_afternoon_pending
    :parameters (?day_slot_a - day_slot_morning ?day_slot_b - day_slot_afternoon ?time_window_a - time_window_morning_type ?time_window_b - time_window_afternoon_type ?visit_candidate - visit_candidate)
    :precondition
      (and
        (day_slot_morning_ready ?day_slot_a)
        (day_slot_afternoon_ready ?day_slot_b)
        (day_slot_morning_time_window_link ?day_slot_a ?time_window_a)
        (day_slot_afternoon_time_window_link ?day_slot_b ?time_window_b)
        (time_window_morning_booked ?time_window_a)
        (time_window_afternoon_pending ?time_window_b)
        (day_slot_morning_completed ?day_slot_a)
        (not
          (day_slot_afternoon_completed ?day_slot_b)
        )
        (visit_candidate_available ?visit_candidate)
      )
    :effect
      (and
        (visit_candidate_reserved ?visit_candidate)
        (visit_candidate_has_time_window_morning ?visit_candidate ?time_window_a)
        (visit_candidate_has_time_window_afternoon ?visit_candidate ?time_window_b)
        (visit_candidate_flag_option_b ?visit_candidate)
        (not
          (visit_candidate_available ?visit_candidate)
        )
      )
  )
  (:action compose_visit_candidate_with_both_windows_pending
    :parameters (?day_slot_a - day_slot_morning ?day_slot_b - day_slot_afternoon ?time_window_a - time_window_morning_type ?time_window_b - time_window_afternoon_type ?visit_candidate - visit_candidate)
    :precondition
      (and
        (day_slot_morning_ready ?day_slot_a)
        (day_slot_afternoon_ready ?day_slot_b)
        (day_slot_morning_time_window_link ?day_slot_a ?time_window_a)
        (day_slot_afternoon_time_window_link ?day_slot_b ?time_window_b)
        (time_window_morning_pending ?time_window_a)
        (time_window_afternoon_pending ?time_window_b)
        (not
          (day_slot_morning_completed ?day_slot_a)
        )
        (not
          (day_slot_afternoon_completed ?day_slot_b)
        )
        (visit_candidate_available ?visit_candidate)
      )
    :effect
      (and
        (visit_candidate_reserved ?visit_candidate)
        (visit_candidate_has_time_window_morning ?visit_candidate ?time_window_a)
        (visit_candidate_has_time_window_afternoon ?visit_candidate ?time_window_b)
        (visit_candidate_flag_option_a ?visit_candidate)
        (visit_candidate_flag_option_b ?visit_candidate)
        (not
          (visit_candidate_available ?visit_candidate)
        )
      )
  )
  (:action confirm_visit_candidate_timing
    :parameters (?visit_candidate - visit_candidate ?day_slot_a - day_slot_morning ?activity - activity)
    :precondition
      (and
        (visit_candidate_reserved ?visit_candidate)
        (day_slot_morning_ready ?day_slot_a)
        (activity_assigned_to_itinerary_item ?day_slot_a ?activity)
        (not
          (visit_candidate_timing_confirmed ?visit_candidate)
        )
      )
    :effect (visit_candidate_timing_confirmed ?visit_candidate)
  )
  (:action allocate_resource_option_to_itinerary
    :parameters (?itinerary_instance - itinerary_instance ?resource_option - resource_option ?visit_candidate - visit_candidate)
    :precondition
      (and
        (itinerary_item_activity_confirmed ?itinerary_instance)
        (itinerary_includes_visit_candidate ?itinerary_instance ?visit_candidate)
        (itinerary_has_resource_option ?itinerary_instance ?resource_option)
        (resource_option_available ?resource_option)
        (visit_candidate_reserved ?visit_candidate)
        (visit_candidate_timing_confirmed ?visit_candidate)
        (not
          (resource_option_locked ?resource_option)
        )
      )
    :effect
      (and
        (resource_option_locked ?resource_option)
        (resource_option_attached_to_visit_candidate ?resource_option ?visit_candidate)
        (not
          (resource_option_available ?resource_option)
        )
      )
  )
  (:action confirm_resource_option_allocation
    :parameters (?itinerary_instance - itinerary_instance ?resource_option - resource_option ?visit_candidate - visit_candidate ?activity - activity)
