--- domain.pddl (as passed to VAL) ---
(define (domain pilot_batch_learning_integration_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types category_holder - object category_holder_alt - object category_holder_sub - object top_candidate_type - object formulation_candidate - top_candidate_type laboratory_resource - category_holder analytical_method_slot - category_holder process_reviewer - category_holder regulatory_input - category_holder acceptance_criterion - category_holder evidence_document - category_holder equipment_qualification - category_holder subject_matter_expert - category_holder formulation_component - category_holder_alt stability_protocol - category_holder_alt external_data_source - category_holder_alt manufacturing_condition_set - category_holder_sub scaleup_parameter_set - category_holder_sub batch_record - category_holder_sub candidate_branch_a - formulation_candidate candidate_branch_b - formulation_candidate pilot_batch_run_a - candidate_branch_a pilot_batch_run_b - candidate_branch_a development_workstream - candidate_branch_b)
  (:predicates
    (candidate_registered ?formulation_candidate - formulation_candidate)
    (entity_characterized ?formulation_candidate - formulation_candidate)
    (candidate_resource_allocated ?formulation_candidate - formulation_candidate)
    (scaleup_ready ?formulation_candidate - formulation_candidate)
    (integration_confirmed ?formulation_candidate - formulation_candidate)
    (entity_evidence_compiled ?formulation_candidate - formulation_candidate)
    (resource_available ?laboratory_resource - laboratory_resource)
    (entity_resource_assignment ?formulation_candidate - formulation_candidate ?laboratory_resource - laboratory_resource)
    (method_slot_available ?analytical_method_slot - analytical_method_slot)
    (entity_method_assignment ?formulation_candidate - formulation_candidate ?analytical_method_slot - analytical_method_slot)
    (reviewer_available ?process_reviewer - process_reviewer)
    (entity_reviewer_assignment ?formulation_candidate - formulation_candidate ?process_reviewer - process_reviewer)
    (component_available ?formulation_component - formulation_component)
    (batch_component_assignment ?pilot_batch_run_a - pilot_batch_run_a ?formulation_component - formulation_component)
    (batch_b_component_assignment ?pilot_batch_run_b - pilot_batch_run_b ?formulation_component - formulation_component)
    (batch_condition_assignment ?pilot_batch_run_a - pilot_batch_run_a ?manufacturing_condition_set - manufacturing_condition_set)
    (condition_tested ?manufacturing_condition_set - manufacturing_condition_set)
    (condition_component_assigned ?manufacturing_condition_set - manufacturing_condition_set)
    (batch_ready_flag ?pilot_batch_run_a - pilot_batch_run_a)
    (batch_scaleup_parameter_assignment ?pilot_batch_run_b - pilot_batch_run_b ?scaleup_parameter_set - scaleup_parameter_set)
    (scaleup_parameter_validated ?scaleup_parameter_set - scaleup_parameter_set)
    (scaleup_parameter_has_component ?scaleup_parameter_set - scaleup_parameter_set)
    (batch_b_ready_flag ?pilot_batch_run_b - pilot_batch_run_b)
    (batch_record_open ?batch_record - batch_record)
    (batch_record_assembled ?batch_record - batch_record)
    (batch_record_has_condition ?batch_record - batch_record ?manufacturing_condition_set - manufacturing_condition_set)
    (batch_record_has_scaleup_parameters ?batch_record - batch_record ?scaleup_parameter_set - scaleup_parameter_set)
    (batch_record_review_stage_a ?batch_record - batch_record)
    (batch_record_review_stage_b ?batch_record - batch_record)
    (batch_record_qc_ready ?batch_record - batch_record)
    (workstream_associated_batch_a ?development_workstream - development_workstream ?pilot_batch_run_a - pilot_batch_run_a)
    (workstream_associated_batch_b ?development_workstream - development_workstream ?pilot_batch_run_b - pilot_batch_run_b)
    (workstream_has_batch_record ?development_workstream - development_workstream ?batch_record - batch_record)
    (stability_protocol_available ?stability_protocol - stability_protocol)
    (workstream_stability_protocol_assigned ?development_workstream - development_workstream ?stability_protocol - stability_protocol)
    (stability_protocol_enrolled ?stability_protocol - stability_protocol)
    (stability_protocol_linked_to_batch_record ?stability_protocol - stability_protocol ?batch_record - batch_record)
    (workstream_stability_initiated ?development_workstream - development_workstream)
    (workstream_stability_trial_activated ?development_workstream - development_workstream)
    (workstream_qc_started ?development_workstream - development_workstream)
    (workstream_regulatory_input_assigned ?development_workstream - development_workstream)
    (workstream_regulatory_reviewed ?development_workstream - development_workstream)
    (workstream_data_integrated ?development_workstream - development_workstream)
    (workstream_ready_for_integration ?development_workstream - development_workstream)
    (external_data_available ?external_data_source - external_data_source)
    (workstream_has_external_data ?development_workstream - development_workstream ?external_data_source - external_data_source)
    (external_data_processed ?development_workstream - development_workstream)
    (workstream_review_started ?development_workstream - development_workstream)
    (workstream_review_completed ?development_workstream - development_workstream)
    (regulatory_input_available ?regulatory_input - regulatory_input)
    (workstream_regulatory_assignment ?development_workstream - development_workstream ?regulatory_input - regulatory_input)
    (acceptance_criterion_available ?acceptance_criterion - acceptance_criterion)
    (workstream_acceptance_criterion_assigned ?development_workstream - development_workstream ?acceptance_criterion - acceptance_criterion)
    (equipment_qualification_available ?equipment_qualification - equipment_qualification)
    (workstream_equipment_qualification_linked ?development_workstream - development_workstream ?equipment_qualification - equipment_qualification)
    (sme_available ?subject_matter_expert - subject_matter_expert)
    (workstream_has_sme_assigned ?development_workstream - development_workstream ?subject_matter_expert - subject_matter_expert)
    (evidence_document_available ?evidence_document - evidence_document)
    (entity_has_evidence ?formulation_candidate - formulation_candidate ?evidence_document - evidence_document)
    (batch_a_ready_for_assembly ?pilot_batch_run_a - pilot_batch_run_a)
    (batch_b_ready_for_assembly ?pilot_batch_run_b - pilot_batch_run_b)
    (workstream_integration_finalized ?development_workstream - development_workstream)
  )
  (:action register_formulation_candidate
    :parameters (?formulation_candidate - formulation_candidate)
    :precondition
      (and
        (not
          (candidate_registered ?formulation_candidate)
        )
        (not
          (scaleup_ready ?formulation_candidate)
        )
      )
    :effect (candidate_registered ?formulation_candidate)
  )
  (:action allocate_laboratory_resource
    :parameters (?formulation_candidate - formulation_candidate ?laboratory_resource - laboratory_resource)
    :precondition
      (and
        (candidate_registered ?formulation_candidate)
        (not
          (candidate_resource_allocated ?formulation_candidate)
        )
        (resource_available ?laboratory_resource)
      )
    :effect
      (and
        (candidate_resource_allocated ?formulation_candidate)
        (entity_resource_assignment ?formulation_candidate ?laboratory_resource)
        (not
          (resource_available ?laboratory_resource)
        )
      )
  )
  (:action reserve_analytical_method_slot
    :parameters (?formulation_candidate - formulation_candidate ?analytical_method_slot - analytical_method_slot)
    :precondition
      (and
        (candidate_registered ?formulation_candidate)
        (candidate_resource_allocated ?formulation_candidate)
        (method_slot_available ?analytical_method_slot)
      )
    :effect
      (and
        (entity_method_assignment ?formulation_candidate ?analytical_method_slot)
        (not
          (method_slot_available ?analytical_method_slot)
        )
      )
  )
  (:action confirm_candidate_characterization
    :parameters (?formulation_candidate - formulation_candidate ?analytical_method_slot - analytical_method_slot)
    :precondition
      (and
        (candidate_registered ?formulation_candidate)
        (candidate_resource_allocated ?formulation_candidate)
        (entity_method_assignment ?formulation_candidate ?analytical_method_slot)
        (not
          (entity_characterized ?formulation_candidate)
        )
      )
    :effect (entity_characterized ?formulation_candidate)
  )
  (:action release_analytical_method_slot
    :parameters (?formulation_candidate - formulation_candidate ?analytical_method_slot - analytical_method_slot)
    :precondition
      (and
        (entity_method_assignment ?formulation_candidate ?analytical_method_slot)
      )
    :effect
      (and
        (method_slot_available ?analytical_method_slot)
        (not
          (entity_method_assignment ?formulation_candidate ?analytical_method_slot)
        )
      )
  )
  (:action assign_process_reviewer
    :parameters (?formulation_candidate - formulation_candidate ?process_reviewer - process_reviewer)
    :precondition
      (and
        (entity_characterized ?formulation_candidate)
        (reviewer_available ?process_reviewer)
      )
    :effect
      (and
        (entity_reviewer_assignment ?formulation_candidate ?process_reviewer)
        (not
          (reviewer_available ?process_reviewer)
        )
      )
  )
  (:action release_process_reviewer
    :parameters (?formulation_candidate - formulation_candidate ?process_reviewer - process_reviewer)
    :precondition
      (and
        (entity_reviewer_assignment ?formulation_candidate ?process_reviewer)
      )
    :effect
      (and
        (reviewer_available ?process_reviewer)
        (not
          (entity_reviewer_assignment ?formulation_candidate ?process_reviewer)
        )
      )
  )
  (:action assign_equipment_qualification_to_workstream
    :parameters (?development_workstream - development_workstream ?equipment_qualification - equipment_qualification)
    :precondition
      (and
        (entity_characterized ?development_workstream)
        (equipment_qualification_available ?equipment_qualification)
      )
    :effect
      (and
        (workstream_equipment_qualification_linked ?development_workstream ?equipment_qualification)
        (not
          (equipment_qualification_available ?equipment_qualification)
        )
      )
  )
  (:action release_equipment_qualification_from_workstream
    :parameters (?development_workstream - development_workstream ?equipment_qualification - equipment_qualification)
    :precondition
      (and
        (workstream_equipment_qualification_linked ?development_workstream ?equipment_qualification)
      )
    :effect
      (and
        (equipment_qualification_available ?equipment_qualification)
        (not
          (workstream_equipment_qualification_linked ?development_workstream ?equipment_qualification)
        )
      )
  )
  (:action assign_subject_matter_expert_to_workstream
    :parameters (?development_workstream - development_workstream ?subject_matter_expert - subject_matter_expert)
    :precondition
      (and
        (entity_characterized ?development_workstream)
        (sme_available ?subject_matter_expert)
      )
    :effect
      (and
        (workstream_has_sme_assigned ?development_workstream ?subject_matter_expert)
        (not
          (sme_available ?subject_matter_expert)
        )
      )
  )
  (:action release_subject_matter_expert_from_workstream
    :parameters (?development_workstream - development_workstream ?subject_matter_expert - subject_matter_expert)
    :precondition
      (and
        (workstream_has_sme_assigned ?development_workstream ?subject_matter_expert)
      )
    :effect
      (and
        (sme_available ?subject_matter_expert)
        (not
          (workstream_has_sme_assigned ?development_workstream ?subject_matter_expert)
        )
      )
  )
  (:action execute_condition_test_on_batch
    :parameters (?pilot_batch_run_a - pilot_batch_run_a ?manufacturing_condition_set - manufacturing_condition_set ?analytical_method_slot - analytical_method_slot)
    :precondition
      (and
        (entity_characterized ?pilot_batch_run_a)
        (entity_method_assignment ?pilot_batch_run_a ?analytical_method_slot)
        (batch_condition_assignment ?pilot_batch_run_a ?manufacturing_condition_set)
        (not
          (condition_tested ?manufacturing_condition_set)
        )
        (not
          (condition_component_assigned ?manufacturing_condition_set)
        )
      )
    :effect (condition_tested ?manufacturing_condition_set)
  )
  (:action record_process_review_for_batch_a
    :parameters (?pilot_batch_run_a - pilot_batch_run_a ?manufacturing_condition_set - manufacturing_condition_set ?process_reviewer - process_reviewer)
    :precondition
      (and
        (entity_characterized ?pilot_batch_run_a)
        (entity_reviewer_assignment ?pilot_batch_run_a ?process_reviewer)
        (batch_condition_assignment ?pilot_batch_run_a ?manufacturing_condition_set)
        (condition_tested ?manufacturing_condition_set)
        (not
          (batch_a_ready_for_assembly ?pilot_batch_run_a)
        )
      )
    :effect
      (and
        (batch_a_ready_for_assembly ?pilot_batch_run_a)
        (batch_ready_flag ?pilot_batch_run_a)
      )
  )
  (:action attach_component_to_batch_condition
    :parameters (?pilot_batch_run_a - pilot_batch_run_a ?manufacturing_condition_set - manufacturing_condition_set ?formulation_component - formulation_component)
    :precondition
      (and
        (entity_characterized ?pilot_batch_run_a)
        (batch_condition_assignment ?pilot_batch_run_a ?manufacturing_condition_set)
        (component_available ?formulation_component)
        (not
          (batch_a_ready_for_assembly ?pilot_batch_run_a)
        )
      )
    :effect
      (and
        (condition_component_assigned ?manufacturing_condition_set)
        (batch_a_ready_for_assembly ?pilot_batch_run_a)
        (batch_component_assignment ?pilot_batch_run_a ?formulation_component)
        (not
          (component_available ?formulation_component)
        )
      )
  )
  (:action finalize_component_evaluation_on_batch
    :parameters (?pilot_batch_run_a - pilot_batch_run_a ?manufacturing_condition_set - manufacturing_condition_set ?analytical_method_slot - analytical_method_slot ?formulation_component - formulation_component)
    :precondition
      (and
        (entity_characterized ?pilot_batch_run_a)
        (entity_method_assignment ?pilot_batch_run_a ?analytical_method_slot)
        (batch_condition_assignment ?pilot_batch_run_a ?manufacturing_condition_set)
        (condition_component_assigned ?manufacturing_condition_set)
        (batch_component_assignment ?pilot_batch_run_a ?formulation_component)
        (not
          (batch_ready_flag ?pilot_batch_run_a)
        )
      )
    :effect
      (and
        (condition_tested ?manufacturing_condition_set)
        (batch_ready_flag ?pilot_batch_run_a)
        (component_available ?formulation_component)
        (not
          (batch_component_assignment ?pilot_batch_run_a ?formulation_component)
        )
      )
  )
  (:action execute_scaleup_parameter_test_for_batch_b
    :parameters (?pilot_batch_run_b - pilot_batch_run_b ?scaleup_parameter_set - scaleup_parameter_set ?analytical_method_slot - analytical_method_slot)
    :precondition
      (and
        (entity_characterized ?pilot_batch_run_b)
        (entity_method_assignment ?pilot_batch_run_b ?analytical_method_slot)
        (batch_scaleup_parameter_assignment ?pilot_batch_run_b ?scaleup_parameter_set)
        (not
          (scaleup_parameter_validated ?scaleup_parameter_set)
        )
        (not
          (scaleup_parameter_has_component ?scaleup_parameter_set)
        )
      )
    :effect (scaleup_parameter_validated ?scaleup_parameter_set)
  )
  (:action record_process_review_for_batch_b
    :parameters (?pilot_batch_run_b - pilot_batch_run_b ?scaleup_parameter_set - scaleup_parameter_set ?process_reviewer - process_reviewer)
    :precondition
      (and
        (entity_characterized ?pilot_batch_run_b)
        (entity_reviewer_assignment ?pilot_batch_run_b ?process_reviewer)
        (batch_scaleup_parameter_assignment ?pilot_batch_run_b ?scaleup_parameter_set)
        (scaleup_parameter_validated ?scaleup_parameter_set)
        (not
          (batch_b_ready_for_assembly ?pilot_batch_run_b)
        )
      )
    :effect
      (and
        (batch_b_ready_for_assembly ?pilot_batch_run_b)
        (batch_b_ready_flag ?pilot_batch_run_b)
      )
  )
  (:action attach_component_to_batch_condition_b
    :parameters (?pilot_batch_run_b - pilot_batch_run_b ?scaleup_parameter_set - scaleup_parameter_set ?formulation_component - formulation_component)
    :precondition
      (and
        (entity_characterized ?pilot_batch_run_b)
        (batch_scaleup_parameter_assignment ?pilot_batch_run_b ?scaleup_parameter_set)
        (component_available ?formulation_component)
        (not
          (batch_b_ready_for_assembly ?pilot_batch_run_b)
        )
      )
    :effect
      (and
        (scaleup_parameter_has_component ?scaleup_parameter_set)
        (batch_b_ready_for_assembly ?pilot_batch_run_b)
        (batch_b_component_assignment ?pilot_batch_run_b ?formulation_component)
        (not
          (component_available ?formulation_component)
        )
      )
  )
  (:action finalize_component_evaluation_on_batch_b
    :parameters (?pilot_batch_run_b - pilot_batch_run_b ?scaleup_parameter_set - scaleup_parameter_set ?analytical_method_slot - analytical_method_slot ?formulation_component - formulation_component)
    :precondition
      (and
        (entity_characterized ?pilot_batch_run_b)
        (entity_method_assignment ?pilot_batch_run_b ?analytical_method_slot)
        (batch_scaleup_parameter_assignment ?pilot_batch_run_b ?scaleup_parameter_set)
        (scaleup_parameter_has_component ?scaleup_parameter_set)
        (batch_b_component_assignment ?pilot_batch_run_b ?formulation_component)
        (not
          (batch_b_ready_flag ?pilot_batch_run_b)
        )
      )
    :effect
      (and
        (scaleup_parameter_validated ?scaleup_parameter_set)
        (batch_b_ready_flag ?pilot_batch_run_b)
        (component_available ?formulation_component)
        (not
          (batch_b_component_assignment ?pilot_batch_run_b ?formulation_component)
        )
      )
  )
  (:action assemble_batch_record_from_runs
    :parameters (?pilot_batch_run_a - pilot_batch_run_a ?pilot_batch_run_b - pilot_batch_run_b ?manufacturing_condition_set - manufacturing_condition_set ?scaleup_parameter_set - scaleup_parameter_set ?batch_record - batch_record)
    :precondition
      (and
        (batch_a_ready_for_assembly ?pilot_batch_run_a)
        (batch_b_ready_for_assembly ?pilot_batch_run_b)
        (batch_condition_assignment ?pilot_batch_run_a ?manufacturing_condition_set)
        (batch_scaleup_parameter_assignment ?pilot_batch_run_b ?scaleup_parameter_set)
        (condition_tested ?manufacturing_condition_set)
        (scaleup_parameter_validated ?scaleup_parameter_set)
        (batch_ready_flag ?pilot_batch_run_a)
        (batch_b_ready_flag ?pilot_batch_run_b)
        (batch_record_open ?batch_record)
      )
    :effect
      (and
        (batch_record_assembled ?batch_record)
        (batch_record_has_condition ?batch_record ?manufacturing_condition_set)
        (batch_record_has_scaleup_parameters ?batch_record ?scaleup_parameter_set)
        (not
          (batch_record_open ?batch_record)
        )
      )
  )
  (:action assemble_batch_record_with_qc_flag_a
    :parameters (?pilot_batch_run_a - pilot_batch_run_a ?pilot_batch_run_b - pilot_batch_run_b ?manufacturing_condition_set - manufacturing_condition_set ?scaleup_parameter_set - scaleup_parameter_set ?batch_record - batch_record)
    :precondition
      (and
        (batch_a_ready_for_assembly ?pilot_batch_run_a)
        (batch_b_ready_for_assembly ?pilot_batch_run_b)
        (batch_condition_assignment ?pilot_batch_run_a ?manufacturing_condition_set)
        (batch_scaleup_parameter_assignment ?pilot_batch_run_b ?scaleup_parameter_set)
        (condition_component_assigned ?manufacturing_condition_set)
        (scaleup_parameter_validated ?scaleup_parameter_set)
        (not
          (batch_ready_flag ?pilot_batch_run_a)
        )
        (batch_b_ready_flag ?pilot_batch_run_b)
        (batch_record_open ?batch_record)
      )
    :effect
      (and
        (batch_record_assembled ?batch_record)
        (batch_record_has_condition ?batch_record ?manufacturing_condition_set)
        (batch_record_has_scaleup_parameters ?batch_record ?scaleup_parameter_set)
        (batch_record_review_stage_a ?batch_record)
        (not
          (batch_record_open ?batch_record)
        )
      )
  )
  (:action assemble_batch_record_with_qc_flag_b
    :parameters (?pilot_batch_run_a - pilot_batch_run_a ?pilot_batch_run_b - pilot_batch_run_b ?manufacturing_condition_set - manufacturing_condition_set ?scaleup_parameter_set - scaleup_parameter_set ?batch_record - batch_record)
    :precondition
      (and
        (batch_a_ready_for_assembly ?pilot_batch_run_a)
        (batch_b_ready_for_assembly ?pilot_batch_run_b)
        (batch_condition_assignment ?pilot_batch_run_a ?manufacturing_condition_set)
        (batch_scaleup_parameter_assignment ?pilot_batch_run_b ?scaleup_parameter_set)
        (condition_tested ?manufacturing_condition_set)
        (scaleup_parameter_has_component ?scaleup_parameter_set)
        (batch_ready_flag ?pilot_batch_run_a)
        (not
          (batch_b_ready_flag ?pilot_batch_run_b)
        )
        (batch_record_open ?batch_record)
      )
    :effect
      (and
        (batch_record_assembled ?batch_record)
        (batch_record_has_condition ?batch_record ?manufacturing_condition_set)
        (batch_record_has_scaleup_parameters ?batch_record ?scaleup_parameter_set)
        (batch_record_review_stage_b ?batch_record)
        (not
          (batch_record_open ?batch_record)
        )
      )
  )
  (:action assemble_batch_record_with_qc_flags_a_b
    :parameters (?pilot_batch_run_a - pilot_batch_run_a ?pilot_batch_run_b - pilot_batch_run_b ?manufacturing_condition_set - manufacturing_condition_set ?scaleup_parameter_set - scaleup_parameter_set ?batch_record - batch_record)
    :precondition
      (and
        (batch_a_ready_for_assembly ?pilot_batch_run_a)
        (batch_b_ready_for_assembly ?pilot_batch_run_b)
        (batch_condition_assignment ?pilot_batch_run_a ?manufacturing_condition_set)
        (batch_scaleup_parameter_assignment ?pilot_batch_run_b ?scaleup_parameter_set)
        (condition_component_assigned ?manufacturing_condition_set)
        (scaleup_parameter_has_component ?scaleup_parameter_set)
        (not
          (batch_ready_flag ?pilot_batch_run_a)
        )
        (not
          (batch_b_ready_flag ?pilot_batch_run_b)
        )
        (batch_record_open ?batch_record)
      )
    :effect
      (and
        (batch_record_assembled ?batch_record)
        (batch_record_has_condition ?batch_record ?manufacturing_condition_set)
        (batch_record_has_scaleup_parameters ?batch_record ?scaleup_parameter_set)
        (batch_record_review_stage_a ?batch_record)
        (batch_record_review_stage_b ?batch_record)
        (not
          (batch_record_open ?batch_record)
        )
      )
  )
  (:action finalize_batch_record_qc
    :parameters (?batch_record - batch_record ?pilot_batch_run_a - pilot_batch_run_a ?analytical_method_slot - analytical_method_slot)
    :precondition
      (and
        (batch_record_assembled ?batch_record)
        (batch_a_ready_for_assembly ?pilot_batch_run_a)
        (entity_method_assignment ?pilot_batch_run_a ?analytical_method_slot)
        (not
          (batch_record_qc_ready ?batch_record)
        )
      )
    :effect (batch_record_qc_ready ?batch_record)
  )
  (:action plan_stability_protocol_for_workstream
    :parameters (?development_workstream - development_workstream ?stability_protocol - stability_protocol ?batch_record - batch_record)
    :precondition
      (and
        (entity_characterized ?development_workstream)
        (workstream_has_batch_record ?development_workstream ?batch_record)
        (workstream_stability_protocol_assigned ?development_workstream ?stability_protocol)
        (stability_protocol_available ?stability_protocol)
        (batch_record_assembled ?batch_record)
        (batch_record_qc_ready ?batch_record)
        (not
          (stability_protocol_enrolled ?stability_protocol)
        )
      )
    :effect
      (and
        (stability_protocol_enrolled ?stability_protocol)
        (stability_protocol_linked_to_batch_record ?stability_protocol ?batch_record)
        (not
          (stability_protocol_available ?stability_protocol)
        )
      )
  )
  (:action initiate_stability_trial
    :parameters (?development_workstream - development_workstream ?stability_protocol - stability_protocol ?batch_record - batch_record ?analytical_method_slot - analytical_method_slot)
    :precondition
      (and
