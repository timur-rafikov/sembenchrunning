--- domain.pddl (as passed to VAL) ---
(define (domain beneficiary_account_change_control)
  (:requirements :strips :typing :negative-preconditions)
  (:types operational_resource_group - object reference_data_group - object settlement_instrument_group - object case_root - object beneficiary_change_case - case_root operator_endpoint - operational_resource_group validator_service - operational_resource_group authorizer_service - operational_resource_group compliance_rule - operational_resource_group approval_profile - operational_resource_group audit_record - operational_resource_group signature_credential - operational_resource_group regulatory_token - operational_resource_group document - reference_data_group attachment_bundle - reference_data_group external_reference - reference_data_group clearing_channel - settlement_instrument_group routing_path - settlement_instrument_group settlement_message - settlement_instrument_group case_subtype_origin - beneficiary_change_case case_subtype_receiving - beneficiary_change_case originating_case - case_subtype_origin receiving_case - case_subtype_origin work_item - case_subtype_receiving)
  (:predicates
    (case_registered ?beneficiary_change_case - beneficiary_change_case)
    (validated ?beneficiary_change_case - beneficiary_change_case)
    (case_operator_assigned ?beneficiary_change_case - beneficiary_change_case)
    (change_executed ?beneficiary_change_case - beneficiary_change_case)
    (released_to_settlement ?beneficiary_change_case - beneficiary_change_case)
    (change_committed ?beneficiary_change_case - beneficiary_change_case)
    (endpoint_available ?operator_endpoint - operator_endpoint)
    (assigned_endpoint ?beneficiary_change_case - beneficiary_change_case ?operator_endpoint - operator_endpoint)
    (validator_available ?validator_service - validator_service)
    (assigned_validator ?beneficiary_change_case - beneficiary_change_case ?validator_service - validator_service)
    (authorizer_available ?authorizer_service - authorizer_service)
    (assigned_authorizer ?beneficiary_change_case - beneficiary_change_case ?authorizer_service - authorizer_service)
    (document_available ?document - document)
    (originating_case_document_bound ?originating_case - originating_case ?document - document)
    (receiving_case_document_bound ?receiving_case - receiving_case ?document - document)
    (originating_case_channel_binding ?originating_case - originating_case ?clearing_channel - clearing_channel)
    (clearing_channel_primary_ready ?clearing_channel - clearing_channel)
    (clearing_channel_requires_attachment ?clearing_channel - clearing_channel)
    (originating_case_ready ?originating_case - originating_case)
    (receiving_case_routing_binding ?receiving_case - receiving_case ?routing_path - routing_path)
    (routing_path_ready ?routing_path - routing_path)
    (routing_path_requires_attachment ?routing_path - routing_path)
    (receiving_case_ready ?receiving_case - receiving_case)
    (settlement_message_pending ?settlement_message - settlement_message)
    (message_composed ?settlement_message - settlement_message)
    (message_bound_channel ?settlement_message - settlement_message ?clearing_channel - clearing_channel)
    (message_bound_routing_path ?settlement_message - settlement_message ?routing_path - routing_path)
    (message_flag_originating_attachment ?settlement_message - settlement_message)
    (message_flag_receiving_attachment ?settlement_message - settlement_message)
    (message_locked ?settlement_message - settlement_message)
    (workitem_originating_case_link ?work_item - work_item ?originating_case - originating_case)
    (workitem_receiving_case_link ?work_item - work_item ?receiving_case - receiving_case)
    (workitem_bound_message ?work_item - work_item ?settlement_message - settlement_message)
    (attachment_bundle_available ?attachment_bundle - attachment_bundle)
    (workitem_has_attachment ?work_item - work_item ?attachment_bundle - attachment_bundle)
    (attachment_bundle_sealed ?attachment_bundle - attachment_bundle)
    (attachment_bound_message ?attachment_bundle - attachment_bundle ?settlement_message - settlement_message)
    (workitem_authorization_ready ?work_item - work_item)
    (workitem_authorization_engaged ?work_item - work_item)
    (workitem_authorized ?work_item - work_item)
    (workitem_has_compliance_rule ?work_item - work_item)
    (workitem_compliance_flag ?work_item - work_item)
    (workitem_ready_for_approval_profile ?work_item - work_item)
    (workitem_final_checks_passed ?work_item - work_item)
    (external_reference_available ?external_reference - external_reference)
    (workitem_bound_external_reference ?work_item - work_item ?external_reference - external_reference)
    (workitem_exception_approval_initiated ?work_item - work_item)
    (workitem_exception_authorizer_engaged ?work_item - work_item)
    (workitem_exception_authorized ?work_item - work_item)
    (compliance_rule_available ?compliance_rule - compliance_rule)
    (workitem_assigned_compliance_rule ?work_item - work_item ?compliance_rule - compliance_rule)
    (approval_profile_available ?approval_profile - approval_profile)
    (workitem_assigned_approval_profile ?work_item - work_item ?approval_profile - approval_profile)
    (signature_credential_available ?signature_credential - signature_credential)
    (workitem_bound_signature_credential ?work_item - work_item ?signature_credential - signature_credential)
    (regulatory_token_available ?regulatory_token - regulatory_token)
    (workitem_bound_regulatory_token ?work_item - work_item ?regulatory_token - regulatory_token)
    (audit_record_available ?audit_record - audit_record)
    (bound_audit_record ?beneficiary_change_case - beneficiary_change_case ?audit_record - audit_record)
    (originating_case_composition_ready ?originating_case - originating_case)
    (receiving_case_composition_ready ?receiving_case - receiving_case)
    (workitem_completion_recorded ?work_item - work_item)
  )
  (:action register_beneficiary_change_case
    :parameters (?beneficiary_change_case - beneficiary_change_case)
    :precondition
      (and
        (not
          (case_registered ?beneficiary_change_case)
        )
        (not
          (change_executed ?beneficiary_change_case)
        )
      )
    :effect (case_registered ?beneficiary_change_case)
  )
  (:action assign_operator_endpoint_to_case
    :parameters (?beneficiary_change_case - beneficiary_change_case ?operator_endpoint - operator_endpoint)
    :precondition
      (and
        (case_registered ?beneficiary_change_case)
        (not
          (case_operator_assigned ?beneficiary_change_case)
        )
        (endpoint_available ?operator_endpoint)
      )
    :effect
      (and
        (case_operator_assigned ?beneficiary_change_case)
        (assigned_endpoint ?beneficiary_change_case ?operator_endpoint)
        (not
          (endpoint_available ?operator_endpoint)
        )
      )
  )
  (:action assign_validator_to_case
    :parameters (?beneficiary_change_case - beneficiary_change_case ?validator_service - validator_service)
    :precondition
      (and
        (case_registered ?beneficiary_change_case)
        (case_operator_assigned ?beneficiary_change_case)
        (validator_available ?validator_service)
      )
    :effect
      (and
        (assigned_validator ?beneficiary_change_case ?validator_service)
        (not
          (validator_available ?validator_service)
        )
      )
  )
  (:action confirm_case_validation
    :parameters (?beneficiary_change_case - beneficiary_change_case ?validator_service - validator_service)
    :precondition
      (and
        (case_registered ?beneficiary_change_case)
        (case_operator_assigned ?beneficiary_change_case)
        (assigned_validator ?beneficiary_change_case ?validator_service)
        (not
          (validated ?beneficiary_change_case)
        )
      )
    :effect (validated ?beneficiary_change_case)
  )
  (:action release_validator_from_case
    :parameters (?beneficiary_change_case - beneficiary_change_case ?validator_service - validator_service)
    :precondition
      (and
        (assigned_validator ?beneficiary_change_case ?validator_service)
      )
    :effect
      (and
        (validator_available ?validator_service)
        (not
          (assigned_validator ?beneficiary_change_case ?validator_service)
        )
      )
  )
  (:action assign_authorizer_to_case
    :parameters (?beneficiary_change_case - beneficiary_change_case ?authorizer_service - authorizer_service)
    :precondition
      (and
        (validated ?beneficiary_change_case)
        (authorizer_available ?authorizer_service)
      )
    :effect
      (and
        (assigned_authorizer ?beneficiary_change_case ?authorizer_service)
        (not
          (authorizer_available ?authorizer_service)
        )
      )
  )
  (:action release_authorizer_from_case
    :parameters (?beneficiary_change_case - beneficiary_change_case ?authorizer_service - authorizer_service)
    :precondition
      (and
        (assigned_authorizer ?beneficiary_change_case ?authorizer_service)
      )
    :effect
      (and
        (authorizer_available ?authorizer_service)
        (not
          (assigned_authorizer ?beneficiary_change_case ?authorizer_service)
        )
      )
  )
  (:action bind_signature_credential_to_work_item
    :parameters (?work_item - work_item ?signature_credential - signature_credential)
    :precondition
      (and
        (validated ?work_item)
        (signature_credential_available ?signature_credential)
      )
    :effect
      (and
        (workitem_bound_signature_credential ?work_item ?signature_credential)
        (not
          (signature_credential_available ?signature_credential)
        )
      )
  )
  (:action unbind_signature_credential_from_work_item
    :parameters (?work_item - work_item ?signature_credential - signature_credential)
    :precondition
      (and
        (workitem_bound_signature_credential ?work_item ?signature_credential)
      )
    :effect
      (and
        (signature_credential_available ?signature_credential)
        (not
          (workitem_bound_signature_credential ?work_item ?signature_credential)
        )
      )
  )
  (:action bind_regulatory_token_to_work_item
    :parameters (?work_item - work_item ?regulatory_token - regulatory_token)
    :precondition
      (and
        (validated ?work_item)
        (regulatory_token_available ?regulatory_token)
      )
    :effect
      (and
        (workitem_bound_regulatory_token ?work_item ?regulatory_token)
        (not
          (regulatory_token_available ?regulatory_token)
        )
      )
  )
  (:action unbind_regulatory_token_from_work_item
    :parameters (?work_item - work_item ?regulatory_token - regulatory_token)
    :precondition
      (and
        (workitem_bound_regulatory_token ?work_item ?regulatory_token)
      )
    :effect
      (and
        (regulatory_token_available ?regulatory_token)
        (not
          (workitem_bound_regulatory_token ?work_item ?regulatory_token)
        )
      )
  )
  (:action collect_originating_evidence
    :parameters (?originating_case - originating_case ?clearing_channel - clearing_channel ?validator_service - validator_service)
    :precondition
      (and
        (validated ?originating_case)
        (assigned_validator ?originating_case ?validator_service)
        (originating_case_channel_binding ?originating_case ?clearing_channel)
        (not
          (clearing_channel_primary_ready ?clearing_channel)
        )
        (not
          (clearing_channel_requires_attachment ?clearing_channel)
        )
      )
    :effect (clearing_channel_primary_ready ?clearing_channel)
  )
  (:action apply_originating_authorization
    :parameters (?originating_case - originating_case ?clearing_channel - clearing_channel ?authorizer_service - authorizer_service)
    :precondition
      (and
        (validated ?originating_case)
        (assigned_authorizer ?originating_case ?authorizer_service)
        (originating_case_channel_binding ?originating_case ?clearing_channel)
        (clearing_channel_primary_ready ?clearing_channel)
        (not
          (originating_case_composition_ready ?originating_case)
        )
      )
    :effect
      (and
        (originating_case_composition_ready ?originating_case)
        (originating_case_ready ?originating_case)
      )
  )
  (:action bind_document_to_originating_case
    :parameters (?originating_case - originating_case ?clearing_channel - clearing_channel ?document - document)
    :precondition
      (and
        (validated ?originating_case)
        (originating_case_channel_binding ?originating_case ?clearing_channel)
        (document_available ?document)
        (not
          (originating_case_composition_ready ?originating_case)
        )
      )
    :effect
      (and
        (clearing_channel_requires_attachment ?clearing_channel)
        (originating_case_composition_ready ?originating_case)
        (originating_case_document_bound ?originating_case ?document)
        (not
          (document_available ?document)
        )
      )
  )
  (:action finalize_originating_evidence_collection
    :parameters (?originating_case - originating_case ?clearing_channel - clearing_channel ?validator_service - validator_service ?document - document)
    :precondition
      (and
        (validated ?originating_case)
        (assigned_validator ?originating_case ?validator_service)
        (originating_case_channel_binding ?originating_case ?clearing_channel)
        (clearing_channel_requires_attachment ?clearing_channel)
        (originating_case_document_bound ?originating_case ?document)
        (not
          (originating_case_ready ?originating_case)
        )
      )
    :effect
      (and
        (clearing_channel_primary_ready ?clearing_channel)
        (originating_case_ready ?originating_case)
        (document_available ?document)
        (not
          (originating_case_document_bound ?originating_case ?document)
        )
      )
  )
  (:action collect_receiving_evidence
    :parameters (?receiving_case - receiving_case ?routing_path - routing_path ?validator_service - validator_service)
    :precondition
      (and
        (validated ?receiving_case)
        (assigned_validator ?receiving_case ?validator_service)
        (receiving_case_routing_binding ?receiving_case ?routing_path)
        (not
          (routing_path_ready ?routing_path)
        )
        (not
          (routing_path_requires_attachment ?routing_path)
        )
      )
    :effect (routing_path_ready ?routing_path)
  )
  (:action apply_receiving_authorization
    :parameters (?receiving_case - receiving_case ?routing_path - routing_path ?authorizer_service - authorizer_service)
    :precondition
      (and
        (validated ?receiving_case)
        (assigned_authorizer ?receiving_case ?authorizer_service)
        (receiving_case_routing_binding ?receiving_case ?routing_path)
        (routing_path_ready ?routing_path)
        (not
          (receiving_case_composition_ready ?receiving_case)
        )
      )
    :effect
      (and
        (receiving_case_composition_ready ?receiving_case)
        (receiving_case_ready ?receiving_case)
      )
  )
  (:action bind_document_to_receiving_case
    :parameters (?receiving_case - receiving_case ?routing_path - routing_path ?document - document)
    :precondition
      (and
        (validated ?receiving_case)
        (receiving_case_routing_binding ?receiving_case ?routing_path)
        (document_available ?document)
        (not
          (receiving_case_composition_ready ?receiving_case)
        )
      )
    :effect
      (and
        (routing_path_requires_attachment ?routing_path)
        (receiving_case_composition_ready ?receiving_case)
        (receiving_case_document_bound ?receiving_case ?document)
        (not
          (document_available ?document)
        )
      )
  )
  (:action finalize_receiving_evidence_collection
    :parameters (?receiving_case - receiving_case ?routing_path - routing_path ?validator_service - validator_service ?document - document)
    :precondition
      (and
        (validated ?receiving_case)
        (assigned_validator ?receiving_case ?validator_service)
        (receiving_case_routing_binding ?receiving_case ?routing_path)
        (routing_path_requires_attachment ?routing_path)
        (receiving_case_document_bound ?receiving_case ?document)
        (not
          (receiving_case_ready ?receiving_case)
        )
      )
    :effect
      (and
        (routing_path_ready ?routing_path)
        (receiving_case_ready ?receiving_case)
        (document_available ?document)
        (not
          (receiving_case_document_bound ?receiving_case ?document)
        )
      )
  )
  (:action compose_and_bind_settlement_message
    :parameters (?originating_case - originating_case ?receiving_case - receiving_case ?clearing_channel - clearing_channel ?routing_path - routing_path ?settlement_message - settlement_message)
    :precondition
      (and
        (originating_case_composition_ready ?originating_case)
        (receiving_case_composition_ready ?receiving_case)
        (originating_case_channel_binding ?originating_case ?clearing_channel)
        (receiving_case_routing_binding ?receiving_case ?routing_path)
        (clearing_channel_primary_ready ?clearing_channel)
        (routing_path_ready ?routing_path)
        (originating_case_ready ?originating_case)
        (receiving_case_ready ?receiving_case)
        (settlement_message_pending ?settlement_message)
      )
    :effect
      (and
        (message_composed ?settlement_message)
        (message_bound_channel ?settlement_message ?clearing_channel)
        (message_bound_routing_path ?settlement_message ?routing_path)
        (not
          (settlement_message_pending ?settlement_message)
        )
      )
  )
  (:action compose_settlement_message_with_originating_attachment
    :parameters (?originating_case - originating_case ?receiving_case - receiving_case ?clearing_channel - clearing_channel ?routing_path - routing_path ?settlement_message - settlement_message)
    :precondition
      (and
        (originating_case_composition_ready ?originating_case)
        (receiving_case_composition_ready ?receiving_case)
        (originating_case_channel_binding ?originating_case ?clearing_channel)
        (receiving_case_routing_binding ?receiving_case ?routing_path)
        (clearing_channel_requires_attachment ?clearing_channel)
        (routing_path_ready ?routing_path)
        (not
          (originating_case_ready ?originating_case)
        )
        (receiving_case_ready ?receiving_case)
        (settlement_message_pending ?settlement_message)
      )
    :effect
      (and
        (message_composed ?settlement_message)
        (message_bound_channel ?settlement_message ?clearing_channel)
        (message_bound_routing_path ?settlement_message ?routing_path)
        (message_flag_originating_attachment ?settlement_message)
        (not
          (settlement_message_pending ?settlement_message)
        )
      )
  )
  (:action compose_settlement_message_with_receiving_attachment
    :parameters (?originating_case - originating_case ?receiving_case - receiving_case ?clearing_channel - clearing_channel ?routing_path - routing_path ?settlement_message - settlement_message)
    :precondition
      (and
        (originating_case_composition_ready ?originating_case)
        (receiving_case_composition_ready ?receiving_case)
        (originating_case_channel_binding ?originating_case ?clearing_channel)
        (receiving_case_routing_binding ?receiving_case ?routing_path)
        (clearing_channel_primary_ready ?clearing_channel)
        (routing_path_requires_attachment ?routing_path)
        (originating_case_ready ?originating_case)
        (not
          (receiving_case_ready ?receiving_case)
        )
        (settlement_message_pending ?settlement_message)
      )
    :effect
      (and
        (message_composed ?settlement_message)
        (message_bound_channel ?settlement_message ?clearing_channel)
        (message_bound_routing_path ?settlement_message ?routing_path)
        (message_flag_receiving_attachment ?settlement_message)
        (not
          (settlement_message_pending ?settlement_message)
        )
      )
  )
  (:action compose_settlement_message_with_both_attachments
    :parameters (?originating_case - originating_case ?receiving_case - receiving_case ?clearing_channel - clearing_channel ?routing_path - routing_path ?settlement_message - settlement_message)
    :precondition
      (and
        (originating_case_composition_ready ?originating_case)
        (receiving_case_composition_ready ?receiving_case)
        (originating_case_channel_binding ?originating_case ?clearing_channel)
        (receiving_case_routing_binding ?receiving_case ?routing_path)
        (clearing_channel_requires_attachment ?clearing_channel)
        (routing_path_requires_attachment ?routing_path)
        (not
          (originating_case_ready ?originating_case)
        )
        (not
          (receiving_case_ready ?receiving_case)
        )
        (settlement_message_pending ?settlement_message)
      )
    :effect
      (and
        (message_composed ?settlement_message)
        (message_bound_channel ?settlement_message ?clearing_channel)
        (message_bound_routing_path ?settlement_message ?routing_path)
        (message_flag_originating_attachment ?settlement_message)
        (message_flag_receiving_attachment ?settlement_message)
        (not
          (settlement_message_pending ?settlement_message)
        )
      )
  )
  (:action lock_settlement_message
    :parameters (?settlement_message - settlement_message ?originating_case - originating_case ?validator_service - validator_service)
    :precondition
      (and
        (message_composed ?settlement_message)
        (originating_case_composition_ready ?originating_case)
        (assigned_validator ?originating_case ?validator_service)
        (not
          (message_locked ?settlement_message)
        )
      )
    :effect (message_locked ?settlement_message)
  )
  (:action seal_and_attach_bundle_to_message
    :parameters (?work_item - work_item ?attachment_bundle - attachment_bundle ?settlement_message - settlement_message)
    :precondition
      (and
        (validated ?work_item)
        (workitem_bound_message ?work_item ?settlement_message)
        (workitem_has_attachment ?work_item ?attachment_bundle)
        (attachment_bundle_available ?attachment_bundle)
        (message_composed ?settlement_message)
        (message_locked ?settlement_message)
        (not
          (attachment_bundle_sealed ?attachment_bundle)
        )
      )
    :effect
      (and
        (attachment_bundle_sealed ?attachment_bundle)
        (attachment_bound_message ?attachment_bundle ?settlement_message)
        (not
          (attachment_bundle_available ?attachment_bundle)
        )
      )
  )
  (:action prepare_workitem_for_authorization
    :parameters (?work_item - work_item ?attachment_bundle - attachment_bundle ?settlement_message - settlement_message ?validator_service - validator_service)
    :precondition
      (and
        (validated ?work_item)
        (workitem_has_attachment ?work_item ?attachment_bundle)
        (attachment_bundle_sealed ?attachment_bundle)
        (attachment_bound_message ?attachment_bundle ?settlement_message)
        (assigned_validator ?work_item ?validator_service)
        (not
          (message_flag_originating_attachment ?settlement_message)
        )
        (not
          (workitem_authorization_ready ?work_item)
        )
      )
    :effect (workitem_authorization_ready ?work_item)
  )
  (:action assign_compliance_rule_to_workitem
    :parameters (?work_item - work_item ?compliance_rule - compliance_rule)
    :precondition
      (and
        (validated ?work_item)
        (compliance_rule_available ?compliance_rule)
        (not
          (workitem_has_compliance_rule ?work_item)
        )
      )
    :effect
      (and
        (workitem_has_compliance_rule ?work_item)
        (workitem_assigned_compliance_rule ?work_item ?compliance_rule)
        (not
          (compliance_rule_available ?compliance_rule)
        )
      )
  )
  (:action initiate_authorization_with_compliance_and_attachment
    :parameters (?work_item - work_item ?attachment_bundle - attachment_bundle ?settlement_message - settlement_message ?validator_service - validator_service ?compliance_rule - compliance_rule)
    :precondition
      (and
        (validated ?work_item)
        (workitem_has_attachment ?work_item ?attachment_bundle)
        (attachment_bundle_sealed ?attachment_bundle)
        (attachment_bound_message ?attachment_bundle ?settlement_message)
        (assigned_validator ?work_item ?validator_service)
        (message_flag_originating_attachment ?settlement_message)
        (workitem_has_compliance_rule ?work_item)
        (workitem_assigned_compliance_rule ?work_item ?compliance_rule)
        (not
          (workitem_authorization_ready ?work_item)
        )
      )
    :effect
      (and
        (workitem_authorization_ready ?work_item)
        (workitem_compliance_flag ?work_item)
      )
  )
  (:action request_primary_authorizer_signature
    :parameters (?work_item - work_item ?signature_credential - signature_credential ?authorizer_service - authorizer_service ?attachment_bundle - attachment_bundle ?settlement_message - settlement_message)
    :precondition
      (and
        (workitem_authorization_ready ?work_item)
        (workitem_bound_signature_credential ?work_item ?signature_credential)
        (assigned_authorizer ?work_item ?authorizer_service)
        (workitem_has_attachment ?work_item ?attachment_bundle)
        (attachment_bound_message ?attachment_bundle ?settlement_message)
        (not
          (message_flag_receiving_attachment ?settlement_message)
        )
        (not
          (workitem_authorization_engaged ?work_item)
        )
      )
    :effect (workitem_authorization_engaged ?work_item)
  )
  (:action request_secondary_authorizer_signature
    :parameters (?work_item - work_item ?signature_credential - signature_credential ?authorizer_service - authorizer_service ?attachment_bundle - attachment_bundle ?settlement_message - settlement_message)
    :precondition
