--- domain.pddl (as passed to VAL) ---
(define (domain end_of_season_transition_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types operational_resource - object material_resource - object scheduling_resource - object field_entity - object field_parcel - field_entity equipment_unit - operational_resource operation_task - operational_resource crew_team - operational_resource site_inspection_kit - operational_resource compliance_certificate - operational_resource transition_checklist - operational_resource calibration_tool - operational_resource regulatory_approval - operational_resource input_material_batch - material_resource supply_container - material_resource inspection_report - material_resource primary_time_window - scheduling_resource secondary_time_window - scheduling_resource staging_unit - scheduling_resource field_section_group - field_parcel plan_assignment_group - field_parcel primary_field_section - field_section_group secondary_field_section - field_section_group transition_plan - plan_assignment_group)
  (:predicates
    (parcel_identified ?field_parcel - field_parcel)
    (entity_ready_for_transition ?field_parcel - field_parcel)
    (equipment_reserved_for_parcel ?field_parcel - field_parcel)
    (transition_finalized ?field_parcel - field_parcel)
    (handover_ready ?field_parcel - field_parcel)
    (checklist_completed ?field_parcel - field_parcel)
    (equipment_available ?equipment_unit - equipment_unit)
    (equipment_assigned ?field_parcel - field_parcel ?equipment_unit - equipment_unit)
    (operation_task_available ?operation_task - operation_task)
    (task_assigned ?field_parcel - field_parcel ?operation_task - operation_task)
    (crew_available ?crew_team - crew_team)
    (crew_assigned ?field_parcel - field_parcel ?crew_team - crew_team)
    (material_batch_available ?input_material_batch - input_material_batch)
    (material_allocated_to_section ?primary_field_section - primary_field_section ?input_material_batch - input_material_batch)
    (material_allocated_to_secondary_section ?secondary_field_section - secondary_field_section ?input_material_batch - input_material_batch)
    (section_primary_time_window_assigned ?primary_field_section - primary_field_section ?time_window_a - primary_time_window)
    (primary_time_window_reserved ?time_window_a - primary_time_window)
    (primary_time_window_material_allocated ?time_window_a - primary_time_window)
    (primary_section_ready ?primary_field_section - primary_field_section)
    (section_secondary_time_window_assigned ?secondary_field_section - secondary_field_section ?time_window_b - secondary_time_window)
    (secondary_time_window_reserved ?time_window_b - secondary_time_window)
    (secondary_time_window_material_allocated ?time_window_b - secondary_time_window)
    (secondary_section_ready ?secondary_field_section - secondary_field_section)
    (staging_unit_available ?staging_unit - staging_unit)
    (staging_unit_prepared ?staging_unit - staging_unit)
    (staging_unit_time_window_primary_attached ?staging_unit - staging_unit ?time_window_a - primary_time_window)
    (staging_unit_time_window_secondary_attached ?staging_unit - staging_unit ?time_window_b - secondary_time_window)
    (staging_unit_tag_a ?staging_unit - staging_unit)
    (staging_unit_tag_b ?staging_unit - staging_unit)
    (staging_unit_confirmed_for_material_handling ?staging_unit - staging_unit)
    (plan_associated_with_primary_section ?transition_plan - transition_plan ?primary_field_section - primary_field_section)
    (plan_associated_with_secondary_section ?transition_plan - transition_plan ?secondary_field_section - secondary_field_section)
    (plan_assigned_to_staging_unit ?transition_plan - transition_plan ?staging_unit - staging_unit)
    (supply_container_available ?supply_container - supply_container)
    (plan_has_supply_container ?transition_plan - transition_plan ?supply_container - supply_container)
    (supply_container_prepared ?supply_container - supply_container)
    (supply_container_staged_in_staging_unit ?supply_container - supply_container ?staging_unit - staging_unit)
    (plan_validated ?transition_plan - transition_plan)
    (plan_calibrated ?transition_plan - transition_plan)
    (plan_ready_for_final_verification ?transition_plan - transition_plan)
    (site_inspection_attached ?transition_plan - transition_plan)
    (site_inspection_verified ?transition_plan - transition_plan)
    (plan_ready_for_certification ?transition_plan - transition_plan)
    (plan_final_checks_passed ?transition_plan - transition_plan)
    (inspection_report_available ?inspection_report - inspection_report)
    (plan_has_inspection_report ?transition_plan - transition_plan ?inspection_report - inspection_report)
    (inspection_report_attached_to_plan ?transition_plan - transition_plan)
    (inspection_review_complete ?transition_plan - transition_plan)
    (regulatory_review_completed ?transition_plan - transition_plan)
    (inspection_kit_available ?site_inspection_kit - site_inspection_kit)
    (plan_has_inspectionkit ?transition_plan - transition_plan ?site_inspection_kit - site_inspection_kit)
    (compliance_certificate_available ?compliance_certificate - compliance_certificate)
    (plan_has_compliance_certificate ?transition_plan - transition_plan ?compliance_certificate - compliance_certificate)
    (calibration_tool_available ?calibration_tool - calibration_tool)
    (plan_has_calibration_tool ?transition_plan - transition_plan ?calibration_tool - calibration_tool)
    (regulatory_approval_available ?regulatory_approval - regulatory_approval)
    (plan_has_regulatory_approval ?transition_plan - transition_plan ?regulatory_approval - regulatory_approval)
    (transition_checklist_available ?transition_checklist - transition_checklist)
    (entity_has_transition_checklist ?field_parcel - field_parcel ?transition_checklist - transition_checklist)
    (primary_section_prepared ?primary_field_section - primary_field_section)
    (secondary_section_prepared ?secondary_field_section - secondary_field_section)
    (plan_handover_recorded ?transition_plan - transition_plan)
  )
  (:action identify_field_parcel
    :parameters (?field_parcel - field_parcel)
    :precondition
      (and
        (not
          (parcel_identified ?field_parcel)
        )
        (not
          (transition_finalized ?field_parcel)
        )
      )
    :effect (parcel_identified ?field_parcel)
  )
  (:action reserve_equipment_for_parcel
    :parameters (?field_parcel - field_parcel ?equipment_unit - equipment_unit)
    :precondition
      (and
        (parcel_identified ?field_parcel)
        (not
          (equipment_reserved_for_parcel ?field_parcel)
        )
        (equipment_available ?equipment_unit)
      )
    :effect
      (and
        (equipment_reserved_for_parcel ?field_parcel)
        (equipment_assigned ?field_parcel ?equipment_unit)
        (not
          (equipment_available ?equipment_unit)
        )
      )
  )
  (:action assign_operation_task_to_parcel
    :parameters (?field_parcel - field_parcel ?operation_task - operation_task)
    :precondition
      (and
        (parcel_identified ?field_parcel)
        (equipment_reserved_for_parcel ?field_parcel)
        (operation_task_available ?operation_task)
      )
    :effect
      (and
        (task_assigned ?field_parcel ?operation_task)
        (not
          (operation_task_available ?operation_task)
        )
      )
  )
  (:action confirm_operation_schedule_for_parcel
    :parameters (?field_parcel - field_parcel ?operation_task - operation_task)
    :precondition
      (and
        (parcel_identified ?field_parcel)
        (equipment_reserved_for_parcel ?field_parcel)
        (task_assigned ?field_parcel ?operation_task)
        (not
          (entity_ready_for_transition ?field_parcel)
        )
      )
    :effect (entity_ready_for_transition ?field_parcel)
  )
  (:action release_operation_task_from_parcel
    :parameters (?field_parcel - field_parcel ?operation_task - operation_task)
    :precondition
      (and
        (task_assigned ?field_parcel ?operation_task)
      )
    :effect
      (and
        (operation_task_available ?operation_task)
        (not
          (task_assigned ?field_parcel ?operation_task)
        )
      )
  )
  (:action assign_crew_to_parcel
    :parameters (?field_parcel - field_parcel ?crew_team - crew_team)
    :precondition
      (and
        (entity_ready_for_transition ?field_parcel)
        (crew_available ?crew_team)
      )
    :effect
      (and
        (crew_assigned ?field_parcel ?crew_team)
        (not
          (crew_available ?crew_team)
        )
      )
  )
  (:action release_crew_from_parcel
    :parameters (?field_parcel - field_parcel ?crew_team - crew_team)
    :precondition
      (and
        (crew_assigned ?field_parcel ?crew_team)
      )
    :effect
      (and
        (crew_available ?crew_team)
        (not
          (crew_assigned ?field_parcel ?crew_team)
        )
      )
  )
  (:action attach_calibration_tool_to_plan
    :parameters (?transition_plan - transition_plan ?calibration_tool - calibration_tool)
    :precondition
      (and
        (entity_ready_for_transition ?transition_plan)
        (calibration_tool_available ?calibration_tool)
      )
    :effect
      (and
        (plan_has_calibration_tool ?transition_plan ?calibration_tool)
        (not
          (calibration_tool_available ?calibration_tool)
        )
      )
  )
  (:action detach_calibration_tool_from_plan
    :parameters (?transition_plan - transition_plan ?calibration_tool - calibration_tool)
    :precondition
      (and
        (plan_has_calibration_tool ?transition_plan ?calibration_tool)
      )
    :effect
      (and
        (calibration_tool_available ?calibration_tool)
        (not
          (plan_has_calibration_tool ?transition_plan ?calibration_tool)
        )
      )
  )
  (:action attach_regulatory_approval_to_plan
    :parameters (?transition_plan - transition_plan ?regulatory_approval - regulatory_approval)
    :precondition
      (and
        (entity_ready_for_transition ?transition_plan)
        (regulatory_approval_available ?regulatory_approval)
      )
    :effect
      (and
        (plan_has_regulatory_approval ?transition_plan ?regulatory_approval)
        (not
          (regulatory_approval_available ?regulatory_approval)
        )
      )
  )
  (:action detach_regulatory_approval_from_plan
    :parameters (?transition_plan - transition_plan ?regulatory_approval - regulatory_approval)
    :precondition
      (and
        (plan_has_regulatory_approval ?transition_plan ?regulatory_approval)
      )
    :effect
      (and
        (regulatory_approval_available ?regulatory_approval)
        (not
          (plan_has_regulatory_approval ?transition_plan ?regulatory_approval)
        )
      )
  )
  (:action reserve_primary_time_window_for_section
    :parameters (?primary_field_section - primary_field_section ?time_window_a - primary_time_window ?operation_task - operation_task)
    :precondition
      (and
        (entity_ready_for_transition ?primary_field_section)
        (task_assigned ?primary_field_section ?operation_task)
        (section_primary_time_window_assigned ?primary_field_section ?time_window_a)
        (not
          (primary_time_window_reserved ?time_window_a)
        )
        (not
          (primary_time_window_material_allocated ?time_window_a)
        )
      )
    :effect (primary_time_window_reserved ?time_window_a)
  )
  (:action assign_crew_to_primary_section_for_primary_time_window
    :parameters (?primary_field_section - primary_field_section ?time_window_a - primary_time_window ?crew_team - crew_team)
    :precondition
      (and
        (entity_ready_for_transition ?primary_field_section)
        (crew_assigned ?primary_field_section ?crew_team)
        (section_primary_time_window_assigned ?primary_field_section ?time_window_a)
        (primary_time_window_reserved ?time_window_a)
        (not
          (primary_section_prepared ?primary_field_section)
        )
      )
    :effect
      (and
        (primary_section_prepared ?primary_field_section)
        (primary_section_ready ?primary_field_section)
      )
  )
  (:action allocate_material_to_primary_section
    :parameters (?primary_field_section - primary_field_section ?time_window_a - primary_time_window ?input_material_batch - input_material_batch)
    :precondition
      (and
        (entity_ready_for_transition ?primary_field_section)
        (section_primary_time_window_assigned ?primary_field_section ?time_window_a)
        (material_batch_available ?input_material_batch)
        (not
          (primary_section_prepared ?primary_field_section)
        )
      )
    :effect
      (and
        (primary_time_window_material_allocated ?time_window_a)
        (primary_section_prepared ?primary_field_section)
        (material_allocated_to_section ?primary_field_section ?input_material_batch)
        (not
          (material_batch_available ?input_material_batch)
        )
      )
  )
  (:action confirm_material_assignment_for_primary_section
    :parameters (?primary_field_section - primary_field_section ?time_window_a - primary_time_window ?operation_task - operation_task ?input_material_batch - input_material_batch)
    :precondition
      (and
        (entity_ready_for_transition ?primary_field_section)
        (task_assigned ?primary_field_section ?operation_task)
        (section_primary_time_window_assigned ?primary_field_section ?time_window_a)
        (primary_time_window_material_allocated ?time_window_a)
        (material_allocated_to_section ?primary_field_section ?input_material_batch)
        (not
          (primary_section_ready ?primary_field_section)
        )
      )
    :effect
      (and
        (primary_time_window_reserved ?time_window_a)
        (primary_section_ready ?primary_field_section)
        (material_batch_available ?input_material_batch)
        (not
          (material_allocated_to_section ?primary_field_section ?input_material_batch)
        )
      )
  )
  (:action reserve_secondary_time_window_for_section
    :parameters (?secondary_field_section - secondary_field_section ?time_window_b - secondary_time_window ?operation_task - operation_task)
    :precondition
      (and
        (entity_ready_for_transition ?secondary_field_section)
        (task_assigned ?secondary_field_section ?operation_task)
        (section_secondary_time_window_assigned ?secondary_field_section ?time_window_b)
        (not
          (secondary_time_window_reserved ?time_window_b)
        )
        (not
          (secondary_time_window_material_allocated ?time_window_b)
        )
      )
    :effect (secondary_time_window_reserved ?time_window_b)
  )
  (:action assign_crew_to_secondary_section_for_secondary_time_window
    :parameters (?secondary_field_section - secondary_field_section ?time_window_b - secondary_time_window ?crew_team - crew_team)
    :precondition
      (and
        (entity_ready_for_transition ?secondary_field_section)
        (crew_assigned ?secondary_field_section ?crew_team)
        (section_secondary_time_window_assigned ?secondary_field_section ?time_window_b)
        (secondary_time_window_reserved ?time_window_b)
        (not
          (secondary_section_prepared ?secondary_field_section)
        )
      )
    :effect
      (and
        (secondary_section_prepared ?secondary_field_section)
        (secondary_section_ready ?secondary_field_section)
      )
  )
  (:action allocate_material_to_secondary_section
    :parameters (?secondary_field_section - secondary_field_section ?time_window_b - secondary_time_window ?input_material_batch - input_material_batch)
    :precondition
      (and
        (entity_ready_for_transition ?secondary_field_section)
        (section_secondary_time_window_assigned ?secondary_field_section ?time_window_b)
        (material_batch_available ?input_material_batch)
        (not
          (secondary_section_prepared ?secondary_field_section)
        )
      )
    :effect
      (and
        (secondary_time_window_material_allocated ?time_window_b)
        (secondary_section_prepared ?secondary_field_section)
        (material_allocated_to_secondary_section ?secondary_field_section ?input_material_batch)
        (not
          (material_batch_available ?input_material_batch)
        )
      )
  )
  (:action confirm_material_assignment_for_secondary_section
    :parameters (?secondary_field_section - secondary_field_section ?time_window_b - secondary_time_window ?operation_task - operation_task ?input_material_batch - input_material_batch)
    :precondition
      (and
        (entity_ready_for_transition ?secondary_field_section)
        (task_assigned ?secondary_field_section ?operation_task)
        (section_secondary_time_window_assigned ?secondary_field_section ?time_window_b)
        (secondary_time_window_material_allocated ?time_window_b)
        (material_allocated_to_secondary_section ?secondary_field_section ?input_material_batch)
        (not
          (secondary_section_ready ?secondary_field_section)
        )
      )
    :effect
      (and
        (secondary_time_window_reserved ?time_window_b)
        (secondary_section_ready ?secondary_field_section)
        (material_batch_available ?input_material_batch)
        (not
          (material_allocated_to_secondary_section ?secondary_field_section ?input_material_batch)
        )
      )
  )
  (:action assemble_staging_unit_standard
    :parameters (?primary_field_section - primary_field_section ?secondary_field_section - secondary_field_section ?time_window_a - primary_time_window ?time_window_b - secondary_time_window ?staging_unit - staging_unit)
    :precondition
      (and
        (primary_section_prepared ?primary_field_section)
        (secondary_section_prepared ?secondary_field_section)
        (section_primary_time_window_assigned ?primary_field_section ?time_window_a)
        (section_secondary_time_window_assigned ?secondary_field_section ?time_window_b)
        (primary_time_window_reserved ?time_window_a)
        (secondary_time_window_reserved ?time_window_b)
        (primary_section_ready ?primary_field_section)
        (secondary_section_ready ?secondary_field_section)
        (staging_unit_available ?staging_unit)
      )
    :effect
      (and
        (staging_unit_prepared ?staging_unit)
        (staging_unit_time_window_primary_attached ?staging_unit ?time_window_a)
        (staging_unit_time_window_secondary_attached ?staging_unit ?time_window_b)
        (not
          (staging_unit_available ?staging_unit)
        )
      )
  )
  (:action assemble_staging_unit_with_primary_material
    :parameters (?primary_field_section - primary_field_section ?secondary_field_section - secondary_field_section ?time_window_a - primary_time_window ?time_window_b - secondary_time_window ?staging_unit - staging_unit)
    :precondition
      (and
        (primary_section_prepared ?primary_field_section)
        (secondary_section_prepared ?secondary_field_section)
        (section_primary_time_window_assigned ?primary_field_section ?time_window_a)
        (section_secondary_time_window_assigned ?secondary_field_section ?time_window_b)
        (primary_time_window_material_allocated ?time_window_a)
        (secondary_time_window_reserved ?time_window_b)
        (not
          (primary_section_ready ?primary_field_section)
        )
        (secondary_section_ready ?secondary_field_section)
        (staging_unit_available ?staging_unit)
      )
    :effect
      (and
        (staging_unit_prepared ?staging_unit)
        (staging_unit_time_window_primary_attached ?staging_unit ?time_window_a)
        (staging_unit_time_window_secondary_attached ?staging_unit ?time_window_b)
        (staging_unit_tag_a ?staging_unit)
        (not
          (staging_unit_available ?staging_unit)
        )
      )
  )
  (:action assemble_staging_unit_with_secondary_material
    :parameters (?primary_field_section - primary_field_section ?secondary_field_section - secondary_field_section ?time_window_a - primary_time_window ?time_window_b - secondary_time_window ?staging_unit - staging_unit)
    :precondition
      (and
        (primary_section_prepared ?primary_field_section)
        (secondary_section_prepared ?secondary_field_section)
        (section_primary_time_window_assigned ?primary_field_section ?time_window_a)
        (section_secondary_time_window_assigned ?secondary_field_section ?time_window_b)
        (primary_time_window_reserved ?time_window_a)
        (secondary_time_window_material_allocated ?time_window_b)
        (primary_section_ready ?primary_field_section)
        (not
          (secondary_section_ready ?secondary_field_section)
        )
        (staging_unit_available ?staging_unit)
      )
    :effect
      (and
        (staging_unit_prepared ?staging_unit)
        (staging_unit_time_window_primary_attached ?staging_unit ?time_window_a)
        (staging_unit_time_window_secondary_attached ?staging_unit ?time_window_b)
        (staging_unit_tag_b ?staging_unit)
        (not
          (staging_unit_available ?staging_unit)
        )
      )
  )
  (:action assemble_staging_unit_with_both_materials
    :parameters (?primary_field_section - primary_field_section ?secondary_field_section - secondary_field_section ?time_window_a - primary_time_window ?time_window_b - secondary_time_window ?staging_unit - staging_unit)
    :precondition
      (and
        (primary_section_prepared ?primary_field_section)
        (secondary_section_prepared ?secondary_field_section)
        (section_primary_time_window_assigned ?primary_field_section ?time_window_a)
        (section_secondary_time_window_assigned ?secondary_field_section ?time_window_b)
        (primary_time_window_material_allocated ?time_window_a)
        (secondary_time_window_material_allocated ?time_window_b)
        (not
          (primary_section_ready ?primary_field_section)
        )
        (not
          (secondary_section_ready ?secondary_field_section)
        )
        (staging_unit_available ?staging_unit)
      )
    :effect
      (and
        (staging_unit_prepared ?staging_unit)
        (staging_unit_time_window_primary_attached ?staging_unit ?time_window_a)
        (staging_unit_time_window_secondary_attached ?staging_unit ?time_window_b)
        (staging_unit_tag_a ?staging_unit)
        (staging_unit_tag_b ?staging_unit)
        (not
          (staging_unit_available ?staging_unit)
        )
      )
  )
  (:action confirm_staging_unit_prepared
    :parameters (?staging_unit - staging_unit ?primary_field_section - primary_field_section ?operation_task - operation_task)
    :precondition
      (and
        (staging_unit_prepared ?staging_unit)
        (primary_section_prepared ?primary_field_section)
        (task_assigned ?primary_field_section ?operation_task)
        (not
          (staging_unit_confirmed_for_material_handling ?staging_unit)
        )
      )
    :effect (staging_unit_confirmed_for_material_handling ?staging_unit)
  )
  (:action prepare_supply_container
    :parameters (?transition_plan - transition_plan ?supply_container - supply_container ?staging_unit - staging_unit)
    :precondition
      (and
        (entity_ready_for_transition ?transition_plan)
        (plan_assigned_to_staging_unit ?transition_plan ?staging_unit)
        (plan_has_supply_container ?transition_plan ?supply_container)
        (supply_container_available ?supply_container)
        (staging_unit_prepared ?staging_unit)
        (staging_unit_confirmed_for_material_handling ?staging_unit)
        (not
          (supply_container_prepared ?supply_container)
        )
      )
    :effect
      (and
        (supply_container_prepared ?supply_container)
        (supply_container_staged_in_staging_unit ?supply_container ?staging_unit)
        (not
          (supply_container_available ?supply_container)
        )
      )
  )
  (:action validate_supply_container_for_plan
    :parameters (?transition_plan - transition_plan ?supply_container - supply_container ?staging_unit - staging_unit ?operation_task - operation_task)
    :precondition
      (and
        (entity_ready_for_transition ?transition_plan)
        (plan_has_supply_container ?transition_plan ?supply_container)
        (supply_container_prepared ?supply_container)
        (supply_container_staged_in_staging_unit ?supply_container ?staging_unit)
        (task_assigned ?transition_plan ?operation_task)
        (not
          (staging_unit_tag_a ?staging_unit)
        )
        (not
          (plan_validated ?transition_plan)
        )
      )
    :effect (plan_validated ?transition_plan)
  )
  (:action attach_site_inspection_kit_to_plan
    :parameters (?transition_plan - transition_plan ?site_inspection_kit - site_inspection_kit)
    :precondition
      (and
        (entity_ready_for_transition ?transition_plan)
        (inspection_kit_available ?site_inspection_kit)
        (not
          (site_inspection_attached ?transition_plan)
        )
      )
    :effect
      (and
        (site_inspection_attached ?transition_plan)
        (plan_has_inspectionkit ?transition_plan ?site_inspection_kit)
        (not
          (inspection_kit_available ?site_inspection_kit)
        )
      )
  )
  (:action finalize_supply_container_attachment_to_plan
    :parameters (?transition_plan - transition_plan ?supply_container - supply_container ?staging_unit - staging_unit ?operation_task - operation_task ?site_inspection_kit - site_inspection_kit)
    :precondition
