  (:action mark_branch_available
    :parameters (?branch_entity - branch_entity)
    :precondition
      (and
        (not
          (branch_available ?branch_entity)
        )
        (not
          (entity_committed ?branch_entity)
        )
      )
    :effect (branch_available ?branch_entity)
  )
  (:action allocate_assignment_token_to_branch
    :parameters (?branch_entity - branch_entity ?assignment_token - assignment_token)
    :precondition
      (and
        (branch_available ?branch_entity)
        (not
          (token_assigned_flag ?branch_entity)
        )
        (assignment_token_available ?assignment_token)
      )
    :effect
      (and
        (token_assigned_flag ?branch_entity)
        (assignment_token_bound_to_branch ?branch_entity ?assignment_token)
        (not
          (assignment_token_available ?assignment_token)
        )
      )
  )
  (:action attach_objective_to_branch
    :parameters (?branch_entity - branch_entity ?objective_template - objective_template)
    :precondition
      (and
        (branch_available ?branch_entity)
        (token_assigned_flag ?branch_entity)
        (objective_available ?objective_template)
      )
    :effect
      (and
        (objective_bound_to_branch ?branch_entity ?objective_template)
        (not
          (objective_available ?objective_template)
        )
      )
  )
  (:action activate_branch_after_objective
    :parameters (?branch_entity - branch_entity ?objective_template - objective_template)

      (and
        (branch_available ?branch_entity)
        (token_assigned_flag ?branch_entity)
        (objective_bound_to_branch ?branch_entity ?objective_template)
        (not
          (activated_flag ?branch_entity)
        )
      )
    :effect (activated_flag ?branch_entity)
  )
  (:action detach_objective_from_branch
    :parameters (?branch_entity - branch_entity ?objective_template - objective_template)
    :precondition
      (and
        (objective_bound_to_branch ?branch_entity ?objective_template)
      )
    :effect
      (and
        (objective_available ?objective_template)
        (not
          (objective_bound_to_branch ?branch_entity ?objective_template)
        )
      )
  )
  (:action bind_agent_to_branch
    :parameters (?branch_entity - branch_entity ?agent_role - agent_role)
    :precondition
      (and
        (activated_flag ?branch_entity)
        (agent_available ?agent_role)
      )
    :effect
      (and
        (agent_bound_to_branch ?branch_entity ?agent_role)
        (not
          (agent_available ?agent_role)
        )
      )
  )
  (:action unbind_agent_from_branch
    :parameters (?branch_entity - branch_entity ?agent_role - agent_role)

      (and
        (agent_bound_to_branch ?branch_entity ?agent_role)

        (agent_available ?agent_role)

          (agent_bound_to_branch ?branch_entity ?agent_role)

      )
  )

        (slot_a_site_link ?faction_slot_a ?site_context_a)
        (not
          (site_signal_a_positive ?site_context_a)

        (not
          (site_signal_a_negative ?site_context_a)

    :parameters (?faction_slot_a - faction_slot_a ?site_context_a - site_context_a ?agent_role - agent_role)

      (and
        (activated_flag ?faction_slot_a)

        (slot_a_site_link ?faction_slot_a ?site_context_a)
        (site_signal_a_positive ?site_context_a)

          (slot_a_ready ?faction_slot_a)

          (slot_a_ready ?faction_slot_a)

        (not
          (stage_item_available ?stage_item)

        (not

          (optional_trigger_available ?optional_trigger)

        (activated_flag ?story_branch_instance)
        (artifact_bound_to_branch ?story_branch_instance ?artifact)
