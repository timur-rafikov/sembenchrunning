--- domain.pddl (as passed to VAL) ---
(define (domain finance_management_reporting_pack_assembly)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_group - object document_category - object supporting_artifact - object report_element_type - object schedule - report_element_type preparer - resource_group data_extract - resource_group reviewer - resource_group approver_role - resource_group presentation_template - resource_group audit_query - resource_group accounting_policy - resource_group audit_partner - resource_group variance_issue_record - document_category supporting_file - document_category external_reviewer_organization - document_category control_point - supporting_artifact disclosure_item - supporting_artifact section_draft - supporting_artifact entity_schedule_group - schedule sub_schedule_group - schedule legal_entity_schedule - entity_schedule_group business_unit_schedule - entity_schedule_group reporting_pack - sub_schedule_group)
  (:predicates
    (schedule_in_scope ?schedule - schedule)
    (is_prepared ?schedule - schedule)
    (preparer_assigned ?schedule - schedule)
    (included_in_pack ?schedule - schedule)
    (ready_for_pack ?schedule - schedule)
    (preparation_complete ?schedule - schedule)
    (preparer_available ?preparer - preparer)
    (assigned_to_preparer ?schedule - schedule ?preparer - preparer)
    (data_extract_available ?data_extract - data_extract)
    (linked_to_data_extract ?schedule - schedule ?data_extract - data_extract)
    (reviewer_available ?reviewer - reviewer)
    (assigned_to_reviewer ?schedule - schedule ?reviewer - reviewer)
    (variance_issue_open ?variance_issue - variance_issue_record)
    (entity_schedule_linked_to_issue ?legal_entity_schedule - legal_entity_schedule ?variance_issue - variance_issue_record)
    (business_unit_schedule_linked_to_issue ?business_unit_schedule - business_unit_schedule ?variance_issue - variance_issue_record)
    (entity_schedule_linked_to_control_point ?legal_entity_schedule - legal_entity_schedule ?control_point - control_point)
    (control_point_valid ?control_point - control_point)
    (control_point_flagged ?control_point - control_point)
    (control_validation_complete_for_entity ?legal_entity_schedule - legal_entity_schedule)
    (business_unit_schedule_linked_to_disclosure ?business_unit_schedule - business_unit_schedule ?disclosure_item - disclosure_item)
    (disclosure_valid ?disclosure_item - disclosure_item)
    (disclosure_flagged ?disclosure_item - disclosure_item)
    (disclosure_validation_complete_for_bu ?business_unit_schedule - business_unit_schedule)
    (section_draft_slot_available ?section_draft - section_draft)
    (section_draft_reserved ?section_draft - section_draft)
    (section_draft_linked_to_control_point ?section_draft - section_draft ?control_point - control_point)
    (section_draft_linked_to_disclosure ?section_draft - section_draft ?disclosure_item - disclosure_item)
    (section_draft_ready_for_template ?section_draft - section_draft)
    (section_draft_ready_for_presentation ?section_draft - section_draft)
    (section_draft_supporting_files_validated ?section_draft - section_draft)
    (pack_contains_entity_schedule ?reporting_pack - reporting_pack ?legal_entity_schedule - legal_entity_schedule)
    (pack_contains_business_unit_schedule ?reporting_pack - reporting_pack ?business_unit_schedule - business_unit_schedule)
    (pack_contains_section_draft ?reporting_pack - reporting_pack ?section_draft - section_draft)
    (supporting_file_available ?supporting_file - supporting_file)
    (supporting_file_linked_to_pack ?reporting_pack - reporting_pack ?supporting_file - supporting_file)
    (supporting_file_validated ?supporting_file - supporting_file)
    (supporting_file_bound_to_section_draft ?supporting_file - supporting_file ?section_draft - section_draft)
    (pack_attachments_validated ?reporting_pack - reporting_pack)
    (pack_template_applied ?reporting_pack - reporting_pack)
    (pack_ready_for_approval ?reporting_pack - reporting_pack)
    (approver_assigned ?reporting_pack - reporting_pack)
    (pack_template_confirmation ?reporting_pack - reporting_pack)
    (pack_presentation_confirmed ?reporting_pack - reporting_pack)
    (pack_ready_for_final_checks ?reporting_pack - reporting_pack)
    (external_reviewer_available ?external_reviewer - external_reviewer_organization)
    (pack_linked_to_external_reviewer ?reporting_pack - reporting_pack ?external_reviewer - external_reviewer_organization)
    (approver_engaged ?reporting_pack - reporting_pack)
    (approver_stage_one_complete ?reporting_pack - reporting_pack)
    (approver_stage_two_complete ?reporting_pack - reporting_pack)
    (approver_role_available ?approver_role - approver_role)
    (pack_linked_to_approver_role ?reporting_pack - reporting_pack ?approver_role - approver_role)
    (presentation_template_available ?presentation_template - presentation_template)
    (pack_linked_to_presentation_template ?reporting_pack - reporting_pack ?presentation_template - presentation_template)
    (accounting_policy_available ?accounting_policy - accounting_policy)
    (pack_linked_to_accounting_policy ?reporting_pack - reporting_pack ?accounting_policy - accounting_policy)
    (audit_partner_available ?audit_partner - audit_partner)
    (pack_linked_to_audit_partner ?reporting_pack - reporting_pack ?audit_partner - audit_partner)
    (audit_query_open ?audit_query - audit_query)
    (linked_to_audit_query ?schedule - schedule ?audit_query - audit_query)
    (entity_control_passed ?legal_entity_schedule - legal_entity_schedule)
    (business_unit_disclosure_passed ?business_unit_schedule - business_unit_schedule)
    (pack_finalized ?reporting_pack - reporting_pack)
  )
  (:action mark_schedule_in_scope
    :parameters (?schedule - schedule)
    :precondition
      (and
        (not
          (schedule_in_scope ?schedule)
        )
        (not
          (included_in_pack ?schedule)
        )
      )
    :effect (schedule_in_scope ?schedule)
  )
  (:action assign_preparer_to_schedule
    :parameters (?schedule - schedule ?preparer - preparer)
    :precondition
      (and
        (schedule_in_scope ?schedule)
        (not
          (preparer_assigned ?schedule)
        )
        (preparer_available ?preparer)
      )
    :effect
      (and
        (preparer_assigned ?schedule)
        (assigned_to_preparer ?schedule ?preparer)
        (not
          (preparer_available ?preparer)
        )
      )
  )
  (:action attach_data_extract_to_schedule
    :parameters (?schedule - schedule ?data_extract - data_extract)
    :precondition
      (and
        (schedule_in_scope ?schedule)
        (preparer_assigned ?schedule)
        (data_extract_available ?data_extract)
      )
    :effect
      (and
        (linked_to_data_extract ?schedule ?data_extract)
        (not
          (data_extract_available ?data_extract)
        )
      )
  )
  (:action mark_schedule_prepared
    :parameters (?schedule - schedule ?data_extract - data_extract)
    :precondition
      (and
        (schedule_in_scope ?schedule)
        (preparer_assigned ?schedule)
        (linked_to_data_extract ?schedule ?data_extract)
        (not
          (is_prepared ?schedule)
        )
      )
    :effect (is_prepared ?schedule)
  )
  (:action release_data_extract_from_schedule
    :parameters (?schedule - schedule ?data_extract - data_extract)
    :precondition
      (and
        (linked_to_data_extract ?schedule ?data_extract)
      )
    :effect
      (and
        (data_extract_available ?data_extract)
        (not
          (linked_to_data_extract ?schedule ?data_extract)
        )
      )
  )
  (:action assign_reviewer_to_schedule
    :parameters (?schedule - schedule ?reviewer - reviewer)
    :precondition
      (and
        (is_prepared ?schedule)
        (reviewer_available ?reviewer)
      )
    :effect
      (and
        (assigned_to_reviewer ?schedule ?reviewer)
        (not
          (reviewer_available ?reviewer)
        )
      )
  )
  (:action unassign_reviewer_from_schedule
    :parameters (?schedule - schedule ?reviewer - reviewer)
    :precondition
      (and
        (assigned_to_reviewer ?schedule ?reviewer)
      )
    :effect
      (and
        (reviewer_available ?reviewer)
        (not
          (assigned_to_reviewer ?schedule ?reviewer)
        )
      )
  )
  (:action assign_accounting_policy_to_pack
    :parameters (?reporting_pack - reporting_pack ?accounting_policy - accounting_policy)
    :precondition
      (and
        (is_prepared ?reporting_pack)
        (accounting_policy_available ?accounting_policy)
      )
    :effect
      (and
        (pack_linked_to_accounting_policy ?reporting_pack ?accounting_policy)
        (not
          (accounting_policy_available ?accounting_policy)
        )
      )
  )
  (:action unassign_accounting_policy_from_pack
    :parameters (?reporting_pack - reporting_pack ?accounting_policy - accounting_policy)
    :precondition
      (and
        (pack_linked_to_accounting_policy ?reporting_pack ?accounting_policy)
      )
    :effect
      (and
        (accounting_policy_available ?accounting_policy)
        (not
          (pack_linked_to_accounting_policy ?reporting_pack ?accounting_policy)
        )
      )
  )
  (:action assign_audit_partner_to_pack
    :parameters (?reporting_pack - reporting_pack ?audit_partner - audit_partner)
    :precondition
      (and
        (is_prepared ?reporting_pack)
        (audit_partner_available ?audit_partner)
      )
    :effect
      (and
        (pack_linked_to_audit_partner ?reporting_pack ?audit_partner)
        (not
          (audit_partner_available ?audit_partner)
        )
      )
  )
  (:action unassign_audit_partner_from_pack
    :parameters (?reporting_pack - reporting_pack ?audit_partner - audit_partner)
    :precondition
      (and
        (pack_linked_to_audit_partner ?reporting_pack ?audit_partner)
      )
    :effect
      (and
        (audit_partner_available ?audit_partner)
        (not
          (pack_linked_to_audit_partner ?reporting_pack ?audit_partner)
        )
      )
  )
  (:action validate_control_point_for_entity
    :parameters (?legal_entity_schedule - legal_entity_schedule ?control_point - control_point ?data_extract - data_extract)
    :precondition
      (and
        (is_prepared ?legal_entity_schedule)
        (linked_to_data_extract ?legal_entity_schedule ?data_extract)
        (entity_schedule_linked_to_control_point ?legal_entity_schedule ?control_point)
        (not
          (control_point_valid ?control_point)
        )
        (not
          (control_point_flagged ?control_point)
        )
      )
    :effect (control_point_valid ?control_point)
  )
  (:action complete_entity_control_validation
    :parameters (?legal_entity_schedule - legal_entity_schedule ?control_point - control_point ?reviewer - reviewer)
    :precondition
      (and
        (is_prepared ?legal_entity_schedule)
        (assigned_to_reviewer ?legal_entity_schedule ?reviewer)
        (entity_schedule_linked_to_control_point ?legal_entity_schedule ?control_point)
        (control_point_valid ?control_point)
        (not
          (entity_control_passed ?legal_entity_schedule)
        )
      )
    :effect
      (and
        (entity_control_passed ?legal_entity_schedule)
        (control_validation_complete_for_entity ?legal_entity_schedule)
      )
  )
  (:action record_variance_issue_for_entity_schedule
    :parameters (?legal_entity_schedule - legal_entity_schedule ?control_point - control_point ?variance_issue - variance_issue_record)
    :precondition
      (and
        (is_prepared ?legal_entity_schedule)
        (entity_schedule_linked_to_control_point ?legal_entity_schedule ?control_point)
        (variance_issue_open ?variance_issue)
        (not
          (entity_control_passed ?legal_entity_schedule)
        )
      )
    :effect
      (and
        (control_point_flagged ?control_point)
        (entity_control_passed ?legal_entity_schedule)
        (entity_schedule_linked_to_issue ?legal_entity_schedule ?variance_issue)
        (not
          (variance_issue_open ?variance_issue)
        )
      )
  )
  (:action resolve_variance_issue_for_entity_schedule
    :parameters (?legal_entity_schedule - legal_entity_schedule ?control_point - control_point ?data_extract - data_extract ?variance_issue - variance_issue_record)
    :precondition
      (and
        (is_prepared ?legal_entity_schedule)
        (linked_to_data_extract ?legal_entity_schedule ?data_extract)
        (entity_schedule_linked_to_control_point ?legal_entity_schedule ?control_point)
        (control_point_flagged ?control_point)
        (entity_schedule_linked_to_issue ?legal_entity_schedule ?variance_issue)
        (not
          (control_validation_complete_for_entity ?legal_entity_schedule)
        )
      )
    :effect
      (and
        (control_point_valid ?control_point)
        (control_validation_complete_for_entity ?legal_entity_schedule)
        (variance_issue_open ?variance_issue)
        (not
          (entity_schedule_linked_to_issue ?legal_entity_schedule ?variance_issue)
        )
      )
  )
  (:action validate_disclosure_for_business_unit_schedule
    :parameters (?business_unit_schedule - business_unit_schedule ?disclosure_item - disclosure_item ?data_extract - data_extract)
    :precondition
      (and
        (is_prepared ?business_unit_schedule)
        (linked_to_data_extract ?business_unit_schedule ?data_extract)
        (business_unit_schedule_linked_to_disclosure ?business_unit_schedule ?disclosure_item)
        (not
          (disclosure_valid ?disclosure_item)
        )
        (not
          (disclosure_flagged ?disclosure_item)
        )
      )
    :effect (disclosure_valid ?disclosure_item)
  )
  (:action complete_disclosure_review_for_business_unit
    :parameters (?business_unit_schedule - business_unit_schedule ?disclosure_item - disclosure_item ?reviewer - reviewer)
    :precondition
      (and
        (is_prepared ?business_unit_schedule)
        (assigned_to_reviewer ?business_unit_schedule ?reviewer)
        (business_unit_schedule_linked_to_disclosure ?business_unit_schedule ?disclosure_item)
        (disclosure_valid ?disclosure_item)
        (not
          (business_unit_disclosure_passed ?business_unit_schedule)
        )
      )
    :effect
      (and
        (business_unit_disclosure_passed ?business_unit_schedule)
        (disclosure_validation_complete_for_bu ?business_unit_schedule)
      )
  )
  (:action record_variance_issue_for_business_unit_schedule
    :parameters (?business_unit_schedule - business_unit_schedule ?disclosure_item - disclosure_item ?variance_issue - variance_issue_record)
    :precondition
      (and
        (is_prepared ?business_unit_schedule)
        (business_unit_schedule_linked_to_disclosure ?business_unit_schedule ?disclosure_item)
        (variance_issue_open ?variance_issue)
        (not
          (business_unit_disclosure_passed ?business_unit_schedule)
        )
      )
    :effect
      (and
        (disclosure_flagged ?disclosure_item)
        (business_unit_disclosure_passed ?business_unit_schedule)
        (business_unit_schedule_linked_to_issue ?business_unit_schedule ?variance_issue)
        (not
          (variance_issue_open ?variance_issue)
        )
      )
  )
  (:action resolve_variance_issue_for_business_unit_schedule
    :parameters (?business_unit_schedule - business_unit_schedule ?disclosure_item - disclosure_item ?data_extract - data_extract ?variance_issue - variance_issue_record)
    :precondition
      (and
        (is_prepared ?business_unit_schedule)
        (linked_to_data_extract ?business_unit_schedule ?data_extract)
        (business_unit_schedule_linked_to_disclosure ?business_unit_schedule ?disclosure_item)
        (disclosure_flagged ?disclosure_item)
        (business_unit_schedule_linked_to_issue ?business_unit_schedule ?variance_issue)
        (not
          (disclosure_validation_complete_for_bu ?business_unit_schedule)
        )
      )
    :effect
      (and
        (disclosure_valid ?disclosure_item)
        (disclosure_validation_complete_for_bu ?business_unit_schedule)
        (variance_issue_open ?variance_issue)
        (not
          (business_unit_schedule_linked_to_issue ?business_unit_schedule ?variance_issue)
        )
      )
  )
  (:action assemble_section_draft
    :parameters (?legal_entity_schedule - legal_entity_schedule ?business_unit_schedule - business_unit_schedule ?control_point - control_point ?disclosure_item - disclosure_item ?section_draft - section_draft)
    :precondition
      (and
        (entity_control_passed ?legal_entity_schedule)
        (business_unit_disclosure_passed ?business_unit_schedule)
        (entity_schedule_linked_to_control_point ?legal_entity_schedule ?control_point)
        (business_unit_schedule_linked_to_disclosure ?business_unit_schedule ?disclosure_item)
        (control_point_valid ?control_point)
        (disclosure_valid ?disclosure_item)
        (control_validation_complete_for_entity ?legal_entity_schedule)
        (disclosure_validation_complete_for_bu ?business_unit_schedule)
        (section_draft_slot_available ?section_draft)
      )
    :effect
      (and
        (section_draft_reserved ?section_draft)
        (section_draft_linked_to_control_point ?section_draft ?control_point)
        (section_draft_linked_to_disclosure ?section_draft ?disclosure_item)
        (not
          (section_draft_slot_available ?section_draft)
        )
      )
  )
  (:action assemble_section_draft_control_flagged
    :parameters (?legal_entity_schedule - legal_entity_schedule ?business_unit_schedule - business_unit_schedule ?control_point - control_point ?disclosure_item - disclosure_item ?section_draft - section_draft)
    :precondition
      (and
        (entity_control_passed ?legal_entity_schedule)
        (business_unit_disclosure_passed ?business_unit_schedule)
        (entity_schedule_linked_to_control_point ?legal_entity_schedule ?control_point)
        (business_unit_schedule_linked_to_disclosure ?business_unit_schedule ?disclosure_item)
        (control_point_flagged ?control_point)
        (disclosure_valid ?disclosure_item)
        (not
          (control_validation_complete_for_entity ?legal_entity_schedule)
        )
        (disclosure_validation_complete_for_bu ?business_unit_schedule)
        (section_draft_slot_available ?section_draft)
      )
    :effect
      (and
        (section_draft_reserved ?section_draft)
        (section_draft_linked_to_control_point ?section_draft ?control_point)
        (section_draft_linked_to_disclosure ?section_draft ?disclosure_item)
        (section_draft_ready_for_template ?section_draft)
        (not
          (section_draft_slot_available ?section_draft)
        )
      )
  )
  (:action assemble_section_draft_disclosure_flagged
    :parameters (?legal_entity_schedule - legal_entity_schedule ?business_unit_schedule - business_unit_schedule ?control_point - control_point ?disclosure_item - disclosure_item ?section_draft - section_draft)
    :precondition
      (and
        (entity_control_passed ?legal_entity_schedule)
        (business_unit_disclosure_passed ?business_unit_schedule)
        (entity_schedule_linked_to_control_point ?legal_entity_schedule ?control_point)
        (business_unit_schedule_linked_to_disclosure ?business_unit_schedule ?disclosure_item)
        (control_point_valid ?control_point)
        (disclosure_flagged ?disclosure_item)
        (control_validation_complete_for_entity ?legal_entity_schedule)
        (not
          (disclosure_validation_complete_for_bu ?business_unit_schedule)
        )
        (section_draft_slot_available ?section_draft)
      )
    :effect
      (and
        (section_draft_reserved ?section_draft)
        (section_draft_linked_to_control_point ?section_draft ?control_point)
        (section_draft_linked_to_disclosure ?section_draft ?disclosure_item)
        (section_draft_ready_for_presentation ?section_draft)
        (not
          (section_draft_slot_available ?section_draft)
        )
      )
  )
  (:action assemble_section_draft_with_issues
    :parameters (?legal_entity_schedule - legal_entity_schedule ?business_unit_schedule - business_unit_schedule ?control_point - control_point ?disclosure_item - disclosure_item ?section_draft - section_draft)
    :precondition
      (and
        (entity_control_passed ?legal_entity_schedule)
        (business_unit_disclosure_passed ?business_unit_schedule)
        (entity_schedule_linked_to_control_point ?legal_entity_schedule ?control_point)
        (business_unit_schedule_linked_to_disclosure ?business_unit_schedule ?disclosure_item)
        (control_point_flagged ?control_point)
        (disclosure_flagged ?disclosure_item)
        (not
          (control_validation_complete_for_entity ?legal_entity_schedule)
        )
        (not
          (disclosure_validation_complete_for_bu ?business_unit_schedule)
        )
        (section_draft_slot_available ?section_draft)
      )
    :effect
      (and
        (section_draft_reserved ?section_draft)
        (section_draft_linked_to_control_point ?section_draft ?control_point)
        (section_draft_linked_to_disclosure ?section_draft ?disclosure_item)
        (section_draft_ready_for_template ?section_draft)
        (section_draft_ready_for_presentation ?section_draft)
        (not
          (section_draft_slot_available ?section_draft)
        )
      )
  )
  (:action validate_section_draft_supporting_files
    :parameters (?section_draft - section_draft ?legal_entity_schedule - legal_entity_schedule ?data_extract - data_extract)
    :precondition
      (and
        (section_draft_reserved ?section_draft)
        (entity_control_passed ?legal_entity_schedule)
        (linked_to_data_extract ?legal_entity_schedule ?data_extract)
        (not
          (section_draft_supporting_files_validated ?section_draft)
        )
      )
    :effect (section_draft_supporting_files_validated ?section_draft)
  )
  (:action attach_supporting_file_to_section_draft_and_pack
    :parameters (?reporting_pack - reporting_pack ?supporting_file - supporting_file ?section_draft - section_draft)
    :precondition
      (and
        (is_prepared ?reporting_pack)
        (pack_contains_section_draft ?reporting_pack ?section_draft)
        (supporting_file_linked_to_pack ?reporting_pack ?supporting_file)
        (supporting_file_available ?supporting_file)
        (section_draft_reserved ?section_draft)
        (section_draft_supporting_files_validated ?section_draft)
        (not
          (supporting_file_validated ?supporting_file)
        )
      )
    :effect
      (and
        (supporting_file_validated ?supporting_file)
        (supporting_file_bound_to_section_draft ?supporting_file ?section_draft)
        (not
          (supporting_file_available ?supporting_file)
        )
      )
  )
  (:action validate_supporting_file_and_mark_pack_attachments
    :parameters (?reporting_pack - reporting_pack ?supporting_file - supporting_file ?section_draft - section_draft ?data_extract - data_extract)
    :precondition
      (and
        (is_prepared ?reporting_pack)
        (supporting_file_linked_to_pack ?reporting_pack ?supporting_file)
        (supporting_file_validated ?supporting_file)
        (supporting_file_bound_to_section_draft ?supporting_file ?section_draft)
        (linked_to_data_extract ?reporting_pack ?data_extract)
        (not
          (section_draft_ready_for_template ?section_draft)
        )
        (not
          (pack_attachments_validated ?reporting_pack)
        )
      )
    :effect (pack_attachments_validated ?reporting_pack)
  )
  (:action assign_approver_role_to_pack
    :parameters (?reporting_pack - reporting_pack ?approver_role - approver_role)
    :precondition
      (and
        (is_prepared ?reporting_pack)
        (approver_role_available ?approver_role)
        (not
          (approver_assigned ?reporting_pack)
        )
      )
    :effect
      (and
        (approver_assigned ?reporting_pack)
        (pack_linked_to_approver_role ?reporting_pack ?approver_role)
        (not
          (approver_role_available ?approver_role)
        )
      )
  )
  (:action apply_presentation_template_to_pack
    :parameters (?reporting_pack - reporting_pack ?supporting_file - supporting_file ?section_draft - section_draft ?data_extract - data_extract ?approver_role - approver_role)
    :precondition
      (and
        (is_prepared ?reporting_pack)
        (supporting_file_linked_to_pack ?reporting_pack ?supporting_file)
        (supporting_file_validated ?supporting_file)
        (supporting_file_bound_to_section_draft ?supporting_file ?section_draft)
        (linked_to_data_extract ?reporting_pack ?data_extract)
        (section_draft_ready_for_template ?section_draft)
        (approver_assigned ?reporting_pack)
        (pack_linked_to_approver_role ?reporting_pack ?approver_role)
        (not
          (pack_attachments_validated ?reporting_pack)
        )
      )
    :effect
      (and
        (pack_attachments_validated ?reporting_pack)
        (pack_template_confirmation ?reporting_pack)
      )
  )
  (:action start_presentation_review_for_pack
    :parameters (?reporting_pack - reporting_pack ?accounting_policy - accounting_policy ?reviewer - reviewer ?supporting_file - supporting_file ?section_draft - section_draft)
    :precondition
      (and
        (pack_attachments_validated ?reporting_pack)
        (pack_linked_to_accounting_policy ?reporting_pack ?accounting_policy)
        (assigned_to_reviewer ?reporting_pack ?reviewer)
        (supporting_file_linked_to_pack ?reporting_pack ?supporting_file)
        (supporting_file_bound_to_section_draft ?supporting_file ?section_draft)
        (not
          (section_draft_ready_for_presentation ?section_draft)
        )
        (not
          (pack_template_applied ?reporting_pack)
        )
      )
