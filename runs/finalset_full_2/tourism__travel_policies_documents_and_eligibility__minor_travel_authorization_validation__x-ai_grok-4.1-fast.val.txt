  (:action create_application_record
    :parameters (?case_subject - case_entity)
    :precondition
      (and
        (not
          (subject_application_record_created ?case_subject)
        )
        (not
          (verifier_assignment_confirmed ?case_subject)
        )
      )
    :effect (subject_application_record_created ?case_subject)
  )
  (:action assign_verifier_agent
    :parameters (?case_subject - case_entity ?verifier_agent - verifier_agent)
    :precondition
      (and
        (subject_application_record_created ?case_subject)
        (not
          (verifier_reserved_for_case_entity ?case_subject)
        )
        (verifier_agent_available ?verifier_agent)
      )
    :effect
      (and
        (verifier_reserved_for_case_entity ?case_subject)
        (assigned_verifier ?case_subject ?verifier_agent)
        (not
          (verifier_agent_available ?verifier_agent)
        )
      )
  )
  (:action submit_document_for_application
    :parameters (?case_subject - case_entity ?document_type - document_type)
    :precondition
      (and
        (subject_application_record_created ?case_subject)
        (verifier_reserved_for_case_entity ?case_subject)
        (submitted_document_available ?document_type)
      )
    :effect
      (and
        (document_attached ?case_subject ?document_type)
        (not
          (submitted_document_available ?document_type)
        )
      )
  )
  (:action confirm_document_verification
    :parameters (?case_subject - case_entity ?document_type - document_type)
    :precondition
      (and
        (subject_application_record_created ?case_subject)
        (verifier_reserved_for_case_entity ?case_subject)
        (document_attached ?case_subject ?document_type)
        (not
          (primary_verification_complete ?case_subject)
        )
      )
    :effect (primary_verification_complete ?case_subject)
  )
  (:action release_document_attachment
    :parameters (?case_subject - case_entity ?document_type - document_type)
    :precondition
      (and
        (document_attached ?case_subject ?document_type)
      )
    :effect
      (and
        (submitted_document_available ?document_type)
        (not
          (document_attached ?case_subject ?document_type)
        )
      )
  )
  (:action attach_certifying_witness
    :parameters (?case_subject - case_entity ?certifying_witness - certifying_witness)
    :precondition
      (and
        (primary_verification_complete ?case_subject)
        (witness_available ?certifying_witness)
      )
    :effect
      (and
        (certifying_witness_attached ?case_subject ?certifying_witness)
        (not
          (witness_available ?certifying_witness)
        )
      )
  )
  (:action detach_certifying_witness
    :parameters (?case_subject - case_entity ?certifying_witness - certifying_witness)
    :precondition
      (and
        (certifying_witness_attached ?case_subject ?certifying_witness)
      )
    :effect
      (and
        (witness_available ?certifying_witness)
        (not
          (certifying_witness_attached ?case_subject ?certifying_witness)
        )
      )
  )

          (requirement_processing_mark ?minor_traveler)

          (entry_requirement_satisfied_for_traveler ?minor_traveler)

        (entry_requirement_confirmed ?entry_requirement)
        (entry_requirement_satisfied_for_traveler ?minor_traveler)

          (guardian_requirement_processing_mark ?guardian)
        )

        (fare_rule_confirmed_by_consent ?fare_category_rule)

          (consent_document_available ?consent_document)

  (:action resolve_fare_rule

        (fare_rule_confirmed ?fare_category_rule)

      (and
        (requirement_processing_mark ?minor_traveler)

        (fare_rule_applies ?guardian ?fare_category_rule)
        (entry_requirement_confirmed ?entry_requirement)
        (fare_rule_confirmed ?fare_category_rule)
        (entry_requirement_satisfied_for_traveler ?minor_traveler)
        (fare_requirement_satisfied_for_guardian ?guardian)
        (product_available ?travel_product)
      )
    :effect
      (and
        (product_evaluated ?travel_product)
        (product_linked_to_entry_requirement ?travel_product ?entry_requirement)
        (product_linked_to_fare_rule ?travel_product ?fare_category_rule)
        (not
          (product_available ?travel_product)
        )
      )
  )
  (:action select_and_link_travel_product_entry_by_consent

    :effect
      (and
        (product_evaluated ?travel_product)

        (product_linked_to_fare_rule ?travel_product ?fare_category_rule)

        (not
          (product_available ?travel_product)
        )
      )
  )
  (:action select_and_link_travel_product_fare_by_consent

      (and
        (issuer_checks_passed ?authorization_application)

          (health_certificate_associated ?authorization_application)
