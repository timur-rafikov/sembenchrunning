--- domain.pddl (as passed to VAL) ---
(define (domain equipment_durability_maintenance)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_category - object resource_unit_category - object workstation_or_activity_category - object equipment_category - object equipment_unit - equipment_category repair_supply_item - resource_category repair_tool - resource_category maintenance_technician - resource_category calibration_kit - resource_category support_supply - resource_category pre_deployment_supply - resource_category upgrade_module - resource_category verification_tool - resource_category replacement_part - resource_unit_category upgrade_component - resource_unit_category inspection_tag - resource_unit_category maintenance_job_primary - workstation_or_activity_category maintenance_job_secondary - workstation_or_activity_category repair_bay - workstation_or_activity_category unit_category_primary - equipment_unit unit_category_module - equipment_unit primary_unit - unit_category_primary secondary_unit - unit_category_primary assembly_module - unit_category_module)
  (:predicates
    (asset_registered_for_maintenance ?equipment_unit - equipment_unit)
    (asset_accepted_for_maintenance ?equipment_unit - equipment_unit)
    (asset_supply_assigned ?equipment_unit - equipment_unit)
    (asset_validated_for_deployment ?equipment_unit - equipment_unit)
    (asset_ready_for_predeployment ?equipment_unit - equipment_unit)
    (asset_preserved_for_deployment ?equipment_unit - equipment_unit)
    (repair_supply_available ?repair_supply_item - repair_supply_item)
    (asset_assigned_supply ?equipment_unit - equipment_unit ?repair_supply_item - repair_supply_item)
    (repair_tool_available ?repair_tool - repair_tool)
    (asset_assigned_tool ?equipment_unit - equipment_unit ?repair_tool - repair_tool)
    (technician_available ?technician - maintenance_technician)
    (asset_assigned_technician ?equipment_unit - equipment_unit ?technician - maintenance_technician)
    (replacement_part_available ?replacement_part - replacement_part)
    (primary_unit_attached_replacement_part ?primary_unit - primary_unit ?replacement_part - replacement_part)
    (secondary_unit_attached_replacement_part ?secondary_unit - secondary_unit ?replacement_part - replacement_part)
    (primary_unit_assigned_task ?primary_unit - primary_unit ?maintenance_task_primary - maintenance_job_primary)
    (maintenance_task_primary_ready ?maintenance_task_primary - maintenance_job_primary)
    (maintenance_task_primary_staged_with_part ?maintenance_task_primary - maintenance_job_primary)
    (primary_unit_ready_for_bay ?primary_unit - primary_unit)
    (secondary_unit_assigned_task ?secondary_unit - secondary_unit ?maintenance_task_secondary - maintenance_job_secondary)
    (maintenance_task_secondary_ready ?maintenance_task_secondary - maintenance_job_secondary)
    (maintenance_task_secondary_staged_with_part ?maintenance_task_secondary - maintenance_job_secondary)
    (secondary_unit_ready_for_bay ?secondary_unit - secondary_unit)
    (repair_bay_available ?repair_bay - repair_bay)
    (repair_bay_reserved ?repair_bay - repair_bay)
    (repair_bay_assigned_primary_task ?repair_bay - repair_bay ?maintenance_task_primary - maintenance_job_primary)
    (repair_bay_assigned_secondary_task ?repair_bay - repair_bay ?maintenance_task_secondary - maintenance_job_secondary)
    (repair_bay_flag_calibration_required ?repair_bay - repair_bay)
    (repair_bay_flag_additional_validation_required ?repair_bay - repair_bay)
    (repair_bay_prepped_for_assembly ?repair_bay - repair_bay)
    (module_attached_to_primary_unit ?assembly_module - assembly_module ?primary_unit - primary_unit)
    (module_attached_to_secondary_unit ?assembly_module - assembly_module ?secondary_unit - secondary_unit)
    (module_assigned_to_repair_bay ?assembly_module - assembly_module ?repair_bay - repair_bay)
    (upgrade_component_available ?upgrade_component - upgrade_component)
    (module_has_upgrade_component ?assembly_module - assembly_module ?upgrade_component - upgrade_component)
    (upgrade_component_in_use ?upgrade_component - upgrade_component)
    (upgrade_component_assigned_to_bay ?upgrade_component - upgrade_component ?repair_bay - repair_bay)
    (module_staged_for_upgrade ?assembly_module - assembly_module)
    (module_ready_for_validation ?assembly_module - assembly_module)
    (module_validation_passed ?assembly_module - assembly_module)
    (module_attached_calibration_kit ?assembly_module - assembly_module)
    (module_marked_for_calibration_review ?assembly_module - assembly_module)
    (module_support_ready ?assembly_module - assembly_module)
    (module_integration_complete ?assembly_module - assembly_module)
    (inspection_tag_available ?inspection_tag - inspection_tag)
    (module_assigned_inspection_tag ?assembly_module - assembly_module ?inspection_tag - inspection_tag)
    (module_inspection_tag_locked ?assembly_module - assembly_module)
    (module_inspected ?assembly_module - assembly_module)
    (module_inspection_validated ?assembly_module - assembly_module)
    (calibration_kit_available ?calibration_kit - calibration_kit)
    (module_assigned_calibration_kit ?assembly_module - assembly_module ?calibration_kit - calibration_kit)
    (support_supply_available ?support_supply - support_supply)
    (module_assigned_support_supply ?assembly_module - assembly_module ?support_supply - support_supply)
    (upgrade_module_available ?upgrade_module - upgrade_module)
    (module_assigned_upgrade_module ?assembly_module - assembly_module ?upgrade_module - upgrade_module)
    (verification_tool_available ?verification_tool - verification_tool)
    (module_assigned_verification_tool ?assembly_module - assembly_module ?verification_tool - verification_tool)
    (pre_deployment_supply_available ?pre_deployment_supply - pre_deployment_supply)
    (asset_assigned_pre_deployment_supply ?equipment_unit - equipment_unit ?pre_deployment_supply - pre_deployment_supply)
    (primary_unit_task_ready ?primary_unit - primary_unit)
    (secondary_unit_task_ready ?secondary_unit - secondary_unit)
    (module_finalized ?assembly_module - assembly_module)
  )
  (:action register_asset_for_maintenance
    :parameters (?equipment_unit - equipment_unit)
    :precondition
      (and
        (not
          (asset_registered_for_maintenance ?equipment_unit)
        )
        (not
          (asset_validated_for_deployment ?equipment_unit)
        )
      )
    :effect (asset_registered_for_maintenance ?equipment_unit)
  )
  (:action allocate_repair_supply_to_asset
    :parameters (?equipment_unit - equipment_unit ?repair_supply_item - repair_supply_item)
    :precondition
      (and
        (asset_registered_for_maintenance ?equipment_unit)
        (not
          (asset_supply_assigned ?equipment_unit)
        )
        (repair_supply_available ?repair_supply_item)
      )
    :effect
      (and
        (asset_supply_assigned ?equipment_unit)
        (asset_assigned_supply ?equipment_unit ?repair_supply_item)
        (not
          (repair_supply_available ?repair_supply_item)
        )
      )
  )
  (:action assign_repair_tool_to_asset
    :parameters (?equipment_unit - equipment_unit ?repair_tool - repair_tool)
    :precondition
      (and
        (asset_registered_for_maintenance ?equipment_unit)
        (asset_supply_assigned ?equipment_unit)
        (repair_tool_available ?repair_tool)
      )
    :effect
      (and
        (asset_assigned_tool ?equipment_unit ?repair_tool)
        (not
          (repair_tool_available ?repair_tool)
        )
      )
  )
  (:action accept_asset_for_maintenance
    :parameters (?equipment_unit - equipment_unit ?repair_tool - repair_tool)
    :precondition
      (and
        (asset_registered_for_maintenance ?equipment_unit)
        (asset_supply_assigned ?equipment_unit)
        (asset_assigned_tool ?equipment_unit ?repair_tool)
        (not
          (asset_accepted_for_maintenance ?equipment_unit)
        )
      )
    :effect (asset_accepted_for_maintenance ?equipment_unit)
  )
  (:action return_repair_tool_from_asset
    :parameters (?equipment_unit - equipment_unit ?repair_tool - repair_tool)
    :precondition
      (and
        (asset_assigned_tool ?equipment_unit ?repair_tool)
      )
    :effect
      (and
        (repair_tool_available ?repair_tool)
        (not
          (asset_assigned_tool ?equipment_unit ?repair_tool)
        )
      )
  )
  (:action allocate_technician_to_asset
    :parameters (?equipment_unit - equipment_unit ?technician - maintenance_technician)
    :precondition
      (and
        (asset_accepted_for_maintenance ?equipment_unit)
        (technician_available ?technician)
      )
    :effect
      (and
        (asset_assigned_technician ?equipment_unit ?technician)
        (not
          (technician_available ?technician)
        )
      )
  )
  (:action release_technician_from_asset
    :parameters (?equipment_unit - equipment_unit ?technician - maintenance_technician)
    :precondition
      (and
        (asset_assigned_technician ?equipment_unit ?technician)
      )
    :effect
      (and
        (technician_available ?technician)
        (not
          (asset_assigned_technician ?equipment_unit ?technician)
        )
      )
  )
  (:action attach_upgrade_module_to_module
    :parameters (?assembly_module - assembly_module ?upgrade_module - upgrade_module)
    :precondition
      (and
        (asset_accepted_for_maintenance ?assembly_module)
        (upgrade_module_available ?upgrade_module)
      )
    :effect
      (and
        (module_assigned_upgrade_module ?assembly_module ?upgrade_module)
        (not
          (upgrade_module_available ?upgrade_module)
        )
      )
  )
  (:action detach_upgrade_module_from_module
    :parameters (?assembly_module - assembly_module ?upgrade_module - upgrade_module)
    :precondition
      (and
        (module_assigned_upgrade_module ?assembly_module ?upgrade_module)
      )
    :effect
      (and
        (upgrade_module_available ?upgrade_module)
        (not
          (module_assigned_upgrade_module ?assembly_module ?upgrade_module)
        )
      )
  )
  (:action assign_verification_tool_to_module
    :parameters (?assembly_module - assembly_module ?verification_tool - verification_tool)
    :precondition
      (and
        (asset_accepted_for_maintenance ?assembly_module)
        (verification_tool_available ?verification_tool)
      )
    :effect
      (and
        (module_assigned_verification_tool ?assembly_module ?verification_tool)
        (not
          (verification_tool_available ?verification_tool)
        )
      )
  )
  (:action release_verification_tool_from_module
    :parameters (?assembly_module - assembly_module ?verification_tool - verification_tool)
    :precondition
      (and
        (module_assigned_verification_tool ?assembly_module ?verification_tool)
      )
    :effect
      (and
        (verification_tool_available ?verification_tool)
        (not
          (module_assigned_verification_tool ?assembly_module ?verification_tool)
        )
      )
  )
  (:action prepare_primary_task
    :parameters (?primary_unit - primary_unit ?maintenance_task_primary - maintenance_job_primary ?repair_tool - repair_tool)
    :precondition
      (and
        (asset_accepted_for_maintenance ?primary_unit)
        (asset_assigned_tool ?primary_unit ?repair_tool)
        (primary_unit_assigned_task ?primary_unit ?maintenance_task_primary)
        (not
          (maintenance_task_primary_ready ?maintenance_task_primary)
        )
        (not
          (maintenance_task_primary_staged_with_part ?maintenance_task_primary)
        )
      )
    :effect (maintenance_task_primary_ready ?maintenance_task_primary)
  )
  (:action stage_primary_task_with_technician
    :parameters (?primary_unit - primary_unit ?maintenance_task_primary - maintenance_job_primary ?technician - maintenance_technician)
    :precondition
      (and
        (asset_accepted_for_maintenance ?primary_unit)
        (asset_assigned_technician ?primary_unit ?technician)
        (primary_unit_assigned_task ?primary_unit ?maintenance_task_primary)
        (maintenance_task_primary_ready ?maintenance_task_primary)
        (not
          (primary_unit_task_ready ?primary_unit)
        )
      )
    :effect
      (and
        (primary_unit_task_ready ?primary_unit)
        (primary_unit_ready_for_bay ?primary_unit)
      )
  )
  (:action assign_replacement_part_to_primary_task
    :parameters (?primary_unit - primary_unit ?maintenance_task_primary - maintenance_job_primary ?replacement_part - replacement_part)
    :precondition
      (and
        (asset_accepted_for_maintenance ?primary_unit)
        (primary_unit_assigned_task ?primary_unit ?maintenance_task_primary)
        (replacement_part_available ?replacement_part)
        (not
          (primary_unit_task_ready ?primary_unit)
        )
      )
    :effect
      (and
        (maintenance_task_primary_staged_with_part ?maintenance_task_primary)
        (primary_unit_task_ready ?primary_unit)
        (primary_unit_attached_replacement_part ?primary_unit ?replacement_part)
        (not
          (replacement_part_available ?replacement_part)
        )
      )
  )
  (:action finalize_primary_task_with_part
    :parameters (?primary_unit - primary_unit ?maintenance_task_primary - maintenance_job_primary ?repair_tool - repair_tool ?replacement_part - replacement_part)
    :precondition
      (and
        (asset_accepted_for_maintenance ?primary_unit)
        (asset_assigned_tool ?primary_unit ?repair_tool)
        (primary_unit_assigned_task ?primary_unit ?maintenance_task_primary)
        (maintenance_task_primary_staged_with_part ?maintenance_task_primary)
        (primary_unit_attached_replacement_part ?primary_unit ?replacement_part)
        (not
          (primary_unit_ready_for_bay ?primary_unit)
        )
      )
    :effect
      (and
        (maintenance_task_primary_ready ?maintenance_task_primary)
        (primary_unit_ready_for_bay ?primary_unit)
        (replacement_part_available ?replacement_part)
        (not
          (primary_unit_attached_replacement_part ?primary_unit ?replacement_part)
        )
      )
  )
  (:action prepare_secondary_task
    :parameters (?secondary_unit - secondary_unit ?maintenance_task_secondary - maintenance_job_secondary ?repair_tool - repair_tool)
    :precondition
      (and
        (asset_accepted_for_maintenance ?secondary_unit)
        (asset_assigned_tool ?secondary_unit ?repair_tool)
        (secondary_unit_assigned_task ?secondary_unit ?maintenance_task_secondary)
        (not
          (maintenance_task_secondary_ready ?maintenance_task_secondary)
        )
        (not
          (maintenance_task_secondary_staged_with_part ?maintenance_task_secondary)
        )
      )
    :effect (maintenance_task_secondary_ready ?maintenance_task_secondary)
  )
  (:action stage_secondary_task_with_technician
    :parameters (?secondary_unit - secondary_unit ?maintenance_task_secondary - maintenance_job_secondary ?technician - maintenance_technician)
    :precondition
      (and
        (asset_accepted_for_maintenance ?secondary_unit)
        (asset_assigned_technician ?secondary_unit ?technician)
        (secondary_unit_assigned_task ?secondary_unit ?maintenance_task_secondary)
        (maintenance_task_secondary_ready ?maintenance_task_secondary)
        (not
          (secondary_unit_task_ready ?secondary_unit)
        )
      )
    :effect
      (and
        (secondary_unit_task_ready ?secondary_unit)
        (secondary_unit_ready_for_bay ?secondary_unit)
      )
  )
  (:action assign_replacement_part_to_secondary_task
    :parameters (?secondary_unit - secondary_unit ?maintenance_task_secondary - maintenance_job_secondary ?replacement_part - replacement_part)
    :precondition
      (and
        (asset_accepted_for_maintenance ?secondary_unit)
        (secondary_unit_assigned_task ?secondary_unit ?maintenance_task_secondary)
        (replacement_part_available ?replacement_part)
        (not
          (secondary_unit_task_ready ?secondary_unit)
        )
      )
    :effect
      (and
        (maintenance_task_secondary_staged_with_part ?maintenance_task_secondary)
        (secondary_unit_task_ready ?secondary_unit)
        (secondary_unit_attached_replacement_part ?secondary_unit ?replacement_part)
        (not
          (replacement_part_available ?replacement_part)
        )
      )
  )
  (:action finalize_secondary_task_with_part
    :parameters (?secondary_unit - secondary_unit ?maintenance_task_secondary - maintenance_job_secondary ?repair_tool - repair_tool ?replacement_part - replacement_part)
    :precondition
      (and
        (asset_accepted_for_maintenance ?secondary_unit)
        (asset_assigned_tool ?secondary_unit ?repair_tool)
        (secondary_unit_assigned_task ?secondary_unit ?maintenance_task_secondary)
        (maintenance_task_secondary_staged_with_part ?maintenance_task_secondary)
        (secondary_unit_attached_replacement_part ?secondary_unit ?replacement_part)
        (not
          (secondary_unit_ready_for_bay ?secondary_unit)
        )
      )
    :effect
      (and
        (maintenance_task_secondary_ready ?maintenance_task_secondary)
        (secondary_unit_ready_for_bay ?secondary_unit)
        (replacement_part_available ?replacement_part)
        (not
          (secondary_unit_attached_replacement_part ?secondary_unit ?replacement_part)
        )
      )
  )
  (:action reserve_repair_bay_standard
    :parameters (?primary_unit - primary_unit ?secondary_unit - secondary_unit ?maintenance_task_primary - maintenance_job_primary ?maintenance_task_secondary - maintenance_job_secondary ?repair_bay - repair_bay)
    :precondition
      (and
        (primary_unit_task_ready ?primary_unit)
        (secondary_unit_task_ready ?secondary_unit)
        (primary_unit_assigned_task ?primary_unit ?maintenance_task_primary)
        (secondary_unit_assigned_task ?secondary_unit ?maintenance_task_secondary)
        (maintenance_task_primary_ready ?maintenance_task_primary)
        (maintenance_task_secondary_ready ?maintenance_task_secondary)
        (primary_unit_ready_for_bay ?primary_unit)
        (secondary_unit_ready_for_bay ?secondary_unit)
        (repair_bay_available ?repair_bay)
      )
    :effect
      (and
        (repair_bay_reserved ?repair_bay)
        (repair_bay_assigned_primary_task ?repair_bay ?maintenance_task_primary)
        (repair_bay_assigned_secondary_task ?repair_bay ?maintenance_task_secondary)
        (not
          (repair_bay_available ?repair_bay)
        )
      )
  )
  (:action reserve_repair_bay_set_calibration_flag
    :parameters (?primary_unit - primary_unit ?secondary_unit - secondary_unit ?maintenance_task_primary - maintenance_job_primary ?maintenance_task_secondary - maintenance_job_secondary ?repair_bay - repair_bay)
    :precondition
      (and
        (primary_unit_task_ready ?primary_unit)
        (secondary_unit_task_ready ?secondary_unit)
        (primary_unit_assigned_task ?primary_unit ?maintenance_task_primary)
        (secondary_unit_assigned_task ?secondary_unit ?maintenance_task_secondary)
        (maintenance_task_primary_staged_with_part ?maintenance_task_primary)
        (maintenance_task_secondary_ready ?maintenance_task_secondary)
        (not
          (primary_unit_ready_for_bay ?primary_unit)
        )
        (secondary_unit_ready_for_bay ?secondary_unit)
        (repair_bay_available ?repair_bay)
      )
    :effect
      (and
        (repair_bay_reserved ?repair_bay)
        (repair_bay_assigned_primary_task ?repair_bay ?maintenance_task_primary)
        (repair_bay_assigned_secondary_task ?repair_bay ?maintenance_task_secondary)
        (repair_bay_flag_calibration_required ?repair_bay)
        (not
          (repair_bay_available ?repair_bay)
        )
      )
  )
  (:action reserve_repair_bay_set_validation_flag
    :parameters (?primary_unit - primary_unit ?secondary_unit - secondary_unit ?maintenance_task_primary - maintenance_job_primary ?maintenance_task_secondary - maintenance_job_secondary ?repair_bay - repair_bay)
    :precondition
      (and
        (primary_unit_task_ready ?primary_unit)
        (secondary_unit_task_ready ?secondary_unit)
        (primary_unit_assigned_task ?primary_unit ?maintenance_task_primary)
        (secondary_unit_assigned_task ?secondary_unit ?maintenance_task_secondary)
        (maintenance_task_primary_ready ?maintenance_task_primary)
        (maintenance_task_secondary_staged_with_part ?maintenance_task_secondary)
        (primary_unit_ready_for_bay ?primary_unit)
        (not
          (secondary_unit_ready_for_bay ?secondary_unit)
        )
        (repair_bay_available ?repair_bay)
      )
    :effect
      (and
        (repair_bay_reserved ?repair_bay)
        (repair_bay_assigned_primary_task ?repair_bay ?maintenance_task_primary)
        (repair_bay_assigned_secondary_task ?repair_bay ?maintenance_task_secondary)
        (repair_bay_flag_additional_validation_required ?repair_bay)
        (not
          (repair_bay_available ?repair_bay)
        )
      )
  )
  (:action reserve_repair_bay_set_calibration_and_validation_flags
    :parameters (?primary_unit - primary_unit ?secondary_unit - secondary_unit ?maintenance_task_primary - maintenance_job_primary ?maintenance_task_secondary - maintenance_job_secondary ?repair_bay - repair_bay)
    :precondition
      (and
        (primary_unit_task_ready ?primary_unit)
        (secondary_unit_task_ready ?secondary_unit)
        (primary_unit_assigned_task ?primary_unit ?maintenance_task_primary)
        (secondary_unit_assigned_task ?secondary_unit ?maintenance_task_secondary)
        (maintenance_task_primary_staged_with_part ?maintenance_task_primary)
        (maintenance_task_secondary_staged_with_part ?maintenance_task_secondary)
        (not
          (primary_unit_ready_for_bay ?primary_unit)
        )
        (not
          (secondary_unit_ready_for_bay ?secondary_unit)
        )
        (repair_bay_available ?repair_bay)
      )
    :effect
      (and
        (repair_bay_reserved ?repair_bay)
        (repair_bay_assigned_primary_task ?repair_bay ?maintenance_task_primary)
        (repair_bay_assigned_secondary_task ?repair_bay ?maintenance_task_secondary)
        (repair_bay_flag_calibration_required ?repair_bay)
        (repair_bay_flag_additional_validation_required ?repair_bay)
        (not
          (repair_bay_available ?repair_bay)
        )
      )
  )
  (:action prepare_repair_bay_for_assembly
    :parameters (?repair_bay - repair_bay ?primary_unit - primary_unit ?repair_tool - repair_tool)
    :precondition
      (and
        (repair_bay_reserved ?repair_bay)
        (primary_unit_task_ready ?primary_unit)
        (asset_assigned_tool ?primary_unit ?repair_tool)
        (not
          (repair_bay_prepped_for_assembly ?repair_bay)
        )
      )
    :effect (repair_bay_prepped_for_assembly ?repair_bay)
  )
  (:action install_upgrade_component_to_module
    :parameters (?assembly_module - assembly_module ?upgrade_component - upgrade_component ?repair_bay - repair_bay)
    :precondition
      (and
        (asset_accepted_for_maintenance ?assembly_module)
        (module_assigned_to_repair_bay ?assembly_module ?repair_bay)
        (module_has_upgrade_component ?assembly_module ?upgrade_component)
        (upgrade_component_available ?upgrade_component)
        (repair_bay_reserved ?repair_bay)
        (repair_bay_prepped_for_assembly ?repair_bay)
        (not
          (upgrade_component_in_use ?upgrade_component)
        )
      )
    :effect
      (and
        (upgrade_component_in_use ?upgrade_component)
        (upgrade_component_assigned_to_bay ?upgrade_component ?repair_bay)
        (not
          (upgrade_component_available ?upgrade_component)
        )
      )
  )
  (:action prepare_module_for_upgrade
    :parameters (?assembly_module - assembly_module ?upgrade_component - upgrade_component ?repair_bay - repair_bay ?repair_tool - repair_tool)
    :precondition
      (and
        (asset_accepted_for_maintenance ?assembly_module)
        (module_has_upgrade_component ?assembly_module ?upgrade_component)
        (upgrade_component_in_use ?upgrade_component)
        (upgrade_component_assigned_to_bay ?upgrade_component ?repair_bay)
        (asset_assigned_tool ?assembly_module ?repair_tool)
        (not
          (repair_bay_flag_calibration_required ?repair_bay)
        )
        (not
          (module_staged_for_upgrade ?assembly_module)
        )
      )
    :effect (module_staged_for_upgrade ?assembly_module)
  )
  (:action attach_calibration_kit_to_module
    :parameters (?assembly_module - assembly_module ?calibration_kit - calibration_kit)
    :precondition
      (and
        (asset_accepted_for_maintenance ?assembly_module)
        (calibration_kit_available ?calibration_kit)
        (not
          (module_attached_calibration_kit ?assembly_module)
        )
      )
    :effect
      (and
        (module_attached_calibration_kit ?assembly_module)
        (module_assigned_calibration_kit ?assembly_module ?calibration_kit)
        (not
          (calibration_kit_available ?calibration_kit)
        )
      )
  )
  (:action prepare_module_with_component_and_calibration
    :parameters (?assembly_module - assembly_module ?upgrade_component - upgrade_component ?repair_bay - repair_bay ?repair_tool - repair_tool ?calibration_kit - calibration_kit)
    :precondition
