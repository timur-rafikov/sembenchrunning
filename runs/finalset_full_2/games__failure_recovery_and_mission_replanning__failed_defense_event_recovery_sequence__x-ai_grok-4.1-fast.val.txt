--- domain.pddl (as passed to VAL) ---
(define (domain failed_defense_recovery)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_group - object asset_group - object plan_element_group - object entity_supertype - object recoverable_entity - entity_supertype reinforcement_asset - resource_group route_option - resource_group support_role - resource_group tactical_marker - resource_group upgrade_module - resource_group salvage_token - resource_group equipment_variant - resource_group specialist_upgrade - resource_group consumable_supply - asset_group fortification_module - asset_group mission_flag - asset_group approach_primary_lane - plan_element_group approach_secondary_lane - plan_element_group recovery_plan_profile - plan_element_group defender_subtype - recoverable_entity support_subtype - recoverable_entity local_defender_element - defender_subtype support_element - defender_subtype squad_leader - support_subtype)
  (:predicates
    (entity_triaged ?defense_site - recoverable_entity)
    (entity_prepared ?defense_site - recoverable_entity)
    (reinforcement_reserved ?defense_site - recoverable_entity)
    (entity_restored ?defense_site - recoverable_entity)
    (ready_for_recovery ?defense_site - recoverable_entity)
    (entity_salvage_spent ?defense_site - recoverable_entity)
    (reinforcement_available ?reinforcement_asset - reinforcement_asset)
    (entity_assigned_reinforcement ?defense_site - recoverable_entity ?reinforcement_asset - reinforcement_asset)
    (route_available ?route_option - route_option)
    (entity_assigned_route ?defense_site - recoverable_entity ?route_option - route_option)
    (support_role_available ?support_role - support_role)
    (entity_assigned_support_role ?defense_site - recoverable_entity ?support_role - support_role)
    (supply_available ?consumable_supply - consumable_supply)
    (defender_supply_assigned ?local_defender_element - local_defender_element ?consumable_supply - consumable_supply)
    (support_supply_assigned ?support_element - support_element ?consumable_supply - consumable_supply)
    (defender_assigned_primary_lane ?local_defender_element - local_defender_element ?approach_primary_lane - approach_primary_lane)
    (primary_lane_reserved ?approach_primary_lane - approach_primary_lane)
    (primary_lane_validated ?approach_primary_lane - approach_primary_lane)
    (defender_ready ?local_defender_element - local_defender_element)
    (support_assigned_secondary_lane ?support_element - support_element ?approach_secondary_lane - approach_secondary_lane)
    (secondary_lane_reserved ?approach_secondary_lane - approach_secondary_lane)
    (secondary_lane_validated ?approach_secondary_lane - approach_secondary_lane)
    (support_ready ?support_element - support_element)
    (plan_profile_available ?recovery_plan_profile - recovery_plan_profile)
    (plan_profile_reserved ?recovery_plan_profile - recovery_plan_profile)
    (plan_assigns_primary_lane ?recovery_plan_profile - recovery_plan_profile ?approach_primary_lane - approach_primary_lane)
    (plan_assigns_secondary_lane ?recovery_plan_profile - recovery_plan_profile ?approach_secondary_lane - approach_secondary_lane)
    (plan_option_alpha ?recovery_plan_profile - recovery_plan_profile)
    (plan_option_beta ?recovery_plan_profile - recovery_plan_profile)
    (plan_validated ?recovery_plan_profile - recovery_plan_profile)
    (leader_assigned_to_defender ?squad_leader - squad_leader ?local_defender_element - local_defender_element)
    (leader_assigned_to_support ?squad_leader - squad_leader ?support_element - support_element)
    (leader_assigned_plan ?squad_leader - squad_leader ?recovery_plan_profile - recovery_plan_profile)
    (module_available ?fortification_module - fortification_module)
    (leader_reserved_module ?squad_leader - squad_leader ?fortification_module - fortification_module)
    (module_installed ?fortification_module - fortification_module)
    (module_linked_to_plan ?fortification_module - fortification_module ?recovery_plan_profile - recovery_plan_profile)
    (leader_module_ready ?squad_leader - squad_leader)
    (leader_module_progressed ?squad_leader - squad_leader)
    (leader_prepped ?squad_leader - squad_leader)
    (leader_marker_applied ?squad_leader - squad_leader)
    (leader_marker_confirmed ?squad_leader - squad_leader)
    (leader_upgrades_applied ?squad_leader - squad_leader)
    (leader_activation_ready ?squad_leader - squad_leader)
    (mission_flag_available ?mission_flag - mission_flag)
    (leader_has_mission_flag ?squad_leader - squad_leader ?mission_flag - mission_flag)
    (leader_flag_attached ?squad_leader - squad_leader)
    (leader_marker_locked ?squad_leader - squad_leader)
    (leader_activation_prereq_met ?squad_leader - squad_leader)
    (tactical_marker_available ?tactical_marker - tactical_marker)
    (leader_has_tactical_marker ?squad_leader - squad_leader ?tactical_marker - tactical_marker)
    (upgrade_module_available ?upgrade_module - upgrade_module)
    (leader_has_upgrade_module ?squad_leader - squad_leader ?upgrade_module - upgrade_module)
    (equipment_variant_available ?equipment_variant - equipment_variant)
    (leader_equipped_variant ?squad_leader - squad_leader ?equipment_variant - equipment_variant)
    (specialist_upgrade_available ?specialist_upgrade - specialist_upgrade)
    (leader_has_specialist_upgrade ?squad_leader - squad_leader ?specialist_upgrade - specialist_upgrade)
    (salvage_token_available ?salvage_token - salvage_token)
    (entity_consumed_salvage ?defense_site - recoverable_entity ?salvage_token - salvage_token)
    (defender_lane_confirmed ?local_defender_element - local_defender_element)
    (support_lane_confirmed ?support_element - support_element)
    (activation_acknowledged ?squad_leader - squad_leader)
  )
  (:action mark_site_for_recovery
    :parameters (?defense_site - recoverable_entity)
    :precondition
      (and
        (not
          (entity_triaged ?defense_site)
        )
        (not
          (entity_restored ?defense_site)
        )
      )
    :effect (entity_triaged ?defense_site)
  )
  (:action reserve_reinforcement_for_site
    :parameters (?defense_site - recoverable_entity ?reinforcement_asset - reinforcement_asset)
    :precondition
      (and
        (entity_triaged ?defense_site)
        (not
          (reinforcement_reserved ?defense_site)
        )
        (reinforcement_available ?reinforcement_asset)
      )
    :effect
      (and
        (reinforcement_reserved ?defense_site)
        (entity_assigned_reinforcement ?defense_site ?reinforcement_asset)
        (not
          (reinforcement_available ?reinforcement_asset)
        )
      )
  )
  (:action reserve_route_for_site
    :parameters (?defense_site - recoverable_entity ?route_option - route_option)
    :precondition
      (and
        (entity_triaged ?defense_site)
        (reinforcement_reserved ?defense_site)
        (route_available ?route_option)
      )
    :effect
      (and
        (entity_assigned_route ?defense_site ?route_option)
        (not
          (route_available ?route_option)
        )
      )
  )
  (:action finalize_site_preparation
    :parameters (?defense_site - recoverable_entity ?route_option - route_option)
    :precondition
      (and
        (entity_triaged ?defense_site)
        (reinforcement_reserved ?defense_site)
        (entity_assigned_route ?defense_site ?route_option)
        (not
          (entity_prepared ?defense_site)
        )
      )
    :effect (entity_prepared ?defense_site)
  )
  (:action release_route_reservation
    :parameters (?defense_site - recoverable_entity ?route_option - route_option)
    :precondition
      (and
        (entity_assigned_route ?defense_site ?route_option)
      )
    :effect
      (and
        (route_available ?route_option)
        (not
          (entity_assigned_route ?defense_site ?route_option)
        )
      )
  )
  (:action assign_support_role_to_site
    :parameters (?defense_site - recoverable_entity ?support_role - support_role)
    :precondition
      (and
        (entity_prepared ?defense_site)
        (support_role_available ?support_role)
      )
    :effect
      (and
        (entity_assigned_support_role ?defense_site ?support_role)
        (not
          (support_role_available ?support_role)
        )
      )
  )
  (:action release_support_role_from_site
    :parameters (?defense_site - recoverable_entity ?support_role - support_role)
    :precondition
      (and
        (entity_assigned_support_role ?defense_site ?support_role)
      )
    :effect
      (and
        (support_role_available ?support_role)
        (not
          (entity_assigned_support_role ?defense_site ?support_role)
        )
      )
  )
  (:action equip_leader_with_variant
    :parameters (?squad_leader - squad_leader ?equipment_variant - equipment_variant)
    :precondition
      (and
        (entity_prepared ?squad_leader)
        (equipment_variant_available ?equipment_variant)
      )
    :effect
      (and
        (leader_equipped_variant ?squad_leader ?equipment_variant)
        (not
          (equipment_variant_available ?equipment_variant)
        )
      )
  )
  (:action unequip_leader_variant
    :parameters (?squad_leader - squad_leader ?equipment_variant - equipment_variant)
    :precondition
      (and
        (leader_equipped_variant ?squad_leader ?equipment_variant)
      )
    :effect
      (and
        (equipment_variant_available ?equipment_variant)
        (not
          (leader_equipped_variant ?squad_leader ?equipment_variant)
        )
      )
  )
  (:action install_specialist_upgrade
    :parameters (?squad_leader - squad_leader ?specialist_upgrade - specialist_upgrade)
    :precondition
      (and
        (entity_prepared ?squad_leader)
        (specialist_upgrade_available ?specialist_upgrade)
      )
    :effect
      (and
        (leader_has_specialist_upgrade ?squad_leader ?specialist_upgrade)
        (not
          (specialist_upgrade_available ?specialist_upgrade)
        )
      )
  )
  (:action remove_specialist_upgrade
    :parameters (?squad_leader - squad_leader ?specialist_upgrade - specialist_upgrade)
    :precondition
      (and
        (leader_has_specialist_upgrade ?squad_leader ?specialist_upgrade)
      )
    :effect
      (and
        (specialist_upgrade_available ?specialist_upgrade)
        (not
          (leader_has_specialist_upgrade ?squad_leader ?specialist_upgrade)
        )
      )
  )
  (:action reserve_defender_primary_lane
    :parameters (?local_defender_element - local_defender_element ?approach_primary_lane - approach_primary_lane ?route_option - route_option)
    :precondition
      (and
        (entity_prepared ?local_defender_element)
        (entity_assigned_route ?local_defender_element ?route_option)
        (defender_assigned_primary_lane ?local_defender_element ?approach_primary_lane)
        (not
          (primary_lane_reserved ?approach_primary_lane)
        )
        (not
          (primary_lane_validated ?approach_primary_lane)
        )
      )
    :effect (primary_lane_reserved ?approach_primary_lane)
  )
  (:action confirm_defender_lane_with_support
    :parameters (?local_defender_element - local_defender_element ?approach_primary_lane - approach_primary_lane ?support_role - support_role)
    :precondition
      (and
        (entity_prepared ?local_defender_element)
        (entity_assigned_support_role ?local_defender_element ?support_role)
        (defender_assigned_primary_lane ?local_defender_element ?approach_primary_lane)
        (primary_lane_reserved ?approach_primary_lane)
        (not
          (defender_lane_confirmed ?local_defender_element)
        )
      )
    :effect
      (and
        (defender_lane_confirmed ?local_defender_element)
        (defender_ready ?local_defender_element)
      )
  )
  (:action apply_supply_to_defender_lane
    :parameters (?local_defender_element - local_defender_element ?approach_primary_lane - approach_primary_lane ?consumable_supply - consumable_supply)
    :precondition
      (and
        (entity_prepared ?local_defender_element)
        (defender_assigned_primary_lane ?local_defender_element ?approach_primary_lane)
        (supply_available ?consumable_supply)
        (not
          (defender_lane_confirmed ?local_defender_element)
        )
      )
    :effect
      (and
        (primary_lane_validated ?approach_primary_lane)
        (defender_lane_confirmed ?local_defender_element)
        (defender_supply_assigned ?local_defender_element ?consumable_supply)
        (not
          (supply_available ?consumable_supply)
        )
      )
  )
  (:action finalize_defender_lane_validation
    :parameters (?local_defender_element - local_defender_element ?approach_primary_lane - approach_primary_lane ?route_option - route_option ?consumable_supply - consumable_supply)
    :precondition
      (and
        (entity_prepared ?local_defender_element)
        (entity_assigned_route ?local_defender_element ?route_option)
        (defender_assigned_primary_lane ?local_defender_element ?approach_primary_lane)
        (primary_lane_validated ?approach_primary_lane)
        (defender_supply_assigned ?local_defender_element ?consumable_supply)
        (not
          (defender_ready ?local_defender_element)
        )
      )
    :effect
      (and
        (primary_lane_reserved ?approach_primary_lane)
        (defender_ready ?local_defender_element)
        (supply_available ?consumable_supply)
        (not
          (defender_supply_assigned ?local_defender_element ?consumable_supply)
        )
      )
  )
  (:action reserve_support_secondary_lane
    :parameters (?support_element - support_element ?approach_secondary_lane - approach_secondary_lane ?route_option - route_option)
    :precondition
      (and
        (entity_prepared ?support_element)
        (entity_assigned_route ?support_element ?route_option)
        (support_assigned_secondary_lane ?support_element ?approach_secondary_lane)
        (not
          (secondary_lane_reserved ?approach_secondary_lane)
        )
        (not
          (secondary_lane_validated ?approach_secondary_lane)
        )
      )
    :effect (secondary_lane_reserved ?approach_secondary_lane)
  )
  (:action confirm_support_lane_with_role
    :parameters (?support_element - support_element ?approach_secondary_lane - approach_secondary_lane ?support_role - support_role)
    :precondition
      (and
        (entity_prepared ?support_element)
        (entity_assigned_support_role ?support_element ?support_role)
        (support_assigned_secondary_lane ?support_element ?approach_secondary_lane)
        (secondary_lane_reserved ?approach_secondary_lane)
        (not
          (support_lane_confirmed ?support_element)
        )
      )
    :effect
      (and
        (support_lane_confirmed ?support_element)
        (support_ready ?support_element)
      )
  )
  (:action apply_supply_to_support_lane
    :parameters (?support_element - support_element ?approach_secondary_lane - approach_secondary_lane ?consumable_supply - consumable_supply)
    :precondition
      (and
        (entity_prepared ?support_element)
        (support_assigned_secondary_lane ?support_element ?approach_secondary_lane)
        (supply_available ?consumable_supply)
        (not
          (support_lane_confirmed ?support_element)
        )
      )
    :effect
      (and
        (secondary_lane_validated ?approach_secondary_lane)
        (support_lane_confirmed ?support_element)
        (support_supply_assigned ?support_element ?consumable_supply)
        (not
          (supply_available ?consumable_supply)
        )
      )
  )
  (:action finalize_support_lane_validation
    :parameters (?support_element - support_element ?approach_secondary_lane - approach_secondary_lane ?route_option - route_option ?consumable_supply - consumable_supply)
    :precondition
      (and
        (entity_prepared ?support_element)
        (entity_assigned_route ?support_element ?route_option)
        (support_assigned_secondary_lane ?support_element ?approach_secondary_lane)
        (secondary_lane_validated ?approach_secondary_lane)
        (support_supply_assigned ?support_element ?consumable_supply)
        (not
          (support_ready ?support_element)
        )
      )
    :effect
      (and
        (secondary_lane_reserved ?approach_secondary_lane)
        (support_ready ?support_element)
        (supply_available ?consumable_supply)
        (not
          (support_supply_assigned ?support_element ?consumable_supply)
        )
      )
  )
  (:action assemble_plan_profile_standard
    :parameters (?local_defender_element - local_defender_element ?support_element - support_element ?approach_primary_lane - approach_primary_lane ?approach_secondary_lane - approach_secondary_lane ?recovery_plan_profile - recovery_plan_profile)
    :precondition
      (and
        (defender_lane_confirmed ?local_defender_element)
        (support_lane_confirmed ?support_element)
        (defender_assigned_primary_lane ?local_defender_element ?approach_primary_lane)
        (support_assigned_secondary_lane ?support_element ?approach_secondary_lane)
        (primary_lane_reserved ?approach_primary_lane)
        (secondary_lane_reserved ?approach_secondary_lane)
        (defender_ready ?local_defender_element)
        (support_ready ?support_element)
        (plan_profile_available ?recovery_plan_profile)
      )
    :effect
      (and
        (plan_profile_reserved ?recovery_plan_profile)
        (plan_assigns_primary_lane ?recovery_plan_profile ?approach_primary_lane)
        (plan_assigns_secondary_lane ?recovery_plan_profile ?approach_secondary_lane)
        (not
          (plan_profile_available ?recovery_plan_profile)
        )
      )
  )
  (:action assemble_plan_profile_with_primary_lane_supplied
    :parameters (?local_defender_element - local_defender_element ?support_element - support_element ?approach_primary_lane - approach_primary_lane ?approach_secondary_lane - approach_secondary_lane ?recovery_plan_profile - recovery_plan_profile)
    :precondition
      (and
        (defender_lane_confirmed ?local_defender_element)
        (support_lane_confirmed ?support_element)
        (defender_assigned_primary_lane ?local_defender_element ?approach_primary_lane)
        (support_assigned_secondary_lane ?support_element ?approach_secondary_lane)
        (primary_lane_validated ?approach_primary_lane)
        (secondary_lane_reserved ?approach_secondary_lane)
        (not
          (defender_ready ?local_defender_element)
        )
        (support_ready ?support_element)
        (plan_profile_available ?recovery_plan_profile)
      )
    :effect
      (and
        (plan_profile_reserved ?recovery_plan_profile)
        (plan_assigns_primary_lane ?recovery_plan_profile ?approach_primary_lane)
        (plan_assigns_secondary_lane ?recovery_plan_profile ?approach_secondary_lane)
        (plan_option_alpha ?recovery_plan_profile)
        (not
          (plan_profile_available ?recovery_plan_profile)
        )
      )
  )
  (:action assemble_plan_profile_with_secondary_lane_supplied
    :parameters (?local_defender_element - local_defender_element ?support_element - support_element ?approach_primary_lane - approach_primary_lane ?approach_secondary_lane - approach_secondary_lane ?recovery_plan_profile - recovery_plan_profile)
    :precondition
      (and
        (defender_lane_confirmed ?local_defender_element)
        (support_lane_confirmed ?support_element)
        (defender_assigned_primary_lane ?local_defender_element ?approach_primary_lane)
        (support_assigned_secondary_lane ?support_element ?approach_secondary_lane)
        (primary_lane_reserved ?approach_primary_lane)
        (secondary_lane_validated ?approach_secondary_lane)
        (defender_ready ?local_defender_element)
        (not
          (support_ready ?support_element)
        )
        (plan_profile_available ?recovery_plan_profile)
      )
    :effect
      (and
        (plan_profile_reserved ?recovery_plan_profile)
        (plan_assigns_primary_lane ?recovery_plan_profile ?approach_primary_lane)
        (plan_assigns_secondary_lane ?recovery_plan_profile ?approach_secondary_lane)
        (plan_option_beta ?recovery_plan_profile)
        (not
          (plan_profile_available ?recovery_plan_profile)
        )
      )
  )
  (:action assemble_full_plan_profile
    :parameters (?local_defender_element - local_defender_element ?support_element - support_element ?approach_primary_lane - approach_primary_lane ?approach_secondary_lane - approach_secondary_lane ?recovery_plan_profile - recovery_plan_profile)
    :precondition
      (and
        (defender_lane_confirmed ?local_defender_element)
        (support_lane_confirmed ?support_element)
        (defender_assigned_primary_lane ?local_defender_element ?approach_primary_lane)
        (support_assigned_secondary_lane ?support_element ?approach_secondary_lane)
        (primary_lane_validated ?approach_primary_lane)
        (secondary_lane_validated ?approach_secondary_lane)
        (not
          (defender_ready ?local_defender_element)
        )
        (not
          (support_ready ?support_element)
        )
        (plan_profile_available ?recovery_plan_profile)
      )
    :effect
      (and
        (plan_profile_reserved ?recovery_plan_profile)
        (plan_assigns_primary_lane ?recovery_plan_profile ?approach_primary_lane)
        (plan_assigns_secondary_lane ?recovery_plan_profile ?approach_secondary_lane)
        (plan_option_alpha ?recovery_plan_profile)
        (plan_option_beta ?recovery_plan_profile)
        (not
          (plan_profile_available ?recovery_plan_profile)
        )
      )
  )
  (:action validate_plan_profile
    :parameters (?recovery_plan_profile - recovery_plan_profile ?local_defender_element - local_defender_element ?route_option - route_option)
    :precondition
      (and
        (plan_profile_reserved ?recovery_plan_profile)
        (defender_lane_confirmed ?local_defender_element)
        (entity_assigned_route ?local_defender_element ?route_option)
        (not
          (plan_validated ?recovery_plan_profile)
        )
      )
    :effect (plan_validated ?recovery_plan_profile)
  )
  (:action install_fortification_module
    :parameters (?squad_leader - squad_leader ?fortification_module - fortification_module ?recovery_plan_profile - recovery_plan_profile)
    :precondition
      (and
        (entity_prepared ?squad_leader)
        (leader_assigned_plan ?squad_leader ?recovery_plan_profile)
        (leader_reserved_module ?squad_leader ?fortification_module)
        (module_available ?fortification_module)
        (plan_profile_reserved ?recovery_plan_profile)
        (plan_validated ?recovery_plan_profile)
        (not
          (module_installed ?fortification_module)
        )
      )
    :effect
      (and
        (module_installed ?fortification_module)
        (module_linked_to_plan ?fortification_module ?recovery_plan_profile)
        (not
          (module_available ?fortification_module)
        )
      )
  )
  (:action prepare_leader_for_installation
    :parameters (?squad_leader - squad_leader ?fortification_module - fortification_module ?recovery_plan_profile - recovery_plan_profile ?route_option - route_option)
    :precondition
      (and
        (entity_prepared ?squad_leader)
        (leader_reserved_module ?squad_leader ?fortification_module)
        (module_installed ?fortification_module)
        (module_linked_to_plan ?fortification_module ?recovery_plan_profile)
        (entity_assigned_route ?squad_leader ?route_option)
        (not
          (plan_option_alpha ?recovery_plan_profile)
        )
        (not
          (leader_module_ready ?squad_leader)
        )
      )
    :effect (leader_module_ready ?squad_leader)
  )
  (:action apply_tactical_marker_to_leader
    :parameters (?squad_leader - squad_leader ?tactical_marker - tactical_marker)
    :precondition
      (and
        (entity_prepared ?squad_leader)
        (tactical_marker_available ?tactical_marker)
        (not
          (leader_marker_applied ?squad_leader)
        )
      )
    :effect
      (and
        (leader_marker_applied ?squad_leader)
        (leader_has_tactical_marker ?squad_leader ?tactical_marker)
        (not
          (tactical_marker_available ?tactical_marker)
        )
      )
  )
  (:action confirm_tactical_marker_and_prepare_leader
    :parameters (?squad_leader - squad_leader ?fortification_module - fortification_module ?recovery_plan_profile - recovery_plan_profile ?route_option - route_option ?tactical_marker - tactical_marker)
    :precondition
      (and
        (entity_prepared ?squad_leader)
        (leader_reserved_module ?squad_leader ?fortification_module)
        (module_installed ?fortification_module)
        (module_linked_to_plan ?fortification_module ?recovery_plan_profile)
        (entity_assigned_route ?squad_leader ?route_option)
        (plan_option_alpha ?recovery_plan_profile)
        (leader_marker_applied ?squad_leader)
        (leader_has_tactical_marker ?squad_leader ?tactical_marker)
        (not
          (leader_module_ready ?squad_leader)
        )
      )
    :effect
      (and
        (leader_module_ready ?squad_leader)
        (leader_marker_confirmed ?squad_leader)
      )
  )
  (:action leader_install_module_step_one
    :parameters (?squad_leader - squad_leader ?equipment_variant - equipment_variant ?support_role - support_role ?fortification_module - fortification_module ?recovery_plan_profile - recovery_plan_profile)
    :precondition
      (and
        (leader_module_ready ?squad_leader)
