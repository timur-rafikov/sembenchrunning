(define (domain target_allocation_construction_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types template_holder_type - object asset_classification_type - object execution_classification_type - object mandate_container_type - object investment_mandate - mandate_container_type allocation_template - template_holder_type instrument - template_holder_type execution_desk - template_holder_type benchmark_clause - template_holder_type execution_algorithm - template_holder_type allocation_round_token - template_holder_type broker_routing_instruction - template_holder_type venue_preference - template_holder_type allocation_constraint - asset_classification_type position_lot_record - asset_classification_type approval_token - asset_classification_type asset_slot - execution_classification_type execution_slot - execution_classification_type proposed_trade_ticket - execution_classification_type model_container_type - investment_mandate bucket_container_type - investment_mandate model_portfolio - model_container_type portfolio_bucket - model_container_type allocation_job - bucket_container_type)
  (:predicates
    (entity_registered ?mandate - investment_mandate)
    (entity_onboarded ?mandate - investment_mandate)
    (entity_template_bound ?mandate - investment_mandate)
    (entity_assigned ?mandate - investment_mandate)
    (entity_allocation_committed ?mandate - investment_mandate)
    (entity_has_execution_fill ?mandate - investment_mandate)
    (allocation_template_available ?allocation_template - allocation_template)
    (entity_bound_to_allocation_template ?mandate - investment_mandate ?allocation_template - allocation_template)
    (instrument_available ?instrument - instrument)
    (entity_bound_to_instrument ?mandate - investment_mandate ?instrument - instrument)
    (execution_desk_available ?execution_desk - execution_desk)
    (entity_assigned_to_execution_desk ?mandate - investment_mandate ?execution_desk - execution_desk)
    (allocation_constraint_available ?allocation_constraint - allocation_constraint)
    (model_portfolio_has_constraint ?model_portfolio - model_portfolio ?allocation_constraint - allocation_constraint)
    (portfolio_bucket_has_constraint ?portfolio_bucket - portfolio_bucket ?allocation_constraint - allocation_constraint)
    (model_portfolio_targets_asset_slot ?model_portfolio - model_portfolio ?asset_slot - asset_slot)
    (asset_slot_tagged_primary ?asset_slot - asset_slot)
    (asset_slot_tagged_secondary ?asset_slot - asset_slot)
    (model_portfolio_exposure_tagged ?model_portfolio - model_portfolio)
    (portfolio_bucket_targets_execution_slot ?portfolio_bucket - portfolio_bucket ?execution_slot - execution_slot)
    (execution_slot_tagged_primary ?execution_slot - execution_slot)
    (execution_slot_tagged_secondary ?execution_slot - execution_slot)
    (portfolio_bucket_exposure_tagged ?portfolio_bucket - portfolio_bucket)
    (proposed_trade_ticket_available ?proposed_trade_ticket - proposed_trade_ticket)
    (proposed_trade_ticket_prepared ?proposed_trade_ticket - proposed_trade_ticket)
    (ticket_assigned_asset_slot ?proposed_trade_ticket - proposed_trade_ticket ?asset_slot - asset_slot)
    (ticket_assigned_execution_slot ?proposed_trade_ticket - proposed_trade_ticket ?execution_slot - execution_slot)
    (proposed_trade_ticket_requires_benchmark_enrichment ?proposed_trade_ticket - proposed_trade_ticket)
    (proposed_trade_ticket_requires_broker_enrichment ?proposed_trade_ticket - proposed_trade_ticket)
    (proposed_trade_ticket_validated ?proposed_trade_ticket - proposed_trade_ticket)
    (entity_assigned_to_model_portfolio ?allocation_job - allocation_job ?model_portfolio - model_portfolio)
    (entity_assigned_to_portfolio_bucket ?allocation_job - allocation_job ?portfolio_bucket - portfolio_bucket)
    (entity_has_proposed_trade_ticket ?allocation_job - allocation_job ?proposed_trade_ticket - proposed_trade_ticket)
    (position_lot_available ?position_lot - position_lot_record)
    (entity_has_position_lot ?allocation_job - allocation_job ?position_lot - position_lot_record)
    (position_lot_reserved ?position_lot - position_lot_record)
    (position_lot_allocated_to_ticket ?position_lot - position_lot_record ?proposed_trade_ticket - proposed_trade_ticket)
    (allocation_job_enrichment_claimed ?allocation_job - allocation_job)
    (entity_has_routing_instruction ?allocation_job - allocation_job)
    (allocation_job_ready_for_finalization ?allocation_job - allocation_job)
    (allocation_job_benchmark_attached ?allocation_job - allocation_job)
    (allocation_job_benchmark_enriched ?allocation_job - allocation_job)
    (allocation_job_enrichment_complete ?allocation_job - allocation_job)
    (allocation_job_validated ?allocation_job - allocation_job)
    (approval_token_available ?approval_token - approval_token)
    (allocation_job_has_approval_token ?allocation_job - allocation_job ?approval_token - approval_token)
    (allocation_job_approved ?allocation_job - allocation_job)
    (allocation_job_ready_for_venue_attachment ?allocation_job - allocation_job)
    (allocation_job_venue_attached ?allocation_job - allocation_job)
    (benchmark_clause_available ?benchmark_clause - benchmark_clause)
    (allocation_job_has_benchmark_clause ?allocation_job - allocation_job ?benchmark_clause - benchmark_clause)
    (execution_algorithm_available ?execution_algorithm - execution_algorithm)
    (entity_assigned_to_execution_algorithm ?allocation_job - allocation_job ?execution_algorithm - execution_algorithm)
    (broker_routing_instruction_available ?broker_instruction - broker_routing_instruction)
    (entity_assigned_to_broker_instruction ?allocation_job - allocation_job ?broker_instruction - broker_routing_instruction)
    (venue_preference_available ?venue_preference - venue_preference)
    (entity_assigned_to_venue_preference ?allocation_job - allocation_job ?venue_preference - venue_preference)
    (allocation_round_token_available ?allocation_round_token - allocation_round_token)
    (entity_assigned_allocation_round_token ?mandate - investment_mandate ?allocation_round_token - allocation_round_token)
    (model_portfolio_ready_for_synthesis ?model_portfolio - model_portfolio)
    (portfolio_bucket_ready_for_synthesis ?portfolio_bucket - portfolio_bucket)
    (allocation_job_finalized ?allocation_job - allocation_job)
  )
  (:action register_investment_mandate
    :parameters (?mandate - investment_mandate)
    :precondition
      (and
        (not
          (entity_registered ?mandate)
        )
        (not
          (entity_assigned ?mandate)
        )
      )
    :effect (entity_registered ?mandate)
  )
  (:action attach_allocation_template_to_mandate
    :parameters (?mandate - investment_mandate ?allocation_template - allocation_template)
    :precondition
      (and
        (entity_registered ?mandate)
        (not
          (entity_template_bound ?mandate)
        )
        (allocation_template_available ?allocation_template)
      )
    :effect
      (and
        (entity_template_bound ?mandate)
        (entity_bound_to_allocation_template ?mandate ?allocation_template)
        (not
          (allocation_template_available ?allocation_template)
        )
      )
  )
  (:action reserve_instrument_for_mandate
    :parameters (?mandate - investment_mandate ?instrument - instrument)
    :precondition
      (and
        (entity_registered ?mandate)
        (entity_template_bound ?mandate)
        (instrument_available ?instrument)
      )
    :effect
      (and
        (entity_bound_to_instrument ?mandate ?instrument)
        (not
          (instrument_available ?instrument)
        )
      )
  )
  (:action complete_mandate_onboarding
    :parameters (?mandate - investment_mandate ?instrument - instrument)
    :precondition
      (and
        (entity_registered ?mandate)
        (entity_template_bound ?mandate)
        (entity_bound_to_instrument ?mandate ?instrument)
        (not
          (entity_onboarded ?mandate)
        )
      )
    :effect (entity_onboarded ?mandate)
  )
  (:action release_instrument_reservation
    :parameters (?mandate - investment_mandate ?instrument - instrument)
    :precondition
      (and
        (entity_bound_to_instrument ?mandate ?instrument)
      )
    :effect
      (and
        (instrument_available ?instrument)
        (not
          (entity_bound_to_instrument ?mandate ?instrument)
        )
      )
  )
  (:action assign_execution_desk_to_mandate
    :parameters (?mandate - investment_mandate ?execution_desk - execution_desk)
    :precondition
      (and
        (entity_onboarded ?mandate)
        (execution_desk_available ?execution_desk)
      )
    :effect
      (and
        (entity_assigned_to_execution_desk ?mandate ?execution_desk)
        (not
          (execution_desk_available ?execution_desk)
        )
      )
  )
  (:action release_execution_desk_from_mandate
    :parameters (?mandate - investment_mandate ?execution_desk - execution_desk)
    :precondition
      (and
        (entity_assigned_to_execution_desk ?mandate ?execution_desk)
      )
    :effect
      (and
        (execution_desk_available ?execution_desk)
        (not
          (entity_assigned_to_execution_desk ?mandate ?execution_desk)
        )
      )
  )
  (:action reserve_broker_for_allocation_job
    :parameters (?allocation_job - allocation_job ?broker_instruction - broker_routing_instruction)

  (:action tag_primary_asset_slot_for_model_portfolio
    :parameters (?model_portfolio - model_portfolio ?asset_slot - asset_slot ?instrument - instrument)

      (and
        (entity_onboarded ?model_portfolio)
        (entity_bound_to_instrument ?model_portfolio ?instrument)
        (model_portfolio_targets_asset_slot ?model_portfolio ?asset_slot)
        (not
          (asset_slot_tagged_primary ?asset_slot)
        )
        (not
          (asset_slot_tagged_secondary ?asset_slot)
        )
      )
    :effect (asset_slot_tagged_primary ?asset_slot)
  )
  (:action reserve_model_portfolio_asset_slot_with_desk
    :parameters (?model_portfolio - model_portfolio ?asset_slot - asset_slot ?execution_desk - execution_desk)
    :precondition
      (and
        (entity_onboarded ?model_portfolio)
        (entity_assigned_to_execution_desk ?model_portfolio ?execution_desk)
        (model_portfolio_targets_asset_slot ?model_portfolio ?asset_slot)
        (asset_slot_tagged_primary ?asset_slot)
        (not
          (model_portfolio_ready_for_synthesis ?model_portfolio)
        )
      )
    :effect
      (and
        (model_portfolio_ready_for_synthesis ?model_portfolio)
        (model_portfolio_exposure_tagged ?model_portfolio)

  )

      (and
        (asset_slot_tagged_secondary ?asset_slot)
        (model_portfolio_ready_for_synthesis ?model_portfolio)

          (allocation_constraint_available ?allocation_constraint)

          (model_portfolio_exposure_tagged ?model_portfolio)

        (asset_slot_tagged_primary ?asset_slot)

        (allocation_constraint_available ?allocation_constraint)
        (not
          (model_portfolio_has_constraint ?model_portfolio ?allocation_constraint)
        )
      )
  )

    :precondition
      (and
        (model_portfolio_ready_for_synthesis ?model_portfolio)

        (model_portfolio_exposure_tagged ?model_portfolio)

        (proposed_trade_ticket_available ?proposed_trade_ticket)
      )
    :effect
      (and
        (proposed_trade_ticket_prepared ?proposed_trade_ticket)
        (ticket_assigned_asset_slot ?proposed_trade_ticket ?asset_slot)
        (ticket_assigned_execution_slot ?proposed_trade_ticket ?execution_slot)
        (not
          (proposed_trade_ticket_available ?proposed_trade_ticket)
        )
      )
  )
  (:action assemble_proposed_trade_ticket_with_benchmark_path

          (model_portfolio_exposure_tagged ?model_portfolio)

        (portfolio_bucket_exposure_tagged ?portfolio_bucket)
        (proposed_trade_ticket_available ?proposed_trade_ticket)

    :effect

          (allocation_job_enrichment_claimed ?allocation_job)

        (not
          (allocation_job_benchmark_attached ?allocation_job)

      (and
        (allocation_job_benchmark_attached ?allocation_job)

        (not
          (benchmark_clause_available ?benchmark_clause)
