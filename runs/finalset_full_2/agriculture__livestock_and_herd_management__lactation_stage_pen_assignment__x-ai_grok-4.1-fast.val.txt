    :parameters (?animal - animal)
    :precondition
      (and

          (animal_assigned_pen ?animal)

  )
  (:action reserve_pen_candidate
    :parameters (?animal - animal ?pen - pen)

      (and

        (pen_available ?pen)

    :effect

        (animal_has_pen_candidate ?animal)

          (pen_available ?pen)
        )
      )
  )
