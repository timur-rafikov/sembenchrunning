--- domain.pddl (as passed to VAL) ---
(define (domain anomaly_detection_pipeline)
  (:requirements :strips :typing :negative-preconditions)
  (:types processing_resource - object feature_type - object bin_type - object data_item_root - object data_item - data_item_root detector_worker - processing_resource computed_feature - processing_resource validator - processing_resource explanation_method - processing_resource postprocessor - processing_resource evidence_token - processing_resource scoring_model - processing_resource explanation_template - processing_resource calibration_param - feature_type data_segment - feature_type reference_profile - feature_type feature_bin_raw - bin_type feature_bin_agg - bin_type anomaly_candidate - bin_type record_subtype_a - data_item record_subtype_b - data_item raw_record - record_subtype_a aggregated_record - record_subtype_a detection_pipeline - record_subtype_b)
  (:predicates
    (data_item_discovered ?data_item - data_item)
    (evaluation_completed ?data_item - data_item)
    (evaluation_allocated ?data_item - data_item)
    (anomaly_flagged ?data_item - data_item)
    (ready_for_evidence ?data_item - data_item)
    (evidence_bound ?data_item - data_item)
    (worker_available ?detector_worker - detector_worker)
    (assigned_to_worker ?data_item - data_item ?detector_worker - detector_worker)
    (feature_available ?feature - computed_feature)
    (feature_computed ?data_item - data_item ?feature - computed_feature)
    (validator_available ?validator - validator)
    (assigned_validator ?data_item - data_item ?validator - validator)
    (calibration_param_available ?calibration_param - calibration_param)
    (raw_record_has_calibration ?raw_record - raw_record ?calibration_param - calibration_param)
    (agg_record_has_calibration ?aggregated_record - aggregated_record ?calibration_param - calibration_param)
    (has_feature_bin_raw ?raw_record - raw_record ?feature_bin_raw - feature_bin_raw)
    (feature_bin_raw_vote_yes ?feature_bin_raw - feature_bin_raw)
    (feature_bin_raw_vote_no ?feature_bin_raw - feature_bin_raw)
    (raw_record_vote_confirmed ?raw_record - raw_record)
    (has_feature_bin_agg ?aggregated_record - aggregated_record ?feature_bin_agg - feature_bin_agg)
    (feature_bin_agg_vote_yes ?feature_bin_agg - feature_bin_agg)
    (feature_bin_agg_vote_no ?feature_bin_agg - feature_bin_agg)
    (agg_record_vote_confirmed ?aggregated_record - aggregated_record)
    (candidate_slot_available ?anomaly_candidate - anomaly_candidate)
    (candidate_allocated ?anomaly_candidate - anomaly_candidate)
    (candidate_has_raw_bin ?anomaly_candidate - anomaly_candidate ?feature_bin_raw - feature_bin_raw)
    (candidate_has_agg_bin ?anomaly_candidate - anomaly_candidate ?feature_bin_agg - feature_bin_agg)
    (candidate_confirmed_by_raw_votes ?anomaly_candidate - anomaly_candidate)
    (candidate_confirmed_by_agg_votes ?anomaly_candidate - anomaly_candidate)
    (candidate_ready_for_segment_attachment ?anomaly_candidate - anomaly_candidate)
    (pipeline_includes_raw_record ?detection_pipeline - detection_pipeline ?raw_record - raw_record)
    (pipeline_includes_agg_record ?detection_pipeline - detection_pipeline ?aggregated_record - aggregated_record)
    (pipeline_associated_with_candidate ?detection_pipeline - detection_pipeline ?anomaly_candidate - anomaly_candidate)
    (data_segment_available ?data_segment - data_segment)
    (pipeline_has_data_segment ?detection_pipeline - detection_pipeline ?data_segment - data_segment)
    (data_segment_consumed ?data_segment - data_segment)
    (data_segment_attached_to_candidate ?data_segment - data_segment ?anomaly_candidate - anomaly_candidate)
    (pipeline_segment_binding_complete ?detection_pipeline - detection_pipeline)
    (pipeline_model_bound ?detection_pipeline - detection_pipeline)
    (pipeline_ready_for_postprocessing ?detection_pipeline - detection_pipeline)
    (explanation_method_reserved ?detection_pipeline - detection_pipeline)
    (pipeline_explanation_prepared ?detection_pipeline - detection_pipeline)
    (postprocessing_configured ?detection_pipeline - detection_pipeline)
    (postprocessing_complete ?detection_pipeline - detection_pipeline)
    (reference_profile_available ?reference_profile - reference_profile)
    (pipeline_has_reference_profile ?detection_pipeline - detection_pipeline ?reference_profile - reference_profile)
    (reference_profile_assigned ?detection_pipeline - detection_pipeline)
    (reference_prepared ?detection_pipeline - detection_pipeline)
    (reference_explanation_applied ?detection_pipeline - detection_pipeline)
    (explanation_method_available ?explanation_method - explanation_method)
    (pipeline_has_explanation_method ?detection_pipeline - detection_pipeline ?explanation_method - explanation_method)
    (postprocessor_available ?postprocessor - postprocessor)
    (pipeline_has_postprocessor ?detection_pipeline - detection_pipeline ?postprocessor - postprocessor)
    (scoring_model_available ?scoring_model - scoring_model)
    (pipeline_has_scoring_model ?detection_pipeline - detection_pipeline ?scoring_model - scoring_model)
    (explanation_template_available ?explanation_template - explanation_template)
    (pipeline_has_explanation_template ?detection_pipeline - detection_pipeline ?explanation_template - explanation_template)
    (evidence_token_available ?evidence_token - evidence_token)
    (data_item_has_evidence ?data_item - data_item ?evidence_token - evidence_token)
    (raw_record_vote_flagged ?raw_record - raw_record)
    (agg_record_vote_flagged ?aggregated_record - aggregated_record)
    (pipeline_finalized ?detection_pipeline - detection_pipeline)
  )
  (:action discover_data_item
    :parameters (?data_item - data_item)
    :precondition
      (and
        (not
          (data_item_discovered ?data_item)
        )
        (not
          (anomaly_flagged ?data_item)
        )
      )
    :effect (data_item_discovered ?data_item)
  )
  (:action allocate_detector_worker
    :parameters (?data_item - data_item ?detector_worker - detector_worker)
    :precondition
      (and
        (data_item_discovered ?data_item)
        (not
          (evaluation_allocated ?data_item)
        )
        (worker_available ?detector_worker)
      )
    :effect
      (and
        (evaluation_allocated ?data_item)
        (assigned_to_worker ?data_item ?detector_worker)
        (not
          (worker_available ?detector_worker)
        )
      )
  )
  (:action compute_feature_for_item
    :parameters (?data_item - data_item ?feature - computed_feature)
    :precondition
      (and
        (data_item_discovered ?data_item)
        (evaluation_allocated ?data_item)
        (feature_available ?feature)
      )
    :effect
      (and
        (feature_computed ?data_item ?feature)
        (not
          (feature_available ?feature)
        )
      )
  )
  (:action complete_feature_evaluation
    :parameters (?data_item - data_item ?feature - computed_feature)
    :precondition
      (and
        (data_item_discovered ?data_item)
        (evaluation_allocated ?data_item)
        (feature_computed ?data_item ?feature)
        (not
          (evaluation_completed ?data_item)
        )
      )
    :effect (evaluation_completed ?data_item)
  )
  (:action release_feature_assignment
    :parameters (?data_item - data_item ?feature - computed_feature)
    :precondition
      (and
        (feature_computed ?data_item ?feature)
      )
    :effect
      (and
        (feature_available ?feature)
        (not
          (feature_computed ?data_item ?feature)
        )
      )
  )
  (:action assign_validator_to_item
    :parameters (?data_item - data_item ?validator - validator)
    :precondition
      (and
        (evaluation_completed ?data_item)
        (validator_available ?validator)
      )
    :effect
      (and
        (assigned_validator ?data_item ?validator)
        (not
          (validator_available ?validator)
        )
      )
  )
  (:action release_validator_from_item
    :parameters (?data_item - data_item ?validator - validator)
    :precondition
      (and
        (assigned_validator ?data_item ?validator)
      )
    :effect
      (and
        (validator_available ?validator)
        (not
          (assigned_validator ?data_item ?validator)
        )
      )
  )
  (:action bind_scoring_model_to_pipeline
    :parameters (?detection_pipeline - detection_pipeline ?scoring_model - scoring_model)
    :precondition
      (and
        (evaluation_completed ?detection_pipeline)
        (scoring_model_available ?scoring_model)
      )
    :effect
      (and
        (pipeline_has_scoring_model ?detection_pipeline ?scoring_model)
        (not
          (scoring_model_available ?scoring_model)
        )
      )
  )
  (:action unbind_scoring_model_from_pipeline
    :parameters (?detection_pipeline - detection_pipeline ?scoring_model - scoring_model)
    :precondition
      (and
        (pipeline_has_scoring_model ?detection_pipeline ?scoring_model)
      )
    :effect
      (and
        (scoring_model_available ?scoring_model)
        (not
          (pipeline_has_scoring_model ?detection_pipeline ?scoring_model)
        )
      )
  )
  (:action bind_explanation_template_to_pipeline
    :parameters (?detection_pipeline - detection_pipeline ?explanation_template - explanation_template)
    :precondition
      (and
        (evaluation_completed ?detection_pipeline)
        (explanation_template_available ?explanation_template)
      )
    :effect
      (and
        (pipeline_has_explanation_template ?detection_pipeline ?explanation_template)
        (not
          (explanation_template_available ?explanation_template)
        )
      )
  )
  (:action unbind_explanation_template_from_pipeline
    :parameters (?detection_pipeline - detection_pipeline ?explanation_template - explanation_template)
    :precondition
      (and
        (pipeline_has_explanation_template ?detection_pipeline ?explanation_template)
      )
    :effect
      (and
        (explanation_template_available ?explanation_template)
        (not
          (pipeline_has_explanation_template ?detection_pipeline ?explanation_template)
        )
      )
  )
  (:action vote_raw_feature_bin_yes
    :parameters (?raw_record - raw_record ?feature_bin_raw - feature_bin_raw ?feature - computed_feature)
    :precondition
      (and
        (evaluation_completed ?raw_record)
        (feature_computed ?raw_record ?feature)
        (has_feature_bin_raw ?raw_record ?feature_bin_raw)
        (not
          (feature_bin_raw_vote_yes ?feature_bin_raw)
        )
        (not
          (feature_bin_raw_vote_no ?feature_bin_raw)
        )
      )
    :effect (feature_bin_raw_vote_yes ?feature_bin_raw)
  )
  (:action confirm_raw_vote_with_validator
    :parameters (?raw_record - raw_record ?feature_bin_raw - feature_bin_raw ?validator - validator)
    :precondition
      (and
        (evaluation_completed ?raw_record)
        (assigned_validator ?raw_record ?validator)
        (has_feature_bin_raw ?raw_record ?feature_bin_raw)
        (feature_bin_raw_vote_yes ?feature_bin_raw)
        (not
          (raw_record_vote_flagged ?raw_record)
        )
      )
    :effect
      (and
        (raw_record_vote_flagged ?raw_record)
        (raw_record_vote_confirmed ?raw_record)
      )
  )
  (:action apply_calibration_to_raw_record
    :parameters (?raw_record - raw_record ?feature_bin_raw - feature_bin_raw ?calibration_param - calibration_param)
    :precondition
      (and
        (evaluation_completed ?raw_record)
        (has_feature_bin_raw ?raw_record ?feature_bin_raw)
        (calibration_param_available ?calibration_param)
        (not
          (raw_record_vote_flagged ?raw_record)
        )
      )
    :effect
      (and
        (feature_bin_raw_vote_no ?feature_bin_raw)
        (raw_record_vote_flagged ?raw_record)
        (raw_record_has_calibration ?raw_record ?calibration_param)
        (not
          (calibration_param_available ?calibration_param)
        )
      )
  )
  (:action reconcile_raw_bin_votes
    :parameters (?raw_record - raw_record ?feature_bin_raw - feature_bin_raw ?feature - computed_feature ?calibration_param - calibration_param)
    :precondition
      (and
        (evaluation_completed ?raw_record)
        (feature_computed ?raw_record ?feature)
        (has_feature_bin_raw ?raw_record ?feature_bin_raw)
        (feature_bin_raw_vote_no ?feature_bin_raw)
        (raw_record_has_calibration ?raw_record ?calibration_param)
        (not
          (raw_record_vote_confirmed ?raw_record)
        )
      )
    :effect
      (and
        (feature_bin_raw_vote_yes ?feature_bin_raw)
        (raw_record_vote_confirmed ?raw_record)
        (calibration_param_available ?calibration_param)
        (not
          (raw_record_has_calibration ?raw_record ?calibration_param)
        )
      )
  )
  (:action vote_agg_feature_bin_yes
    :parameters (?aggregated_record - aggregated_record ?feature_bin_agg - feature_bin_agg ?feature - computed_feature)
    :precondition
      (and
        (evaluation_completed ?aggregated_record)
        (feature_computed ?aggregated_record ?feature)
        (has_feature_bin_agg ?aggregated_record ?feature_bin_agg)
        (not
          (feature_bin_agg_vote_yes ?feature_bin_agg)
        )
        (not
          (feature_bin_agg_vote_no ?feature_bin_agg)
        )
      )
    :effect (feature_bin_agg_vote_yes ?feature_bin_agg)
  )
  (:action confirm_agg_vote_with_validator
    :parameters (?aggregated_record - aggregated_record ?feature_bin_agg - feature_bin_agg ?validator - validator)
    :precondition
      (and
        (evaluation_completed ?aggregated_record)
        (assigned_validator ?aggregated_record ?validator)
        (has_feature_bin_agg ?aggregated_record ?feature_bin_agg)
        (feature_bin_agg_vote_yes ?feature_bin_agg)
        (not
          (agg_record_vote_flagged ?aggregated_record)
        )
      )
    :effect
      (and
        (agg_record_vote_flagged ?aggregated_record)
        (agg_record_vote_confirmed ?aggregated_record)
      )
  )
  (:action apply_calibration_to_agg_record
    :parameters (?aggregated_record - aggregated_record ?feature_bin_agg - feature_bin_agg ?calibration_param - calibration_param)
    :precondition
      (and
        (evaluation_completed ?aggregated_record)
        (has_feature_bin_agg ?aggregated_record ?feature_bin_agg)
        (calibration_param_available ?calibration_param)
        (not
          (agg_record_vote_flagged ?aggregated_record)
        )
      )
    :effect
      (and
        (feature_bin_agg_vote_no ?feature_bin_agg)
        (agg_record_vote_flagged ?aggregated_record)
        (agg_record_has_calibration ?aggregated_record ?calibration_param)
        (not
          (calibration_param_available ?calibration_param)
        )
      )
  )
  (:action reconcile_agg_bin_votes
    :parameters (?aggregated_record - aggregated_record ?feature_bin_agg - feature_bin_agg ?feature - computed_feature ?calibration_param - calibration_param)
    :precondition
      (and
        (evaluation_completed ?aggregated_record)
        (feature_computed ?aggregated_record ?feature)
        (has_feature_bin_agg ?aggregated_record ?feature_bin_agg)
        (feature_bin_agg_vote_no ?feature_bin_agg)
        (agg_record_has_calibration ?aggregated_record ?calibration_param)
        (not
          (agg_record_vote_confirmed ?aggregated_record)
        )
      )
    :effect
      (and
        (feature_bin_agg_vote_yes ?feature_bin_agg)
        (agg_record_vote_confirmed ?aggregated_record)
        (calibration_param_available ?calibration_param)
        (not
          (agg_record_has_calibration ?aggregated_record ?calibration_param)
        )
      )
  )
  (:action assemble_candidate_from_bins
    :parameters (?raw_record - raw_record ?aggregated_record - aggregated_record ?feature_bin_raw - feature_bin_raw ?feature_bin_agg - feature_bin_agg ?anomaly_candidate - anomaly_candidate)
    :precondition
      (and
        (raw_record_vote_flagged ?raw_record)
        (agg_record_vote_flagged ?aggregated_record)
        (has_feature_bin_raw ?raw_record ?feature_bin_raw)
        (has_feature_bin_agg ?aggregated_record ?feature_bin_agg)
        (feature_bin_raw_vote_yes ?feature_bin_raw)
        (feature_bin_agg_vote_yes ?feature_bin_agg)
        (raw_record_vote_confirmed ?raw_record)
        (agg_record_vote_confirmed ?aggregated_record)
        (candidate_slot_available ?anomaly_candidate)
      )
    :effect
      (and
        (candidate_allocated ?anomaly_candidate)
        (candidate_has_raw_bin ?anomaly_candidate ?feature_bin_raw)
        (candidate_has_agg_bin ?anomaly_candidate ?feature_bin_agg)
        (not
          (candidate_slot_available ?anomaly_candidate)
        )
      )
  )
  (:action assemble_candidate_with_raw_votes
    :parameters (?raw_record - raw_record ?aggregated_record - aggregated_record ?feature_bin_raw - feature_bin_raw ?feature_bin_agg - feature_bin_agg ?anomaly_candidate - anomaly_candidate)
    :precondition
      (and
        (raw_record_vote_flagged ?raw_record)
        (agg_record_vote_flagged ?aggregated_record)
        (has_feature_bin_raw ?raw_record ?feature_bin_raw)
        (has_feature_bin_agg ?aggregated_record ?feature_bin_agg)
        (feature_bin_raw_vote_no ?feature_bin_raw)
        (feature_bin_agg_vote_yes ?feature_bin_agg)
        (not
          (raw_record_vote_confirmed ?raw_record)
        )
        (agg_record_vote_confirmed ?aggregated_record)
        (candidate_slot_available ?anomaly_candidate)
      )
    :effect
      (and
        (candidate_allocated ?anomaly_candidate)
        (candidate_has_raw_bin ?anomaly_candidate ?feature_bin_raw)
        (candidate_has_agg_bin ?anomaly_candidate ?feature_bin_agg)
        (candidate_confirmed_by_raw_votes ?anomaly_candidate)
        (not
          (candidate_slot_available ?anomaly_candidate)
        )
      )
  )
  (:action assemble_candidate_with_agg_votes
    :parameters (?raw_record - raw_record ?aggregated_record - aggregated_record ?feature_bin_raw - feature_bin_raw ?feature_bin_agg - feature_bin_agg ?anomaly_candidate - anomaly_candidate)
    :precondition
      (and
        (raw_record_vote_flagged ?raw_record)
        (agg_record_vote_flagged ?aggregated_record)
        (has_feature_bin_raw ?raw_record ?feature_bin_raw)
        (has_feature_bin_agg ?aggregated_record ?feature_bin_agg)
        (feature_bin_raw_vote_yes ?feature_bin_raw)
        (feature_bin_agg_vote_no ?feature_bin_agg)
        (raw_record_vote_confirmed ?raw_record)
        (not
          (agg_record_vote_confirmed ?aggregated_record)
        )
        (candidate_slot_available ?anomaly_candidate)
      )
    :effect
      (and
        (candidate_allocated ?anomaly_candidate)
        (candidate_has_raw_bin ?anomaly_candidate ?feature_bin_raw)
        (candidate_has_agg_bin ?anomaly_candidate ?feature_bin_agg)
        (candidate_confirmed_by_agg_votes ?anomaly_candidate)
        (not
          (candidate_slot_available ?anomaly_candidate)
        )
      )
  )
  (:action assemble_candidate_with_both_votes
    :parameters (?raw_record - raw_record ?aggregated_record - aggregated_record ?feature_bin_raw - feature_bin_raw ?feature_bin_agg - feature_bin_agg ?anomaly_candidate - anomaly_candidate)
    :precondition
      (and
        (raw_record_vote_flagged ?raw_record)
        (agg_record_vote_flagged ?aggregated_record)
        (has_feature_bin_raw ?raw_record ?feature_bin_raw)
        (has_feature_bin_agg ?aggregated_record ?feature_bin_agg)
        (feature_bin_raw_vote_no ?feature_bin_raw)
        (feature_bin_agg_vote_no ?feature_bin_agg)
        (not
          (raw_record_vote_confirmed ?raw_record)
        )
        (not
          (agg_record_vote_confirmed ?aggregated_record)
        )
        (candidate_slot_available ?anomaly_candidate)
      )
    :effect
      (and
        (candidate_allocated ?anomaly_candidate)
        (candidate_has_raw_bin ?anomaly_candidate ?feature_bin_raw)
        (candidate_has_agg_bin ?anomaly_candidate ?feature_bin_agg)
        (candidate_confirmed_by_raw_votes ?anomaly_candidate)
        (candidate_confirmed_by_agg_votes ?anomaly_candidate)
        (not
          (candidate_slot_available ?anomaly_candidate)
        )
      )
  )
  (:action mark_candidate_ready_for_attachment
    :parameters (?anomaly_candidate - anomaly_candidate ?raw_record - raw_record ?feature - computed_feature)
    :precondition
      (and
        (candidate_allocated ?anomaly_candidate)
        (raw_record_vote_flagged ?raw_record)
        (feature_computed ?raw_record ?feature)
        (not
          (candidate_ready_for_segment_attachment ?anomaly_candidate)
        )
      )
    :effect (candidate_ready_for_segment_attachment ?anomaly_candidate)
  )
  (:action attach_data_segment_to_candidate
    :parameters (?detection_pipeline - detection_pipeline ?data_segment - data_segment ?anomaly_candidate - anomaly_candidate)
    :precondition
      (and
        (evaluation_completed ?detection_pipeline)
        (pipeline_associated_with_candidate ?detection_pipeline ?anomaly_candidate)
        (pipeline_has_data_segment ?detection_pipeline ?data_segment)
        (data_segment_available ?data_segment)
        (candidate_allocated ?anomaly_candidate)
        (candidate_ready_for_segment_attachment ?anomaly_candidate)
        (not
          (data_segment_consumed ?data_segment)
        )
      )
    :effect
      (and
        (data_segment_consumed ?data_segment)
        (data_segment_attached_to_candidate ?data_segment ?anomaly_candidate)
        (not
          (data_segment_available ?data_segment)
        )
      )
  )
  (:action complete_pipeline_segment_binding
    :parameters (?detection_pipeline - detection_pipeline ?data_segment - data_segment ?anomaly_candidate - anomaly_candidate ?feature - computed_feature)
    :precondition
      (and
        (evaluation_completed ?detection_pipeline)
        (pipeline_has_data_segment ?detection_pipeline ?data_segment)
        (data_segment_consumed ?data_segment)
        (data_segment_attached_to_candidate ?data_segment ?anomaly_candidate)
        (feature_computed ?detection_pipeline ?feature)
        (not
          (candidate_confirmed_by_raw_votes ?anomaly_candidate)
        )
        (not
          (pipeline_segment_binding_complete ?detection_pipeline)
        )
      )
    :effect (pipeline_segment_binding_complete ?detection_pipeline)
  )
  (:action assign_explanation_method_to_pipeline
    :parameters (?detection_pipeline - detection_pipeline ?explanation_method - explanation_method)
    :precondition
      (and
        (evaluation_completed ?detection_pipeline)
        (explanation_method_available ?explanation_method)
        (not
          (explanation_method_reserved ?detection_pipeline)
        )
      )
    :effect
      (and
        (explanation_method_reserved ?detection_pipeline)
        (pipeline_has_explanation_method ?detection_pipeline ?explanation_method)
        (not
          (explanation_method_available ?explanation_method)
        )
      )
  )
  (:action prepare_pipeline_with_explanation_method
    :parameters (?detection_pipeline - detection_pipeline ?data_segment - data_segment ?anomaly_candidate - anomaly_candidate ?feature - computed_feature ?explanation_method - explanation_method)
    :precondition
      (and
        (evaluation_completed ?detection_pipeline)
        (pipeline_has_data_segment ?detection_pipeline ?data_segment)
        (data_segment_consumed ?data_segment)
        (data_segment_attached_to_candidate ?data_segment ?anomaly_candidate)
        (feature_computed ?detection_pipeline ?feature)
        (candidate_confirmed_by_raw_votes ?anomaly_candidate)
        (explanation_method_reserved ?detection_pipeline)
        (pipeline_has_explanation_method ?detection_pipeline ?explanation_method)
        (not
          (pipeline_segment_binding_complete ?detection_pipeline)
        )
      )
    :effect
      (and
        (pipeline_segment_binding_complete ?detection_pipeline)
        (pipeline_explanation_prepared ?detection_pipeline)
      )
  )
  (:action bind_model_to_pipeline_stage1
    :parameters (?detection_pipeline - detection_pipeline ?scoring_model - scoring_model ?validator - validator ?data_segment - data_segment ?anomaly_candidate - anomaly_candidate)
    :precondition
      (and
        (pipeline_segment_binding_complete ?detection_pipeline)
        (pipeline_has_scoring_model ?detection_pipeline ?scoring_model)
        (assigned_validator ?detection_pipeline ?validator)
        (pipeline_has_data_segment ?detection_pipeline ?data_segment)
        (data_segment_attached_to_candidate ?data_segment ?anomaly_candidate)
        (not
