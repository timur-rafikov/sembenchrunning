--- domain.pddl (as passed to VAL) ---
(define (domain attendance_based_assessment_risk_monitoring_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_actor - object schedule_sequence - object document_asset - object enrollment_supertype - object enrollment_record - enrollment_supertype assessment_slot - resource_actor assessment_component - resource_actor instructor_proctor - resource_actor administrative_approval - resource_actor remediation_resource - resource_actor notification_token - resource_actor grading_override - resource_actor appeal_record - resource_actor makeup_attempt - schedule_sequence evidence_document - schedule_sequence academic_supervisor - schedule_sequence attendance_risk_profile - document_asset exam_session - document_asset intervention_case - document_asset enrollment_subcategory - enrollment_record cohort_association - enrollment_record student_primary - enrollment_subcategory student_secondary - enrollment_subcategory module_enrollment - cohort_association)
  (:predicates
    (entity_registered ?enrollment_record - enrollment_record)
    (entity_confirmed ?enrollment_record - enrollment_record)
    (attempt_scheduled ?enrollment_record - enrollment_record)
    (outcome_released ?enrollment_record - enrollment_record)
    (finalization_ready ?enrollment_record - enrollment_record)
    (ready_for_outcome_release ?enrollment_record - enrollment_record)
    (assessment_slot_available ?assessment_slot - assessment_slot)
    (entity_assigned_assessment_slot ?enrollment_record - enrollment_record ?assessment_slot - assessment_slot)
    (assessment_component_available ?assessment_component - assessment_component)
    (entity_assigned_assessment_component ?enrollment_record - enrollment_record ?assessment_component - assessment_component)
    (instructor_available ?instructor_proctor - instructor_proctor)
    (instructor_assigned_to_entity ?enrollment_record - enrollment_record ?instructor_proctor - instructor_proctor)
    (makeup_attempt_available ?makeup_attempt - makeup_attempt)
    (student_makeup_assignment_primary ?student_primary - student_primary ?makeup_attempt - makeup_attempt)
    (student_makeup_assignment_secondary ?student_secondary - student_secondary ?makeup_attempt - makeup_attempt)
    (student_meets_attendance_criterion ?student_primary - student_primary ?attendance_risk_profile - attendance_risk_profile)
    (attendance_criterion_triggered ?attendance_risk_profile - attendance_risk_profile)
    (attendance_criterion_remediation_required ?attendance_risk_profile - attendance_risk_profile)
    (student_risk_flagged ?student_primary - student_primary)
    (student_assigned_exam_session ?student_secondary - student_secondary ?exam_session - exam_session)
    (exam_session_ready ?exam_session - exam_session)
    (exam_session_remediation_required ?exam_session - exam_session)
    (student_secondary_risk_flagged ?student_secondary - student_secondary)
    (case_staged_for_packaging ?intervention_case - intervention_case)
    (case_packaged ?intervention_case - intervention_case)
    (case_includes_attendance_criterion ?intervention_case - intervention_case ?attendance_risk_profile - attendance_risk_profile)
    (case_includes_exam_session ?intervention_case - intervention_case ?exam_session - exam_session)
    (case_requires_supervisor_review ?intervention_case - intervention_case)
    (case_requires_additional_review ?intervention_case - intervention_case)
    (case_ready_for_evidence_intake ?intervention_case - intervention_case)
    (module_linked_to_primary_student ?course_module - module_enrollment ?student_primary - student_primary)
    (module_linked_to_secondary_student ?course_module - module_enrollment ?student_secondary - student_secondary)
    (module_linked_to_case ?course_module - module_enrollment ?intervention_case - intervention_case)
    (evidence_available ?evidence_document - evidence_document)
    (module_linked_to_evidence ?course_module - module_enrollment ?evidence_document - evidence_document)
    (evidence_consumed ?evidence_document - evidence_document)
    (evidence_linked_to_case ?evidence_document - evidence_document ?intervention_case - intervention_case)
    (module_review_started ?course_module - module_enrollment)
    (module_ready_for_supervisor_review ?course_module - module_enrollment)
    (module_supervisor_reviewed ?course_module - module_enrollment)
    (module_admin_approval_flag ?course_module - module_enrollment)
    (module_review_advanced ?course_module - module_enrollment)
    (module_ready_for_finalization ?course_module - module_enrollment)
    (module_remediation_applied ?course_module - module_enrollment)
    (supervisor_available ?academic_supervisor - academic_supervisor)
    (module_assigned_supervisor ?course_module - module_enrollment ?academic_supervisor - academic_supervisor)
    (module_supervisor_engaged ?course_module - module_enrollment)
    (supervisor_action_recorded ?course_module - module_enrollment)
    (supervisor_approval_granted ?course_module - module_enrollment)
    (administrative_approval_available ?administrative_approval - administrative_approval)
    (module_linked_administrative_approval ?course_module - module_enrollment ?administrative_approval - administrative_approval)
    (remediation_resource_available ?remediation_resource - remediation_resource)
    (module_assigned_remediation_resource ?course_module - module_enrollment ?remediation_resource - remediation_resource)
    (grading_override_available ?grading_override - grading_override)
    (module_has_grading_override ?course_module - module_enrollment ?grading_override - grading_override)
    (appeal_record_available ?appeal_record - appeal_record)
    (module_has_appeal_record ?course_module - module_enrollment ?appeal_record - appeal_record)
    (notification_template_available ?notification_token - notification_token)
    (entity_assigned_notification ?enrollment_record - enrollment_record ?notification_token - notification_token)
    (student_intervention_flag ?student_primary - student_primary)
    (student_secondary_intervention_flag ?student_secondary - student_secondary)
    (module_finalization_marker ?course_module - module_enrollment)
  )
  (:action create_enrollment_record
    :parameters (?enrollment_record - enrollment_record)
    :precondition
      (and
        (not
          (entity_registered ?enrollment_record)
        )
        (not
          (outcome_released ?enrollment_record)
        )
      )
    :effect (entity_registered ?enrollment_record)
  )
  (:action assign_assessment_slot_to_enrollment
    :parameters (?enrollment_record - enrollment_record ?assessment_slot - assessment_slot)
    :precondition
      (and
        (entity_registered ?enrollment_record)
        (not
          (attempt_scheduled ?enrollment_record)
        )
        (assessment_slot_available ?assessment_slot)
      )
    :effect
      (and
        (attempt_scheduled ?enrollment_record)
        (entity_assigned_assessment_slot ?enrollment_record ?assessment_slot)
        (not
          (assessment_slot_available ?assessment_slot)
        )
      )
  )
  (:action assign_assessment_component_to_enrollment
    :parameters (?enrollment_record - enrollment_record ?assessment_component - assessment_component)
    :precondition
      (and
        (entity_registered ?enrollment_record)
        (attempt_scheduled ?enrollment_record)
        (assessment_component_available ?assessment_component)
      )
    :effect
      (and
        (entity_assigned_assessment_component ?enrollment_record ?assessment_component)
        (not
          (assessment_component_available ?assessment_component)
        )
      )
  )
  (:action confirm_assessment_attempt_for_enrollment
    :parameters (?enrollment_record - enrollment_record ?assessment_component - assessment_component)
    :precondition
      (and
        (entity_registered ?enrollment_record)
        (attempt_scheduled ?enrollment_record)
        (entity_assigned_assessment_component ?enrollment_record ?assessment_component)
        (not
          (entity_confirmed ?enrollment_record)
        )
      )
    :effect (entity_confirmed ?enrollment_record)
  )
  (:action cancel_assessment_component_assignment
    :parameters (?enrollment_record - enrollment_record ?assessment_component - assessment_component)
    :precondition
      (and
        (entity_assigned_assessment_component ?enrollment_record ?assessment_component)
      )
    :effect
      (and
        (assessment_component_available ?assessment_component)
        (not
          (entity_assigned_assessment_component ?enrollment_record ?assessment_component)
        )
      )
  )
  (:action assign_instructor_to_enrollment
    :parameters (?enrollment_record - enrollment_record ?instructor_proctor - instructor_proctor)
    :precondition
      (and
        (entity_confirmed ?enrollment_record)
        (instructor_available ?instructor_proctor)
      )
    :effect
      (and
        (instructor_assigned_to_entity ?enrollment_record ?instructor_proctor)
        (not
          (instructor_available ?instructor_proctor)
        )
      )
  )
  (:action release_instructor_from_enrollment
    :parameters (?enrollment_record - enrollment_record ?instructor_proctor - instructor_proctor)
    :precondition
      (and
        (instructor_assigned_to_entity ?enrollment_record ?instructor_proctor)
      )
    :effect
      (and
        (instructor_available ?instructor_proctor)
        (not
          (instructor_assigned_to_entity ?enrollment_record ?instructor_proctor)
        )
      )
  )
  (:action apply_grading_override_to_module
    :parameters (?course_module - module_enrollment ?grading_override - grading_override)
    :precondition
      (and
        (entity_confirmed ?course_module)
        (grading_override_available ?grading_override)
      )
    :effect
      (and
        (module_has_grading_override ?course_module ?grading_override)
        (not
          (grading_override_available ?grading_override)
        )
      )
  )
  (:action revoke_grading_override_from_module
    :parameters (?course_module - module_enrollment ?grading_override - grading_override)
    :precondition
      (and
        (module_has_grading_override ?course_module ?grading_override)
      )
    :effect
      (and
        (grading_override_available ?grading_override)
        (not
          (module_has_grading_override ?course_module ?grading_override)
        )
      )
  )
  (:action attach_appeal_to_module
    :parameters (?course_module - module_enrollment ?appeal_record - appeal_record)
    :precondition
      (and
        (entity_confirmed ?course_module)
        (appeal_record_available ?appeal_record)
      )
    :effect
      (and
        (module_has_appeal_record ?course_module ?appeal_record)
        (not
          (appeal_record_available ?appeal_record)
        )
      )
  )
  (:action detach_appeal_from_module
    :parameters (?course_module - module_enrollment ?appeal_record - appeal_record)
    :precondition
      (and
        (module_has_appeal_record ?course_module ?appeal_record)
      )
    :effect
      (and
        (appeal_record_available ?appeal_record)
        (not
          (module_has_appeal_record ?course_module ?appeal_record)
        )
      )
  )
  (:action trigger_attendance_criterion_for_student
    :parameters (?student_primary - student_primary ?attendance_risk_profile - attendance_risk_profile ?assessment_component - assessment_component)
    :precondition
      (and
        (entity_confirmed ?student_primary)
        (entity_assigned_assessment_component ?student_primary ?assessment_component)
        (student_meets_attendance_criterion ?student_primary ?attendance_risk_profile)
        (not
          (attendance_criterion_triggered ?attendance_risk_profile)
        )
        (not
          (attendance_criterion_remediation_required ?attendance_risk_profile)
        )
      )
    :effect (attendance_criterion_triggered ?attendance_risk_profile)
  )
  (:action instructor_confirm_intervention_for_student
    :parameters (?student_primary - student_primary ?attendance_risk_profile - attendance_risk_profile ?instructor_proctor - instructor_proctor)
    :precondition
      (and
        (entity_confirmed ?student_primary)
        (instructor_assigned_to_entity ?student_primary ?instructor_proctor)
        (student_meets_attendance_criterion ?student_primary ?attendance_risk_profile)
        (attendance_criterion_triggered ?attendance_risk_profile)
        (not
          (student_intervention_flag ?student_primary)
        )
      )
    :effect
      (and
        (student_intervention_flag ?student_primary)
        (student_risk_flagged ?student_primary)
      )
  )
  (:action allocate_makeup_attempt_for_student
    :parameters (?student_primary - student_primary ?attendance_risk_profile - attendance_risk_profile ?makeup_attempt - makeup_attempt)
    :precondition
      (and
        (entity_confirmed ?student_primary)
        (student_meets_attendance_criterion ?student_primary ?attendance_risk_profile)
        (makeup_attempt_available ?makeup_attempt)
        (not
          (student_intervention_flag ?student_primary)
        )
      )
    :effect
      (and
        (attendance_criterion_remediation_required ?attendance_risk_profile)
        (student_intervention_flag ?student_primary)
        (student_makeup_assignment_primary ?student_primary ?makeup_attempt)
        (not
          (makeup_attempt_available ?makeup_attempt)
        )
      )
  )
  (:action complete_makeup_attempt_and_update_flags
    :parameters (?student_primary - student_primary ?attendance_risk_profile - attendance_risk_profile ?assessment_component - assessment_component ?makeup_attempt - makeup_attempt)
    :precondition
      (and
        (entity_confirmed ?student_primary)
        (entity_assigned_assessment_component ?student_primary ?assessment_component)
        (student_meets_attendance_criterion ?student_primary ?attendance_risk_profile)
        (attendance_criterion_remediation_required ?attendance_risk_profile)
        (student_makeup_assignment_primary ?student_primary ?makeup_attempt)
        (not
          (student_risk_flagged ?student_primary)
        )
      )
    :effect
      (and
        (attendance_criterion_triggered ?attendance_risk_profile)
        (student_risk_flagged ?student_primary)
        (makeup_attempt_available ?makeup_attempt)
        (not
          (student_makeup_assignment_primary ?student_primary ?makeup_attempt)
        )
      )
  )
  (:action trigger_exam_session_for_secondary_student
    :parameters (?student_secondary - student_secondary ?exam_session - exam_session ?assessment_component - assessment_component)
    :precondition
      (and
        (entity_confirmed ?student_secondary)
        (entity_assigned_assessment_component ?student_secondary ?assessment_component)
        (student_assigned_exam_session ?student_secondary ?exam_session)
        (not
          (exam_session_ready ?exam_session)
        )
        (not
          (exam_session_remediation_required ?exam_session)
        )
      )
    :effect (exam_session_ready ?exam_session)
  )
  (:action instructor_confirm_intervention_for_secondary_student
    :parameters (?student_secondary - student_secondary ?exam_session - exam_session ?instructor_proctor - instructor_proctor)
    :precondition
      (and
        (entity_confirmed ?student_secondary)
        (instructor_assigned_to_entity ?student_secondary ?instructor_proctor)
        (student_assigned_exam_session ?student_secondary ?exam_session)
        (exam_session_ready ?exam_session)
        (not
          (student_secondary_intervention_flag ?student_secondary)
        )
      )
    :effect
      (and
        (student_secondary_intervention_flag ?student_secondary)
        (student_secondary_risk_flagged ?student_secondary)
      )
  )
  (:action allocate_makeup_attempt_for_secondary_student
    :parameters (?student_secondary - student_secondary ?exam_session - exam_session ?makeup_attempt - makeup_attempt)
    :precondition
      (and
        (entity_confirmed ?student_secondary)
        (student_assigned_exam_session ?student_secondary ?exam_session)
        (makeup_attempt_available ?makeup_attempt)
        (not
          (student_secondary_intervention_flag ?student_secondary)
        )
      )
    :effect
      (and
        (exam_session_remediation_required ?exam_session)
        (student_secondary_intervention_flag ?student_secondary)
        (student_makeup_assignment_secondary ?student_secondary ?makeup_attempt)
        (not
          (makeup_attempt_available ?makeup_attempt)
        )
      )
  )
  (:action complete_makeup_attempt_and_update_flags_secondary
    :parameters (?student_secondary - student_secondary ?exam_session - exam_session ?assessment_component - assessment_component ?makeup_attempt - makeup_attempt)
    :precondition
      (and
        (entity_confirmed ?student_secondary)
        (entity_assigned_assessment_component ?student_secondary ?assessment_component)
        (student_assigned_exam_session ?student_secondary ?exam_session)
        (exam_session_remediation_required ?exam_session)
        (student_makeup_assignment_secondary ?student_secondary ?makeup_attempt)
        (not
          (student_secondary_risk_flagged ?student_secondary)
        )
      )
    :effect
      (and
        (exam_session_ready ?exam_session)
        (student_secondary_risk_flagged ?student_secondary)
        (makeup_attempt_available ?makeup_attempt)
        (not
          (student_makeup_assignment_secondary ?student_secondary ?makeup_attempt)
        )
      )
  )
  (:action assemble_intervention_case_package
    :parameters (?student_primary - student_primary ?student_secondary - student_secondary ?attendance_risk_profile - attendance_risk_profile ?exam_session - exam_session ?intervention_case - intervention_case)
    :precondition
      (and
        (student_intervention_flag ?student_primary)
        (student_secondary_intervention_flag ?student_secondary)
        (student_meets_attendance_criterion ?student_primary ?attendance_risk_profile)
        (student_assigned_exam_session ?student_secondary ?exam_session)
        (attendance_criterion_triggered ?attendance_risk_profile)
        (exam_session_ready ?exam_session)
        (student_risk_flagged ?student_primary)
        (student_secondary_risk_flagged ?student_secondary)
        (case_staged_for_packaging ?intervention_case)
      )
    :effect
      (and
        (case_packaged ?intervention_case)
        (case_includes_attendance_criterion ?intervention_case ?attendance_risk_profile)
        (case_includes_exam_session ?intervention_case ?exam_session)
        (not
          (case_staged_for_packaging ?intervention_case)
        )
      )
  )
  (:action assemble_intervention_case_package_with_attendance_remediation
    :parameters (?student_primary - student_primary ?student_secondary - student_secondary ?attendance_risk_profile - attendance_risk_profile ?exam_session - exam_session ?intervention_case - intervention_case)
    :precondition
      (and
        (student_intervention_flag ?student_primary)
        (student_secondary_intervention_flag ?student_secondary)
        (student_meets_attendance_criterion ?student_primary ?attendance_risk_profile)
        (student_assigned_exam_session ?student_secondary ?exam_session)
        (attendance_criterion_remediation_required ?attendance_risk_profile)
        (exam_session_ready ?exam_session)
        (not
          (student_risk_flagged ?student_primary)
        )
        (student_secondary_risk_flagged ?student_secondary)
        (case_staged_for_packaging ?intervention_case)
      )
    :effect
      (and
        (case_packaged ?intervention_case)
        (case_includes_attendance_criterion ?intervention_case ?attendance_risk_profile)
        (case_includes_exam_session ?intervention_case ?exam_session)
        (case_requires_supervisor_review ?intervention_case)
        (not
          (case_staged_for_packaging ?intervention_case)
        )
      )
  )
  (:action assemble_intervention_case_package_with_session_remediation
    :parameters (?student_primary - student_primary ?student_secondary - student_secondary ?attendance_risk_profile - attendance_risk_profile ?exam_session - exam_session ?intervention_case - intervention_case)
    :precondition
      (and
        (student_intervention_flag ?student_primary)
        (student_secondary_intervention_flag ?student_secondary)
        (student_meets_attendance_criterion ?student_primary ?attendance_risk_profile)
        (student_assigned_exam_session ?student_secondary ?exam_session)
        (attendance_criterion_triggered ?attendance_risk_profile)
        (exam_session_remediation_required ?exam_session)
        (student_risk_flagged ?student_primary)
        (not
          (student_secondary_risk_flagged ?student_secondary)
        )
        (case_staged_for_packaging ?intervention_case)
      )
    :effect
      (and
        (case_packaged ?intervention_case)
        (case_includes_attendance_criterion ?intervention_case ?attendance_risk_profile)
        (case_includes_exam_session ?intervention_case ?exam_session)
        (case_requires_additional_review ?intervention_case)
        (not
          (case_staged_for_packaging ?intervention_case)
        )
      )
  )
  (:action assemble_intervention_case_package_with_both_remediations
    :parameters (?student_primary - student_primary ?student_secondary - student_secondary ?attendance_risk_profile - attendance_risk_profile ?exam_session - exam_session ?intervention_case - intervention_case)
    :precondition
      (and
        (student_intervention_flag ?student_primary)
        (student_secondary_intervention_flag ?student_secondary)
        (student_meets_attendance_criterion ?student_primary ?attendance_risk_profile)
        (student_assigned_exam_session ?student_secondary ?exam_session)
        (attendance_criterion_remediation_required ?attendance_risk_profile)
        (exam_session_remediation_required ?exam_session)
        (not
          (student_risk_flagged ?student_primary)
        )
        (not
          (student_secondary_risk_flagged ?student_secondary)
        )
        (case_staged_for_packaging ?intervention_case)
      )
    :effect
      (and
        (case_packaged ?intervention_case)
        (case_includes_attendance_criterion ?intervention_case ?attendance_risk_profile)
        (case_includes_exam_session ?intervention_case ?exam_session)
        (case_requires_supervisor_review ?intervention_case)
        (case_requires_additional_review ?intervention_case)
        (not
          (case_staged_for_packaging ?intervention_case)
        )
      )
  )
  (:action mark_case_ready_for_evidence_intake
    :parameters (?intervention_case - intervention_case ?student_primary - student_primary ?assessment_component - assessment_component)
    :precondition
      (and
        (case_packaged ?intervention_case)
        (student_intervention_flag ?student_primary)
        (entity_assigned_assessment_component ?student_primary ?assessment_component)
        (not
          (case_ready_for_evidence_intake ?intervention_case)
        )
      )
    :effect (case_ready_for_evidence_intake ?intervention_case)
  )
  (:action ingest_evidence_and_link_to_case
    :parameters (?course_module - module_enrollment ?evidence_document - evidence_document ?intervention_case - intervention_case)
    :precondition
      (and
        (entity_confirmed ?course_module)
        (module_linked_to_case ?course_module ?intervention_case)
        (module_linked_to_evidence ?course_module ?evidence_document)
        (evidence_available ?evidence_document)
        (case_packaged ?intervention_case)
        (case_ready_for_evidence_intake ?intervention_case)
        (not
          (evidence_consumed ?evidence_document)
        )
      )
    :effect
      (and
        (evidence_consumed ?evidence_document)
        (evidence_linked_to_case ?evidence_document ?intervention_case)
        (not
          (evidence_available ?evidence_document)
        )
      )
  )
  (:action start_module_review_after_evidence
    :parameters (?course_module - module_enrollment ?evidence_document - evidence_document ?intervention_case - intervention_case ?assessment_component - assessment_component)
    :precondition
      (and
        (entity_confirmed ?course_module)
        (module_linked_to_evidence ?course_module ?evidence_document)
        (evidence_consumed ?evidence_document)
        (evidence_linked_to_case ?evidence_document ?intervention_case)
        (entity_assigned_assessment_component ?course_module ?assessment_component)
        (not
          (case_requires_supervisor_review ?intervention_case)
        )
        (not
          (module_review_started ?course_module)
        )
      )
    :effect (module_review_started ?course_module)
  )
  (:action attach_administrative_approval_to_module
    :parameters (?course_module - module_enrollment ?administrative_approval - administrative_approval)
    :precondition
      (and
        (entity_confirmed ?course_module)
        (administrative_approval_available ?administrative_approval)
        (not
          (module_admin_approval_flag ?course_module)
        )
      )
    :effect
      (and
        (module_admin_approval_flag ?course_module)
        (module_linked_administrative_approval ?course_module ?administrative_approval)
        (not
          (administrative_approval_available ?administrative_approval)
        )
      )
  )
  (:action advance_module_to_supervisor_review
    :parameters (?course_module - module_enrollment ?evidence_document - evidence_document ?intervention_case - intervention_case ?assessment_component - assessment_component ?administrative_approval - administrative_approval)
    :precondition
      (and
        (entity_confirmed ?course_module)
        (module_linked_to_evidence ?course_module ?evidence_document)
        (evidence_consumed ?evidence_document)
        (evidence_linked_to_case ?evidence_document ?intervention_case)
        (entity_assigned_assessment_component ?course_module ?assessment_component)
        (case_requires_supervisor_review ?intervention_case)
        (module_admin_approval_flag ?course_module)
        (module_linked_administrative_approval ?course_module ?administrative_approval)
        (not
          (module_review_started ?course_module)
        )
      )
    :effect
      (and
        (module_review_started ?course_module)
        (module_review_advanced ?course_module)
      )
  )
