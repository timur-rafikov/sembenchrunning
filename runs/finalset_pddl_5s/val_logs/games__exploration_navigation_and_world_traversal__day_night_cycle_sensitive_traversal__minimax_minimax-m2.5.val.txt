--- domain.pddl (as passed to VAL) ---
(define (domain day_night_traversal)
  (:requirements :strips :typing :negative-preconditions)
  (:types resource_category - object env_object_category - object spatial_zone_category - object traversal_entity_parent - object traversal_entity - traversal_entity_parent route_permit - resource_category scout_report - resource_category companion_unit - resource_category landmark_modifier - resource_category activation_credential - resource_category return_marker - resource_category equipment_module - resource_category time_of_day_tool - resource_category consumable_resource - env_object_category map_fragment - env_object_category objective_token - env_object_category day_zone_token - spatial_zone_category night_zone_token - spatial_zone_category map_tile - spatial_zone_category traversal_subgroup - traversal_entity waypoint_subgroup - traversal_entity day_node - traversal_subgroup night_node - traversal_subgroup waypoint - waypoint_subgroup)
  (:predicates
    (discovered ?traversal_entity - traversal_entity)
    (prepared ?traversal_entity - traversal_entity)
    (permit_assignment ?traversal_entity - traversal_entity)
    (traversal_unlocked ?traversal_entity - traversal_entity)
    (completion_marker ?traversal_entity - traversal_entity)
    (return_prepared ?traversal_entity - traversal_entity)
    (permit_available ?route_permit - route_permit)
    (has_permit ?traversal_entity - traversal_entity ?route_permit - route_permit)
    (scout_available ?scout_report - scout_report)
    (has_scout_report ?traversal_entity - traversal_entity ?scout_report - scout_report)
    (companion_available ?companion_unit - companion_unit)
    (companion_assigned ?traversal_entity - traversal_entity ?companion_unit - companion_unit)
    (consumable_available ?consumable_resource - consumable_resource)
    (consumed_on_day_node ?day_node - day_node ?consumable_resource - consumable_resource)
    (consumed_on_night_node ?night_node - night_node ?consumable_resource - consumable_resource)
    (day_node_zone_link ?day_node - day_node ?day_zone_token - day_zone_token)
    (day_zone_active ?day_zone_token - day_zone_token)
    (day_zone_hazard_handled ?day_zone_token - day_zone_token)
    (day_node_cleared ?day_node - day_node)
    (night_node_zone_link ?night_node - night_node ?night_zone_token - night_zone_token)
    (night_zone_active ?night_zone_token - night_zone_token)
    (night_zone_hazard_handled ?night_zone_token - night_zone_token)
    (night_node_cleared ?night_node - night_node)
    (tile_unprocessed ?map_tile - map_tile)
    (tile_registered ?map_tile - map_tile)
    (tile_has_day_zone ?map_tile - map_tile ?day_zone_token - day_zone_token)
    (tile_has_night_zone ?map_tile - map_tile ?night_zone_token - night_zone_token)
    (tile_day_confirmed ?map_tile - map_tile)
    (tile_night_confirmed ?map_tile - map_tile)
    (tile_validated ?map_tile - map_tile)
    (waypoint_link_day_node ?waypoint - waypoint ?day_node - day_node)
    (waypoint_link_night_node ?waypoint - waypoint ?night_node - night_node)
    (waypoint_on_tile ?waypoint - waypoint ?map_tile - map_tile)
    (fragment_available ?map_fragment - map_fragment)
    (waypoint_has_fragment ?waypoint - waypoint ?map_fragment - map_fragment)
    (fragment_collected ?map_fragment - map_fragment)
    (fragment_on_tile ?map_fragment - map_fragment ?map_tile - map_tile)
    (waypoint_intermediate ?waypoint - waypoint)
    (waypoint_pre_activation ?waypoint - waypoint)
    (waypoint_inspection_ready ?waypoint - waypoint)
    (landmark_modifier_applied ?waypoint - waypoint)
    (landmark_secondary_active ?waypoint - waypoint)
    (waypoint_fully_activated ?waypoint - waypoint)
    (activation_executed ?waypoint - waypoint)
    (objective_available ?objective_token - objective_token)
    (waypoint_has_objective ?waypoint - waypoint ?objective_token - objective_token)
    (objective_acknowledged ?waypoint - waypoint)
    (objective_queued ?waypoint - waypoint)
    (objective_completed ?waypoint - waypoint)
    (landmark_modifier_available ?landmark_modifier - landmark_modifier)
    (waypoint_has_landmark_modifier ?waypoint - waypoint ?landmark_modifier - landmark_modifier)
    (credential_available ?activation_credential - activation_credential)
    (credential_bound ?waypoint - waypoint ?activation_credential - activation_credential)
    (equipment_available ?equipment_module - equipment_module)
    (equipment_installed ?waypoint - waypoint ?equipment_module - equipment_module)
    (time_tool_available ?time_of_day_tool - time_of_day_tool)
    (time_tool_installed ?waypoint - waypoint ?time_of_day_tool - time_of_day_tool)
    (return_marker_available ?return_marker - return_marker)
    (return_marker_bound ?traversal_entity - traversal_entity ?return_marker - return_marker)
    (day_node_contributed ?day_node - day_node)
    (night_node_contributed ?night_node - night_node)
    (waypoint_finalized ?waypoint - waypoint)
  )
  (:action discover_entity
    :parameters (?traversal_entity - traversal_entity)
    :precondition
      (and
        (not
          (discovered ?traversal_entity)
        )
        (not
          (traversal_unlocked ?traversal_entity)
        )
      )
    :effect (discovered ?traversal_entity)
  )
  (:action attach_route_permit
    :parameters (?traversal_entity - traversal_entity ?route_permit - route_permit)
    :precondition
      (and
        (discovered ?traversal_entity)
        (not
          (permit_assignment ?traversal_entity)
        )
        (permit_available ?route_permit)
      )
    :effect
      (and
        (permit_assignment ?traversal_entity)
        (has_permit ?traversal_entity ?route_permit)
        (not
          (permit_available ?route_permit)
        )
      )
  )
  (:action apply_scout_report
    :parameters (?traversal_entity - traversal_entity ?scout_report - scout_report)
    :precondition
      (and
        (discovered ?traversal_entity)
        (permit_assignment ?traversal_entity)
        (scout_available ?scout_report)
      )
    :effect
      (and
        (has_scout_report ?traversal_entity ?scout_report)
        (not
          (scout_available ?scout_report)
        )
      )
  )
  (:action prepare_entity
    :parameters (?traversal_entity - traversal_entity ?scout_report - scout_report)
    :precondition
      (and
        (discovered ?traversal_entity)
        (permit_assignment ?traversal_entity)
        (has_scout_report ?traversal_entity ?scout_report)
        (not
          (prepared ?traversal_entity)
        )
      )
    :effect (prepared ?traversal_entity)
  )
  (:action release_scout_report
    :parameters (?traversal_entity - traversal_entity ?scout_report - scout_report)
    :precondition
      (and
        (has_scout_report ?traversal_entity ?scout_report)
      )
    :effect
      (and
        (scout_available ?scout_report)
        (not
          (has_scout_report ?traversal_entity ?scout_report)
        )
      )
  )
  (:action assign_companion
    :parameters (?traversal_entity - traversal_entity ?companion_unit - companion_unit)
    :precondition
      (and
        (prepared ?traversal_entity)
        (companion_available ?companion_unit)
      )
    :effect
      (and
        (companion_assigned ?traversal_entity ?companion_unit)
        (not
          (companion_available ?companion_unit)
        )
      )
  )
  (:action release_companion
    :parameters (?traversal_entity - traversal_entity ?companion_unit - companion_unit)
    :precondition
      (and
        (companion_assigned ?traversal_entity ?companion_unit)
      )
    :effect
      (and
        (companion_available ?companion_unit)
        (not
          (companion_assigned ?traversal_entity ?companion_unit)
        )
      )
  )
  (:action install_equipment
    :parameters (?waypoint - waypoint ?equipment_module - equipment_module)
    :precondition
      (and
        (prepared ?waypoint)
        (equipment_available ?equipment_module)
      )
    :effect
      (and
        (equipment_installed ?waypoint ?equipment_module)
        (not
          (equipment_available ?equipment_module)
        )
      )
  )
  (:action remove_equipment
    :parameters (?waypoint - waypoint ?equipment_module - equipment_module)
    :precondition
      (and
        (equipment_installed ?waypoint ?equipment_module)
      )
    :effect
      (and
        (equipment_available ?equipment_module)
        (not
          (equipment_installed ?waypoint ?equipment_module)
        )
      )
  )
  (:action install_time_tool
    :parameters (?waypoint - waypoint ?time_of_day_tool - time_of_day_tool)
    :precondition
      (and
        (prepared ?waypoint)
        (time_tool_available ?time_of_day_tool)
      )
    :effect
      (and
        (time_tool_installed ?waypoint ?time_of_day_tool)
        (not
          (time_tool_available ?time_of_day_tool)
        )
      )
  )
  (:action remove_time_tool
    :parameters (?waypoint - waypoint ?time_of_day_tool - time_of_day_tool)
    :precondition
      (and
        (time_tool_installed ?waypoint ?time_of_day_tool)
      )
    :effect
      (and
        (time_tool_available ?time_of_day_tool)
        (not
          (time_tool_installed ?waypoint ?time_of_day_tool)
        )
      )
  )
  (:action activate_day_zone
    :parameters (?day_node - day_node ?day_zone_token - day_zone_token ?scout_report - scout_report)
    :precondition
      (and
        (prepared ?day_node)
        (has_scout_report ?day_node ?scout_report)
        (day_node_zone_link ?day_node ?day_zone_token)
        (not
          (day_zone_active ?day_zone_token)
        )
        (not
          (day_zone_hazard_handled ?day_zone_token)
        )
      )
    :effect (day_zone_active ?day_zone_token)
  )
  (:action clear_day_node_with_companion
    :parameters (?day_node - day_node ?day_zone_token - day_zone_token ?companion_unit - companion_unit)
    :precondition
      (and
        (prepared ?day_node)
        (companion_assigned ?day_node ?companion_unit)
        (day_node_zone_link ?day_node ?day_zone_token)
        (day_zone_active ?day_zone_token)
        (not
          (day_node_contributed ?day_node)
        )
      )
    :effect
      (and
        (day_node_contributed ?day_node)
        (day_node_cleared ?day_node)
      )
  )
  (:action use_consumable_on_day_node
    :parameters (?day_node - day_node ?day_zone_token - day_zone_token ?consumable_resource - consumable_resource)
    :precondition
      (and
        (prepared ?day_node)
        (day_node_zone_link ?day_node ?day_zone_token)
        (consumable_available ?consumable_resource)
        (not
          (day_node_contributed ?day_node)
        )
      )
    :effect
      (and
        (day_zone_hazard_handled ?day_zone_token)
        (day_node_contributed ?day_node)
        (consumed_on_day_node ?day_node ?consumable_resource)
        (not
          (consumable_available ?consumable_resource)
        )
      )
  )
  (:action finalize_day_node_clearing
    :parameters (?day_node - day_node ?day_zone_token - day_zone_token ?scout_report - scout_report ?consumable_resource - consumable_resource)
    :precondition
      (and
        (prepared ?day_node)
        (has_scout_report ?day_node ?scout_report)
        (day_node_zone_link ?day_node ?day_zone_token)
        (day_zone_hazard_handled ?day_zone_token)
        (consumed_on_day_node ?day_node ?consumable_resource)
        (not
          (day_node_cleared ?day_node)
        )
      )
    :effect
      (and
        (day_zone_active ?day_zone_token)
        (day_node_cleared ?day_node)
        (consumable_available ?consumable_resource)
        (not
          (consumed_on_day_node ?day_node ?consumable_resource)
        )
      )
  )
  (:action activate_night_zone
    :parameters (?night_node - night_node ?night_zone_token - night_zone_token ?scout_report - scout_report)
    :precondition
      (and
        (prepared ?night_node)
        (has_scout_report ?night_node ?scout_report)
        (night_node_zone_link ?night_node ?night_zone_token)
        (not
          (night_zone_active ?night_zone_token)
        )
        (not
          (night_zone_hazard_handled ?night_zone_token)
        )
      )
    :effect (night_zone_active ?night_zone_token)
  )
  (:action clear_night_node_with_companion
    :parameters (?night_node - night_node ?night_zone_token - night_zone_token ?companion_unit - companion_unit)
    :precondition
      (and
        (prepared ?night_node)
        (companion_assigned ?night_node ?companion_unit)
        (night_node_zone_link ?night_node ?night_zone_token)
        (night_zone_active ?night_zone_token)
        (not
          (night_node_contributed ?night_node)
        )
      )
    :effect
      (and
        (night_node_contributed ?night_node)
        (night_node_cleared ?night_node)
      )
  )
  (:action use_consumable_on_night_node
    :parameters (?night_node - night_node ?night_zone_token - night_zone_token ?consumable_resource - consumable_resource)
    :precondition
      (and
        (prepared ?night_node)
        (night_node_zone_link ?night_node ?night_zone_token)
        (consumable_available ?consumable_resource)
        (not
          (night_node_contributed ?night_node)
        )
      )
    :effect
      (and
        (night_zone_hazard_handled ?night_zone_token)
        (night_node_contributed ?night_node)
        (consumed_on_night_node ?night_node ?consumable_resource)
        (not
          (consumable_available ?consumable_resource)
        )
      )
  )
  (:action finalize_night_node_clearing
    :parameters (?night_node - night_node ?night_zone_token - night_zone_token ?scout_report - scout_report ?consumable_resource - consumable_resource)
    :precondition
      (and
        (prepared ?night_node)
        (has_scout_report ?night_node ?scout_report)
        (night_node_zone_link ?night_node ?night_zone_token)
        (night_zone_hazard_handled ?night_zone_token)
        (consumed_on_night_node ?night_node ?consumable_resource)
        (not
          (night_node_cleared ?night_node)
        )
      )
    :effect
      (and
        (night_zone_active ?night_zone_token)
        (night_node_cleared ?night_node)
        (consumable_available ?consumable_resource)
        (not
          (consumed_on_night_node ?night_node ?consumable_resource)
        )
      )
  )
  (:action register_map_tile
    :parameters (?day_node - day_node ?night_node - night_node ?day_zone_token - day_zone_token ?night_zone_token - night_zone_token ?map_tile - map_tile)
    :precondition
      (and
        (day_node_contributed ?day_node)
        (night_node_contributed ?night_node)
        (day_node_zone_link ?day_node ?day_zone_token)
        (night_node_zone_link ?night_node ?night_zone_token)
        (day_zone_active ?day_zone_token)
        (night_zone_active ?night_zone_token)
        (day_node_cleared ?day_node)
        (night_node_cleared ?night_node)
        (tile_unprocessed ?map_tile)
      )
    :effect
      (and
        (tile_registered ?map_tile)
        (tile_has_day_zone ?map_tile ?day_zone_token)
        (tile_has_night_zone ?map_tile ?night_zone_token)
        (not
          (tile_unprocessed ?map_tile)
        )
      )
  )
  (:action register_map_tile_day_variant
    :parameters (?day_node - day_node ?night_node - night_node ?day_zone_token - day_zone_token ?night_zone_token - night_zone_token ?map_tile - map_tile)
    :precondition
      (and
        (day_node_contributed ?day_node)
        (night_node_contributed ?night_node)
        (day_node_zone_link ?day_node ?day_zone_token)
        (night_node_zone_link ?night_node ?night_zone_token)
        (day_zone_hazard_handled ?day_zone_token)
        (night_zone_active ?night_zone_token)
        (not
          (day_node_cleared ?day_node)
        )
        (night_node_cleared ?night_node)
        (tile_unprocessed ?map_tile)
      )
    :effect
      (and
        (tile_registered ?map_tile)
        (tile_has_day_zone ?map_tile ?day_zone_token)
        (tile_has_night_zone ?map_tile ?night_zone_token)
        (tile_day_confirmed ?map_tile)
        (not
          (tile_unprocessed ?map_tile)
        )
      )
  )
  (:action register_map_tile_night_variant
    :parameters (?day_node - day_node ?night_node - night_node ?day_zone_token - day_zone_token ?night_zone_token - night_zone_token ?map_tile - map_tile)
    :precondition
      (and
        (day_node_contributed ?day_node)
        (night_node_contributed ?night_node)
        (day_node_zone_link ?day_node ?day_zone_token)
        (night_node_zone_link ?night_node ?night_zone_token)
        (day_zone_active ?day_zone_token)
        (night_zone_hazard_handled ?night_zone_token)
        (day_node_cleared ?day_node)
        (not
          (night_node_cleared ?night_node)
        )
        (tile_unprocessed ?map_tile)
      )
    :effect
      (and
        (tile_registered ?map_tile)
        (tile_has_day_zone ?map_tile ?day_zone_token)
        (tile_has_night_zone ?map_tile ?night_zone_token)
        (tile_night_confirmed ?map_tile)
        (not
          (tile_unprocessed ?map_tile)
        )
      )
  )
  (:action register_map_tile_both_variant
    :parameters (?day_node - day_node ?night_node - night_node ?day_zone_token - day_zone_token ?night_zone_token - night_zone_token ?map_tile - map_tile)
    :precondition
      (and
        (day_node_contributed ?day_node)
        (night_node_contributed ?night_node)
        (day_node_zone_link ?day_node ?day_zone_token)
        (night_node_zone_link ?night_node ?night_zone_token)
        (day_zone_hazard_handled ?day_zone_token)
        (night_zone_hazard_handled ?night_zone_token)
        (not
          (day_node_cleared ?day_node)
        )
        (not
          (night_node_cleared ?night_node)
        )
        (tile_unprocessed ?map_tile)
      )
    :effect
      (and
        (tile_registered ?map_tile)
        (tile_has_day_zone ?map_tile ?day_zone_token)
        (tile_has_night_zone ?map_tile ?night_zone_token)
        (tile_day_confirmed ?map_tile)
        (tile_night_confirmed ?map_tile)
        (not
          (tile_unprocessed ?map_tile)
        )
      )
  )
  (:action validate_map_tile
    :parameters (?map_tile - map_tile ?day_node - day_node ?scout_report - scout_report)
    :precondition
      (and
        (tile_registered ?map_tile)
        (day_node_contributed ?day_node)
        (has_scout_report ?day_node ?scout_report)
        (not
          (tile_validated ?map_tile)
        )
      )
    :effect (tile_validated ?map_tile)
  )
  (:action collect_map_fragment
    :parameters (?waypoint - waypoint ?map_fragment - map_fragment ?map_tile - map_tile)
    :precondition
      (and
        (prepared ?waypoint)
        (waypoint_on_tile ?waypoint ?map_tile)
        (waypoint_has_fragment ?waypoint ?map_fragment)
        (fragment_available ?map_fragment)
        (tile_registered ?map_tile)
        (tile_validated ?map_tile)
        (not
          (fragment_collected ?map_fragment)
        )
      )
    :effect
      (and
        (fragment_collected ?map_fragment)
        (fragment_on_tile ?map_fragment ?map_tile)
        (not
          (fragment_available ?map_fragment)
        )
      )
  )
  (:action enter_waypoint_intermediate
    :parameters (?waypoint - waypoint ?map_fragment - map_fragment ?map_tile - map_tile ?scout_report - scout_report)
    :precondition
      (and
        (prepared ?waypoint)
        (waypoint_has_fragment ?waypoint ?map_fragment)
        (fragment_collected ?map_fragment)
        (fragment_on_tile ?map_fragment ?map_tile)
        (has_scout_report ?waypoint ?scout_report)
        (not
          (tile_day_confirmed ?map_tile)
        )
        (not
          (waypoint_intermediate ?waypoint)
        )
      )
    :effect (waypoint_intermediate ?waypoint)
  )
  (:action apply_landmark_modifier
    :parameters (?waypoint - waypoint ?landmark_modifier - landmark_modifier)
    :precondition
      (and
        (prepared ?waypoint)
        (landmark_modifier_available ?landmark_modifier)
        (not
          (landmark_modifier_applied ?waypoint)
        )
      )
    :effect
      (and
        (landmark_modifier_applied ?waypoint)
        (waypoint_has_landmark_modifier ?waypoint ?landmark_modifier)
        (not
          (landmark_modifier_available ?landmark_modifier)
        )
      )
  )
  (:action configure_waypoint_with_landmark
    :parameters (?waypoint - waypoint ?map_fragment - map_fragment ?map_tile - map_tile ?scout_report - scout_report ?landmark_modifier - landmark_modifier)
    :precondition
      (and
        (prepared ?waypoint)
        (waypoint_has_fragment ?waypoint ?map_fragment)
        (fragment_collected ?map_fragment)
        (fragment_on_tile ?map_fragment ?map_tile)
        (has_scout_report ?waypoint ?scout_report)
        (tile_day_confirmed ?map_tile)
        (landmark_modifier_applied ?waypoint)
        (waypoint_has_landmark_modifier ?waypoint ?landmark_modifier)
        (not
          (waypoint_intermediate ?waypoint)
        )
      )
    :effect
      (and
        (waypoint_intermediate ?waypoint)
        (landmark_secondary_active ?waypoint)
      )
  )
  (:action start_waypoint_pre_activation
    :parameters (?waypoint - waypoint ?equipment_module - equipment_module ?companion_unit - companion_unit ?map_fragment - map_fragment ?map_tile - map_tile)
    :precondition
      (and
        (waypoint_intermediate ?waypoint)
        (equipment_installed ?waypoint ?equipment_module)
        (companion_assigned ?waypoint ?companion_unit)
        (waypoint_has_fragment ?waypoint ?map_fragment)
        (fragment_on_tile ?map_fragment ?map_tile)
        (not
          (tile_night_confirmed ?map_tile)
        )
        (not
          (waypoint_pre_activation ?waypoint)
        )
      )
    :effect (waypoint_pre_activation ?waypoint)
  )
  (:action confirm_waypoint_pre_activation
    :parameters (?waypoint - waypoint ?equipment_module - equipment_module ?companion_unit - companion_unit ?map_fragment - map_fragment ?map_tile - map_tile)
    :precondition
      (and
        (waypoint_intermediate ?waypoint)
        (equipment_installed ?waypoint ?equipment_module)
        (companion_assigned ?waypoint ?companion_unit)
        (waypoint_has_fragment ?waypoint ?map_fragment)
        (fragment_on_tile ?map_fragment ?map_tile)
        (tile_night_confirmed ?map_tile)
        (not
          (waypoint_pre_activation ?waypoint)
        )
      )
    :effect (waypoint_pre_activation ?waypoint)
  )
  (:action mark_waypoint_inspection_ready
    :parameters (?waypoint - waypoint ?time_of_day_tool - time_of_day_tool ?map_fragment - map_fragment ?map_tile - map_tile)
    :precondition
      (and
        (waypoint_pre_activation ?waypoint)
        (time_tool_installed ?waypoint ?time_of_day_tool)
        (waypoint_has_fragment ?waypoint ?map_fragment)
        (fragment_on_tile ?map_fragment ?map_tile)
        (not
          (tile_day_confirmed ?map_tile)
        )
        (not
          (tile_night_confirmed ?map_tile)
        )
        (not
          (waypoint_inspection_ready ?waypoint)
        )
      )
    :effect (waypoint_inspection_ready ?waypoint)
  )
  (:action confirm_waypoint_full_activation_day
    :parameters (?waypoint - waypoint ?time_of_day_tool - time_of_day_tool ?map_fragment - map_fragment ?map_tile - map_tile)
    :precondition
