    (entity_input_spec_attached ?allocatable_entity - allocatable_entity ?input_specification - input_specification)
    (crew_unit_available ?crew_unit - crew_unit)
    (crew_assigned_to_request ?allocatable_entity - allocatable_entity ?crew_unit - crew_unit)
    (consumable_available ?consumable_batch - consumable_batch)
    (consumable_allocated_to_field ?field_unit - field_unit ?consumable_batch - consumable_batch)
    (consumable_allocated_to_block ?crop_block - crop_block ?consumable_batch - consumable_batch)
    (local_window_for_field ?field_unit - field_unit ?local_time_window - local_time_window)
    (local_window_selected ?local_time_window - local_time_window)
    (local_window_material_reserved ?local_time_window - local_time_window)

    (regional_window_material_reserved ?regional_time_window - regional_time_window)

    (slot_pending_bundling ?service_slot_record - service_slot_record)
    (slot_bundled ?service_slot_record - service_slot_record)
    (slot_local_window_attached ?service_slot_record - service_slot_record ?local_time_window - local_time_window)
    (slot_regional_window_attached ?service_slot_record - service_slot_record ?regional_time_window - regional_time_window)
    (slot_local_readiness_marker ?service_slot_record - service_slot_record)
    (slot_regional_readiness_marker ?service_slot_record - service_slot_record)
    (slot_availability_confirmed ?service_slot_record - service_slot_record)
    (operation_assigned_field ?service_operation - service_operation ?field_unit - field_unit)
    (operation_assigned_block ?service_operation - service_operation ?crop_block - crop_block)
    (operation_slot_link ?service_operation - service_operation ?service_slot_record - service_slot_record)
    (material_batch_available ?material_batch - material_batch)
    (operation_material_link ?service_operation - service_operation ?material_batch - material_batch)
    (material_reserved ?material_batch - material_batch)
    (material_allocated_to_slot ?material_batch - material_batch ?service_slot_record - service_slot_record)
    (operation_prepared ?service_operation - service_operation)
    (operation_materials_confirmed ?service_operation - service_operation)
    (operation_ready_for_final_checks ?service_operation - service_operation)

    (operation_permit_verified ?service_operation - service_operation)
    (priority_tag_attached ?service_operation - service_operation)
    (operation_final_checks_passed ?service_operation - service_operation)
    (voucher_available ?incentive_voucher - incentive_voucher)
    (voucher_attached_to_operation ?service_operation - service_operation ?incentive_voucher - incentive_voucher)
    (voucher_allocated_to_operation ?service_operation - service_operation)
    (operation_voucher_and_crew_confirmed ?service_operation - service_operation)
    (operation_specialist_tool_confirmed ?service_operation - service_operation)
    (permit_available ?permit_certification - permit_certification)
    (operation_permit_attached ?service_operation - service_operation ?permit_certification - permit_certification)

    (operation_machinery_allocated ?service_operation - service_operation ?machinery_asset - machinery_asset)

    (contingency_available ?contingency_reserve - contingency_reserve)
    (contingency_allocated_to_entity ?allocatable_entity - allocatable_entity ?contingency_reserve - contingency_reserve)

    (operation_finalized ?service_operation - service_operation)
  )
  (:action register_contract_request
    :parameters (?allocatable_entity - allocatable_entity)
    :precondition
      (and
        (not
          (request_registered ?allocatable_entity)
        )
        (not
          (entity_allocation_committed ?allocatable_entity)
        )
      )
    :effect (request_registered ?allocatable_entity)
  )
  (:action provisionally_assign_provider
    :parameters (?allocatable_entity - allocatable_entity ?service_provider - service_provider)
    :precondition
      (and
        (request_registered ?allocatable_entity)
        (not
          (provider_provisionally_assigned ?allocatable_entity)
        )
        (provider_available ?service_provider)
      )
    :effect
      (and
        (provider_provisionally_assigned ?allocatable_entity)
        (provider_assigned_to_entity ?allocatable_entity ?service_provider)
        (not
          (provider_available ?service_provider)
        )
      )
  )
  (:action attach_input_spec_to_request
    :parameters (?allocatable_entity - allocatable_entity ?input_specification - input_specification)
    :precondition
      (and
        (request_registered ?allocatable_entity)
        (provider_provisionally_assigned ?allocatable_entity)
        (input_spec_available ?input_specification)
      )
    :effect
      (and
        (entity_input_spec_attached ?allocatable_entity ?input_specification)
        (not
          (input_spec_available ?input_specification)
        )
      )
  )
  (:action confirm_input_spec_attachment
    :parameters (?allocatable_entity - allocatable_entity ?input_specification - input_specification)
    :precondition
      (and
        (request_registered ?allocatable_entity)
        (provider_provisionally_assigned ?allocatable_entity)
        (entity_input_spec_attached ?allocatable_entity ?input_specification)
        (not
          (entity_confirmed ?allocatable_entity)
        )
      )
    :effect (entity_confirmed ?allocatable_entity)
  )
  (:action release_input_spec_from_request
    :parameters (?allocatable_entity - allocatable_entity ?input_specification - input_specification)
    :precondition
      (and
        (entity_input_spec_attached ?allocatable_entity ?input_specification)
      )
    :effect
      (and
        (input_spec_available ?input_specification)
        (not
          (entity_input_spec_attached ?allocatable_entity ?input_specification)
        )
      )
  )
  (:action assign_crew_unit_to_request
    :parameters (?allocatable_entity - allocatable_entity ?crew_unit - crew_unit)
    :precondition
      (and
        (entity_confirmed ?allocatable_entity)
        (crew_unit_available ?crew_unit)
      )
    :effect
      (and
        (crew_assigned_to_request ?allocatable_entity ?crew_unit)
        (not
          (crew_unit_available ?crew_unit)
        )
      )
  )
  (:action release_crew_from_request
    :parameters (?allocatable_entity - allocatable_entity ?crew_unit - crew_unit)
    :precondition
      (and
        (crew_assigned_to_request ?allocatable_entity ?crew_unit)
      )
    :effect
      (and
        (crew_unit_available ?crew_unit)
        (not
          (crew_assigned_to_request ?allocatable_entity ?crew_unit)
        )
      )
  )
  (:action reserve_machinery_for_operation
    :parameters (?service_operation - service_operation ?machinery_asset - machinery_asset)
    :precondition
      (and
        (entity_confirmed ?service_operation)
        (machinery_available ?machinery_asset)
      )
    :effect
      (and
        (operation_machinery_allocated ?service_operation ?machinery_asset)
        (not
          (machinery_available ?machinery_asset)
        )
      )
  )
  (:action release_machinery_from_operation
    :parameters (?service_operation - service_operation ?machinery_asset - machinery_asset)
    :precondition
      (and
        (operation_machinery_allocated ?service_operation ?machinery_asset)
      )
    :effect
      (and
        (machinery_available ?machinery_asset)
        (not
          (operation_machinery_allocated ?service_operation ?machinery_asset)
        )
      )
  )
  (:action reserve_specialist_tool_for_operation
    :parameters (?service_operation - service_operation ?specialist_tool - specialist_tool)
    :precondition
      (and
        (entity_confirmed ?service_operation)
        (specialist_tool_available ?specialist_tool)
      )
    :effect
      (and
        (tool_allocated_to_operation ?service_operation ?specialist_tool)
        (not
          (specialist_tool_available ?specialist_tool)
        )
      )
  )
  (:action release_specialist_tool_from_operation
    :parameters (?service_operation - service_operation ?specialist_tool - specialist_tool)
    :precondition
      (and
        (tool_allocated_to_operation ?service_operation ?specialist_tool)
      )
    :effect
      (and
        (specialist_tool_available ?specialist_tool)
        (not
          (tool_allocated_to_operation ?service_operation ?specialist_tool)
        )
      )
  )
  (:action create_field_level_window_candidate
    :parameters (?field_unit - field_unit ?local_time_window - local_time_window ?input_specification - input_specification)
    :precondition
      (and
        (entity_confirmed ?field_unit)
        (entity_input_spec_attached ?field_unit ?input_specification)
        (local_window_for_field ?field_unit ?local_time_window)
        (not
          (local_window_selected ?local_time_window)
        )
        (not
          (local_window_material_reserved ?local_time_window)
        )
      )
    :effect (local_window_selected ?local_time_window)
  )
  (:action confirm_field_candidate_with_crew
    :parameters (?field_unit - field_unit ?local_time_window - local_time_window ?crew_unit - crew_unit)
    :precondition
      (and
        (entity_confirmed ?field_unit)
        (crew_assigned_to_request ?field_unit ?crew_unit)
        (local_window_for_field ?field_unit ?local_time_window)
        (local_window_selected ?local_time_window)
        (not
          (field_candidate_selected ?field_unit)
        )
      )
    :effect
      (and
        (field_candidate_selected ?field_unit)
        (field_ready ?field_unit)
      )
  )
  (:action reserve_consumable_for_field_candidate
    :parameters (?field_unit - field_unit ?local_time_window - local_time_window ?consumable_batch - consumable_batch)
    :precondition
      (and
        (entity_confirmed ?field_unit)
        (local_window_for_field ?field_unit ?local_time_window)
        (consumable_available ?consumable_batch)
        (not
          (field_candidate_selected ?field_unit)
        )
      )
    :effect
      (and
        (local_window_material_reserved ?local_time_window)
        (field_candidate_selected ?field_unit)
        (consumable_allocated_to_field ?field_unit ?consumable_batch)
        (not
          (consumable_available ?consumable_batch)
        )
      )
  )
  (:action finalize_field_candidate_and_release_consumable
    :parameters (?field_unit - field_unit ?local_time_window - local_time_window ?input_specification - input_specification ?consumable_batch - consumable_batch)
    :precondition
      (and
        (entity_confirmed ?field_unit)
        (entity_input_spec_attached ?field_unit ?input_specification)
        (local_window_for_field ?field_unit ?local_time_window)
        (local_window_material_reserved ?local_time_window)
        (consumable_allocated_to_field ?field_unit ?consumable_batch)
        (not
          (field_ready ?field_unit)
        )
      )
    :effect
      (and
        (local_window_selected ?local_time_window)
        (field_ready ?field_unit)
        (consumable_available ?consumable_batch)
        (not
          (consumable_allocated_to_field ?field_unit ?consumable_batch)
        )
      )
  )
  (:action create_block_level_window_candidate
    :parameters (?crop_block - crop_block ?regional_time_window - regional_time_window ?input_specification - input_specification)
    :precondition
      (and
        (entity_confirmed ?crop_block)
        (entity_input_spec_attached ?crop_block ?input_specification)
        (regional_window_for_block ?crop_block ?regional_time_window)
        (not
          (regional_window_selected ?regional_time_window)
        )
        (not
          (regional_window_material_reserved ?regional_time_window)
        )
      )
    :effect (regional_window_selected ?regional_time_window)
  )
  (:action confirm_block_candidate_with_crew
    :parameters (?crop_block - crop_block ?regional_time_window - regional_time_window ?crew_unit - crew_unit)
    :precondition
      (and
        (entity_confirmed ?crop_block)
        (crew_assigned_to_request ?crop_block ?crew_unit)
        (regional_window_for_block ?crop_block ?regional_time_window)
        (regional_window_selected ?regional_time_window)
        (not
          (block_candidate_selected ?crop_block)
        )
      )
    :effect
      (and
        (block_candidate_selected ?crop_block)
        (block_ready ?crop_block)
      )
  )
  (:action reserve_consumable_for_block_candidate
    :parameters (?crop_block - crop_block ?regional_time_window - regional_time_window ?consumable_batch - consumable_batch)
    :precondition
      (and
        (entity_confirmed ?crop_block)
        (regional_window_for_block ?crop_block ?regional_time_window)
        (consumable_available ?consumable_batch)
        (not
          (block_candidate_selected ?crop_block)
        )
      )
    :effect
      (and
        (regional_window_material_reserved ?regional_time_window)
        (block_candidate_selected ?crop_block)
        (consumable_allocated_to_block ?crop_block ?consumable_batch)
        (not
          (consumable_available ?consumable_batch)
        )
      )
  )
  (:action finalize_block_candidate_and_release_consumable
    :parameters (?crop_block - crop_block ?regional_time_window - regional_time_window ?input_specification - input_specification ?consumable_batch - consumable_batch)
    :precondition
      (and
        (entity_confirmed ?crop_block)
        (entity_input_spec_attached ?crop_block ?input_specification)
        (regional_window_for_block ?crop_block ?regional_time_window)
        (regional_window_material_reserved ?regional_time_window)
        (consumable_allocated_to_block ?crop_block ?consumable_batch)
        (not
          (block_ready ?crop_block)
        )
      )
    :effect
      (and
        (regional_window_selected ?regional_time_window)
        (block_ready ?crop_block)
        (consumable_available ?consumable_batch)
        (not
          (consumable_allocated_to_block ?crop_block ?consumable_batch)
        )
      )
  )
  (:action bundle_service_slot_primary
    :parameters (?field_unit - field_unit ?crop_block - crop_block ?local_time_window - local_time_window ?regional_time_window - regional_time_window ?service_slot_record - service_slot_record)
    :precondition
      (and
        (field_candidate_selected ?field_unit)
        (block_candidate_selected ?crop_block)
        (local_window_for_field ?field_unit ?local_time_window)
        (regional_window_for_block ?crop_block ?regional_time_window)
        (local_window_selected ?local_time_window)
        (regional_window_selected ?regional_time_window)
        (field_ready ?field_unit)
        (block_ready ?crop_block)
        (slot_pending_bundling ?service_slot_record)
      )
    :effect
      (and
        (slot_bundled ?service_slot_record)
        (slot_local_window_attached ?service_slot_record ?local_time_window)
        (slot_regional_window_attached ?service_slot_record ?regional_time_window)
        (not
          (slot_pending_bundling ?service_slot_record)
        )
      )
  )
  (:action bundle_service_slot_variant_local
    :parameters (?field_unit - field_unit ?crop_block - crop_block ?local_time_window - local_time_window ?regional_time_window - regional_time_window ?service_slot_record - service_slot_record)
    :precondition
      (and
        (field_candidate_selected ?field_unit)
        (block_candidate_selected ?crop_block)
        (local_window_for_field ?field_unit ?local_time_window)
        (regional_window_for_block ?crop_block ?regional_time_window)
        (local_window_material_reserved ?local_time_window)
        (regional_window_selected ?regional_time_window)
        (not
          (field_ready ?field_unit)
        )
        (block_ready ?crop_block)
        (slot_pending_bundling ?service_slot_record)
      )
    :effect
      (and
        (slot_bundled ?service_slot_record)
        (slot_local_window_attached ?service_slot_record ?local_time_window)
        (slot_regional_window_attached ?service_slot_record ?regional_time_window)
        (slot_local_readiness_marker ?service_slot_record)
        (not
          (slot_pending_bundling ?service_slot_record)
        )
      )
  )
  (:action bundle_service_slot_variant_regional
    :parameters (?field_unit - field_unit ?crop_block - crop_block ?local_time_window - local_time_window ?regional_time_window - regional_time_window ?service_slot_record - service_slot_record)
    :precondition
      (and
        (field_candidate_selected ?field_unit)
        (block_candidate_selected ?crop_block)
        (local_window_for_field ?field_unit ?local_time_window)
        (regional_window_for_block ?crop_block ?regional_time_window)
        (local_window_selected ?local_time_window)
        (regional_window_material_reserved ?regional_time_window)
        (field_ready ?field_unit)
        (not
          (block_ready ?crop_block)
        )
        (slot_pending_bundling ?service_slot_record)
      )
    :effect
      (and
        (slot_bundled ?service_slot_record)
        (slot_local_window_attached ?service_slot_record ?local_time_window)
        (slot_regional_window_attached ?service_slot_record ?regional_time_window)
        (slot_regional_readiness_marker ?service_slot_record)
        (not
          (slot_pending_bundling ?service_slot_record)
        )
      )
  )
  (:action bundle_service_slot_variant_local_and_regional
    :parameters (?field_unit - field_unit ?crop_block - crop_block ?local_time_window - local_time_window ?regional_time_window - regional_time_window ?service_slot_record - service_slot_record)
    :precondition
      (and
        (field_candidate_selected ?field_unit)
        (block_candidate_selected ?crop_block)
        (local_window_for_field ?field_unit ?local_time_window)
        (regional_window_for_block ?crop_block ?regional_time_window)
        (local_window_material_reserved ?local_time_window)
        (regional_window_material_reserved ?regional_time_window)
        (not
          (field_ready ?field_unit)
        )
        (not
          (block_ready ?crop_block)
        )
        (slot_pending_bundling ?service_slot_record)
      )
    :effect
      (and
        (slot_bundled ?service_slot_record)
        (slot_local_window_attached ?service_slot_record ?local_time_window)
        (slot_regional_window_attached ?service_slot_record ?regional_time_window)
        (slot_local_readiness_marker ?service_slot_record)
        (slot_regional_readiness_marker ?service_slot_record)
        (not
          (slot_pending_bundling ?service_slot_record)
        )
      )
  )
  (:action validate_slot_availability_for_deployment
    :parameters (?service_slot_record - service_slot_record ?field_unit - field_unit ?input_specification - input_specification)
    :precondition
      (and
        (slot_bundled ?service_slot_record)
        (field_candidate_selected ?field_unit)
        (entity_input_spec_attached ?field_unit ?input_specification)
        (not
          (slot_availability_confirmed ?service_slot_record)
        )
      )
    :effect (slot_availability_confirmed ?service_slot_record)
  )
  (:action attach_material_batch_to_slot
    :parameters (?service_operation - service_operation ?material_batch - material_batch ?service_slot_record - service_slot_record)
    :precondition
      (and
        (entity_confirmed ?service_operation)
        (operation_slot_link ?service_operation ?service_slot_record)
        (operation_material_link ?service_operation ?material_batch)
        (material_batch_available ?material_batch)
        (slot_bundled ?service_slot_record)
        (slot_availability_confirmed ?service_slot_record)
        (not
          (material_reserved ?material_batch)
        )
      )
    :effect
      (and
        (material_reserved ?material_batch)
        (material_allocated_to_slot ?material_batch ?service_slot_record)
        (not
          (material_batch_available ?material_batch)
        )
      )
  )
  (:action confirm_operation_materials
    :parameters (?service_operation - service_operation ?material_batch - material_batch ?service_slot_record - service_slot_record ?input_specification - input_specification)
    :precondition
      (and
        (entity_confirmed ?service_operation)
        (operation_material_link ?service_operation ?material_batch)
        (material_reserved ?material_batch)
        (material_allocated_to_slot ?material_batch ?service_slot_record)
        (entity_input_spec_attached ?service_operation ?input_specification)
        (not
          (slot_local_readiness_marker ?service_slot_record)
        )
        (not
          (operation_prepared ?service_operation)
        )
      )
    :effect (operation_prepared ?service_operation)
  )
  (:action assign_permit_to_operation
    :parameters (?service_operation - service_operation ?permit_certification - permit_certification)
    :precondition
      (and
        (entity_confirmed ?service_operation)
        (permit_available ?permit_certification)
        (not
          (operation_has_permit ?service_operation)
        )
      )
    :effect
      (and
        (operation_has_permit ?service_operation)
        (operation_permit_attached ?service_operation ?permit_certification)
        (not
          (permit_available ?permit_certification)
        )
      )
  )
  (:action verify_operation_permit_and_readiness
    :parameters (?service_operation - service_operation ?material_batch - material_batch ?service_slot_record - service_slot_record ?input_specification - input_specification ?permit_certification - permit_certification)
    :precondition
      (and
        (entity_confirmed ?service_operation)
        (operation_material_link ?service_operation ?material_batch)
        (material_reserved ?material_batch)
        (material_allocated_to_slot ?material_batch ?service_slot_record)
        (entity_input_spec_attached ?service_operation ?input_specification)
        (slot_local_readiness_marker ?service_slot_record)
        (operation_has_permit ?service_operation)
        (operation_permit_attached ?service_operation ?permit_certification)
        (not
          (operation_prepared ?service_operation)
        )
      )
    :effect
      (and
        (operation_prepared ?service_operation)
        (operation_permit_verified ?service_operation)
      )
  )
  (:action confirm_operation_machinery_and_crew_local_flow
    :parameters (?service_operation - service_operation ?machinery_asset - machinery_asset ?crew_unit - crew_unit ?material_batch - material_batch ?service_slot_record - service_slot_record)
    :precondition
      (and
        (operation_prepared ?service_operation)
        (operation_machinery_allocated ?service_operation ?machinery_asset)
