  (:predicates
    (suspect_case_reported ?suspect_product_case - suspect_product_case)

    (containment_executed ?suspect_product_case - suspect_product_case)
    (case_ready_for_notification ?suspect_product_case - suspect_product_case)
    (regulatory_authorization_received ?suspect_product_case - suspect_product_case)
    (investigator_available ?investigator_resource - investigator_resource)
    (investigator_assigned_to_case ?suspect_product_case - suspect_product_case ?investigator_resource - investigator_resource)
    (lab_sample_available ?lab_sample - lab_sample)
    (sample_collected_for_case ?suspect_product_case - suspect_product_case ?lab_sample - lab_sample)
    (response_team_available ?response_team_resource - response_team_resource)
    (response_team_assigned_to_case ?suspect_product_case - suspect_product_case ?response_team_resource - response_team_resource)
    (mitigation_resource_available ?mitigation_resource - mitigation_resource)
    (mitigation_allocated_to_manufacturer_site_case ?manufacturer_site_case - manufacturer_site_case ?mitigation_resource - mitigation_resource)
    (mitigation_allocated_to_distributor_site_case ?distributor_site_case - distributor_site_case ?mitigation_resource - mitigation_resource)
    (manufacturer_site_case_affects_site_node ?manufacturer_site_case - manufacturer_site_case ?manufacturer_site_node - manufacturer_site_node)
    (manufacturer_node_quarantined ?manufacturer_site_node - manufacturer_site_node)
    (manufacturer_node_mitigation_applied ?manufacturer_site_node - manufacturer_site_node)
    (manufacturer_site_case_ready_for_recall ?manufacturer_site_case - manufacturer_site_case)
    (distributor_site_case_affects_site_node ?distributor_site_case - distributor_site_case ?distributor_site_node - distributor_site_node)

    (distributor_node_mitigation_applied ?distributor_site_node - distributor_site_node)
    (distributor_site_case_ready_for_recall ?distributor_site_case - distributor_site_case)
    (recall_event_available ?recall_event - recall_event)
    (recall_event_active ?recall_event - recall_event)
    (recall_event_targets_manufacturer_node ?recall_event - recall_event ?manufacturer_site_node - manufacturer_site_node)
    (recall_event_targets_distributor_node ?recall_event - recall_event ?distributor_site_node - distributor_site_node)
    (recall_event_includes_manufacturer_nodes ?recall_event - recall_event)
    (recall_event_includes_distributor_nodes ?recall_event - recall_event)
    (recall_event_finalized ?recall_event - recall_event)
    (coordinated_case_includes_manufacturer_site_case ?coordinated_response_case - coordinated_response_case ?manufacturer_site_case - manufacturer_site_case)
    (coordinated_case_includes_distributor_site_case ?coordinated_response_case - coordinated_response_case ?distributor_site_case - distributor_site_case)

    (coordinated_case_has_lab_report ?coordinated_response_case - coordinated_response_case ?lab_result_report - lab_result_report)
    (lab_report_bound ?lab_result_report - lab_result_report)
    (lab_report_linked_to_recall ?lab_result_report - lab_result_report ?recall_event - recall_event)

    (communication_channel_selected ?coordinated_response_case - coordinated_response_case)
    (final_approval_recorded ?coordinated_response_case - coordinated_response_case)

    (legal_signoff_obtained ?coordinated_response_case - coordinated_response_case)

    (coordinated_case_has_crisis_plan ?coordinated_response_case - coordinated_response_case ?crisis_plan - crisis_plan)

    (manufacturer_site_case_quarantined ?manufacturer_site_case - manufacturer_site_case)
    (distributor_site_case_quarantined ?distributor_site_case - distributor_site_case)
    (final_disposition_recorded ?coordinated_response_case - coordinated_response_case)
  )
  (:action record_suspect_product_case
    :parameters (?suspect_product_case - suspect_product_case)
    :precondition
      (and
        (not
          (suspect_case_reported ?suspect_product_case)
        )
        (not
          (containment_executed ?suspect_product_case)
        )
      )
    :effect (suspect_case_reported ?suspect_product_case)
  )
  (:action assign_investigator_to_case
    :parameters (?suspect_product_case - suspect_product_case ?investigator_resource - investigator_resource)
    :precondition
      (and
        (suspect_case_reported ?suspect_product_case)
        (not
          (investigator_assigned ?suspect_product_case)
        )
        (investigator_available ?investigator_resource)
      )
    :effect
      (and
        (investigator_assigned ?suspect_product_case)
        (investigator_assigned_to_case ?suspect_product_case ?investigator_resource)
        (not
          (investigator_available ?investigator_resource)
        )
      )
  )
  (:action collect_lab_sample
    :parameters (?suspect_product_case - suspect_product_case ?lab_sample - lab_sample)
    :precondition
      (and
        (suspect_case_reported ?suspect_product_case)
        (investigator_assigned ?suspect_product_case)
        (lab_sample_available ?lab_sample)
      )
    :effect
      (and
        (sample_collected_for_case ?suspect_product_case ?lab_sample)
        (not
          (lab_sample_available ?lab_sample)
        )
      )
  )
  (:action escalate_case_after_lab_result
    :parameters (?suspect_product_case - suspect_product_case ?lab_sample - lab_sample)
    :precondition
      (and
        (suspect_case_reported ?suspect_product_case)
        (investigator_assigned ?suspect_product_case)
        (sample_collected_for_case ?suspect_product_case ?lab_sample)
        (not
          (case_escalated ?suspect_product_case)
        )
      )
    :effect (case_escalated ?suspect_product_case)
  )
  (:action return_lab_sample_to_pool
    :parameters (?suspect_product_case - suspect_product_case ?lab_sample - lab_sample)
    :precondition
      (and
        (sample_collected_for_case ?suspect_product_case ?lab_sample)
      )
    :effect
      (and
        (lab_sample_available ?lab_sample)
        (not
          (sample_collected_for_case ?suspect_product_case ?lab_sample)
        )
      )
  )
  (:action assign_response_team_to_case
    :parameters (?suspect_product_case - suspect_product_case ?response_team_resource - response_team_resource)
    :precondition
      (and
        (case_escalated ?suspect_product_case)
        (response_team_available ?response_team_resource)
      )
    :effect
      (and
        (response_team_assigned_to_case ?suspect_product_case ?response_team_resource)
        (not
          (response_team_available ?response_team_resource)
        )
      )
  )
  (:action release_investigator_from_case
    :parameters (?suspect_product_case - suspect_product_case ?response_team_resource - response_team_resource)
    :precondition
      (and
        (response_team_assigned_to_case ?suspect_product_case ?response_team_resource)
      )
    :effect
      (and
        (response_team_available ?response_team_resource)
        (not
          (response_team_assigned_to_case ?suspect_product_case ?response_team_resource)
        )
      )
  )
  (:action assign_disposal_facility_to_case
    :parameters (?coordinated_response_case - coordinated_response_case ?disposal_facility - disposal_facility)
    :precondition
      (and
        (case_escalated ?coordinated_response_case)
        (disposal_facility_available ?disposal_facility)
      )
    :effect
      (and
        (coordinated_case_has_disposal_facility ?coordinated_response_case ?disposal_facility)
        (not
          (disposal_facility_available ?disposal_facility)
        )
      )
  )
  (:action release_disposal_facility_from_case
    :parameters (?coordinated_response_case - coordinated_response_case ?disposal_facility - disposal_facility)
    :precondition
      (and
        (coordinated_case_has_disposal_facility ?coordinated_response_case ?disposal_facility)
      )
    :effect
      (and
        (disposal_facility_available ?disposal_facility)
        (not
          (coordinated_case_has_disposal_facility ?coordinated_response_case ?disposal_facility)
        )
      )
  )
  (:action assign_quality_expert_to_case
    :parameters (?coordinated_response_case - coordinated_response_case ?quality_expert - quality_expert)
    :precondition
      (and
        (case_escalated ?coordinated_response_case)
        (quality_expert_available ?quality_expert)
      )
    :effect
      (and
        (quality_expert_assigned_to_coordinated_case ?coordinated_response_case ?quality_expert)
        (not
          (quality_expert_available ?quality_expert)
        )
      )
  )
  (:action release_quality_expert_from_case
    :parameters (?coordinated_response_case - coordinated_response_case ?quality_expert - quality_expert)
    :precondition
      (and
        (quality_expert_assigned_to_coordinated_case ?coordinated_response_case ?quality_expert)
      )
    :effect
      (and
        (quality_expert_available ?quality_expert)
        (not
          (quality_expert_assigned_to_coordinated_case ?coordinated_response_case ?quality_expert)
        )
      )
  )
  (:action flag_manufacturer_node_quarantine
    :parameters (?manufacturer_site_case - manufacturer_site_case ?manufacturer_site_node - manufacturer_site_node ?lab_sample - lab_sample)
    :precondition
      (and
        (case_escalated ?manufacturer_site_case)
        (sample_collected_for_case ?manufacturer_site_case ?lab_sample)
        (manufacturer_site_case_affects_site_node ?manufacturer_site_case ?manufacturer_site_node)
        (not
          (manufacturer_node_quarantined ?manufacturer_site_node)
        )
        (not
          (manufacturer_node_mitigation_applied ?manufacturer_site_node)
        )
      )
    :effect (manufacturer_node_quarantined ?manufacturer_site_node)
  )
  (:action confirm_manufacturer_case_quarantine_with_response_team
    :parameters (?manufacturer_site_case - manufacturer_site_case ?manufacturer_site_node - manufacturer_site_node ?response_team_resource - response_team_resource)
    :precondition
      (and
        (case_escalated ?manufacturer_site_case)
        (response_team_assigned_to_case ?manufacturer_site_case ?response_team_resource)
        (manufacturer_site_case_affects_site_node ?manufacturer_site_case ?manufacturer_site_node)
        (manufacturer_node_quarantined ?manufacturer_site_node)
        (not
          (manufacturer_site_case_quarantined ?manufacturer_site_case)
        )
      )
    :effect
      (and
        (manufacturer_site_case_quarantined ?manufacturer_site_case)
        (manufacturer_site_case_ready_for_recall ?manufacturer_site_case)
      )
  )
  (:action apply_mitigation_resource_to_manufacturer_node
    :parameters (?manufacturer_site_case - manufacturer_site_case ?manufacturer_site_node - manufacturer_site_node ?mitigation_resource - mitigation_resource)
    :precondition
      (and
        (case_escalated ?manufacturer_site_case)
        (manufacturer_site_case_affects_site_node ?manufacturer_site_case ?manufacturer_site_node)
        (mitigation_resource_available ?mitigation_resource)
        (not
          (manufacturer_site_case_quarantined ?manufacturer_site_case)
        )
      )
    :effect
      (and
        (manufacturer_node_mitigation_applied ?manufacturer_site_node)
        (manufacturer_site_case_quarantined ?manufacturer_site_case)
        (mitigation_allocated_to_manufacturer_site_case ?manufacturer_site_case ?mitigation_resource)
        (not
          (mitigation_resource_available ?mitigation_resource)
        )
      )
  )
  (:action finalize_manufacturer_node_mitigation_and_mark_case_ready
    :parameters (?manufacturer_site_case - manufacturer_site_case ?manufacturer_site_node - manufacturer_site_node ?lab_sample - lab_sample ?mitigation_resource - mitigation_resource)

        (case_escalated ?manufacturer_site_case)
        (sample_collected_for_case ?manufacturer_site_case ?lab_sample)
        (manufacturer_site_case_affects_site_node ?manufacturer_site_case ?manufacturer_site_node)
        (manufacturer_node_mitigation_applied ?manufacturer_site_node)
        (mitigation_allocated_to_manufacturer_site_case ?manufacturer_site_case ?mitigation_resource)
        (not
          (manufacturer_site_case_ready_for_recall ?manufacturer_site_case)
        )
      )
    :effect
      (and
        (manufacturer_node_quarantined ?manufacturer_site_node)
        (manufacturer_site_case_ready_for_recall ?manufacturer_site_case)
        (mitigation_resource_available ?mitigation_resource)
        (not
          (mitigation_allocated_to_manufacturer_site_case ?manufacturer_site_case ?mitigation_resource)
        )
      )
  )
  (:action flag_distributor_node_quarantine
    :parameters (?distributor_site_case - distributor_site_case ?distributor_site_node - distributor_site_node ?lab_sample - lab_sample)

      (and
        (case_escalated ?distributor_site_case)

          (distributor_node_quarantined ?distributor_site_node)

  (:action confirm_distributor_case_quarantine_with_response_team
    :parameters (?distributor_site_case - distributor_site_case ?distributor_site_node - distributor_site_node ?response_team_resource - response_team_resource)
    :precondition
      (and
        (case_escalated ?distributor_site_case)

        (distributor_site_case_affects_site_node ?distributor_site_case ?distributor_site_node)
        (distributor_node_quarantined ?distributor_site_node)
        (not
          (distributor_site_case_quarantined ?distributor_site_case)
        )

    :effect
      (and
        (distributor_site_case_quarantined ?distributor_site_case)
        (distributor_site_case_ready_for_recall ?distributor_site_case)

  (:action apply_mitigation_resource_to_distributor_node
    :parameters (?distributor_site_case - distributor_site_case ?distributor_site_node - distributor_site_node ?mitigation_resource - mitigation_resource)

      (and

        (mitigation_resource_available ?mitigation_resource)

          (distributor_site_case_quarantined ?distributor_site_case)

    :effect
      (and
        (distributor_node_mitigation_applied ?distributor_site_node)
        (distributor_site_case_quarantined ?distributor_site_case)
        (mitigation_allocated_to_distributor_site_case ?distributor_site_case ?mitigation_resource)
        (not
          (mitigation_resource_available ?mitigation_resource)
        )
      )
  )
  (:action finalize_distributor_node_mitigation_and_mark_case_ready
    :parameters (?distributor_site_case - distributor_site_case ?distributor_site_node - distributor_site_node ?lab_sample - lab_sample ?mitigation_resource - mitigation_resource)
    :precondition
      (and
        (case_escalated ?distributor_site_case)
        (sample_collected_for_case ?distributor_site_case ?lab_sample)
        (distributor_site_case_affects_site_node ?distributor_site_case ?distributor_site_node)
        (distributor_node_mitigation_applied ?distributor_site_node)
        (mitigation_allocated_to_distributor_site_case ?distributor_site_case ?mitigation_resource)

    :effect
      (and
        (distributor_node_quarantined ?distributor_site_node)

        (mitigation_resource_available ?mitigation_resource)
        (not
          (mitigation_allocated_to_distributor_site_case ?distributor_site_case ?mitigation_resource)
        )
      )
  )
  (:action activate_recall_event_with_target_nodes

    :precondition
      (and
        (manufacturer_site_case_quarantined ?manufacturer_site_case)
        (distributor_site_case_quarantined ?distributor_site_case)
        (manufacturer_site_case_affects_site_node ?manufacturer_site_case ?manufacturer_site_node)
        (distributor_site_case_affects_site_node ?distributor_site_case ?distributor_site_node)
        (manufacturer_node_quarantined ?manufacturer_site_node)
        (distributor_node_quarantined ?distributor_site_node)
        (manufacturer_site_case_ready_for_recall ?manufacturer_site_case)
        (distributor_site_case_ready_for_recall ?distributor_site_case)
        (recall_event_available ?recall_event)
      )
    :effect
      (and
        (recall_event_active ?recall_event)
        (recall_event_targets_manufacturer_node ?recall_event ?manufacturer_site_node)
        (recall_event_targets_distributor_node ?recall_event ?distributor_site_node)
        (not
          (recall_event_available ?recall_event)
        )
      )

  (:action activate_recall_event_include_manufacturer_nodes

    :precondition
      (and
        (manufacturer_site_case_quarantined ?manufacturer_site_case)
        (distributor_site_case_quarantined ?distributor_site_case)

        (distributor_site_case_affects_site_node ?distributor_site_case ?distributor_site_node)

        (distributor_node_quarantined ?distributor_site_node)

          (manufacturer_site_case_ready_for_recall ?manufacturer_site_case)

        (distributor_site_case_ready_for_recall ?distributor_site_case)
        (recall_event_available ?recall_event)
      )
    :effect
      (and
        (recall_event_active ?recall_event)
        (recall_event_targets_manufacturer_node ?recall_event ?manufacturer_site_node)
        (recall_event_targets_distributor_node ?recall_event ?distributor_site_node)
        (recall_event_includes_manufacturer_nodes ?recall_event)
        (not
          (recall_event_available ?recall_event)

  (:action activate_recall_event_include_distributor_nodes

    :precondition
      (and
        (manufacturer_site_case_quarantined ?manufacturer_site_case)

          (distributor_site_case_ready_for_recall ?distributor_site_case)

        (recall_event_includes_distributor_nodes ?recall_event)
        (not
          (recall_event_available ?recall_event)
        )
      )

          (manufacturer_site_case_ready_for_recall ?manufacturer_site_case)

        (recall_event_includes_distributor_nodes ?recall_event)

          (recall_event_available ?recall_event)
        )
      )
  )
  (:action finalize_recall_event_execution
    :parameters (?recall_event - recall_event ?manufacturer_site_case - manufacturer_site_case ?lab_sample - lab_sample)
    :precondition
      (and
        (recall_event_active ?recall_event)
