--- domain.pddl (as passed to VAL) ---
(define (domain pharmaceutics_controlled_document_revision_rollout)
  (:requirements :strips :typing :negative-preconditions)
  (:types entity - object resource_category - entity org_asset_category - entity artifact_category - entity document_type - entity document_revision - document_type reviewer - resource_category validator - resource_category approver - resource_category regulatory_attachment - resource_category implementation_plan - resource_category change_control_record - resource_category qualification_task - resource_category validation_task - resource_category validation_protocol - org_asset_category qa_checklist - org_asset_category regulatory_submission_item - org_asset_category validation_activity - artifact_category training_activity - artifact_category release_package - artifact_category deployment_instance - document_revision document_family - document_revision site_revision_instance - deployment_instance department_revision_instance - deployment_instance controlled_document - document_family)

  (:predicates
    (entity_initiated ?document_revision - document_revision)
    (entity_validated ?document_revision - document_revision)
    (review_recorded ?document_revision - document_revision)
    (entity_deployed ?document_revision - document_revision)
    (entity_release_applied ?document_revision - document_revision)
    (change_control_linked_flag ?document_revision - document_revision)
    (reviewer_available ?reviewer - reviewer)
    (assigned_reviewer ?document_revision - document_revision ?reviewer - reviewer)
    (validator_available ?validator - validator)
    (assigned_validator ?document_revision - document_revision ?validator - validator)
    (approver_available ?approver - approver)
    (assigned_approver ?document_revision - document_revision ?approver - approver)
    (validation_protocol_available ?validation_protocol - validation_protocol)
    (site_assigned_validation_protocol ?site_revision_instance - site_revision_instance ?validation_protocol - validation_protocol)
    (dept_assigned_validation_protocol ?department_revision_instance - department_revision_instance ?validation_protocol - validation_protocol)
    (site_has_validation_activity ?site_revision_instance - site_revision_instance ?validation_activity - validation_activity)
    (validation_activity_ready ?validation_activity - validation_activity)
    (validation_activity_configured ?validation_activity - validation_activity)
    (site_validation_activity_completed ?site_revision_instance - site_revision_instance)
    (department_has_training_activity ?department_revision_instance - department_revision_instance ?training_activity - training_activity)
    (training_activity_ready ?training_activity - training_activity)
    (training_activity_materials_assigned ?training_activity - training_activity)
    (department_training_completed ?department_revision_instance - department_revision_instance)
    (release_package_available ?release_package - release_package)
    (release_package_assembled ?release_package - release_package)
    (package_includes_validation_activity ?release_package - release_package ?validation_activity - validation_activity)
    (package_includes_training_activity ?release_package - release_package ?training_activity - training_activity)
    (package_includes_protocol_evidence ?release_package - release_package)
    (package_includes_training_evidence ?release_package - release_package)
    (release_package_finalized ?release_package - release_package)
    (document_registered_at_site ?controlled_document - controlled_document ?site_revision_instance - site_revision_instance)
    (document_registered_in_department ?controlled_document - controlled_document ?department_revision_instance - department_revision_instance)
    (document_linked_to_release_package ?controlled_document - controlled_document ?release_package - release_package)
    (qa_checklist_available ?qa_checklist - qa_checklist)
    (document_has_qa_checklist ?controlled_document - controlled_document ?qa_checklist - qa_checklist)
    (qa_checklist_integrated ?qa_checklist - qa_checklist)
    (qa_checklist_added_to_package ?qa_checklist - qa_checklist ?release_package - release_package)
    (document_ready_for_qa_review ?controlled_document - controlled_document)
    (document_qa_checks_passed ?controlled_document - controlled_document)
    (document_final_review_ready ?controlled_document - controlled_document)
    (document_regulatory_attachment_assigned ?controlled_document - controlled_document)
    (document_regulatory_attachment_integrated ?controlled_document - controlled_document)
    (implementation_plan_required ?controlled_document - controlled_document)
    (final_crossfunctional_review_completed ?controlled_document - controlled_document)
    (regulatory_submission_item_available ?regulatory_submission_item - regulatory_submission_item)
    (document_linked_regulatory_submission_item ?controlled_document - controlled_document ?regulatory_submission_item - regulatory_submission_item)
    (document_regulatory_submission_initiated ?controlled_document - controlled_document)
    (document_regulatory_submission_in_review ?controlled_document - controlled_document)
    (document_regulatory_submission_approved ?controlled_document - controlled_document)
    (regulatory_attachment_available ?regulatory_attachment - regulatory_attachment)
    (document_has_regulatory_attachment ?controlled_document - controlled_document ?regulatory_attachment - regulatory_attachment)
    (implementation_plan_available ?implementation_plan - implementation_plan)
    (document_has_implementation_plan ?controlled_document - controlled_document ?implementation_plan - implementation_plan)
    (qualification_task_available ?qualification_task - qualification_task)
    (document_has_qualification_task ?controlled_document - controlled_document ?qualification_task - qualification_task)
    (validation_task_available ?validation_task - validation_task)
    (document_has_validation_task ?controlled_document - controlled_document ?validation_task - validation_task)
    (change_control_record_available ?change_control_record - change_control_record)
    (entity_linked_to_change_control ?document_revision - document_revision ?change_control_record - change_control_record)
    (site_ready ?site_revision_instance - site_revision_instance)
    (department_ready ?department_revision_instance - department_revision_instance)
    (controlled_document_release_flag ?controlled_document - controlled_document)
  )
  (:action initiate_document_revision
    :parameters (?document_revision - document_revision)
    :precondition
      (and
        (not
          (entity_initiated ?document_revision)
        )
        (not
          (entity_deployed ?document_revision)
        )
      )
    :effect (entity_initiated ?document_revision)
  )
  (:action assign_and_record_reviewer
    :parameters (?document_revision - document_revision ?reviewer - reviewer)
    :precondition
      (and
        (entity_initiated ?document_revision)
        (not
          (review_recorded ?document_revision)
        )
        (reviewer_available ?reviewer)
      )
    :effect
      (and
        (review_recorded ?document_revision)
        (assigned_reviewer ?document_revision ?reviewer)
        (not
          (reviewer_available ?reviewer)
        )
      )
  )
  (:action assign_validator_and_record
    :parameters (?document_revision - document_revision ?validator - validator)
    :precondition
      (and
        (entity_initiated ?document_revision)
        (review_recorded ?document_revision)
        (validator_available ?validator)
      )
    :effect
      (and
        (assigned_validator ?document_revision ?validator)
        (not
          (validator_available ?validator)
        )
      )
  )
  (:action complete_validation
    :parameters (?document_revision - document_revision ?validator - validator)
    :precondition
      (and
        (entity_initiated ?document_revision)
        (review_recorded ?document_revision)
        (assigned_validator ?document_revision ?validator)
        (not
          (entity_validated ?document_revision)
        )
      )
    :effect (entity_validated ?document_revision)
  )
  (:action release_validator_to_pool
    :parameters (?document_revision - document_revision ?validator - validator)
    :precondition
      (and
        (assigned_validator ?document_revision ?validator)
      )
    :effect
      (and
        (validator_available ?validator)
        (not
          (assigned_validator ?document_revision ?validator)
        )
      )
  )
  (:action assign_approver
    :parameters (?document_revision - document_revision ?approver - approver)
    :precondition
      (and
        (entity_validated ?document_revision)
        (approver_available ?approver)
      )
    :effect
      (and
        (assigned_approver ?document_revision ?approver)
        (not
          (approver_available ?approver)
        )
      )
  )
  (:action release_approver_to_pool
    :parameters (?document_revision - document_revision ?approver - approver)
    :precondition
      (and
        (assigned_approver ?document_revision ?approver)
      )
    :effect
      (and
        (approver_available ?approver)
        (not
          (assigned_approver ?document_revision ?approver)
        )
      )
  )
  (:action assign_qualification_task
    :parameters (?controlled_document - controlled_document ?qualification_task - qualification_task)
    :precondition
      (and
        (entity_validated ?controlled_document)
        (qualification_task_available ?qualification_task)
      )
    :effect
      (and
        (document_has_qualification_task ?controlled_document ?qualification_task)
        (not
          (qualification_task_available ?qualification_task)
        )
      )
  )
  (:action release_qualification_task_to_pool
    :parameters (?controlled_document - controlled_document ?qualification_task - qualification_task)
    :precondition
      (and
        (document_has_qualification_task ?controlled_document ?qualification_task)
      )
    :effect
      (and
        (qualification_task_available ?qualification_task)
        (not
          (document_has_qualification_task ?controlled_document ?qualification_task)
        )
      )
  )
  (:action assign_validation_task
    :parameters (?controlled_document - controlled_document ?validation_task - validation_task)
    :precondition
      (and
        (entity_validated ?controlled_document)
        (validation_task_available ?validation_task)
      )
    :effect
      (and
        (document_has_validation_task ?controlled_document ?validation_task)
        (not
          (validation_task_available ?validation_task)
        )
      )
  )
  (:action release_validation_task_to_pool
    :parameters (?controlled_document - controlled_document ?validation_task - validation_task)
    :precondition
      (and
        (document_has_validation_task ?controlled_document ?validation_task)
      )
    :effect
      (and
        (validation_task_available ?validation_task)
        (not
          (document_has_validation_task ?controlled_document ?validation_task)
        )
      )
  )
  (:action mark_validation_activity_ready
    :parameters (?site_revision_instance - site_revision_instance ?validation_activity - validation_activity ?validator - validator)
    :precondition
      (and
        (entity_validated ?site_revision_instance)
        (assigned_validator ?site_revision_instance ?validator)
        (site_has_validation_activity ?site_revision_instance ?validation_activity)
        (not
          (validation_activity_ready ?validation_activity)
        )
        (not
          (validation_activity_configured ?validation_activity)
        )
      )
    :effect (validation_activity_ready ?validation_activity)
  )
  (:action execute_validation_activity
    :parameters (?site_revision_instance - site_revision_instance ?validation_activity - validation_activity ?approver - approver)
    :precondition
      (and
        (entity_validated ?site_revision_instance)
        (assigned_approver ?site_revision_instance ?approver)
        (site_has_validation_activity ?site_revision_instance ?validation_activity)
        (validation_activity_ready ?validation_activity)
        (not
          (site_ready ?site_revision_instance)
        )
      )
    :effect
      (and
        (site_ready ?site_revision_instance)
        (site_validation_activity_completed ?site_revision_instance)
      )
  )
  (:action assign_validation_protocol_to_site
    :parameters (?site_revision_instance - site_revision_instance ?validation_activity - validation_activity ?validation_protocol - validation_protocol)
    :precondition
      (and
        (entity_validated ?site_revision_instance)
        (site_has_validation_activity ?site_revision_instance ?validation_activity)
        (validation_protocol_available ?validation_protocol)
        (not
          (site_ready ?site_revision_instance)
        )
      )
    :effect
      (and
        (validation_activity_configured ?validation_activity)
        (site_ready ?site_revision_instance)
        (site_assigned_validation_protocol ?site_revision_instance ?validation_protocol)
        (not
          (validation_protocol_available ?validation_protocol)
        )
      )
  )
  (:action finalize_validation_activity
    :parameters (?site_revision_instance - site_revision_instance ?validation_activity - validation_activity ?validator - validator ?validation_protocol - validation_protocol)
    :precondition
      (and
        (entity_validated ?site_revision_instance)
        (assigned_validator ?site_revision_instance ?validator)
        (site_has_validation_activity ?site_revision_instance ?validation_activity)
        (validation_activity_configured ?validation_activity)
        (site_assigned_validation_protocol ?site_revision_instance ?validation_protocol)
        (not
          (site_validation_activity_completed ?site_revision_instance)
        )
      )
    :effect
      (and
        (validation_activity_ready ?validation_activity)
        (site_validation_activity_completed ?site_revision_instance)
        (validation_protocol_available ?validation_protocol)
        (not
          (site_assigned_validation_protocol ?site_revision_instance ?validation_protocol)
        )
      )
  )
  (:action mark_training_activity_ready
    :parameters (?department_revision_instance - department_revision_instance ?training_activity - training_activity ?validator - validator)
    :precondition
      (and
        (entity_validated ?department_revision_instance)
        (assigned_validator ?department_revision_instance ?validator)
        (department_has_training_activity ?department_revision_instance ?training_activity)
        (not
          (training_activity_ready ?training_activity)
        )
        (not
          (training_activity_materials_assigned ?training_activity)
        )
      )
    :effect (training_activity_ready ?training_activity)
  )
  (:action execute_training_activity_with_approver
    :parameters (?department_revision_instance - department_revision_instance ?training_activity - training_activity ?approver - approver)
    :precondition
      (and
        (entity_validated ?department_revision_instance)
        (assigned_approver ?department_revision_instance ?approver)
        (department_has_training_activity ?department_revision_instance ?training_activity)
        (training_activity_ready ?training_activity)
        (not
          (department_ready ?department_revision_instance)
        )
      )
    :effect
      (and
        (department_ready ?department_revision_instance)
        (department_training_completed ?department_revision_instance)
      )
  )
  (:action assign_validation_protocol_to_department
    :parameters (?department_revision_instance - department_revision_instance ?training_activity - training_activity ?validation_protocol - validation_protocol)
    :precondition
      (and
        (entity_validated ?department_revision_instance)
        (department_has_training_activity ?department_revision_instance ?training_activity)
        (validation_protocol_available ?validation_protocol)
        (not
          (department_ready ?department_revision_instance)
        )
      )
    :effect
      (and
        (training_activity_materials_assigned ?training_activity)
        (department_ready ?department_revision_instance)
        (dept_assigned_validation_protocol ?department_revision_instance ?validation_protocol)
        (not
          (validation_protocol_available ?validation_protocol)
        )
      )
  )
  (:action finalize_training_activity
    :parameters (?department_revision_instance - department_revision_instance ?training_activity - training_activity ?validator - validator ?validation_protocol - validation_protocol)
    :precondition
      (and
        (entity_validated ?department_revision_instance)
        (assigned_validator ?department_revision_instance ?validator)
        (department_has_training_activity ?department_revision_instance ?training_activity)
        (training_activity_materials_assigned ?training_activity)
        (dept_assigned_validation_protocol ?department_revision_instance ?validation_protocol)
        (not
          (department_training_completed ?department_revision_instance)
        )
      )
    :effect
      (and
        (training_activity_ready ?training_activity)
        (department_training_completed ?department_revision_instance)
        (validation_protocol_available ?validation_protocol)
        (not
          (dept_assigned_validation_protocol ?department_revision_instance ?validation_protocol)
        )
      )
  )
  (:action assemble_release_package_standard
    :parameters (?site_revision_instance - site_revision_instance ?department_revision_instance - department_revision_instance ?validation_activity - validation_activity ?training_activity - training_activity ?release_package - release_package)
    :precondition
      (and
        (site_ready ?site_revision_instance)
        (department_ready ?department_revision_instance)
        (site_has_validation_activity ?site_revision_instance ?validation_activity)
        (department_has_training_activity ?department_revision_instance ?training_activity)
        (validation_activity_ready ?validation_activity)
        (training_activity_ready ?training_activity)
        (site_validation_activity_completed ?site_revision_instance)
        (department_training_completed ?department_revision_instance)
        (release_package_available ?release_package)
      )
    :effect
      (and
        (release_package_assembled ?release_package)
        (package_includes_validation_activity ?release_package ?validation_activity)
        (package_includes_training_activity ?release_package ?training_activity)
        (not
          (release_package_available ?release_package)
        )
      )
  )
  (:action assemble_release_package_with_protocol_evidence
    :parameters (?site_revision_instance - site_revision_instance ?department_revision_instance - department_revision_instance ?validation_activity - validation_activity ?training_activity - training_activity ?release_package - release_package)
    :precondition
      (and
        (site_ready ?site_revision_instance)
        (department_ready ?department_revision_instance)
        (site_has_validation_activity ?site_revision_instance ?validation_activity)
        (department_has_training_activity ?department_revision_instance ?training_activity)
        (validation_activity_configured ?validation_activity)
        (training_activity_ready ?training_activity)
        (not
          (site_validation_activity_completed ?site_revision_instance)
        )
        (department_training_completed ?department_revision_instance)
        (release_package_available ?release_package)
      )
    :effect
      (and
        (release_package_assembled ?release_package)
        (package_includes_validation_activity ?release_package ?validation_activity)
        (package_includes_training_activity ?release_package ?training_activity)
        (package_includes_protocol_evidence ?release_package)
        (not
          (release_package_available ?release_package)
        )
      )
  )
  (:action assemble_release_package_with_training_evidence
    :parameters (?site_revision_instance - site_revision_instance ?department_revision_instance - department_revision_instance ?validation_activity - validation_activity ?training_activity - training_activity ?release_package - release_package)
    :precondition
      (and
        (site_ready ?site_revision_instance)
        (department_ready ?department_revision_instance)
        (site_has_validation_activity ?site_revision_instance ?validation_activity)
        (department_has_training_activity ?department_revision_instance ?training_activity)
        (validation_activity_ready ?validation_activity)
        (training_activity_materials_assigned ?training_activity)
        (site_validation_activity_completed ?site_revision_instance)
        (not
          (department_training_completed ?department_revision_instance)
        )
        (release_package_available ?release_package)
      )
    :effect
      (and
        (release_package_assembled ?release_package)
        (package_includes_validation_activity ?release_package ?validation_activity)
        (package_includes_training_activity ?release_package ?training_activity)
        (package_includes_training_evidence ?release_package)
        (not
          (release_package_available ?release_package)
        )
      )
  )
  (:action assemble_release_package_comprehensive
    :parameters (?site_revision_instance - site_revision_instance ?department_revision_instance - department_revision_instance ?validation_activity - validation_activity ?training_activity - training_activity ?release_package - release_package)
    :precondition
      (and
        (site_ready ?site_revision_instance)
        (department_ready ?department_revision_instance)
        (site_has_validation_activity ?site_revision_instance ?validation_activity)
        (department_has_training_activity ?department_revision_instance ?training_activity)
        (validation_activity_configured ?validation_activity)
        (training_activity_materials_assigned ?training_activity)
        (not
          (site_validation_activity_completed ?site_revision_instance)
        )
        (not
          (department_training_completed ?department_revision_instance)
        )
        (release_package_available ?release_package)
      )
    :effect
      (and
        (release_package_assembled ?release_package)
        (package_includes_validation_activity ?release_package ?validation_activity)
        (package_includes_training_activity ?release_package ?training_activity)
        (package_includes_protocol_evidence ?release_package)
        (package_includes_training_evidence ?release_package)
        (not
          (release_package_available ?release_package)
        )
      )
  )
  (:action finalize_release_package
    :parameters (?release_package - release_package ?site_revision_instance - site_revision_instance ?validator - validator)
    :precondition
      (and
        (release_package_assembled ?release_package)
        (site_ready ?site_revision_instance)
        (assigned_validator ?site_revision_instance ?validator)
        (not
          (release_package_finalized ?release_package)
        )
      )
    :effect (release_package_finalized ?release_package)
  )
  (:action integrate_qa_checklist_into_package
    :parameters (?controlled_document - controlled_document ?qa_checklist - qa_checklist ?release_package - release_package)
    :precondition
      (and
        (entity_validated ?controlled_document)
        (document_linked_to_release_package ?controlled_document ?release_package)
        (document_has_qa_checklist ?controlled_document ?qa_checklist)
        (qa_checklist_available ?qa_checklist)
        (release_package_assembled ?release_package)
        (release_package_finalized ?release_package)
        (not
          (qa_checklist_integrated ?qa_checklist)
        )
      )
    :effect
      (and
        (qa_checklist_integrated ?qa_checklist)
        (qa_checklist_added_to_package ?qa_checklist ?release_package)
        (not
          (qa_checklist_available ?qa_checklist)
        )
      )
  )
  (:action prepare_document_for_qa_review
    :parameters (?controlled_document - controlled_document ?qa_checklist - qa_checklist ?release_package - release_package ?validator - validator)
    :precondition
      (and
        (entity_validated ?controlled_document)
        (document_has_qa_checklist ?controlled_document ?qa_checklist)
        (qa_checklist_integrated ?qa_checklist)
        (qa_checklist_added_to_package ?qa_checklist ?release_package)
        (assigned_validator ?controlled_document ?validator)
        (not
          (package_includes_protocol_evidence ?release_package)
        )
        (not
          (document_ready_for_qa_review ?controlled_document)
        )
      )
    :effect (document_ready_for_qa_review ?controlled_document)
  )
  (:action assign_regulatory_attachment_to_document
    :parameters (?controlled_document - controlled_document ?regulatory_attachment - regulatory_attachment)
    :precondition
      (and
        (entity_validated ?controlled_document)
        (regulatory_attachment_available ?regulatory_attachment)
        (not
          (document_regulatory_attachment_assigned ?controlled_document)
        )
      )
    :effect
      (and
        (document_regulatory_attachment_assigned ?controlled_document)
        (document_has_regulatory_attachment ?controlled_document ?regulatory_attachment)
        (not
          (regulatory_attachment_available ?regulatory_attachment)
        )
      )
  )
  (:action attach_regulatory_attachment_to_document
    :parameters (?controlled_document - controlled_document ?qa_checklist - qa_checklist ?release_package - release_package ?validator - validator ?regulatory_attachment - regulatory_attachment)
    :precondition
      (and
        (entity_validated ?controlled_document)
        (document_has_qa_checklist ?controlled_document ?qa_checklist)
        (qa_checklist_integrated ?qa_checklist)
        (qa_checklist_added_to_package ?qa_checklist ?release_package)
        (assigned_validator ?controlled_document ?validator)
        (package_includes_protocol_evidence ?release_package)
        (document_regulatory_attachment_assigned ?controlled_document)
        (document_has_regulatory_attachment ?controlled_document ?regulatory_attachment)
        (not
          (document_ready_for_qa_review ?controlled_document)
        )
      )
    :effect
      (and
        (document_ready_for_qa_review ?controlled_document)
        (document_regulatory_attachment_integrated ?controlled_document)
      )
  )
  (:action execute_qa_acceptance_without_training
    :parameters (?controlled_document - controlled_document ?qualification_task - qualification_task ?approver - approver ?qa_checklist - qa_checklist ?release_package - release_package)
    :precondition
      (and
        (document_ready_for_qa_review ?controlled_document)
        (document_has_qualification_task ?controlled_document ?qualification_task)
        (assigned_approver ?controlled_document ?approver)
        (document_has_qa_checklist ?controlled_document ?qa_checklist)
        (qa_checklist_added_to_package ?qa_checklist ?release_package)
        (not
          (package_includes_training_evidence ?release_package)
        )
        (not
          (document_qa_checks_passed ?controlled_document)
        )
      )
    :effect (document_qa_checks_passed ?controlled_document)
  )
  (:action execute_qa_acceptance_with_training
    :parameters (?controlled_document - controlled_document ?qualification_task - qualification_task ?approver - approver ?qa_checklist - qa_checklist ?release_package - release_package)
    :precondition
      (and
        (document_ready_for_qa_review ?controlled_document)
