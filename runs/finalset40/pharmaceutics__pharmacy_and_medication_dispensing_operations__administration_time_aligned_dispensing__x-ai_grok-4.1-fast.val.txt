    :precondition
      (and

    :parameters (?prescription_order - pharmacy_order ?formulary_option - formulary_product_option)
    :precondition
      (and
        (order_registered ?prescription_order)
        (not
          (product_choice_confirmed ?prescription_order)
        )
        (formulary_option_available ?formulary_option)
      )
    :effect
      (and
        (product_choice_confirmed ?prescription_order)
        (order_has_formulary_option ?prescription_order ?formulary_option)
        (not
          (formulary_option_available ?formulary_option)
        )
      )
  )

    :parameters (?prescription_order - pharmacy_order ?clinical_check - clinical_check_record)

      (and
        (order_registered ?prescription_order)

      (and
        (order_has_clinical_check ?prescription_order ?clinical_check)

          (clinical_check_available ?clinical_check)
        )
      )

        (clinical_check_available ?clinical_check)

          (order_has_clinical_check ?prescription_order ?clinical_check)
        )
      )
  )
