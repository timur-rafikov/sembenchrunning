--- domain.pddl (as passed to VAL) ---
(define (domain baggage_sensitive_transfer_planning_domain)
  (:requirements :strips :typing :negative-preconditions)
  (:types operational_asset - object location_entity - object auxiliary_attribute - object task_root - object baggage_transfer_task - task_root vehicle_asset - operational_asset transport_leg - operational_asset baggage_handler - operational_asset priority_designator - operational_asset special_handling_tag - operational_asset baggage_tag - operational_asset handling_equipment - operational_asset inspection_certificate - operational_asset baggage_unit - location_entity baggage_container - location_entity airline_operator - location_entity arrival_zone - auxiliary_attribute departure_zone - auxiliary_attribute transfer_vehicle_unit - auxiliary_attribute segment_group - baggage_transfer_task plan_group - baggage_transfer_task arrival_segment - segment_group departure_segment - segment_group transfer_plan - plan_group)
  (:predicates
    (transfer_entity_registered ?baggage_transfer_task - baggage_transfer_task)
    (transfer_entity_activated ?baggage_transfer_task - baggage_transfer_task)
    (transfer_entity_vehicle_bound ?baggage_transfer_task - baggage_transfer_task)
    (transfer_entity_allocation_confirmed ?baggage_transfer_task - baggage_transfer_task)
    (transfer_entity_execution_commenced ?baggage_transfer_task - baggage_transfer_task)
    (transfer_entity_baggage_tag_attached ?baggage_transfer_task - baggage_transfer_task)
    (vehicle_available ?vehicle_asset - vehicle_asset)
    (transfer_entity_assigned_vehicle ?baggage_transfer_task - baggage_transfer_task ?vehicle_asset - vehicle_asset)
    (transport_leg_available ?transport_leg - transport_leg)
    (transfer_entity_assigned_transport_leg ?baggage_transfer_task - baggage_transfer_task ?transport_leg - transport_leg)
    (handler_available ?baggage_handler - baggage_handler)
    (transfer_entity_assigned_handler ?baggage_transfer_task - baggage_transfer_task ?baggage_handler - baggage_handler)
    (baggage_unit_available ?baggage_unit - baggage_unit)
    (arrival_segment_offered_baggage_unit ?arrival_segment - arrival_segment ?baggage_unit - baggage_unit)
    (departure_segment_offered_baggage_unit ?departure_segment - departure_segment ?baggage_unit - baggage_unit)
    (segment_assigned_arrival_zone ?arrival_segment - arrival_segment ?arrival_zone - arrival_zone)
    (arrival_zone_ready ?arrival_zone - arrival_zone)
    (arrival_zone_has_container ?arrival_zone - arrival_zone)
    (arrival_segment_ready ?arrival_segment - arrival_segment)
    (segment_assigned_departure_zone ?departure_segment - departure_segment ?departure_zone - departure_zone)
    (departure_zone_ready ?departure_zone - departure_zone)
    (departure_zone_has_container ?departure_zone - departure_zone)
    (departure_segment_ready ?departure_segment - departure_segment)
    (transfer_unit_available ?transfer_vehicle_unit - transfer_vehicle_unit)
    (transfer_unit_allocated ?transfer_vehicle_unit - transfer_vehicle_unit)
    (transfer_unit_assigned_arrival_zone ?transfer_vehicle_unit - transfer_vehicle_unit ?arrival_zone - arrival_zone)
    (transfer_unit_assigned_departure_zone ?transfer_vehicle_unit - transfer_vehicle_unit ?departure_zone - departure_zone)
    (transfer_unit_arrival_prepared ?transfer_vehicle_unit - transfer_vehicle_unit)
    (transfer_unit_departure_prepared ?transfer_vehicle_unit - transfer_vehicle_unit)
    (transfer_unit_loading_commenced ?transfer_vehicle_unit - transfer_vehicle_unit)
    (plan_includes_arrival_segment ?transfer_plan - transfer_plan ?arrival_segment - arrival_segment)
    (plan_includes_departure_segment ?transfer_plan - transfer_plan ?departure_segment - departure_segment)
    (plan_assigned_transfer_unit ?transfer_plan - transfer_plan ?transfer_vehicle_unit - transfer_vehicle_unit)
    (baggage_container_available ?baggage_container - baggage_container)
    (plan_assigned_container ?transfer_plan - transfer_plan ?baggage_container - baggage_container)
    (container_staged ?baggage_container - baggage_container)
    (container_assigned_to_unit ?baggage_container - baggage_container ?transfer_vehicle_unit - transfer_vehicle_unit)
    (plan_container_verified ?transfer_plan - transfer_plan)
    (plan_equipment_verified ?transfer_plan - transfer_plan)
    (plan_inspection_ready ?transfer_plan - transfer_plan)
    (plan_priority_attached ?transfer_plan - transfer_plan)
    (plan_priority_confirmed ?transfer_plan - transfer_plan)
    (plan_certificate_verified ?transfer_plan - transfer_plan)
    (plan_finalized ?transfer_plan - transfer_plan)
    (operator_available ?airline_operator - airline_operator)
    (plan_assigned_operator ?transfer_plan - transfer_plan ?airline_operator - airline_operator)
    (plan_priority_reserved ?transfer_plan - transfer_plan)
    (plan_operator_confirmed ?transfer_plan - transfer_plan)
    (plan_certified ?transfer_plan - transfer_plan)
    (priority_designator_available ?priority_designator - priority_designator)
    (plan_assigned_priority ?transfer_plan - transfer_plan ?priority_designator - priority_designator)
    (special_handling_tag_available ?special_handling_tag - special_handling_tag)
    (plan_assigned_special_handling_tag ?transfer_plan - transfer_plan ?special_handling_tag - special_handling_tag)
    (handling_equipment_available ?handling_equipment - handling_equipment)
    (plan_assigned_handling_equipment ?transfer_plan - transfer_plan ?handling_equipment - handling_equipment)
    (inspection_certificate_available ?inspection_certificate - inspection_certificate)
    (plan_assigned_inspection_certificate ?transfer_plan - transfer_plan ?inspection_certificate - inspection_certificate)
    (baggage_tag_available ?baggage_tag - baggage_tag)
    (transfer_entity_assigned_baggage_tag ?baggage_transfer_task - baggage_transfer_task ?baggage_tag - baggage_tag)
    (arrival_segment_prepared ?arrival_segment - arrival_segment)
    (departure_segment_prepared ?departure_segment - departure_segment)
    (plan_execution_ready ?transfer_plan - transfer_plan)
  )
  (:action create_transfer_task
    :parameters (?baggage_transfer_task - baggage_transfer_task)
    :precondition
      (and
        (not
          (transfer_entity_registered ?baggage_transfer_task)
        )
        (not
          (transfer_entity_allocation_confirmed ?baggage_transfer_task)
        )
      )
    :effect (transfer_entity_registered ?baggage_transfer_task)
  )
  (:action assign_vehicle_to_task
    :parameters (?baggage_transfer_task - baggage_transfer_task ?vehicle_asset - vehicle_asset)
    :precondition
      (and
        (transfer_entity_registered ?baggage_transfer_task)
        (not
          (transfer_entity_vehicle_bound ?baggage_transfer_task)
        )
        (vehicle_available ?vehicle_asset)
      )
    :effect
      (and
        (transfer_entity_vehicle_bound ?baggage_transfer_task)
        (transfer_entity_assigned_vehicle ?baggage_transfer_task ?vehicle_asset)
        (not
          (vehicle_available ?vehicle_asset)
        )
      )
  )
  (:action bind_task_to_transport_leg
    :parameters (?baggage_transfer_task - baggage_transfer_task ?transport_leg - transport_leg)
    :precondition
      (and
        (transfer_entity_registered ?baggage_transfer_task)
        (transfer_entity_vehicle_bound ?baggage_transfer_task)
        (transport_leg_available ?transport_leg)
      )
    :effect
      (and
        (transfer_entity_assigned_transport_leg ?baggage_transfer_task ?transport_leg)
        (not
          (transport_leg_available ?transport_leg)
        )
      )
  )
  (:action activate_transfer_task
    :parameters (?baggage_transfer_task - baggage_transfer_task ?transport_leg - transport_leg)
    :precondition
      (and
        (transfer_entity_registered ?baggage_transfer_task)
        (transfer_entity_vehicle_bound ?baggage_transfer_task)
        (transfer_entity_assigned_transport_leg ?baggage_transfer_task ?transport_leg)
        (not
          (transfer_entity_activated ?baggage_transfer_task)
        )
      )
    :effect (transfer_entity_activated ?baggage_transfer_task)
  )
  (:action release_transport_leg_binding
    :parameters (?baggage_transfer_task - baggage_transfer_task ?transport_leg - transport_leg)
    :precondition
      (and
        (transfer_entity_assigned_transport_leg ?baggage_transfer_task ?transport_leg)
      )
    :effect
      (and
        (transport_leg_available ?transport_leg)
        (not
          (transfer_entity_assigned_transport_leg ?baggage_transfer_task ?transport_leg)
        )
      )
  )
  (:action reserve_handler_for_task
    :parameters (?baggage_transfer_task - baggage_transfer_task ?baggage_handler - baggage_handler)
    :precondition
      (and
        (transfer_entity_activated ?baggage_transfer_task)
        (handler_available ?baggage_handler)
      )
    :effect
      (and
        (transfer_entity_assigned_handler ?baggage_transfer_task ?baggage_handler)
        (not
          (handler_available ?baggage_handler)
        )
      )
  )
  (:action release_handler_from_task
    :parameters (?baggage_transfer_task - baggage_transfer_task ?baggage_handler - baggage_handler)
    :precondition
      (and
        (transfer_entity_assigned_handler ?baggage_transfer_task ?baggage_handler)
      )
    :effect
      (and
        (handler_available ?baggage_handler)
        (not
          (transfer_entity_assigned_handler ?baggage_transfer_task ?baggage_handler)
        )
      )
  )
  (:action assign_handling_equipment_to_plan
    :parameters (?transfer_plan - transfer_plan ?handling_equipment - handling_equipment)
    :precondition
      (and
        (transfer_entity_activated ?transfer_plan)
        (handling_equipment_available ?handling_equipment)
      )
    :effect
      (and
        (plan_assigned_handling_equipment ?transfer_plan ?handling_equipment)
        (not
          (handling_equipment_available ?handling_equipment)
        )
      )
  )
  (:action release_handling_equipment_from_plan
    :parameters (?transfer_plan - transfer_plan ?handling_equipment - handling_equipment)
    :precondition
      (and
        (plan_assigned_handling_equipment ?transfer_plan ?handling_equipment)
      )
    :effect
      (and
        (handling_equipment_available ?handling_equipment)
        (not
          (plan_assigned_handling_equipment ?transfer_plan ?handling_equipment)
        )
      )
  )
  (:action assign_inspection_certificate_to_plan
    :parameters (?transfer_plan - transfer_plan ?inspection_certificate - inspection_certificate)
    :precondition
      (and
        (transfer_entity_activated ?transfer_plan)
        (inspection_certificate_available ?inspection_certificate)
      )
    :effect
      (and
        (plan_assigned_inspection_certificate ?transfer_plan ?inspection_certificate)
        (not
          (inspection_certificate_available ?inspection_certificate)
        )
      )
  )
  (:action release_inspection_certificate_from_plan
    :parameters (?transfer_plan - transfer_plan ?inspection_certificate - inspection_certificate)
    :precondition
      (and
        (plan_assigned_inspection_certificate ?transfer_plan ?inspection_certificate)
      )
    :effect
      (and
        (inspection_certificate_available ?inspection_certificate)
        (not
          (plan_assigned_inspection_certificate ?transfer_plan ?inspection_certificate)
        )
      )
  )
  (:action offer_arrival_zone
    :parameters (?arrival_segment - arrival_segment ?arrival_zone - arrival_zone ?transport_leg - transport_leg)
    :precondition
      (and
        (transfer_entity_activated ?arrival_segment)
        (transfer_entity_assigned_transport_leg ?arrival_segment ?transport_leg)
        (segment_assigned_arrival_zone ?arrival_segment ?arrival_zone)
        (not
          (arrival_zone_ready ?arrival_zone)
        )
        (not
          (arrival_zone_has_container ?arrival_zone)
        )
      )
    :effect (arrival_zone_ready ?arrival_zone)
  )
  (:action accept_arrival_zone_with_handler
    :parameters (?arrival_segment - arrival_segment ?arrival_zone - arrival_zone ?baggage_handler - baggage_handler)
    :precondition
      (and
        (transfer_entity_activated ?arrival_segment)
        (transfer_entity_assigned_handler ?arrival_segment ?baggage_handler)
        (segment_assigned_arrival_zone ?arrival_segment ?arrival_zone)
        (arrival_zone_ready ?arrival_zone)
        (not
          (arrival_segment_prepared ?arrival_segment)
        )
      )
    :effect
      (and
        (arrival_segment_prepared ?arrival_segment)
        (arrival_segment_ready ?arrival_segment)
      )
  )
  (:action stage_baggage_unit_in_arrival_zone
    :parameters (?arrival_segment - arrival_segment ?arrival_zone - arrival_zone ?baggage_unit - baggage_unit)
    :precondition
      (and
        (transfer_entity_activated ?arrival_segment)
        (segment_assigned_arrival_zone ?arrival_segment ?arrival_zone)
        (baggage_unit_available ?baggage_unit)
        (not
          (arrival_segment_prepared ?arrival_segment)
        )
      )
    :effect
      (and
        (arrival_zone_has_container ?arrival_zone)
        (arrival_segment_prepared ?arrival_segment)
        (arrival_segment_offered_baggage_unit ?arrival_segment ?baggage_unit)
        (not
          (baggage_unit_available ?baggage_unit)
        )
      )
  )
  (:action finalize_arrival_unit_staging
    :parameters (?arrival_segment - arrival_segment ?arrival_zone - arrival_zone ?transport_leg - transport_leg ?baggage_unit - baggage_unit)
    :precondition
      (and
        (transfer_entity_activated ?arrival_segment)
        (transfer_entity_assigned_transport_leg ?arrival_segment ?transport_leg)
        (segment_assigned_arrival_zone ?arrival_segment ?arrival_zone)
        (arrival_zone_has_container ?arrival_zone)
        (arrival_segment_offered_baggage_unit ?arrival_segment ?baggage_unit)
        (not
          (arrival_segment_ready ?arrival_segment)
        )
      )
    :effect
      (and
        (arrival_zone_ready ?arrival_zone)
        (arrival_segment_ready ?arrival_segment)
        (baggage_unit_available ?baggage_unit)
        (not
          (arrival_segment_offered_baggage_unit ?arrival_segment ?baggage_unit)
        )
      )
  )
  (:action offer_departure_zone
    :parameters (?departure_segment - departure_segment ?departure_zone - departure_zone ?transport_leg - transport_leg)
    :precondition
      (and
        (transfer_entity_activated ?departure_segment)
        (transfer_entity_assigned_transport_leg ?departure_segment ?transport_leg)
        (segment_assigned_departure_zone ?departure_segment ?departure_zone)
        (not
          (departure_zone_ready ?departure_zone)
        )
        (not
          (departure_zone_has_container ?departure_zone)
        )
      )
    :effect (departure_zone_ready ?departure_zone)
  )
  (:action accept_departure_zone_with_handler
    :parameters (?departure_segment - departure_segment ?departure_zone - departure_zone ?baggage_handler - baggage_handler)
    :precondition
      (and
        (transfer_entity_activated ?departure_segment)
        (transfer_entity_assigned_handler ?departure_segment ?baggage_handler)
        (segment_assigned_departure_zone ?departure_segment ?departure_zone)
        (departure_zone_ready ?departure_zone)
        (not
          (departure_segment_prepared ?departure_segment)
        )
      )
    :effect
      (and
        (departure_segment_prepared ?departure_segment)
        (departure_segment_ready ?departure_segment)
      )
  )
  (:action stage_baggage_unit_in_departure_zone
    :parameters (?departure_segment - departure_segment ?departure_zone - departure_zone ?baggage_unit - baggage_unit)
    :precondition
      (and
        (transfer_entity_activated ?departure_segment)
        (segment_assigned_departure_zone ?departure_segment ?departure_zone)
        (baggage_unit_available ?baggage_unit)
        (not
          (departure_segment_prepared ?departure_segment)
        )
      )
    :effect
      (and
        (departure_zone_has_container ?departure_zone)
        (departure_segment_prepared ?departure_segment)
        (departure_segment_offered_baggage_unit ?departure_segment ?baggage_unit)
        (not
          (baggage_unit_available ?baggage_unit)
        )
      )
  )
  (:action finalize_departure_unit_staging
    :parameters (?departure_segment - departure_segment ?departure_zone - departure_zone ?transport_leg - transport_leg ?baggage_unit - baggage_unit)
    :precondition
      (and
        (transfer_entity_activated ?departure_segment)
        (transfer_entity_assigned_transport_leg ?departure_segment ?transport_leg)
        (segment_assigned_departure_zone ?departure_segment ?departure_zone)
        (departure_zone_has_container ?departure_zone)
        (departure_segment_offered_baggage_unit ?departure_segment ?baggage_unit)
        (not
          (departure_segment_ready ?departure_segment)
        )
      )
    :effect
      (and
        (departure_zone_ready ?departure_zone)
        (departure_segment_ready ?departure_segment)
        (baggage_unit_available ?baggage_unit)
        (not
          (departure_segment_offered_baggage_unit ?departure_segment ?baggage_unit)
        )
      )
  )
  (:action allocate_transfer_unit_to_segments
    :parameters (?arrival_segment - arrival_segment ?departure_segment - departure_segment ?arrival_zone - arrival_zone ?departure_zone - departure_zone ?transfer_vehicle_unit - transfer_vehicle_unit)
    :precondition
      (and
        (arrival_segment_prepared ?arrival_segment)
        (departure_segment_prepared ?departure_segment)
        (segment_assigned_arrival_zone ?arrival_segment ?arrival_zone)
        (segment_assigned_departure_zone ?departure_segment ?departure_zone)
        (arrival_zone_ready ?arrival_zone)
        (departure_zone_ready ?departure_zone)
        (arrival_segment_ready ?arrival_segment)
        (departure_segment_ready ?departure_segment)
        (transfer_unit_available ?transfer_vehicle_unit)
      )
    :effect
      (and
        (transfer_unit_allocated ?transfer_vehicle_unit)
        (transfer_unit_assigned_arrival_zone ?transfer_vehicle_unit ?arrival_zone)
        (transfer_unit_assigned_departure_zone ?transfer_vehicle_unit ?departure_zone)
        (not
          (transfer_unit_available ?transfer_vehicle_unit)
        )
      )
  )
  (:action allocate_transfer_unit_arrival_ready
    :parameters (?arrival_segment - arrival_segment ?departure_segment - departure_segment ?arrival_zone - arrival_zone ?departure_zone - departure_zone ?transfer_vehicle_unit - transfer_vehicle_unit)
    :precondition
      (and
        (arrival_segment_prepared ?arrival_segment)
        (departure_segment_prepared ?departure_segment)
        (segment_assigned_arrival_zone ?arrival_segment ?arrival_zone)
        (segment_assigned_departure_zone ?departure_segment ?departure_zone)
        (arrival_zone_has_container ?arrival_zone)
        (departure_zone_ready ?departure_zone)
        (not
          (arrival_segment_ready ?arrival_segment)
        )
        (departure_segment_ready ?departure_segment)
        (transfer_unit_available ?transfer_vehicle_unit)
      )
    :effect
      (and
        (transfer_unit_allocated ?transfer_vehicle_unit)
        (transfer_unit_assigned_arrival_zone ?transfer_vehicle_unit ?arrival_zone)
        (transfer_unit_assigned_departure_zone ?transfer_vehicle_unit ?departure_zone)
        (transfer_unit_arrival_prepared ?transfer_vehicle_unit)
        (not
          (transfer_unit_available ?transfer_vehicle_unit)
        )
      )
  )
  (:action allocate_transfer_unit_departure_ready
    :parameters (?arrival_segment - arrival_segment ?departure_segment - departure_segment ?arrival_zone - arrival_zone ?departure_zone - departure_zone ?transfer_vehicle_unit - transfer_vehicle_unit)
    :precondition
      (and
        (arrival_segment_prepared ?arrival_segment)
        (departure_segment_prepared ?departure_segment)
        (segment_assigned_arrival_zone ?arrival_segment ?arrival_zone)
        (segment_assigned_departure_zone ?departure_segment ?departure_zone)
        (arrival_zone_ready ?arrival_zone)
        (departure_zone_has_container ?departure_zone)
        (arrival_segment_ready ?arrival_segment)
        (not
          (departure_segment_ready ?departure_segment)
        )
        (transfer_unit_available ?transfer_vehicle_unit)
      )
    :effect
      (and
        (transfer_unit_allocated ?transfer_vehicle_unit)
        (transfer_unit_assigned_arrival_zone ?transfer_vehicle_unit ?arrival_zone)
        (transfer_unit_assigned_departure_zone ?transfer_vehicle_unit ?departure_zone)
        (transfer_unit_departure_prepared ?transfer_vehicle_unit)
        (not
          (transfer_unit_available ?transfer_vehicle_unit)
        )
      )
  )
  (:action allocate_transfer_unit_both_sides_ready
    :parameters (?arrival_segment - arrival_segment ?departure_segment - departure_segment ?arrival_zone - arrival_zone ?departure_zone - departure_zone ?transfer_vehicle_unit - transfer_vehicle_unit)
    :precondition
      (and
        (arrival_segment_prepared ?arrival_segment)
        (departure_segment_prepared ?departure_segment)
        (segment_assigned_arrival_zone ?arrival_segment ?arrival_zone)
        (segment_assigned_departure_zone ?departure_segment ?departure_zone)
        (arrival_zone_has_container ?arrival_zone)
        (departure_zone_has_container ?departure_zone)
        (not
          (arrival_segment_ready ?arrival_segment)
        )
        (not
          (departure_segment_ready ?departure_segment)
        )
        (transfer_unit_available ?transfer_vehicle_unit)
      )
    :effect
      (and
        (transfer_unit_allocated ?transfer_vehicle_unit)
        (transfer_unit_assigned_arrival_zone ?transfer_vehicle_unit ?arrival_zone)
        (transfer_unit_assigned_departure_zone ?transfer_vehicle_unit ?departure_zone)
        (transfer_unit_arrival_prepared ?transfer_vehicle_unit)
        (transfer_unit_departure_prepared ?transfer_vehicle_unit)
        (not
          (transfer_unit_available ?transfer_vehicle_unit)
        )
      )
  )
  (:action prepare_transfer_unit_for_loading
    :parameters (?transfer_vehicle_unit - transfer_vehicle_unit ?arrival_segment - arrival_segment ?transport_leg - transport_leg)
    :precondition
      (and
        (transfer_unit_allocated ?transfer_vehicle_unit)
        (arrival_segment_prepared ?arrival_segment)
        (transfer_entity_assigned_transport_leg ?arrival_segment ?transport_leg)
        (not
          (transfer_unit_loading_commenced ?transfer_vehicle_unit)
        )
      )
    :effect (transfer_unit_loading_commenced ?transfer_vehicle_unit)
  )
  (:action stage_container_for_plan
    :parameters (?transfer_plan - transfer_plan ?baggage_container - baggage_container ?transfer_vehicle_unit - transfer_vehicle_unit)
    :precondition
      (and
        (transfer_entity_activated ?transfer_plan)
        (plan_assigned_transfer_unit ?transfer_plan ?transfer_vehicle_unit)
        (plan_assigned_container ?transfer_plan ?baggage_container)
        (baggage_container_available ?baggage_container)
        (transfer_unit_allocated ?transfer_vehicle_unit)
        (transfer_unit_loading_commenced ?transfer_vehicle_unit)
        (not
          (container_staged ?baggage_container)
        )
      )
    :effect
      (and
        (container_staged ?baggage_container)
        (container_assigned_to_unit ?baggage_container ?transfer_vehicle_unit)
        (not
          (baggage_container_available ?baggage_container)
        )
      )
  )
  (:action attach_container_to_plan
    :parameters (?transfer_plan - transfer_plan ?baggage_container - baggage_container ?transfer_vehicle_unit - transfer_vehicle_unit ?transport_leg - transport_leg)
    :precondition
      (and
        (transfer_entity_activated ?transfer_plan)
        (plan_assigned_container ?transfer_plan ?baggage_container)
        (container_staged ?baggage_container)
        (container_assigned_to_unit ?baggage_container ?transfer_vehicle_unit)
        (transfer_entity_assigned_transport_leg ?transfer_plan ?transport_leg)
        (not
          (transfer_unit_arrival_prepared ?transfer_vehicle_unit)
        )
        (not
          (plan_container_verified ?transfer_plan)
        )
      )
    :effect (plan_container_verified ?transfer_plan)
  )
  (:action attach_priority_designator_to_plan
    :parameters (?transfer_plan - transfer_plan ?priority_designator - priority_designator)
    :precondition
      (and
        (transfer_entity_activated ?transfer_plan)
        (priority_designator_available ?priority_designator)
        (not
          (plan_priority_attached ?transfer_plan)
        )
      )
    :effect
      (and
        (plan_priority_attached ?transfer_plan)
        (plan_assigned_priority ?transfer_plan ?priority_designator)
        (not
          (priority_designator_available ?priority_designator)
        )
      )
  )
  (:action apply_priority_to_plan
    :parameters (?transfer_plan - transfer_plan ?baggage_container - baggage_container ?transfer_vehicle_unit - transfer_vehicle_unit ?transport_leg - transport_leg ?priority_designator - priority_designator)
    :precondition
      (and
        (transfer_entity_activated ?transfer_plan)
        (plan_assigned_container ?transfer_plan ?baggage_container)
        (container_staged ?baggage_container)
        (container_assigned_to_unit ?baggage_container ?transfer_vehicle_unit)
