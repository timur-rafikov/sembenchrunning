--- domain.pddl (as passed to VAL) ---
(define (domain pharmaceutics_capa_action_sequence_planning)
  (:requirements :strips :typing :negative-preconditions)
  (:types personnel - object artifact - object package_supertype - object case_supertype - object capa_record - case_supertype workstream_owner - personnel investigation_lead - personnel functional_reviewer - personnel training_certificate - personnel regulatory_document - personnel inspection_token - personnel resource_allocation - personnel external_validator - personnel support_resource - artifact evidence_artifact - artifact authorized_signatory - artifact validated_system - package_supertype external_reviewer_candidate - package_supertype approval_package - package_supertype capa_substream - capa_record document_substream - capa_record corrective_action - capa_substream preventive_action - capa_substream document_package - document_substream)
  (:predicates
    (capa_initiated ?capa - capa_record)
    (ready_for_implementation ?capa - capa_record)
    (owner_assignment_confirmed ?capa - capa_record)
    (status_completed ?capa - capa_record)
    (ready_for_closure ?capa - capa_record)
    (closure_initiated ?capa - capa_record)
    (owner_available ?owner - workstream_owner)
    (owner_assigned_to ?capa - capa_record ?owner - workstream_owner)
    (investigator_available ?investigator - investigation_lead)
    (investigator_assigned_to ?capa - capa_record ?investigator - investigation_lead)
    (reviewer_available ?reviewer - functional_reviewer)
    (reviewer_assigned_to ?capa - capa_record ?reviewer - functional_reviewer)
    (support_resource_available ?support_resource - support_resource)
    (support_resource_allocated_to_corrective ?corrective_action - corrective_action ?support_resource - support_resource)
    (support_resource_allocated_to_preventive ?preventive_action - preventive_action ?support_resource - support_resource)
    (validated_system_candidate_for_action ?corrective_action - corrective_action ?validated_system - validated_system)
    (validated_system_reserved ?validated_system - validated_system)
    (validated_system_allocated ?validated_system - validated_system)
    (system_allocation_confirmed ?corrective_action - corrective_action)
    (review_channel_assigned_to_preventive ?preventive_action - preventive_action ?external_reviewer - external_reviewer_candidate)
    (external_reviewer_reserved ?external_reviewer - external_reviewer_candidate)
    (external_reviewer_allocated ?external_reviewer - external_reviewer_candidate)
    (preventive_action_system_confirmed ?preventive_action - preventive_action)
    (approval_package_draft_available ?approval_package - approval_package)
    (approval_package_active ?approval_package - approval_package)
    (package_includes_validated_system ?approval_package - approval_package ?validated_system - validated_system)
    (package_includes_external_reviewer ?approval_package - approval_package ?external_reviewer - external_reviewer_candidate)
    (approval_package_requires_training ?approval_package - approval_package)
    (approval_package_requires_external_validation ?approval_package - approval_package)
    (approval_package_locked_for_evidence ?approval_package - approval_package)
    (document_linked_to_corrective_action ?document_package - document_package ?corrective_action - corrective_action)
    (document_linked_to_preventive_action ?document_package - document_package ?preventive_action - preventive_action)
    (document_associated_with_approval_package ?document_package - document_package ?approval_package - approval_package)
    (evidence_available ?evidence_artifact - evidence_artifact)
    (evidence_attached_to_document ?document_package - document_package ?evidence_artifact - evidence_artifact)
    (evidence_committed ?evidence_artifact - evidence_artifact)
    (evidence_linked_to_approval_package ?evidence_artifact - evidence_artifact ?approval_package - approval_package)
    (document_review_ready ?document_package - document_package)
    (document_authorized_for_signoff ?document_package - document_package)
    (document_verified ?document_package - document_package)
    (document_reserved_for_training_attachment ?document_package - document_package)
    (document_training_attached ?document_package - document_package)
    (document_requires_additional_evidence ?document_package - document_package)
    (document_consolidated_verification_done ?document_package - document_package)
    (signatory_available ?authorized_signatory - authorized_signatory)
    (document_linked_to_signatory ?document_package - document_package ?authorized_signatory - authorized_signatory)
    (document_signatory_applied ?document_package - document_package)
    (document_signoff_in_progress ?document_package - document_package)
    (document_signoff_completed ?document_package - document_package)
    (training_record_available ?training_record - training_certificate)
    (document_linked_to_training_record ?document_package - document_package ?training_record - training_certificate)
    (regulatory_artifact_available ?regulatory_artifact - regulatory_document)
    (document_linked_to_regulatory_artifact ?document_package - document_package ?regulatory_artifact - regulatory_document)
    (resource_allocation_available ?resource_allocation - resource_allocation)
    (document_linked_to_resource_allocation ?document_package - document_package ?resource_allocation - resource_allocation)
    (external_validator_available ?external_validator - external_validator)
    (document_linked_to_external_validator ?document_package - document_package ?external_validator - external_validator)
    (inspection_token_available ?inspection_token - inspection_token)
    (inspection_token_assigned_to ?capa - capa_record ?inspection_token - inspection_token)
    (corrective_action_ready ?corrective_action - corrective_action)
    (preventive_action_ready ?preventive_action - preventive_action)
    (document_closure_authorized ?document_package - document_package)
  )
  (:action initiate_capa_record
    :parameters (?capa - capa_record)
    :precondition
      (and
        (not
          (capa_initiated ?capa)
        )
        (not
          (status_completed ?capa)
        )
      )
    :effect (capa_initiated ?capa)
  )
  (:action assign_workstream_owner
    :parameters (?capa - capa_record ?owner - workstream_owner)
    :precondition
      (and
        (capa_initiated ?capa)
        (not
          (owner_assignment_confirmed ?capa)
        )
        (owner_available ?owner)
      )
    :effect
      (and
        (owner_assignment_confirmed ?capa)
        (owner_assigned_to ?capa ?owner)
        (not
          (owner_available ?owner)
        )
      )
  )
  (:action assign_investigation_lead
    :parameters (?capa - capa_record ?investigator - investigation_lead)
    :precondition
      (and
        (capa_initiated ?capa)
        (owner_assignment_confirmed ?capa)
        (investigator_available ?investigator)
      )
    :effect
      (and
        (investigator_assigned_to ?capa ?investigator)
        (not
          (investigator_available ?investigator)
        )
      )
  )
  (:action finalize_investigation
    :parameters (?capa - capa_record ?investigator - investigation_lead)
    :precondition
      (and
        (capa_initiated ?capa)
        (owner_assignment_confirmed ?capa)
        (investigator_assigned_to ?capa ?investigator)
        (not
          (ready_for_implementation ?capa)
        )
      )
    :effect (ready_for_implementation ?capa)
  )
  (:action unassign_investigation_lead
    :parameters (?capa - capa_record ?investigator - investigation_lead)
    :precondition
      (and
        (investigator_assigned_to ?capa ?investigator)
      )
    :effect
      (and
        (investigator_available ?investigator)
        (not
          (investigator_assigned_to ?capa ?investigator)
        )
      )
  )
  (:action reserve_reviewer
    :parameters (?capa - capa_record ?reviewer - functional_reviewer)
    :precondition
      (and
        (ready_for_implementation ?capa)
        (reviewer_available ?reviewer)
      )
    :effect
      (and
        (reviewer_assigned_to ?capa ?reviewer)
        (not
          (reviewer_available ?reviewer)
        )
      )
  )
  (:action release_reviewer
    :parameters (?capa - capa_record ?reviewer - functional_reviewer)
    :precondition
      (and
        (reviewer_assigned_to ?capa ?reviewer)
      )
    :effect
      (and
        (reviewer_available ?reviewer)
        (not
          (reviewer_assigned_to ?capa ?reviewer)
        )
      )
  )
  (:action reserve_resource_for_document_package
    :parameters (?document_package - document_package ?resource_allocation - resource_allocation)
    :precondition
      (and
        (ready_for_implementation ?document_package)
        (resource_allocation_available ?resource_allocation)
      )
    :effect
      (and
        (document_linked_to_resource_allocation ?document_package ?resource_allocation)
        (not
          (resource_allocation_available ?resource_allocation)
        )
      )
  )
  (:action release_resource_from_document_package
    :parameters (?document_package - document_package ?resource_allocation - resource_allocation)
    :precondition
      (and
        (document_linked_to_resource_allocation ?document_package ?resource_allocation)
      )
    :effect
      (and
        (resource_allocation_available ?resource_allocation)
        (not
          (document_linked_to_resource_allocation ?document_package ?resource_allocation)
        )
      )
  )
  (:action reserve_external_validator_for_document
    :parameters (?document_package - document_package ?external_validator - external_validator)
    :precondition
      (and
        (ready_for_implementation ?document_package)
        (external_validator_available ?external_validator)
      )
    :effect
      (and
        (document_linked_to_external_validator ?document_package ?external_validator)
        (not
          (external_validator_available ?external_validator)
        )
      )
  )
  (:action release_external_validator_from_document
    :parameters (?document_package - document_package ?external_validator - external_validator)
    :precondition
      (and
        (document_linked_to_external_validator ?document_package ?external_validator)
      )
    :effect
      (and
        (external_validator_available ?external_validator)
        (not
          (document_linked_to_external_validator ?document_package ?external_validator)
        )
      )
  )
  (:action reserve_validated_system_for_action
    :parameters (?corrective_action - corrective_action ?validated_system - validated_system ?investigator - investigation_lead)
    :precondition
      (and
        (ready_for_implementation ?corrective_action)
        (investigator_assigned_to ?corrective_action ?investigator)
        (validated_system_candidate_for_action ?corrective_action ?validated_system)
        (not
          (validated_system_reserved ?validated_system)
        )
        (not
          (validated_system_allocated ?validated_system)
        )
      )
    :effect (validated_system_reserved ?validated_system)
  )
  (:action confirm_system_allocation
    :parameters (?corrective_action - corrective_action ?validated_system - validated_system ?reviewer - functional_reviewer)
    :precondition
      (and
        (ready_for_implementation ?corrective_action)
        (reviewer_assigned_to ?corrective_action ?reviewer)
        (validated_system_candidate_for_action ?corrective_action ?validated_system)
        (validated_system_reserved ?validated_system)
        (not
          (corrective_action_ready ?corrective_action)
        )
      )
    :effect
      (and
        (corrective_action_ready ?corrective_action)
        (system_allocation_confirmed ?corrective_action)
      )
  )
  (:action allocate_support_resource_to_corrective_action
    :parameters (?corrective_action - corrective_action ?validated_system - validated_system ?support_resource - support_resource)
    :precondition
      (and
        (ready_for_implementation ?corrective_action)
        (validated_system_candidate_for_action ?corrective_action ?validated_system)
        (support_resource_available ?support_resource)
        (not
          (corrective_action_ready ?corrective_action)
        )
      )
    :effect
      (and
        (validated_system_allocated ?validated_system)
        (corrective_action_ready ?corrective_action)
        (support_resource_allocated_to_corrective ?corrective_action ?support_resource)
        (not
          (support_resource_available ?support_resource)
        )
      )
  )
  (:action finalize_system_and_resource_allocation_for_corrective_action
    :parameters (?corrective_action - corrective_action ?validated_system - validated_system ?investigator - investigation_lead ?support_resource - support_resource)
    :precondition
      (and
        (ready_for_implementation ?corrective_action)
        (investigator_assigned_to ?corrective_action ?investigator)
        (validated_system_candidate_for_action ?corrective_action ?validated_system)
        (validated_system_allocated ?validated_system)
        (support_resource_allocated_to_corrective ?corrective_action ?support_resource)
        (not
          (system_allocation_confirmed ?corrective_action)
        )
      )
    :effect
      (and
        (validated_system_reserved ?validated_system)
        (system_allocation_confirmed ?corrective_action)
        (support_resource_available ?support_resource)
        (not
          (support_resource_allocated_to_corrective ?corrective_action ?support_resource)
        )
      )
  )
  (:action reserve_external_reviewer_for_preventive_action
    :parameters (?preventive_action - preventive_action ?external_reviewer - external_reviewer_candidate ?investigator - investigation_lead)
    :precondition
      (and
        (ready_for_implementation ?preventive_action)
        (investigator_assigned_to ?preventive_action ?investigator)
        (review_channel_assigned_to_preventive ?preventive_action ?external_reviewer)
        (not
          (external_reviewer_reserved ?external_reviewer)
        )
        (not
          (external_reviewer_allocated ?external_reviewer)
        )
      )
    :effect (external_reviewer_reserved ?external_reviewer)
  )
  (:action confirm_preventive_reviewer_and_mark_ready
    :parameters (?preventive_action - preventive_action ?external_reviewer - external_reviewer_candidate ?reviewer - functional_reviewer)
    :precondition
      (and
        (ready_for_implementation ?preventive_action)
        (reviewer_assigned_to ?preventive_action ?reviewer)
        (review_channel_assigned_to_preventive ?preventive_action ?external_reviewer)
        (external_reviewer_reserved ?external_reviewer)
        (not
          (preventive_action_ready ?preventive_action)
        )
      )
    :effect
      (and
        (preventive_action_ready ?preventive_action)
        (preventive_action_system_confirmed ?preventive_action)
      )
  )
  (:action allocate_support_and_reserve_reviewer_for_preventive_action
    :parameters (?preventive_action - preventive_action ?external_reviewer - external_reviewer_candidate ?support_resource - support_resource)
    :precondition
      (and
        (ready_for_implementation ?preventive_action)
        (review_channel_assigned_to_preventive ?preventive_action ?external_reviewer)
        (support_resource_available ?support_resource)
        (not
          (preventive_action_ready ?preventive_action)
        )
      )
    :effect
      (and
        (external_reviewer_allocated ?external_reviewer)
        (preventive_action_ready ?preventive_action)
        (support_resource_allocated_to_preventive ?preventive_action ?support_resource)
        (not
          (support_resource_available ?support_resource)
        )
      )
  )
  (:action finalize_support_and_reviewer_allocation_for_preventive_action
    :parameters (?preventive_action - preventive_action ?external_reviewer - external_reviewer_candidate ?investigator - investigation_lead ?support_resource - support_resource)
    :precondition
      (and
        (ready_for_implementation ?preventive_action)
        (investigator_assigned_to ?preventive_action ?investigator)
        (review_channel_assigned_to_preventive ?preventive_action ?external_reviewer)
        (external_reviewer_allocated ?external_reviewer)
        (support_resource_allocated_to_preventive ?preventive_action ?support_resource)
        (not
          (preventive_action_system_confirmed ?preventive_action)
        )
      )
    :effect
      (and
        (external_reviewer_reserved ?external_reviewer)
        (preventive_action_system_confirmed ?preventive_action)
        (support_resource_available ?support_resource)
        (not
          (support_resource_allocated_to_preventive ?preventive_action ?support_resource)
        )
      )
  )
  (:action synthesize_approval_package_with_resources
    :parameters (?corrective_action - corrective_action ?preventive_action - preventive_action ?validated_system - validated_system ?external_reviewer - external_reviewer_candidate ?approval_package - approval_package)
    :precondition
      (and
        (corrective_action_ready ?corrective_action)
        (preventive_action_ready ?preventive_action)
        (validated_system_candidate_for_action ?corrective_action ?validated_system)
        (review_channel_assigned_to_preventive ?preventive_action ?external_reviewer)
        (validated_system_reserved ?validated_system)
        (external_reviewer_reserved ?external_reviewer)
        (system_allocation_confirmed ?corrective_action)
        (preventive_action_system_confirmed ?preventive_action)
        (approval_package_draft_available ?approval_package)
      )
    :effect
      (and
        (approval_package_active ?approval_package)
        (package_includes_validated_system ?approval_package ?validated_system)
        (package_includes_external_reviewer ?approval_package ?external_reviewer)
        (not
          (approval_package_draft_available ?approval_package)
        )
      )
  )
  (:action synthesize_approval_package_with_training_requirement
    :parameters (?corrective_action - corrective_action ?preventive_action - preventive_action ?validated_system - validated_system ?external_reviewer - external_reviewer_candidate ?approval_package - approval_package)
    :precondition
      (and
        (corrective_action_ready ?corrective_action)
        (preventive_action_ready ?preventive_action)
        (validated_system_candidate_for_action ?corrective_action ?validated_system)
        (review_channel_assigned_to_preventive ?preventive_action ?external_reviewer)
        (validated_system_allocated ?validated_system)
        (external_reviewer_reserved ?external_reviewer)
        (not
          (system_allocation_confirmed ?corrective_action)
        )
        (preventive_action_system_confirmed ?preventive_action)
        (approval_package_draft_available ?approval_package)
      )
    :effect
      (and
        (approval_package_active ?approval_package)
        (package_includes_validated_system ?approval_package ?validated_system)
        (package_includes_external_reviewer ?approval_package ?external_reviewer)
        (approval_package_requires_training ?approval_package)
        (not
          (approval_package_draft_available ?approval_package)
        )
      )
  )
  (:action synthesize_approval_package_with_external_validation_requirement
    :parameters (?corrective_action - corrective_action ?preventive_action - preventive_action ?validated_system - validated_system ?external_reviewer - external_reviewer_candidate ?approval_package - approval_package)
    :precondition
      (and
        (corrective_action_ready ?corrective_action)
        (preventive_action_ready ?preventive_action)
        (validated_system_candidate_for_action ?corrective_action ?validated_system)
        (review_channel_assigned_to_preventive ?preventive_action ?external_reviewer)
        (validated_system_reserved ?validated_system)
        (external_reviewer_allocated ?external_reviewer)
        (system_allocation_confirmed ?corrective_action)
        (not
          (preventive_action_system_confirmed ?preventive_action)
        )
        (approval_package_draft_available ?approval_package)
      )
    :effect
      (and
        (approval_package_active ?approval_package)
        (package_includes_validated_system ?approval_package ?validated_system)
        (package_includes_external_reviewer ?approval_package ?external_reviewer)
        (approval_package_requires_external_validation ?approval_package)
        (not
          (approval_package_draft_available ?approval_package)
        )
      )
  )
  (:action synthesize_approval_package_with_training_and_external_validation
    :parameters (?corrective_action - corrective_action ?preventive_action - preventive_action ?validated_system - validated_system ?external_reviewer - external_reviewer_candidate ?approval_package - approval_package)
    :precondition
      (and
        (corrective_action_ready ?corrective_action)
        (preventive_action_ready ?preventive_action)
        (validated_system_candidate_for_action ?corrective_action ?validated_system)
        (review_channel_assigned_to_preventive ?preventive_action ?external_reviewer)
        (validated_system_allocated ?validated_system)
        (external_reviewer_allocated ?external_reviewer)
        (not
          (system_allocation_confirmed ?corrective_action)
        )
        (not
          (preventive_action_system_confirmed ?preventive_action)
        )
        (approval_package_draft_available ?approval_package)
      )
    :effect
      (and
        (approval_package_active ?approval_package)
        (package_includes_validated_system ?approval_package ?validated_system)
        (package_includes_external_reviewer ?approval_package ?external_reviewer)
        (approval_package_requires_training ?approval_package)
        (approval_package_requires_external_validation ?approval_package)
        (not
          (approval_package_draft_available ?approval_package)
        )
      )
  )
  (:action lock_approval_package_for_evidence_attachment
    :parameters (?approval_package - approval_package ?corrective_action - corrective_action ?investigator - investigation_lead)
    :precondition
      (and
        (approval_package_active ?approval_package)
        (corrective_action_ready ?corrective_action)
        (investigator_assigned_to ?corrective_action ?investigator)
        (not
          (approval_package_locked_for_evidence ?approval_package)
        )
      )
    :effect (approval_package_locked_for_evidence ?approval_package)
  )
  (:action attach_evidence_and_commit_to_package
    :parameters (?document_package - document_package ?evidence_artifact - evidence_artifact ?approval_package - approval_package)
    :precondition
      (and
        (ready_for_implementation ?document_package)
        (document_associated_with_approval_package ?document_package ?approval_package)
        (evidence_attached_to_document ?document_package ?evidence_artifact)
        (evidence_available ?evidence_artifact)
        (approval_package_active ?approval_package)
        (approval_package_locked_for_evidence ?approval_package)
        (not
          (evidence_committed ?evidence_artifact)
        )
      )
    :effect
      (and
        (evidence_committed ?evidence_artifact)
        (evidence_linked_to_approval_package ?evidence_artifact ?approval_package)
        (not
          (evidence_available ?evidence_artifact)
        )
      )
  )
  (:action finalize_document_and_mark_review_ready
    :parameters (?document_package - document_package ?evidence_artifact - evidence_artifact ?approval_package - approval_package ?investigator - investigation_lead)
    :precondition
      (and
        (ready_for_implementation ?document_package)
        (evidence_attached_to_document ?document_package ?evidence_artifact)
        (evidence_committed ?evidence_artifact)
        (evidence_linked_to_approval_package ?evidence_artifact ?approval_package)
        (investigator_assigned_to ?document_package ?investigator)
        (not
          (approval_package_requires_training ?approval_package)
        )
        (not
          (document_review_ready ?document_package)
        )
      )
    :effect (document_review_ready ?document_package)
  )
  (:action reserve_training_record_for_document
    :parameters (?document_package - document_package ?training_record - training_certificate)
    :precondition
      (and
        (ready_for_implementation ?document_package)
        (training_record_available ?training_record)
        (not
          (document_reserved_for_training_attachment ?document_package)
        )
      )
    :effect
      (and
        (document_reserved_for_training_attachment ?document_package)
        (document_linked_to_training_record ?document_package ?training_record)
        (not
          (training_record_available ?training_record)
        )
      )
  )
  (:action attach_training_and_finalize_document_review
    :parameters (?document_package - document_package ?evidence_artifact - evidence_artifact ?approval_package - approval_package ?investigator - investigation_lead ?training_record - training_certificate)
    :precondition
      (and
        (ready_for_implementation ?document_package)
        (evidence_attached_to_document ?document_package ?evidence_artifact)
        (evidence_committed ?evidence_artifact)
        (evidence_linked_to_approval_package ?evidence_artifact ?approval_package)
        (investigator_assigned_to ?document_package ?investigator)
        (approval_package_requires_training ?approval_package)
        (document_reserved_for_training_attachment ?document_package)
        (document_linked_to_training_record ?document_package ?training_record)
        (not
          (document_review_ready ?document_package)
        )
      )
    :effect
      (and
        (document_review_ready ?document_package)
        (document_training_attached ?document_package)
      )
  )
  (:action initiate_document_signoff_sequence
    :parameters (?document_package - document_package ?resource_allocation - resource_allocation ?reviewer - functional_reviewer ?evidence_artifact - evidence_artifact ?approval_package - approval_package)
    :precondition
      (and
        (document_review_ready ?document_package)
        (document_linked_to_resource_allocation ?document_package ?resource_allocation)
        (reviewer_assigned_to ?document_package ?reviewer)
        (evidence_attached_to_document ?document_package ?evidence_artifact)
        (evidence_linked_to_approval_package ?evidence_artifact ?approval_package)
        (not
          (approval_package_requires_external_validation ?approval_package)
        )
        (not
          (document_authorized_for_signoff ?document_package)
        )
      )
    :effect (document_authorized_for_signoff ?document_package)
