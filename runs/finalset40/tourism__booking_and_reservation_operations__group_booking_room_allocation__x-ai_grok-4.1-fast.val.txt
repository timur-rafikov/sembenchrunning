  (:requirements :strips :typing :negative-preconditions)
  (:types inventory_domain - object auxiliary_resource_domain - object classification_domain - object booking_entity_root - object group_booking - booking_entity_root room_unit - inventory_domain rate_plan_option - inventory_domain allocation_agent - inventory_domain payment_instrument - inventory_domain billing_account - inventory_domain confirmation_token - inventory_domain supplier_confirmation - inventory_domain insurance_option - inventory_domain amenity_package - auxiliary_resource_domain guest_profile - auxiliary_resource_domain voucher_or_promo - auxiliary_resource_domain room_type - classification_domain room_grouping - classification_domain proposal_package - classification_domain lead_booking - group_booking sub_booking_unit - group_booking primary_room_request - lead_booking secondary_room_request - lead_booking room_assignment_record - sub_booking_unit)
  (:predicates
    (group_booking_initiated ?group_booking - group_booking)
    (booking_confirmed ?group_booking - group_booking)
    (provisional_allocation ?group_booking - group_booking)

    (ready_for_enrichment ?group_booking - group_booking)
    (enriched_with_confirmation ?group_booking - group_booking)
    (room_unit_available ?room_unit - room_unit)

    (rate_option_available ?rate_plan_option - rate_plan_option)
    (has_rate_option ?group_booking - group_booking ?rate_plan_option - rate_plan_option)
    (allocation_agent_available ?allocation_agent - allocation_agent)
    (assigned_allocation_agent ?group_booking - group_booking ?allocation_agent - allocation_agent)
    (amenity_package_available ?amenity_package - amenity_package)
    (primary_request_has_amenity ?primary_room_request - primary_room_request ?amenity_package - amenity_package)
    (secondary_request_has_amenity ?secondary_room_request - secondary_room_request ?amenity_package - amenity_package)
    (primary_request_prefers_room_type ?primary_room_request - primary_room_request ?room_type - room_type)
    (room_type_hold_primary ?room_type - room_type)
    (room_type_hold_auxiliary ?room_type - room_type)
    (primary_request_ready_for_package ?primary_room_request - primary_room_request)
    (secondary_request_prefers_room_group ?secondary_room_request - secondary_room_request ?room_grouping - room_grouping)
    (room_group_hold_primary ?room_grouping - room_grouping)
    (room_group_hold_auxiliary ?room_grouping - room_grouping)
    (secondary_request_ready_for_package ?secondary_room_request - secondary_room_request)
    (proposal_package_proposed ?proposal_package - proposal_package)
    (proposal_package_assembled ?proposal_package - proposal_package)
    (proposal_includes_room_type ?proposal_package - proposal_package ?room_type - room_type)
    (proposal_includes_room_group ?proposal_package - proposal_package ?room_grouping - room_grouping)
    (proposal_locked_for_assignment ?proposal_package - proposal_package)
    (proposal_locked_for_supplier_confirmation ?proposal_package - proposal_package)
    (proposal_profiles_ready ?proposal_package - proposal_package)
    (assignment_links_primary_request ?room_assignment_record - room_assignment_record ?primary_room_request - primary_room_request)
    (assignment_links_secondary_request ?room_assignment_record - room_assignment_record ?secondary_room_request - secondary_room_request)
    (assignment_part_of_proposal ?room_assignment_record - room_assignment_record ?proposal_package - proposal_package)
    (guest_profile_available ?guest_profile - guest_profile)
    (assignment_has_guest_profile ?room_assignment_record - room_assignment_record ?guest_profile - guest_profile)
    (guest_profile_in_use ?guest_profile - guest_profile)
    (guest_profile_attached_to_proposal ?guest_profile - guest_profile ?proposal_package - proposal_package)
    (assignment_profile_locked ?room_assignment_record - room_assignment_record)
    (assignment_supplier_attached ?room_assignment_record - room_assignment_record)

    (assignment_payment_instrument_attached_flag ?room_assignment_record - room_assignment_record)
    (assignment_package_enriched ?room_assignment_record - room_assignment_record)
    (assignment_needs_billing_account ?room_assignment_record - room_assignment_record)
    (assignment_prepared_for_finalization ?room_assignment_record - room_assignment_record)
    (voucher_available ?voucher_or_promo - voucher_or_promo)
    (assignment_link_voucher ?room_assignment_record - room_assignment_record ?voucher_or_promo - voucher_or_promo)

    (assignment_voucher_authorized ?room_assignment_record - room_assignment_record)
    (assignment_voucher_approved ?room_assignment_record - room_assignment_record)
    (payment_instrument_available ?payment_instrument - payment_instrument)
    (assignment_attached_payment_instrument ?room_assignment_record - room_assignment_record ?payment_instrument - payment_instrument)
    (billing_account_available ?billing_account - billing_account)
    (assignment_attached_billing_account ?room_assignment_record - room_assignment_record ?billing_account - billing_account)
    (supplier_confirmation_available ?supplier_confirmation - supplier_confirmation)
    (assignment_has_supplier_confirmation ?room_assignment_record - room_assignment_record ?supplier_confirmation - supplier_confirmation)
    (insurance_option_available ?insurance_option - insurance_option)
    (assignment_has_insurance_option ?room_assignment_record - room_assignment_record ?insurance_option - insurance_option)

    (primary_request_optioned ?primary_room_request - primary_room_request)
    (secondary_request_optioned ?secondary_room_request - secondary_room_request)
    (assignment_finalized ?room_assignment_record - room_assignment_record)
  )
  (:action initiate_group_booking
    :parameters (?group_booking - group_booking)
    :precondition
      (and
        (not
          (group_booking_initiated ?group_booking)
        )
        (not
          (committed ?group_booking)
        )
      )
    :effect (group_booking_initiated ?group_booking)
  )
  (:action hold_room_unit_for_booking
    :parameters (?group_booking - group_booking ?room_unit - room_unit)
    :precondition
      (and
        (group_booking_initiated ?group_booking)
        (not
          (provisional_allocation ?group_booking)
        )
        (room_unit_available ?room_unit)
      )
    :effect
      (and
        (provisional_allocation ?group_booking)
        (holds_room_unit ?group_booking ?room_unit)
        (not
          (room_unit_available ?room_unit)
        )
      )
  )
  (:action attach_rate_option_to_booking
    :parameters (?group_booking - group_booking ?rate_plan_option - rate_plan_option)
    :precondition
      (and
        (group_booking_initiated ?group_booking)
        (provisional_allocation ?group_booking)
        (rate_option_available ?rate_plan_option)
      )
    :effect
      (and
        (has_rate_option ?group_booking ?rate_plan_option)
        (not
          (rate_option_available ?rate_plan_option)
        )
      )
  )
  (:action confirm_group_booking
    :parameters (?group_booking - group_booking ?rate_plan_option - rate_plan_option)
    :precondition
      (and
        (group_booking_initiated ?group_booking)
        (provisional_allocation ?group_booking)
        (has_rate_option ?group_booking ?rate_plan_option)
        (not
          (booking_confirmed ?group_booking)
        )
      )
    :effect (booking_confirmed ?group_booking)
  )
  (:action release_rate_option_from_booking
    :parameters (?group_booking - group_booking ?rate_plan_option - rate_plan_option)
    :precondition
      (and
        (has_rate_option ?group_booking ?rate_plan_option)
      )
    :effect
      (and
        (rate_option_available ?rate_plan_option)
        (not
          (has_rate_option ?group_booking ?rate_plan_option)
        )
      )
  )
  (:action assign_allocation_agent_to_booking
    :parameters (?group_booking - group_booking ?allocation_agent - allocation_agent)
    :precondition
      (and
        (booking_confirmed ?group_booking)
        (allocation_agent_available ?allocation_agent)
      )
    :effect
      (and
        (assigned_allocation_agent ?group_booking ?allocation_agent)
        (not
          (allocation_agent_available ?allocation_agent)
        )
      )
  )
  (:action unassign_allocation_agent_from_booking
    :parameters (?group_booking - group_booking ?allocation_agent - allocation_agent)
    :precondition
      (and
        (assigned_allocation_agent ?group_booking ?allocation_agent)
      )
    :effect
      (and
        (allocation_agent_available ?allocation_agent)
        (not
          (assigned_allocation_agent ?group_booking ?allocation_agent)
        )
      )
  )
  (:action attach_supplier_confirmation_to_assignment_record
    :parameters (?room_assignment_record - room_assignment_record ?supplier_confirmation - supplier_confirmation)
    :precondition
      (and
        (booking_confirmed ?room_assignment_record)
        (supplier_confirmation_available ?supplier_confirmation)
      )
    :effect
      (and
        (assignment_has_supplier_confirmation ?room_assignment_record ?supplier_confirmation)

          (supplier_confirmation_available ?supplier_confirmation)
        )
      )
  )
  (:action detach_supplier_confirmation_from_assignment
    :parameters (?room_assignment_record - room_assignment_record ?supplier_confirmation - supplier_confirmation)
    :precondition
      (and
        (assignment_has_supplier_confirmation ?room_assignment_record ?supplier_confirmation)
      )
    :effect
      (and
        (supplier_confirmation_available ?supplier_confirmation)
        (not
          (assignment_has_supplier_confirmation ?room_assignment_record ?supplier_confirmation)
        )
      )
  )
  (:action attach_insurance_option_to_assignment
    :parameters (?room_assignment_record - room_assignment_record ?insurance_option - insurance_option)
    :precondition
      (and
        (booking_confirmed ?room_assignment_record)
        (insurance_option_available ?insurance_option)
      )
    :effect
      (and
        (assignment_has_insurance_option ?room_assignment_record ?insurance_option)
        (not
          (insurance_option_available ?insurance_option)
        )
      )
  )
  (:action detach_insurance_option_from_assignment
    :parameters (?room_assignment_record - room_assignment_record ?insurance_option - insurance_option)
    :precondition
      (and
        (assignment_has_insurance_option ?room_assignment_record ?insurance_option)
      )
    :effect
      (and
        (insurance_option_available ?insurance_option)
        (not
          (assignment_has_insurance_option ?room_assignment_record ?insurance_option)
        )
      )
  )
  (:action place_primary_hold_on_room_type
    :parameters (?primary_room_request - primary_room_request ?room_type - room_type ?rate_plan_option - rate_plan_option)
    :precondition
      (and
        (booking_confirmed ?primary_room_request)
        (has_rate_option ?primary_room_request ?rate_plan_option)
        (primary_request_prefers_room_type ?primary_room_request ?room_type)
        (not
          (room_type_hold_primary ?room_type)
        )
        (not
          (room_type_hold_auxiliary ?room_type)
        )
      )
    :effect (room_type_hold_primary ?room_type)
  )
  (:action finalize_primary_hold_with_agent
    :parameters (?primary_room_request - primary_room_request ?room_type - room_type ?allocation_agent - allocation_agent)
    :precondition
      (and
        (booking_confirmed ?primary_room_request)
        (assigned_allocation_agent ?primary_room_request ?allocation_agent)
        (primary_request_prefers_room_type ?primary_room_request ?room_type)
        (room_type_hold_primary ?room_type)
        (not
          (primary_request_optioned ?primary_room_request)
        )
      )
    :effect
      (and
        (primary_request_optioned ?primary_room_request)
        (primary_request_ready_for_package ?primary_room_request)
      )
  )
  (:action attach_amenity_and_option_to_primary_request
    :parameters (?primary_room_request - primary_room_request ?room_type - room_type ?amenity_package - amenity_package)
    :precondition
      (and
        (booking_confirmed ?primary_room_request)
        (primary_request_prefers_room_type ?primary_room_request ?room_type)
        (amenity_package_available ?amenity_package)
        (not
          (primary_request_optioned ?primary_room_request)
        )
      )
    :effect
      (and
        (room_type_hold_auxiliary ?room_type)
        (primary_request_optioned ?primary_room_request)
        (primary_request_has_amenity ?primary_room_request ?amenity_package)
        (not
          (amenity_package_available ?amenity_package)
        )
      )
  )
  (:action promote_primary_option_to_hold
    :parameters (?primary_room_request - primary_room_request ?room_type - room_type ?rate_plan_option - rate_plan_option ?amenity_package - amenity_package)
    :precondition
      (and
        (booking_confirmed ?primary_room_request)
        (has_rate_option ?primary_room_request ?rate_plan_option)
        (primary_request_prefers_room_type ?primary_room_request ?room_type)
        (room_type_hold_auxiliary ?room_type)
        (primary_request_has_amenity ?primary_room_request ?amenity_package)
        (not
          (primary_request_ready_for_package ?primary_room_request)
        )
      )
    :effect
      (and
        (room_type_hold_primary ?room_type)
        (primary_request_ready_for_package ?primary_room_request)
        (amenity_package_available ?amenity_package)
        (not
          (primary_request_has_amenity ?primary_room_request ?amenity_package)
        )
      )
  )
  (:action place_secondary_hold_on_room_group
    :parameters (?secondary_room_request - secondary_room_request ?room_grouping - room_grouping ?rate_plan_option - rate_plan_option)
    :precondition
      (and

  (:action finalize_secondary_hold_with_agent
    :parameters (?secondary_room_request - secondary_room_request ?room_grouping - room_grouping ?allocation_agent - allocation_agent)

      (and
        (booking_confirmed ?secondary_room_request)
        (assigned_allocation_agent ?secondary_room_request ?allocation_agent)
        (secondary_request_prefers_room_group ?secondary_room_request ?room_grouping)

        (not
          (secondary_request_optioned ?secondary_room_request)
        )
      )
    :effect
      (and
        (secondary_request_optioned ?secondary_room_request)
        (secondary_request_ready_for_package ?secondary_room_request)
      )
  )
  (:action attach_amenity_to_secondary_request
    :parameters (?secondary_room_request - secondary_room_request ?room_grouping - room_grouping ?amenity_package - amenity_package)
    :precondition
      (and
        (booking_confirmed ?secondary_room_request)
        (secondary_request_prefers_room_group ?secondary_room_request ?room_grouping)
        (amenity_package_available ?amenity_package)
        (not
          (secondary_request_optioned ?secondary_room_request)
        )
      )
    :effect
      (and
        (room_group_hold_auxiliary ?room_grouping)
        (secondary_request_optioned ?secondary_room_request)
        (secondary_request_has_amenity ?secondary_room_request ?amenity_package)
        (not
          (amenity_package_available ?amenity_package)
        )
      )
  )
  (:action promote_secondary_option_to_hold
    :parameters (?secondary_room_request - secondary_room_request ?room_grouping - room_grouping ?rate_plan_option - rate_plan_option ?amenity_package - amenity_package)
    :precondition
      (and
        (booking_confirmed ?secondary_room_request)
        (has_rate_option ?secondary_room_request ?rate_plan_option)
        (secondary_request_prefers_room_group ?secondary_room_request ?room_grouping)
        (room_group_hold_auxiliary ?room_grouping)
        (secondary_request_has_amenity ?secondary_room_request ?amenity_package)
        (not
          (secondary_request_ready_for_package ?secondary_room_request)
        )
      )
    :effect
      (and
        (room_group_hold_primary ?room_grouping)
        (secondary_request_ready_for_package ?secondary_room_request)
        (amenity_package_available ?amenity_package)
        (not
          (secondary_request_has_amenity ?secondary_room_request ?amenity_package)
        )
      )
  )
  (:action assemble_proposal_package
    :parameters (?primary_room_request - primary_room_request ?secondary_room_request - secondary_room_request ?room_type - room_type ?room_grouping - room_grouping ?proposal_package - proposal_package)
    :precondition
      (and
        (primary_request_optioned ?primary_room_request)
        (secondary_request_optioned ?secondary_room_request)
        (primary_request_prefers_room_type ?primary_room_request ?room_type)
        (secondary_request_prefers_room_group ?secondary_room_request ?room_grouping)
        (room_type_hold_primary ?room_type)
        (room_group_hold_primary ?room_grouping)
        (primary_request_ready_for_package ?primary_room_request)
        (secondary_request_ready_for_package ?secondary_room_request)
        (proposal_package_proposed ?proposal_package)
      )
    :effect
      (and
        (proposal_package_assembled ?proposal_package)
        (proposal_includes_room_type ?proposal_package ?room_type)
        (proposal_includes_room_group ?proposal_package ?room_grouping)
        (not
          (proposal_package_proposed ?proposal_package)
        )
      )
  )
  (:action assemble_and_lock_proposal_package
    :parameters (?primary_room_request - primary_room_request ?secondary_room_request - secondary_room_request ?room_type - room_type ?room_grouping - room_grouping ?proposal_package - proposal_package)
    :precondition
      (and
        (primary_request_optioned ?primary_room_request)
        (secondary_request_optioned ?secondary_room_request)
        (primary_request_prefers_room_type ?primary_room_request ?room_type)
        (secondary_request_prefers_room_group ?secondary_room_request ?room_grouping)
        (room_type_hold_auxiliary ?room_type)
        (room_group_hold_primary ?room_grouping)
        (not
          (primary_request_ready_for_package ?primary_room_request)
        )
        (secondary_request_ready_for_package ?secondary_room_request)
        (proposal_package_proposed ?proposal_package)
      )
    :effect
      (and
        (proposal_package_assembled ?proposal_package)
        (proposal_includes_room_type ?proposal_package ?room_type)
        (proposal_includes_room_group ?proposal_package ?room_grouping)
        (proposal_locked_for_assignment ?proposal_package)
        (not
          (proposal_package_proposed ?proposal_package)
        )
      )
  )
  (:action assemble_and_lock_proposal_package_supplier_variant
    :parameters (?primary_room_request - primary_room_request ?secondary_room_request - secondary_room_request ?room_type - room_type ?room_grouping - room_grouping ?proposal_package - proposal_package)
    :precondition
      (and
        (primary_request_optioned ?primary_room_request)
        (secondary_request_optioned ?secondary_room_request)
        (primary_request_prefers_room_type ?primary_room_request ?room_type)
        (secondary_request_prefers_room_group ?secondary_room_request ?room_grouping)
        (room_type_hold_primary ?room_type)
        (room_group_hold_auxiliary ?room_grouping)
        (primary_request_ready_for_package ?primary_room_request)
        (not
          (secondary_request_ready_for_package ?secondary_room_request)
        )
        (proposal_package_proposed ?proposal_package)
      )
    :effect
      (and
        (proposal_package_assembled ?proposal_package)
        (proposal_includes_room_type ?proposal_package ?room_type)
        (proposal_includes_room_group ?proposal_package ?room_grouping)
        (proposal_locked_for_supplier_confirmation ?proposal_package)
        (not
          (proposal_package_proposed ?proposal_package)
        )
      )
  )
  (:action assemble_and_lock_proposal_package_fully
    :parameters (?primary_room_request - primary_room_request ?secondary_room_request - secondary_room_request ?room_type - room_type ?room_grouping - room_grouping ?proposal_package - proposal_package)
    :precondition
      (and
        (primary_request_optioned ?primary_room_request)
        (secondary_request_optioned ?secondary_room_request)
        (primary_request_prefers_room_type ?primary_room_request ?room_type)
        (secondary_request_prefers_room_group ?secondary_room_request ?room_grouping)
        (room_type_hold_auxiliary ?room_type)
        (room_group_hold_auxiliary ?room_grouping)
        (not
          (primary_request_ready_for_package ?primary_room_request)

        (not
          (secondary_request_ready_for_package ?secondary_room_request)

        (proposal_package_proposed ?proposal_package)
      )
    :effect
      (and
        (proposal_package_assembled ?proposal_package)
        (proposal_includes_room_type ?proposal_package ?room_type)
        (proposal_includes_room_group ?proposal_package ?room_grouping)
        (proposal_locked_for_assignment ?proposal_package)
        (proposal_locked_for_supplier_confirmation ?proposal_package)
        (not
          (proposal_package_proposed ?proposal_package)
        )
      )
  )
  (:action prepare_proposal_profiles
    :parameters (?proposal_package - proposal_package ?primary_room_request - primary_room_request ?rate_plan_option - rate_plan_option)
    :precondition
      (and
        (proposal_package_assembled ?proposal_package)
        (primary_request_optioned ?primary_room_request)
        (has_rate_option ?primary_room_request ?rate_plan_option)
        (not
          (proposal_profiles_ready ?proposal_package)
        )
      )
    :effect (proposal_profiles_ready ?proposal_package)
  )
  (:action attach_guest_profile_to_assignment_from_proposal
    :parameters (?room_assignment_record - room_assignment_record ?guest_profile - guest_profile ?proposal_package - proposal_package)
    :precondition
      (and
        (booking_confirmed ?room_assignment_record)
        (assignment_part_of_proposal ?room_assignment_record ?proposal_package)
        (assignment_has_guest_profile ?room_assignment_record ?guest_profile)
        (guest_profile_available ?guest_profile)
        (proposal_package_assembled ?proposal_package)
        (proposal_profiles_ready ?proposal_package)
        (not
          (guest_profile_in_use ?guest_profile)
        )
      )
    :effect
      (and
        (guest_profile_in_use ?guest_profile)
        (guest_profile_attached_to_proposal ?guest_profile ?proposal_package)
        (not
          (guest_profile_available ?guest_profile)
        )
      )
  )
  (:action lock_assignment_profile
    :parameters (?room_assignment_record - room_assignment_record ?guest_profile - guest_profile ?proposal_package - proposal_package ?rate_plan_option - rate_plan_option)
    :precondition
      (and
        (booking_confirmed ?room_assignment_record)
        (assignment_has_guest_profile ?room_assignment_record ?guest_profile)
        (guest_profile_in_use ?guest_profile)
        (guest_profile_attached_to_proposal ?guest_profile ?proposal_package)
        (has_rate_option ?room_assignment_record ?rate_plan_option)
        (not
          (proposal_locked_for_assignment ?proposal_package)
        )
        (not
          (assignment_profile_locked ?room_assignment_record)
        )
      )
    :effect (assignment_profile_locked ?room_assignment_record)
  )
  (:action attach_payment_instrument_to_assignment
    :parameters (?room_assignment_record - room_assignment_record ?payment_instrument - payment_instrument)
    :precondition
      (and
        (booking_confirmed ?room_assignment_record)
        (payment_instrument_available ?payment_instrument)
        (not
          (assignment_payment_instrument_attached_flag ?room_assignment_record)
        )
      )
    :effect
