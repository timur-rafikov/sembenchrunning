    (functional_reviewer_available ?functional_reviewer - functional_reviewer)
    (assigned_functional_reviewer ?change_request - change_control_request ?functional_reviewer - functional_reviewer)
    (supporting_document_available ?supporting_document - supporting_document)
    (technical_owner_has_supporting_document ?technical_owner - technical_owner ?supporting_document - supporting_document)
    (process_owner_has_supporting_document ?process_owner - process_owner ?supporting_document - supporting_document)
    (technical_owner_impacts_technical_area ?technical_owner - technical_owner ?technical_impact_area - technical_impact_area)
    (technical_impact_identified ?technical_impact_area - technical_impact_area)
    (technical_impact_has_supporting_document ?technical_impact_area - technical_impact_area)
    (technical_owner_analysis_complete ?technical_owner - technical_owner)
    (process_owner_impacts_quality_area ?process_owner - process_owner ?quality_impact_area - quality_impact_area)

    (change_dossier_registered ?change_dossier - change_dossier)
    (change_dossier_links_technical_area ?change_dossier - change_dossier ?technical_impact_area - technical_impact_area)
    (change_dossier_links_quality_area ?change_dossier - change_dossier ?quality_impact_area - quality_impact_area)
    (dossier_requires_validation_protocol ?change_dossier - change_dossier)
    (dossier_requires_regulatory_attachment ?change_dossier - change_dossier)
    (change_dossier_evidence_reviewed ?change_dossier - change_dossier)
    (assessment_associates_technical_owner ?impact_assessment - impact_assessment ?technical_owner - technical_owner)
    (assessment_associates_process_owner ?impact_assessment - impact_assessment ?process_owner - process_owner)
    (assessment_associated_with_dossier ?impact_assessment - impact_assessment ?change_dossier - change_dossier)
    (evidence_item_available ?evidence_item - evidence_item)
    (assessment_has_evidence_item ?impact_assessment - impact_assessment ?evidence_item - evidence_item)

    (evidence_item_attached_to_dossier ?evidence_item - evidence_item ?change_dossier - change_dossier)

  (:action record_change_request
    :parameters (?change_request - change_control_request)
    :precondition
      (and
        (not
          (change_request_registered ?change_request)
        )
        (not
          (change_closed ?change_request)
        )
      )
    :effect (change_request_registered ?change_request)
  )
  (:action assign_assessor_to_request
    :parameters (?change_request - change_control_request ?assessor_candidate - assessor_candidate)
    :precondition
      (and
        (change_request_registered ?change_request)
        (not
          (assessor_assignment_recorded ?change_request)
        )
        (assessor_candidate_available ?assessor_candidate)
      )
    :effect
      (and
        (assessor_assignment_recorded ?change_request)
        (assigned_assessor ?change_request ?assessor_candidate)
        (not
          (assessor_candidate_available ?assessor_candidate)
        )
      )
  )
  (:action engage_sme_for_request
    :parameters (?change_request - change_control_request ?subject_matter_expert - subject_matter_expert)
    :precondition
      (and
        (change_request_registered ?change_request)
        (assessor_assignment_recorded ?change_request)
        (sme_available ?subject_matter_expert)
      )
    :effect
      (and
        (assigned_sme ?change_request ?subject_matter_expert)
        (not
          (sme_available ?subject_matter_expert)
        )
      )
  )
  (:action finalize_sme_engagement_for_request
    :parameters (?change_request - change_control_request ?subject_matter_expert - subject_matter_expert)
    :precondition
      (and
        (change_request_registered ?change_request)
        (assessor_assignment_recorded ?change_request)
        (assigned_sme ?change_request ?subject_matter_expert)
        (not
          (ready_for_assessment ?change_request)
        )
      )
    :effect (ready_for_assessment ?change_request)
  )
  (:action release_sme_from_request
    :parameters (?change_request - change_control_request ?subject_matter_expert - subject_matter_expert)
    :precondition
      (and
        (assigned_sme ?change_request ?subject_matter_expert)
      )
    :effect
      (and
        (sme_available ?subject_matter_expert)
        (not
          (assigned_sme ?change_request ?subject_matter_expert)
        )
      )
  )
  (:action assign_functional_reviewer_to_request
    :parameters (?change_request - change_control_request ?functional_reviewer - functional_reviewer)
    :precondition
      (and
        (ready_for_assessment ?change_request)
        (functional_reviewer_available ?functional_reviewer)
      )
    :effect
      (and
        (assigned_functional_reviewer ?change_request ?functional_reviewer)
        (not
          (functional_reviewer_available ?functional_reviewer)
        )
      )
  )
  (:action release_functional_reviewer_from_request
    :parameters (?change_request - change_control_request ?functional_reviewer - functional_reviewer)
    :precondition
      (and
        (assigned_functional_reviewer ?change_request ?functional_reviewer)
      )
    :effect
      (and
        (functional_reviewer_available ?functional_reviewer)
        (not
          (assigned_functional_reviewer ?change_request ?functional_reviewer)
        )
      )
  )
  (:action assign_validation_protocol_to_assessment
    :parameters (?impact_assessment - impact_assessment ?validation_protocol - validation_protocol)

        (ready_for_assessment ?impact_assessment)
        (validation_protocol_available ?validation_protocol)

        )
      )

  (:action unassign_validation_protocol_from_assessment
    :parameters (?impact_assessment - impact_assessment ?validation_protocol - validation_protocol)

        (assessment_has_validation_protocol ?impact_assessment ?validation_protocol)

      (and
        (validation_protocol_available ?validation_protocol)

          (assessment_has_validation_protocol ?impact_assessment ?validation_protocol)
        )
      )
  )
  (:action assign_technical_assessment_to_assessment
    :parameters (?impact_assessment - impact_assessment ?technical_assessment - technical_assessment)
    :precondition
      (and
        (ready_for_assessment ?impact_assessment)
        (technical_assessment_available ?technical_assessment)
      )
    :effect

        (assessment_has_technical_assessment ?impact_assessment ?technical_assessment)

          (technical_assessment_available ?technical_assessment)
        )
      )
  )
  (:action unassign_technical_assessment_from_assessment
    :parameters (?impact_assessment - impact_assessment ?technical_assessment - technical_assessment)
    :precondition
      (and
        (assessment_has_technical_assessment ?impact_assessment ?technical_assessment)
      )
    :effect
      (and
        (technical_assessment_available ?technical_assessment)
        (not
          (assessment_has_technical_assessment ?impact_assessment ?technical_assessment)

  (:action identify_technical_impact_area
    :parameters (?technical_owner - technical_owner ?technical_impact_area - technical_impact_area ?subject_matter_expert - subject_matter_expert)
    :precondition
      (and
        (ready_for_assessment ?technical_owner)
        (assigned_sme ?technical_owner ?subject_matter_expert)
        (technical_owner_impacts_technical_area ?technical_owner ?technical_impact_area)
        (not
          (technical_impact_identified ?technical_impact_area)
        )
        (not
          (technical_impact_has_supporting_document ?technical_impact_area)
        )
      )
    :effect (technical_impact_identified ?technical_impact_area)
  )
  (:action record_technical_owner_review
    :parameters (?technical_owner - technical_owner ?technical_impact_area - technical_impact_area ?functional_reviewer - functional_reviewer)
    :precondition
      (and
        (ready_for_assessment ?technical_owner)
        (assigned_functional_reviewer ?technical_owner ?functional_reviewer)
        (technical_owner_impacts_technical_area ?technical_owner ?technical_impact_area)
        (technical_impact_identified ?technical_impact_area)
        (not
          (technical_owner_review_completed ?technical_owner)
        )
      )
    :effect
      (and
        (technical_owner_review_completed ?technical_owner)
        (technical_owner_analysis_complete ?technical_owner)
      )
  )
  (:action attach_supporting_document_to_technical_owner
    :parameters (?technical_owner - technical_owner ?technical_impact_area - technical_impact_area ?supporting_document - supporting_document)

      (and
        (ready_for_assessment ?technical_owner)
        (technical_owner_impacts_technical_area ?technical_owner ?technical_impact_area)

          (technical_owner_review_completed ?technical_owner)

        (technical_impact_has_supporting_document ?technical_impact_area)

          (supporting_document_available ?supporting_document)

        (technical_owner_has_supporting_document ?technical_owner ?supporting_document)
        (not
          (technical_owner_analysis_complete ?technical_owner)
        )
      )
    :effect
      (and
        (technical_impact_identified ?technical_impact_area)
        (technical_owner_analysis_complete ?technical_owner)
        (supporting_document_available ?supporting_document)
        (not
          (technical_owner_has_supporting_document ?technical_owner ?supporting_document)
        )
      )
  )
  (:action identify_quality_impact_area
    :parameters (?process_owner - process_owner ?quality_impact_area - quality_impact_area ?subject_matter_expert - subject_matter_expert)

          (process_owner_has_supporting_document ?process_owner ?supporting_document)

      (and
        (technical_owner_review_completed ?technical_owner)

        (process_owner_impacts_quality_area ?process_owner ?quality_impact_area)

          (change_dossier_available ?change_dossier)

      (and
        (technical_owner_review_completed ?technical_owner)

        (process_owner_impacts_quality_area ?process_owner ?quality_impact_area)
        (technical_impact_identified ?technical_impact_area)
        (quality_impact_has_supporting_document ?quality_impact_area)
        (technical_owner_analysis_complete ?technical_owner)

          (process_owner_analysis_complete ?process_owner)

        (change_dossier_available ?change_dossier)

        (dossier_requires_regulatory_attachment ?change_dossier)
        (not
          (change_dossier_available ?change_dossier)

  (:action finalize_evidence_and_attach_to_dossier

    :precondition
      (and
        (ready_for_assessment ?impact_assessment)

        (evidence_item_attached_to_assessment ?evidence_item)
        (evidence_item_attached_to_dossier ?evidence_item ?change_dossier)
        (not
          (evidence_item_available ?evidence_item)
        )
      )
  )

    :precondition
      (and
        (ready_for_assessment ?impact_assessment)

        (assigned_sme ?impact_assessment ?subject_matter_expert)
        (not
          (dossier_requires_validation_protocol ?change_dossier)
        )
        (not
          (assessment_ready_for_validation ?impact_assessment)
        )
      )
    :effect (assessment_ready_for_validation ?impact_assessment)
  )
  (:action attach_regulatory_attachment_to_assessment
    :parameters (?impact_assessment - impact_assessment ?regulatory_attachment - regulatory_attachment)
    :precondition
      (and
        (ready_for_assessment ?impact_assessment)
        (regulatory_attachment_available ?regulatory_attachment)
        (not
          (assessment_has_regulatory_attachment_attached ?impact_assessment)
        )
      )
    :effect
      (and
        (assessment_has_regulatory_attachment_attached ?impact_assessment)
        (assessment_has_regulatory_attachment ?impact_assessment ?regulatory_attachment)
        (not
          (regulatory_attachment_available ?regulatory_attachment)

      )

  (:action integrate_regulatory_attachment_with_assessment

    :precondition
      (and
        (ready_for_assessment ?impact_assessment)
        (assessment_has_evidence_item ?impact_assessment ?evidence_item)
