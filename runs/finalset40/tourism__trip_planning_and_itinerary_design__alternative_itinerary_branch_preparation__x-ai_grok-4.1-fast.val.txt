--- domain.pddl (as passed to VAL) ---
(define (domain alternative_itinerary_branch_preparation)
  (:requirements :strips :typing :negative-preconditions)
  (:types planning_asset - object offerable - object service_resource - object itinerary_root - object itinerary_element - itinerary_root transport_option - planning_asset activity_template - planning_asset service_provider - planning_asset attraction_category - planning_asset logistics_option - planning_asset traveler_resource - planning_asset amenity - planning_asset availability_constraint - planning_asset activity_option - offerable venue_feature - offerable promotion - offerable location_context_primary - service_resource location_context_secondary - service_resource proposal_bundle - service_resource day_group - itinerary_element segment_group - itinerary_element day_slot_primary - day_group day_slot_secondary - day_group segment_proposal - segment_group)
  (:predicates
    (itinerary_element_initialized ?itinerary_element - itinerary_element)
    (itinerary_element_confirmed ?itinerary_element - itinerary_element)
    (itinerary_element_transport_reserved ?itinerary_element - itinerary_element)
    (itinerary_element_finalized ?itinerary_element - itinerary_element)
    (itinerary_element_validated ?itinerary_element - itinerary_element)
    (itinerary_element_traveler_resource_linked ?itinerary_element - itinerary_element)
    (transport_option_available ?transport_option - transport_option)
    (itinerary_element_assigned_transport ?itinerary_element - itinerary_element ?transport_option - transport_option)
    (activity_template_available ?activity_template - activity_template)
    (itinerary_element_assigned_activity_template ?itinerary_element - itinerary_element ?activity_template - activity_template)
    (service_provider_available ?service_provider - service_provider)
    (itinerary_element_assigned_service_provider ?itinerary_element - itinerary_element ?service_provider - service_provider)
    (activity_option_available ?activity_option - activity_option)
    (day_slot_primary_has_activity_option ?day_slot_a - day_slot_primary ?activity_option - activity_option)
    (day_slot_secondary_has_activity_option ?day_slot_b - day_slot_secondary ?activity_option - activity_option)
    (day_slot_primary_has_location_context_primary ?day_slot_a - day_slot_primary ?location_context_a - location_context_primary)
    (location_context_primary_is_chosen ?location_context_a - location_context_primary)
    (location_context_primary_is_alternative ?location_context_a - location_context_primary)
    (day_slot_primary_pacing_indicator ?day_slot_a - day_slot_primary)
    (day_slot_secondary_has_location_context_secondary ?day_slot_b - day_slot_secondary ?location_context_b - location_context_secondary)
    (location_context_secondary_is_chosen ?location_context_b - location_context_secondary)
    (location_context_secondary_is_alternative ?location_context_b - location_context_secondary)
    (day_slot_secondary_pacing_indicator ?day_slot_b - day_slot_secondary)
    (proposal_bundle_available ?proposal_bundle - proposal_bundle)
    (proposal_bundle_assembled ?proposal_bundle - proposal_bundle)
    (proposal_bundle_associated_location_context_primary ?proposal_bundle - proposal_bundle ?location_context_a - location_context_primary)
    (proposal_bundle_associated_location_context_secondary ?proposal_bundle - proposal_bundle ?location_context_b - location_context_secondary)
    (proposal_bundle_primary_context_flag ?proposal_bundle - proposal_bundle)
    (proposal_bundle_secondary_context_flag ?proposal_bundle - proposal_bundle)
    (proposal_bundle_ready_for_venue_assignment ?proposal_bundle - proposal_bundle)
    (segment_proposal_has_day_slot_primary ?segment_proposal - segment_proposal ?day_slot_a - day_slot_primary)
    (segment_proposal_has_day_slot_secondary ?segment_proposal - segment_proposal ?day_slot_b - day_slot_secondary)
    (segment_proposal_in_proposal_bundle ?segment_proposal - segment_proposal ?proposal_bundle - proposal_bundle)
    (venue_feature_available ?venue_feature - venue_feature)
    (segment_proposal_has_venue_feature ?segment_proposal - segment_proposal ?venue_feature - venue_feature)
    (venue_feature_claimed ?venue_feature - venue_feature)
    (venue_feature_assigned_to_proposal_bundle ?venue_feature - venue_feature ?proposal_bundle - proposal_bundle)
    (segment_venue_linked ?segment_proposal - segment_proposal)
    (segment_venue_requirements_satisfied ?segment_proposal - segment_proposal)
    (segment_prevalidation_complete ?segment_proposal - segment_proposal)
    (segment_category_attached ?segment_proposal - segment_proposal)
    (segment_venue_secondary_verified ?segment_proposal - segment_proposal)
    (segment_post_venue_checks_passed ?segment_proposal - segment_proposal)
    (segment_final_checks_passed ?segment_proposal - segment_proposal)
    (promotion_available ?promotion - promotion)
    (segment_assigned_promotion ?segment_proposal - segment_proposal ?promotion - promotion)
    (promotion_applied_to_segment ?segment_proposal - segment_proposal)
    (segment_provider_enriched ?segment_proposal - segment_proposal)
    (segment_provider_enrichment_confirmed ?segment_proposal - segment_proposal)
    (attraction_category_available ?attraction_category - attraction_category)
    (segment_assigned_attraction_category ?segment_proposal - segment_proposal ?attraction_category - attraction_category)
    (logistics_option_available ?logistics_option - logistics_option)
    (segment_assigned_logistics_option ?segment_proposal - segment_proposal ?logistics_option - logistics_option)
    (amenity_available ?amenity - amenity)
    (segment_assigned_amenity ?segment_proposal - segment_proposal ?amenity - amenity)
    (availability_constraint_available ?availability_constraint - availability_constraint)
    (segment_assigned_availability_constraint ?segment_proposal - segment_proposal ?availability_constraint - availability_constraint)
    (traveler_resource_available ?traveler_resource - traveler_resource)
    (itinerary_element_assigned_traveler_resource ?itinerary_element - itinerary_element ?traveler_resource - traveler_resource)
    (day_slot_primary_pacing_locked ?day_slot_a - day_slot_primary)
    (day_slot_secondary_pacing_locked ?day_slot_b - day_slot_secondary)
    (segment_validation_marker ?segment_proposal - segment_proposal)
  )
  (:action initialize_itinerary_element
    :parameters (?itinerary_element - itinerary_element)
    :precondition
      (and
        (not
          (itinerary_element_initialized ?itinerary_element)
        )
        (not
          (itinerary_element_finalized ?itinerary_element)
        )
      )
    :effect (itinerary_element_initialized ?itinerary_element)
  )
  (:action attach_transport_to_itinerary_element
    :parameters (?itinerary_element - itinerary_element ?transport_option - transport_option)
    :precondition
      (and
        (itinerary_element_initialized ?itinerary_element)
        (not
          (itinerary_element_transport_reserved ?itinerary_element)
        )
        (transport_option_available ?transport_option)
      )
    :effect
      (and
        (itinerary_element_transport_reserved ?itinerary_element)
        (itinerary_element_assigned_transport ?itinerary_element ?transport_option)
        (not
          (transport_option_available ?transport_option)
        )
      )
  )
  (:action assign_activity_template_to_itinerary_element
    :parameters (?itinerary_element - itinerary_element ?activity_template - activity_template)
    :precondition
      (and
        (itinerary_element_initialized ?itinerary_element)
        (itinerary_element_transport_reserved ?itinerary_element)
        (activity_template_available ?activity_template)
      )
    :effect
      (and
        (itinerary_element_assigned_activity_template ?itinerary_element ?activity_template)
        (not
          (activity_template_available ?activity_template)
        )
      )
  )
  (:action confirm_itinerary_element
    :parameters (?itinerary_element - itinerary_element ?activity_template - activity_template)
    :precondition
      (and
        (itinerary_element_initialized ?itinerary_element)
        (itinerary_element_transport_reserved ?itinerary_element)
        (itinerary_element_assigned_activity_template ?itinerary_element ?activity_template)
        (not
          (itinerary_element_confirmed ?itinerary_element)
        )
      )
    :effect (itinerary_element_confirmed ?itinerary_element)
  )
  (:action release_activity_template_from_itinerary_element
    :parameters (?itinerary_element - itinerary_element ?activity_template - activity_template)
    :precondition
      (and
        (itinerary_element_assigned_activity_template ?itinerary_element ?activity_template)
      )
    :effect
      (and
        (activity_template_available ?activity_template)
        (not
          (itinerary_element_assigned_activity_template ?itinerary_element ?activity_template)
        )
      )
  )
  (:action attach_service_provider_to_itinerary_element
    :parameters (?itinerary_element - itinerary_element ?service_provider - service_provider)
    :precondition
      (and
        (itinerary_element_confirmed ?itinerary_element)
        (service_provider_available ?service_provider)
      )
    :effect
      (and
        (itinerary_element_assigned_service_provider ?itinerary_element ?service_provider)
        (not
          (service_provider_available ?service_provider)
        )
      )
  )
  (:action detach_service_provider_from_itinerary_element
    :parameters (?itinerary_element - itinerary_element ?service_provider - service_provider)
    :precondition
      (and
        (itinerary_element_assigned_service_provider ?itinerary_element ?service_provider)
      )
    :effect
      (and
        (service_provider_available ?service_provider)
        (not
          (itinerary_element_assigned_service_provider ?itinerary_element ?service_provider)
        )
      )
  )
  (:action assign_amenity_to_segment_proposal
    :parameters (?segment_proposal - segment_proposal ?amenity - amenity)
    :precondition
      (and
        (itinerary_element_confirmed ?segment_proposal)
        (amenity_available ?amenity)
      )
    :effect
      (and
        (segment_assigned_amenity ?segment_proposal ?amenity)
        (not
          (amenity_available ?amenity)
        )
      )
  )
  (:action remove_amenity_from_segment_proposal
    :parameters (?segment_proposal - segment_proposal ?amenity - amenity)
    :precondition
      (and
        (segment_assigned_amenity ?segment_proposal ?amenity)
      )
    :effect
      (and
        (amenity_available ?amenity)
        (not
          (segment_assigned_amenity ?segment_proposal ?amenity)
        )
      )
  )
  (:action assign_availability_constraint_to_segment_proposal
    :parameters (?segment_proposal - segment_proposal ?availability_constraint - availability_constraint)
    :precondition
      (and
        (itinerary_element_confirmed ?segment_proposal)
        (availability_constraint_available ?availability_constraint)
      )
    :effect
      (and
        (segment_assigned_availability_constraint ?segment_proposal ?availability_constraint)
        (not
          (availability_constraint_available ?availability_constraint)
        )
      )
  )
  (:action remove_availability_constraint_from_segment_proposal
    :parameters (?segment_proposal - segment_proposal ?availability_constraint - availability_constraint)
    :precondition
      (and
        (segment_assigned_availability_constraint ?segment_proposal ?availability_constraint)
      )
    :effect
      (and
        (availability_constraint_available ?availability_constraint)
        (not
          (segment_assigned_availability_constraint ?segment_proposal ?availability_constraint)
        )
      )
  )
  (:action mark_location_context_primary_primary_for_day_slot
    :parameters (?day_slot_a - day_slot_primary ?location_context_a - location_context_primary ?activity_template - activity_template)
    :precondition
      (and
        (itinerary_element_confirmed ?day_slot_a)
        (itinerary_element_assigned_activity_template ?day_slot_a ?activity_template)
        (day_slot_primary_has_location_context_primary ?day_slot_a ?location_context_a)
        (not
          (location_context_primary_is_chosen ?location_context_a)
        )
        (not
          (location_context_primary_is_alternative ?location_context_a)
        )
      )
    :effect (location_context_primary_is_chosen ?location_context_a)
  )
  (:action allocate_provider_and_mark_pacing_for_day_slot_primary
    :parameters (?day_slot_a - day_slot_primary ?location_context_a - location_context_primary ?service_provider - service_provider)
    :precondition
      (and
        (itinerary_element_confirmed ?day_slot_a)
        (itinerary_element_assigned_service_provider ?day_slot_a ?service_provider)
        (day_slot_primary_has_location_context_primary ?day_slot_a ?location_context_a)
        (location_context_primary_is_chosen ?location_context_a)
        (not
          (day_slot_primary_pacing_locked ?day_slot_a)
        )
      )
    :effect
      (and
        (day_slot_primary_pacing_locked ?day_slot_a)
        (day_slot_primary_pacing_indicator ?day_slot_a)
      )
  )
  (:action assign_activity_option_and_mark_context_secondary_for_day_slot_primary
    :parameters (?day_slot_a - day_slot_primary ?location_context_a - location_context_primary ?activity_option - activity_option)
    :precondition
      (and
        (itinerary_element_confirmed ?day_slot_a)
        (day_slot_primary_has_location_context_primary ?day_slot_a ?location_context_a)
        (activity_option_available ?activity_option)
        (not
          (day_slot_primary_pacing_locked ?day_slot_a)
        )
      )
    :effect
      (and
        (location_context_primary_is_alternative ?location_context_a)
        (day_slot_primary_pacing_locked ?day_slot_a)
        (day_slot_primary_has_activity_option ?day_slot_a ?activity_option)
        (not
          (activity_option_available ?activity_option)
        )
      )
  )
  (:action promote_activity_option_and_update_pacing_for_day_slot_primary
    :parameters (?day_slot_a - day_slot_primary ?location_context_a - location_context_primary ?activity_template - activity_template ?activity_option - activity_option)
    :precondition
      (and
        (itinerary_element_confirmed ?day_slot_a)
        (itinerary_element_assigned_activity_template ?day_slot_a ?activity_template)
        (day_slot_primary_has_location_context_primary ?day_slot_a ?location_context_a)
        (location_context_primary_is_alternative ?location_context_a)
        (day_slot_primary_has_activity_option ?day_slot_a ?activity_option)
        (not
          (day_slot_primary_pacing_indicator ?day_slot_a)
        )
      )
    :effect
      (and
        (location_context_primary_is_chosen ?location_context_a)
        (day_slot_primary_pacing_indicator ?day_slot_a)
        (activity_option_available ?activity_option)
        (not
          (day_slot_primary_has_activity_option ?day_slot_a ?activity_option)
        )
      )
  )
  (:action mark_location_context_secondary_primary_for_day_slot
    :parameters (?day_slot_b - day_slot_secondary ?location_context_b - location_context_secondary ?activity_template - activity_template)
    :precondition
      (and
        (itinerary_element_confirmed ?day_slot_b)
        (itinerary_element_assigned_activity_template ?day_slot_b ?activity_template)
        (day_slot_secondary_has_location_context_secondary ?day_slot_b ?location_context_b)
        (not
          (location_context_secondary_is_chosen ?location_context_b)
        )
        (not
          (location_context_secondary_is_alternative ?location_context_b)
        )
      )
    :effect (location_context_secondary_is_chosen ?location_context_b)
  )
  (:action allocate_provider_and_mark_pacing_for_day_slot_secondary
    :parameters (?day_slot_b - day_slot_secondary ?location_context_b - location_context_secondary ?service_provider - service_provider)
    :precondition
      (and
        (itinerary_element_confirmed ?day_slot_b)
        (itinerary_element_assigned_service_provider ?day_slot_b ?service_provider)
        (day_slot_secondary_has_location_context_secondary ?day_slot_b ?location_context_b)
        (location_context_secondary_is_chosen ?location_context_b)
        (not
          (day_slot_secondary_pacing_locked ?day_slot_b)
        )
      )
    :effect
      (and
        (day_slot_secondary_pacing_locked ?day_slot_b)
        (day_slot_secondary_pacing_indicator ?day_slot_b)
      )
  )
  (:action assign_activity_option_and_mark_context_secondary_for_day_slot_secondary
    :parameters (?day_slot_b - day_slot_secondary ?location_context_b - location_context_secondary ?activity_option - activity_option)
    :precondition
      (and
        (itinerary_element_confirmed ?day_slot_b)
        (day_slot_secondary_has_location_context_secondary ?day_slot_b ?location_context_b)
        (activity_option_available ?activity_option)
        (not
          (day_slot_secondary_pacing_locked ?day_slot_b)
        )
      )
    :effect
      (and
        (location_context_secondary_is_alternative ?location_context_b)
        (day_slot_secondary_pacing_locked ?day_slot_b)
        (day_slot_secondary_has_activity_option ?day_slot_b ?activity_option)
        (not
          (activity_option_available ?activity_option)
        )
      )
  )
  (:action promote_activity_option_and_update_pacing_for_day_slot_secondary
    :parameters (?day_slot_b - day_slot_secondary ?location_context_b - location_context_secondary ?activity_template - activity_template ?activity_option - activity_option)
    :precondition
      (and
        (itinerary_element_confirmed ?day_slot_b)
        (itinerary_element_assigned_activity_template ?day_slot_b ?activity_template)
        (day_slot_secondary_has_location_context_secondary ?day_slot_b ?location_context_b)
        (location_context_secondary_is_alternative ?location_context_b)
        (day_slot_secondary_has_activity_option ?day_slot_b ?activity_option)
        (not
          (day_slot_secondary_pacing_indicator ?day_slot_b)
        )
      )
    :effect
      (and
        (location_context_secondary_is_chosen ?location_context_b)
        (day_slot_secondary_pacing_indicator ?day_slot_b)
        (activity_option_available ?activity_option)
        (not
          (day_slot_secondary_has_activity_option ?day_slot_b ?activity_option)
        )
      )
  )
  (:action assemble_proposal_bundle_primary_primary
    :parameters (?day_slot_a - day_slot_primary ?day_slot_b - day_slot_secondary ?location_context_a - location_context_primary ?location_context_b - location_context_secondary ?proposal_bundle - proposal_bundle)
    :precondition
      (and
        (day_slot_primary_pacing_locked ?day_slot_a)
        (day_slot_secondary_pacing_locked ?day_slot_b)
        (day_slot_primary_has_location_context_primary ?day_slot_a ?location_context_a)
        (day_slot_secondary_has_location_context_secondary ?day_slot_b ?location_context_b)
        (location_context_primary_is_chosen ?location_context_a)
        (location_context_secondary_is_chosen ?location_context_b)
        (day_slot_primary_pacing_indicator ?day_slot_a)
        (day_slot_secondary_pacing_indicator ?day_slot_b)
        (proposal_bundle_available ?proposal_bundle)
      )
    :effect
      (and
        (proposal_bundle_assembled ?proposal_bundle)
        (proposal_bundle_associated_location_context_primary ?proposal_bundle ?location_context_a)
        (proposal_bundle_associated_location_context_secondary ?proposal_bundle ?location_context_b)
        (not
          (proposal_bundle_available ?proposal_bundle)
        )
      )
  )
  (:action assemble_proposal_bundle_secondary_a_primary_b
    :parameters (?day_slot_a - day_slot_primary ?day_slot_b - day_slot_secondary ?location_context_a - location_context_primary ?location_context_b - location_context_secondary ?proposal_bundle - proposal_bundle)
    :precondition
      (and
        (day_slot_primary_pacing_locked ?day_slot_a)
        (day_slot_secondary_pacing_locked ?day_slot_b)
        (day_slot_primary_has_location_context_primary ?day_slot_a ?location_context_a)
        (day_slot_secondary_has_location_context_secondary ?day_slot_b ?location_context_b)
        (location_context_primary_is_alternative ?location_context_a)
        (location_context_secondary_is_chosen ?location_context_b)
        (not
          (day_slot_primary_pacing_indicator ?day_slot_a)
        )
        (day_slot_secondary_pacing_indicator ?day_slot_b)
        (proposal_bundle_available ?proposal_bundle)
      )
    :effect
      (and
        (proposal_bundle_assembled ?proposal_bundle)
        (proposal_bundle_associated_location_context_primary ?proposal_bundle ?location_context_a)
        (proposal_bundle_associated_location_context_secondary ?proposal_bundle ?location_context_b)
        (proposal_bundle_primary_context_flag ?proposal_bundle)
        (not
          (proposal_bundle_available ?proposal_bundle)
        )
      )
  )
  (:action assemble_proposal_bundle_primary_a_secondary_b
    :parameters (?day_slot_a - day_slot_primary ?day_slot_b - day_slot_secondary ?location_context_a - location_context_primary ?location_context_b - location_context_secondary ?proposal_bundle - proposal_bundle)
    :precondition
      (and
        (day_slot_primary_pacing_locked ?day_slot_a)
        (day_slot_secondary_pacing_locked ?day_slot_b)
        (day_slot_primary_has_location_context_primary ?day_slot_a ?location_context_a)
        (day_slot_secondary_has_location_context_secondary ?day_slot_b ?location_context_b)
        (location_context_primary_is_chosen ?location_context_a)
        (location_context_secondary_is_alternative ?location_context_b)
        (day_slot_primary_pacing_indicator ?day_slot_a)
        (not
          (day_slot_secondary_pacing_indicator ?day_slot_b)
        )
        (proposal_bundle_available ?proposal_bundle)
      )
    :effect
      (and
        (proposal_bundle_assembled ?proposal_bundle)
        (proposal_bundle_associated_location_context_primary ?proposal_bundle ?location_context_a)
        (proposal_bundle_associated_location_context_secondary ?proposal_bundle ?location_context_b)
        (proposal_bundle_secondary_context_flag ?proposal_bundle)
        (not
          (proposal_bundle_available ?proposal_bundle)
        )
      )
  )
  (:action assemble_proposal_bundle_secondary_both
    :parameters (?day_slot_a - day_slot_primary ?day_slot_b - day_slot_secondary ?location_context_a - location_context_primary ?location_context_b - location_context_secondary ?proposal_bundle - proposal_bundle)
    :precondition
      (and
        (day_slot_primary_pacing_locked ?day_slot_a)
        (day_slot_secondary_pacing_locked ?day_slot_b)
        (day_slot_primary_has_location_context_primary ?day_slot_a ?location_context_a)
        (day_slot_secondary_has_location_context_secondary ?day_slot_b ?location_context_b)
        (location_context_primary_is_alternative ?location_context_a)
        (location_context_secondary_is_alternative ?location_context_b)
        (not
          (day_slot_primary_pacing_indicator ?day_slot_a)
        )
        (not
          (day_slot_secondary_pacing_indicator ?day_slot_b)
        )
        (proposal_bundle_available ?proposal_bundle)
      )
    :effect
      (and
        (proposal_bundle_assembled ?proposal_bundle)
        (proposal_bundle_associated_location_context_primary ?proposal_bundle ?location_context_a)
        (proposal_bundle_associated_location_context_secondary ?proposal_bundle ?location_context_b)
        (proposal_bundle_primary_context_flag ?proposal_bundle)
        (proposal_bundle_secondary_context_flag ?proposal_bundle)
        (not
          (proposal_bundle_available ?proposal_bundle)
        )
      )
  )
  (:action mark_proposal_bundle_ready_for_venue_assignment
    :parameters (?proposal_bundle - proposal_bundle ?day_slot_a - day_slot_primary ?activity_template - activity_template)
    :precondition
      (and
        (proposal_bundle_assembled ?proposal_bundle)
        (day_slot_primary_pacing_locked ?day_slot_a)
        (itinerary_element_assigned_activity_template ?day_slot_a ?activity_template)
        (not
          (proposal_bundle_ready_for_venue_assignment ?proposal_bundle)
        )
      )
    :effect (proposal_bundle_ready_for_venue_assignment ?proposal_bundle)
  )
  (:action attach_venue_feature_to_segment_proposal
    :parameters (?segment_proposal - segment_proposal ?venue_feature - venue_feature ?proposal_bundle - proposal_bundle)
    :precondition
      (and
        (itinerary_element_confirmed ?segment_proposal)
        (segment_proposal_in_proposal_bundle ?segment_proposal ?proposal_bundle)
        (segment_proposal_has_venue_feature ?segment_proposal ?venue_feature)
        (venue_feature_available ?venue_feature)
        (proposal_bundle_assembled ?proposal_bundle)
        (proposal_bundle_ready_for_venue_assignment ?proposal_bundle)
        (not
          (venue_feature_claimed ?venue_feature)
        )
      )
    :effect
      (and
        (venue_feature_claimed ?venue_feature)
        (venue_feature_assigned_to_proposal_bundle ?venue_feature ?proposal_bundle)
        (not
          (venue_feature_available ?venue_feature)
        )
      )
  )
